11   @pytest.mark.xfail(_IS_WASM, reason="cannot start subprocess")
12   def test_import_raises_warning():
13       code = """ 
14       import pytest 
15       with pytest.warns(UserWarning, match="it is not needed to import"): 
16           from sklearn.experimental import enable_hist_gradient_boosting  # noqa 
17       """
18       pattern = "it is not needed to import enable_hist_gradient_boosting anymore"
19       assert_run_python_script_without_output(textwrap.dedent(code), pattern=pattern)
