31   def _gs_decorrelation(w, W, j):
32       """ 
33       Orthonormalize w wrt the first j rows of W. 
34    
35       Parameters 
36       ---------- 
37       w : ndarray of shape (n,) 
38           Array to be orthogonalized 
39    
40       W : ndarray of shape (p, n) 
41           Null space definition 
42    
43       j : int < p 
44           The no of (from the first) rows of Null space W wrt which w is 
45           orthogonalized. 
46    
47       Notes 
48       ----- 
49       Assumes that W is orthogonal 
50       w changed in place 
51       """
52       w -= np.linalg.multi_dot([w, W[:j].T, W[:j]])
53       return w

56   def _sym_decorrelation(W):
57       """Symmetric decorrelation 
58       i.e. W <- (W * W.T) ^{-1/2} * W 
59       """
60       s, u = linalg.eigh(np.dot(W, W.T))
61       # Avoid sqrt of negative values because of rounding errors. Note that
62       # np.sqrt(tiny) is larger than tiny and therefore this clipping also
63       # prevents division by zero in the next step.
64       s = np.clip(s, a_min=np.finfo(W.dtype).tiny, a_max=None)
65   
66       # u (resp. s) contains the eigenvectors (resp. square roots of
67       # the eigenvalues) of W * W.T
68       return np.linalg.multi_dot([u * (1.0 / np.sqrt(s)), u.T, W])

71   def _ica_def(X, tol, g, fun_args, max_iter, w_init):
72       """Deflationary FastICA using fun approx to neg-entropy function 
73    
74       Used internally by FastICA. 
75       """
76   
77       n_components = w_init.shape[0]
78       W = np.zeros((n_components, n_components), dtype=X.dtype)
79       n_iter = []
80   
81       # j is the index of the extracted component
82       for j in range(n_components):
83           w = w_init[j, :].copy()
84           w /= np.sqrt((w**2).sum())
85   
86           for i in range(max_iter):
87               gwtx, g_wtx = g(np.dot(w.T, X), fun_args)
88   
89               w1 = (X * gwtx).mean(axis=1) - g_wtx.mean() * w
90   
91               _gs_decorrelation(w1, W, j)
92   
93               w1 /= np.sqrt((w1**2).sum())
94   
95               lim = np.abs(np.abs((w1 * w).sum()) - 1)
96               w = w1
97               if lim < tol:
98                   break
99   
100          n_iter.append(i + 1)
101          W[j, :] = w
102  
103      return W, max(n_iter)

106  def _ica_par(X, tol, g, fun_args, max_iter, w_init):
107      """Parallel FastICA. 
108   
109      Used internally by FastICA --main loop 
110   
111      """
112      W = _sym_decorrelation(w_init)
113      del w_init
114      p_ = float(X.shape[1])
115      for ii in range(max_iter):
116          gwtx, g_wtx = g(np.dot(W, X), fun_args)
117          W1 = _sym_decorrelation(np.dot(gwtx, X.T) / p_ - g_wtx[:, np.newaxis] * W)
118          del gwtx, g_wtx
119          # builtin max, abs are faster than numpy counter parts.
120          # np.einsum allows having the lowest memory footprint.
121          # It is faster than np.diag(np.dot(W1, W.T)).
122          lim = max(abs(abs(np.einsum("ij,ij->i", W1, W)) - 1))
123          W = W1
124          if lim < tol:
125              break
126      else:
127          warnings.warn(
128              (
129                  "FastICA did not converge. Consider increasing "
130                  "tolerance or the maximum number of iterations."
131              ),
132              ConvergenceWarning,
133          )
134  
135      return W, ii + 1

140  def _logcosh(x, fun_args=None):
141      alpha = fun_args.get("alpha", 1.0)  # comment it out?
142  
143      x *= alpha
144      gx = np.tanh(x, x)  # apply the tanh inplace
145      g_x = np.empty(x.shape[0], dtype=x.dtype)
146      # XXX compute in chunks to avoid extra allocation
147      for i, gx_i in enumerate(gx):  # please don't vectorize.
148          g_x[i] = (alpha * (1 - gx_i**2)).mean()
149      return gx, g_x

152  def _exp(x, fun_args):
153      exp = np.exp(-(x**2) / 2)
154      gx = x * exp
155      g_x = (1 - x**2) * exp
156      return gx, g_x.mean(axis=-1)

159  def _cube(x, fun_args):
160      return x**3, (3 * x**2).mean(axis=-1)

172  def fastica(
173      X,
174      n_components=None,
175      *,
176      algorithm="parallel",
177      whiten="unit-variance",
178      fun="logcosh",
179      fun_args=None,
180      max_iter=200,
181      tol=1e-04,
182      w_init=None,
183      whiten_solver="svd",
184      random_state=None,
185      return_X_mean=False,
186      compute_sources=True,
187      return_n_iter=False,
188  ):
189      """Perform Fast Independent Component Analysis. 
190   
191      The implementation is based on [1]_. 
192   
193      Read more in the :ref:`User Guide <ICA>`. 
194   
195      Parameters 
196      ---------- 
197      X : array-like of shape (n_samples, n_features) 
198          Training vector, where `n_samples` is the number of samples and 
199          `n_features` is the number of features. 
200   
201      n_components : int, default=None 
202          Number of components to use. If None is passed, all are used. 
203   
204      algorithm : {'parallel', 'deflation'}, default='parallel' 
205          Specify which algorithm to use for FastICA. 
206   
207      whiten : str or bool, default='unit-variance' 
208          Specify the whitening strategy to use. 
209   
210          - If 'arbitrary-variance', a whitening with variance 
211            arbitrary is used. 
212          - If 'unit-variance', the whitening matrix is rescaled to ensure that 
213            each recovered source has unit variance. 
214          - If False, the data is already considered to be whitened, and no 
215            whitening is performed. 
216   
217          .. versionchanged:: 1.3 
218              The default value of `whiten` changed to 'unit-variance' in 1.3. 
219   
220      fun : {'logcosh', 'exp', 'cube'} or callable, default='logcosh' 
221          The functional form of the G function used in the 
222          approximation to neg-entropy. Could be either 'logcosh', 'exp', 
223          or 'cube'. 
224          You can also provide your own function. It should return a tuple 
225          containing the value of the function, and of its derivative, in the 
226          point. The derivative should be averaged along its last dimension. 
227          Example:: 
228   
229              def my_g(x): 
230                  return x ** 3, (3 * x ** 2).mean(axis=-1) 
231   
232      fun_args : dict, default=None 
233          Arguments to send to the functional form. 
234          If empty or None and if fun='logcosh', fun_args will take value 
235          {'alpha' : 1.0}. 
236   
237      max_iter : int, default=200 
238          Maximum number of iterations to perform. 
239   
240      tol : float, default=1e-4 
241          A positive scalar giving the tolerance at which the 
242          un-mixing matrix is considered to have converged. 
243   
244      w_init : ndarray of shape (n_components, n_components), default=None 
245          Initial un-mixing array. If `w_init=None`, then an array of values 
246          drawn from a normal distribution is used. 
247   
248      whiten_solver : {"eigh", "svd"}, default="svd" 
249          The solver to use for whitening. 
250   
251          - "svd" is more stable numerically if the problem is degenerate, and 
252            often faster when `n_samples <= n_features`. 
253   
254          - "eigh" is generally more memory efficient when 
255            `n_samples >= n_features`, and can be faster when 
256            `n_samples >= 50 * n_features`. 
257   
258          .. versionadded:: 1.2 
259   
260      random_state : int, RandomState instance or None, default=None 
261          Used to initialize ``w_init`` when not specified, with a 
262          normal distribution. Pass an int, for reproducible results 
263          across multiple function calls. 
264          See :term:`Glossary <random_state>`. 
265   
266      return_X_mean : bool, default=False 
267          If True, X_mean is returned too. 
268   
269      compute_sources : bool, default=True 
270          If False, sources are not computed, but only the rotation matrix. 
271          This can save memory when working with big data. Defaults to True. 
272   
273      return_n_iter : bool, default=False 
274          Whether or not to return the number of iterations. 
275   
276      Returns 
277      ------- 
278      K : ndarray of shape (n_components, n_features) or None 
279          If whiten is 'True', K is the pre-whitening matrix that projects data 
280          onto the first n_components principal components. If whiten is 'False', 
281          K is 'None'. 
282   
283      W : ndarray of shape (n_components, n_components) 
284          The square matrix that unmixes the data after whitening. 
285          The mixing matrix is the pseudo-inverse of matrix ``W K`` 
286          if K is not None, else it is the inverse of W. 
287   
288      S : ndarray of shape (n_samples, n_components) or None 
289          Estimated source matrix. 
290   
291      X_mean : ndarray of shape (n_features,) 
292          The mean over features. Returned only if return_X_mean is True. 
293   
294      n_iter : int 
295          If the algorithm is "deflation", n_iter is the 
296          maximum number of iterations run across all components. Else 
297          they are just the number of iterations taken to converge. This is 
298          returned only when return_n_iter is set to `True`. 
299   
300      Notes 
301      ----- 
302      The data matrix X is considered to be a linear combination of 
303      non-Gaussian (independent) components i.e. X = AS where columns of S 
304      contain the independent components and A is a linear mixing 
305      matrix. In short ICA attempts to `un-mix' the data by estimating an 
306      un-mixing matrix W where ``S = W K X.`` 
307      While FastICA was proposed to estimate as many sources 
308      as features, it is possible to estimate less by setting 
309      n_components < n_features. It this case K is not a square matrix 
310      and the estimated A is the pseudo-inverse of ``W K``. 
311   
312      This implementation was originally made for data of shape 
313      [n_features, n_samples]. Now the input is transposed 
314      before the algorithm is applied. This makes it slightly 
315      faster for Fortran-ordered input. 
316   
317      References 
318      ---------- 
319      .. [1] A. Hyvarinen and E. Oja, "Fast Independent Component Analysis", 
320             Algorithms and Applications, Neural Networks, 13(4-5), 2000, 
321             pp. 411-430. 
322   
323      Examples 
324      -------- 
325      >>> from sklearn.datasets import load_digits 
326      >>> from sklearn.decomposition import fastica 
327      >>> X, _ = load_digits(return_X_y=True) 
328      >>> K, W, S = fastica(X, n_components=7, random_state=0, whiten='unit-variance') 
329      >>> K.shape 
330      (7, 64) 
331      >>> W.shape 
332      (7, 7) 
333      >>> S.shape 
334      (1797, 7) 
335      """
336      est = FastICA(
337          n_components=n_components,
338          algorithm=algorithm,
339          whiten=whiten,
340          fun=fun,
341          fun_args=fun_args,
342          max_iter=max_iter,
343          tol=tol,
344          w_init=w_init,
345          whiten_solver=whiten_solver,
346          random_state=random_state,
347      )
348      est._validate_params()
349      S = est._fit_transform(X, compute_sources=compute_sources)
350  
351      if est.whiten in ["unit-variance", "arbitrary-variance"]:
352          K = est.whitening_
353          X_mean = est.mean_
354      else:
355          K = None
356          X_mean = None
357  
358      returned_values = [K, est._unmixing, S]
359      if return_X_mean:
360          returned_values.append(X_mean)
361      if return_n_iter:
362          returned_values.append(est.n_iter_)
363  
364      return returned_values

367  class FastICA(ClassNamePrefixFeaturesOutMixin, TransformerMixin, BaseEstimator):
368      """FastICA: a fast algorithm for Independent Component Analysis. 
369   
370      The implementation is based on [1]_. 
371   
372      Read more in the :ref:`User Guide <ICA>`. 
373   
374      Parameters 
375      ---------- 
376      n_components : int, default=None 
377          Number of components to use. If None is passed, all are used. 
378   
379      algorithm : {'parallel', 'deflation'}, default='parallel' 
380          Specify which algorithm to use for FastICA. 
381   
382      whiten : str or bool, default='unit-variance' 
383          Specify the whitening strategy to use. 
384   
385          - If 'arbitrary-variance', a whitening with variance 
386            arbitrary is used. 
387          - If 'unit-variance', the whitening matrix is rescaled to ensure that 
388            each recovered source has unit variance. 
389          - If False, the data is already considered to be whitened, and no 
390            whitening is performed. 
391   
392          .. versionchanged:: 1.3 
393              The default value of `whiten` changed to 'unit-variance' in 1.3. 
394   
395      fun : {'logcosh', 'exp', 'cube'} or callable, default='logcosh' 
396          The functional form of the G function used in the 
397          approximation to neg-entropy. Could be either 'logcosh', 'exp', 
398          or 'cube'. 
399          You can also provide your own function. It should return a tuple 
400          containing the value of the function, and of its derivative, in the 
401          point. The derivative should be averaged along its last dimension. 
402          Example:: 
403   
404              def my_g(x): 
405                  return x ** 3, (3 * x ** 2).mean(axis=-1) 
406   
407      fun_args : dict, default=None 
408          Arguments to send to the functional form. 
409          If empty or None and if fun='logcosh', fun_args will take value 
410          {'alpha' : 1.0}. 
411   
412      max_iter : int, default=200 
413          Maximum number of iterations during fit. 
414   
415      tol : float, default=1e-4 
416          A positive scalar giving the tolerance at which the 
417          un-mixing matrix is considered to have converged. 
418   
419      w_init : array-like of shape (n_components, n_components), default=None 
420          Initial un-mixing array. If `w_init=None`, then an array of values 
421          drawn from a normal distribution is used. 
422   
423      whiten_solver : {"eigh", "svd"}, default="svd" 
424          The solver to use for whitening. 
425   
426          - "svd" is more stable numerically if the problem is degenerate, and 
427            often faster when `n_samples <= n_features`. 
428   
429          - "eigh" is generally more memory efficient when 
430            `n_samples >= n_features`, and can be faster when 
431            `n_samples >= 50 * n_features`. 
432   
433          .. versionadded:: 1.2 
434   
435      random_state : int, RandomState instance or None, default=None 
436          Used to initialize ``w_init`` when not specified, with a 
437          normal distribution. Pass an int, for reproducible results 
438          across multiple function calls. 
439          See :term:`Glossary <random_state>`. 
440   
441      Attributes 
442      ---------- 
443      components_ : ndarray of shape (n_components, n_features) 
444          The linear operator to apply to the data to get the independent 
445          sources. This is equal to the unmixing matrix when ``whiten`` is 
446          False, and equal to ``np.dot(unmixing_matrix, self.whitening_)`` when 
447          ``whiten`` is True. 
448   
449      mixing_ : ndarray of shape (n_features, n_components) 
450          The pseudo-inverse of ``components_``. It is the linear operator 
451          that maps independent sources to the data. 
452   
453      mean_ : ndarray of shape(n_features,) 
454          The mean over features. Only set if `self.whiten` is True. 
455   
456      n_features_in_ : int 
457          Number of features seen during :term:`fit`. 
458   
459          .. versionadded:: 0.24 
460   
461      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
462          Names of features seen during :term:`fit`. Defined only when `X` 
463          has feature names that are all strings. 
464   
465          .. versionadded:: 1.0 
466   
467      n_iter_ : int 
468          If the algorithm is "deflation", n_iter is the 
469          maximum number of iterations run across all components. Else 
470          they are just the number of iterations taken to converge. 
471   
472      whitening_ : ndarray of shape (n_components, n_features) 
473          Only set if whiten is 'True'. This is the pre-whitening matrix 
474          that projects data onto the first `n_components` principal components. 
475   
476      See Also 
477      -------- 
478      PCA : Principal component analysis (PCA). 
479      IncrementalPCA : Incremental principal components analysis (IPCA). 
480      KernelPCA : Kernel Principal component analysis (KPCA). 
481      MiniBatchSparsePCA : Mini-batch Sparse Principal Components Analysis. 
482      SparsePCA : Sparse Principal Components Analysis (SparsePCA). 
483   
484      References 
485      ---------- 
486      .. [1] A. Hyvarinen and E. Oja, Independent Component Analysis: 
487             Algorithms and Applications, Neural Networks, 13(4-5), 2000, 
488             pp. 411-430. 
489   
490      Examples 
491      -------- 
492      >>> from sklearn.datasets import load_digits 
493      >>> from sklearn.decomposition import FastICA 
494      >>> X, _ = load_digits(return_X_y=True) 
495      >>> transformer = FastICA(n_components=7, 
496      ...         random_state=0, 
497      ...         whiten='unit-variance') 
498      >>> X_transformed = transformer.fit_transform(X) 
499      >>> X_transformed.shape 
500      (1797, 7) 
501      """
502  
503      _parameter_constraints: dict = {
504          "n_components": [Interval(Integral, 1, None, closed="left"), None],
505          "algorithm": [StrOptions({"parallel", "deflation"})],
506          "whiten": [
507              StrOptions({"arbitrary-variance", "unit-variance"}),
508              Options(bool, {False}),
509          ],
510          "fun": [StrOptions({"logcosh", "exp", "cube"}), callable],
511          "fun_args": [dict, None],
512          "max_iter": [Interval(Integral, 1, None, closed="left")],
513          "tol": [Interval(Real, 0.0, None, closed="left")],
514          "w_init": ["array-like", None],
515          "whiten_solver": [StrOptions({"eigh", "svd"})],
516          "random_state": ["random_state"],
517      }

519      def __init__(
520          self,
521          n_components=None,
522          *,
523          algorithm="parallel",
524          whiten="unit-variance",
525          fun="logcosh",
526          fun_args=None,
527          max_iter=200,
528          tol=1e-4,
529          w_init=None,
530          whiten_solver="svd",
531          random_state=None,
532      ):
533          super().__init__()
534          self.n_components = n_components
535          self.algorithm = algorithm
536          self.whiten = whiten
537          self.fun = fun
538          self.fun_args = fun_args
539          self.max_iter = max_iter
540          self.tol = tol
541          self.w_init = w_init
542          self.whiten_solver = whiten_solver
543          self.random_state = random_state

545      def _fit_transform(self, X, compute_sources=False):
546          """Fit the model. 
547   
548          Parameters 
549          ---------- 
550          X : array-like of shape (n_samples, n_features) 
551              Training data, where `n_samples` is the number of samples 
552              and `n_features` is the number of features. 
553   
554          compute_sources : bool, default=False 
555              If False, sources are not computes but only the rotation matrix. 
556              This can save memory when working with big data. Defaults to False. 
557   
558          Returns 
559          ------- 
560          S : ndarray of shape (n_samples, n_components) or None 
561              Sources matrix. `None` if `compute_sources` is `False`. 
562          """
563          XT = validate_data(
564              self,
565              X,
566              copy=self.whiten,
567              dtype=[np.float64, np.float32],
568              ensure_min_samples=2,
569          ).T
570          fun_args = {} if self.fun_args is None else self.fun_args
571          random_state = check_random_state(self.random_state)
572  
573          alpha = fun_args.get("alpha", 1.0)
574          if not 1 <= alpha <= 2:
575              raise ValueError("alpha must be in [1,2]")
576  
577          if self.fun == "logcosh":
578              g = _logcosh
579          elif self.fun == "exp":
580              g = _exp
581          elif self.fun == "cube":
582              g = _cube
583          elif callable(self.fun):
584  
585              def g(x, fun_args):
586                  return self.fun(x, **fun_args)
587  
588          n_features, n_samples = XT.shape
589          n_components = self.n_components
590          if not self.whiten and n_components is not None:
591              n_components = None
592              warnings.warn("Ignoring n_components with whiten=False.")
593  
594          if n_components is None:
595              n_components = min(n_samples, n_features)
596          if n_components > min(n_samples, n_features):
597              n_components = min(n_samples, n_features)
598              warnings.warn(
599                  "n_components is too large: it will be set to %s" % n_components
600              )
601  
602          if self.whiten:
603              # Centering the features of X
604              X_mean = XT.mean(axis=-1)
605              XT -= X_mean[:, np.newaxis]
606  
607              # Whitening and preprocessing by PCA
608              if self.whiten_solver == "eigh":
609                  # Faster when num_samples >> n_features
610                  d, u = linalg.eigh(XT.dot(X))
611                  sort_indices = np.argsort(d)[::-1]
612                  eps = np.finfo(d.dtype).eps * 10
613                  degenerate_idx = d < eps
614                  if np.any(degenerate_idx):
615                      warnings.warn(
616                          "There are some small singular values, using "
617                          "whiten_solver = 'svd' might lead to more "
618                          "accurate results."
619                      )
620                  d[degenerate_idx] = eps  # For numerical issues
621                  np.sqrt(d, out=d)
622                  d, u = d[sort_indices], u[:, sort_indices]
623              elif self.whiten_solver == "svd":
624                  u, d = linalg.svd(XT, full_matrices=False, check_finite=False)[:2]
625  
626              # Give consistent eigenvectors for both svd solvers
627              u *= np.sign(u[0])
628  
629              K = (u / d).T[:n_components]  # see (6.33) p.140
630              del u, d
631              X1 = np.dot(K, XT)
632              # see (13.6) p.267 Here X1 is white and data
633              # in X has been projected onto a subspace by PCA
634              X1 *= np.sqrt(n_samples)
635          else:
636              # X must be casted to floats to avoid typing issues with numpy
637              # 2.0 and the line below
638              X1 = as_float_array(XT, copy=False)  # copy has been taken care of
639  
640          w_init = self.w_init
641          if w_init is None:
642              w_init = np.asarray(
643                  random_state.normal(size=(n_components, n_components)), dtype=X1.dtype
644              )
645  
646          else:
647              w_init = np.asarray(w_init)
648              if w_init.shape != (n_components, n_components):
649                  raise ValueError(
650                      "w_init has invalid shape -- should be %(shape)s"
651                      % {"shape": (n_components, n_components)}
652                  )
653  
654          kwargs = {
655              "tol": self.tol,
656              "g": g,
657              "fun_args": fun_args,
658              "max_iter": self.max_iter,
659              "w_init": w_init,
660          }
661  
662          if self.algorithm == "parallel":
663              W, n_iter = _ica_par(X1, **kwargs)
664          elif self.algorithm == "deflation":
665              W, n_iter = _ica_def(X1, **kwargs)
666          del X1
667  
668          self.n_iter_ = n_iter
669  
670          if compute_sources:
671              if self.whiten:
672                  S = np.linalg.multi_dot([W, K, XT]).T
673              else:
674                  S = np.dot(W, XT).T
675          else:
676              S = None
677  
678          if self.whiten:
679              if self.whiten == "unit-variance":
680                  if not compute_sources:
681                      S = np.linalg.multi_dot([W, K, XT]).T
682                  S_std = np.std(S, axis=0, keepdims=True)
683                  S /= S_std
684                  W /= S_std.T
685  
686              self.components_ = np.dot(W, K)
687              self.mean_ = X_mean
688              self.whitening_ = K
689          else:
690              self.components_ = W
691  
692          self.mixing_ = linalg.pinv(self.components_, check_finite=False)
693          self._unmixing = W
694  
695          return S
696  
697      @_fit_context(prefer_skip_nested_validation=True)
698      def fit_transform(self, X, y=None):
699          """Fit the model and recover the sources from X. 
700   
701          Parameters 
702          ---------- 
703          X : array-like of shape (n_samples, n_features) 
704              Training data, where `n_samples` is the number of samples 
705              and `n_features` is the number of features. 
706   
707          y : Ignored 
708              Not used, present for API consistency by convention. 
709   
710          Returns 
711          ------- 
712          X_new : ndarray of shape (n_samples, n_components) 
713              Estimated sources obtained by transforming the data with the 
714              estimated unmixing matrix. 
715          """
716          return self._fit_transform(X, compute_sources=True)
717  
718      @_fit_context(prefer_skip_nested_validation=True)
719      def fit(self, X, y=None):
720          """Fit the model to X. 
721   
722          Parameters 
723          ---------- 
724          X : array-like of shape (n_samples, n_features) 
725              Training data, where `n_samples` is the number of samples 
726              and `n_features` is the number of features. 
727   
728          y : Ignored 
729              Not used, present for API consistency by convention. 
730   
731          Returns 
732          ------- 
733          self : object 
734              Returns the instance itself. 
735          """
736          self._fit_transform(X, compute_sources=False)
737          return self

739      def transform(self, X, copy=True):
740          """Recover the sources from X (apply the unmixing matrix). 
741   
742          Parameters 
743          ---------- 
744          X : array-like of shape (n_samples, n_features) 
745              Data to transform, where `n_samples` is the number of samples 
746              and `n_features` is the number of features. 
747   
748          copy : bool, default=True 
749              If False, data passed to fit can be overwritten. Defaults to True. 
750   
751          Returns 
752          ------- 
753          X_new : ndarray of shape (n_samples, n_components) 
754              Estimated sources obtained by transforming the data with the 
755              estimated unmixing matrix. 
756          """
757          check_is_fitted(self)
758  
759          X = validate_data(
760              self,
761              X,
762              copy=(copy and self.whiten),
763              dtype=[np.float64, np.float32],
764              reset=False,
765          )
766          if self.whiten:
767              X -= self.mean_
768  
769          return np.dot(X, self.components_.T)
770  
771      def inverse_transform(self, X, copy=True):
772          """Transform the sources back to the mixed data (apply mixing matrix). 
773   
774          Parameters 
775          ---------- 
776          X : array-like of shape (n_samples, n_components) 
777              Sources, where `n_samples` is the number of samples 
778              and `n_components` is the number of components. 
779          copy : bool, default=True 
780              If False, data passed to fit are overwritten. Defaults to True. 
781   
782          Returns 
783          ------- 
784          X_new : ndarray of shape (n_samples, n_features) 
785              Reconstructed data obtained with the mixing matrix. 
786          """
787          check_is_fitted(self)
788  
789          X = check_array(X, copy=(copy and self.whiten), dtype=[np.float64, np.float32])
790          X = np.dot(X, self.mixing_.T)
791          if self.whiten:
792              X += self.mean_
793  
794          return X

798          """Number of transformed output features."""
799          return self.components_.shape[0]
