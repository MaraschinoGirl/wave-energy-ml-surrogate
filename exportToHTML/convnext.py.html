141  class StochasticDepth(Layer):
142      """Stochastic Depth module. 
143   
144      It performs batch-wise dropping rather than sample-wise. In libraries like 
145      `timm`, it's similar to `DropPath` layers that drops residual paths 
146      sample-wise. 
147   
148      References: 
149      - https://github.com/rwightman/pytorch-image-models 
150   
151      Args: 
152        drop_path_rate (float): Probability of dropping paths. Should be within 
153          [0, 1]. 
154   
155      Returns: 
156        Tensor either with the residual path dropped or kept. 
157      """
158  
159      def __init__(self, drop_path_rate, **kwargs):
160          super().__init__(**kwargs)
161          self.drop_path_rate = drop_path_rate
162  
163      def call(self, x, training=None):
164          if training:
165              keep_prob = 1 - self.drop_path_rate
166              shape = (ops.shape(x)[0],) + (1,) * (len(ops.shape(x)) - 1)
167              random_tensor = keep_prob + random.uniform(shape, 0, 1)
168              random_tensor = ops.floor(random_tensor)
169              return (x / keep_prob) * random_tensor
170          return x
171  
172      def get_config(self):
173          config = super().get_config()
174          config.update({"drop_path_rate": self.drop_path_rate})
175          return config

178  class LayerScale(Layer):
179      """Layer scale module. 
180   
181      References: 
182   
183      - https://arxiv.org/abs/2103.17239 
184   
185      Args: 
186          init_values (float): Initial value for layer scale. Should be within 
187              [0, 1]. 
188          projection_dim (int): Projection dimensionality. 
189   
190      Returns: 
191          Tensor multiplied to the scale. 
192      """
193  
194      def __init__(self, init_values, projection_dim, **kwargs):
195          super().__init__(**kwargs)
196          self.init_values = init_values
197          self.projection_dim = projection_dim
198  
199      def build(self, _):
200          self.gamma = self.add_weight(
201              shape=(self.projection_dim,),
202              initializer=initializers.Constant(self.init_values),
203              trainable=True,
204          )
205  
206      def call(self, x):
207          return x * self.gamma
208  
209      def get_config(self):
210          config = super().get_config()
211          config.update(
212              {
213                  "init_values": self.init_values,
214                  "projection_dim": self.projection_dim,
215              }
216          )
217          return config

220  def ConvNeXtBlock(
221      projection_dim, drop_path_rate=0.0, layer_scale_init_value=1e-6, name=None
222  ):
223      """ConvNeXt block. 
224   
225      References: 
226      - https://arxiv.org/abs/2201.03545 
227      - https://github.com/facebookresearch/ConvNeXt/blob/main/models/convnext.py 
228   
229      Notes: 
230          In the original ConvNeXt implementation (linked above), the authors use 
231          `Dense` layers for pointwise convolutions for increased efficiency. 
232          Following that, this implementation also uses the same. 
233   
234      Args: 
235          projection_dim (int): Number of filters for convolution layers. In the 
236              ConvNeXt paper, this is referred to as projection dimension. 
237          drop_path_rate (float): Probability of dropping paths. Should be within 
238              [0, 1]. 
239          layer_scale_init_value (float): Layer scale value. 
240              Should be a small float number. 
241          name: name to path to the keras layer. 
242   
243      Returns: 
244          A function representing a ConvNeXtBlock block. 
245      """
246      if name is None:
247          name = "prestem" + str(backend.get_uid("prestem"))
248  
249      def apply(inputs):
250          x = inputs
251  
252          x = layers.Conv2D(
253              filters=projection_dim,
254              kernel_size=7,
255              padding="same",
256              groups=projection_dim,
257              name=name + "_depthwise_conv",
258          )(x)
259          x = layers.LayerNormalization(epsilon=1e-6, name=name + "_layernorm")(x)
260          x = layers.Dense(4 * projection_dim, name=name + "_pointwise_conv_1")(x)
261          x = layers.Activation("gelu", name=name + "_gelu")(x)
262          x = layers.Dense(projection_dim, name=name + "_pointwise_conv_2")(x)
263  
264          if layer_scale_init_value is not None:
265              x = LayerScale(
266                  layer_scale_init_value,
267                  projection_dim,
268                  name=name + "_layer_scale",
269              )(x)
270          if drop_path_rate:
271              layer = StochasticDepth(
272                  drop_path_rate, name=name + "_stochastic_depth"
273              )
274          else:
275              layer = layers.Activation("linear", name=name + "_identity")
276  
277          return inputs + layer(x)
278  
279      return apply

282  def PreStem(name=None):
283      """Normalizes inputs with ImageNet-1k mean and std."""
284      if name is None:
285          name = "prestem" + str(backend.get_uid("prestem"))
286  
287      def apply(x):
288          x = layers.Normalization(
289              mean=[0.485 * 255, 0.456 * 255, 0.406 * 255],
290              variance=[
291                  (0.229 * 255) ** 2,
292                  (0.224 * 255) ** 2,
293                  (0.225 * 255) ** 2,
294              ],
295              name=name + "_prestem_normalization",
296          )(x)
297          return x
298  
299      return apply

302  def Head(num_classes=1000, classifier_activation=None, name=None):
303      """Implementation of classification head of ConvNeXt. 
304   
305      Args: 
306          num_classes: number of classes for Dense layer 
307          classifier_activation: activation function for the Dense layer 
308          name: name prefix 
309   
310      Returns: 
311          Classification head function. 
312      """
313      if name is None:
314          name = str(backend.get_uid("head"))
315  
316      def apply(x):
317          x = layers.GlobalAveragePooling2D(name=name + "_head_gap")(x)
318          x = layers.LayerNormalization(
319              epsilon=1e-6, name=name + "_head_layernorm"
320          )(x)
321          x = layers.Dense(
322              num_classes,
323              activation=classifier_activation,
324              name=name + "_head_dense",
325          )(x)
326          return x
327  
328      return apply

331  def ConvNeXt(
332      depths,
333      projection_dims,
334      drop_path_rate=0.0,
335      layer_scale_init_value=1e-6,
336      default_size=224,
337      name="convnext",
338      include_preprocessing=True,
339      include_top=True,
340      weights=None,
341      input_tensor=None,
342      input_shape=None,
343      pooling=None,
344      classes=1000,
345      classifier_activation="softmax",
346      weights_name=None,
347  ):
348      """Instantiates ConvNeXt architecture given specific configuration. 
349   
350      Args: 
351          depths: An iterable containing depths for each individual stages. 
352          projection_dims: An iterable containing output number of channels of 
353          each individual stages. 
354          drop_path_rate: Stochastic depth probability. If 0.0, then stochastic 
355              depth won't be used. 
356          layer_scale_init_value: Layer scale coefficient. If 0.0, layer scaling 
357              won't be used. 
358          default_size: Default input image size. 
359          name: An optional name for the model. 
360          include_preprocessing: boolean denoting whether to 
361              include preprocessing in the model. 
362              When `weights="imagenet"` this should always be `True`. 
363              But for other models (e.g., randomly initialized) you should set it 
364              to `False` and apply preprocessing to data accordingly. 
365          include_top: Boolean denoting whether to include classification 
366              head to the model. 
367          weights: one of `None` (random initialization), `"imagenet"` 
368              (pre-training on ImageNet-1k), 
369              or the path to the weights file to be loaded. 
370          input_tensor: optional Keras tensor (i.e. output of `layers.Input()`) to 
371              use as image input for the model. 
372          input_shape: optional shape tuple, only to be specified if `include_top` 
373              is `False`. It should have exactly 3 inputs channels. 
374          pooling: optional pooling mode for feature extraction when `include_top` 
375              is `False`. 
376              - `None` means that the output of the model will be 
377                  the 4D tensor output of the last convolutional layer. 
378              - `avg` means that global average pooling will be applied 
379                  to the output of the last convolutional layer, 
380                  and thus the output of the model will be a 2D tensor. 
381              - `max` means that global max pooling will be applied. 
382          classes: optional number of classes to classify images into, 
383              only to be specified if `include_top` is `True`, 
384              and if no `weights` argument is specified. 
385          classifier_activation: A `str` or callable. 
386              The activation function to use 
387              on the "top" layer. Ignored unless `include_top=True`. 
388              Set `classifier_activation=None` to return the logits 
389              of the "top" layer. 
390   
391      Returns: 
392          A model instance. 
393      """
394      if backend.image_data_format() == "channels_first":
395          raise ValueError(
396              "ConvNeXt does not support the `channels_first` image data "
397              "format. Switch to `channels_last` by editing your local "
398              "config file at ~/.keras/keras.json"
399          )
400      if not (weights in {"imagenet", None} or file_utils.exists(weights)):
401          raise ValueError(
402              "The `weights` argument should be either "
403              "`None` (random initialization), `imagenet` "
404              "(pre-training on ImageNet), "
405              "or the path to the weights file to be loaded."
406          )
407  
408      if weights == "imagenet" and include_top and classes != 1000:
409          raise ValueError(
410              'If using `weights="imagenet"` with `include_top=True`, '
411              "`classes` should be 1000. "
412              f"Received classes={classes}"
413          )
414  
415      # Determine proper input shape.
416      input_shape = imagenet_utils.obtain_input_shape(
417          input_shape,
418          default_size=default_size,
419          min_size=32,
420          data_format=backend.image_data_format(),
421          require_flatten=include_top,
422          weights=weights,
423      )
424  
425      if input_tensor is None:
426          img_input = layers.Input(shape=input_shape)
427      else:
428          if not backend.is_keras_tensor(input_tensor):
429              img_input = layers.Input(tensor=input_tensor, shape=input_shape)
430          else:
431              img_input = input_tensor
432  
433      if input_tensor is not None:
434          inputs = operation_utils.get_source_inputs(input_tensor)[0]
435          x = input_tensor
436      else:
437          inputs = img_input
438          x = inputs
439  
440      if include_preprocessing:
441          channel_axis = (
442              3 if backend.image_data_format() == "channels_last" else 1
443          )
444          num_channels = input_shape[channel_axis - 1]
445          if num_channels == 3:
446              x = PreStem(name=name)(x)
447  
448      # Stem block.
449      stem = Sequential(
450          [
451              layers.Conv2D(
452                  projection_dims[0],
453                  kernel_size=4,
454                  strides=4,
455                  name=name + "_stem_conv",
456              ),
457              layers.LayerNormalization(
458                  epsilon=1e-6, name=name + "_stem_layernorm"
459              ),
460          ],
461          name=name + "_stem",
462      )
463  
464      # Downsampling blocks.
465      downsample_layers = []
466      downsample_layers.append(stem)
467  
468      num_downsample_layers = 3
469      for i in range(num_downsample_layers):
470          downsample_layer = Sequential(
471              [
472                  layers.LayerNormalization(
473                      epsilon=1e-6,
474                      name=name + "_downsampling_layernorm_" + str(i),
475                  ),
476                  layers.Conv2D(
477                      projection_dims[i + 1],
478                      kernel_size=2,
479                      strides=2,
480                      name=name + "_downsampling_conv_" + str(i),
481                  ),
482              ],
483              name=name + "_downsampling_block_" + str(i),
484          )
485          downsample_layers.append(downsample_layer)
486  
487      # Stochastic depth schedule.
488      # This is referred from the original ConvNeXt codebase:
489      # https://github.com/facebookresearch/ConvNeXt/blob/main/models/convnext.py#L86
490      depth_drop_rates = [
491          float(x) for x in np.linspace(0.0, drop_path_rate, sum(depths))
492      ]
493  
494      # First apply downsampling blocks and then apply ConvNeXt stages.
495      cur = 0
496  
497      num_convnext_blocks = 4
498      for i in range(num_convnext_blocks):
499          x = downsample_layers[i](x)
500          for j in range(depths[i]):
501              x = ConvNeXtBlock(
502                  projection_dim=projection_dims[i],
503                  drop_path_rate=depth_drop_rates[cur + j],
504                  layer_scale_init_value=layer_scale_init_value,
505                  name=name + f"_stage_{i}_block_{j}",
506              )(x)
507          cur += depths[i]
508  
509      if include_top:
510          imagenet_utils.validate_activation(classifier_activation, weights)
511          x = Head(
512              num_classes=classes,
513              classifier_activation=classifier_activation,
514              name=name,
515          )(x)
516  
517      else:
518          if pooling == "avg":
519              x = layers.GlobalAveragePooling2D()(x)
520          elif pooling == "max":
521              x = layers.GlobalMaxPooling2D()(x)
522          x = layers.LayerNormalization(epsilon=1e-6)(x)
523  
524      model = Functional(inputs=inputs, outputs=x, name=name)
525  
526      # Validate weights before requesting them from the API
527      if weights == "imagenet":
528          expected_config = MODEL_CONFIGS[weights_name.split("convnext_")[-1]]
529          if (
530              depths != expected_config["depths"]
531              or projection_dims != expected_config["projection_dims"]
532          ):
533              raise ValueError(
534                  f"Architecture configuration does not match {weights_name} "
535                  f"variant. When using pre-trained weights, the model "
536                  f"architecture must match the pre-trained configuration "
537                  f"exactly. Expected depths: {expected_config['depths']}, "
538                  f"got: {depths}. Expected projection_dims: "
539                  f"{expected_config['projection_dims']}, got: {projection_dims}."
540              )
541  
542          if weights_name not in name:
543              raise ValueError(
544                  f'Model name "{name}" does not match weights variant '
545                  f'"{weights_name}". When using imagenet weights, model name '
546                  f'must contain the weights variant (e.g., "convnext_'
547                  f'{weights_name.split("convnext_")[-1]}").'
548              )
549  
550      # Load weights.
551      if weights == "imagenet":
552          if include_top:
553              file_suffix = ".h5"
554              file_hash = WEIGHTS_HASHES[weights_name][0]
555          else:
556              file_suffix = "_notop.h5"
557              file_hash = WEIGHTS_HASHES[weights_name][1]
558          file_name = name + file_suffix
559          weights_path = file_utils.get_file(
560              file_name,
561              BASE_WEIGHTS_PATH + file_name,
562              cache_subdir="models",
563              file_hash=file_hash,
564          )
565          model.load_weights(weights_path)
566      elif weights is not None:
567          model.load_weights(weights)
568  
569      return model
