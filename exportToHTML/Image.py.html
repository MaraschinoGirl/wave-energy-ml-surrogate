31   def _make_edges_3d(n_x, n_y, n_z=1):
32       """Returns a list of edges for a 3D image. 
33    
34       Parameters 
35       ---------- 
36       n_x : int 
37           The size of the grid in the x direction. 
38       n_y : int 
39           The size of the grid in the y direction. 
40       n_z : integer, default=1 
41           The size of the grid in the z direction, defaults to 1 
42       """
43       vertices = np.arange(n_x * n_y * n_z).reshape((n_x, n_y, n_z))
44       edges_deep = np.vstack((vertices[:, :, :-1].ravel(), vertices[:, :, 1:].ravel()))
45       edges_right = np.vstack((vertices[:, :-1].ravel(), vertices[:, 1:].ravel()))
46       edges_down = np.vstack((vertices[:-1].ravel(), vertices[1:].ravel()))
47       edges = np.hstack((edges_deep, edges_right, edges_down))
48       return edges

51   def _compute_gradient_3d(edges, img):
52       _, n_y, n_z = img.shape
53       gradient = np.abs(
54           img[
55               edges[0] // (n_y * n_z),
56               (edges[0] % (n_y * n_z)) // n_z,
57               (edges[0] % (n_y * n_z)) % n_z,
58           ]
59           - img[
60               edges[1] // (n_y * n_z),
61               (edges[1] % (n_y * n_z)) // n_z,
62               (edges[1] % (n_y * n_z)) % n_z,
63           ]
64       )
65       return gradient

71   def _mask_edges_weights(mask, edges, weights=None):
72       """Apply a mask to edges (weighted or not)"""
73       inds = np.arange(mask.size)
74       inds = inds[mask.ravel()]
75       ind_mask = np.logical_and(np.isin(edges[0], inds), np.isin(edges[1], inds))
76       edges = edges[:, ind_mask]
77       if weights is not None:
78           weights = weights[ind_mask]
79       if len(edges.ravel()):
80           maxval = edges.max()
81       else:
82           maxval = 0
83       order = np.searchsorted(np.flatnonzero(mask), np.arange(maxval + 1))
84       edges = order[edges]
85       if weights is None:
86           return edges
87       else:
88           return edges, weights

91   def _to_graph(
92       n_x, n_y, n_z, mask=None, img=None, return_as=sparse.coo_matrix, dtype=None
93   ):
94       """Auxiliary function for img_to_graph and grid_to_graph"""
95       edges = _make_edges_3d(n_x, n_y, n_z)
96   
97       if dtype is None:  # To not overwrite input dtype
98           if img is None:
99               dtype = int
100          else:
101              dtype = img.dtype
102  
103      if img is not None:
104          img = np.atleast_3d(img)
105          weights = _compute_gradient_3d(edges, img)
106          if mask is not None:
107              edges, weights = _mask_edges_weights(mask, edges, weights)
108              diag = img.squeeze()[mask]
109          else:
110              diag = img.ravel()
111          n_voxels = diag.size
112      else:
113          if mask is not None:
114              mask = mask.astype(dtype=bool, copy=False)
115              edges = _mask_edges_weights(mask, edges)
116              n_voxels = np.sum(mask)
117          else:
118              n_voxels = n_x * n_y * n_z
119          weights = np.ones(edges.shape[1], dtype=dtype)
120          diag = np.ones(n_voxels, dtype=dtype)
121  
122      diag_idx = np.arange(n_voxels)
123      i_idx = np.hstack((edges[0], edges[1]))
124      j_idx = np.hstack((edges[1], edges[0]))
125      graph = sparse.coo_matrix(
126          (
127              np.hstack((weights, weights, diag)),
128              (np.hstack((i_idx, diag_idx)), np.hstack((j_idx, diag_idx))),
129          ),
130          (n_voxels, n_voxels),
131          dtype=dtype,
132      )
133      if return_as is np.ndarray:
134          return graph.toarray()
135      return return_as(graph)

147  def img_to_graph(img, *, mask=None, return_as=sparse.coo_matrix, dtype=None):
148      """Graph of the pixel-to-pixel gradient connections. 
149   
150      Edges are weighted with the gradient values. 
151   
152      Read more in the :ref:`User Guide <image_feature_extraction>`. 
153   
154      Parameters 
155      ---------- 
156      img : array-like of shape (height, width) or (height, width, channel) 
157          2D or 3D image. 
158      mask : ndarray of shape (height, width) or \ 
159              (height, width, channel), dtype=bool, default=None 
160          An optional mask of the image, to consider only part of the 
161          pixels. 
162      return_as : np.ndarray or a sparse matrix class, \ 
163              default=sparse.coo_matrix 
164          The class to use to build the returned adjacency matrix. 
165      dtype : dtype, default=None 
166          The data of the returned sparse matrix. By default it is the 
167          dtype of img. 
168   
169      Returns 
170      ------- 
171      graph : ndarray or a sparse matrix class 
172          The computed adjacency matrix. 
173   
174      Examples 
175      -------- 
176      >>> import numpy as np 
177      >>> from sklearn.feature_extraction.image import img_to_graph 
178      >>> img = np.array([[0, 0], [0, 1]]) 
179      >>> img_to_graph(img, return_as=np.ndarray) 
180      array([[0, 0, 0, 0], 
181             [0, 0, 0, 1], 
182             [0, 0, 0, 1], 
183             [0, 1, 1, 1]]) 
184      """
185      img = np.atleast_3d(img)
186      n_x, n_y, n_z = img.shape
187      return _to_graph(n_x, n_y, n_z, mask, img, return_as, dtype)

202      n_x, n_y, n_z=1, *, mask=None, return_as=sparse.coo_matrix, dtype=int
203  ):
204      """Graph of the pixel-to-pixel connections. 
205   
206      Edges exist if 2 voxels are connected. 
207   
208      Parameters 
209      ---------- 
210      n_x : int 
211          Dimension in x axis. 
212      n_y : int 
213          Dimension in y axis. 
214      n_z : int, default=1 
215          Dimension in z axis. 
216      mask : ndarray of shape (n_x, n_y, n_z), dtype=bool, default=None 
217          An optional mask of the image, to consider only part of the 
218          pixels. 
219      return_as : np.ndarray or a sparse matrix class, \ 
220              default=sparse.coo_matrix 
221          The class to use to build the returned adjacency matrix. 
222      dtype : dtype, default=int 
223          The data of the returned sparse matrix. By default it is int. 
224   
225      Returns 
226      ------- 
227      graph : np.ndarray or a sparse matrix class 
228          The computed adjacency matrix. 
229   
230      Examples 
231      -------- 
232      >>> import numpy as np 
233      >>> from sklearn.feature_extraction.image import grid_to_graph 
234      >>> shape_img = (4, 4, 1) 
235      >>> mask = np.zeros(shape=shape_img, dtype=bool) 
236      >>> mask[[1, 2], [1, 2], :] = True 
237      >>> graph = grid_to_graph(*shape_img, mask=mask) 
238      >>> print(graph) 
239      <COOrdinate sparse matrix of dtype 'int64' 
240        with 2 stored elements and shape (2, 2)> 
241        Coords    Values 
242        (0, 0)    1 
243        (1, 1)    1 
244      """
245      return _to_graph(n_x, n_y, n_z, mask=mask, return_as=return_as, dtype=dtype)

252  def _compute_n_patches(i_h, i_w, p_h, p_w, max_patches=None):
253      """Compute the number of patches that will be extracted in an image. 
254   
255      Read more in the :ref:`User Guide <image_feature_extraction>`. 
256   
257      Parameters 
258      ---------- 
259      i_h : int 
260          The image height 
261      i_w : int 
262          The image with 
263      p_h : int 
264          The height of a patch 
265      p_w : int 
266          The width of a patch 
267      max_patches : int or float, default=None 
268          The maximum number of patches to extract. If `max_patches` is a float 
269          between 0 and 1, it is taken to be a proportion of the total number 
270          of patches. If `max_patches` is None, all possible patches are extracted. 
271      """
272      n_h = i_h - p_h + 1
273      n_w = i_w - p_w + 1
274      all_patches = n_h * n_w
275  
276      if max_patches:
277          if isinstance(max_patches, (Integral)) and max_patches < all_patches:
278              return max_patches
279          elif isinstance(max_patches, (Integral)) and max_patches >= all_patches:
280              return all_patches
281          elif isinstance(max_patches, (Real)) and 0 < max_patches < 1:
282              return int(max_patches * all_patches)
283          else:
284              raise ValueError("Invalid value for max_patches: %r" % max_patches)
285      else:
286          return all_patches

289  def _extract_patches(arr, patch_shape=8, extraction_step=1):
290      """Extracts patches of any n-dimensional array in place using strides. 
291   
292      Given an n-dimensional array it will return a 2n-dimensional array with 
293      the first n dimensions indexing patch position and the last n indexing 
294      the patch content. This operation is immediate (O(1)). A reshape 
295      performed on the first n dimensions will cause numpy to copy data, leading 
296      to a list of extracted patches. 
297   
298      Read more in the :ref:`User Guide <image_feature_extraction>`. 
299   
300      Parameters 
301      ---------- 
302      arr : ndarray 
303          n-dimensional array of which patches are to be extracted 
304   
305      patch_shape : int or tuple of length arr.ndim.default=8 
306          Indicates the shape of the patches to be extracted. If an 
307          integer is given, the shape will be a hypercube of 
308          sidelength given by its value. 
309   
310      extraction_step : int or tuple of length arr.ndim, default=1 
311          Indicates step size at which extraction shall be performed. 
312          If integer is given, then the step is uniform in all dimensions. 
313   
314   
315      Returns 
316      ------- 
317      patches : strided ndarray 
318          2n-dimensional array indexing patches on first n dimensions and 
319          containing patches on the last n dimensions. These dimensions 
320          are fake, but this way no data is copied. A simple reshape invokes 
321          a copying operation to obtain a list of patches: 
322          result.reshape([-1] + list(patch_shape)) 
323      """
324  
325      arr_ndim = arr.ndim
326  
327      if isinstance(patch_shape, Number):
328          patch_shape = tuple([patch_shape] * arr_ndim)
329      if isinstance(extraction_step, Number):
330          extraction_step = tuple([extraction_step] * arr_ndim)
331  
332      patch_strides = arr.strides
333  
334      slices = tuple(slice(None, None, st) for st in extraction_step)
335      indexing_strides = arr[slices].strides
336  
337      patch_indices_shape = (
338          (np.array(arr.shape) - np.array(patch_shape)) // np.array(extraction_step)
339      ) + 1
340  
341      shape = tuple(list(patch_indices_shape) + list(patch_shape))
342      strides = tuple(list(indexing_strides) + list(patch_strides))
343  
344      patches = as_strided(arr, shape=shape, strides=strides)
345      return patches

361  def extract_patches_2d(image, patch_size, *, max_patches=None, random_state=None):
362      """Reshape a 2D image into a collection of patches. 
363   
364      The resulting patches are allocated in a dedicated array. 
365   
366      Read more in the :ref:`User Guide <image_feature_extraction>`. 
367   
368      Parameters 
369      ---------- 
370      image : ndarray of shape (image_height, image_width) or \ 
371          (image_height, image_width, n_channels) 
372          The original image data. For color images, the last dimension specifies 
373          the channel: a RGB image would have `n_channels=3`. 
374   
375      patch_size : tuple of int (patch_height, patch_width) 
376          The dimensions of one patch. 
377   
378      max_patches : int or float, default=None 
379          The maximum number of patches to extract. If `max_patches` is a float 
380          between 0 and 1, it is taken to be a proportion of the total number 
381          of patches. If `max_patches` is None it corresponds to the total number 
382          of patches that can be extracted. 
383   
384      random_state : int, RandomState instance, default=None 
385          Determines the random number generator used for random sampling when 
386          `max_patches` is not None. Use an int to make the randomness 
387          deterministic. 
388          See :term:`Glossary <random_state>`. 
389   
390      Returns 
391      ------- 
392      patches : array of shape (n_patches, patch_height, patch_width) or \ 
393          (n_patches, patch_height, patch_width, n_channels) 
394          The collection of patches extracted from the image, where `n_patches` 
395          is either `max_patches` or the total number of patches that can be 
396          extracted. 
397   
398      Examples 
399      -------- 
400      >>> from sklearn.datasets import load_sample_image 
401      >>> from sklearn.feature_extraction import image 
402      >>> # Use the array data from the first image in this dataset: 
403      >>> one_image = load_sample_image("china.jpg") 
404      >>> print('Image shape: {}'.format(one_image.shape)) 
405      Image shape: (427, 640, 3) 
406      >>> patches = image.extract_patches_2d(one_image, (2, 2)) 
407      >>> print('Patches shape: {}'.format(patches.shape)) 
408      Patches shape: (272214, 2, 2, 3) 
409      >>> # Here are just two of these patches: 
410      >>> print(patches[1]) 
411      [[[174 201 231] 
412        [174 201 231]] 
413       [[173 200 230] 
414        [173 200 230]]] 
415      >>> print(patches[800]) 
416      [[[187 214 243] 
417        [188 215 244]] 
418       [[187 214 243] 
419        [188 215 244]]] 
420      """
421      i_h, i_w = image.shape[:2]
422      p_h, p_w = patch_size
423  
424      if p_h > i_h:
425          raise ValueError(
426              "Height of the patch should be less than the height of the image."
427          )
428  
429      if p_w > i_w:
430          raise ValueError(
431              "Width of the patch should be less than the width of the image."
432          )
433  
434      image = check_array(image, allow_nd=True)
435      image = image.reshape((i_h, i_w, -1))
436      n_colors = image.shape[-1]
437  
438      extracted_patches = _extract_patches(
439          image, patch_shape=(p_h, p_w, n_colors), extraction_step=1
440      )
441  
442      n_patches = _compute_n_patches(i_h, i_w, p_h, p_w, max_patches)
443      if max_patches:
444          rng = check_random_state(random_state)
445          i_s = rng.randint(i_h - p_h + 1, size=n_patches)
446          j_s = rng.randint(i_w - p_w + 1, size=n_patches)
447          patches = extracted_patches[i_s, j_s, 0]
448      else:
449          patches = extracted_patches
450  
451      patches = patches.reshape(-1, p_h, p_w, n_colors)
452      # remove the color dimension if useless
453      if patches.shape[-1] == 1:
454          return patches.reshape((n_patches, p_h, p_w))
455      else:
456          return patches

459  @validate_params(

462  )
463  def reconstruct_from_patches_2d(patches, image_size):
464      """Reconstruct the image from all of its patches. 
465   
466      Patches are assumed to overlap and the image is constructed by filling in 
467      the patches from left to right, top to bottom, averaging the overlapping 
468      regions. 
469   
470      Read more in the :ref:`User Guide <image_feature_extraction>`. 
471   
472      Parameters 
473      ---------- 
474      patches : ndarray of shape (n_patches, patch_height, patch_width) or \ 
475          (n_patches, patch_height, patch_width, n_channels) 
476          The complete set of patches. If the patches contain colour information, 
477          channels are indexed along the last dimension: RGB patches would 
478          have `n_channels=3`. 
479   
480      image_size : tuple of int (image_height, image_width) or \ 
481          (image_height, image_width, n_channels) 
482          The size of the image that will be reconstructed. 
483   
484      Returns 
485      ------- 
486      image : ndarray of shape image_size 
487          The reconstructed image. 
488   
489      Examples 
490      -------- 
491      >>> from sklearn.datasets import load_sample_image 
492      >>> from sklearn.feature_extraction import image 
493      >>> one_image = load_sample_image("china.jpg") 
494      >>> print('Image shape: {}'.format(one_image.shape)) 
495      Image shape: (427, 640, 3) 
496      >>> image_patches = image.extract_patches_2d(image=one_image, patch_size=(10, 10)) 
497      >>> print('Patches shape: {}'.format(image_patches.shape)) 
498      Patches shape: (263758, 10, 10, 3) 
499      >>> image_reconstructed = image.reconstruct_from_patches_2d( 
500      ...     patches=image_patches, 
501      ...     image_size=one_image.shape 
502      ... ) 
503      >>> print(f"Reconstructed shape: {image_reconstructed.shape}") 
504      Reconstructed shape: (427, 640, 3) 
505      """
506      i_h, i_w = image_size[:2]
507      p_h, p_w = patches.shape[1:3]
508      img = np.zeros(image_size)
509      # compute the dimensions of the patches array
510      n_h = i_h - p_h + 1
511      n_w = i_w - p_w + 1
512      for p, (i, j) in zip(patches, product(range(n_h), range(n_w))):
513          img[i : i + p_h, j : j + p_w] += p
514  
515      for i in range(i_h):
516          for j in range(i_w):
517              # divide by the amount of overlap
518              # XXX: is this the most efficient way? memory-wise yes, cpu wise?
519              img[i, j] /= float(min(i + 1, p_h, i_h - i) * min(j + 1, p_w, i_w - j))
520      return img

523  class PatchExtractor(TransformerMixin, BaseEstimator):
524      """Extracts patches from a collection of images. 
525   
526      Read more in the :ref:`User Guide <image_feature_extraction>`. 
527   
528      .. versionadded:: 0.9 
529   
530      Parameters 
531      ---------- 
532      patch_size : tuple of int (patch_height, patch_width), default=None 
533          The dimensions of one patch. If set to None, the patch size will be 
534          automatically set to `(img_height // 10, img_width // 10)`, where 
535          `img_height` and `img_width` are the dimensions of the input images. 
536   
537      max_patches : int or float, default=None 
538          The maximum number of patches per image to extract. If `max_patches` is 
539          a float in (0, 1), it is taken to mean a proportion of the total number 
540          of patches. If set to None, extract all possible patches. 
541   
542      random_state : int, RandomState instance, default=None 
543          Determines the random number generator used for random sampling when 
544          `max_patches is not None`. Use an int to make the randomness 
545          deterministic. 
546          See :term:`Glossary <random_state>`. 
547   
548      See Also 
549      -------- 
550      reconstruct_from_patches_2d : Reconstruct image from all of its patches. 
551   
552      Notes 
553      ----- 
554      This estimator is stateless and does not need to be fitted. However, we 
555      recommend to call :meth:`fit_transform` instead of :meth:`transform`, as 
556      parameter validation is only performed in :meth:`fit`. 
557   
558      Examples 
559      -------- 
560      >>> from sklearn.datasets import load_sample_images 
561      >>> from sklearn.feature_extraction import image 
562      >>> # Use the array data from the second image in this dataset: 
563      >>> X = load_sample_images().images[1] 
564      >>> X = X[None, ...] 
565      >>> print(f"Image shape: {X.shape}") 
566      Image shape: (1, 427, 640, 3) 
567      >>> pe = image.PatchExtractor(patch_size=(10, 10)) 
568      >>> pe_trans = pe.transform(X) 
569      >>> print(f"Patches shape: {pe_trans.shape}") 
570      Patches shape: (263758, 10, 10, 3) 
571      >>> X_reconstructed = image.reconstruct_from_patches_2d(pe_trans, X.shape[1:]) 
572      >>> print(f"Reconstructed shape: {X_reconstructed.shape}") 
573      Reconstructed shape: (427, 640, 3) 
574      """
575  
576      _parameter_constraints: dict = {
577          "patch_size": [tuple, None],
578          "max_patches": [
579              None,
580              Interval(RealNotInt, 0, 1, closed="neither"),
581              Interval(Integral, 1, None, closed="left"),
582          ],
583          "random_state": ["random_state"],
584      }
585  
586      def __init__(self, *, patch_size=None, max_patches=None, random_state=None):
587          self.patch_size = patch_size
588          self.max_patches = max_patches
589          self.random_state = random_state
590  
591      @_fit_context(prefer_skip_nested_validation=True)
592      def fit(self, X, y=None):
593          """Only validate the parameters of the estimator. 
594   
595          This method allows to: (i) validate the parameters of the estimator  and 
596          (ii) be consistent with the scikit-learn transformer API. 
597   
598          Parameters 
599          ---------- 
600          X : ndarray of shape (n_samples, image_height, image_width) or \ 
601                  (n_samples, image_height, image_width, n_channels) 
602              Array of images from which to extract patches. For color images, 
603              the last dimension specifies the channel: a RGB image would have 
604              `n_channels=3`. 
605   
606          y : Ignored 
607              Not used, present for API consistency by convention. 
608   
609          Returns 
610          ------- 
611          self : object 
612              Returns the instance itself. 
613          """
614          return self
615  
616      def transform(self, X):
617          """Transform the image samples in `X` into a matrix of patch data. 
618   
619          Parameters 
620          ---------- 
621          X : ndarray of shape (n_samples, image_height, image_width) or \ 
622                  (n_samples, image_height, image_width, n_channels) 
623              Array of images from which to extract patches. For color images, 
624              the last dimension specifies the channel: a RGB image would have 
625              `n_channels=3`. 
626   
627          Returns 
628          ------- 
629          patches : array of shape (n_patches, patch_height, patch_width) or \ 
630                  (n_patches, patch_height, patch_width, n_channels) 
631              The collection of patches extracted from the images, where 
632              `n_patches` is either `n_samples * max_patches` or the total 
633              number of patches that can be extracted. 
634          """
635          X = validate_data(
636              self,
637              X=X,
638              ensure_2d=False,
639              allow_nd=True,
640              ensure_min_samples=1,
641              ensure_min_features=1,
642              reset=False,
643          )
644          random_state = check_random_state(self.random_state)
645          n_imgs, img_height, img_width = X.shape[:3]
646          if self.patch_size is None:
647              patch_size = img_height // 10, img_width // 10
648          else:
649              if len(self.patch_size) != 2:
650                  raise ValueError(
651                      "patch_size must be a tuple of two integers. Got"
652                      f" {self.patch_size} instead."
653                  )
654              patch_size = self.patch_size
655  
656          n_imgs, img_height, img_width = X.shape[:3]
657          X = np.reshape(X, (n_imgs, img_height, img_width, -1))
658          n_channels = X.shape[-1]
659  
660          # compute the dimensions of the patches array
661          patch_height, patch_width = patch_size
662          n_patches = _compute_n_patches(
663              img_height, img_width, patch_height, patch_width, self.max_patches
664          )
665          patches_shape = (n_imgs * n_patches,) + patch_size
666          if n_channels > 1:
667              patches_shape += (n_channels,)
668  
669          # extract the patches
670          patches = np.empty(patches_shape)
671          for ii, image in enumerate(X):
672              patches[ii * n_patches : (ii + 1) * n_patches] = extract_patches_2d(
673                  image,
674                  patch_size,
675                  max_patches=self.max_patches,
676                  random_state=random_state,
677              )
678          return patches

681          tags = super().__sklearn_tags__()
682          tags.input_tags.two_d_array = False
683          tags.input_tags.three_d_array = True
684          tags.requires_fit = False
685          return tags
