30   def _calculate_threshold(estimator, importances, threshold):
31       """Interpret the threshold value"""
32   
33       if threshold is None:
34           # determine default from estimator
35           est_name = estimator.__class__.__name__
36           is_l1_penalized = hasattr(estimator, "penalty") and estimator.penalty == "l1"
37           is_lasso = "Lasso" in est_name
38           is_elasticnet_l1_penalized = "ElasticNet" in est_name and (
39               (hasattr(estimator, "l1_ratio_") and np.isclose(estimator.l1_ratio_, 1.0))
40               or (hasattr(estimator, "l1_ratio") and np.isclose(estimator.l1_ratio, 1.0))
41           )
42           if is_l1_penalized or is_lasso or is_elasticnet_l1_penalized:
43               # the natural default threshold is 0 when l1 penalty was used
44               threshold = 1e-5
45           else:
46               threshold = "mean"
47   
48       if isinstance(threshold, str):
49           if "*" in threshold:
50               scale, reference = threshold.split("*")
51               scale = float(scale.strip())
52               reference = reference.strip()
53   
54               if reference == "median":
55                   reference = np.median(importances)
56               elif reference == "mean":
57                   reference = np.mean(importances)
58               else:
59                   raise ValueError("Unknown reference: " + reference)
60   
61               threshold = scale * reference
62   
63           elif threshold == "median":
64               threshold = np.median(importances)
65   
66           elif threshold == "mean":
67               threshold = np.mean(importances)
68   
69           else:
70               raise ValueError(
71                   "Expected threshold='mean' or threshold='median' got %s" % threshold
72               )
73   
74       else:
75           threshold = float(threshold)
76   
77       return threshold

80   class SelectFromModel(MetaEstimatorMixin, SelectorMixin, BaseEstimator):
81       """Meta-transformer for selecting features based on importance weights. 
82    
83       .. versionadded:: 0.17 
84    
85       Read more in the :ref:`User Guide <select_from_model>`. 
86    
87       Parameters 
88       ---------- 
89       estimator : object 
90           The base estimator from which the transformer is built. 
91           This can be both a fitted (if ``prefit`` is set to True) 
92           or a non-fitted estimator. The estimator should have a 
93           ``feature_importances_`` or ``coef_`` attribute after fitting. 
94           Otherwise, the ``importance_getter`` parameter should be used. 
95    
96       threshold : str or float, default=None 
97           The threshold value to use for feature selection. Features whose 
98           absolute importance value is greater or equal are kept while the others 
99           are discarded. If "median" (resp. "mean"), then the ``threshold`` value 
100          is the median (resp. the mean) of the feature importances. A scaling 
101          factor (e.g., "1.25*mean") may also be used. If None and if the 
102          estimator has a parameter penalty set to l1, either explicitly 
103          or implicitly (e.g, Lasso), the threshold used is 1e-5. 
104          Otherwise, "mean" is used by default. 
105   
106      prefit : bool, default=False 
107          Whether a prefit model is expected to be passed into the constructor 
108          directly or not. 
109          If `True`, `estimator` must be a fitted estimator. 
110          If `False`, `estimator` is fitted and updated by calling 
111          `fit` and `partial_fit`, respectively. 
112   
113      norm_order : non-zero int, inf, -inf, default=1 
114          Order of the norm used to filter the vectors of coefficients below 
115          ``threshold`` in the case where the ``coef_`` attribute of the 
116          estimator is of dimension 2. 
117   
118      max_features : int, callable, default=None 
119          The maximum number of features to select. 
120   
121          - If an integer, then it specifies the maximum number of features to 
122            allow. 
123          - If a callable, then it specifies how to calculate the maximum number of 
124            features allowed by using the output of `max_features(X)`. 
125          - If `None`, then all features are kept. 
126   
127          To only select based on ``max_features``, set ``threshold=-np.inf``. 
128   
129          .. versionadded:: 0.20 
130          .. versionchanged:: 1.1 
131             `max_features` accepts a callable. 
132   
133      importance_getter : str or callable, default='auto' 
134          If 'auto', uses the feature importance either through a ``coef_`` 
135          attribute or ``feature_importances_`` attribute of estimator. 
136   
137          Also accepts a string that specifies an attribute name/path 
138          for extracting feature importance (implemented with `attrgetter`). 
139          For example, give `regressor_.coef_` in case of 
140          :class:`~sklearn.compose.TransformedTargetRegressor`  or 
141          `named_steps.clf.feature_importances_` in case of 
142          :class:`~sklearn.pipeline.Pipeline` with its last step named `clf`. 
143   
144          If `callable`, overrides the default feature importance getter. 
145          The callable is passed with the fitted estimator and it should 
146          return importance for each feature. 
147   
148          .. versionadded:: 0.24 
149   
150      Attributes 
151      ---------- 
152      estimator_ : estimator 
153          The base estimator from which the transformer is built. This attribute 
154          exist only when `fit` has been called. 
155   
156          - If `prefit=True`, it is a deep copy of `estimator`. 
157          - If `prefit=False`, it is a clone of `estimator` and fit on the data 
158            passed to `fit` or `partial_fit`. 
159   
160      n_features_in_ : int 
161          Number of features seen during :term:`fit`. Only defined if the 
162          underlying estimator exposes such an attribute when fit. 
163   
164          .. versionadded:: 0.24 
165   
166      max_features_ : int 
167          Maximum number of features calculated during :term:`fit`. Only defined 
168          if the ``max_features`` is not `None`. 
169   
170          - If `max_features` is an `int`, then `max_features_ = max_features`. 
171          - If `max_features` is a callable, then `max_features_ = max_features(X)`. 
172   
173          .. versionadded:: 1.1 
174   
175      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
176          Names of features seen during :term:`fit`. Defined only when `X` 
177          has feature names that are all strings. 
178   
179          .. versionadded:: 1.0 
180   
181      threshold_ : float 
182          The threshold value used for feature selection. 
183   
184      See Also 
185      -------- 
186      RFE : Recursive feature elimination based on importance weights. 
187      RFECV : Recursive feature elimination with built-in cross-validated 
188          selection of the best number of features. 
189      SequentialFeatureSelector : Sequential cross-validation based feature 
190          selection. Does not rely on importance weights. 
191   
192      Notes 
193      ----- 
194      Allows NaN/Inf in the input if the underlying estimator does as well. 
195   
196      Examples 
197      -------- 
198      >>> from sklearn.feature_selection import SelectFromModel 
199      >>> from sklearn.linear_model import LogisticRegression 
200      >>> X = [[ 0.87, -1.34,  0.31 ], 
201      ...      [-2.79, -0.02, -0.85 ], 
202      ...      [-1.34, -0.48, -2.55 ], 
203      ...      [ 1.92,  1.48,  0.65 ]] 
204      >>> y = [0, 1, 0, 1] 
205      >>> selector = SelectFromModel(estimator=LogisticRegression()).fit(X, y) 
206      >>> selector.estimator_.coef_ 
207      array([[-0.3252...,  0.8345...,  0.4976...]]) 
208      >>> selector.threshold_ 
209      np.float64(0.55249...) 
210      >>> selector.get_support() 
211      array([False,  True, False]) 
212      >>> selector.transform(X) 
213      array([[-1.34], 
214             [-0.02], 
215             [-0.48], 
216             [ 1.48]]) 
217   
218      Using a callable to create a selector that can use no more than half 
219      of the input features. 
220   
221      >>> def half_callable(X): 
222      ...     return round(len(X[0]) / 2) 
223      >>> half_selector = SelectFromModel(estimator=LogisticRegression(), 
224      ...                                 max_features=half_callable) 
225      >>> _ = half_selector.fit(X, y) 
226      >>> half_selector.max_features_ 
227      2 
228      """

230      _parameter_constraints: dict = {
231          "estimator": [HasMethods("fit")],
232          "threshold": [Interval(Real, None, None, closed="both"), str, None],
233          "prefit": ["boolean"],
234          "norm_order": [
235              Interval(Integral, None, -1, closed="right"),
236              Interval(Integral, 1, None, closed="left"),
237              Options(Real, {np.inf, -np.inf}),
238          ],
239          "max_features": [Interval(Integral, 0, None, closed="left"), callable, None],
240          "importance_getter": [str, callable],
241      }

243      def __init__(
244          self,
245          estimator,
246          *,
247          threshold=None,
248          prefit=False,
249          norm_order=1,
250          max_features=None,
251          importance_getter="auto",
252      ):
253          self.estimator = estimator
254          self.threshold = threshold
255          self.prefit = prefit
256          self.importance_getter = importance_getter
257          self.norm_order = norm_order
258          self.max_features = max_features

260      def _get_support_mask(self):
261          estimator = getattr(self, "estimator_", self.estimator)
262          max_features = getattr(self, "max_features_", self.max_features)
263  
264          if self.prefit:
265              try:
266                  check_is_fitted(self.estimator)
267              except NotFittedError as exc:
268                  raise NotFittedError(
269                      "When `prefit=True`, `estimator` is expected to be a fitted "
270                      "estimator."
271                  ) from exc
272          if callable(max_features):
273              # This branch is executed when `transform` is called directly and thus
274              # `max_features_` is not set and we fallback using `self.max_features`
275              # that is not validated
276              raise NotFittedError(
277                  "When `prefit=True` and `max_features` is a callable, call `fit` "
278                  "before calling `transform`."
279              )
280          elif max_features is not None and not isinstance(max_features, Integral):
281              raise ValueError(
282                  f"`max_features` must be an integer. Got `max_features={max_features}` "
283                  "instead."
284              )
285  
286          scores = _get_feature_importances(
287              estimator=estimator,
288              getter=self.importance_getter,
289              transform_func="norm",
290              norm_order=self.norm_order,
291          )
292          threshold = _calculate_threshold(estimator, scores, self.threshold)
293          if self.max_features is not None:
294              mask = np.zeros_like(scores, dtype=bool)
295              candidate_indices = np.argsort(-scores, kind="mergesort")[:max_features]
296              mask[candidate_indices] = True
297          else:
298              mask = np.ones_like(scores, dtype=bool)
299          mask[scores < threshold] = False
300          return mask
301  
302      def _check_max_features(self, X):
303          if self.max_features is not None:
304              n_features = _num_features(X)
305  
306              if callable(self.max_features):
307                  max_features = self.max_features(X)
308              else:  # int
309                  max_features = self.max_features
310  
311              check_scalar(
312                  max_features,
313                  "max_features",
314                  Integral,
315                  min_val=0,
316                  max_val=n_features,
317              )
318              self.max_features_ = max_features

320      @_fit_context(
321          # SelectFromModel.estimator is not validated yet
322          prefer_skip_nested_validation=False
323      )
324      def fit(self, X, y=None, **fit_params):
325          """Fit the SelectFromModel meta-transformer. 
326   
327          Parameters 
328          ---------- 
329          X : array-like of shape (n_samples, n_features) 
330              The training input samples. 
331   
332          y : array-like of shape (n_samples,), default=None 
333              The target values (integers that correspond to classes in 
334              classification, real numbers in regression). 
335   
336          **fit_params : dict 
337              - If `enable_metadata_routing=False` (default): Parameters directly passed 
338                to the `fit` method of the sub-estimator. They are ignored if 
339                `prefit=True`. 
340   
341              - If `enable_metadata_routing=True`: Parameters safely routed to the `fit` 
342                method of the sub-estimator. They are ignored if `prefit=True`. 
343   
344              .. versionchanged:: 1.4 
345                  See :ref:`Metadata Routing User Guide <metadata_routing>` for 
346                  more details. 
347   
348          Returns 
349          ------- 
350          self : object 
351              Fitted estimator. 
352          """
353          self._check_max_features(X)
354  
355          if self.prefit:
356              try:
357                  check_is_fitted(self.estimator)
358              except NotFittedError as exc:
359                  raise NotFittedError(
360                      "When `prefit=True`, `estimator` is expected to be a fitted "
361                      "estimator."
362                  ) from exc
363              self.estimator_ = deepcopy(self.estimator)
364          else:
365              if _routing_enabled():
366                  routed_params = process_routing(self, "fit", **fit_params)
367                  self.estimator_ = clone(self.estimator)
368                  self.estimator_.fit(X, y, **routed_params.estimator.fit)
369              else:
370                  # TODO(SLEP6): remove when metadata routing cannot be disabled.
371                  self.estimator_ = clone(self.estimator)
372                  self.estimator_.fit(X, y, **fit_params)
373  
374          if hasattr(self.estimator_, "feature_names_in_"):
375              self.feature_names_in_ = self.estimator_.feature_names_in_
376          else:
377              _check_feature_names(self, X, reset=True)
378  
379          return self
380  
381      @property
382      def threshold_(self):
383          """Threshold value used for feature selection."""
384          scores = _get_feature_importances(
385              estimator=self.estimator_,
386              getter=self.importance_getter,
387              transform_func="norm",
388              norm_order=self.norm_order,
389          )
390          return _calculate_threshold(self.estimator, scores, self.threshold)

392      @available_if(_estimator_has("partial_fit"))
393      @_fit_context(
394          # SelectFromModel.estimator is not validated yet
395          prefer_skip_nested_validation=False
396      )
397      def partial_fit(self, X, y=None, **partial_fit_params):
398          """Fit the SelectFromModel meta-transformer only once. 
399   
400          Parameters 
401          ---------- 
402          X : array-like of shape (n_samples, n_features) 
403              The training input samples. 
404   
405          y : array-like of shape (n_samples,), default=None 
406              The target values (integers that correspond to classes in 
407              classification, real numbers in regression). 
408   
409          **partial_fit_params : dict 
410              - If `enable_metadata_routing=False` (default): Parameters directly passed 
411                to the `partial_fit` method of the sub-estimator. 
412   
413              - If `enable_metadata_routing=True`: Parameters passed to the `partial_fit` 
414                method of the sub-estimator. They are ignored if `prefit=True`. 
415   
416              .. versionchanged:: 1.4 
417   
418                  `**partial_fit_params` are routed to the sub-estimator, if 
419                  `enable_metadata_routing=True` is set via 
420                  :func:`~sklearn.set_config`, which allows for aliasing. 
421   
422                  See :ref:`Metadata Routing User Guide <metadata_routing>` for 
423                  more details. 
424   
425          Returns 
426          ------- 
427          self : object 
428              Fitted estimator. 
429          """
430          first_call = not hasattr(self, "estimator_")
431  
432          if first_call:
433              self._check_max_features(X)
434  
435          if self.prefit:
436              if first_call:
437                  try:
438                      check_is_fitted(self.estimator)
439                  except NotFittedError as exc:
440                      raise NotFittedError(
441                          "When `prefit=True`, `estimator` is expected to be a fitted "
442                          "estimator."
443                      ) from exc
444                  self.estimator_ = deepcopy(self.estimator)
445              return self
446  
447          if first_call:
448              self.estimator_ = clone(self.estimator)
449          if _routing_enabled():
450              routed_params = process_routing(self, "partial_fit", **partial_fit_params)
451              self.estimator_ = clone(self.estimator)
452              self.estimator_.partial_fit(X, y, **routed_params.estimator.partial_fit)
453          else:
454              # TODO(SLEP6): remove when metadata routing cannot be disabled.
455              self.estimator_.partial_fit(X, y, **partial_fit_params)
456  
457          if hasattr(self.estimator_, "feature_names_in_"):
458              self.feature_names_in_ = self.estimator_.feature_names_in_
459          else:
460              _check_feature_names(self, X, reset=first_call)
461  
462          return self
463  
464      @property
465      def n_features_in_(self):
466          """Number of features seen during `fit`."""
467          # For consistency with other estimators we raise a AttributeError so
468          # that hasattr() fails if the estimator isn't fitted.
469          try:
470              check_is_fitted(self)
471          except NotFittedError as nfe:
472              raise AttributeError(
473                  "{} object has no n_features_in_ attribute.".format(
474                      self.__class__.__name__
475                  )
476              ) from nfe
477  
478          return self.estimator_.n_features_in_

480      def get_metadata_routing(self):
481          """Get metadata routing of this object. 
482   
483          Please check :ref:`User Guide <metadata_routing>` on how the routing 
484          mechanism works. 
485   
486          .. versionadded:: 1.4 
487   
488          Returns 
489          ------- 
490          routing : MetadataRouter 
491              A :class:`~sklearn.utils.metadata_routing.MetadataRouter` encapsulating 
492              routing information. 
493          """
494          router = MetadataRouter(owner=self.__class__.__name__).add(
495              estimator=self.estimator,
496              method_mapping=MethodMapping()
497              .add(caller="partial_fit", callee="partial_fit")
498              .add(caller="fit", callee="fit"),
499          )
500          return router

503          tags = super().__sklearn_tags__()
504          tags.input_tags.sparse = get_tags(self.estimator).input_tags.sparse
505          tags.input_tags.allow_nan = get_tags(self.estimator).input_tags.allow_nan
