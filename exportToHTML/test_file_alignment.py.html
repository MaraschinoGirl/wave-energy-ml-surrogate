14   class TestFileAlignment(TestCase):
15       """ 
16           Ensure that setting the file alignment has the desired effect 
17           in the internal structure. 
18       """
19       def test_no_alignment_set(self):
20           fname = self.mktemp()
21           # 881 is a prime number, so hopefully this help randomize the alignment
22           # enough
23           # A nice even number might give a pathological case where
24           # While we don't want the data to be aligned, it ends up aligned...
25           shape = (881,)
26   
27           with h5py.File(fname, 'w') as h5file:
28               # Create up to 1000 datasets
29               # At least one of them should be misaligned.
30               # While this isn't perfect, it seems that there
31               # The case where 1000 datasets get created is one where the data
32               # is aligned. Therefore, during correct operation, this test is
33               # expected to finish quickly
34               for i in range(1000):
35                   dataset = h5file.create_dataset(
36                       dataset_name(i), shape, dtype='uint8')
37                   # Assign data so that the dataset is instantiated in
38                   # the file
39                   dataset[...] = i
40                   if not is_aligned(dataset):
41                       # Break early asserting that the file is not aligned
42                       break
43               else:
44                   raise RuntimeError("Data was all found to be aligned to 4096")

46       def test_alignment_set_above_threshold(self):
47           # 2022/01/19 hmaarrfk
48           # UnitTest (TestCase) doesn't play well with pytest parametrization.
49           alignment_threshold = 1000
50           alignment_interval = 4096
51   
52           for shape in [
53               (1033,),  # A prime number above the threshold
54               (1000,),  # Exactly equal to the threshold
55               (1001,),  # one above the threshold
56           ]:
57               fname = self.mktemp()
58               with h5py.File(fname, 'w',
59                              alignment_threshold=alignment_threshold,
60                              alignment_interval=alignment_interval) as h5file:
61                   # Create up to 1000 datasets
62                   # They are all expected to be aligned
63                   for i in range(1000):
64                       dataset = h5file.create_dataset(
65                           dataset_name(i), shape, dtype='uint8')
66                       # Assign data so that the dataset is instantiated in
67                       # the file
68                       dataset[...] = (i % 256)  # Truncate to uint8
69                       assert is_aligned(dataset, offset=alignment_interval)
