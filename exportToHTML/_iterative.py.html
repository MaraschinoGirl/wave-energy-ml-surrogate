40   def _assign_where(X1, X2, cond):
41       """Assign X2 to X1 where cond is True. 
42    
43       Parameters 
44       ---------- 
45       X1 : ndarray or dataframe of shape (n_samples, n_features) 
46           Data. 
47    
48       X2 : ndarray of shape (n_samples, n_features) 
49           Data to be assigned. 
50    
51       cond : ndarray of shape (n_samples, n_features) 
52           Boolean mask to assign data. 
53       """
54       if hasattr(X1, "mask"):  # pandas dataframes
55           X1.mask(cond=cond, other=X2, inplace=True)
56       else:  # ndarrays
57           X1[cond] = X2[cond]

60   class IterativeImputer(_BaseImputer):
61       """Multivariate imputer that estimates each feature from all the others. 
62    
63       A strategy for imputing missing values by modeling each feature with 
64       missing values as a function of other features in a round-robin fashion. 
65    
66       Read more in the :ref:`User Guide <iterative_imputer>`. 
67    
68       .. versionadded:: 0.21 
69    
70       .. note:: 
71    
72         This estimator is still **experimental** for now: the predictions 
73         and the API might change without any deprecation cycle. To use it, 
74         you need to explicitly import `enable_iterative_imputer`:: 
75    
76           >>> # explicitly require this experimental feature 
77           >>> from sklearn.experimental import enable_iterative_imputer  # noqa 
78           >>> # now you can import normally from sklearn.impute 
79           >>> from sklearn.impute import IterativeImputer 
80    
81       Parameters 
82       ---------- 
83       estimator : estimator object, default=BayesianRidge() 
84           The estimator to use at each step of the round-robin imputation. 
85           If `sample_posterior=True`, the estimator must support 
86           `return_std` in its `predict` method. 
87    
88       missing_values : int or np.nan, default=np.nan 
89           The placeholder for the missing values. All occurrences of 
90           `missing_values` will be imputed. For pandas' dataframes with 
91           nullable integer dtypes with missing values, `missing_values` 
92           should be set to `np.nan`, since `pd.NA` will be converted to `np.nan`. 
93    
94       sample_posterior : bool, default=False 
95           Whether to sample from the (Gaussian) predictive posterior of the 
96           fitted estimator for each imputation. Estimator must support 
97           `return_std` in its `predict` method if set to `True`. Set to 
98           `True` if using `IterativeImputer` for multiple imputations. 
99    
100      max_iter : int, default=10 
101          Maximum number of imputation rounds to perform before returning the 
102          imputations computed during the final round. A round is a single 
103          imputation of each feature with missing values. The stopping criterion 
104          is met once `max(abs(X_t - X_{t-1}))/max(abs(X[known_vals])) < tol`, 
105          where `X_t` is `X` at iteration `t`. Note that early stopping is only 
106          applied if `sample_posterior=False`. 
107   
108      tol : float, default=1e-3 
109          Tolerance of the stopping condition. 
110   
111      n_nearest_features : int, default=None 
112          Number of other features to use to estimate the missing values of 
113          each feature column. Nearness between features is measured using 
114          the absolute correlation coefficient between each feature pair (after 
115          initial imputation). To ensure coverage of features throughout the 
116          imputation process, the neighbor features are not necessarily nearest, 
117          but are drawn with probability proportional to correlation for each 
118          imputed target feature. Can provide significant speed-up when the 
119          number of features is huge. If `None`, all features will be used. 
120   
121      initial_strategy : {'mean', 'median', 'most_frequent', 'constant'}, \ 
122              default='mean' 
123          Which strategy to use to initialize the missing values. Same as the 
124          `strategy` parameter in :class:`~sklearn.impute.SimpleImputer`. 
125   
126      fill_value : str or numerical value, default=None 
127          When `strategy="constant"`, `fill_value` is used to replace all 
128          occurrences of missing_values. For string or object data types, 
129          `fill_value` must be a string. 
130          If `None`, `fill_value` will be 0 when imputing numerical 
131          data and "missing_value" for strings or object data types. 
132   
133          .. versionadded:: 1.3 
134   
135      imputation_order : {'ascending', 'descending', 'roman', 'arabic', \ 
136              'random'}, default='ascending' 
137          The order in which the features will be imputed. Possible values: 
138   
139          - `'ascending'`: From features with fewest missing values to most. 
140          - `'descending'`: From features with most missing values to fewest. 
141          - `'roman'`: Left to right. 
142          - `'arabic'`: Right to left. 
143          - `'random'`: A random order for each round. 
144   
145      skip_complete : bool, default=False 
146          If `True` then features with missing values during :meth:`transform` 
147          which did not have any missing values during :meth:`fit` will be 
148          imputed with the initial imputation method only. Set to `True` if you 
149          have many features with no missing values at both :meth:`fit` and 
150          :meth:`transform` time to save compute. 
151   
152      min_value : float or array-like of shape (n_features,), default=-np.inf 
153          Minimum possible imputed value. Broadcast to shape `(n_features,)` if 
154          scalar. If array-like, expects shape `(n_features,)`, one min value for 
155          each feature. The default is `-np.inf`. 
156   
157          .. versionchanged:: 0.23 
158             Added support for array-like. 
159   
160      max_value : float or array-like of shape (n_features,), default=np.inf 
161          Maximum possible imputed value. Broadcast to shape `(n_features,)` if 
162          scalar. If array-like, expects shape `(n_features,)`, one max value for 
163          each feature. The default is `np.inf`. 
164   
165          .. versionchanged:: 0.23 
166             Added support for array-like. 
167   
168      verbose : int, default=0 
169          Verbosity flag, controls the debug messages that are issued 
170          as functions are evaluated. The higher, the more verbose. Can be 0, 1, 
171          or 2. 
172   
173      random_state : int, RandomState instance or None, default=None 
174          The seed of the pseudo random number generator to use. Randomizes 
175          selection of estimator features if `n_nearest_features` is not `None`, 
176          the `imputation_order` if `random`, and the sampling from posterior if 
177          `sample_posterior=True`. Use an integer for determinism. 
178          See :term:`the Glossary <random_state>`. 
179   
180      add_indicator : bool, default=False 
181          If `True`, a :class:`MissingIndicator` transform will stack onto output 
182          of the imputer's transform. This allows a predictive estimator 
183          to account for missingness despite imputation. If a feature has no 
184          missing values at fit/train time, the feature won't appear on 
185          the missing indicator even if there are missing values at 
186          transform/test time. 
187   
188      keep_empty_features : bool, default=False 
189          If True, features that consist exclusively of missing values when 
190          `fit` is called are returned in results when `transform` is called. 
191          The imputed value is always `0` except when 
192          `initial_strategy="constant"` in which case `fill_value` will be 
193          used instead. 
194   
195          .. versionadded:: 1.2 
196   
197      Attributes 
198      ---------- 
199      initial_imputer_ : object of type :class:`~sklearn.impute.SimpleImputer` 
200          Imputer used to initialize the missing values. 
201   
202      imputation_sequence_ : list of tuples 
203          Each tuple has `(feat_idx, neighbor_feat_idx, estimator)`, where 
204          `feat_idx` is the current feature to be imputed, 
205          `neighbor_feat_idx` is the array of other features used to impute the 
206          current feature, and `estimator` is the trained estimator used for 
207          the imputation. Length is `self.n_features_with_missing_ * 
208          self.n_iter_`. 
209   
210      n_iter_ : int 
211          Number of iteration rounds that occurred. Will be less than 
212          `self.max_iter` if early stopping criterion was reached. 
213   
214      n_features_in_ : int 
215          Number of features seen during :term:`fit`. 
216   
217          .. versionadded:: 0.24 
218   
219      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
220          Names of features seen during :term:`fit`. Defined only when `X` 
221          has feature names that are all strings. 
222   
223          .. versionadded:: 1.0 
224   
225      n_features_with_missing_ : int 
226          Number of features with missing values. 
227   
228      indicator_ : :class:`~sklearn.impute.MissingIndicator` 
229          Indicator used to add binary indicators for missing values. 
230          `None` if `add_indicator=False`. 
231   
232      random_state_ : RandomState instance 
233          RandomState instance that is generated either from a seed, the random 
234          number generator or by `np.random`. 
235   
236      See Also 
237      -------- 
238      SimpleImputer : Univariate imputer for completing missing values 
239          with simple strategies. 
240      KNNImputer : Multivariate imputer that estimates missing features using 
241          nearest samples. 
242   
243      Notes 
244      ----- 
245      To support imputation in inductive mode we store each feature's estimator 
246      during the :meth:`fit` phase, and predict without refitting (in order) 
247      during the :meth:`transform` phase. 
248   
249      Features which contain all missing values at :meth:`fit` are discarded upon 
250      :meth:`transform`. 
251   
252      Using defaults, the imputer scales in :math:`\\mathcal{O}(knp^3\\min(n,p))` 
253      where :math:`k` = `max_iter`, :math:`n` the number of samples and 
254      :math:`p` the number of features. It thus becomes prohibitively costly when 
255      the number of features increases. Setting 
256      `n_nearest_features << n_features`, `skip_complete=True` or increasing `tol` 
257      can help to reduce its computational cost. 
258   
259      Depending on the nature of missing values, simple imputers can be 
260      preferable in a prediction context. 
261   
262      References 
263      ---------- 
264      .. [1] `Stef van Buuren, Karin Groothuis-Oudshoorn (2011). "mice: 
265          Multivariate Imputation by Chained Equations in R". Journal of 
266          Statistical Software 45: 1-67. 
267          <https://www.jstatsoft.org/article/view/v045i03>`_ 
268   
269      .. [2] `S. F. Buck, (1960). "A Method of Estimation of Missing Values in 
270          Multivariate Data Suitable for use with an Electronic Computer". 
271          Journal of the Royal Statistical Society 22(2): 302-306. 
272          <https://www.jstor.org/stable/2984099>`_ 
273   
274      Examples 
275      -------- 
276      >>> import numpy as np 
277      >>> from sklearn.experimental import enable_iterative_imputer 
278      >>> from sklearn.impute import IterativeImputer 
279      >>> imp_mean = IterativeImputer(random_state=0) 
280      >>> imp_mean.fit([[7, 2, 3], [4, np.nan, 6], [10, 5, 9]]) 
281      IterativeImputer(random_state=0) 
282      >>> X = [[np.nan, 2, 3], [4, np.nan, 6], [10, np.nan, 9]] 
283      >>> imp_mean.transform(X) 
284      array([[ 6.9584...,  2.       ,  3.        ], 
285             [ 4.       ,  2.6000...,  6.        ], 
286             [10.       ,  4.9999...,  9.        ]]) 
287   
288      For a more detailed example see 
289      :ref:`sphx_glr_auto_examples_impute_plot_missing_values.py` or 
290      :ref:`sphx_glr_auto_examples_impute_plot_iterative_imputer_variants_comparison.py`. 
291      """

354      def _impute_one_feature(
355          self,
356          X_filled,
357          mask_missing_values,
358          feat_idx,
359          neighbor_feat_idx,
360          estimator=None,
361          fit_mode=True,
362          params=None,
363      ):
364          """Impute a single feature from the others provided. 
365   
366          This function predicts the missing values of one of the features using 
367          the current estimates of all the other features. The `estimator` must 
368          support `return_std=True` in its `predict` method for this function 
369          to work. 
370   
371          Parameters 
372          ---------- 
373          X_filled : ndarray 
374              Input data with the most recent imputations. 
375   
376          mask_missing_values : ndarray 
377              Input data's missing indicator matrix. 
378   
379          feat_idx : int 
380              Index of the feature currently being imputed. 
381   
382          neighbor_feat_idx : ndarray 
383              Indices of the features to be used in imputing `feat_idx`. 
384   
385          estimator : object 
386              The estimator to use at this step of the round-robin imputation. 
387              If `sample_posterior=True`, the estimator must support 
388              `return_std` in its `predict` method. 
389              If None, it will be cloned from self._estimator. 
390   
391          fit_mode : boolean, default=True 
392              Whether to fit and predict with the estimator or just predict. 
393   
394          params : dict 
395              Additional params routed to the individual estimator. 
396   
397          Returns 
398          ------- 
399          X_filled : ndarray 
400              Input data with `X_filled[missing_row_mask, feat_idx]` updated. 
401   
402          estimator : estimator with sklearn API 
403              The fitted estimator used to impute 
404              `X_filled[missing_row_mask, feat_idx]`. 
405          """
406          if estimator is None and fit_mode is False:
407              raise ValueError(
408                  "If fit_mode is False, then an already-fitted "
409                  "estimator should be passed in."
410              )
411  
412          if estimator is None:
413              estimator = clone(self._estimator)
414  
415          missing_row_mask = mask_missing_values[:, feat_idx]
416          if fit_mode:
417              X_train = _safe_indexing(
418                  _safe_indexing(X_filled, neighbor_feat_idx, axis=1),
419                  ~missing_row_mask,
420                  axis=0,
421              )
422              y_train = _safe_indexing(
423                  _safe_indexing(X_filled, feat_idx, axis=1),
424                  ~missing_row_mask,
425                  axis=0,
426              )
427              estimator.fit(X_train, y_train, **params)
428  
429          # if no missing values, don't predict
430          if np.sum(missing_row_mask) == 0:
431              return X_filled, estimator
432  
433          # get posterior samples if there is at least one missing value
434          X_test = _safe_indexing(
435              _safe_indexing(X_filled, neighbor_feat_idx, axis=1),
436              missing_row_mask,
437              axis=0,
438          )
439          if self.sample_posterior:
440              mus, sigmas = estimator.predict(X_test, return_std=True)
441              imputed_values = np.zeros(mus.shape, dtype=X_filled.dtype)
442              # two types of problems: (1) non-positive sigmas
443              # (2) mus outside legal range of min_value and max_value
444              # (results in inf sample)
445              positive_sigmas = sigmas > 0
446              imputed_values[~positive_sigmas] = mus[~positive_sigmas]
447              mus_too_low = mus < self._min_value[feat_idx]
448              imputed_values[mus_too_low] = self._min_value[feat_idx]
449              mus_too_high = mus > self._max_value[feat_idx]
450              imputed_values[mus_too_high] = self._max_value[feat_idx]
451              # the rest can be sampled without statistical issues
452              inrange_mask = positive_sigmas & ~mus_too_low & ~mus_too_high
453              mus = mus[inrange_mask]
454              sigmas = sigmas[inrange_mask]
455              a = (self._min_value[feat_idx] - mus) / sigmas
456              b = (self._max_value[feat_idx] - mus) / sigmas
457  
458              truncated_normal = stats.truncnorm(a=a, b=b, loc=mus, scale=sigmas)
459              imputed_values[inrange_mask] = truncated_normal.rvs(
460                  random_state=self.random_state_
461              )
462          else:
463              imputed_values = estimator.predict(X_test)
464              imputed_values = np.clip(
465                  imputed_values, self._min_value[feat_idx], self._max_value[feat_idx]
466              )
467  
468          # update the feature
469          _safe_assign(
470              X_filled,
471              imputed_values,
472              row_indexer=missing_row_mask,
473              column_indexer=feat_idx,
474          )
475          return X_filled, estimator

477      def _get_neighbor_feat_idx(self, n_features, feat_idx, abs_corr_mat):
478          """Get a list of other features to predict `feat_idx`. 
479   
480          If `self.n_nearest_features` is less than or equal to the total 
481          number of features, then use a probability proportional to the absolute 
482          correlation between `feat_idx` and each other feature to randomly 
483          choose a subsample of the other features (without replacement). 
484   
485          Parameters 
486          ---------- 
487          n_features : int 
488              Number of features in `X`. 
489   
490          feat_idx : int 
491              Index of the feature currently being imputed. 
492   
493          abs_corr_mat : ndarray, shape (n_features, n_features) 
494              Absolute correlation matrix of `X`. The diagonal has been zeroed 
495              out and each feature has been normalized to sum to 1. Can be None. 
496   
497          Returns 
498          ------- 
499          neighbor_feat_idx : array-like 
500              The features to use to impute `feat_idx`. 
501          """
502          if self.n_nearest_features is not None and self.n_nearest_features < n_features:
503              p = abs_corr_mat[:, feat_idx]
504              neighbor_feat_idx = self.random_state_.choice(
505                  np.arange(n_features), self.n_nearest_features, replace=False, p=p
506              )
507          else:
508              inds_left = np.arange(feat_idx)
509              inds_right = np.arange(feat_idx + 1, n_features)
510              neighbor_feat_idx = np.concatenate((inds_left, inds_right))
511          return neighbor_feat_idx

513      def _get_ordered_idx(self, mask_missing_values):
514          """Decide in what order we will update the features. 
515   
516          As a homage to the MICE R package, we will have 4 main options of 
517          how to order the updates, and use a random order if anything else 
518          is specified. 
519   
520          Also, this function skips features which have no missing values. 
521   
522          Parameters 
523          ---------- 
524          mask_missing_values : array-like, shape (n_samples, n_features) 
525              Input data's missing indicator matrix, where `n_samples` is the 
526              number of samples and `n_features` is the number of features. 
527   
528          Returns 
529          ------- 
530          ordered_idx : ndarray, shape (n_features,) 
531              The order in which to impute the features. 
532          """
533          frac_of_missing_values = mask_missing_values.mean(axis=0)
534          if self.skip_complete:
535              missing_values_idx = np.flatnonzero(frac_of_missing_values)
536          else:
537              missing_values_idx = np.arange(np.shape(frac_of_missing_values)[0])
538          if self.imputation_order == "roman":
539              ordered_idx = missing_values_idx
540          elif self.imputation_order == "arabic":
541              ordered_idx = missing_values_idx[::-1]
542          elif self.imputation_order == "ascending":
543              n = len(frac_of_missing_values) - len(missing_values_idx)
544              ordered_idx = np.argsort(frac_of_missing_values, kind="mergesort")[n:]
545          elif self.imputation_order == "descending":
546              n = len(frac_of_missing_values) - len(missing_values_idx)
547              ordered_idx = np.argsort(frac_of_missing_values, kind="mergesort")[n:][::-1]
548          elif self.imputation_order == "random":
549              ordered_idx = missing_values_idx
550              self.random_state_.shuffle(ordered_idx)
551          return ordered_idx

553      def _get_abs_corr_mat(self, X_filled, tolerance=1e-6):
554          """Get absolute correlation matrix between features. 
555   
556          Parameters 
557          ---------- 
558          X_filled : ndarray, shape (n_samples, n_features) 
559              Input data with the most recent imputations. 
560   
561          tolerance : float, default=1e-6 
562              `abs_corr_mat` can have nans, which will be replaced 
563              with `tolerance`. 
564   
565          Returns 
566          ------- 
567          abs_corr_mat : ndarray, shape (n_features, n_features) 
568              Absolute correlation matrix of `X` at the beginning of the 
569              current round. The diagonal has been zeroed out and each feature's 
570              absolute correlations with all others have been normalized to sum 
571              to 1. 
572          """
573          n_features = X_filled.shape[1]
574          if self.n_nearest_features is None or self.n_nearest_features >= n_features:
575              return None
576          with np.errstate(invalid="ignore"):
577              # if a feature in the neighborhood has only a single value
578              # (e.g., categorical feature), the std. dev. will be null and
579              # np.corrcoef will raise a warning due to a division by zero
580              abs_corr_mat = np.abs(np.corrcoef(X_filled.T))
581          # np.corrcoef is not defined for features with zero std
582          abs_corr_mat[np.isnan(abs_corr_mat)] = tolerance
583          # ensures exploration, i.e. at least some probability of sampling
584          np.clip(abs_corr_mat, tolerance, None, out=abs_corr_mat)
585          # features are not their own neighbors
586          np.fill_diagonal(abs_corr_mat, 0)
587          # needs to sum to 1 for np.random.choice sampling
588          abs_corr_mat = normalize(abs_corr_mat, norm="l1", axis=0, copy=False)
589          return abs_corr_mat

591      def _initial_imputation(self, X, in_fit=False):
592          """Perform initial imputation for input `X`. 
593   
594          Parameters 
595          ---------- 
596          X : ndarray of shape (n_samples, n_features) 
597              Input data, where `n_samples` is the number of samples and 
598              `n_features` is the number of features. 
599   
600          in_fit : bool, default=False 
601              Whether function is called in :meth:`fit`. 
602   
603          Returns 
604          ------- 
605          Xt : ndarray of shape (n_samples, n_features) 
606              Input data, where `n_samples` is the number of samples and 
607              `n_features` is the number of features. 
608   
609          X_filled : ndarray of shape (n_samples, n_features) 
610              Input data with the most recent imputations. 
611   
612          mask_missing_values : ndarray of shape (n_samples, n_features) 
613              Input data's missing indicator matrix, where `n_samples` is the 
614              number of samples and `n_features` is the number of features, 
615              masked by non-missing features. 
616   
617          X_missing_mask : ndarray, shape (n_samples, n_features) 
618              Input data's mask matrix indicating missing datapoints, where 
619              `n_samples` is the number of samples and `n_features` is the 
620              number of features. 
621          """
622          if is_scalar_nan(self.missing_values):
623              ensure_all_finite = "allow-nan"
624          else:
625              ensure_all_finite = True
626  
627          X = validate_data(
628              self,
629              X,
630              dtype=FLOAT_DTYPES,
631              order="F",
632              reset=in_fit,
633              ensure_all_finite=ensure_all_finite,
634          )
635          _check_inputs_dtype(X, self.missing_values)
636  
637          X_missing_mask = _get_mask(X, self.missing_values)
638          mask_missing_values = X_missing_mask.copy()
639  
640          # TODO (1.8): remove this once the deprecation is removed. In the meantime,
641          # we need to catch the warning to avoid false positives.
642          catch_warning = (
643              self.initial_strategy == "constant" and not self.keep_empty_features
644          )
645  
646          if self.initial_imputer_ is None:
647              self.initial_imputer_ = SimpleImputer(
648                  missing_values=self.missing_values,
649                  strategy=self.initial_strategy,
650                  fill_value=self.fill_value,
651                  keep_empty_features=self.keep_empty_features,
652              ).set_output(transform="default")
653  
654              # TODO (1.8): remove this once the deprecation is removed to keep only
655              # the code in the else case.
656              if catch_warning:
657                  with warnings.catch_warnings():
658                      warnings.simplefilter("ignore", FutureWarning)
659                      X_filled = self.initial_imputer_.fit_transform(X)
660              else:
661                  X_filled = self.initial_imputer_.fit_transform(X)
662          else:
663              # TODO (1.8): remove this once the deprecation is removed to keep only
664              # the code in the else case.
665              if catch_warning:
666                  with warnings.catch_warnings():
667                      warnings.simplefilter("ignore", FutureWarning)
668                      X_filled = self.initial_imputer_.transform(X)
669              else:
670                  X_filled = self.initial_imputer_.transform(X)
671  
672          if in_fit:
673              self._is_empty_feature = np.all(mask_missing_values, axis=0)
674  
675          if not self.keep_empty_features:
676              # drop empty features
677              Xt = X[:, ~self._is_empty_feature]
678              mask_missing_values = mask_missing_values[:, ~self._is_empty_feature]
679  
680              if self.initial_imputer_.get_params()["strategy"] == "constant":
681                  # The constant strategy has a specific behavior and preserve empty
682                  # features even with ``keep_empty_features=False``. We need to drop
683                  # the column for consistency.
684                  # TODO (1.8): remove this `if` branch once the following issue is
685                  # addressed:
686                  # https://github.com/scikit-learn/scikit-learn/issues/29827
687                  X_filled = X_filled[:, ~self._is_empty_feature]
688  
689          else:
690              # mark empty features as not missing and keep the original
691              # imputation
692              mask_missing_values[:, self._is_empty_feature] = False
693              Xt = X
694              Xt[:, self._is_empty_feature] = X_filled[:, self._is_empty_feature]
695  
696          return Xt, X_filled, mask_missing_values, X_missing_mask
697  
698      @staticmethod
699      def _validate_limit(
700          limit, limit_type, n_features, is_empty_feature, keep_empty_feature
701      ):
702          """Validate the limits (min/max) of the feature values. 
703   
704          Converts scalar min/max limits to vectors of shape `(n_features,)`. 
705   
706          Parameters 
707          ---------- 
708          limit: scalar or array-like 
709              The user-specified limit (i.e, min_value or max_value). 
710          limit_type: {'max', 'min'} 
711              Type of limit to validate. 
712          n_features: int 
713              Number of features in the dataset. 
714          is_empty_feature: ndarray, shape (n_features, ) 
715              Mask array indicating empty feature imputer has seen during fit. 
716          keep_empty_feature: bool 
717              If False, remove empty-feature indices from the limit. 
718   
719          Returns 
720          ------- 
721          limit: ndarray, shape(n_features,) 
722              Array of limits, one for each feature. 
723          """
724          n_features_in = _num_samples(is_empty_feature)
725          if (
726              limit is not None
727              and not np.isscalar(limit)
728              and _num_samples(limit) != n_features_in
729          ):
730              raise ValueError(
731                  f"'{limit_type}_value' should be of shape ({n_features_in},) when an"
732                  f" array-like is provided. Got {len(limit)}, instead."
733              )
734  
735          limit_bound = np.inf if limit_type == "max" else -np.inf
736          limit = limit_bound if limit is None else limit
737          if np.isscalar(limit):
738              limit = np.full(n_features, limit)
739          limit = check_array(limit, ensure_all_finite=False, copy=False, ensure_2d=False)
740  
741          # Make sure to remove the empty feature elements from the bounds
742          if not keep_empty_feature and len(limit) == len(is_empty_feature):
743              limit = limit[~is_empty_feature]
744  
745          return limit

747      @_fit_context(
748          # IterativeImputer.estimator is not validated yet
749          prefer_skip_nested_validation=False
750      )
751      def fit_transform(self, X, y=None, **params):
752          """Fit the imputer on `X` and return the transformed `X`. 
753   
754          Parameters 
755          ---------- 
756          X : array-like, shape (n_samples, n_features) 
757              Input data, where `n_samples` is the number of samples and 
758              `n_features` is the number of features. 
759   
760          y : Ignored 
761              Not used, present for API consistency by convention. 
762   
763          **params : dict 
764              Parameters routed to the `fit` method of the sub-estimator via the 
765              metadata routing API. 
766   
767              .. versionadded:: 1.5 
768                Only available if 
769                `sklearn.set_config(enable_metadata_routing=True)` is set. See 
770                :ref:`Metadata Routing User Guide <metadata_routing>` for more 
771                details. 
772   
773          Returns 
774          ------- 
775          Xt : array-like, shape (n_samples, n_features) 
776              The imputed input data. 
777          """
778          _raise_for_params(params, self, "fit")
779  
780          routed_params = process_routing(
781              self,
782              "fit",
783              **params,
784          )
785  
786          self.random_state_ = getattr(
787              self, "random_state_", check_random_state(self.random_state)
788          )
789  
790          if self.estimator is None:
791              from ..linear_model import BayesianRidge
792  
793              self._estimator = BayesianRidge()
794          else:
795              self._estimator = clone(self.estimator)
796  
797          self.imputation_sequence_ = []
798  
799          self.initial_imputer_ = None
800  
801          X, Xt, mask_missing_values, complete_mask = self._initial_imputation(
802              X, in_fit=True
803          )
804  
805          super()._fit_indicator(complete_mask)
806          X_indicator = super()._transform_indicator(complete_mask)
807  
808          if self.max_iter == 0 or np.all(mask_missing_values):
809              self.n_iter_ = 0
810              return super()._concatenate_indicator(Xt, X_indicator)
811  
812          # Edge case: a single feature, we return the initial imputation.
813          if Xt.shape[1] == 1:
814              self.n_iter_ = 0
815              return super()._concatenate_indicator(Xt, X_indicator)
816  
817          self._min_value = self._validate_limit(
818              self.min_value,
819              "min",
820              X.shape[1],
821              self._is_empty_feature,
822              self.keep_empty_features,
823          )
824          self._max_value = self._validate_limit(
825              self.max_value,
826              "max",
827              X.shape[1],
828              self._is_empty_feature,
829              self.keep_empty_features,
830          )
831  
832          if not np.all(np.greater(self._max_value, self._min_value)):
833              raise ValueError("One (or more) features have min_value >= max_value.")
834  
835          # order in which to impute
836          # note this is probably too slow for large feature data (d > 100000)
837          # and a better way would be good.
838          # see: https://goo.gl/KyCNwj and subsequent comments
839          ordered_idx = self._get_ordered_idx(mask_missing_values)
840          self.n_features_with_missing_ = len(ordered_idx)
841  
842          abs_corr_mat = self._get_abs_corr_mat(Xt)
843  
844          n_samples, n_features = Xt.shape
845          if self.verbose > 0:
846              print("[IterativeImputer] Completing matrix with shape %s" % (X.shape,))
847          start_t = time()
848          if not self.sample_posterior:
849              Xt_previous = Xt.copy()
850              normalized_tol = self.tol * np.max(np.abs(X[~mask_missing_values]))
851          for self.n_iter_ in range(1, self.max_iter + 1):
852              if self.imputation_order == "random":
853                  ordered_idx = self._get_ordered_idx(mask_missing_values)
854  
855              for feat_idx in ordered_idx:
856                  neighbor_feat_idx = self._get_neighbor_feat_idx(
857                      n_features, feat_idx, abs_corr_mat
858                  )
859                  Xt, estimator = self._impute_one_feature(
860                      Xt,
861                      mask_missing_values,
862                      feat_idx,
863                      neighbor_feat_idx,
864                      estimator=None,
865                      fit_mode=True,
866                      params=routed_params.estimator.fit,
867                  )
868                  estimator_triplet = _ImputerTriplet(
869                      feat_idx, neighbor_feat_idx, estimator
870                  )
871                  self.imputation_sequence_.append(estimator_triplet)
872  
873              if self.verbose > 1:
874                  print(
875                      "[IterativeImputer] Ending imputation round "
876                      "%d/%d, elapsed time %0.2f"
877                      % (self.n_iter_, self.max_iter, time() - start_t)
878                  )
879  
880              if not self.sample_posterior:
881                  inf_norm = np.linalg.norm(Xt - Xt_previous, ord=np.inf, axis=None)
882                  if self.verbose > 0:
883                      print(
884                          "[IterativeImputer] Change: {}, scaled tolerance: {} ".format(
885                              inf_norm, normalized_tol
886                          )
887                      )
888                  if inf_norm < normalized_tol:
889                      if self.verbose > 0:
890                          print("[IterativeImputer] Early stopping criterion reached.")
891                      break
892                  Xt_previous = Xt.copy()
893          else:
894              if not self.sample_posterior:
895                  warnings.warn(
896                      "[IterativeImputer] Early stopping criterion not reached.",
897                      ConvergenceWarning,
898                  )
899          _assign_where(Xt, X, cond=~mask_missing_values)
900  
901          return super()._concatenate_indicator(Xt, X_indicator)
902  
903      def transform(self, X):
904          """Impute all missing values in `X`. 
905   
906          Note that this is stochastic, and that if `random_state` is not fixed, 
907          repeated calls, or permuted input, results will differ. 
908   
909          Parameters 
910          ---------- 
911          X : array-like of shape (n_samples, n_features) 
912              The input data to complete. 
913   
914          Returns 
915          ------- 
916          Xt : array-like, shape (n_samples, n_features) 
917               The imputed input data. 
918          """
919          check_is_fitted(self)
920  
921          X, Xt, mask_missing_values, complete_mask = self._initial_imputation(
922              X, in_fit=False
923          )
924  
925          X_indicator = super()._transform_indicator(complete_mask)
926  
927          if self.n_iter_ == 0 or np.all(mask_missing_values):
928              return super()._concatenate_indicator(Xt, X_indicator)
929  
930          imputations_per_round = len(self.imputation_sequence_) // self.n_iter_
931          i_rnd = 0
932          if self.verbose > 0:
933              print("[IterativeImputer] Completing matrix with shape %s" % (X.shape,))
934          start_t = time()
935          for it, estimator_triplet in enumerate(self.imputation_sequence_):
936              Xt, _ = self._impute_one_feature(
937                  Xt,
938                  mask_missing_values,
939                  estimator_triplet.feat_idx,
940                  estimator_triplet.neighbor_feat_idx,
941                  estimator=estimator_triplet.estimator,
942                  fit_mode=False,
943              )
944              if not (it + 1) % imputations_per_round:
945                  if self.verbose > 1:
946                      print(
947                          "[IterativeImputer] Ending imputation round "
948                          "%d/%d, elapsed time %0.2f"
949                          % (i_rnd + 1, self.n_iter_, time() - start_t)
950                      )
951                  i_rnd += 1
952  
953          _assign_where(Xt, X, cond=~mask_missing_values)
954  
955          return super()._concatenate_indicator(Xt, X_indicator)
956  
957      def fit(self, X, y=None, **fit_params):
958          """Fit the imputer on `X` and return self. 
959   
960          Parameters 
961          ---------- 
962          X : array-like, shape (n_samples, n_features) 
963              Input data, where `n_samples` is the number of samples and 
964              `n_features` is the number of features. 
965   
966          y : Ignored 
967              Not used, present for API consistency by convention. 
968   
969          **fit_params : dict 
970              Parameters routed to the `fit` method of the sub-estimator via the 
971              metadata routing API. 
972   
973              .. versionadded:: 1.5 
974                Only available if 
975                `sklearn.set_config(enable_metadata_routing=True)` is set. See 
976                :ref:`Metadata Routing User Guide <metadata_routing>` for more 
977                details. 
978   
979          Returns 
980          ------- 
981          self : object 
982              Fitted estimator. 
983          """
984          self.fit_transform(X, **fit_params)
985          return self

988          """Get output feature names for transformation. 
989   
990          Parameters 
991          ---------- 
992          input_features : array-like of str or None, default=None 
993              Input features. 
994   
995              - If `input_features` is `None`, then `feature_names_in_` is 
996                used as feature names in. If `feature_names_in_` is not defined,

1010         return self._concatenate_indicator_feature_names_out(names, input_features)
