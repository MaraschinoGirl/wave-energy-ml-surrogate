23   def check_dir(module, module_name=None):
24       """Returns a mapping of all objects with the wrong __module__ attribute."""
25       if module_name is None:
26           module_name = module.__name__
27       results = {}
28       for name in dir(module):
29           if name == "core":
30               continue
31           item = getattr(module, name)
32           if (hasattr(item, '__module__') and hasattr(item, '__name__')
33                   and item.__module__ != module_name):
34               results[name] = item.__module__ + '.' + item.__name__
35       return results

285          return False
286  
287      return True

341  def test_all_modules_are_expected_2():
342      """ 
343      Method checking all objects. The pkgutil-based method in 
344      `test_all_modules_are_expected` does not catch imports into a namespace, 
345      only filenames.  So this test is more thorough, and checks this like: 
346   
347          import .lib.scimath as emath 
348   
349      To check if something in a module is (effectively) public, one can check if 
350      there's anything in that namespace that's a public function/object but is 
351      not exposed in a higher-level namespace.  For example for a `numpy.lib` 
352      submodule:: 
353   
354          mod = np.lib.mixins 
355          for obj in mod.__all__: 
356              if obj in np.__all__: 
357                  continue 
358              elif obj in np.lib.__all__: 
359                  continue 
360   
361              else: 
362                  print(obj) 
363   
364      """
365  
366      def find_unexpected_members(mod_name):
367          members = []
368          module = importlib.import_module(mod_name)
369          if hasattr(module, '__all__'):
370              objnames = module.__all__
371          else:
372              objnames = dir(module)
373  
374          for objname in objnames:
375              if not objname.startswith('_'):
376                  fullobjname = mod_name + '.' + objname
377                  if isinstance(getattr(module, objname), types.ModuleType):
378                      if is_unexpected(fullobjname):
379                          if fullobjname not in SKIP_LIST_2:
380                              members.append(fullobjname)
381  
382          return members
383  
384      unexpected_members = find_unexpected_members("numpy")
385      for modname in PUBLIC_MODULES:
386          unexpected_members.extend(find_unexpected_members(modname))
387  
388      if unexpected_members:
389          raise AssertionError("Found unexpected object(s) that look like "
390                               "modules: {}".format(unexpected_members))

393  def test_api_importable():
394      """ 
395      Check that all submodules listed higher up in this file can be imported 
396   
397      Note that if a PRIVATE_BUT_PRESENT_MODULES entry goes missing, it may 
398      simply need to be removed from the list (deprecation may or may not be 
399      needed - apply common sense). 
400      """
401      def check_importable(module_name):
402          try:
403              importlib.import_module(module_name)
404          except (ImportError, AttributeError):
405              return False
406  
407          return True
408  
409      module_names = []
410      for module_name in PUBLIC_MODULES:
411          if not check_importable(module_name):
412              module_names.append(module_name)
413  
414      if module_names:
415          raise AssertionError("Modules in the public API that cannot be "
416                               "imported: {}".format(module_names))
417  
418      for module_name in PUBLIC_ALIASED_MODULES:
419          try:
420              eval(module_name)
421          except AttributeError:
422              module_names.append(module_name)
423  
424      if module_names:
425          raise AssertionError("Modules in the public API that were not "
426                               "found: {}".format(module_names))
427  
428      with warnings.catch_warnings(record=True) as w:
429          warnings.filterwarnings('always', category=DeprecationWarning)
430          warnings.filterwarnings('always', category=ImportWarning)
431          for module_name in PRIVATE_BUT_PRESENT_MODULES:
432              if not check_importable(module_name):
433                  module_names.append(module_name)
434  
435      if module_names:
436          raise AssertionError("Modules that are not really public but looked "
437                               "public and can not be imported: "
438                               "{}".format(module_names))

441  @pytest.mark.xfail(
442      sysconfig.get_config_var("Py_DEBUG") not in (None, 0, "0"),
443      reason=(
444          "NumPy possibly built with `USE_DEBUG=True ./tools/travis-test.sh`, "

449  def test_array_api_entry_point():
450      """ 
451      Entry point for Array API implementation can be found with importlib and 
452      returns the main numpy namespace. 
453      """
454      # For a development install that did not go through meson-python,
455      # the entrypoint will not have been installed. So ensure this test fails
456      # only if numpy is inside site-packages.
457      numpy_in_sitepackages = sysconfig.get_path('platlib') in np.__file__
458  
459      eps = importlib.metadata.entry_points()
460      try:
461          xp_eps = eps.select(group="array_api")
462      except AttributeError:
463          # The select interface for entry_points was introduced in py3.10,
464          # deprecating its dict interface. We fallback to dict keys for finding
465          # Array API entry points so that running this test in <=3.9 will
466          # still work - see https://github.com/numpy/numpy/pull/19800.
467          xp_eps = eps.get("array_api", [])
468      if len(xp_eps) == 0:
469          if numpy_in_sitepackages:
470              msg = "No entry points for 'array_api' found"
471              raise AssertionError(msg) from None
472          return
473  
474      try:
475          ep = next(ep for ep in xp_eps if ep.name == "numpy")
476      except StopIteration:
477          if numpy_in_sitepackages:
478              msg = "'numpy' not in array_api entry points"
479              raise AssertionError(msg) from None
480          return
481  
482      if ep.value == 'numpy.array_api':
483          # Looks like the entrypoint for the current numpy build isn't
484          # installed, but an older numpy is also installed and hence the
485          # entrypoint is pointing to the old (no longer existing) location.
486          # This isn't a problem except for when running tests with `spin` or an
487          # in-place build.
488          return
489  
490      xp = ep.load()
491      msg = (
492          f"numpy entry point value '{ep.value}' "
493          "does not point to our Array API implementation"
494      )
495      assert xp is numpy, msg

498  def test_main_namespace_all_dir_coherence():
499      """ 
500      Checks if `dir(np)` and `np.__all__` are consistent and return 
501      the same content, excluding exceptions and private members. 
502      """
503      def _remove_private_members(member_set):
504          return {m for m in member_set if not m.startswith('_')}
505  
506      def _remove_exceptions(member_set):
507          return member_set.difference({
508              "bool"  # included only in __dir__
509          })
510  
511      all_members = _remove_private_members(np.__all__)
512      all_members = _remove_exceptions(all_members)
513  
514      dir_members = _remove_private_members(np.__dir__())
515      dir_members = _remove_exceptions(dir_members)
516  
517      assert all_members == dir_members, (
518          "Members that break symmetry: "
519          f"{all_members.symmetric_difference(dir_members)}"
520      )

523  @pytest.mark.filterwarnings(
524      r"ignore:numpy.core(\.\w+)? is deprecated:DeprecationWarning"
525  )
526  def test_core_shims_coherence():
527      """ 
528      Check that all "semi-public" members of `numpy._core` are also accessible 
529      from `numpy.core` shims. 
530      """
531      import numpy.core as core
532  
533      for member_name in dir(np._core):
534          # Skip private and test members. Also if a module is aliased,
535          # no need to add it to np.core
536          if (
537              member_name.startswith("_")
538              or member_name in ["tests", "strings"]
539              or f"numpy.{member_name}" in PUBLIC_ALIASED_MODULES
540          ):
541              continue
542  
543          member = getattr(np._core, member_name)
544  
545          # np.core is a shim and all submodules of np.core are shims
546          # but we should be able to import everything in those shims
547          # that are available in the "real" modules in np._core
548          if inspect.ismodule(member):
549              submodule = member
550              submodule_name = member_name
551              for submodule_member_name in dir(submodule):
552                  # ignore dunder names
553                  if submodule_member_name.startswith("__"):
554                      continue
555                  submodule_member = getattr(submodule, submodule_member_name)
556  
557                  core_submodule = __import__(
558                      f"numpy.core.{submodule_name}",
559                      fromlist=[submodule_member_name]
560                  )
561  
562                  assert submodule_member is getattr(
563                      core_submodule, submodule_member_name
564                  )
565  
566          else:
567              assert member is getattr(core, member_name)

570  def test_functions_single_location():
571      """ 
572      Check that each public function is available from one location only. 
573   
574      Test performs BFS search traversing NumPy's public API. It flags 
575      any function-like object that is accessible from more that one place. 
576      """
577      from typing import Any, Callable, Dict, List, Set, Tuple
578      from numpy._core._multiarray_umath import (
579          _ArrayFunctionDispatcher as dispatched_function
580      )
581  
582      visited_modules: Set[types.ModuleType] = {np}
583      visited_functions: Set[Callable[..., Any]] = set()
584      # Functions often have `__name__` overridden, therefore we need
585      # to keep track of locations where functions have been found.
586      functions_original_paths: Dict[Callable[..., Any], str] = dict()
587  
588      # Here we aggregate functions with more than one location.
589      # It must be empty for the test to pass.
590      duplicated_functions: List[Tuple] = []
591  
592      modules_queue = [np]
593  
594      while len(modules_queue) > 0:
595  
596          module = modules_queue.pop()
597  
598          for member_name in dir(module):
599              member = getattr(module, member_name)
600  
601              # first check if we got a module
602              if (
603                  inspect.ismodule(member) and  # it's a module
604                  "numpy" in member.__name__ and  # inside NumPy
605                  not member_name.startswith("_") and  # not private
606                  "numpy._core" not in member.__name__ and  # outside _core
607                  # not a legacy or testing module
608                  member_name not in ["f2py", "ma", "testing", "tests"] and
609                  member not in visited_modules  # not visited yet
610              ):
611                  modules_queue.append(member)
612                  visited_modules.add(member)
613  
614              # else check if we got a function-like object
615              elif (
616                  inspect.isfunction(member) or
617                  isinstance(member, (dispatched_function, np.ufunc))
618              ):
619                  if member in visited_functions:
620  
621                      # skip main namespace functions with aliases
622                      if (
623                          member.__name__ in [
624                              "absolute",  # np.abs
625                              "arccos",  # np.acos
626                              "arccosh",  # np.acosh
627                              "arcsin",  # np.asin
628                              "arcsinh",  # np.asinh
629                              "arctan",  # np.atan
630                              "arctan2",  # np.atan2
631                              "arctanh",  # np.atanh
632                              "left_shift",  # np.bitwise_left_shift
633                              "right_shift",  # np.bitwise_right_shift
634                              "conjugate",  # np.conj
635                              "invert",  # np.bitwise_not & np.bitwise_invert
636                              "remainder",  # np.mod
637                              "divide",  # np.true_divide
638                              "concatenate",  # np.concat
639                              "power",  # np.pow
640                              "transpose",  # np.permute_dims
641                          ] and
642                          module.__name__ == "numpy"
643                      ):
644                          continue
645                      # skip trimcoef from numpy.polynomial as it is
646                      # duplicated by design.
647                      if (
648                          member.__name__ == "trimcoef" and
649                          module.__name__.startswith("numpy.polynomial")
650                      ):
651                          continue
652  
653                      # skip ufuncs that are exported in np.strings as well
654                      if member.__name__ in (
655                          "add",
656                          "equal",
657                          "not_equal",
658                          "greater",
659                          "greater_equal",
660                          "less",
661                          "less_equal",
662                      ) and module.__name__ == "numpy.strings":
663                          continue
664  
665                      # numpy.char reexports all numpy.strings functions for
666                      # backwards-compatibility
667                      if module.__name__ == "numpy.char":
668                          continue
669  
670                      # function is present in more than one location!
671                      duplicated_functions.append(
672                          (member.__name__,
673                           module.__name__,
674                           functions_original_paths[member])
675                      )
676                  else:
677                      visited_functions.add(member)
678                      functions_original_paths[member] = module.__name__
679  
680      del visited_functions, visited_modules, functions_original_paths
681  
682      assert len(duplicated_functions) == 0, duplicated_functions

689      incorrect_entries = []
690  
691      while len(modules_queue) > 0:
692          module = modules_queue.pop()
693          for member_name in dir(module):
694              member = getattr(module, member_name)
695              # first check if we got a module
696              if (
697                  inspect.ismodule(member) and  # it's a module
698                  "numpy" in member.__name__ and  # inside NumPy
699                  not member_name.startswith("_") and  # not private
700                  "numpy._core" not in member.__name__ and  # outside _core
701                  # not in a skip module list
702                  member_name not in [
703                      "char", "core", "ctypeslib", "f2py", "ma", "lapack_lite",
704                      "mrecords", "testing", "tests", "polynomial", "typing",
705                      "mtrand", "bit_generator",
706                  ] and
707                  member not in visited_modules  # not visited yet
708              ):
709                  modules_queue.append(member)
710                  visited_modules.add(member)
711              elif (
712                  not inspect.ismodule(member) and
713                  hasattr(member, "__name__") and
714                  not member.__name__.startswith("_") and
715                  member.__module__ != module.__name__ and
716                  member not in visited_functions
717              ):
718                  # skip ufuncs that are exported in np.strings as well
719                  if member.__name__ in (
720                      "add", "equal", "not_equal", "greater", "greater_equal",
721                      "less", "less_equal",
722                  ) and module.__name__ == "numpy.strings":
723                      continue
724  
725                  # recarray and record are exported in np and np.rec
726                  if (
727                      (member.__name__ == "recarray" and module.__name__ == "numpy") or
728                      (member.__name__ == "record" and module.__name__ == "numpy.rec")
729                  ):
730                      continue
731  
732                  # skip cdef classes
733                  if member.__name__ in (
734                      "BitGenerator", "Generator", "MT19937", "PCG64", "PCG64DXSM",
735                      "Philox", "RandomState", "SFC64", "SeedSequence",
736                  ):
737                      continue

739                  incorrect_entries.append(
740                      dict(
741                          Func=member.__name__,
742                          actual=member.__module__,
743                          expected=module.__name__,
744                      )
745                  )
746                  visited_functions.add(member)

749          assert len(incorrect_entries) == 0, incorrect_entries

752  def _check___qualname__(obj) -> bool:
753      qualname = obj.__qualname__
754      name = obj.__name__
755      module_name = obj.__module__
756      assert name == qualname.split(".")[-1]
757  
758      module = sys.modules[module_name]
759      actual_obj = functools.reduce(getattr, qualname.split("."), module)
760      return (
761          actual_obj is obj or
762          (
763              # for bound methods check qualname match
764              module_name.startswith("numpy.random") and
765              actual_obj.__qualname__ == qualname
766          )
767      )
