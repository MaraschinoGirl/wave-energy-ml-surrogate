14   def _compute_area(box):
15       """Computes area for bounding boxes 
16    
17       Args: 
18         box: [N, 4] or [batch_size, N, 4] float Tensor, either batched 
19           or unbatched boxes. 
20       Returns: 
21         a float Tensor of [N] or [batch_size, N] 
22       """
23       y_min, x_min, y_max, x_max = ops.split(box[..., :4], 4, axis=-1)
24       return ops.squeeze((y_max - y_min) * (x_max - x_min), axis=-1)

27   def _compute_intersection(boxes1, boxes2):
28       """Computes intersection area between two sets of boxes. 
29    
30       Args: 
31         boxes1: [N, 4] or [batch_size, N, 4] float Tensor boxes. 
32         boxes2: [M, 4] or [batch_size, M, 4] float Tensor boxes. 
33       Returns: 
34         a [N, M] or [batch_size, N, M] float Tensor. 
35       """
36       y_min1, x_min1, y_max1, x_max1 = ops.split(boxes1[..., :4], 4, axis=-1)
37       y_min2, x_min2, y_max2, x_max2 = ops.split(boxes2[..., :4], 4, axis=-1)
38       boxes2_rank = len(boxes2.shape)
39       perm = [1, 0] if boxes2_rank == 2 else [0, 2, 1]
40       # [N, M] or [batch_size, N, M]
41       intersect_ymax = ops.minimum(y_max1, ops.transpose(y_max2, perm))
42       intersect_ymin = ops.maximum(y_min1, ops.transpose(y_min2, perm))
43       intersect_xmax = ops.minimum(x_max1, ops.transpose(x_max2, perm))
44       intersect_xmin = ops.maximum(x_min1, ops.transpose(x_min2, perm))
45   
46       intersect_height = intersect_ymax - intersect_ymin
47       intersect_width = intersect_xmax - intersect_xmin
48       zeros_t = ops.cast(0, intersect_height.dtype)
49       intersect_height = ops.maximum(zeros_t, intersect_height)
50       intersect_width = ops.maximum(zeros_t, intersect_width)
51   
52       return intersect_height * intersect_width

55   @keras_export("keras.utils.bounding_boxes.compute_iou")
56   def compute_iou(
57       boxes1,
58       boxes2,
59       bounding_box_format,
60       use_masking=False,
61       mask_val=-1,
62       image_shape=None,
63   ):
64       """Computes a lookup table vector containing the ious for a given set boxes. 
65    
66       The lookup vector is to be indexed by [`boxes1_index`,`boxes2_index`] if 
67       boxes are unbatched and by [`batch`, `boxes1_index`,`boxes2_index`] if the 
68       boxes are batched. 
69    
70       The users can pass `boxes1` and `boxes2` to be different ranks. For example: 
71       1) `boxes1`: [batch_size, M, 4], `boxes2`: [batch_size, N, 4] -> return 
72           [batch_size, M, N]. 
73       2) `boxes1`: [batch_size, M, 4], `boxes2`: [N, 4] -> return 
74           [batch_size, M, N] 
75       3) `boxes1`: [M, 4], `boxes2`: [batch_size, N, 4] -> return 
76           [batch_size, M, N] 
77       4) `boxes1`: [M, 4], `boxes2`: [N, 4] -> return [M, N] 
78    
79       Args: 
80           boxes1: a list of bounding boxes in 'corners' format. Can be batched or 
81               unbatched. 
82           boxes2: a list of bounding boxes in 'corners' format. Can be batched or 
83               unbatched. 
84           bounding_box_format: a case-insensitive string which is one of `"xyxy"`, 
85               `"rel_xyxy"`, `"xyWH"`, `"center_xyWH"`, `"yxyx"`, `"rel_yxyx"`. 
86               For detailed information on the supported format, see the 
87           use_masking: whether masking will be applied. This will mask all 
88               `boxes1` or `boxes2` that have values less than 0 in all its 4 
89               dimensions. Default to `False`. 
90           mask_val: int to mask those returned IOUs if the masking is True, 
91               defaults to -1. 
92           image_shape: `Tuple[int]`. The shape of the image (height, width, 3). 
93               When using relative bounding box format for `box_format` the 
94               `image_shape` is used for normalization. 
95    
96       Returns: 
97           iou_lookup_table: a vector containing the pairwise ious of boxes1 and 
98               boxes2. 
99       """  # noqa: E501
100  
101      boxes1_rank = len(ops.shape(boxes1))
102      boxes2_rank = len(ops.shape(boxes2))
103  
104      if boxes1_rank not in [2, 3]:
105          raise ValueError(
106              "compute_iou() expects boxes1 to be batched, or to be unbatched. "
107              f"Received len(boxes1.shape)={boxes1_rank}, "
108              f"len(boxes2.shape)={boxes2_rank}. Expected either "
109              "len(boxes1.shape)=2 AND or len(boxes1.shape)=3."
110          )
111      if boxes2_rank not in [2, 3]:
112          raise ValueError(
113              "compute_iou() expects boxes2 to be batched, or to be unbatched. "
114              f"Received len(boxes1.shape)={boxes1_rank}, "
115              f"len(boxes2.shape)={boxes2_rank}. Expected either "
116              "len(boxes2.shape)=2 AND or len(boxes2.shape)=3."
117          )
118  
119      target_format = "yxyx"
120      if "rel" in bounding_box_format and image_shape is None:
121          raise ValueError(
122              "When using relative bounding box formats (e.g. `rel_yxyx`) "
123              "the `image_shape` argument must be provided."
124              f"Received `image_shape`: {image_shape}"
125          )
126  
127      if image_shape is None:
128          height, width = None, None
129      else:
130          height, width, _ = image_shape
131  
132      boxes1 = converters.convert_format(
133          boxes1,
134          source=bounding_box_format,
135          target=target_format,
136          height=height,
137          width=width,
138      )
139  
140      boxes2 = converters.convert_format(
141          boxes2,
142          source=bounding_box_format,
143          target=target_format,
144          height=height,
145          width=width,
146      )
147  
148      intersect_area = _compute_intersection(boxes1, boxes2)
149      boxes1_area = _compute_area(boxes1)
150      boxes2_area = _compute_area(boxes2)
151      boxes2_area_rank = len(boxes2_area.shape)
152      boxes2_axis = 1 if (boxes2_area_rank == 2) else 0
153      boxes1_area = ops.expand_dims(boxes1_area, axis=-1)
154      boxes2_area = ops.expand_dims(boxes2_area, axis=boxes2_axis)
155      union_area = boxes1_area + boxes2_area - intersect_area
156      res = ops.divide(intersect_area, union_area + backend.epsilon())
157  
158      if boxes1_rank == 2:
159          perm = [1, 0]
160      else:
161          perm = [0, 2, 1]
162  
163      if not use_masking:
164          return res
165  
166      mask_val_t = ops.cast(mask_val, res.dtype) * ops.ones_like(res)
167      boxes1_mask = ops.less(ops.max(boxes1, axis=-1, keepdims=True), 0.0)
168      boxes2_mask = ops.less(ops.max(boxes2, axis=-1, keepdims=True), 0.0)
169      background_mask = ops.logical_or(
170          boxes1_mask, ops.transpose(boxes2_mask, perm)
171      )
172      iou_lookup_table = ops.where(background_mask, mask_val_t, res)
173      return iou_lookup_table

176  @keras_export("keras.utils.bounding_boxes.compute_ciou")
177  def compute_ciou(boxes1, boxes2, bounding_box_format, image_shape=None):
178      """ 
179      Computes the Complete IoU (CIoU) between two bounding boxes or between 
180      two batches of bounding boxes. 
181   
182      CIoU loss is an extension of GIoU loss, which further improves the IoU 
183      optimization for object detection. CIoU loss not only penalizes the 
184      bounding box coordinates but also considers the aspect ratio and center 
185      distance of the boxes. The length of the last dimension should be 4 to 
186      represent the bounding boxes. 
187   
188      Args: 
189          box1 (tensor): tensor representing the first bounding box with 
190              shape (..., 4). 
191          box2 (tensor): tensor representing the second bounding box with 
192              shape (..., 4). 
193          bounding_box_format: a case-insensitive string (for example, "xyxy"). 
194              Each bounding box is defined by these 4 values. For detailed 
195              information on the supported formats, see the [KerasCV bounding box 
196              documentation](https://keras.io/api/keras_cv/bounding_box/formats/). 
197          image_shape: `Tuple[int]`. The shape of the image (height, width, 3). 
198              When using relative bounding box format for `box_format` the 
199              `image_shape` is used for normalization. 
200   
201      Returns: 
202          tensor: The CIoU distance between the two bounding boxes. 
203      """
204      target_format = "xyxy"
205      if "rel" in bounding_box_format:
206          raise ValueError(
207              "When using relative bounding box formats (e.g. `rel_yxyx`) "
208              "the `image_shape` argument must be provided."
209              f"Received `image_shape`: {image_shape}"
210          )
211  
212      if image_shape is None:
213          height, width = None, None
214      else:
215          height, width, _ = image_shape
216  
217      boxes1 = converters.convert_format(
218          boxes1,
219          source=bounding_box_format,
220          target=target_format,
221          height=height,
222          width=width,
223      )
224  
225      boxes2 = converters.convert_format(
226          boxes2,
227          source=bounding_box_format,
228          target=target_format,
229          height=height,
230          width=width,
231      )
232  
233      x_min1, y_min1, x_max1, y_max1 = ops.split(boxes1[..., :4], 4, axis=-1)
234      x_min2, y_min2, x_max2, y_max2 = ops.split(boxes2[..., :4], 4, axis=-1)
235  
236      width_1 = x_max1 - x_min1
237      height_1 = y_max1 - y_min1 + keras.backend.epsilon()
238      width_2 = x_max2 - x_min2
239      height_2 = y_max2 - y_min2 + keras.backend.epsilon()
240  
241      intersection_area = ops.maximum(
242          ops.minimum(x_max1, x_max2) - ops.maximum(x_min1, x_min2), 0
243      ) * ops.maximum(
244          ops.minimum(y_max1, y_max2) - ops.maximum(y_min1, y_min2), 0
245      )
246      union_area = (
247          width_1 * height_1
248          + width_2 * height_2
249          - intersection_area
250          + keras.backend.epsilon()
251      )
252      iou = ops.squeeze(
253          ops.divide(intersection_area, union_area + keras.backend.epsilon()),
254          axis=-1,
255      )
256  
257      convex_width = ops.maximum(x_max1, x_max2) - ops.minimum(x_min1, x_min2)
258      convex_height = ops.maximum(y_max1, y_max2) - ops.minimum(y_min1, y_min2)
259      convex_diagonal_squared = ops.squeeze(
260          convex_width**2 + convex_height**2 + keras.backend.epsilon(),
261          axis=-1,
262      )
263      centers_distance_squared = ops.squeeze(
264          ((x_min1 + x_max1) / 2 - (x_min2 + x_max2) / 2) ** 2
265          + ((y_min1 + y_max1) / 2 - (y_min2 + y_max2) / 2) ** 2,
266          axis=-1,
267      )
268  
269      v = ops.squeeze(
270          (4 / math.pi**2)
271          * ops.power(
272              (ops.arctan(width_2 / height_2) - ops.arctan(width_1 / height_1)),
273              2,
274          ),
275          axis=-1,
276      )
277      alpha = v / (v - iou + (1 + keras.backend.epsilon()))
278  
279      return iou - (
280          centers_distance_squared / convex_diagonal_squared + v * alpha
281      )
