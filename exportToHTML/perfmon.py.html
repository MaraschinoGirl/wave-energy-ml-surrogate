1    """A COM Server which exposes the NT Performance monitor in a very rudimentary way 
2     
3    Usage from VB: 
4        set ob = CreateObject("Python.PerfmonQuery") 
5        freeBytes = ob.Query("Memory", "Available Bytes") 
6    """
7    
8    import pythoncom
9    import win32pdhutil
10   import winerror
11   from win32com.server import register
12   from win32com.server.exception import COMException
13   
14   
15   class PerfMonQuery:
16       _reg_verprogid_ = "Python.PerfmonQuery.1"
17       _reg_progid_ = "Python.PerfmonQuery"
18       _reg_desc_ = "Python Performance Monitor query object"
19       _reg_clsid_ = "{64cef7a0-8ece-11d1-a65a-00aa00125a98}"
20       _reg_class_spec_ = "win32com.servers.perfmon.PerfMonQuery"
21       _public_methods_ = ["Query"]
22   
23       def Query(self, object, counter, instance=None, machine=None):
24           try:
25               return win32pdhutil.GetPerformanceAttributes(
26                   object, counter, instance, machine=machine
27               )
28           except win32pdhutil.error as exc:
29               raise COMException(desc=exc.strerror)
30           except TypeError as desc:
31               raise COMException(desc=desc, scode=winerror.DISP_E_TYPEMISMATCH)
32   
33   
34   if __name__ == "__main__":
35       print("Registering COM server...")
36       register.UseCommandLine(PerfMonQuery)
