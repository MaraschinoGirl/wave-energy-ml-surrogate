5    from libcpp.vector cimport vector
6    
7    from ..utils._typedefs cimport uint8_t, intp_t
8    
9    
10   def dbscan_inner(const uint8_t[::1] is_core,
11                    object[:] neighborhoods,
12                    intp_t[::1] labels):
13       cdef intp_t i, label_num = 0, v
14       cdef intp_t[:] neighb
15       cdef vector[intp_t] stack
16   
17       for i in range(labels.shape[0]):
18           if labels[i] != -1 or not is_core[i]:
19               continue
20   
21           # Depth-first search starting from i, ending at the non-core points.
22           # This is very similar to the classic algorithm for computing connected
23           # components, the difference being that we label non-core points as
24           # part of a cluster (component), but don't expand their neighborhoods.
25           while True:
26               if labels[i] == -1:
27                   labels[i] = label_num
28                   if is_core[i]:
29                       neighb = neighborhoods[i]
30                       for i in range(neighb.shape[0]):
31                           v = neighb[i]
32                           if labels[v] == -1:
33                               stack.push_back(v)
34   
35               if stack.size() == 0:
36                   break
37               i = stack.back()
38               stack.pop_back()
39   
40           label_num += 1
