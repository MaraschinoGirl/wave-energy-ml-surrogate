runproc.py 
1    """runproc.py 
2     
3    start a process with three inherited pipes. 
4    Try to write to and read from those. 
5    """
6    
7    import msvcrt
8    import os
9    
10   import win32api
11   import win32con
12   import win32file
13   import win32pipe
14   import win32process
15   import win32security
16   
17   
18   class Process:
19       def run(self, cmdline):
20           # security attributes for pipes
21           sAttrs = win32security.SECURITY_ATTRIBUTES()
22           sAttrs.bInheritHandle = 1
23   
24           # create pipes
25           hStdin_r, self.hStdin_w = win32pipe.CreatePipe(sAttrs, 0)
26           self.hStdout_r, hStdout_w = win32pipe.CreatePipe(sAttrs, 0)
27           self.hStderr_r, hStderr_w = win32pipe.CreatePipe(sAttrs, 0)
28   
29           # set the info structure for the new process.
30           StartupInfo = win32process.STARTUPINFO()
31           StartupInfo.hStdInput = hStdin_r
32           StartupInfo.hStdOutput = hStdout_w
33           StartupInfo.hStdError = hStderr_w
34           StartupInfo.dwFlags = win32process.STARTF_USESTDHANDLES
35           # Mark doesn't support wShowWindow yet.
36           # StartupInfo.dwFlags = StartupInfo.dwFlags | win32process.STARTF_USESHOWWINDOW
37           # StartupInfo.wShowWindow = win32con.SW_HIDE
38   
39           # Create new output read handles and the input write handle. Set
40           # the inheritance properties to FALSE. Otherwise, the child inherits
41           # the these handles; resulting in non-closeable handles to the pipes
42           # being created.
43           pid = win32api.GetCurrentProcess()
44   
45           tmp = win32api.DuplicateHandle(
46               pid,
47               self.hStdin_w,
48               pid,
49               0,
50               0,  # non-inheritable!!
51               win32con.DUPLICATE_SAME_ACCESS,
52           )
53           # Close the inhertible version of the handle
54           win32file.CloseHandle(self.hStdin_w)
55           self.hStdin_w = tmp
56           tmp = win32api.DuplicateHandle(
57               pid,
58               self.hStdout_r,
59               pid,
60               0,
61               0,  # non-inheritable!
62               win32con.DUPLICATE_SAME_ACCESS,
63           )
64           # Close the inhertible version of the handle
65           win32file.CloseHandle(self.hStdout_r)
66           self.hStdout_r = tmp
67   
68           # start the process.
69           hProcess, hThread, dwPid, dwTid = win32process.CreateProcess(
70               None,  # program
71               cmdline,  # command line
72               None,  # process security attributes
73               None,  # thread attributes
74               1,  # inherit handles, or USESTDHANDLES won't work.
75               # creation flags. Don't access the console.
76               0,  # Don't need anything here.
77               # If you're in a GUI app, you should use
78               # CREATE_NEW_CONSOLE here, or any subprocesses
79               # might fall victim to the problem described in:
80               # KB article: Q156755, cmd.exe requires
81               # an NT console in order to perform redirection..
82               None,  # no new environment
83               None,  # current directory (stay where we are)
84               StartupInfo,
85           )
86           # normally, we would save the pid etc. here...
87   
88           # Child is launched. Close the parents copy of those pipe handles
89           # that only the child should have open.
90           # You need to make sure that no handles to the write end of the
91           # output pipe are maintained in this process or else the pipe will
92           # not close when the child process exits and the ReadFile will hang.
93           win32file.CloseHandle(hStderr_w)
94           win32file.CloseHandle(hStdout_w)
95           win32file.CloseHandle(hStdin_r)
96   
97           self.stdin = os.fdopen(msvcrt.open_osfhandle(self.hStdin_w, 0), "wb")
98           self.stdin.write("hmmmmm\r\n")
99           self.stdin.flush()
100          self.stdin.close()
101  
102          self.stdout = os.fdopen(msvcrt.open_osfhandle(self.hStdout_r, 0), "rb")
103          print(f"Read on stdout: {self.stdout.read()!r}")
104  
105          self.stderr = os.fdopen(msvcrt.open_osfhandle(self.hStderr_r, 0), "rb")
106          print(f"Read on stderr: {self.stderr.read()!r}")
107  
108  
109  if __name__ == "__main__":
110      p = Process()
111      exe = win32api.GetModuleFileName(0)
112      p.run(exe + " cat.py")
113  
114  # end of runproc.py
