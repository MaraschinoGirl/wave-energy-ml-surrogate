90   def _brute_mst(mutual_reachability, min_samples):
91       """ 
92       Builds a minimum spanning tree (MST) from the provided mutual-reachability 
93       values. This function dispatches to a custom Cython implementation for 
94       dense arrays, and `scipy.sparse.csgraph.minimum_spanning_tree` for sparse 
95       arrays/matrices. 
96    
97       Parameters 
98       ---------- 
99       mututal_reachability_graph: {ndarray, sparse matrix} of shape \ 
100              (n_samples, n_samples) 
101          Weighted adjacency matrix of the mutual reachability graph. 
102   
103      min_samples : int, default=None 
104          The number of samples in a neighborhood for a point 
105          to be considered as a core point. This includes the point itself. 
106   
107      Returns 
108      ------- 
109      mst : ndarray of shape (n_samples - 1,), dtype=MST_edge_dtype 
110          The MST representation of the mutual-reachability graph. The MST is 
111          represented as a collection of edges. 
112      """
113      if not issparse(mutual_reachability):
114          return mst_from_mutual_reachability(mutual_reachability)
115  
116      # Check if the mutual reachability matrix has any rows which have
117      # less than `min_samples` non-zero elements.
118      indptr = mutual_reachability.indptr
119      num_points = mutual_reachability.shape[0]
120      if any((indptr[i + 1] - indptr[i]) < min_samples for i in range(num_points)):
121          raise ValueError(
122              f"There exists points with fewer than {min_samples} neighbors. Ensure"
123              " your distance matrix has non-zero values for at least"
124              f" `min_sample`={min_samples} neighbors for each points (i.e. K-nn"
125              " graph), or specify a `max_distance` in `metric_params` to use when"
126              " distances are missing."
127          )
128      # Check connected component on mutual reachability.
129      # If more than one connected component is present,
130      # it means that the graph is disconnected.
131      n_components = csgraph.connected_components(
132          mutual_reachability, directed=False, return_labels=False
133      )
134      if n_components > 1:
135          raise ValueError(
136              f"Sparse mutual reachability matrix has {n_components} connected"
137              " components. HDBSCAN cannot be perfomed on a disconnected graph. Ensure"
138              " that the sparse distance matrix has only one connected component."
139          )
140  
141      # Compute the minimum spanning tree for the sparse graph
142      sparse_min_spanning_tree = csgraph.minimum_spanning_tree(mutual_reachability)
143      rows, cols = sparse_min_spanning_tree.nonzero()
144      mst = np.rec.fromarrays(
145          [rows, cols, sparse_min_spanning_tree.data],
146          dtype=MST_edge_dtype,
147      )
148      return mst

151  def _process_mst(min_spanning_tree):
152      """ 
153      Builds a single-linkage tree (SLT) from the provided minimum spanning tree 
154      (MST). The MST is first sorted then processed by a custom Cython routine. 
155   
156      Parameters 
157      ---------- 
158      min_spanning_tree : ndarray of shape (n_samples - 1,), dtype=MST_edge_dtype 
159          The MST representation of the mutual-reachability graph. The MST is 
160          represented as a collection of edges. 
161   
162      Returns 
163      ------- 
164      single_linkage : ndarray of shape (n_samples - 1,), dtype=HIERARCHY_dtype 
165          The single-linkage tree tree (dendrogram) built from the MST. 
166      """
167      # Sort edges of the min_spanning_tree by weight
168      row_order = np.argsort(min_spanning_tree["distance"])
169      min_spanning_tree = min_spanning_tree[row_order]
170      # Convert edge list into standard hierarchical clustering format
171      return make_single_linkage(min_spanning_tree)

174  def _hdbscan_brute(
175      X,
176      min_samples=5,
177      alpha=None,
178      metric="euclidean",
179      n_jobs=None,
180      copy=False,
181      **metric_params,
182  ):
183      """ 
184      Builds a single-linkage tree (SLT) from the input data `X`. If 
185      `metric="precomputed"` then `X` must be a symmetric array of distances. 
186      Otherwise, the pairwise distances are calculated directly and passed to 
187      `mutual_reachability_graph`. 
188   
189      Parameters 
190      ---------- 
191      X : ndarray of shape (n_samples, n_features) or (n_samples, n_samples) 
192          Either the raw data from which to compute the pairwise distances, 
193          or the precomputed distances. 
194   
195      min_samples : int, default=None 
196          The number of samples in a neighborhood for a point 
197          to be considered as a core point. This includes the point itself. 
198   
199      alpha : float, default=1.0 
200          A distance scaling parameter as used in robust single linkage. 
201   
202      metric : str or callable, default='euclidean' 
203          The metric to use when calculating distance between instances in a 
204          feature array. 
205   
206          - If metric is a string or callable, it must be one of 
207            the options allowed by :func:`~sklearn.metrics.pairwise_distances` 
208            for its metric parameter. 
209   
210          - If metric is "precomputed", X is assumed to be a distance matrix and 
211            must be square. 
212   
213      n_jobs : int, default=None 
214          The number of jobs to use for computing the pairwise distances. This 
215          works by breaking down the pairwise matrix into n_jobs even slices and 
216          computing them in parallel. This parameter is passed directly to 
217          :func:`~sklearn.metrics.pairwise_distances`. 
218   
219          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
220          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
221          for more details. 
222   
223      copy : bool, default=False 
224          If `copy=True` then any time an in-place modifications would be made 
225          that would overwrite `X`, a copy will first be made, guaranteeing that 
226          the original data will be unchanged. Currently, it only applies when 
227          `metric="precomputed"`, when passing a dense array or a CSR sparse 
228          array/matrix. 
229   
230      metric_params : dict, default=None 
231          Arguments passed to the distance metric. 
232   
233      Returns 
234      ------- 
235      single_linkage : ndarray of shape (n_samples - 1,), dtype=HIERARCHY_dtype 
236          The single-linkage tree tree (dendrogram) built from the MST. 
237      """
238      if metric == "precomputed":
239          if X.shape[0] != X.shape[1]:
240              raise ValueError(
241                  "The precomputed distance matrix is expected to be symmetric, however"
242                  f" it has shape {X.shape}. Please verify that the"
243                  " distance matrix was constructed correctly."
244              )
245          if not _allclose_dense_sparse(X, X.T):
246              raise ValueError(
247                  "The precomputed distance matrix is expected to be symmetric, however"
248                  " its values appear to be asymmetric. Please verify that the distance"
249                  " matrix was constructed correctly."
250              )
251  
252          distance_matrix = X.copy() if copy else X
253      else:
254          distance_matrix = pairwise_distances(
255              X, metric=metric, n_jobs=n_jobs, **metric_params
256          )
257      distance_matrix /= alpha
258  
259      max_distance = metric_params.get("max_distance", 0.0)
260      if issparse(distance_matrix) and distance_matrix.format != "csr":
261          # we need CSR format to avoid a conversion in `_brute_mst` when calling
262          # `csgraph.connected_components`
263          distance_matrix = distance_matrix.tocsr()
264  
265      # Note that `distance_matrix` is manipulated in-place, however we do not
266      # need it for anything else past this point, hence the operation is safe.
267      mutual_reachability_ = mutual_reachability_graph(
268          distance_matrix, min_samples=min_samples, max_distance=max_distance
269      )
270      min_spanning_tree = _brute_mst(mutual_reachability_, min_samples=min_samples)
271      # Warn if the MST couldn't be constructed around the missing distances
272      if np.isinf(min_spanning_tree["distance"]).any():
273          warn(
274              (
275                  "The minimum spanning tree contains edge weights with value "
276                  "infinity. Potentially, you are missing too many distances "
277                  "in the initial distance matrix for the given neighborhood "
278                  "size."
279              ),
280              UserWarning,
281          )
282      return _process_mst(min_spanning_tree)

285  def _hdbscan_prims(
286      X,
287      algo,
288      min_samples=5,
289      alpha=1.0,
290      metric="euclidean",
291      leaf_size=40,
292      n_jobs=None,
293      **metric_params,
294  ):
295      """ 
296      Builds a single-linkage tree (SLT) from the input data `X`. If 
297      `metric="precomputed"` then `X` must be a symmetric array of distances. 
298      Otherwise, the pairwise distances are calculated directly and passed to 
299      `mutual_reachability_graph`. 
300   
301      Parameters 
302      ---------- 
303      X : ndarray of shape (n_samples, n_features) 
304          The raw data. 
305   
306      min_samples : int, default=None 
307          The number of samples in a neighborhood for a point 
308          to be considered as a core point. This includes the point itself. 
309   
310      alpha : float, default=1.0 
311          A distance scaling parameter as used in robust single linkage. 
312   
313      metric : str or callable, default='euclidean' 
314          The metric to use when calculating distance between instances in a 
315          feature array. `metric` must be one of the options allowed by 
316          :func:`~sklearn.metrics.pairwise_distances` for its metric 
317          parameter. 
318   
319      n_jobs : int, default=None 
320          The number of jobs to use for computing the pairwise distances. This 
321          works by breaking down the pairwise matrix into n_jobs even slices and 
322          computing them in parallel. This parameter is passed directly to 
323          :func:`~sklearn.metrics.pairwise_distances`. 
324   
325          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
326          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
327          for more details. 
328   
329      copy : bool, default=False 
330          If `copy=True` then any time an in-place modifications would be made 
331          that would overwrite `X`, a copy will first be made, guaranteeing that 
332          the original data will be unchanged. Currently, it only applies when 
333          `metric="precomputed"`, when passing a dense array or a CSR sparse 
334          array/matrix. 
335   
336      metric_params : dict, default=None 
337          Arguments passed to the distance metric. 
338   
339      Returns 
340      ------- 
341      single_linkage : ndarray of shape (n_samples - 1,), dtype=HIERARCHY_dtype 
342          The single-linkage tree tree (dendrogram) built from the MST. 
343      """
344      # The Cython routines used require contiguous arrays
345      X = np.asarray(X, order="C")
346  
347      # Get distance to kth nearest neighbour
348      nbrs = NearestNeighbors(
349          n_neighbors=min_samples,
350          algorithm=algo,
351          leaf_size=leaf_size,
352          metric=metric,
353          metric_params=metric_params,
354          n_jobs=n_jobs,
355          p=None,
356      ).fit(X)
357  
358      neighbors_distances, _ = nbrs.kneighbors(X, min_samples, return_distance=True)
359      core_distances = np.ascontiguousarray(neighbors_distances[:, -1])
360      dist_metric = DistanceMetric.get_metric(metric, **metric_params)
361  
362      # Mutual reachability distance is implicit in mst_from_data_matrix
363      min_spanning_tree = mst_from_data_matrix(X, core_distances, dist_metric, alpha)
364      return _process_mst(min_spanning_tree)

367  def remap_single_linkage_tree(tree, internal_to_raw, non_finite):
368      """ 
369      Takes an internal single_linkage_tree structure and adds back in a set of points 
370      that were initially detected as non-finite and returns that new tree. 
371      These points will all be merged into the final node at np.inf distance and 
372      considered noise points. 
373   
374      Parameters 
375      ---------- 
376      tree : ndarray of shape (n_samples - 1,), dtype=HIERARCHY_dtype 
377          The single-linkage tree tree (dendrogram) built from the MST. 
378      internal_to_raw: dict 
379          A mapping from internal integer index to the raw integer index 
380      non_finite : ndarray 
381          Boolean array of which entries in the raw data are non-finite 
382      """
383      finite_count = len(internal_to_raw)
384  
385      outlier_count = len(non_finite)
386      for i, _ in enumerate(tree):
387          left = tree[i]["left_node"]
388          right = tree[i]["right_node"]
389  
390          if left < finite_count:
391              tree[i]["left_node"] = internal_to_raw[left]
392          else:
393              tree[i]["left_node"] = left + outlier_count
394          if right < finite_count:
395              tree[i]["right_node"] = internal_to_raw[right]
396          else:
397              tree[i]["right_node"] = right + outlier_count
398  
399      outlier_tree = np.zeros(len(non_finite), dtype=HIERARCHY_dtype)
400      last_cluster_id = max(
401          tree[tree.shape[0] - 1]["left_node"], tree[tree.shape[0] - 1]["right_node"]
402      )
403      last_cluster_size = tree[tree.shape[0] - 1]["cluster_size"]
404      for i, outlier in enumerate(non_finite):
405          outlier_tree[i] = (outlier, last_cluster_id + 1, np.inf, last_cluster_size + 1)
406          last_cluster_id += 1
407          last_cluster_size += 1
408      tree = np.concatenate([tree, outlier_tree])
409      return tree

412  def _get_finite_row_indices(matrix):
413      """ 
414      Returns the indices of the purely finite rows of a 
415      sparse matrix or dense ndarray 
416      """
417      if issparse(matrix):
418          row_indices = np.array(
419              [i for i, row in enumerate(matrix.tolil().data) if np.all(np.isfinite(row))]
420          )
421      else:
422          (row_indices,) = np.isfinite(matrix.sum(axis=1)).nonzero()
423      return row_indices

426  class HDBSCAN(ClusterMixin, BaseEstimator):
427      """Cluster data using hierarchical density-based clustering. 
428   
429      HDBSCAN - Hierarchical Density-Based Spatial Clustering of Applications 
430      with Noise. Performs :class:`~sklearn.cluster.DBSCAN` over varying epsilon 
431      values and integrates the result to find a clustering that gives the best 
432      stability over epsilon. 
433      This allows HDBSCAN to find clusters of varying densities (unlike 
434      :class:`~sklearn.cluster.DBSCAN`), and be more robust to parameter selection. 
435      Read more in the :ref:`User Guide <hdbscan>`. 
436   
437      For an example of how to use HDBSCAN, as well as a comparison to 
438      :class:`~sklearn.cluster.DBSCAN`, please see the :ref:`plotting demo 
439      <sphx_glr_auto_examples_cluster_plot_hdbscan.py>`. 
440   
441      .. versionadded:: 1.3 
442   
443      Parameters 
444      ---------- 
445      min_cluster_size : int, default=5 
446          The minimum number of samples in a group for that group to be 
447          considered a cluster; groupings smaller than this size will be left 
448          as noise. 
449   
450      min_samples : int, default=None 
451          The parameter `k` used to calculate the distance between a point 
452          `x_p` and its k-th nearest neighbor. 
453          When `None`, defaults to `min_cluster_size`. 
454   
455      cluster_selection_epsilon : float, default=0.0 
456          A distance threshold. Clusters below this value will be merged. 
457          See [5]_ for more information. 
458   
459      max_cluster_size : int, default=None 
460          A limit to the size of clusters returned by the `"eom"` cluster 
461          selection algorithm. There is no limit when `max_cluster_size=None`. 
462          Has no effect if `cluster_selection_method="leaf"`. 
463   
464      metric : str or callable, default='euclidean' 
465          The metric to use when calculating distance between instances in a 
466          feature array. 
467   
468          - If metric is a string or callable, it must be one of 
469            the options allowed by :func:`~sklearn.metrics.pairwise_distances` 
470            for its metric parameter. 
471   
472          - If metric is "precomputed", X is assumed to be a distance matrix and 
473            must be square. 
474   
475      metric_params : dict, default=None 
476          Arguments passed to the distance metric. 
477   
478      alpha : float, default=1.0 
479          A distance scaling parameter as used in robust single linkage. 
480          See [3]_ for more information. 
481   
482      algorithm : {"auto", "brute", "kd_tree", "ball_tree"}, default="auto" 
483          Exactly which algorithm to use for computing core distances; By default 
484          this is set to `"auto"` which attempts to use a 
485          :class:`~sklearn.neighbors.KDTree` tree if possible, otherwise it uses 
486          a :class:`~sklearn.neighbors.BallTree` tree. Both `"kd_tree"` and 
487          `"ball_tree"` algorithms use the 
488          :class:`~sklearn.neighbors.NearestNeighbors` estimator. 
489   
490          If the `X` passed during `fit` is sparse or `metric` is invalid for 
491          both :class:`~sklearn.neighbors.KDTree` and 
492          :class:`~sklearn.neighbors.BallTree`, then it resolves to use the 
493          `"brute"` algorithm. 
494   
495      leaf_size : int, default=40 
496          Leaf size for trees responsible for fast nearest neighbour queries when 
497          a KDTree or a BallTree are used as core-distance algorithms. A large 
498          dataset size and small `leaf_size` may induce excessive memory usage. 
499          If you are running out of memory consider increasing the `leaf_size` 
500          parameter. Ignored for `algorithm="brute"`. 
501   
502      n_jobs : int, default=None 
503          Number of jobs to run in parallel to calculate distances. 
504          `None` means 1 unless in a :obj:`joblib.parallel_backend` context. 
505          `-1` means using all processors. See :term:`Glossary <n_jobs>` 
506          for more details. 
507   
508      cluster_selection_method : {"eom", "leaf"}, default="eom" 
509          The method used to select clusters from the condensed tree. The 
510          standard approach for HDBSCAN* is to use an Excess of Mass (`"eom"`) 
511          algorithm to find the most persistent clusters. Alternatively you can 
512          instead select the clusters at the leaves of the tree -- this provides 
513          the most fine grained and homogeneous clusters. 
514   
515      allow_single_cluster : bool, default=False 
516          By default HDBSCAN* will not produce a single cluster, setting this 
517          to True will override this and allow single cluster results in 
518          the case that you feel this is a valid result for your dataset. 
519   
520      store_centers : str, default=None 
521          Which, if any, cluster centers to compute and store. The options are: 
522   
523          - `None` which does not compute nor store any centers. 
524          - `"centroid"` which calculates the center by taking the weighted 
525            average of their positions. Note that the algorithm uses the 
526            euclidean metric and does not guarantee that the output will be 
527            an observed data point. 
528          - `"medoid"` which calculates the center by taking the point in the 
529            fitted data which minimizes the distance to all other points in 
530            the cluster. This is slower than "centroid" since it requires 
531            computing additional pairwise distances between points of the 
532            same cluster but guarantees the output is an observed data point. 
533            The medoid is also well-defined for arbitrary metrics, and does not 
534            depend on a euclidean metric. 
535          - `"both"` which computes and stores both forms of centers. 
536   
537      copy : bool, default=False 
538          If `copy=True` then any time an in-place modifications would be made 
539          that would overwrite data passed to :term:`fit`, a copy will first be 
540          made, guaranteeing that the original data will be unchanged. 
541          Currently, it only applies when `metric="precomputed"`, when passing 
542          a dense array or a CSR sparse matrix and when `algorithm="brute"`. 
543   
544      Attributes 
545      ---------- 
546      labels_ : ndarray of shape (n_samples,) 
547          Cluster labels for each point in the dataset given to :term:`fit`. 
548          Outliers are labeled as follows: 
549   
550          - Noisy samples are given the label -1. 
551          - Samples with infinite elements (+/- np.inf) are given the label -2. 
552          - Samples with missing data are given the label -3, even if they 
553            also have infinite elements. 
554   
555      probabilities_ : ndarray of shape (n_samples,) 
556          The strength with which each sample is a member of its assigned 
557          cluster. 
558   
559          - Clustered samples have probabilities proportional to the degree that 
560            they persist as part of the cluster. 
561          - Noisy samples have probability zero. 
562          - Samples with infinite elements (+/- np.inf) have probability 0. 
563          - Samples with missing data have probability `np.nan`. 
564   
565      n_features_in_ : int 
566          Number of features seen during :term:`fit`. 
567   
568      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
569          Names of features seen during :term:`fit`. Defined only when `X` 
570          has feature names that are all strings. 
571   
572      centroids_ : ndarray of shape (n_clusters, n_features) 
573          A collection containing the centroid of each cluster calculated under 
574          the standard euclidean metric. The centroids may fall "outside" their 
575          respective clusters if the clusters themselves are non-convex. 
576   
577          Note that `n_clusters` only counts non-outlier clusters. That is to 
578          say, the `-1, -2, -3` labels for the outlier clusters are excluded. 
579   
580      medoids_ : ndarray of shape (n_clusters, n_features) 
581          A collection containing the medoid of each cluster calculated under 
582          the whichever metric was passed to the `metric` parameter. The 
583          medoids are points in the original cluster which minimize the average 
584          distance to all other points in that cluster under the chosen metric. 
585          These can be thought of as the result of projecting the `metric`-based 
586          centroid back onto the cluster. 
587   
588          Note that `n_clusters` only counts non-outlier clusters. That is to 
589          say, the `-1, -2, -3` labels for the outlier clusters are excluded. 
590   
591      See Also 
592      -------- 
593      DBSCAN : Density-Based Spatial Clustering of Applications 
594          with Noise. 
595      OPTICS : Ordering Points To Identify the Clustering Structure. 
596      Birch : Memory-efficient, online-learning algorithm. 
597   
598      Notes 
599      ----- 
600      The `min_samples` parameter includes the point itself, whereas the implementation in 
601      `scikit-learn-contrib/hdbscan <https://github.com/scikit-learn-contrib/hdbscan>`_ 
602      does not. To get the same results in both versions, the value of `min_samples` here 
603      must be 1 greater than the value used in `scikit-learn-contrib/hdbscan

700      @_fit_context(
701          # HDBSCAN.metric is not validated yet
702          prefer_skip_nested_validation=False
703      )
704      def fit(self, X, y=None):
705          """Find clusters based on hierarchical density-based clustering. 
706   
707          Parameters 
708          ---------- 
709          X : {array-like, sparse matrix} of shape (n_samples, n_features), or \ 
710                  ndarray of shape (n_samples, n_samples) 
711              A feature array, or array of distances between samples if 
712              `metric='precomputed'`. 
713   
714          y : None 
715              Ignored. 
716   
717          Returns 
718          ------- 
719          self : object 
720              Returns self. 
721          """
722          if self.metric == "precomputed" and self.store_centers is not None:
723              raise ValueError(
724                  "Cannot store centers when using a precomputed distance matrix."
725              )
726  
727          self._metric_params = self.metric_params or {}
728          if self.metric != "precomputed":
729              # Non-precomputed matrices may contain non-finite values.
730              X = validate_data(
731                  self,
732                  X,
733                  accept_sparse=["csr", "lil"],
734                  ensure_all_finite=False,
735                  dtype=np.float64,
736              )
737              self._raw_data = X
738              all_finite = True
739              try:
740                  _assert_all_finite(X.data if issparse(X) else X)
741              except ValueError:
742                  all_finite = False
743  
744              if not all_finite:
745                  # Pass only the purely finite indices into hdbscan
746                  # We will later assign all non-finite points their
747                  # corresponding labels, as specified in `_OUTLIER_ENCODING`
748  
749                  # Reduce X to make the checks for missing/outlier samples more
750                  # convenient.
751                  reduced_X = X.sum(axis=1)
752  
753                  # Samples with missing data are denoted by the presence of
754                  # `np.nan`
755                  missing_index = np.isnan(reduced_X).nonzero()[0]
756  
757                  # Outlier samples are denoted by the presence of `np.inf`
758                  infinite_index = np.isinf(reduced_X).nonzero()[0]
759  
760                  # Continue with only finite samples
761                  finite_index = _get_finite_row_indices(X)
762                  internal_to_raw = {x: y for x, y in enumerate(finite_index)}
763                  X = X[finite_index]
764          elif issparse(X):
765              # Handle sparse precomputed distance matrices separately
766              X = validate_data(
767                  self,
768                  X,
769                  accept_sparse=["csr", "lil"],
770                  dtype=np.float64,
771                  force_writeable=True,
772              )
773          else:
774              # Only non-sparse, precomputed distance matrices are handled here
775              # and thereby allowed to contain numpy.inf for missing distances
776  
777              # Perform data validation after removing infinite values (numpy.inf)
778              # from the given distance matrix.
779              X = validate_data(
780                  self, X, ensure_all_finite=False, dtype=np.float64, force_writeable=True
781              )
782              if np.isnan(X).any():
783                  # TODO: Support np.nan in Cython implementation for precomputed
784                  # dense HDBSCAN
785                  raise ValueError("np.nan values found in precomputed-dense")
786          if X.shape[0] == 1:
787              raise ValueError("n_samples=1 while HDBSCAN requires more than one sample")
788          self._min_samples = (
789              self.min_cluster_size if self.min_samples is None else self.min_samples
790          )
791  
792          if self._min_samples > X.shape[0]:
793              raise ValueError(
794                  f"min_samples ({self._min_samples}) must be at most the number of"
795                  f" samples in X ({X.shape[0]})"
796              )
797  
798          mst_func = None
799          kwargs = dict(
800              X=X,
801              min_samples=self._min_samples,
802              alpha=self.alpha,
803              metric=self.metric,
804              n_jobs=self.n_jobs,
805              **self._metric_params,
806          )
807          if self.algorithm == "kd_tree" and self.metric not in KDTree.valid_metrics:
808              raise ValueError(
809                  f"{self.metric} is not a valid metric for a KDTree-based algorithm."
810                  " Please select a different metric."
811              )
812          elif (
813              self.algorithm == "ball_tree" and self.metric not in BallTree.valid_metrics
814          ):
815              raise ValueError(
816                  f"{self.metric} is not a valid metric for a BallTree-based algorithm."
817                  " Please select a different metric."
818              )
819  
820          if self.algorithm != "auto":
821              if (
822                  self.metric != "precomputed"
823                  and issparse(X)
824                  and self.algorithm != "brute"
825              ):
826                  raise ValueError("Sparse data matrices only support algorithm `brute`.")
827  
828              if self.algorithm == "brute":
829                  mst_func = _hdbscan_brute
830                  kwargs["copy"] = self.copy
831              elif self.algorithm == "kd_tree":
832                  mst_func = _hdbscan_prims
833                  kwargs["algo"] = "kd_tree"
834                  kwargs["leaf_size"] = self.leaf_size
835              else:
836                  mst_func = _hdbscan_prims
837                  kwargs["algo"] = "ball_tree"
838                  kwargs["leaf_size"] = self.leaf_size
839          else:
840              if issparse(X) or self.metric not in FAST_METRICS:
841                  # We can't do much with sparse matrices ...
842                  mst_func = _hdbscan_brute
843                  kwargs["copy"] = self.copy
844              elif self.metric in KDTree.valid_metrics:
845                  # TODO: Benchmark KD vs Ball Tree efficiency
846                  mst_func = _hdbscan_prims
847                  kwargs["algo"] = "kd_tree"
848                  kwargs["leaf_size"] = self.leaf_size
849              else:
850                  # Metric is a valid BallTree metric
851                  mst_func = _hdbscan_prims
852                  kwargs["algo"] = "ball_tree"
853                  kwargs["leaf_size"] = self.leaf_size
854  
855          self._single_linkage_tree_ = mst_func(**kwargs)
856  
857          self.labels_, self.probabilities_ = tree_to_labels(
858              self._single_linkage_tree_,
859              self.min_cluster_size,
860              self.cluster_selection_method,
861              self.allow_single_cluster,
862              self.cluster_selection_epsilon,
863              self.max_cluster_size,
864          )
865          if self.metric != "precomputed" and not all_finite:
866              # Remap indices to align with original data in the case of
867              # non-finite entries. Samples with np.inf are mapped to -1 and
868              # those with np.nan are mapped to -2.
869              self._single_linkage_tree_ = remap_single_linkage_tree(
870                  self._single_linkage_tree_,
871                  internal_to_raw,
872                  # There may be overlap for points w/ both `np.inf` and `np.nan`
873                  non_finite=set(np.hstack([infinite_index, missing_index])),
874              )
875              new_labels = np.empty(self._raw_data.shape[0], dtype=np.int32)
876              new_labels[finite_index] = self.labels_
877              new_labels[infinite_index] = _OUTLIER_ENCODING["infinite"]["label"]
878              new_labels[missing_index] = _OUTLIER_ENCODING["missing"]["label"]
879              self.labels_ = new_labels
880  
881              new_probabilities = np.zeros(self._raw_data.shape[0], dtype=np.float64)
882              new_probabilities[finite_index] = self.probabilities_
883              # Infinite outliers have probability 0 by convention, though this
884              # is arbitrary.
885              new_probabilities[infinite_index] = _OUTLIER_ENCODING["infinite"]["prob"]
886              new_probabilities[missing_index] = _OUTLIER_ENCODING["missing"]["prob"]
887              self.probabilities_ = new_probabilities
888  
889          if self.store_centers:
890              self._weighted_cluster_center(X)
891          return self

893      def fit_predict(self, X, y=None):
894          """Cluster X and return the associated cluster labels. 
895   
896          Parameters 
897          ---------- 
898          X : {array-like, sparse matrix} of shape (n_samples, n_features), or \ 
899                  ndarray of shape (n_samples, n_samples) 
900              A feature array, or array of distances between samples if 
901              `metric='precomputed'`. 
902   
903          y : None 
904              Ignored. 
905   
906          Returns 
907          ------- 
908          y : ndarray of shape (n_samples,) 
909              Cluster labels. 
910          """
911          self.fit(X)
912          return self.labels_

914      def _weighted_cluster_center(self, X):
915          """Calculate and store the centroids/medoids of each cluster. 
916   
917          This requires `X` to be a raw feature array, not precomputed 
918          distances. Rather than return outputs directly, this helper method 
919          instead stores them in the `self.{centroids, medoids}_` attributes. 
920          The choice for which attributes are calculated and stored is mediated 
921          by the value of `self.store_centers`. 
922   
923          Parameters 
924          ---------- 
925          X : ndarray of shape (n_samples, n_features) 
926              The feature array that the estimator was fit with. 
927   
928          """
929          # Number of non-noise clusters
930          n_clusters = len(set(self.labels_) - {-1, -2})
931          mask = np.empty((X.shape[0],), dtype=np.bool_)
932          make_centroids = self.store_centers in ("centroid", "both")
933          make_medoids = self.store_centers in ("medoid", "both")
934  
935          if make_centroids:
936              self.centroids_ = np.empty((n_clusters, X.shape[1]), dtype=np.float64)
937          if make_medoids:
938              self.medoids_ = np.empty((n_clusters, X.shape[1]), dtype=np.float64)
939  
940          # Need to handle iteratively seen each cluster may have a different
941          # number of samples, hence we can't create a homogeneous 3D array.
942          for idx in range(n_clusters):
943              mask = self.labels_ == idx
944              data = X[mask]
945              strength = self.probabilities_[mask]
946              if make_centroids:
947                  self.centroids_[idx] = np.average(data, weights=strength, axis=0)
948              if make_medoids:
949                  # TODO: Implement weighted argmin PWD backend
950                  dist_mat = pairwise_distances(
951                      data, metric=self.metric, **self._metric_params
952                  )
953                  dist_mat = dist_mat * strength
954                  medoid_index = np.argmin(dist_mat.sum(axis=1))
955                  self.medoids_[idx] = data[medoid_index]
956          return

958      def dbscan_clustering(self, cut_distance, min_cluster_size=5):
959          """Return clustering given by DBSCAN without border points. 
960   
961          Return clustering that would be equivalent to running DBSCAN* for a 
962          particular cut_distance (or epsilon) DBSCAN* can be thought of as 
963          DBSCAN without the border points.  As such these results may differ 
964          slightly from `cluster.DBSCAN` due to the difference in implementation 
965          over the non-core points. 
966   
967          This can also be thought of as a flat clustering derived from constant 
968          height cut through the single linkage tree. 
969   
970          This represents the result of selecting a cut value for robust single linkage 
971          clustering. The `min_cluster_size` allows the flat clustering to declare noise 
972          points (and cluster smaller than `min_cluster_size`). 
973   
974          Parameters 
975          ---------- 
976          cut_distance : float 
977              The mutual reachability distance cut value to use to generate a 
978              flat clustering. 
979   
980          min_cluster_size : int, default=5 
981              Clusters smaller than this value with be called 'noise' and remain 
982              unclustered in the resulting flat clustering. 
983   
984          Returns 
985          ------- 
986          labels : ndarray of shape (n_samples,) 
987              An array of cluster labels, one per datapoint. 
988              Outliers are labeled as follows: 
989   
990              - Noisy samples are given the label -1. 
991              - Samples with infinite elements (+/- np.inf) are given the label -2. 
992              - Samples with missing data are given the label -3, even if they 
993                also have infinite elements. 
994          """
995          labels = labelling_at_cut(
996              self._single_linkage_tree_, cut_distance, min_cluster_size
997          )
998          # Infer indices from labels generated during `fit`
999          infinite_index = self.labels_ == _OUTLIER_ENCODING["infinite"]["label"]
1000         missing_index = self.labels_ == _OUTLIER_ENCODING["missing"]["label"]
1001 
1002         # Overwrite infinite/missing outlier samples (otherwise simple noise)
1003         labels[infinite_index] = _OUTLIER_ENCODING["infinite"]["label"]
1004         labels[missing_index] = _OUTLIER_ENCODING["missing"]["label"]
1005         return labels
