13   # GLUT is quite an old library and it is difficult to ensure proper
14   # integration within IPython since original GLUT does not allow to handle
15   # events one by one. Instead, it requires for the mainloop to be entered
16   # and never returned (there is not even a function to exit he
17   # mainloop). Fortunately, there are alternatives such as freeglut
18   # (available for linux and windows) and the OSX implementation gives
19   # access to a glutCheckLoop() function that blocks itself until a new
20   # event is received. This means we have to setup the idle callback to
21   # ensure we got at least one event that will unblock the function.
22   #
23   # Furthermore, it is not possible to install these handlers without a window
24   # being first created. We choose to make this window invisible. This means that
25   # display mode options are set at this level and user won't be able to change
26   # them later without modifying the code. This should probably be made available
27   # via IPython options system.

103  def inputhook_glut():
104      """Run the pyglet event loop by processing pending events only. 
105   
106      This keeps processing pending events until stdin is ready.  After 
107      processing all pending events, a call to time.sleep is inserted.  This is 
108      needed, otherwise, CPU usage is at 100%.  This sleep time should be tuned 
109      though for best performance. 
110      """
111      # We need to protect against a user pressing Control-C when IPython is
112      # idle and this is running. We trap KeyboardInterrupt and pass.
113  
114      signal.signal(signal.SIGINT, glut_int_handler)
115  
116      try:
117          t = clock()
118  
119          # Make sure the default window is set after a window has been closed
120          if glut.glutGetWindow() == 0:
121              glut.glutSetWindow(1)
122              glutMainLoopEvent()
123              return 0
124  
125          while not stdin_ready():
126              glutMainLoopEvent()
127              # We need to sleep at this point to keep the idle CPU load
128              # low.  However, if sleep to long, GUI response is poor.  As
129              # a compromise, we watch how often GUI events are being processed
130              # and switch between a short and long sleep time.  Here are some
131              # stats useful in helping to tune this.
132              # time    CPU load
133              # 0.001   13%
134              # 0.005   3%
135              # 0.01    1.5%
136              # 0.05    0.5%
137              used_time = clock() - t
138              if used_time > 10.0:
139                  # print 'Sleep for 1 s'  # dbg
140                  time.sleep(1.0)
141              elif used_time > 0.1:
142                  # Few GUI events coming in, so we can sleep longer
143                  # print 'Sleep for 0.05 s'  # dbg
144                  time.sleep(0.05)
145              else:
146                  # Many GUI events coming in, so sleep only very little
147                  time.sleep(0.001)
148      except KeyboardInterrupt:
149          pass
150      return 0
