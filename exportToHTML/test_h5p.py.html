53   class TestDA(TestCase):
54       ''' 
55       Feature: setting/getting chunk cache size on a dataset access property list 
56       '''
57       def test_chunk_cache(self):
58           '''test get/set chunk cache '''
59           dalist = h5p.create(h5p.DATASET_ACCESS)
60           nslots = 10000  # 40kb hash table
61           nbytes = 1000000  # 1MB cache size
62           w0 = .5  # even blend of eviction strategy
63   
64           dalist.set_chunk_cache(nslots, nbytes, w0)
65           self.assertEqual((nslots, nbytes, w0),
66                            dalist.get_chunk_cache())
67   
68       def test_efile_prefix(self):
69           '''test get/set efile prefix '''
70           dalist = h5p.create(h5p.DATASET_ACCESS)
71           self.assertEqual(dalist.get_efile_prefix().decode(), '')
72   
73           efile_prefix = "path/to/external/dataset"
74           dalist.set_efile_prefix(efile_prefix.encode('utf-8'))
75           self.assertEqual(dalist.get_efile_prefix().decode(),
76                            efile_prefix)
77   
78           efile_prefix = "${ORIGIN}"
79           dalist.set_efile_prefix(efile_prefix.encode('utf-8'))
80           self.assertEqual(dalist.get_efile_prefix().decode(),
81                            efile_prefix)

105      def test_set_alignment(self):
106          '''test get/set chunk cache '''
107          falist = h5p.create(h5p.FILE_ACCESS)
108          threshold = 10 * 1024  # threshold of 10kiB
109          alignment = 1024 * 1024  # threshold of 1kiB
110  
111          falist.set_alignment(threshold, alignment)
112          self.assertEqual((threshold, alignment),
113                           falist.get_alignment())

130  class TestPL(TestCase):
131      def test_obj_track_times(self):
132          """ 
133          tests if the object track times  set/get 
134          """
135          # test for groups
136          gcid = h5p.create(h5p.GROUP_CREATE)
137          gcid.set_obj_track_times(False)
138          self.assertEqual(False, gcid.get_obj_track_times())
139  
140          gcid.set_obj_track_times(True)
141          self.assertEqual(True, gcid.get_obj_track_times())
142          # test for datasets
143          dcid = h5p.create(h5p.DATASET_CREATE)
144          dcid.set_obj_track_times(False)
145          self.assertEqual(False, dcid.get_obj_track_times())
146  
147          dcid.set_obj_track_times(True)
148          self.assertEqual(True, dcid.get_obj_track_times())
149  
150          # test for generic objects
151          ocid = h5p.create(h5p.OBJECT_CREATE)
152          ocid.set_obj_track_times(False)
153          self.assertEqual(False, ocid.get_obj_track_times())
154  
155          ocid.set_obj_track_times(True)
156          self.assertEqual(True, ocid.get_obj_track_times())

158      def test_link_creation_tracking(self):
159          """ 
160          tests the link creation order set/get 
161          """
162  
163          gcid = h5p.create(h5p.GROUP_CREATE)
164          gcid.set_link_creation_order(0)
165          self.assertEqual(0, gcid.get_link_creation_order())
166  
167          flags = h5p.CRT_ORDER_TRACKED | h5p.CRT_ORDER_INDEXED
168          gcid.set_link_creation_order(flags)
169          self.assertEqual(flags, gcid.get_link_creation_order())
170  
171          # test for file creation
172          fcpl = h5p.create(h5p.FILE_CREATE)
173          fcpl.set_link_creation_order(flags)
174          self.assertEqual(flags, fcpl.get_link_creation_order())

176      def test_attr_phase_change(self):
177          """ 
178          test the attribute phase change 
179          """
180  
181          cid = h5p.create(h5p.OBJECT_CREATE)
182          # test default value
183          ret = cid.get_attr_phase_change()
184          self.assertEqual((8,6), ret)
185  
186          # max_compact must < 65536 (64kb)
187          with self.assertRaises(ValueError):
188              cid.set_attr_phase_change(65536, 6)
189  
190          # Using dense attributes storage to avoid 64kb size limitation
191          # for a single attribute in compact attribute storage.
192          cid.set_attr_phase_change(0, 0)
193          self.assertEqual((0,0), cid.get_attr_phase_change())
