151      # This code is mostly taken from scipy 0.14 and extended to handle nans, see
152      # https://github.com/scikit-learn/scikit-learn/pull/11196
153      def _minor_reduce(X, ufunc):
154          major_index = np.flatnonzero(np.diff(X.indptr))
155  
156          # reduceat tries casts X.indptr to intp, which errors
157          # if it is int64 on a 32 bit system.
158          # Reinitializing prevents this where possible, see #13737
159          X = type(X)((X.data, X.indices, X.indptr), shape=X.shape)
160          value = ufunc.reduceat(X.data, X.indptr[major_index])
161          return major_index, value
162  
163      def _min_or_max_axis(X, axis, min_or_max):
164          N = X.shape[axis]
165          if N == 0:
166              raise ValueError("zero-size array to reduction operation")
167          M = X.shape[1 - axis]
168          mat = X.tocsc() if axis == 0 else X.tocsr()
169          mat.sum_duplicates()
170          major_index, value = _minor_reduce(mat, min_or_max)
171          not_full = np.diff(mat.indptr)[major_index] < N
172          value[not_full] = min_or_max(value[not_full], 0)
173          mask = value != 0
174          major_index = np.compress(mask, major_index)
175          value = np.compress(mask, value)
176  
177          if axis == 0:
178              res = scipy.sparse.coo_matrix(
179                  (value, (np.zeros(len(value)), major_index)),
180                  dtype=X.dtype,
181                  shape=(1, M),
182              )
183          else:
184              res = scipy.sparse.coo_matrix(
185                  (value, (major_index, np.zeros(len(value)))),
186                  dtype=X.dtype,
187                  shape=(M, 1),
188              )
189          return res.A.ravel()
190  
191      def _sparse_min_or_max(X, axis, min_or_max):
192          if axis is None:
193              if 0 in X.shape:
194                  raise ValueError("zero-size array to reduction operation")
195              zero = X.dtype.type(0)
196              if X.nnz == 0:
197                  return zero
198              m = min_or_max.reduce(X.data.ravel())
199              if X.nnz != np.prod(X.shape):
200                  m = min_or_max(zero, m)
201              return m
202          if axis < 0:
203              axis += 2
204          if (axis == 0) or (axis == 1):
205              return _min_or_max_axis(X, axis, min_or_max)
206          else:
207              raise ValueError("invalid axis, use 0 for rows, or 1 for columns")
208  
209      def _sparse_min_max(X, axis):
210          return (
211              _sparse_min_or_max(X, axis, np.minimum),
212              _sparse_min_or_max(X, axis, np.maximum),
213          )
214  
215      def _sparse_nan_min_max(X, axis):
216          return (
217              _sparse_min_or_max(X, axis, np.fmin),
218              _sparse_min_or_max(X, axis, np.fmax),
219          )

245  def _preserve_dia_indices_dtype(
246      sparse_container, original_container_format, requested_sparse_format
247  ):
248      """Preserve indices dtype for SciPy < 1.12 when converting from DIA to CSR/CSC. 
249   
250      For SciPy < 1.12, DIA arrays indices are upcasted to `np.int64` that is 
251      inconsistent with DIA matrices. We downcast the indices dtype to `np.int32` to 
252      be consistent with DIA matrices. 
253   
254      The converted indices arrays are affected back inplace to the sparse container. 
255   
256      Parameters 
257      ---------- 
258      sparse_container : sparse container 
259          Sparse container to be checked. 
260      requested_sparse_format : str or bool 
261          The type of format of `sparse_container`. 
262   
263      Notes 
264      ----- 
265      See https://github.com/scipy/scipy/issues/19245 for more details. 
266      """
267      if original_container_format == "dia_array" and requested_sparse_format in (
268          "csr",
269          "coo",
270      ):
271          if requested_sparse_format == "csr":
272              index_dtype = _smallest_admissible_index_dtype(
273                  arrays=(sparse_container.indptr, sparse_container.indices),
274                  maxval=max(sparse_container.nnz, sparse_container.shape[1]),
275                  check_contents=True,
276              )
277              sparse_container.indices = sparse_container.indices.astype(
278                  index_dtype, copy=False
279              )
280              sparse_container.indptr = sparse_container.indptr.astype(
281                  index_dtype, copy=False
282              )
283          else:  # requested_sparse_format == "coo"
284              index_dtype = _smallest_admissible_index_dtype(
285                  maxval=max(sparse_container.shape)
286              )
287              sparse_container.row = sparse_container.row.astype(index_dtype, copy=False)
288              sparse_container.col = sparse_container.col.astype(index_dtype, copy=False)

292  def _smallest_admissible_index_dtype(arrays=(), maxval=None, check_contents=False):
293      """Based on input (integer) arrays `a`, determine a suitable index data 
294      type that can hold the data in the arrays. 
295   
296      This function returns `np.int64` if it either required by `maxval` or based on the 
297      largest precision of the dtype of the arrays passed as argument, or by the their 
298      contents (when `check_contents is True`). If none of the condition requires 
299      `np.int64` then this function returns `np.int32`. 
300   
301      Parameters 
302      ---------- 
303      arrays : ndarray or tuple of ndarrays, default=() 
304          Input arrays whose types/contents to check. 
305   
306      maxval : float, default=None 
307          Maximum value needed. 
308   
309      check_contents : bool, default=False 
310          Whether to check the values in the arrays and not just their types. 
311          By default, check only the types. 
312   
313      Returns 
314      ------- 
315      dtype : {np.int32, np.int64} 
316          Suitable index data type (int32 or int64). 
317      """
318  
319      int32min = np.int32(np.iinfo(np.int32).min)
320      int32max = np.int32(np.iinfo(np.int32).max)
321  
322      if maxval is not None:
323          if maxval > np.iinfo(np.int64).max:
324              raise ValueError(
325                  f"maxval={maxval} is to large to be represented as np.int64."
326              )
327          if maxval > int32max:
328              return np.int64
329  
330      if isinstance(arrays, np.ndarray):
331          arrays = (arrays,)
332  
333      for arr in arrays:
334          if not isinstance(arr, np.ndarray):
335              raise TypeError(
336                  f"Arrays should be of type np.ndarray, got {type(arr)} instead."
337              )
338          if not np.issubdtype(arr.dtype, np.integer):
339              raise ValueError(
340                  f"Array dtype {arr.dtype} is not supported for index dtype. We expect "
341                  "integral values."
342              )
343          if not np.can_cast(arr.dtype, np.int32):
344              if not check_contents:
345                  # when `check_contents` is False, we stay on the safe side and return
346                  # np.int64.
347                  return np.int64
348              if arr.size == 0:
349                  # a bigger type not needed yet, let's look at the next array
350                  continue
351              else:
352                  maxval = arr.max()
353                  minval = arr.min()
354                  if minval < int32min or maxval > int32max:
355                      # a big index type is actually needed
356                      return np.int64
357  
358      return np.int32

380  def _in_unstable_openblas_configuration():
381      """Return True if in an unstable configuration for OpenBLAS"""
382  
383      # Import libraries which might load OpenBLAS.
384      import numpy  # noqa
385      import scipy  # noqa
386  
387      modules_info = _get_threadpool_controller().info()
388  
389      open_blas_used = any(info["internal_api"] == "openblas" for info in modules_info)
390      if not open_blas_used:
391          return False
392  
393      # OpenBLAS 0.3.16 fixed instability for arm64, see:
394      # https://github.com/xianyi/OpenBLAS/blob/1b6db3dbba672b4f8af935bd43a1ff6cff4d20b7/Changelog.txt#L56-L58 # noqa
395      openblas_arm64_stable_version = parse_version("0.3.16")
396      for info in modules_info:
397          if info["internal_api"] != "openblas":
398              continue
399          openblas_version = info.get("version")
400          openblas_architecture = info.get("architecture")
401          if openblas_version is None or openblas_architecture is None:
402              # Cannot be sure that OpenBLAS is good enough. Assume unstable:
403              return True  # pragma: no cover
404          if (
405              openblas_architecture == "neoversen1"
406              and parse_version(openblas_version) < openblas_arm64_stable_version
407          ):
408              # See discussions in https://github.com/numpy/numpy/issues/19411
409              return True  # pragma: no cover
410      return False
