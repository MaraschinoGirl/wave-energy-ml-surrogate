30   def solve_sylvester(a, b, q):
31       """ 
32       Computes a solution (X) to the Sylvester equation :math:`AX + XB = Q`. 
33    
34       Parameters 
35       ---------- 
36       a : (M, M) array_like 
37           Leading matrix of the Sylvester equation 
38       b : (N, N) array_like 
39           Trailing matrix of the Sylvester equation 
40       q : (M, N) array_like 
41           Right-hand side 
42    
43       Returns 
44       ------- 
45       x : (M, N) ndarray 
46           The solution to the Sylvester equation. 
47    
48       Raises 
49       ------ 
50       LinAlgError 
51           If solution was not found 
52    
53       Notes 
54       ----- 
55       Computes a solution to the Sylvester matrix equation via the Bartels- 
56       Stewart algorithm. The A and B matrices first undergo Schur 
57       decompositions. The resulting matrices are used to construct an 
58       alternative Sylvester equation (``RY + YS^T = F``) where the R and S 
59       matrices are in quasi-triangular form (or, when R, S or F are complex, 
60       triangular form). The simplified equation is then solved using 
61       ``*TRSYL`` from LAPACK directly. 
62    
63       .. versionadded:: 0.11.0 
64    
65       Examples 
66       -------- 
67       Given `a`, `b`, and `q` solve for `x`: 
68    
69       >>> import numpy as np 
70       >>> from scipy import linalg 
71       >>> a = np.array([[-3, -2, 0], [-1, -1, 3], [3, -5, -1]]) 
72       >>> b = np.array([[1]]) 
73       >>> q = np.array([[1],[2],[3]]) 
74       >>> x = linalg.solve_sylvester(a, b, q) 
75       >>> x 
76       array([[ 0.0625], 
77              [-0.5625], 
78              [ 0.6875]]) 
79       >>> np.allclose(a.dot(x) + x.dot(b), q) 
80       True 
81    
82       """
83       # Accommodate empty a
84       if a.size == 0 or b.size == 0:
85           tdict = {'s': np.float32, 'd': np.float64,
86                    'c': np.complex64, 'z': np.complex128}
87           func, = get_lapack_funcs(('trsyl',), arrays=(a, b, q))
88           return np.empty(q.shape, dtype=tdict[func.typecode])
89   
90       # Compute the Schur decomposition form of a
91       r, u = schur(a, output='real')
92   
93       # Compute the Schur decomposition of b
94       s, v = schur(b.conj().transpose(), output='real')
95   
96       # Construct f = u'*q*v
97       f = np.dot(np.dot(u.conj().transpose(), q), v)
98   
99       # Call the Sylvester equation solver
100      trsyl, = get_lapack_funcs(('trsyl',), (r, s, f))
101      if trsyl is None:
102          raise RuntimeError('LAPACK implementation does not contain a proper '
103                             'Sylvester equation solver (TRSYL)')
104      y, scale, info = trsyl(r, s, f, tranb='C')
105  
106      y = scale*y
107  
108      if info < 0:
109          raise LinAlgError("Illegal value encountered in "
110                            "the %d term" % (-info,))
111  
112      return np.dot(np.dot(u, y), v.conj().transpose())

115  def solve_continuous_lyapunov(a, q):
116      """ 
117      Solves the continuous Lyapunov equation :math:`AX + XA^H = Q`. 
118   
119      Uses the Bartels-Stewart algorithm to find :math:`X`. 
120   
121      Parameters 
122      ---------- 
123      a : array_like 
124          A square matrix 
125   
126      q : array_like 
127          Right-hand side square matrix 
128   
129      Returns 
130      ------- 
131      x : ndarray 
132          Solution to the continuous Lyapunov equation 
133   
134      See Also 
135      -------- 
136      solve_discrete_lyapunov : computes the solution to the discrete-time 
137          Lyapunov equation 
138      solve_sylvester : computes the solution to the Sylvester equation 
139   
140      Notes 
141      ----- 
142      The continuous Lyapunov equation is a special form of the Sylvester 
143      equation, hence this solver relies on LAPACK routine ?TRSYL. 
144   
145      .. versionadded:: 0.11.0 
146   
147      Examples 
148      -------- 
149      Given `a` and `q` solve for `x`: 
150   
151      >>> import numpy as np 
152      >>> from scipy import linalg 
153      >>> a = np.array([[-3, -2, 0], [-1, -1, 0], [0, -5, -1]]) 
154      >>> b = np.array([2, 4, -1]) 
155      >>> q = np.eye(3) 
156      >>> x = linalg.solve_continuous_lyapunov(a, q) 
157      >>> x 
158      array([[ -0.75  ,   0.875 ,  -3.75  ], 
159             [  0.875 ,  -1.375 ,   5.3125], 
160             [ -3.75  ,   5.3125, -27.0625]]) 
161      >>> np.allclose(a.dot(x) + x.dot(a.T), q) 
162      True 
163      """
164  
165      a = np.atleast_2d(_asarray_validated(a, check_finite=True))
166      q = np.atleast_2d(_asarray_validated(q, check_finite=True))
167  
168      r_or_c = float
169  
170      for ind, _ in enumerate((a, q)):
171          if np.iscomplexobj(_):
172              r_or_c = complex
173  
174          if not np.equal(*_.shape):
175              raise ValueError(f"Matrix {'aq'[ind]} should be square.")
176  
177      # Shape consistency check
178      if a.shape != q.shape:
179          raise ValueError("Matrix a and q should have the same shape.")
180  
181      # Accommodate empty array
182      if a.size == 0:
183          tdict = {'s': np.float32, 'd': np.float64,
184                   'c': np.complex64, 'z': np.complex128}
185          func, = get_lapack_funcs(('trsyl',), arrays=(a, q))
186          return np.empty(a.shape, dtype=tdict[func.typecode])
187  
188      # Compute the Schur decomposition form of a
189      r, u = schur(a, output='real')
190  
191      # Construct f = u'*q*u
192      f = u.conj().T.dot(q.dot(u))
193  
194      # Call the Sylvester equation solver
195      trsyl = get_lapack_funcs('trsyl', (r, f))
196  
197      dtype_string = 'T' if r_or_c is float else 'C'
198      y, scale, info = trsyl(r, r, f, tranb=dtype_string)
199  
200      if info < 0:
201          raise ValueError('?TRSYL exited with the internal error '
202                           f'"illegal value in argument number {-info}.". See '
203                           'LAPACK documentation for the ?TRSYL error codes.')
204      elif info == 1:
205          warnings.warn('Input "a" has an eigenvalue pair whose sum is '
206                        'very close to or exactly zero. The solution is '
207                        'obtained via perturbing the coefficients.',
208                        RuntimeWarning, stacklevel=2)
209      y *= scale
210  
211      return u.dot(y).dot(u.conj().T)

218  def _solve_discrete_lyapunov_direct(a, q):
219      """ 
220      Solves the discrete Lyapunov equation directly. 
221   
222      This function is called by the `solve_discrete_lyapunov` function with 
223      `method=direct`. It is not supposed to be called directly. 
224      """
225  
226      lhs = np.kron(a, a.conj())
227      lhs = np.eye(lhs.shape[0]) - lhs
228      x = solve(lhs, q.flatten())
229  
230      return np.reshape(x, q.shape)

233  def _solve_discrete_lyapunov_bilinear(a, q):
234      """ 
235      Solves the discrete Lyapunov equation using a bilinear transformation. 
236   
237      This function is called by the `solve_discrete_lyapunov` function with 
238      `method=bilinear`. It is not supposed to be called directly. 
239      """
240      eye = np.eye(a.shape[0])
241      aH = a.conj().transpose()
242      aHI_inv = inv(aH + eye)
243      b = np.dot(aH - eye, aHI_inv)
244      c = 2*np.dot(np.dot(inv(a + eye), q), aHI_inv)
245      return solve_lyapunov(b.conj().transpose(), -c)

248  def solve_discrete_lyapunov(a, q, method=None):
249      """ 
250      Solves the discrete Lyapunov equation :math:`AXA^H - X + Q = 0`. 
251   
252      Parameters 
253      ---------- 
254      a, q : (M, M) array_like 
255          Square matrices corresponding to A and Q in the equation 
256          above respectively. Must have the same shape. 
257   
258      method : {'direct', 'bilinear'}, optional 
259          Type of solver. 
260   
261          If not given, chosen to be ``direct`` if ``M`` is less than 10 and 
262          ``bilinear`` otherwise. 
263   
264      Returns 
265      ------- 
266      x : ndarray 
267          Solution to the discrete Lyapunov equation 
268   
269      See Also 
270      -------- 
271      solve_continuous_lyapunov : computes the solution to the continuous-time 
272          Lyapunov equation 
273   
274      Notes 
275      ----- 
276      This section describes the available solvers that can be selected by the 
277      'method' parameter. The default method is *direct* if ``M`` is less than 10 
278      and ``bilinear`` otherwise. 
279   
280      Method *direct* uses a direct analytical solution to the discrete Lyapunov 
281      equation. The algorithm is given in, for example, [1]_. However, it requires 
282      the linear solution of a system with dimension :math:`M^2` so that 
283      performance degrades rapidly for even moderately sized matrices. 
284   
285      Method *bilinear* uses a bilinear transformation to convert the discrete 
286      Lyapunov equation to a continuous Lyapunov equation :math:`(BX+XB'=-C)` 
287      where :math:`B=(A-I)(A+I)^{-1}` and 
288      :math:`C=2(A' + I)^{-1} Q (A + I)^{-1}`. The continuous equation can be 
289      efficiently solved since it is a special case of a Sylvester equation. 
290      The transformation algorithm is from Popov (1964) as described in [2]_. 
291   
292      .. versionadded:: 0.11.0 
293   
294      References 
295      ---------- 
296      .. [1] "Lyapunov equation", Wikipedia, 
297         https://en.wikipedia.org/wiki/Lyapunov_equation#Discrete_time 
298      .. [2] Gajic, Z., and M.T.J. Qureshi. 2008. 
299         Lyapunov Matrix Equation in System Stability and Control. 
300         Dover Books on Engineering Series. Dover Publications. 
301   
302      Examples 
303      -------- 
304      Given `a` and `q` solve for `x`: 
305   
306      >>> import numpy as np 
307      >>> from scipy import linalg 
308      >>> a = np.array([[0.2, 0.5],[0.7, -0.9]]) 
309      >>> q = np.eye(2) 
310      >>> x = linalg.solve_discrete_lyapunov(a, q) 
311      >>> x 
312      array([[ 0.70872893,  1.43518822], 
313             [ 1.43518822, -2.4266315 ]]) 
314      >>> np.allclose(a.dot(x).dot(a.T)-x, -q) 
315      True 
316   
317      """
318      a = np.asarray(a)
319      q = np.asarray(q)
320      if method is None:
321          # Select automatically based on size of matrices
322          if a.shape[0] >= 10:
323              method = 'bilinear'
324          else:
325              method = 'direct'
326  
327      meth = method.lower()
328  
329      if meth == 'direct':
330          x = _solve_discrete_lyapunov_direct(a, q)
331      elif meth == 'bilinear':
332          x = _solve_discrete_lyapunov_bilinear(a, q)
333      else:
334          raise ValueError(f'Unknown solver {method}')
335  
336      return x

339  def solve_continuous_are(a, b, q, r, e=None, s=None, balanced=True):
340      r""" 
341      Solves the continuous-time algebraic Riccati equation (CARE). 
342   
343      The CARE is defined as 
344   
345      .. math:: 
346   
347            X A + A^H X - X B R^{-1} B^H X + Q = 0 
348   
349      The limitations for a solution to exist are : 
350   
351          * All eigenvalues of :math:`A` on the right half plane, should be 
352            controllable. 
353   
354          * The associated hamiltonian pencil (See Notes), should have 
355            eigenvalues sufficiently away from the imaginary axis. 
356   
357      Moreover, if ``e`` or ``s`` is not precisely ``None``, then the 
358      generalized version of CARE 
359   
360      .. math:: 
361   
362            E^HXA + A^HXE - (E^HXB + S) R^{-1} (B^HXE + S^H) + Q = 0 
363   
364      is solved. When omitted, ``e`` is assumed to be the identity and ``s`` 
365      is assumed to be the zero matrix with sizes compatible with ``a`` and 
366      ``b``, respectively. 
367   
368      Parameters 
369      ---------- 
370      a : (M, M) array_like 
371          Square matrix 
372      b : (M, N) array_like 
373          Input 
374      q : (M, M) array_like 
375          Input 
376      r : (N, N) array_like 
377          Nonsingular square matrix 
378      e : (M, M) array_like, optional 
379          Nonsingular square matrix 
380      s : (M, N) array_like, optional 
381          Input 
382      balanced : bool, optional 
383          The boolean that indicates whether a balancing step is performed 
384          on the data. The default is set to True. 
385   
386      Returns 
387      ------- 
388      x : (M, M) ndarray 
389          Solution to the continuous-time algebraic Riccati equation. 
390   
391      Raises 
392      ------ 
393      LinAlgError 
394          For cases where the stable subspace of the pencil could not be 
395          isolated. See Notes section and the references for details. 
396   
397      See Also 
398      -------- 
399      solve_discrete_are : Solves the discrete-time algebraic Riccati equation 
400   
401      Notes 
402      ----- 
403      The equation is solved by forming the extended hamiltonian matrix pencil, 
404      as described in [1]_, :math:`H - \lambda J` given by the block matrices :: 
405   
406          [ A    0    B ]             [ E   0    0 ] 
407          [-Q  -A^H  -S ] - \lambda * [ 0  E^H   0 ] 
408          [ S^H B^H   R ]             [ 0   0    0 ] 
409   
410      and using a QZ decomposition method. 
411   
412      In this algorithm, the fail conditions are linked to the symmetry 
413      of the product :math:`U_2 U_1^{-1}` and condition number of 
414      :math:`U_1`. Here, :math:`U` is the 2m-by-m matrix that holds the 
415      eigenvectors spanning the stable subspace with 2-m rows and partitioned 
416      into two m-row matrices. See [1]_ and [2]_ for more details. 
417   
418      In order to improve the QZ decomposition accuracy, the pencil goes 
419      through a balancing step where the sum of absolute values of 
420      :math:`H` and :math:`J` entries (after removing the diagonal entries of 
421      the sum) is balanced following the recipe given in [3]_. 
422   
423      .. versionadded:: 0.11.0 
424   
425      References 
426      ---------- 
427      .. [1]  P. van Dooren , "A Generalized Eigenvalue Approach For Solving 
428         Riccati Equations.", SIAM Journal on Scientific and Statistical 
429         Computing, Vol.2(2), :doi:`10.1137/0902010` 
430   
431      .. [2] A.J. Laub, "A Schur Method for Solving Algebraic Riccati 
432         Equations.", Massachusetts Institute of Technology. Laboratory for 
433         Information and Decision Systems. LIDS-R ; 859. Available online : 
434         http://hdl.handle.net/1721.1/1301 
435   
436      .. [3] P. Benner, "Symplectic Balancing of Hamiltonian Matrices", 2001, 
437         SIAM J. Sci. Comput., 2001, Vol.22(5), :doi:`10.1137/S1064827500367993` 
438   
439      Examples 
440      -------- 
441      Given `a`, `b`, `q`, and `r` solve for `x`: 
442   
443      >>> import numpy as np 
444      >>> from scipy import linalg 
445      >>> a = np.array([[4, 3], [-4.5, -3.5]]) 
446      >>> b = np.array([[1], [-1]]) 
447      >>> q = np.array([[9, 6], [6, 4.]]) 
448      >>> r = 1 
449      >>> x = linalg.solve_continuous_are(a, b, q, r) 
450      >>> x 
451      array([[ 21.72792206,  14.48528137], 
452             [ 14.48528137,   9.65685425]]) 
453      >>> np.allclose(a.T.dot(x) + x.dot(a)-x.dot(b).dot(b.T).dot(x), -q) 
454      True 
455   
456      """
457  
458      # Validate input arguments
459      a, b, q, r, e, s, m, n, r_or_c, gen_are = _are_validate_args(
460                                                       a, b, q, r, e, s, 'care')
461  
462      H = np.empty((2*m+n, 2*m+n), dtype=r_or_c)
463      H[:m, :m] = a
464      H[:m, m:2*m] = 0.
465      H[:m, 2*m:] = b
466      H[m:2*m, :m] = -q
467      H[m:2*m, m:2*m] = -a.conj().T
468      H[m:2*m, 2*m:] = 0. if s is None else -s
469      H[2*m:, :m] = 0. if s is None else s.conj().T
470      H[2*m:, m:2*m] = b.conj().T
471      H[2*m:, 2*m:] = r
472  
473      if gen_are and e is not None:
474          J = block_diag(e, e.conj().T, np.zeros_like(r, dtype=r_or_c))
475      else:
476          J = block_diag(np.eye(2*m), np.zeros_like(r, dtype=r_or_c))
477  
478      if balanced:
479          # xGEBAL does not remove the diagonals before scaling. Also
480          # to avoid destroying the Symplectic structure, we follow Ref.3
481          M = np.abs(H) + np.abs(J)
482          np.fill_diagonal(M, 0.)
483          _, (sca, _) = matrix_balance(M, separate=1, permute=0)
484          # do we need to bother?
485          if not np.allclose(sca, np.ones_like(sca)):
486              # Now impose diag(D,inv(D)) from Benner where D is
487              # square root of s_i/s_(n+i) for i=0,....
488              sca = np.log2(sca)
489              # NOTE: Py3 uses "Bankers Rounding: round to the nearest even" !!
490              s = np.round((sca[m:2*m] - sca[:m])/2)
491              sca = 2 ** np.r_[s, -s, sca[2*m:]]
492              # Elementwise multiplication via broadcasting.
493              elwisescale = sca[:, None] * np.reciprocal(sca)
494              H *= elwisescale
495              J *= elwisescale
496  
497      # Deflate the pencil to 2m x 2m ala Ref.1, eq.(55)
498      q, r = qr(H[:, -n:])
499      H = q[:, n:].conj().T.dot(H[:, :2*m])
500      J = q[:2*m, n:].conj().T.dot(J[:2*m, :2*m])
501  
502      # Decide on which output type is needed for QZ
503      out_str = 'real' if r_or_c is float else 'complex'
504  
505      _, _, _, _, _, u = ordqz(H, J, sort='lhp', overwrite_a=True,
506                               overwrite_b=True, check_finite=False,
507                               output=out_str)
508  
509      # Get the relevant parts of the stable subspace basis
510      if e is not None:
511          u, _ = qr(np.vstack((e.dot(u[:m, :m]), u[m:, :m])))
512      u00 = u[:m, :m]
513      u10 = u[m:, :m]
514  
515      # Solve via back-substituion after checking the condition of u00
516      up, ul, uu = lu(u00)
517      if 1/cond(uu) < np.spacing(1.):
518          raise LinAlgError('Failed to find a finite solution.')
519  
520      # Exploit the triangular structure
521      x = solve_triangular(ul.conj().T,
522                           solve_triangular(uu.conj().T,
523                                            u10.conj().T,
524                                            lower=True),
525                           unit_diagonal=True,
526                           ).conj().T.dot(up.conj().T)
527      if balanced:
528          x *= sca[:m, None] * sca[:m]
529  
530      # Check the deviation from symmetry for lack of success
531      # See proof of Thm.5 item 3 in [2]
532      u_sym = u00.conj().T.dot(u10)
533      n_u_sym = norm(u_sym, 1)
534      u_sym = u_sym - u_sym.conj().T
535      sym_threshold = np.max([np.spacing(1000.), 0.1*n_u_sym])
536  
537      if norm(u_sym, 1) > sym_threshold:
538          raise LinAlgError('The associated Hamiltonian pencil has eigenvalues '
539                            'too close to the imaginary axis')
540  
541      return (x + x.conj().T)/2
542

544  def solve_discrete_are(a, b, q, r, e=None, s=None, balanced=True):
545      r""" 
546      Solves the discrete-time algebraic Riccati equation (DARE). 
547   
548      The DARE is defined as 
549   
550      .. math:: 
551   
552            A^HXA - X - (A^HXB) (R + B^HXB)^{-1} (B^HXA) + Q = 0 
553   
554      The limitations for a solution to exist are : 
555   
556          * All eigenvalues of :math:`A` outside the unit disc, should be 
557            controllable. 
558   
559          * The associated symplectic pencil (See Notes), should have 
560            eigenvalues sufficiently away from the unit circle. 
561   
562      Moreover, if ``e`` and ``s`` are not both precisely ``None``, then the 
563      generalized version of DARE 
564   
565      .. math:: 
566   
567            A^HXA - E^HXE - (A^HXB+S) (R+B^HXB)^{-1} (B^HXA+S^H) + Q = 0 
568   
569      is solved. When omitted, ``e`` is assumed to be the identity and ``s`` 
570      is assumed to be the zero matrix. 
571   
572      Parameters 
573      ---------- 
574      a : (M, M) array_like 
575          Square matrix 
576      b : (M, N) array_like 
577          Input 
578      q : (M, M) array_like 
579          Input 
580      r : (N, N) array_like 
581          Square matrix 
582      e : (M, M) array_like, optional 
583          Nonsingular square matrix 
584      s : (M, N) array_like, optional 
585          Input 
586      balanced : bool 
587          The boolean that indicates whether a balancing step is performed 
588          on the data. The default is set to True. 
589   
590      Returns 
591      ------- 
592      x : (M, M) ndarray 
593          Solution to the discrete algebraic Riccati equation. 
594   
595      Raises 
596      ------ 
597      LinAlgError 
598          For cases where the stable subspace of the pencil could not be 
599          isolated. See Notes section and the references for details. 
600   
601      See Also 
602      -------- 
603      solve_continuous_are : Solves the continuous algebraic Riccati equation 
604   
605      Notes 
606      ----- 
607      The equation is solved by forming the extended symplectic matrix pencil, 
608      as described in [1]_, :math:`H - \lambda J` given by the block matrices :: 
609   
610             [  A   0   B ]             [ E   0   B ] 
611             [ -Q  E^H -S ] - \lambda * [ 0  A^H  0 ] 
612             [ S^H  0   R ]             [ 0 -B^H  0 ] 
613   
614      and using a QZ decomposition method. 
615   
616      In this algorithm, the fail conditions are linked to the symmetry 
617      of the product :math:`U_2 U_1^{-1}` and condition number of 
618      :math:`U_1`. Here, :math:`U` is the 2m-by-m matrix that holds the 
619      eigenvectors spanning the stable subspace with 2-m rows and partitioned 
620      into two m-row matrices. See [1]_ and [2]_ for more details. 
621   
622      In order to improve the QZ decomposition accuracy, the pencil goes 
623      through a balancing step where the sum of absolute values of 
624      :math:`H` and :math:`J` rows/cols (after removing the diagonal entries) 
625      is balanced following the recipe given in [3]_. If the data has small 
626      numerical noise, balancing may amplify their effects and some clean up 
627      is required. 
628   
629      .. versionadded:: 0.11.0 
630   
631      References 
632      ---------- 
633      .. [1]  P. van Dooren , "A Generalized Eigenvalue Approach For Solving 
634         Riccati Equations.", SIAM Journal on Scientific and Statistical 
635         Computing, Vol.2(2), :doi:`10.1137/0902010` 
636   
637      .. [2] A.J. Laub, "A Schur Method for Solving Algebraic Riccati 
638         Equations.", Massachusetts Institute of Technology. Laboratory for 
639         Information and Decision Systems. LIDS-R ; 859. Available online : 
640         http://hdl.handle.net/1721.1/1301 
641   
642      .. [3] P. Benner, "Symplectic Balancing of Hamiltonian Matrices", 2001, 
643         SIAM J. Sci. Comput., 2001, Vol.22(5), :doi:`10.1137/S1064827500367993` 
644   
645      Examples 
646      -------- 
647      Given `a`, `b`, `q`, and `r` solve for `x`: 
648   
649      >>> import numpy as np 
650      >>> from scipy import linalg as la 
651      >>> a = np.array([[0, 1], [0, -1]]) 
652      >>> b = np.array([[1, 0], [2, 1]]) 
653      >>> q = np.array([[-4, -4], [-4, 7]]) 
654      >>> r = np.array([[9, 3], [3, 1]]) 
655      >>> x = la.solve_discrete_are(a, b, q, r) 
656      >>> x 
657      array([[-4., -4.], 
658             [-4.,  7.]]) 
659      >>> R = la.solve(r + b.T.dot(x).dot(b), b.T.dot(x).dot(a)) 
660      >>> np.allclose(a.T.dot(x).dot(a) - x - a.T.dot(x).dot(b).dot(R), -q) 
661      True 
662   
663      """
664  
665      # Validate input arguments
666      a, b, q, r, e, s, m, n, r_or_c, gen_are = _are_validate_args(
667                                                       a, b, q, r, e, s, 'dare')
668  
669      # Form the matrix pencil
670      H = np.zeros((2*m+n, 2*m+n), dtype=r_or_c)
671      H[:m, :m] = a
672      H[:m, 2*m:] = b
673      H[m:2*m, :m] = -q
674      H[m:2*m, m:2*m] = np.eye(m) if e is None else e.conj().T
675      H[m:2*m, 2*m:] = 0. if s is None else -s
676      H[2*m:, :m] = 0. if s is None else s.conj().T
677      H[2*m:, 2*m:] = r
678  
679      J = np.zeros_like(H, dtype=r_or_c)
680      J[:m, :m] = np.eye(m) if e is None else e
681      J[m:2*m, m:2*m] = a.conj().T
682      J[2*m:, m:2*m] = -b.conj().T
683  
684      if balanced:
685          # xGEBAL does not remove the diagonals before scaling. Also
686          # to avoid destroying the Symplectic structure, we follow Ref.3
687          M = np.abs(H) + np.abs(J)
688          np.fill_diagonal(M, 0.)
689          _, (sca, _) = matrix_balance(M, separate=1, permute=0)
690          # do we need to bother?
691          if not np.allclose(sca, np.ones_like(sca)):
692              # Now impose diag(D,inv(D)) from Benner where D is
693              # square root of s_i/s_(n+i) for i=0,....
694              sca = np.log2(sca)
695              # NOTE: Py3 uses "Bankers Rounding: round to the nearest even" !!
696              s = np.round((sca[m:2*m] - sca[:m])/2)
697              sca = 2 ** np.r_[s, -s, sca[2*m:]]
698              # Elementwise multiplication via broadcasting.
699              elwisescale = sca[:, None] * np.reciprocal(sca)
700              H *= elwisescale
701              J *= elwisescale
702  
703      # Deflate the pencil by the R column ala Ref.1
704      q_of_qr, _ = qr(H[:, -n:])
705      H = q_of_qr[:, n:].conj().T.dot(H[:, :2*m])
706      J = q_of_qr[:, n:].conj().T.dot(J[:, :2*m])
707  
708      # Decide on which output type is needed for QZ
709      out_str = 'real' if r_or_c is float else 'complex'
710  
711      _, _, _, _, _, u = ordqz(H, J, sort='iuc',
712                               overwrite_a=True,
713                               overwrite_b=True,
714                               check_finite=False,
715                               output=out_str)
716  
717      # Get the relevant parts of the stable subspace basis
718      if e is not None:
719          u, _ = qr(np.vstack((e.dot(u[:m, :m]), u[m:, :m])))
720      u00 = u[:m, :m]
721      u10 = u[m:, :m]
722  
723      # Solve via back-substituion after checking the condition of u00
724      up, ul, uu = lu(u00)
725  
726      if 1/cond(uu) < np.spacing(1.):
727          raise LinAlgError('Failed to find a finite solution.')
728  
729      # Exploit the triangular structure
730      x = solve_triangular(ul.conj().T,
731                           solve_triangular(uu.conj().T,
732                                            u10.conj().T,
733                                            lower=True),
734                           unit_diagonal=True,
735                           ).conj().T.dot(up.conj().T)
736      if balanced:
737          x *= sca[:m, None] * sca[:m]
738  
739      # Check the deviation from symmetry for lack of success
740      # See proof of Thm.5 item 3 in [2]
741      u_sym = u00.conj().T.dot(u10)
742      n_u_sym = norm(u_sym, 1)
743      u_sym = u_sym - u_sym.conj().T
744      sym_threshold = np.max([np.spacing(1000.), 0.1*n_u_sym])
745  
746      if norm(u_sym, 1) > sym_threshold:
747          raise LinAlgError('The associated symplectic pencil has eigenvalues '
748                            'too close to the unit circle')
749  
750      return (x + x.conj().T)/2
751

753  def _are_validate_args(a, b, q, r, e, s, eq_type='care'):
754      """ 
755      A helper function to validate the arguments supplied to the 
756      Riccati equation solvers. Any discrepancy found in the input 
757      matrices leads to a ``ValueError`` exception. 
758   
759      Essentially, it performs: 
760   
761          - a check whether the input is free of NaN and Infs 
762          - a pass for the data through ``numpy.atleast_2d()`` 
763          - squareness check of the relevant arrays 
764          - shape consistency check of the arrays 
765          - singularity check of the relevant arrays 
766          - symmetricity check of the relevant matrices 
767          - a check whether the regular or the generalized version is asked. 
768   
769      This function is used by ``solve_continuous_are`` and 
770      ``solve_discrete_are``. 
771   
772      Parameters 
773      ---------- 
774      a, b, q, r, e, s : array_like 
775          Input data 
776      eq_type : str 
777          Accepted arguments are 'care' and 'dare'. 
778   
779      Returns 
780      ------- 
781      a, b, q, r, e, s : ndarray 
782          Regularized input data 
783      m, n : int 
784          shape of the problem 
785      r_or_c : type 
786          Data type of the problem, returns float or complex 
787      gen_or_not : bool 
788          Type of the equation, True for generalized and False for regular ARE. 
789   
790      """
791  
792      if eq_type.lower() not in ("dare", "care"):
793          raise ValueError("Equation type unknown. "
794                           "Only 'care' and 'dare' is understood")
795  
796      a = np.atleast_2d(_asarray_validated(a, check_finite=True))
797      b = np.atleast_2d(_asarray_validated(b, check_finite=True))
798      q = np.atleast_2d(_asarray_validated(q, check_finite=True))
799      r = np.atleast_2d(_asarray_validated(r, check_finite=True))
800  
801      # Get the correct data types otherwise NumPy complains
802      # about pushing complex numbers into real arrays.
803      r_or_c = complex if np.iscomplexobj(b) else float
804  
805      for ind, mat in enumerate((a, q, r)):
806          if np.iscomplexobj(mat):
807              r_or_c = complex
808  
809          if not np.equal(*mat.shape):
810              raise ValueError(f"Matrix {'aqr'[ind]} should be square.")
811  
812      # Shape consistency checks
813      m, n = b.shape
814      if m != a.shape[0]:
815          raise ValueError("Matrix a and b should have the same number of rows.")
816      if m != q.shape[0]:
817          raise ValueError("Matrix a and q should have the same shape.")
818      if n != r.shape[0]:
819          raise ValueError("Matrix b and r should have the same number of cols.")
820  
821      # Check if the data matrices q, r are (sufficiently) hermitian
822      for ind, mat in enumerate((q, r)):
823          if norm(mat - mat.conj().T, 1) > np.spacing(norm(mat, 1))*100:
824              raise ValueError(f"Matrix {'qr'[ind]} should be symmetric/hermitian.")
825  
826      # Continuous time ARE should have a nonsingular r matrix.
827      if eq_type == 'care':
828          min_sv = svd(r, compute_uv=False)[-1]
829          if min_sv == 0. or min_sv < np.spacing(1.)*norm(r, 1):
830              raise ValueError('Matrix r is numerically singular.')
831  
832      # Check if the generalized case is required with omitted arguments
833      # perform late shape checking etc.
834      generalized_case = e is not None or s is not None
835  
836      if generalized_case:
837          if e is not None:
838              e = np.atleast_2d(_asarray_validated(e, check_finite=True))
839              if not np.equal(*e.shape):
840                  raise ValueError("Matrix e should be square.")
841              if m != e.shape[0]:
842                  raise ValueError("Matrix a and e should have the same shape.")
843              # numpy.linalg.cond doesn't check for exact zeros and
844              # emits a runtime warning. Hence the following manual check.
845              min_sv = svd(e, compute_uv=False)[-1]
846              if min_sv == 0. or min_sv < np.spacing(1.) * norm(e, 1):
847                  raise ValueError('Matrix e is numerically singular.')
848              if np.iscomplexobj(e):
849                  r_or_c = complex
850          if s is not None:
851              s = np.atleast_2d(_asarray_validated(s, check_finite=True))
852              if s.shape != b.shape:
853                  raise ValueError("Matrix b and s should have the same shape.")
854              if np.iscomplexobj(s):
855                  r_or_c = complex
856  
857      return a, b, q, r, e, s, m, n, r_or_c, generalized_case
