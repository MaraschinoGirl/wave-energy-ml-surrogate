mypy_plugin.py 
1    """A mypy_ plugin for managing a number of platform-specific annotations. 
2    Its functionality can be split into three distinct parts: 
3     
4    * Assigning the (platform-dependent) precisions of certain `~numpy.number` 
5      subclasses, including the likes of `~numpy.int_`, `~numpy.intp` and 
6      `~numpy.longlong`. See the documentation on 
7      :ref:`scalar types <arrays.scalars.built-in>` for a comprehensive overview 
8      of the affected classes. Without the plugin the precision of all relevant 
9      classes will be inferred as `~typing.Any`. 
10   * Removing all extended-precision `~numpy.number` subclasses that are 
11     unavailable for the platform in question. Most notably this includes the 
12     likes of `~numpy.float128` and `~numpy.complex256`. Without the plugin *all* 
13     extended-precision types will, as far as mypy is concerned, be available 
14     to all platforms. 
15   * Assigning the (platform-dependent) precision of `~numpy.ctypeslib.c_intp`. 
16     Without the plugin the type will default to `ctypes.c_int64`. 
17    
18     .. versionadded:: 1.22 
19    
20   Examples 
21   -------- 
22   To enable the plugin, one must add it to their mypy `configuration file`_: 
23    
24   .. code-block:: ini 
25    
26       [mypy] 
27       plugins = numpy.typing.mypy_plugin 
28    
29   .. _mypy: https://mypy-lang.org/ 
30   .. _configuration file: https://mypy.readthedocs.io/en/stable/config_file.html 
31    
32   """
33

51       MYPY_EX: None | ModuleNotFoundError = None

58   def _get_precision_dict() -> dict[str, str]:
59       names = [
60           ("_NBitByte", np.byte),
61           ("_NBitShort", np.short),
62           ("_NBitIntC", np.intc),
63           ("_NBitIntP", np.intp),
64           ("_NBitInt", np.int_),
65           ("_NBitLong", np.long),
66           ("_NBitLongLong", np.longlong),
67   
68           ("_NBitHalf", np.half),
69           ("_NBitSingle", np.single),
70           ("_NBitDouble", np.double),
71           ("_NBitLongDouble", np.longdouble),
72       ]
73       ret = {}
74       module = "numpy._typing"
75       for name, typ in names:
76           n: int = 8 * typ().dtype.itemsize
77           ret[f'{module}._nbit.{name}'] = f"{module}._nbit_base._{n}Bit"
78       return ret

81   def _get_extended_precision_list() -> list[str]:
82       extended_names = [
83           "uint128",
84           "uint256",
85           "int128",
86           "int256",
87           "float80",
88           "float96",
89           "float128",
90           "float256",
91           "complex160",
92           "complex192",
93           "complex256",
94           "complex512",
95       ]
96       return [i for i in extended_names if hasattr(np, i)]
97   
98   def _get_c_intp_name() -> str:
99       # Adapted from `np.core._internal._getintp_ctype`
100      char = np.dtype('n').char
101      if char == 'i':
102          return "c_int"
103      elif char == 'l':
104          return "c_long"
105      elif char == 'q':
106          return "c_longlong"
107      else:
108          return "c_long"

111  #: A dictionary mapping type-aliases in `numpy._typing._nbit` to
112  #: concrete `numpy.typing.NBitBase` subclasses.
113  _PRECISION_DICT: Final = _get_precision_dict()
114  
115  #: A list with the names of all extended precision `np.number` subclasses.
116  _EXTENDED_PRECISION_LIST: Final = _get_extended_precision_list()
117  
118  #: The name of the ctypes equivalent of `np.intp`
119  _C_INTP: Final = _get_c_intp_name()
120

122  def _hook(ctx: AnalyzeTypeContext) -> Type:
123      """Replace a type-alias with a concrete ``NBitBase`` subclass."""
124      typ, _, api = ctx
125      name = typ.name.split(".")[-1]
126      name_new = _PRECISION_DICT[f"numpy._typing._nbit.{name}"]
127      return api.named_type(name_new)

130  if TYPE_CHECKING or MYPY_EX is None:
131      def _index(iterable: Iterable[Statement], id: str) -> int:
132          """Identify the first ``ImportFrom`` instance the specified `id`."""
133          for i, value in enumerate(iterable):
134              if getattr(value, "id", None) == id:
135                  return i
136          raise ValueError("Failed to identify a `ImportFrom` instance "
137                           f"with the following id: {id!r}")

139      def _override_imports(
140          file: MypyFile,
141          module: str,
142          imports: list[tuple[str, None | str]],
143      ) -> None:
144          """Override the first `module`-based import with new `imports`."""
145          # Construct a new `from module import y` statement
146          import_obj = ImportFrom(module, 0, names=imports)
147          import_obj.is_top_level = True
148  
149          # Replace the first `module`-based import statement with `import_obj`
150          for lst in [file.defs, file.imports]:  # type: list[Statement]
151              i = _index(lst, module)
152              lst[i] = import_obj
153  
154      class _NumpyPlugin(Plugin):
155          """A mypy plugin for handling versus numpy-specific typing tasks."""
156  
157          def get_type_analyze_hook(self, fullname: str) -> None | _HookFunc:
158              """Set the precision of platform-specific `numpy.number` 
159              subclasses. 
160   
161              For example: `numpy.int_`, `numpy.longlong` and `numpy.longdouble`. 
162              """
163              if fullname in _PRECISION_DICT:
164                  return _hook
165              return None
166  
167          def get_additional_deps(
168              self, file: MypyFile
169          ) -> list[tuple[int, str, int]]:
170              """Handle all import-based overrides. 
171   
172              * Import platform-specific extended-precision `numpy.number` 
173                subclasses (*e.g.* `numpy.float96`, `numpy.float128` and 
174                `numpy.complex256`). 
175              * Import the appropriate `ctypes` equivalent to `numpy.intp`. 
176   
177              """
178              ret = [(PRI_MED, file.fullname, -1)]
179  
180              if file.fullname == "numpy":
181                  _override_imports(
182                      file, "numpy._typing._extended_precision",
183                      imports=[(v, v) for v in _EXTENDED_PRECISION_LIST],
184                  )
185              elif file.fullname == "numpy.ctypeslib":
186                  _override_imports(
187                      file, "ctypes",
188                      imports=[(_C_INTP, "_c_intp")],
189                  )
190              return ret
191  
192      def plugin(version: str) -> type[_NumpyPlugin]:
193          """An entry-point for mypy."""
194          return _NumpyPlugin
195  
196  else:
197      def plugin(version: str) -> type[_NumpyPlugin]:
198          """An entry-point for mypy."""
199          raise MYPY_EX
