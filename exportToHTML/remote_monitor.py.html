1    import json
2    import warnings
3    
4    import numpy as np
5    
6    from keras.src.api_export import keras_export
7    from keras.src.callbacks.callback import Callback
8    
9    try:
10       import requests
11   except ImportError:
12       requests = None
13   
14   
15   @keras_export("keras.callbacks.RemoteMonitor")
16   class RemoteMonitor(Callback):
17       """Callback used to stream events to a server. 
18    
19       Requires the `requests` library. 
20       Events are sent to `root + '/publish/epoch/end/'` by default. Calls are 
21       HTTP POST, with a `data` argument which is a 
22       JSON-encoded dictionary of event data. 
23       If `send_as_json=True`, the content type of the request will be 
24       `"application/json"`. 
25       Otherwise the serialized JSON will be sent within a form. 
26    
27       Args: 
28           root: String; root url of the target server. 
29           path: String; path relative to `root` to which the events will be sent. 
30           field: String; JSON field under which the data will be stored. 
31               The field is used only if the payload is sent within a form 
32               (i.e. when `send_as_json=False`). 
33           headers: Dictionary; optional custom HTTP headers. 
34           send_as_json: Boolean; whether the request should be 
35               sent as `"application/json"`. 
36       """
37   
38       def __init__(
39           self,
40           root="http://localhost:9000",
41           path="/publish/epoch/end/",
42           field="data",
43           headers=None,
44           send_as_json=False,
45       ):
46           super().__init__()
47   
48           self.root = root
49           self.path = path
50           self.field = field
51           self.headers = headers
52           self.send_as_json = send_as_json
53   
54       def on_epoch_end(self, epoch, logs=None):
55           if requests is None:
56               raise ImportError("RemoteMonitor requires the `requests` library.")
57           logs = logs or {}
58           send = {}
59           send["epoch"] = epoch
60           for k, v in logs.items():
61               # np.ndarray and np.generic are not scalar types
62               # therefore we must unwrap their scalar values and
63               # pass to the json-serializable dict 'send'
64               if isinstance(v, (np.ndarray, np.generic)):
65                   send[k] = v.item()
66               else:
67                   send[k] = v
68           try:
69               if self.send_as_json:
70                   requests.post(
71                       self.root + self.path, json=send, headers=self.headers
72                   )
73               else:
74                   requests.post(
75                       self.root + self.path,
76                       {self.field: json.dumps(send)},
77                       headers=self.headers,
78                   )
79           except requests.exceptions.RequestException:
80               warnings.warn(
81                   f"Could not reach RemoteMonitor root server at {self.root}",
82                   stacklevel=2,
83               )
