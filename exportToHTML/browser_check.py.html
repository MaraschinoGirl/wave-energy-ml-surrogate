40   class LogErrorHandler(logging.StreamHandler):
41       """A handler that exits with 1 on a logged error."""
42   
43       def __init__(self):
44           super().__init__(stream=sys.stderr)
45           self.setLevel(logging.ERROR)
46           self.errored = False
47   
48       def filter(self, record):
49           # Handle known StreamClosedError from Tornado
50           # These occur when we forcibly close Websockets or
51           # browser connections during the test.
52           # https://github.com/tornadoweb/tornado/issues/2834
53           if (
54               hasattr(record, "exc_info")
55               and record.exc_info is not None
56               and isinstance(record.exc_info[1], (StreamClosedError, WebSocketClosedError))
57           ):
58               return False
59           return super().filter(record)
60   
61       def emit(self, record):
62           self.errored = True
63           super().emit(record)

66   def run_test(app, func):
67       """Synchronous entry point to run a test function. 
68       func is a function that accepts an app url as a parameter and returns a result. 
69       func can be synchronous or asynchronous.  If it is synchronous, it will be run 
70       in a thread, so asynchronous is preferred. 
71       """
72       IOLoop.current().spawn_callback(run_test_async, app, func)

75   async def run_test_async(app, func):
76       """Run a test against the application. 
77       func is a function that accepts an app url as a parameter and returns a result. 
78       func can be synchronous or asynchronous.  If it is synchronous, it will be run 
79       in a thread, so asynchronous is preferred. 
80       """
81       handler = LogErrorHandler()
82       app.log.addHandler(handler)
83   
84       env_patch = TestEnv()
85       env_patch.start()
86   
87       app.log.info("Running async test")
88   
89       # The entry URL for browser tests is different in notebook >= 6.0,
90       # since that uses a local HTML file to point the user at the app.
91       if hasattr(app, "browser_open_file"):
92           url = urljoin("file:", pathname2url(app.browser_open_file))
93       else:
94           url = app.display_url
95   
96       # Allow a synchronous function to be passed in.
97       if inspect.iscoroutinefunction(func):
98           test = func(url)
99       else:
100          app.log.info("Using thread pool executor to run test")
101          loop = asyncio.get_event_loop()
102          executor = ThreadPoolExecutor()
103          task = loop.run_in_executor(executor, func, url)
104          test = asyncio.wait([task])
105  
106      try:
107          await test
108      except Exception as e:
109          app.log.critical("Caught exception during the test:")
110          app.log.error(str(e))
111  
112      app.log.info("Test Complete")
113  
114      result = 0
115      if handler.errored:
116          result = 1
117          app.log.critical("Exiting with 1 due to errors")
118      else:
119          app.log.info("Exiting normally")
120  
121      app.log.info("Stopping server...")
122      try:
123          app.http_server.stop()
124          app.io_loop.stop()
125          env_patch.stop()
126      except Exception as e:
127          app.log.error(str(e))
128          result = 1
129      finally:
130          time.sleep(2)
131          os._exit(result)

134  async def run_async_process(cmd, **kwargs):
135      """Run an asynchronous command"""
136      proc = await asyncio.create_subprocess_exec(*cmd, **kwargs)
137      stdout, stderr = await proc.communicate()
138      if proc.returncode != 0:
139          raise RuntimeError(str(cmd) + " exited with " + str(proc.returncode))
140      return stdout, stderr

143  async def run_browser(url):
144      """Run the browser test and return an exit code."""
145      browser = os.environ.get("JLAB_BROWSER_TYPE", "chromium")
146      if browser not in {"chromium", "firefox", "webkit"}:
147          browser = "chromium"
148  
149      target = osp.join(get_app_dir(), "browser_test")
150      if not osp.exists(osp.join(target, "node_modules")):
151          if not osp.exists(target):
152              os.makedirs(osp.join(target))
153          await run_async_process(["npm", "init", "-y"], cwd=target)
154          await run_async_process(["npm", "install", "playwright@^1.9.2"], cwd=target)
155      await run_async_process(["npx", "playwright", "install", browser], cwd=target)
156      shutil.copy(osp.join(here, "browser-test.js"), osp.join(target, "browser-test.js"))
157      await run_async_process(["node", "browser-test.js", url], cwd=target)

160  def run_browser_sync(url):
161      """Run the browser test and return an exit code."""
162      browser = os.environ.get("JLAB_BROWSER_TYPE", "chromium")
163      if browser not in {"chromium", "firefox", "webkit"}:
164          browser = "chromium"
165  
166      target = osp.join(get_app_dir(), "browser_test")
167      if not osp.exists(osp.join(target, "node_modules")):
168          os.makedirs(target)
169          subprocess.call(["npm", "init", "-y"], cwd=target)  # noqa S603 S607
170          subprocess.call(["npm", "install", "playwright@^1.9.2"], cwd=target)  # noqa S603 S607
171      subprocess.call(["npx", "playwright", "install", browser], cwd=target)  # noqa S603 S607
172      shutil.copy(osp.join(here, "browser-test.js"), osp.join(target, "browser-test.js"))
173      return subprocess.check_call(["node", "browser-test.js", url], cwd=target)  # noqa S603 S607

176  class BrowserApp(LabApp):
177      """An app the launches JupyterLab and waits for it to start up, checking for 
178      JS console errors, JS errors, and Python logged errors. 
179      """
180  
181      name = __name__
182      open_browser = False
183  
184      serverapp_config = {"base_url": "/foo/"}
185      default_url = Unicode("/lab?reset", config=True, help="The default URL to redirect to from `/`")
186      ip = "127.0.0.1"
187      flags = test_flags
188      aliases = test_aliases
189      test_browser = Bool(True)
190  
191      def initialize_settings(self):
192          self.settings.setdefault("page_config_data", {})
193          self.settings["page_config_data"]["browserTest"] = True
194          self.settings["page_config_data"]["buildAvailable"] = False
195          self.settings["page_config_data"]["exposeAppInBrowser"] = True
196          super().initialize_settings()
197  
198      def initialize_handlers(self):
199          def func(*args, **kwargs):
200              return 0
201  
202          if self.test_browser:
203              func = run_browser_sync if os.name == "nt" else run_browser
204  
205          run_test(self.serverapp, func)
206          super().initialize_handlers()
