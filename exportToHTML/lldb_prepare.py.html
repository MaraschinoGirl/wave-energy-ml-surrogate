3    # Also it marks process threads to to distinguish them from debugger
4    # threads later while settings trace in threads

7    def load_lib_and_attach(debugger, command, result, internal_dict):
8        import shlex
9    
10       args = shlex.split(command)
11   
12       dll = args[0]
13       is_debug = args[1]
14       python_code = args[2]
15       show_debug_info = args[3]
16   
17       import lldb
18   
19       options = lldb.SBExpressionOptions()
20       options.SetFetchDynamicValue()
21       options.SetTryAllThreads(run_others=False)
22       options.SetTimeoutInMicroSeconds(timeout=10000000)
23   
24       print(dll)
25       target = debugger.GetSelectedTarget()
26       res = target.EvaluateExpression('(void*)dlopen("%s", 2);' % (dll), options)
27       error = res.GetError()
28       if error:
29           print(error)
30   
31       print(python_code)
32       res = target.EvaluateExpression('(int)DoAttach(%s, "%s", %s);' % (is_debug, python_code.replace('"', "'"), show_debug_info), options)
33       error = res.GetError()
34       if error:
35           print(error)
36

38   def __lldb_init_module(debugger, internal_dict):
39       import lldb
40   
41       debugger.HandleCommand("command script add -f lldb_prepare.load_lib_and_attach load_lib_and_attach")
42   
43       try:
44           target = debugger.GetSelectedTarget()
45           if target:
46               process = target.GetProcess()
47               if process:
48                   for thread in process:
49                       # print('Marking process thread %d'%thread.GetThreadID())
50                       internal_dict["_thread_%d" % thread.GetThreadID()] = True
51                       # thread.Suspend()
52       except:
53           import traceback
54   
55           traceback.print_exc()
