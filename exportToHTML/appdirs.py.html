73   def user_cache_dir(appname=None, appauthor=None, version=None, opinion=True):
74       r"""Return full path to the user-specific cache dir for this application. 
75    
76           "appname" is the name of application. 
77               If None, just the system directory is returned. 
78           "appauthor" (only used on Windows) is the name of the 
79               appauthor or distributing body for this application. Typically 
80               it is the owning company name. This falls back to appname. You may 
81               pass False to disable it. 
82           "version" is an optional version path element to append to the 
83               path. You might want to use this if you want multiple versions 
84               of your app to be able to run independently. If used, this 
85               would typically be "<major>.<minor>". 
86               Only applied when appname is present. 
87           "opinion" (boolean) can be False to disable the appending of 
88               "Cache" to the base app data dir for Windows. See 
89               discussion below. 
90    
91       Typical user cache directories are: 
92           Mac OS X:   ~/Library/Caches/<AppName> 
93           Unix:       ~/.cache/<AppName> (XDG default) 
94           Win XP:     C:\Documents and Settings\<username>\Local Settings\Application Data\<AppAuthor>\<AppName>\Cache 
95           Vista:      C:\Users\<username>\AppData\Local\<AppAuthor>\<AppName>\Cache 
96    
97       On Windows the only suggestion in the MSDN docs is that local settings go in 
98       the `CSIDL_LOCAL_APPDATA` directory. This is identical to the non-roaming 
99       app data dir (the default returned by `user_data_dir` above). Apps typically 
100      put cache data somewhere *under* the given dir here. Some examples: 
101          ...\Mozilla\Firefox\Profiles\<ProfileName>\Cache 
102          ...\Acme\SuperApp\Cache\1.0 
103      OPINION: This function appends "Cache" to the `CSIDL_LOCAL_APPDATA` value. 
104      This can be disabled with the `opinion=False` option. 
105      """
106      if system == "win32":
107          if appauthor is None:
108              appauthor = appname
109          path = os.path.normpath(_get_win_folder("CSIDL_LOCAL_APPDATA"))
110          if appname:
111              if appauthor is not False:
112                  path = os.path.join(path, appauthor, appname)
113              else:
114                  path = os.path.join(path, appname)
115              if opinion:
116                  path = os.path.join(path, "Cache")
117      elif system == 'darwin':
118          path = os.path.expanduser('~/Library/Caches')
119          if appname:
120              path = os.path.join(path, appname)
121      else:
122          path = os.getenv('XDG_CACHE_HOME', os.path.expanduser('~/.cache'))
123          if appname:
124              path = os.path.join(path, appname)
125      if appname and version:
126          path = os.path.join(path, version)
127      return path

132  def _get_win_folder_from_registry(csidl_name):
133      """This is a fallback technique at best. I'm not sure if using the 
134      registry for this guarantees us the correct answer for all CSIDL_* 
135      names. 
136      """
137      import winreg as _winreg
138  
139      shell_folder_name = {
140          "CSIDL_APPDATA": "AppData",
141          "CSIDL_COMMON_APPDATA": "Common AppData",
142          "CSIDL_LOCAL_APPDATA": "Local AppData",
143      }[csidl_name]
144  
145      key = _winreg.OpenKey(
146          _winreg.HKEY_CURRENT_USER,
147          r"Software\Microsoft\Windows\CurrentVersion\Explorer\Shell Folders"
148      )
149      dir, type = _winreg.QueryValueEx(key, shell_folder_name)
150      return dir

153  def _get_win_folder_with_pywin32(csidl_name):
154      from win32com.shell import shellcon, shell
155      dir = shell.SHGetFolderPath(0, getattr(shellcon, csidl_name), 0, 0)
156      # Try to make this a unicode path because SHGetFolderPath does
157      # not return unicode strings when there is unicode data in the
158      # path.
159      try:
160          dir = unicode(dir)
161  
162          # Downgrade to short path name if have highbit chars. See
163          # <http://bugs.activestate.com/show_bug.cgi?id=85099>.
164          has_high_char = False
165          for c in dir:
166              if ord(c) > 255:
167                  has_high_char = True
168                  break
169          if has_high_char:
170              try:
171                  import win32api
172                  dir = win32api.GetShortPathName(dir)
173              except ImportError:
174                  pass
175      except UnicodeError:
176          pass
177      return dir

180  def _get_win_folder_with_ctypes(csidl_name):
181      import ctypes
182  
183      csidl_const = {
184          "CSIDL_APPDATA": 26,
185          "CSIDL_COMMON_APPDATA": 35,
186          "CSIDL_LOCAL_APPDATA": 28,
187      }[csidl_name]
188  
189      buf = ctypes.create_unicode_buffer(1024)
190      ctypes.windll.shell32.SHGetFolderPathW(None, csidl_const, None, 0, buf)
191  
192      # Downgrade to short path name if have highbit chars. See
193      # <http://bugs.activestate.com/show_bug.cgi?id=85099>.
194      has_high_char = False
195      for c in buf:
196          if ord(c) > 255:
197              has_high_char = True
198              break
199      if has_high_char:
200          buf2 = ctypes.create_unicode_buffer(1024)
201          if ctypes.windll.kernel32.GetShortPathNameW(buf.value, buf2, 1024):
202              buf = buf2
203  
204      return buf.value

206  def _get_win_folder_with_jna(csidl_name):
207      import array
208      from com.sun import jna
209      from com.sun.jna.platform import win32
210  
211      buf_size = win32.WinDef.MAX_PATH * 2
212      buf = array.zeros('c', buf_size)
213      shell = win32.Shell32.INSTANCE
214      shell.SHGetFolderPath(None, getattr(win32.ShlObj, csidl_name), None, win32.ShlObj.SHGFP_TYPE_CURRENT, buf)
215      dir = jna.Native.toString(buf.tostring()).rstrip("\0")
216  
217      # Downgrade to short path name if have highbit chars. See
218      # <http://bugs.activestate.com/show_bug.cgi?id=85099>.
219      has_high_char = False
220      for c in dir:
221          if ord(c) > 255:
222              has_high_char = True
223              break
224      if has_high_char:
225          buf = array.zeros('c', buf_size)
226          kernel = win32.Kernel32.INSTANCE
227          if kernel.GetShortPathName(dir, buf, buf_size):
228              dir = jna.Native.toString(buf.tostring()).rstrip("\0")
229  
230      return dir
