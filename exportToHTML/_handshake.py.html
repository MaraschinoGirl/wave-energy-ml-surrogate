33   # websocket supported version.
34   VERSION = 13

36   SUPPORTED_REDIRECT_STATUSES = (
37       HTTPStatus.MOVED_PERMANENTLY,
38       HTTPStatus.FOUND,
39       HTTPStatus.SEE_OTHER,
40       HTTPStatus.TEMPORARY_REDIRECT,
41       HTTPStatus.PERMANENT_REDIRECT,
42   )
43   SUCCESS_STATUSES = SUPPORTED_REDIRECT_STATUSES + (HTTPStatus.SWITCHING_PROTOCOLS,)
44   
45   CookieJar = SimpleCookieJar()

48   class handshake_response:
49       def __init__(self, status: int, headers: dict, subprotocol):
50           self.status = status
51           self.headers = headers
52           self.subprotocol = subprotocol
53           CookieJar.add(headers.get("set-cookie"))
54

56   def handshake(
57       sock, url: str, hostname: str, port: int, resource: str, **options
58   ) -> handshake_response:
59       headers, key = _get_handshake_headers(resource, url, hostname, port, options)
60   
61       header_str = "\r\n".join(headers)
62       send(sock, header_str)
63       dump("request header", header_str)
64   
65       status, resp = _get_resp_headers(sock)
66       if status in SUPPORTED_REDIRECT_STATUSES:
67           return handshake_response(status, resp, None)
68       success, subproto = _validate(resp, key, options.get("subprotocols"))
69       if not success:
70           raise WebSocketException("Invalid WebSocket Header")
71   
72       return handshake_response(status, resp, subproto)

75   def _pack_hostname(hostname: str) -> str:
76       # IPv6 address
77       if ":" in hostname:
78           return f"[{hostname}]"
79       return hostname

82   def _get_handshake_headers(
83       resource: str, url: str, host: str, port: int, options: dict
84   ) -> tuple:
85       headers = [f"GET {resource} HTTP/1.1", "Upgrade: websocket"]
86       if port in [80, 443]:
87           hostport = _pack_hostname(host)
88       else:
89           hostport = f"{_pack_hostname(host)}:{port}"
90       if options.get("host"):
91           headers.append(f'Host: {options["host"]}')
92       else:
93           headers.append(f"Host: {hostport}")
94   
95       # scheme indicates whether http or https is used in Origin
96       # The same approach is used in parse_url of _url.py to set default port
97       scheme, url = url.split(":", 1)
98       if not options.get("suppress_origin"):
99           if "origin" in options and options["origin"] is not None:
100              headers.append(f'Origin: {options["origin"]}')
101          elif scheme == "wss":
102              headers.append(f"Origin: https://{hostport}")
103          else:
104              headers.append(f"Origin: http://{hostport}")
105  
106      key = _create_sec_websocket_key()
107  
108      # Append Sec-WebSocket-Key & Sec-WebSocket-Version if not manually specified
109      if not options.get("header") or "Sec-WebSocket-Key" not in options["header"]:
110          headers.append(f"Sec-WebSocket-Key: {key}")
111      else:
112          key = options["header"]["Sec-WebSocket-Key"]
113  
114      if not options.get("header") or "Sec-WebSocket-Version" not in options["header"]:
115          headers.append(f"Sec-WebSocket-Version: {VERSION}")
116  
117      if not options.get("connection"):
118          headers.append("Connection: Upgrade")
119      else:
120          headers.append(options["connection"])
121  
122      if subprotocols := options.get("subprotocols"):
123          headers.append(f'Sec-WebSocket-Protocol: {",".join(subprotocols)}')
124  
125      if header := options.get("header"):
126          if isinstance(header, dict):
127              header = [": ".join([k, v]) for k, v in header.items() if v is not None]
128          headers.extend(header)
129  
130      server_cookie = CookieJar.get(host)
131      client_cookie = options.get("cookie", None)
132  
133      if cookie := "; ".join(filter(None, [server_cookie, client_cookie])):
134          headers.append(f"Cookie: {cookie}")
135  
136      headers.extend(("", ""))
137      return headers, key
138

140  def _get_resp_headers(sock, success_statuses: tuple = SUCCESS_STATUSES) -> tuple:
141      status, resp_headers, status_message = read_headers(sock)
142      if status not in success_statuses:
143          content_len = resp_headers.get("content-length")
144          if content_len:
145              response_body = sock.recv(
146                  int(content_len)
147              )  # read the body of the HTTP error message response and include it in the exception
148          else:
149              response_body = None
150          raise WebSocketBadStatusException(
151              f"Handshake status {status} {status_message} -+-+- {resp_headers} -+-+- {response_body}",
152              status,
153              status_message,
154              resp_headers,
155              response_body,
156          )
157      return status, resp_headers

160  _HEADERS_TO_CHECK = {
161      "upgrade": "websocket",
162      "connection": "upgrade",
163  }
164  
165  
166  def _validate(headers, key: str, subprotocols) -> tuple:
167      subproto = None
168      for k, v in _HEADERS_TO_CHECK.items():
169          r = headers.get(k, None)
170          if not r:
171              return False, None
172          r = [x.strip().lower() for x in r.split(",")]
173          if v not in r:
174              return False, None
175  
176      if subprotocols:
177          subproto = headers.get("sec-websocket-protocol", None)
178          if not subproto or subproto.lower() not in [s.lower() for s in subprotocols]:
179              error(f"Invalid subprotocol: {subprotocols}")
180              return False, None
181          subproto = subproto.lower()
182  
183      result = headers.get("sec-websocket-accept", None)
184      if not result:
185          return False, None
186      result = result.lower()
187  
188      if isinstance(result, str):
189          result = result.encode("utf-8")
190  
191      value = f"{key}258EAFA5-E914-47DA-95CA-C5AB0DC85B11".encode("utf-8")
192      hashed = base64encode(hashlib.sha1(value).digest()).strip().lower()
193  
194      if hmac.compare_digest(hashed, result):
195          return True, subproto
196      else:
197          return False, None
198  
199  
200  def _create_sec_websocket_key() -> str:
201      randomness = os.urandom(16)
202      return base64encode(randomness).decode("utf-8").strip()
