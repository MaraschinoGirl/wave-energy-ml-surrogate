48   def _load_latest_dbghelp_dll():
49       from os import getenv
50       from os.path import join, exists
51   
52       program_files_location = getenv("ProgramFiles")
53       if not program_files_location:
54           program_files_location = "C:\\Program Files"
55   
56       program_files_x86_location = getenv("ProgramFiles(x86)")
57   
58       if arch == ARCH_AMD64:
59           if wow64:
60               pathname = join(program_files_x86_location or program_files_location, "Debugging Tools for Windows (x86)", "dbghelp.dll")
61           else:
62               pathname = join(program_files_location, "Debugging Tools for Windows (x64)", "dbghelp.dll")
63       elif arch == ARCH_I386:
64           pathname = join(program_files_location, "Debugging Tools for Windows (x86)", "dbghelp.dll")
65       else:
66           pathname = None
67   
68       if pathname and exists(pathname):
69           try:
70               _dbghelp = ctypes.windll.LoadLibrary(pathname)
71               ctypes.windll.dbghelp = _dbghelp
72           except Exception:
73               pass

400  # DWORD WINAPI SymLoadModule(
401  #   __in      HANDLE hProcess,
402  #   __in_opt  HANDLE hFile,
403  #   __in_opt  PCSTR ImageName,
404  #   __in_opt  PCSTR ModuleName,
405  #   __in      DWORD BaseOfDll,
406  #   __in      DWORD SizeOfDll
407  # );
408  def SymLoadModuleA(hProcess, hFile=None, ImageName=None, ModuleName=None, BaseOfDll=None, SizeOfDll=None):
409      _SymLoadModule = windll.dbghelp.SymLoadModule
410      _SymLoadModule.argtypes = [HANDLE, HANDLE, LPSTR, LPSTR, DWORD, DWORD]
411      _SymLoadModule.restype = DWORD
412  
413      if not ImageName:
414          ImageName = None
415      if not ModuleName:
416          ModuleName = None
417      if not BaseOfDll:
418          BaseOfDll = 0
419      if not SizeOfDll:
420          SizeOfDll = 0
421      SetLastError(ERROR_SUCCESS)
422      lpBaseAddress = _SymLoadModule(hProcess, hFile, ImageName, ModuleName, BaseOfDll, SizeOfDll)
423      if lpBaseAddress == NULL:
424          dwErrorCode = GetLastError()
425          if dwErrorCode != ERROR_SUCCESS:
426              raise ctypes.WinError(dwErrorCode)
427      return lpBaseAddress
428

431  SymLoadModule = GuessStringType(SymLoadModuleA, SymLoadModuleW)
