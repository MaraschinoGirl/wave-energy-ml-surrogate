12   # pylint: disable=too-many-locals
13   def run_training_continuation_model_output(device: str, tree_method: str) -> None:
14       """Run training continuation test."""
15       datasets = pytest.importorskip("sklearn.datasets")
16       n_samples = 64
17       n_features = 32
18       X, y = datasets.make_regression(n_samples, n_features, random_state=1)
19   
20       dtrain = xgb.DMatrix(X, y)
21       params = {
22           "tree_method": tree_method,
23           "max_depth": "2",
24           "gamma": "0.1",
25           "alpha": "0.01",
26           "device": device,
27       }
28       bst_0 = xgb.train(params, dtrain, num_boost_round=64)
29       dump_0 = bst_0.get_dump(dump_format="json")
30   
31       bst_1 = xgb.train(params, dtrain, num_boost_round=32)
32       bst_1 = xgb.train(params, dtrain, num_boost_round=32, xgb_model=bst_1)
33       dump_1 = bst_1.get_dump(dump_format="json")
34   
35       T = TypeVar("T", Dict[str, Any], float, str, int, list)
36   
37       def recursive_compare(obj_0: T, obj_1: T) -> None:
38           if isinstance(obj_0, float):
39               assert np.isclose(obj_0, obj_1, atol=1e-6)
40           elif isinstance(obj_0, str):
41               assert obj_0 == obj_1
42           elif isinstance(obj_0, int):
43               assert obj_0 == obj_1
44           elif isinstance(obj_0, dict):
45               for i in range(len(obj_0.items())):
46                   assert list(obj_0.keys())[i] == list(obj_1.keys())[i]
47                   if list(obj_0.keys())[i] != "missing":
48                       recursive_compare(list(obj_0.values()), list(obj_1.values()))
49           else:
50               for i, lhs in enumerate(obj_0):
51                   rhs = obj_1[i]
52                   recursive_compare(lhs, rhs)
53   
54       assert len(dump_0) == len(dump_1)
55   
56       for i, lhs in enumerate(dump_0):
57           obj_0 = json.loads(lhs)
58           obj_1 = json.loads(dump_1[i])
59           recursive_compare(obj_0, obj_1)
