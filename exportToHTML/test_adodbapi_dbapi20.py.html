164      def test_nextset(self):
165          con = self._connect()
166          try:
167              cur = con.cursor()
168  
169              stmts = [self.ddl1] + self._populate()
170              for sql in stmts:
171                  cur.execute(sql)
172  
173              self.help_nextset_setUp(cur)
174  
175              cur.callproc("deleteme")
176              numberofrows = cur.fetchone()
177              assert numberofrows[0] == 6
178              assert cur.nextset()
179              names = cur.fetchall()
180              assert len(names) == len(self.samples)
181              s = cur.nextset()
182              assert s is None, "No more return sets, should return None"
183          finally:
184              try:
185                  self.help_nextset_tearDown(cur)
186              finally:
