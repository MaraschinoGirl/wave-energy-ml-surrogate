12   class CodeTests(unittest.TestCase):
13       """Check that bytecode.from_code(code).to_code() returns code."""
14   
15       def check(self, source, function=False):
16           ref_code = get_code(source, function=function)
17   
18           code = ConcreteBytecode.from_code(ref_code).to_code()
19           self.assertEqual(code, ref_code)
20   
21           code = Bytecode.from_code(ref_code).to_code()
22           self.assertEqual(code, ref_code)
23   
24           bytecode = Bytecode.from_code(ref_code)
25           blocks = ControlFlowGraph.from_bytecode(bytecode)
26           code = blocks.to_bytecode().to_code()
27           self.assertEqual(code, ref_code)
