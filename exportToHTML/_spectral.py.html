23   def cluster_qr(vectors):
24       """Find the discrete partition closest to the eigenvector embedding. 
25    
26           This implementation was proposed in [1]_. 
27    
28       .. versionadded:: 1.1 
29    
30           Parameters 
31           ---------- 
32           vectors : array-like, shape: (n_samples, n_clusters) 
33               The embedding space of the samples. 
34    
35           Returns 
36           ------- 
37           labels : array of integers, shape: n_samples 
38               The cluster labels of vectors. 
39    
40           References 
41           ---------- 
42           .. [1] :doi:`Simple, direct, and efficient multi-way spectral clustering, 2019 
43               Anil Damle, Victor Minden, Lexing Ying 
44               <10.1093/imaiai/iay008>` 
45    
46       """
47   
48       k = vectors.shape[1]
49       _, _, piv = qr(vectors.T, pivoting=True)
50       ut, _, v = svd(vectors[piv[:k], :].T)
51       vectors = abs(np.dot(vectors, np.dot(ut, v.conj())))
52       return vectors.argmax(axis=1)

55   def discretize(
56       vectors, *, copy=True, max_svd_restarts=30, n_iter_max=20, random_state=None
57   ):
58       """Search for a partition matrix which is closest to the eigenvector embedding. 
59    
60       This implementation was proposed in [1]_. 
61    
62       Parameters 
63       ---------- 
64       vectors : array-like of shape (n_samples, n_clusters) 
65           The embedding space of the samples. 
66    
67       copy : bool, default=True 
68           Whether to copy vectors, or perform in-place normalization. 
69    
70       max_svd_restarts : int, default=30 
71           Maximum number of attempts to restart SVD if convergence fails 
72    
73       n_iter_max : int, default=30 
74           Maximum number of iterations to attempt in rotation and partition 
75           matrix search if machine precision convergence is not reached 
76    
77       random_state : int, RandomState instance, default=None 
78           Determines random number generation for rotation matrix initialization. 
79           Use an int to make the randomness deterministic. 
80           See :term:`Glossary <random_state>`. 
81    
82       Returns 
83       ------- 
84       labels : array of integers, shape: n_samples 
85           The labels of the clusters. 
86    
87       References 
88       ---------- 
89    
90       .. [1] `Multiclass spectral clustering, 2003 
91              Stella X. Yu, Jianbo Shi 
92              <https://people.eecs.berkeley.edu/~jordan/courses/281B-spring04/readings/yu-shi.pdf>`_ 
93    
94       Notes 
95       ----- 
96    
97       The eigenvector embedding is used to iteratively search for the 
98       closest discrete partition.  First, the eigenvector embedding is 
99       normalized to the space of partition matrices. An optimal discrete 
100      partition matrix closest to this normalized embedding multiplied by 
101      an initial rotation is calculated.  Fixing this discrete partition 
102      matrix, an optimal rotation matrix is calculated.  These two 
103      calculations are performed until convergence.  The discrete partition 
104      matrix is returned as the clustering solution.  Used in spectral 
105      clustering, this method tends to be faster and more robust to random 
106      initialization than k-means. 
107   
108      """
109  
110      random_state = check_random_state(random_state)
111  
112      vectors = as_float_array(vectors, copy=copy)
113  
114      eps = np.finfo(float).eps
115      n_samples, n_components = vectors.shape
116  
117      # Normalize the eigenvectors to an equal length of a vector of ones.
118      # Reorient the eigenvectors to point in the negative direction with respect
119      # to the first element.  This may have to do with constraining the
120      # eigenvectors to lie in a specific quadrant to make the discretization
121      # search easier.
122      norm_ones = np.sqrt(n_samples)
123      for i in range(vectors.shape[1]):
124          vectors[:, i] = (vectors[:, i] / np.linalg.norm(vectors[:, i])) * norm_ones
125          if vectors[0, i] != 0:
126              vectors[:, i] = -1 * vectors[:, i] * np.sign(vectors[0, i])
127  
128      # Normalize the rows of the eigenvectors.  Samples should lie on the unit
129      # hypersphere centered at the origin.  This transforms the samples in the
130      # embedding space to the space of partition matrices.
131      vectors = vectors / np.sqrt((vectors**2).sum(axis=1))[:, np.newaxis]
132  
133      svd_restarts = 0
134      has_converged = False
135  
136      # If there is an exception we try to randomize and rerun SVD again
137      # do this max_svd_restarts times.
138      while (svd_restarts < max_svd_restarts) and not has_converged:
139          # Initialize first column of rotation matrix with a row of the
140          # eigenvectors
141          rotation = np.zeros((n_components, n_components))
142          rotation[:, 0] = vectors[random_state.randint(n_samples), :].T
143  
144          # To initialize the rest of the rotation matrix, find the rows
145          # of the eigenvectors that are as orthogonal to each other as
146          # possible
147          c = np.zeros(n_samples)
148          for j in range(1, n_components):
149              # Accumulate c to ensure row is as orthogonal as possible to
150              # previous picks as well as current one
151              c += np.abs(np.dot(vectors, rotation[:, j - 1]))
152              rotation[:, j] = vectors[c.argmin(), :].T
153  
154          last_objective_value = 0.0
155          n_iter = 0
156  
157          while not has_converged:
158              n_iter += 1
159  
160              t_discrete = np.dot(vectors, rotation)
161  
162              labels = t_discrete.argmax(axis=1)
163              vectors_discrete = csc_matrix(
164                  (np.ones(len(labels)), (np.arange(0, n_samples), labels)),
165                  shape=(n_samples, n_components),
166              )
167  
168              t_svd = vectors_discrete.T * vectors
169  
170              try:
171                  U, S, Vh = np.linalg.svd(t_svd)
172              except LinAlgError:
173                  svd_restarts += 1
174                  print("SVD did not converge, randomizing and trying again")
175                  break
176  
177              ncut_value = 2.0 * (n_samples - S.sum())
178              if (abs(ncut_value - last_objective_value) < eps) or (n_iter > n_iter_max):
179                  has_converged = True
180              else:
181                  # otherwise calculate rotation and continue
182                  last_objective_value = ncut_value
183                  rotation = np.dot(Vh.T, U.T)
184  
185      if not has_converged:
186          raise LinAlgError("SVD did not converge")
187      return labels

194  def spectral_clustering(
195      affinity,
196      *,
197      n_clusters=8,
198      n_components=None,
199      eigen_solver=None,
200      random_state=None,
201      n_init=10,
202      eigen_tol="auto",
203      assign_labels="kmeans",
204      verbose=False,
205  ):
206      """Apply clustering to a projection of the normalized Laplacian. 
207   
208      In practice Spectral Clustering is very useful when the structure of 
209      the individual clusters is highly non-convex or more generally when 
210      a measure of the center and spread of the cluster is not a suitable 
211      description of the complete cluster. For instance, when clusters are 
212      nested circles on the 2D plane. 
213   
214      If affinity is the adjacency matrix of a graph, this method can be 
215      used to find normalized graph cuts [1]_, [2]_. 
216   
217      Read more in the :ref:`User Guide <spectral_clustering>`. 
218   
219      Parameters 
220      ---------- 
221      affinity : {array-like, sparse matrix} of shape (n_samples, n_samples) 
222          The affinity matrix describing the relationship of the samples to 
223          embed. **Must be symmetric**. 
224   
225          Possible examples: 
226            - adjacency matrix of a graph, 
227            - heat kernel of the pairwise distance matrix of the samples, 
228            - symmetric k-nearest neighbours connectivity matrix of the samples. 
229   
230      n_clusters : int, default=None 
231          Number of clusters to extract. 
232   
233      n_components : int, default=n_clusters 
234          Number of eigenvectors to use for the spectral embedding. 
235   
236      eigen_solver : {None, 'arpack', 'lobpcg', or 'amg'} 
237          The eigenvalue decomposition method. If None then ``'arpack'`` is used. 
238          See [4]_ for more details regarding ``'lobpcg'``. 
239          Eigensolver ``'amg'`` runs ``'lobpcg'`` with optional 
240          Algebraic MultiGrid preconditioning and requires pyamg to be installed. 
241          It can be faster on very large sparse problems [6]_ and [7]_. 
242   
243      random_state : int, RandomState instance, default=None 
244          A pseudo random number generator used for the initialization 
245          of the lobpcg eigenvectors decomposition when `eigen_solver == 
246          'amg'`, and for the K-Means initialization. Use an int to make 
247          the results deterministic across calls (See 
248          :term:`Glossary <random_state>`). 
249   
250          .. note:: 
251              When using `eigen_solver == 'amg'`, 
252              it is necessary to also fix the global numpy seed with 
253              `np.random.seed(int)` to get deterministic results. See 
254              https://github.com/pyamg/pyamg/issues/139 for further 
255              information. 
256   
257      n_init : int, default=10 
258          Number of time the k-means algorithm will be run with different 
259          centroid seeds. The final results will be the best output of n_init 
260          consecutive runs in terms of inertia. Only used if 
261          ``assign_labels='kmeans'``. 
262   
263      eigen_tol : float, default="auto" 
264          Stopping criterion for eigendecomposition of the Laplacian matrix. 
265          If `eigen_tol="auto"` then the passed tolerance will depend on the 
266          `eigen_solver`: 
267   
268          - If `eigen_solver="arpack"`, then `eigen_tol=0.0`; 
269          - If `eigen_solver="lobpcg"` or `eigen_solver="amg"`, then 
270            `eigen_tol=None` which configures the underlying `lobpcg` solver to 
271            automatically resolve the value according to their heuristics. See, 
272            :func:`scipy.sparse.linalg.lobpcg` for details. 
273   
274          Note that when using `eigen_solver="lobpcg"` or `eigen_solver="amg"` 
275          values of `tol<1e-5` may lead to convergence issues and should be 
276          avoided. 
277   
278          .. versionadded:: 1.2 
279             Added 'auto' option. 
280   
281      assign_labels : {'kmeans', 'discretize', 'cluster_qr'}, default='kmeans' 
282          The strategy to use to assign labels in the embedding 
283          space.  There are three ways to assign labels after the Laplacian 
284          embedding.  k-means can be applied and is a popular choice. But it can 
285          also be sensitive to initialization. Discretization is another 
286          approach which is less sensitive to random initialization [3]_. 
287          The cluster_qr method [5]_ directly extracts clusters from eigenvectors 
288          in spectral clustering. In contrast to k-means and discretization, cluster_qr 
289          has no tuning parameters and is not an iterative method, yet may outperform 
290          k-means and discretization in terms of both quality and speed. For a detailed 
291          comparison of clustering strategies, refer to the following example: 
292          :ref:`sphx_glr_auto_examples_cluster_plot_coin_segmentation.py`. 
293   
294          .. versionchanged:: 1.1 
295             Added new labeling method 'cluster_qr'. 
296   
297      verbose : bool, default=False 
298          Verbosity mode. 
299   
300          .. versionadded:: 0.24 
301   
302      Returns 
303      ------- 
304      labels : array of integers, shape: n_samples 
305          The labels of the clusters. 
306   
307      Notes 
308      ----- 
309      The graph should contain only one connected component, elsewhere 
310      the results make little sense. 
311   
312      This algorithm solves the normalized cut for `k=2`: it is a 
313      normalized spectral clustering. 
314   
315      References 
316      ---------- 
317   
318      .. [1] :doi:`Normalized cuts and image segmentation, 2000 
319             Jianbo Shi, Jitendra Malik 
320             <10.1109/34.868688>` 
321   
322      .. [2] :doi:`A Tutorial on Spectral Clustering, 2007 
323             Ulrike von Luxburg 
324             <10.1007/s11222-007-9033-z>` 
325   
326      .. [3] `Multiclass spectral clustering, 2003 
327             Stella X. Yu, Jianbo Shi 
328             <https://people.eecs.berkeley.edu/~jordan/courses/281B-spring04/readings/yu-shi.pdf>`_ 
329   
330      .. [4] :doi:`Toward the Optimal Preconditioned Eigensolver: 
331             Locally Optimal Block Preconditioned Conjugate Gradient Method, 2001 
332             A. V. Knyazev 
333             SIAM Journal on Scientific Computing 23, no. 2, pp. 517-541. 
334             <10.1137/S1064827500366124>` 
335   
336      .. [5] :doi:`Simple, direct, and efficient multi-way spectral clustering, 2019 
337             Anil Damle, Victor Minden, Lexing Ying 
338             <10.1093/imaiai/iay008>` 
339   
340      .. [6] :doi:`Multiscale Spectral Image Segmentation Multiscale preconditioning 
341             for computing eigenvalues of graph Laplacians in image segmentation, 2006 
342             Andrew Knyazev 
343             <10.13140/RG.2.2.35280.02565>` 
344   
345      .. [7] :doi:`Preconditioned spectral clustering for stochastic block partition 
346             streaming graph challenge (Preliminary version at arXiv.) 
347             David Zhuzhunashvili, Andrew Knyazev 
348             <10.1109/HPEC.2017.8091045>` 
349   
350      Examples 
351      -------- 
352      >>> import numpy as np 
353      >>> from sklearn.metrics.pairwise import pairwise_kernels 
354      >>> from sklearn.cluster import spectral_clustering 
355      >>> X = np.array([[1, 1], [2, 1], [1, 0], 
356      ...               [4, 7], [3, 5], [3, 6]]) 
357      >>> affinity = pairwise_kernels(X, metric='rbf') 
358      >>> spectral_clustering( 
359      ...     affinity=affinity, n_clusters=2, assign_labels="discretize", random_state=0 
360      ... ) 
361      array([1, 1, 1, 0, 0, 0]) 
362      """
363  
364      clusterer = SpectralClustering(
365          n_clusters=n_clusters,
366          n_components=n_components,
367          eigen_solver=eigen_solver,
368          random_state=random_state,
369          n_init=n_init,
370          affinity="precomputed",
371          eigen_tol=eigen_tol,
372          assign_labels=assign_labels,
373          verbose=verbose,
374      ).fit(affinity)
375  
376      return clusterer.labels_

379  class SpectralClustering(ClusterMixin, BaseEstimator):
380      """Apply clustering to a projection of the normalized Laplacian. 
381   
382      In practice Spectral Clustering is very useful when the structure of 
383      the individual clusters is highly non-convex, or more generally when 
384      a measure of the center and spread of the cluster is not a suitable 
385      description of the complete cluster, such as when clusters are 
386      nested circles on the 2D plane. 
387   
388      If the affinity matrix is the adjacency matrix of a graph, this method 
389      can be used to find normalized graph cuts [1]_, [2]_. 
390   
391      When calling ``fit``, an affinity matrix is constructed using either 
392      a kernel function such the Gaussian (aka RBF) kernel with Euclidean 
393      distance ``d(X, X)``:: 
394   
395              np.exp(-gamma * d(X,X) ** 2) 
396   
397      or a k-nearest neighbors connectivity matrix. 
398   
399      Alternatively, a user-provided affinity matrix can be specified by 
400      setting ``affinity='precomputed'``. 
401   
402      Read more in the :ref:`User Guide <spectral_clustering>`. 
403   
404      Parameters 
405      ---------- 
406      n_clusters : int, default=8 
407          The dimension of the projection subspace. 
408   
409      eigen_solver : {'arpack', 'lobpcg', 'amg'}, default=None 
410          The eigenvalue decomposition strategy to use. AMG requires pyamg 
411          to be installed. It can be faster on very large, sparse problems, 
412          but may also lead to instabilities. If None, then ``'arpack'`` is 
413          used. See [4]_ for more details regarding `'lobpcg'`. 
414   
415      n_components : int, default=None 
416          Number of eigenvectors to use for the spectral embedding. If None, 
417          defaults to `n_clusters`. 
418   
419      random_state : int, RandomState instance, default=None 
420          A pseudo random number generator used for the initialization 
421          of the lobpcg eigenvectors decomposition when `eigen_solver == 
422          'amg'`, and for the K-Means initialization. Use an int to make 
423          the results deterministic across calls (See 
424          :term:`Glossary <random_state>`). 
425   
426          .. note:: 
427              When using `eigen_solver == 'amg'`, 
428              it is necessary to also fix the global numpy seed with 
429              `np.random.seed(int)` to get deterministic results. See 
430              https://github.com/pyamg/pyamg/issues/139 for further 
431              information. 
432   
433      n_init : int, default=10 
434          Number of time the k-means algorithm will be run with different 
435          centroid seeds. The final results will be the best output of n_init 
436          consecutive runs in terms of inertia. Only used if 
437          ``assign_labels='kmeans'``. 
438   
439      gamma : float, default=1.0 
440          Kernel coefficient for rbf, poly, sigmoid, laplacian and chi2 kernels. 
441          Ignored for ``affinity='nearest_neighbors'``, ``affinity='precomputed'`` 
442          or ``affinity='precomputed_nearest_neighbors'``. 
443   
444      affinity : str or callable, default='rbf' 
445          How to construct the affinity matrix. 
446           - 'nearest_neighbors': construct the affinity matrix by computing a 
447             graph of nearest neighbors. 
448           - 'rbf': construct the affinity matrix using a radial basis function 
449             (RBF) kernel. 
450           - 'precomputed': interpret ``X`` as a precomputed affinity matrix, 
451             where larger values indicate greater similarity between instances. 
452           - 'precomputed_nearest_neighbors': interpret ``X`` as a sparse graph 
453             of precomputed distances, and construct a binary affinity matrix 
454             from the ``n_neighbors`` nearest neighbors of each instance. 
455           - one of the kernels supported by 
456             :func:`~sklearn.metrics.pairwise.pairwise_kernels`. 
457   
458          Only kernels that produce similarity scores (non-negative values that 
459          increase with similarity) should be used. This property is not checked 
460          by the clustering algorithm. 
461   
462      n_neighbors : int, default=10 
463          Number of neighbors to use when constructing the affinity matrix using 
464          the nearest neighbors method. Ignored for ``affinity='rbf'``. 
465   
466      eigen_tol : float, default="auto" 
467          Stopping criterion for eigen decomposition of the Laplacian matrix. 
468          If `eigen_tol="auto"` then the passed tolerance will depend on the 
469          `eigen_solver`: 
470   
471          - If `eigen_solver="arpack"`, then `eigen_tol=0.0`; 
472          - If `eigen_solver="lobpcg"` or `eigen_solver="amg"`, then 
473            `eigen_tol=None` which configures the underlying `lobpcg` solver to 
474            automatically resolve the value according to their heuristics. See, 
475            :func:`scipy.sparse.linalg.lobpcg` for details. 
476   
477          Note that when using `eigen_solver="lobpcg"` or `eigen_solver="amg"` 
478          values of `tol<1e-5` may lead to convergence issues and should be 
479          avoided. 
480   
481          .. versionadded:: 1.2 
482             Added 'auto' option. 
483   
484      assign_labels : {'kmeans', 'discretize', 'cluster_qr'}, default='kmeans' 
485          The strategy for assigning labels in the embedding space. There are two 
486          ways to assign labels after the Laplacian embedding. k-means is a 
487          popular choice, but it can be sensitive to initialization. 
488          Discretization is another approach which is less sensitive to random 
489          initialization [3]_. 
490          The cluster_qr method [5]_ directly extract clusters from eigenvectors 
491          in spectral clustering. In contrast to k-means and discretization, cluster_qr 
492          has no tuning parameters and runs no iterations, yet may outperform 
493          k-means and discretization in terms of both quality and speed. 
494   
495          .. versionchanged:: 1.1 
496             Added new labeling method 'cluster_qr'. 
497   
498      degree : float, default=3 
499          Degree of the polynomial kernel. Ignored by other kernels. 
500   
501      coef0 : float, default=1 
502          Zero coefficient for polynomial and sigmoid kernels. 
503          Ignored by other kernels. 
504   
505      kernel_params : dict of str to any, default=None 
506          Parameters (keyword arguments) and values for kernel passed as 
507          callable object. Ignored by other kernels. 
508   
509      n_jobs : int, default=None 
510          The number of parallel jobs to run when `affinity='nearest_neighbors'` 
511          or `affinity='precomputed_nearest_neighbors'`. The neighbors search 
512          will be done in parallel. 
513          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
514          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
515          for more details. 
516   
517      verbose : bool, default=False 
518          Verbosity mode. 
519   
520          .. versionadded:: 0.24 
521   
522      Attributes 
523      ---------- 
524      affinity_matrix_ : array-like of shape (n_samples, n_samples) 
525          Affinity matrix used for clustering. Available only after calling 
526          ``fit``. 
527   
528      labels_ : ndarray of shape (n_samples,) 
529          Labels of each point 
530   
531      n_features_in_ : int 
532          Number of features seen during :term:`fit`. 
533   
534          .. versionadded:: 0.24 
535   
536      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
537          Names of features seen during :term:`fit`. Defined only when `X` 
538          has feature names that are all strings. 
539   
540          .. versionadded:: 1.0 
541   
542      See Also 
543      -------- 
544      sklearn.cluster.KMeans : K-Means clustering. 
545      sklearn.cluster.DBSCAN : Density-Based Spatial Clustering of 
546          Applications with Noise. 
547   
548      Notes 
549      ----- 
550      A distance matrix for which 0 indicates identical elements and high values 
551      indicate very dissimilar elements can be transformed into an affinity / 
552      similarity matrix that is well-suited for the algorithm by 
553      applying the Gaussian (aka RBF, heat) kernel:: 
554   
555          np.exp(- dist_matrix ** 2 / (2. * delta ** 2)) 
556   
557      where ``delta`` is a free parameter representing the width of the Gaussian 
558      kernel. 
559   
560      An alternative is to take a symmetric version of the k-nearest neighbors 
561      connectivity matrix of the points. 
562   
563      If the pyamg package is installed, it is used: this greatly 
564      speeds up computation. 
565   
566      References 
567      ---------- 
568      .. [1] :doi:`Normalized cuts and image segmentation, 2000 
569             Jianbo Shi, Jitendra Malik 
570             <10.1109/34.868688>` 
571   
572      .. [2] :doi:`A Tutorial on Spectral Clustering, 2007 
573             Ulrike von Luxburg 
574             <10.1007/s11222-007-9033-z>` 
575   
576      .. [3] `Multiclass spectral clustering, 2003 
577             Stella X. Yu, Jianbo Shi 
578             <https://people.eecs.berkeley.edu/~jordan/courses/281B-spring04/readings/yu-shi.pdf>`_ 
579   
580      .. [4] :doi:`Toward the Optimal Preconditioned Eigensolver: 
581             Locally Optimal Block Preconditioned Conjugate Gradient Method, 2001 
582             A. V. Knyazev 
583             SIAM Journal on Scientific Computing 23, no. 2, pp. 517-541. 
584             <10.1137/S1064827500366124>` 
585   
586      .. [5] :doi:`Simple, direct, and efficient multi-way spectral clustering, 2019 
587             Anil Damle, Victor Minden, Lexing Ying 
588             <10.1093/imaiai/iay008>`

590      Examples 
591      -------- 
592      >>> from sklearn.cluster import SpectralClustering 
593      >>> import numpy as np 
594      >>> X = np.array([[1, 1], [2, 1], [1, 0], 
595      ...               [4, 7], [3, 5], [3, 6]]) 
596      >>> clustering = SpectralClustering(n_clusters=2, 
597      ...         assign_labels='discretize', 
598      ...         random_state=0).fit(X) 
599      >>> clustering.labels_ 
600      array([1, 1, 1, 0, 0, 0]) 
601      >>> clustering 
602      SpectralClustering(assign_labels='discretize', n_clusters=2, 
603          random_state=0) 
604      """
605  
606      _parameter_constraints: dict = {
607          "n_clusters": [Interval(Integral, 1, None, closed="left")],
608          "eigen_solver": [StrOptions({"arpack", "lobpcg", "amg"}), None],
609          "n_components": [Interval(Integral, 1, None, closed="left"), None],
610          "random_state": ["random_state"],
611          "n_init": [Interval(Integral, 1, None, closed="left")],
612          "gamma": [Interval(Real, 0, None, closed="left")],
613          "affinity": [
614              callable,
615              StrOptions(
616                  set(KERNEL_PARAMS)
617                  | {"nearest_neighbors", "precomputed", "precomputed_nearest_neighbors"}
618              ),
619          ],
620          "n_neighbors": [Interval(Integral, 1, None, closed="left")],
621          "eigen_tol": [
622              Interval(Real, 0.0, None, closed="left"),
623              StrOptions({"auto"}),
624          ],
625          "assign_labels": [StrOptions({"kmeans", "discretize", "cluster_qr"})],

668      @_fit_context(prefer_skip_nested_validation=True)
669      def fit(self, X, y=None):
670          """Perform spectral clustering from features, or affinity matrix. 
671   
672          Parameters 
673          ---------- 
674          X : {array-like, sparse matrix} of shape (n_samples, n_features) or \ 
675                  (n_samples, n_samples) 
676              Training instances to cluster, similarities / affinities between 
677              instances if ``affinity='precomputed'``, or distances between 
678              instances if ``affinity='precomputed_nearest_neighbors``. If a 
679              sparse matrix is provided in a format other than ``csr_matrix``, 
680              ``csc_matrix``, or ``coo_matrix``, it will be converted into a 
681              sparse ``csr_matrix``. 
682   
683          y : Ignored 
684              Not used, present here for API consistency by convention. 
685   
686          Returns 
687          ------- 
688          self : object 
689              A fitted instance of the estimator. 
690          """
691          X = validate_data(
692              self,
693              X,
694              accept_sparse=["csr", "csc", "coo"],
695              dtype=np.float64,
696              ensure_min_samples=2,
697          )
698          allow_squared = self.affinity in [
699              "precomputed",
700              "precomputed_nearest_neighbors",
701          ]
702          if X.shape[0] == X.shape[1] and not allow_squared:
703              warnings.warn(
704                  "The spectral clustering API has changed. ``fit``"
705                  "now constructs an affinity matrix from data. To use"
706                  " a custom affinity matrix, "
707                  "set ``affinity=precomputed``."
708              )
709  
710          if self.affinity == "nearest_neighbors":
711              connectivity = kneighbors_graph(
712                  X, n_neighbors=self.n_neighbors, include_self=True, n_jobs=self.n_jobs
713              )
714              self.affinity_matrix_ = 0.5 * (connectivity + connectivity.T)
715          elif self.affinity == "precomputed_nearest_neighbors":
716              estimator = NearestNeighbors(
717                  n_neighbors=self.n_neighbors, n_jobs=self.n_jobs, metric="precomputed"
718              ).fit(X)
719              connectivity = estimator.kneighbors_graph(X=X, mode="connectivity")
720              self.affinity_matrix_ = 0.5 * (connectivity + connectivity.T)
721          elif self.affinity == "precomputed":
722              self.affinity_matrix_ = X
723          else:
724              params = self.kernel_params
725              if params is None:
726                  params = {}
727              if not callable(self.affinity):
728                  params["gamma"] = self.gamma
729                  params["degree"] = self.degree
730                  params["coef0"] = self.coef0
731              self.affinity_matrix_ = pairwise_kernels(
732                  X, metric=self.affinity, filter_params=True, **params
733              )
734  
735          random_state = check_random_state(self.random_state)
736          n_components = (
737              self.n_clusters if self.n_components is None else self.n_components
738          )
739          # We now obtain the real valued solution matrix to the
740          # relaxed Ncut problem, solving the eigenvalue problem
741          # L_sym x = lambda x  and recovering u = D^-1/2 x.
742          # The first eigenvector is constant only for fully connected graphs
743          # and should be kept for spectral clustering (drop_first = False)
744          # See spectral_embedding documentation.
745          maps = _spectral_embedding(
746              self.affinity_matrix_,
747              n_components=n_components,
748              eigen_solver=self.eigen_solver,
749              random_state=random_state,
750              eigen_tol=self.eigen_tol,
751              drop_first=False,
752          )
753          if self.verbose:
754              print(f"Computing label assignment using {self.assign_labels}")
755  
756          if self.assign_labels == "kmeans":
757              _, self.labels_, _ = k_means(
758                  maps,
759                  self.n_clusters,
760                  random_state=random_state,
761                  n_init=self.n_init,
762                  verbose=self.verbose,
763              )
764          elif self.assign_labels == "cluster_qr":
765              self.labels_ = cluster_qr(maps)
766          else:
767              self.labels_ = discretize(maps, random_state=random_state)
768  
769          return self
770  
771      def fit_predict(self, X, y=None):
772          """Perform spectral clustering on `X` and return cluster labels. 
773   
774          Parameters 
775          ---------- 
776          X : {array-like, sparse matrix} of shape (n_samples, n_features) or \ 
777                  (n_samples, n_samples) 
778              Training instances to cluster, similarities / affinities between 
779              instances if ``affinity='precomputed'``, or distances between 
780              instances if ``affinity='precomputed_nearest_neighbors``. If a 
781              sparse matrix is provided in a format other than ``csr_matrix``, 
782              ``csc_matrix``, or ``coo_matrix``, it will be converted into a 
783              sparse ``csr_matrix``. 
784   
785          y : Ignored 
786              Not used, present here for API consistency by convention. 
787   
788          Returns 
789          ------- 
790          labels : ndarray of shape (n_samples,) 
791              Cluster labels. 
792          """
793          return super().fit_predict(X, y)
