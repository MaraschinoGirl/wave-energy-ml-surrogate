29   class NearestCentroid(
30       DiscriminantAnalysisPredictionMixin, ClassifierMixin, BaseEstimator
31   ):
32       """Nearest centroid classifier. 
33    
34       Each class is represented by its centroid, with test samples classified to 
35       the class with the nearest centroid. 
36    
37       Read more in the :ref:`User Guide <nearest_centroid_classifier>`. 
38    
39       Parameters 
40       ---------- 
41       metric : {"euclidean", "manhattan"}, default="euclidean" 
42           Metric to use for distance computation. 
43    
44           If `metric="euclidean"`, the centroid for the samples corresponding to each 
45           class is the arithmetic mean, which minimizes the sum of squared L1 distances. 
46           If `metric="manhattan"`, the centroid is the feature-wise median, which 
47           minimizes the sum of L1 distances. 
48    
49           .. versionchanged:: 1.5 
50               All metrics but `"euclidean"` and `"manhattan"` were deprecated and 
51               now raise an error. 
52    
53           .. versionchanged:: 0.19 
54               `metric='precomputed'` was deprecated and now raises an error 
55    
56       shrink_threshold : float, default=None 
57           Threshold for shrinking centroids to remove features. 
58    
59       priors : {"uniform", "empirical"} or array-like of shape (n_classes,), \ 
60           default="uniform" 
61           The class prior probabilities. By default, the class proportions are 
62           inferred from the training data. 
63    
64           .. versionadded:: 1.6 
65    
66       Attributes 
67       ---------- 
68       centroids_ : array-like of shape (n_classes, n_features) 
69           Centroid of each class. 
70    
71       classes_ : array of shape (n_classes,) 
72           The unique classes labels. 
73    
74       n_features_in_ : int 
75           Number of features seen during :term:`fit`. 
76    
77           .. versionadded:: 0.24 
78    
79       feature_names_in_ : ndarray of shape (`n_features_in_`,) 
80           Names of features seen during :term:`fit`. Defined only when `X` 
81           has feature names that are all strings. 
82    
83           .. versionadded:: 1.0 
84    
85       deviations_ : ndarray of shape (n_classes, n_features) 
86           Deviations (or shrinkages) of the centroids of each class from the 
87           overall centroid. Equal to eq. (18.4) if `shrink_threshold=None`, 
88           else (18.5) p. 653 of [2]. Can be used to identify features used 
89           for classification. 
90    
91           .. versionadded:: 1.6 
92    
93       within_class_std_dev_ : ndarray of shape (n_features,) 
94           Pooled or within-class standard deviation of input data. 
95    
96           .. versionadded:: 1.6 
97    
98       class_prior_ : ndarray of shape (n_classes,) 
99           The class prior probabilities. 
100   
101          .. versionadded:: 1.6 
102   
103      See Also 
104      -------- 
105      KNeighborsClassifier : Nearest neighbors classifier. 
106   
107      Notes 
108      ----- 
109      When used for text classification with tf-idf vectors, this classifier is 
110      also known as the Rocchio classifier. 
111   
112      References 
113      ---------- 
114      [1] Tibshirani, R., Hastie, T., Narasimhan, B., & Chu, G. (2002). Diagnosis of 
115      multiple cancer types by shrunken centroids of gene expression. Proceedings 
116      of the National Academy of Sciences of the United States of America, 
117      99(10), 6567-6572. The National Academy of Sciences. 
118   
119      [2] Hastie, T., Tibshirani, R., Friedman, J. (2009). The Elements of Statistical 
120      Learning Data Mining, Inference, and Prediction. 2nd Edition. New York, Springer. 
121   
122      Examples 
123      -------- 
124      >>> from sklearn.neighbors import NearestCentroid 
125      >>> import numpy as np 
126      >>> X = np.array([[-1, -1], [-2, -1], [-3, -2], [1, 1], [2, 1], [3, 2]]) 
127      >>> y = np.array([1, 1, 1, 2, 2, 2]) 
128      >>> clf = NearestCentroid() 
129      >>> clf.fit(X, y) 
130      NearestCentroid() 
131      >>> print(clf.predict([[-0.8, -1]])) 
132      [1] 
133      """
134  
135      _parameter_constraints: dict = {
136          "metric": [StrOptions({"manhattan", "euclidean"})],
137          "shrink_threshold": [Interval(Real, 0, None, closed="neither"), None],
138          "priors": ["array-like", StrOptions({"empirical", "uniform"})],
139      }
140  
141      def __init__(
142          self,
143          metric="euclidean",
144          *,
145          shrink_threshold=None,
146          priors="uniform",
147      ):
148          self.metric = metric
149          self.shrink_threshold = shrink_threshold
150          self.priors = priors
151  
152      @_fit_context(prefer_skip_nested_validation=True)
153      def fit(self, X, y):
154          """ 
155          Fit the NearestCentroid model according to the given training data. 
156   
157          Parameters 
158          ---------- 
159          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
160              Training vector, where `n_samples` is the number of samples and 
161              `n_features` is the number of features. 
162              Note that centroid shrinking cannot be used with sparse matrices. 
163          y : array-like of shape (n_samples,) 
164              Target values. 
165   
166          Returns 
167          ------- 
168          self : object 
169              Fitted estimator. 
170          """
171          # If X is sparse and the metric is "manhattan", store it in a csc
172          # format is easier to calculate the median.
173          if self.metric == "manhattan":
174              X, y = validate_data(self, X, y, accept_sparse=["csc"])
175          else:
176              ensure_all_finite = (
177                  "allow-nan" if get_tags(self).input_tags.allow_nan else True
178              )
179              X, y = validate_data(
180                  self,
181                  X,
182                  y,
183                  ensure_all_finite=ensure_all_finite,
184                  accept_sparse=["csr", "csc"],
185              )
186          is_X_sparse = sp.issparse(X)
187          check_classification_targets(y)
188  
189          n_samples, n_features = X.shape
190          le = LabelEncoder()
191          y_ind = le.fit_transform(y)
192          self.classes_ = classes = le.classes_
193          n_classes = classes.size
194          if n_classes < 2:
195              raise ValueError(
196                  "The number of classes has to be greater than one; got %d class"
197                  % (n_classes)
198              )
199  
200          if self.priors == "empirical":  # estimate priors from sample
201              _, class_counts = np.unique(y, return_inverse=True)  # non-negative ints
202              self.class_prior_ = np.bincount(class_counts) / float(len(y))
203          elif self.priors == "uniform":
204              self.class_prior_ = np.asarray([1 / n_classes] * n_classes)
205          else:
206              self.class_prior_ = np.asarray(self.priors)
207  
208          if (self.class_prior_ < 0).any():
209              raise ValueError("priors must be non-negative")
210          if not np.isclose(self.class_prior_.sum(), 1.0):
211              warnings.warn(
212                  "The priors do not sum to 1. Normalizing such that it sums to one.",
213                  UserWarning,
214              )
215              self.class_prior_ = self.class_prior_ / self.class_prior_.sum()
216  
217          # Mask mapping each class to its members.
218          self.centroids_ = np.empty((n_classes, n_features), dtype=np.float64)
219  
220          # Number of clusters in each class.
221          nk = np.zeros(n_classes)
222  
223          for cur_class in range(n_classes):
224              center_mask = y_ind == cur_class
225              nk[cur_class] = np.sum(center_mask)
226              if is_X_sparse:
227                  center_mask = np.where(center_mask)[0]
228  
229              if self.metric == "manhattan":
230                  # NumPy does not calculate median of sparse matrices.
231                  if not is_X_sparse:
232                      self.centroids_[cur_class] = np.median(X[center_mask], axis=0)
233                  else:
234                      self.centroids_[cur_class] = csc_median_axis_0(X[center_mask])
235              else:  # metric == "euclidean"
236                  self.centroids_[cur_class] = X[center_mask].mean(axis=0)
237  
238          # Compute within-class std_dev with unshrunked centroids
239          variance = np.array(X - self.centroids_[y_ind], copy=False) ** 2
240          self.within_class_std_dev_ = np.array(
241              np.sqrt(variance.sum(axis=0) / (n_samples - n_classes)), copy=False
242          )
243          if any(self.within_class_std_dev_ == 0):
244              warnings.warn(
245                  "self.within_class_std_dev_ has at least 1 zero standard deviation."
246                  "Inputs within the same classes for at least 1 feature are identical."
247              )
248  
249          err_msg = "All features have zero variance. Division by zero."
250          if is_X_sparse and np.all((X.max(axis=0) - X.min(axis=0)).toarray() == 0):
251              raise ValueError(err_msg)
252          elif not is_X_sparse and np.all(np.ptp(X, axis=0) == 0):
253              raise ValueError(err_msg)
254  
255          dataset_centroid_ = X.mean(axis=0)
256          # m parameter for determining deviation
257          m = np.sqrt((1.0 / nk) - (1.0 / n_samples))
258          # Calculate deviation using the standard deviation of centroids.
259          # To deter outliers from affecting the results.
260          s = self.within_class_std_dev_ + np.median(self.within_class_std_dev_)
261          mm = m.reshape(len(m), 1)  # Reshape to allow broadcasting.
262          ms = mm * s
263          self.deviations_ = np.array(
264              (self.centroids_ - dataset_centroid_) / ms, copy=False
265          )
266          # Soft thresholding: if the deviation crosses 0 during shrinking,
267          # it becomes zero.
268          if self.shrink_threshold:
269              signs = np.sign(self.deviations_)
270              self.deviations_ = np.abs(self.deviations_) - self.shrink_threshold
271              np.clip(self.deviations_, 0, None, out=self.deviations_)
272              self.deviations_ *= signs
273              # Now adjust the centroids using the deviation
274              msd = ms * self.deviations_
275              self.centroids_ = np.array(dataset_centroid_ + msd, copy=False)
276          return self
277  
278      def predict(self, X):
279          """Perform classification on an array of test vectors `X`. 
280   
281          The predicted class `C` for each sample in `X` is returned. 
282   
283          Parameters 
284          ---------- 
285          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
286              Input data. 
287   
288          Returns 
289          ------- 
290          y_pred : ndarray of shape (n_samples,) 
291              The predicted classes. 
292          """
293          check_is_fitted(self)
294          if np.isclose(self.class_prior_, 1 / len(self.classes_)).all():
295              # `validate_data` is called here since we are not calling `super()`
296              ensure_all_finite = (
297                  "allow-nan" if get_tags(self).input_tags.allow_nan else True
298              )
299              X = validate_data(
300                  self,
301                  X,
302                  ensure_all_finite=ensure_all_finite,
303                  accept_sparse="csr",
304                  reset=False,
305              )
306              return self.classes_[
307                  pairwise_distances_argmin(X, self.centroids_, metric=self.metric)
308              ]
309          else:
310              return super().predict(X)
311  
312      def _decision_function(self, X):
313          # return discriminant scores, see eq. (18.2) p. 652 of the ESL.
314          check_is_fitted(self, "centroids_")
315  
316          X_normalized = validate_data(
317              self, X, copy=True, reset=False, accept_sparse="csr", dtype=np.float64
318          )
319  
320          discriminant_score = np.empty(
321              (X_normalized.shape[0], self.classes_.size), dtype=np.float64
322          )
323  
324          mask = self.within_class_std_dev_ != 0
325          X_normalized[:, mask] /= self.within_class_std_dev_[mask]
326          centroids_normalized = self.centroids_.copy()
327          centroids_normalized[:, mask] /= self.within_class_std_dev_[mask]
328  
329          for class_idx in range(self.classes_.size):
330              distances = pairwise_distances(
331                  X_normalized, centroids_normalized[[class_idx]], metric=self.metric
332              ).ravel()
333              distances **= 2
334              discriminant_score[:, class_idx] = np.squeeze(
335                  -distances + 2.0 * np.log(self.class_prior_[class_idx])
336              )
337  
338          return discriminant_score
339  
340      def _check_euclidean_metric(self):
341          return self.metric == "euclidean"
