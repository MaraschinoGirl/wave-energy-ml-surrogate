7    class TestGetSetClipboardData(unittest.TestCase):
8        def copyData(self, data, format_):
9            win32clipboard.OpenClipboard()
10           ret = None
11           try:
12               win32clipboard.SetClipboardData(format_, data)
13               ret = win32clipboard.GetClipboardData(format_)
14           finally:
15               win32clipboard.CloseClipboard()
16           return ret
17   
18       def copyText(self, data, format_):
19           win32clipboard.OpenClipboard()
20           ret = None
21           try:
22               win32clipboard.SetClipboardText(data, format_)
23               ret = win32clipboard.GetClipboardData(format_)
24           finally:
25               win32clipboard.CloseClipboard()
26           return ret
27   
28       def test_data(self):
29           test_data = {
30               "Dummy str": win32clipboard.CF_UNICODETEXT,
31               b"Dummy bytes text": win32clipboard.CF_TEXT,
32               b"Dummy\x00\xff bytes": win32clipboard.CF_DIB,
33           }
34           for data, fmt in test_data.items():
35               self.assertEqual(data, self.copyData(data, fmt))
36           test_data = {
37               "Dummy str": (win32clipboard.CF_TEXT, win32clipboard.CF_DIB),
38               b"Dummy\x00\xff bytes": (win32clipboard.CF_UNICODETEXT,),
39           }
40           for data, formats in test_data.items():
41               for fmt in formats:
42                   self.assertNotEqual(data, self.copyData(data, fmt))

45           test_data = {
