26   class BayesianRidge(RegressorMixin, LinearModel):
27       """Bayesian ridge regression. 
28    
29       Fit a Bayesian ridge model. See the Notes section for details on this 
30       implementation and the optimization of the regularization parameters 
31       lambda (precision of the weights) and alpha (precision of the noise). 
32    
33       Read more in the :ref:`User Guide <bayesian_regression>`. 
34       For an intuitive visualization of how the sinusoid is approximated by 
35       a polynomial using different pairs of initial values, see 
36       :ref:`sphx_glr_auto_examples_linear_model_plot_bayesian_ridge_curvefit.py`. 
37    
38       Parameters 
39       ---------- 
40       max_iter : int, default=300 
41           Maximum number of iterations over the complete dataset before 
42           stopping independently of any early stopping criterion. 
43    
44           .. versionchanged:: 1.3 
45    
46       tol : float, default=1e-3 
47           Stop the algorithm if w has converged. 
48    
49       alpha_1 : float, default=1e-6 
50           Hyper-parameter : shape parameter for the Gamma distribution prior 
51           over the alpha parameter. 
52    
53       alpha_2 : float, default=1e-6 
54           Hyper-parameter : inverse scale parameter (rate parameter) for the 
55           Gamma distribution prior over the alpha parameter. 
56    
57       lambda_1 : float, default=1e-6 
58           Hyper-parameter : shape parameter for the Gamma distribution prior 
59           over the lambda parameter. 
60    
61       lambda_2 : float, default=1e-6 
62           Hyper-parameter : inverse scale parameter (rate parameter) for the 
63           Gamma distribution prior over the lambda parameter. 
64    
65       alpha_init : float, default=None 
66           Initial value for alpha (precision of the noise). 
67           If not set, alpha_init is 1/Var(y). 
68    
69           .. versionadded:: 0.22 
70    
71       lambda_init : float, default=None 
72           Initial value for lambda (precision of the weights). 
73           If not set, lambda_init is 1. 
74    
75           .. versionadded:: 0.22 
76    
77       compute_score : bool, default=False 
78           If True, compute the log marginal likelihood at each iteration of the 
79           optimization. 
80    
81       fit_intercept : bool, default=True 
82           Whether to calculate the intercept for this model. 
83           The intercept is not treated as a probabilistic parameter 
84           and thus has no associated variance. If set 
85           to False, no intercept will be used in calculations 
86           (i.e. data is expected to be centered). 
87    
88       copy_X : bool, default=True 
89           If True, X will be copied; else, it may be overwritten. 
90    
91       verbose : bool, default=False 
92           Verbose mode when fitting the model. 
93    
94       Attributes 
95       ---------- 
96       coef_ : array-like of shape (n_features,) 
97           Coefficients of the regression model (mean of distribution) 
98    
99       intercept_ : float 
100          Independent term in decision function. Set to 0.0 if 
101          `fit_intercept = False`. 
102   
103      alpha_ : float 
104         Estimated precision of the noise. 
105   
106      lambda_ : float 
107         Estimated precision of the weights. 
108   
109      sigma_ : array-like of shape (n_features, n_features) 
110          Estimated variance-covariance matrix of the weights 
111   
112      scores_ : array-like of shape (n_iter_+1,) 
113          If computed_score is True, value of the log marginal likelihood (to be 
114          maximized) at each iteration of the optimization. The array starts 
115          with the value of the log marginal likelihood obtained for the initial 
116          values of alpha and lambda and ends with the value obtained for the 
117          estimated alpha and lambda. 
118   
119      n_iter_ : int 
120          The actual number of iterations to reach the stopping criterion. 
121   
122      X_offset_ : ndarray of shape (n_features,) 
123          If `fit_intercept=True`, offset subtracted for centering data to a 
124          zero mean. Set to np.zeros(n_features) otherwise. 
125   
126      X_scale_ : ndarray of shape (n_features,) 
127          Set to np.ones(n_features). 
128   
129      n_features_in_ : int 
130          Number of features seen during :term:`fit`. 
131   
132          .. versionadded:: 0.24 
133   
134      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
135          Names of features seen during :term:`fit`. Defined only when `X` 
136          has feature names that are all strings. 
137   
138          .. versionadded:: 1.0 
139   
140      See Also 
141      -------- 
142      ARDRegression : Bayesian ARD regression. 
143   
144      Notes 
145      ----- 
146      There exist several strategies to perform Bayesian ridge regression. This 
147      implementation is based on the algorithm described in Appendix A of 
148      (Tipping, 2001) where updates of the regularization parameters are done as 
149      suggested in (MacKay, 1992). Note that according to A New 
150      View of Automatic Relevance Determination (Wipf and Nagarajan, 2008) these 
151      update rules do not guarantee that the marginal likelihood is increasing 
152      between two consecutive iterations of the optimization. 
153   
154      References 
155      ---------- 
156      D. J. C. MacKay, Bayesian Interpolation, Computation and Neural Systems, 
157      Vol. 4, No. 3, 1992. 
158   
159      M. E. Tipping, Sparse Bayesian Learning and the Relevance Vector Machine, 
160      Journal of Machine Learning Research, Vol. 1, 2001. 
161   
162      Examples 
163      -------- 
164      >>> from sklearn import linear_model 
165      >>> clf = linear_model.BayesianRidge() 
166      >>> clf.fit([[0,0], [1, 1], [2, 2]], [0, 1, 2]) 
167      BayesianRidge() 
168      >>> clf.predict([[1, 1]]) 
169      array([1.]) 
170      """
171

216      @_fit_context(prefer_skip_nested_validation=True)
217      def fit(self, X, y, sample_weight=None):
218          """Fit the model. 
219   
220          Parameters 
221          ---------- 
222          X : ndarray of shape (n_samples, n_features) 
223              Training data. 
224          y : ndarray of shape (n_samples,) 
225              Target values. Will be cast to X's dtype if necessary. 
226   
227          sample_weight : ndarray of shape (n_samples,), default=None 
228              Individual weights for each sample. 
229   
230              .. versionadded:: 0.20 
231                 parameter *sample_weight* support to BayesianRidge. 
232   
233          Returns 
234          ------- 
235          self : object 
236              Returns the instance itself. 
237          """
238          X, y = validate_data(
239              self,
240              X,
241              y,
242              dtype=[np.float64, np.float32],
243              force_writeable=True,
244              y_numeric=True,
245          )
246          dtype = X.dtype
247  
248          if sample_weight is not None:
249              sample_weight = _check_sample_weight(sample_weight, X, dtype=dtype)
250  
251          X, y, X_offset_, y_offset_, X_scale_ = _preprocess_data(
252              X,
253              y,
254              fit_intercept=self.fit_intercept,
255              copy=self.copy_X,
256              sample_weight=sample_weight,
257          )
258  
259          if sample_weight is not None:
260              # Sample weight can be implemented via a simple rescaling.
261              X, y, _ = _rescale_data(X, y, sample_weight)
262  
263          self.X_offset_ = X_offset_
264          self.X_scale_ = X_scale_
265          n_samples, n_features = X.shape
266  
267          # Initialization of the values of the parameters
268          eps = np.finfo(np.float64).eps
269          # Add `eps` in the denominator to omit division by zero if `np.var(y)`
270          # is zero
271          alpha_ = self.alpha_init
272          lambda_ = self.lambda_init
273          if alpha_ is None:
274              alpha_ = 1.0 / (np.var(y) + eps)
275          if lambda_ is None:
276              lambda_ = 1.0
277  
278          # Avoid unintended type promotion to float64 with numpy 2
279          alpha_ = np.asarray(alpha_, dtype=dtype)
280          lambda_ = np.asarray(lambda_, dtype=dtype)
281  
282          verbose = self.verbose
283          lambda_1 = self.lambda_1
284          lambda_2 = self.lambda_2
285          alpha_1 = self.alpha_1
286          alpha_2 = self.alpha_2
287  
288          self.scores_ = list()
289          coef_old_ = None
290  
291          XT_y = np.dot(X.T, y)
292          U, S, Vh = linalg.svd(X, full_matrices=False)
293          eigen_vals_ = S**2
294  
295          # Convergence loop of the bayesian ridge regression
296          for iter_ in range(self.max_iter):
297              # update posterior mean coef_ based on alpha_ and lambda_ and
298              # compute corresponding rmse
299              coef_, rmse_ = self._update_coef_(
300                  X, y, n_samples, n_features, XT_y, U, Vh, eigen_vals_, alpha_, lambda_
301              )
302              if self.compute_score:
303                  # compute the log marginal likelihood
304                  s = self._log_marginal_likelihood(
305                      n_samples, n_features, eigen_vals_, alpha_, lambda_, coef_, rmse_
306                  )
307                  self.scores_.append(s)
308  
309              # Update alpha and lambda according to (MacKay, 1992)
310              gamma_ = np.sum((alpha_ * eigen_vals_) / (lambda_ + alpha_ * eigen_vals_))
311              lambda_ = (gamma_ + 2 * lambda_1) / (np.sum(coef_**2) + 2 * lambda_2)
312              alpha_ = (n_samples - gamma_ + 2 * alpha_1) / (rmse_ + 2 * alpha_2)
313  
314              # Check for convergence
315              if iter_ != 0 and np.sum(np.abs(coef_old_ - coef_)) < self.tol:
316                  if verbose:
317                      print("Convergence after ", str(iter_), " iterations")
318                  break
319              coef_old_ = np.copy(coef_)
320  
321          self.n_iter_ = iter_ + 1
322  
323          # return regularization parameters and corresponding posterior mean,
324          # log marginal likelihood and posterior covariance
325          self.alpha_ = alpha_
326          self.lambda_ = lambda_
327          self.coef_, rmse_ = self._update_coef_(
328              X, y, n_samples, n_features, XT_y, U, Vh, eigen_vals_, alpha_, lambda_
329          )
330          if self.compute_score:
331              # compute the log marginal likelihood
332              s = self._log_marginal_likelihood(
333                  n_samples, n_features, eigen_vals_, alpha_, lambda_, coef_, rmse_
334              )
335              self.scores_.append(s)
336              self.scores_ = np.array(self.scores_)
337  
338          # posterior covariance is given by 1/alpha_ * scaled_sigma_
339          scaled_sigma_ = np.dot(
340              Vh.T, Vh / (eigen_vals_ + lambda_ / alpha_)[:, np.newaxis]
341          )
342          self.sigma_ = (1.0 / alpha_) * scaled_sigma_
343  
344          self._set_intercept(X_offset_, y_offset_, X_scale_)
345  
346          return self
347  
348      def predict(self, X, return_std=False):
349          """Predict using the linear model. 
350   
351          In addition to the mean of the predictive distribution, also its 
352          standard deviation can be returned. 
353   
354          Parameters 
355          ---------- 
356          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
357              Samples. 
358   
359          return_std : bool, default=False 
360              Whether to return the standard deviation of posterior prediction. 
361   
362          Returns 
363          ------- 
364          y_mean : array-like of shape (n_samples,) 
365              Mean of predictive distribution of query points. 
366   
367          y_std : array-like of shape (n_samples,) 
368              Standard deviation of predictive distribution of query points. 
369          """
370          y_mean = self._decision_function(X)
371          if not return_std:
372              return y_mean
373          else:
374              sigmas_squared_data = (np.dot(X, self.sigma_) * X).sum(axis=1)
375              y_std = np.sqrt(sigmas_squared_data + (1.0 / self.alpha_))
376              return y_mean, y_std
377  
378      def _update_coef_(
379          self, X, y, n_samples, n_features, XT_y, U, Vh, eigen_vals_, alpha_, lambda_
380      ):
381          """Update posterior mean and compute corresponding rmse. 
382   
383          Posterior mean is given by coef_ = scaled_sigma_ * X.T * y where 
384          scaled_sigma_ = (lambda_/alpha_ * np.eye(n_features) 
385                           + np.dot(X.T, X))^-1 
386          """
387  
388          if n_samples > n_features:
389              coef_ = np.linalg.multi_dot(
390                  [Vh.T, Vh / (eigen_vals_ + lambda_ / alpha_)[:, np.newaxis], XT_y]
391              )
392          else:
393              coef_ = np.linalg.multi_dot(
394                  [X.T, U / (eigen_vals_ + lambda_ / alpha_)[None, :], U.T, y]
395              )
396  
397          rmse_ = np.sum((y - np.dot(X, coef_)) ** 2)
398  
399          return coef_, rmse_
400  
401      def _log_marginal_likelihood(
402          self, n_samples, n_features, eigen_vals, alpha_, lambda_, coef, rmse
403      ):
404          """Log marginal likelihood."""
405          alpha_1 = self.alpha_1
406          alpha_2 = self.alpha_2
407          lambda_1 = self.lambda_1
408          lambda_2 = self.lambda_2
409  
410          # compute the log of the determinant of the posterior covariance.
411          # posterior covariance is given by
412          # sigma = (lambda_ * np.eye(n_features) + alpha_ * np.dot(X.T, X))^-1
413          if n_samples > n_features:
414              logdet_sigma = -np.sum(np.log(lambda_ + alpha_ * eigen_vals))
415          else:
416              logdet_sigma = np.full(n_features, lambda_, dtype=np.array(lambda_).dtype)
417              logdet_sigma[:n_samples] += alpha_ * eigen_vals
418              logdet_sigma = -np.sum(np.log(logdet_sigma))
419  
420          score = lambda_1 * log(lambda_) - lambda_2 * lambda_
421          score += alpha_1 * log(alpha_) - alpha_2 * alpha_
422          score += 0.5 * (
423              n_features * log(lambda_)
424              + n_samples * log(alpha_)
425              - alpha_ * rmse
426              - lambda_ * np.sum(coef**2)
427              + logdet_sigma
428              - n_samples * log(2 * np.pi)
429          )
430  
431          return score

438  class ARDRegression(RegressorMixin, LinearModel):
439      """Bayesian ARD regression. 
440   
441      Fit the weights of a regression model, using an ARD prior. The weights of 
442      the regression model are assumed to be in Gaussian distributions. 
443      Also estimate the parameters lambda (precisions of the distributions of the 
444      weights) and alpha (precision of the distribution of the noise). 
445      The estimation is done by an iterative procedures (Evidence Maximization) 
446   
447      Read more in the :ref:`User Guide <bayesian_regression>`. 
448   
449      Parameters 
450      ---------- 
451      max_iter : int, default=300 
452          Maximum number of iterations. 
453   
454          .. versionchanged:: 1.3 
455   
456      tol : float, default=1e-3 
457          Stop the algorithm if w has converged. 
458   
459      alpha_1 : float, default=1e-6 
460          Hyper-parameter : shape parameter for the Gamma distribution prior 
461          over the alpha parameter. 
462   
463      alpha_2 : float, default=1e-6 
464          Hyper-parameter : inverse scale parameter (rate parameter) for the 
465          Gamma distribution prior over the alpha parameter. 
466   
467      lambda_1 : float, default=1e-6 
468          Hyper-parameter : shape parameter for the Gamma distribution prior 
469          over the lambda parameter. 
470   
471      lambda_2 : float, default=1e-6 
472          Hyper-parameter : inverse scale parameter (rate parameter) for the 
473          Gamma distribution prior over the lambda parameter. 
474   
475      compute_score : bool, default=False 
476          If True, compute the objective function at each step of the model. 
477   
478      threshold_lambda : float, default=10 000 
479          Threshold for removing (pruning) weights with high precision from 
480          the computation. 
481   
482      fit_intercept : bool, default=True 
483          Whether to calculate the intercept for this model. If set 
484          to false, no intercept will be used in calculations 
485          (i.e. data is expected to be centered). 
486   
487      copy_X : bool, default=True 
488          If True, X will be copied; else, it may be overwritten. 
489   
490      verbose : bool, default=False 
491          Verbose mode when fitting the model. 
492   
493      Attributes 
494      ---------- 
495      coef_ : array-like of shape (n_features,) 
496          Coefficients of the regression model (mean of distribution) 
497   
498      alpha_ : float 
499         estimated precision of the noise. 
500   
501      lambda_ : array-like of shape (n_features,) 
502         estimated precisions of the weights. 
503   
504      sigma_ : array-like of shape (n_features, n_features) 
505          estimated variance-covariance matrix of the weights 
506   
507      scores_ : float 
508          if computed, value of the objective function (to be maximized) 
509   
510      n_iter_ : int 
511          The actual number of iterations to reach the stopping criterion. 
512   
513          .. versionadded:: 1.3 
514   
515      intercept_ : float 
516          Independent term in decision function. Set to 0.0 if 
517          ``fit_intercept = False``. 
518   
519      X_offset_ : float 
520          If `fit_intercept=True`, offset subtracted for centering data to a 
521          zero mean. Set to np.zeros(n_features) otherwise. 
522   
523      X_scale_ : float 
524          Set to np.ones(n_features). 
525   
526      n_features_in_ : int 
527          Number of features seen during :term:`fit`. 
528   
529          .. versionadded:: 0.24 
530   
531      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
532          Names of features seen during :term:`fit`. Defined only when `X` 
533          has feature names that are all strings. 
534   
535          .. versionadded:: 1.0 
536   
537      See Also 
538      -------- 
539      BayesianRidge : Bayesian ridge regression. 
540   
541      Notes 
542      ----- 
543      For an example, see :ref:`examples/linear_model/plot_ard.py 
544      <sphx_glr_auto_examples_linear_model_plot_ard.py>`. 
545   
546      References 
547      ---------- 
548      D. J. C. MacKay, Bayesian nonlinear modeling for the prediction 
549      competition, ASHRAE Transactions, 1994. 
550   
551      R. Salakhutdinov, Lecture notes on Statistical Machine Learning, 
552      http://www.utstat.toronto.edu/~rsalakhu/sta4273/notes/Lecture2.pdf#page=15 
553      Their beta is our ``self.alpha_`` 
554      Their alpha is our ``self.lambda_`` 
555      ARD is a little different than the slide: only dimensions/features for 
556      which ``self.lambda_ < self.threshold_lambda`` are kept and the rest are 
557      discarded. 
558   
559      Examples 
560      -------- 
561      >>> from sklearn import linear_model 
562      >>> clf = linear_model.ARDRegression() 
563      >>> clf.fit([[0,0], [1, 1], [2, 2]], [0, 1, 2]) 
564      ARDRegression() 
565      >>> clf.predict([[1, 1]]) 
566      array([1.]) 
567      """

573          "alpha_2": [Interval(Real, 0, None, closed="left")],
574          "lambda_1": [Interval(Real, 0, None, closed="left")],
575          "lambda_2": [Interval(Real, 0, None, closed="left")],

596          verbose=False,

610      @_fit_context(prefer_skip_nested_validation=True)
611      def fit(self, X, y):
612          """Fit the model according to the given training data and parameters. 
613   
614          Iterative procedure to maximize the evidence 
615   
616          Parameters 
617          ---------- 
618          X : array-like of shape (n_samples, n_features) 
619              Training vector, where `n_samples` is the number of samples and 
620              `n_features` is the number of features. 
621          y : array-like of shape (n_samples,) 
622              Target values (integers). Will be cast to X's dtype if necessary. 
623   
624          Returns 
625          ------- 
626          self : object 
627              Fitted estimator. 
628          """
629          X, y = validate_data(
630              self,
631              X,
632              y,
633              dtype=[np.float64, np.float32],
634              force_writeable=True,
635              y_numeric=True,
636              ensure_min_samples=2,
637          )
638          dtype = X.dtype
639  
640          n_samples, n_features = X.shape
641          coef_ = np.zeros(n_features, dtype=dtype)
642  
643          X, y, X_offset_, y_offset_, X_scale_ = _preprocess_data(
644              X, y, fit_intercept=self.fit_intercept, copy=self.copy_X
645          )
646  
647          self.X_offset_ = X_offset_
648          self.X_scale_ = X_scale_
649  
650          # Launch the convergence loop
651          keep_lambda = np.ones(n_features, dtype=bool)
652  
653          lambda_1 = self.lambda_1
654          lambda_2 = self.lambda_2
655          alpha_1 = self.alpha_1
656          alpha_2 = self.alpha_2
657          verbose = self.verbose
658  
659          # Initialization of the values of the parameters
660          eps = np.finfo(np.float64).eps
661          # Add `eps` in the denominator to omit division by zero if `np.var(y)`
662          # is zero.
663          # Explicitly set dtype to avoid unintended type promotion with numpy 2.
664          alpha_ = np.asarray(1.0 / (np.var(y) + eps), dtype=dtype)
665          lambda_ = np.ones(n_features, dtype=dtype)
666  
667          self.scores_ = list()
668          coef_old_ = None
669  
670          def update_coeff(X, y, coef_, alpha_, keep_lambda, sigma_):
671              coef_[keep_lambda] = alpha_ * np.linalg.multi_dot(
672                  [sigma_, X[:, keep_lambda].T, y]
673              )
674              return coef_
675  
676          update_sigma = (
677              self._update_sigma
678              if n_samples >= n_features
679              else self._update_sigma_woodbury
680          )
681          # Iterative procedure of ARDRegression
682          for iter_ in range(self.max_iter):
683              sigma_ = update_sigma(X, alpha_, lambda_, keep_lambda)
684              coef_ = update_coeff(X, y, coef_, alpha_, keep_lambda, sigma_)
685  
686              # Update alpha and lambda
687              rmse_ = np.sum((y - np.dot(X, coef_)) ** 2)
688              gamma_ = 1.0 - lambda_[keep_lambda] * np.diag(sigma_)
689              lambda_[keep_lambda] = (gamma_ + 2.0 * lambda_1) / (
690                  (coef_[keep_lambda]) ** 2 + 2.0 * lambda_2
691              )
692              alpha_ = (n_samples - gamma_.sum() + 2.0 * alpha_1) / (
693                  rmse_ + 2.0 * alpha_2
694              )
695  
696              # Prune the weights with a precision over a threshold
697              keep_lambda = lambda_ < self.threshold_lambda
698              coef_[~keep_lambda] = 0
699  
700              # Compute the objective function
701              if self.compute_score:
702                  s = (lambda_1 * np.log(lambda_) - lambda_2 * lambda_).sum()
703                  s += alpha_1 * log(alpha_) - alpha_2 * alpha_
704                  s += 0.5 * (
705                      fast_logdet(sigma_)
706                      + n_samples * log(alpha_)
707                      + np.sum(np.log(lambda_))
708                  )
709                  s -= 0.5 * (alpha_ * rmse_ + (lambda_ * coef_**2).sum())
710                  self.scores_.append(s)
711  
712              # Check for convergence
713              if iter_ > 0 and np.sum(np.abs(coef_old_ - coef_)) < self.tol:
714                  if verbose:
715                      print("Converged after %s iterations" % iter_)
716                  break
717              coef_old_ = np.copy(coef_)
718  
719              if not keep_lambda.any():
720                  break
721  
722          self.n_iter_ = iter_ + 1
723  
724          if keep_lambda.any():
725              # update sigma and mu using updated params from the last iteration
726              sigma_ = update_sigma(X, alpha_, lambda_, keep_lambda)
727              coef_ = update_coeff(X, y, coef_, alpha_, keep_lambda, sigma_)
728          else:
729              sigma_ = np.array([]).reshape(0, 0)
730  
731          self.coef_ = coef_
732          self.alpha_ = alpha_
733          self.sigma_ = sigma_
734          self.lambda_ = lambda_
735          self._set_intercept(X_offset_, y_offset_, X_scale_)
736          return self
737  
738      def _update_sigma_woodbury(self, X, alpha_, lambda_, keep_lambda):
739          # See slides as referenced in the docstring note
740          # this function is used when n_samples < n_features and will invert
741          # a matrix of shape (n_samples, n_samples) making use of the
742          # woodbury formula:
743          # https://en.wikipedia.org/wiki/Woodbury_matrix_identity
744          n_samples = X.shape[0]
745          X_keep = X[:, keep_lambda]
746          inv_lambda = 1 / lambda_[keep_lambda].reshape(1, -1)
747          sigma_ = pinvh(
748              np.eye(n_samples, dtype=X.dtype) / alpha_
749              + np.dot(X_keep * inv_lambda, X_keep.T)
750          )
751          sigma_ = np.dot(sigma_, X_keep * inv_lambda)
752          sigma_ = -np.dot(inv_lambda.reshape(-1, 1) * X_keep.T, sigma_)
753          sigma_[np.diag_indices(sigma_.shape[1])] += 1.0 / lambda_[keep_lambda]
754          return sigma_
755  
756      def _update_sigma(self, X, alpha_, lambda_, keep_lambda):
757          # See slides as referenced in the docstring note
758          # this function is used when n_samples >= n_features and will
759          # invert a matrix of shape (n_features, n_features)
760          X_keep = X[:, keep_lambda]
761          gram = np.dot(X_keep.T, X_keep)
762          eye = np.eye(gram.shape[0], dtype=X.dtype)
763          sigma_inv = lambda_[keep_lambda] * eye + alpha_ * gram
764          sigma_ = pinvh(sigma_inv)
765          return sigma_
766  
767      def predict(self, X, return_std=False):
768          """Predict using the linear model. 
769   
770          In addition to the mean of the predictive distribution, also its 
771          standard deviation can be returned. 
772   
773          Parameters 
774          ---------- 
775          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
776              Samples. 
777   
778          return_std : bool, default=False 
779              Whether to return the standard deviation of posterior prediction. 
780   
781          Returns 
782          ------- 
783          y_mean : array-like of shape (n_samples,) 
784              Mean of predictive distribution of query points. 
785   
786          y_std : array-like of shape (n_samples,) 
787              Standard deviation of predictive distribution of query points. 
788          """
789          y_mean = self._decision_function(X)
790          if return_std is False:
791              return y_mean
792          else:
793              col_index = self.lambda_ < self.threshold_lambda
794              X = _safe_indexing(X, indices=col_index, axis=1)
795              sigmas_squared_data = (np.dot(X, self.sigma_) * X).sum(axis=1)
796              y_std = np.sqrt(sigmas_squared_data + (1.0 / self.alpha_))
797              return y_mean, y_std
