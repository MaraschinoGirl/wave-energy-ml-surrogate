68   @pytest.mark.parametrize("class_method", ["from_estimator", "from_predictions"])
69   @pytest.mark.parametrize("kind", ["actual_vs_predicted", "residual_vs_predicted"])
70   def test_prediction_error_display(pyplot, regressor_fitted, class_method, kind):
71       """Check the default behaviour of the display."""
72       if class_method == "from_estimator":
73           display = PredictionErrorDisplay.from_estimator(
74               regressor_fitted, X, y, kind=kind
75           )
76       else:
77           y_pred = regressor_fitted.predict(X)
78           display = PredictionErrorDisplay.from_predictions(
79               y_true=y, y_pred=y_pred, kind=kind
80           )
81   
82       if kind == "actual_vs_predicted":
83           assert_allclose(display.line_.get_xdata(), display.line_.get_ydata())
84           assert display.ax_.get_xlabel() == "Predicted values"
85           assert display.ax_.get_ylabel() == "Actual values"
86           assert display.line_ is not None
87       else:
88           assert display.ax_.get_xlabel() == "Predicted values"
89           assert display.ax_.get_ylabel() == "Residuals (actual - predicted)"
90           assert display.line_ is not None
91   
92       assert display.ax_.get_legend() is None

130  @pytest.mark.parametrize("class_method", ["from_estimator", "from_predictions"])
131  @pytest.mark.parametrize(
132      "scatter_kwargs",
133      [None, {"color": "blue", "alpha": 0.9}, {"c": "blue", "alpha": 0.9}],

136      "line_kwargs", [None, {"color": "red", "linestyle": "-"}, {"c": "red", "ls": "-"}]
137  )
138  def test_prediction_error_custom_artist(
139      pyplot, regressor_fitted, class_method, scatter_kwargs, line_kwargs
140  ):
141      """Check that we can tune the style of the line and the scatter."""
142      extra_params = {
143          "kind": "actual_vs_predicted",
144          "scatter_kwargs": scatter_kwargs,
145          "line_kwargs": line_kwargs,
146      }
147      if class_method == "from_estimator":
148          display = PredictionErrorDisplay.from_estimator(
149              regressor_fitted, X, y, **extra_params
150          )
151      else:
152          y_pred = regressor_fitted.predict(X)
153          display = PredictionErrorDisplay.from_predictions(
154              y_true=y, y_pred=y_pred, **extra_params
155          )
156  
157      if line_kwargs is not None:
158          assert display.line_.get_linestyle() == "-"
159          assert display.line_.get_color() == "red"
160      else:
161          assert display.line_.get_linestyle() == "--"
162          assert display.line_.get_color() == "black"
163          assert display.line_.get_alpha() == 0.7
164  
165      if scatter_kwargs is not None:
166          assert_allclose(display.scatter_.get_facecolor(), [[0.0, 0.0, 1.0, 0.9]])
167          assert_allclose(display.scatter_.get_edgecolor(), [[0.0, 0.0, 1.0, 0.9]])
168      else:
169          assert display.scatter_.get_alpha() == 0.8
