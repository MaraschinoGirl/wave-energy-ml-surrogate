294      def test_discrepancy_alternative_implementation(self):
295          """Alternative definitions from Matt Haberland."""
296  
297          def disc_c2(x):
298              n, s = x.shape
299              xij = x
300              disc1 = np.sum(np.prod((1
301                                      + 1/2*np.abs(xij-0.5)
302                                      - 1/2*np.abs(xij-0.5)**2), axis=1))
303              xij = x[None, :, :]
304              xkj = x[:, None, :]
305              disc2 = np.sum(np.sum(np.prod(1
306                                            + 1/2*np.abs(xij - 0.5)
307                                            + 1/2*np.abs(xkj - 0.5)
308                                            - 1/2*np.abs(xij - xkj), axis=2),
309                                    axis=0))
310              return (13/12)**s - 2/n * disc1 + 1/n**2*disc2
311  
312          def disc_wd(x):
313              n, s = x.shape
314              xij = x[None, :, :]
315              xkj = x[:, None, :]
316              disc = np.sum(np.sum(np.prod(3/2
317                                           - np.abs(xij - xkj)
318                                           + np.abs(xij - xkj)**2, axis=2),
319                                   axis=0))
320              return -(4/3)**s + 1/n**2 * disc
321  
322          def disc_md(x):
323              n, s = x.shape
324              xij = x
325              disc1 = np.sum(np.prod((5/3
326                                      - 1/4*np.abs(xij-0.5)
327                                      - 1/4*np.abs(xij-0.5)**2), axis=1))
328              xij = x[None, :, :]
329              xkj = x[:, None, :]
330              disc2 = np.sum(np.sum(np.prod(15/8
331                                            - 1/4*np.abs(xij - 0.5)
332                                            - 1/4*np.abs(xkj - 0.5)
333                                            - 3/4*np.abs(xij - xkj)
334                                            + 1/2*np.abs(xij - xkj)**2,
335                                            axis=2), axis=0))
336              return (19/12)**s - 2/n * disc1 + 1/n**2*disc2
337  
338          def disc_star_l2(x):
339              n, s = x.shape
340              return np.sqrt(
341                  3 ** (-s) - 2 ** (1 - s) / n
342                  * np.sum(np.prod(1 - x ** 2, axis=1))
343                  + np.sum([
344                      np.prod(1 - np.maximum(x[k, :], x[j, :]))
345                      for k in range(n) for j in range(n)
346                  ]) / n ** 2
347              )
348  
349          rng = np.random.default_rng(117065081482921065782761407107747179201)
350          sample = rng.random((30, 10))
351  
352          disc_curr = qmc.discrepancy(sample, method='CD')
353          disc_alt = disc_c2(sample)
354          assert_allclose(disc_curr, disc_alt)
355  
356          disc_curr = qmc.discrepancy(sample, method='WD')
357          disc_alt = disc_wd(sample)
358          assert_allclose(disc_curr, disc_alt)
359  
360          disc_curr = qmc.discrepancy(sample, method='MD')
361          disc_alt = disc_md(sample)
362          assert_allclose(disc_curr, disc_alt)
363  
364          disc_curr = qmc.discrepancy(sample, method='L2-star')
365          disc_alt = disc_star_l2(sample)
366          assert_allclose(disc_curr, disc_alt)
