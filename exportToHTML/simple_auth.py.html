1    # A demo of basic SSPI authentication.
2    # There is a 'client' context and a 'server' context - typically these will
3    # be on different machines (here they are in the same process, but the same
4    # concepts apply)
5    import sspi
6    import sspicon
7    import win32api
8    import win32security
9    
10   
11   def lookup_ret_code(err):
12       for k, v in sspicon.__dict__.items():
13           if k[0:6] in ("SEC_I_", "SEC_E_") and v == err:
14               return k
15   
16   
17   """ 
18   pkg_name='Kerberos' 
19   sspiclient=SSPIClient(pkg_name, win32api.GetUserName(),  ## target spn is ourself 
20       None, None,   ## use none for client name and authentication information for current context 
21       ## 'username', ('username','domain.com','passwd'), 
22       sspicon.ISC_REQ_INTEGRITY|sspicon.ISC_REQ_SEQUENCE_DETECT|sspicon.ISC_REQ_REPLAY_DETECT|    \ 
23           sspicon.ISC_REQ_DELEGATE|sspicon.ISC_REQ_CONFIDENTIALITY|sspicon.ISC_REQ_USE_SESSION_KEY) 
24   sspiserver=SSPIServer(pkg_name, None, 
25       sspicon.ASC_REQ_INTEGRITY|sspicon.ASC_REQ_SEQUENCE_DETECT|sspicon.ASC_REQ_REPLAY_DETECT|   \ 
26           sspicon.ASC_REQ_DELEGATE|sspicon.ASC_REQ_CONFIDENTIALITY|sspicon.ASC_REQ_STREAM|sspicon.ASC_REQ_USE_SESSION_KEY) 
27   """
28   
29   pkg_name = "NTLM"
30   
31   # Setup the 2 contexts.
32   sspiclient = sspi.ClientAuth(pkg_name)
33   sspiserver = sspi.ServerAuth(pkg_name)
34   
35   # Perform the authentication dance, each loop exchanging more information
36   # on the way to completing authentication.
37   sec_buffer = None
38   while 1:
39       err, sec_buffer = sspiclient.authorize(sec_buffer)
40       err, sec_buffer = sspiserver.authorize(sec_buffer)
41       if err == 0:
42           break
43   
44   # The server can now impersonate the client.  In this demo the 2 users will
45   # always be the same.
46   sspiserver.ctxt.ImpersonateSecurityContext()
47   print("Impersonated user: ", win32api.GetUserNameEx(win32api.NameSamCompatible))
48   sspiserver.ctxt.RevertSecurityContext()
49   print("Reverted to self: ", win32api.GetUserName())
50   
51   pkg_size_info = sspiclient.ctxt.QueryContextAttributes(sspicon.SECPKG_ATTR_SIZES)
52   # Now sign some data
53   msg = "some data to be encrypted ......"
54   
55   sigsize = pkg_size_info["MaxSignature"]
56   sigbuf = win32security.PySecBufferDescType()
57   sigbuf.append(win32security.PySecBufferType(len(msg), sspicon.SECBUFFER_DATA))
58   sigbuf.append(win32security.PySecBufferType(sigsize, sspicon.SECBUFFER_TOKEN))
59   sigbuf[0].Buffer = msg
60   sspiclient.ctxt.MakeSignature(0, sigbuf, 1)
61   sspiserver.ctxt.VerifySignature(sigbuf, 1)
62   
63   # And finally encrypt some.
64   trailersize = pkg_size_info["SecurityTrailer"]
65   encbuf = win32security.PySecBufferDescType()
66   encbuf.append(win32security.PySecBufferType(len(msg), sspicon.SECBUFFER_DATA))
67   encbuf.append(win32security.PySecBufferType(trailersize, sspicon.SECBUFFER_TOKEN))
68   encbuf[0].Buffer = msg
69   sspiclient.ctxt.EncryptMessage(0, encbuf, 1)
70   print(f"Encrypted data: {encbuf[0].Buffer!r}")
71   sspiserver.ctxt.DecryptMessage(encbuf, 1)
72   print(f"Encrypted data: {encbuf[0].Buffer}")
