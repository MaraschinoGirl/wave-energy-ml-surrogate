21   # safe_realloc(&p, n) resizes the allocation of p to n * sizeof(*p) bytes or
22   # raises a MemoryError. It never calls free, since that's __dealloc__'s job.
23   #   cdef float32_t *p = NULL
24   #   safe_realloc(&p, n)
25   # is equivalent to p = malloc(n * sizeof(*p)) with error checking.
