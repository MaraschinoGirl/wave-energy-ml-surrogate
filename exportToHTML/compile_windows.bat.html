compile_windows.bat 
1    :: This script compiles the attach and inject DLLs for x86 and x64 architectures.
2    
3    setlocal
4    @cd /d %~dp0
5    
6    @set VSWHERE=%ProgramFiles(x86)%\Microsoft Visual Studio\Installer\vswhere.exe 
7    @echo Using vswhere at %VSWHERE%
8    @for /f "usebackq tokens=*" %%i in (`"%VSWHERE%" -prerelease -latest -products * -requires Microsoft.VisualStudio.Component.VC.Tools.x86.x64 -property installationPath`) do set VSDIR=%%i 
9    @echo Using Visual C++ at %VSDIR%
10                                    
11   call "%VSDIR%\VC\Auxiliary\Build\vcvarsall.bat" x86 -vcvars_spectre_libs=spectre 
12   
13   cl -DUNICODE -D_UNICODE /EHsc /Zi /O1 /W3 /LD /MD /Qspectre attach.cpp /link /PROFILE /GUARD:CF /CETCOMPAT /out:attach_x86.dll 
14   copy attach_x86.dll ..\attach_x86.dll /Y 
15   copy attach_x86.pdb ..\attach_x86.pdb /Y 
16   
17   cl -DUNICODE -D_UNICODE /EHsc /Zi /O1 /W3 /LD /MD /D BITS_32 /Qspectre run_code_on_dllmain.cpp /link /PROFILE /GUARD:CF /CETCOMPAT /out:run_code_on_dllmain_x86.dll 
18   copy run_code_on_dllmain_x86.dll ..\run_code_on_dllmain_x86.dll /Y 
19   copy run_code_on_dllmain_x86.pdb ..\run_code_on_dllmain_x86.pdb /Y 
20   
21   cl /EHsc /Zi /O1 /W3 /Qspectre inject_dll.cpp /link /PROFILE /GUARD:CF /CETCOMPAT /out:inject_dll_x86.exe 
22   copy inject_dll_x86.exe ..\inject_dll_x86.exe /Y 
23   copy inject_dll_x86.pdb ..\inject_dll_x86.pdb /Y 
24   
25   call "%VSDIR%\VC\Auxiliary\Build\vcvarsall.bat" x86_amd64 -vcvars_spectre_libs=spectre 
26   
27   cl -DUNICODE -D_UNICODE /EHsc /Zi /O1 /W3 /LD /MD /Qspectre attach.cpp /link /PROFILE /GUARD:CF /CETCOMPAT /out:attach_amd64.dll 
28   copy attach_amd64.dll ..\attach_amd64.dll /Y 
29   copy attach_amd64.pdb ..\attach_amd64.pdb /Y 
30   
31   cl -DUNICODE -D_UNICODE /EHsc /Zi /O1 /W3 /LD /MD /D BITS_64 /Qspectre run_code_on_dllmain.cpp /link /PROFILE /GUARD:CF /CETCOMPAT /out:run_code_on_dllmain_amd64.dll 
32   copy run_code_on_dllmain_amd64.dll ..\run_code_on_dllmain_amd64.dll /Y 
33   copy run_code_on_dllmain_amd64.pdb ..\run_code_on_dllmain_amd64.pdb /Y 
34   
35   cl /EHsc /Zi /O1 /W3 /Qspectre inject_dll.cpp /link /PROFILE /GUARD:CF /CETCOMPAT /out:inject_dll_amd64.exe 
36   copy inject_dll_amd64.exe ..\inject_dll_amd64.exe /Y 
37   copy inject_dll_amd64.pdb ..\inject_dll_amd64.pdb /Y 
38   
39   del *.exe 
40   del *.lib 
41   del *.obj 
42   del *.pdb 
43   del *.dll 
44   del *.exp
