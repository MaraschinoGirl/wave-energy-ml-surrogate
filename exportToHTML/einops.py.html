13   def _create_axes_map(axes, input_shape, axes_lengths):
14       axes_map = {}
15   
16       for axis, dim in zip(axes, input_shape):
17           # Check for grouped axes pattern, e.g., "(h1 h)"
18           grouped_axes = re.match(r"\(([\w\s]+)\)", axis)
19   
20           if grouped_axes:
21               inner_axes = grouped_axes.group(1).split()
22               known_axes = [a for a in inner_axes if a in axes_lengths]
23               inferred_axes = [a for a in inner_axes if a not in axes_lengths]
24   
25               if inferred_axes:
26                   inferred_axis = inferred_axes[0]
27                   known_product = prod([axes_lengths[a] for a in known_axes])
28                   axes_lengths[inferred_axis] = dim // known_product
29   
30               axes_map.update({a: axes_lengths[a] for a in inner_axes})
31           else:
32               axes_map[axis] = dim
33   
34       return axes_map

37   def _create_grouped_axes(axes):
38       grouped_output_axes = []
39       for axis in axes:
40           grouped_axes = re.match(r"\(([\w\s]+)\)", axis)
41   
42           if grouped_axes:
43               inner_axes = grouped_axes.group(1).split()
44               grouped_output_axes.append(inner_axes)
45           else:
46               grouped_output_axes.append([axis])
47   
48       return grouped_output_axes

52       return [x for xs in axes for x in xs]

55   def _get_transpose_order(from_shape, to_shape):
56       flattened_from_shape = _flatten_group(_create_grouped_axes(from_shape))
57   
58       return [flattened_from_shape.index(dim) for dim in to_shape]

61   def _compute_output_shape(axes_map, grouped_axes):
62       output_shape = []
63       for group in grouped_axes:
64           size = 1
65           for axis in group:
66               size *= axes_map[axis]
67           output_shape.append(size)
68   
69       return tuple(output_shape)

72   def _compute_decomposed_shape(input_axes, axes_lengths, axes_map):
73       reshaped_input_axes = []
74       reshaped_sizes = []
75   
76       for axis in input_axes:
77           if "(" in axis:  # Decomposed axis
78               inner_axes = re.findall(r"\w+", axis)
79               sizes = [axes_lengths[a] for a in inner_axes]
80               reshaped_input_axes.extend(inner_axes)
81               reshaped_sizes.extend(sizes)
82           else:
83               reshaped_input_axes.append(axis)
84               reshaped_sizes.append(axes_map[axis])
85   
86       return reshaped_sizes

89   class Rearrange(Operation):
90       def call(self, tensor, pattern, **axes_lengths):
91           return rearrange(tensor, pattern, **axes_lengths)
92   
93       def compute_output_spec(self, tensor, pattern, **axes_lengths):
94           input_pattern, output_pattern = re.split(r"\s*->\s*", pattern)
95           input_axes = re.findall(r"\w+|\(.*?\)", input_pattern)
96           output_axes = re.findall(r"\w+|\(.*?\)", output_pattern)
97           input_shape = shape(tensor)
98   
99           axes_map = _create_axes_map(input_axes, input_shape, axes_lengths)
100          grouped_output_axes = _create_grouped_axes(output_axes)
101          output_shape = _compute_output_shape(axes_map, grouped_output_axes)
102  
103          return KerasTensor(shape=output_shape, dtype=tensor.dtype)

106  @keras_export("keras.ops.rearrange")
107  def rearrange(tensor, pattern, **axes_lengths):
108      """Rearranges the axes of a Keras tensor according to a specified pattern, 
109      einops-style. 
110   
111      Args: 
112          tensor: Input Keras tensor. 
113          pattern: String describing the rearrangement in einops notation. 
114          **axes_lengths: Keyword arguments specifying lengths of axes 
115              when axes decomposition is used. 
116   
117      Returns: 
118          Tensor: A Keras tensor with rearranged axes. 
119   
120      Follows the logic of: 
121   
122      1. If decomposition is needed, reshape to match decomposed dimensions. 
123      2. Permute known and inferred axes to match the form of the output. 
124      3. Reshape to match the desired output shape. 
125   
126   
127      Example Usage: 
128   
129      ``` 
130      >>> import numpy as np 
131      >>> from keras.ops import rearrange 
132      >>> images = np.random.rand(32, 30, 40, 3) # BHWC format 
133   
134      # Reordering to BCHW 
135      >>> rearrange(images, 'b h w c -> b c h w').shape 
136      TensorShape([32, 3, 30, 40]) 
137   
138      # "Merge" along first axis - concat images from a batch 
139      >>> rearrange(images, 'b h w c -> (b h) w c').shape 
140      TensorShape([960, 40, 3]) 
141   
142      # "Merge" along second axis - concat images horizontally 
143      >>> rearrange(images, 'b h w c -> h (b w) c').shape 
144      TensorShape([30, 1280, 3]) 
145   
146      # Flatten images into a CHW vector 
147      >>> rearrange(images, 'b h w c -> b (c h w)').shape 
148      TensorShape([32, 3600]) 
149   
150      # Decompose H and W axes into 4 smaller patches 
151      >>> rearrange(images, 'b (h1 h) (w1 w) c -> (b h1 w1) h w c', h1=2, w1=2).shape 
152      TensorShape([128, 15, 20, 3]) 
153   
154      # Space-to-depth decomposition of input axes 
155      >>> rearrange(images, 'b (h h1) (w w1) c -> b h w (c h1 w1)', h1=2, w1=2).shape 
156      TensorShape([32, 15, 20, 12]) 
157      ``` 
158      """  # noqa: E501
159  
160      if any_symbolic_tensors((tensor,)):
161          return Rearrange().symbolic_call(tensor, pattern, **axes_lengths)
162  
163      # Split the input and output patterns
164      input_pattern, output_pattern = re.split(r"\s*->\s*", pattern)
165      input_axes = re.findall(r"\w+|\(.*?\)", input_pattern)
166      output_axes = re.findall(r"\w+|\(.*?\)", output_pattern)
167      input_shape = shape(tensor)
168  
169      # Create axes map, and flattened output group
170      axes_map = _create_axes_map(input_axes, input_shape, axes_lengths)
171      grouped_output_axes = _create_grouped_axes(output_axes)
172      flattened_output_axes = _flatten_group(grouped_output_axes)
173  
174      # 1. Axes decomposition
175      decomposed_shapes = _compute_decomposed_shape(
176          input_axes, axes_lengths, axes_map
177      )
178      if decomposed_shapes != tensor.shape:
179          tensor = reshape(tensor, decomposed_shapes)
180  
181      # 2. Transpose to match target shape
182      permute_order = _get_transpose_order(input_axes, flattened_output_axes)
183      tensor = transpose(tensor, permute_order)
184  
185      # 3. Reshape to final target shape
186      output_shape = _compute_output_shape(axes_map, grouped_output_axes)
187      tensor = reshape(tensor, output_shape)
188  
189      return tensor
