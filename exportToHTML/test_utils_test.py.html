17   from __future__ import absolute_import
18   from __future__ import division
19   from __future__ import print_function
20   
21   import ast
22   import unittest
23   
24   from pasta.base import test_utils
25   
26   
27   class CheckAstEqualityTest(test_utils.TestCase):
28   
29     def test_empty(self):
30       src = ""
31       t = ast.parse(src)
32       self.checkAstsEqual(t, t)
33   
34     def test_one_global(self):
35       src = "X = 1\n"
36       t = ast.parse(src)
37       self.checkAstsEqual(t, t)
38   
39     def test_two_globals(self):
40       src = "X = 1\nY = 2\n"
41       t = ast.parse(src)
42       self.checkAstsEqual(t, t)
43   
44     def test_different_number_of_nodes(self):
45       src1 = "X = 1\ndef Foo():\n  return None\n"
46       src2 = src1 + "Y = 2\n"
47       t1 = ast.parse(src1)
48       t2 = ast.parse(src2)
49       with self.assertRaises(AssertionError):
50         self.checkAstsEqual(t1, t2)
51   
52     def test_simple_function_def(self):
53       code = ("def foo(x):\n"
54               "  return x + 1\n")
55       t = ast.parse(code)
56       self.checkAstsEqual(t, t)
57   
58   
59   def suite():
60     result = unittest.TestSuite()
61     result.addTests(unittest.makeSuite(CheckAstEqualityTest))
62     return result
63   
64   
65   if __name__ == '__main__':
66     unittest.main()
