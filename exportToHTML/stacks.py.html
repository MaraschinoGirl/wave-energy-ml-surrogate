5    """Provides facilities to dump all stacks of all threads in the process. 
6    """
7    
8    import os
9    import sys
10   import time
11   import threading
12   import traceback
13   
14   from debugpy.common import log
15   
16   
17   def dump():
18       """Dump stacks of all threads in this process, except for the current thread."""
19   
20       tid = threading.current_thread().ident
21       pid = os.getpid()
22   
23       log.info("Dumping stacks for process {0}...", pid)
24   
25       for t_ident, frame in sys._current_frames().items():
26           if t_ident == tid:
27               continue
28   
29           for t in threading.enumerate():
30               if t.ident == tid:
31                   t_name = t.name
32                   t_daemon = t.daemon
33                   break
34           else:
35               t_name = t_daemon = "<unknown>"
36   
37           stack = "".join(traceback.format_stack(frame))
38           log.info(
39               "Stack of thread {0} (tid={1}, pid={2}, daemon={3}):\n\n{4}",
40               t_name,
41               t_ident,
42               pid,
43               t_daemon,
44               stack,
45           )
46   
47       log.info("Finished dumping stacks for process {0}.", pid)
48   
49   
50   def dump_after(secs):
51       """Invokes dump() on a background thread after waiting for the specified time."""
52   
53       def dumper():
54           time.sleep(secs)
55           try:
56               dump()
57           except:
58               log.swallow_exception()
59   
60       thread = threading.Thread(target=dumper)
61       thread.daemon = True
62       thread.start()
