27   class UtilsTest(test_utils.TestCase):
28   
29     def test_sanitize_source(self):
30       coding_lines = (
31           '# -*- coding: latin-1 -*-',
32           '# -*- coding: iso-8859-15 -*-',
33           '# vim: set fileencoding=ascii :',
34           '# This Python file uses the following encoding: utf-8',
35       )
36       src_template = '{coding}\na = 123\n'
37       sanitized_src = '# (removed coding)\na = 123\n'
38       for line in coding_lines:
39         src = src_template.format(coding=line)
40   
41         # Replaced on lines 1 and 2
42         self.assertEqual(sanitized_src, ast_utils.sanitize_source(src))
43         src_prefix = '"""Docstring."""\n'
44         self.assertEqual(src_prefix + sanitized_src,
45                          ast_utils.sanitize_source(src_prefix + src))
46   
47         # Unchanged on line 3
48         src_prefix = '"""Docstring."""\n# line 2\n'
49         self.assertEqual(src_prefix + src,
50                          ast_utils.sanitize_source(src_prefix + src))

53   class AlterChildTest(test_utils.TestCase):
54   
55     def testRemoveChildMethod(self):
56       src = """\ 
57   class C(): 
58     def f(x): 
59       return x + 2 
60     def g(x): 
61       return x + 3 
62   """
63       tree = pasta.parse(src)
64       class_node = tree.body[0]
65       meth1_node = class_node.body[0]
66   
67       ast_utils.remove_child(class_node, meth1_node)
68   
69       result = pasta.dump(tree)
70       expected = """\ 
71   class C(): 
72     def g(x): 
73       return x + 3 
74   """
75       self.assertEqual(result, expected)
76   
77     def testRemoveAlias(self):
78       src = "from a import b, c"
79       tree = pasta.parse(src)
80       import_node = tree.body[0]
81       alias1 = import_node.names[0]
82       ast_utils.remove_child(import_node, alias1)
83   
84       self.assertEqual(pasta.dump(tree), "from a import c")
85   
86     def testRemoveFromBlock(self):
87       src = """\ 
88   if a: 
89     print("foo!") 
90     x = 1 
91   """
92       tree = pasta.parse(src)
93       if_block = tree.body[0]
94       print_stmt = if_block.body[0]
95       ast_utils.remove_child(if_block, print_stmt)
96   
97       expected = """\ 
98   if a: 
99     x = 1 
100  """
101      self.assertEqual(pasta.dump(tree), expected)

103    def testReplaceChildInBody(self):
104      src = 'def foo():\n  a = 0\n  a += 1 # replace this\n  return a\n'
105      replace_with = pasta.parse('foo(a + 1)  # trailing comment\n').body[0]
106      expected = 'def foo():\n  a = 0\n  foo(a + 1) # replace this\n  return a\n'
107      t = pasta.parse(src)
108  
109      parent = t.body[0]
110      node_to_replace = parent.body[1]
111      ast_utils.replace_child(parent, node_to_replace, replace_with)
112  
113      self.assertEqual(expected, pasta.dump(t))
114  
115    def testReplaceChildInvalid(self):
116      src = 'def foo():\n  return 1\nx = 1\n'
117      replace_with = pasta.parse('bar()').body[0]
118      t = pasta.parse(src)
119  
120      parent = t.body[0]
121      node_to_replace = t.body[1]
122      with self.assertRaises(errors.InvalidAstError):
123        ast_utils.replace_child(parent, node_to_replace, replace_with)
