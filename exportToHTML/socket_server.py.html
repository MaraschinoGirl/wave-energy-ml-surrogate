1    """A sample socket server and client using SSPI authentication and encryption. 
2     
3    You must run with either 'client' or 'server' as arguments.  A server must be 
4    running before a client can connect. 
5     
6    To use with Kerberos you should include in the client options 
7    --target-spn=username, where 'username' is the user under which the server is 
8    being run. 
9     
10   Running either the client or server as a different user can be informative. 
11   A command-line such as the following may be useful: 
12   `runas /user:{user} {fqp}\\python.exe {fqp}\\socket_server.py --wait client|server` 
13    
14   {fqp} should specify the relevant fully-qualified path names. 
15    
16   To use 'runas' with Kerberos, the client program will need to 
17   specify --target-spn with the username under which the *server* is running. 
18    
19   See the SSPI documentation for more details. 
20   """
21

24   import socketserver
25   import struct
26   import traceback

28   import sspi
29   import win32api

31   options = None  # set to optparse object.

34   def GetUserName():
35       try:
36           return win32api.GetUserName()
37       except win32api.error as details:
38           # Seeing 'access denied' errors here for non-local users (presumably
39           # without permission to login locally).  Get the fully-qualified
40           # username, although a side-effect of these permission-denied errors
41           # is a lack of Python codecs - so printing the Unicode value fails.
42           # So just return the repr(), and avoid codecs completely.
43           return repr(win32api.GetUserNameEx(win32api.NameSamCompatible))
44   
45   
46   # Send a simple "message" over a socket - send the number of bytes first,
47   # then the string.  Ditto for receive.
48   def _send_msg(s, m):
49       s.send(struct.pack("i", len(m)))
50       s.send(m)
51   
52   
53   def _get_msg(s):
54       size_data = s.recv(struct.calcsize("i"))
55       if not size_data:
56           return None
57       cb = struct.unpack("i", size_data)[0]
58       return s.recv(cb)
59   
60   
61   class SSPISocketServer(socketserver.TCPServer):
62       def __init__(self, *args, **kw):
63           socketserver.TCPServer.__init__(self, *args, **kw)
64           self.sa = sspi.ServerAuth(options.package)
65   
66       def verify_request(self, sock, ca):
67           # Do the sspi auth dance
68           self.sa.reset()
69           while 1:
70               data = _get_msg(sock)
71               if data is None:
72                   return False
73               try:
74                   err, sec_buffer = self.sa.authorize(data)
75               except sspi.error as details:
76                   print("FAILED to authorize client:", details)
77                   return False
78   
79               if err == 0:
80                   break
81               _send_msg(sock, sec_buffer[0].Buffer)
82           return True
83   
84       def process_request(self, request, client_address):
85           # An example using the connection once it is established.
86           print("The server is running as user", GetUserName())
87           self.sa.ctxt.ImpersonateSecurityContext()
88           try:
89               print("Having conversation with client as user", GetUserName())
90               while 1:
91                   # we need to grab 2 bits of data - the encrypted data, and the
92                   # 'key'
93                   data = _get_msg(request)
94                   key = _get_msg(request)
95                   if data is None or key is None:
96                       break
97                   data = self.sa.decrypt(data, key)
98                   print(f"Client sent: {data!r}")
99           finally:
100              self.sa.ctxt.RevertSecurityContext()
101          self.close_request(request)
102          print("The server is back to user", GetUserName())
103  
104  
105  def serve():
106      s = SSPISocketServer(("localhost", options.port), None)
107      print("Running test server...")
108      s.serve_forever()
109  
110  
111  def sspi_client():
112      c = http.client.HTTPConnection("localhost", options.port)
113      c.connect()
114      # Do the auth dance.
115      ca = sspi.ClientAuth(options.package, targetspn=options.target_spn)
116      data = None
117      while 1:
118          err, out_buf = ca.authorize(data)
119          _send_msg(c.sock, out_buf[0].Buffer)
120          if err == 0:
121              break
122          data = _get_msg(c.sock)
123      print("Auth dance complete - sending a few encryted messages")
124      # Assume out data is sensitive - encrypt the message.
125      for data in "Hello from the client".split():
126          blob, key = ca.encrypt(data)
127          _send_msg(c.sock, blob)
128          _send_msg(c.sock, key)
129      c.sock.close()
130      print("Client completed.")

152                        process.  For NTLM this must be blank.""",
153      )

156          "",
157          "--port",
158          action="store",
159          default="8181",
160          help="The port number to use (default=8181)",

164          "",

166          action="store_true",

169                                any error messages before termination. 
170                             """,
171      )
172  
173      options, args = parser.parse_args()
174      try:
175          options.port = int(options.port)
176      except (ValueError, TypeError):
177          parser.error("--port must be an integer")
178  
179      try:
180          try:
181              if not args:
182                  args = [""]
183              if args[0] == "client":
184                  sspi_client()
185              elif args[0] == "server":
186                  serve()
187              else:
188                  parser.error(
189                      "You must supply 'client' or 'server' - use --help for details"
190                  )
191          except KeyboardInterrupt:
192              pass
193          except SystemExit:
194              pass
195          except:
196              traceback.print_exc()
197      finally:
198          if options.wait:
