27       def test_soupstrainer(self):
28           # The html5lib tree builder does not support parse_only.
29           strainer = SoupStrainer("b")
30           markup = "<p>A <b>bold</b> statement.</p>"
31           with warnings.catch_warnings(record=True) as w:
32               soup = BeautifulSoup(markup, "html5lib", parse_only=strainer)
33           assert soup.decode() == self.document_for(markup)
34   
35           [warning] = w
36           assert warning.filename == __file__
37           assert "the html5lib tree builder doesn't support parse_only" in str(
38               warning.message
39           )

41       def test_correctly_nested_tables(self):
42           """html5lib inserts <tbody> tags where other parsers don't."""
43           markup = (
44               '<table id="1">'
45               "<tr>"
46               "<td>Here's another table:"
47               '<table id="2">'
48               "<tr><td>foo</td></tr>"
49               "</table></td>"
50           )
51   
52           self.assert_soup(
53               markup,
54               '<table id="1"><tbody><tr><td>Here\'s another table:'
55               '<table id="2"><tbody><tr><td>foo</td></tr></tbody></table>'
56               "</td></tr></tbody></table>",
57           )
58   
59           self.assert_soup(
60               "<table><thead><tr><td>Foo</td></tr></thead>"
61               "<tbody><tr><td>Bar</td></tr></tbody>"
62               "<tfoot><tr><td>Baz</td></tr></tfoot></table>"
63           )

97       def test_reparented_markup_containing_identical_whitespace_nodes(self):
98           """Verify that we keep the two whitespace nodes in this 
99           document distinct when reparenting the adjacent <tbody> tags. 
100          """
101          markup = "<table> <tbody><tbody><ims></tbody> </table>"
102          soup = self.soup(markup)
103          space1, space2 = soup.find_all(string=" ")
104          tbody1, tbody2 = soup.find_all("tbody")
105          assert space1.next_element is tbody1
106          assert tbody2.next_element is space2

108      def test_reparented_markup_containing_children(self):
109          markup = (
110              "<div><a>aftermath<p><noscript>target</noscript>aftermath</a></p></div>"
111          )
112          soup = self.soup(markup)
113          noscript = soup.noscript
114          assert "target" == noscript.next_element
115          target = soup.find(string="target")
116  
117          # The 'aftermath' string was duplicated; we want the second one.
118          final_aftermath = soup.find_all(string="aftermath")[-1]
119  
120          # The <noscript> tag was moved beneath a copy of the <a> tag,
121          # but the 'target' string within is still connected to the
122          # (second) 'aftermath' string.
123          assert final_aftermath == target.next_element
124          assert target == final_aftermath.previous_element

187      def test_tracking_line_numbers(self):
188          # The html.parser TreeBuilder keeps track of line number and
189          # position of each element.
190          markup = "\n   <p>\n\n<sourceline>\n<b>text</b></sourceline><sourcepos></p>"
191          soup = self.soup(markup)
192          assert 2 == soup.p.sourceline
193          assert 5 == soup.p.sourcepos
194          assert "sourceline" == soup.p.find("sourceline").name
195  
196          # You can deactivate this behavior.
197          soup = self.soup(markup, store_line_numbers=False)
198          assert None is soup.p.sourceline
199          assert None is soup.p.sourcepos

207      def test_html5_attributes(self):
208          # The html5lib TreeBuilder can convert any entity named in
209          # the HTML5 spec to a sequence of Unicode characters, and
210          # convert those Unicode characters to a (potentially
211          # different) named entity on the way out.
212          #
213          # This is a copy of the same test from
214          # HTMLParserTreeBuilderSmokeTest.  It's not in the superclass
215          # because the lxml HTML TreeBuilder _doesn't_ work this way.
216          for input_element, output_unicode, output_element in (
217              ("&RightArrowLeftArrow;", "\u21c4", b"&rlarr;"),
218              ("&models;", "\u22a7", b"&models;"),
219              ("&Nfr;", "\U0001d511", b"&Nfr;"),
220              ("&ngeqq;", "\u2267\u0338", b"&ngeqq;"),
221              ("&not;", "\xac", b"&not;"),
222              ("&Not;", "\u2aec", b"&Not;"),
223              ("&quot;", '"', b'"'),
224              ("&there4;", "\u2234", b"&there4;"),
225              ("&Therefore;", "\u2234", b"&there4;"),
226              ("&therefore;", "\u2234", b"&there4;"),
227              ("&fjlig;", "fj", b"fj"),
228              ("&sqcup;", "\u2294", b"&sqcup;"),
229              ("&sqcups;", "\u2294\ufe00", b"&sqcups;"),
230              ("&apos;", "'", b"'"),
231              ("&verbar;", "|", b"|"),
232          ):
233              markup = "<div>%s</div>" % input_element
234              div = self.soup(markup).div
235              without_element = div.encode()
236              expect = b"<div>%s</div>" % output_unicode.encode("utf8")
237              assert without_element == expect
238  
239              with_element = div.encode(formatter="html")
240              expect = b"<div>%s</div>" % output_element
241              assert with_element == expect

243      @pytest.mark.parametrize(
244          "name,value",
245          [("document_declared_encoding", "utf8"), ("exclude_encodings", ["utf8"])],
246      )
247      def test_prepare_markup_warnings(self, name, value):
248          # html5lib doesn't support a couple of the common arguments to
249          # prepare_markup.
250          builder = self.default_builder()
251          kwargs = {name: value}
252          with warnings.catch_warnings(record=True) as w:
253              list(builder.prepare_markup("a", **kwargs))
254          [warning] = w
255          msg = str(warning.message)
256          assert (
257              msg
258              == f"You provided a value for {name}, but the html5lib tree builder doesn't support {name}."
259          )
