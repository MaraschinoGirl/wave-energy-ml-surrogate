1    # This is a simple serial port terminal demo.
2    #
3    # Its primary purpose is to demonstrate the native serial port access offered via
4    # win32file.
5    
6    # It uses 3 threads:
7    # - The main thread, which cranks up the other 2 threads, then simply waits for them to exit.
8    # - The user-input thread - blocks waiting for a keyboard character, and when found sends it
9    #   out the COM port.  If the character is Ctrl+C, it stops, signalling the COM port thread to stop.
10   # - The COM port thread is simply listening for input on the COM port, and prints it to the screen.
11   
12   # This demo uses userlapped IO, so that none of the read or write operations actually block (however,
13   # in this sample, the very next thing we do _is_ block - so it shows off the concepts even though it
14   # doesn't exploit them.
15

55   def FindModem():
56       # Snoop over the comports, seeing if it is likely we have a modem.
57       for i in range(1, 5):
58           port = "COM%d" % (i,)
59           try:
60               handle = CreateFile(
61                   port,
62                   win32con.GENERIC_READ | win32con.GENERIC_WRITE,
63                   0,  # exclusive access
64                   None,  # no security
65                   win32con.OPEN_EXISTING,
66                   win32con.FILE_ATTRIBUTE_NORMAL,
67                   None,
68               )
69               # It appears that an available COM port will always success here,
70               # just return 0 for the status flags.  We only care that it has _any_ status
71               # flags (and therefore probably a real modem)
72               if GetCommModemStatus(handle) != 0:
73                   return port
74           except error:
75               pass  # No port, or modem status failed.
76       return None

79   # A basic synchronous COM port file-like object
80   class SerialTTY:
81       def __init__(self, port):
82           if isinstance(port, int):
83               port = "COM%d" % (port,)
84           self.handle = CreateFile(
85               port,
86               win32con.GENERIC_READ | win32con.GENERIC_WRITE,
87               0,  # exclusive access
88               None,  # no security
89               win32con.OPEN_EXISTING,
90               win32con.FILE_ATTRIBUTE_NORMAL | win32con.FILE_FLAG_OVERLAPPED,
91               None,
92           )
93           # Tell the port we want a notification on each char.
94           SetCommMask(self.handle, EV_RXCHAR)
95           # Setup a 4k buffer
96           SetupComm(self.handle, 4096, 4096)
97           # Remove anything that was there
98           PurgeComm(
99               self.handle, PURGE_TXABORT | PURGE_RXABORT | PURGE_TXCLEAR | PURGE_RXCLEAR
100          )
101          # Setup for overlapped IO.
102          timeouts = 0xFFFFFFFF, 0, 1000, 0, 1000
103          SetCommTimeouts(self.handle, timeouts)
104          # Setup the connection info.
105          dcb = GetCommState(self.handle)
106          dcb.BaudRate = CBR_115200
107          dcb.ByteSize = 8
108          dcb.Parity = NOPARITY
109          dcb.StopBits = ONESTOPBIT
110          SetCommState(self.handle, dcb)
111          print(f"Connected to {port} at {dcb.BaudRate} baud")
112  
113      def _UserInputReaderThread(self):
114          overlapped = OVERLAPPED()
115          overlapped.hEvent = CreateEvent(None, 1, 0, None)
116          try:
117              while 1:
118                  ch = msvcrt.getch()
119                  if ord(ch) == 3:
120                      break
121                  WriteFile(self.handle, ch, overlapped)
122                  # Wait for the write to complete.
123                  WaitForSingleObject(overlapped.hEvent, INFINITE)
124          finally:
125              SetEvent(self.eventStop)
126  
127      def _ComPortThread(self):
128          overlapped = OVERLAPPED()
129          overlapped.hEvent = CreateEvent(None, 1, 0, None)
130          while 1:
131              # XXX - note we could _probably_ just use overlapped IO on the win32file.ReadFile() statement
132              # XXX but this tests the COM stuff!
133              rc, mask = WaitCommEvent(self.handle, overlapped)
134              if rc == 0:  # Character already ready!
135                  SetEvent(overlapped.hEvent)
136              rc = WaitForMultipleObjects(
137                  [overlapped.hEvent, self.eventStop], 0, INFINITE
138              )
139              if rc == WAIT_OBJECT_0:
140                  # Some input - read and print it
141                  flags, comstat = ClearCommError(self.handle)
142                  rc, data = ReadFile(self.handle, comstat.cbInQue, overlapped)
143                  WaitForSingleObject(overlapped.hEvent, INFINITE)
144                  sys.stdout.write(data)
145              else:
146                  # Stop the thread!
147                  # Just incase the user input thread uis still going, close it
148                  sys.stdout.close()
149                  break
150  
151      def Run(self):
152          self.eventStop = CreateEvent(None, 0, 0, None)
153          # Start the reader and writer threads.
154          user_thread = threading.Thread(target=self._UserInputReaderThread)
155          user_thread.start()
156          com_thread = threading.Thread(target=self._ComPortThread)
157          com_thread.start()
158          user_thread.join()
159          com_thread.join()

162  if __name__ == "__main__":
163      print("Serial port terminal demo - press Ctrl+C to exit")
164      if len(sys.argv) <= 1:
165          port = FindModem()
166          if port is None:
167              print("No COM port specified, and no modem could be found")
168              print("Please re-run this script with the name of a COM port (eg COM3)")
169              sys.exit(1)
170      else:
171          port = sys.argv[1]
172  
173      tty = SerialTTY(port)
174      tty.Run()
