48   class _BaseVoting(TransformerMixin, _BaseHeterogeneousEnsemble):
49       """Base class for voting. 
50    
51       Warning: This class should not be used directly. Use derived classes 
52       instead. 
53       """
54   
55       _parameter_constraints: dict = {
56           "estimators": [list],
57           "weights": ["array-like", None],
58           "n_jobs": [None, Integral],
59           "verbose": ["verbose"],
60       }
61   
62       def _log_message(self, name, idx, total):
63           if not self.verbose:
64               return None
65           return f"({idx} of {total}) Processing {name}"
66   
67       @property
68       def _weights_not_none(self):
69           """Get the weights of not `None` estimators."""
70           if self.weights is None:
71               return None
72           return [w for est, w in zip(self.estimators, self.weights) if est[1] != "drop"]
73   
74       def _predict(self, X):
75           """Collect results from clf.predict calls."""
76           return np.asarray([est.predict(X) for est in self.estimators_]).T
77   
78       @abstractmethod
79       def fit(self, X, y, **fit_params):
80           """Get common fit operations."""
81           names, clfs = self._validate_estimators()
82   
83           if self.weights is not None and len(self.weights) != len(self.estimators):
84               raise ValueError(
85                   "Number of `estimators` and weights must be equal; got"
86                   f" {len(self.weights)} weights, {len(self.estimators)} estimators"
87               )
88   
89           if _routing_enabled():
90               routed_params = process_routing(self, "fit", **fit_params)
91           else:
92               routed_params = Bunch()
93               for name in names:
94                   routed_params[name] = Bunch(fit={})
95                   if "sample_weight" in fit_params:
96                       routed_params[name].fit["sample_weight"] = fit_params[
97                           "sample_weight"
98                       ]
99   
100          self.estimators_ = Parallel(n_jobs=self.n_jobs)(
101              delayed(_fit_single_estimator)(
102                  clone(clf),
103                  X,
104                  y,
105                  fit_params=routed_params[name]["fit"],
106                  message_clsname="Voting",
107                  message=self._log_message(name, idx + 1, len(clfs)),
108              )
109              for idx, (name, clf) in enumerate(zip(names, clfs))
110              if clf != "drop"
111          )
112  
113          self.named_estimators_ = Bunch()
114  
115          # Uses 'drop' as placeholder for dropped estimators
116          est_iter = iter(self.estimators_)
117          for name, est in self.estimators:
118              current_est = est if est == "drop" else next(est_iter)
119              self.named_estimators_[name] = current_est
120  
121              if hasattr(current_est, "feature_names_in_"):
122                  self.feature_names_in_ = current_est.feature_names_in_
123  
124          return self
125  
126      def fit_transform(self, X, y=None, **fit_params):
127          """Return class labels or probabilities for each estimator. 
128   
129          Return predictions for X for each estimator. 
130   
131          Parameters 
132          ---------- 
133          X : {array-like, sparse matrix, dataframe} of shape \ 
134                  (n_samples, n_features) 
135              Input samples. 
136   
137          y : ndarray of shape (n_samples,), default=None 
138              Target values (None for unsupervised transformations). 
139   
140          **fit_params : dict 
141              Additional fit parameters. 
142   
143          Returns 
144          ------- 
145          X_new : ndarray array of shape (n_samples, n_features_new) 
146              Transformed array. 
147          """
148          return super().fit_transform(X, y, **fit_params)
149  
150      @property
151      def n_features_in_(self):
152          """Number of features seen during :term:`fit`."""
153          # For consistency with other estimators we raise a AttributeError so
154          # that hasattr() fails if the estimator isn't fitted.
155          try:
156              check_is_fitted(self)
157          except NotFittedError as nfe:
158              raise AttributeError(
159                  "{} object has no n_features_in_ attribute.".format(
160                      self.__class__.__name__
161                  )
162              ) from nfe
163  
164          return self.estimators_[0].n_features_in_
165  
166      def _sk_visual_block_(self):
167          names, estimators = zip(*self.estimators)
168          return _VisualBlock("parallel", estimators, names=names)
169  
170      def get_metadata_routing(self):
171          """Get metadata routing of this object. 
172   
173          Please check :ref:`User Guide <metadata_routing>` on how the routing 
174          mechanism works. 
175   
176          .. versionadded:: 1.5 
177   
178          Returns 
179          ------- 
180          routing : MetadataRouter 
181              A :class:`~sklearn.utils.metadata_routing.MetadataRouter` encapsulating 
182              routing information. 
183          """
184          router = MetadataRouter(owner=self.__class__.__name__)
185  
186          # `self.estimators` is a list of (name, est) tuples
187          for name, estimator in self.estimators:
188              router.add(
189                  **{name: estimator},
190                  method_mapping=MethodMapping().add(callee="fit", caller="fit"),
191              )
192          return router

195  class VotingClassifier(ClassifierMixin, _BaseVoting):
196      """Soft Voting/Majority Rule classifier for unfitted estimators. 
197   
198      Read more in the :ref:`User Guide <voting_classifier>`. 
199   
200      .. versionadded:: 0.17 
201   
202      Parameters 
203      ---------- 
204      estimators : list of (str, estimator) tuples 
205          Invoking the ``fit`` method on the ``VotingClassifier`` will fit clones 
206          of those original estimators that will be stored in the class attribute 
207          ``self.estimators_``. An estimator can be set to ``'drop'`` using 
208          :meth:`set_params`. 
209   
210          .. versionchanged:: 0.21 
211              ``'drop'`` is accepted. Using None was deprecated in 0.22 and 
212              support was removed in 0.24. 
213   
214      voting : {'hard', 'soft'}, default='hard' 
215          If 'hard', uses predicted class labels for majority rule voting. 
216          Else if 'soft', predicts the class label based on the argmax of 
217          the sums of the predicted probabilities, which is recommended for 
218          an ensemble of well-calibrated classifiers. 
219   
220      weights : array-like of shape (n_classifiers,), default=None 
221          Sequence of weights (`float` or `int`) to weight the occurrences of 
222          predicted class labels (`hard` voting) or class probabilities 
223          before averaging (`soft` voting). Uses uniform weights if `None`. 
224   
225      n_jobs : int, default=None 
226          The number of jobs to run in parallel for ``fit``. 
227          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
228          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
229          for more details. 
230   
231          .. versionadded:: 0.18 
232   
233      flatten_transform : bool, default=True 
234          Affects shape of transform output only when voting='soft' 
235          If voting='soft' and flatten_transform=True, transform method returns 
236          matrix with shape (n_samples, n_classifiers * n_classes). If 
237          flatten_transform=False, it returns 
238          (n_classifiers, n_samples, n_classes). 
239   
240      verbose : bool, default=False 
241          If True, the time elapsed while fitting will be printed as it 
242          is completed. 
243   
244          .. versionadded:: 0.23 
245   
246      Attributes 
247      ---------- 
248      estimators_ : list of classifiers 
249          The collection of fitted sub-estimators as defined in ``estimators`` 
250          that are not 'drop'. 
251   
252      named_estimators_ : :class:`~sklearn.utils.Bunch` 
253          Attribute to access any fitted sub-estimators by name. 
254   
255          .. versionadded:: 0.20 
256   
257      le_ : :class:`~sklearn.preprocessing.LabelEncoder` 
258          Transformer used to encode the labels during fit and decode during 
259          prediction. 
260   
261      classes_ : ndarray of shape (n_classes,) 
262          The classes labels. 
263   
264      n_features_in_ : int 
265          Number of features seen during :term:`fit`. Only defined if the 
266          underlying classifier exposes such an attribute when fit. 
267   
268          .. versionadded:: 0.24 
269   
270      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
271          Names of features seen during :term:`fit`. Only defined if the 
272          underlying estimators expose such an attribute when fit. 
273   
274          .. versionadded:: 1.0 
275   
276      See Also 
277      -------- 
278      VotingRegressor : Prediction voting regressor. 
279   
280      Examples 
281      -------- 
282      >>> import numpy as np 
283      >>> from sklearn.linear_model import LogisticRegression 
284      >>> from sklearn.naive_bayes import GaussianNB 
285      >>> from sklearn.ensemble import RandomForestClassifier, VotingClassifier 
286      >>> clf1 = LogisticRegression(random_state=1) 
287      >>> clf2 = RandomForestClassifier(n_estimators=50, random_state=1) 
288      >>> clf3 = GaussianNB() 
289      >>> X = np.array([[-1, -1], [-2, -1], [-3, -2], [1, 1], [2, 1], [3, 2]]) 
290      >>> y = np.array([1, 1, 1, 2, 2, 2]) 
291      >>> eclf1 = VotingClassifier(estimators=[ 
292      ...         ('lr', clf1), ('rf', clf2), ('gnb', clf3)], voting='hard') 
293      >>> eclf1 = eclf1.fit(X, y) 
294      >>> print(eclf1.predict(X)) 
295      [1 1 1 2 2 2] 
296      >>> np.array_equal(eclf1.named_estimators_.lr.predict(X), 
297      ...                eclf1.named_estimators_['lr'].predict(X)) 
298      True 
299      >>> eclf2 = VotingClassifier(estimators=[ 
300      ...         ('lr', clf1), ('rf', clf2), ('gnb', clf3)], 
301      ...         voting='soft') 
302      >>> eclf2 = eclf2.fit(X, y) 
303      >>> print(eclf2.predict(X)) 
304      [1 1 1 2 2 2] 
305   
306      To drop an estimator, :meth:`set_params` can be used to remove it. Here we 
307      dropped one of the estimators, resulting in 2 fitted estimators: 
308   
309      >>> eclf2 = eclf2.set_params(lr='drop') 
310      >>> eclf2 = eclf2.fit(X, y) 
311      >>> len(eclf2.estimators_) 
312      2 
313   
314      Setting `flatten_transform=True` with `voting='soft'` flattens output shape of 
315      `transform`: 
316   
317      >>> eclf3 = VotingClassifier(estimators=[ 
318      ...        ('lr', clf1), ('rf', clf2), ('gnb', clf3)], 
319      ...        voting='soft', weights=[2,1,1], 
320      ...        flatten_transform=True) 
321      >>> eclf3 = eclf3.fit(X, y) 
322      >>> print(eclf3.predict(X)) 
323      [1 1 1 2 2 2] 
324      >>> print(eclf3.transform(X).shape) 
325      (6, 6) 
326      """
327  
328      _parameter_constraints: dict = {
329          **_BaseVoting._parameter_constraints,
330          "voting": [StrOptions({"hard", "soft"})],
331          "flatten_transform": ["boolean"],
332      }
333  
334      def __init__(
335          self,
336          estimators,
337          *,
338          voting="hard",
339          weights=None,
340          n_jobs=None,
341          flatten_transform=True,
342          verbose=False,
343      ):
344          super().__init__(estimators=estimators)
345          self.voting = voting
346          self.weights = weights
347          self.n_jobs = n_jobs
348          self.flatten_transform = flatten_transform
349          self.verbose = verbose
350  
351      @_fit_context(
352          # estimators in VotingClassifier.estimators are not validated yet
353          prefer_skip_nested_validation=False
354      )
355      # TODO(1.7): remove `sample_weight` from the signature after deprecation
356      # cycle; pop it from `fit_params` before the `_raise_for_params` check and
357      # reinsert later, for backwards compatibility
358      @_deprecate_positional_args(version="1.7")
359      def fit(self, X, y, *, sample_weight=None, **fit_params):
360          """Fit the estimators. 
361   
362          Parameters 
363          ---------- 
364          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
365              Training vectors, where `n_samples` is the number of samples and 
366              `n_features` is the number of features. 
367   
368          y : array-like of shape (n_samples,) 
369              Target values. 
370   
371          sample_weight : array-like of shape (n_samples,), default=None 
372              Sample weights. If None, then samples are equally weighted. 
373              Note that this is supported only if all underlying estimators 
374              support sample weights. 
375   
376              .. versionadded:: 0.18 
377   
378          **fit_params : dict 
379              Parameters to pass to the underlying estimators. 
380   
381              .. versionadded:: 1.5 
382   
383                  Only available if `enable_metadata_routing=True`, 
384                  which can be set by using 
385                  ``sklearn.set_config(enable_metadata_routing=True)``. 
386                  See :ref:`Metadata Routing User Guide <metadata_routing>` for 
387                  more details. 
388   
389          Returns 
390          ------- 
391          self : object 
392              Returns the instance itself. 
393          """
394          _raise_for_params(fit_params, self, "fit")
395          y_type = type_of_target(y, input_name="y")
396          if y_type in ("unknown", "continuous"):
397              # raise a specific ValueError for non-classification tasks
398              raise ValueError(
399                  f"Unknown label type: {y_type}. Maybe you are trying to fit a "
400                  "classifier, which expects discrete classes on a "
401                  "regression target with continuous values."
402              )
403          elif y_type not in ("binary", "multiclass"):
404              # raise a NotImplementedError for backward compatibility for non-supported
405              # classification tasks
406              raise NotImplementedError(
407                  f"{self.__class__.__name__} only supports binary or multiclass "
408                  "classification. Multilabel and multi-output classification are not "
409                  "supported."
410              )
411  
412          self.le_ = LabelEncoder().fit(y)
413          self.classes_ = self.le_.classes_
414          transformed_y = self.le_.transform(y)
415  
416          if sample_weight is not None:
417              fit_params["sample_weight"] = sample_weight
418  
419          return super().fit(X, transformed_y, **fit_params)
420  
421      def predict(self, X):
422          """Predict class labels for X. 
423   
424          Parameters 
425          ---------- 
426          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
427              The input samples. 
428   
429          Returns 
430          ------- 
431          maj : array-like of shape (n_samples,) 
432              Predicted class labels. 
433          """
434          check_is_fitted(self)
435          if self.voting == "soft":
436              maj = np.argmax(self.predict_proba(X), axis=1)
437  
438          else:  # 'hard' voting
439              predictions = self._predict(X)
440              maj = np.apply_along_axis(
441                  lambda x: np.argmax(np.bincount(x, weights=self._weights_not_none)),
442                  axis=1,
443                  arr=predictions,
444              )
445  
446          maj = self.le_.inverse_transform(maj)
447  
448          return maj
449  
450      def _collect_probas(self, X):
451          """Collect results from clf.predict calls."""
452          return np.asarray([clf.predict_proba(X) for clf in self.estimators_])
453  
454      def _check_voting(self):
455          if self.voting == "hard":
456              raise AttributeError(
457                  f"predict_proba is not available when voting={repr(self.voting)}"
458              )
459          return True
460  
461      @available_if(_check_voting)
462      def predict_proba(self, X):
463          """Compute probabilities of possible outcomes for samples in X. 
464   
465          Parameters 
466          ---------- 
467          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
468              The input samples. 
469   
470          Returns 
471          ------- 
472          avg : array-like of shape (n_samples, n_classes) 
473              Weighted average probability for each class per sample. 
474          """
475          check_is_fitted(self)
476          avg = np.average(
477              self._collect_probas(X), axis=0, weights=self._weights_not_none
478          )
479          return avg
480  
481      def transform(self, X):
482          """Return class labels or probabilities for X for each estimator. 
483   
484          Parameters 
485          ---------- 
486          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
487              Training vectors, where `n_samples` is the number of samples and 
488              `n_features` is the number of features. 
489   
490          Returns 
491          ------- 
492          probabilities_or_labels 
493              If `voting='soft'` and `flatten_transform=True`: 
494                  returns ndarray of shape (n_samples, n_classifiers * n_classes), 
495                  being class probabilities calculated by each classifier. 
496              If `voting='soft' and `flatten_transform=False`: 
497                  ndarray of shape (n_classifiers, n_samples, n_classes) 
498              If `voting='hard'`: 
499                  ndarray of shape (n_samples, n_classifiers), being 
500                  class labels predicted by each classifier. 
501          """
502          check_is_fitted(self)
503  
504          if self.voting == "soft":
505              probas = self._collect_probas(X)
506              if not self.flatten_transform:
507                  return probas
508              return np.hstack(probas)
509  
510          else:
511              return self._predict(X)
512  
513      def get_feature_names_out(self, input_features=None):
514          """Get output feature names for transformation. 
515   
516          Parameters 
517          ---------- 
518          input_features : array-like of str or None, default=None 
519              Not used, present here for API consistency by convention. 
520   
521          Returns 
522          ------- 
523          feature_names_out : ndarray of str objects 
524              Transformed feature names. 
525          """
526          check_is_fitted(self, "n_features_in_")
527          if self.voting == "soft" and not self.flatten_transform:
528              raise ValueError(
529                  "get_feature_names_out is not supported when `voting='soft'` and "
530                  "`flatten_transform=False`"
531              )
532  
533          _check_feature_names_in(self, input_features, generate_names=False)
534          class_name = self.__class__.__name__.lower()
535  
536          active_names = [name for name, est in self.estimators if est != "drop"]
537  
538          if self.voting == "hard":
539              return np.asarray(
540                  [f"{class_name}_{name}" for name in active_names], dtype=object
541              )
542  
543          # voting == "soft"
544          n_classes = len(self.classes_)
545          names_out = [
546              f"{class_name}_{name}{i}" for name in active_names for i in range(n_classes)
547          ]
548          return np.asarray(names_out, dtype=object)

552          tags.transformer_tags.preserves_dtype = []
553          return tags

556  class VotingRegressor(RegressorMixin, _BaseVoting):
557      """Prediction voting regressor for unfitted estimators. 
558   
559      A voting regressor is an ensemble meta-estimator that fits several base 
560      regressors, each on the whole dataset. Then it averages the individual 
561      predictions to form a final prediction. 
562   
563      For a detailed example, refer to 
564      :ref:`sphx_glr_auto_examples_ensemble_plot_voting_regressor.py`. 
565   
566      Read more in the :ref:`User Guide <voting_regressor>`. 
567   
568      .. versionadded:: 0.21 
569   
570      Parameters 
571      ---------- 
572      estimators : list of (str, estimator) tuples 
573          Invoking the ``fit`` method on the ``VotingRegressor`` will fit clones 
574          of those original estimators that will be stored in the class attribute 
575          ``self.estimators_``. An estimator can be set to ``'drop'`` using 
576          :meth:`set_params`. 
577   
578          .. versionchanged:: 0.21 
579              ``'drop'`` is accepted. Using None was deprecated in 0.22 and 
580              support was removed in 0.24. 
581   
582      weights : array-like of shape (n_regressors,), default=None 
583          Sequence of weights (`float` or `int`) to weight the occurrences of 
584          predicted values before averaging. Uses uniform weights if `None`. 
585   
586      n_jobs : int, default=None 
587          The number of jobs to run in parallel for ``fit``. 
588          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
589          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
590          for more details. 
591   
592      verbose : bool, default=False 
593          If True, the time elapsed while fitting will be printed as it 
594          is completed. 
595   
596          .. versionadded:: 0.23 
597   
598      Attributes 
599      ---------- 
600      estimators_ : list of regressors 
601          The collection of fitted sub-estimators as defined in ``estimators`` 
602          that are not 'drop'. 
603   
604      named_estimators_ : :class:`~sklearn.utils.Bunch` 
605          Attribute to access any fitted sub-estimators by name. 
606   
607          .. versionadded:: 0.20 
608   
609      n_features_in_ : int 
610          Number of features seen during :term:`fit`. Only defined if the 
611          underlying regressor exposes such an attribute when fit. 
612   
613          .. versionadded:: 0.24 
614   
615      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
616          Names of features seen during :term:`fit`. Only defined if the 
617          underlying estimators expose such an attribute when fit. 
618   
619          .. versionadded:: 1.0 
620   
621      See Also 
622      -------- 
623      VotingClassifier : Soft Voting/Majority Rule classifier. 
624   
625      Examples 
626      -------- 
627      >>> import numpy as np 
628      >>> from sklearn.linear_model import LinearRegression 
629      >>> from sklearn.ensemble import RandomForestRegressor 
630      >>> from sklearn.ensemble import VotingRegressor 
631      >>> from sklearn.neighbors import KNeighborsRegressor 
632      >>> r1 = LinearRegression() 
633      >>> r2 = RandomForestRegressor(n_estimators=10, random_state=1) 
634      >>> r3 = KNeighborsRegressor() 
635      >>> X = np.array([[1, 1], [2, 4], [3, 9], [4, 16], [5, 25], [6, 36]]) 
636      >>> y = np.array([2, 6, 12, 20, 30, 42]) 
637      >>> er = VotingRegressor([('lr', r1), ('rf', r2), ('r3', r3)]) 
638      >>> print(er.fit(X, y).predict(X)) 
639      [ 6.8...  8.4... 12.5... 17.8... 26...  34...] 
640   
641      In the following example, we drop the `'lr'` estimator with 
642      :meth:`~VotingRegressor.set_params` and fit the remaining two estimators: 
643   
644      >>> er = er.set_params(lr='drop') 
645      >>> er = er.fit(X, y) 
646      >>> len(er.estimators_) 
647      2 
648      """
649  
650      def __init__(self, estimators, *, weights=None, n_jobs=None, verbose=False):
651          super().__init__(estimators=estimators)
652          self.weights = weights
653          self.n_jobs = n_jobs
654          self.verbose = verbose
655  
656      @_fit_context(
657          # estimators in VotingRegressor.estimators are not validated yet
658          prefer_skip_nested_validation=False
659      )
660      # TODO(1.7): remove `sample_weight` from the signature after deprecation cycle;
661      # pop it from `fit_params` before the `_raise_for_params` check and reinsert later,
662      # for backwards compatibility
663      @_deprecate_positional_args(version="1.7")
664      def fit(self, X, y, *, sample_weight=None, **fit_params):
665          """Fit the estimators. 
666   
667          Parameters 
668          ---------- 
669          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
670              Training vectors, where `n_samples` is the number of samples and 
671              `n_features` is the number of features. 
672   
673          y : array-like of shape (n_samples,) 
674              Target values. 
675   
676          sample_weight : array-like of shape (n_samples,), default=None 
677              Sample weights. If None, then samples are equally weighted. 
678              Note that this is supported only if all underlying estimators 
679              support sample weights. 
680   
681          **fit_params : dict 
682              Parameters to pass to the underlying estimators. 
683   
684              .. versionadded:: 1.5 
685   
686                  Only available if `enable_metadata_routing=True`, 
687                  which can be set by using 
688                  ``sklearn.set_config(enable_metadata_routing=True)``. 
689                  See :ref:`Metadata Routing User Guide <metadata_routing>` for 
690                  more details. 
691   
692          Returns 
693          ------- 
694          self : object 
695              Fitted estimator. 
696          """
697          _raise_for_params(fit_params, self, "fit")
698          y = column_or_1d(y, warn=True)
699          if sample_weight is not None:
700              fit_params["sample_weight"] = sample_weight
701          return super().fit(X, y, **fit_params)
702  
703      def predict(self, X):
704          """Predict regression target for X. 
705   
706          The predicted regression target of an input sample is computed as the 
707          mean predicted regression targets of the estimators in the ensemble. 
708   
709          Parameters 
710          ---------- 
711          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
712              The input samples. 
713   
714          Returns 
715          ------- 
716          y : ndarray of shape (n_samples,) 
717              The predicted values. 
718          """
719          check_is_fitted(self)
720          return np.average(self._predict(X), axis=1, weights=self._weights_not_none)
721  
722      def transform(self, X):
723          """Return predictions for X for each estimator. 
724   
725          Parameters 
726          ---------- 
727          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
728              The input samples. 
729   
730          Returns 
731          ------- 
732          predictions : ndarray of shape (n_samples, n_classifiers) 
733              Values predicted by each regressor. 
734          """
735          check_is_fitted(self)
736          return self._predict(X)

738      def get_feature_names_out(self, input_features=None):
739          """Get output feature names for transformation. 
740   
741          Parameters 
742          ---------- 
743          input_features : array-like of str or None, default=None 
744              Not used, present here for API consistency by convention. 
745   
746          Returns 
747          ------- 
748          feature_names_out : ndarray of str objects 
749              Transformed feature names. 
750          """
751          check_is_fitted(self, "n_features_in_")
752          _check_feature_names_in(self, input_features, generate_names=False)
753          class_name = self.__class__.__name__.lower()
754          return np.asarray(
755              [f"{class_name}_{name}" for name, est in self.estimators if est != "drop"],
756              dtype=object,
757          )
