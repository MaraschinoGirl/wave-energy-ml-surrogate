15   class TagRemovePreprocessor(Preprocessor):
16       """ 
17       Removes inputs, outputs, or cells from a notebook that 
18       have tags that designate they are to be removed prior to exporting 
19       the notebook. 
20    
21       remove_cell_tags 
22           removes cells tagged with these values 
23    
24       remove_all_outputs_tags 
25           removes entire output areas on cells 
26           tagged with these values 
27    
28       remove_single_output_tags 
29           removes individual output objects on 
30           outputs tagged with these values 
31    
32       remove_input_tags 
33           removes inputs tagged with these values 
34       """
35   
36       remove_cell_tags: set[str] = Set(  # type:ignore[assignment]
37           Unicode(),
38           default_value=[],
39           help=(
40               "Tags indicating which cells are to be removed,"
41               "matches tags in ``cell.metadata.tags``."
42           ),
43       ).tag(config=True)
44       remove_all_outputs_tags: set[str] = Set(  # type:ignore[assignment]
45           Unicode(),
46           default_value=[],
47           help=(
48               "Tags indicating cells for which the outputs are to be removed,"
49               "matches tags in ``cell.metadata.tags``."
50           ),
51       ).tag(config=True)
52       remove_single_output_tags: set[str] = Set(  # type:ignore[assignment]
53           Unicode(),
54           default_value=[],
55           help=(
56               "Tags indicating which individual outputs are to be removed,"
57               "matches output *i* tags in ``cell.outputs[i].metadata.tags``."
58           ),
59       ).tag(config=True)
60       remove_input_tags: set[str] = Set(  # type:ignore[assignment]
61           Unicode(),
62           default_value=[],
63           help=(
64               "Tags indicating cells for which input is to be removed,"
65               "matches tags in ``cell.metadata.tags``."
66           ),
67       ).tag(config=True)
68       remove_metadata_fields: set[str] = Set({"collapsed", "scrolled"}).tag(config=True)  # type:ignore[assignment]
69   
70       def check_cell_conditions(self, cell, resources, index):
71           """ 
72           Checks that a cell has a tag that is to be removed 
73    
74           Returns: Boolean. 
75           True means cell should *not* be removed. 
76           """
77   
78           # Return true if any of the tags in the cell are removable.
79           return not self.remove_cell_tags.intersection(cell.get("metadata", {}).get("tags", []))
80   
81       def preprocess(self, nb, resources):
82           """ 
83           Preprocessing to apply to each notebook. See base.py for details. 
84           """
85           # Skip preprocessing if the list of patterns is empty
86           if not any(
87               [
88                   self.remove_cell_tags,
89                   self.remove_all_outputs_tags,
90                   self.remove_single_output_tags,
91                   self.remove_input_tags,
92               ]
93           ):
94               return nb, resources
95   
96           # Filter out cells that meet the conditions
97           nb.cells = [
98               self.preprocess_cell(cell, resources, index)[0]
99               for index, cell in enumerate(nb.cells)
100              if self.check_cell_conditions(cell, resources, index)
101          ]
102  
103          return nb, resources
104  
105      def preprocess_cell(self, cell, resources, cell_index):
106          """ 
107          Apply a transformation on each cell. See base.py for details. 
108          """
109  
110          if (
111              self.remove_all_outputs_tags.intersection(cell.get("metadata", {}).get("tags", []))
112              and cell.cell_type == "code"
113          ):
114              cell.outputs = []
115              cell.execution_count = None
116              # Remove metadata associated with output
117              if "metadata" in cell:
118                  for field in self.remove_metadata_fields:
119                      cell.metadata.pop(field, None)
120  
121          if self.remove_input_tags.intersection(cell.get("metadata", {}).get("tags", [])):
122              cell.metadata["transient"] = {"remove_source": True}
123  
124          if cell.get("outputs", []):
125              cell.outputs = [
126                  output
127                  for output_index, output in enumerate(cell.outputs)
128                  if self.check_output_conditions(output, resources, cell_index, output_index)
129              ]
130          return cell, resources
131  
132      def check_output_conditions(self, output, resources, cell_index, output_index):
133          """ 
134          Checks that an output has a tag that indicates removal. 
135   
136          Returns: Boolean. 
137          True means output should *not* be removed. 
138          """
139          return not self.remove_single_output_tags.intersection(
140              output.get("metadata", {}).get("tags", [])
141          )
