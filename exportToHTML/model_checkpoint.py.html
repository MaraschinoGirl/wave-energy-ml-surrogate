14   @keras_export("keras.callbacks.ModelCheckpoint")
15   class ModelCheckpoint(Callback):
16       """Callback to save the Keras model or model weights at some frequency. 
17    
18       `ModelCheckpoint` callback is used in conjunction with training using 
19       `model.fit()` to save a model or weights (in a checkpoint file) at some 
20       interval, so the model or weights can be loaded later to continue the 
21       training from the state saved. 
22    
23       A few options this callback provides include: 
24    
25       - Whether to only keep the model that has achieved the "best performance" so 
26         far, or whether to save the model at the end of every epoch regardless of 
27         performance. 
28       - Definition of "best"; which quantity to monitor and whether it should be 
29         maximized or minimized. 
30       - The frequency it should save at. Currently, the callback supports saving 
31         at the end of every epoch, or after a fixed number of training batches. 
32       - Whether only weights are saved, or the whole model is saved. 
33    
34       Example: 
35    
36       ```python 
37       model.compile(loss=..., optimizer=..., 
38                     metrics=['accuracy']) 
39    
40       EPOCHS = 10 
41       checkpoint_filepath = '/tmp/ckpt/checkpoint.model.keras' 
42       model_checkpoint_callback = keras.callbacks.ModelCheckpoint( 
43           filepath=checkpoint_filepath, 
44           monitor='val_accuracy', 
45           mode='max', 
46           save_best_only=True) 
47    
48       # Model is saved at the end of every epoch, if it's the best seen so far. 
49       model.fit(epochs=EPOCHS, callbacks=[model_checkpoint_callback]) 
50    
51       # The model (that are considered the best) can be loaded as - 
52       keras.models.load_model(checkpoint_filepath) 
53    
54       # Alternatively, one could checkpoint just the model weights as - 
55       checkpoint_filepath = '/tmp/ckpt/checkpoint.weights.h5' 
56       model_checkpoint_callback = keras.callbacks.ModelCheckpoint( 
57           filepath=checkpoint_filepath, 
58           save_weights_only=True, 
59           monitor='val_accuracy', 
60           mode='max', 
61           save_best_only=True) 
62    
63       # Model weights are saved at the end of every epoch, if it's the best seen 
64       # so far. 
65       model.fit(epochs=EPOCHS, callbacks=[model_checkpoint_callback]) 
66    
67       # The model weights (that are considered the best) can be loaded as - 
68       model.load_weights(checkpoint_filepath) 
69       ``` 
70    
71       Args: 
72           filepath: string or `PathLike`, path to save the model file. 
73               `filepath` can contain named formatting options, 
74               which will be filled the value of `epoch` and keys in `logs` 
75               (passed in `on_epoch_end`). 
76               The `filepath` name needs to end with `".weights.h5"` when 
77               `save_weights_only=True` or should end with `".keras"` or `".h5"` 
78               when checkpoint saving the whole model (default). 
79               For example: 
80               if `filepath` is `"{epoch:02d}-{val_loss:.2f}.keras"` or 
81               "{epoch:02d}-{val_loss:.2f}.weights.h5"`, then the model 
82               checkpoints will be saved with the epoch number and the validation 
83               loss in the filename. The directory of the filepath 
84               should not be reused by any other callbacks to avoid conflicts. 
85           monitor: The metric name to monitor. Typically the metrics are set by 
86               the `Model.compile` method. Note: 
87               * Prefix the name with `"val_"` to monitor validation metrics. 
88               * Use `"loss"` or `"val_loss"` to monitor the model's total loss. 
89               * If you specify metrics as strings, like `"accuracy"`, pass the 
90                   same string (with or without the `"val_"` prefix). 
91               * If you pass `metrics.Metric` objects, `monitor` should be set to 
92                   `metric.name` 
93               * If you're not sure about the metric names you can check the 
94                   contents of the `history.history` dictionary returned by 
95                   `history = model.fit()` 
96               * Multi-output models set additional prefixes on the metric names. 
97           verbose: Verbosity mode, 0 or 1. Mode 0 is silent, and mode 1 
98               displays messages when the callback takes an action. 
99           save_best_only: if `save_best_only=True`, it only saves when the model 
100              is considered the "best" and the latest best model according to the 
101              quantity monitored will not be overwritten. If `filepath` doesn't 
102              contain formatting options like `{epoch}` then `filepath` will be 
103              overwritten by each new better model. 
104          mode: one of {`"auto"`, `"min"`, `"max"`}. If `save_best_only=True`, the 
105              decision to overwrite the current save file is made based on either 
106              the maximization or the minimization of the monitored quantity. 
107              For `val_acc`, this should be `"max"`, for `val_loss` this should be 
108              `"min"`, etc. In `"auto"` mode, the mode is set to `"max"` if the 
109              quantities monitored are `"acc"` or start with `"fmeasure"` and are 
110              set to `"min"` for the rest of the quantities. 
111          save_weights_only: if `True`, then only the model's weights will be 
112              saved (`model.save_weights(filepath)`), else the full model is 
113              saved (`model.save(filepath)`). 
114          save_freq: `"epoch"` or integer. When using `"epoch"`, the callback 
115              saves the model after each epoch. When using integer, the callback 
116              saves the model at end of this many batches. If the `Model` is 
117              compiled with `steps_per_execution=N`, then the saving criteria will 
118              be checked every Nth batch. Note that if the saving isn't aligned to 
119              epochs, the monitored metric may potentially be less reliable (it 
120              could reflect as little as 1 batch, since the metrics get reset 
121              every epoch). Defaults to `"epoch"`. 
122          initial_value_threshold: Floating point initial "best" value of the 
123              metric to be monitored. Only applies if `save_best_value=True`. Only 
124              overwrites the model weights already saved if the performance of 
125              current model is better than this value. 
126      """
127  
128      def __init__(
129          self,
130          filepath,
131          monitor="val_loss",
132          verbose=0,
133          save_best_only=False,
134          save_weights_only=False,
135          mode="auto",
136          save_freq="epoch",
137          initial_value_threshold=None,
138      ):
139          super().__init__()
140          self.monitor = monitor
141          self.verbose = verbose
142          self.filepath = file_utils.path_to_string(filepath)
143          self.save_best_only = save_best_only
144          self.save_weights_only = save_weights_only
145          self.save_freq = save_freq
146          self._batches_seen_since_last_saving = 0
147          self._last_batch_seen = 0
148          self.best = initial_value_threshold
149  
150          if mode not in ["auto", "min", "max"]:
151              warnings.warn(
152                  f"ModelCheckpoint mode '{mode}' is unknown, "
153                  "fallback to auto mode.",
154                  stacklevel=2,
155              )
156              mode = "auto"
157  
158          if mode == "min":
159              self.monitor_op = np.less
160              if self.best is None:
161                  self.best = np.inf
162          elif mode == "max":
163              self.monitor_op = np.greater
164              if self.best is None:
165                  self.best = -np.inf
166          else:
167              if "acc" in self.monitor or self.monitor.startswith("fmeasure"):
168                  self.monitor_op = np.greater
169                  if self.best is None:
170                      self.best = -np.inf
171              else:
172                  self.monitor_op = np.less
173                  if self.best is None:
174                      self.best = np.inf
175  
176          if self.save_freq != "epoch" and not isinstance(self.save_freq, int):
177              raise ValueError(
178                  f"Unrecognized save_freq: {self.save_freq}. "
179                  "Expected save_freq are 'epoch' or integer values"
180              )
181  
182          if save_weights_only:
183              if not self.filepath.endswith(".weights.h5"):
184                  raise ValueError(
185                      "When using `save_weights_only=True` in `ModelCheckpoint`"
186                      ", the filepath provided must end in `.weights.h5` "
187                      "(Keras weights format). Received: "
188                      f"filepath={self.filepath}"
189                  )
190          else:
191              if not any(
192                  self.filepath.endswith(ext) for ext in (".keras", ".h5")
193              ):
194                  raise ValueError(
195                      "The filepath provided must end in `.keras` "
196                      "(Keras model format). Received: "
197                      f"filepath={self.filepath}"
198                  )
199  
200      def on_train_batch_end(self, batch, logs=None):
201          if self._should_save_on_batch(batch):
202              self._save_model(epoch=self._current_epoch, batch=batch, logs=logs)
203  
204      def on_epoch_begin(self, epoch, logs=None):
205          self._current_epoch = epoch
206  
207      def on_epoch_end(self, epoch, logs=None):
208          if self.save_freq == "epoch":
209              self._save_model(epoch=epoch, batch=None, logs=logs)
210  
211      def _should_save_on_batch(self, batch):
212          """Handles batch-level saving logic, supports steps_per_execution."""
213          if self.save_freq == "epoch":
214              return False
215          if batch <= self._last_batch_seen:  # New epoch.
216              add_batches = batch + 1  # batches are zero-indexed.
217          else:
218              add_batches = batch - self._last_batch_seen
219          self._batches_seen_since_last_saving += add_batches
220          self._last_batch_seen = batch
221  
222          if self._batches_seen_since_last_saving >= self.save_freq:
223              self._batches_seen_since_last_saving = 0
224              return True
225          return False
226  
227      def _save_model(self, epoch, batch, logs):
228          """Saves the model. 
229   
230          Args: 
231              epoch: the epoch this iteration is in. 
232              batch: the batch this iteration is in. `None` if the `save_freq` 
233                  is set to `"epoch"`. 
234              logs: the `logs` dict passed in to `on_batch_end` or `on_epoch_end`. 
235          """
236          logs = logs or {}
237  
238          filepath = self._get_file_path(epoch, batch, logs)
239          # Create host directory if it doesn't exist.
240          dirname = os.path.dirname(filepath)
241          if dirname and not file_utils.exists(dirname):
242              file_utils.makedirs(dirname)
243  
244          try:
245              if self.save_best_only:
246                  current = logs.get(self.monitor)
247                  if current is None:
248                      warnings.warn(
249                          f"Can save best model only with {self.monitor} "
250                          "available, skipping.",
251                          stacklevel=2,
252                      )
253                  elif (
254                      isinstance(current, np.ndarray)
255                      or backend.is_tensor(current)
256                  ) and len(current.shape) > 0:
257                      warnings.warn(
258                          "Can save best model only when `monitor` is "
259                          f"a scalar value. Received: {current}. "
260                          "Falling back to `save_best_only=False`."
261                      )
262                      self.model.save(filepath, overwrite=True)
263                  else:
264                      if self.monitor_op(current, self.best):
265                          if self.verbose > 0:
266                              io_utils.print_msg(
267                                  f"\nEpoch {epoch + 1}: {self.monitor} "
268                                  "improved "
269                                  f"from {self.best:.5f} to {current:.5f}, "
270                                  f"saving model to {filepath}"
271                              )
272                          self.best = current
273                          if self.save_weights_only:
274                              self.model.save_weights(filepath, overwrite=True)
275                          else:
276                              self.model.save(filepath, overwrite=True)
277                      else:
278                          if self.verbose > 0:
279                              io_utils.print_msg(
280                                  f"\nEpoch {epoch + 1}: "
281                                  f"{self.monitor} did not improve "
282                                  f"from {self.best:.5f}"
283                              )
284              else:
285                  if self.verbose > 0:
286                      io_utils.print_msg(
287                          f"\nEpoch {epoch + 1}: saving model to {filepath}"
288                      )
289                  if self.save_weights_only:
290                      self.model.save_weights(filepath, overwrite=True)
291                  else:
292                      self.model.save(filepath, overwrite=True)
293          except IsADirectoryError:  # h5py 3.x
294              raise IOError(
295                  "Please specify a non-directory filepath for "
296                  "ModelCheckpoint. Filepath used is an existing "
297                  f"directory: {filepath}"
298              )
299          except IOError as e:  # h5py 2.x
300              # `e.errno` appears to be `None` so checking the content of
301              # `e.args[0]`.
302              if "is a directory" in str(e.args[0]).lower():
303                  raise IOError(
304                      "Please specify a non-directory filepath for "
305                      "ModelCheckpoint. Filepath used is an existing "
306                      f"directory: f{filepath}"
307                  )
308              # Re-throw the error for any other causes.
309              raise e
310  
311      def _get_file_path(self, epoch, batch, logs):
312          """Returns the file path for checkpoint."""
313  
314          try:
315              # `filepath` may contain placeholders such as
316              # `{epoch:02d}`,`{batch:02d}` and `{mape:.2f}`. A mismatch between
317              # logged metrics and the path's placeholders can cause formatting to
318              # fail.
319              if batch is None or "batch" in logs:
320                  file_path = self.filepath.format(epoch=epoch + 1, **logs)
321              else:
322                  file_path = self.filepath.format(
323                      epoch=epoch + 1, batch=batch + 1, **logs
324                  )
325          except KeyError as e:
326              raise KeyError(
327                  f'Failed to format this callback filepath: "{self.filepath}". '
328                  f"Reason: {e}"
329              )
330          return file_path
331  
332      def _checkpoint_exists(self, filepath):
333          """Returns whether the checkpoint `filepath` refers to exists."""
334          return file_utils.exists(filepath)

336      def _get_most_recently_modified_file_matching_pattern(self, pattern):
337          """Returns the most recently modified filepath matching pattern. 
338   
339          In the rare case where there are more than one pattern-matching file 
340          having the same modified time that is most recent among all, return the 
341          filepath that is largest (by `>` operator, lexicographically using the 
342          numeric equivalents). This provides a tie-breaker when multiple files 
343          are most recent. Note that a larger `filepath` can sometimes indicate a 
344          later time of modification (for instance, when epoch/batch is used as 
345          formatting option), but not necessarily (when accuracy or loss is used). 
346          The tie-breaker is put in the logic as best effort to return the most 
347          recent, and to avoid nondeterministic result. 
348   
349          Modified time of a file is obtained with `os.path.getmtime()`. 
350   
351          This utility function is best demonstrated via an example: 
352   
353          ```python 
354          file_pattern = 'batch{batch:02d}epoch{epoch:02d}.keras' 
355          test_dir = self.get_temp_dir() 
356          path_pattern = os.path.join(test_dir, file_pattern) 
357          file_paths = [ 
358              os.path.join(test_dir, file_name) for file_name in 
359              ['batch03epoch02.keras', 
360               'batch02epoch02.keras', 'batch01epoch01.keras'] 
361          ] 
362          for file_path in file_paths: 
363              # Write something to each of the files 
364              ... 
365          self.assertEqual( 
366              _get_most_recently_modified_file_matching_pattern(path_pattern), 
367              file_paths[-1]) 
368          ``` 
369   
370          Args: 
371              pattern: The file pattern that may optionally contain python 
372                  placeholder such as `{epoch:02d}`. 
373   
374          Returns: 
375              The most recently modified file's full filepath matching `pattern`. 
376              If `pattern` does not contain any placeholder, this returns the 
377              filepath that exactly matches `pattern`. Returns `None` if no match 
378              is found. 
379          """
380          dir_name = os.path.dirname(pattern)
381          base_name = os.path.basename(pattern)
382          base_name_regex = "^" + re.sub(r"{.*}", r".*", base_name) + "$"
383  
384          latest_mod_time = 0
385          file_path_with_latest_mod_time = None
386          n_file_with_latest_mod_time = 0
387          file_path_with_largest_file_name = None
388  
389          if file_utils.exists(dir_name):
390              for file_name in os.listdir(dir_name):
391                  # Only consider if `file_name` matches the pattern.
392                  if re.match(base_name_regex, file_name):
393                      file_path = os.path.join(dir_name, file_name)
394                      mod_time = os.path.getmtime(file_path)
395                      if (
396                          file_path_with_largest_file_name is None
397                          or file_path > file_path_with_largest_file_name
398                      ):
399                          file_path_with_largest_file_name = file_path
400                      if mod_time > latest_mod_time:
401                          latest_mod_time = mod_time
402                          file_path_with_latest_mod_time = file_path
403                          # In the case a file with later modified time is found,
404                          # reset the counter for the number of files with latest
405                          # modified time.
406                          n_file_with_latest_mod_time = 1
407                      elif mod_time == latest_mod_time:
408                          # In the case a file has modified time tied with the
409                          # most recent, increment the counter for the number of
410                          # files with latest modified time by 1.
411                          n_file_with_latest_mod_time += 1
412  
413          if n_file_with_latest_mod_time == 1:
414              # Return the sole file that has most recent modified time.
415              return file_path_with_latest_mod_time
416          else:
417              # If there are more than one file having latest modified time,
418              # return the file path with the largest file name.
419              return file_path_with_largest_file_name
