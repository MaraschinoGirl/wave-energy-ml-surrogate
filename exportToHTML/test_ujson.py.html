33       """ 
34       Sanitize dictionary for JSON by converting all keys to strings. 
35    
36       Parameters 
37       ---------- 
38       d : dict 
39           The dictionary to convert. 
40    
41       Returns 
42       ------- 
43       cleaned_dict : dict 
44       """
45       return {str(k): v for k, v in d.items()}

114          string_input = "A string \\ / \b \f \n \r \t </script> &"
