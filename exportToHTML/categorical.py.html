56       def __init__(
57           self,
58           data=None,
59           variables={},
60           order=None,
61           orient=None,
62           require_numeric=False,
63           color=None,
64           legend="auto",
65       ):
66   
67           super().__init__(data=data, variables=variables)
68   
69           # This method takes care of some bookkeeping that is necessary because the
70           # original categorical plots (prior to the 2021 refactor) had some rules that
71           # don't fit exactly into VectorPlotter logic. It may be wise to have a second
72           # round of refactoring that moves the logic deeper, but this will keep things
73           # relatively sensible for now.
74   
75           # For wide data, orient determines assignment to x/y differently from the
76           # default VectorPlotter rules. If we do decide to make orient part of the
77           # _base variable assignment, we'll want to figure out how to express that.
78           if self.input_format == "wide" and orient in ["h", "y"]:
79               self.plot_data = self.plot_data.rename(columns={"x": "y", "y": "x"})
80               orig_variables = set(self.variables)
81               orig_x = self.variables.pop("x", None)
82               orig_y = self.variables.pop("y", None)
83               orig_x_type = self.var_types.pop("x", None)
84               orig_y_type = self.var_types.pop("y", None)
85               if "x" in orig_variables:
86                   self.variables["y"] = orig_x
87                   self.var_types["y"] = orig_x_type
88               if "y" in orig_variables:
89                   self.variables["x"] = orig_y
90                   self.var_types["x"] = orig_y_type
91   
92           # Initially there was more special code for wide-form data where plots were
93           # multi-colored by default and then either palette or color could be used.
94           # We want to provide backwards compatibility for this behavior in a relatively
95           # simply way, so we delete the hue information when color is specified.
96           if (
97               self.input_format == "wide"
98               and "hue" in self.variables
99               and color is not None
100          ):
101              self.plot_data.drop("hue", axis=1)
102              self.variables.pop("hue")
103  
104          # The concept of an "orientation" is important to the original categorical
105          # plots, but there's no provision for it in VectorPlotter, so we need it here.
106          # Note that it could be useful for the other functions in at least two ways
107          # (orienting a univariate distribution plot from long-form data and selecting
108          # the aggregation axis in lineplot), so we may want to eventually refactor it.
109          self.orient = infer_orient(
110              x=self.plot_data.get("x", None),
111              y=self.plot_data.get("y", None),
112              orient=orient,
113              require_numeric=False,
114          )
115  
116          self.legend = legend
117  
118          # Short-circuit in the case of an empty plot
119          if not self.has_xy_data:
120              return
121  
122          # Categorical plots can be "univariate" in which case they get an anonymous
123          # category label on the opposite axis. Note: this duplicates code in the core
124          # scale_categorical function. We need to do it here because of the next line.
125          if self.orient not in self.variables:
126              self.variables[self.orient] = None
127              self.var_types[self.orient] = "categorical"
128              self.plot_data[self.orient] = ""
129  
130          # Categorical variables have discrete levels that we need to track
131          cat_levels = categorical_order(self.plot_data[self.orient], order)
132          self.var_levels[self.orient] = cat_levels

134      def _hue_backcompat(self, color, palette, hue_order, force_hue=False):
135          """Implement backwards compatibility for hue parametrization. 
136   
137          Note: the force_hue parameter is used so that functions can be shown to 
138          pass existing tests during refactoring and then tested for new behavior. 
139          It can be removed after completion of the work. 
140   
141          """
142          # The original categorical functions applied a palette to the categorical axis
143          # by default. We want to require an explicit hue mapping, to be more consistent
144          # with how things work elsewhere now. I don't think there's any good way to
145          # do this gently -- because it's triggered by the default value of hue=None,
146          # users would always get a warning, unless we introduce some sentinel "default"
147          # argument for this change. That's possible, but asking users to set `hue=None`
148          # on every call is annoying.
149          # We are keeping the logic for implementing the old behavior in with the current
150          # system so that (a) we can punt on that decision and (b) we can ensure that
151          # refactored code passes old tests.
152          default_behavior = color is None or palette is not None
153          if force_hue and "hue" not in self.variables and default_behavior:
154              self._redundant_hue = True
155              self.plot_data["hue"] = self.plot_data[self.orient]
156              self.variables["hue"] = self.variables[self.orient]
157              self.var_types["hue"] = "categorical"
158              hue_order = self.var_levels[self.orient]
159  
160              # Because we convert the categorical axis variable to string,
161              # we need to update a dictionary palette too
162              if isinstance(palette, dict):
163                  palette = {str(k): v for k, v in palette.items()}
164  
165          else:
166              if "hue" in self.variables:
167                  redundant = (self.plot_data["hue"] == self.plot_data[self.orient]).all()
168              else:
169                  redundant = False
170              self._redundant_hue = redundant
171  
172          # Previously, categorical plots had a trick where color= could seed the palette.
173          # Because that's an explicit parameterization, we are going to give it one
174          # release cycle with a warning before removing.
175          if "hue" in self.variables and palette is None and color is not None:
176              if not isinstance(color, str):
177                  color = mpl.colors.to_hex(color)
178              palette = f"dark:{color}"
179              msg = (
180                  "\n\nSetting a gradient palette using color= is deprecated and will be "
181                  f"removed in v0.14.0. Set `palette='{palette}'` for the same effect.\n"
182              )
183              warnings.warn(msg, FutureWarning, stacklevel=3)
184  
185          return palette, hue_order

312      def _complement_color(self, color, base_color, hue_map):
313          """Allow a color to be set automatically using a basis of comparison."""
314          if color == "gray":
315              msg = (
316                  'Use "auto" to set automatic grayscale colors. From v0.14.0, '
317                  '"gray" will default to matplotlib\'s definition.'
318              )
319              warnings.warn(msg, FutureWarning, stacklevel=3)
320              color = "auto"
321          elif color is None or color is default:
322              color = "auto"
323  
324          if color != "auto":
325              return color
326  
327          if hue_map.lookup_table is None:
328              if base_color is None:
329                  return None
330              basis = [mpl.colors.to_rgb(base_color)]
331          else:
332              basis = [mpl.colors.to_rgb(c) for c in hue_map.lookup_table.values()]
333          unique_colors = np.unique(basis, axis=0)
334          light_vals = [rgb_to_hls(*rgb[:3])[1] for rgb in unique_colors]
335          lum = min(light_vals) * .6
336          return (lum, lum, lum)

354      def _adjust_cat_axis(self, ax, axis):
355          """Set ticks and limits for a categorical variable."""
356          # Note: in theory, this could happen in _attach for all categorical axes
357          # But two reasons not to do that:
358          # - If it happens before plotting, autoscaling messes up the plot limits
359          # - It would change existing plots from other seaborn functions
360          if self.var_types[axis] != "categorical":
361              return
362  
363          # If both x/y data are empty, the correct way to set up the plot is
364          # somewhat undefined; because we don't add null category data to the plot in
365          # this case we don't *have* a categorical axis (yet), so best to just bail.
366          if self.plot_data[axis].empty:
367              return
368  
369          # We can infer the total number of categories (including those from previous
370          # plots that are not part of the plot we are currently making) from the number
371          # of ticks, which matplotlib sets up while doing unit conversion. This feels
372          # slightly risky, as if we are relying on something that may be a matplotlib
373          # implementation detail. But I cannot think of a better way to keep track of
374          # the state from previous categorical calls (see GH2516 for context)
375          n = len(getattr(ax, f"get_{axis}ticks")())
376  
377          if axis == "x":
378              ax.xaxis.grid(False)
379              ax.set_xlim(-.5, n - .5, auto=None)
380          else:
381              ax.yaxis.grid(False)
382              # Note limits that correspond to previously-inverted y axis
383              ax.set_ylim(n - .5, -.5, auto=None)

385      def _dodge_needed(self):
386          """Return True when use of `hue` would cause overlaps."""
387          groupers = list({self.orient, "col", "row"} & set(self.variables))
388          if "hue" in self.variables:
389              orient = self.plot_data[groupers].value_counts()
390              paired = self.plot_data[[*groupers, "hue"]].value_counts()
391              return orient.size != paired.size
392          return False

394      def _dodge(self, keys, data):
395          """Apply a dodge transform to coordinates in place."""
396          if "hue" not in self.variables:
397              # Short-circuit if hue variable was not assigned
398              # We could potentially warn when hue=None, dodge=True, user may be confused
399              # But I think it's fine to just treat it as a no-op.
400              return
401          hue_idx = self._hue_map.levels.index(keys["hue"])
402          n = len(self._hue_map.levels)
403          data["width"] /= n
404  
405          full_width = data["width"] * n
406          offset = data["width"] * hue_idx + data["width"] / 2 - full_width / 2
407          data[self.orient] += offset

409      def _invert_scale(self, ax, data, vars=("x", "y")):
410          """Undo scaling after computation so data are plotted correctly."""
411          for var in vars:
412              _, inv = _get_transform_functions(ax, var[0])
413              if var == self.orient and "width" in data:
414                  hw = data["width"] / 2
415                  data["edge"] = inv(data[var] - hw)
416                  data["width"] = inv(data[var] + hw) - data["edge"].to_numpy()
417              for suf in ["", "min", "max"]:
418                  if (col := f"{var}{suf}") in data:
419                      data[col] = inv(data[col])

433      def _native_width(self):
434          """Return unit of width separating categories on native numeric scale."""
435          # Categorical data always have a unit width
436          if self.var_types[self.orient] == "categorical":
437              return 1
438  
439          # Otherwise, define the width as the smallest space between observations
440          unique_values = np.unique(self.comp_data[self.orient])
441          if len(unique_values) > 1:
442              native_width = np.nanmin(np.diff(unique_values))
443          else:
444              native_width = 1
445          return native_width

447      def _nested_offsets(self, width, dodge):
448          """Return offsets for each hue level for dodged plots."""
449          offsets = None
450          if "hue" in self.variables and self._hue_map.levels is not None:
451              n_levels = len(self._hue_map.levels)
452              if dodge:
453                  each_width = width / n_levels
454                  offsets = np.linspace(0, width - each_width, n_levels)
455                  offsets -= offsets.mean()
456              else:
457                  offsets = np.zeros(n_levels)
458          return offsets

466      def plot_strips(
467          self,
468          jitter,
469          dodge,
470          color,
471          plot_kws,
472      ):
473  
474          width = .8 * self._native_width
475          offsets = self._nested_offsets(width, dodge)
476  
477          if jitter is True:
478              jlim = 0.1
479          else:
480              jlim = float(jitter)
481          if "hue" in self.variables and dodge and self._hue_map.levels is not None:
482              jlim /= len(self._hue_map.levels)
483          jlim *= self._native_width
484          jitterer = partial(np.random.uniform, low=-jlim, high=+jlim)
485  
486          iter_vars = [self.orient]
487          if dodge:
488              iter_vars.append("hue")
489  
490          ax = self.ax
491          dodge_move = jitter_move = 0
492  
493          if "marker" in plot_kws and not MarkerStyle(plot_kws["marker"]).is_filled():
494              plot_kws.pop("edgecolor", None)
495  
496          for sub_vars, sub_data in self.iter_data(iter_vars,
497                                                   from_comp_data=True,
498                                                   allow_empty=True):
499  
500              ax = self._get_axes(sub_vars)
501  
502              if offsets is not None and (offsets != 0).any():
503                  dodge_move = offsets[sub_data["hue"].map(self._hue_map.levels.index)]
504  
505              jitter_move = jitterer(size=len(sub_data)) if len(sub_data) > 1 else 0
506  
507              adjusted_data = sub_data[self.orient] + dodge_move + jitter_move
508              sub_data[self.orient] = adjusted_data
509              self._invert_scale(ax, sub_data)
510  
511              points = ax.scatter(sub_data["x"], sub_data["y"], color=color, **plot_kws)
512              if "hue" in self.variables:
513                  points.set_facecolors(self._hue_map(sub_data["hue"]))
514  
515          self._configure_legend(ax, _scatter_legend_artist, common_kws=plot_kws)

517      def plot_swarms(
518          self,
519          dodge,
520          color,
521          warn_thresh,
522          plot_kws,
523      ):
524  
525          width = .8 * self._native_width
526          offsets = self._nested_offsets(width, dodge)
527  
528          iter_vars = [self.orient]
529          if dodge:
530              iter_vars.append("hue")
531  
532          ax = self.ax
533          point_collections = {}
534          dodge_move = 0
535  
536          if "marker" in plot_kws and not MarkerStyle(plot_kws["marker"]).is_filled():
537              plot_kws.pop("edgecolor", None)
538  
539          for sub_vars, sub_data in self.iter_data(iter_vars,
540                                                   from_comp_data=True,
541                                                   allow_empty=True):
542  
543              ax = self._get_axes(sub_vars)

546                  dodge_move = offsets[sub_data["hue"].map(self._hue_map.levels.index)]
547  
548              if not sub_data.empty:
549                  sub_data[self.orient] = sub_data[self.orient] + dodge_move
550  
551              self._invert_scale(ax, sub_data)
552  
553              points = ax.scatter(sub_data["x"], sub_data["y"], color=color, **plot_kws)
554              if "hue" in self.variables:
555                  points.set_facecolors(self._hue_map(sub_data["hue"]))
556  
557              if not sub_data.empty:
558                  point_collections[(ax, sub_data[self.orient].iloc[0])] = points
559  
560          beeswarm = Beeswarm(width=width, orient=self.orient, warn_thresh=warn_thresh)
561          for (ax, center), points in point_collections.items():
562              if points.get_offsets().shape[0] > 1:
563  
564                  def draw(points, renderer, *, center=center):
565  
566                      beeswarm(points, center)
567  
568                      if self.orient == "y":
569                          scalex = False
570                          scaley = ax.get_autoscaley_on()
571                      else:
572                          scalex = ax.get_autoscalex_on()
573                          scaley = False
574  
575                      # This prevents us from undoing the nice categorical axis limits
576                      # set in _adjust_cat_axis, because that method currently leave
577                      # the autoscale flag in its original setting. It may be better
578                      # to disable autoscaling there to avoid needing to do this.
579                      fixed_scale = self.var_types[self.orient] == "categorical"
580                      ax.update_datalim(points.get_datalim(ax.transData))
581                      if not fixed_scale and (scalex or scaley):
582                          ax.autoscale_view(scalex=scalex, scaley=scaley)
583  
584                      super(points.__class__, points).draw(renderer)
585  
586                  points.draw = draw.__get__(points)
587  
588          _draw_figure(ax.figure)
589          self._configure_legend(ax, _scatter_legend_artist, plot_kws)

747      def plot_boxens(
748          self,
749          width,
750          dodge,
751          gap,
752          fill,
753          color,
754          linecolor,
755          linewidth,
756          width_method,
757          k_depth,
758          outlier_prop,
759          trust_alpha,
760          showfliers,
761          box_kws,
762          flier_kws,
763          line_kws,
764          plot_kws,
765      ):
766  
767          iter_vars = [self.orient, "hue"]
768          value_var = {"x": "y", "y": "x"}[self.orient]
769  
770          estimator = LetterValues(k_depth, outlier_prop, trust_alpha)
771  
772          width_method_options = ["exponential", "linear", "area"]
773          _check_argument("width_method", width_method_options, width_method)
774  
775          box_kws = plot_kws if box_kws is None else {**plot_kws, **box_kws}
776          flier_kws = {} if flier_kws is None else flier_kws.copy()
777          line_kws = {} if line_kws is None else line_kws.copy()
778  
779          if linewidth is None:
780              if fill:
781                  linewidth = 0.5 * mpl.rcParams["lines.linewidth"]
782              else:
783                  linewidth = mpl.rcParams["lines.linewidth"]
784  
785          ax = self.ax
786  
787          for sub_vars, sub_data in self.iter_data(iter_vars,
788                                                   from_comp_data=True,
789                                                   allow_empty=False):
790  
791              ax = self._get_axes(sub_vars)
792              _, inv_ori = _get_transform_functions(ax, self.orient)
793              _, inv_val = _get_transform_functions(ax, value_var)
794  
795              # Statistics
796              lv_data = estimator(sub_data[value_var])
797              n = lv_data["k"] * 2 - 1
798              vals = lv_data["values"]
799  
800              pos_data = pd.DataFrame({
801                  self.orient: [sub_vars[self.orient]],
802                  "width": [width * self._native_width],
803              })
804              if dodge:
805                  self._dodge(sub_vars, pos_data)
806              if gap:
807                  pos_data["width"] *= 1 - gap
808  
809              # Letter-value boxes
810              levels = lv_data["levels"]
811              exponent = (levels - 1 - lv_data["k"]).astype(float)
812              if width_method == "linear":
813                  rel_widths = levels + 1
814              elif width_method == "exponential":
815                  rel_widths = 2 ** exponent
816              elif width_method == "area":
817                  tails = levels < (lv_data["k"] - 1)
818                  rel_widths = 2 ** (exponent - tails) / np.diff(lv_data["values"])
819  
820              center = pos_data[self.orient].item()
821              widths = rel_widths / rel_widths.max() * pos_data["width"].item()
822  
823              box_vals = inv_val(vals)
824              box_pos = inv_ori(center - widths / 2)
825              box_heights = inv_val(vals[1:]) - inv_val(vals[:-1])
826              box_widths = inv_ori(center + widths / 2) - inv_ori(center - widths / 2)
827  
828              maincolor = self._hue_map(sub_vars["hue"]) if "hue" in sub_vars else color
829              flier_colors = {
830                  "facecolor": "none", "edgecolor": ".45" if fill else maincolor
831              }
832              if fill:
833                  cmap = light_palette(maincolor, as_cmap=True)
834                  boxcolors = cmap(2 ** ((exponent + 2) / 3))
835              else:
836                  boxcolors = maincolor
837  
838              boxen = []
839              for i in range(n):
840                  if self.orient == "x":
841                      xy = (box_pos[i], box_vals[i])
842                      w, h = (box_widths[i], box_heights[i])
843                  else:
844                      xy = (box_vals[i], box_pos[i])
845                      w, h = (box_heights[i], box_widths[i])
846                  boxen.append(Rectangle(xy, w, h))
847  
848              if fill:
849                  box_colors = {"facecolors": boxcolors, "edgecolors": linecolor}
850              else:
851                  box_colors = {"facecolors": "none", "edgecolors": boxcolors}
852  
853              collection_kws = {**box_colors, "linewidth": linewidth, **box_kws}
854              ax.add_collection(PatchCollection(boxen, **collection_kws), autolim=False)
855              ax.update_datalim(
856                  np.column_stack([box_vals, box_vals]),
857                  updatex=self.orient == "y",
858                  updatey=self.orient == "x",
859              )
860  
861              # Median line
862              med = lv_data["median"]
863              hw = pos_data["width"].item() / 2
864              if self.orient == "x":
865                  x, y = inv_ori([center - hw, center + hw]), inv_val([med, med])
866              else:
867                  x, y = inv_val([med, med]), inv_ori([center - hw, center + hw])
868              default_kws = {
869                  "color": linecolor if fill else maincolor,
870                  "solid_capstyle": "butt",
871                  "linewidth": 1.25 * linewidth,
872              }
873              ax.plot(x, y, **{**default_kws, **line_kws})
874  
875              # Outliers ("fliers")
876              if showfliers:
877                  vals = inv_val(lv_data["fliers"])
878                  pos = np.full(len(vals), inv_ori(pos_data[self.orient].item()))
879                  x, y = (pos, vals) if self.orient == "x" else (vals, pos)
880                  ax.scatter(x, y, **{**flier_colors, "s": 25, **flier_kws})
881  
882          ax.autoscale_view(scalex=self.orient == "y", scaley=self.orient == "x")
883  
884          legend_artist = _get_patch_legend_artist(fill)
885          common_kws = {**box_kws, "linewidth": linewidth, "edgecolor": linecolor}
886          self._configure_legend(ax, legend_artist, common_kws)

906          iter_vars = [self.orient, "hue"]
907          value_var = {"x": "y", "y": "x"}[self.orient]

911          _check_argument("density_norm", ["area", "count", "width"], density_norm)

928          # Iterate through all the data splits once to compute the KDEs

930                                                   from_comp_data=True,
931                                                   allow_empty=False):
932  
933              sub_data["weight"] = sub_data.get("weights", 1)

940              default_kws = dict(

952                  "sub_vars": sub_vars,
953                  "ax": self._get_axes(sub_vars),
954              })
955  
956          # Once we've computed all the KDEs, get statistics for normalization
957          def vars_to_key(sub_vars):
958              return tuple((k, v) for k, v in sub_vars.items() if k != self.orient)
959  
960          norm_keys = [vars_to_key(violin["sub_vars"]) for violin in violin_data]
961          if common_norm:
962              common_max_density = np.nanmax([v["density"].max() for v in violin_data])
963              common_max_count = np.nanmax([len(v["observations"]) for v in violin_data])

965              max_count = {key: common_max_count for key in norm_keys}

967              with warnings.catch_warnings():

971                      key: np.nanmax([
972                          v["density"].max() for v in violin_data
973                          if vars_to_key(v["sub_vars"]) == key
974                      ]) for key in norm_keys
975                  }
976              max_count = {
977                  key: np.nanmax([
978                      len(v["observations"]) for v in violin_data
979                      if vars_to_key(v["sub_vars"]) == key

1003             hw = data["width"] / 2
1004             peak_density = violin["density"].max()

1009             elif density_norm == "count":
1010                 count = len(violin["observations"])

1018                 0 if "hue" not in self.variables

1116                 stats = mpl.cbook.boxplot_stats(obs)[0]
1117                 pos = np.array(pos_dict[self.orient])
1118                 if split:
1119                     pos += (-1 if right_side else 1) * pos_dict["width"] / 2

1121                 val = (
1122                     [stats["whislo"], stats["whishi"]],
1123                     [stats["q1"], stats["q3"]],
1124                     [stats["med"]]

1137                 else:
1138                     trans = ax.transData

1140                     "color": linecolor,
1141                     "transform": trans,
1142                     **inner_kws,
1143                     "linewidth": whis_width,

1158                 ax.plot(invx(x2), invy(y2), **dot_kws)
1159 
1160         legend_artist = _get_patch_legend_artist(fill)
1161         common_kws = {**plot_kws, "linewidth": linewidth, "edgecolor": linecolor}

3285     def beeswarm(self, orig_xyr):
3286         """Adjust x position of points to avoid overlaps."""
3287         # In this method, `x` is always the categorical axis
3288         # Center of the swarm, in point coordinates
3289         midline = orig_xyr[0, 0]
3290 
3291         # Start the swarm with the first point
3292         swarm = np.atleast_2d(orig_xyr[0])
3293 
3294         # Loop over the remaining points
3295         for xyr_i in orig_xyr[1:]:
3296 
3297             # Find the points in the swarm that could possibly
3298             # overlap with the point we are currently placing
3299             neighbors = self.could_overlap(xyr_i, swarm)
3300 
3301             # Find positions that would be valid individually
3302             # with respect to each of the swarm neighbors
3303             candidates = self.position_candidates(xyr_i, neighbors)
3304 
3305             # Sort candidates by their centrality
3306             offsets = np.abs(candidates[:, 0] - midline)
3307             candidates = candidates[np.argsort(offsets)]
3308 
3309             # Find the first candidate that does not overlap any neighbors
3310             new_xyr_i = self.first_non_overlapping_candidate(candidates, neighbors)

3312             # Place it into the swarm

3317     def could_overlap(self, xyr_i, swarm):
3318         """Return a list of all swarm points that could overlap with target."""
3319         # Because we work backwards through the swarm and can short-circuit,
3320         # the for-loop is faster than vectorization
3321         _, y_i, r_i = xyr_i
3322         neighbors = []
3323         for xyr_j in reversed(swarm):
3324             _, y_j, r_j = xyr_j
3325             if (y_i - y_j) < (r_i + r_j):
3326                 neighbors.append(xyr_j)
3327             else:
3328                 break
3329         return np.array(neighbors)[::-1]

3331     def position_candidates(self, xyr_i, neighbors):
3332         """Return a list of coordinates that might be valid by adjusting x."""
3333         candidates = [xyr_i]
3334         x_i, y_i, r_i = xyr_i
3335         left_first = True
3336         for x_j, y_j, r_j in neighbors:
3337             dy = y_i - y_j
3338             dx = np.sqrt(max((r_i + r_j) ** 2 - dy ** 2, 0)) * 1.05
3339             cl, cr = (x_j - dx, y_i, r_i), (x_j + dx, y_i, r_i)
3340             if left_first:
3341                 new_candidates = [cl, cr]
3342             else:
3343                 new_candidates = [cr, cl]
3344             candidates.extend(new_candidates)
3345             left_first = not left_first
3346         return np.array(candidates)

3348     def first_non_overlapping_candidate(self, candidates, neighbors):
3349         """Find the first candidate that does not overlap with the swarm."""
3350 
3351         # If we have no neighbors, all candidates are good.
3352         if len(neighbors) == 0:
3353             return candidates[0]
3354 
3355         neighbors_x = neighbors[:, 0]
3356         neighbors_y = neighbors[:, 1]
3357         neighbors_r = neighbors[:, 2]
3358 
3359         for xyr_i in candidates:
3360 
3361             x_i, y_i, r_i = xyr_i
3362 
3363             dx = neighbors_x - x_i
3364             dy = neighbors_y - y_i
3365             sq_distances = np.square(dx) + np.square(dy)
3366 
3367             sep_needed = np.square(neighbors_r + r_i)

3369             # Good candidate does not overlap any of neighbors which means that
3370             # squared distance between candidate and any of the neighbors has
3371             # to be at least square of the summed radii
3372             good_candidate = np.all(sq_distances >= sep_needed)

3386         if off_low.any():

3388         high_gutter = trans_inv(trans_fwd(center) + half_width)
3389         off_high = points > high_gutter
3390         if off_high.any():
3391             points[off_high] = high_gutter
3392

3397                 "to decrease the size of the markers or use stripplot."
