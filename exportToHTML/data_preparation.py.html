2    import pandas as pd
3    import numpy as np
4    def prepare_lstm_data(df, target='qW', sequence_length=10, top_features=None):
5        df = df.copy()
6    
7        # If top_features is given, select only those (including target column)
8        if top_features is not None:
9            missing = [col for col in top_features if col not in df.columns]
10           if missing:
11               raise KeyError(f"These top_features are missing from df: {missing}")
12           features = df[top_features].values
13       else:
14           # Drop irrelevant power cols if not using top features
15           power_cols = [col for col in df.columns if "Power" in col and col != target]
16           df = df.drop(columns=power_cols + ['Total_Power'])
17           features = df.drop(columns=[target]).values
18   
19       labels = df[target].values
20   
21       X, y = [], []
22       for i in range(sequence_length, len(df)):
23           X.append(features[i - sequence_length:i])
24           y.append(labels[i])
25   
26       return np.array(X), np.array(y)
