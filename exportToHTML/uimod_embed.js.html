5    window.addEventListener(
6      "load",
7      function () {
8        var containers = document.getElementsByClassName("terminado-container");
9        var container, rows, cols, protocol, ws_url;
10       for (var i = 0; i < containers.length; i++) {
11         container = containers[i];
12         rows = parseInt(container.dataset.rows);
13         cols = parseInt(container.dataset.cols);
14         protocol = window.location.protocol.indexOf("https") === 0 ? "wss" : "ws";
15         ws_url =
16           protocol + "://" + window.location.host + container.dataset.wsUrl;
17         make_terminal(container, { rows: rows, cols: cols }, ws_url);
18       }
19     },
20     false,
21   );
