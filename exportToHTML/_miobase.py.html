86   ''' 
87    
88    Note on architecture 
89   ====================== 
90    
91   There are three sets of parameters relevant for reading files. The 
92   first are *file read parameters* - containing options that are common 
93   for reading the whole file, and therefore every variable within that 
94   file. At the moment these are: 
95    
96   * mat_stream 
97   * dtypes (derived from byte code) 
98   * byte_order 
99   * chars_as_strings 
100  * squeeze_me 
101  * struct_as_record (MATLAB 5 files) 
102  * class_dtypes (derived from order code, MATLAB 5 files) 
103  * codecs (MATLAB 5 files) 
104  * uint16_codec (MATLAB 5 files) 
105   
106  Another set of parameters are those that apply only to the current 
107  variable being read - the *header*: 
108   
109  * header related variables (different for v4 and v5 mat files) 
110  * is_complex 
111  * mclass 
112  * var_stream 
113   
114  With the header, we need ``next_position`` to tell us where the next 
115  variable in the stream is. 
116   
117  Then, for each element in a matrix, there can be *element read 
118  parameters*. An element is, for example, one element in a MATLAB cell 
119  array. At the moment, these are: 
120   
121  * mat_dtype 
122   
123  The file-reading object contains the *file read parameters*. The 
124  *header* is passed around as a data object, or may be read and discarded 
125  in a single function. The *element read parameters* - the mat_dtype in 
126  this instance, is passed into a general post-processing function - see 
127  ``mio_utils`` for details. 
128  '''

131  def convert_dtypes(dtype_template, order_code):
132      ''' Convert dtypes in mapping to given order 
133   
134      Parameters 
135      ---------- 
136      dtype_template : mapping 
137         mapping with values returning numpy dtype from ``np.dtype(val)`` 
138      order_code : str 
139         an order code suitable for using in ``dtype.newbyteorder()`` 
140   
141      Returns 
142      ------- 
143      dtypes : mapping 
144         mapping where values have been replaced by 
145         ``np.dtype(val).newbyteorder(order_code)`` 
146   
147      '''
148      dtypes = dtype_template.copy()
149      for k in dtypes:
150          dtypes[k] = np.dtype(dtypes[k]).newbyteorder(order_code)
151      return dtypes

154  def read_dtype(mat_stream, a_dtype):
155      """ 
156      Generic get of byte stream data of known type 
157   
158      Parameters 
159      ---------- 
160      mat_stream : file_like object 
161          MATLAB (tm) mat file stream 
162      a_dtype : dtype 
163          dtype of array to read. `a_dtype` is assumed to be correct 
164          endianness. 
165   
166      Returns 
167      ------- 
168      arr : ndarray 
169          Array of dtype `a_dtype` read from stream. 
170   
171      """
172      num_bytes = a_dtype.itemsize
173      arr = np.ndarray(shape=(),
174                       dtype=a_dtype,
175                       buffer=mat_stream.read(num_bytes),
176                       order='F')
177      return arr

180  def matfile_version(file_name, *, appendmat=True):
181      """ 
182      Return major, minor tuple depending on apparent mat file type 
183   
184      Where: 
185   
186       #. 0,x -> version 4 format mat files 
187       #. 1,x -> version 5 format mat files 
188       #. 2,x -> version 7.3 format mat files (HDF format) 
189   
190      Parameters 
191      ---------- 
192      file_name : str 
193         Name of the mat file (do not need .mat extension if 
194         appendmat==True). Can also pass open file-like object. 
195      appendmat : bool, optional 
196         True to append the .mat extension to the end of the given 
197         filename, if not already present. Default is True. 
198   
199      Returns 
200      ------- 
201      major_version : {0, 1, 2} 
202          major MATLAB File format version 
203      minor_version : int 
204          minor MATLAB file format version 
205   
206      Raises 
207      ------ 
208      MatReadError 
209          If the file is empty. 
210      ValueError 
211          The matfile version is unknown. 
212   
213      Notes 
214      ----- 
215      Has the side effect of setting the file read pointer to 0 
216      """
217      from ._mio import _open_file_context
218      with _open_file_context(file_name, appendmat=appendmat) as fileobj:
219          return _get_matfile_version(fileobj)

225  _HDR_N_BYTES = 20
226  
227  
228  def _get_matfile_version(fileobj):
229      # Mat4 files have a zero somewhere in first 4 bytes
230      fileobj.seek(0)
231      hdr_bytes = fileobj.read(_HDR_N_BYTES)
232      if len(hdr_bytes) < _HDR_N_BYTES:
233          raise MatReadError("Mat file appears to be truncated")
234      if hdr_bytes.count(0) == _HDR_N_BYTES:
235          raise MatReadError("Mat file appears to be corrupt "
236                             f"(first {_HDR_N_BYTES} bytes == 0)")
237      mopt_ints = np.ndarray(shape=(4,), dtype=np.uint8, buffer=hdr_bytes[:4])
238      if 0 in mopt_ints:
239          fileobj.seek(0)
240          return (0,0)
241      # For 5 format or 7.3 format we need to read an integer in the
242      # header. Bytes 124 through 128 contain a version integer and an
243      # endian test string
244      fileobj.seek(124)
245      tst_str = fileobj.read(4)
246      fileobj.seek(0)
247      maj_ind = int(tst_str[2] == b'I'[0])
248      maj_val = int(tst_str[maj_ind])
249      min_val = int(tst_str[1 - maj_ind])
250      ret = (maj_val, min_val)
251      if maj_val in (1, 2):
252          return ret
253      raise ValueError('Unknown mat file type, version {}, {}'.format(*ret))

256  def matdims(arr, oned_as='column'):
257      """ 
258      Determine equivalent MATLAB dimensions for given array 
259   
260      Parameters 
261      ---------- 
262      arr : ndarray 
263          Input array 
264      oned_as : {'column', 'row'}, optional 
265          Whether 1-D arrays are returned as MATLAB row or column matrices. 
266          Default is 'column'. 
267   
268      Returns 
269      ------- 
270      dims : tuple 
271          Shape tuple, in the form MATLAB expects it. 
272   
273      Notes 
274      ----- 
275      We had to decide what shape a 1 dimensional array would be by 
276      default. ``np.atleast_2d`` thinks it is a row vector. The 
277      default for a vector in MATLAB (e.g., ``>> 1:12``) is a row vector. 
278   
279      Versions of scipy up to and including 0.11 resulted (accidentally) 
280      in 1-D arrays being read as column vectors. For the moment, we 
281      maintain the same tradition here. 
282   
283      Examples 
284      -------- 
285      >>> import numpy as np 
286      >>> from scipy.io.matlab._miobase import matdims 
287      >>> matdims(np.array(1)) # NumPy scalar 
288      (1, 1) 
289      >>> matdims(np.array([1])) # 1-D array, 1 element 
290      (1, 1) 
291      >>> matdims(np.array([1,2])) # 1-D array, 2 elements 
292      (2, 1) 
293      >>> matdims(np.array([[2],[3]])) # 2-D array, column vector 
294      (2, 1) 
295      >>> matdims(np.array([[2,3]])) # 2-D array, row vector 
296      (1, 2) 
297      >>> matdims(np.array([[[2,3]]])) # 3-D array, rowish vector 
298      (1, 1, 2) 
299      >>> matdims(np.array([])) # empty 1-D array 
300      (0, 0) 
301      >>> matdims(np.array([[]])) # empty 2-D array 
302      (0, 0) 
303      >>> matdims(np.array([[[]]])) # empty 3-D array 
304      (0, 0, 0) 
305   
306      Optional argument flips 1-D shape behavior. 
307   
308      >>> matdims(np.array([1,2]), 'row') # 1-D array, 2 elements 
309      (1, 2) 
310   
311      The argument has to make sense though 
312   
313      >>> matdims(np.array([1,2]), 'bizarre') 
314      Traceback (most recent call last): 
315         ... 
316      ValueError: 1-D option "bizarre" is strange 
317   
318      """
319      shape = arr.shape
320      if shape == ():  # scalar
321          return (1, 1)
322      if len(shape) == 1:  # 1D
323          if shape[0] == 0:
324              return (0, 0)
325          elif oned_as == 'column':
326              return shape + (1,)
327          elif oned_as == 'row':
328              return (1,) + shape
329          else:
330              raise ValueError(f'1-D option "{oned_as}" is strange')
331      return shape

348  class MatFileReader:
349      """ Base object for reading mat files 
350   
351      To make this class functional, you will need to override the 
352      following methods: 
353   
354      matrix_getter_factory   - gives object to fetch next matrix from stream 
355      guess_byte_order        - guesses file byte order from file 
356      """
357  
358      @docfiller
359      def __init__(self, mat_stream,
360                   byte_order=None,
361                   mat_dtype=False,
362                   squeeze_me=False,
363                   chars_as_strings=True,
364                   matlab_compatible=False,
365                   struct_as_record=True,
366                   verify_compressed_data_integrity=True,
367                   simplify_cells=False):
368          ''' 
369          Initializer for mat file reader 
370   
371          mat_stream : file-like 
372              object with file API, open for reading 
373      %(load_args)s 
374          '''
375          # Initialize stream
376          self.mat_stream = mat_stream
377          self.dtypes = {}
378          if not byte_order:
379              byte_order = self.guess_byte_order()
380          else:
381              byte_order = boc.to_numpy_code(byte_order)
382          self.byte_order = byte_order
383          self.struct_as_record = struct_as_record
384          if matlab_compatible:
385              self.set_matlab_compatible()
386          else:
387              self.squeeze_me = squeeze_me
388              self.chars_as_strings = chars_as_strings
389              self.mat_dtype = mat_dtype
390          self.verify_compressed_data_integrity = verify_compressed_data_integrity
391          self.simplify_cells = simplify_cells
392          if simplify_cells:
393              self.squeeze_me = True
394              self.struct_as_record = False
395  
396      def set_matlab_compatible(self):
397          ''' Sets options to return arrays as MATLAB loads them '''
398          self.mat_dtype = True
399          self.squeeze_me = False
400          self.chars_as_strings = False
401  
402      def guess_byte_order(self):
403          ''' As we do not know what file type we have, assume native '''
404          return boc.native_code
405  
406      def end_of_stream(self):
407          b = self.mat_stream.read(1)
408          curpos = self.mat_stream.tell()
409          self.mat_stream.seek(curpos-1)
410          return len(b) == 0

413  def arr_dtype_number(arr, num):
414      ''' Return dtype for given number of items per element'''
415      return np.dtype(arr.dtype.str[:2] + str(num))

418  def arr_to_chars(arr):
419      ''' Convert string array to char array '''
420      dims = list(arr.shape)
421      if not dims:
422          dims = [1]
423      dims.append(int(arr.dtype.str[2:]))
424      arr = np.ndarray(shape=dims,
425                       dtype=arr_dtype_number(arr, 1),
426                       buffer=arr)
427      empties = [arr == np.array('', dtype=arr.dtype)]
428      if not np.any(empties):
429          return arr
430      arr = arr.copy()
431      arr[tuple(empties)] = ' '
432      return arr
