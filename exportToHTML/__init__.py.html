4    import getpass
5    import os
6    from pathlib import Path
7    from tempfile import mkdtemp

9    
10   def configure_jupyter_server(c):
11       """Helper to configure the Jupyter Server for integration testing 
12       with Galata. 
13    
14       By default the tests will be executed in the OS temporary folder. You 
15       can override that folder by setting the environment variable ``JUPYTERLAB_GALATA_ROOT_DIR``. 
16    
17       .. warning:: 
18    
19           Never use this configuration in production as it will remove all security protections. 
20       """
21       # Test if we are running in a docker
22       if getpass.getuser() == "jovyan":
23           c.ServerApp.ip = "0.0.0.0"  # noqa S104
24   
25       c.ServerApp.port = 8888
26       c.ServerApp.port_retries = 0
27       c.ServerApp.open_browser = False
28       # Add test helpers extension shipped with JupyterLab.
29       # You can replace the following line by the two following one
30       #   import jupyterlab
31       #   c.LabServerApp.extra_labextensions_path = str(Path(jupyterlab.__file__).parent / "galata")
32       c.LabServerApp.extra_labextensions_path = str(Path(__file__).parent)
33   
34       c.LabApp.workspaces_dir = mkdtemp(prefix="galata-workspaces-")
35   
36       c.ServerApp.root_dir = os.environ.get(
37           "JUPYTERLAB_GALATA_ROOT_DIR", mkdtemp(prefix="galata-test-")
38       )
39       c.IdentityProvider.token = ""
40       c.ServerApp.password = ""
41       c.ServerApp.disable_check_xsrf = True
42       c.LabApp.expose_app_in_browser = True
