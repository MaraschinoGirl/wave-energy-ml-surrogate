37       @torch_utils.no_grad
38       def _apply_weight_decay(self, variables):
39           if self.weight_decay is None:
40               return
41   
42           torch._foreach_mul_(
43               [v.value for v in variables if self._use_weight_decay(v)],
44               1 - self.weight_decay * self._get_current_learning_rate(),
45           )
