25   class KBinsDiscretizer(TransformerMixin, BaseEstimator):
26       """ 
27       Bin continuous data into intervals. 
28    
29       Read more in the :ref:`User Guide <preprocessing_discretization>`. 
30    
31       .. versionadded:: 0.20 
32    
33       Parameters 
34       ---------- 
35       n_bins : int or array-like of shape (n_features,), default=5 
36           The number of bins to produce. Raises ValueError if ``n_bins < 2``. 
37    
38       encode : {'onehot', 'onehot-dense', 'ordinal'}, default='onehot' 
39           Method used to encode the transformed result. 
40    
41           - 'onehot': Encode the transformed result with one-hot encoding 
42             and return a sparse matrix. Ignored features are always 
43             stacked to the right. 
44           - 'onehot-dense': Encode the transformed result with one-hot encoding 
45             and return a dense array. Ignored features are always 
46             stacked to the right. 
47           - 'ordinal': Return the bin identifier encoded as an integer value. 
48    
49       strategy : {'uniform', 'quantile', 'kmeans'}, default='quantile' 
50           Strategy used to define the widths of the bins. 
51    
52           - 'uniform': All bins in each feature have identical widths. 
53           - 'quantile': All bins in each feature have the same number of points. 
54           - 'kmeans': Values in each bin have the same nearest center of a 1D 
55             k-means cluster. 
56    
57           For an example of the different strategies see: 
58           :ref:`sphx_glr_auto_examples_preprocessing_plot_discretization_strategies.py`. 
59    
60       dtype : {np.float32, np.float64}, default=None 
61           The desired data-type for the output. If None, output dtype is 
62           consistent with input dtype. Only np.float32 and np.float64 are 
63           supported. 
64    
65           .. versionadded:: 0.24 
66    
67       subsample : int or None, default=200_000 
68           Maximum number of samples, used to fit the model, for computational 
69           efficiency. 
70           `subsample=None` means that all the training samples are used when 
71           computing the quantiles that determine the binning thresholds. 
72           Since quantile computation relies on sorting each column of `X` and 
73           that sorting has an `n log(n)` time complexity, 
74           it is recommended to use subsampling on datasets with a 
75           very large number of samples. 
76    
77           .. versionchanged:: 1.3 
78               The default value of `subsample` changed from `None` to `200_000` when 
79               `strategy="quantile"`. 
80    
81           .. versionchanged:: 1.5 
82               The default value of `subsample` changed from `None` to `200_000` when 
83               `strategy="uniform"` or `strategy="kmeans"`. 
84    
85       random_state : int, RandomState instance or None, default=None 
86           Determines random number generation for subsampling. 
87           Pass an int for reproducible results across multiple function calls. 
88           See the `subsample` parameter for more details. 
89           See :term:`Glossary <random_state>`. 
90    
91           .. versionadded:: 1.1 
92    
93       Attributes 
94       ---------- 
95       bin_edges_ : ndarray of ndarray of shape (n_features,) 
96           The edges of each bin. Contain arrays of varying shapes ``(n_bins_, )`` 
97           Ignored features will have empty arrays. 
98    
99       n_bins_ : ndarray of shape (n_features,), dtype=np.int64 
100          Number of bins per feature. Bins whose width are too small 
101          (i.e., <= 1e-8) are removed with a warning. 
102   
103      n_features_in_ : int 
104          Number of features seen during :term:`fit`. 
105   
106          .. versionadded:: 0.24 
107   
108      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
109          Names of features seen during :term:`fit`. Defined only when `X` 
110          has feature names that are all strings. 
111   
112          .. versionadded:: 1.0 
113   
114      See Also 
115      -------- 
116      Binarizer : Class used to bin values as ``0`` or 
117          ``1`` based on a parameter ``threshold``. 
118   
119      Notes 
120      ----- 
121   
122      For a visualization of discretization on different datasets refer to 
123      :ref:`sphx_glr_auto_examples_preprocessing_plot_discretization_classification.py`. 
124      On the effect of discretization on linear models see: 
125      :ref:`sphx_glr_auto_examples_preprocessing_plot_discretization.py`. 
126   
127      In bin edges for feature ``i``, the first and last values are used only for 
128      ``inverse_transform``. During transform, bin edges are extended to:: 
129   
130        np.concatenate([-np.inf, bin_edges_[i][1:-1], np.inf]) 
131   
132      You can combine ``KBinsDiscretizer`` with 
133      :class:`~sklearn.compose.ColumnTransformer` if you only want to preprocess 
134      part of the features. 
135   
136      ``KBinsDiscretizer`` might produce constant features (e.g., when 
137      ``encode = 'onehot'`` and certain bins do not contain any data). 
138      These features can be removed with feature selection algorithms 
139      (e.g., :class:`~sklearn.feature_selection.VarianceThreshold`). 
140   
141      Examples 
142      -------- 
143      >>> from sklearn.preprocessing import KBinsDiscretizer 
144      >>> X = [[-2, 1, -4,   -1], 
145      ...      [-1, 2, -3, -0.5], 
146      ...      [ 0, 3, -2,  0.5], 
147      ...      [ 1, 4, -1,    2]] 
148      >>> est = KBinsDiscretizer( 
149      ...     n_bins=3, encode='ordinal', strategy='uniform' 
150      ... ) 
151      >>> est.fit(X) 
152      KBinsDiscretizer(...) 
153      >>> Xt = est.transform(X) 
154      >>> Xt  # doctest: +SKIP 
155      array([[ 0., 0., 0., 0.], 
156             [ 1., 1., 1., 0.], 
157             [ 2., 2., 2., 1.], 
158             [ 2., 2., 2., 2.]]) 
159   
160      Sometimes it may be useful to convert the data back into the original 
161      feature space. The ``inverse_transform`` function converts the binned 
162      data into the original feature space. Each value will be equal to the mean 
163      of the two bin edges. 
164   
165      >>> est.bin_edges_[0] 
166      array([-2., -1.,  0.,  1.]) 
167      >>> est.inverse_transform(Xt) 
168      array([[-1.5,  1.5, -3.5, -0.5], 
169             [-0.5,  2.5, -2.5, -0.5], 
170             [ 0.5,  3.5, -1.5,  0.5], 
171             [ 0.5,  3.5, -1.5,  1.5]]) 
172      """
173  
174      _parameter_constraints: dict = {
175          "n_bins": [Interval(Integral, 2, None, closed="left"), "array-like"],
176          "encode": [StrOptions({"onehot", "onehot-dense", "ordinal"})],
177          "strategy": [StrOptions({"uniform", "quantile", "kmeans"})],
178          "dtype": [Options(type, {np.float64, np.float32}), None],
179          "subsample": [Interval(Integral, 1, None, closed="left"), None],
180          "random_state": ["random_state"],
181      }
182  
183      def __init__(
184          self,
185          n_bins=5,
186          *,
187          encode="onehot",
188          strategy="quantile",
189          dtype=None,
190          subsample=200_000,
191          random_state=None,
192      ):
193          self.n_bins = n_bins
194          self.encode = encode
195          self.strategy = strategy
196          self.dtype = dtype
197          self.subsample = subsample
198          self.random_state = random_state
199  
200      @_fit_context(prefer_skip_nested_validation=True)
201      def fit(self, X, y=None, sample_weight=None):
202          """ 
203          Fit the estimator. 
204   
205          Parameters 
206          ---------- 
207          X : array-like of shape (n_samples, n_features) 
208              Data to be discretized. 
209   
210          y : None 
211              Ignored. This parameter exists only for compatibility with 
212              :class:`~sklearn.pipeline.Pipeline`. 
213   
214          sample_weight : ndarray of shape (n_samples,) 
215              Contains weight values to be associated with each sample. 
216              Cannot be used when `strategy` is set to `"uniform"`. 
217   
218              .. versionadded:: 1.3 
219   
220          Returns 
221          ------- 
222          self : object 
223              Returns the instance itself. 
224          """
225          X = validate_data(self, X, dtype="numeric")
226  
227          if self.dtype in (np.float64, np.float32):
228              output_dtype = self.dtype
229          else:  # self.dtype is None
230              output_dtype = X.dtype
231  
232          n_samples, n_features = X.shape
233  
234          if sample_weight is not None and self.strategy == "uniform":
235              raise ValueError(
236                  "`sample_weight` was provided but it cannot be "
237                  "used with strategy='uniform'. Got strategy="
238                  f"{self.strategy!r} instead."
239              )
240  
241          if self.subsample is not None and n_samples > self.subsample:
242              # Take a subsample of `X`
243              X = resample(
244                  X,
245                  replace=False,
246                  n_samples=self.subsample,
247                  random_state=self.random_state,
248              )
249  
250          n_features = X.shape[1]
251          n_bins = self._validate_n_bins(n_features)
252  
253          if sample_weight is not None:
254              sample_weight = _check_sample_weight(sample_weight, X, dtype=X.dtype)
255  
256          bin_edges = np.zeros(n_features, dtype=object)
257          for jj in range(n_features):
258              column = X[:, jj]
259              col_min, col_max = column.min(), column.max()
260  
261              if col_min == col_max:
262                  warnings.warn(
263                      "Feature %d is constant and will be replaced with 0." % jj
264                  )
265                  n_bins[jj] = 1
266                  bin_edges[jj] = np.array([-np.inf, np.inf])
267                  continue
268  
269              if self.strategy == "uniform":
270                  bin_edges[jj] = np.linspace(col_min, col_max, n_bins[jj] + 1)
271  
272              elif self.strategy == "quantile":
273                  quantiles = np.linspace(0, 100, n_bins[jj] + 1)
274                  if sample_weight is None:
275                      bin_edges[jj] = np.asarray(np.percentile(column, quantiles))
276                  else:
277                      bin_edges[jj] = np.asarray(
278                          [
279                              _weighted_percentile(column, sample_weight, q)
280                              for q in quantiles
281                          ],
282                          dtype=np.float64,
283                      )
284              elif self.strategy == "kmeans":
285                  from ..cluster import KMeans  # fixes import loops
286  
287                  # Deterministic initialization with uniform spacing
288                  uniform_edges = np.linspace(col_min, col_max, n_bins[jj] + 1)
289                  init = (uniform_edges[1:] + uniform_edges[:-1])[:, None] * 0.5
290  
291                  # 1D k-means procedure
292                  km = KMeans(n_clusters=n_bins[jj], init=init, n_init=1)
293                  centers = km.fit(
294                      column[:, None], sample_weight=sample_weight
295                  ).cluster_centers_[:, 0]
296                  # Must sort, centers may be unsorted even with sorted init
297                  centers.sort()
298                  bin_edges[jj] = (centers[1:] + centers[:-1]) * 0.5
299                  bin_edges[jj] = np.r_[col_min, bin_edges[jj], col_max]
300  
301              # Remove bins whose width are too small (i.e., <= 1e-8)
302              if self.strategy in ("quantile", "kmeans"):
303                  mask = np.ediff1d(bin_edges[jj], to_begin=np.inf) > 1e-8
304                  bin_edges[jj] = bin_edges[jj][mask]
305                  if len(bin_edges[jj]) - 1 != n_bins[jj]:
306                      warnings.warn(
307                          "Bins whose width are too small (i.e., <= "
308                          "1e-8) in feature %d are removed. Consider "
309                          "decreasing the number of bins." % jj
310                      )
311                      n_bins[jj] = len(bin_edges[jj]) - 1
312  
313          self.bin_edges_ = bin_edges
314          self.n_bins_ = n_bins
315  
316          if "onehot" in self.encode:
317              self._encoder = OneHotEncoder(
318                  categories=[np.arange(i) for i in self.n_bins_],
319                  sparse_output=self.encode == "onehot",
320                  dtype=output_dtype,
321              )
322              # Fit the OneHotEncoder with toy datasets
323              # so that it's ready for use after the KBinsDiscretizer is fitted
324              self._encoder.fit(np.zeros((1, len(self.n_bins_))))
325  
326          return self
327  
328      def _validate_n_bins(self, n_features):
329          """Returns n_bins_, the number of bins per feature."""
330          orig_bins = self.n_bins
331          if isinstance(orig_bins, Integral):
332              return np.full(n_features, orig_bins, dtype=int)
333  
334          n_bins = check_array(orig_bins, dtype=int, copy=True, ensure_2d=False)
335  
336          if n_bins.ndim > 1 or n_bins.shape[0] != n_features:
337              raise ValueError("n_bins must be a scalar or array of shape (n_features,).")
338  
339          bad_nbins_value = (n_bins < 2) | (n_bins != orig_bins)
340  
341          violating_indices = np.where(bad_nbins_value)[0]
342          if violating_indices.shape[0] > 0:
343              indices = ", ".join(str(i) for i in violating_indices)
344              raise ValueError(
345                  "{} received an invalid number "
346                  "of bins at indices {}. Number of bins "
347                  "must be at least 2, and must be an int.".format(
348                      KBinsDiscretizer.__name__, indices
349                  )
350              )
351          return n_bins
352  
353      def transform(self, X):
354          """ 
355          Discretize the data. 
356   
357          Parameters 
358          ---------- 
359          X : array-like of shape (n_samples, n_features) 
360              Data to be discretized. 
361   
362          Returns 
363          ------- 
364          Xt : {ndarray, sparse matrix}, dtype={np.float32, np.float64} 
365              Data in the binned space. Will be a sparse matrix if 
366              `self.encode='onehot'` and ndarray otherwise. 
367          """
368          check_is_fitted(self)
369  
370          # check input and attribute dtypes
371          dtype = (np.float64, np.float32) if self.dtype is None else self.dtype
372          Xt = validate_data(self, X, copy=True, dtype=dtype, reset=False)
373  
374          bin_edges = self.bin_edges_
375          for jj in range(Xt.shape[1]):
376              Xt[:, jj] = np.searchsorted(bin_edges[jj][1:-1], Xt[:, jj], side="right")
377  
378          if self.encode == "ordinal":
379              return Xt
380  
381          dtype_init = None
382          if "onehot" in self.encode:
383              dtype_init = self._encoder.dtype
384              self._encoder.dtype = Xt.dtype
385          try:
386              Xt_enc = self._encoder.transform(Xt)
387          finally:
388              # revert the initial dtype to avoid modifying self.
389              self._encoder.dtype = dtype_init
390          return Xt_enc
391  
392      def inverse_transform(self, X=None, *, Xt=None):
393          """ 
394          Transform discretized data back to original feature space. 
395   
396          Note that this function does not regenerate the original data 
397          due to discretization rounding. 
398   
399          Parameters 
400          ---------- 
401          X : array-like of shape (n_samples, n_features) 
402              Transformed data in the binned space. 
403   
404          Xt : array-like of shape (n_samples, n_features) 
405              Transformed data in the binned space. 
406   
407              .. deprecated:: 1.5 
408                  `Xt` was deprecated in 1.5 and will be removed in 1.7. Use `X` instead. 
409   
410          Returns 
411          ------- 
412          Xinv : ndarray, dtype={np.float32, np.float64} 
413              Data in the original feature space. 
414          """
415          X = _deprecate_Xt_in_inverse_transform(X, Xt)
416  
417          check_is_fitted(self)
418  
419          if "onehot" in self.encode:
420              X = self._encoder.inverse_transform(X)
421  
422          Xinv = check_array(X, copy=True, dtype=(np.float64, np.float32))
423          n_features = self.n_bins_.shape[0]
424          if Xinv.shape[1] != n_features:
425              raise ValueError(
426                  "Incorrect number of features. Expecting {}, received {}.".format(
427                      n_features, Xinv.shape[1]
428                  )
429              )
430  
431          for jj in range(n_features):
432              bin_edges = self.bin_edges_[jj]
433              bin_centers = (bin_edges[1:] + bin_edges[:-1]) * 0.5
434              Xinv[:, jj] = bin_centers[(Xinv[:, jj]).astype(np.int64)]
435  
436          return Xinv
437  
438      def get_feature_names_out(self, input_features=None):
439          """Get output feature names. 
440   
441          Parameters 
442          ---------- 
443          input_features : array-like of str or None, default=None 
444              Input features. 
445   
446              - If `input_features` is `None`, then `feature_names_in_` is 
447                used as feature names in. If `feature_names_in_` is not defined, 
448                then the following input feature names are generated: 
449                `["x0", "x1", ..., "x(n_features_in_ - 1)"]`. 
450              - If `input_features` is an array-like, then `input_features` must 
451                match `feature_names_in_` if `feature_names_in_` is defined. 
452   
453          Returns 
454          ------- 
455          feature_names_out : ndarray of str objects 
456              Transformed feature names. 
457          """
458          check_is_fitted(self, "n_features_in_")
459          input_features = _check_feature_names_in(self, input_features)
460          if hasattr(self, "_encoder"):
461              return self._encoder.get_feature_names_out(input_features)
462  
463          # ordinal encoding
464          return input_features
