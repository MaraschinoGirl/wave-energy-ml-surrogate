makegw.py 
1    """Utility functions for writing out gateway C++ files 
2     
3    This module will generate a C++/Python binding for a specific COM 
4    interface. 
5     
6    Can be run from command line (passing required arguments) or the old way 
7    (start Python,  import this module,  change to the directory where the generated code 
8    should be written,  and run the public function). 
9     
10   This module is capable of generating both 'Interfaces' (ie, Python 
11   client side support for the interface) and 'Gateways' (ie, Python 
12   server side support for the interface).  Many COM interfaces are useful 
13   both as Client and Server.  Other interfaces, however, really only make 
14   sense to implement one side or the other.  For example, it would be pointless 
15   for Python to implement Server side for 'IRunningObjectTable', unless we were 
16   implementing core COM for an operating system in Python (hey - now there's an idea!) 
17    
18   Most COM interface code is totally boiler-plate - it consists of 
19   converting arguments, dispatching the call to Python, and processing 
20   any result values. 
21    
22   This module automates the generation of such code.  It has the ability to 
23   parse a .h file generated by the MIDL tool (ie, almost all COM .h files) 
24   and build almost totally complete C++ code. 
25    
26   The module understands some of the well known data types, and how to 
27   convert them.  There are only a couple of places where hand-editing is 
28   necessary, as detailed below: 
29    
30   unsupported types -- If a type is not known, the generator will 
31   pretty much ignore it, but write a comment to the generated code.  You 
32   may want to add custom support for this type.  In some cases, C++ compile errors 
33   will result.  These are intentional - generating code to remove these errors would 
34   imply a false sense of security that the generator has done the right thing. 
35    
36   other return policies -- By default, Python never sees the return SCODE from 
37   a COM function.  The interface usually returns None if OK, else a COM exception 
38   if "FAILED(scode)" is TRUE.  You may need to change this if: 
39   * EXCEPINFO is passed to the COM function.  This is not detected and handled 
40   * For some reason Python should always see the result SCODE, even if it 
41     did fail or succeed.  For example, some functions return a BOOLEAN result 
42     in the SCODE, meaning Python should always see it. 
43   * FAILED(scode) for the interface still has valid data to return (by default, 
44     the code generated does not process the return values, and raise an exception 
45     to Python/COM 
46   """

55   def make_framework_support(
56       header_file_name,
57       interface_name,
58       bMakeInterface=1,
59       bMakeGateway=1,
60       output_directory=None,
61   ):
62       """Generate C++ code for a Python Interface and Gateway 
63    
64       header_file_name -- The full path to the .h file which defines the interface. 
65       interface_name -- The name of the interface to search for, and to generate. 
66       bMakeInterface = 1 -- Should interface (ie, client) support be generated. 
67       bMakeGatewayInterface = 1 -- Should gateway (ie, server) support be generated. 
68    
69       This method will write a .cpp and .h file into the current directory, 
70       (using the name of the interface to build the file name. 
71    
72       """
73       fin = open(header_file_name)
74       try:
75           interface = makegwparse.parse_interface_info(interface_name, fin)
76       finally:
77           fin.close()
78   
79       if bMakeInterface and bMakeGateway:
80           desc = "Interface and Gateway"
81       elif bMakeInterface and not bMakeGateway:
82           desc = "Interface"
83       else:
84           desc = "Gateway"
85       if interface.name[:5] == "IEnum":  # IEnum - use my really simple template-based one
86           import win32com.makegw.makegwenum
87   
88           ifc_cpp_writer = win32com.makegw.makegwenum._write_enumifc_cpp
89           gw_cpp_writer = win32com.makegw.makegwenum._write_enumgw_cpp
90       else:  # Use my harder working ones.
91           ifc_cpp_writer = _write_ifc_cpp
92           gw_cpp_writer = _write_gw_cpp
93   
94       fout = open(
95           os.path.join(
96               directory if directory else os.getcwd(), f"Py{interface.name}.cpp"
97           ),
98           "w",
99       )
100      try:
101          fout.write(
102              f"""\ 
103  // This file implements the {interface.name} {desc} for Python.
104  // Generated by makegw.py
105  
106  #include "shell_pch.h"
107  """
108          )
109          #    if bMakeGateway:
110          #      fout.write('#include "PythonCOMServer.h"\n')
111          #    if interface.base not in ["IUnknown", "IDispatch"]:
112          #      fout.write('#include "Py%s.h"\n' % interface.base)
113          fout.write(
114              '#include "Py%s.h"\n\n// @doc - This file contains autoduck documentation\n'
115              % interface.name
116          )
117          if bMakeInterface:
118              ifc_cpp_writer(fout, interface)
119          if bMakeGateway:
120              gw_cpp_writer(fout, interface)
121      finally:
122          fout.close()
123      fout = open(
124          os.path.join(directory if directory else os.getcwd(), f"Py{interface.name}.h"),
125          "w",
126      )
127      try:
128          fout.write(
129              f"""\ 
130  // This file declares the {interface.name} {desc} for Python.
131  // Generated by makegw.py
132  """
133          )
134  
135          if bMakeInterface:
136              _write_ifc_h(fout, interface)
137          if bMakeGateway:
138              _write_gw_h(fout, interface)
139      finally:
140          fout.close()

150  def _write_ifc_h(f, interface):
151      f.write(
152          f"""\ 
153  // ---------------------------------------------------
154  //
155  // Interface Declaration
156  
157  class Py{interface.name} : public Py{interface.base}
158  {{
159  public:
160      MAKE_PYCOM_CTOR(Py{interface.name});
161      static {interface.name} *GetI(PyObject *self);
162      static PyComTypeObject type;
163  
164      // The Python methods
165  """
166      )
167      for method in interface.methods:
168          f.write(
169              "\tstatic PyObject *%s(PyObject *self, PyObject *args);\n" % method.name
170          )
171      f.write(
172          f"""\ 
173  
174  protected:
175      Py{interface.name}(IUnknown *pdisp);
176      ~Py{interface.name}();
177  }};
178  """
179      )

182  def _write_ifc_cpp(f, interface):
183      name = interface.name
184      f.write(
185          """\ 
186  // --------------------------------------------------- 
187  // 
188  // Interface Implementation 
189   
190  Py{name}::Py{name}(IUnknown *pdisp): 
191      Py{base}(pdisp) 
192  {{ 
193      ob_type = &type; 
194  }} 
195   
196  Py{name}::~Py{name}() 
197  {{ 
198  }} 
199   
200  /* static */ {name} *Py{name}::GetI(PyObject *self) 
201  {{ 
202      return ({name} *)Py{base}::GetI(self); 
203  }} 
204   
205  """.format(**interface.__dict__)
206      )
207  
208      ptr = re.sub(r"[a-z]", "", interface.name)
209      strdict = {"interfacename": interface.name, "ptr": ptr}
210      for method in interface.methods:
211          strdict["method"] = method.name
212          f.write(
213              """\ 
214  // @pymethod |Py{interfacename}|{method}|Description of {method}. 
215  PyObject *Py{interfacename}::{method}(PyObject *self, PyObject *args) 
216  {{ 
217      {interfacename} *p{ptr} = GetI(self); 
218      if ( p{ptr} == NULL ) 
219          return NULL; 
220  """.format(**strdict)
221          )
222          argsParseTuple = argsCOM = formatChars = codePost = codePobjects = (
223              codeCobjects
224          ) = cleanup = cleanup_gil = ""
225          needConversion = 0
226          #    if method.name=="Stat": import win32dbg;win32dbg.brk()
227          for arg in method.args:
228              try:
229                  argCvt = makegwparse.make_arg_converter(arg)
230                  if arg.HasAttribute("in"):
231                      val = argCvt.GetFormatChar()
232                      if val:
233                          f.write("\t" + argCvt.GetAutoduckString() + "\n")
234                          formatChars += val
235                          argsParseTuple += ", " + argCvt.GetParseTupleArg()
236                          codePobjects += argCvt.DeclareParseArgTupleInputConverter()
237                          codePost += argCvt.GetParsePostCode()
238                          needConversion = needConversion or argCvt.NeedUSES_CONVERSION()
239                          cleanup += argCvt.GetInterfaceArgCleanup()
240                          cleanup_gil += argCvt.GetInterfaceArgCleanupGIL()
241                  comArgName, comArgDeclString = argCvt.GetInterfaceCppObjectInfo()
242                  if comArgDeclString:  # If we should declare a variable
243                      codeCobjects += "\t%s;\n" % comArgDeclString
244                  argsCOM += ", " + comArgName
245              except makegwparse.error_not_supported as why:
246                  f.write(
247                      '// *** The input argument {} of type "{}" was not processed ***\n//     Please check the conversion function is appropriate and exists!\n'.format(
248                          arg.name, arg.raw_type
249                      )
250                  )
251  
252                  f.write(f"\t{arg.type} {arg.name};\n\tPyObject *ob{arg.name};\n")
253                  f.write(
254                      "\t// @pyparm <o Py{}>|{}||Description for {}\n".format(
255                          arg.type, arg.name, arg.name
256                      )
257                  )
258                  codePost += "\tif (bPythonIsHappy && !PyObject_As{}( ob{}, &{} )) bPythonIsHappy = FALSE;\n".format(
259                      arg.type, arg.name, arg.name
260                  )
261  
262                  formatChars += "O"
263                  argsParseTuple += ", &ob%s" % arg.name
264  
265                  argsCOM += ", " + arg.name
266                  cleanup += f"\tPyObject_Free{arg.type}({arg.name});\n"
267  
268          if needConversion:
269              f.write("\tUSES_CONVERSION;\n")
270          f.write(codePobjects)
271          f.write(codeCobjects)
272          f.write(
273              '\tif ( !PyArg_ParseTuple(args, "{}:{}"{}) )\n\t\treturn NULL;\n'.format(
274                  formatChars, method.name, argsParseTuple
275              )
276          )
277          if codePost:
278              f.write("\tBOOL bPythonIsHappy = TRUE;\n")
279              f.write(codePost)
280              f.write("\tif (!bPythonIsHappy) return NULL;\n")
281          strdict["argsCOM"] = argsCOM[1:]
282          strdict["cleanup"] = cleanup
283          strdict["cleanup_gil"] = cleanup_gil
284          f.write(
285              """    HRESULT hr; 
286      PY_INTERFACE_PRECALL; 
287      hr = p{ptr}->{method}({argsCOM} ); 
288  {cleanup} 
289      PY_INTERFACE_POSTCALL; 
290  {cleanup_gil} 
291      if ( FAILED(hr) ) 
292          return PyCom_BuildPyException(hr, p{ptr}, IID_{interfacename} ); 
293  """.format(**strdict)
294          )
295          codePre = codePost = formatChars = codeVarsPass = codeDecl = ""
296          for arg in method.args:
297              if not arg.HasAttribute("out"):
298                  continue
299              try:
300                  argCvt = makegwparse.make_arg_converter(arg)
301                  formatChar = argCvt.GetFormatChar()
302                  if formatChar:
303                      formatChars += formatChar
304                      codePre += argCvt.GetBuildForInterfacePreCode()
305                      codePost += argCvt.GetBuildForInterfacePostCode()
306                      codeVarsPass += ", " + argCvt.GetBuildValueArg()
307                      codeDecl += argCvt.DeclareParseArgTupleInputConverter()
308              except makegwparse.error_not_supported as why:
309                  f.write(
310                      '// *** The output argument {} of type "{}" was not processed ***\n//     {}\n'.format(
311                          arg.name, arg.raw_type, why
312                      )
313                  )
314                  continue
315          if formatChars:
316              f.write(
317                  '{}\n{}\tPyObject *pyretval = Py_BuildValue("{}"{});\n{}\treturn pyretval;'.format(
318                      codeDecl, codePre, formatChars, codeVarsPass, codePost
319                  )
320              )
321          else:
322              f.write("\tPy_RETURN_NONE;\n")
323          f.write("\n}\n\n")
324  
325      f.write("// @object Py%s|Description of the interface\n" % (name))
326      f.write("static struct PyMethodDef Py%s_methods[] =\n{\n" % name)
327      for method in interface.methods:
328          f.write(
329              '\t{{ "{}", Py{}::{}, 1 }}, // @pymeth {}|Description of {}\n'.format(
330                  method.name, interface.name, method.name, method.name, method.name
331              )
332          )
333  
334      interfacebase = interface.base
335      f.write(
336          """\ 
337      {{ NULL }} 
338  }}; 
339   
340  PyComTypeObject Py{name}::type("Py{name}", 
341          &Py{interfacebase}::type, 
342          sizeof(Py{name}), 
343          Py{name}_methods, 
344          GET_PYCOM_CTOR(Py{name})); 
345  """.format(**locals())
346      )

349  def _write_gw_h(f, interface):
350      if interface.name[0] == "I":
351          gname = "PyG" + interface.name[1:]
352      else:
353          gname = "PyG" + interface.name
354      name = interface.name
355      if interface.base == "IUnknown" or interface.base == "IDispatch":
356          base_name = "PyGatewayBase"
357      else:
358          if interface.base[0] == "I":
359              base_name = "PyG" + interface.base[1:]
360          else:
361              base_name = "PyG" + interface.base
362      f.write(
363          f"""\ 
364  // ---------------------------------------------------
365  //
366  // Gateway Declaration
367  
368  class {gname} : public {base_name}, public {name}
369  {{
370  protected:
371      {gname}(PyObject *instance) : {base_name}(instance) {{ ; }}
372      PYGATEWAY_MAKE_SUPPORT2({gname}, {name}, IID_{name}, {base_name})
373  
374  """
375      )
376      if interface.base != "IUnknown":
377          f.write(
378              "\t// {}\n\t// *** Manually add {} method decls here\n\n".format(
379                  interface.base, interface.base
380              )
381          )
382      else:
383          f.write("\n\n")
384  
385      f.write("\t// %s\n" % name)
386  
387      for method in interface.methods:
388          f.write("\tSTDMETHOD(%s)(\n" % method.name)
389          if method.args:
390              for arg in method.args[:-1]:
391                  f.write("\t\t%s,\n" % (arg.GetRawDeclaration()))
392              arg = method.args[-1]
393              f.write("\t\t%s);\n\n" % (arg.GetRawDeclaration()))
394          else:
395              f.write("\t\tvoid);\n\n")
396  
397      f.write("};\n")
398      f.close()

401  def _write_gw_cpp(f, interface):
402      if interface.name[0] == "I":
403          gname = "PyG" + interface.name[1:]
404      else:
405          gname = "PyG" + interface.name
406      name = interface.name
407      if interface.base == "IUnknown" or interface.base == "IDispatch":
408          base_name = "PyGatewayBase"
409      else:
410          if interface.base[0] == "I":
411              base_name = "PyG" + interface.base[1:]
412          else:
413              base_name = "PyG" + interface.base
414      f.write(
415          """\ 
416  // --------------------------------------------------- 
417  // 
418  // Gateway Implementation 
419  """
420      )
421  
422      for method in interface.methods:
423          f.write(
424              f"""\ 
425  STDMETHODIMP {gname}::{method.name}(
426  """
427          )
428  
429          if method.args:
430              for arg in method.args[:-1]:
431                  inoutstr = "][".join(arg.inout)
432                  f.write(f"\t\t/* [{inoutstr}] */ {arg.GetRawDeclaration()},\n")
433  
434              arg = method.args[-1]
435              inoutstr = "][".join(arg.inout)
436              f.write(f"\t\t/* [{inoutstr}] */ {arg.GetRawDeclaration()})\n")
437          else:
438              f.write("\t\tvoid)\n")
439  
440          f.write("{\n\tPY_GATEWAY_METHOD;\n")
441          cout = 0
442          codePre = codePost = codeVars = ""
443          argStr = ""
444          needConversion = 0
445          formatChars = ""
446          if method.args:
447              for arg in method.args:
448                  if arg.HasAttribute("out"):
449                      cout += 1
450                      if arg.indirectionLevel == 2:
451                          f.write("\tif (%s==NULL) return E_POINTER;\n" % arg.name)
452                  if arg.HasAttribute("in"):
453                      try:
454                          argCvt = makegwparse.make_arg_converter(arg)
455                          argCvt.SetGatewayMode()
456                          formatchar = argCvt.GetFormatChar()
457                          needConversion = needConversion or argCvt.NeedUSES_CONVERSION()
458  
459                          if formatchar:
460                              formatChars += formatchar
461                              codeVars += argCvt.DeclareParseArgTupleInputConverter()
462                              argStr += ", " + argCvt.GetBuildValueArg()
463                          codePre += argCvt.GetBuildForGatewayPreCode()
464                          codePost += argCvt.GetBuildForGatewayPostCode()
465                      except makegwparse.error_not_supported as why:
466                          f.write(
467                              '// *** The input argument {} of type "{}" was not processed ***\n//   - Please ensure this conversion function exists, and is appropriate\n//   - {}\n'.format(
468                                  arg.name, arg.raw_type, why
469                              )
470                          )
471                          f.write(
472                              "\tPyObject *ob{} = PyObject_From{}({});\n".format(
473                                  arg.name, arg.type, arg.name
474                              )
475                          )
476                          f.write(
477                              '\tif (ob{}==NULL) return MAKE_PYCOM_GATEWAY_FAILURE_CODE("{}");\n'.format(
478                                  arg.name, method.name
479                              )
480                          )
481                          codePost += "\tPy_DECREF(ob%s);\n" % arg.name
482                          formatChars += "O"
483                          argStr += ", ob%s" % arg.name
484  
485          if needConversion:
486              f.write("\tUSES_CONVERSION;\n")
487          f.write(codeVars)
488          f.write(codePre)
489          if cout:
490              f.write("\tPyObject *result;\n")
491              resStr = "&result"
492          else:
493              resStr = "NULL"
494  
495          if formatChars:
496              fullArgStr = f'{resStr}, "{formatChars}"{argStr}'
497          else:
498              fullArgStr = resStr
499  
500          f.write(f'\tHRESULT hr=InvokeViaPolicy("{method.name}", {fullArgStr});\n')
501          f.write(codePost)
502          if cout:
503              f.write("\tif (FAILED(hr)) return hr;\n")
504              f.write(
505                  "\t// Process the Python results, and convert back to the real params\n"
506              )
507              # process the output arguments.
508              formatChars = codePobjects = codePost = argsParseTuple = ""
509              needConversion = 0
510              for arg in method.args:
511                  if not arg.HasAttribute("out"):
512                      continue
513                  try:
514                      argCvt = makegwparse.make_arg_converter(arg)
515                      argCvt.SetGatewayMode()
516                      val = argCvt.GetFormatChar()
517                      if val:
518                          formatChars += val
519                          argsParseTuple += ", " + argCvt.GetParseTupleArg()
520                          codePobjects += argCvt.DeclareParseArgTupleInputConverter()
521                          codePost += argCvt.GetParsePostCode()
522                          needConversion = needConversion or argCvt.NeedUSES_CONVERSION()
523                  except makegwparse.error_not_supported as why:
524                      f.write(
525                          '// *** The output argument {} of type "{}" was not processed ***\n//     {}\n'.format(
526                              arg.name, arg.raw_type, why
527                          )
528                      )
529  
530              if formatChars:  # If I have any to actually process.
531                  if len(formatChars) == 1:
532                      parseFn = "PyArg_Parse"
533                  else:
534                      parseFn = "PyArg_ParseTuple"
535                  if codePobjects:
536                      f.write(codePobjects)
537                  f.write(
538                      '\tif (!{}(result, "{}" {}))\n\t\treturn MAKE_PYCOM_GATEWAY_FAILURE_CODE("{}");\n'.format(
539                          parseFn, formatChars, argsParseTuple, method.name
540                      )
541                  )
542              if codePost:
543                  f.write("\tBOOL bPythonIsHappy = TRUE;\n")
544                  f.write(codePost)
545                  f.write(
546                      '\tif (!bPythonIsHappy) hr = MAKE_PYCOM_GATEWAY_FAILURE_CODE("%s");\n'
547                      % method.name
548                  )
549              f.write("\tPy_DECREF(result);\n")
550          f.write("\treturn hr;\n}\n\n")

562  def parse_arguments():
563      parser = argparse.ArgumentParser(description="COM interface wrapper generator")
564      parser.add_argument(
565          "--header_file", "-f", required=True, help="header file (system) to parse"
566      )
567      parser.add_argument(
568          "--interface_name", "-n", required=True, help="interface name to search for"
569      )
570      parser.add_argument(
571          "--no_create_interface",
572          "-i",
573          action="store_false",
574          dest="create_interface",
575          help="do not generate interface code",
576      )
577      parser.add_argument(
578          "--no_create_gateway",
579          "-g",
580          action="store_false",
581          dest="create_gateway",
582          help="do not generate gateway code",
583      )
584      parser.add_argument(
585          "--output_directory", "-o", help="directory where to generate files"
586      )
587  
588      args, unk = parser.parse_known_args()
589      if unk:
590          print(f"Warning: Ignoring unknown arguments: {unk}")
591  
592      if not args.header_file or not os.path.isfile(args.header_file):
593          parser.exit(status=-1, message="Invalid header file\n")
594      if not args.interface_name:
595          parser.exit(status=-1, message="Invalid interface name\n")
596  
597      return (
598          args.header_file,
599          args.interface_name,
600          args.create_interface,
601          args.create_gateway,
602          args.output_directory or None,
603      )
604  
605  
606  if __name__ == "__main__":
607      header_file, interface_name, create_interface, create_gateway, directory = (
608          parse_arguments()
609      )
610      if directory:
611          os.makedirs(directory, exist_ok=True)
612      make_framework_support(
613          header_file_name=header_file,
614          interface_name=interface_name,
615          bMakeInterface=create_interface,
616          bMakeGateway=create_gateway,
617          output_directory=directory,
618      )
