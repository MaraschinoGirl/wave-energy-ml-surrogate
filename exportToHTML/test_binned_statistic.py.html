553      def test_dd_all(self, dtype, statistic):
554          def ref_statistic(x):
555              return len(x) if statistic == 'count' else statistic(x)
556  
557          rng = np.random.default_rng(3704743126639371)
558          n = 10
559          x = rng.random(size=n)
560          i = x >= 0.5
