19       Wrap an integer to the interval [-128, 127]. 
20       """
21       return (n + 128) % 256 - 128

28   def test_threads():
29       # Smoke test for parallel threaded execution; doesn't actually
30       # check that code runs in parallel, but just that it produces
31       # expected results.
32       nthreads = 10
33       niter = 100
34   
35       n = 20
36       a = csr_matrix(np.ones([n, n]))
37       bres = []
38   
39       class Worker(threading.Thread):
40           def run(self):
41               b = a.copy()
42               for j in range(niter):
43                   _sparsetools.csr_plus_csr(n, n,
44                                             a.indptr, a.indices, a.data,
45                                             a.indptr, a.indices, a.data,
46                                             b.indptr, b.indices, b.data)
47               bres.append(b)
48   
49       threads = [Worker() for _ in range(nthreads)]
50       for thread in threads:
51           thread.start()
52       for thread in threads:
53           thread.join()
54   
55       for b in bres:
56           assert_(np.all(b.toarray() == 2))

70   @pytest.mark.slow
71   @pytest.mark.xfail_on_32bit("Can't create large array for test")
72   def test_nnz_overflow():
73       # Regression test for gh-7230 / gh-7871, checking that coo_toarray
74       # with nnz > int32max doesn't overflow.
75       nnz = np.iinfo(np.int32).max + 1
76       # Ensure ~20 GB of RAM is free to run this test.
77       check_free_memory((4 + 4 + 1) * nnz / 1e6 + 0.5)
78   
79       # Use nnz duplicate entries to keep the dense version small.
80       row = np.zeros(nnz, dtype=np.int32)
81       col = np.zeros(nnz, dtype=np.int32)
82       data = np.zeros(nnz, dtype=np.int8)
83       data[-1] = 4
84       s = coo_matrix((data, (row, col)), shape=(1, 1), copy=False)
85       # Sums nnz duplicates to produce a 1x1 array containing 4.
86       d = s.toarray()
87   
88       assert_allclose(d, [[4]])

95   class TestInt32Overflow:
96       """ 
97       Some of the sparsetools routines use dense 2D matrices whose 
98       total size is not bounded by the nnz of the sparse matrix. These 
99       routines used to suffer from int32 wraparounds; here, we try to 
100      check that the wraparounds don't occur any more.

103      n = 50000

105      def setup_method(self):
106          assert self.n**2 > np.iinfo(np.int32).max

108          # check there's enough memory even if everything is run at the
109          # same time
110          try:
111              parallel_count = int(os.environ.get('PYTEST_XDIST_WORKER_COUNT', '1'))
112          except ValueError:
113              parallel_count = np.inf
114  
115          check_free_memory(3000 * parallel_count)

291  def test_upcast():
292      a0 = csr_matrix([[np.pi, np.pi*1j], [3, 4]], dtype=complex)
293      b0 = np.array([256+1j, 2**32], dtype=complex)
294  
295      for a_dtype in supported_dtypes:
296          for b_dtype in supported_dtypes:
297              msg = f"({a_dtype!r}, {b_dtype!r})"
298  
299              if np.issubdtype(a_dtype, np.complexfloating):
300                  a = a0.copy().astype(a_dtype)
301              else:
302                  a = a0.real.copy().astype(a_dtype)
303  
304              if np.issubdtype(b_dtype, np.complexfloating):
305                  b = b0.copy().astype(b_dtype)
306              else:
307                  with np.errstate(invalid="ignore"):
308                      # Casting a large value (2**32) to int8 causes a warning in
309                      # numpy >1.23
310                      b = b0.real.copy().astype(b_dtype)
311  
312              if not (a_dtype == np.bool_ and b_dtype == np.bool_):
313                  c = np.zeros((2,), dtype=np.bool_)
314                  assert_raises(ValueError, _sparsetools.csr_matvec,
315                                2, 2, a.indptr, a.indices, a.data, b, c)
316  
317              if ((np.issubdtype(a_dtype, np.complexfloating) and
318                   not np.issubdtype(b_dtype, np.complexfloating)) or
319                  (not np.issubdtype(a_dtype, np.complexfloating) and
320                   np.issubdtype(b_dtype, np.complexfloating))):
321                  c = np.zeros((2,), dtype=np.float64)
322                  assert_raises(ValueError, _sparsetools.csr_matvec,
323                                2, 2, a.indptr, a.indices, a.data, b, c)
324  
325              c = np.zeros((2,), dtype=np.result_type(a_dtype, b_dtype))
326              _sparsetools.csr_matvec(2, 2, a.indptr, a.indices, a.data, b, c)
327              assert_allclose(c, np.dot(a.toarray(), b), err_msg=msg)

330  def test_endianness():
331      d = np.ones((3,4))
332      offsets = [-1,0,1]
333  
334      a = dia_matrix((d.astype('<f8'), offsets), (4, 4))
335      b = dia_matrix((d.astype('>f8'), offsets), (4, 4))
336      v = np.arange(4)
