61   # create inverse mapping for setuptools
62   tag_to_packages: dict = defaultdict(list)
63   for package, (min_version, extras) in dependent_packages.items():
64       for extra in extras.split(", "):
65           tag_to_packages[extra].append("{}>={}".format(package, min_version))
66   
67   
68   # Used by CI to get the min dependencies
69   if __name__ == "__main__":
70       parser = argparse.ArgumentParser(description="Get min dependencies for a package")
71   
72       parser.add_argument("package", choices=dependent_packages)
73       args = parser.parse_args()
74       min_version = dependent_packages[args.package][0]
75       print(min_version)
