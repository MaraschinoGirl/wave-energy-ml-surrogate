5    /*                                                     j0.c 
6     * 
7     *     Bessel function of order zero 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double x, y, j0(); 
14    * 
15    * y = j0( x ); 
16    * 
17    * 
18    * 
19    * DESCRIPTION: 
20    * 
21    * Returns Bessel function of order zero of the argument. 
22    * 
23    * The domain is divided into the intervals [0, 5] and 
24    * (5, infinity). In the first interval the following rational 
25    * approximation is used: 
26    * 
27    * 
28    *        2         2 
29    * (w - r  ) (w - r  ) P (w) / Q (w) 
30    *       1         2    3       8 
31    * 
32    *            2 
33    * where w = x  and the two r's are zeros of the function. 
34    * 
35    * In the second interval, the Hankel asymptotic expansion 
36    * is employed with two rational functions of degree 6/6 
37    * and 7/7. 
38    * 
39    * 
40    * 
41    * ACCURACY: 
42    * 
43    *                      Absolute error: 
44    * arithmetic   domain     # trials      peak         rms 
45    *    IEEE      0, 30       60000       4.2e-16     1.1e-16 
46    * 
47    */
48   /*                          y0.c 
49    * 
50    *  Bessel function of the second kind, order zero 
51    * 
52    * 
53    * 
54    * SYNOPSIS: 
55    * 
56    * double x, y, y0(); 
57    * 
58    * y = y0( x ); 
59    * 
60    * 
61    * 
62    * DESCRIPTION: 
63    * 
64    * Returns Bessel function of the second kind, of order 
65    * zero, of the argument. 
66    * 
67    * The domain is divided into the intervals [0, 5] and 
68    * (5, infinity). In the first interval a rational approximation 
69    * R(x) is employed to compute 
70    *   y0(x)  = R(x)  +   2 * log(x) * j0(x) / M_PI. 
71    * Thus a call to j0() is required. 
72    * 
73    * In the second interval, the Hankel asymptotic expansion 
74    * is employed with two rational functions of degree 6/6 
75    * and 7/7. 
76    * 
77    * 
78    * 
79    * ACCURACY: 
80    * 
81    *  Absolute error, when y0(x) < 1; else relative error: 
82    * 
83    * arithmetic   domain     # trials      peak         rms 
84    *    IEEE      0, 30       30000       1.3e-15     1.6e-16 
85    * 
86    */

162      XSF_HOST_DEVICE inline double j0(double x) {
163          double w, z, p, q, xn;
164  
165          if (x < 0) {
166              x = -x;
167          }
168  
169          if (x <= 5.0) {
170              z = x * x;
171              if (x < 1.0e-5) {
172                  return (1.0 - z / 4.0);
173              }
174  
175              p = (z - detail::j0_DR1) * (z - detail::j0_DR2);
176              p = p * polevl(z, detail::j0_RP, 3) / p1evl(z, detail::j0_RQ, 8);
177              return (p);
178          }
179  
180          w = 5.0 / x;
181          q = 25.0 / (x * x);
182          p = polevl(q, detail::j0_PP, 6) / polevl(q, detail::j0_PQ, 6);
183          q = polevl(q, detail::j0_QP, 7) / p1evl(q, detail::j0_QQ, 7);
184          xn = x - M_PI_4;
185          p = p * std::cos(xn) - w * q * std::sin(xn);
186          return (p * detail::SQRT2OPI / std::sqrt(x));
187      }

189      /*                                                     y0() 2  */
190      /* Bessel function of second kind, order zero  */
191  
192      /* Rational approximation coefficients YP[], YQ[] are used here. 
193       * The function computed is  y0(x)  -  2 * log(x) * j0(x) / M_PI, 
194       * whose value at x = 0 is  2 * ( log(0.5) + EUL ) / M_PI 
195       * = 0.073804295108687225. 
196       */
197  
198      XSF_HOST_DEVICE inline double y0(double x) {
199          double w, z, p, q, xn;
200  
201          if (x <= 5.0) {
202              if (x == 0.0) {
203                  set_error("y0", SF_ERROR_SINGULAR, NULL);
204                  return -std::numeric_limits<double>::infinity();
205              } else if (x < 0.0) {
206                  set_error("y0", SF_ERROR_DOMAIN, NULL);
207                  return std::numeric_limits<double>::quiet_NaN();
208              }
209              z = x * x;
210              w = polevl(z, detail::j0_YP, 7) / p1evl(z, detail::j0_YQ, 7);
211              w += M_2_PI * std::log(x) * j0(x);
212              return (w);
213          }
214  
215          w = 5.0 / x;
216          z = 25.0 / (x * x);
217          p = polevl(z, detail::j0_PP, 6) / polevl(z, detail::j0_PQ, 6);
218          q = polevl(z, detail::j0_QP, 7) / p1evl(z, detail::j0_QQ, 7);
219          xn = x - M_PI_4;
220          p = p * std::sin(xn) + w * q * std::cos(xn);
221          return (p * detail::SQRT2OPI / std::sqrt(x));
222      }
