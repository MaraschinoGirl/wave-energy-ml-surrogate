21   def min_pos(const floating[:] X):
22       """Find the minimum value of an array over positive values. 
23    
24       Returns the maximum representable value of the input dtype if none of the 
25       values are positive. 
26    
27       Parameters 
28       ---------- 
29       X : ndarray of shape (n,) 
30           Input array. 
31    
32       Returns 
33       ------- 
34       min_val : float 
35           The smallest positive value in the array, or the maximum representable value 
36            of the input dtype if no positive values are found. 
37    
38       Examples 
39       -------- 
40       >>> import numpy as np 
41       >>> from sklearn.utils.arrayfuncs import min_pos 
42       >>> X = np.array([0, -1, 2, 3, -4, 5]) 
43       >>> min_pos(X) 
44       2.0 
45       """
46       cdef Py_ssize_t i
47       cdef floating min_val = FLT_MAX if floating is float else DBL_MAX
48       for i in range(X.size):
49           if 0. < X[i] < min_val:
50               min_val = X[i]
51       return min_val

54   def _all_with_any_reduction_axis_1(real_numeric[:, :] array, real_numeric value):
55       """Check whether any row contains all values equal to `value`. 
56    
57       It is equivalent to `np.any(np.all(X == value, axis=1))`, but it avoids to 
58       materialize the temporary boolean matrices in memory. 
59    
60       Parameters 
61       ---------- 
62       array: array-like 
63           The array to be checked. 
64       value: short, int, long, float, or double 
65           The value to use for the comparison. 
66    
67       Returns 
68       ------- 
69       any_all_equal: bool 
70           Whether or not any rows contains all values equal to `value`. 
71       """
72       cdef Py_ssize_t i, j
73   
74       for i in range(array.shape[0]):
75           for j in range(array.shape[1]):
76               if array[i, j] != value:
77                   break
78           else:  # no break
79               return True
80       return False

89   def cholesky_delete(floating[:, :] L, int go_out):
90       cdef:
91           int n = L.shape[0]
92           int m = L.strides[0]
93           floating c, s
94           floating *L1
95           int i
96   
97       if floating is float:
98           m /= sizeof(float)
99       else:
100          m /= sizeof(double)
101  
102      # delete row go_out
103      L1 = &L[0, 0] + (go_out * m)
104      for i in range(go_out, n-1):
105          _copy(i + 2, L1 + m, 1, L1, 1)
106          L1 += m
107  
108      L1 = &L[0, 0] + (go_out * m)
109      for i in range(go_out, n-1):
110          _rotg(L1 + i, L1 + i + 1, &c, &s)
111          if L1[i] < 0:
112              # Diagonals cannot be negative
113              L1[i] = fabs(L1[i])
114              c = -c
115              s = -s
116  
117          L1[i + 1] = 0.  # just for cleanup
118          L1 += m
119  
120          _rot(n - i - 2, L1 + i, m, L1 + i + 1, m, c, s)

123  def sum_parallel(const floating [:] array, int n_threads):
124      """Parallel sum, always using float64 internally."""
125      cdef:
126          float64_t out = 0.
127          int i = 0
128  
129      for i in prange(
130          array.shape[0], schedule='static', nogil=True, num_threads=n_threads
131      ):
132          out += array[i]
133  
134      return out
