4     
5    This module contains low-level functions from the LAPACK library. 
6     
7    .. versionadded:: 0.12.0 
8     
9    .. note:: 
10    
11       The common ``overwrite_<>`` option in many routines, allows the 
12       input arrays to be overwritten to avoid extra memory allocation. 
13       However this requires the array to satisfy two conditions 
14       which are memory order and the data type to match exactly the 
15       order and the type expected by the routine. 
16    
17       As an example, if you pass a double precision float array to any 
18       ``S....`` routine which expects single precision arguments, f2py 
19       will create an intermediate array to match the argument types and 
20       overwriting will be performed on that intermediate array. 
21    
22       Similarly, if a C-contiguous array is passed, f2py will pass a 
23       FORTRAN-contiguous array internally. Please make sure that these 
24       details are satisfied. More information can be found in the f2py 
25       documentation. 
26    
27   .. warning:: 
28    
29      These functions do little to no error checking. 
30      It is possible to cause crashes by mis-using them, 
31      so prefer using the higher-level routines in `scipy.linalg`.

906  @_memoize_get_funcs
907  def get_lapack_funcs(names, arrays=(), dtype=None, ilp64=False):
908      """Return available LAPACK function objects from names. 
909   
910      Arrays are used to determine the optimal prefix of LAPACK routines. 
911   
912      Parameters 
913      ---------- 
914      names : str or sequence of str 
915          Name(s) of LAPACK functions without type prefix. 
916   
917      arrays : sequence of ndarrays, optional 
918          Arrays can be given to determine optimal prefix of LAPACK 
919          routines. If not given, double-precision routines will be 
920          used, otherwise the most generic type in arrays will be used. 
921   
922      dtype : str or dtype, optional 
923          Data-type specifier. Not used if `arrays` is non-empty. 
924   
925      ilp64 : {True, False, 'preferred'}, optional 
926          Whether to return ILP64 routine variant. 
927          Choosing 'preferred' returns ILP64 routine if available, and 
928          otherwise the 32-bit routine. Default: False 
929   
930      Returns 
931      ------- 
932      funcs : list 
933          List containing the found function(s). 
934   
935      Notes 
936      ----- 
937      This routine automatically chooses between Fortran/C 
938      interfaces. Fortran code is used whenever possible for arrays with 
939      column major order. In all other cases, C code is preferred. 
940   
941      In LAPACK, the naming convention is that all functions start with a 
942      type prefix, which depends on the type of the principal 
943      matrix. These can be one of {'s', 'd', 'c', 'z'} for the NumPy 
944      types {float32, float64, complex64, complex128} respectively, and 
945      are stored in attribute ``typecode`` of the returned functions. 
946   
947      Examples 
948      -------- 
949      Suppose we would like to use '?lange' routine which computes the selected 
950      norm of an array. We pass our array in order to get the correct 'lange' 
951      flavor. 
952   
953      >>> import numpy as np 
954      >>> import scipy.linalg as LA 
955      >>> rng = np.random.default_rng() 
956   
957      >>> a = rng.random((3,2)) 
958      >>> x_lange = LA.get_lapack_funcs('lange', (a,)) 
959      >>> x_lange.typecode 
960      'd' 
961      >>> x_lange = LA.get_lapack_funcs('lange',(a*1j,)) 
962      >>> x_lange.typecode 
963      'z' 
964   
965      Several LAPACK routines work best when its internal WORK array has 
966      the optimal size (big enough for fast computation and small enough to 
967      avoid waste of memory). This size is determined also by a dedicated query 
968      to the function which is often wrapped as a standalone function and 
969      commonly denoted as ``###_lwork``. Below is an example for ``?sysv`` 
970   
971      >>> a = rng.random((1000, 1000)) 
972      >>> b = rng.random((1000, 1)) * 1j 
973      >>> # We pick up zsysv and zsysv_lwork due to b array 
974      ... xsysv, xlwork = LA.get_lapack_funcs(('sysv', 'sysv_lwork'), (a, b)) 
975      >>> opt_lwork, _ = xlwork(a.shape[0])  # returns a complex for 'z' prefix 
976      >>> udut, ipiv, x, info = xsysv(a, b, lwork=int(opt_lwork.real)) 
977   
978      """
979      if isinstance(ilp64, str):
980          if ilp64 == 'preferred':
981              ilp64 = HAS_ILP64
982          else:
983              raise ValueError("Invalid value for 'ilp64'")
984  
985      if not ilp64:
986          return _get_funcs(names, arrays, dtype,
987                            "LAPACK", _flapack, _clapack,
988                            "flapack", "clapack", _lapack_alias,
989                            ilp64=False)
990      else:
991          if not HAS_ILP64:
992              raise RuntimeError("LAPACK ILP64 routine requested, but Scipy "
993                                 "compiled only with 32-bit BLAS")
994          return _get_funcs(names, arrays, dtype,
995                            "LAPACK", _flapack_64, None,
996                            "flapack_64", None, _lapack_alias,
997                            ilp64=True)

1004 def _compute_lwork(routine, *args, **kwargs):
1005     """ 
1006     Round floating-point lwork returned by lapack to integer. 
1007  
1008     Several LAPACK routines compute optimal values for LWORK, which 
1009     they return in a floating-point variable. However, for large 
1010     values of LWORK, single-precision floating point is not sufficient 
1011     to hold the exact value --- some LAPACK versions (<= 3.5.0 at 
1012     least) truncate the returned integer to single precision and in 
1013     some cases this can be smaller than the required value. 
1014  
1015     Examples 
1016     -------- 
1017     >>> from scipy.linalg import lapack 
1018     >>> n = 5000 
1019     >>> s_r, s_lw = lapack.get_lapack_funcs(('sysvx', 'sysvx_lwork')) 
1020     >>> lwork = lapack._compute_lwork(s_lw, n) 
1021     >>> lwork 
1022     32000 
1023  
1024     """
1025     dtype = getattr(routine, 'dtype', None)
1026     int_dtype = getattr(routine, 'int_dtype', None)
1027     ret = routine(*args, **kwargs)
1028     if ret[-1] != 0:
1029         raise ValueError("Internal work array size computation failed: "
1030                          "%d" % (ret[-1],))
1031 
1032     if len(ret) == 2:
1033         return _check_work_float(ret[0].real, dtype, int_dtype)
1034     else:
1035         return tuple(_check_work_float(x.real, dtype, int_dtype)
1036                      for x in ret[:-1])

1039 def _check_work_float(value, dtype, int_dtype):
1040     """ 
1041     Convert LAPACK-returned work array size float to integer, 
1042     carefully for single-precision types. 
1043     """
1044 
1045     if dtype == np.float32 or dtype == np.complex64:
1046         # Single-precision routine -- take next fp value to work
1047         # around possible truncation in LAPACK code
1048         value = np.nextafter(value, np.inf, dtype=np.float32)
1049 
1050     value = int(value)
1051     if int_dtype.itemsize == 4:
1052         if value < 0 or value > _int32_max:
1053             raise ValueError("Too large work array required -- computation "
1054                              "cannot be performed with standard 32-bit"
1055                              " LAPACK.")
1056     elif int_dtype.itemsize == 8:
1057         if value < 0 or value > _int64_max:
1058             raise ValueError("Too large work array required -- computation"
1059                              " cannot be performed with standard 64-bit"
1060                              " LAPACK.")
1061     return value
