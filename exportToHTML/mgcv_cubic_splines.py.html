25   def _get_natural_f(knots):
26       """Returns mapping of natural cubic spline values to 2nd derivatives. 
27    
28       .. note:: See 'Generalized Additive Models', Simon N. Wood, 2006, pp 145-146 
29    
30       :param knots: The 1-d array knots used for cubic spline parametrization, 
31        must be sorted in ascending order. 
32       :return: A 2-d array mapping natural cubic spline values at 
33        knots to second derivatives. 
34    
35       :raise ImportError: if scipy is not found, required for 
36        ``linalg.solve_banded()`` 
37       """
38       try:
39           from scipy import linalg
40       except ImportError:  # pragma: no cover
41           raise ImportError("Cubic spline functionality requires scipy.")
42   
43       h = knots[1:] - knots[:-1]
44       diag = (h[:-1] + h[1:]) / 3.0
45       ul_diag = h[1:-1] / 6.0
46       banded_b = np.array([np.r_[0.0, ul_diag], diag, np.r_[ul_diag, 0.0]])
47       d = np.zeros((knots.size - 2, knots.size))
48       for i in range(knots.size - 2):
49           d[i, i] = 1.0 / h[i]
50           d[i, i + 2] = 1.0 / h[i + 1]
51           d[i, i + 1] = -d[i, i] - d[i, i + 2]
52   
53       fm = linalg.solve_banded((1, 1), banded_b, d)
54   
55       return np.vstack([np.zeros(knots.size), fm, np.zeros(knots.size)])

61   def _map_cyclic(x, lbound, ubound):
62       """Maps values into the interval [lbound, ubound] in a cyclic fashion. 
63    
64       :param x: The 1-d array values to be mapped. 
65       :param lbound: The lower bound of the interval. 
66       :param ubound: The upper bound of the interval. 
67       :return: A new 1-d array containing mapped x values. 
68    
69       :raise ValueError: if lbound >= ubound. 
70       """
71       if lbound >= ubound:
72           raise ValueError(
73               "Invalid argument: lbound (%r) should be "
74               "less than ubound (%r)." % (lbound, ubound)
75           )
76   
77       x = np.copy(x)
78       x[x > ubound] = lbound + (x[x > ubound] - ubound) % (ubound - lbound)
79       x[x < lbound] = ubound - (lbound - x[x < lbound]) % (ubound - lbound)
80   
81       return x

101  def _get_cyclic_f(knots):
102      """Returns mapping of cyclic cubic spline values to 2nd derivatives. 
103   
104      .. note:: See 'Generalized Additive Models', Simon N. Wood, 2006, pp 146-147 
105   
106      :param knots: The 1-d array knots used for cubic spline parametrization, 
107       must be sorted in ascending order. 
108      :return: A 2-d array mapping cyclic cubic spline values at 
109       knots to second derivatives. 
110      """
111      h = knots[1:] - knots[:-1]
112      n = knots.size - 1
113      b = np.zeros((n, n))
114      d = np.zeros((n, n))
115  
116      b[0, 0] = (h[n - 1] + h[0]) / 3.0
117      b[0, n - 1] = h[n - 1] / 6.0
118      b[n - 1, 0] = h[n - 1] / 6.0
119  
120      d[0, 0] = -1.0 / h[0] - 1.0 / h[n - 1]
121      d[0, n - 1] = 1.0 / h[n - 1]
122      d[n - 1, 0] = 1.0 / h[n - 1]
123  
124      for i in range(1, n):
125          b[i, i] = (h[i - 1] + h[i]) / 3.0
126          b[i, i - 1] = h[i - 1] / 6.0
127          b[i - 1, i] = h[i - 1] / 6.0
128  
129          d[i, i] = -1.0 / h[i - 1] - 1.0 / h[i]
130          d[i, i - 1] = 1.0 / h[i - 1]
131          d[i - 1, i] = 1.0 / h[i - 1]
132  
133      return np.linalg.solve(b, d)

139  def _row_tensor_product(dms):
140      """Computes row-wise tensor product of given arguments. 
141   
142      .. note:: Custom algorithm to precisely match what is done in 'mgcv', 
143      in particular look out for order of result columns! 
144      For reference implementation see 'mgcv' source code, 
145      file 'mat.c', mgcv_tensor_mm(), l.62 
146   
147      :param dms: A sequence of 2-d arrays (marginal design matrices). 
148      :return: The 2-d array row-wise tensor product of given arguments. 
149   
150      :raise ValueError: if argument sequence is empty, does not contain only 
151       2-d arrays or if the arrays number of rows does not match. 
152      """
153      if len(dms) == 0:
154          raise ValueError("Tensor product arrays sequence should not be empty.")
155      for dm in dms:
156          if dm.ndim != 2:
157              raise ValueError("Tensor product arguments should be 2-d arrays.")
158  
159      tp_nrows = dms[0].shape[0]
160      tp_ncols = 1
161      for dm in dms:
162          if dm.shape[0] != tp_nrows:
163              raise ValueError(
164                  "Tensor product arguments should have " "same number of rows."
165              )
166          tp_ncols *= dm.shape[1]
167      tp = np.zeros((tp_nrows, tp_ncols))
168      tp[:, -dms[-1].shape[1] :] = dms[-1]
169      filled_tp_ncols = dms[-1].shape[1]
170      for dm in dms[-2::-1]:
171          p = -filled_tp_ncols * dm.shape[1]
172          for j in range(dm.shape[1]):
173              xj = dm[:, j]
174              for t in range(-filled_tp_ncols, 0):
175                  tp[:, p] = tp[:, t] * xj
176                  p += 1
177          filled_tp_ncols *= dm.shape[1]
178  
179      return tp

226  def _find_knots_lower_bounds(x, knots):
227      """Finds knots lower bounds for given values. 
228   
229      Returns an array of indices ``I`` such that 
230      ``0 <= I[i] <= knots.size - 2`` for all ``i`` 
231      and 
232      ``knots[I[i]] < x[i] <= knots[I[i] + 1]`` if 
233      ``np.min(knots) < x[i] <= np.max(knots)``, 
234      ``I[i] = 0`` if ``x[i] <= np.min(knots)`` 
235      ``I[i] = knots.size - 2`` if ``np.max(knots) < x[i]`` 
236   
237      :param x: The 1-d array values whose knots lower bounds are to be found. 
238      :param knots: The 1-d array knots used for cubic spline parametrization, 
239       must be sorted in ascending order. 
240      :return: An array of knots lower bounds indices. 
241      """
242      lb = np.searchsorted(knots, x) - 1
243  
244      # I[i] = 0 for x[i] <= np.min(knots)
245      lb[lb == -1] = 0
246  
247      # I[i] = knots.size - 2 for x[i] > np.max(knots)
248      lb[lb == knots.size - 1] = knots.size - 2
249  
250      return lb

253  def _compute_base_functions(x, knots):
254      """Computes base functions used for building cubic splines basis. 
255   
256      .. note:: See 'Generalized Additive Models', Simon N. Wood, 2006, p. 146 
257        and for the special treatment of ``x`` values outside ``knots`` range 
258        see 'mgcv' source code, file 'mgcv.c', function 'crspl()', l.249 
259   
260      :param x: The 1-d array values for which base functions should be computed. 
261      :param knots: The 1-d array knots used for cubic spline parametrization, 
262       must be sorted in ascending order. 
263      :return: 4 arrays corresponding to the 4 base functions ajm, ajp, cjm, cjp 
264       + the 1-d array of knots lower bounds indices corresponding to 
265       the given ``x`` values. 
266      """
267      j = _find_knots_lower_bounds(x, knots)
268  
269      h = knots[1:] - knots[:-1]
270      hj = h[j]
271      xj1_x = knots[j + 1] - x
272      x_xj = x - knots[j]
273  
274      ajm = xj1_x / hj
275      ajp = x_xj / hj
276  
277      cjm_3 = xj1_x * xj1_x * xj1_x / (6.0 * hj)
278      cjm_3[x > np.max(knots)] = 0.0
279      cjm_1 = hj * xj1_x / 6.0
280      cjm = cjm_3 - cjm_1
281  
282      cjp_3 = x_xj * x_xj * x_xj / (6.0 * hj)
283      cjp_3[x < np.min(knots)] = 0.0
284      cjp_1 = hj * x_xj / 6.0
285      cjp = cjp_3 - cjp_1
286  
287      return ajm, ajp, cjm, cjp, j

290  def _absorb_constraints(design_matrix, constraints):
291      """Absorb model parameters constraints into the design matrix. 
292   
293      :param design_matrix: The (2-d array) initial design matrix. 
294      :param constraints: The 2-d array defining initial model parameters 
295       (``betas``) constraints (``np.dot(constraints, betas) = 0``). 
296      :return: The new design matrix with absorbed parameters constraints. 
297   
298      :raise ImportError: if scipy is not found, used for ``scipy.linalg.qr()`` 
299        which is cleaner than numpy's version requiring a call like 
300        ``qr(..., mode='complete')`` to get a full QR decomposition. 
301      """
302      try:
303          from scipy import linalg
304      except ImportError:  # pragma: no cover
305          raise ImportError("Cubic spline functionality requires scipy.")
306  
307      m = constraints.shape[0]
308      q, r = linalg.qr(np.transpose(constraints))
309  
310      return np.dot(design_matrix, q[:, m:])

313  def _get_free_crs_dmatrix(x, knots, cyclic=False):
314      """Builds an unconstrained cubic regression spline design matrix. 
315   
316      Returns design matrix with dimensions ``len(x) x n`` 
317      for a cubic regression spline smoother 
318      where 
319       - ``n = len(knots)`` for natural CRS 
320       - ``n = len(knots) - 1`` for cyclic CRS 
321   
322      .. note:: See 'Generalized Additive Models', Simon N. Wood, 2006, p. 145 
323   
324      :param x: The 1-d array values. 
325      :param knots: The 1-d array knots used for cubic spline parametrization, 
326       must be sorted in ascending order. 
327      :param cyclic: Indicates whether used cubic regression splines should 
328       be cyclic or not. Default is ``False``. 
329      :return: The (2-d array) design matrix. 
330      """
331      n = knots.size
332      if cyclic:
333          x = _map_cyclic(x, min(knots), max(knots))
334          n -= 1
335  
336      ajm, ajp, cjm, cjp, j = _compute_base_functions(x, knots)
337  
338      j1 = j + 1
339      if cyclic:
340          j1[j1 == n] = 0
341  
342      i = np.identity(n)
343  
344      if cyclic:
345          f = _get_cyclic_f(knots)
346      else:
347          f = _get_natural_f(knots)
348  
349      dmt = ajm * i[j, :].T + ajp * i[j1, :].T + cjm * f[j, :].T + cjp * f[j1, :].T
350  
351      return dmt.T

354  def _get_crs_dmatrix(x, knots, constraints=None, cyclic=False):
355      """Builds a cubic regression spline design matrix. 
356   
357      Returns design matrix with dimensions len(x) x n 
358      where: 
359       - ``n = len(knots) - nrows(constraints)`` for natural CRS 
360       - ``n = len(knots) - nrows(constraints) - 1`` for cyclic CRS 
361      for a cubic regression spline smoother 
362   
363      :param x: The 1-d array values. 
364      :param knots: The 1-d array knots used for cubic spline parametrization, 
365       must be sorted in ascending order. 
366      :param constraints: The 2-d array defining model parameters (``betas``) 
367       constraints (``np.dot(constraints, betas) = 0``). 
368      :param cyclic: Indicates whether used cubic regression splines should 
369       be cyclic or not. Default is ``False``. 
370      :return: The (2-d array) design matrix. 
371      """
372      dm = _get_free_crs_dmatrix(x, knots, cyclic)
373      if constraints is not None:
374          dm = _absorb_constraints(dm, constraints)
375  
376      return dm

379  def _get_te_dmatrix(design_matrices, constraints=None):
380      """Builds tensor product design matrix, given the marginal design matrices. 
381   
382      :param design_matrices: A sequence of 2-d arrays (marginal design matrices). 
383      :param constraints: The 2-d array defining model parameters (``betas``) 
384       constraints (``np.dot(constraints, betas) = 0``). 
385      :return: The (2-d array) design matrix. 
386      """
387      dm = _row_tensor_product(design_matrices)
388      if constraints is not None:
389          dm = _absorb_constraints(dm, constraints)
390  
391      return dm

397  def _get_all_sorted_knots(
398      x, n_inner_knots=None, inner_knots=None, lower_bound=None, upper_bound=None
399  ):
400      """Gets all knots locations with lower and upper exterior knots included. 
401   
402      If needed, inner knots are computed as equally spaced quantiles of the 
403      input data falling between given lower and upper bounds. 
404   
405      :param x: The 1-d array data values. 
406      :param n_inner_knots: Number of inner knots to compute. 
407      :param inner_knots: Provided inner knots if any. 
408      :param lower_bound: The lower exterior knot location. If unspecified, the 
409       minimum of ``x`` values is used. 
410      :param upper_bound: The upper exterior knot location. If unspecified, the 
411       maximum of ``x`` values is used. 
412      :return: The array of ``n_inner_knots + 2`` distinct knots. 
413   
414      :raise ValueError: for various invalid parameters sets or if unable to 
415       compute ``n_inner_knots + 2`` distinct knots. 
416      """
417      if lower_bound is None and x.size == 0:
418          raise ValueError(
419              "Cannot set lower exterior knot location: empty "
420              "input data and lower_bound not specified."
421          )
422      elif lower_bound is None and x.size != 0:
423          lower_bound = np.min(x)
424  
425      if upper_bound is None and x.size == 0:
426          raise ValueError(
427              "Cannot set upper exterior knot location: empty "
428              "input data and upper_bound not specified."
429          )
430      elif upper_bound is None and x.size != 0:
431          upper_bound = np.max(x)
432  
433      if upper_bound < lower_bound:
434          raise ValueError(
435              "lower_bound > upper_bound (%r > %r)" % (lower_bound, upper_bound)
436          )
437  
438      if inner_knots is None and n_inner_knots is not None:
439          if n_inner_knots < 0:
440              raise ValueError(
441                  "Invalid requested number of inner knots: %r" % (n_inner_knots,)
442              )
443  
444          x = x[(lower_bound <= x) & (x <= upper_bound)]
445          x = np.unique(x)
446  
447          if x.size != 0:
448              inner_knots_q = np.linspace(0, 100, n_inner_knots + 2)[1:-1]
449              # .tolist() is necessary to work around a bug in numpy 1.8
450              inner_knots = np.asarray(np.percentile(x, inner_knots_q.tolist()))
451          elif n_inner_knots == 0:
452              inner_knots = np.array([])
453          else:
454              raise ValueError(
455                  "No data values between lower_bound(=%r) and "
456                  "upper_bound(=%r): cannot compute requested "
457                  "%r inner knot(s)." % (lower_bound, upper_bound, n_inner_knots)
458              )
459      elif inner_knots is not None:
460          inner_knots = np.unique(inner_knots)
461          if n_inner_knots is not None and n_inner_knots != inner_knots.size:
462              raise ValueError(
463                  "Needed number of inner knots=%r does not match "
464                  "provided number of inner knots=%r." % (n_inner_knots, inner_knots.size)
465              )
466          n_inner_knots = inner_knots.size
467          if np.any(inner_knots < lower_bound):
468              raise ValueError(
469                  "Some knot values (%s) fall below lower bound "
470                  "(%r)." % (inner_knots[inner_knots < lower_bound], lower_bound)
471              )
472          if np.any(inner_knots > upper_bound):
473              raise ValueError(
474                  "Some knot values (%s) fall above upper bound "
475                  "(%r)." % (inner_knots[inner_knots > upper_bound], upper_bound)
476              )
477      else:
478          raise ValueError("Must specify either 'n_inner_knots' or 'inner_knots'.")
479  
480      all_knots = np.concatenate(([lower_bound, upper_bound], inner_knots))
481      all_knots = np.unique(all_knots)
482      if all_knots.size != n_inner_knots + 2:
483          raise ValueError(
484              "Unable to compute n_inner_knots(=%r) + 2 distinct "
485              "knots: %r data value(s) found between "
486              "lower_bound(=%r) and upper_bound(=%r)."
487              % (n_inner_knots, x.size, lower_bound, upper_bound)
488          )
489  
490      return all_knots

540  def _get_centering_constraint_from_dmatrix(design_matrix):
541      """Computes the centering constraint from the given design matrix. 
542   
543      We want to ensure that if ``b`` is the array of parameters, our 
544      model is centered, ie ``np.mean(np.dot(design_matrix, b))`` is zero. 
545      We can rewrite this as ``np.dot(c, b)`` being zero with ``c`` a 1-row 
546      constraint matrix containing the mean of each column of ``design_matrix``. 
547   
548      :param design_matrix: The 2-d array design matrix. 
549      :return: A 2-d array (1 x ncols(design_matrix)) defining the 
550       centering constraint. 
551      """
552      return design_matrix.mean(axis=0).reshape((1, design_matrix.shape[1]))

555  class CubicRegressionSpline(object):
556      """Base class for cubic regression spline stateful transforms 
557   
558      This class contains all the functionality for the following stateful 
559      transforms: 
560       - ``cr(x, df=None, knots=None, lower_bound=None, upper_bound=None, constraints=None)`` 
561         for natural cubic regression spline 
562       - ``cc(x, df=None, knots=None, lower_bound=None, upper_bound=None, constraints=None)`` 
563         for cyclic cubic regression spline 
564      """
565  
566      common_doc = """ 
567      :arg df: The number of degrees of freedom to use for this spline. The 
568        return value will have this many columns. You must specify at least one 
569        of ``df`` and ``knots``. 
570      :arg knots: The interior knots to use for the spline. If unspecified, then 
571        equally spaced quantiles of the input data are used. You must specify at 
572        least one of ``df`` and ``knots``. 
573      :arg lower_bound: The lower exterior knot location. 
574      :arg upper_bound: The upper exterior knot location. 
575      :arg constraints: Either a 2-d array defining general linear constraints 
576       (that is ``np.dot(constraints, betas)`` is zero, where ``betas`` denotes 
577       the array of *initial* parameters, corresponding to the *initial* 
578       unconstrained design matrix), or the string 
579       ``'center'`` indicating that we should apply a centering constraint 
580       (this constraint will be computed from the input data, remembered and 
581       re-used for prediction from the fitted model). 
582       The constraints are absorbed in the resulting design matrix which means 
583       that the model is actually rewritten in terms of 
584       *unconstrained* parameters. For more details see :ref:`spline-regression`. 
585   
586      This is a stateful transforms (for details see 
587      :ref:`stateful-transforms`). If ``knots``, ``lower_bound``, or 
588      ``upper_bound`` are not specified, they will be calculated from the data 
589      and then the chosen values will be remembered and re-used for prediction 
590      from the fitted model. 
591   
592      Using this function requires scipy be installed. 
593   
594      .. versionadded:: 0.3.0 
595      """
596  
597      def __init__(self, name, cyclic):
598          self._name = name
599          self._cyclic = cyclic
600          self._tmp = {}
601          self._all_knots = None
602          self._constraints = None
603  
604      def memorize_chunk(
605          self,
606          x,
607          df=None,
608          knots=None,
609          lower_bound=None,
610          upper_bound=None,
611          constraints=None,
612      ):
613          args = {
614              "df": df,
615              "knots": knots,
616              "lower_bound": lower_bound,
617              "upper_bound": upper_bound,
618              "constraints": constraints,
619          }
620          self._tmp["args"] = args
621  
622          x = np.atleast_1d(x)
623          if x.ndim == 2 and x.shape[1] == 1:
624              x = x[:, 0]
625          if x.ndim > 1:
626              raise ValueError(
627                  "Input to %r must be 1-d, " "or a 2-d column vector." % (self._name,)
628              )
629  
630          self._tmp.setdefault("xs", []).append(x)
631  
632      def memorize_finish(self):
633          args = self._tmp["args"]
634          xs = self._tmp["xs"]
635          # Guards against invalid subsequent memorize_chunk() calls.
636          del self._tmp
637  
638          x = np.concatenate(xs)
639          if args["df"] is None and args["knots"] is None:
640              raise ValueError("Must specify either 'df' or 'knots'.")
641  
642          constraints = args["constraints"]
643          n_constraints = 0
644          if constraints is not None:
645              if safe_string_eq(constraints, "center"):
646                  # Here we collect only number of constraints,
647                  # actual centering constraint will be computed after all_knots
648                  n_constraints = 1
649              else:
650                  constraints = np.atleast_2d(constraints)
651                  if constraints.ndim != 2:
652                      raise ValueError("Constraints must be 2-d array or " "1-d vector.")
653                  n_constraints = constraints.shape[0]
654  
655          n_inner_knots = None
656          if args["df"] is not None:
657              min_df = 1
658              if not self._cyclic and n_constraints == 0:
659                  min_df = 2
660              if args["df"] < min_df:
661                  raise ValueError(
662                      "'df'=%r must be greater than or equal to %r."
663                      % (args["df"], min_df)
664                  )
665              n_inner_knots = args["df"] - 2 + n_constraints
666              if self._cyclic:
667                  n_inner_knots += 1
668          self._all_knots = _get_all_sorted_knots(
669              x,
670              n_inner_knots=n_inner_knots,
671              inner_knots=args["knots"],
672              lower_bound=args["lower_bound"],
673              upper_bound=args["upper_bound"],
674          )
675          if constraints is not None:
676              if safe_string_eq(constraints, "center"):
677                  # Now we can compute centering constraints
678                  constraints = _get_centering_constraint_from_dmatrix(
679                      _get_free_crs_dmatrix(x, self._all_knots, cyclic=self._cyclic)
680                  )
681  
682              df_before_constraints = self._all_knots.size
683              if self._cyclic:
684                  df_before_constraints -= 1
685              if constraints.shape[1] != df_before_constraints:
686                  raise ValueError(
687                      "Constraints array should have %r columns but"
688                      " %r found." % (df_before_constraints, constraints.shape[1])
689                  )
690              self._constraints = constraints
691  
692      def transform(
693          self,
694          x,
695          df=None,
696          knots=None,
697          lower_bound=None,
698          upper_bound=None,
699          constraints=None,
700      ):
701          x_orig = x
702          x = np.atleast_1d(x)
703          if x.ndim == 2 and x.shape[1] == 1:
704              x = x[:, 0]
705          if x.ndim > 1:
706              raise ValueError(
707                  "Input to %r must be 1-d, " "or a 2-d column vector." % (self._name,)
708              )
709          dm = _get_crs_dmatrix(
710              x, self._all_knots, self._constraints, cyclic=self._cyclic
711          )
712          if have_pandas:
713              if isinstance(x_orig, (pandas.Series, pandas.DataFrame)):
714                  dm = pandas.DataFrame(dm)
715                  dm.index = x_orig.index
716          return dm

910  class TE(object):
911      """te(s1, .., sn, constraints=None) 
912   
913      Generates smooth of several covariates as a tensor product of the bases 
914      of marginal univariate smooths ``s1, .., sn``. The marginal smooths are 
915      required to transform input univariate data into some kind of smooth 
916      functions basis producing a 2-d array output with the ``(i, j)`` element 
917      corresponding to the value of the ``j`` th basis function at the ``i`` th 
918      data point. 
919      The resulting basis dimension is the product of the basis dimensions of 
920      the marginal smooths. The usual usage is something like:: 
921   
922        y ~ 1 + te(cr(x1, df=5), cc(x2, df=6), constraints='center') 
923   
924      to fit ``y`` as a smooth function of both ``x1`` and ``x2``, with a natural 
925      cubic spline for ``x1`` marginal smooth and a cyclic cubic spline for 
926      ``x2`` (and centering constraint absorbed in the resulting design matrix). 
927   
928      :arg constraints: Either a 2-d array defining general linear constraints 
929       (that is ``np.dot(constraints, betas)`` is zero, where ``betas`` denotes 
930       the array of *initial* parameters, corresponding to the *initial* 
931       unconstrained design matrix), or the string 
932       ``'center'`` indicating that we should apply a centering constraint 
933       (this constraint will be computed from the input data, remembered and 
934       re-used for prediction from the fitted model). 
935       The constraints are absorbed in the resulting design matrix which means 
936       that the model is actually rewritten in terms of 
937       *unconstrained* parameters. For more details see :ref:`spline-regression`. 
938   
939      Using this function requires scipy be installed. 
940   
941      .. note:: This function reproduce the tensor product smooth 'te' as 
942        implemented in the R package 'mgcv' (GAM modelling). 
943        See also 'Generalized Additive Models', Simon N. Wood, 2006, pp 158-163 
944   
945      .. versionadded:: 0.3.0 
946      """
947  
948      def __init__(self):
949          self._tmp = {}
950          self._constraints = None
951  
952      def memorize_chunk(self, *args, **kwargs):
953          constraints = self._tmp.setdefault("constraints", kwargs.get("constraints"))
954          if safe_string_eq(constraints, "center"):
955              args_2d = []
956              for arg in args:
957                  arg = atleast_2d_column_default(arg)
958                  if arg.ndim != 2:
959                      raise ValueError(
960                          "Each tensor product argument must be "
961                          "a 2-d array or 1-d vector."
962                      )
963                  args_2d.append(arg)
964  
965              tp = _row_tensor_product(args_2d)
966              self._tmp.setdefault("count", 0)
967              self._tmp["count"] += tp.shape[0]
968  
969              chunk_sum = np.atleast_2d(tp.sum(axis=0))
970              self._tmp.setdefault("sum", np.zeros(chunk_sum.shape))
971              self._tmp["sum"] += chunk_sum
972  
973      def memorize_finish(self):
974          tmp = self._tmp
975          constraints = self._tmp["constraints"]
976          # Guards against invalid subsequent memorize_chunk() calls.
977          del self._tmp
978  
979          if constraints is not None:
980              if safe_string_eq(constraints, "center"):
981                  constraints = np.atleast_2d(tmp["sum"] / tmp["count"])
982              else:
983                  constraints = np.atleast_2d(constraints)
984                  if constraints.ndim != 2:
985                      raise ValueError("Constraints must be 2-d array or " "1-d vector.")
986  
987          self._constraints = constraints
988  
989      def transform(self, *args, **kwargs):
990          args_2d = []
991          for arg in args:
992              arg = atleast_2d_column_default(arg)
993              if arg.ndim != 2:
994                  raise ValueError(
995                      "Each tensor product argument must be " "a 2-d array or 1-d vector."
996                  )
997              args_2d.append(arg)
998  
999          return _get_te_dmatrix(args_2d, self._constraints)
