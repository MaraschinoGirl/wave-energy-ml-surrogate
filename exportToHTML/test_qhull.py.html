94       """ 
95       Generate incremental datasets from basic data sets 
96       """
97       points = DATASETS[name]
98       ndim = points.shape[1]

100      opts = None
101      nmin = ndim + 2

107          # include enough points so that we get different x-coordinates
108          nmin = 12

111      for j in range(nmin, len(points), chunksize):
112          chunks.append(points[j:j+chunksize])

114      new_name = "%s-chunk-%d" % (name, chunksize)
115      assert new_name not in INCREMENTAL_DATASETS

126          # Check that Qhull state swapping works
127  
128          x = qhull._Qhull(b'v',
129                           np.array([[0,0],[0,1],[1,0],[1,1.],[0.5,0.5]]),
130                           b'Qz')

136          yd = copy.deepcopy(y.get_voronoi_diagram())
137  
138          xd2 = copy.deepcopy(x.get_voronoi_diagram())
139          x.close()
140          yd2 = copy.deepcopy(y.get_voronoi_diagram())

143          assert_raises(RuntimeError, x.get_voronoi_diagram)
144          assert_raises(RuntimeError, y.get_voronoi_diagram)

146          assert_allclose(xd[0], xd2[0])
147          assert_unordered_tuple_list_equal(xd[1], xd2[1], tpl=sorted_tuple)
148          assert_unordered_tuple_list_equal(xd[2], xd2[2], tpl=sorted_tuple)
149          assert_unordered_tuple_list_equal(xd[3], xd2[3], tpl=sorted_tuple)
150          assert_array_equal(xd[4], xd2[4])

197      def test_plane_distance(self):
198          # Compare plane distance from hyperplane equations obtained from Qhull
199          # to manually computed plane equations
200          x = np.array([(0,0), (1, 1), (1, 0), (0.99189033, 0.37674127),
201                        (0.99440079, 0.45182168)], dtype=np.float64)
202          p = np.array([0.99966555, 0.15685619], dtype=np.float64)
203  
204          tri = qhull.Delaunay(x)
205  
206          z = tri.lift_points(x)
207          pz = tri.lift_points(p)
208  
209          dist = tri.plane_distance(p)
210  
211          for j, v in enumerate(tri.simplices):
212              x1 = z[v[0]]
213              x2 = z[v[1]]
214              x3 = z[v[2]]
215  
216              n = np.cross(x1 - x3, x2 - x3)
217              n /= np.sqrt(np.dot(n, n))
218              n *= -np.sign(n[2])
219  
220              d = np.dot(n, pz - x3)
221  
222              assert_almost_equal(dist[j], d)

285      def _check_barycentric_transforms(self, tri, err_msg="",
286                                        unit_cube=False,
287                                        unit_cube_tol=0):
288          """Check that a triangulation has reasonable barycentric transforms"""
289          vertices = tri.points[tri.simplices]
290          sc = 1/(tri.ndim + 1.0)
291          centroids = vertices.sum(axis=1) * sc
292  
293          # Either: (i) the simplex has a `nan` barycentric transform,
294          # or, (ii) the centroid is in the simplex
295  
296          def barycentric_transform(tr, x):
297              r = tr[:,-1,:]
298              Tinv = tr[:,:-1,:]
299              return np.einsum('ijk,ik->ij', Tinv, x - r)
300  
301          eps = np.finfo(float).eps
302  
303          c = barycentric_transform(tri.transform, centroids)
304          with np.errstate(invalid="ignore"):
305              ok = np.isnan(c).all(axis=1) | (abs(c - sc)/sc < 0.1).all(axis=1)
306  
307          assert_(ok.all(), f"{err_msg} {np.nonzero(~ok)}")
308  
309          # Invalid simplices must be (nearly) zero volume
310          q = vertices[:,:-1,:] - vertices[:,-1,None,:]
311          volume = np.array([np.linalg.det(q[k,:,:])
312                             for k in range(tri.nsimplex)])
313          ok = np.isfinite(tri.transform[:,0,0]) | (volume < np.sqrt(eps))
314          assert_(ok.all(), f"{err_msg} {np.nonzero(~ok)}")
315  
316          # Also, find_simplex for the centroid should end up in some
317          # simplex for the non-degenerate cases
318          j = tri.find_simplex(centroids)
319          ok = (j != -1) | np.isnan(tri.transform[:,0,0])
320          assert_(ok.all(), f"{err_msg} {np.nonzero(~ok)}")
321  
322          if unit_cube:
323              # If in unit cube, no interior point should be marked out of hull
324              at_boundary = (centroids <= unit_cube_tol).any(axis=1)
325              at_boundary |= (centroids >= 1 - unit_cube_tol).any(axis=1)
326  
327              ok = (j != -1) | at_boundary
328              assert_(ok.all(), f"{err_msg} {np.nonzero(~ok)}")

550  def assert_hulls_equal(points, facets_1, facets_2):
551      # Check that two convex hulls constructed from the same point set
552      # are equal
553  
554      facets_1 = set(map(sorted_tuple, facets_1))
555      facets_2 = set(map(sorted_tuple, facets_2))
556  
557      if facets_1 != facets_2 and points.shape[1] == 2:
558          # The direct check fails for the pathological cases
559          # --- then the convex hull from Delaunay differs (due
560          # to rounding error etc.) from the hull computed
561          # otherwise, by the question whether (tricoplanar)
562          # points that lie almost exactly on the hull are
563          # included as vertices of the hull or not.
564          #
565          # So we check the result, and accept it if the Delaunay
566          # hull line segments are a subset of the usual hull.
567  
568          eps = 1000 * np.finfo(float).eps
569  
570          for a, b in facets_1:
571              for ap, bp in facets_2:
572                  t = points[bp] - points[ap]
573                  t /= np.linalg.norm(t)       # tangent
574                  n = np.array([-t[1], t[0]])  # normal
575  
576                  # check that the two line segments are parallel
577                  # to the same line
578                  c1 = np.dot(n, points[b] - points[ap])
579                  c2 = np.dot(n, points[a] - points[ap])
580                  if not np.allclose(np.dot(c1, n), 0):
581                      continue
582                  if not np.allclose(np.dot(c2, n), 0):
583                      continue
584  
585                  # Check that the segment (a, b) is contained in (ap, bp)
586                  c1 = np.dot(t, points[a] - points[ap])
587                  c2 = np.dot(t, points[b] - points[ap])
588                  c3 = np.dot(t, points[bp] - points[ap])
589                  if c1 < -eps or c1 > c3 + eps:
590                      continue
591                  if c2 < -eps or c2 > c3 + eps:
592                      continue
593  
594                  # OK:
595                  break
596              else:
597                  raise AssertionError("comparison fails")
598  
599          # it was OK
600          return
601  
602      assert_equal(facets_1, facets_2)

874      def _compare_qvoronoi(self, points, output, **kw):
875          """Compare to output from 'qvoronoi o Fv < data' to Voronoi()"""
876  
877          # Parse output
878          output = [list(map(float, x.split())) for x in output.strip().splitlines()]
879          nvertex = int(output[1][0])
880          vertices = list(map(tuple, output[3:2+nvertex]))  # exclude inf
881          nregion = int(output[1][1])
882          regions = [[int(y)-1 for y in x[1:]]
883                     for x in output[2+nvertex:2+nvertex+nregion]]
884          ridge_points = [[int(y) for y in x[1:3]]
885                          for x in output[3+nvertex+nregion:]]
886          ridge_vertices = [[int(y)-1 for y in x[3:]]
887                            for x in output[3+nvertex+nregion:]]
888  
889          # Compare results
890          vor = qhull.Voronoi(points, **kw)
891  
892          def sorttuple(x):
893              return tuple(sorted(x))
894  
895          assert_allclose(vor.vertices, vertices)
896          assert_equal(set(map(tuple, vor.regions)),
897                       set(map(tuple, regions)))
898  
899          p1 = list(zip(list(map(sorttuple, ridge_points)),
900                        list(map(sorttuple, ridge_vertices))))
901          p2 = list(zip(list(map(sorttuple, vor.ridge_points.tolist())),
902                        list(map(sorttuple, vor.ridge_vertices))))
903          p1.sort()
904          p2.sort()
905  
906          assert_equal(p1, p2)

1108                                [0., -1., 0., -0.5],
1109                                [-1., 0., 0., -0.5],
1110                                [1., 0., 0., -0.5],

1112                                [0., 0., 1., -0.5]])
1113         #Cut each summit
1114         extra_normals = np.array([[1., 1., 1.],
1115                                   [1., 1., -1.],
1116                                   [1., -1., 1.],
1117                                   [1, -1., -1.]])

1120                                       offsets))
1121 
1122         feas_point = np.array([0., 0., 0.])
1123 
1124         inc_hs = qhull.HalfspaceIntersection(halfspaces, feas_point, incremental=True)
1125 
1126         inc_res_hs = qhull.HalfspaceIntersection(halfspaces, feas_point,
1127                                                  incremental=True)
1128 
1129         for i, ehs in enumerate(extra_halfspaces):
1130             inc_hs.add_halfspaces(ehs[np.newaxis, :])
1131
