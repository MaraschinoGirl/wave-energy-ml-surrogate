77   def test_docstring_parameters():
78       # Test module docstring formatting
79   
80       # Skip test if numpydoc is not found
81       pytest.importorskip(
82           "numpydoc", reason="numpydoc is required to test the docstrings"
83       )
84   
85       # XXX unreached code as of v0.22
86       from numpydoc import docscrape
87   
88       incorrect = []
89       for name in PUBLIC_MODULES:
90           if name.endswith(".conftest"):
91               # pytest tooling, not part of the scikit-learn API
92               continue
93           if name == "sklearn.utils.fixes":
94               # We cannot always control these docstrings
95               continue
96           with warnings.catch_warnings(record=True):
97               module = importlib.import_module(name)
98           classes = inspect.getmembers(module, inspect.isclass)
99           # Exclude non-scikit-learn classes
100          classes = [cls for cls in classes if cls[1].__module__.startswith("sklearn")]
101          for cname, cls in classes:
102              this_incorrect = []
103              if cname in _DOCSTRING_IGNORES or cname.startswith("_"):
104                  continue
105              if inspect.isabstract(cls):
106                  continue
107              with warnings.catch_warnings(record=True) as w:
108                  cdoc = docscrape.ClassDoc(cls)
109              if len(w):
110                  raise RuntimeError(
111                      "Error for __init__ of %s in %s:\n%s" % (cls, name, w[0])
112                  )
113  
114              # Skip checks on deprecated classes
115              if _is_deprecated(cls.__new__):
116                  continue
117  
118              this_incorrect += check_docstring_parameters(cls.__init__, cdoc)
119  
120              for method_name in cdoc.methods:
121                  method = getattr(cls, method_name)
122                  if _is_deprecated(method):
123                      continue
124                  param_ignore = None
125                  # Now skip docstring test for y when y is None
126                  # by default for API reason
127                  if method_name in _METHODS_IGNORE_NONE_Y:
128                      sig = signature(method)
129                      if "y" in sig.parameters and sig.parameters["y"].default is None:
130                          param_ignore = ["y"]  # ignore y for fit and score
131                  result = check_docstring_parameters(method, ignore=param_ignore)
132                  this_incorrect += result
133  
134              incorrect += this_incorrect
135  
136          functions = inspect.getmembers(module, inspect.isfunction)
137          # Exclude imported functions
138          functions = [fn for fn in functions if fn[1].__module__ == name]
139          for fname, func in functions:
140              # Don't test private methods / functions
141              if fname.startswith("_"):
142                  continue
143              if fname == "configuration" and name.endswith("setup"):
144                  continue
145              name_ = _get_func_name(func)
146              if not any(d in name_ for d in _DOCSTRING_IGNORES) and not _is_deprecated(
147                  func
148              ):
149                  incorrect += check_docstring_parameters(func)
150  
151      msg = "\n".join(incorrect)
152      if len(incorrect) > 0:
153          raise AssertionError("Docstring Error:\n" + msg)

156  def _construct_searchcv_instance(SearchCV):
157      return SearchCV(LogisticRegression(), {"C": [0.1, 1]})

160  def _construct_compose_pipeline_instance(Estimator):
161      # Minimal / degenerate instances: only useful to test the docstrings.
162      if Estimator.__name__ == "ColumnTransformer":
163          return Estimator(transformers=[("transformer", "passthrough", [0, 1])])
164      elif Estimator.__name__ == "Pipeline":
165          return Estimator(steps=[("clf", LogisticRegression())])
166      elif Estimator.__name__ == "FeatureUnion":
167          return Estimator(transformer_list=[("transformer", FunctionTransformer())])

170  def _construct_sparse_coder(Estimator):
171      # XXX: hard-coded assumption that n_features=3
172      dictionary = np.array(
173          [[0, 1, 0], [-1, -1, 2], [1, 1, 1], [0, 1, 1], [0, 2, 1]],
174          dtype=np.float64,
175      )
176      return Estimator(dictionary=dictionary)

179  @pytest.mark.filterwarnings("ignore::sklearn.exceptions.ConvergenceWarning")
180  @pytest.mark.parametrize("name, Estimator", all_estimators())
181  def test_fit_docstring_attributes(name, Estimator):
182      pytest.importorskip("numpydoc")
183      from numpydoc import docscrape
184  
185      doc = docscrape.ClassDoc(Estimator)
186      attributes = doc["Attributes"]
187  
188      if Estimator.__name__ in (
189          "HalvingRandomSearchCV",
190          "RandomizedSearchCV",
191          "HalvingGridSearchCV",
192          "GridSearchCV",
193      ):
194          est = _construct_searchcv_instance(Estimator)
195      elif Estimator.__name__ in (
196          "ColumnTransformer",
197          "Pipeline",
198          "FeatureUnion",
199      ):
200          est = _construct_compose_pipeline_instance(Estimator)
201      elif Estimator.__name__ == "SparseCoder":
202          est = _construct_sparse_coder(Estimator)
203      elif Estimator.__name__ == "FrozenEstimator":
204          X, y = make_classification(n_samples=20, n_features=5, random_state=0)
205          est = Estimator(LogisticRegression().fit(X, y))
206      else:
207          # TODO(devtools): use _tested_estimators instead of all_estimators in the
208          # decorator
209          est = next(_construct_instances(Estimator))
210  
211      if Estimator.__name__ == "SelectKBest":
212          est.set_params(k=2)
213      elif Estimator.__name__ == "DummyClassifier":
214          est.set_params(strategy="stratified")
215      elif Estimator.__name__ == "CCA" or Estimator.__name__.startswith("PLS"):
216          # default = 2 is invalid for single target
217          est.set_params(n_components=1)
218      elif Estimator.__name__ in (
219          "GaussianRandomProjection",
220          "SparseRandomProjection",
221      ):
222          # default="auto" raises an error with the shape of `X`
223          est.set_params(n_components=2)
224      elif Estimator.__name__ == "TSNE":
225          # default raises an error, perplexity must be less than n_samples
226          est.set_params(perplexity=2)
227  
228      # Low max iter to speed up tests: we are only interested in checking the existence
229      # of fitted attributes. This should be invariant to whether it has converged or not.
230      if "max_iter" in est.get_params():
231          est.set_params(max_iter=2)
232          # min value for `TSNE` is 250
233          if Estimator.__name__ == "TSNE":
234              est.set_params(max_iter=250)
235  
236      if "random_state" in est.get_params():
237          est.set_params(random_state=0)
238  
239      # In case we want to deprecate some attributes in the future
240      skipped_attributes = {}
241  
242      if Estimator.__name__.endswith("Vectorizer"):
243          # Vectorizer require some specific input data
244          if Estimator.__name__ in (
245              "CountVectorizer",
246              "HashingVectorizer",
247              "TfidfVectorizer",
248          ):
249              X = [
250                  "This is the first document.",
251                  "This document is the second document.",
252                  "And this is the third one.",
253                  "Is this the first document?",
254              ]
255          elif Estimator.__name__ == "DictVectorizer":
256              X = [{"foo": 1, "bar": 2}, {"foo": 3, "baz": 1}]
257          y = None
258      else:
259          X, y = make_classification(
260              n_samples=20,
261              n_features=3,
262              n_redundant=0,
263              n_classes=2,
264              random_state=2,
265          )
266  
267          y = _enforce_estimator_tags_y(est, y)
268          X = _enforce_estimator_tags_X(est, X)
269  
270      if est.__sklearn_tags__().target_tags.one_d_labels:
271          est.fit(y)
272      elif est.__sklearn_tags__().target_tags.two_d_labels:
273          est.fit(np.c_[y, y])
274      elif est.__sklearn_tags__().input_tags.three_d_array:
275          est.fit(X[np.newaxis, ...], y)
276      else:
277          est.fit(X, y)
278  
279      for attr in attributes:
280          if attr.name in skipped_attributes:
281              continue
282          desc = " ".join(attr.desc).lower()
283          # As certain attributes are present "only" if a certain parameter is
284          # provided, this checks if the word "only" is present in the attribute
285          # description, and if not the attribute is required to be present.
286          if "only " in desc:
287              continue
288          # ignore deprecation warnings
289          with ignore_warnings(category=FutureWarning):
290              assert hasattr(est, attr.name)
291  
292      fit_attr = _get_all_fitted_attributes(est)
293      fit_attr_names = [attr.name for attr in attributes]
294      undocumented_attrs = set(fit_attr).difference(fit_attr_names)
295      undocumented_attrs = set(undocumented_attrs).difference(skipped_attributes)
296      if undocumented_attrs:
297          raise AssertionError(
298              f"Undocumented attributes for {Estimator.__name__}: {undocumented_attrs}"
299          )

302  def _get_all_fitted_attributes(estimator):
303      "Get all the fitted attributes of an estimator including properties"
304      # attributes
305      fit_attr = list(estimator.__dict__.keys())
306  
307      # properties
308      with warnings.catch_warnings():
309          warnings.filterwarnings("error", category=FutureWarning)
310  
311          for name in dir(estimator.__class__):
312              obj = getattr(estimator.__class__, name)
313              if not isinstance(obj, property):
314                  continue
315  
316              # ignore properties that raises an AttributeError and deprecated
317              # properties
318              try:
319                  getattr(estimator, name)
320              except (AttributeError, FutureWarning):
321                  continue
322              fit_attr.append(name)
323  
324      return [k for k in fit_attr if k.endswith("_") and not k.startswith("_")]

327  @skip_if_no_numpydoc
328  def test_precision_recall_f_score_docstring_consistency():
329      """Check docstrings parameters of related metrics are consistent."""
330      metrics_to_check = [
331          metrics.precision_recall_fscore_support,
332          metrics.f1_score,
333          metrics.fbeta_score,
334          metrics.precision_score,
335          metrics.recall_score,
336      ]
337      assert_docstring_consistency(
338          metrics_to_check,
339          include_params=True,
340          # "zero_division" - the reason for zero division differs between f scores,
341          # precision and recall.
342          exclude_params=["average", "zero_division"],
343      )
344      description_regex = (
345          r"""This parameter is required for multiclass/multilabel targets\. 
346          If ``None``, the metrics for each class are returned\. Otherwise, this 
347          determines the type of averaging performed on the data: 
348          ``'binary'``: 
349              Only report results for the class specified by ``pos_label``\. 
350              This is applicable only if targets \(``y_\{true,pred\}``\) are binary\. 
351          ``'micro'``: 
352              Calculate metrics globally by counting the total true positives, 
353              false negatives and false positives\. 
354          ``'macro'``: 
355              Calculate metrics for each label, and find their unweighted 
356              mean\.  This does not take label imbalance into account\. 
357          ``'weighted'``: 
358              Calculate metrics for each label, and find their average weighted 
359              by support \(the number of true instances for each label\)\. This 
360              alters 'macro' to account for label imbalance; it can result in an 
361              F-score that is not between precision and recall\."""
362          + r"[\s\w]*\.*"  # optionally match additonal sentence
363          + r""" 
364          ``'samples'``: 
365              Calculate metrics for each instance, and find their average \(only 
366              meaningful for multilabel classification where this differs from 
367              :func:`accuracy_score`\)\."""
368      )
369      assert_docstring_consistency(
370          metrics_to_check,
371          include_params=["average"],
372          descr_regex_pattern=" ".join(description_regex.split()),
373      )
