10   @keras_export("keras.layers.LayerNormalization")
11   class LayerNormalization(Layer):
12       """Layer normalization layer (Ba et al., 2016). 
13    
14       Normalize the activations of the previous layer for each given example in a 
15       batch independently, rather than across a batch like Batch Normalization. 
16       i.e. applies a transformation that maintains the mean activation within each 
17       example close to 0 and the activation standard deviation close to 1. 
18    
19       If `scale` or `center` are enabled, the layer will scale the normalized 
20       outputs by broadcasting them with a trainable variable `gamma`, and center 
21       the outputs by broadcasting with a trainable variable `beta`. `gamma` will 
22       default to a ones tensor and `beta` will default to a zeros tensor, so that 
23       centering and scaling are no-ops before training has begun. 
24    
25       So, with scaling and centering enabled the normalization equations 
26       are as follows: 
27    
28       Let the intermediate activations for a mini-batch to be the `inputs`. 
29    
30       For each sample `x_i` in `inputs` with `k` features, we compute the mean and 
31       variance of the sample: 
32    
33       ```python 
34       mean_i = sum(x_i[j] for j in range(k)) / k 
35       var_i = sum((x_i[j] - mean_i) ** 2 for j in range(k)) / k 
36       ``` 
37    
38       and then compute a normalized `x_i_normalized`, including a small factor 
39       `epsilon` for numerical stability. 
40    
41       ```python 
42       x_i_normalized = (x_i - mean_i) / sqrt(var_i + epsilon) 
43       ``` 
44    
45       And finally `x_i_normalized ` is linearly transformed by `gamma` and `beta`, 
46       which are learned parameters: 
47    
48       ```python 
49       output_i = x_i_normalized * gamma + beta 
50       ``` 
51    
52       `gamma` and `beta` will span the axes of `inputs` specified in `axis`, and 
53       this part of the inputs' shape must be fully defined. 
54    
55       For example: 
56    
57       >>> layer = keras.layers.LayerNormalization(axis=[1, 2, 3]) 
58       >>> layer.build([5, 20, 30, 40]) 
59       >>> print(layer.beta.shape) 
60       (20, 30, 40) 
61       >>> print(layer.gamma.shape) 
62       (20, 30, 40) 
63    
64       Note that other implementations of layer normalization may choose to define 
65       `gamma` and `beta` over a separate set of axes from the axes being 
66       normalized across. For example, Group Normalization 
67       ([Wu et al. 2018](https://arxiv.org/abs/1803.08494)) with group size of 1 
68       corresponds to a Layer Normalization that normalizes across height, width, 
69       and channel and has `gamma` and `beta` span only the channel dimension. 
70       So, this Layer Normalization implementation will not match a Group 
71       Normalization layer with group size set to 1. 
72    
73       Args: 
74           axis: Integer or List/Tuple. The axis or axes to normalize across. 
75               Typically, this is the features axis/axes. The left-out axes are 
76               typically the batch axis/axes. `-1` is the last dimension in the 
77               input. Defaults to `-1`. 
78           epsilon: Small float added to variance to avoid dividing by zero. 
79               Defaults to 1e-3. 
80           center: If True, add offset of `beta` to normalized tensor. If False, 
81               `beta` is ignored. Defaults to `True`. 
82           scale: If True, multiply by `gamma`. If False, `gamma` is not used. 
83               When the next layer is linear (also e.g. `nn.relu`), this can be 
84               disabled since the scaling will be done by the next layer. 
85               Defaults to `True`. 
86           rms_scaling: If True, `center` and `scale` are ignored, and the 
87               inputs are scaled by `gamma` and the inverse square root 
88               of the square of all inputs. This is an approximate and faster 
89               approach that avoids ever computing the mean of the input. Note that 
90               this *isn't* equivalent to the computation that the 
91               `keras.layers.RMSNormalization` layer performs. 
92           beta_initializer: Initializer for the beta weight. Defaults to zeros. 
93           gamma_initializer: Initializer for the gamma weight. Defaults to ones. 
94           beta_regularizer: Optional regularizer for the beta weight. 
95               None by default. 
96           gamma_regularizer: Optional regularizer for the gamma weight. 
97               None by default. 
98           beta_constraint: Optional constraint for the beta weight. 
99               None by default. 
100          gamma_constraint: Optional constraint for the gamma weight. 
101              None by default. 
102          **kwargs: Base layer keyword arguments (e.g. `name` and `dtype`). 
103   
104   
105      Reference: 
106   
107      - [Lei Ba et al., 2016](https://arxiv.org/abs/1607.06450). 
108      """
109  
110      def __init__(
111          self,
112          axis=-1,
113          epsilon=1e-3,
114          center=True,
115          scale=True,
116          rms_scaling=False,
117          beta_initializer="zeros",
118          gamma_initializer="ones",
119          beta_regularizer=None,
120          gamma_regularizer=None,
121          beta_constraint=None,
122          gamma_constraint=None,
123          **kwargs,
124      ):
125          super().__init__(**kwargs)
126          if isinstance(axis, (list, tuple)):
127              self.axis = list(axis)
128          elif isinstance(axis, int):
129              self.axis = axis
130          else:
131              raise TypeError(
132                  "Expected an int or a list/tuple of ints for the "
133                  "argument 'axis', but received: %r" % axis
134              )
135  
136          self.epsilon = epsilon
137          self.center = center
138          self.scale = scale
139          self.rms_scaling = rms_scaling
140          self.beta_initializer = initializers.get(beta_initializer)
141          self.gamma_initializer = initializers.get(gamma_initializer)
142          self.beta_regularizer = regularizers.get(beta_regularizer)
143          self.gamma_regularizer = regularizers.get(gamma_regularizer)
144          self.beta_constraint = constraints.get(beta_constraint)
145          self.gamma_constraint = constraints.get(gamma_constraint)
146  
147          self.supports_masking = True
148          self.autocast = False
149  
150      def build(self, input_shape):
151          if isinstance(self.axis, list):
152              shape = tuple([input_shape[dim] for dim in self.axis])
153          else:
154              shape = (input_shape[self.axis],)
155              self.axis = [self.axis]
156          if self.scale or self.rms_scaling:
157              self.gamma = self.add_weight(
158                  name="gamma",
159                  shape=shape,
160                  initializer=self.gamma_initializer,
161                  regularizer=self.gamma_regularizer,
162                  constraint=self.gamma_constraint,
163                  trainable=True,
164                  autocast=False,
165              )
166          else:
167              self.gamma = None
168  
169          if self.center and not self.rms_scaling:
170              self.beta = self.add_weight(
171                  name="beta",
172                  shape=shape,
173                  initializer=self.beta_initializer,
174                  regularizer=self.beta_regularizer,
175                  constraint=self.beta_constraint,
176                  trainable=True,
177                  autocast=False,
178              )
179          else:
180              self.beta = None
181  
182          self.built = True
183  
184      def call(self, inputs):
185          # Compute the axes along which to reduce the mean / variance
186          input_shape = inputs.shape
187          ndims = len(input_shape)
188  
189          # Broadcasting only necessary for norm when the axis is not just
190          # the last dimension
191          broadcast_shape = [1] * ndims
192          for dim in self.axis:
193              broadcast_shape[dim] = input_shape[dim]
194  
195          def _broadcast(v):
196              if (
197                  v is not None
198                  and len(v.shape) != ndims
199                  and self.axis != [ndims - 1]
200              ):
201                  return ops.reshape(v, broadcast_shape)
202              return v
203  
204          compute_dtype = backend.result_type(inputs.dtype, "float32")
205          # LN is prone to overflow with float16/bfloat16 inputs, so we upcast to
206          # float32 for the subsequent computations.
207          inputs = ops.cast(inputs, compute_dtype)
208  
209          if self.rms_scaling:
210              # Calculate outputs with only variance and gamma if rms scaling
211              # is enabled
212              # Calculate the variance along self.axis (layer activations).
213              variance = ops.var(inputs, axis=self.axis, keepdims=True)
214              inv = ops.rsqrt(variance + self.epsilon)
215  
216              outputs = (
217                  inputs * inv * ops.cast(_broadcast(self.gamma), inputs.dtype)
218              )
219          else:
220              # Calculate the mean & variance along self.axis (layer activations).
221              mean, variance = ops.moments(inputs, axes=self.axis, keepdims=True)
222              gamma, beta = _broadcast(self.gamma), _broadcast(self.beta)
223  
224              inv = ops.rsqrt(variance + self.epsilon)
225              if gamma is not None:
226                  gamma = ops.cast(gamma, inputs.dtype)
227                  inv = inv * gamma
228  
229              res = -mean * inv
230              if beta is not None:
231                  beta = ops.cast(beta, inputs.dtype)
232                  res = res + beta
233  
234              outputs = inputs * inv + res
235          return ops.cast(outputs, self.compute_dtype)
236  
237      def compute_output_shape(self, input_shape):
238          if isinstance(self.axis, int):
239              axes = [self.axis]
240          else:
241              axes = self.axis
242  
243          for axis in axes:
244              if axis >= len(input_shape) or axis < -len(input_shape):
245                  raise ValueError(
246                      f"Axis {axis} is out of bounds for "
247                      f"input shape {input_shape}. "
248                      f"Received: axis={self.axis}"
249                  )
250          return input_shape
251  
252      def get_config(self):
253          config = {
254              "axis": self.axis,
255              "epsilon": self.epsilon,
256              "center": self.center,
257              "scale": self.scale,
258              "rms_scaling": self.rms_scaling,
259              "beta_initializer": initializers.serialize(self.beta_initializer),
260              "gamma_initializer": initializers.serialize(self.gamma_initializer),
261              "beta_regularizer": regularizers.serialize(self.beta_regularizer),
262              "gamma_regularizer": regularizers.serialize(self.gamma_regularizer),
263              "beta_constraint": constraints.serialize(self.beta_constraint),
264              "gamma_constraint": constraints.serialize(self.gamma_constraint),
265          }
266          base_config = super().get_config()
267          return {**base_config, **config}
