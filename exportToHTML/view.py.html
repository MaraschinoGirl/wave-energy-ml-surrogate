108  def DoBraceMatch(control):
109      curPos = control.SCIGetCurrentPos()
110      charBefore = " "
111      if curPos:
112          charBefore = control.SCIGetCharAt(curPos - 1)
113      charAt = control.SCIGetCharAt(curPos)
114      braceAtPos = braceOpposite = -1
115      if charBefore in "[](){}":
116          braceAtPos = curPos - 1
117      if braceAtPos == -1:
118          if charAt in "[](){}":
119              braceAtPos = curPos
120      if braceAtPos != -1:
121          braceOpposite = control.SCIBraceMatch(braceAtPos, 0)
122      if braceAtPos != -1 and braceOpposite == -1:
123          control.SCIBraceBadHighlight(braceAtPos)
124      else:
125          # either clear them both or set them both.
126          control.SCIBraceHighlight(braceAtPos, braceOpposite)

129  def _get_class_attributes(ob):
130      # Recurse into base classes looking for attributes
131      items = []
132      try:
133          items.extend(dir(ob))
134          for i in ob.__bases__:
135              for item in _get_class_attributes(i):
136                  if item not in items:
137                      items.append(item)
138      except AttributeError:
139          pass
140      return items

318      def EnsureCharsVisible(self, start, end=None):
319          if end is None:
320              end = start
321          lineStart = self.LineFromChar(min(start, end))
322          lineEnd = self.LineFromChar(max(start, end))
323          while lineStart <= lineEnd:
324              self.SCIEnsureVisible(lineStart)

430      def OnCmdFileLocate(self, cmd, id):
431          line = self.GetLine().strip()
432          import pywin.framework.scriptutils
433  
434          m = patImport.match(line)
435          if m:
436              # Module name on this line - locate that!
437              modName = m.group("name")
438              fileName = pywin.framework.scriptutils.LocatePythonFile(modName)
439              if fileName is None:
440                  win32ui.SetStatusText("Can't locate module %s" % modName)
441                  return 1  # Let the default get it.
442              else:
443                  win32ui.GetApp().OpenDocumentFile(fileName)
444          else:
445              # Just to a "normal" locate - let the default handler get it.
446              return 1
447          return 0

464      def _AutoComplete(self):
465          self.SCIAutoCCancel()  # Cancel old auto-complete lists.
466          # First try and get an object without evaluating calls
467          ob = self._GetObjectAtPos(bAllowCalls=0)
468          # If that failed, try and process call or indexing to get the object.
469          if ob is None:
470              ob = self._GetObjectAtPos(bAllowCalls=1)
471          items_dict = {}
472          if ob is not None:
473              try:  # Catch unexpected errors when fetching attribute names from the object
474                  # extra attributes of win32ui objects
475                  if hasattr(ob, "_obj_"):
476                      try:
477                          items_dict.update(dict.fromkeys(dir(ob._obj_)))
478                      except AttributeError:
479                          pass  # object has no __dict__
480  
481                  # normal attributes
482                  try:
483                      items_dict.update(dict.fromkeys(dir(ob)))
484                  except AttributeError:
485                      pass  # object has no __dict__
486                  if hasattr(ob, "__class__"):
487                      items_dict.update(
488                          dict.fromkeys(_get_class_attributes(ob.__class__))
489                      )
490                  # The object may be a COM object with typelib support - let's see if we can get its props.
491                  # (contributed by Stefan Migowsky)
492                  try:
493                      # Get the automation attributes
494                      items_dict.update(ob.__class__._prop_map_get_)
495                      # See if there is an write only property
496                      # could be optimized
497                      items_dict.update(ob.__class__._prop_map_put_)
498                      # append to the already evaluated list
499                  except AttributeError:
500                      pass
501                  # The object might be a pure COM dynamic dispatch with typelib support - let's see if we can get its props.
502                  if hasattr(ob, "_oleobj_"):
503                      try:
504                          for iTI in range(0, ob._oleobj_.GetTypeInfoCount()):
505                              typeInfo = ob._oleobj_.GetTypeInfo(iTI)
506                              self._UpdateWithITypeInfo(items_dict, typeInfo)
507                      except:
508                          pass
509              except:
510                  win32ui.SetStatusText(
511                      f"Error attempting to get object attributes - {sys.exc_info()[0]!r}"
512                  )
513  
514          items = [
515              k
516              for k in
517              # ensure all keys are strings.
518              map(str, items_dict)
519              # All names that start with "_" go!
520              if not k.startswith("_")
521          ]
522  
523          if not items:
524              # Heuristics a-la AutoExpand
525              # The idea is to find other usages of the current binding
526              # and assume, that it refers to the same object (or at least,
527              # to an object of the same type)
528              # Contributed by Vadim Chugunov [vadimch@yahoo.com]
529              left, right = self._GetWordSplit()
530              if left == "":  # Ignore standalone dots
531                  return None
532              # We limit our search to the current class, if that
533              # information is available
534              minline, maxline, curclass = self._GetClassInfoFromBrowser()
535              endpos = self.LineIndex(maxline)
536              text = self.GetTextRange(self.LineIndex(minline), endpos)
537              try:
538                  l = re.findall(r"\b" + left + r"\.\w+", text)
539              except re.error:
540                  # parens etc may make an invalid RE, but this code wouldnt
541                  # benefit even if the RE did work :-)
542                  l = []
543              prefix = len(left) + 1
544              unique = {}
545              for li in l:
546                  unique[li[prefix:]] = 1
547              # Assuming traditional usage of self...
548              if curclass and left == "self":
549                  self._UpdateWithClassMethods(unique, curclass)
550  
551              items = [word for word in unique if word[:2] != "__" or word[-2:] != "__"]
552              # Ignore the word currently to the right of the dot - probably a red-herring.
553              try:
554                  items.remove(right[1:])
555              except ValueError:
556                  pass
557          if items:
558              items.sort()
559              self.SCIAutoCSetAutoHide(0)
560              self.SCIAutoCShow(items)

562      def _UpdateWithITypeInfo(self, items_dict, typeInfo):
563          import pythoncom
564  
565          typeInfos = [typeInfo]
566          # suppress IDispatch and IUnknown methods
567          inspectedIIDs = {pythoncom.IID_IDispatch: None}
568  
569          while len(typeInfos) > 0:
570              typeInfo = typeInfos.pop()
571              typeAttr = typeInfo.GetTypeAttr()
572  
573              if typeAttr.iid not in inspectedIIDs:
574                  inspectedIIDs[typeAttr.iid] = None
575                  for iFun in range(0, typeAttr.cFuncs):
576                      funDesc = typeInfo.GetFuncDesc(iFun)
577                      funName = typeInfo.GetNames(funDesc.memid)[0]
578                      if funName not in items_dict:
579                          items_dict[funName] = None
580  
581                  # Inspect the type info of all implemented types
582                  # E.g. IShellDispatch5 implements IShellDispatch4 which implements IShellDispatch3 ...
583                  for iImplType in range(0, typeAttr.cImplTypes):
584                      iRefType = typeInfo.GetRefTypeOfImplType(iImplType)
585                      refTypeInfo = typeInfo.GetRefTypeInfo(iRefType)
586                      typeInfos.append(refTypeInfo)

591      def _UpdateWithClassMethods(self, dict, classinfo):
592          if not hasattr(classinfo, "methods"):
593              # No 'methods' - probably not what we think it is.
594              return
595          dict.update(classinfo.methods)
596          for super in classinfo.super:
597              if hasattr(super, "methods"):
598                  self._UpdateWithClassMethods(dict, super)

600      # Find which class definition caret is currently in and return
601      # indexes of the the first and the last lines of that class definition
602      # Data is obtained from module browser (if enabled)
603      def _GetClassInfoFromBrowser(self, pos=-1):
604          minline = 0
605          maxline = self.GetLineCount() - 1
606          doc = self.GetParentFrame().GetActiveDocument()
607          browser = None
608          try:
609              if doc is not None:
610                  browser = doc.GetAllViews()[1]
611          except IndexError:
612              pass
613          if browser is None:
614              return (minline, maxline, None)  # Current window has no browser
615          if not browser.list:
616              return (minline, maxline, None)  # Not initialized
617          path = self.GetDocument().GetPathName()
618          if not path:
619              return (minline, maxline, None)  # No current path
620  
621          import pywin.framework.scriptutils
622  
623          curmodule, path = pywin.framework.scriptutils.GetPackageModuleName(path)
624          try:
625              clbrdata = browser.list.root.clbrdata
626          except AttributeError:
627              return (minline, maxline, None)  # No class data for this module.
628          curline = self.LineFromChar(pos)
629          curclass = None
630          # Find out which class we are in
631          for item in clbrdata.values():
632              if item.module == curmodule:
633                  item_lineno = (
634                      item.lineno - 1
635                  )  # Scintilla counts lines from 0, whereas pyclbr - from 1
636                  if minline < item_lineno <= curline:
637                      minline = item_lineno
638                      curclass = item
639                  if curline < item_lineno < maxline:
640                      maxline = item_lineno
641          return (minline, maxline, curclass)

643      def _GetObjectAtPos(self, pos=-1, bAllowCalls=0):
644          left, right = self._GetWordSplit(pos, bAllowCalls)
645          if left:  # It is an attribute lookup
646              # How is this for a hack!
647              namespace = sys.modules.copy()
648              namespace.update(__main__.__dict__)
649              # Get the debugger's context.
650              try:
651                  from pywin.framework import interact
652  
653                  if interact.edit is not None and interact.edit.currentView is not None:
654                      globs, locs = interact.edit.currentView.GetContext()[:2]
655                      if globs:
656                          namespace.update(globs)
657                      if locs:
658                          namespace.update(locs)
659              except ImportError:
660                  pass
661              try:
662                  return eval(left, namespace)
663              except:
664                  pass
665          return None

667      def _GetWordSplit(self, pos=-1, bAllowCalls=0):
668          if pos == -1:
669              pos = self.GetSel()[0] - 1  # Character before current one
670          limit = self.GetTextLength()
671          before = []
672          after = []
673          index = pos - 1
674          wordbreaks_use = wordbreaks
675          if bAllowCalls:
676              wordbreaks_use += "()[]"
677          while index >= 0:
678              char = self.SCIGetCharAt(index)
679              if char not in wordbreaks_use:
680                  break
681              before.insert(0, char)
682              index -= 1
683          index = pos
684          while index <= limit:
685              char = self.SCIGetCharAt(index)
686              if char not in wordbreaks_use:
687                  break
688              after.append(char)
689              index += 1
690          return "".join(before), "".join(after)

732      def CalculatePageRanges(self, dc, pInfo):
733          # Calculate page ranges and max page
734          self.starts = {0: 0}
735          metrics = dc.GetTextMetrics()
736          left, top, right, bottom = pInfo.GetDraw()
737          # Leave space at the top for the header.
738          rc = (left, top + int((9 * metrics["tmHeight"]) / 2), right, bottom)
739          pageStart = 0
740          maxPage = 0
741          textLen = self.GetTextLength()
742          while pageStart < textLen:
743              pageStart = self.FormatRange(dc, pageStart, textLen, rc, 0)
744              maxPage += 1
745              self.starts[maxPage] = pageStart
746          # And a sentinel for one page past the end
747          self.starts[maxPage + 1] = textLen
748          # When actually printing, maxPage doesn't have any effect at this late state.
749          # but is needed to make the Print Preview work correctly.
750          pInfo.SetMaxPage(maxPage)

758      def FormatRange(self, dc, pageStart, lengthDoc, rc, draw):
759          """ 
760          typedef struct _formatrange { 
761                  HDC hdc; 
762                  HDC hdcTarget; 
763                  RECT rc; 
764                  RECT rcPage; 
765                  CHARRANGE chrg;} FORMATRANGE; 
766          """
767          fmt = "PPIIIIIIIIll"
768          hdcRender = dc.GetHandleOutput()
769          hdcFormat = dc.GetHandleAttrib()
770          fr = struct.pack(
771              fmt,
772              hdcRender,
773              hdcFormat,
774              rc[0],
775              rc[1],
776              rc[2],
777              rc[3],
778              rc[0],
779              rc[1],
780              rc[2],
781              rc[3],
782              pageStart,
783              lengthDoc,
784          )
785          nextPageStart = self.SendScintilla(EM_FORMATRANGE, draw, fr)
786          return nextPageStart

788      def OnPrint(self, dc, pInfo):
789          metrics = dc.GetTextMetrics()
790          # print("dev", w, h, l, metrics["tmAscent"], metrics["tmDescent"])
791          if self.starts is None:
792              self.CalculatePageRanges(dc, pInfo)
793          pageNum = pInfo.GetCurPage() - 1
794          # Setup the header of the page - docname on left, pagenum on right.
795          doc = self.GetDocument()
796          cxChar = metrics["tmAveCharWidth"]
797          cyChar = metrics["tmHeight"]
798          left, top, right, bottom = pInfo.GetDraw()
799          dc.TextOut(0, 2 * cyChar, doc.GetTitle())
800          pagenum_str = win32ui.LoadString(afxres.AFX_IDS_PRINTPAGENUM) % (pageNum + 1,)
801          dc.SetTextAlign(win32con.TA_RIGHT)
802          dc.TextOut(right, 2 * cyChar, pagenum_str)
803          dc.SetTextAlign(win32con.TA_LEFT)
804          top += int(7 * cyChar / 2)
805          dc.MoveTo(left, top)
806          dc.LineTo(right, top)
807          top += cyChar
808          rc = (left, top, right, bottom)
809          nextPageStart = self.FormatRange(
810              dc, self.starts[pageNum], self.starts[pageNum + 1], rc, 1
811          )
