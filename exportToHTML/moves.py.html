27   @dataclass
28   class Jitter(Move):
29       """ 
30       Random displacement along one or both axes to reduce overplotting. 
31    
32       Parameters 
33       ---------- 
34       width : float 
35           Magnitude of jitter, relative to mark width, along the orientation axis. 
36           If not provided, the default value will be 0 when `x` or `y` are set, otherwise 
37           there will be a small amount of jitter applied by default. 
38       x : float 
39           Magnitude of jitter, in data units, along the x axis. 
40       y : float 
41           Magnitude of jitter, in data units, along the y axis. 
42    
43       Examples 
44       -------- 
45       .. include:: ../docstrings/objects.Jitter.rst 
46    
47       """
48       width: float | Default = default
49       x: float = 0
50       y: float = 0
51       seed: int | None = None
52   
53       def __call__(
54           self, data: DataFrame, groupby: GroupBy, orient: str, scales: dict[str, Scale],
55       ) -> DataFrame:
56   
57           data = data.copy()
58           rng = np.random.default_rng(self.seed)
59   
60           def jitter(data, col, scale):
61               noise = rng.uniform(-.5, +.5, len(data))
62               offsets = noise * scale
63               return data[col] + offsets
64   
65           if self.width is default:
66               width = 0.0 if self.x or self.y else 0.2
67           else:
68               width = cast(float, self.width)
69   
70           if self.width:
71               data[orient] = jitter(data, orient, width * data["width"])
72           if self.x:
73               data["x"] = jitter(data, "x", self.x)
74           if self.y:
75               data["y"] = jitter(data, "y", self.y)
76   
77           return data

80   @dataclass
81   class Dodge(Move):
82       """ 
83       Displacement and narrowing of overlapping marks along orientation axis. 
84    
85       Parameters 
86       ---------- 
87       empty : {'keep', 'drop', 'fill'} 
88       gap : float 
89           Size of gap between dodged marks. 
90       by : list of variable names 
91           Variables to apply the movement to, otherwise use all. 
92    
93       Examples 
94       -------- 
95       .. include:: ../docstrings/objects.Dodge.rst 
96    
97       """
98       empty: str = "keep"  # Options: keep, drop, fill
99       gap: float = 0
100  
101      # TODO accept just a str here?
102      # TODO should this always be present?
103      # TODO should the default be an "all" singleton?
104      by: Optional[list[str]] = None
105  
106      def __call__(
107          self, data: DataFrame, groupby: GroupBy, orient: str, scales: dict[str, Scale],
108      ) -> DataFrame:
109  
110          grouping_vars = [v for v in groupby.order if v in data]
111          groups = groupby.agg(data, {"width": "max"})
112          if self.empty == "fill":
113              groups = groups.dropna()
114  
115          def groupby_pos(s):
116              grouper = [groups[v] for v in [orient, "col", "row"] if v in data]
117              return s.groupby(grouper, sort=False, observed=True)
118  
119          def scale_widths(w):
120              # TODO what value to fill missing widths??? Hard problem...
121              # TODO short circuit this if outer widths has no variance?
122              empty = 0 if self.empty == "fill" else w.mean()
123              filled = w.fillna(empty)
124              scale = filled.max()
125              norm = filled.sum()
126              if self.empty == "keep":
127                  w = filled
128              return w / norm * scale
129  
130          def widths_to_offsets(w):
131              return w.shift(1).fillna(0).cumsum() + (w - w.sum()) / 2
132  
133          new_widths = groupby_pos(groups["width"]).transform(scale_widths)
134          offsets = groupby_pos(new_widths).transform(widths_to_offsets)
135  
136          if self.gap:
137              new_widths *= 1 - self.gap
138  
139          groups["_dodged"] = groups[orient] + offsets
140          groups["width"] = new_widths
141  
142          out = (
143              data
144              .drop("width", axis=1)
145              .merge(groups, on=grouping_vars, how="left")
146              .drop(orient, axis=1)
147              .rename(columns={"_dodged": orient})
148          )
149  
150          return out

153  @dataclass
154  class Stack(Move):
155      """ 
156      Displacement of overlapping bar or area marks along the value axis. 
157   
158      Examples 
159      -------- 
160      .. include:: ../docstrings/objects.Stack.rst 
161   
162      """
163      # TODO center? (or should this be a different move, eg. Stream())
164  
165      def _stack(self, df, orient):
166  
167          # TODO should stack do something with ymin/ymax style marks?
168          # Should there be an upstream conversion to baseline/height parameterization?
169  
170          if df["baseline"].nunique() > 1:
171              err = "Stack move cannot be used when baselines are already heterogeneous"
172              raise RuntimeError(err)
173  
174          other = {"x": "y", "y": "x"}[orient]
175          stacked_lengths = (df[other] - df["baseline"]).dropna().cumsum()
176          offsets = stacked_lengths.shift(1).fillna(0)
177  
178          df[other] = stacked_lengths
179          df["baseline"] = df["baseline"] + offsets
180  
181          return df
182  
183      def __call__(
184          self, data: DataFrame, groupby: GroupBy, orient: str, scales: dict[str, Scale],
185      ) -> DataFrame:
186  
187          # TODO where to ensure that other semantic variables are sorted properly?
188          # TODO why are we not using the passed in groupby here?
189          groupers = ["col", "row", orient]
190          return GroupBy(groupers).apply(data, self._stack, orient)

194  class Shift(Move):
195      """ 
196      Displacement of all marks with the same magnitude / direction. 
197   
198      Parameters 
199      ---------- 
200      x, y : float 
201          Magnitude of shift, in data units, along each axis. 
202   
203      Examples 
204      -------- 
205      .. include:: ../docstrings/objects.Shift.rst 
206   
207      """
208      x: float = 0
209      y: float = 0
210  
211      def __call__(
212          self, data: DataFrame, groupby: GroupBy, orient: str, scales: dict[str, Scale],
213      ) -> DataFrame:
214  
215          data = data.copy(deep=False)
216          data["x"] = data["x"] + self.x
217          data["y"] = data["y"] + self.y
218          return data

221  @dataclass
222  class Norm(Move):
223      """ 
224      Divisive scaling on the value axis after aggregating within groups. 
225   
226      Parameters 
227      ---------- 
228      func : str or callable 
229          Function called on each group to define the comparison value. 
230      where : str 
231          Query string defining the subset used to define the comparison values. 
232      by : list of variables 
233          Variables used to define aggregation groups. 
234      percent : bool 
235          If True, multiply the result by 100. 
236   
237      Examples 
238      -------- 
239      .. include:: ../docstrings/objects.Norm.rst 
240   
241      """
242  
243      func: Union[Callable, str] = "max"
244      where: Optional[str] = None
245      by: Optional[list[str]] = None
246      percent: bool = False
247  
248      group_by_orient: ClassVar[bool] = False
249  
250      def _norm(self, df, var):
251  
252          if self.where is None:
253              denom_data = df[var]
254          else:
255              denom_data = df.query(self.where)[var]
256          df[var] = df[var] / denom_data.agg(self.func)
257  
258          if self.percent:
259              df[var] = df[var] * 100
260  
261          return df
262  
263      def __call__(
264          self, data: DataFrame, groupby: GroupBy, orient: str, scales: dict[str, Scale],
265      ) -> DataFrame:
266  
267          other = {"x": "y", "y": "x"}[orient]
268          return groupby.apply(data, self._norm, other)
