1    from functools import lru_cache

5    @lru_cache(maxsize=None)
6    def camel_to_snake(s: str) -> str:
7        """ 
8        Convert CamelCase to snake_case. 
9        """
10       return "".join("_" + c.lower() if c.isupper() else c for c in s).lstrip("_")

41   pre_pep8_method_name = pp.one_of(pre_pep8_method_names, as_keyword=True)
42   pre_pep8_method_name.set_parse_action(lambda t: camel_to_snake(t[0]))
43   special_pre_pep8_name = pp.one_of(special_changes, as_keyword=True)
44   special_pre_pep8_name.set_parse_action(lambda t: special_changes[t[0]])
45   # only replace arg names if part of an arg list
46   pre_pep8_arg_name = pp.Regex(
47       rf"{pp.util.make_compressed_re(pre_pep8_arg_names)}\s*="
48   )
49   pre_pep8_arg_name.set_parse_action(lambda t: camel_to_snake(t[0]))

51   pep8_converter = pre_pep8_method_name | special_pre_pep8_name | pre_pep8_arg_name

53   if __name__ == "__main__":
54       import argparse
55       from pathlib import Path
56       import sys
57   
58       argparser = argparse.ArgumentParser(
59           description = (
60               "Utility to convert Python pyparsing scripts using legacy"
61               " camelCase names to use PEP8 snake_case names."
62               "\nBy default, this script will only show whether this script would make any changes."
63           )
64       )
65       argparser.add_argument("--verbose", "-v", action="store_true", help="Show unified diff for each source file")
66       argparser.add_argument("-vv", action="store_true", dest="verbose2", help="Show unified diff for each source file, plus names of scanned files with no changes")
67       argparser.add_argument("--update", "-u", action="store_true", help="Update source files in-place")
68       argparser.add_argument("--encoding", type=str, default="utf-8", help="Encoding of source files (default: utf-8)")
69       argparser.add_argument("--exit-zero-even-if-changed", "-exit0", action="store_true", help="Exit with status code 0 even if changes were made")
70       argparser.add_argument("source_filename", nargs="+", help="Source filenames or filename patterns of Python files to be converted")
71       args = argparser.parse_args()
72   
73   
74       def show_diffs(original, modified):
75           import difflib
76   
77           diff = difflib.unified_diff(
78               original.splitlines(), modified.splitlines(), lineterm=""
79           )
80           sys.stdout.writelines(f"{diff_line}\n" for diff_line in diff)
81   
82       exit_status = 0
83   
84       for filename_pattern in args.source_filename:
85   
86           for filename in Path().glob(filename_pattern):
87               if not Path(filename).is_file():
88                   continue
89   
90               try:
91                   original_contents = Path(filename).read_text(encoding=args.encoding)
92                   modified_contents = pep8_converter.transform_string(
93                       original_contents
94                   )
95   
96                   if modified_contents != original_contents:
97                       if args.update:
98                           Path(filename).write_text(modified_contents, encoding=args.encoding)
99                           print(f"Converted {filename}")
100                      else:
101                          print(f"Found required changes in {filename}")
102  
103                      if args.verbose:
104                          show_diffs(original_contents, modified_contents)
105                          print()
106  
107                      exit_status = 1
108  
109                  else:
110                      if args.verbose2:
111                          print(f"No required changes in {filename}")
112  
113              except Exception as e:
114                  print(f"Failed to convert {filename}: {type(e).__name__}: {e}")
115  
116      sys.exit(exit_status if not args.exit_zero_even_if_changed else 0)
