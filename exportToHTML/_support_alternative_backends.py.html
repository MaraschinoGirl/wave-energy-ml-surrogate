22   def get_array_special_func(f_name, xp, n_array_args):
23       spx = scipy_namespace_for(xp)
24       f = None
25       if is_numpy(xp):
26           f = getattr(_ufuncs, f_name, None)
27       elif spx is not None:
28           f = getattr(spx.special, f_name, None)
29   
30       if f is not None:
31           return f
32   
33       # if generic array-API implementation is available, use that;
34       # otherwise, fall back to NumPy/SciPy
35       if f_name in _generic_implementations:
36           _f = _generic_implementations[f_name](xp=xp, spx=spx)
37           if _f is not None:
38               return _f
39   
40       _f = getattr(_ufuncs, f_name, None)
41       def __f(*args, _f=_f, _xp=xp, **kwargs):
42           array_args = args[:n_array_args]
43           other_args = args[n_array_args:]
44           array_args = [np.asarray(arg) for arg in array_args]
45           out = _f(*array_args, *other_args, **kwargs)
46           return _xp.asarray(out)
47   
48       return __f

51   def _get_shape_dtype(*args, xp):
52       args = xp.broadcast_arrays(*args)
53       shape = args[0].shape
54       dtype = xp.result_type(*args)
55       if xp.isdtype(dtype, 'integral'):
56           dtype = xp.float64
57           args = [xp.asarray(arg, dtype=dtype) for arg in args]
58       return args, shape, dtype

61   def _rel_entr(xp, spx):
62       def __rel_entr(x, y, *, xp=xp):
63           args, shape, dtype = _get_shape_dtype(x, y, xp=xp)
64           x, y = args
65           res = xp.full(x.shape, xp.inf, dtype=dtype)
66           res[(x == 0) & (y >= 0)] = xp.asarray(0, dtype=dtype)
67           i = (x > 0) & (y > 0)
68           res[i] = x[i] * (xp.log(x[i]) - xp.log(y[i]))
69           return res
70       return __rel_entr

73   def _xlogy(xp, spx):
74       def __xlogy(x, y, *, xp=xp):
75           with np.errstate(divide='ignore', invalid='ignore'):
76               temp = x * xp.log(y)
77           return xp.where(x == 0., xp.asarray(0., dtype=temp.dtype), temp)
78       return __xlogy

81   def _chdtr(xp, spx):
82       # The difference between this and just using `gammainc`
83       # defined by `get_array_special_func` is that if `gammainc`
84       # isn't found, we don't want to use the SciPy version; we'll
85       # return None here and use the SciPy version of `chdtr`.
86       gammainc = getattr(spx.special, 'gammainc', None) if spx else None  # noqa: F811
87       if gammainc is None and hasattr(xp, 'special'):
88           gammainc = getattr(xp.special, 'gammainc', None)
89       if gammainc is None:
90           return None
91   
92       def __chdtr(v, x):
93           res = gammainc(v / 2, x / 2)  # this is almost all we need
94           # The rest can be removed when google/jax#20507 is resolved
95           mask = (v == 0) & (x > 0)  # JAX returns NaN
96           res = xp.where(mask, 1., res)
97           mask = xp.isinf(v) & xp.isinf(x)  # JAX returns 1.0
98           return xp.where(mask, xp.nan, res)
99       return __chdtr

102  def _chdtrc(xp, spx):
103      # The difference between this and just using `gammaincc`
104      # defined by `get_array_special_func` is that if `gammaincc`
105      # isn't found, we don't want to use the SciPy version; we'll
106      # return None here and use the SciPy version of `chdtrc`.
107      gammaincc = getattr(spx.special, 'gammaincc', None) if spx else None  # noqa: F811
108      if gammaincc is None and hasattr(xp, 'special'):
109          gammaincc = getattr(xp.special, 'gammaincc', None)
110      if gammaincc is None:
111          return None
112  
113      def __chdtrc(v, x):
114          res = xp.where(x >= 0, gammaincc(v/2, x/2), 1)
115          i_nan = ((x == 0) & (v == 0)) | xp.isnan(x) | xp.isnan(v) | (v <= 0)
116          res = xp.where(i_nan, xp.nan, res)
117          return res
118      return __chdtrc

121  def _betaincc(xp, spx):
122      betainc = getattr(spx.special, 'betainc', None) if spx else None  # noqa: F811
123      if betainc is None and hasattr(xp, 'special'):
124          betainc = getattr(xp.special, 'betainc', None)
125      if betainc is None:
126          return None
127  
128      def __betaincc(a, b, x):
129          # not perfect; might want to just rely on SciPy
130          return betainc(b, a, 1-x)
131      return __betaincc

134  def _stdtr(xp, spx):
135      betainc = getattr(spx.special, 'betainc', None) if spx else None  # noqa: F811
136      if betainc is None and hasattr(xp, 'special'):
137          betainc = getattr(xp.special, 'betainc', None)
138      if betainc is None:
139          return None
140  
141      def __stdtr(df, t):
142          x = df / (t ** 2 + df)
143          tail = betainc(df / 2, 0.5, x) / 2
144          return xp.where(t < 0, tail, 1 - tail)
145  
146      return __stdtr

158  # functools.wraps doesn't work because:
159  # 'numpy.ufunc' object has no attribute '__module__'
160  def support_alternative_backends(f_name, n_array_args):
161      func = getattr(_ufuncs, f_name)
162  
163      @functools.wraps(func)
164      def wrapped(*args, **kwargs):
165          xp = array_namespace(*args[:n_array_args])
166          f = get_array_special_func(f_name, xp, n_array_args)
167          return f(*args, **kwargs)
168  
169      return wrapped
