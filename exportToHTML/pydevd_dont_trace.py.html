1    """ 
2    Support for a tag that allows skipping over functions while debugging. 
3    """
4    import linecache
5    import re
6    
7    # To suppress tracing a method, add the tag @DontTrace
8    # to a comment either preceding or on the same line as
9    # the method definition
10   #
11   # E.g.:
12   # #@DontTrace
13   # def test1():
14   #     pass
15   #
16   #  ... or ...
17   #
18   # def test2(): #@DontTrace
19   #     pass
20   DONT_TRACE_TAG = "@DontTrace"
21   
22   # Regular expression to match a decorator (at the beginning
23   # of a line).
24   RE_DECORATOR = re.compile(r"^\s*@")
25   
26   # Mapping from code object to bool.
27   # If the key exists, the value is the cached result of should_trace_hook
28   _filename_to_ignored_lines = {}
29   
30   
31   def default_should_trace_hook(code, absolute_filename):
32       """ 
33       Return True if this frame should be traced, False if tracing should be blocked. 
34       """
35       # First, check whether this code object has a cached value
36       ignored_lines = _filename_to_ignored_lines.get(absolute_filename)
37       if ignored_lines is None:
38           # Now, look up that line of code and check for a @DontTrace
39           # preceding or on the same line as the method.
40           # E.g.:
41           # #@DontTrace
42           # def test():
43           #     pass
44           #  ... or ...
45           # def test(): #@DontTrace
46           #     pass
47           ignored_lines = {}
48           lines = linecache.getlines(absolute_filename)
49           for i_line, line in enumerate(lines):
50               j = line.find("#")
51               if j >= 0:
52                   comment = line[j:]
53                   if DONT_TRACE_TAG in comment:
54                       ignored_lines[i_line] = 1
55   
56                       # Note: when it's found in the comment, mark it up and down for the decorator lines found.
57                       k = i_line - 1
58                       while k >= 0:
59                           if RE_DECORATOR.match(lines[k]):
60                               ignored_lines[k] = 1
61                               k -= 1
62                           else:
63                               break
64   
65                       k = i_line + 1
66                       while k <= len(lines):
67                           if RE_DECORATOR.match(lines[k]):
68                               ignored_lines[k] = 1
69                               k += 1
70                           else:
71                               break
72   
73           _filename_to_ignored_lines[absolute_filename] = ignored_lines
74   
75       func_line = code.co_firstlineno - 1  # co_firstlineno is 1-based, so -1 is needed
76       return not (
77           func_line - 1 in ignored_lines  # -1 to get line before method
78           or func_line in ignored_lines
79       )  # method line
80   
81   
82   should_trace_hook = None
83   
84   
85   def clear_trace_filter_cache():
86       """ 
87       Clear the trace filter cache. 
88       Call this after reloading. 
89       """
90       global should_trace_hook
91       try:
92           # Need to temporarily disable a hook because otherwise
93           # _filename_to_ignored_lines.clear() will never complete.
94           old_hook = should_trace_hook
95           should_trace_hook = None
96   
97           # Clear the linecache
98           linecache.clearcache()
99           _filename_to_ignored_lines.clear()
100  
101      finally:
102          should_trace_hook = old_hook
103  
104  
105  def trace_filter(mode):
106      """ 
107      Set the trace filter mode. 
108   
109      mode: Whether to enable the trace hook. 
110        True: Trace filtering on (skipping methods tagged @DontTrace) 
111        False: Trace filtering off (trace methods tagged @DontTrace) 
112        None/default: Toggle trace filtering. 
113      """
114      global should_trace_hook
115      if mode is None:
116          mode = should_trace_hook is None
117  
118      if mode:
119          should_trace_hook = default_should_trace_hook
120      else:
121          should_trace_hook = None
122  
123      return mode
