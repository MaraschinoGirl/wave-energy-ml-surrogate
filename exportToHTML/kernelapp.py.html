223      def _try_bind_socket(self, s, port):
224          iface = f"{self.transport}://{self.ip}"
225          if self.transport == "tcp":
226              if port <= 0:
227                  port = s.bind_to_random_port(iface)
228              else:
229                  s.bind("tcp://%s:%i" % (self.ip, port))
230          elif self.transport == "ipc":
231              if port <= 0:
232                  port = 1
233                  path = "%s-%i" % (self.ip, port)
234                  while Path(path).exists():
235                      port = port + 1
236                      path = "%s-%i" % (self.ip, port)
237              else:
238                  path = "%s-%i" % (self.ip, port)
239              s.bind("ipc://%s" % path)
240          return port
241  
242      def _bind_socket(self, s, port):
243          try:
244              win_in_use = errno.WSAEADDRINUSE  # type:ignore[attr-defined]
245          except AttributeError:
246              win_in_use = None
247  
248          # Try up to 100 times to bind a port when in conflict to avoid
249          # infinite attempts in bad setups
250          max_attempts = 1 if port else 100
251          for attempt in range(max_attempts):
252              try:
253                  return self._try_bind_socket(s, port)
254              except zmq.ZMQError as ze:
255                  # Raise if we have any error not related to socket binding
256                  if ze.errno != errno.EADDRINUSE and ze.errno != win_in_use:
257                      raise
258                  if attempt == max_attempts - 1:
259                      raise
260          return None

262      def write_connection_file(self):
263          """write connection info to JSON file"""
264          cf = self.abs_connection_file
265          connection_info = dict(
266              ip=self.ip,
267              key=self.session.key,
268              transport=self.transport,
269              shell_port=self.shell_port,
270              stdin_port=self.stdin_port,
271              hb_port=self.hb_port,
272              iopub_port=self.iopub_port,
273              control_port=self.control_port,
274          )
275          if Path(cf).exists():
276              # If the file exists, merge our info into it. For example, if the
277              # original file had port number 0, we update with the actual port
278              # used.
279              existing_connection_info = get_connection_info(cf, unpack=True)
280              assert isinstance(existing_connection_info, dict)
281              connection_info = dict(existing_connection_info, **connection_info)
282              if connection_info == existing_connection_info:
283                  self.log.debug("Connection file %s with current information already exists", cf)
284                  return
285  
286          self.log.debug("Writing connection file: %s", cf)
287  
288          write_connection_file(cf, **connection_info)

301      def init_connection_file(self):
302          """Initialize our connection file."""
303          if not self.connection_file:
304              self.connection_file = "kernel-%s.json" % os.getpid()
305          try:
306              self.connection_file = filefind(self.connection_file, [".", self.connection_dir])
307          except OSError:
308              self.log.debug("Connection file not found: %s", self.connection_file)
309              # This means I own it, and I'll create it in this directory:
310              Path(self.abs_connection_file).parent.mkdir(mode=0o700, exist_ok=True, parents=True)
311              # Also, I will clean it up:
312              atexit.register(self.cleanup_connection_file)
313              return
314          try:
315              self.load_connection_file()
316          except Exception:
317              self.log.error(  # noqa: G201
318                  "Failed to load connection file: %r", self.connection_file, exc_info=True
319              )
320              self.exit(1)

322      def init_sockets(self):
323          """Create a context, a session, and the kernel sockets."""
324          self.log.info("Starting the kernel at pid: %i", os.getpid())
325          assert self.context is None, "init_sockets cannot be called twice!"
326          self.context = context = zmq.Context()
327          atexit.register(self.close)
328  
329          self.shell_socket = context.socket(zmq.ROUTER)
330          self.shell_socket.linger = 1000
331          self.shell_port = self._bind_socket(self.shell_socket, self.shell_port)
332          self.log.debug("shell ROUTER Channel on port: %i" % self.shell_port)
333  
334          self.stdin_socket = context.socket(zmq.ROUTER)
335          self.stdin_socket.linger = 1000
336          self.stdin_port = self._bind_socket(self.stdin_socket, self.stdin_port)
337          self.log.debug("stdin ROUTER Channel on port: %i" % self.stdin_port)
338  
339          if hasattr(zmq, "ROUTER_HANDOVER"):
340              # set router-handover to workaround zeromq reconnect problems
341              # in certain rare circumstances
342              # see ipython/ipykernel#270 and zeromq/libzmq#2892
343              self.shell_socket.router_handover = self.stdin_socket.router_handover = 1
344  
345          self.init_control(context)
346          self.init_iopub(context)
347  
348      def init_control(self, context):
349          """Initialize the control channel."""
350          self.control_socket = context.socket(zmq.ROUTER)
351          self.control_socket.linger = 1000
352          self.control_port = self._bind_socket(self.control_socket, self.control_port)
353          self.log.debug("control ROUTER Channel on port: %i" % self.control_port)
354  
355          self.debugpy_socket = context.socket(zmq.STREAM)
356          self.debugpy_socket.linger = 1000
357  
358          self.debug_shell_socket = context.socket(zmq.DEALER)
359          self.debug_shell_socket.linger = 1000
360          if self.shell_socket.getsockopt(zmq.LAST_ENDPOINT):
361              self.debug_shell_socket.connect(self.shell_socket.getsockopt(zmq.LAST_ENDPOINT))
362  
363          if hasattr(zmq, "ROUTER_HANDOVER"):
364              # set router-handover to workaround zeromq reconnect problems
365              # in certain rare circumstances
366              # see ipython/ipykernel#270 and zeromq/libzmq#2892
367              self.control_socket.router_handover = 1
368  
369          self.control_thread = ControlThread(daemon=True)
370  
371      def init_iopub(self, context):
372          """Initialize the iopub channel."""
373          self.iopub_socket = context.socket(zmq.PUB)
374          self.iopub_socket.linger = 1000
375          self.iopub_port = self._bind_socket(self.iopub_socket, self.iopub_port)
376          self.log.debug("iopub PUB Channel on port: %i" % self.iopub_port)
377          self.configure_tornado_logger()
378          self.iopub_thread = IOPubThread(self.iopub_socket, pipe=True)
379          self.iopub_thread.start()
380          # backward-compat: wrap iopub socket API in background thread
381          self.iopub_socket = self.iopub_thread.background_socket

383      def init_heartbeat(self):
384          """start the heart beating"""
385          # heartbeat doesn't share context, because it mustn't be blocked
386          # by the GIL, which is accessed by libzmq when freeing zero-copy messages
387          hb_ctx = zmq.Context()
388          self.heartbeat = Heartbeat(hb_ctx, (self.transport, self.ip, self.hb_port))
389          self.hb_port = self.heartbeat.port
390          self.log.debug("Heartbeat REP Channel on port: %i" % self.hb_port)
391          self.heartbeat.start()

393      def close(self):
394          """Close zmq sockets in an orderly fashion"""
395          # un-capture IO before we start closing channels
396          self.reset_io()
397          self.log.info("Cleaning up sockets")
398          if self.heartbeat:
399              self.log.debug("Closing heartbeat channel")
400              self.heartbeat.context.term()
401          if self.iopub_thread:
402              self.log.debug("Closing iopub channel")
403              self.iopub_thread.stop()
404              self.iopub_thread.close()
405          if self.control_thread and self.control_thread.is_alive():
406              self.log.debug("Closing control thread")
407              self.control_thread.stop()
408              self.control_thread.join()
409  
410          if self.debugpy_socket and not self.debugpy_socket.closed:
411              self.debugpy_socket.close()
412          if self.debug_shell_socket and not self.debug_shell_socket.closed:
413              self.debug_shell_socket.close()
414  
415          for channel in ("shell", "control", "stdin"):
416              self.log.debug("Closing %s channel", channel)
417              socket = getattr(self, channel + "_socket", None)
418              if socket and not socket.closed:
419                  socket.close()
420          self.log.debug("Terminating zmq context")
421          if self.context:
422              self.context.term()
423          self.log.debug("Terminated zmq context")

425      def log_connection_info(self):
426          """display connection info, and store ports"""
427          basename = Path(self.connection_file).name
428          if (
429              basename == self.connection_file
430              or str(Path(self.connection_file).parent) == self.connection_dir
431          ):
432              # use shortname
433              tail = basename
434          else:
435              tail = self.connection_file
436          lines = [
437              "To connect another client to this kernel, use:",
438              "    --existing %s" % tail,
439          ]
440          # log connection info
441          # info-level, so often not shown.
442          # frontends should use the %connect_info magic
443          # to see the connection info
444          for line in lines:
445              self.log.info(line)
446          # also raw print to the terminal if no parent_handle (`ipython kernel`)
447          # unless log-level is CRITICAL (--quiet)
448          if not self.parent_handle and int(self.log_level) < logging.CRITICAL:  # type:ignore[call-overload]
449              print(_ctrl_c_message, file=sys.__stdout__)
450              for line in lines:
451                  print(line, file=sys.__stdout__)
452  
453          self._ports = dict(
454              shell=self.shell_port,
455              iopub=self.iopub_port,
456              stdin=self.stdin_port,
457              hb=self.hb_port,
458              control=self.control_port,
459          )

461      def init_blackhole(self):
462          """redirects stdout/stderr to devnull if necessary"""
463          if self.no_stdout or self.no_stderr:
464              blackhole = open(os.devnull, "w")  # noqa: SIM115
465              if self.no_stdout:
466                  sys.stdout = sys.__stdout__ = blackhole  # type:ignore[misc]
467              if self.no_stderr:
468                  sys.stderr = sys.__stderr__ = blackhole  # type:ignore[misc]

470      def init_io(self):
471          """Redirect input streams and set a display hook."""
472          if self.outstream_class:
473              outstream_factory = import_item(str(self.outstream_class))
474              if sys.stdout is not None:
475                  sys.stdout.flush()
476  
477              e_stdout = None if self.quiet else sys.__stdout__
478              e_stderr = None if self.quiet else sys.__stderr__
479  
480              if not self.capture_fd_output:
481                  outstream_factory = partial(outstream_factory, watchfd=False)
482  
483              sys.stdout = outstream_factory(self.session, self.iopub_thread, "stdout", echo=e_stdout)
484              if sys.stderr is not None:
485                  sys.stderr.flush()
486              sys.stderr = outstream_factory(self.session, self.iopub_thread, "stderr", echo=e_stderr)
487              if hasattr(sys.stderr, "_original_stdstream_copy"):
488                  for handler in self.log.handlers:
489                      if isinstance(handler, StreamHandler) and (handler.stream.buffer.fileno() == 2):
490                          self.log.debug("Seeing logger to stderr, rerouting to raw filedescriptor.")
491  
492                          handler.stream = TextIOWrapper(
493                              FileIO(
494                                  sys.stderr._original_stdstream_copy,
495                                  "w",
496                              )
497                          )
498          if self.displayhook_class:
499              displayhook_factory = import_item(str(self.displayhook_class))
500              self.displayhook = displayhook_factory(self.session, self.iopub_socket)
501              sys.displayhook = self.displayhook
502  
503          self.patch_io()
504  
505      def reset_io(self):
506          """restore original io 
507   
508          restores state after init_io 
509          """
510          sys.stdout = sys.__stdout__
511          sys.stderr = sys.__stderr__
512          sys.displayhook = sys.__displayhook__

514      def patch_io(self):
515          """Patch important libraries that can't handle sys.stdout forwarding"""
516          try:
517              import faulthandler
518          except ImportError:
519              pass
520          else:
521              # Warning: this is a monkeypatch of `faulthandler.enable`, watch for possible
522              # updates to the upstream API and update accordingly (up-to-date as of Python 3.5):
523              # https://docs.python.org/3/library/faulthandler.html#faulthandler.enable
524  
525              # change default file to __stderr__ from forwarded stderr
526              faulthandler_enable = faulthandler.enable
527  
528              def enable(file=sys.__stderr__, all_threads=True, **kwargs):
529                  return faulthandler_enable(file=file, all_threads=all_threads, **kwargs)
530  
531              faulthandler.enable = enable
532  
533              if hasattr(faulthandler, "register"):
534                  faulthandler_register = faulthandler.register
535  
536                  def register(signum, file=sys.__stderr__, all_threads=True, chain=False, **kwargs):
537                      return faulthandler_register(
538                          signum, file=file, all_threads=all_threads, chain=chain, **kwargs
539                      )
540  
541                  faulthandler.register = register

547      def init_kernel(self):
548          """Create the Kernel object itself"""
549          shell_stream = ZMQStream(self.shell_socket)
550          control_stream = ZMQStream(self.control_socket, self.control_thread.io_loop)
551          debugpy_stream = ZMQStream(self.debugpy_socket, self.control_thread.io_loop)
552          self.control_thread.start()
553          kernel_factory = self.kernel_class.instance  # type:ignore[attr-defined]
554  
555          kernel = kernel_factory(
556              parent=self,
557              session=self.session,
558              control_stream=control_stream,
559              debugpy_stream=debugpy_stream,
560              debug_shell_socket=self.debug_shell_socket,
561              shell_stream=shell_stream,
562              control_thread=self.control_thread,
563              iopub_thread=self.iopub_thread,
564              iopub_socket=self.iopub_socket,
565              stdin_socket=self.stdin_socket,
566              log=self.log,
567              profile_dir=self.profile_dir,
568              user_ns=self.user_ns,
569          )
570          kernel.record_ports({name + "_port": port for name, port in self._ports.items()})
571          self.kernel = kernel
572  
573          # Allow the displayhook to get the execution count
574          self.displayhook.get_execution_count = lambda: kernel.execution_count

576      def init_gui_pylab(self):
577          """Enable GUI event loop integration, taking pylab into account."""
578  
579          # Register inline backend as default
580          # this is higher priority than matplotlibrc,
581          # but lower priority than anything else (mpl.use() for instance).
582          # This only affects matplotlib >= 1.5
583          if not os.environ.get("MPLBACKEND"):
584              os.environ["MPLBACKEND"] = "module://matplotlib_inline.backend_inline"
585  
586          # Provide a wrapper for :meth:`InteractiveShellApp.init_gui_pylab`
587          # to ensure that any exception is printed straight to stderr.
588          # Normally _showtraceback associates the reply with an execution,
589          # which means frontends will never draw it, as this exception
590          # is not associated with any execute request.
591  
592          shell = self.shell
593          assert shell is not None
594          _showtraceback = shell._showtraceback
595          try:
596              # replace error-sending traceback with stderr
597              def print_tb(etype, evalue, stb):
598                  print("GUI event loop or pylab initialization failed", file=sys.stderr)
599                  assert shell is not None
600                  print(shell.InteractiveTB.stb2text(stb), file=sys.stderr)
601  
602              shell._showtraceback = print_tb
603              InteractiveShellApp.init_gui_pylab(self)
604          finally:
605              shell._showtraceback = _showtraceback

613      def configure_tornado_logger(self):
614          """Configure the tornado logging.Logger. 
615   
616          Must set up the tornado logger or else tornado will call 
617          basicConfig for the root logger which makes the root logger 
618          go to the real sys.stderr instead of the capture streams. 
619          This function mimics the setup of logging.basicConfig. 
620          """
621          logger = logging.getLogger("tornado")
622          handler = logging.StreamHandler()
623          formatter = logging.Formatter(logging.BASIC_FORMAT)
624          handler.setFormatter(formatter)
625          logger.addHandler(handler)
626  
627      def _init_asyncio_patch(self):
628          """set default asyncio policy to be compatible with tornado 
629   
630          Tornado 6 (at least) is not compatible with the default 
631          asyncio implementation on Windows 
632   
633          Pick the older SelectorEventLoopPolicy on Windows 
634          if the known-incompatible default policy is in use. 
635   
636          Support for Proactor via a background thread is available in tornado 6.1, 
637          but it is still preferable to run the Selector in the main thread 
638          instead of the background. 
639   
640          do this as early as possible to make it a low priority and overridable 
641   
642          ref: https://github.com/tornadoweb/tornado/issues/2608 
643   
644          FIXME: if/when tornado supports the defaults in asyncio without threads, 
645                 remove and bump tornado requirement for py38. 
646                 Most likely, this will mean a new Python version 
647                 where asyncio.ProactorEventLoop supports add_reader and friends. 
648   
649          """
650          if sys.platform.startswith("win") and sys.version_info >= (3, 8):
651              import asyncio
652  
653              try:
654                  from asyncio import WindowsProactorEventLoopPolicy, WindowsSelectorEventLoopPolicy
655              except ImportError:
656                  pass
657                  # not affected
658              else:
659                  if type(asyncio.get_event_loop_policy()) is WindowsProactorEventLoopPolicy:
660                      # WindowsProactorEventLoopPolicy is not compatible with tornado 6
661                      # fallback to the pre-3.8 default of Selector
662                      asyncio.set_event_loop_policy(WindowsSelectorEventLoopPolicy())
663  
664      def init_pdb(self):
665          """Replace pdb with IPython's version that is interruptible. 
666   
667          With the non-interruptible version, stopping pdb() locks up the kernel in a 
668          non-recoverable state. 
669          """
670          import pdb
671  
672          from IPython.core import debugger
673  
674          if hasattr(debugger, "InterruptiblePdb"):
675              # Only available in newer IPython releases:
676              debugger.Pdb = debugger.InterruptiblePdb  # type:ignore[misc]
677              pdb.Pdb = debugger.Pdb  # type:ignore[assignment,misc]
678              pdb.set_trace = debugger.set_trace
679  
680      @catch_config_error
681      def initialize(self, argv=None):
682          """Initialize the application."""
683          self._init_asyncio_patch()
684          super().initialize(argv)
685          if self.subapp is not None:
686              return
687  
688          self.init_pdb()
689          self.init_blackhole()
690          self.init_connection_file()
691          self.init_poller()
692          self.init_sockets()
693          self.init_heartbeat()
694          # writing/displaying connection info must be *after* init_sockets/heartbeat
695          self.write_connection_file()
696          # Log connection info after writing connection file, so that the connection
697          # file is definitely available at the time someone reads the log.
698          self.log_connection_info()
699          self.init_io()
700          try:
701              self.init_signal()
702          except Exception:
703              # Catch exception when initializing signal fails, eg when running the
704              # kernel on a separate thread
705              if int(self.log_level) < logging.CRITICAL:  # type:ignore[call-overload]
706                  self.log.error("Unable to initialize signal:", exc_info=True)  # noqa: G201
707          self.init_kernel()
708          # shell init steps
709          self.init_path()
710          self.init_shell()
711          if self.shell:
712              self.init_gui_pylab()
713              self.init_extensions()
714              self.init_code()
715          # flush stdout/stderr, so that anything written to these streams during
716          # initialization do not get associated with the first execution request
717          sys.stdout.flush()
718          sys.stderr.flush()
719  
720      def start(self):
721          """Start the application."""
722          if self.subapp is not None:
723              return self.subapp.start()
724          if self.poller is not None:
725              self.poller.start()
726          self.kernel.start()
727          self.io_loop = ioloop.IOLoop.current()
728          if self.trio_loop:
729              from ipykernel.trio_runner import TrioRunner
730  
731              tr = TrioRunner()
732              tr.initialize(self.kernel, self.io_loop)
733              try:
734                  tr.run()
735              except KeyboardInterrupt:
736                  pass
737          else:
738              try:
739                  self.io_loop.start()
740              except KeyboardInterrupt:
741                  pass

748      """Run an IPKernel as an application"""
749      app = IPKernelApp.instance()

751      app.start()
