8    @keras_export("keras.metrics.Hinge")
9    class Hinge(reduction_metrics.MeanMetricWrapper):
10       """Computes the hinge metric between `y_true` and `y_pred`. 
11    
12       `y_true` values are expected to be -1 or 1. If binary (0 or 1) labels are 
13       provided we will convert them to -1 or 1. 
14    
15       Args: 
16           name: (Optional) string name of the metric instance. 
17           dtype: (Optional) data type of the metric result. 
18    
19       Examples: 
20    
21       >>> m = keras.metrics.Hinge() 
22       >>> m.update_state([[0, 1], [0, 0]], [[0.6, 0.4], [0.4, 0.6]]) 
23       >>> m.result() 
24       1.3 
25       >>> m.reset_state() 
26       >>> m.update_state([[0, 1], [0, 0]], [[0.6, 0.4], [0.4, 0.6]], 
27       ...                sample_weight=[1, 0]) 
28       >>> m.result() 
29       1.1 
30       """
31   
32       def __init__(self, name="hinge", dtype=None):
33           super().__init__(fn=hinge, name=name, dtype=dtype)
34           # Metric should be minimized during optimization.
35           self._direction = "down"
36   
37       def get_config(self):
38           return {"name": self.name, "dtype": self.dtype}
