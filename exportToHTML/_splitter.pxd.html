13   cdef struct SplitRecord:
14       # Data to track sample split
15       intp_t feature         # Which feature to split on.
16       intp_t pos             # Split samples array at the given position,
17       #                      # i.e. count of samples below threshold for feature.
18       #                      # pos is >= end if the node is a leaf.
19       float64_t threshold       # Threshold to split at.
20       float64_t improvement     # Impurity improvement given parent node.
21       float64_t impurity_left   # Impurity of the left split.
22       float64_t impurity_right  # Impurity of the right split.
23       float64_t lower_bound     # Lower bound on value of both children for monotonicity
24       float64_t upper_bound     # Upper bound on value of both children for monotonicity
25       uint8_t missing_go_to_left  # Controls if missing values go to the left node.
26       intp_t n_missing            # Number of missing values for the feature being split on
27   
28   cdef class Splitter:
29       # The splitter searches in the input space for a feature and a threshold
30       # to split the samples samples[start:end].
31       #
32       # The impurity computations are delegated to a criterion object.
33   
34       # Internal structures
35       cdef public Criterion criterion      # Impurity criterion
36       cdef public intp_t max_features      # Number of features to test
37       cdef public intp_t min_samples_leaf  # Min samples in a leaf
38       cdef public float64_t min_weight_leaf   # Minimum weight in a leaf
39   
40       cdef object random_state             # Random state
41       cdef uint32_t rand_r_state           # sklearn_rand_r random number state
42   
43       cdef intp_t[::1] samples             # Sample indices in X, y
44       cdef intp_t n_samples                # X.shape[0]
45       cdef float64_t weighted_n_samples       # Weighted number of samples
46       cdef intp_t[::1] features            # Feature indices in X
47       cdef intp_t[::1] constant_features   # Constant features indices
48       cdef intp_t n_features               # X.shape[1]
49       cdef float32_t[::1] feature_values   # temp. array holding feature values
50   
51       cdef intp_t start                    # Start position for the current node
52       cdef intp_t end                      # End position for the current node
53   
54       cdef const float64_t[:, ::1] y
55       # Monotonicity constraints for each feature.
56       # The encoding is as follows:
57       #   -1: monotonic decrease
58       #    0: no constraint
59       #   +1: monotonic increase
60       cdef const int8_t[:] monotonic_cst
61       cdef bint with_monotonic_cst
62       cdef const float64_t[:] sample_weight
63   
64       # The samples vector `samples` is maintained by the Splitter object such
65       # that the samples contained in a node are contiguous. With this setting,
66       # `node_split` reorganizes the node samples `samples[start:end]` in two
67       # subsets `samples[start:pos]` and `samples[pos:end]`.
68   
69       # The 1-d  `features` array of size n_features contains the features
70       # indices and allows fast sampling without replacement of features.
71   
72       # The 1-d `constant_features` array of size n_features holds in
73       # `constant_features[:n_constant_features]` the feature ids with
74       # constant values for all the samples that reached a specific node.
75       # The value `n_constant_features` is given by the parent node to its
76       # child nodes.  The content of the range `[n_constant_features:]` is left
77       # undefined, but preallocated for performance reasons
78       # This allows optimization with depth-based tree building.
79   
80       # Methods
81       cdef int init(
82           self,
83           object X,
84           const float64_t[:, ::1] y,
85           const float64_t[:] sample_weight,
86           const uint8_t[::1] missing_values_in_feature_mask,
87       ) except -1
88

90           self,
91           intp_t start,
92           intp_t end,
93           float64_t* weighted_n_node_samples
94       ) except -1 nogil
95   
96       cdef int node_split(
97           self,
98           ParentInfo* parent,
99           SplitRecord* split,
100      ) except -1 nogil
