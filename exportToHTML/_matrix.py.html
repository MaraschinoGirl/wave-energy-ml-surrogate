60           # Make sure copy is False since this is in place

116      def todense(self, order=None, out=None):
117          """ 
118          Return a dense representation of this sparse matrix. 
119   
120          Parameters 
121          ---------- 
122          order : {'C', 'F'}, optional 
123              Whether to store multi-dimensional data in C (row-major) 
124              or Fortran (column-major) order in memory. The default 
125              is 'None', which provides no ordering guarantees. 
126              Cannot be specified in conjunction with the `out` 
127              argument. 
128   
129          out : ndarray, 2-D, optional 
130              If specified, uses this array (or `numpy.matrix`) as the 
131              output buffer instead of allocating a new array to 
132              return. The provided array must have the same shape and 
133              dtype as the sparse matrix on which you are calling the 
134              method. 
135   
136          Returns 
137          ------- 
138          arr : numpy.matrix, 2-D 
139              A NumPy matrix object with the same shape and containing 
140              the same data represented by the sparse matrix, with the 
141              requested memory order. If `out` was passed and was an 
142              array (rather than a `numpy.matrix`), it will be filled 
143              with the appropriate values and returned wrapped in a 
144              `numpy.matrix` object that shares the same memory. 
145          """
146          return super().todense(order, out)
