6    import contextlib
7    import logging
8    import os
9    import sys
10   import tempfile
11   import traceback
12   import warnings
13   
14   from traitlets import Unicode
15   from traitlets.config import Configurable
16   
17   
18   class DebugLogFileMixin(Configurable):
19       debug_log_path = Unicode("", config=True, help="Path to use for the debug log file")
20   
21       @contextlib.contextmanager
22       def debug_logging(self):
23           log_path = self.debug_log_path
24           if os.path.isdir(log_path):
25               log_path = os.path.join(log_path, "jupyterlab-debug.log")
26           if not log_path:
27               handle, log_path = tempfile.mkstemp(prefix="jupyterlab-debug-", suffix=".log")
28               os.close(handle)
29           log = self.log
30   
31           # Transfer current log level to the handlers:
32           for h in log.handlers:
33               h.setLevel(self.log_level)
34           log.setLevel("DEBUG")
35   
36           # Create our debug-level file handler:
37           _debug_handler = logging.FileHandler(log_path, "w", "utf8", delay=True)
38           _log_formatter = self._log_formatter_cls(fmt=self.log_format, datefmt=self.log_datefmt)
39           _debug_handler.setFormatter(_log_formatter)
40           _debug_handler.setLevel("DEBUG")
41   
42           log.addHandler(_debug_handler)
43   
44           try:
45               yield
46           except Exception as ex:
47               _, _, exc_traceback = sys.exc_info()
48               msg = traceback.format_exception(ex.__class__, ex, exc_traceback)
49               for line in msg:
50                   self.log.debug(line)
51               if isinstance(ex, SystemExit):
52                   warnings.warn(f"An error occurred. See the log file for details: {log_path!s}")
53                   raise
54               warnings.warn("An error occurred.")
55               warnings.warn(msg[-1].strip())
56               warnings.warn(f"See the log file for details: {log_path!s}")
57               self.exit(1)
58           else:
59               log.removeHandler(_debug_handler)
60               _debug_handler.flush()
61               _debug_handler.close()
62               try:
63                   os.remove(log_path)
64               except FileNotFoundError:
65                   pass
66           log.removeHandler(_debug_handler)
