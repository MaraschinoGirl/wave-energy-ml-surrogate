27   def _exec_demo(cmd):
28       try:
29           exec(cmd)
30       except Exception as error:
31           print(f"Demo of {cmd} failed - {type(error)}:{error}")
32   
33   
34   def demo():
35       if "/go" in sys.argv:
36           for name, cmd in demos:
37               _exec_demo(cmd)
38           return
39   
40       # Otherwise allow the user to select the demo to run
41       while True:
42           rc = pywin.dialogs.list.SelectFromLists("Select a Demo", demos, ["Demo Title"])
43           if rc is None:
44               break
45           title, cmd = demos[rc]
46           _exec_demo(cmd)
47   
48   
49   if __name__ == __main__.__name__:
50       import demoutils
51   
52       if demoutils.NeedGoodGUI():
53           demo()
