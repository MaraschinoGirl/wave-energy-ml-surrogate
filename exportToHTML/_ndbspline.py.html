18   def _get_dtype(dtype):
19       """Return np.complex128 for complex dtypes, np.float64 otherwise."""
20       if np.issubdtype(dtype, np.complexfloating):
21           return np.complex128
22       else:
23           return np.float64

26   class NdBSpline:
27       """Tensor product spline object. 
28    
29       The value at point ``xp = (x1, x2, ..., xN)`` is evaluated as a linear 
30       combination of products of one-dimensional b-splines in each of the ``N`` 
31       dimensions:: 
32    
33          c[i1, i2, ..., iN] * B(x1; i1, t1) * B(x2; i2, t2) * ... * B(xN; iN, tN) 
34    
35    
36       Here ``B(x; i, t)`` is the ``i``-th b-spline defined by the knot vector 
37       ``t`` evaluated at ``x``. 
38    
39       Parameters 
40       ---------- 
41       t : tuple of 1D ndarrays 
42           knot vectors in directions 1, 2, ... N, 
43           ``len(t[i]) == n[i] + k + 1`` 
44       c : ndarray, shape (n1, n2, ..., nN, ...) 
45           b-spline coefficients 
46       k : int or length-d tuple of integers 
47           spline degrees. 
48           A single integer is interpreted as having this degree for 
49           all dimensions. 
50       extrapolate : bool, optional 
51           Whether to extrapolate out-of-bounds inputs, or return `nan`. 
52           Default is to extrapolate. 
53    
54       Attributes 
55       ---------- 
56       t : tuple of ndarrays 
57           Knots vectors. 
58       c : ndarray 
59           Coefficients of the tensor-product spline. 
60       k : tuple of integers 
61           Degrees for each dimension. 
62       extrapolate : bool, optional 
63           Whether to extrapolate or return nans for out-of-bounds inputs. 
64           Defaults to true. 
65    
66       Methods 
67       ------- 
68       __call__ 
69       design_matrix 
70    
71       See Also 
72       -------- 
73       BSpline : a one-dimensional B-spline object 
74       NdPPoly : an N-dimensional piecewise tensor product polynomial 
75    
76       """
77       def __init__(self, t, c, k, *, extrapolate=None):
78           self._k, self._indices_k1d, (self._t, self._len_t) = _preprocess_inputs(k, t)
79   
80           if extrapolate is None:
81               extrapolate = True
82           self.extrapolate = bool(extrapolate)
83   
84           self.c = np.asarray(c)
85   
86           ndim = self._t.shape[0]   # == len(self.t)
87           if self.c.ndim < ndim:
88               raise ValueError(f"Coefficients must be at least {ndim}-dimensional.")
89   
90           for d in range(ndim):
91               td = self.t[d]
92               kd = self.k[d]
93               n = td.shape[0] - kd - 1
94   
95               if self.c.shape[d] != n:
96                   raise ValueError(f"Knots, coefficients and degree in dimension"
97                                    f" {d} are inconsistent:"
98                                    f" got {self.c.shape[d]} coefficients for"
99                                    f" {len(td)} knots, need at least {n} for"
100                                   f" k={k}.")
101  
102          dt = _get_dtype(self.c.dtype)
103          self.c = np.ascontiguousarray(self.c, dtype=dt)
104  
105      @property
106      def k(self):
107          return tuple(self._k)
108  
109      @property
110      def t(self):
111          # repack the knots into a tuple
112          return tuple(self._t[d, :self._len_t[d]] for d in range(self._t.shape[0]))

114      def __call__(self, xi, *, nu=None, extrapolate=None):
115          """Evaluate the tensor product b-spline at ``xi``. 
116   
117          Parameters 
118          ---------- 
119          xi : array_like, shape(..., ndim) 
120              The coordinates to evaluate the interpolator at. 
121              This can be a list or tuple of ndim-dimensional points 
122              or an array with the shape (num_points, ndim). 
123          nu : array_like, optional, shape (ndim,) 
124              Orders of derivatives to evaluate. Each must be non-negative. 
125              Defaults to the zeroth derivivative. 
126          extrapolate : bool, optional 
127              Whether to exrapolate based on first and last intervals in each 
128              dimension, or return `nan`. Default is to ``self.extrapolate``. 
129   
130          Returns 
131          ------- 
132          values : ndarray, shape ``xi.shape[:-1] + self.c.shape[ndim:]`` 
133              Interpolated values at ``xi`` 
134          """
135          ndim = self._t.shape[0]  # == len(self.t)
136  
137          if extrapolate is None:
138              extrapolate = self.extrapolate
139          extrapolate = bool(extrapolate)
140  
141          if nu is None:
142              nu = np.zeros((ndim,), dtype=np.intc)
143          else:
144              nu = np.asarray(nu, dtype=np.intc)
145              if nu.ndim != 1 or nu.shape[0] != ndim:
146                  raise ValueError(
147                      f"invalid number of derivative orders {nu = } for "
148                      f"ndim = {len(self.t)}.")
149              if any(nu < 0):
150                  raise ValueError(f"derivatives must be positive, got {nu = }")
151  
152          # prepare xi : shape (..., m1, ..., md) -> (1, m1, ..., md)
153          xi = np.asarray(xi, dtype=float)
154          xi_shape = xi.shape
155          xi = xi.reshape(-1, xi_shape[-1])
156          xi = np.ascontiguousarray(xi)
157  
158          if xi_shape[-1] != ndim:
159              raise ValueError(f"Shapes: xi.shape={xi_shape} and ndim={ndim}")
160  
161          # complex -> double
162          was_complex = self.c.dtype.kind == 'c'
163          cc = self.c
164          if was_complex and self.c.ndim == ndim:
165              # make sure that core dimensions are intact, and complex->float
166              # size doubling only adds a trailing dimension
167              cc = self.c[..., None]
168          cc = cc.view(float)
169  
170          # prepare the coefficients: flatten the trailing dimensions
171          c1 = cc.reshape(cc.shape[:ndim] + (-1,))
172          c1r = c1.ravel()
173  
174          # replacement for np.ravel_multi_index for indexing of `c1`:
175          _strides_c1 = np.asarray([s // c1.dtype.itemsize
176                                    for s in c1.strides], dtype=np.intp)
177  
178          num_c_tr = c1.shape[-1]  # # of trailing coefficients
179          out = np.empty(xi.shape[:-1] + (num_c_tr,), dtype=c1.dtype)
180  
181          _bspl.evaluate_ndbspline(xi,
182                                   self._t,
183                                   self._len_t,
184                                   self._k,
185                                   nu,
186                                   extrapolate,
187                                   c1r,
188                                   num_c_tr,
189                                   _strides_c1,
190                                   self._indices_k1d,
191                                   out,)
192          out = out.view(self.c.dtype)
193          return out.reshape(xi_shape[:-1] + self.c.shape[ndim:])

195      @classmethod
196      def design_matrix(cls, xvals, t, k, extrapolate=True):
197          """Construct the design matrix as a CSR format sparse array. 
198   
199          Parameters 
200          ---------- 
201          xvals :  ndarray, shape(npts, ndim) 
202              Data points. ``xvals[j, :]`` gives the ``j``-th data point as an 
203              ``ndim``-dimensional array. 
204          t : tuple of 1D ndarrays, length-ndim 
205              Knot vectors in directions 1, 2, ... ndim, 
206          k : int 
207              B-spline degree. 
208          extrapolate : bool, optional 
209              Whether to extrapolate out-of-bounds values of raise a `ValueError` 
210   
211          Returns 
212          ------- 
213          design_matrix : a CSR array 
214              Each row of the design matrix corresponds to a value in `xvals` and 
215              contains values of b-spline basis elements which are non-zero 
216              at this value. 
217   
218          """
219          xvals = np.asarray(xvals, dtype=float)
220          ndim = xvals.shape[-1]
221          if len(t) != ndim:
222              raise ValueError(
223                  f"Data and knots are inconsistent: len(t) = {len(t)} for "
224                  f" {ndim = }."
225              )
226  
227          # tabulate the flat indices for iterating over the (k+1)**ndim subarray
228          k, _indices_k1d, (_t, len_t) = _preprocess_inputs(k, t)
229  
230          # Precompute the shape and strides of the 'coefficients array'.
231          # This would have been the NdBSpline coefficients; in the present context
232          # this is a helper to compute the indices into the colocation matrix.
233          c_shape = tuple(len_t[d] - k[d] - 1 for d in range(ndim))
234  
235          # The strides of the coeffs array: the computation is equivalent to
236          # >>> cstrides = [s // 8 for s in np.empty(c_shape).strides]
237          cs = c_shape[1:] + (1,)
238          cstrides = np.cumprod(cs[::-1], dtype=np.intp)[::-1].copy()
239  
240          # heavy lifting happens here
241          data, indices, indptr = _bspl._colloc_nd(xvals,
242                                                  _t,
243                                                  len_t,
244                                                  k,
245                                                  _indices_k1d,
246                                                  cstrides)
247          return csr_array((data, indices, indptr))

250  def _preprocess_inputs(k, t_tpl):
251      """Helpers: validate and preprocess NdBSpline inputs. 
252   
253         Parameters 
254         ---------- 
255         k : int or tuple 
256            Spline orders 
257         t_tpl : tuple or array-likes 
258            Knots. 
259      """
260      # 1. Make sure t_tpl is a tuple
261      if not isinstance(t_tpl, tuple):
262          raise ValueError(f"Expect `t` to be a tuple of array-likes. "
263                           f"Got {t_tpl} instead."
264          )
265  
266      # 2. Make ``k`` a tuple of integers
267      ndim = len(t_tpl)
268      try:
269          len(k)
270      except TypeError:
271          # make k a tuple
272          k = (k,)*ndim
273  
274      k = np.asarray([operator.index(ki) for ki in k], dtype=np.int32)
275  
276      if len(k) != ndim:
277          raise ValueError(f"len(t) = {len(t_tpl)} != {len(k) = }.")
278  
279      # 3. Validate inputs
280      ndim = len(t_tpl)
281      for d in range(ndim):
282          td = np.asarray(t_tpl[d])
283          kd = k[d]
284          n = td.shape[0] - kd - 1
285          if kd < 0:
286              raise ValueError(f"Spline degree in dimension {d} cannot be"
287                               f" negative.")
288          if td.ndim != 1:
289              raise ValueError(f"Knot vector in dimension {d} must be"
290                               f" one-dimensional.")
291          if n < kd + 1:
292              raise ValueError(f"Need at least {2*kd + 2} knots for degree"
293                               f" {kd} in dimension {d}.")
294          if (np.diff(td) < 0).any():
295              raise ValueError(f"Knots in dimension {d} must be in a"
296                               f" non-decreasing order.")
297          if len(np.unique(td[kd:n + 1])) < 2:
298              raise ValueError(f"Need at least two internal knots in"
299                               f" dimension {d}.")
300          if not np.isfinite(td).all():
301              raise ValueError(f"Knots in dimension {d} should not have"
302                               f" nans or infs.")
303  
304      # 4. tabulate the flat indices for iterating over the (k+1)**ndim subarray
305      # non-zero b-spline elements
306      shape = tuple(kd + 1 for kd in k)
307      indices = np.unravel_index(np.arange(prod(shape)), shape)
308      _indices_k1d = np.asarray(indices, dtype=np.intp).T.copy()
309  
310      # 5. pack the knots into a single array:
311      #    ([1, 2, 3, 4], [5, 6], (7, 8, 9)) -->
312      #    array([[1, 2, 3, 4],
313      #           [5, 6, nan, nan],
314      #           [7, 8, 9, nan]])
315      ndim = len(t_tpl)
316      len_t = [len(ti) for ti in t_tpl]
317      _t = np.empty((ndim, max(len_t)), dtype=float)
318      _t.fill(np.nan)
319      for d in range(ndim):
320          _t[d, :len(t_tpl[d])] = t_tpl[d]
321      len_t = np.asarray(len_t, dtype=np.int32)
322  
323      return k, _indices_k1d, (_t, len_t)

326  def _iter_solve(a, b, solver=ssl.gcrotmk, **solver_args):
327      # work around iterative solvers not accepting multiple r.h.s.
328  
329      # also work around a.dtype == float64 and b.dtype == complex128
330      # cf https://github.com/scipy/scipy/issues/19644
331      if np.issubdtype(b.dtype, np.complexfloating):
332          real = _iter_solve(a, b.real, solver, **solver_args)
333          imag = _iter_solve(a, b.imag, solver, **solver_args)
334          return real + 1j*imag
335  
336      if b.ndim == 2 and b.shape[1] !=1:
337          res = np.empty_like(b)
338          for j in range(b.shape[1]):
339              res[:, j], info = solver(a, b[:, j], **solver_args)
340              if info != 0:
341                  raise ValueError(f"{solver = } returns {info =} for column {j}.")
342          return res
343      else:
344          res, info = solver(a, b, **solver_args)
345          if info != 0:
346              raise ValueError(f"{solver = } returns {info = }.")
347          return res

350  def make_ndbspl(points, values, k=3, *, solver=ssl.gcrotmk, **solver_args):
351      """Construct an interpolating NdBspline. 
352   
353      Parameters 
354      ---------- 
355      points : tuple of ndarrays of float, with shapes (m1,), ... (mN,) 
356          The points defining the regular grid in N dimensions. The points in 
357          each dimension (i.e. every element of the `points` tuple) must be 
358          strictly ascending or descending.       
359      values : ndarray of float, shape (m1, ..., mN, ...) 
360          The data on the regular grid in n dimensions. 
361      k : int, optional 
362          The spline degree. Must be odd. Default is cubic, k=3 
363      solver : a `scipy.sparse.linalg` solver (iterative or direct), optional. 
364          An iterative solver from `scipy.sparse.linalg` or a direct one, 
365          `sparse.sparse.linalg.spsolve`. 
366          Used to solve the sparse linear system 
367          ``design_matrix @ coefficients = rhs`` for the coefficients. 
368          Default is `scipy.sparse.linalg.gcrotmk` 
369      solver_args : dict, optional 
370          Additional arguments for the solver. The call signature is 
371          ``solver(csr_array, rhs_vector, **solver_args)`` 
372   
373      Returns 
374      ------- 
375      spl : NdBSpline object 
376   
377      Notes 
378      ----- 
379      Boundary conditions are not-a-knot in all dimensions. 
380      """
381      ndim = len(points)
382      xi_shape = tuple(len(x) for x in points)
383  
384      try:
385          len(k)
386      except TypeError:
387          # make k a tuple
388          k = (k,)*ndim
389  
390      for d, point in enumerate(points):
391          numpts = len(np.atleast_1d(point))
392          if numpts <= k[d]:
393              raise ValueError(f"There are {numpts} points in dimension {d},"
394                               f" but order {k[d]} requires at least "
395                               f" {k[d]+1} points per dimension.")
396  
397      t = tuple(_not_a_knot(np.asarray(points[d], dtype=float), k[d])
398                for d in range(ndim))
399      xvals = np.asarray([xv for xv in itertools.product(*points)], dtype=float)
400  
401      # construct the colocation matrix
402      matr = NdBSpline.design_matrix(xvals, t, k)
403  
404      # Solve for the coefficients given `values`.
405      # Trailing dimensions: first ndim dimensions are data, the rest are batch
406      # dimensions, so stack `values` into a 2D array for `spsolve` to undestand.
407      v_shape = values.shape
408      vals_shape = (prod(v_shape[:ndim]), prod(v_shape[ndim:]))
409      vals = values.reshape(vals_shape)
410  
411      if solver != ssl.spsolve:
412          solver = functools.partial(_iter_solve, solver=solver)
413          if "atol" not in solver_args:
414              # avoid a DeprecationWarning, grumble grumble
415              solver_args["atol"] = 1e-6
416  
417      coef = solver(matr, vals, **solver_args)
418      coef = coef.reshape(xi_shape + v_shape[ndim:])
419      return NdBSpline(t, coef, k)
