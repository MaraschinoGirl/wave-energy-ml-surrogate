4        else:
5            # On Python 2 we don't really have an API to get the main thread,

10   def get_main_thread_id(unlikely_thread_id=None):
11       """ 
12       :param unlikely_thread_id: 
13           Pass to mark some thread id as not likely the main thread. 
14    
15       :return tuple(thread_id, critical_warning) 
16       """
17       import sys
18       import os
19   
20       current_frames = sys._current_frames()
21       possible_thread_ids = []
22       for thread_ident, frame in current_frames.items():
23           while frame.f_back is not None:
24               frame = frame.f_back
25   
26           basename = os.path.basename(frame.f_code.co_filename)
27           if basename.endswith((".pyc", ".pyo")):
28               basename = basename[:-1]
29   
30           if (frame.f_code.co_name, basename) in [
31               ("_run_module_as_main", "runpy.py"),
32               ("_run_module_as_main", "<frozen runpy>"),
33               ("run_module_as_main", "runpy.py"),
34               ("run_module", "runpy.py"),
35               ("run_path", "runpy.py"),
36           ]:
37               # This is the case for python -m <module name> (this is an ideal match, so,
38               # let's return it).
39               return thread_ident, ""
40   
41           if frame.f_code.co_name == "<module>":
42               if frame.f_globals.get("__name__") == "__main__":
43                   possible_thread_ids.insert(0, thread_ident)  # Add with higher priority
44                   continue
45   
46               # Usually the main thread will be started in the <module>, whereas others would
47               # be started in another place (but when Python is embedded, this may not be
48               # correct, so, just add to the available possibilities as we'll have to choose
49               # one if there are multiple).
50               possible_thread_ids.append(thread_ident)
51   
52       if len(possible_thread_ids) > 0:
53           if len(possible_thread_ids) == 1:
54               return possible_thread_ids[0], ""  # Ideal: only one match
55   
56           while unlikely_thread_id in possible_thread_ids:
57               possible_thread_ids.remove(unlikely_thread_id)
58   
59           if len(possible_thread_ids) == 1:
60               return possible_thread_ids[0], ""  # Ideal: only one match
61   
62           elif len(possible_thread_ids) > 1:
63               # Bad: we can't really be certain of anything at this point.
64               return possible_thread_ids[0], "Multiple thread ids found (%s). Choosing main thread id randomly (%s)." % (
65                   possible_thread_ids,
66                   possible_thread_ids[0],
67               )
68   
69       # If we got here we couldn't discover the main thread id.
70       return None, "Unable to discover main thread id."

73   def fix_main_thread_id(on_warn=lambda msg: None, on_exception=lambda msg: None, on_critical=lambda msg: None):
74       # This means that we weren't able to import threading in the main thread (which most
75       # likely means that the main thread is paused or in some very long operation).
76       # In this case we'll import threading here and hotfix what may be wrong in the threading
77       # module (if we're on Windows where we create a thread to do the attach and on Linux
78       # we are not certain on which thread we're executing this code).
79       #
80       # The code below is a workaround for https://bugs.python.org/issue37416
81       import sys
82       import threading
83   
84       # This is no longer needed in Py 3.13 (as the related issue is already fixed).
85       if sys.version_info[:2] >= (3, 13):
86           return
87   
88       try:
89           with threading._active_limbo_lock:
90               main_thread_instance = get_main_thread_instance(threading)
91   
92               if sys.platform == "win32":
93                   # On windows this code would be called in a secondary thread, so,
94                   # the current thread is unlikely to be the main thread.
95                   if hasattr(threading, "_get_ident"):
96                       unlikely_thread_id = threading._get_ident()  # py2
97                   else:
98                       unlikely_thread_id = threading.get_ident()  # py3
99               else:
100                  unlikely_thread_id = None
101  
102              main_thread_id, critical_warning = get_main_thread_id(unlikely_thread_id)
103  
104              if main_thread_id is not None:
105                  main_thread_id_attr = "_ident"
106                  if not hasattr(main_thread_instance, main_thread_id_attr):
107                      main_thread_id_attr = "_Thread__ident"
108                      assert hasattr(main_thread_instance, main_thread_id_attr)
109  
110                  if main_thread_id != getattr(main_thread_instance, main_thread_id_attr):
111                      # Note that we also have to reset the '_tstack_lock' for a regular lock.
112                      # This is needed to avoid an error on shutdown because this lock is bound
113                      # to the thread state and will be released when the secondary thread
114                      # that initialized the lock is finished -- making an assert fail during
115                      # process shutdown.
116                      main_thread_instance._tstate_lock = threading._allocate_lock()
117                      main_thread_instance._tstate_lock.acquire()
118  
119                      # Actually patch the thread ident as well as the threading._active dict
120                      # (we should have the _active_limbo_lock to do that).
121                      threading._active.pop(getattr(main_thread_instance, main_thread_id_attr), None)
122                      setattr(main_thread_instance, main_thread_id_attr, main_thread_id)
123                      threading._active[getattr(main_thread_instance, main_thread_id_attr)] = main_thread_instance
124  
125          # Note: only import from pydevd after the patching is done (we want to do the minimum
126          # possible when doing that patching).
127          on_warn(
128              "The threading module was not imported by user code in the main thread. The debugger will attempt to work around https://bugs.python.org/issue37416."
129          )
130  
131          if critical_warning:
132              on_critical("Issue found when debugger was trying to work around https://bugs.python.org/issue37416:\n%s" % (critical_warning,))
133      except:
134          on_exception("Error patching main thread id.")

137  def attach(port, host, protocol="", debug_mode=""):
138      try:
139          import sys
140  
141          fix_main_thread = "threading" not in sys.modules
142  
143          if fix_main_thread:
144  
145              def on_warn(msg):
146                  from _pydev_bundle import pydev_log
147  
148                  pydev_log.warn(msg)
149  
150              def on_exception(msg):
151                  from _pydev_bundle import pydev_log
152  
153                  pydev_log.exception(msg)
154  
155              def on_critical(msg):
156                  from _pydev_bundle import pydev_log
157  
158                  pydev_log.critical(msg)
159  
160              fix_main_thread_id(on_warn=on_warn, on_exception=on_exception, on_critical=on_critical)
161  
162          else:
163              from _pydev_bundle import pydev_log  # @Reimport
164  
165              pydev_log.debug("The threading module is already imported by user code.")
166  
167          if protocol:
168              from _pydevd_bundle import pydevd_defaults
169  
170              pydevd_defaults.PydevdCustomization.DEFAULT_PROTOCOL = protocol
171  
172          if debug_mode:
173              from _pydevd_bundle import pydevd_defaults
174  
175              pydevd_defaults.PydevdCustomization.DEBUG_MODE = debug_mode
176  
177          import pydevd
178  
179          # I.e.: disconnect/reset if already connected.
180  
181          pydevd.SetupHolder.setup = None
182  
183          py_db = pydevd.get_global_debugger()
184          if py_db is not None:
185              py_db.dispose_and_kill_all_pydevd_threads(wait=False)
186  
187          # pydevd.DebugInfoHolder.DEBUG_TRACE_LEVEL = 3
188          pydevd.settrace(
189              port=port,
190              host=host,
191              stdoutToServer=True,
192              stderrToServer=True,
193              overwrite_prev_trace=True,
194              suspend=False,
195              trace_only_current_thread=False,
196              patch_multiprocessing=False,
197          )
198      except:
199          import traceback
200  
201          traceback.print_exc()
