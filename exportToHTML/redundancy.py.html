5    # This file has the code that figures out how each factor in some given Term
6    # should be coded. This is complicated by dealing with models with categorical
7    # factors like:
8    #   1 + a + a:b
9    # then technically 'a' (which represents the space of vectors that can be
10   # produced as linear combinations of the dummy coding of the levels of the
11   # factor a) is collinear with the intercept, and 'a:b' (which represents the
12   # space of vectors that can be produced as linear combinations of the dummy
13   # coding *of a new factor whose levels are the cartesian product of a and b)
14   # is collinear with both 'a' and the intercept.
15   #
16   # In such a case, the rule is that we find some way to code each term so that
17   # the full space of vectors that it represents *is present in the model* BUT
18   # there is no collinearity between the different terms. In effect, we have to
19   # choose a set of vectors that spans everything that that term wants to span,
20   # *except* that part of the vector space which was already spanned by earlier
21   # terms.
22   
23   # How? We replace each term with the set of "subterms" that it covers, like
24   # so:
25   #   1 -> ()
26   #   a -> (), a-
27   #   a:b -> (), a-, b-, a-:b-
28   # where "-" means "coded so as not to span the intercept". So that example
29   # above expands to
30   #   [()] + [() + a-] + [() + a- + b- + a-:b-]
31   # so we go through from left to right, and for each term we:
32   #   1) toss out all the subterms that have already been used (this is a simple
33   #      equality test, no magic)
34   #   2) simplify the terms that are left, according to rules like
35   #        () + a- = a+
36   #      (here + means, "coded to span the intercept")
37   #   3) use the resulting subterm list as our coding for this term!
38   # So in the above, we go:
39   #   (): stays the same, coded as intercept
40   #   () + a-: reduced to just a-, which is what we code
41   #   () + a- + b- + a-:b-: reduced to b- + a-:b-, which is simplified to a+:b-.
42

46   # This should really be a named tuple, but those don't exist until Python
47   # 2.6...
48   class _ExpandedFactor(object):
49       """A factor, with an additional annotation for whether it is coded 
50       full-rank (includes_intercept=True) or not. 
51    
52       These objects are treated as immutable."""
53   
54       def __init__(self, includes_intercept, factor):
55           self.includes_intercept = includes_intercept
56           self.factor = factor
57   
58       def __hash__(self):
59           return hash((_ExpandedFactor, self.includes_intercept, self.factor))
60   
61       def __eq__(self, other):
62           return (
63               isinstance(other, _ExpandedFactor)
64               and other.includes_intercept == self.includes_intercept
65               and other.factor == self.factor
66           )
67   
68       def __ne__(self, other):
69           return not self == other
70   
71       def __repr__(self):
72           if self.includes_intercept:
73               suffix = "+"
74           else:
75               suffix = "-"
76           return "%r%s" % (self.factor, suffix)
77   
78       __getstate__ = no_pickling
79

81   class _Subterm(object):
82       "Also immutable."
83   
84       def __init__(self, efactors):
85           self.efactors = frozenset(efactors)
86   
87       def can_absorb(self, other):
88           # returns True if 'self' is like a-:b-, and 'other' is like a-
89           return len(self.efactors) - len(
90               other.efactors
91           ) == 1 and self.efactors.issuperset(other.efactors)
92   
93       def absorb(self, other):
94           diff = self.efactors.difference(other.efactors)
95           assert len(diff) == 1
96           efactor = list(diff)[0]
97           assert not efactor.includes_intercept
98           new_factors = set(other.efactors)
99           new_factors.add(_ExpandedFactor(True, efactor.factor))
100          return _Subterm(new_factors)
101  
102      def __hash__(self):
103          return hash((_Subterm, self.efactors))
104  
105      def __eq__(self, other):
106          return isinstance(other, _Subterm) and self.efactors == self.efactors
107  
108      def __ne__(self, other):
109          return not self == other
110  
111      def __repr__(self):
112          return "%s(%r)" % (self.__class__.__name__, list(self.efactors))
113  
114      __getstate__ = no_pickling

117  # For testing: takes a shorthand description of a list of subterms like
118  #   [(), ("a-",), ("a-", "b+")]
119  # and expands it into a list of _Subterm and _ExpandedFactor objects.
120  def _expand_test_abbrevs(short_subterms):
121      subterms = []
122      for subterm in short_subterms:
123          factors = []
124          for factor_name in subterm:
125              assert factor_name[-1] in ("+", "-")
126              factors.append(_ExpandedFactor(factor_name[-1] == "+", factor_name[:-1]))
127          subterms.append(_Subterm(factors))
128      return subterms

131  def test__Subterm():
132      s_ab = _expand_test_abbrevs([["a-", "b-"]])[0]
133      s_abc = _expand_test_abbrevs([["a-", "b-", "c-"]])[0]
134      s_null = _expand_test_abbrevs([[]])[0]
135      s_cd = _expand_test_abbrevs([["c-", "d-"]])[0]
136      s_a = _expand_test_abbrevs([["a-"]])[0]
137      s_ap = _expand_test_abbrevs([["a+"]])[0]
138      s_abp = _expand_test_abbrevs([["a-", "b+"]])[0]
139      for bad in s_abc, s_null, s_cd, s_ap, s_abp:
140          assert not s_ab.can_absorb(bad)
141      assert s_ab.can_absorb(s_a)
142      assert s_ab.absorb(s_a) == s_abp

145  # Importantly, this preserves the order of the input. Both the items inside
146  # each subset are in the order they were in the original tuple, and the tuples
147  # are emitted so that they're sorted with respect to their elements position
148  # in the original tuple.
149  def _subsets_sorted(tupl):
150      def helper(seq):
151          if not seq:
152              yield ()
153          else:
154              obj = seq[0]
155              for subset in _subsets_sorted(seq[1:]):
156                  yield subset
157                  yield (obj,) + subset
158  
159      # Transform each obj -> (idx, obj) tuple, so that we can later sort them
160      # by their position in the original list.
161      expanded = list(enumerate(tupl))
162      expanded_subsets = list(helper(expanded))
163      # This exploits Python's stable sort: we want short before long, and ties
164      # broken by natural ordering on the (idx, obj) entries in each subset. So
165      # we sort by the latter first, then by the former.
166      expanded_subsets.sort()
167      expanded_subsets.sort(key=len)
168      # And finally, we strip off the idx's:
169      for subset in expanded_subsets:
170          yield tuple([obj for (idx, obj) in subset])
171  
172  
173  def test__subsets_sorted():
174      assert list(_subsets_sorted((1, 2))) == [(), (1,), (2,), (1, 2)]
175      assert list(_subsets_sorted((1, 2, 3))) == [
176          (),
177          (1,),
178          (2,),
179          (3,),
180          (1, 2),
181          (1, 3),
182          (2, 3),
183          (1, 2, 3),
184      ]
185      assert len(list(_subsets_sorted(range(5)))) == 2**5

188  def _simplify_one_subterm(subterms):
189      # We simplify greedily from left to right.
190      # Returns True if succeeded, False otherwise
191      for short_i, short_subterm in enumerate(subterms):
192          for long_i, long_subterm in enumerate(subterms[short_i + 1 :]):
193              if long_subterm.can_absorb(short_subterm):
194                  new_subterm = long_subterm.absorb(short_subterm)
195                  subterms[short_i + 1 + long_i] = new_subterm
196                  subterms.pop(short_i)
197                  return True
198      return False
199  
200  
201  def _simplify_subterms(subterms):
202      while _simplify_one_subterm(subterms):
203          pass

206  def test__simplify_subterms():
207      def t(given, expected):
208          given = _expand_test_abbrevs(given)
209          expected = _expand_test_abbrevs(expected)
210          print("testing if:", given, "->", expected)
211          _simplify_subterms(given)
212          assert given == expected

215      t([(), ("a-",)], [("a+",)])
216      t([(), ("a-",), ("b-",), ("a-", "b-")], [("a+", "b+")])

221  # 'term' is a Term
222  # 'numeric_factors' is any set-like object which lists the
223  #   numeric/non-categorical factors in this term. Such factors are just
224  #   ignored by this routine.
225  # 'used_subterms' is a set which records which subterms have previously been
226  #   used. E.g., a:b has subterms (), a, b, a:b, and if we're processing
227  #    y ~ a + a:b
228  #   then by the time we reach a:b, the () and a subterms will have already
229  #   been used. This is an in/out argument, and should be treated as opaque by
230  #   callers -- really it is a way for multiple invocations of this routine to
231  #   talk to each other. Each time it is called, this routine adds the subterms
232  #   of each factor to this set in place. So the first time this routine is
233  #   called, pass in an empty set, and then just keep passing the same set to
234  #   any future calls.
235  # Returns: a list of dicts. Each dict maps from factors to booleans. The
236  # coding for the given term should use a full-rank contrast for those factors
237  # which map to True, a (n-1)-rank contrast for those factors which map to
238  # False, and any factors which are not mentioned are numeric and should be
239  # added back in. These dicts should add columns to the design matrix from left
240  # to right.
241  def pick_contrasts_for_term(term, numeric_factors, used_subterms):
242      categorical_factors = [f for f in term.factors if f not in numeric_factors]
243      # Converts a term into an expanded list of subterms like:
244      #   a:b  ->  1 + a- + b- + a-:b-
245      # and discards the ones that have already been used.
246      subterms = []
247      for subset in _subsets_sorted(categorical_factors):
248          subterm = _Subterm([_ExpandedFactor(False, f) for f in subset])
249          if subterm not in used_subterms:
250              subterms.append(subterm)
251      used_subterms.update(subterms)
252      _simplify_subterms(subterms)
253      factor_codings = []
254      for subterm in subterms:
255          factor_coding = {}
256          for expanded in subterm.efactors:
257              factor_coding[expanded.factor] = expanded.includes_intercept
258          factor_codings.append(factor_coding)
259      return factor_codings

262  def test_pick_contrasts_for_term():
263      from patsy.desc import Term
264  
265      used = set()
266      codings = pick_contrasts_for_term(Term([]), set(), used)
267      assert codings == [{}]
268      codings = pick_contrasts_for_term(Term(["a", "x"]), set(["x"]), used)
269      assert codings == [{"a": False}]
270      codings = pick_contrasts_for_term(Term(["a", "b"]), set(), used)
271      assert codings == [{"a": True, "b": False}]
272      used_snapshot = set(used)
273      codings = pick_contrasts_for_term(Term(["c", "d"]), set(), used)
274      assert codings == [{"d": False}, {"c": False, "d": True}]
275      # Do it again backwards, to make sure we're deterministic with respect to
276      # order:
277      codings = pick_contrasts_for_term(Term(["d", "c"]), set(), used_snapshot)
278      assert codings == [{"c": False}, {"c": True, "d": False}]
