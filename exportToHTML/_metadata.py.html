14   """API metadata conversion utilities."""
15   
16   import collections
17   
18   _Metadatum = collections.namedtuple(
19       "_Metadatum",
20       (
21           "key",
22           "value",
23       ),
24   )
25   
26   
27   def _beta_metadatum(key, value):
28       beta_key = key if isinstance(key, (bytes,)) else key.encode("ascii")
29       beta_value = value if isinstance(value, (bytes,)) else value.encode("ascii")
30       return _Metadatum(beta_key, beta_value)
31   
32   
33   def _metadatum(beta_key, beta_value):
34       key = beta_key if isinstance(beta_key, (str,)) else beta_key.decode("utf8")
35       if isinstance(beta_value, (str,)) or key[-4:] == "-bin":
36           value = beta_value
37       else:
38           value = beta_value.decode("utf8")
39       return _Metadatum(key, value)
40   
41   
42   def beta(metadata):
43       if metadata is None:
44           return ()
45       else:
46           return tuple(_beta_metadatum(key, value) for key, value in metadata)
47   
48   
49   def unbeta(beta_metadata):
50       if beta_metadata is None:
51           return ()
52       else:
53           return tuple(
54               _metadatum(beta_key, beta_value)
55               for beta_key, beta_value in beta_metadata
56           )
