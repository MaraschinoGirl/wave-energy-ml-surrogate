58   def _fit_estimator(estimator, X, y, sample_weight=None, **fit_params):
59       estimator = clone(estimator)
60       if sample_weight is not None:
61           estimator.fit(X, y, sample_weight=sample_weight, **fit_params)
62       else:
63           estimator.fit(X, y, **fit_params)
64       return estimator
65

67   def _partial_fit_estimator(
68       estimator, X, y, classes=None, partial_fit_params=None, first_time=True
69   ):
70       partial_fit_params = {} if partial_fit_params is None else partial_fit_params
71       if first_time:
72           estimator = clone(estimator)
73   
74       if classes is not None:
75           estimator.partial_fit(X, y, classes=classes, **partial_fit_params)
76       else:
77           estimator.partial_fit(X, y, **partial_fit_params)
78       return estimator

81   def _available_if_estimator_has(attr):
82       """Return a function to check if the sub-estimator(s) has(have) `attr`. 
83    
84       Helper for Chain implementations. 
85       """
86   
87       def _check(self):
88           if hasattr(self, "estimators_"):
89               return all(hasattr(est, attr) for est in self.estimators_)
90   
91           if hasattr(self.estimator, attr):
92               return True
93   
94           return False
95   
96       return available_if(_check)

99   class _MultiOutputEstimator(MetaEstimatorMixin, BaseEstimator, metaclass=ABCMeta):
100      _parameter_constraints: dict = {
101          "estimator": [HasMethods(["fit", "predict"])],
102          "n_jobs": [Integral, None],
103      }
104  
105      @abstractmethod
106      def __init__(self, estimator, *, n_jobs=None):
107          self.estimator = estimator
108          self.n_jobs = n_jobs
109  
110      @_available_if_estimator_has("partial_fit")
111      @_fit_context(
112          # MultiOutput*.estimator is not validated yet
113          prefer_skip_nested_validation=False
114      )
115      def partial_fit(self, X, y, classes=None, sample_weight=None, **partial_fit_params):
116          """Incrementally fit a separate model for each class output. 
117   
118          Parameters 
119          ---------- 
120          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
121              The input data. 
122   
123          y : {array-like, sparse matrix} of shape (n_samples, n_outputs) 
124              Multi-output targets. 
125   
126          classes : list of ndarray of shape (n_outputs,), default=None 
127              Each array is unique classes for one output in str/int. 
128              Can be obtained via 
129              ``[np.unique(y[:, i]) for i in range(y.shape[1])]``, where `y` 
130              is the target matrix of the entire dataset. 
131              This argument is required for the first call to partial_fit 
132              and can be omitted in the subsequent calls. 
133              Note that `y` doesn't need to contain all labels in `classes`. 
134   
135          sample_weight : array-like of shape (n_samples,), default=None 
136              Sample weights. If `None`, then samples are equally weighted. 
137              Only supported if the underlying regressor supports sample 
138              weights. 
139   
140          **partial_fit_params : dict of str -> object 
141              Parameters passed to the ``estimator.partial_fit`` method of each 
142              sub-estimator. 
143   
144              Only available if `enable_metadata_routing=True`. See the 
145              :ref:`User Guide <metadata_routing>`. 
146   
147              .. versionadded:: 1.3 
148   
149          Returns 
150          ------- 
151          self : object 
152              Returns a fitted instance. 
153          """
154          _raise_for_params(partial_fit_params, self, "partial_fit")
155  
156          first_time = not hasattr(self, "estimators_")
157  
158          y = validate_data(self, X="no_validation", y=y, multi_output=True)
159  
160          if y.ndim == 1:
161              raise ValueError(
162                  "y must have at least two dimensions for "
163                  "multi-output regression but has only one."
164              )
165  
166          if _routing_enabled():
167              if sample_weight is not None:
168                  partial_fit_params["sample_weight"] = sample_weight
169              routed_params = process_routing(
170                  self,
171                  "partial_fit",
172                  **partial_fit_params,
173              )
174          else:
175              if sample_weight is not None and not has_fit_parameter(
176                  self.estimator, "sample_weight"
177              ):
178                  raise ValueError(
179                      "Underlying estimator does not support sample weights."
180                  )
181  
182              if sample_weight is not None:
183                  routed_params = Bunch(
184                      estimator=Bunch(partial_fit=Bunch(sample_weight=sample_weight))
185                  )
186              else:
187                  routed_params = Bunch(estimator=Bunch(partial_fit=Bunch()))
188  
189          self.estimators_ = Parallel(n_jobs=self.n_jobs)(
190              delayed(_partial_fit_estimator)(
191                  self.estimators_[i] if not first_time else self.estimator,
192                  X,
193                  y[:, i],
194                  classes[i] if classes is not None else None,
195                  partial_fit_params=routed_params.estimator.partial_fit,
196                  first_time=first_time,
197              )
198              for i in range(y.shape[1])
199          )
200  
201          if first_time and hasattr(self.estimators_[0], "n_features_in_"):
202              self.n_features_in_ = self.estimators_[0].n_features_in_
203          if first_time and hasattr(self.estimators_[0], "feature_names_in_"):
204              self.feature_names_in_ = self.estimators_[0].feature_names_in_
205  
206          return self
207  
208      @_fit_context(
209          # MultiOutput*.estimator is not validated yet
210          prefer_skip_nested_validation=False
211      )
212      def fit(self, X, y, sample_weight=None, **fit_params):
213          """Fit the model to data, separately for each output variable. 
214   
215          Parameters 
216          ---------- 
217          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
218              The input data. 
219   
220          y : {array-like, sparse matrix} of shape (n_samples, n_outputs) 
221              Multi-output targets. An indicator matrix turns on multilabel 
222              estimation. 
223   
224          sample_weight : array-like of shape (n_samples,), default=None 
225              Sample weights. If `None`, then samples are equally weighted. 
226              Only supported if the underlying regressor supports sample 
227              weights. 
228   
229          **fit_params : dict of string -> object 
230              Parameters passed to the ``estimator.fit`` method of each step. 
231   
232              .. versionadded:: 0.23 
233   
234          Returns 
235          ------- 
236          self : object 
237              Returns a fitted instance. 
238          """
239          if not hasattr(self.estimator, "fit"):
240              raise ValueError("The base estimator should implement a fit method")
241  
242          y = validate_data(self, X="no_validation", y=y, multi_output=True)
243  
244          if is_classifier(self):
245              check_classification_targets(y)
246  
247          if y.ndim == 1:
248              raise ValueError(
249                  "y must have at least two dimensions for "
250                  "multi-output regression but has only one."
251              )
252  
253          if _routing_enabled():
254              if sample_weight is not None:
255                  fit_params["sample_weight"] = sample_weight
256              routed_params = process_routing(
257                  self,
258                  "fit",
259                  **fit_params,
260              )
261          else:
262              if sample_weight is not None and not has_fit_parameter(
263                  self.estimator, "sample_weight"
264              ):
265                  raise ValueError(
266                      "Underlying estimator does not support sample weights."
267                  )
268  
269              fit_params_validated = _check_method_params(X, params=fit_params)
270              routed_params = Bunch(estimator=Bunch(fit=fit_params_validated))
271              if sample_weight is not None:
272                  routed_params.estimator.fit["sample_weight"] = sample_weight
273  
274          self.estimators_ = Parallel(n_jobs=self.n_jobs)(
275              delayed(_fit_estimator)(
276                  self.estimator, X, y[:, i], **routed_params.estimator.fit
277              )
278              for i in range(y.shape[1])
279          )
280  
281          if hasattr(self.estimators_[0], "n_features_in_"):
282              self.n_features_in_ = self.estimators_[0].n_features_in_
283          if hasattr(self.estimators_[0], "feature_names_in_"):
284              self.feature_names_in_ = self.estimators_[0].feature_names_in_
285  
286          return self
287  
288      def predict(self, X):
289          """Predict multi-output variable using model for each target variable. 
290   
291          Parameters 
292          ---------- 
293          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
294              The input data. 
295   
296          Returns 
297          ------- 
298          y : {array-like, sparse matrix} of shape (n_samples, n_outputs) 
299              Multi-output targets predicted across multiple predictors. 
300              Note: Separate models are generated for each predictor. 
301          """
302          check_is_fitted(self)
303          if not hasattr(self.estimators_[0], "predict"):
304              raise ValueError("The base estimator should implement a predict method")
305  
306          y = Parallel(n_jobs=self.n_jobs)(
307              delayed(e.predict)(X) for e in self.estimators_
308          )
309  
310          return np.asarray(y).T

312      def __sklearn_tags__(self):
313          tags = super().__sklearn_tags__()
314          tags.input_tags.sparse = get_tags(self.estimator).input_tags.sparse
315          tags.target_tags.single_output = False
316          tags.target_tags.multi_output = True

321

325          .. versionadded:: 1.3 
326

342  class MultiOutputRegressor(RegressorMixin, _MultiOutputEstimator):
343      """Multi target regression. 
344   
345      This strategy consists of fitting one regressor per target. This is a 
346      simple strategy for extending regressors that do not natively support 
347      multi-target regression. 
348   
349      .. versionadded:: 0.18 
350   
351      Parameters 
352      ---------- 
353      estimator : estimator object 
354          An estimator object implementing :term:`fit` and :term:`predict`. 
355   
356      n_jobs : int or None, optional (default=None) 
357          The number of jobs to run in parallel. 
358          :meth:`fit`, :meth:`predict` and :meth:`partial_fit` (if supported 
359          by the passed estimator) will be parallelized for each target. 
360   
361          When individual estimators are fast to train or predict, 
362          using ``n_jobs > 1`` can result in slower performance due 
363          to the parallelism overhead. 
364   
365          ``None`` means `1` unless in a :obj:`joblib.parallel_backend` context. 
366          ``-1`` means using all available processes / threads. 
367          See :term:`Glossary <n_jobs>` for more details. 
368   
369          .. versionchanged:: 0.20 
370              `n_jobs` default changed from `1` to `None`. 
371   
372      Attributes 
373      ---------- 
374      estimators_ : list of ``n_output`` estimators 
375          Estimators used for predictions. 
376   
377      n_features_in_ : int 
378          Number of features seen during :term:`fit`. Only defined if the 
379          underlying `estimator` exposes such an attribute when fit. 
380   
381          .. versionadded:: 0.24 
382   
383      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
384          Names of features seen during :term:`fit`. Only defined if the 
385          underlying estimators expose such an attribute when fit. 
386   
387          .. versionadded:: 1.0 
388   
389      See Also 
390      -------- 
391      RegressorChain : A multi-label model that arranges regressions into a 
392          chain. 
393      MultiOutputClassifier : Classifies each output independently rather than 
394          chaining. 
395   
396      Examples 
397      -------- 
398      >>> import numpy as np 
399      >>> from sklearn.datasets import load_linnerud 
400      >>> from sklearn.multioutput import MultiOutputRegressor 
401      >>> from sklearn.linear_model import Ridge 
402      >>> X, y = load_linnerud(return_X_y=True) 
403      >>> regr = MultiOutputRegressor(Ridge(random_state=123)).fit(X, y) 
404      >>> regr.predict(X[[0]]) 
405      array([[176..., 35..., 57...]]) 
406      """
407  
408      def __init__(self, estimator, *, n_jobs=None):
409          super().__init__(estimator, n_jobs=n_jobs)
410  
411      @_available_if_estimator_has("partial_fit")
412      def partial_fit(self, X, y, sample_weight=None, **partial_fit_params):
413          """Incrementally fit the model to data, for each output variable. 
414   
415          Parameters 
416          ---------- 
417          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
418              The input data. 
419   
420          y : {array-like, sparse matrix} of shape (n_samples, n_outputs) 
421              Multi-output targets. 
422   
423          sample_weight : array-like of shape (n_samples,), default=None 
424              Sample weights. If `None`, then samples are equally weighted. 
425              Only supported if the underlying regressor supports sample 
426              weights. 
427   
428          **partial_fit_params : dict of str -> object 
429              Parameters passed to the ``estimator.partial_fit`` method of each 
430              sub-estimator. 
431   
432              Only available if `enable_metadata_routing=True`. See the 
433              :ref:`User Guide <metadata_routing>`. 
434   
435              .. versionadded:: 1.3 
436   
437          Returns 
438          ------- 
439          self : object 
440              Returns a fitted instance. 
441          """
442          super().partial_fit(X, y, sample_weight=sample_weight, **partial_fit_params)
443

445  class MultiOutputClassifier(ClassifierMixin, _MultiOutputEstimator):
446      """Multi target classification. 
447   
448      This strategy consists of fitting one classifier per target. This is a 
449      simple strategy for extending classifiers that do not natively support 
450      multi-target classification. 
451   
452      Parameters 
453      ---------- 
454      estimator : estimator object 
455          An estimator object implementing :term:`fit` and :term:`predict`. 
456          A :term:`predict_proba` method will be exposed only if `estimator` implements 
457          it. 
458   
459      n_jobs : int or None, optional (default=None) 
460          The number of jobs to run in parallel. 
461          :meth:`fit`, :meth:`predict` and :meth:`partial_fit` (if supported 
462          by the passed estimator) will be parallelized for each target. 
463   
464          When individual estimators are fast to train or predict, 
465          using ``n_jobs > 1`` can result in slower performance due 
466          to the parallelism overhead. 
467   
468          ``None`` means `1` unless in a :obj:`joblib.parallel_backend` context. 
469          ``-1`` means using all available processes / threads. 
470          See :term:`Glossary <n_jobs>` for more details. 
471   
472          .. versionchanged:: 0.20 
473              `n_jobs` default changed from `1` to `None`. 
474   
475      Attributes 
476      ---------- 
477      classes_ : ndarray of shape (n_classes,) 
478          Class labels. 
479   
480      estimators_ : list of ``n_output`` estimators 
481          Estimators used for predictions. 
482   
483      n_features_in_ : int 
484          Number of features seen during :term:`fit`. Only defined if the 
485          underlying `estimator` exposes such an attribute when fit. 
486   
487          .. versionadded:: 0.24 
488   
489      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
490          Names of features seen during :term:`fit`. Only defined if the 
491          underlying estimators expose such an attribute when fit. 
492   
493          .. versionadded:: 1.0 
494   
495      See Also 
496      -------- 
497      ClassifierChain : A multi-label model that arranges binary classifiers 
498          into a chain. 
499      MultiOutputRegressor : Fits one regressor per target variable. 
500   
501      Examples 
502      -------- 
503      >>> import numpy as np 
504      >>> from sklearn.datasets import make_multilabel_classification 
505      >>> from sklearn.multioutput import MultiOutputClassifier 
506      >>> from sklearn.linear_model import LogisticRegression 
507      >>> X, y = make_multilabel_classification(n_classes=3, random_state=0) 
508      >>> clf = MultiOutputClassifier(LogisticRegression()).fit(X, y) 
509      >>> clf.predict(X[-2:]) 
510      array([[1, 1, 1], 
511             [1, 0, 1]]) 
512      """
513  
514      def __init__(self, estimator, *, n_jobs=None):
515          super().__init__(estimator, n_jobs=n_jobs)
516  
517      def fit(self, X, Y, sample_weight=None, **fit_params):
518          """Fit the model to data matrix X and targets Y. 
519   
520          Parameters 
521          ---------- 
522          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
523              The input data. 
524   
525          Y : array-like of shape (n_samples, n_classes) 
526              The target values. 
527   
528          sample_weight : array-like of shape (n_samples,), default=None 
529              Sample weights. If `None`, then samples are equally weighted. 
530              Only supported if the underlying classifier supports sample 
531              weights. 
532   
533          **fit_params : dict of string -> object 
534              Parameters passed to the ``estimator.fit`` method of each step. 
535   
536              .. versionadded:: 0.23 
537   
538          Returns 
539          ------- 
540          self : object 
541              Returns a fitted instance. 
542          """
543          super().fit(X, Y, sample_weight=sample_weight, **fit_params)
544          self.classes_ = [estimator.classes_ for estimator in self.estimators_]
545          return self
546  
547      def _check_predict_proba(self):
548          if hasattr(self, "estimators_"):
549              # raise an AttributeError if `predict_proba` does not exist for
550              # each estimator
551              [getattr(est, "predict_proba") for est in self.estimators_]
552              return True
553          # raise an AttributeError if `predict_proba` does not exist for the
554          # unfitted estimator
555          getattr(self.estimator, "predict_proba")
556          return True
557  
558      @available_if(_check_predict_proba)
559      def predict_proba(self, X):
560          """Return prediction probabilities for each class of each output. 
561   
562          This method will raise a ``ValueError`` if any of the 
563          estimators do not have ``predict_proba``. 
564   
565          Parameters 
566          ---------- 
567          X : array-like of shape (n_samples, n_features) 
568              The input data. 
569   
570          Returns 
571          ------- 
572          p : array of shape (n_samples, n_classes), or a list of n_outputs \ 
573                  such arrays if n_outputs > 1. 
574              The class probabilities of the input samples. The order of the 
575              classes corresponds to that in the attribute :term:`classes_`. 
576   
577              .. versionchanged:: 0.19 
578                  This function now returns a list of arrays where the length of 
579                  the list is ``n_outputs``, and each array is (``n_samples``, 
580                  ``n_classes``) for that particular output. 
581          """
582          check_is_fitted(self)
583          results = [estimator.predict_proba(X) for estimator in self.estimators_]
584          return results
585  
586      def score(self, X, y):
587          """Return the mean accuracy on the given test data and labels. 
588   
589          Parameters 
590          ---------- 
591          X : array-like of shape (n_samples, n_features) 
592              Test samples. 
593   
594          y : array-like of shape (n_samples, n_outputs) 
595              True values for X. 
596   
597          Returns 
598          ------- 
599          scores : float 
600              Mean accuracy of predicted target versus true target. 
601          """
602          check_is_fitted(self)
603          n_outputs_ = len(self.estimators_)
604          if y.ndim == 1:
605              raise ValueError(
606                  "y must have at least two dimensions for "
607                  "multi target classification but has only one"
608              )
609          if y.shape[1] != n_outputs_:
610              raise ValueError(
611                  "The number of outputs of Y for fit {0} and"
612                  " score {1} should be same".format(n_outputs_, y.shape[1])
613              )
614          y_pred = self.predict(X)
615          return np.mean(np.all(y == y_pred, axis=1))
616  
617      def __sklearn_tags__(self):
618          tags = super().__sklearn_tags__()
619          # FIXME
620          tags._skip_test = True
621          return tags

638  class _BaseChain(BaseEstimator, metaclass=ABCMeta):
639      _parameter_constraints: dict = {
640          "base_estimator": [HasMethods(["fit", "predict"])],
641          "order": ["array-like", StrOptions({"random"}), None],
642          "cv": ["cv_object", StrOptions({"prefit"})],
643          "random_state": ["random_state"],
644          "verbose": ["boolean"],
645      }
646  
647      def __init__(
648          self, base_estimator, *, order=None, cv=None, random_state=None, verbose=False
649      ):
650          self.base_estimator = base_estimator
651          self.order = order
652          self.cv = cv
653          self.random_state = random_state
654          self.verbose = verbose
655  
656      def _log_message(self, *, estimator_idx, n_estimators, processing_msg):
657          if not self.verbose:
658              return None
659          return f"({estimator_idx} of {n_estimators}) {processing_msg}"
660  
661      def _get_predictions(self, X, *, output_method):
662          """Get predictions for each model in the chain."""
663          check_is_fitted(self)
664          X = validate_data(self, X, accept_sparse=True, reset=False)
665          Y_output_chain = np.zeros((X.shape[0], len(self.estimators_)))
666          Y_feature_chain = np.zeros((X.shape[0], len(self.estimators_)))
667  
668          # `RegressorChain` does not have a `chain_method_` parameter so we
669          # default to "predict"
670          chain_method = getattr(self, "chain_method_", "predict")
671          hstack = sp.hstack if sp.issparse(X) else np.hstack
672          for chain_idx, estimator in enumerate(self.estimators_):
673              previous_predictions = Y_feature_chain[:, :chain_idx]
674              # if `X` is a scipy sparse dok_array, we convert it to a sparse
675              # coo_array format before hstacking, it's faster; see
676              # https://github.com/scipy/scipy/issues/20060#issuecomment-1937007039:
677              if sp.issparse(X) and not sp.isspmatrix(X) and X.format == "dok":
678                  X = sp.coo_array(X)
679              X_aug = hstack((X, previous_predictions))
680  
681              feature_predictions, _ = _get_response_values(
682                  estimator,
683                  X_aug,
684                  response_method=chain_method,
685              )
686              Y_feature_chain[:, chain_idx] = feature_predictions
687  
688              output_predictions, _ = _get_response_values(
689                  estimator,
690                  X_aug,
691                  response_method=output_method,
692              )
693              Y_output_chain[:, chain_idx] = output_predictions
694  
695          inv_order = np.empty_like(self.order_)
696          inv_order[self.order_] = np.arange(len(self.order_))
697          Y_output = Y_output_chain[:, inv_order]
698  
699          return Y_output
700  
701      @abstractmethod
702      def fit(self, X, Y, **fit_params):
703          """Fit the model to data matrix X and targets Y. 
704   
705          Parameters 
706          ---------- 
707          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
708              The input data. 
709   
710          Y : array-like of shape (n_samples, n_classes) 
711              The target values. 
712   
713          **fit_params : dict of string -> object 
714              Parameters passed to the `fit` method of each step. 
715   
716              .. versionadded:: 0.23 
717   
718          Returns 
719          ------- 
720          self : object 
721              Returns a fitted instance. 
722          """
723          X, Y = validate_data(self, X, Y, multi_output=True, accept_sparse=True)
724  
725          random_state = check_random_state(self.random_state)
726          self.order_ = self.order
727          if isinstance(self.order_, tuple):
728              self.order_ = np.array(self.order_)
729  
730          if self.order_ is None:
731              self.order_ = np.array(range(Y.shape[1]))
732          elif isinstance(self.order_, str):
733              if self.order_ == "random":
734                  self.order_ = random_state.permutation(Y.shape[1])
735          elif sorted(self.order_) != list(range(Y.shape[1])):
736              raise ValueError("invalid order")
737  
738          self.estimators_ = [clone(self.base_estimator) for _ in range(Y.shape[1])]
739  
740          if self.cv is None:
741              Y_pred_chain = Y[:, self.order_]
742              if sp.issparse(X):
743                  X_aug = sp.hstack((X, Y_pred_chain), format="lil")
744                  X_aug = X_aug.tocsr()
745              else:
746                  X_aug = np.hstack((X, Y_pred_chain))
747  
748          elif sp.issparse(X):
749              # TODO: remove this condition check when the minimum supported scipy version
750              # doesn't support sparse matrices anymore
751              if not sp.isspmatrix(X):
752                  # if `X` is a scipy sparse dok_array, we convert it to a sparse
753                  # coo_array format before hstacking, it's faster; see
754                  # https://github.com/scipy/scipy/issues/20060#issuecomment-1937007039:
755                  if X.format == "dok":
756                      X = sp.coo_array(X)
757                  # in case that `X` is a sparse array we create `Y_pred_chain` as a
758                  # sparse array format:
759                  Y_pred_chain = sp.coo_array((X.shape[0], Y.shape[1]))
760              else:
761                  Y_pred_chain = sp.coo_matrix((X.shape[0], Y.shape[1]))
762              X_aug = sp.hstack((X, Y_pred_chain), format="lil")
763  
764          else:
765              Y_pred_chain = np.zeros((X.shape[0], Y.shape[1]))
766              X_aug = np.hstack((X, Y_pred_chain))
767  
768          del Y_pred_chain
769  
770          if _routing_enabled():
771              routed_params = process_routing(self, "fit", **fit_params)
772          else:
773              routed_params = Bunch(estimator=Bunch(fit=fit_params))
774  
775          if hasattr(self, "chain_method"):
776              chain_method = _check_response_method(
777                  self.base_estimator,
778                  self.chain_method,
779              ).__name__
780              self.chain_method_ = chain_method
781          else:
782              # `RegressorChain` does not have a `chain_method` parameter
783              chain_method = "predict"
784  
785          for chain_idx, estimator in enumerate(self.estimators_):
786              message = self._log_message(
787                  estimator_idx=chain_idx + 1,
788                  n_estimators=len(self.estimators_),
789                  processing_msg=f"Processing order {self.order_[chain_idx]}",
790              )
791              y = Y[:, self.order_[chain_idx]]
792              with _print_elapsed_time("Chain", message):
793                  estimator.fit(
794                      X_aug[:, : (X.shape[1] + chain_idx)],
795                      y,
796                      **routed_params.estimator.fit,
797                  )
798  
799              if self.cv is not None and chain_idx < len(self.estimators_) - 1:
800                  col_idx = X.shape[1] + chain_idx
801                  cv_result = cross_val_predict(
802                      self.base_estimator,
803                      X_aug[:, :col_idx],
804                      y=y,
805                      cv=self.cv,
806                      method=chain_method,
807                  )
808                  # `predict_proba` output is 2D, we use only output for classes[-1]
809                  if cv_result.ndim > 1:
810                      cv_result = cv_result[:, 1]
811                  if sp.issparse(X_aug):
812                      X_aug[:, col_idx] = np.expand_dims(cv_result, 1)
813                  else:
814                      X_aug[:, col_idx] = cv_result
815  
816          return self
817  
818      def predict(self, X):
819          """Predict on the data matrix X using the ClassifierChain model. 
820   
821          Parameters 
822          ---------- 
823          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
824              The input data. 
825   
826          Returns 
827          ------- 
828          Y_pred : array-like of shape (n_samples, n_classes) 
829              The predicted values. 
830          """
831          return self._get_predictions(X, output_method="predict")

839  class ClassifierChain(MetaEstimatorMixin, ClassifierMixin, _BaseChain):
840      """A multi-label model that arranges binary classifiers into a chain. 
841   
842      Each model makes a prediction in the order specified by the chain using 
843      all of the available features provided to the model plus the predictions 
844      of models that are earlier in the chain. 
845   
846      For an example of how to use ``ClassifierChain`` and benefit from its 
847      ensemble, see 
848      :ref:`ClassifierChain on a yeast dataset 
849      <sphx_glr_auto_examples_multioutput_plot_classifier_chain_yeast.py>` example. 
850   
851      Read more in the :ref:`User Guide <classifierchain>`. 
852   
853      .. versionadded:: 0.19 
854   
855      Parameters 
856      ---------- 
857      base_estimator : estimator 
858          The base estimator from which the classifier chain is built. 
859   
860      order : array-like of shape (n_outputs,) or 'random', default=None 
861          If `None`, the order will be determined by the order of columns in 
862          the label matrix Y.:: 
863   
864              order = [0, 1, 2, ..., Y.shape[1] - 1] 
865   
866          The order of the chain can be explicitly set by providing a list of 
867          integers. For example, for a chain of length 5.:: 
868   
869              order = [1, 3, 2, 4, 0] 
870   
871          means that the first model in the chain will make predictions for 
872          column 1 in the Y matrix, the second model will make predictions 
873          for column 3, etc. 
874   
875          If order is `random` a random ordering will be used. 
876   
877      cv : int, cross-validation generator or an iterable, default=None 
878          Determines whether to use cross validated predictions or true 
879          labels for the results of previous estimators in the chain. 
880          Possible inputs for cv are: 
881   
882          - None, to use true labels when fitting, 
883          - integer, to specify the number of folds in a (Stratified)KFold, 
884          - :term:`CV splitter`, 
885          - An iterable yielding (train, test) splits as arrays of indices. 
886   
887      chain_method : {'predict', 'predict_proba', 'predict_log_proba', \ 
888              'decision_function'} or list of such str's, default='predict' 
889   
890          Prediction method to be used by estimators in the chain for 
891          the 'prediction' features of previous estimators in the chain. 
892   
893          - if `str`, name of the method; 
894          - if a list of `str`, provides the method names in order of 
895            preference. The method used corresponds to the first method in 
896            the list that is implemented by `base_estimator`. 
897   
898          .. versionadded:: 1.5 
899   
900      random_state : int, RandomState instance or None, optional (default=None) 
901          If ``order='random'``, determines random number generation for the 
902          chain order. 
903          In addition, it controls the random seed given at each `base_estimator` 
904          at each chaining iteration. Thus, it is only used when `base_estimator` 
905          exposes a `random_state`. 
906          Pass an int for reproducible output across multiple function calls. 
907          See :term:`Glossary <random_state>`. 
908   
909      verbose : bool, default=False 
910          If True, chain progress is output as each model is completed. 
911   
912          .. versionadded:: 1.2 
913   
914      Attributes 
915      ---------- 
916      classes_ : list 
917          A list of arrays of length ``len(estimators_)`` containing the 
918          class labels for each estimator in the chain. 
919   
920      estimators_ : list 
921          A list of clones of base_estimator. 
922   
923      order_ : list 
924          The order of labels in the classifier chain. 
925   
926      chain_method_ : str 
927          Prediction method used by estimators in the chain for the prediction 
928          features. 
929   
930      n_features_in_ : int 
931          Number of features seen during :term:`fit`. Only defined if the 
932          underlying `base_estimator` exposes such an attribute when fit. 
933   
934          .. versionadded:: 0.24 
935   
936      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
937          Names of features seen during :term:`fit`. Defined only when `X` 
938          has feature names that are all strings. 
939   
940          .. versionadded:: 1.0 
941   
942      See Also 
943      -------- 
944      RegressorChain : Equivalent for regression. 
945      MultiOutputClassifier : Classifies each output independently rather than 
946          chaining. 
947   
948      References 
949      ---------- 
950      Jesse Read, Bernhard Pfahringer, Geoff Holmes, Eibe Frank, "Classifier 
951      Chains for Multi-label Classification", 2009. 
952   
953      Examples 
954      -------- 
955      >>> from sklearn.datasets import make_multilabel_classification 
956      >>> from sklearn.linear_model import LogisticRegression 
957      >>> from sklearn.model_selection import train_test_split 
958      >>> from sklearn.multioutput import ClassifierChain 
959      >>> X, Y = make_multilabel_classification( 
960      ...    n_samples=12, n_classes=3, random_state=0 
961      ... ) 
962      >>> X_train, X_test, Y_train, Y_test = train_test_split( 
963      ...    X, Y, random_state=0 
964      ... ) 
965      >>> base_lr = LogisticRegression(solver='lbfgs', random_state=0) 
966      >>> chain = ClassifierChain(base_lr, order='random', random_state=0) 
967      >>> chain.fit(X_train, Y_train).predict(X_test) 
968      array([[1., 1., 0.], 
969             [1., 0., 0.], 
970             [0., 1., 0.]]) 
971      >>> chain.predict_proba(X_test) 
972      array([[0.8387..., 0.9431..., 0.4576...], 
973             [0.8878..., 0.3684..., 0.2640...], 
974             [0.0321..., 0.9935..., 0.0626...]]) 
975      """
