5    import zmq
6    from zmq.green import Poller
7    
8    
9    def device(device_type, isocket, osocket):
10       """Start a zeromq device (gevent-compatible). 
11    
12       Unlike the true zmq.device, this does not release the GIL. 
13    
14       Parameters 
15       ---------- 
16       device_type : (QUEUE, FORWARDER, STREAMER) 
17           The type of device to start (ignored). 
18       isocket : Socket 
19           The Socket instance for the incoming traffic. 
20       osocket : Socket 
21           The Socket instance for the outbound traffic. 
22       """
23       p = Poller()
24       if osocket == -1:
25           osocket = isocket
26       p.register(isocket, zmq.POLLIN)
27       p.register(osocket, zmq.POLLIN)
28   
29       while True:
30           events = dict(p.poll())
31           if isocket in events:
32               osocket.send_multipart(isocket.recv_multipart())
33           if osocket in events:
34               isocket.send_multipart(osocket.recv_multipart())
