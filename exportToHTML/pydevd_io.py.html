7    class IORedirector:
8        """ 
9        This class works to wrap a stream (stdout/stderr) with an additional redirect. 
10       """
11   
12       def __init__(self, original, new_redirect, wrap_buffer=False):
13           """ 
14           :param stream original: 
15               The stream to be wrapped (usually stdout/stderr, but could be None). 
16    
17           :param stream new_redirect: 
18               Usually IOBuf (below). 
19    
20           :param bool wrap_buffer: 
21               Whether to create a buffer attribute (needed to mimick python 3 s 
22               tdout/stderr which has a buffer to write binary data). 
23           """
24           self._lock = ForkSafeLock(rlock=True)
25           self._writing = False
26           self._redirect_to = (original, new_redirect)
27           if wrap_buffer and hasattr(original, "buffer"):
28               self.buffer = IORedirector(original.buffer, new_redirect.buffer, False)
29   
30       def write(self, s):
31           # Note that writing to the original stream may fail for some reasons
32           # (such as trying to write something that's not a string or having it closed).
33           with self._lock:
34               if self._writing:
35                   return
36               self._writing = True
37               try:
38                   for r in self._redirect_to:
39                       if hasattr(r, "write"):
40                           r.write(s)
41               finally:
42                   self._writing = False
43   
44       def isatty(self):
45           for r in self._redirect_to:
46               if hasattr(r, "isatty"):
47                   return r.isatty()
48           return False
49   
50       def flush(self):
51           for r in self._redirect_to:
52               if hasattr(r, "flush"):
53                   r.flush()
54   
55       def __getattr__(self, name):
56           for r in self._redirect_to:
57               if hasattr(r, name):
58                   return getattr(r, name)
59           raise AttributeError(name)

62   class RedirectToPyDBIoMessages(object):
63       def __init__(self, out_ctx, wrap_stream, wrap_buffer, on_write=None):
64           """ 
65           :param out_ctx: 
66               1=stdout and 2=stderr 
67    
68           :param wrap_stream: 
69               Either sys.stdout or sys.stderr. 
70    
71           :param bool wrap_buffer: 
72               If True the buffer attribute (which wraps writing bytes) should be 
73               wrapped. 
74    
75           :param callable(str) on_write: 
76               May be a custom callable to be called when to write something. 
77               If not passed the default implementation will create an io message 
78               and send it through the debugger. 
79           """
80           encoding = getattr(wrap_stream, "encoding", None)
81           if not encoding:
82               encoding = os.environ.get("PYTHONIOENCODING", "utf-8")
83           self.encoding = encoding
84           self._out_ctx = out_ctx
85           if wrap_buffer:
86               self.buffer = RedirectToPyDBIoMessages(out_ctx, wrap_stream, wrap_buffer=False, on_write=on_write)
87           self._on_write = on_write
88   
89       def get_pydb(self):
90           # Note: separate method for mocking on tests.
91           return get_global_debugger()
92   
93       def flush(self):
94           pass  # no-op here
95   
96       def write(self, s):
97           if self._on_write is not None:
98               self._on_write(s)
99               return
100  
101          if s:
102              # Need s in str
103              if isinstance(s, bytes):
104                  s = s.decode(self.encoding, errors="replace")
105  
106              py_db = self.get_pydb()
107              if py_db is not None:
108                  # Note that the actual message contents will be a xml with utf-8, although
109                  # the entry is str on py3 and bytes on py2.
110                  cmd = py_db.cmd_factory.make_io_message(s, self._out_ctx)
111                  if py_db.writer is not None:
112                      py_db.writer.add_command(cmd)

115  class IOBuf:
116      """This class works as a replacement for stdio and stderr. 
117      It is a buffer and when its contents are requested, it will erase what 
118      it has so far so that the next return will not return the same contents again. 
119      """
120  
121      def __init__(self):
122          self.buflist = []
123          import os
124  
125          self.encoding = os.environ.get("PYTHONIOENCODING", "utf-8")
126  
127      def getvalue(self):
128          b = self.buflist
129          self.buflist = []  # clear it
130          return "".join(b)  # bytes on py2, str on py3.
131  
132      def write(self, s):
133          if isinstance(s, bytes):
134              s = s.decode(self.encoding, errors="replace")
135          self.buflist.append(s)
136  
137      def isatty(self):
138          return False
139  
140      def flush(self):
141          pass
142  
143      def empty(self):
144          return len(self.buflist) == 0

154      _lock = ForkSafeLock(rlock=True)
155      _stack_stdout = []
156      _stack_stderr = []
157  
158      _pydevd_stdout_redirect_ = None
159      _pydevd_stderr_redirect_ = None

162  def start_redirect(keep_original_redirection=False, std="stdout", redirect_to=None):
163      """ 
164      @param std: 'stdout', 'stderr', or 'both' 
165      """
166      with _RedirectionsHolder._lock:
167          if redirect_to is None:
168              redirect_to = IOBuf()
169  
170          if std == "both":
171              config_stds = ["stdout", "stderr"]
172          else:
173              config_stds = [std]
174  
175          for std in config_stds:
176              original = getattr(sys, std)
177              stack = getattr(_RedirectionsHolder, "_stack_%s" % std)
178  
179              if keep_original_redirection:
180                  wrap_buffer = True if hasattr(redirect_to, "buffer") else False
181                  new_std_instance = IORedirector(getattr(sys, std), redirect_to, wrap_buffer=wrap_buffer)
182                  setattr(sys, std, new_std_instance)
183              else:
184                  new_std_instance = redirect_to
185                  setattr(sys, std, redirect_to)
186  
187              stack.append(_RedirectInfo(original, new_std_instance))
188  
189          return redirect_to
190  
191  
192  def end_redirect(std="stdout"):
193      with _RedirectionsHolder._lock:
194          if std == "both":
195              config_stds = ["stdout", "stderr"]
196          else:
197              config_stds = [std]
198          for std in config_stds:
199              stack = getattr(_RedirectionsHolder, "_stack_%s" % std)
200              redirect_info = stack.pop()
201              setattr(sys, std, redirect_info.original)

204  def redirect_stream_to_pydb_io_messages(std):
205      """ 
206      :param std: 
207          'stdout' or 'stderr' 
208      """
209      with _RedirectionsHolder._lock:
210          redirect_to_name = "_pydevd_%s_redirect_" % (std,)
211          if getattr(_RedirectionsHolder, redirect_to_name) is None:
212              wrap_buffer = True
213              original = getattr(sys, std)
214  
215              redirect_to = RedirectToPyDBIoMessages(1 if std == "stdout" else 2, original, wrap_buffer)
216              start_redirect(keep_original_redirection=True, std=std, redirect_to=redirect_to)
217  
218              stack = getattr(_RedirectionsHolder, "_stack_%s" % std)
219              setattr(_RedirectionsHolder, redirect_to_name, stack[-1])
220              return True
221  
222          return False

225  def stop_redirect_stream_to_pydb_io_messages(std):
226      """ 
227      :param std: 
228          'stdout' or 'stderr' 
229      """
230      with _RedirectionsHolder._lock:
231          redirect_to_name = "_pydevd_%s_redirect_" % (std,)
232          redirect_info = getattr(_RedirectionsHolder, redirect_to_name)
233          if redirect_info is not None:  # :type redirect_info: _RedirectInfo
234              setattr(_RedirectionsHolder, redirect_to_name, None)
235  
236              stack = getattr(_RedirectionsHolder, "_stack_%s" % std)
237              prev_info = stack.pop()
238  
239              curr = getattr(sys, std)
240              if curr is redirect_info.redirect_to:
241                  setattr(sys, std, redirect_info.original)

244  @contextmanager
245  def redirect_stream_to_pydb_io_messages_context():
246      with _RedirectionsHolder._lock:
247          redirecting = []
248          for std in ("stdout", "stderr"):
249              if redirect_stream_to_pydb_io_messages(std):
250                  redirecting.append(std)
251  
252          try:
253              yield
254          finally:
255              for std in redirecting:
256                  stop_redirect_stream_to_pydb_io_messages(std)
