52   def inputhook_pyglet():
53       """Run the pyglet event loop by processing pending events only. 
54    
55       This keeps processing pending events until stdin is ready.  After 
56       processing all pending events, a call to time.sleep is inserted.  This is 
57       needed, otherwise, CPU usage is at 100%.  This sleep time should be tuned 
58       though for best performance. 
59       """
60       # We need to protect against a user pressing Control-C when IPython is
61       # idle and this is running. We trap KeyboardInterrupt and pass.
62       try:
63           t = clock()
64           while not stdin_ready():
65               pyglet.clock.tick()
66               for window in pyglet.app.windows:
67                   window.switch_to()
68                   window.dispatch_events()
69                   window.dispatch_event("on_draw")
70                   flip(window)
71   
72               # We need to sleep at this point to keep the idle CPU load
73               # low.  However, if sleep to long, GUI response is poor.  As
74               # a compromise, we watch how often GUI events are being processed
75               # and switch between a short and long sleep time.  Here are some
76               # stats useful in helping to tune this.
77               # time    CPU load
78               # 0.001   13%
79               # 0.005   3%
80               # 0.01    1.5%
81               # 0.05    0.5%
82               used_time = clock() - t
83               if used_time > 10.0:
84                   # print 'Sleep for 1 s'  # dbg
85                   time.sleep(1.0)
86               elif used_time > 0.1:
87                   # Few GUI events coming in, so we can sleep longer
88                   # print 'Sleep for 0.05 s'  # dbg
89                   time.sleep(0.05)
90               else:
91                   # Many GUI events coming in, so sleep only very little
92                   time.sleep(0.001)
93       except KeyboardInterrupt:
94           pass
95       return 0
