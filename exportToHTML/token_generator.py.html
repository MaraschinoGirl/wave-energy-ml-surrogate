36   TOKENS = tokenize
37   Token = collections.namedtuple('Token', ('type', 'src', 'start', 'end', 'line'))
38   FORMATTING_TOKENS = (TOKENS.INDENT, TOKENS.DEDENT, TOKENS.NL, TOKENS.NEWLINE,
39                        TOKENS.COMMENT)

42   class TokenGenerator(object):
43     """Helper for sequentially parsing Python source code, token by token. 
44    
45     Holds internal state during parsing, including: 
46     _tokens: List of tokens in the source code, as parsed by `tokenize` module. 
47     _parens: Stack of open parenthesis at the current point in parsing. 
48     _hints: Number of open parentheses, brackets, etc. at the current point. 
49     _scope_stack: Stack containing tuples of nodes where the last parenthesis that 
50       was open is related to one of the nodes on the top of the stack. 
51     _lines: Full lines of the source code. 
52     _i: Index of the last token that was parsed. Initially -1. 
53     _loc: (lineno, column_offset) pair of the position in the source that has been 
54        parsed to. This should be either the start or end of the token at index _i. 
55    
56     Arguments: 
57       ignore_error_tokens: If True, will ignore error tokens. Otherwise, an error 
58         token will cause an exception. This is useful when the source being parsed 
59         contains invalid syntax, e.g. if it is in an fstring context. 
60     """
61   
62     def __init__(self, source, ignore_error_token=False):
63       self.lines = source.splitlines(True)
64       self._tokens = list(_generate_tokens(source, ignore_error_token))
65       self._parens = []
66       self._hints = 0
67       self._scope_stack = []
68       self._len = len(self._tokens)
69       self._i = -1
70       self._loc = self.loc_begin()
71   
72     def chars_consumed(self):
73       return len(self._space_between((1, 0), self._tokens[self._i].end))
74   
75     def loc_begin(self):
76       """Get the start column of the current location parsed to."""
77       if self._i < 0:
78         return (1, 0)
79       return self._tokens[self._i].start
80   
81     def loc_end(self):
82       """Get the end column of the current location parsed to."""
83       if self._i < 0:
84         return (1, 0)
85       return self._tokens[self._i].end
86   
87     def peek(self):
88       """Get the next token without advancing."""
89       if self._i + 1 >= self._len:
90         return None
91       return self._tokens[self._i + 1]
92   
93     def peek_non_whitespace(self):
94       """Get the next non-whitespace token without advancing."""
95       return self.peek_conditional(lambda t: t.type not in FORMATTING_TOKENS)
96   
97     def peek_conditional(self, condition):
98       """Get the next token of the given type without advancing."""
99       return next((t for t in self._tokens[self._i + 1:] if condition(t)), None)
100  
101    def next(self, advance=True):
102      """Consume the next token and optionally advance the current location."""
103      self._i += 1
104      if self._i >= self._len:
105        return None
106      if advance:
107        self._loc = self._tokens[self._i].end
108      return self._tokens[self._i]
109  
110    def rewind(self, amount=1):
111      """Rewind the token iterator."""
112      self._i -= amount
113  
114    def whitespace(self, max_lines=None, comment=False):
115      """Parses whitespace from the current _loc to the next non-whitespace. 
116   
117      Arguments: 
118        max_lines: (optional int) Maximum number of lines to consider as part of 
119          the whitespace. Valid values are None, 0 and 1. 
120        comment: (boolean) If True, look for a trailing comment even when not in 
121          a parenthesized scope. 
122   
123      Pre-condition: 
124        `_loc' represents the point before which everything has been parsed and 
125        after which nothing has been parsed. 
126      Post-condition: 
127        `_loc' is exactly at the character that was parsed to. 
128      """
129      next_token = self.peek()
130      if not comment and next_token and next_token.type == TOKENS.COMMENT:
131        return ''
132      def predicate(token):
133        return (token.type in (TOKENS.INDENT, TOKENS.DEDENT) or
134                token.type == TOKENS.COMMENT and (comment or self._hints) or
135                token.type == TOKENS.ERRORTOKEN and token.src == ' ' or
136                max_lines is None and token.type in (TOKENS.NL, TOKENS.NEWLINE))
137      whitespace = list(self.takewhile(predicate, advance=False))
138      next_token = self.peek()
139  
140      result = ''
141      for tok in itertools.chain(whitespace,
142                                 ((next_token,) if next_token else ())):
143        result += self._space_between(self._loc, tok.start)
144        if tok != next_token:
145          result += tok.src
146          self._loc = tok.end
147        else:
148          self._loc = tok.start
149  
150      # Eat a single newline character
151      if ((max_lines is None or max_lines > 0) and
152          next_token and next_token.type in (TOKENS.NL, TOKENS.NEWLINE)):
153        result += self.next().src
154  
155      return result
156  
157    def block_whitespace(self, indent_level):
158      """Parses whitespace from the current _loc to the end of the block."""
159      # Get the normal suffix lines, but don't advance the token index unless
160      # there is no indentation to account for
161      start_i = self._i
162      full_whitespace = self.whitespace(comment=True)
163      if not indent_level:
164        return full_whitespace
165      self._i = start_i
166  
167      # Trim the full whitespace into only lines that match the indentation level
168      lines = full_whitespace.splitlines(True)
169      try:
170        last_line_idx = next(i for i, line in reversed(list(enumerate(lines)))
171                             if line.startswith(indent_level + '#'))
172      except StopIteration:
173        # No comment lines at the end of this block
174        self._loc = self._tokens[self._i].end
175        return ''
176      lines = lines[:last_line_idx + 1]
177  
178      # Advance the current location to the last token in the lines we've read
179      end_line = self._tokens[self._i].end[0] + 1 + len(lines)
180      list(self.takewhile(lambda tok: tok.start[0] < end_line))
181      self._loc = self._tokens[self._i].end
182      return ''.join(lines)

184    def dots(self, num_dots):
185      """Parse a number of dots. 
186       
187      This is to work around an oddity in python3's tokenizer, which treats three 
188      `.` tokens next to each other in a FromImport's level as an ellipsis. This 
189      parses until the expected number of dots have been seen. 
190      """
191      result = ''
192      dots_seen = 0
193      prev_loc = self._loc
194      while dots_seen < num_dots:
195        tok = self.next()
196        assert tok.src in ('.', '...')
197        result += self._space_between(prev_loc, tok.start) + tok.src
198        dots_seen += tok.src.count('.')
199        prev_loc = self._loc
200      return result

202    def open_scope(self, node, single_paren=False):
203      """Open a parenthesized scope on the given node."""
204      result = ''
205      parens = []
206      start_i = self._i
207      start_loc = prev_loc = self._loc
208  
209      # Eat whitespace or '(' tokens one at a time
210      for tok in self.takewhile(
211          lambda t: t.type in FORMATTING_TOKENS or t.src == '('):
212        # Stores all the code up to and including this token
213        result += self._space_between(prev_loc, tok.start)
214  
215        if tok.src == '(' and single_paren and parens:
216          self.rewind()
217          self._loc = tok.start
218          break
219  
220        result += tok.src
221        if tok.src == '(':
222          # Start a new scope
223          parens.append(result)
224          result = ''
225          start_i = self._i
226          start_loc = self._loc
227        prev_loc = self._loc
228  
229      if parens:
230        # Add any additional whitespace on to the last open-paren
231        next_tok = self.peek()
232        parens[-1] += result + self._space_between(self._loc, next_tok.start)
233        self._loc = next_tok.start
234        # Add each paren onto the stack
235        for paren in parens:
236          self._parens.append(paren)
237          self._scope_stack.append(_scope_helper(node))
238      else:
239        # No parens were encountered, then reset like this method did nothing
240        self._i = start_i
241        self._loc = start_loc

243    def close_scope(self, node, prefix_attr='prefix', suffix_attr='suffix',
244                    trailing_comma=False, single_paren=False):
245      """Close a parenthesized scope on the given node, if one is open."""
246      # Ensures the prefix + suffix are not None
247      if fmt.get(node, prefix_attr) is None:
248        fmt.set(node, prefix_attr, '')
249      if fmt.get(node, suffix_attr) is None:
250        fmt.set(node, suffix_attr, '')
251  
252      if not self._parens or node not in self._scope_stack[-1]:
253        return
254      symbols = {')'}
255      if trailing_comma:
256        symbols.add(',')
257      parsed_to_i = self._i
258      parsed_to_loc = prev_loc = self._loc
259      encountered_paren = False
260      result = ''
261  
262      for tok in self.takewhile(
263          lambda t: t.type in FORMATTING_TOKENS or t.src in symbols):
264        # Consume all space up to this token
265        result += self._space_between(prev_loc, tok.start)
266        if tok.src == ')' and single_paren and encountered_paren:
267          self.rewind()
268          parsed_to_i = self._i
269          parsed_to_loc = tok.start
270          fmt.append(node, suffix_attr, result)
271          break
272  
273        # Consume the token itself
274        result += tok.src
275  
276        if tok.src == ')':
277          # Close out the open scope
278          encountered_paren = True
279          self._scope_stack.pop()
280          fmt.prepend(node, prefix_attr, self._parens.pop())
281          fmt.append(node, suffix_attr, result)
282          result = ''
283          parsed_to_i = self._i
284          parsed_to_loc = tok.end
285          if not self._parens or node not in self._scope_stack[-1]:
286            break
287        prev_loc = tok.end
288  
289      # Reset back to the last place where we parsed anything
290      self._i = parsed_to_i
291      self._loc = parsed_to_loc

293    def hint_open(self):
294      """Indicates opening a group of parentheses or brackets."""
295      self._hints += 1
296  
297    def hint_closed(self):
298      """Indicates closing a group of parentheses or brackets."""
299      self._hints -= 1
300      if self._hints < 0:
301        raise ValueError('Hint value negative')

303    @contextlib.contextmanager
304    def scope(self, node, attr=None, trailing_comma=False):
305      """Context manager to handle a parenthesized scope."""
306      self.open_scope(node, single_paren=(attr is not None))
307      yield
308      if attr:
309        self.close_scope(node, prefix_attr=attr + '_prefix',
310                         suffix_attr=attr + '_suffix',
311                         trailing_comma=trailing_comma,
312                         single_paren=True)
313      else:
314        self.close_scope(node, trailing_comma=trailing_comma)
315  
316    def is_in_scope(self):
317      """Return True iff there is a scope open."""
318      return self._parens or self._hints

320    def str(self):
321      """Parse a full string literal from the input."""
322      def predicate(token):
323        return (token.type in (TOKENS.STRING, TOKENS.COMMENT) or
324                self.is_in_scope() and token.type in (TOKENS.NL, TOKENS.NEWLINE))
325  
326      return self.eat_tokens(predicate)
327  
328    def eat_tokens(self, predicate):
329      """Parse input from tokens while a given condition is met."""
330      content = ''
331      prev_loc = self._loc
332      tok = None
333      for tok in self.takewhile(predicate, advance=False):
334        content += self._space_between(prev_loc, tok.start)
335        content += tok.src
336        prev_loc = tok.end
337  
338      if tok:
339        self._loc = tok.end
340      return content

342    def fstr(self):
343      """Parses an fstring, including subexpressions. 
344   
345      Returns: 
346        A generator function which, when repeatedly reads a chunk of the fstring 
347        up until the next subexpression and yields that chunk, plus a new token 
348        generator to use to parse the subexpression. The subexpressions in the 
349        original fstring data are replaced by placeholders to make it possible to 
350        fill them in with new values, if desired. 
351      """
352      def fstr_parser():
353        # Reads the whole fstring as a string, then parses it char by char
354        if self.peek_non_whitespace().type == TOKENS.STRING:
355          # Normal fstrings are one ore more STRING tokens, maybe mixed with
356          # spaces, e.g.: f"Hello, {name}"
357          str_content = self.str()
358        else:
359          # Format specifiers in fstrings are also JoinedStr nodes, but these are
360          # arbitrary expressions, e.g. in: f"{value:{width}.{precision}}", the
361          # format specifier is an fstring: "{width}.{precision}" but these are
362          # not STRING tokens.
363          def fstr_eater(tok):
364            if tok.type == TOKENS.OP and tok.src == '}':
365              if fstr_eater.level <= 0:
366                return False
367              fstr_eater.level -= 1
368            if tok.type == TOKENS.OP and tok.src == '{':
369              fstr_eater.level += 1
370            return True
371          fstr_eater.level = 0
372          str_content = self.eat_tokens(fstr_eater)
373  
374        indexed_chars = enumerate(str_content)
375        val_idx = 0
376        i = -1
377        result = ''
378        while i < len(str_content) - 1:
379          i, c = next(indexed_chars)
380          result += c
381  
382          # When an open bracket is encountered, start parsing a subexpression
383          if c == '{':
384            # First check if this is part of an escape sequence
385            # (f"{{" is used to escape a bracket literal)
386            nexti, nextc = next(indexed_chars)
387            if nextc == '{':
388              result += c
389              continue
390            indexed_chars = itertools.chain([(nexti, nextc)], indexed_chars)
391  
392            # Add a placeholder onto the result
393            result += fstring_utils.placeholder(val_idx) + '}'
394            val_idx += 1
395  
396            # Yield a new token generator to parse the subexpression only
397            tg = TokenGenerator(str_content[i+1:], ignore_error_token=True)
398            yield (result, tg)
399            result = ''
400  
401            # Skip the number of characters consumed by the subexpression
402            for tg_i in range(tg.chars_consumed()):
403              i, c = next(indexed_chars)
404  
405            # Eat up to and including the close bracket
406            i, c = next(indexed_chars)
407            while c != '}':
408              i, c = next(indexed_chars)
409        # Yield the rest of the fstring, when done
410        yield (result, None)
411      return fstr_parser

413    def _space_between(self, start_loc, end_loc):
414      """Parse the space between a location and the next token"""
415      if start_loc > end_loc:
416        raise ValueError('start_loc > end_loc', start_loc, end_loc)
417      if start_loc[0] > len(self.lines):
418        return ''
419  
420      prev_row, prev_col = start_loc
421      end_row, end_col = end_loc
422      if prev_row == end_row:
423        return self.lines[prev_row - 1][prev_col:end_col]
424  
425      return ''.join(itertools.chain(
426          (self.lines[prev_row - 1][prev_col:],),
427          self.lines[prev_row:end_row - 1],
428          (self.lines[end_row - 1][:end_col],) if end_col > 0 else '',
429      ))

431    def next_name(self):
432      """Parse the next name token."""
433      last_i = self._i
434      def predicate(token):
435        return token.type != TOKENS.NAME
436  
437      unused_tokens = list(self.takewhile(predicate, advance=False))
438      result = self.next(advance=False)
439      self._i = last_i
440      return result

442    def next_of_type(self, token_type):
443      """Parse a token of the given type and return it."""
444      token = self.next()
445      if token.type != token_type:
446        raise ValueError("Expected %r but found %r\nline %d: %s" % (
447            tokenize.tok_name[token_type], token.src, token.start[0],
448            self.lines[token.start[0] - 1]))
449      return token

451    def takewhile(self, condition, advance=True):
452      """Parse tokens as long as a condition holds on the next token."""
453      prev_loc = self._loc
454      token = self.next(advance=advance)
455      while token is not None and condition(token):
456        yield token
457        prev_loc = self._loc
458        token = self.next(advance=advance)
459      self.rewind()
460      self._loc = prev_loc

463  def _scope_helper(node):
464    """Get the closure of nodes that could begin a scope at this point. 
465   
466    For instance, when encountering a `(` when parsing a BinOp node, this could 
467    indicate that the BinOp itself is parenthesized OR that the BinOp's left node 
468    could be parenthesized. 
469   
470    E.g.: (a + b * c)   or   (a + b) * c   or   (a) + b * c 
471          ^                  ^                  ^ 
472   
473    Arguments: 
474      node: (ast.AST) Node encountered when opening a scope. 
475   
476    Returns: 
477      A closure of nodes which that scope might apply to. 
478    """
479    if isinstance(node, ast.Attribute):
480      return (node,) + _scope_helper(node.value)
481    if isinstance(node, ast.Subscript):
482      return (node,) + _scope_helper(node.value)
483    if isinstance(node, ast.Assign):
484      return (node,) + _scope_helper(node.targets[0])
485    if isinstance(node, ast.AugAssign):
486      return (node,) + _scope_helper(node.target)
487    if isinstance(node, ast.Expr):
488      return (node,) + _scope_helper(node.value)
489    if isinstance(node, ast.Compare):
490      return (node,) + _scope_helper(node.left)
491    if isinstance(node, ast.BoolOp):
492      return (node,) + _scope_helper(node.values[0])
493    if isinstance(node, ast.BinOp):
494      return (node,) + _scope_helper(node.left)
495    if isinstance(node, ast.Tuple) and node.elts:
496      return (node,) + _scope_helper(node.elts[0])
497    if isinstance(node, ast.Call):
498      return (node,) + _scope_helper(node.func)
499    if isinstance(node, ast.GeneratorExp):
500      return (node,) + _scope_helper(node.elt)
501    if isinstance(node, ast.IfExp):
502      return (node,) + _scope_helper(node.body)
503    return (node,)

506  def _generate_tokens(source, ignore_error_token=False):
507    token_generator = tokenize.generate_tokens(StringIO(source).readline)
508    try:
509      for tok in token_generator:
510        yield Token(*tok) 
511    except tokenize.TokenError:
512      if not ignore_error_token:
513        raise
