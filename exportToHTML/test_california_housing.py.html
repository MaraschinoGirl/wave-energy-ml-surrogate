12   def test_fetch(fetch_california_housing_fxt):

15       assert (20640,) == data.target.shape
16       assert data.DESCR.startswith(".. _california_housing_dataset:")

18       # test return_X_y option
19       fetch_func = partial(fetch_california_housing_fxt)
20       check_return_X_y(data, fetch_func)

23   def test_fetch_asframe(fetch_california_housing_fxt):
24       pd = pytest.importorskip("pandas")
25       bunch = fetch_california_housing_fxt(as_frame=True)
26       frame = bunch.frame
27       assert hasattr(bunch, "frame") is True
28       assert frame.shape == (20640, 9)
29       assert isinstance(bunch.data, pd.DataFrame)
30       assert isinstance(bunch.target, pd.Series)
