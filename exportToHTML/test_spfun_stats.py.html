18       def test2(self):
19           # A test of the identity
20           #     Gamma_2(a) = sqrt(pi) * Gamma(a) * Gamma(a - 0.5)
21           a = np.array([2.5, 10.0])
22           result = multigammaln(a, 2)
23           expected = np.log(np.sqrt(np.pi)) + gammaln(a) + gammaln(a - 0.5)
24           assert_almost_equal(result, expected)
