9    @keras_export("keras.layers.InputLayer")
10   class InputLayer(Layer):
11       def __init__(
12           self,
13           shape=None,
14           batch_size=None,
15           dtype=None,
16           sparse=None,
17           ragged=None,
18           batch_shape=None,
19           input_tensor=None,
20           optional=False,
21           name=None,
22           **kwargs,
23       ):
24           super().__init__(name=name)
25   
26           if "input_shape" in kwargs:
27               warnings.warn(
28                   "Argument `input_shape` is deprecated. Use `shape` instead."
29               )
30               shape = kwargs.pop("input_shape")
31           if "batch_input_shape" in kwargs:
32               batch_shape = kwargs.pop("batch_input_shape")
33   
34           if input_tensor is not None:
35               if not isinstance(input_tensor, backend.KerasTensor):
36                   raise ValueError(
37                       "Argument `input_tensor` must be a KerasTensor. "
38                       f"Received invalid type: input_tensor={input_tensor} "
39                       f"(of type {type(input_tensor)})"
40                   )
41               if batch_size is not None:
42                   if (
43                       len(input_tensor.shape) < 1
44                       or input_tensor.shape[0] != batch_size
45                   ):
46                       raise ValueError(
47                           "When providing the `input_tensor` argument, you "
48                           "cannot provide an incompatible `batch_size` argument."
49                       )
50               if shape is not None:
51                   if (
52                       len(shape) != len(input_tensor.shape) - 1
53                       or shape != input_tensor.shape[1:]
54                   ):
55                       raise ValueError(
56                           "When providing the `input_tensor` argument, you "
57                           "cannot provide an incompatible `shape` argument."
58                       )
59               if batch_shape is not None and batch_shape != input_tensor.shape:
60                   raise ValueError(
61                       "When providing the `input_tensor` argument, you "
62                       "cannot provide an incompatible `batch_shape` argument."
63                   )
64               if dtype is not None and input_tensor.dtype != dtype:
65                   raise ValueError(
66                       "When providing the `input_tensor` argument, you "
67                       "cannot provide an incompatible `dtype` argument."
68                   )
69               if sparse is not None and input_tensor.sparse != sparse:
70                   raise ValueError(
71                       "When providing the `input_tensor` argument, you "
72                       "cannot provide an incompatible `sparse` argument."
73                   )
74               batch_shape = input_tensor.shape
75               dtype = input_tensor.dtype
76               sparse = input_tensor.sparse
77           else:
78               if shape is not None and batch_shape is not None:
79                   raise ValueError(
80                       "You cannot pass both `shape` and `batch_shape` at the "
81                       "same time."
82                   )
83               if batch_size is not None and batch_shape is not None:
84                   raise ValueError(
85                       "You cannot pass both `batch_size` and `batch_shape` "
86                       "at the same time."
87                   )
88               if shape is None and batch_shape is None:
89                   raise ValueError("You must pass a `shape` argument.")
90   
91               if shape is not None:
92                   shape = backend.standardize_shape(shape)
93                   batch_shape = (batch_size,) + shape
94   
95           self._batch_shape = backend.standardize_shape(batch_shape)
96           self._dtype = backend.standardize_dtype(dtype)
97           self.sparse = bool(sparse)
98           if self.sparse and not backend.SUPPORTS_SPARSE_TENSORS:
99               raise ValueError(
100                  f"`sparse=True` is not supported with the {backend.backend()} "
101                  "backend"
102              )
103          self.ragged = bool(ragged)
104          if self.ragged and not backend.SUPPORTS_RAGGED_TENSORS:
105              raise ValueError(
106                  f"`ragged=True` is not supported with the {backend.backend()} "
107                  "backend"
108              )
109  
110          if input_tensor is None:
111              input_tensor = backend.KerasTensor(
112                  shape=batch_shape,
113                  dtype=dtype,
114                  sparse=sparse,
115                  ragged=ragged,
116                  name=name,
117              )
118          self._input_tensor = input_tensor
119          Node(operation=self, call_args=(), call_kwargs={}, outputs=input_tensor)
120          self.built = True
121          self.optional = optional
122  
123      def call(self):
124          return
125  
126      @property
127      def batch_shape(self):
128          return self._batch_shape
129  
130      @property
131      def dtype(self):
132          return self._dtype
133  
134      def get_config(self):
135          return {
136              "batch_shape": self.batch_shape,
137              "dtype": self.dtype,
138              "sparse": self.sparse,
139              "ragged": self.ragged,
140              "name": self.name,
141          }

144  @keras_export(["keras.layers.Input", "keras.Input"])
145  def Input(
146      shape=None,
147      batch_size=None,
148      dtype=None,
149      sparse=None,
150      ragged=None,
151      batch_shape=None,
152      name=None,
153      tensor=None,
154      optional=False,
155  ):
156      """Used to instantiate a Keras tensor. 
157   
158      A Keras tensor is a symbolic tensor-like object, which we augment with 
159      certain attributes that allow us to build a Keras model just by knowing the 
160      inputs and outputs of the model. 
161   
162      For instance, if `a`, `b` and `c` are Keras tensors, 
163      it becomes possible to do: 
164      `model = Model(input=[a, b], output=c)` 
165   
166      Args: 
167          shape: A shape tuple (tuple of integers or `None` objects), 
168              not including the batch size. 
169              For instance, `shape=(32,)` indicates that the expected input 
170              will be batches of 32-dimensional vectors. Elements of this tuple 
171              can be `None`; `None` elements represent dimensions where the shape 
172              is not known and may vary (e.g. sequence length). 
173          batch_size: Optional static batch size (integer). 
174          dtype: The data type expected by the input, as a string 
175              (e.g. `"float32"`, `"int32"`...) 
176          sparse: A boolean specifying whether the expected input will be sparse 
177              tensors. Note that, if `sparse` is `False`, sparse tensors can still 
178              be passed into the input - they will be densified with a default 
179              value of 0. This feature is only supported with the TensorFlow and 
180              the JAX backends. Defaults to `False`. 
181          ragged: A boolean specifying whether the expected input will be ragged 
182              tensors. Note that, if `ragged` is `False`, ragged tensors can still 
183              be passed into the input - they will be densified with a default 
184              value of 0. This feature is only supported with the TensorFlow 
185              backend. Defaults to `False`. 
186          batch_shape: Optional shape tuple (tuple of integers or `None` objects), 
187              including the batch size. 
188          name: Optional name string for the layer. 
189              Should be unique in a model (do not reuse the same name twice). 
190              It will be autogenerated if it isn't provided. 
191          tensor: Optional existing tensor to wrap into the `Input` layer. 
192              If set, the layer will use this tensor rather 
193              than creating a new placeholder tensor. 
194          optional: Boolean, whether the input is optional or not. 
195              An optional input can accept `None` values. 
196   
197      Returns: 
198        A Keras tensor. 
199   
200      Example: 
201   
202      ```python 
203      # This is a logistic regression in Keras 
204      x = Input(shape=(32,)) 
205      y = Dense(16, activation='softmax')(x) 
206      model = Model(x, y) 
207      ``` 
208      """
209      layer = InputLayer(
210          shape=shape,
211          batch_size=batch_size,
212          dtype=dtype,
213          sparse=sparse,
214          ragged=ragged,
215          batch_shape=batch_shape,
216          name=name,
217          input_tensor=tensor,
218          optional=optional,
219      )
220      return layer.output
