CreateFileTransacted_MiniVersion.py 
1    """ 
2    This demonstrates the creation of miniversions of a file during a transaction. 
3    The FSCTL_TXFS_CREATE_MINIVERSION control code saves any changes to a new 
4    miniversion (effectively a savepoint within a transaction). 
5    """
6    
7    import os
8    import struct
9    
10   import win32api
11   import win32con
12   import win32file
13   import win32transaction
14   import winerror
15   import winioctlcon
16   
17   
18   def demo():
19       """ 
20       Definition of buffer used with FSCTL_TXFS_CREATE_MINIVERSION: 
21       typedef struct _TXFS_CREATE_MINIVERSION_INFO{ 
22           USHORT StructureVersion; 
23           USHORT StructureLength; 
24           ULONG BaseVersion; 
25           USHORT MiniVersion;} 
26       """
27       buf_fmt = "HHLH0L"  ## buffer size must include struct padding
28       buf_size = struct.calcsize(buf_fmt)
29   
30       tempdir = win32api.GetTempPath()
31       tempfile = win32api.GetTempFileName(tempdir, "cft")[0]
32       print("Demonstrating transactions on tempfile", tempfile)
33       f = open(tempfile, "w")
34       f.write("This is original file.\n")
35       f.close()
36   
37       trans = win32transaction.CreateTransaction(
38           Description="Test creating miniversions of a file"
39       )
40       hfile = win32file.CreateFileW(
41           tempfile,
42           win32con.GENERIC_READ | win32con.GENERIC_WRITE,
43           win32con.FILE_SHARE_READ | win32con.FILE_SHARE_WRITE,
44           None,
45           win32con.OPEN_EXISTING,
46           0,
47           None,
48           Transaction=trans,
49       )
50   
51       win32file.WriteFile(hfile, b"This is first miniversion.\n")
52       buf = win32file.DeviceIoControl(
53           hfile, winioctlcon.FSCTL_TXFS_CREATE_MINIVERSION, None, buf_size, None
54       )
55       struct_ver, struct_len, base_ver, ver_1 = struct.unpack(buf_fmt, buf)
56   
57       win32file.SetFilePointer(hfile, 0, win32con.FILE_BEGIN)
58       win32file.WriteFile(hfile, b"This is second miniversion!\n")
59       buf = win32file.DeviceIoControl(
60           hfile, winioctlcon.FSCTL_TXFS_CREATE_MINIVERSION, None, buf_size, None
61       )
62       struct_ver, struct_len, base_ver, ver_2 = struct.unpack(buf_fmt, buf)
63       hfile.Close()
64   
65       ## miniversions can't be opened with write access
66       hfile_0 = win32file.CreateFileW(
67           tempfile,
68           win32con.GENERIC_READ,
69           win32con.FILE_SHARE_READ | win32con.FILE_SHARE_WRITE,
70           None,
71           win32con.OPEN_EXISTING,
72           0,
73           None,
74           Transaction=trans,
75           MiniVersion=base_ver,
76       )
77       print("version:", base_ver, win32file.ReadFile(hfile_0, 100))
78       hfile_0.Close()
79   
80       hfile_1 = win32file.CreateFileW(
81           tempfile,
82           win32con.GENERIC_READ,
83           win32con.FILE_SHARE_READ | win32con.FILE_SHARE_WRITE,
84           None,
85           win32con.OPEN_EXISTING,
86           0,
87           None,
88           Transaction=trans,
89           MiniVersion=ver_1,
90       )
91       print("version:", ver_1, win32file.ReadFile(hfile_1, 100))
92       hfile_1.Close()
93   
94       hfile_2 = win32file.CreateFileW(
95           tempfile,
96           win32con.GENERIC_READ,
97           win32con.FILE_SHARE_READ | win32con.FILE_SHARE_WRITE,
98           None,
99           win32con.OPEN_EXISTING,
100          0,
101          None,
102          Transaction=trans,
103          MiniVersion=ver_2,
104      )
105      print("version:", ver_2, win32file.ReadFile(hfile_2, 100))
106      hfile_2.Close()
107  
108      ## MiniVersions are destroyed when transaction is committed or rolled back
109      win32transaction.CommitTransaction(trans)
110  
111      os.unlink(tempfile)
112  
113  
114  if __name__ == "__main__":
115      # When run on CI, this fails with NOT_SUPPORTED, so don't have that cause "failure"
116      try:
117          demo()
118      except win32file.error as e:
119          if e.winerror == winerror.ERROR_NOT_SUPPORTED:
120              print("These features are not supported by this filesystem.")
121          else:
122              raise
