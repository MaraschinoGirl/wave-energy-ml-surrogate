25   @dataclass
26   class DunnettResult:
27       """Result object returned by `scipy.stats.dunnett`. 
28    
29       Attributes 
30       ---------- 
31       statistic : float ndarray 
32           The computed statistic of the test for each comparison. The element 
33           at index ``i`` is the statistic for the comparison between 
34           groups ``i`` and the control. 
35       pvalue : float ndarray 
36           The computed p-value of the test for each comparison. The element 
37           at index ``i`` is the p-value for the comparison between 
38           group ``i`` and the control. 
39       """
40       statistic: np.ndarray
41       pvalue: np.ndarray
42       _alternative: Literal['two-sided', 'less', 'greater'] = field(repr=False)
43       _rho: np.ndarray = field(repr=False)
44       _df: int = field(repr=False)
45       _std: float = field(repr=False)
46       _mean_samples: np.ndarray = field(repr=False)
47       _mean_control: np.ndarray = field(repr=False)
48       _n_samples: np.ndarray = field(repr=False)
49       _n_control: int = field(repr=False)
50       _rng: SeedType = field(repr=False)
51       _ci: ConfidenceInterval | None = field(default=None, repr=False)
52       _ci_cl: DecimalNumber | None = field(default=None, repr=False)
53   
54       def __str__(self):
55           # Note: `__str__` prints the confidence intervals from the most
56           # recent call to `confidence_interval`. If it has not been called,
57           # it will be called with the default CL of .95.
58           if self._ci is None:
59               self.confidence_interval(confidence_level=.95)
60           s = (
61               "Dunnett's test"
62               f" ({self._ci_cl*100:.1f}% Confidence Interval)\n"
63               "Comparison               Statistic  p-value  Lower CI  Upper CI\n"
64           )
65           for i in range(self.pvalue.size):
66               s += (f" (Sample {i} - Control) {self.statistic[i]:>10.3f}"
67                     f"{self.pvalue[i]:>10.3f}"
68                     f"{self._ci.low[i]:>10.3f}"
69                     f"{self._ci.high[i]:>10.3f}\n")
70   
71           return s
72   
73       def _allowance(
74           self, confidence_level: DecimalNumber = 0.95, tol: DecimalNumber = 1e-3
75       ) -> float:
76           """Allowance. 
77    
78           It is the quantity to add/subtract from the observed difference 
79           between the means of observed groups and the mean of the control 
80           group. The result gives confidence limits. 
81    
82           Parameters 
83           ---------- 
84           confidence_level : float, optional 
85               Confidence level for the computed confidence interval. 
86               Default is .95. 
87           tol : float, optional 
88               A tolerance for numerical optimization: the allowance will produce 
89               a confidence within ``10*tol*(1 - confidence_level)`` of the 
90               specified level, or a warning will be emitted. Tight tolerances 
91               may be impractical due to noisy evaluation of the objective. 
92               Default is 1e-3. 
93    
94           Returns 
95           ------- 
96           allowance : float 
97               Allowance around the mean. 
98           """
99           alpha = 1 - confidence_level
100  
101          def pvalue_from_stat(statistic):
102              statistic = np.array(statistic)
103              sf = _pvalue_dunnett(
104                  rho=self._rho, df=self._df,
105                  statistic=statistic, alternative=self._alternative,
106                  rng=self._rng
107              )
108              return abs(sf - alpha)/alpha
109  
110          # Evaluation of `pvalue_from_stat` is noisy due to the use of RQMC to
111          # evaluate `multivariate_t.cdf`. `minimize_scalar` is not designed
112          # to tolerate a noisy objective function and may fail to find the
113          # minimum accurately. We mitigate this possibility with the validation
114          # step below, but implementation of a noise-tolerant root finder or
115          # minimizer would be a welcome enhancement. See gh-18150.
116          res = minimize_scalar(pvalue_from_stat, method='brent', tol=tol)
117          critical_value = res.x
118  
119          # validation
120          # tol*10 because tol=1e-3 means we tolerate a 1% change at most
121          if res.success is False or res.fun >= tol*10:
122              warnings.warn(
123                  "Computation of the confidence interval did not converge to "
124                  "the desired level. The confidence level corresponding with "
125                  f"the returned interval is approximately {alpha*(1+res.fun)}.",
126                  stacklevel=3
127              )
128  
129          # From [1] p. 1101 between (1) and (3)
130          allowance = critical_value*self._std*np.sqrt(
131              1/self._n_samples + 1/self._n_control
132          )
133          return abs(allowance)
134  
135      def confidence_interval(
136          self, confidence_level: DecimalNumber = 0.95
137      ) -> ConfidenceInterval:
138          """Compute the confidence interval for the specified confidence level. 
139   
140          Parameters 
141          ---------- 
142          confidence_level : float, optional 
143              Confidence level for the computed confidence interval. 
144              Default is .95. 
145   
146          Returns 
147          ------- 
148          ci : ``ConfidenceInterval`` object 
149              The object has attributes ``low`` and ``high`` that hold the 
150              lower and upper bounds of the confidence intervals for each 
151              comparison. The high and low values are accessible for each 
152              comparison at index ``i`` for each group ``i``. 
153   
154          """
155          # check to see if the supplied confidence level matches that of the
156          # previously computed CI.
157          if (self._ci is not None) and (confidence_level == self._ci_cl):
158              return self._ci
159  
160          if not (0 < confidence_level < 1):
161              raise ValueError("Confidence level must be between 0 and 1.")
162  
163          allowance = self._allowance(confidence_level=confidence_level)
164          diff_means = self._mean_samples - self._mean_control
165  
166          low = diff_means-allowance
167          high = diff_means+allowance
168  
169          if self._alternative == 'greater':
170              high = [np.inf] * len(diff_means)
171          elif self._alternative == 'less':
172              low = [-np.inf] * len(diff_means)
173  
174          self._ci_cl = confidence_level
175          self._ci = ConfidenceInterval(
176              low=low,
177              high=high
178          )
179          return self._ci

182  @_transition_to_rng('random_state', replace_doc=False)
183  def dunnett(
184      *samples: "npt.ArrayLike",  # noqa: D417
185      control: "npt.ArrayLike",
186      alternative: Literal['two-sided', 'less', 'greater'] = "two-sided",
187      rng: SeedType = None
188  ) -> DunnettResult:
189      """Dunnett's test: multiple comparisons of means against a control group. 
190   
191      This is an implementation of Dunnett's original, single-step test as 
192      described in [1]_. 
193   
194      Parameters 
195      ---------- 
196      sample1, sample2, ... : 1D array_like 
197          The sample measurements for each experimental group. 
198      control : 1D array_like 
199          The sample measurements for the control group. 
200      alternative : {'two-sided', 'less', 'greater'}, optional 
201          Defines the alternative hypothesis. 
202   
203          The null hypothesis is that the means of the distributions underlying 
204          the samples and control are equal. The following alternative 
205          hypotheses are available (default is 'two-sided'): 
206   
207          * 'two-sided': the means of the distributions underlying the samples 
208            and control are unequal. 
209          * 'less': the means of the distributions underlying the samples 
210            are less than the mean of the distribution underlying the control. 
211          * 'greater': the means of the distributions underlying the 
212            samples are greater than the mean of the distribution underlying 
213            the control. 
214      rng : `numpy.random.Generator`, optional 
215          Pseudorandom number generator state. When `rng` is None, a new 
216          `numpy.random.Generator` is created using entropy from the 
217          operating system. Types other than `numpy.random.Generator` are 
218          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
219   
220          .. versionchanged:: 1.15.0 
221   
222              As part of the `SPEC-007 <https://scientific-python.org/specs/spec-0007/>`_ 
223              transition from use of `numpy.random.RandomState` to 
224              `numpy.random.Generator`, this keyword was changed from `random_state` to 
225              `rng`. For an interim period, both keywords will continue to work, although 
226              only one may be specified at a time. After the interim period, function 
227              calls using the `random_state` keyword will emit warnings. Following a 
228              deprecation period, the `random_state` keyword will be removed. 
229   
230      Returns 
231      ------- 
232      res : `~scipy.stats._result_classes.DunnettResult` 
233          An object containing attributes: 
234   
235          statistic : float ndarray 
236              The computed statistic of the test for each comparison. The element 
237              at index ``i`` is the statistic for the comparison between 
238              groups ``i`` and the control. 
239          pvalue : float ndarray 
240              The computed p-value of the test for each comparison. The element 
241              at index ``i`` is the p-value for the comparison between 
242              group ``i`` and the control. 
243   
244          And the following method: 
245   
246          confidence_interval(confidence_level=0.95) : 
247              Compute the difference in means of the groups 
248              with the control +- the allowance. 
249   
250      See Also 
251      -------- 
252      tukey_hsd : performs pairwise comparison of means. 
253      :ref:`hypothesis_dunnett` : Extended example 
254   
255      Notes 
256      ----- 
257      Like the independent-sample t-test, Dunnett's test [1]_ is used to make 
258      inferences about the means of distributions from which samples were drawn. 
259      However, when multiple t-tests are performed at a fixed significance level, 
260      the "family-wise error rate" - the probability of incorrectly rejecting the 
261      null hypothesis in at least one test - will exceed the significance level. 
262      Dunnett's test is designed to perform multiple comparisons while 
263      controlling the family-wise error rate. 
264   
265      Dunnett's test compares the means of multiple experimental groups 
266      against a single control group. Tukey's Honestly Significant Difference Test 
267      is another multiple-comparison test that controls the family-wise error 
268      rate, but `tukey_hsd` performs *all* pairwise comparisons between groups. 
269      When pairwise comparisons between experimental groups are not needed, 
270      Dunnett's test is preferable due to its higher power. 
271   
272      The use of this test relies on several assumptions. 
273   
274      1. The observations are independent within and among groups. 
275      2. The observations within each group are normally distributed. 
276      3. The distributions from which the samples are drawn have the same finite 
277         variance. 
278   
279      References 
280      ---------- 
281      .. [1] Dunnett, Charles W. (1955) "A Multiple Comparison Procedure for 
282             Comparing Several Treatments with a Control." Journal of the American 
283             Statistical Association, 50:272, 1096-1121, 
284             :doi:`10.1080/01621459.1955.10501294` 
285      .. [2] Thomson, M. L., & Short, M. D. (1969). Mucociliary function in 
286             health, chronic obstructive airway disease, and asbestosis. Journal 
287             of applied physiology, 26(5), 535-539. 
288             :doi:`10.1152/jappl.1969.26.5.535` 
289   
290      Examples 
291      -------- 
292      We'll use data from [2]_, Table 1. The null hypothesis is that the means of 
293      the distributions underlying the samples and control are equal. 
294   
295      First, we test that the means of the distributions underlying the samples 
296      and control are unequal (``alternative='two-sided'``, the default). 
297   
298      >>> import numpy as np 
299      >>> from scipy.stats import dunnett 
300      >>> samples = [[3.8, 2.7, 4.0, 2.4], [2.8, 3.4, 3.7, 2.2, 2.0]] 
301      >>> control = [2.9, 3.0, 2.5, 2.6, 3.2] 
302      >>> res = dunnett(*samples, control=control) 
303      >>> res.statistic 
304      array([ 0.90874545, -0.05007117]) 
305      >>> res.pvalue 
306      array([0.58325114, 0.99819341]) 
307   
308      Now, we test that the means of the distributions underlying the samples are 
309      greater than the mean of the distribution underlying the control. 
310   
311      >>> res = dunnett(*samples, control=control, alternative='greater') 
312      >>> res.statistic 
313      array([ 0.90874545, -0.05007117]) 
314      >>> res.pvalue 
315      array([0.30230596, 0.69115597]) 
316   
317      For a more detailed example, see :ref:`hypothesis_dunnett`. 
318      """
319      samples_, control_, rng = _iv_dunnett(
320          samples=samples, control=control,
321          alternative=alternative, rng=rng
322      )
323  
324      rho, df, n_group, n_samples, n_control = _params_dunnett(
325          samples=samples_, control=control_
326      )
327  
328      statistic, std, mean_control, mean_samples = _statistic_dunnett(
329          samples_, control_, df, n_samples, n_control
330      )
331  
332      pvalue = _pvalue_dunnett(
333          rho=rho, df=df, statistic=statistic, alternative=alternative, rng=rng
334      )
335  
336      return DunnettResult(
337          statistic=statistic, pvalue=pvalue,
338          _alternative=alternative,
339          _rho=rho, _df=df, _std=std,
340          _mean_samples=mean_samples,
341          _mean_control=mean_control,
342          _n_samples=n_samples,
343          _n_control=n_control,
344          _rng=rng
345      )

348  def _iv_dunnett(
349      samples: Sequence["npt.ArrayLike"],
350      control: "npt.ArrayLike",
351      alternative: Literal['two-sided', 'less', 'greater'],
352      rng: SeedType
353  ) -> tuple[list[np.ndarray], np.ndarray, SeedType]:
354      """Input validation for Dunnett's test."""
355      rng = check_random_state(rng)
356  
357      if alternative not in {'two-sided', 'less', 'greater'}:
358          raise ValueError(
359              "alternative must be 'less', 'greater' or 'two-sided'"
360          )
361  
362      ndim_msg = "Control and samples groups must be 1D arrays"
363      n_obs_msg = "Control and samples groups must have at least 1 observation"
364  
365      control = np.asarray(control)
366      samples_ = [np.asarray(sample) for sample in samples]
367  
368      # samples checks
369      samples_control: list[np.ndarray] = samples_ + [control]
370      for sample in samples_control:
371          if sample.ndim > 1:
372              raise ValueError(ndim_msg)
373  
374          if sample.size < 1:
375              raise ValueError(n_obs_msg)
376  
377      return samples_, control, rng

380  def _params_dunnett(
381      samples: list[np.ndarray], control: np.ndarray
382  ) -> tuple[np.ndarray, int, int, np.ndarray, int]:
383      """Specific parameters for Dunnett's test. 
384   
385      Degree of freedom is the number of observations minus the number of groups 
386      including the control. 
387      """
388      n_samples = np.array([sample.size for sample in samples])
389  
390      # From [1] p. 1100 d.f. = (sum N)-(p+1)
391      n_sample = n_samples.sum()
392      n_control = control.size
393      n = n_sample + n_control
394      n_groups = len(samples)
395      df = n - n_groups - 1
396  
397      # From [1] p. 1103 rho_ij = 1/sqrt((N0/Ni+1)(N0/Nj+1))
398      rho = n_control/n_samples + 1
399      rho = 1/np.sqrt(rho[:, None] * rho[None, :])
400      np.fill_diagonal(rho, 1)
401  
402      return rho, df, n_groups, n_samples, n_control

405  def _statistic_dunnett(
406      samples: list[np.ndarray], control: np.ndarray, df: int,
407      n_samples: np.ndarray, n_control: int
408  ) -> tuple[np.ndarray, float, np.ndarray, np.ndarray]:
409      """Statistic of Dunnett's test. 
410   
411      Computation based on the original single-step test from [1]. 
412      """
413      mean_control = np.mean(control)
414      mean_samples = np.array([np.mean(sample) for sample in samples])
415      all_samples = [control] + samples
416      all_means = np.concatenate([[mean_control], mean_samples])
417  
418      # Variance estimate s^2 from [1] Eq. 1
419      s2 = np.sum([_var(sample, mean=mean)*sample.size
420                   for sample, mean in zip(all_samples, all_means)]) / df
421      std = np.sqrt(s2)
422  
423      # z score inferred from [1] unlabeled equation after Eq. 1
424      z = (mean_samples - mean_control) / np.sqrt(1/n_samples + 1/n_control)
425  
426      return z / std, std, mean_control, mean_samples

429  def _pvalue_dunnett(
430      rho: np.ndarray, df: int, statistic: np.ndarray,
431      alternative: Literal['two-sided', 'less', 'greater'],
432      rng: SeedType = None
433  ) -> np.ndarray:
434      """pvalue from the multivariate t-distribution. 
435   
436      Critical values come from the multivariate student-t distribution. 
437      """
438      statistic = statistic.reshape(-1, 1)
439  
440      mvt = stats.multivariate_t(shape=rho, df=df, seed=rng)
441      if alternative == "two-sided":
442          statistic = abs(statistic)
443          pvalue = 1 - mvt.cdf(statistic, lower_limit=-statistic)
444      elif alternative == "greater":
445          pvalue = 1 - mvt.cdf(statistic, lower_limit=-np.inf)
446      else:
447          pvalue = 1 - mvt.cdf(np.inf, lower_limit=statistic)
448  
449      return np.atleast_1d(pvalue)
