3    from keras.src.api_export import keras_export
4    from keras.src.callbacks.callback import Callback
5    from keras.src.utils import file_utils

8    @keras_export("keras.callbacks.BackupAndRestore")
9    class BackupAndRestore(Callback):
10       """Callback to back up and restore the training state. 
11    
12       `BackupAndRestore` callback is intended to recover training from an 
13       interruption that has happened in the middle of a `Model.fit` execution, by 
14       backing up the training states in a temporary checkpoint file, at the end of 
15       each epoch. Each backup overwrites the previously written checkpoint file, 
16       so at any given time there is at most one such checkpoint file for 
17       backup/restoring purpose. 
18    
19       If training restarts before completion, the training state (which includes 
20       the `Model` weights and epoch number) is restored to the most recently saved 
21       state at the beginning of a new `Model.fit` run. At the completion of a 
22       `Model.fit` run, the temporary checkpoint file is deleted. 
23    
24       Note that the user is responsible to bring jobs back after the interruption. 
25       This callback is important for the backup and restore mechanism for fault 
26       tolerance purpose, and the model to be restored from a previous checkpoint 
27       is expected to be the same as the one used to back up. If user changes 
28       arguments passed to compile or fit, the checkpoint saved for fault tolerance 
29       can become invalid. 
30    
31       Example: 
32    
33       >>> class InterruptingCallback(keras.callbacks.Callback): 
34       ...   def on_epoch_begin(self, epoch, logs=None): 
35       ...     if epoch == 4: 
36       ...       raise RuntimeError('Interrupting!') 
37       >>> callback = keras.callbacks.BackupAndRestore(backup_dir="/tmp/backup") 
38       >>> model = keras.models.Sequential([keras.layers.Dense(10)]) 
39       >>> model.compile(keras.optimizers.SGD(), loss='mse') 
40       >>> model.build(input_shape=(None, 20)) 
41       >>> try: 
42       ...   model.fit(np.arange(100).reshape(5, 20), np.zeros(5), epochs=10, 
43       ...             batch_size=1, callbacks=[callback, InterruptingCallback()], 
44       ...             verbose=0) 
45       ... except: 
46       ...   pass 
47       >>> history = model.fit(np.arange(100).reshape(5, 20), np.zeros(5), 
48       ...                     epochs=10, batch_size=1, callbacks=[callback], 
49       ...                     verbose=0) 
50       >>> # Only 6 more epochs are run, since first training got interrupted at 
51       >>> # zero-indexed epoch 4, second training will continue from 4 to 9. 
52       >>> len(history.history['loss']) 
53       >>> 6 
54    
55       Args: 
56           backup_dir: String, path of directory where to store the data 
57               needed to restore the model. The directory 
58               cannot be reused elsewhere to store other files, e.g. by the 
59               `BackupAndRestore` callback of another training run, 
60               or by another callback (e.g. `ModelCheckpoint`) 
61               of the same training run. 
62           save_freq: `"epoch"`, integer, or `False`. When set to `"epoch"` 
63             the callback saves the checkpoint at the end of each epoch. 
64             When set to an integer, the callback saves the checkpoint every 
65             `save_freq` batches. Set `save_freq=False` only if using 
66             preemption checkpointing (i.e. with `save_before_preemption=True`). 
67           double_checkpoint: Boolean. If enabled, `BackupAndRestore` callback 
68             will save 2 last training states (current and previous). After 
69             interruption if current state can't be loaded due to IO error 
70             (e.g. file corrupted) it will try to restore previous one. Such 
71             behaviour will consume twice more space on disk, but increase fault 
72             tolerance. Defaults to `False`. 
73           delete_checkpoint: Boolean. This `BackupAndRestore` 
74             callback works by saving a checkpoint to back up the training state. 
75             If `delete_checkpoint=True`, the checkpoint will be deleted after 
76             training is finished. Use `False` if you'd like to keep the checkpoint 
77             for future usage. Defaults to `True`. 
78       """
79   
80       def __init__(
81           self,
82           backup_dir,
83           save_freq="epoch",
84           double_checkpoint=False,
85           delete_checkpoint=True,
86       ):
87           super().__init__()
88           self.save_freq = save_freq
89           self.double_checkpoint = double_checkpoint
90           self.delete_checkpoint = delete_checkpoint
91           self._batches_seen_since_last_saving = 0
92           self._last_batch_seen = 0
93           self._current_epoch = 0
94   
95           if not backup_dir:
96               raise ValueError("Empty `backup_dir` argument passed")
97           self.backup_dir = backup_dir
98           self._weights_path = file_utils.join(backup_dir, "latest.weights.h5")
99           self._training_metadata_path = file_utils.join(
100              backup_dir, "training_metadata.json"
101          )
102          self._prev_weights_path = self._weights_path + ".bkp"
103          self._prev_training_metadata_path = (
104              self._training_metadata_path + ".bkp"
105          )
106          if save_freq != "epoch" and not isinstance(save_freq, int):
107              raise ValueError(
108                  "Invalid value for argument `save_freq`. "
109                  f"Received: save_freq={save_freq}. "
110                  "Expected either 'epoch' or an integer value."
111              )
112  
113      def on_train_begin(self, logs=None):
114          try:
115              self._load_model()
116          except OSError as e:
117              # Weights may be corrupted. Trying to load previous one.
118              if not file_utils.exists(self._prev_weights_path):
119                  raise e
120              file_utils.copy(self._prev_weights_path, self._weights_path)
121              if file_utils.exists(self._prev_training_metadata_path):
122                  file_utils.copy(
123                      self._prev_training_metadata_path,
124                      self._training_metadata_path,
125                  )
126              elif file_utils.exists(self._training_metadata_path):
127                  file_utils.remove(self._training_metadata_path)
128              self._load_model()
129  
130      def _load_model(self):
131          """Get training state from temporary file and restore it."""
132          if not self.model.built:
133              raise ValueError(
134                  "To use the BackupAndRestore callback, "
135                  "you model must be built before you call `fit()`. "
136                  f"Model {self.model} is unbuilt. You can build it "
137                  "beforehand by calling it on a batch of data."
138              )
139          if file_utils.exists(self._weights_path):
140              if (
141                  self.model.optimizer is not None
142                  and not self.model.optimizer.built
143              ):
144                  # Make sure optimizer weights exist before loading.
145                  self.model.optimizer.build(self.model.trainable_variables)
146              self.model.load_weights(self._weights_path)
147  
148          if file_utils.exists(self._training_metadata_path):
149              with file_utils.File(self._training_metadata_path, "r") as f:
150                  training_metadata = json.loads(f.read())
151              epoch = training_metadata["epoch"]
152              self.model._initial_epoch = epoch
153  
154      def on_epoch_end(self, epoch, logs=None):
155          self._current_epoch = epoch + 1
156          self._last_batch_seen = 0
157          if self.save_freq == "epoch":
158              self._save_model()
159  
160      def on_train_batch_end(self, batch, logs=None):
161          if self._should_save_on_batch(batch):
162              self._save_model()
163  
164      def _save_model(self):
165          """Saves the model. 
166   
167          Args: 
168              epoch: the epoch this iteration is in. 
169              batch: the batch this iteration is in. `None` if the `save_freq` 
170                  is set to `"epoch"`. 
171              logs: the `logs` dict passed in to `on_batch_end` or `on_epoch_end`. 
172          """
173          # Create host directory if it doesn't exist.
174          if not file_utils.exists(self.backup_dir):
175              file_utils.makedirs(self.backup_dir)
176          if self.double_checkpoint and file_utils.exists(self._weights_path):
177              file_utils.copy(self._weights_path, self._prev_weights_path)
178          if self.double_checkpoint and file_utils.exists(
179              self._training_metadata_path
180          ):
181              file_utils.copy(
182                  self._training_metadata_path, self._prev_training_metadata_path
183              )
184          self.model.save_weights(filepath=self._weights_path, overwrite=True)
185          with file_utils.File(self._training_metadata_path, "w") as f:
186              training_metadata = {
187                  "epoch": self._current_epoch,
188                  "batch": self._last_batch_seen,
189              }
190              f.write(json.dumps(training_metadata))
191  
192      def _should_save_on_batch(self, batch):
193          """Handles batch-level saving logic, supports steps_per_execution."""
194          if self.save_freq == "epoch":
195              return False
196          if batch <= self._last_batch_seen:  # New epoch.
197              add_batches = batch + 1  # batches are zero-indexed.
198          else:
199              add_batches = batch - self._last_batch_seen
200          self._batches_seen_since_last_saving += add_batches
201          self._last_batch_seen = batch
202  
203          if self._batches_seen_since_last_saving >= self.save_freq:
204              self._batches_seen_since_last_saving = 0
205              return True
206          return False
207  
208      def on_train_end(self, logs=None):
209          if self.delete_checkpoint and file_utils.exists(self.backup_dir):
210              file_utils.rmtree(self.backup_dir)
