13   To demonstrate: 
14   * Execute this script to register the server. 
15   * Open Pythonwin's Tools -> Trace Collector Debugging Tool, so you can

17   * Open a new IE instance.  The toolbar should have a new "scissors" icon, 
18     with tooltip text "IE Button" - this is our new button - click it. 
19   * Switch back to the Pythonwin window - you should see: 
20      IOleCommandTarget::Exec called. 
21     This is the button being clicked.  Extending this to do something more

59   class Stub:
60       """ 
61       this class serves as a method stub, 
62       outputting debug info whenever the object 
63       is being called. 
64       """
65   
66       def __init__(self, name):
67           self.name = name
68   
69       def __call__(self, *args):
70           print("STUB: ", self.name, args)

73   class IEButton:
74       """ 
75       The actual COM server class 
76       """
77   
78       _com_interfaces_ = _iebutton_com_interfaces_
79       _public_methods_ = _iebutton_methods_
80       _reg_clsctx_ = pythoncom.CLSCTX_INPROC_SERVER
81       _button_text_ = "IE Button"
82       _tool_tip_ = "An example implementation for an IE Button."
83       _icon_ = ""
84       _hot_icon_ = ""
85   
86       def __init__(self):
87           # put stubs for non-implemented methods
88           for method in self._public_methods_:
89               if not hasattr(self, method):
90                   print("providing default stub for %s" % method)
91                   setattr(self, method, Stub(method))
92   
93       def QueryStatus(self, pguidCmdGroup, prgCmds, cmdtextf):
94           # 'cmdtextf' is the 'cmdtextf' element from the OLECMDTEXT structure,
95           # or None if a NULL pointer was passed.
96           result = []
97           for id, flags in prgCmds:
98               flags |= axcontrol.OLECMDF_SUPPORTED | axcontrol.OLECMDF_ENABLED
99               result.append((id, flags))
100          if cmdtextf is None:
101              cmdtext = None  # must return None if nothing requested.
102          # IE never seems to want any text - this code is here for
103          # demo purposes only
104          elif cmdtextf == axcontrol.OLECMDTEXTF_NAME:
105              cmdtext = "IEButton Name"
106          else:
107              cmdtext = "IEButton State"
108          return result, cmdtext
109  
110      def Exec(self, pguidCmdGroup, nCmdID, nCmdExecOpt, pvaIn):
111          print(pguidCmdGroup, nCmdID, nCmdExecOpt, pvaIn)
112          print("IOleCommandTarget::Exec called.")
113          # self.webbrowser.ShowBrowserBar(GUID_IETOOLBAR, not is_ietoolbar_visible())
114  
115      def SetSite(self, unknown):
116          if unknown:
117              # first get a command target
118              cmdtarget = unknown.QueryInterface(axcontrol.IID_IOleCommandTarget)
119              # then travel over to a service provider
120              serviceprovider = cmdtarget.QueryInterface(pythoncom.IID_IServiceProvider)
121              # finally ask for the internet explorer application, returned as a dispatch object
122              self.webbrowser = win32com.client.Dispatch(
123                  serviceprovider.QueryService(
124                      "{0002DF05-0000-0000-C000-000000000046}", pythoncom.IID_IDispatch
125                  )
126              )
127          else:
128              # lose all references
129              self.webbrowser = None

131      def GetClassID(self):
132          return self._reg_clsid_

135  def register(classobj):
136      import winreg
137  
138      subKeyCLSID = (
139          "SOFTWARE\\Microsoft\\Internet Explorer\\Extensions\\%38s"
140          % classobj._reg_clsid_
141      )
142      try:
143          hKey = winreg.CreateKey(winreg.HKEY_LOCAL_MACHINE, subKeyCLSID)
144          subKey = winreg.SetValueEx(
145              hKey, "ButtonText", 0, winreg.REG_SZ, classobj._button_text_
146          )
147          winreg.SetValueEx(
148              hKey, "ClsidExtension", 0, winreg.REG_SZ, classobj._reg_clsid_
149          )  # reg value for calling COM object
150          winreg.SetValueEx(
151              hKey, "CLSID", 0, winreg.REG_SZ, "{1FBA04EE-3024-11D2-8F1F-0000F87ABD16}"
152          )  # CLSID for button that sends command to COM object
153          winreg.SetValueEx(hKey, "Default Visible", 0, winreg.REG_SZ, "Yes")
154          winreg.SetValueEx(hKey, "ToolTip", 0, winreg.REG_SZ, classobj._tool_tip_)
155          winreg.SetValueEx(hKey, "Icon", 0, winreg.REG_SZ, classobj._icon_)
156          winreg.SetValueEx(hKey, "HotIcon", 0, winreg.REG_SZ, classobj._hot_icon_)
157      except OSError:
158          print("Couldn't set standard toolbar reg keys.")
159      else:
160          print("Set standard toolbar reg keys.")
161

163  def unregister(classobj):
164      import winreg
165  
166      subKeyCLSID = (
167          "SOFTWARE\\Microsoft\\Internet Explorer\\Extensions\\%38s"
168          % classobj._reg_clsid_
169      )
170      try:
171          hKey = winreg.CreateKey(winreg.HKEY_LOCAL_MACHINE, subKeyCLSID)
172          subKey = winreg.DeleteValue(hKey, "ButtonText")
173          winreg.DeleteValue(hKey, "ClsidExtension")  # for calling COM object
174          winreg.DeleteValue(hKey, "CLSID")
175          winreg.DeleteValue(hKey, "Default Visible")
176          winreg.DeleteValue(hKey, "ToolTip")
177          winreg.DeleteValue(hKey, "Icon")
178          winreg.DeleteValue(hKey, "HotIcon")
179          winreg.DeleteKey(winreg.HKEY_LOCAL_MACHINE, subKeyCLSID)
180      except OSError:
181          print("Couldn't delete Standard toolbar regkey.")
182      else:
183          print("Deleted Standard toolbar regkey.")
