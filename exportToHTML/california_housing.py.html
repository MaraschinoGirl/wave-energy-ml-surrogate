3    import numpy as np
4    
5    from keras.src.api_export import keras_export
6    from keras.src.utils.file_utils import get_file
7    
8    
9    @keras_export("keras.datasets.california_housing.load_data")
10   def load_data(
11       version="large", path="california_housing.npz", test_split=0.2, seed=113
12   ):
13       """Loads the California Housing dataset. 
14    
15       This dataset was obtained from the [StatLib repository]( 
16       https://www.dcc.fc.up.pt/~ltorgo/Regression/cal_housing.html). 
17    
18       It's a continuous regression dataset with 20,640 samples with 
19       8 features each. 
20    
21       The target variable is a scalar: the median house value 
22       for California districts, in dollars. 
23    
24       The 8 input features are the following: 
25    
26       - MedInc: median income in block group 
27       - HouseAge: median house age in block group 
28       - AveRooms: average number of rooms per household 
29       - AveBedrms: average number of bedrooms per household 
30       - Population: block group population 
31       - AveOccup: average number of household members 
32       - Latitude: block group latitude 
33       - Longitude: block group longitude 
34    
35       This dataset was derived from the 1990 U.S. census, using one row 
36       per census block group. A block group is the smallest geographical 
37       unit for which the U.S. Census Bureau publishes sample data 
38       (a block group typically has a population of 600 to 3,000 people). 
39    
40       A household is a group of people residing within a home. 
41       Since the average number of rooms and bedrooms in this dataset are 
42       provided per household, these columns may take surprisingly large 
43       values for block groups with few households and many empty houses, 
44       such as vacation resorts. 
45    
46       Args: 
47           version: `"small"` or `"large"`. The small version 
48               contains 600 samples, the large version contains 
49               20,640 samples. The purpose of the small version is 
50               to serve as an approximate replacement for the 
51               deprecated `boston_housing` dataset. 
52           path: path where to cache the dataset locally 
53               (relative to `~/.keras/datasets`). 
54           test_split: fraction of the data to reserve as test set. 
55           seed: Random seed for shuffling the data 
56               before computing the test split. 
57    
58       Returns: 
59           Tuple of Numpy arrays: `(x_train, y_train), (x_test, y_test)`. 
60    
61       **`x_train`, `x_test`**: numpy arrays with shape `(num_samples, 8)` 
62         containing either the training samples (for `x_train`), 
63         or test samples (for `y_train`). 
64    
65       **`y_train`, `y_test`**: numpy arrays of shape `(num_samples,)` 
66           containing the target scalars. The targets are float scalars 
67           typically between 25,000 and 500,000 that represent 
68           the home prices in dollars. 
69       """
70       assert 0 <= test_split < 1
71       origin_folder = (
72           "https://storage.googleapis.com/tensorflow/tf-keras-datasets/"
73       )
74       path = get_file(
75           path,
76           origin=origin_folder + "california_housing.npz",
77           file_hash=(  # noqa: E501
78               "1a2e3a52e0398de6463aebe6f4a8da34fb21fbb6b934cf88c3425e766f2a1a6f"
79           ),
80       )
81       with np.load(path, allow_pickle=True) as f:
82           x = f["x"]
83           y = f["y"]
84   
85       if version == "small":
86           x = x[:600]
87           y = y[:600]
88       elif version != "large":
89           raise ValueError(
90               "Argument `version` must be one of 'small', 'large'. "
91               f"Received: version={version}"
92           )
93   
94       rng = np.random.RandomState(seed)
95       indices = np.arange(len(x))
96       rng.shuffle(indices)
97       x = x[indices]
98       y = y[indices]
99   
100      x_train = np.array(x[: int(len(x) * (1 - test_split))])
101      y_train = np.array(y[: int(len(x) * (1 - test_split))])
102      x_test = np.array(x[int(len(x) * (1 - test_split)) :])
103      y_test = np.array(y[int(len(x) * (1 - test_split)) :])
104      return (x_train, y_train), (x_test, y_test)
