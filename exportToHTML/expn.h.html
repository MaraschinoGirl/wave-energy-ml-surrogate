5    /*                                                     expn.c 
6     * 
7     *             Exponential integral En 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * int n; 
14    * double x, y, expn(); 
15    * 
16    * y = expn( n, x ); 
17    * 
18    * 
19    * 
20    * DESCRIPTION: 
21    * 
22    * Evaluates the exponential integral 
23    * 
24    *                 inf. 
25    *                   - 
26    *                  | |   -xt 
27    *                  |    e 
28    *      E (x)  =    |    ----  dt. 
29    *       n          |      n 
30    *                | |     t 
31    *                 - 
32    *                  1 
33    * 
34    * 
35    * Both n and x must be nonnegative. 
36    * 
37    * The routine employs either a power series, a continued 
38    * fraction, or an asymptotic formula depending on the 
39    * relative values of n and x. 
40    * 
41    * ACCURACY: 
42    * 
43    *                      Relative error: 
44    * arithmetic   domain     # trials      peak         rms 
45    *    IEEE      0, 30       10000       1.7e-15     3.6e-16 
46    * 
47    */

112          /* Asymptotic expansion for large n, DLMF 8.20(ii) */
113          XSF_HOST_DEVICE double expn_large_n(int n, double x) {
114              int k;
115              double p = n;
116              double lambda = x / p;
117              double multiplier = 1 / p / (lambda + 1) / (lambda + 1);
118              double fac = 1;
119              double res = 1; /* A[0] = 1 */
120              double expfac, term;
121  
122              expfac = std::exp(-lambda * p) / (lambda + 1) / p;
123              if (expfac == 0) {
124                  set_error("expn", SF_ERROR_UNDERFLOW, NULL);
125                  return 0;
126              }
127  
128              /* Do the k = 1 term outside the loop since A[1] = 1 */
129              fac *= multiplier;
130              res += fac;
131  
132              for (k = 2; k < expn_nA; k++) {
133                  fac *= multiplier;
134                  term = fac * polevl(lambda, expn_A[k], expn_Adegs[k]);
135                  res += term;
136                  if (std::abs(term) < MACHEP * std::abs(res)) {
137                      break;
138                  }
139              }
140  
141              return expfac * res;
142          }

145      XSF_HOST_DEVICE double expn(int n, double x) {
146          double ans, r, t, yk, xk;
147          double pk, pkm1, pkm2, qk, qkm1, qkm2;
148          double psi, z;
149          int i, k;
150          constexpr double big = 1.44115188075855872E+17;
151  
152          if (std::isnan(x)) {
153              return std::numeric_limits<double>::quiet_NaN();
154          } else if (n < 0 || x < 0) {
155              set_error("expn", SF_ERROR_DOMAIN, NULL);
156              return std::numeric_limits<double>::quiet_NaN();
157          }
158  
159          if (x > detail::MAXLOG) {
160              return (0.0);
161          }
162  
163          if (x == 0.0) {
164              if (n < 2) {
165                  set_error("expn", SF_ERROR_SINGULAR, NULL);
166                  return std::numeric_limits<double>::infinity();
167              } else {
168                  return (1.0 / (n - 1.0));
169              }
170          }
171  
172          if (n == 0) {
173              return (std::exp(-x) / x);
174          }
175  
176          /* Asymptotic expansion for large n, DLMF 8.20(ii) */
177          if (n > 50) {
178              ans = detail::expn_large_n(n, x);
179              return ans;
180          }
181  
182          /* Continued fraction, DLMF 8.19.17 */
183          if (x > 1.0) {
184              k = 1;
185              pkm2 = 1.0;
186              qkm2 = x;
187              pkm1 = 1.0;
188              qkm1 = x + n;
189              ans = pkm1 / qkm1;
190  
191              do {
192                  k += 1;
193                  if (k & 1) {
194                      yk = 1.0;
195                      xk = n + (k - 1) / 2;
196                  } else {
197                      yk = x;
198                      xk = k / 2;
199                  }
200                  pk = pkm1 * yk + pkm2 * xk;
201                  qk = qkm1 * yk + qkm2 * xk;
202                  if (qk != 0) {
203                      r = pk / qk;
204                      t = std::abs((ans - r) / r);
205                      ans = r;
206                  } else {
207                      t = 1.0;
208                  }
209                  pkm2 = pkm1;
210                  pkm1 = pk;
211                  qkm2 = qkm1;
212                  qkm1 = qk;
213                  if (std::abs(pk) > big) {
214                      pkm2 /= big;
215                      pkm1 /= big;
216                      qkm2 /= big;
217                      qkm1 /= big;
218                  }
219              } while (t > detail::MACHEP);
220  
221              ans *= std::exp(-x);
222              return ans;
223          }
224  
225          /* Power series expansion, DLMF 8.19.8 */
226          psi = -detail::SCIPY_EULER - std::log(x);
227          for (i = 1; i < n; i++) {
228              psi = psi + 1.0 / i;
229          }
230  
231          z = -x;
232          xk = 0.0;
233          yk = 1.0;
234          pk = 1.0 - n;
235          if (n == 1) {
236              ans = 0.0;
237          } else {
238              ans = 1.0 / pk;
239          }
240          do {
241              xk += 1.0;
242              yk *= z / xk;
243              pk += 1.0;
244              if (pk != 0.0) {
245                  ans += yk / pk;
246              }
247              if (ans != 0.0)
248                  t = std::abs(yk / ans);
249              else
250                  t = 1.0;
251          } while (t > detail::MACHEP);
252          k = xk;
253          t = n;
254          r = n - 1;
255          ans = (std::pow(z, r) * psi * rgamma(t)) - ans;
256          return ans;
257      }
