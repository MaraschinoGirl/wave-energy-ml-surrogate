15   def get_global_attribute(name, default=None, set_to_default=False):
16       attr = getattr(GLOBAL_STATE_TRACKER, name, None)
17       if attr is None and default is not None:
18           attr = default
19           if set_to_default:
20               set_global_attribute(name, attr)
21       return attr

24   @keras_export(["keras.utils.clear_session", "keras.backend.clear_session"])
25   def clear_session(free_memory=True):
26       """Resets all state generated by Keras. 
27    
28       Keras manages a global state, which it uses to implement the Functional 
29       model-building API and to uniquify autogenerated layer names. 
30    
31       If you are creating many models in a loop, this global state will consume 
32       an increasing amount of memory over time, and you may want to clear it. 
33       Calling `clear_session()` releases the global state: this helps avoid 
34       clutter from old models and layers, especially when memory is limited. 
35    
36       Args: 
37           free_memory: Whether to call Python garbage collection. 
38               It's usually a good practice to call it to make sure 
39               memory used by deleted objects is immediately freed. 
40               However, it may take a few seconds to execute, so 
41               when using `clear_session()` in a short loop, 
42               you may want to skip it. 
43    
44       Example 1: calling `clear_session()` when creating models in a loop 
45    
46       ```python 
47       for _ in range(100): 
48         # Without `clear_session()`, each iteration of this loop will 
49         # slightly increase the size of the global state managed by Keras 
50         model = keras.Sequential([ 
51             keras.layers.Dense(10) for _ in range(10)]) 
52    
53       for _ in range(100): 
54         # With `clear_session()` called at the beginning, 
55         # Keras starts with a blank state at each iteration 
56         # and memory consumption is constant over time. 
57         keras.backend.clear_session() 
58         model = keras.Sequential([ 
59             keras.layers.Dense(10) for _ in range(10)]) 
60       ``` 
61    
62       Example 2: resetting the layer name generation counter 
63    
64       >>> layers = [keras.layers.Dense(10) for _ in range(10)] 
65       >>> new_layer = keras.layers.Dense(10) 
66       >>> print(new_layer.name) 
67       dense_10 
68       >>> keras.backend.clear_session() 
69       >>> new_layer = keras.layers.Dense(10) 
70       >>> print(new_layer.name) 
71       dense 
72       """
73       global GLOBAL_STATE_TRACKER
74       global GLOBAL_SETTINGS_TRACKER
75   
76       GLOBAL_STATE_TRACKER = threading.local()
77       GLOBAL_SETTINGS_TRACKER = threading.local()
78   
79       if backend.backend() == "tensorflow":
80           from keras.src.utils.module_utils import tensorflow as tf
81   
82           tf.compat.v1.reset_default_graph()
83           if tf.executing_eagerly():
84               # Clear pending nodes in eager executors, kernel caches and
85               # step_containers.
86               from tensorflow.python.eager import context
87   
88               context.context().clear_kernel_cache()
89       elif backend.backend() == "torch":
90           import torch._dynamo as dynamo
91   
92           # reset's torchdynamo's cache so that  cached guards, compiled fn, etc
93           # do not persist between clear_session() calls
94           dynamo.reset()
95   
96       if free_memory:
97           # Manually trigger garbage collection.
98           gc.collect()
