5    import asyncio
6    import os

8    import websockets

10   LOCAL_WS_SERVER_PORT = int(os.environ.get("LOCAL_WS_SERVER_PORT", "8765"))
11   
12   
13   async def echo(websocket):
14       async for message in websocket:
15           await websocket.send(message)
16   
17   
18   async def main():
19       async with websockets.serve(echo, "localhost", LOCAL_WS_SERVER_PORT):
20           await asyncio.Future()  # run forever
21   
22   
23   asyncio.run(main())
