33   def _check_action(fun, args, action):
34       # TODO: special expert should correct
35       # the coercion at the true location?
36       args = np.asarray(args, dtype=np.dtype("long"))
37       if action == 'warn':
38           with pytest.warns(sc.SpecialFunctionWarning):
39               fun(*args)
40       elif action == 'raise':
41           with assert_raises(sc.SpecialFunctionError):
42               fun(*args)
43       else:
44           # action == 'ignore', make sure there are no warnings/exceptions
45           with warnings.catch_warnings():
46               warnings.simplefilter("error")
47               fun(*args)

57   @pytest.mark.thread_unsafe
58   def test_seterr():
59       entry_err = sc.geterr()
60       try:
61           for category, error_code in _sf_error_code_map.items():
62               for action in _sf_error_actions:
63                   geterr_olderr = sc.geterr()
64                   seterr_olderr = sc.seterr(**{category: action})
65                   assert_(geterr_olderr == seterr_olderr)
66                   newerr = sc.geterr()
67                   assert_(newerr[category] == action)
68                   geterr_olderr.pop(category)
69                   newerr.pop(category)
70                   assert_(geterr_olderr == newerr)
71                   _check_action(_sf_error_test_function, (error_code,), action)
72       finally:
73           sc.seterr(**entry_err)

76   @pytest.mark.skipif(IS_PYPY, reason="Test not meaningful on PyPy")
77   def test_sf_error_special_refcount():
78       # Regression test for gh-16233.
79       # Check that the reference count of scipy.special is not increased
80       # when a SpecialFunctionError is raised.
81       refcount_before = sys.getrefcount(sc)
82       with sc.errstate(all='raise'):
83           with pytest.raises(sc.SpecialFunctionError, match='domain error'):
84               sc.ndtri(2.0)
85       refcount_after = sys.getrefcount(sc)
86       assert refcount_after == refcount_before
