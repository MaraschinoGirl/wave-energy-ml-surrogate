32   class TestBuiltInRegistry(object):
33       """Test the built-in registry with the default builders registered."""
34   
35       def test_combination(self):
36           assert registry.lookup("strict", "html") == HTMLParserTreeBuilder
37           if LXML_PRESENT:
38               assert registry.lookup("fast", "html") == LXMLTreeBuilder
39               assert registry.lookup("permissive", "xml") == LXMLTreeBuilderForXML
40           if HTML5LIB_PRESENT:
41               assert registry.lookup("html5lib", "html") == HTML5TreeBuilder
42   
43       def test_lookup_by_markup_type(self):
44           if LXML_PRESENT:
45               assert registry.lookup("html") == LXMLTreeBuilder
46               assert registry.lookup("xml") == LXMLTreeBuilderForXML
47           else:
48               assert registry.lookup("xml") is None
49               if HTML5LIB_PRESENT:
50                   assert registry.lookup("html") == HTML5TreeBuilder
51               else:
52                   assert registry.lookup("html") == HTMLParserTreeBuilder
53   
54       def test_named_library(self):
55           if LXML_PRESENT:
56               assert registry.lookup("lxml", "xml") == LXMLTreeBuilderForXML
57               assert registry.lookup("lxml", "html") == LXMLTreeBuilder
58           if HTML5LIB_PRESENT:
59               assert registry.lookup("html5lib") == HTML5TreeBuilder
60   
61           assert registry.lookup("html.parser") == HTMLParserTreeBuilder
62   
63       def test_beautifulsoup_constructor_does_lookup(self):
64           with warnings.catch_warnings(record=True):
65               # This will create a warning about not explicitly
66               # specifying a parser, but we'll ignore it.
67   
68               # You can pass in a string.
69               BeautifulSoup("", features="html")
70               # Or a list of strings.
71               BeautifulSoup("", features=["html", "fast"])
72               pass
73   
74           # You'll get an exception if BS can't find an appropriate
75           # builder.
76           with pytest.raises(ValueError):
77               BeautifulSoup("", features="no-such-feature")

80   class TestRegistry(object):
81       """Test the TreeBuilderRegistry class in general."""
82   
83       def setup_method(self):
84           self.registry = TreeBuilderRegistry()
85   
86       def builder_for_features(self, *feature_list: str) -> Type[TreeBuilder]:
87           cls = type(
88               "Builder_" + "_".join(feature_list), (object,), {"features": feature_list}
89           )
90   
91           self.registry.register(cls)
92           return cls
93   
94       def test_register_with_no_features(self):
95           builder = self.builder_for_features()
96   
97           # Since the builder advertises no features, you can't find it
98           # by looking up features.
99           assert self.registry.lookup("foo") is None
100  
101          # But you can find it by doing a lookup with no features, if
102          # this happens to be the only registered builder.
103          assert self.registry.lookup() == builder
104  
105      def test_register_with_features_makes_lookup_succeed(self):
106          builder = self.builder_for_features("foo", "bar")
107          assert self.registry.lookup("foo") is builder
108          assert self.registry.lookup("bar") is builder
109  
110      def test_lookup_fails_when_no_builder_implements_feature(self):
111          assert self.registry.lookup("baz") is None
112  
113      def test_lookup_gets_most_recent_registration_when_no_feature_specified(self):
114          self.builder_for_features("foo")
115          builder2 = self.builder_for_features("bar")
116          assert self.registry.lookup() == builder2
117  
118      def test_lookup_fails_when_no_tree_builders_registered(self):
119          assert self.registry.lookup() is None
120  
121      def test_lookup_gets_most_recent_builder_supporting_all_features(self):
122          self.builder_for_features("foo")
123          self.builder_for_features("bar")
124          has_both_early = self.builder_for_features("foo", "bar", "baz")
125          has_both_late = self.builder_for_features("foo", "bar", "quux")
126          self.builder_for_features("bar")
127          self.builder_for_features("foo")
128  
129          # There are two builders featuring 'foo' and 'bar', but
130          # the one that also features 'quux' was registered later.
131          assert self.registry.lookup("foo", "bar") == has_both_late
132  
133          # There is only one builder featuring 'foo', 'bar', and 'baz'.
134          assert self.registry.lookup("foo", "bar", "baz") == has_both_early
135  
136      def test_lookup_fails_when_cannot_reconcile_requested_features(self):
137          self.builder_for_features("foo", "bar")
138          self.builder_for_features("foo", "baz")
139          assert self.registry.lookup("bar", "baz") is None
