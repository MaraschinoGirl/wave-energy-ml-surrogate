24   class table__a_v_a_r(BaseTTXConverter):
25       """Axis Variations table 
26    
27       This class represents the ``avar`` table of a variable font. The object has one 
28       substantive attribute, ``segments``, which maps axis tags to a segments dictionary:: 
29    
30           >>> font["avar"].segments   # doctest: +SKIP 
31           {'wght': {-1.0: -1.0, 
32             0.0: 0.0, 
33             0.125: 0.11444091796875, 
34             0.25: 0.23492431640625, 
35             0.5: 0.35540771484375, 
36             0.625: 0.5, 
37             0.75: 0.6566162109375, 
38             0.875: 0.81927490234375, 
39             1.0: 1.0}, 
40            'ital': {-1.0: -1.0, 0.0: 0.0, 1.0: 1.0}} 
41    
42       Notice that the segments dictionary is made up of normalized values. A valid 
43       ``avar`` segment mapping must contain the entries ``-1.0: -1.0, 0.0: 0.0, 1.0: 1.0``. 
44       fontTools does not enforce this, so it is your responsibility to ensure that 
45       mappings are valid. 
46    
47       See also https://learn.microsoft.com/en-us/typography/opentype/spec/avar 
48       """
49   
50       dependencies = ["fvar"]
51   
52       def __init__(self, tag=None):
53           super().__init__(tag)
54           self.segments = {}
55   
56       def compile(self, ttFont):
57           axisTags = [axis.axisTag for axis in ttFont["fvar"].axes]
58           if not hasattr(self, "table"):
59               self.table = otTables.avar()
60           if not hasattr(self.table, "Reserved"):
61               self.table.Reserved = 0
62           self.table.Version = (getattr(self, "majorVersion", 1) << 16) | getattr(
63               self, "minorVersion", 0
64           )
65           self.table.AxisCount = len(axisTags)
66           self.table.AxisSegmentMap = []
67           for axis in axisTags:
68               mappings = self.segments[axis]
69               segmentMap = otTables.AxisSegmentMap()
70               segmentMap.PositionMapCount = len(mappings)
71               segmentMap.AxisValueMap = []
72               for key, value in sorted(mappings.items()):
73                   valueMap = otTables.AxisValueMap()
74                   valueMap.FromCoordinate = key
75                   valueMap.ToCoordinate = value
76                   segmentMap.AxisValueMap.append(valueMap)
77               self.table.AxisSegmentMap.append(segmentMap)
78           return super().compile(ttFont)
79   
80       def decompile(self, data, ttFont):
81           super().decompile(data, ttFont)
82           self.majorVersion = self.table.Version >> 16
83           self.minorVersion = self.table.Version & 0xFFFF
84           if self.majorVersion not in (1, 2):
85               raise NotImplementedError("Unknown avar table version")
86           axisTags = [axis.axisTag for axis in ttFont["fvar"].axes]
87           for axis in axisTags:
88               self.segments[axis] = {}
89           for axis, segmentMap in zip(axisTags, self.table.AxisSegmentMap):
90               segments = self.segments[axis] = {}
91               for segment in segmentMap.AxisValueMap:
92                   segments[segment.FromCoordinate] = segment.ToCoordinate
93   
94       def toXML(self, writer, ttFont):
95           writer.simpletag(
96               "version",
97               major=getattr(self, "majorVersion", 1),
98               minor=getattr(self, "minorVersion", 0),
99           )
100          writer.newline()
101          axisTags = [axis.axisTag for axis in ttFont["fvar"].axes]
102          for axis in axisTags:
103              writer.begintag("segment", axis=axis)
104              writer.newline()
105              for key, value in sorted(self.segments[axis].items()):
106                  key = fl2str(key, 14)
107                  value = fl2str(value, 14)
108                  writer.simpletag("mapping", **{"from": key, "to": value})
109                  writer.newline()
110              writer.endtag("segment")
111              writer.newline()
112          if getattr(self, "majorVersion", 1) >= 2:
113              if self.table.VarIdxMap:
114                  self.table.VarIdxMap.toXML(writer, ttFont, name="VarIdxMap")
115              if self.table.VarStore:
116                  self.table.VarStore.toXML(writer, ttFont)
117  
118      def fromXML(self, name, attrs, content, ttFont):
119          if not hasattr(self, "table"):
120              self.table = otTables.avar()
121          if not hasattr(self.table, "Reserved"):
122              self.table.Reserved = 0
123          if name == "version":
124              self.majorVersion = safeEval(attrs["major"])
125              self.minorVersion = safeEval(attrs["minor"])
126              self.table.Version = (getattr(self, "majorVersion", 1) << 16) | getattr(
127                  self, "minorVersion", 0
128              )
129          elif name == "segment":
130              axis = attrs["axis"]
131              segment = self.segments[axis] = {}
132              for element in content:
133                  if isinstance(element, tuple):
134                      elementName, elementAttrs, _ = element
135                      if elementName == "mapping":
136                          fromValue = str2fl(elementAttrs["from"], 14)
137                          toValue = str2fl(elementAttrs["to"], 14)
138                          if fromValue in segment:
139                              log.warning(
140                                  "duplicate entry for %s in axis '%s'", fromValue, axis
141                              )
142                          segment[fromValue] = toValue
143          else:
144              super().fromXML(name, attrs, content, ttFont)

146      def renormalizeLocation(self, location, font):
147  
148          majorVersion = getattr(self, "majorVersion", 1)
149  
150          if majorVersion not in (1, 2):
151              raise NotImplementedError("Unknown avar table version")
152  
153          avarSegments = self.segments
154          mappedLocation = {}
155          for axisTag, value in location.items():
156              avarMapping = avarSegments.get(axisTag, None)
157              if avarMapping is not None:
158                  value = piecewiseLinearMap(value, avarMapping)
159              mappedLocation[axisTag] = value
160  
161          if majorVersion < 2:
162              return mappedLocation
163  
164          # Version 2
165  
166          varIdxMap = self.table.VarIdxMap
167          varStore = self.table.VarStore
168          axes = font["fvar"].axes
169          if varStore is not None:
170              instancer = VarStoreInstancer(varStore, axes, mappedLocation)
171  
172          coords = list(fl2fi(mappedLocation.get(axis.axisTag, 0), 14) for axis in axes)
173  
174          out = []
175          for varIdx, v in enumerate(coords):
176  
177              if varIdxMap is not None:
178                  varIdx = varIdxMap[varIdx]
179  
180              if varStore is not None:
181                  delta = instancer[varIdx]
182                  v += otRound(delta)
183                  v = min(max(v, -(1 << 14)), +(1 << 14))
184  
185              out.append(v)
186  
187          mappedLocation = {
188              axis.axisTag: fi2fl(v, 14) for v, axis in zip(out, axes) if v != 0
189          }
190  
191          return mappedLocation
