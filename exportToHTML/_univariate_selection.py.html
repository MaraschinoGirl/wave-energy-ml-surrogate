23   def _clean_nans(scores):
24       """ 
25       Fixes Issue #1240: NaNs can't be properly compared, so change them to the 
26       smallest value of scores's dtype. -inf seems to be unreliable. 
27       """
28       # XXX where should this function be called? fit? scoring functions
29       # themselves?
30       scores = as_float_array(scores, copy=True)
31       scores[np.isnan(scores)] = np.finfo(scores.dtype).min
32       return scores

39   # The following function is a rewriting of scipy.stats.f_oneway
40   # Contrary to the scipy.stats.f_oneway implementation it does not
41   # copy the data while keeping the inputs unchanged.
42   def f_oneway(*args):
43       """Perform a 1-way ANOVA. 
44    
45       The one-way ANOVA tests the null hypothesis that 2 or more groups have 
46       the same population mean. The test is applied to samples from two or 
47       more groups, possibly with differing sizes. 
48    
49       Read more in the :ref:`User Guide <univariate_feature_selection>`. 
50    
51       Parameters 
52       ---------- 
53       *args : {array-like, sparse matrix} 
54           Sample1, sample2... The sample measurements should be given as 
55           arguments. 
56    
57       Returns 
58       ------- 
59       f_statistic : float 
60           The computed F-value of the test. 
61       p_value : float 
62           The associated p-value from the F-distribution. 
63    
64       Notes 
65       ----- 
66       The ANOVA test has important assumptions that must be satisfied in order 
67       for the associated p-value to be valid. 
68    
69       1. The samples are independent 
70       2. Each sample is from a normally distributed population 
71       3. The population standard deviations of the groups are all equal. This 
72          property is known as homoscedasticity. 
73    
74       If these assumptions are not true for a given set of data, it may still be 
75       possible to use the Kruskal-Wallis H-test (`scipy.stats.kruskal`_) although 
76       with some loss of power. 
77    
78       The algorithm is from Heiman[2], pp.394-7. 
79    
80       See ``scipy.stats.f_oneway`` that should give the same results while 
81       being less efficient. 
82    
83       References 
84       ---------- 
85       .. [1] Lowry, Richard.  "Concepts and Applications of Inferential 
86              Statistics". Chapter 14. 
87              http://vassarstats.net/textbook 
88    
89       .. [2] Heiman, G.W.  Research Methods in Statistics. 2002. 
90       """
91       n_classes = len(args)
92       args = [as_float_array(a) for a in args]
93       n_samples_per_class = np.array([a.shape[0] for a in args])
94       n_samples = np.sum(n_samples_per_class)
95       ss_alldata = sum(safe_sqr(a).sum(axis=0) for a in args)
96       sums_args = [np.asarray(a.sum(axis=0)) for a in args]
97       square_of_sums_alldata = sum(sums_args) ** 2
98       square_of_sums_args = [s**2 for s in sums_args]
99       sstot = ss_alldata - square_of_sums_alldata / float(n_samples)
100      ssbn = 0.0
101      for k, _ in enumerate(args):
102          ssbn += square_of_sums_args[k] / n_samples_per_class[k]
103      ssbn -= square_of_sums_alldata / float(n_samples)
104      sswn = sstot - ssbn
105      dfbn = n_classes - 1
106      dfwn = n_samples - n_classes
107      msb = ssbn / float(dfbn)
108      msw = sswn / float(dfwn)
109      constant_features_idx = np.where(msw == 0.0)[0]
110      if np.nonzero(msb)[0].size != msb.size and constant_features_idx.size:
111          warnings.warn("Features %s are constant." % constant_features_idx, UserWarning)
112      f = msb / msw
113      # flatten matrix to vector in sparse case
114      f = np.asarray(f).ravel()
115      prob = special.fdtrc(dfbn, dfwn, f)
116      return f, prob

126  def f_classif(X, y):
127      """Compute the ANOVA F-value for the provided sample. 
128   
129      Read more in the :ref:`User Guide <univariate_feature_selection>`. 
130   
131      Parameters 
132      ---------- 
133      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
134          The set of regressors that will be tested sequentially. 
135   
136      y : array-like of shape (n_samples,) 
137          The target vector. 
138   
139      Returns 
140      ------- 
141      f_statistic : ndarray of shape (n_features,) 
142          F-statistic for each feature. 
143   
144      p_values : ndarray of shape (n_features,) 
145          P-values associated with the F-statistic. 
146   
147      See Also 
148      -------- 
149      chi2 : Chi-squared stats of non-negative features for classification tasks. 
150      f_regression : F-value between label/feature for regression tasks. 
151   
152      Examples 
153      -------- 
154      >>> from sklearn.datasets import make_classification 
155      >>> from sklearn.feature_selection import f_classif 
156      >>> X, y = make_classification( 
157      ...     n_samples=100, n_features=10, n_informative=2, n_clusters_per_class=1, 
158      ...     shuffle=False, random_state=42 
159      ... ) 
160      >>> f_statistic, p_values = f_classif(X, y) 
161      >>> f_statistic 
162      array([2.2...e+02, 7.0...e-01, 1.6...e+00, 9.3...e-01, 
163             5.4...e+00, 3.2...e-01, 4.7...e-02, 5.7...e-01, 
164             7.5...e-01, 8.9...e-02]) 
165      >>> p_values 
166      array([7.1...e-27, 4.0...e-01, 1.9...e-01, 3.3...e-01, 
167             2.2...e-02, 5.7...e-01, 8.2...e-01, 4.5...e-01, 
168             3.8...e-01, 7.6...e-01]) 
169      """
170      X, y = check_X_y(X, y, accept_sparse=["csr", "csc", "coo"])
171      args = [X[safe_mask(X, y == k)] for k in np.unique(y)]
172      return f_oneway(*args)

175  def _chisquare(f_obs, f_exp):
176      """Fast replacement for scipy.stats.chisquare. 
177   
178      Version from https://github.com/scipy/scipy/pull/2525 with additional 
179      optimizations. 
180      """
181      f_obs = np.asarray(f_obs, dtype=np.float64)
182  
183      k = len(f_obs)
184      # Reuse f_obs for chi-squared statistics
185      chisq = f_obs
186      chisq -= f_exp
187      chisq **= 2
188      with np.errstate(invalid="ignore"):
189          chisq /= f_exp
190      chisq = chisq.sum(axis=0)
191      return chisq, special.chdtrc(k - 1, chisq)

201  def chi2(X, y):
202      """Compute chi-squared stats between each non-negative feature and class. 
203   
204      This score can be used to select the `n_features` features with the 
205      highest values for the test chi-squared statistic from X, which must 
206      contain only **non-negative integer feature values** such as booleans or frequencies 
207      (e.g., term counts in document classification), relative to the classes. 
208   
209      If some of your features are continuous, you need to bin them, for 
210      example by using :class:`~sklearn.preprocessing.KBinsDiscretizer`. 
211   
212      Recall that the chi-square test measures dependence between stochastic 
213      variables, so using this function "weeds out" the features that are the 
214      most likely to be independent of class and therefore irrelevant for 
215      classification. 
216   
217      Read more in the :ref:`User Guide <univariate_feature_selection>`. 
218   
219      Parameters 
220      ---------- 
221      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
222          Sample vectors. 
223   
224      y : array-like of shape (n_samples,) 
225          Target vector (class labels). 
226   
227      Returns 
228      ------- 
229      chi2 : ndarray of shape (n_features,) 
230          Chi2 statistics for each feature. 
231   
232      p_values : ndarray of shape (n_features,) 
233          P-values for each feature. 
234   
235      See Also 
236      -------- 
237      f_classif : ANOVA F-value between label/feature for classification tasks. 
238      f_regression : F-value between label/feature for regression tasks. 
239   
240      Notes 
241      ----- 
242      Complexity of this algorithm is O(n_classes * n_features). 
243   
244      Examples 
245      -------- 
246      >>> import numpy as np 
247      >>> from sklearn.feature_selection import chi2 
248      >>> X = np.array([[1, 1, 3], 
249      ...               [0, 1, 5], 
250      ...               [5, 4, 1], 
251      ...               [6, 6, 2], 
252      ...               [1, 4, 0], 
253      ...               [0, 0, 0]]) 
254      >>> y = np.array([1, 1, 0, 0, 2, 2]) 
255      >>> chi2_stats, p_values = chi2(X, y) 
256      >>> chi2_stats 
257      array([15.3...,  6.5       ,  8.9...]) 
258      >>> p_values 
259      array([0.0004..., 0.0387..., 0.0116... ]) 
260      """
261  
262      # XXX: we might want to do some of the following in logspace instead for
263      # numerical stability.
264      # Converting X to float allows getting better performance for the
265      # safe_sparse_dot call made below.
266      X = check_array(X, accept_sparse="csr", dtype=(np.float64, np.float32))
267      if np.any((X.data if issparse(X) else X) < 0):
268          raise ValueError("Input X must be non-negative.")
269  
270      # Use a sparse representation for Y by default to reduce memory usage when
271      # y has many unique classes.
272      Y = LabelBinarizer(sparse_output=True).fit_transform(y)
273      if Y.shape[1] == 1:
274          Y = Y.toarray()
275          Y = np.append(1 - Y, Y, axis=1)
276  
277      observed = safe_sparse_dot(Y.T, X)  # n_classes * n_features
278  
279      if issparse(observed):
280          # convert back to a dense array before calling _chisquare
281          # XXX: could _chisquare be reimplement to accept sparse matrices for
282          # cases where both n_classes and n_features are large (and X is
283          # sparse)?
284          observed = observed.toarray()
285  
286      feature_count = X.sum(axis=0).reshape(1, -1)
287      class_prob = Y.mean(axis=0).reshape(1, -1)
288      expected = np.dot(class_prob.T, feature_count)
289  
290      return _chisquare(observed, expected)

293  @validate_params(
294      {
295          "X": ["array-like", "sparse matrix"],
296          "y": ["array-like"],
297          "center": ["boolean"],
298          "force_finite": ["boolean"],
299      },
300      prefer_skip_nested_validation=True,
301  )
302  def r_regression(X, y, *, center=True, force_finite=True):
303      """Compute Pearson's r for each features and the target. 
304   
305      Pearson's r is also known as the Pearson correlation coefficient. 
306   
307      Linear model for testing the individual effect of each of many regressors. 
308      This is a scoring function to be used in a feature selection procedure, not 
309      a free standing feature selection procedure. 
310   
311      The cross correlation between each regressor and the target is computed 
312      as:: 
313   
314          E[(X[:, i] - mean(X[:, i])) * (y - mean(y))] / (std(X[:, i]) * std(y)) 
315   
316      For more on usage see the :ref:`User Guide <univariate_feature_selection>`. 
317   
318      .. versionadded:: 1.0 
319   
320      Parameters 
321      ---------- 
322      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
323          The data matrix. 
324   
325      y : array-like of shape (n_samples,) 
326          The target vector. 
327   
328      center : bool, default=True 
329          Whether or not to center the data matrix `X` and the target vector `y`. 
330          By default, `X` and `y` will be centered. 
331   
332      force_finite : bool, default=True 
333          Whether or not to force the Pearson's R correlation to be finite. 
334          In the particular case where some features in `X` or the target `y` 
335          are constant, the Pearson's R correlation is not defined. When 
336          `force_finite=False`, a correlation of `np.nan` is returned to 
337          acknowledge this case. When `force_finite=True`, this value will be 
338          forced to a minimal correlation of `0.0`. 
339   
340          .. versionadded:: 1.1 
341   
342      Returns 
343      ------- 
344      correlation_coefficient : ndarray of shape (n_features,) 
345          Pearson's R correlation coefficients of features. 
346   
347      See Also 
348      -------- 
349      f_regression: Univariate linear regression tests returning f-statistic 
350          and p-values. 
351      mutual_info_regression: Mutual information for a continuous target. 
352      f_classif: ANOVA F-value between label/feature for classification tasks. 
353      chi2: Chi-squared stats of non-negative features for classification tasks. 
354   
355      Examples 
356      -------- 
357      >>> from sklearn.datasets import make_regression 
358      >>> from sklearn.feature_selection import r_regression 
359      >>> X, y = make_regression( 
360      ...     n_samples=50, n_features=3, n_informative=1, noise=1e-4, random_state=42 
361      ... ) 
362      >>> r_regression(X, y) 
363      array([-0.15...,  1.        , -0.22...]) 
364      """
365      X, y = check_X_y(X, y, accept_sparse=["csr", "csc", "coo"], dtype=np.float64)
366      n_samples = X.shape[0]
367  
368      # Compute centered values
369      # Note that E[(x - mean(x))*(y - mean(y))] = E[x*(y - mean(y))], so we
370      # need not center X
371      if center:
372          y = y - np.mean(y)
373          # TODO: for Scipy <= 1.10, `isspmatrix(X)` returns `True` for sparse arrays.
374          # Here, we check the output of the `.mean` operation that returns a `np.matrix`
375          # for sparse matrices while a `np.array` for dense and sparse arrays.
376          # We can reconsider using `isspmatrix` when the minimum version is
377          # SciPy >= 1.11
378          X_means = X.mean(axis=0)
379          X_means = X_means.getA1() if isinstance(X_means, np.matrix) else X_means
380          # Compute the scaled standard deviations via moments
381          X_norms = np.sqrt(row_norms(X.T, squared=True) - n_samples * X_means**2)
382      else:
383          X_norms = row_norms(X.T)
384  
385      correlation_coefficient = safe_sparse_dot(y, X)
386      with np.errstate(divide="ignore", invalid="ignore"):
387          correlation_coefficient /= X_norms
388          correlation_coefficient /= np.linalg.norm(y)
389  
390      if force_finite and not np.isfinite(correlation_coefficient).all():
391          # case where the target or some features are constant
392          # the correlation coefficient(s) is/are set to the minimum (i.e. 0.0)
393          nan_mask = np.isnan(correlation_coefficient)
394          correlation_coefficient[nan_mask] = 0.0
395      return correlation_coefficient

398  @validate_params(
399      {
400          "X": ["array-like", "sparse matrix"],
401          "y": ["array-like"],
402          "center": ["boolean"],
403          "force_finite": ["boolean"],
404      },
405      prefer_skip_nested_validation=True,
406  )
407  def f_regression(X, y, *, center=True, force_finite=True):
408      """Univariate linear regression tests returning F-statistic and p-values. 
409   
410      Quick linear model for testing the effect of a single regressor, 
411      sequentially for many regressors. 
412   
413      This is done in 2 steps: 
414   
415      1. The cross correlation between each regressor and the target is computed 
416         using :func:`r_regression` as:: 
417   
418             E[(X[:, i] - mean(X[:, i])) * (y - mean(y))] / (std(X[:, i]) * std(y)) 
419   
420      2. It is converted to an F score and then to a p-value. 
421   
422      :func:`f_regression` is derived from :func:`r_regression` and will rank 
423      features in the same order if all the features are positively correlated 
424      with the target. 
425   
426      Note however that contrary to :func:`f_regression`, :func:`r_regression` 
427      values lie in [-1, 1] and can thus be negative. :func:`f_regression` is 
428      therefore recommended as a feature selection criterion to identify 
429      potentially predictive feature for a downstream classifier, irrespective of 
430      the sign of the association with the target variable. 
431   
432      Furthermore :func:`f_regression` returns p-values while 
433      :func:`r_regression` does not. 
434   
435      Read more in the :ref:`User Guide <univariate_feature_selection>`. 
436   
437      Parameters 
438      ---------- 
439      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
440          The data matrix. 
441   
442      y : array-like of shape (n_samples,) 
443          The target vector. 
444   
445      center : bool, default=True 
446          Whether or not to center the data matrix `X` and the target vector `y`. 
447          By default, `X` and `y` will be centered. 
448   
449      force_finite : bool, default=True 
450          Whether or not to force the F-statistics and associated p-values to 
451          be finite. There are two cases where the F-statistic is expected to not 
452          be finite: 
453   
454          - when the target `y` or some features in `X` are constant. In this 
455            case, the Pearson's R correlation is not defined leading to obtain 
456            `np.nan` values in the F-statistic and p-value. When 
457            `force_finite=True`, the F-statistic is set to `0.0` and the 
458            associated p-value is set to `1.0`. 
459          - when a feature in `X` is perfectly correlated (or 
460            anti-correlated) with the target `y`. In this case, the F-statistic 
461            is expected to be `np.inf`. When `force_finite=True`, the F-statistic 
462            is set to `np.finfo(dtype).max` and the associated p-value is set to 
463            `0.0`. 
464   
465          .. versionadded:: 1.1 
466   
467      Returns 
468      ------- 
469      f_statistic : ndarray of shape (n_features,) 
470          F-statistic for each feature. 
471   
472      p_values : ndarray of shape (n_features,) 
473          P-values associated with the F-statistic. 
474   
475      See Also 
476      -------- 
477      r_regression: Pearson's R between label/feature for regression tasks. 
478      f_classif: ANOVA F-value between label/feature for classification tasks. 
479      chi2: Chi-squared stats of non-negative features for classification tasks. 
480      SelectKBest: Select features based on the k highest scores. 
481      SelectFpr: Select features based on a false positive rate test. 
482      SelectFdr: Select features based on an estimated false discovery rate. 
483      SelectFwe: Select features based on family-wise error rate. 
484      SelectPercentile: Select features based on percentile of the highest 
485          scores. 
486   
487      Examples 
488      -------- 
489      >>> from sklearn.datasets import make_regression 
490      >>> from sklearn.feature_selection import f_regression 
491      >>> X, y = make_regression( 
492      ...     n_samples=50, n_features=3, n_informative=1, noise=1e-4, random_state=42 
493      ... ) 
494      >>> f_statistic, p_values = f_regression(X, y) 
495      >>> f_statistic 
496      array([1.2...+00, 2.6...+13, 2.6...+00]) 
497      >>> p_values 
498      array([2.7..., 1.5..., 1.0...]) 
499      """
500      correlation_coefficient = r_regression(
501          X, y, center=center, force_finite=force_finite
502      )
503      deg_of_freedom = y.size - (2 if center else 1)
504  
505      corr_coef_squared = correlation_coefficient**2
506  
507      with np.errstate(divide="ignore", invalid="ignore"):
508          f_statistic = corr_coef_squared / (1 - corr_coef_squared) * deg_of_freedom
509          p_values = stats.f.sf(f_statistic, 1, deg_of_freedom)
510  
511      if force_finite and not np.isfinite(f_statistic).all():
512          # case where there is a perfect (anti-)correlation
513          # f-statistics can be set to the maximum and p-values to zero
514          mask_inf = np.isinf(f_statistic)
515          f_statistic[mask_inf] = np.finfo(f_statistic.dtype).max
516          # case where the target or some features are constant
517          # f-statistics would be minimum and thus p-values large
518          mask_nan = np.isnan(f_statistic)
519          f_statistic[mask_nan] = 0.0
520          p_values[mask_nan] = 1.0
521      return f_statistic, p_values

528  class _BaseFilter(SelectorMixin, BaseEstimator):
529      """Initialize the univariate feature selection. 
530   
531      Parameters 
532      ---------- 
533      score_func : callable 
534          Function taking two arrays X and y, and returning a pair of arrays 
535          (scores, pvalues) or a single array with scores. 
536      """
537  
538      _parameter_constraints: dict = {"score_func": [callable]}
539  
540      def __init__(self, score_func):
541          self.score_func = score_func
542  
543      @_fit_context(prefer_skip_nested_validation=True)
544      def fit(self, X, y=None):
545          """Run score function on (X, y) and get the appropriate features. 
546   
547          Parameters 
548          ---------- 
549          X : array-like of shape (n_samples, n_features) 
550              The training input samples. 
551   
552          y : array-like of shape (n_samples,) or None 
553              The target values (class labels in classification, real numbers in 
554              regression). If the selector is unsupervised then `y` can be set to `None`. 
555   
556          Returns 
557          ------- 
558          self : object 
559              Returns the instance itself. 
560          """
561          if y is None:
562              X = validate_data(self, X, accept_sparse=["csr", "csc"])
563          else:
564              X, y = validate_data(
565                  self, X, y, accept_sparse=["csr", "csc"], multi_output=True
566              )
567  
568          self._check_params(X, y)
569          score_func_ret = self.score_func(X, y)
570          if isinstance(score_func_ret, (list, tuple)):
571              self.scores_, self.pvalues_ = score_func_ret
572              self.pvalues_ = np.asarray(self.pvalues_)
573          else:
574              self.scores_ = score_func_ret
575              self.pvalues_ = None
576  
577          self.scores_ = np.asarray(self.scores_)
578  
579          return self
580  
581      def _check_params(self, X, y):
582          pass
583  
584      def __sklearn_tags__(self):
585          tags = super().__sklearn_tags__()
586          tags.target_tags.required = True
587          tags.input_tags.sparse = True
588          return tags

594  class SelectPercentile(_BaseFilter):
595      """Select features according to a percentile of the highest scores. 
596   
597      Read more in the :ref:`User Guide <univariate_feature_selection>`. 
598   
599      Parameters 
600      ---------- 
601      score_func : callable, default=f_classif 
602          Function taking two arrays X and y, and returning a pair of arrays 
603          (scores, pvalues) or a single array with scores. 
604          Default is f_classif (see below "See Also"). The default function only 
605          works with classification tasks. 
606   
607          .. versionadded:: 0.18 
608   
609      percentile : int, default=10 
610          Percent of features to keep. 
611   
612      Attributes 
613      ---------- 
614      scores_ : array-like of shape (n_features,) 
615          Scores of features. 
616   
617      pvalues_ : array-like of shape (n_features,) 
618          p-values of feature scores, None if `score_func` returned only scores. 
619   
620      n_features_in_ : int 
621          Number of features seen during :term:`fit`. 
622   
623          .. versionadded:: 0.24 
624   
625      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
626          Names of features seen during :term:`fit`. Defined only when `X` 
627          has feature names that are all strings. 
628   
629          .. versionadded:: 1.0 
630   
631      See Also 
632      -------- 
633      f_classif : ANOVA F-value between label/feature for classification tasks. 
634      mutual_info_classif : Mutual information for a discrete target. 
635      chi2 : Chi-squared stats of non-negative features for classification tasks. 
636      f_regression : F-value between label/feature for regression tasks. 
637      mutual_info_regression : Mutual information for a continuous target. 
638      SelectKBest : Select features based on the k highest scores. 
639      SelectFpr : Select features based on a false positive rate test. 
640      SelectFdr : Select features based on an estimated false discovery rate. 
641      SelectFwe : Select features based on family-wise error rate. 
642      GenericUnivariateSelect : Univariate feature selector with configurable 
643          mode. 
644   
645      Notes 
646      ----- 
647      Ties between features with equal scores will be broken in an unspecified 
648      way. 
649   
650      This filter supports unsupervised feature selection that only requests `X` for 
651      computing the scores. 
652   
653      Examples 
654      -------- 
655      >>> from sklearn.datasets import load_digits 
656      >>> from sklearn.feature_selection import SelectPercentile, chi2 
657      >>> X, y = load_digits(return_X_y=True) 
658      >>> X.shape 
659      (1797, 64) 
660      >>> X_new = SelectPercentile(chi2, percentile=10).fit_transform(X, y) 
661      >>> X_new.shape 
662      (1797, 7) 
663      """
664  
665      _parameter_constraints: dict = {
666          **_BaseFilter._parameter_constraints,
667          "percentile": [Interval(Real, 0, 100, closed="both")],
668      }
669  
670      def __init__(self, score_func=f_classif, *, percentile=10):
671          super().__init__(score_func=score_func)
672          self.percentile = percentile
673  
674      def _get_support_mask(self):
675          check_is_fitted(self)
676  
677          # Cater for NaNs
678          if self.percentile == 100:
679              return np.ones(len(self.scores_), dtype=bool)
680          elif self.percentile == 0:
681              return np.zeros(len(self.scores_), dtype=bool)
682  
683          scores = _clean_nans(self.scores_)
684          threshold = np.percentile(scores, 100 - self.percentile)
685          mask = scores > threshold
686          ties = np.where(scores == threshold)[0]
687          if len(ties):
688              max_feats = int(len(scores) * self.percentile / 100)
689              kept_ties = ties[: max_feats - mask.sum()]
690              mask[kept_ties] = True
691          return mask
692  
693      def __sklearn_tags__(self):
694          tags = super().__sklearn_tags__()
695          tags.target_tags.required = False
696          return tags

891      """Filter: Select the p-values for an estimated false discovery rate.

897   
898      Parameters 
899      ---------- 
900      score_func : callable, default=f_classif 
901          Function taking two arrays X and y, and returning a pair of arrays 
902          (scores, pvalues). 
903          Default is f_classif (see below "See Also"). The default function only 
904          works with classification tasks. 
905   
906      alpha : float, default=5e-2 
907          The highest uncorrected p-value for features to keep. 
908   
909      Attributes 
910      ---------- 
911      scores_ : array-like of shape (n_features,) 
912          Scores of features. 
913   
914      pvalues_ : array-like of shape (n_features,) 
915          p-values of feature scores.

918          Number of features seen during :term:`fit`.

920          .. versionadded:: 0.24 
921   
922      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
923          Names of features seen during :term:`fit`. Defined only when `X` 
924          has feature names that are all strings. 
925   
926          .. versionadded:: 1.0 
927   
928      See Also 
929      -------- 
930      f_classif : ANOVA F-value between label/feature for classification tasks. 
931      mutual_info_classif : Mutual information for a discrete target. 
932      chi2 : Chi-squared stats of non-negative features for classification tasks. 
933      f_regression : F-value between label/feature for regression tasks. 
934      mutual_info_regression : Mutual information for a continuous target. 
935      SelectPercentile : Select features based on percentile of the highest 
936          scores. 
937      SelectKBest : Select features based on the k highest scores. 
938      SelectFpr : Select features based on a false positive rate test. 
939      SelectFwe : Select features based on family-wise error rate. 
940      GenericUnivariateSelect : Univariate feature selector with configurable 
941          mode. 
942   
943      References 
944      ---------- 
945      https://en.wikipedia.org/wiki/False_discovery_rate 
946   
947      Examples 
948      -------- 
949      >>> from sklearn.datasets import load_breast_cancer 
950      >>> from sklearn.feature_selection import SelectFdr, chi2 
951      >>> X, y = load_breast_cancer(return_X_y=True) 
952      >>> X.shape 
953      (569, 30) 
954      >>> X_new = SelectFdr(chi2, alpha=0.01).fit_transform(X, y) 
955      >>> X_new.shape 
956      (569, 16) 
957      """
958  
959      _parameter_constraints: dict = {
960          **_BaseFilter._parameter_constraints,
961          "alpha": [Interval(Real, 0, 1, closed="both")],
962      }
963  
964      def __init__(self, score_func=f_classif, *, alpha=5e-2):
965          super().__init__(score_func=score_func)
966          self.alpha = alpha
967  
968      def _get_support_mask(self):
969          check_is_fitted(self)
970  
971          n_features = len(self.pvalues_)
972          sv = np.sort(self.pvalues_)
973          selected = sv[
974              sv <= float(self.alpha) / n_features * np.arange(1, n_features + 1)
975          ]
976          if selected.size == 0:
977              return np.zeros_like(self.pvalues_, dtype=bool)
978          return self.pvalues_ <= selected.max()
