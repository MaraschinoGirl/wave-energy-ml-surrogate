6    @keras_export("keras.activations.relu")
7    def relu(x, negative_slope=0.0, max_value=None, threshold=0.0):
8        """Applies the rectified linear unit activation function. 
9     
10       With default values, this returns the standard ReLU activation: 
11       `max(x, 0)`, the element-wise maximum of 0 and the input tensor. 
12    
13       Modifying default parameters allows you to use non-zero thresholds, 
14       change the max value of the activation, 
15       and to use a non-zero multiple of the input for values below the threshold. 
16    
17       Examples: 
18    
19       >>> x = [-10, -5, 0.0, 5, 10] 
20       >>> keras.activations.relu(x) 
21       [ 0.,  0.,  0.,  5., 10.] 
22       >>> keras.activations.relu(x, negative_slope=0.5) 
23       [-5. , -2.5,  0. ,  5. , 10. ] 
24       >>> keras.activations.relu(x, max_value=5.) 
25       [0., 0., 0., 5., 5.] 
26       >>> keras.activations.relu(x, threshold=5.) 
27       [-0., -0.,  0.,  0., 10.] 
28    
29       Args: 
30           x: Input tensor. 
31           negative_slope: A `float` that controls the slope 
32               for values lower than the threshold. 
33           max_value: A `float` that sets the saturation threshold (the largest 
34               value the function will return). 
35           threshold: A `float` giving the threshold value of the activation 
36               function below which values will be damped or set to zero. 
37    
38       Returns: 
39           A tensor with the same shape and dtype as input `x`. 
40       """
41       if backend.any_symbolic_tensors((x,)):
42           return ReLU(
43               negative_slope=negative_slope,
44               max_value=max_value,
45               threshold=threshold,
46           )(x)
47       return ReLU.static_call(
48           x,
49           negative_slope=negative_slope,
50           max_value=max_value,
51           threshold=threshold,
52       )
53   
54   
55   class ReLU(ops.Operation):
56       def __init__(
57           self, negative_slope=0.0, max_value=None, threshold=0.0, name=None
58       ):
59           super().__init__(name=name)
60           self.negative_slope = negative_slope
61           self.max_value = max_value
62           self.threshold = threshold
63   
64       def call(self, x):
65           return self.static_call(
66               x,
67               negative_slope=self.negative_slope,
68               max_value=self.max_value,
69               threshold=self.threshold,
70           )
71   
72       def compute_output_spec(self, x):
73           return backend.KerasTensor(x.shape, x.dtype)
74   
75       @staticmethod
76       def static_call(x, negative_slope=0.0, max_value=None, threshold=0.0):
77           x = backend.convert_to_tensor(x)
78           if negative_slope != 0.0:
79               if max_value is None and threshold == 0:
80                   return backend.nn.leaky_relu(x, negative_slope=negative_slope)
81   
82               if threshold != 0:
83                   negative_part = backend.nn.relu(-x + threshold)
84               else:
85                   negative_part = backend.nn.relu(-x)
86           else:
87               negative_part = 1
88   
89           clip_max = max_value is not None
90           if threshold != 0:
91               # computes x for x > threshold else 0
92               threshold = ops.cast(threshold, dtype=x.dtype)
93               x = x * backend.cast(
94                   backend.numpy.greater(x, threshold), dtype=x.dtype
95               )
96           elif max_value == 6:
97               # if no threshold, then can use nn.relu6 native op for performance
98               x = backend.nn.relu6(x)
99               clip_max = False
100          else:
101              x = backend.nn.relu(x)
102  
103          if clip_max:
104              min_value = ops.cast(0.0, dtype=x.dtype)
105              max_value = ops.cast(max_value, dtype=x.dtype)
106              x = backend.numpy.clip(x, min_value, max_value)
107  
108          if negative_slope != 0.0:
109              x -= negative_slope * negative_part
110          return x

137  @keras_export("keras.activations.softmax")
138  def softmax(x, axis=-1):
139      """Softmax converts a vector of values to a probability distribution. 
140   
141      The elements of the output vector are in range `[0, 1]` and sum to 1. 
142   
143      Each input vector is handled independently. 
144      The `axis` argument sets which axis of the input the function 
145      is applied along. 
146   
147      Softmax is often used as the activation for the last 
148      layer of a classification network because the result could be interpreted as 
149      a probability distribution. 
150   
151      The softmax of each vector x is computed as 
152      `exp(x) / sum(exp(x))`. 
153   
154      The input values in are the log-odds of the resulting probability. 
155   
156      Args: 
157          x: Input tensor. 
158          axis: Integer, axis along which the softmax is applied. 
159      """
160      output = ops.softmax(x, axis=axis)
161      # Cache the logits to use for crossentropy loss.
162      try:
163          output._keras_logits = x
164      except AttributeError:
165          # We're dealing with a C-type.
166          pass
167      return output

170  @keras_export("keras.activations.elu")
171  def elu(x, alpha=1.0):
172      """Exponential Linear Unit. 
173   
174      The exponential linear unit (ELU) with `alpha > 0` is defined as: 
175   
176      - `x` if `x > 0` 
177      - alpha * `exp(x) - 1` if `x < 0` 
178   
179      ELUs have negative values which pushes the mean of the activations 
180      closer to zero. 
181   
182      Mean activations that are closer to zero enable faster learning as they 
183      bring the gradient closer to the natural gradient. 
184      ELUs saturate to a negative value when the argument gets smaller. 
185      Saturation means a small derivative which decreases the variation 
186      and the information that is propagated to the next layer. 
187   
188      Args: 
189          x: Input tensor. 
190   
191      Reference: 
192   
193      - [Clevert et al., 2016](https://arxiv.org/abs/1511.07289) 
194      """
195      return ops.elu(x, alpha=alpha)

198  @keras_export("keras.activations.selu")
199  def selu(x):
200      """Scaled Exponential Linear Unit (SELU). 
201   
202      The Scaled Exponential Linear Unit (SELU) activation function is defined as: 
203   
204      - `scale * x` if `x > 0` 
205      - `scale * alpha * (exp(x) - 1)` if `x < 0` 
206   
207      where `alpha` and `scale` are pre-defined constants 
208      (`alpha=1.67326324` and `scale=1.05070098`). 
209   
210      Basically, the SELU activation function multiplies `scale` (> 1) with the 
211      output of the `keras.activations.elu` function to ensure a slope larger 
212      than one for positive inputs. 
213   
214      The values of `alpha` and `scale` are 
215      chosen so that the mean and variance of the inputs are preserved 
216      between two consecutive layers as long as the weights are initialized 
217      correctly (see `keras.initializers.LecunNormal` initializer) 
218      and the number of input units is "large enough" 
219      (see reference paper for more information). 
220   
221      Args: 
222          x: Input tensor. 
223   
224      Notes: 
225   
226      - To be used together with the 
227          `keras.initializers.LecunNormal` initializer. 
228      - To be used together with the dropout variant 
229          `keras.layers.AlphaDropout` (rather than regular dropout). 
230   
231      Reference: 
232   
233      - [Klambauer et al., 2017](https://arxiv.org/abs/1706.02515) 
234      """
235      return ops.selu(x)

481  @keras_export("keras.activations.sigmoid")
482  def sigmoid(x):
483      """Sigmoid activation function. 
484   
485      It is defined as: `sigmoid(x) = 1 / (1 + exp(-x))`. 
486   
487      For small values (<-5), 
488      `sigmoid` returns a value close to zero, and for large values (>5) 
489      the result of the function gets close to 1. 
490   
491      Sigmoid is equivalent to a 2-element softmax, where the second element is 
492      assumed to be zero. The sigmoid function always returns a value between 
493      0 and 1. 
494   
495      Args: 
496          x: Input tensor. 
497      """
498      output = ops.sigmoid(x)
499      # Cache the logits to use for crossentropy loss.
500      try:
501          output._keras_logits = x
502      except AttributeError:
503          # We're dealing with a C-type.
504          pass
505      return output
