38   # The following classes and functions are mainly intended to be used from
39   # an interactive Python session
40   class UserModuleDeleter:
41       """ 
42       User Module Deleter (UMD) aims at deleting user modules 
43       to force Python to deeply reload them during import 
44    
45       pathlist [list]: ignore list in terms of module path 
46       namelist [list]: ignore list in terms of module name 
47       """
48   
49       def __init__(self, namelist=None, pathlist=None):
50           if namelist is None:
51               namelist = []
52           self.namelist = namelist
53           if pathlist is None:
54               pathlist = []
55           self.pathlist = pathlist
56           try:
57               # ignore all files in org.python.pydev/pysrc
58               import pydev_pysrc, inspect
59   
60               self.pathlist.append(os.path.dirname(pydev_pysrc.__file__))
61           except:
62               pass
63           self.previous_modules = list(sys.modules.keys())
64   
65       def is_module_ignored(self, modname, modpath):
66           for path in [sys.prefix] + self.pathlist:
67               if modpath.startswith(path):
68                   return True
69           else:
70               return set(modname.split(".")) & set(self.namelist)
71   
72       def run(self, verbose=False):
73           """ 
74           Del user modules to force Python to deeply reload them 
75    
76           Do not del modules which are considered as system modules, i.e. 
77           modules installed in subdirectories of Python interpreter's binary 
78           Do not del C modules 
79           """
80           log = []
81           modules_copy = dict(sys.modules)
82           for modname, module in modules_copy.items():
83               if modname == "aaaaa":
84                   print(modname, module)
85                   print(self.previous_modules)
86               if modname not in self.previous_modules:
87                   modpath = getattr(module, "__file__", None)
88                   if modpath is None:
89                       # *module* is a C module that is statically linked into the
90                       # interpreter. There is no way to know its path, so we
91                       # choose to ignore it.
92                       continue
93                   if not self.is_module_ignored(modname, modpath):
94                       log.append(modname)
95                       del sys.modules[modname]
96           if verbose and log:
97               print("\x1b[4;33m%s\x1b[24m%s\x1b[0m" % ("UMD has deleted", ": " + ", ".join(log)))

105  def _set_globals_function(get_globals):
106      global _get_globals_callback
107      _get_globals_callback = get_globals

110  def _get_globals():
111      """Return current Python interpreter globals namespace"""
112      if _get_globals_callback is not None:
113          return _get_globals_callback()
114      else:
115          try:
116              from __main__ import __dict__ as namespace
117          except ImportError:
118              try:
119                  # The import fails on IronPython
120                  import __main__
121  
122                  namespace = __main__.__dict__
123              except:
124                  namespace
125          shell = namespace.get("__ipythonshell__")
126          if shell is not None and hasattr(shell, "user_ns"):
127              # IPython 0.12+ kernel
128              return shell.user_ns
129          else:
130              # Python interpreter
131              return namespace
132          return namespace

135  def runfile(filename, args=None, wdir=None, namespace=None):
136      """ 
137      Run filename 
138      args: command line arguments (string) 
139      wdir: working directory 
140      """
141      try:
142          if hasattr(filename, "decode"):
143              filename = filename.decode("utf-8")
144      except (UnicodeError, TypeError):
145          pass
146      global __umd__
147      if os.environ.get("PYDEV_UMD_ENABLED", "").lower() == "true":
148          if __umd__ is None:
149              namelist = os.environ.get("PYDEV_UMD_NAMELIST", None)
150              if namelist is not None:
151                  namelist = namelist.split(",")
152              __umd__ = UserModuleDeleter(namelist=namelist)
153          else:
154              verbose = os.environ.get("PYDEV_UMD_VERBOSE", "").lower() == "true"
155              __umd__.run(verbose=verbose)
156      if args is not None and not isinstance(args, (bytes, str)):
157          raise TypeError("expected a character buffer object")
158      if namespace is None:
159          namespace = _get_globals()
160      if "__file__" in namespace:
161          old_file = namespace["__file__"]
162      else:
163          old_file = None
164      namespace["__file__"] = filename
165      sys.argv = [filename]
166      if args is not None:
167          for arg in args.split():
168              sys.argv.append(arg)
169      if wdir is not None:
170          try:
171              if hasattr(wdir, "decode"):
172                  wdir = wdir.decode("utf-8")
173          except (UnicodeError, TypeError):
174              pass
175          os.chdir(wdir)
176      execfile(filename, namespace)
177      sys.argv = [""]
178      if old_file is None:
179          del namespace["__file__"]
180      else:
181          namespace["__file__"] = old_file
