1    # This is a sample ISAPI extension written in Python.
2    #
3    # Please see README.txt in this directory, and specifically the
4    # information about the "loader" DLL - installing this sample will create
5    # "_redirector.dll" in the current directory.  The readme explains this.
6    
7    # Executing this script (or any server config script) will install the extension
8    # into your web server. As the server executes, the PyISAPI framework will load
9    # this module and create your Extension and Filter objects.
10   
11   # This is the simplest possible redirector (or proxy) we can write.  The
12   # extension installs with a mask of '*' in the root of the site.
13   # As an added bonus though, we optionally show how, on IIS6 and later, we
14   # can use HSE_ERQ_EXEC_URL to ignore certain requests - in IIS5 and earlier
15   # we can only do this with an ISAPI filter - see redirector_with_filter for
16   # an example.  If this sample is run on IIS5 or earlier it simply ignores
17   # any excludes.
18   
19   import sys
20   from urllib.request import urlopen
21   
22   import win32api
23   
24   from isapi import isapicon, threaded_extension
25   
26   # sys.isapidllhandle will exist when we are loaded by the IIS framework.
27   # In this case we redirect our output to the win32traceutil collector.
28   if hasattr(sys, "isapidllhandle"):
29       import win32traceutil
30   
31   # The site we are proxying.
32   proxy = "https://www.python.org"
33   
34   # Urls we exclude (ie, allow IIS to handle itself) - all are lowered,
35   # and these entries exist by default on Vista...
36   excludes = ["/iisstart.htm", "/welcome.png"]
37   
38   
39   # An "io completion" function, called when ecb.ExecURL completes...
40   def io_callback(ecb, url, cbIO, errcode):
41       # Get the status of our ExecURL
42       httpstatus, substatus, win32 = ecb.GetExecURLStatus()
43       print(
44           "ExecURL of %r finished with http status %d.%d, win32 status %d (%s)"
45           % (url, httpstatus, substatus, win32, win32api.FormatMessage(win32).strip())
46       )
47       # nothing more to do!
48       ecb.DoneWithSession()
49   
50   
51   # The ISAPI extension - handles all requests in the site.
52   class Extension(threaded_extension.ThreadPoolExtension):
53       "Python sample Extension"
54   
55       def Dispatch(self, ecb):
56           # Note that our ThreadPoolExtension base class will catch exceptions
57           # in our Dispatch method, and write the traceback to the client.
58           # That is perfect for this sample, so we don't catch our own.
59           # print(f'IIS dispatching "{ecb.GetServerVariable("URL")}"')
60           url = ecb.GetServerVariable("URL").decode("ascii")
61           for exclude in excludes:
62               if url.lower().startswith(exclude):
63                   print("excluding %s" % url)
64                   if ecb.Version < 0x60000:
65                       print("(but this is IIS5 or earlier - can't do 'excludes')")
66                   else:
67                       ecb.IOCompletion(io_callback, url)
68                       ecb.ExecURL(
69                           None,
70                           None,
71                           None,
72                           None,
73                           None,
74                           isapicon.HSE_EXEC_URL_IGNORE_CURRENT_INTERCEPTOR,
75                       )
76                       return isapicon.HSE_STATUS_PENDING
77   
78           new_url = proxy + url
79           print("Opening %s" % new_url)
80           fp = urlopen(new_url)
81           headers = fp.info()
82           # subtle breakage: str(headers) normalizes \r\n
83           # back to \n and also sticks an extra \n term.
84           # take *all* trailing \n off, replace remaining with
85           # \r\n, then add the 2 trailing \r\n.
86           header_text = str(headers).rstrip("\n").replace("\n", "\r\n") + "\r\n\r\n"
87           ecb.SendResponseHeaders("200 OK", header_text, False)
88           ecb.WriteClient(fp.read())
89           ecb.DoneWithSession()
90           print(f"Returned data from '{new_url}'")
91           return isapicon.HSE_STATUS_SUCCESS
92   
93   
94   # The entry points for the ISAPI extension.
95   def __ExtensionFactory__():
96       return Extension()
97   
98   
99   if __name__ == "__main__":
100      # If run from the command-line, install ourselves.
101      from isapi.install import *
102  
103      params = ISAPIParameters()
104      # Setup the virtual directories - this is a list of directories our
105      # extension uses - in this case only 1.
106      # Each extension has a "script map" - this is the mapping of ISAPI
107      # extensions.
108      sm = [ScriptMapParams(Extension="*", Flags=0)]
109      vd = VirtualDirParameters(
110          Name="/",
111          Description=Extension.__doc__,
112          ScriptMaps=sm,
113          ScriptMapUpdate="replace",
114      )
115      params.VirtualDirs = [vd]
116      HandleCommandLine(params)
