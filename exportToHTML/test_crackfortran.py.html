145  class TestDimSpec(util.F2PyTest):
146      """This test suite tests various expressions that are used as dimension 
147      specifications. 
148   
149      There exists two usage cases where analyzing dimensions 
150      specifications are important. 
151   
152      In the first case, the size of output arrays must be defined based 
153      on the inputs to a Fortran function. Because Fortran supports 
154      arbitrary bases for indexing, for instance, `arr(lower:upper)`, 
155      f2py has to evaluate an expression `upper - lower + 1` where 
156      `lower` and `upper` are arbitrary expressions of input parameters. 
157      The evaluation is performed in C, so f2py has to translate Fortran 
158      expressions to valid C expressions (an alternative approach is 
159      that a developer specifies the corresponding C expressions in a 
160      .pyf file). 
161   
162      In the second case, when user provides an input array with a given 
163      size but some hidden parameters used in dimensions specifications 
164      need to be determined based on the input array size. This is a 
165      harder problem because f2py has to solve the inverse problem: find 
166      a parameter `p` such that `upper(p) - lower(p) + 1` equals to the 
167      size of input array. In the case when this equation cannot be 
168      solved (e.g. because the input array size is wrong), raise an 
169      error before calling the Fortran function (that otherwise would 
170      likely crash Python process when the size of input arrays is 
171      wrong). f2py currently supports this case only when the equation 
172      is linear with respect to unknown parameter. 
173   
174      """
175  
176      suffix = ".f90"
177  
178      code_template = textwrap.dedent(""" 
179        function get_arr_size_{count}(a, n) result (length) 
180          integer, intent(in) :: n 
181          integer, dimension({dimspec}), intent(out) :: a 
182          integer length 
183          length = size(a) 
184        end function 
185   
186        subroutine get_inv_arr_size_{count}(a, n) 
187          integer :: n 
188          ! the value of n is computed in f2py wrapper 
189          !f2py intent(out) n 
190          integer, dimension({dimspec}), intent(in) :: a 
191          if (a({first}).gt.0) then 
192            ! print*, "a=", a 
193          endif 
194        end subroutine 
195      """)
196  
197      linear_dimspecs = [
198          "n", "2*n", "2:n", "n/2", "5 - n/2", "3*n:20", "n*(n+1):n*(n+5)",
199          "2*n, n"
200      ]
201      nonlinear_dimspecs = ["2*n:3*n*n+2*n"]
202      all_dimspecs = linear_dimspecs + nonlinear_dimspecs
203  
204      code = ""
205      for count, dimspec in enumerate(all_dimspecs):
206          lst = [(d.split(":")[0] if ":" in d else "1") for d in dimspec.split(',')]
207          code += code_template.format(
208              count=count,
209              dimspec=dimspec,
210              first=", ".join(lst),
211          )
212  
213      @pytest.mark.parametrize("dimspec", all_dimspecs)
214      @pytest.mark.slow
215      def test_array_size(self, dimspec):
216  
217          count = self.all_dimspecs.index(dimspec)
218          get_arr_size = getattr(self.module, f"get_arr_size_{count}")
219  
220          for n in [1, 2, 3, 4, 5]:
221              sz, a = get_arr_size(n)
222              assert a.size == sz
223  
224      @pytest.mark.parametrize("dimspec", all_dimspecs)
225      def test_inv_array_size(self, dimspec):
226  
227          count = self.all_dimspecs.index(dimspec)
228          get_arr_size = getattr(self.module, f"get_arr_size_{count}")
229          get_inv_arr_size = getattr(self.module, f"get_inv_arr_size_{count}")
230  
231          for n in [1, 2, 3, 4, 5]:
232              sz, a = get_arr_size(n)
233              if dimspec in self.nonlinear_dimspecs:
234                  # one must specify n as input, the call we'll ensure
235                  # that a and n are compatible:
236                  n1 = get_inv_arr_size(a, n)
237              else:
238                  # in case of linear dependence, n can be determined
239                  # from the shape of a:
240                  n1 = get_inv_arr_size(a)
241              # n1 may be different from n (for instance, when `a` size
242              # is a function of some `n` fraction) but it must produce
243              # the same sized array
244              sz1, _ = get_arr_size(n1)
245              assert sz == sz1, (n, n1, sz, sz1)

293  class TestNameArgsPatternBacktracking:
294      @pytest.mark.parametrize(
295          ['adversary'],
296          [
297              ('@)@bind@(@',),
298              ('@)@bind                         @(@',),
299              ('@)@bind foo bar baz@(@',)
300          ]
301      )
302      def test_nameargspattern_backtracking(self, adversary):
303          '''address ReDOS vulnerability: 
304          https://github.com/numpy/numpy/issues/23338'''
305          trials_per_batch = 12
306          batches_per_regex = 4
307          start_reps, end_reps = 15, 25
308          for ii in range(start_reps, end_reps):
309              repeated_adversary = adversary * ii
310              # test times in small batches.
311              # this gives us more chances to catch a bad regex
312              # while still catching it before too long if it is bad
313              for _ in range(batches_per_regex):
314                  times = []
315                  for _ in range(trials_per_batch):
316                      t0 = time.perf_counter()
317                      mtch = nameargspattern.search(repeated_adversary)
318                      times.append(time.perf_counter() - t0)
319                  # our pattern should be much faster than 0.2s per search
320                  # it's unlikely that a bad regex will pass even on fast CPUs
321                  assert np.median(times) < 0.2
322              assert not mtch
323              # if the adversary is capped with @)@, it becomes acceptable
324              # according to the old version of the regex.
325              # that should still be true.
326              good_version_of_adversary = repeated_adversary + '@)@'
327              assert nameargspattern.search(good_version_of_adversary)
