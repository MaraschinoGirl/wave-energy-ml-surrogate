9    def get_sample_problem():
10       # A random 10 x 10 symmetric matrix
11       rng = np.random.RandomState(1234)
12       matrix = rng.rand(10, 10)
13       matrix = matrix + matrix.T
14       # A random vector of length 10
15       vector = rng.rand(10)
16       return matrix, vector

19   def test_singular():
20       A, b = get_sample_problem()
21       A[0, ] = 0
22       b[0] = 0
23       xp, info = minres(A, b)
24       assert_equal(info, 0)
25       assert norm(A @ xp - b) <= 1e-5 * norm(b)

28   def test_x0_is_used_by():
29       A, b = get_sample_problem()
30       # Random x0 to feed minres
31       rng = np.random.RandomState(12345)
32       x0 = rng.rand(10)
33       trace = []
34   
35       def trace_iterates(xk):
36           trace.append(xk)
37       minres(A, b, x0=x0, callback=trace_iterates)
38       trace_with_x0 = trace
39   
40       trace = []
41       minres(A, b, callback=trace_iterates)
42       assert_(not np.array_equal(trace_with_x0[0], trace[0]))

45   def test_shift():
46       A, b = get_sample_problem()
47       shift = 0.5
48       shifted_A = A - shift * np.eye(10)
49       x1, info1 = minres(A, b, shift=shift)
50       x2, info2 = minres(shifted_A, b)
51       assert_equal(info1, 0)
52       assert_allclose(x1, x2, rtol=1e-5)
