99   class TestEnv:
100      """Set Jupyter path variables to a temporary directory 
101   
102      Useful as a context manager or with explicit start/stop 
103      """
104  
105      def start(self):
106          self.test_dir = td = TemporaryDirectory()
107          self.env_patch = patch.dict(
108              os.environ,
109              {
110                  "JUPYTER_CONFIG_DIR": pjoin(td.name, "jupyter"),
111                  "JUPYTER_DATA_DIR": pjoin(td.name, "jupyter_data"),
112                  "JUPYTER_RUNTIME_DIR": pjoin(td.name, "jupyter_runtime"),
113                  "IPYTHONDIR": pjoin(td.name, "ipython"),
114              },
115          )
116          self.env_patch.start()
117          self.path_patch = patch.multiple(
118              jupyter_core.paths,
119              SYSTEM_JUPYTER_PATH=[pjoin(td.name, "share", "jupyter")],
120              ENV_JUPYTER_PATH=[pjoin(td.name, "env", "share", "jupyter")],
121              SYSTEM_CONFIG_PATH=[pjoin(td.name, "etc", "jupyter")],
122              ENV_CONFIG_PATH=[pjoin(td.name, "env", "etc", "jupyter")],
123          )
124          self.path_patch.start()
125  
126      def stop(self):
127          self.env_patch.stop()
128          self.path_patch.stop()
129          try:
130              self.test_dir.cleanup()
131          except OSError:
132              pass
133  
134      def __enter__(self):
135          self.start()
136          return self.test_dir.name
137  
138      def __exit__(self, *exc_info):
139          self.stop()

142  class ProcessTestApp(ProcessApp):
143      """A process app for running tests, includes a mock contents directory."""
144  
145      allow_origin = "*"
146  
147      def initialize_templates(self):
148          self.static_paths = [_create_static_dir()]
149          self.template_paths = [_create_template_dir()]
150  
151      def initialize_settings(self):
152          self.env_patch = TestEnv()
153          self.env_patch.start()
154          ProcessApp.__init__(self)
155  
156          self.settings["allow_origin"] = ProcessTestApp.allow_origin
157  
158          self.static_dir = self.static_paths[0]
159          self.template_dir = self.template_paths[0]
160          self.schemas_dir = _create_schemas_dir()
161          self.user_settings_dir = _create_user_settings_dir()
162          self.workspaces_dir = _create_workspaces_dir()
163  
164          self._install_default_kernels()
165          self.settings["kernel_manager"].default_kernel_name = "echo"
166  
167          super().initialize_settings()
168  
169      def _install_kernel(self, kernel_name, kernel_spec):
170          """Install a kernel spec to the data directory. 
171   
172          Parameters 
173          ---------- 
174          kernel_name: str 
175              Name of the kernel. 
176          kernel_spec: dict 
177              The kernel spec for the kernel 
178          """
179          paths = jupyter_core.paths
180          kernel_dir = pjoin(paths.jupyter_data_dir(), "kernels", kernel_name)
181          os.makedirs(kernel_dir)
182          with open(pjoin(kernel_dir, "kernel.json"), "w") as f:
183              f.write(json.dumps(kernel_spec))
184  
185      def _install_default_kernels(self):
186          # Install echo and ipython kernels - should be done after env patch
187          self._install_kernel(
188              kernel_name="echo",
189              kernel_spec={
190                  "argv": [
191                      sys.executable,
192                      "-m",
193                      "jupyterlab.tests.echo_kernel",
194                      "-f",
195                      "{connection_file}",
196                  ],
197                  "display_name": "Echo Kernel",
198                  "language": "echo",
199              },
200          )
201  
202          paths = jupyter_core.paths
203          ipykernel_dir = pjoin(paths.jupyter_data_dir(), "kernels", "ipython")
204          write_kernel_spec(ipykernel_dir)
205  
206      def _process_finished(self, future):
207          self.serverapp.http_server.stop()
208          self.serverapp.io_loop.stop()
209          self.env_patch.stop()
210          try:
211              os._exit(future.result())
212          except Exception as e:
213              self.log.error(str(e))
214              os._exit(1)

217  class RootedServerApp(ServerApp):
218      @default("root_dir")
219      def _default_root_dir(self):
220          """Create a temporary directory with some file structure."""
221          root_dir = tempfile.mkdtemp(prefix="mock_root")
222          os.mkdir(osp.join(root_dir, "src"))
223          with open(osp.join(root_dir, "src", "temp.txt"), "w") as fid:
224              fid.write("hello")
225  
226          readonly_filepath = osp.join(root_dir, "src", "readonly-temp.txt")
227          with open(readonly_filepath, "w") as fid:
228              fid.write("hello from a readonly file")
229  
230          os.chmod(readonly_filepath, S_IRUSR | S_IRGRP | S_IROTH)
231          atexit.register(lambda: shutil.rmtree(root_dir, True))
232          return root_dir
