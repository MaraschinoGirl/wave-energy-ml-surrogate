170      def _artist_kws(self, kws, fill, element, multiple, color, alpha):
171          """Handle differences between artists in filled/unfilled plots."""
172          kws = kws.copy()
173          if fill:
174              kws = normalize_kwargs(kws, mpl.collections.PolyCollection)
175              kws.setdefault("facecolor", to_rgba(color, alpha))
176  
177              if element == "bars":
178                  # Make bar() interface with property cycle correctly
179                  # https://github.com/matplotlib/matplotlib/issues/19385
180                  kws["color"] = "none"
181  
182              if multiple in ["stack", "fill"] or element == "bars":
183                  kws.setdefault("edgecolor", mpl.rcParams["patch.edgecolor"])
184              else:
185                  kws.setdefault("edgecolor", to_rgba(color, 1))
186          elif element == "bars":
187              kws["facecolor"] = "none"
188              kws["edgecolor"] = to_rgba(color, alpha)
189          else:
190              kws["color"] = to_rgba(color, alpha)
191          return kws

193      def _quantile_to_level(self, data, quantile):
194          """Return data levels corresponding to quantile cuts of mass."""
195          isoprop = np.asarray(quantile)
196          values = np.ravel(data)
197          sorted_values = np.sort(values)[::-1]
198          normalized_values = np.cumsum(sorted_values) / values.sum()
199          idx = np.searchsorted(normalized_values, 1 - isoprop)
200          levels = np.take(sorted_values, idx, mode="clip")
201          return levels

203      def _cmap_from_color(self, color):
204          """Return a sequential colormap given a color seed."""
205          # Like so much else here, this is broadly useful, but keeping it
206          # in this class to signify that I haven't thought overly hard about it...
207          r, g, b, _ = to_rgba(color)
208          h, s, _ = husl.rgb_to_husl(r, g, b)
209          xx = np.linspace(-1, 1, int(1.15 * 256))[:256]
210          ramp = np.zeros((256, 3))
211          ramp[:, 0] = h
212          ramp[:, 1] = s * np.cos(xx)
213          ramp[:, 2] = np.linspace(35, 80, 256)
214          colors = np.clip([husl.husl_to_rgb(*hsl) for hsl in ramp], 0, 1)
215          return mpl.colors.ListedColormap(colors[::-1])

227      def _resolve_multiple(self, curves, multiple):
228          """Modify the density data structure to handle multiple densities."""
229  
230          # Default baselines have all densities starting at 0
231          baselines = {k: np.zeros_like(v) for k, v in curves.items()}
232  
233          # TODO we should have some central clearinghouse for checking if any
234          # "grouping" (terminnology?) semantics have been assigned
235          if "hue" not in self.variables:
236              return curves, baselines
237  
238          if multiple in ("stack", "fill"):
239  
240              # Setting stack or fill means that the curves share a
241              # support grid / set of bin edges, so we can make a dataframe
242              # Reverse the column order to plot from top to bottom
243              curves = pd.DataFrame(curves).iloc[:, ::-1]
244  
245              # Find column groups that are nested within col/row variables
246              column_groups = {}
247              for i, keyd in enumerate(map(dict, curves.columns)):
248                  facet_key = keyd.get("col", None), keyd.get("row", None)
249                  column_groups.setdefault(facet_key, [])
250                  column_groups[facet_key].append(i)
251  
252              baselines = curves.copy()
253  
254              for col_idxs in column_groups.values():
255                  cols = curves.columns[col_idxs]
256  
257                  norm_constant = curves[cols].sum(axis="columns")
258  
259                  # Take the cumulative sum to stack
260                  curves[cols] = curves[cols].cumsum(axis="columns")
261  
262                  # Normalize by row sum to fill
263                  if multiple == "fill":
264                      curves[cols] = curves[cols].div(norm_constant, axis="index")
265  
266                  # Define where each segment starts
267                  baselines[cols] = curves[cols].shift(1, axis=1).fillna(0)
268  
269          if multiple == "dodge":
270  
271              # Account for the unique semantic (non-faceting) levels
272              # This will require rethiniking if we add other semantics!
273              hue_levels = self.var_levels["hue"]
274              n = len(hue_levels)
275              f_fwd, f_inv = self._get_scale_transforms(self.data_variable)
276              for key in curves:
277  
278                  level = dict(key)["hue"]
279                  hist = curves[key].reset_index(name="heights")
280                  level_idx = hue_levels.index(level)
281  
282                  a = f_fwd(hist["edges"])
283                  b = f_fwd(hist["edges"] + hist["widths"])
284                  w = (b - a) / n
285                  new_min = f_inv(a + level_idx * w)
286                  new_max = f_inv(a + (level_idx + 1) * w)
287                  hist["widths"] = new_max - new_min
288                  hist["edges"] = new_min
289  
290                  curves[key] = hist.set_index(["edges", "widths"])["heights"]
291  
292          return curves, baselines

298      def _compute_univariate_density(
299          self,
300          data_variable,
301          common_norm,
302          common_grid,
303          estimate_kws,
304          warn_singular=True,
305      ):
306  
307          # Initialize the estimator object
308          estimator = KDE(**estimate_kws)
309  
310          if set(self.variables) - {"x", "y"}:
311              if common_grid:
312                  all_observations = self.comp_data.dropna()
313                  estimator.define_support(all_observations[data_variable])
314          else:
315              common_norm = False
316  
317          all_data = self.plot_data.dropna()
318          if common_norm and "weights" in all_data:
319              whole_weight = all_data["weights"].sum()
320          else:
321              whole_weight = len(all_data)
322  
323          densities = {}
324  
325          for sub_vars, sub_data in self.iter_data("hue", from_comp_data=True):
326  
327              # Extract the data points from this sub set and remove nulls
328              observations = sub_data[data_variable]
329  
330              # Extract the weights for this subset of observations
331              if "weights" in self.variables:
332                  weights = sub_data["weights"]
333                  part_weight = weights.sum()
334              else:
335                  weights = None
336                  part_weight = len(sub_data)
337  
338              # Estimate the density of observations at this level
339              variance = np.nan_to_num(observations.var())
340              singular = len(observations) < 2 or math.isclose(variance, 0)
341              try:
342                  if not singular:
343                      # Convoluted approach needed because numerical failures
344                      # can manifest in a few different ways.
345                      density, support = estimator(observations, weights=weights)
346              except np.linalg.LinAlgError:
347                  singular = True
348  
349              if singular:
350                  msg = (
351                      "Dataset has 0 variance; skipping density estimate. "
352                      "Pass `warn_singular=False` to disable this warning."
353                  )
354                  if warn_singular:
355                      warnings.warn(msg, UserWarning, stacklevel=4)
356                  continue
357  
358              # Invert the scaling of the support points
359              _, f_inv = self._get_scale_transforms(self.data_variable)
360              support = f_inv(support)
361  
362              # Apply a scaling factor so that the integral over all subsets is 1
363              if common_norm:
364                  density *= part_weight / whole_weight
365  
366              # Store the density for this level
367              key = tuple(sub_vars.items())
368              densities[key] = pd.Series(density, index=support)
369  
370          return densities

376      def plot_univariate_histogram(
377          self,
378          multiple,
379          element,
380          fill,
381          common_norm,
382          common_bins,
383          shrink,
384          kde,
385          kde_kws,
386          color,
387          legend,
388          line_kws,
389          estimate_kws,
390          **plot_kws,
391      ):
392  
393          # -- Default keyword dicts
394          kde_kws = {} if kde_kws is None else kde_kws.copy()
395          line_kws = {} if line_kws is None else line_kws.copy()
396          estimate_kws = {} if estimate_kws is None else estimate_kws.copy()
397  
398          # --  Input checking
399          _check_argument("multiple", ["layer", "stack", "fill", "dodge"], multiple)
400          _check_argument("element", ["bars", "step", "poly"], element)
401  
402          auto_bins_with_weights = (
403              "weights" in self.variables
404              and estimate_kws["bins"] == "auto"
405              and estimate_kws["binwidth"] is None
406              and not estimate_kws["discrete"]
407          )
408          if auto_bins_with_weights:
409              msg = (
410                  "`bins` cannot be 'auto' when using weights. "
411                  "Setting `bins=10`, but you will likely want to adjust."
412              )
413              warnings.warn(msg, UserWarning)
414              estimate_kws["bins"] = 10
415  
416          # Simplify downstream code if we are not normalizing
417          if estimate_kws["stat"] == "count":
418              common_norm = False
419  
420          orient = self.data_variable
421  
422          # Now initialize the Histogram estimator
423          estimator = Hist(**estimate_kws)
424          histograms = {}
425  
426          # Do pre-compute housekeeping related to multiple groups
427          all_data = self.comp_data.dropna()
428          all_weights = all_data.get("weights", None)
429  
430          multiple_histograms = set(self.variables) - {"x", "y"}
431          if multiple_histograms:
432              if common_bins:
433                  bin_kws = estimator._define_bin_params(all_data, orient, None)
434          else:
435              common_norm = False
436  
437          if common_norm and all_weights is not None:
438              whole_weight = all_weights.sum()
439          else:
440              whole_weight = len(all_data)
441  
442          # Estimate the smoothed kernel densities, for use later
443          if kde:
444              # TODO alternatively, clip at min/max bins?
445              kde_kws.setdefault("cut", 0)
446              kde_kws["cumulative"] = estimate_kws["cumulative"]
447              densities = self._compute_univariate_density(
448                  self.data_variable,
449                  common_norm,
450                  common_bins,
451                  kde_kws,
452                  warn_singular=False,
453              )
454  
455          # First pass through the data to compute the histograms
456          for sub_vars, sub_data in self.iter_data("hue", from_comp_data=True):
457  
458              # Prepare the relevant data
459              key = tuple(sub_vars.items())
460              orient = self.data_variable
461  
462              if "weights" in self.variables:
463                  sub_data["weight"] = sub_data.pop("weights")
464                  part_weight = sub_data["weight"].sum()
465              else:
466                  part_weight = len(sub_data)
467  
468              # Do the histogram computation
469              if not (multiple_histograms and common_bins):
470                  bin_kws = estimator._define_bin_params(sub_data, orient, None)
471              res = estimator._normalize(estimator._eval(sub_data, orient, bin_kws))
472              heights = res[estimator.stat].to_numpy()
473              widths = res["space"].to_numpy()
474              edges = res[orient].to_numpy() - widths / 2
475  
476              # Rescale the smoothed curve to match the histogram
477              if kde and key in densities:
478                  density = densities[key]
479                  if estimator.cumulative:
480                      hist_norm = heights.max()
481                  else:
482                      hist_norm = (heights * widths).sum()
483                  densities[key] *= hist_norm
484  
485              # Convert edges back to original units for plotting
486              ax = self._get_axes(sub_vars)
487              _, inv = _get_transform_functions(ax, self.data_variable)
488              widths = inv(edges + widths) - inv(edges)
489              edges = inv(edges)
490  
491              # Pack the histogram data and metadata together
492              edges = edges + (1 - shrink) / 2 * widths
493              widths *= shrink
494              index = pd.MultiIndex.from_arrays([
495                  pd.Index(edges, name="edges"),
496                  pd.Index(widths, name="widths"),
497              ])
498              hist = pd.Series(heights, index=index, name="heights")
499  
500              # Apply scaling to normalize across groups
501              if common_norm:
502                  hist *= part_weight / whole_weight
503  
504              # Store the finalized histogram data for future plotting
505              histograms[key] = hist
506  
507          # Modify the histogram and density data to resolve multiple groups
508          histograms, baselines = self._resolve_multiple(histograms, multiple)
509          if kde:
510              densities, _ = self._resolve_multiple(
511                  densities, None if multiple == "dodge" else multiple
512              )
513  
514          # Set autoscaling-related meta
515          sticky_stat = (0, 1) if multiple == "fill" else (0, np.inf)
516          if multiple == "fill":
517              # Filled plots should not have any margins
518              bin_vals = histograms.index.to_frame()
519              edges = bin_vals["edges"]
520              widths = bin_vals["widths"]
521              sticky_data = (
522                  edges.min(),
523                  edges.max() + widths.loc[edges.idxmax()]
524              )
525          else:
526              sticky_data = []
527  
528          # --- Handle default visual attributes
529  
530          # Note: default linewidth is determined after plotting
531  
532          # Default alpha should depend on other parameters
533          if fill:
534              # Note: will need to account for other grouping semantics if added
535              if "hue" in self.variables and multiple == "layer":
536                  default_alpha = .5 if element == "bars" else .25
537              elif kde:
538                  default_alpha = .5
539              else:
540                  default_alpha = .75
541          else:
542              default_alpha = 1
543          alpha = plot_kws.pop("alpha", default_alpha)  # TODO make parameter?
544  
545          hist_artists = []
546  
547          # Go back through the dataset and draw the plots
548          for sub_vars, _ in self.iter_data("hue", reverse=True):
549  
550              key = tuple(sub_vars.items())
551              hist = histograms[key].rename("heights").reset_index()
552              bottom = np.asarray(baselines[key])
553  
554              ax = self._get_axes(sub_vars)
555  
556              # Define the matplotlib attributes that depend on semantic mapping
557              if "hue" in self.variables:
558                  sub_color = self._hue_map(sub_vars["hue"])
559              else:
560                  sub_color = color
561  
562              artist_kws = self._artist_kws(
563                  plot_kws, fill, element, multiple, sub_color, alpha
564              )
565  
566              if element == "bars":
567  
568                  # Use matplotlib bar plotting
569  
570                  plot_func = ax.bar if self.data_variable == "x" else ax.barh
571                  artists = plot_func(
572                      hist["edges"],
573                      hist["heights"] - bottom,
574                      hist["widths"],
575                      bottom,
576                      align="edge",
577                      **artist_kws,
578                  )
579  
580                  for bar in artists:
581                      if self.data_variable == "x":
582                          bar.sticky_edges.x[:] = sticky_data
583                          bar.sticky_edges.y[:] = sticky_stat
584                      else:
585                          bar.sticky_edges.x[:] = sticky_stat
586                          bar.sticky_edges.y[:] = sticky_data
587  
588                  hist_artists.extend(artists)
589  
590              else:
591  
592                  # Use either fill_between or plot to draw hull of histogram
593                  if element == "step":
594  
595                      final = hist.iloc[-1]
596                      x = np.append(hist["edges"], final["edges"] + final["widths"])
597                      y = np.append(hist["heights"], final["heights"])
598                      b = np.append(bottom, bottom[-1])
599  
600                      if self.data_variable == "x":
601                          step = "post"
602                          drawstyle = "steps-post"
603                      else:
604                          step = "post"  # fillbetweenx handles mapping internally
605                          drawstyle = "steps-pre"
606  
607                  elif element == "poly":
608  
609                      x = hist["edges"] + hist["widths"] / 2
610                      y = hist["heights"]
611                      b = bottom
612  
613                      step = None
614                      drawstyle = None
615  
616                  if self.data_variable == "x":
617                      if fill:
618                          artist = ax.fill_between(x, b, y, step=step, **artist_kws)
619                      else:
620                          artist, = ax.plot(x, y, drawstyle=drawstyle, **artist_kws)
621                      artist.sticky_edges.x[:] = sticky_data
622                      artist.sticky_edges.y[:] = sticky_stat
623                  else:
624                      if fill:
625                          artist = ax.fill_betweenx(x, b, y, step=step, **artist_kws)
626                      else:
627                          artist, = ax.plot(y, x, drawstyle=drawstyle, **artist_kws)
628                      artist.sticky_edges.x[:] = sticky_stat
629                      artist.sticky_edges.y[:] = sticky_data
630  
631                  hist_artists.append(artist)
632  
633              if kde:
634  
635                  # Add in the density curves
636  
637                  try:
638                      density = densities[key]
639                  except KeyError:
640                      continue
641                  support = density.index
642  
643                  if "x" in self.variables:
644                      line_args = support, density
645                      sticky_x, sticky_y = None, (0, np.inf)
646                  else:
647                      line_args = density, support
648                      sticky_x, sticky_y = (0, np.inf), None
649  
650                  line_kws["color"] = to_rgba(sub_color, 1)
651                  line, = ax.plot(
652                      *line_args, **line_kws,
653                  )
654  
655                  if sticky_x is not None:
656                      line.sticky_edges.x[:] = sticky_x
657                  if sticky_y is not None:
658                      line.sticky_edges.y[:] = sticky_y
659  
660          if element == "bars" and "linewidth" not in plot_kws:
661  
662              # Now we handle linewidth, which depends on the scaling of the plot
663  
664              # We will base everything on the minimum bin width
665              hist_metadata = pd.concat([
666                  # Use .items for generality over dict or df
667                  h.index.to_frame() for _, h in histograms.items()
668              ]).reset_index(drop=True)
669              thin_bar_idx = hist_metadata["widths"].idxmin()
670              binwidth = hist_metadata.loc[thin_bar_idx, "widths"]
671              left_edge = hist_metadata.loc[thin_bar_idx, "edges"]
672  
673              # Set initial value
674              default_linewidth = math.inf
675  
676              # Loop through subsets based only on facet variables
677              for sub_vars, _ in self.iter_data():
678  
679                  ax = self._get_axes(sub_vars)
680

2375 def _freedman_diaconis_bins(a):
2376     """Calculate number of hist bins using Freedman-Diaconis rule."""
2377     # From https://stats.stackexchange.com/questions/798/
2378     a = np.asarray(a)
2379     if len(a) < 2:
2380         return 1
2381     iqr = np.subtract.reduce(np.nanpercentile(a, [75, 25]))
2382     h = 2 * iqr / (len(a) ** (1 / 3))
2383     # fall back to sqrt(a) bins if iqr is 0
2384     if h == 0:
2385         return int(np.sqrt(a.size))
2386     else:
2387         return int(np.ceil((a.max() - a.min()) / h))
