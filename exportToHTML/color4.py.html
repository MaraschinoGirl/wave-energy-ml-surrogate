17   class Color:
18       """A specified color in a defined color space. 
19    
20       The color space is one of ``COLOR_SPACES``. 
21    
22       Coordinates are floats with undefined ranges, but alpha channel is clipped 
23       to [0, 1]. Coordinates can also be set to ``None`` when undefined. 
24    
25       """
26       def __init__(self, space, coordinates, alpha):
27           assert space in COLOR_SPACES, f"{space} is not a supported color space"
28           self.space = space
29           self.coordinates = tuple(
30               None if coordinate is None else float(coordinate)
31               for coordinate in coordinates)
32           self.alpha = max(0., min(1., float(alpha)))
33   
34       def __repr__(self):
35           coordinates = ' '.join(str(coordinate) for coordinate in self.coordinates)
36           return f'color({self.space} {coordinates} / {self.alpha})'
37   
38       def __iter__(self):
39           yield from self.coordinates
40           yield self.alpha
41   
42       def __getitem__(self, key):
43           return (*self.coordinates, self.alpha)[key]
44   
45       def __hash__(self):
46           return hash(str(self))
47   
48       def __eq__(self, other):
49           if isinstance(other, str):
50               return False
51           elif isinstance(other, tuple):
52               return tuple(self) == other
53           elif isinstance(other, Color):
54               return self.space == other.space and self.coordinates == other.coordinates
55           return super().__eq__(other)

57       def to(self, space):
58           """Return new instance with coordinates transformed to given ``space``. 
59    
60           The destination color space is one of ``SPACES``. 
61    
62           ``None`` coordinates are always transformed into ``0`` values. 
63    
64           Here are the supported combinations: 
65    
66           - from hsl and hwb to srgb; 
67           - from lab and lch to xyz-d50; 
68           - from oklab and oklch to xyz-d65; 
69           - from xyz-d50, xyz-d65, lch, oklab and oklch to lab. 
70    
71           """
72           coordinates = tuple(coordinate or 0 for coordinate in self.coordinates)
73           if space == 'xyz':
74               space = 'xyz-d65'
75           if space == self.space:
76               return Color(space, coordinates, self.alpha)
77           elif space == 'srgb':
78               if self.space == 'hsl':
79                   rgb = hls_to_rgb(
80                       coordinates[0] / 360,
81                       coordinates[2] / 100,
82                       coordinates[1] / 100,
83                   )
84                   return Color(space, rgb, self.alpha)
85               elif self.space == 'hwb':
86                   white, black = coordinates[1:]
87                   if white + black >= 100:
88                       rgb = (white / (white + black),) * 3
89                   else:
90                       rgb = (
91                           ((channel * (100 - white - black)) + white) / 100
92                           for channel in hls_to_rgb(coordinates[0] / 360, 0.5, 1))
93                   return Color(space, rgb, self.alpha)
94           elif space == 'xyz-d50':
95               if self.space == 'lab':
96                   xyz = _lab_to_xyz(*coordinates, D50)
97                   return Color(space, xyz, self.alpha)
98               elif self.space == 'lch':
99                   a = coordinates[1] * cos(radians(coordinates[2]))
100                  b = coordinates[1] * sin(radians(coordinates[2]))
101                  xyz = _lab_to_xyz(coordinates[0], a, b, D50)
102                  return Color(space, xyz, self.alpha)
103          elif space == 'xyz-d65':
104              if self.space == 'oklab':
105                  xyz = _oklab_to_xyz(*coordinates)
106                  return Color(space, xyz, self.alpha)
107              elif self.space == 'oklch':
108                  a = coordinates[1] * cos(radians(coordinates[2]))
109                  b = coordinates[1] * sin(radians(coordinates[2]))
110                  xyz = _oklab_to_xyz(coordinates[0], a, b)
111                  return Color(space, xyz, self.alpha)
112          elif space == 'lab':
113              if self.space == 'xyz-d50':
114                  lab = _xyz_to_lab(*coordinates, D50)
115                  return Color(space, lab, self.alpha)
116              elif self.space == 'xyz-d65':
117                  lab = _xyz_to_lab(*coordinates, D65)
118                  return Color(space, lab, self.alpha)
119              elif self.space == 'lch':
120                  a = coordinates[1] * cos(radians(coordinates[2]))
121                  b = coordinates[1] * sin(radians(coordinates[2]))
122                  return Color(space, (coordinates[0], a, b), self.alpha)
123              elif self.space == 'oklab':
124                  xyz = _oklab_to_xyz(*coordinates)
125                  lab = _xyz_to_lab(*xyz, D65)
126                  return Color(space, lab, self.alpha)
127              elif self.space == 'oklch':
128                  a = coordinates[1] * cos(radians(coordinates[2]))
129                  b = coordinates[1] * sin(radians(coordinates[2]))
130                  xyz = _oklab_to_xyz(coordinates[0], a, b)
131                  lab = _xyz_to_lab(*xyz, D65)
132                  return Color(space, lab, self.alpha)
133          raise NotImplementedError

136  def parse_color(input):
137      """Parse a color value as defined in CSS Color Level 4. 
138   
139      https://www.w3.org/TR/css-color-4/ 
140   
141      :type input: :obj:`str` or :term:`iterable` 
142      :param input: A string or an iterable of :term:`component values`. 
143      :returns: 
144          * :obj:`None` if the input is not a valid color value. 
145            (No exception is raised.) 
146          * The string ``'currentcolor'`` for the ``currentcolor`` keyword 
147          * A :class:`Color` object for every other values, including keywords. 
148   
149      """
150      if isinstance(input, str):
151          token = parse_one_component_value(input, skip_comments=True)
152      else:
153          token = input
154      if token.type == 'ident':
155          if token.lower_value == 'currentcolor':
156              return 'currentcolor'
157          elif token.lower_value == 'transparent':
158              return Color('srgb', (0, 0, 0), 0)
159          elif color := _COLOR_KEYWORDS.get(token.lower_value):
160              rgb = tuple(channel / 255 for channel in color)
161              return Color('srgb', rgb, 1)
162      elif token.type == 'hash':
163          for multiplier, regexp in _HASH_REGEXPS:
164              match = regexp(token.value)
165              if match:
166                  channels = [
167                      int(group * multiplier, 16) / 255
168                      for group in match.groups()]
169                  alpha = channels.pop() if len(channels) == 4 else 1
170                  return Color('srgb', channels, alpha)
171      elif token.type == 'function':
172          tokens = [
173              token for token in token.arguments
174              if token.type not in ('whitespace', 'comment')]
175          name = token.lower_name
176          if name == 'color':
177              space, *tokens = tokens
178          length = len(tokens)
179          if length in (5, 7) and all(token == ',' for token in tokens[1::2]):
180              old_syntax = True
181              tokens = tokens[::2]
182          elif length == 3:
183              old_syntax = False
184          elif length == 5 and tokens[3] == '/':
185              tokens.pop(3)
186              old_syntax = False
187          else:
188              return
189          args, alpha = tokens[:3], _parse_alpha(tokens[3:])
190          if alpha is None:
191              return
192          if name in ('rgb', 'rgba'):
193              return _parse_rgb(args, alpha)
194          elif name in ('hsl', 'hsla'):
195              return _parse_hsl(args, alpha)
196          elif name == 'hwb':
197              return _parse_hwb(args, alpha)
198          elif name == 'lab' and not old_syntax:
199              return _parse_lab(args, alpha)
200          elif name == 'lch' and not old_syntax:
201              return _parse_lch(args, alpha)
202          elif name == 'oklab' and not old_syntax:
203              return _parse_oklab(args, alpha)
204          elif name == 'oklch' and not old_syntax:
205              return _parse_oklch(args, alpha)
206          elif name == 'color' and not old_syntax:
207              return _parse_color(space, args, alpha)

210  def _parse_alpha(args):
211      """Parse a list of one alpha value. 
212   
213      If args is a list of a single INTEGER, NUMBER or PERCENTAGE token, 
214      return its value clipped to the 0..1 range. Otherwise, return None. 
215   
216      """
217      if len(args) == 0:
218          return 1.
219      elif len(args) == 1:
220          if args[0].type == 'number':
221              return min(1, max(0, args[0].value))
222          elif args[0].type == 'percentage':
223              return min(1, max(0, args[0].value / 100))

226  def _parse_rgb(args, alpha):
227      """Parse a list of RGB channels. 
228   
229      If args is a list of 3 NUMBER tokens or 3 PERCENTAGE tokens, return 
230      sRGB :class:`Color`. Otherwise, return None. 
231   
232      Input R, G, B ranges are [0, 255], output are [0, 1]. 
233   
234      """
235      if _types(args) not in (set(), {'number'}, {'percentage'}):
236          return
237      coordinates = [
238          arg.value / 255 if arg.type == 'number' else
239          arg.value / 100 if arg.type == 'percentage' else None
240          for arg in args]
241      return Color('srgb', coordinates, alpha)

244  def _parse_hsl(args, alpha):
245      """Parse a list of HSL channels. 
246   
247      If args is a list of 1 NUMBER or ANGLE token and 2 PERCENTAGE tokens, 
248      return HSL :class:`Color`. Otherwise, return None. 
249   
250      H range is [0, 360). S, L ranges are [0, 100]. 
251   
252      """
253      if _types(args[1:]) not in (set(), {'number'}, {'percentage'}):
254          return
255      if (hue := _parse_hue(args[0])) is None:
256          return
257      coordinates = [
258          None if args[0].type == 'ident' else hue,
259          None if args[1].type == 'ident' else args[1].value,
260          None if args[2].type == 'ident' else args[2].value,
261      ]
262      return Color('hsl', coordinates, alpha)

287      """Parse a list of CIE Lab channels. 
288   
289      If args is a list of 3 NUMBER or PERCENTAGE tokens, return Lab 
290      :class:`Color`. Otherwise, return None.

297      coordinates = [
298          None if args[0].type == 'ident' else args[0].value,

300              args[1].value * (1 if args[1].type == 'number' else 1.25)),
301          None if args[2].type == 'ident' else (
302              args[2].value * (1 if args[2].type == 'number' else 1.25)),
303      ]
304      return Color('lab', coordinates, alpha)

374  def _parse_color(space, args, alpha):
375      """Parse a color space name list of coordinates. 
376   
377      Ranges are [0, 1]. 
378   
379      """
380      if not _types(args) <= {'number', 'percentage'}:
381          return
382      if space.type != 'ident' or (space := space.lower_value) not in _FUNCTION_SPACES:
383          return
384      if space == 'xyz':
385          space = 'xyz-d65'
386      coordinates = [
387          arg.value if arg.type == 'number' else
388          arg.value / 100 if arg.type == 'percentage' else None
389          for arg in args]
390      return Color(space, coordinates, alpha)

393  def _parse_hue(token):
394      """Parse hue token. 
395   
396      Range is [0, 360). ``none`` value is 0. 
397   
398      """
399      if token.type == 'number':
400          return token.value % 360
401      elif token.type == 'dimension':
402          if token.unit == 'deg':
403              return token.value % 360
404          elif token.unit == 'grad':
405              return token.value / 400 * 360 % 360
406          elif token.unit == 'rad':
407              return degrees(token.value) % 360
408          elif token.unit == 'turn':
409              return token.value * 360 % 360
410      elif token.type == 'ident' and token.lower_value == 'none':
411          return 0

414  def _types(tokens):
415      """Get a set of token types, ignoring ``none`` values."""
416      types = set()
417      for token in tokens:
418          if token.type == 'ident' and token.lower_value == 'none':
419              continue
420          types.add(token.type)
421      return types

425  _κ = 24389 / 27
426  _ε = 216 / 24389
427  _LMS_TO_XYZ = (
428      (1.2268798733741557, -0.5578149965554813, 0.28139105017721583),
429      (-0.04057576262431372, 1.1122868293970594, -0.07171106666151701),

438  def _xyz_to_lab(X, Y, Z, d):
439      x = X / d[0]
440      y = Y / d[1]
441      z = Z / d[2]
442      f0 = x ** (1 / 3) if x > _ε else (_κ * x + 16) / 116
443      f1 = y ** (1 / 3) if y > _ε else (_κ * y + 16) / 116
444      f2 = z ** (1 / 3) if z > _ε else (_κ * z + 16) / 116
445      L = (116 * f1) - 16
446      a = 500 * (f0 - f1)
447      b = 200 * (f1 - f2)
448      return L, a, b
449

451  def _lab_to_xyz(L, a, b, d):
452      f1 = (L + 16) / 116
453      f0 = a / 500 + f1
454      f2 = f1 - b / 200
455      x = (f0 ** 3 if f0 ** 3 > _ε else (116 * f0 - 16) / _κ)
456      y = (((L + 16) / 116) ** 3 if L > _κ * _ε else L / _κ)
457      z = (f2 ** 3 if f2 ** 3 > _ε else (116 * f2 - 16) / _κ)
458      X = x * d[0]
459      Y = y * d[1]
460      Z = z * d[2]
461      return X, Y, Z

464  def _oklab_to_xyz(L, a, b):
465      lab = (L, a, b)
466      lms = [sum(_OKLAB_TO_LMS[i][j] * lab[j] for j in range(3)) for i in range(3)]
467      X, Y, Z = [sum(_LMS_TO_XYZ[i][j] * lms[j]**3 for j in range(3)) for i in range(3)]
468      return X, Y, Z
