26   class TransformedTargetRegressor(RegressorMixin, BaseEstimator):
27       """Meta-estimator to regress on a transformed target. 
28    
29       Useful for applying a non-linear transformation to the target `y` in 
30       regression problems. This transformation can be given as a Transformer 
31       such as the :class:`~sklearn.preprocessing.QuantileTransformer` or as a 
32       function and its inverse such as `np.log` and `np.exp`. 
33    
34       The computation during :meth:`fit` is:: 
35    
36           regressor.fit(X, func(y)) 
37    
38       or:: 
39    
40           regressor.fit(X, transformer.transform(y)) 
41    
42       The computation during :meth:`predict` is:: 
43    
44           inverse_func(regressor.predict(X)) 
45    
46       or:: 
47    
48           transformer.inverse_transform(regressor.predict(X)) 
49    
50       Read more in the :ref:`User Guide <transformed_target_regressor>`. 
51    
52       .. versionadded:: 0.20 
53    
54       Parameters 
55       ---------- 
56       regressor : object, default=None 
57           Regressor object such as derived from 
58           :class:`~sklearn.base.RegressorMixin`. This regressor will 
59           automatically be cloned each time prior to fitting. If `regressor is 
60           None`, :class:`~sklearn.linear_model.LinearRegression` is created and used. 
61    
62       transformer : object, default=None 
63           Estimator object such as derived from 
64           :class:`~sklearn.base.TransformerMixin`. Cannot be set at the same time 
65           as `func` and `inverse_func`. If `transformer is None` as well as 
66           `func` and `inverse_func`, the transformer will be an identity 
67           transformer. Note that the transformer will be cloned during fitting. 
68           Also, the transformer is restricting `y` to be a numpy array. 
69    
70       func : function, default=None 
71           Function to apply to `y` before passing to :meth:`fit`. Cannot be set 
72           at the same time as `transformer`. If `func is None`, the function used will be 
73           the identity function. If `func` is set, `inverse_func` also needs to be 
74           provided. The function needs to return a 2-dimensional array. 
75    
76       inverse_func : function, default=None 
77           Function to apply to the prediction of the regressor. Cannot be set at 
78           the same time as `transformer`. The inverse function is used to return 
79           predictions to the same space of the original training labels. If 
80           `inverse_func` is set, `func` also needs to be provided. The inverse 
81           function needs to return a 2-dimensional array. 
82    
83       check_inverse : bool, default=True 
84           Whether to check that `transform` followed by `inverse_transform` 
85           or `func` followed by `inverse_func` leads to the original targets. 
86    
87       Attributes 
88       ---------- 
89       regressor_ : object 
90           Fitted regressor. 
91    
92       transformer_ : object 
93           Transformer used in :meth:`fit` and :meth:`predict`. 
94    
95       n_features_in_ : int 
96           Number of features seen during :term:`fit`. Only defined if the 
97           underlying regressor exposes such an attribute when fit. 
98    
99           .. versionadded:: 0.24 
100   
101      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
102          Names of features seen during :term:`fit`. Defined only when `X` 
103          has feature names that are all strings. 
104   
105          .. versionadded:: 1.0 
106   
107      See Also 
108      -------- 
109      sklearn.preprocessing.FunctionTransformer : Construct a transformer from an 
110          arbitrary callable. 
111   
112      Notes 
113      ----- 
114      Internally, the target `y` is always converted into a 2-dimensional array 
115      to be used by scikit-learn transformers. At the time of prediction, the 
116      output will be reshaped to a have the same number of dimensions as `y`. 
117   
118      Examples 
119      -------- 
120      >>> import numpy as np 
121      >>> from sklearn.linear_model import LinearRegression 
122      >>> from sklearn.compose import TransformedTargetRegressor 
123      >>> tt = TransformedTargetRegressor(regressor=LinearRegression(), 
124      ...                                 func=np.log, inverse_func=np.exp) 
125      >>> X = np.arange(4).reshape(-1, 1) 
126      >>> y = np.exp(2 * X).ravel() 
127      >>> tt.fit(X, y) 
128      TransformedTargetRegressor(...) 
129      >>> tt.score(X, y) 
130      1.0 
131      >>> tt.regressor_.coef_ 
132      array([2.]) 
133   
134      For a more detailed example use case refer to 
135      :ref:`sphx_glr_auto_examples_compose_plot_transformed_target.py`. 
136      """

138      _parameter_constraints: dict = {
139          "regressor": [HasMethods(["fit", "predict"]), None],
140          "transformer": [HasMethods("transform"), None],
141          "func": [callable, None],
142          "inverse_func": [callable, None],
143          "check_inverse": ["boolean"],
144      }

146      def __init__(
147          self,
148          regressor=None,
149          *,
150          transformer=None,
151          func=None,
152          inverse_func=None,
153          check_inverse=True,
154      ):
155          self.regressor = regressor
156          self.transformer = transformer
157          self.func = func
158          self.inverse_func = inverse_func
159          self.check_inverse = check_inverse
160  
161      def _fit_transformer(self, y):
162          """Check transformer and fit transformer. 
163   
164          Create the default transformer, fit it and make additional inverse 
165          check on a subset (optional). 
166   
167          """
168          if self.transformer is not None and (
169              self.func is not None or self.inverse_func is not None
170          ):
171              raise ValueError(
172                  "'transformer' and functions 'func'/'inverse_func' cannot both be set."
173              )
174          elif self.transformer is not None:
175              self.transformer_ = clone(self.transformer)
176          else:
177              if (self.func is not None and self.inverse_func is None) or (
178                  self.func is None and self.inverse_func is not None
179              ):
180                  lacking_param, existing_param = (
181                      ("func", "inverse_func")
182                      if self.func is None
183                      else ("inverse_func", "func")
184                  )
185                  raise ValueError(
186                      f"When '{existing_param}' is provided, '{lacking_param}' must also"
187                      f" be provided. If {lacking_param} is supposed to be the default,"
188                      " you need to explicitly pass it the identity function."
189                  )
190              self.transformer_ = FunctionTransformer(
191                  func=self.func,
192                  inverse_func=self.inverse_func,
193                  validate=True,
194                  check_inverse=self.check_inverse,
195              )
196              # We are transforming the target here and not the features, so we set the
197              # output of FunctionTransformer() to be a numpy array (default) and to not
198              # depend on the global configuration:
199              self.transformer_.set_output(transform="default")
200          # XXX: sample_weight is not currently passed to the
201          # transformer. However, if transformer starts using sample_weight, the
202          # code should be modified accordingly. At the time to consider the
203          # sample_prop feature, it is also a good use case to be considered.
204          self.transformer_.fit(y)
205          if self.check_inverse:
206              idx_selected = slice(None, None, max(1, y.shape[0] // 10))
207              y_sel = _safe_indexing(y, idx_selected)
208              y_sel_t = self.transformer_.transform(y_sel)
209              if not np.allclose(y_sel, self.transformer_.inverse_transform(y_sel_t)):
210                  warnings.warn(
211                      (
212                          "The provided functions or transformer are"
213                          " not strictly inverse of each other. If"
214                          " you are sure you want to proceed regardless"
215                          ", set 'check_inverse=False'"
216                      ),
217                      UserWarning,
218                  )
219  
220      @_fit_context(
221          # TransformedTargetRegressor.regressor/transformer are not validated yet.
222          prefer_skip_nested_validation=False
223      )
224      def fit(self, X, y, **fit_params):
225          """Fit the model according to the given training data. 
226   
227          Parameters 
228          ---------- 
229          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
230              Training vector, where `n_samples` is the number of samples and 
231              `n_features` is the number of features. 
232   
233          y : array-like of shape (n_samples,) 
234              Target values. 
235   
236          **fit_params : dict 
237              - If `enable_metadata_routing=False` (default): Parameters directly passed 
238                to the `fit` method of the underlying regressor. 
239   
240              - If `enable_metadata_routing=True`: Parameters safely routed to the `fit` 
241                method of the underlying regressor. 
242   
243              .. versionchanged:: 1.6 
244                  See :ref:`Metadata Routing User Guide <metadata_routing>` for 
245                  more details. 
246   
247          Returns 
248          ------- 
249          self : object 
250              Fitted estimator. 
251          """
252          if y is None:
253              raise ValueError(
254                  f"This {self.__class__.__name__} estimator "
255                  "requires y to be passed, but the target y is None."
256              )
257          y = check_array(
258              y,
259              input_name="y",
260              accept_sparse=False,
261              ensure_all_finite=True,
262              ensure_2d=False,
263              dtype="numeric",
264              allow_nd=True,
265          )
266  
267          # store the number of dimension of the target to predict an array of
268          # similar shape at predict
269          self._training_dim = y.ndim
270  
271          # transformers are designed to modify X which is 2d dimensional, we
272          # need to modify y accordingly.
273          if y.ndim == 1:
274              y_2d = y.reshape(-1, 1)
275          else:
276              y_2d = y
277          self._fit_transformer(y_2d)
278  
279          # transform y and convert back to 1d array if needed
280          y_trans = self.transformer_.transform(y_2d)
281          # FIXME: a FunctionTransformer can return a 1D array even when validate
282          # is set to True. Therefore, we need to check the number of dimension
283          # first.
284          if y_trans.ndim == 2 and y_trans.shape[1] == 1:
285              y_trans = y_trans.squeeze(axis=1)
286  
287          self.regressor_ = self._get_regressor(get_clone=True)
288          if _routing_enabled():
289              routed_params = process_routing(self, "fit", **fit_params)
290          else:
291              routed_params = Bunch(regressor=Bunch(fit=fit_params))
292  
293          self.regressor_.fit(X, y_trans, **routed_params.regressor.fit)
294  
295          if hasattr(self.regressor_, "feature_names_in_"):
296              self.feature_names_in_ = self.regressor_.feature_names_in_
297  
298          return self
299  
300      def predict(self, X, **predict_params):
301          """Predict using the base regressor, applying inverse. 
302   
303          The regressor is used to predict and the `inverse_func` or 
304          `inverse_transform` is applied before returning the prediction. 
305   
306          Parameters 
307          ---------- 
308          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
309              Samples. 
310   
311          **predict_params : dict of str -> object 
312              - If `enable_metadata_routing=False` (default): Parameters directly passed 
313                to the `predict` method of the underlying regressor. 
314   
315              - If `enable_metadata_routing=True`: Parameters safely routed to the 
316                `predict` method of the underlying regressor. 
317   
318              .. versionchanged:: 1.6 
319                  See :ref:`Metadata Routing User Guide <metadata_routing>` 
320                  for more details. 
321   
322          Returns 
323          ------- 
324          y_hat : ndarray of shape (n_samples,) 
325              Predicted values. 
326          """
327          check_is_fitted(self)
328          if _routing_enabled():
329              routed_params = process_routing(self, "predict", **predict_params)
330          else:
331              routed_params = Bunch(regressor=Bunch(predict=predict_params))
332  
333          pred = self.regressor_.predict(X, **routed_params.regressor.predict)
334          if pred.ndim == 1:
335              pred_trans = self.transformer_.inverse_transform(pred.reshape(-1, 1))
336          else:
337              pred_trans = self.transformer_.inverse_transform(pred)
338          if (
339              self._training_dim == 1
340              and pred_trans.ndim == 2
341              and pred_trans.shape[1] == 1
342          ):
343              pred_trans = pred_trans.squeeze(axis=1)
344  
345          return pred_trans

347      def __sklearn_tags__(self):
348          regressor = self._get_regressor()
349          tags = super().__sklearn_tags__()
350          tags.regressor_tags.poor_score = True
351          tags.input_tags.sparse = get_tags(regressor).input_tags.sparse
352          tags.target_tags.multi_output = get_tags(regressor).target_tags.multi_output
353          return tags

355      @property
356      def n_features_in_(self):
357          """Number of features seen during :term:`fit`."""
358          # For consistency with other estimators we raise a AttributeError so
359          # that hasattr() returns False the estimator isn't fitted.
360          try:
361              check_is_fitted(self)
362          except NotFittedError as nfe:
363              raise AttributeError(
364                  "{} object has no n_features_in_ attribute.".format(
365                      self.__class__.__name__
366                  )
367              ) from nfe
368  
369          return self.regressor_.n_features_in_

371      def get_metadata_routing(self):
372          """Get metadata routing of this object. 
373   
374          Please check :ref:`User Guide <metadata_routing>` on how the routing 
375          mechanism works. 
376   
377          .. versionadded:: 1.6 
378   
379          Returns 
380          ------- 
381          routing : MetadataRouter 
382              A :class:`~sklearn.utils.metadata_routing.MetadataRouter` encapsulating 
383              routing information. 
384          """
385          router = MetadataRouter(owner=self.__class__.__name__).add(
386              regressor=self._get_regressor(),
387              method_mapping=MethodMapping()
388              .add(caller="fit", callee="fit")
389              .add(caller="predict", callee="predict"),
390          )
391          return router
392  
393      def _get_regressor(self, get_clone=False):
394          if self.regressor is None:
395              return LinearRegression()
396  
397          return clone(self.regressor) if get_clone else self.regressor
