48   def _center_is_true(structure, origin):
49       structure = np.asarray(structure)
50       coor = tuple([oo + ss // 2 for ss, oo in zip(structure.shape,
51                                                    origin)])
52       return bool(structure[coor])

55   def iterate_structure(structure, iterations, origin=None):
56       """ 
57       Iterate a structure by dilating it with itself. 
58    
59       Parameters 
60       ---------- 
61       structure : array_like 
62          Structuring element (an array of bools, for example), to be dilated with 
63          itself. 
64       iterations : int 
65          number of dilations performed on the structure with itself 
66       origin : optional 
67           If origin is None, only the iterated structure is returned. If 
68           not, a tuple of the iterated structure and the modified origin is 
69           returned. 
70    
71       Returns 
72       ------- 
73       iterate_structure : ndarray of bools 
74           A new structuring element obtained by dilating `structure` 
75           (`iterations` - 1) times with itself. 
76    
77       See Also 
78       -------- 
79       generate_binary_structure 
80    
81       Examples 
82       -------- 
83       >>> from scipy import ndimage 
84       >>> struct = ndimage.generate_binary_structure(2, 1) 
85       >>> struct.astype(int) 
86       array([[0, 1, 0], 
87              [1, 1, 1], 
88              [0, 1, 0]]) 
89       >>> ndimage.iterate_structure(struct, 2).astype(int) 
90       array([[0, 0, 1, 0, 0], 
91              [0, 1, 1, 1, 0], 
92              [1, 1, 1, 1, 1], 
93              [0, 1, 1, 1, 0], 
94              [0, 0, 1, 0, 0]]) 
95       >>> ndimage.iterate_structure(struct, 3).astype(int) 
96       array([[0, 0, 0, 1, 0, 0, 0], 
97              [0, 0, 1, 1, 1, 0, 0], 
98              [0, 1, 1, 1, 1, 1, 0], 
99              [1, 1, 1, 1, 1, 1, 1], 
100             [0, 1, 1, 1, 1, 1, 0], 
101             [0, 0, 1, 1, 1, 0, 0], 
102             [0, 0, 0, 1, 0, 0, 0]]) 
103   
104      """
105      structure = np.asarray(structure)
106      if iterations < 2:
107          return structure.copy()
108      ni = iterations - 1
109      shape = [ii + ni * (ii - 1) for ii in structure.shape]
110      pos = [ni * (structure.shape[ii] // 2) for ii in range(len(shape))]
111      slc = tuple(slice(pos[ii], pos[ii] + structure.shape[ii], None)
112                  for ii in range(len(shape)))
113      out = np.zeros(shape, bool)
114      out[slc] = structure != 0
115      out = binary_dilation(out, structure, iterations=ni)
116      if origin is None:
117          return out
118      else:
119          origin = _ni_support._normalize_sequence(origin, structure.ndim)
120          origin = [iterations * o for o in origin]
121          return out, origin

124  def generate_binary_structure(rank, connectivity):
125      """ 
126      Generate a binary structure for binary morphological operations. 
127   
128      Parameters 
129      ---------- 
130      rank : int 
131           Number of dimensions of the array to which the structuring element 
132           will be applied, as returned by `np.ndim`. 
133      connectivity : int 
134           `connectivity` determines which elements of the output array belong 
135           to the structure, i.e., are considered as neighbors of the central 
136           element. Elements up to a squared distance of `connectivity` from 
137           the center are considered neighbors. `connectivity` may range from 1 
138           (no diagonal elements are neighbors) to `rank` (all elements are 
139           neighbors). 
140   
141      Returns 
142      ------- 
143      output : ndarray of bools 
144           Structuring element which may be used for binary morphological 
145           operations, with `rank` dimensions and all dimensions equal to 3. 
146   
147      See Also 
148      -------- 
149      iterate_structure, binary_dilation, binary_erosion 
150   
151      Notes 
152      ----- 
153      `generate_binary_structure` can only create structuring elements with 
154      dimensions equal to 3, i.e., minimal dimensions. For larger structuring 
155      elements, that are useful e.g., for eroding large objects, one may either 
156      use `iterate_structure`, or create directly custom arrays with 
157      numpy functions such as `numpy.ones`. 
158   
159      Examples 
160      -------- 
161      >>> from scipy import ndimage 
162      >>> import numpy as np 
163      >>> struct = ndimage.generate_binary_structure(2, 1) 
164      >>> struct 
165      array([[False,  True, False], 
166             [ True,  True,  True], 
167             [False,  True, False]], dtype=bool) 
168      >>> a = np.zeros((5,5)) 
169      >>> a[2, 2] = 1 
170      >>> a 
171      array([[ 0.,  0.,  0.,  0.,  0.], 
172             [ 0.,  0.,  0.,  0.,  0.], 
173             [ 0.,  0.,  1.,  0.,  0.], 
174             [ 0.,  0.,  0.,  0.,  0.], 
175             [ 0.,  0.,  0.,  0.,  0.]]) 
176      >>> b = ndimage.binary_dilation(a, structure=struct).astype(a.dtype) 
177      >>> b 
178      array([[ 0.,  0.,  0.,  0.,  0.], 
179             [ 0.,  0.,  1.,  0.,  0.], 
180             [ 0.,  1.,  1.,  1.,  0.], 
181             [ 0.,  0.,  1.,  0.,  0.], 
182             [ 0.,  0.,  0.,  0.,  0.]]) 
183      >>> ndimage.binary_dilation(b, structure=struct).astype(a.dtype) 
184      array([[ 0.,  0.,  1.,  0.,  0.], 
185             [ 0.,  1.,  1.,  1.,  0.], 
186             [ 1.,  1.,  1.,  1.,  1.], 
187             [ 0.,  1.,  1.,  1.,  0.], 
188             [ 0.,  0.,  1.,  0.,  0.]]) 
189      >>> struct = ndimage.generate_binary_structure(2, 2) 
190      >>> struct 
191      array([[ True,  True,  True], 
192             [ True,  True,  True], 
193             [ True,  True,  True]], dtype=bool) 
194      >>> struct = ndimage.generate_binary_structure(3, 1) 
195      >>> struct # no diagonal elements 
196      array([[[False, False, False], 
197              [False,  True, False], 
198              [False, False, False]], 
199             [[False,  True, False], 
200              [ True,  True,  True], 
201              [False,  True, False]], 
202             [[False, False, False], 
203              [False,  True, False], 
204              [False, False, False]]], dtype=bool) 
205   
206      """
207      if connectivity < 1:
208          connectivity = 1
209      if rank < 1:
210          return np.array(True, dtype=bool)
211      output = np.fabs(np.indices([3] * rank) - 1)
212      output = np.add.reduce(output, 0)
213      return output <= connectivity

216  def _binary_erosion(input, structure, iterations, mask, output,
217                      border_value, origin, invert, brute_force, axes):
218      try:
219          iterations = operator.index(iterations)
220      except TypeError as e:
221          raise TypeError('iterations parameter should be an integer') from e
222  
223      input = np.asarray(input)
224      ndim = input.ndim
225      if np.iscomplexobj(input):
226          raise TypeError('Complex type not supported')
227      axes = _ni_support._check_axes(axes, input.ndim)
228      num_axes = len(axes)
229      if structure is None:
230          structure = generate_binary_structure(num_axes, 1)
231      else:
232          structure = np.asarray(structure, dtype=bool)
233      if ndim > num_axes:
234          structure = _filters._expand_footprint(ndim, axes, structure,
235                                                 footprint_name="structure")
236  
237      if structure.ndim != input.ndim:
238          raise RuntimeError('structure and input must have same dimensionality')
239      if not structure.flags.contiguous:
240          structure = structure.copy()
241      if structure.size < 1:
242          raise RuntimeError('structure must not be empty')
243      if mask is not None:
244          mask = np.asarray(mask)
245          if mask.shape != input.shape:
246              raise RuntimeError('mask and input must have equal sizes')
247      origin = _ni_support._normalize_sequence(origin, num_axes)
248      origin = _filters._expand_origin(ndim, axes, origin)
249      cit = _center_is_true(structure, origin)
250      if isinstance(output, np.ndarray):
251          if np.iscomplexobj(output):
252              raise TypeError('Complex output type not supported')
253      else:
254          output = bool
255      output = _ni_support._get_output(output, input)
256      temp_needed = np.may_share_memory(input, output)
257      if temp_needed:
258          # input and output arrays cannot share memory
259          temp = output
260          output = _ni_support._get_output(output.dtype, input)
261      if iterations == 1:
262          _nd_image.binary_erosion(input, structure, mask, output,
263                                   border_value, origin, invert, cit, 0)
264      elif cit and not brute_force:
265          changed, coordinate_list = _nd_image.binary_erosion(
266              input, structure, mask, output,
267              border_value, origin, invert, cit, 1)
268          structure = structure[tuple([slice(None, None, -1)] *
269                                      structure.ndim)]
270          for ii in range(len(origin)):
271              origin[ii] = -origin[ii]
272              if not structure.shape[ii] & 1:
273                  origin[ii] -= 1
274          if mask is not None:
275              mask = np.asarray(mask, dtype=np.int8)
276          if not structure.flags.contiguous:
277              structure = structure.copy()
278          _nd_image.binary_erosion2(output, structure, mask, iterations - 1,
279                                    origin, invert, coordinate_list)
280      else:
281          tmp_in = np.empty_like(input, dtype=bool)
282          tmp_out = output
283          if iterations >= 1 and not iterations & 1:
284              tmp_in, tmp_out = tmp_out, tmp_in
285          changed = _nd_image.binary_erosion(
286              input, structure, mask, tmp_out,
287              border_value, origin, invert, cit, 0)
288          ii = 1
289          while ii < iterations or (iterations < 1 and changed):
290              tmp_in, tmp_out = tmp_out, tmp_in
291              changed = _nd_image.binary_erosion(
292                  tmp_in, structure, mask, tmp_out,
293                  border_value, origin, invert, cit, 0)
294              ii += 1
295      if temp_needed:
296          temp[...] = output
297          output = temp
298      return output

301  def binary_erosion(input, structure=None, iterations=1, mask=None, output=None,
302                     border_value=0, origin=0, brute_force=False, *, axes=None):
303      """ 
304      Multidimensional binary erosion with a given structuring element. 
305   
306      Binary erosion is a mathematical morphology operation used for image 
307      processing. 
308   
309      Parameters 
310      ---------- 
311      input : array_like 
312          Binary image to be eroded. Non-zero (True) elements form 
313          the subset to be eroded. 
314      structure : array_like, optional 
315          Structuring element used for the erosion. Non-zero elements are 
316          considered True. If no structuring element is provided, an element 
317          is generated with a square connectivity equal to one. 
318      iterations : int, optional 
319          The erosion is repeated `iterations` times (one, by default). 
320          If iterations is less than 1, the erosion is repeated until the 
321          result does not change anymore. 
322      mask : array_like, optional 
323          If a mask is given, only those elements with a True value at 
324          the corresponding mask element are modified at each iteration. 
325      output : ndarray, optional 
326          Array of the same shape as input, into which the output is placed. 
327          By default, a new array is created. 
328      border_value : int (cast to 0 or 1), optional 
329          Value at the border in the output array. 
330      origin : int or tuple of ints, optional 
331          Placement of the filter, by default 0. 
332      brute_force : boolean, optional 
333          Memory condition: if False, only the pixels whose value was changed in 
334          the last iteration are tracked as candidates to be updated (eroded) in 
335          the current iteration; if True all pixels are considered as candidates 
336          for erosion, regardless of what happened in the previous iteration. 
337          False by default. 
338      axes : tuple of int or None 
339          The axes over which to apply the filter. If None, `input` is filtered 
340          along all axes. If an `origin` tuple is provided, its length must match 
341          the number of axes. 
342   
343      Returns 
344      ------- 
345      binary_erosion : ndarray of bools 
346          Erosion of the input by the structuring element. 
347   
348      See Also 
349      -------- 
350      grey_erosion, binary_dilation, binary_closing, binary_opening, 
351      generate_binary_structure 
352   
353      Notes 
354      ----- 
355      Erosion [1]_ is a mathematical morphology operation [2]_ that uses a 
356      structuring element for shrinking the shapes in an image. The binary 
357      erosion of an image by a structuring element is the locus of the points 
358      where a superimposition of the structuring element centered on the point 
359      is entirely contained in the set of non-zero elements of the image. 
360   
361      References 
362      ---------- 
363      .. [1] https://en.wikipedia.org/wiki/Erosion_%28morphology%29 
364      .. [2] https://en.wikipedia.org/wiki/Mathematical_morphology 
365   
366      Examples 
367      -------- 
368      >>> from scipy import ndimage 
369      >>> import numpy as np 
370      >>> a = np.zeros((7,7), dtype=int) 
371      >>> a[1:6, 2:5] = 1 
372      >>> a 
373      array([[0, 0, 0, 0, 0, 0, 0], 
374             [0, 0, 1, 1, 1, 0, 0], 
375             [0, 0, 1, 1, 1, 0, 0], 
376             [0, 0, 1, 1, 1, 0, 0], 
377             [0, 0, 1, 1, 1, 0, 0], 
378             [0, 0, 1, 1, 1, 0, 0], 
379             [0, 0, 0, 0, 0, 0, 0]]) 
380      >>> ndimage.binary_erosion(a).astype(a.dtype) 
381      array([[0, 0, 0, 0, 0, 0, 0], 
382             [0, 0, 0, 0, 0, 0, 0], 
383             [0, 0, 0, 1, 0, 0, 0], 
384             [0, 0, 0, 1, 0, 0, 0], 
385             [0, 0, 0, 1, 0, 0, 0], 
386             [0, 0, 0, 0, 0, 0, 0], 
387             [0, 0, 0, 0, 0, 0, 0]]) 
388      >>> #Erosion removes objects smaller than the structure 
389      >>> ndimage.binary_erosion(a, structure=np.ones((5,5))).astype(a.dtype) 
390      array([[0, 0, 0, 0, 0, 0, 0], 
391             [0, 0, 0, 0, 0, 0, 0], 
392             [0, 0, 0, 0, 0, 0, 0], 
393             [0, 0, 0, 0, 0, 0, 0], 
394             [0, 0, 0, 0, 0, 0, 0], 
395             [0, 0, 0, 0, 0, 0, 0], 
396             [0, 0, 0, 0, 0, 0, 0]]) 
397   
398      """
399      return _binary_erosion(input, structure, iterations, mask,
400                             output, border_value, origin, 0, brute_force, axes)

403  def binary_dilation(input, structure=None, iterations=1, mask=None,
404                      output=None, border_value=0, origin=0,
405                      brute_force=False, *, axes=None):
406      """ 
407      Multidimensional binary dilation with the given structuring element. 
408   
409      Parameters 
410      ---------- 
411      input : array_like 
412          Binary array_like to be dilated. Non-zero (True) elements form 
413          the subset to be dilated. 
414      structure : array_like, optional 
415          Structuring element used for the dilation. Non-zero elements are 
416          considered True. If no structuring element is provided an element 
417          is generated with a square connectivity equal to one. 
418      iterations : int, optional 
419          The dilation is repeated `iterations` times (one, by default). 
420          If iterations is less than 1, the dilation is repeated until the 
421          result does not change anymore. Only an integer of iterations is 
422          accepted. 
423      mask : array_like, optional 
424          If a mask is given, only those elements with a True value at 
425          the corresponding mask element are modified at each iteration. 
426      output : ndarray, optional 
427          Array of the same shape as input, into which the output is placed. 
428          By default, a new array is created. 
429      border_value : int (cast to 0 or 1), optional 
430          Value at the border in the output array. 
431      origin : int or tuple of ints, optional 
432          Placement of the filter, by default 0. 
433      brute_force : boolean, optional 
434          Memory condition: if False, only the pixels whose value was changed in 
435          the last iteration are tracked as candidates to be updated (dilated) 
436          in the current iteration; if True all pixels are considered as 
437          candidates for dilation, regardless of what happened in the previous 
438          iteration. False by default. 
439      axes : tuple of int or None 
440          The axes over which to apply the filter. If None, `input` is filtered 
441          along all axes. If an `origin` tuple is provided, its length must match 
442          the number of axes. 
443   
444      Returns 
445      ------- 
446      binary_dilation : ndarray of bools 
447          Dilation of the input by the structuring element. 
448   
449      See Also 
450      -------- 
451      grey_dilation, binary_erosion, binary_closing, binary_opening, 
452      generate_binary_structure 
453   
454      Notes 
455      ----- 
456      Dilation [1]_ is a mathematical morphology operation [2]_ that uses a 
457      structuring element for expanding the shapes in an image. The binary 
458      dilation of an image by a structuring element is the locus of the points 
459      covered by the structuring element, when its center lies within the 
460      non-zero points of the image. 
461   
462      References 
463      ---------- 
464      .. [1] https://en.wikipedia.org/wiki/Dilation_%28morphology%29 
465      .. [2] https://en.wikipedia.org/wiki/Mathematical_morphology 
466   
467      Examples 
468      -------- 
469      >>> from scipy import ndimage 
470      >>> import numpy as np 
471      >>> a = np.zeros((5, 5)) 
472      >>> a[2, 2] = 1 
473      >>> a 
474      array([[ 0.,  0.,  0.,  0.,  0.], 
475             [ 0.,  0.,  0.,  0.,  0.], 
476             [ 0.,  0.,  1.,  0.,  0.], 
477             [ 0.,  0.,  0.,  0.,  0.], 
478             [ 0.,  0.,  0.,  0.,  0.]]) 
479      >>> ndimage.binary_dilation(a) 
480      array([[False, False, False, False, False], 
481             [False, False,  True, False, False], 
482             [False,  True,  True,  True, False], 
483             [False, False,  True, False, False], 
484             [False, False, False, False, False]], dtype=bool) 
485      >>> ndimage.binary_dilation(a).astype(a.dtype) 
486      array([[ 0.,  0.,  0.,  0.,  0.], 
487             [ 0.,  0.,  1.,  0.,  0.], 
488             [ 0.,  1.,  1.,  1.,  0.], 
489             [ 0.,  0.,  1.,  0.,  0.], 
490             [ 0.,  0.,  0.,  0.,  0.]]) 
491      >>> # 3x3 structuring element with connectivity 1, used by default 
492      >>> struct1 = ndimage.generate_binary_structure(2, 1) 
493      >>> struct1 
494      array([[False,  True, False], 
495             [ True,  True,  True], 
496             [False,  True, False]], dtype=bool) 
497      >>> # 3x3 structuring element with connectivity 2 
498      >>> struct2 = ndimage.generate_binary_structure(2, 2) 
499      >>> struct2 
500      array([[ True,  True,  True], 
501             [ True,  True,  True], 
502             [ True,  True,  True]], dtype=bool) 
503      >>> ndimage.binary_dilation(a, structure=struct1).astype(a.dtype) 
504      array([[ 0.,  0.,  0.,  0.,  0.], 
505             [ 0.,  0.,  1.,  0.,  0.], 
506             [ 0.,  1.,  1.,  1.,  0.], 
507             [ 0.,  0.,  1.,  0.,  0.], 
508             [ 0.,  0.,  0.,  0.,  0.]]) 
509      >>> ndimage.binary_dilation(a, structure=struct2).astype(a.dtype) 
510      array([[ 0.,  0.,  0.,  0.,  0.], 
511             [ 0.,  1.,  1.,  1.,  0.], 
512             [ 0.,  1.,  1.,  1.,  0.], 
513             [ 0.,  1.,  1.,  1.,  0.], 
514             [ 0.,  0.,  0.,  0.,  0.]]) 
515      >>> ndimage.binary_dilation(a, structure=struct1,\\ 
516      ... iterations=2).astype(a.dtype) 
517      array([[ 0.,  0.,  1.,  0.,  0.], 
518             [ 0.,  1.,  1.,  1.,  0.], 
519             [ 1.,  1.,  1.,  1.,  1.], 
520             [ 0.,  1.,  1.,  1.,  0.], 
521             [ 0.,  0.,  1.,  0.,  0.]]) 
522   
523      """
524      input = np.asarray(input)
525      axes = _ni_support._check_axes(axes, input.ndim)
526      num_axes = len(axes)
527      if structure is None:
528          structure = generate_binary_structure(num_axes, 1)
529      origin = _ni_support._normalize_sequence(origin, num_axes)
530      structure = np.asarray(structure)
531      structure = structure[tuple([slice(None, None, -1)] *
532                                  structure.ndim)]
533      for ii in range(len(origin)):
534          origin[ii] = -origin[ii]
535          if not structure.shape[ii] & 1:
536              origin[ii] -= 1
537  
538      return _binary_erosion(input, structure, iterations, mask,
539                             output, border_value, origin, 1, brute_force, axes)

542  def binary_opening(input, structure=None, iterations=1, output=None,
543                     origin=0, mask=None, border_value=0, brute_force=False, *,
544                     axes=None):
545      """ 
546      Multidimensional binary opening with the given structuring element. 
547   
548      The *opening* of an input image by a structuring element is the 
549      *dilation* of the *erosion* of the image by the structuring element. 
550   
551      Parameters 
552      ---------- 
553      input : array_like 
554          Binary array_like to be opened. Non-zero (True) elements form 
555          the subset to be opened. 
556      structure : array_like, optional 
557          Structuring element used for the opening. Non-zero elements are 
558          considered True. If no structuring element is provided an element 
559          is generated with a square connectivity equal to one (i.e., only 
560          nearest neighbors are connected to the center, diagonally-connected 
561          elements are not considered neighbors). 
562      iterations : int, optional 
563          The erosion step of the opening, then the dilation step are each 
564          repeated `iterations` times (one, by default). If `iterations` is 
565          less than 1, each operation is repeated until the result does 
566          not change anymore. Only an integer of iterations is accepted. 
567      output : ndarray, optional 
568          Array of the same shape as input, into which the output is placed. 
569          By default, a new array is created. 
570      origin : int or tuple of ints, optional 
571          Placement of the filter, by default 0. 
572      mask : array_like, optional 
573          If a mask is given, only those elements with a True value at 
574          the corresponding mask element are modified at each iteration. 
575   
576          .. versionadded:: 1.1.0 
577      border_value : int (cast to 0 or 1), optional 
578          Value at the border in the output array. 
579   
580          .. versionadded:: 1.1.0 
581      brute_force : boolean, optional 
582          Memory condition: if False, only the pixels whose value was changed in 
583          the last iteration are tracked as candidates to be updated in the 
584          current iteration; if true all pixels are considered as candidates for 
585          update, regardless of what happened in the previous iteration. 
586          False by default. 
587   
588          .. versionadded:: 1.1.0 
589      axes : tuple of int or None 
590          The axes over which to apply the filter. If None, `input` is filtered 
591          along all axes. If an `origin` tuple is provided, its length must match 
592          the number of axes. 
593   
594      Returns 
595      ------- 
596      binary_opening : ndarray of bools 
597          Opening of the input by the structuring element. 
598   
599      See Also 
600      -------- 
601      grey_opening, binary_closing, binary_erosion, binary_dilation, 
602      generate_binary_structure 
603   
604      Notes 
605      ----- 
606      *Opening* [1]_ is a mathematical morphology operation [2]_ that 
607      consists in the succession of an erosion and a dilation of the 
608      input with the same structuring element. Opening, therefore, removes 
609      objects smaller than the structuring element. 
610   
611      Together with *closing* (`binary_closing`), opening can be used for 
612      noise removal. 
613   
614      References 
615      ---------- 
616      .. [1] https://en.wikipedia.org/wiki/Opening_%28morphology%29 
617      .. [2] https://en.wikipedia.org/wiki/Mathematical_morphology 
618   
619      Examples 
620      -------- 
621      >>> from scipy import ndimage 
622      >>> import numpy as np 
623      >>> a = np.zeros((5,5), dtype=int) 
624      >>> a[1:4, 1:4] = 1; a[4, 4] = 1 
625      >>> a 
626      array([[0, 0, 0, 0, 0], 
627             [0, 1, 1, 1, 0], 
628             [0, 1, 1, 1, 0], 
629             [0, 1, 1, 1, 0], 
630             [0, 0, 0, 0, 1]]) 
631      >>> # Opening removes small objects 
632      >>> ndimage.binary_opening(a, structure=np.ones((3,3))).astype(int) 
633      array([[0, 0, 0, 0, 0], 
634             [0, 1, 1, 1, 0], 
635             [0, 1, 1, 1, 0], 
636             [0, 1, 1, 1, 0], 
637             [0, 0, 0, 0, 0]]) 
638      >>> # Opening can also smooth corners 
639      >>> ndimage.binary_opening(a).astype(int) 
640      array([[0, 0, 0, 0, 0], 
641             [0, 0, 1, 0, 0], 
642             [0, 1, 1, 1, 0], 
643             [0, 0, 1, 0, 0], 
644             [0, 0, 0, 0, 0]]) 
645      >>> # Opening is the dilation of the erosion of the input 
646      >>> ndimage.binary_erosion(a).astype(int) 
647      array([[0, 0, 0, 0, 0], 
648             [0, 0, 0, 0, 0], 
649             [0, 0, 1, 0, 0], 
650             [0, 0, 0, 0, 0], 
651             [0, 0, 0, 0, 0]]) 
652      >>> ndimage.binary_dilation(ndimage.binary_erosion(a)).astype(int) 
653      array([[0, 0, 0, 0, 0], 
654             [0, 0, 1, 0, 0], 
655             [0, 1, 1, 1, 0], 
656             [0, 0, 1, 0, 0], 
657             [0, 0, 0, 0, 0]]) 
658   
659      """
660      input = np.asarray(input)
661      axes = _ni_support._check_axes(axes, input.ndim)
662      num_axes = len(axes)
663      if structure is None:
664          structure = generate_binary_structure(num_axes, 1)
665  
666      tmp = binary_erosion(input, structure, iterations, mask, None,
667                           border_value, origin, brute_force, axes=axes)
668      return binary_dilation(tmp, structure, iterations, mask, output,
669                             border_value, origin, brute_force, axes=axes)

672  def binary_closing(input, structure=None, iterations=1, output=None,
673                     origin=0, mask=None, border_value=0, brute_force=False, *,
674                     axes=None):
675      """ 
676      Multidimensional binary closing with the given structuring element. 
677   
678      The *closing* of an input image by a structuring element is the 
679      *erosion* of the *dilation* of the image by the structuring element. 
680   
681      Parameters 
682      ---------- 
683      input : array_like 
684          Binary array_like to be closed. Non-zero (True) elements form 
685          the subset to be closed. 
686      structure : array_like, optional 
687          Structuring element used for the closing. Non-zero elements are 
688          considered True. If no structuring element is provided an element 
689          is generated with a square connectivity equal to one (i.e., only 
690          nearest neighbors are connected to the center, diagonally-connected 
691          elements are not considered neighbors). 
692      iterations : int, optional 
693          The dilation step of the closing, then the erosion step are each 
694          repeated `iterations` times (one, by default). If iterations is 
695          less than 1, each operations is repeated until the result does 
696          not change anymore. Only an integer of iterations is accepted. 
697      output : ndarray, optional 
698          Array of the same shape as input, into which the output is placed. 
699          By default, a new array is created. 
700      origin : int or tuple of ints, optional 
701          Placement of the filter, by default 0. 
702      mask : array_like, optional 
703          If a mask is given, only those elements with a True value at 
704          the corresponding mask element are modified at each iteration. 
705   
706          .. versionadded:: 1.1.0 
707      border_value : int (cast to 0 or 1), optional 
708          Value at the border in the output array. 
709   
710          .. versionadded:: 1.1.0 
711      brute_force : boolean, optional 
712          Memory condition: if False, only the pixels whose value was changed in 
713          the last iteration are tracked as candidates to be updated in the 
714          current iteration; if true al pixels are considered as candidates for 
715          update, regardless of what happened in the previous iteration. 
716          False by default. 
717   
718          .. versionadded:: 1.1.0 
719      axes : tuple of int or None 
720          The axes over which to apply the filter. If None, `input` is filtered 
721          along all axes. If an `origin` tuple is provided, its length must match 
722          the number of axes. 
723   
724      Returns 
725      ------- 
726      binary_closing : ndarray of bools 
727          Closing of the input by the structuring element. 
728   
729      See Also 
730      -------- 
731      grey_closing, binary_opening, binary_dilation, binary_erosion, 
732      generate_binary_structure 
733   
734      Notes 
735      ----- 
736      *Closing* [1]_ is a mathematical morphology operation [2]_ that 
737      consists in the succession of a dilation and an erosion of the 
738      input with the same structuring element. Closing therefore fills 
739      holes smaller than the structuring element. 
740   
741      Together with *opening* (`binary_opening`), closing can be used for 
742      noise removal. 
743   
744      References

750      --------

752      >>> import numpy as np 
753      >>> a = np.zeros((5,5), dtype=int) 
754      >>> a[1:-1, 1:-1] = 1; a[2,2] = 0 
755      >>> a 
756      array([[0, 0, 0, 0, 0], 
757             [0, 1, 1, 1, 0], 
758             [0, 1, 0, 1, 0], 
759             [0, 1, 1, 1, 0], 
760             [0, 0, 0, 0, 0]])

765             [0, 1, 1, 1, 0], 
766             [0, 1, 1, 1, 0], 
767             [0, 0, 0, 0, 0]]) 
768      >>> # Closing is the erosion of the dilation of the input 
769      >>> ndimage.binary_dilation(a).astype(int) 
770      array([[0, 1, 1, 1, 0], 
771             [1, 1, 1, 1, 1], 
772             [1, 1, 1, 1, 1], 
773             [1, 1, 1, 1, 1], 
774             [0, 1, 1, 1, 0]]) 
775      >>> ndimage.binary_erosion(ndimage.binary_dilation(a)).astype(int)

787             [0, 0, 1, 0, 1, 0, 0],

789             [0, 0, 1, 1, 1, 0, 0],

794      >>> # coarsen boundaries with fine hollows.

797             [0, 0, 1, 0, 1, 0, 0],

825  def binary_hit_or_miss(input, structure1=None, structure2=None,
826                         output=None, origin1=0, origin2=None, *, axes=None):
827      """ 
828      Multidimensional binary hit-or-miss transform. 
829   
830      The hit-or-miss transform finds the locations of a given pattern 
831      inside the input image. 
832   
833      Parameters 
834      ---------- 
835      input : array_like (cast to booleans) 
836          Binary image where a pattern is to be detected. 
837      structure1 : array_like (cast to booleans), optional 
838          Part of the structuring element to be fitted to the foreground 
839          (non-zero elements) of `input`. If no value is provided, a 
840          structure of square connectivity 1 is chosen. 
841      structure2 : array_like (cast to booleans), optional 
842          Second part of the structuring element that has to miss completely 
843          the foreground. If no value is provided, the complementary of 
844          `structure1` is taken. 
845      output : ndarray, optional 
846          Array of the same shape as input, into which the output is placed. 
847          By default, a new array is created. 
848      origin1 : int or tuple of ints, optional 
849          Placement of the first part of the structuring element `structure1`, 
850          by default 0 for a centered structure. 
851      origin2 : int or tuple of ints, optional 
852          Placement of the second part of the structuring element `structure2`, 
853          by default 0 for a centered structure. If a value is provided for 
854          `origin1` and not for `origin2`, then `origin2` is set to `origin1`. 
855      axes : tuple of int or None 
856          The axes over which to apply the filter. If None, `input` is filtered 
857          along all axes. If `origin1` or `origin2` tuples are provided, their 
858          length must match the number of axes. 
859   
860      Returns 
861      ------- 
862      binary_hit_or_miss : ndarray 
863          Hit-or-miss transform of `input` with the given structuring 
864          element (`structure1`, `structure2`). 
865   
866      See Also 
867      -------- 
868      binary_erosion 
869   
870      References 
871      ---------- 
872      .. [1] https://en.wikipedia.org/wiki/Hit-or-miss_transform 
873   
874      Examples 
875      -------- 
876      >>> from scipy import ndimage 
877      >>> import numpy as np 
878      >>> a = np.zeros((7,7), dtype=int) 
879      >>> a[1, 1] = 1; a[2:4, 2:4] = 1; a[4:6, 4:6] = 1 
880      >>> a 
881      array([[0, 0, 0, 0, 0, 0, 0], 
882             [0, 1, 0, 0, 0, 0, 0], 
883             [0, 0, 1, 1, 0, 0, 0], 
884             [0, 0, 1, 1, 0, 0, 0], 
885             [0, 0, 0, 0, 1, 1, 0], 
886             [0, 0, 0, 0, 1, 1, 0], 
887             [0, 0, 0, 0, 0, 0, 0]]) 
888      >>> structure1 = np.array([[1, 0, 0], [0, 1, 1], [0, 1, 1]]) 
889      >>> structure1 
890      array([[1, 0, 0], 
891             [0, 1, 1], 
892             [0, 1, 1]]) 
893      >>> # Find the matches of structure1 in the array a 
894      >>> ndimage.binary_hit_or_miss(a, structure1=structure1).astype(int) 
895      array([[0, 0, 0, 0, 0, 0, 0], 
896             [0, 0, 0, 0, 0, 0, 0], 
897             [0, 0, 1, 0, 0, 0, 0], 
898             [0, 0, 0, 0, 0, 0, 0], 
899             [0, 0, 0, 0, 1, 0, 0], 
900             [0, 0, 0, 0, 0, 0, 0], 
901             [0, 0, 0, 0, 0, 0, 0]]) 
902      >>> # Change the origin of the filter 
903      >>> # origin1=1 is equivalent to origin1=(1,1) here 
904      >>> ndimage.binary_hit_or_miss(a, structure1=structure1,\\ 
905      ... origin1=1).astype(int) 
906      array([[0, 0, 0, 0, 0, 0, 0], 
907             [0, 0, 0, 0, 0, 0, 0], 
908             [0, 0, 0, 0, 0, 0, 0], 
909             [0, 0, 0, 1, 0, 0, 0], 
910             [0, 0, 0, 0, 0, 0, 0], 
911             [0, 0, 0, 0, 0, 1, 0], 
912             [0, 0, 0, 0, 0, 0, 0]]) 
913   
914      """
915      input = np.asarray(input)
916      axes = _ni_support._check_axes(axes, input.ndim)
917      num_axes = len(axes)
918      if structure1 is None:
919          structure1 = generate_binary_structure(num_axes, 1)
920      else:
921          structure1 = np.asarray(structure1)
922      if structure2 is None:
923          structure2 = np.logical_not(structure1)
924      origin1 = _ni_support._normalize_sequence(origin1, num_axes)
925      if origin2 is None:
926          origin2 = origin1
927      else:
928          origin2 = _ni_support._normalize_sequence(origin2, num_axes)
929  
930      tmp1 = _binary_erosion(input, structure1, 1, None, None, 0, origin1,
931                             0, False, axes)
932      inplace = isinstance(output, np.ndarray)
933      result = _binary_erosion(input, structure2, 1, None, output, 0,
934                               origin2, 1, False, axes)
935      if inplace:
936          np.logical_not(output, output)
937          np.logical_and(tmp1, output, output)
938      else:
939          np.logical_not(result, result)
940          return np.logical_and(tmp1, result)

1201         'constant'. Default is 'reflect'

1293     or a footprint corresponding to a flat structuring element.

1296     simple case of a full and flat structuring element, it can be viewed

1344     over a domain E is given by:

1349     s(y) = 0 for y in E, the grayscale dilation computes the maximum of the

1750 def morphological_laplace(input, size=None, footprint=None, structure=None,
1751                           output=None, mode="reflect", cval=0.0, origin=0, *,
1752                           axes=None):

1754     Multidimensional morphological laplace. 
1755  
1756     Parameters 
1757     ---------- 
1758     input : array_like 
1759         Input.

1763         provided.

1766         used for the morphology operations.

1770         offsets to the pixels in a neighborhood (the offset is additive during 
1771         dilation and subtractive during erosion) 
1772     output : ndarray, optional

1774     mode : {'reflect','constant','nearest','mirror', 'wrap'}, optional 
1775         The mode parameter determines how the array borders are handled. 
1776         For 'constant' mode, values beyond borders are set to be `cval`.

1778     cval : scalar, optional

1780         Default is 0.0 
1781     origin : origin, optional 
1782         The origin parameter controls the placement of the filter. 
1783     axes : tuple of int or None 
1784         The axes over which to apply the filter. If None, `input` is filtered 
1785         along all axes. If an `origin` tuple is provided, its length must match 
1786         the number of axes.

1790     morphological_laplace : ndarray 
1791         Output

1795                          cval, origin, axes=axes)
1796     if isinstance(output, np.ndarray):
1797         grey_erosion(input, size, footprint, structure, output, mode,
1798                      cval, origin, axes=axes)
1799         np.add(tmp1, output, output)
1800         np.subtract(output, input, output)
1801         return np.subtract(output, input, output)
1802     else:
1803         tmp2 = grey_erosion(input, size, footprint, structure, None, mode,
1804                             cval, origin, axes=axes)
1805         np.add(tmp1, tmp2, tmp2)
1806         np.subtract(tmp2, input, tmp2)
1807         np.subtract(tmp2, input, tmp2)
1808         return tmp2

1985     """ 
1986     Distance transform function by a brute force algorithm. 
1987  
1988     This function calculates the distance transform of the `input`, by 
1989     replacing each foreground (non-zero) element, with its 
1990     shortest distance to the background (any zero-valued element). 
1991  
1992     In addition to the distance transform, the feature transform can 
1993     be calculated. In this case the index of the closest background 
1994     element to each foreground element is returned in a separate array. 
1995  
1996     Parameters 
1997     ---------- 
1998     input : array_like 
1999         Input 
2000     metric : {'euclidean', 'taxicab', 'chessboard'}, optional

2002         The default is 'euclidean'. 
2003     sampling : float, or sequence of float, optional

2007         all axes. If not specified, a grid spacing of unity is implied.

2010         Default is True. 
2011     return_indices : bool, optional

2021         An output array to store the calculated feature transform, instead of 
2022         returning it.

2431 def distance_transform_edt(input, sampling=None, return_distances=True,
2432                            return_indices=False, distances=None, indices=None):
2433     """ 
2434     Exact Euclidean distance transform. 
2435  
2436     This function calculates the distance transform of the `input`, by 
2437     replacing each foreground (non-zero) element, with its 
2438     shortest distance to the background (any zero-valued element). 
2439  
2440     In addition to the distance transform, the feature transform can 
2441     be calculated. In this case the index of the closest background 
2442     element to each foreground element is returned in a separate array. 
2443  
2444     Parameters 
2445     ---------- 
2446     input : array_like 
2447         Input data to transform. Can be any type but will be converted 
2448         into binary: 1 wherever input equates to True, 0 elsewhere. 
2449     sampling : float, or sequence of float, optional 
2450         Spacing of elements along each dimension. If a sequence, must be of 
2451         length equal to the input rank; if a single number, this is used for 
2452         all axes. If not specified, a grid spacing of unity is implied. 
2453     return_distances : bool, optional 
2454         Whether to calculate the distance transform. 
2455         Default is True. 
2456     return_indices : bool, optional 
2457         Whether to calculate the feature transform. 
2458         Default is False. 
2459     distances : float64 ndarray, optional 
2460         An output array to store the calculated distance transform, instead of 
2461         returning it. 
2462         `return_distances` must be True. 
2463         It must be the same shape as `input`. 
2464     indices : int32 ndarray, optional 
2465         An output array to store the calculated feature transform, instead of 
2466         returning it. 
2467         `return_indicies` must be True. 
2468         Its shape must be ``(input.ndim,) + input.shape``. 
2469  
2470     Returns 
2471     ------- 
2472     distances : float64 ndarray, optional 
2473         The calculated distance transform. Returned only when 
2474         `return_distances` is True and `distances` is not supplied. 
2475         It will have the same shape as the input array. 
2476     indices : int32 ndarray, optional 
2477         The calculated feature transform. It has an input-shaped array for each 
2478         dimension of the input. See example below. 
2479         Returned only when `return_indices` is True and `indices` is not 
2480         supplied. 
2481  
2482     Notes 
2483     ----- 
2484     The Euclidean distance transform gives values of the Euclidean 
2485     distance:: 
2486  
2487                     n 
2488       y_i = sqrt(sum (x[i]-b[i])**2) 
2489                     i 
2490  
2491     where b[i] is the background point (value 0) with the smallest 
2492     Euclidean distance to input points x[i], and n is the 
2493     number of dimensions. 
2494  
2495     Examples 
2496     -------- 
2497     >>> from scipy import ndimage 
2498     >>> import numpy as np 
2499     >>> a = np.array(([0,1,1,1,1], 
2500     ...               [0,0,1,1,1], 
2501     ...               [0,1,1,1,1], 
2502     ...               [0,1,1,1,0], 
2503     ...               [0,1,1,0,0])) 
2504     >>> ndimage.distance_transform_edt(a) 
2505     array([[ 0.    ,  1.    ,  1.4142,  2.2361,  3.    ], 
2506            [ 0.    ,  0.    ,  1.    ,  2.    ,  2.    ], 
2507            [ 0.    ,  1.    ,  1.4142,  1.4142,  1.    ], 
2508            [ 0.    ,  1.    ,  1.4142,  1.    ,  0.    ], 
2509            [ 0.    ,  1.    ,  1.    ,  0.    ,  0.    ]]) 
2510  
2511     With a sampling of 2 units along x, 1 along y: 
2512  
2513     >>> ndimage.distance_transform_edt(a, sampling=[2,1]) 
2514     array([[ 0.    ,  1.    ,  2.    ,  2.8284,  3.6056], 
2515            [ 0.    ,  0.    ,  1.    ,  2.    ,  3.    ], 
2516            [ 0.    ,  1.    ,  2.    ,  2.2361,  2.    ], 
2517            [ 0.    ,  1.    ,  2.    ,  1.    ,  0.    ], 
2518            [ 0.    ,  1.    ,  1.    ,  0.    ,  0.    ]])

2558     dt_inplace = isinstance(distances, np.ndarray)
2559     _distance_tranform_arg_check(
2560         dt_inplace, ft_inplace, return_distances, return_indices
2561     )
2562 
2563     # calculate the feature transform
2564     input = np.atleast_1d(np.where(input, 1, 0).astype(np.int8))
2565     if sampling is not None:

2571     if ft_inplace:

2595             np.sqrt(dt, distances)

2599 
2600     # construct and return the result
2601     result = []
2602     if return_distances and not dt_inplace:
2603         result.append(dt)
2604     if return_indices and not ft_inplace:
2605         result.append(ft)
2606 
2607     if len(result) == 2:
2608         return tuple(result)
2609     elif len(result) == 1:
2610         return result[0]
