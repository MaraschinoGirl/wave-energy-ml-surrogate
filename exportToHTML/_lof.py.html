19   class LocalOutlierFactor(KNeighborsMixin, OutlierMixin, NeighborsBase):
20       """Unsupervised Outlier Detection using the Local Outlier Factor (LOF). 
21    
22       The anomaly score of each sample is called the Local Outlier Factor. 
23       It measures the local deviation of the density of a given sample with respect 
24       to its neighbors. 
25       It is local in that the anomaly score depends on how isolated the object 
26       is with respect to the surrounding neighborhood. 
27       More precisely, locality is given by k-nearest neighbors, whose distance 
28       is used to estimate the local density. 
29       By comparing the local density of a sample to the local densities of its 
30       neighbors, one can identify samples that have a substantially lower density 
31       than their neighbors. These are considered outliers. 
32    
33       .. versionadded:: 0.19 
34    
35       Parameters 
36       ---------- 
37       n_neighbors : int, default=20 
38           Number of neighbors to use by default for :meth:`kneighbors` queries. 
39           If n_neighbors is larger than the number of samples provided, 
40           all samples will be used. 
41    
42       algorithm : {'auto', 'ball_tree', 'kd_tree', 'brute'}, default='auto' 
43           Algorithm used to compute the nearest neighbors: 
44    
45           - 'ball_tree' will use :class:`BallTree` 
46           - 'kd_tree' will use :class:`KDTree` 
47           - 'brute' will use a brute-force search. 
48           - 'auto' will attempt to decide the most appropriate algorithm 
49             based on the values passed to :meth:`fit` method. 
50    
51           Note: fitting on sparse input will override the setting of 
52           this parameter, using brute force. 
53    
54       leaf_size : int, default=30 
55           Leaf is size passed to :class:`BallTree` or :class:`KDTree`. This can 
56           affect the speed of the construction and query, as well as the memory 
57           required to store the tree. The optimal value depends on the 
58           nature of the problem. 
59    
60       metric : str or callable, default='minkowski' 
61           Metric to use for distance computation. Default is "minkowski", which 
62           results in the standard Euclidean distance when p = 2. See the 
63           documentation of `scipy.spatial.distance 
64           <https://docs.scipy.org/doc/scipy/reference/spatial.distance.html>`_ and 
65           the metrics listed in 
66           :class:`~sklearn.metrics.pairwise.distance_metrics` for valid metric 
67           values. 
68    
69           If metric is "precomputed", X is assumed to be a distance matrix and 
70           must be square during fit. X may be a :term:`sparse graph`, in which 
71           case only "nonzero" elements may be considered neighbors. 
72    
73           If metric is a callable function, it takes two arrays representing 1D 
74           vectors as inputs and must return one value indicating the distance 
75           between those vectors. This works for Scipy's metrics, but is less 
76           efficient than passing the metric name as a string. 
77    
78       p : float, default=2 
79           Parameter for the Minkowski metric from 
80           :func:`sklearn.metrics.pairwise_distances`. When p = 1, this 
81           is equivalent to using manhattan_distance (l1), and euclidean_distance 
82           (l2) for p = 2. For arbitrary p, minkowski_distance (l_p) is used. 
83    
84       metric_params : dict, default=None 
85           Additional keyword arguments for the metric function. 
86    
87       contamination : 'auto' or float, default='auto' 
88           The amount of contamination of the data set, i.e. the proportion 
89           of outliers in the data set. When fitting this is used to define the 
90           threshold on the scores of the samples. 
91    
92           - if 'auto', the threshold is determined as in the 
93             original paper, 
94           - if a float, the contamination should be in the range (0, 0.5]. 
95    
96           .. versionchanged:: 0.22 
97              The default value of ``contamination`` changed from 0.1 
98              to ``'auto'``. 
99    
100      novelty : bool, default=False 
101          By default, LocalOutlierFactor is only meant to be used for outlier 
102          detection (novelty=False). Set novelty to True if you want to use 
103          LocalOutlierFactor for novelty detection. In this case be aware that 
104          you should only use predict, decision_function and score_samples 
105          on new unseen data and not on the training set; and note that the 
106          results obtained this way may differ from the standard LOF results. 
107   
108          .. versionadded:: 0.20 
109   
110      n_jobs : int, default=None 
111          The number of parallel jobs to run for neighbors search. 
112          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
113          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
114          for more details. 
115   
116      Attributes 
117      ---------- 
118      negative_outlier_factor_ : ndarray of shape (n_samples,) 
119          The opposite LOF of the training samples. The higher, the more normal. 
120          Inliers tend to have a LOF score close to 1 
121          (``negative_outlier_factor_`` close to -1), while outliers tend to have 
122          a larger LOF score. 
123   
124          The local outlier factor (LOF) of a sample captures its 
125          supposed 'degree of abnormality'. 
126          It is the average of the ratio of the local reachability density of 
127          a sample and those of its k-nearest neighbors. 
128   
129      n_neighbors_ : int 
130          The actual number of neighbors used for :meth:`kneighbors` queries. 
131   
132      offset_ : float 
133          Offset used to obtain binary labels from the raw scores. 
134          Observations having a negative_outlier_factor smaller than `offset_` 
135          are detected as abnormal. 
136          The offset is set to -1.5 (inliers score around -1), except when a 
137          contamination parameter different than "auto" is provided. In that 
138          case, the offset is defined in such a way we obtain the expected 
139          number of outliers in training. 
140   
141          .. versionadded:: 0.20 
142   
143      effective_metric_ : str 
144          The effective metric used for the distance computation. 
145   
146      effective_metric_params_ : dict 
147          The effective additional keyword arguments for the metric function. 
148   
149      n_features_in_ : int 
150          Number of features seen during :term:`fit`. 
151   
152          .. versionadded:: 0.24 
153   
154      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
155          Names of features seen during :term:`fit`. Defined only when `X` 
156          has feature names that are all strings. 
157   
158          .. versionadded:: 1.0 
159   
160      n_samples_fit_ : int 
161          It is the number of samples in the fitted data. 
162   
163      See Also 
164      -------- 
165      sklearn.svm.OneClassSVM: Unsupervised Outlier Detection using 
166          Support Vector Machine. 
167   
168      References 
169      ---------- 
170      .. [1] Breunig, M. M., Kriegel, H. P., Ng, R. T., & Sander, J. (2000, May). 
171             LOF: identifying density-based local outliers. In ACM sigmod record. 
172   
173      Examples 
174      -------- 
175      >>> import numpy as np 
176      >>> from sklearn.neighbors import LocalOutlierFactor 
177      >>> X = [[-1.1], [0.2], [101.1], [0.3]] 
178      >>> clf = LocalOutlierFactor(n_neighbors=2) 
179      >>> clf.fit_predict(X) 
180      array([ 1,  1, -1,  1]) 
181      >>> clf.negative_outlier_factor_ 
182      array([ -0.9821...,  -1.0370..., -73.3697...,  -0.9821...]) 
183      """

229      @available_if(_check_novelty_fit_predict)
230      def fit_predict(self, X, y=None):
231          """Fit the model to the training set X and return the labels. 
232   
233          **Not available for novelty detection (when novelty is set to True).** 
234          Label is 1 for an inlier and -1 for an outlier according to the LOF 
235          score and the contamination parameter. 
236   
237          Parameters 
238          ---------- 
239          X : {array-like, sparse matrix} of shape (n_samples, n_features), default=None 
240              The query sample or samples to compute the Local Outlier Factor 
241              w.r.t. the training samples. 
242   
243          y : Ignored 
244              Not used, present for API consistency by convention. 
245   
246          Returns 
247          ------- 
248          is_inlier : ndarray of shape (n_samples,) 
249              Returns -1 for anomalies/outliers and 1 for inliers. 
250          """
251  
252          # As fit_predict would be different from fit.predict, fit_predict is
253          # only available for outlier detection (novelty=False)
254  
255          return self.fit(X)._predict()

257      @_fit_context(
258          # LocalOutlierFactor.metric is not validated yet
259          prefer_skip_nested_validation=False
260      )
261      def fit(self, X, y=None):
262          """Fit the local outlier factor detector from the training dataset. 
263   
264          Parameters 
265          ---------- 
266          X : {array-like, sparse matrix} of shape (n_samples, n_features) or \ 
267                  (n_samples, n_samples) if metric='precomputed' 
268              Training data. 
269   
270          y : Ignored 
271              Not used, present for API consistency by convention. 
272   
273          Returns 
274          ------- 
275          self : LocalOutlierFactor 
276              The fitted local outlier factor detector. 
277          """
278          self._fit(X)
279  
280          n_samples = self.n_samples_fit_
281          if self.n_neighbors > n_samples:
282              warnings.warn(
283                  "n_neighbors (%s) is greater than the "
284                  "total number of samples (%s). n_neighbors "
285                  "will be set to (n_samples - 1) for estimation."
286                  % (self.n_neighbors, n_samples)
287              )
288          self.n_neighbors_ = max(1, min(self.n_neighbors, n_samples - 1))
289  
290          self._distances_fit_X_, _neighbors_indices_fit_X_ = self.kneighbors(
291              n_neighbors=self.n_neighbors_
292          )
293  
294          if self._fit_X.dtype == np.float32:
295              self._distances_fit_X_ = self._distances_fit_X_.astype(
296                  self._fit_X.dtype,
297                  copy=False,
298              )
299  
300          self._lrd = self._local_reachability_density(
301              self._distances_fit_X_, _neighbors_indices_fit_X_
302          )
303  
304          # Compute lof score over training samples to define offset_:
305          lrd_ratios_array = (
306              self._lrd[_neighbors_indices_fit_X_] / self._lrd[:, np.newaxis]
307          )
308  
309          self.negative_outlier_factor_ = -np.mean(lrd_ratios_array, axis=1)
310  
311          if self.contamination == "auto":
312              # inliers score around -1 (the higher, the less abnormal).
313              self.offset_ = -1.5
314          else:
315              self.offset_ = np.percentile(
316                  self.negative_outlier_factor_, 100.0 * self.contamination
317              )
318  
319          # Verify if negative_outlier_factor_ values are within acceptable range.
320          # Novelty must also be false to detect outliers
321          if np.min(self.negative_outlier_factor_) < -1e7 and not self.novelty:
322              warnings.warn(
323                  "Duplicate values are leading to incorrect results. "
324                  "Increase the number of neighbors for more accurate results."
325              )
326  
327          return self
328

330          if not self.novelty:
331              msg = (
332                  "predict is not available when novelty=False, use "
333                  "fit_predict if you want to predict on training data. Use "
334                  "novelty=True if you want to use LOF for novelty detection "
335                  "and predict on new unseen data."

340      @available_if(_check_novelty_predict)
341      def predict(self, X=None):
342          """Predict the labels (1 inlier, -1 outlier) of X according to LOF. 
343   
344          **Only available for novelty detection (when novelty is set to True).** 
345          This method allows to generalize prediction to *new observations* (not 
346          in the training set). Note that the result of ``clf.fit(X)`` then 
347          ``clf.predict(X)`` with ``novelty=True`` may differ from the result 
348          obtained by ``clf.fit_predict(X)`` with ``novelty=False``. 
349   
350          Parameters 
351          ---------- 
352          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
353              The query sample or samples to compute the Local Outlier Factor 
354              w.r.t. the training samples. 
355   
356          Returns 
357          ------- 
358          is_inlier : ndarray of shape (n_samples,) 
359              Returns -1 for anomalies/outliers and +1 for inliers. 
360          """
361          return self._predict(X)
362  
363      def _predict(self, X=None):
364          """Predict the labels (1 inlier, -1 outlier) of X according to LOF. 
365   
366          If X is None, returns the same as fit_predict(X_train). 
367   
368          Parameters 
369          ---------- 
370          X : {array-like, sparse matrix} of shape (n_samples, n_features), default=None 
371              The query sample or samples to compute the Local Outlier Factor 
372              w.r.t. the training samples. If None, makes prediction on the 
373              training data without considering them as their own neighbors. 
374   
375          Returns 
376          ------- 
377          is_inlier : ndarray of shape (n_samples,) 
378              Returns -1 for anomalies/outliers and +1 for inliers. 
379          """
380          check_is_fitted(self)
381  
382          if X is not None:
383              shifted_opposite_lof_scores = self.decision_function(X)
384              is_inlier = np.ones(shifted_opposite_lof_scores.shape[0], dtype=int)
385              is_inlier[shifted_opposite_lof_scores < 0] = -1
386          else:
387              is_inlier = np.ones(self.n_samples_fit_, dtype=int)
388              is_inlier[self.negative_outlier_factor_ < self.offset_] = -1
389  
390          return is_inlier

392      def _check_novelty_decision_function(self):
393          if not self.novelty:
394              msg = (
395                  "decision_function is not available when novelty=False. "
396                  "Use novelty=True if you want to use LOF for novelty "
397                  "detection and compute decision_function for new unseen "
398                  "data. Note that the opposite LOF of the training samples "
399                  "is always available by considering the "
400                  "negative_outlier_factor_ attribute."
401              )
402              raise AttributeError(msg)
403          return True

405      @available_if(_check_novelty_decision_function)
406      def decision_function(self, X):
407          """Shifted opposite of the Local Outlier Factor of X. 
408   
409          Bigger is better, i.e. large values correspond to inliers. 
410   
411          **Only available for novelty detection (when novelty is set to True).** 
412          The shift offset allows a zero threshold for being an outlier. 
413          The argument X is supposed to contain *new data*: if X contains a 
414          point from training, it considers the later in its own neighborhood. 
415          Also, the samples in X are not considered in the neighborhood of any 
416          point. 
417   
418          Parameters 
419          ---------- 
420          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
421              The query sample or samples to compute the Local Outlier Factor 
422              w.r.t. the training samples. 
423   
424          Returns 
425          ------- 
426          shifted_opposite_lof_scores : ndarray of shape (n_samples,) 
427              The shifted opposite of the Local Outlier Factor of each input 
428              samples. The lower, the more abnormal. Negative scores represent 
429              outliers, positive scores represent inliers. 
430          """
431          return self.score_samples(X) - self.offset_

445      @available_if(_check_novelty_score_samples)
446      def score_samples(self, X):
447          """Opposite of the Local Outlier Factor of X. 
448   
449          It is the opposite as bigger is better, i.e. large values correspond 
450          to inliers. 
451   
452          **Only available for novelty detection (when novelty is set to True).** 
453          The argument X is supposed to contain *new data*: if X contains a 
454          point from training, it considers the later in its own neighborhood. 
455          Also, the samples in X are not considered in the neighborhood of any 
456          point. Because of this, the scores obtained via ``score_samples`` may 
457          differ from the standard LOF scores. 
458          The standard LOF scores for the training data is available via the 
459          ``negative_outlier_factor_`` attribute. 
460   
461          Parameters 
462          ---------- 
463          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
464              The query sample or samples to compute the Local Outlier Factor 
465              w.r.t. the training samples. 
466   
467          Returns 
468          ------- 
469          opposite_lof_scores : ndarray of shape (n_samples,) 
470              The opposite of the Local Outlier Factor of each input samples. 
471              The lower, the more abnormal. 
472          """
473          check_is_fitted(self)
474          X = check_array(X, accept_sparse="csr")
475  
476          distances_X, neighbors_indices_X = self.kneighbors(
477              X, n_neighbors=self.n_neighbors_
478          )
479  
480          if X.dtype == np.float32:
481              distances_X = distances_X.astype(X.dtype, copy=False)
482  
483          X_lrd = self._local_reachability_density(
484              distances_X,
485              neighbors_indices_X,
486          )
487  
488          lrd_ratios_array = self._lrd[neighbors_indices_X] / X_lrd[:, np.newaxis]
489  
490          # as bigger is better:
491          return -np.mean(lrd_ratios_array, axis=1)
492  
493      def _local_reachability_density(self, distances_X, neighbors_indices):
494          """The local reachability density (LRD) 
495   
496          The LRD of a sample is the inverse of the average reachability 
497          distance of its k-nearest neighbors. 
498   
499          Parameters 
500          ---------- 
501          distances_X : ndarray of shape (n_queries, self.n_neighbors) 
502              Distances to the neighbors (in the training samples `self._fit_X`) 
503              of each query point to compute the LRD. 
504   
505          neighbors_indices : ndarray of shape (n_queries, self.n_neighbors) 
506              Neighbors indices (of each query point) among training samples 
507              self._fit_X. 
508   
509          Returns 
510          ------- 
511          local_reachability_density : ndarray of shape (n_queries,) 
512              The local reachability density of each sample. 
513          """
514          dist_k = self._distances_fit_X_[neighbors_indices, self.n_neighbors_ - 1]
515          reach_dist_array = np.maximum(distances_X, dist_k)
516  
517          # 1e-10 to avoid `nan' when nb of duplicates > n_neighbors_:
518          return 1.0 / (np.mean(reach_dist_array, axis=1) + 1e-10)
