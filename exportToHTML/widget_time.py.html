4    """ 
5    Time picker widget 
6    """
7    
8    from traitlets import Unicode, Bool, Union, CaselessStrEnum, CFloat, validate, TraitError
9    
10   from .trait_types import Time, time_serialization
11   from .valuewidget import ValueWidget
12   from .widget import register
13   from .widget_core import CoreWidget
14   from .widget_description import DescriptionWidget
15   
16   
17   @register
18   class TimePicker(DescriptionWidget, ValueWidget, CoreWidget):
19       """ 
20       Display a widget for picking times. 
21    
22       Parameters 
23       ---------- 
24    
25       value: datetime.time 
26           The current value of the widget. 
27    
28       disabled: bool 
29           Whether to disable user changes. 
30    
31       min: datetime.time 
32           The lower allowed time bound 
33    
34       max: datetime.time 
35           The upper allowed time bound 
36    
37       step: float | 'any' 
38           The time step to use for the picker, in seconds, or "any" 
39    
40       Examples 
41       -------- 
42    
43       >>> import datetime 
44       >>> import ipydatetime 
45       >>> time_pick = ipydatetime.TimePicker() 
46       >>> time_pick.value = datetime.time(12, 34, 3) 
47       """
48   
49       _view_name = Unicode("TimeView").tag(sync=True)
50       _model_name = Unicode("TimeModel").tag(sync=True)
51   
52       value = Time(None, allow_none=True).tag(sync=True, **time_serialization)
53       disabled = Bool(False, help="Enable or disable user changes.").tag(sync=True)
54   
55       min = Time(None, allow_none=True).tag(sync=True, **time_serialization)
56       max = Time(None, allow_none=True).tag(sync=True, **time_serialization)
57       step = Union(
58           (CFloat(60), CaselessStrEnum(["any"])),
59           help='The time step to use for the picker, in seconds, or "any".',
60       ).tag(sync=True)
61   
62       @validate("value")
63       def _validate_value(self, proposal):
64           """Cap and floor value"""
65           value = proposal["value"]
66           if value is None:
67               return value
68           if self.min and self.min > value:
69               value = max(value, self.min)
70           if self.max and self.max < value:
71               value = min(value, self.max)
72           return value
73   
74       @validate("min")
75       def _validate_min(self, proposal):
76           """Enforce min <= value <= max"""
77           min = proposal["value"]
78           if min is None:
79               return min
80           if self.max and min > self.max:
81               raise TraitError("Setting min > max")
82           if self.value and min > self.value:
83               self.value = min
84           return min
85   
86       @validate("max")
87       def _validate_max(self, proposal):
88           """Enforce min <= value <= max"""
89           max = proposal["value"]
90           if max is None:
91               return max
92           if self.min and max < self.min:
93               raise TraitError("setting max < min")
94           if self.value and max < self.value:
95               self.value = max
96           return max
