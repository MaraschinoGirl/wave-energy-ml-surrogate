1    """Python.Dictionary COM Server. 
2     
3    This module implements a simple COM server that acts much like a Python 
4    dictionary or as a standard string-keyed VB Collection.  The keys of 
5    the dictionary are strings and are case-insensitive. 
6     
7    It uses a highly customized policy to fine-tune the behavior exposed to 
8    the COM client. 
9     
10   The object exposes the following properties: 
11    
12       int Count                       (readonly) 
13       VARIANT Item(BSTR key)          (propget for Item) 
14       Item(BSTR key, VARIANT value)   (propput for Item) 
15    
16       Note that 'Item' is the default property, so the following forms of 
17       VB code are acceptable: 
18    
19           set ob = CreateObject("Python.Dictionary") 
20           ob("hello") = "there" 
21           ob.Item("hi") = ob("HELLO") 
22    
23   All keys are defined, returning VT_NULL (None) if a value has not been 
24   stored.  To delete a key, simply assign VT_NULL to the key. 
25    
26   The object responds to the _NewEnum method by returning an enumerator over 
27   the dictionary's keys. This allows for the following type of VB code: 
28    
29       for each name in ob 
30           debug.print(name, ob(name)) 
31       next 
32   """
33   
34   import pythoncom
35   import pywintypes
36   import winerror
37   from pythoncom import DISPATCH_METHOD, DISPATCH_PROPERTYGET
38   from win32com.server import policy, util
39   from win32com.server.exception import COMException
40   from winerror import S_OK
41   
42   
43   class DictionaryPolicy(policy.BasicWrapPolicy):
44       ### BasicWrapPolicy looks for this
45       _com_interfaces_ = []
46   
47       ### BasicWrapPolicy looks for this
48       _name_to_dispid_ = {
49           "item": pythoncom.DISPID_VALUE,
50           "_newenum": pythoncom.DISPID_NEWENUM,
51           "count": 1,
52       }
53   
54       ### Auto-Registration process looks for these...
55       _reg_desc_ = "Python Dictionary"
56       _reg_clsid_ = "{39b61048-c755-11d0-86fa-00c04fc2e03e}"
57       _reg_progid_ = "Python.Dictionary"
58       _reg_verprogid_ = "Python.Dictionary.1"
59       _reg_policy_spec_ = "win32com.servers.dictionary.DictionaryPolicy"
60   
61       def _CreateInstance_(self, clsid, reqIID):
62           self._wrap_({})
63           return pythoncom.WrapObject(self, reqIID)
64   
65       def _wrap_(self, ob):
66           self._obj_ = ob  # ob should be a dictionary
67   
68       def _invokeex_(self, dispid, lcid, wFlags, args, kwargs, serviceProvider):
69           if dispid == 0:  # item
70               l = len(args)
71               if l < 1:
72                   raise COMException(
73                       desc="not enough parameters", scode=winerror.DISP_E_BADPARAMCOUNT
74                   )
75   
76               key = args[0]
77               if not isinstance(key, str):
78                   ### the nArgErr thing should be 0-based, not reversed... sigh
79                   raise COMException(
80                       desc="Key must be a string", scode=winerror.DISP_E_TYPEMISMATCH
81                   )
82   
83               key = key.lower()
84   
85               if wFlags & (DISPATCH_METHOD | DISPATCH_PROPERTYGET):
86                   if l > 1:
87                       raise COMException(scode=winerror.DISP_E_BADPARAMCOUNT)
88                   return self._obj_.get(key)  # unknown keys return None (VT_NULL)
89   
90               if l != 2:
91                   raise COMException(scode=winerror.DISP_E_BADPARAMCOUNT)
92               if args[1] is None:
93                   # delete a key when None is assigned to it
94                   try:
95                       del self._obj_[key]
96                   except KeyError:
97                       pass
98               else:
99                   self._obj_[key] = args[1]
100              return S_OK
101  
102          if dispid == 1:  # count
103              if not wFlags & DISPATCH_PROPERTYGET:
104                  raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)  # not found
105              if len(args) != 0:
106                  raise COMException(scode=winerror.DISP_E_BADPARAMCOUNT)
107              return len(self._obj_)
108  
109          if dispid == pythoncom.DISPID_NEWENUM:
110              return util.NewEnum(list(self._obj_))
111  
112          raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)
113  
114      def _getidsofnames_(self, names, lcid):
115          ### this is a copy of MappedWrapPolicy._getidsofnames_ ...
116  
117          name = names[0].lower()
118          try:
119              return (self._name_to_dispid_[name],)
120          except KeyError:
121              raise COMException(
122                  scode=winerror.DISP_E_MEMBERNOTFOUND, desc="Member not found"
123              )
124  
125  
126  def Register():
127      from win32com.server.register import UseCommandLine
128  
129      return UseCommandLine(DictionaryPolicy)
130  
131  
132  if __name__ == "__main__":
133      Register()
