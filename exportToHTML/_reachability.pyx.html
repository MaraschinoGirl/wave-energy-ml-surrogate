44   def mutual_reachability_graph(
45       distance_matrix, min_samples=5, max_distance=0.0
46   ):
47       """Compute the weighted adjacency matrix of the mutual reachability graph. 
48    
49       The mutual reachability distance used to build the graph is defined as:: 
50    
51           max(d_core(x_p), d_core(x_q), d(x_p, x_q)) 
52    
53       and the core distance `d_core` is defined as the distance between a point 
54       `x_p` and its k-th nearest neighbor. 
55    
56       Note that all computations are done in-place. 
57    
58       Parameters 
59       ---------- 
60       distance_matrix : {ndarray, sparse matrix} of shape (n_samples, n_samples) 
61           Array of distances between samples. If sparse, the array must be in 
62           `CSR` format. 
63    
64       min_samples : int, default=5 
65           The parameter `k` used to calculate the distance between a point 
66           `x_p` and its k-th nearest neighbor. 
67    
68       max_distance : float, default=0.0 
69           The distance which `np.inf` is replaced with. When the true mutual- 
70           reachability distance is measured to be infinite, it is instead 
71           truncated to `max_dist`. Only used when `distance_matrix` is a sparse 
72           matrix. 
73    
74       Returns 
75       ------- 
76       mututal_reachability_graph: {ndarray, sparse matrix} of shape \ 
77               (n_samples, n_samples) 
78           Weighted adjacency matrix of the mutual reachability graph. 
79    
80       References 
81       ---------- 
82       .. [1] Campello, R. J., Moulavi, D., & Sander, J. (2013, April). 
83          Density-based clustering based on hierarchical density estimates. 
84          In Pacific-Asia Conference on Knowledge Discovery and Data Mining 
85          (pp. 160-172). Springer Berlin Heidelberg. 
86       """
87       further_neighbor_idx = min_samples - 1
88       if issparse(distance_matrix):
89           if distance_matrix.format != "csr":
90               raise ValueError(
91                   "Only sparse CSR matrices are supported for `distance_matrix`."
92               )
93           _sparse_mutual_reachability_graph(
94               distance_matrix.data,
95               distance_matrix.indices,
96               distance_matrix.indptr,
97               distance_matrix.shape[0],
98               further_neighbor_idx=further_neighbor_idx,
99               max_distance=max_distance,
100          )
101      else:
102          _dense_mutual_reachability_graph(
103              distance_matrix, further_neighbor_idx=further_neighbor_idx
104          )
105      return distance_matrix

108  def _dense_mutual_reachability_graph(
109      floating[:, :] distance_matrix,
110      intp_t further_neighbor_idx,
111  ):
112      """Dense implementation of mutual reachability graph. 
113   
114      The computation is done in-place, i.e. the distance matrix is modified 
115      directly. 
116   
117      Parameters 
118      ---------- 
119      distance_matrix : ndarray of shape (n_samples, n_samples) 
120          Array of distances between samples. 
121   
122      further_neighbor_idx : int 
123          The index of the furthest neighbor to use to define the core distances. 
124      """
125      cdef:
126          intp_t i, j, n_samples = distance_matrix.shape[0]
127          floating mutual_reachability_distance
128          floating[::1] core_distances
129  
130      # We assume that the distance matrix is symmetric. We choose to sort every
131      # row to have the same implementation than the sparse case that requires
132      # CSR matrix.
133      core_distances = np.ascontiguousarray(
134          np.partition(
135              distance_matrix, further_neighbor_idx, axis=1
136          )[:, further_neighbor_idx]
137      )
138  
139      with nogil:
140          # TODO: Update w/ prange with thread count based on
141          # _openmp_effective_n_threads
142          for i in range(n_samples):
143              for j in range(n_samples):
144                  mutual_reachability_distance = max(
145                      core_distances[i],
146                      core_distances[j],
147                      distance_matrix[i, j],
148                  )
149                  distance_matrix[i, j] = mutual_reachability_distance

152  def _sparse_mutual_reachability_graph(
153      cnp.ndarray[floating, ndim=1, mode="c"] data,
154      cnp.ndarray[integral, ndim=1, mode="c"] indices,
155      cnp.ndarray[integral, ndim=1, mode="c"] indptr,
156      intp_t n_samples,
157      intp_t further_neighbor_idx,
158      floating max_distance,
159  ):
160      """Sparse implementation of mutual reachability graph. 
161   
162      The computation is done in-place, i.e. the distance matrix is modified 
163      directly. This implementation only accepts `CSR` format sparse matrices. 
164   
165      Parameters 
166      ---------- 
167      distance_matrix : sparse matrix of shape (n_samples, n_samples) 
168          Sparse matrix of distances between samples. The sparse format should 
169          be `CSR`. 
170   
171      further_neighbor_idx : int 
172          The index of the furthest neighbor to use to define the core distances. 
173   
174      max_distance : float 
175          The distance which `np.inf` is replaced with. When the true mutual- 
176          reachability distance is measured to be infinite, it is instead 
177          truncated to `max_dist`. Only used when `distance_matrix` is a sparse 
178          matrix. 
179      """
180      cdef:
181          integral i, col_ind, row_ind
182          floating mutual_reachability_distance
183          floating[:] core_distances
184          floating[:] row_data
185  
186      if floating is float:
187          dtype = np.float32
188      else:
189          dtype = np.float64
190  
191      core_distances = np.empty(n_samples, dtype=dtype)
192  
193      for i in range(n_samples):
194          row_data = data[indptr[i]:indptr[i + 1]]
195          if further_neighbor_idx < row_data.size:
196              core_distances[i] = np.partition(
197                  row_data, further_neighbor_idx
198              )[further_neighbor_idx]
199          else:
200              core_distances[i] = INFINITY
201  
202      with nogil:
203          for row_ind in range(n_samples):
204              for i in range(indptr[row_ind], indptr[row_ind + 1]):
205                  col_ind = indices[i]
206                  mutual_reachability_distance = max(
207                      core_distances[row_ind], core_distances[col_ind], data[i]
208                  )
209                  if isfinite(mutual_reachability_distance):
210                      data[i] = mutual_reachability_distance
211                  elif max_distance > 0:
212                      data[i] = max_distance
