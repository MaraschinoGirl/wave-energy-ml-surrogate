6    _import_everything = b""" 
7    # The event loop is not fork-safe, and it's easy to initialize an asyncio.Future 
8    # at startup, which in turn creates the default event loop and prevents forking. 
9    # Explicitly disallow the default event loop so that an error will be raised 
10   # if something tries to touch it. 
11   import asyncio 
12   asyncio.set_event_loop(None) 
13    
14   import importlib 
15   import tornado 
16    
17   for mod in tornado.__all__: 
18       if mod == "curl_httpclient": 
19           # This module has extra dependencies; skip it if they're not installed. 
20           try: 
21               import pycurl 
22           except ImportError: 
23               continue 
24       importlib.import_module(f"tornado.{mod}") 
25   """

28   import sys 
29   import tornado 
30    
31   if "tornado.web" in sys.modules: 
32       raise Exception("unexpected eager import") 
33    
34   # Trigger a lazy import by referring to something in a submodule. 
35   tornado.web.RequestHandler 
36    
37   if "tornado.web" not in sys.modules: 
38       raise Exception("lazy import did not update sys.modules") 
39   """

42   class ImportTest(unittest.TestCase):
43       def test_import_everything(self):
44           # Test that all Tornado modules can be imported without side effects,
45           # specifically without initializing the default asyncio event loop.
46           # Since we can't tell which modules may have already beein imported
47           # in our process, do it in a subprocess for a clean slate.
48           proc = subprocess.Popen([sys.executable], stdin=subprocess.PIPE)
49           proc.communicate(_import_everything)
50           self.assertEqual(proc.returncode, 0)
51   
52       def test_lazy_import(self):
53           # Test that submodules can be referenced lazily after "import tornado"
54           proc = subprocess.Popen([sys.executable], stdin=subprocess.PIPE)
55           proc.communicate(_import_lazy)
56           self.assertEqual(proc.returncode, 0)
57   
58       def test_import_aliases(self):
59           # Ensure we don't delete formerly-documented aliases accidentally.
60           import tornado
61           import asyncio
62   
63           self.assertIs(tornado.ioloop.TimeoutError, tornado.util.TimeoutError)
64           self.assertIs(tornado.gen.TimeoutError, tornado.util.TimeoutError)
65           self.assertIs(tornado.util.TimeoutError, asyncio.TimeoutError)
