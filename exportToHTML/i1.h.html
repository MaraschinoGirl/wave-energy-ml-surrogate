6     * 
7     *     Modified Bessel function of order one 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double x, y, i1(); 
14    * 
15    * y = i1( x ); 
16    * 
17    * 
18    * 
19    * DESCRIPTION: 
20    * 
21    * Returns modified Bessel function of order one of the 
22    * argument. 
23    * 
24    * The function is defined as i1(x) = -i j1( ix ). 
25    * 
26    * The range is partitioned into the two intervals [0,8] and 
27    * (8, infinity).  Chebyshev polynomial expansions are employed 
28    * in each interval. 
29    * 
30    *

33    * 
34    *                      Relative error:

125      XSF_HOST_DEVICE inline double i1(double x) {
126          double y, z;
127  
128          z = std::abs(x);
129          if (z <= 8.0) {
130              y = (z / 2.0) - 2.0;
131              z = chbevl(y, detail::i1_A, 29) * z * std::exp(z);
132          } else {
133              z = std::exp(z) * chbevl(32.0 / z - 2.0, detail::i1_B, 25) / std::sqrt(z);
134          }
135          if (x < 0.0)
136              z = -z;
137          return (z);
138      }

142      XSF_HOST_DEVICE inline double i1e(double x) {
143          double y, z;
144  
145          z = std::abs(x);
146          if (z <= 8.0) {
147              y = (z / 2.0) - 2.0;
148              z = chbevl(y, detail::i1_A, 29) * z;
149          } else {
150              z = chbevl(32.0 / z - 2.0, detail::i1_B, 25) / std::sqrt(z);
151          }
152          if (x < 0.0)
153              z = -z;
154          return (z);
155      }
