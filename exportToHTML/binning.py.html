25   def _find_binning_thresholds(col_data, max_bins):
26       """Extract quantiles from a continuous feature. 
27    
28       Missing values are ignored for finding the thresholds. 
29    
30       Parameters 
31       ---------- 
32       col_data : array-like, shape (n_samples,) 
33           The continuous feature to bin. 
34       max_bins: int 
35           The maximum number of bins to use for non-missing values. If for a 
36           given feature the number of unique values is less than ``max_bins``, 
37           then those unique values will be used to compute the bin thresholds, 
38           instead of the quantiles 
39    
40       Return 
41       ------ 
42       binning_thresholds : ndarray of shape(min(max_bins, n_unique_values) - 1,) 
43           The increasing numeric values that can be used to separate the bins. 
44           A given value x will be mapped into bin value i iff 
45           bining_thresholds[i - 1] < x <= binning_thresholds[i] 
46       """
47       # ignore missing values when computing bin thresholds
48       missing_mask = np.isnan(col_data)
49       if missing_mask.any():
50           col_data = col_data[~missing_mask]
51       # The data will be sorted anyway in np.unique and again in percentile, so we do it
52       # here. Sorting also returns a contiguous array.
53       col_data = np.sort(col_data)
54       distinct_values = np.unique(col_data).astype(X_DTYPE)
55       if len(distinct_values) <= max_bins:
56           midpoints = distinct_values[:-1] + distinct_values[1:]
57           midpoints *= 0.5
58       else:
59           # We could compute approximate midpoint percentiles using the output of
60           # np.unique(col_data, return_counts) instead but this is more
61           # work and the performance benefit will be limited because we
62           # work on a fixed-size subsample of the full data.
63           percentiles = np.linspace(0, 100, num=max_bins + 1)
64           percentiles = percentiles[1:-1]
65           midpoints = percentile(col_data, percentiles, method="midpoint").astype(X_DTYPE)
66           assert midpoints.shape[0] == max_bins - 1
67   
68       # We avoid having +inf thresholds: +inf thresholds are only allowed in
69       # a "split on nan" situation.
70       np.clip(midpoints, a_min=None, a_max=ALMOST_INF, out=midpoints)
71       return midpoints

74   class _BinMapper(TransformerMixin, BaseEstimator):
75       """Transformer that maps a dataset into integer-valued bins. 
76    
77       For continuous features, the bins are created in a feature-wise fashion, 
78       using quantiles so that each bins contains approximately the same number 
79       of samples. For large datasets, quantiles are computed on a subset of the 
80       data to speed-up the binning, but the quantiles should remain stable. 
81    
82       For categorical features, the raw categorical values are expected to be 
83       in [0, 254] (this is not validated here though) and each category 
84       corresponds to a bin. All categorical values must be known at 
85       initialization: transform() doesn't know how to bin unknown categorical 
86       values. Note that transform() is only used on non-training data in the 
87       case of early stopping. 
88    
89       Features with a small number of values may be binned into less than 
90       ``n_bins`` bins. The last bin (at index ``n_bins - 1``) is always reserved 
91       for missing values. 
92    
93       Parameters 
94       ---------- 
95       n_bins : int, default=256 
96           The maximum number of bins to use (including the bin for missing 
97           values). Should be in [3, 256]. Non-missing values are binned on 
98           ``max_bins = n_bins - 1`` bins. The last bin is always reserved for 
99           missing values. If for a given feature the number of unique values is 
100          less than ``max_bins``, then those unique values will be used to 
101          compute the bin thresholds, instead of the quantiles. For categorical 
102          features indicated by ``is_categorical``, the docstring for 
103          ``is_categorical`` details on this procedure. 
104      subsample : int or None, default=2e5 
105          If ``n_samples > subsample``, then ``sub_samples`` samples will be 
106          randomly chosen to compute the quantiles. If ``None``, the whole data 
107          is used. 
108      is_categorical : ndarray of bool of shape (n_features,), default=None 
109          Indicates categorical features. By default, all features are 
110          considered continuous. 
111      known_categories : list of {ndarray, None} of shape (n_features,), \ 
112              default=none 
113          For each categorical feature, the array indicates the set of unique 
114          categorical values. These should be the possible values over all the 
115          data, not just the training data. For continuous features, the 
116          corresponding entry should be None. 
117      random_state: int, RandomState instance or None, default=None 
118          Pseudo-random number generator to control the random sub-sampling. 
119          Pass an int for reproducible output across multiple 
120          function calls. 
121          See :term:`Glossary <random_state>`. 
122      n_threads : int, default=None 
123          Number of OpenMP threads to use. `_openmp_effective_n_threads` is called 
124          to determine the effective number of threads use, which takes cgroups CPU 
125          quotes into account. See the docstring of `_openmp_effective_n_threads` 
126          for details. 
127   
128      Attributes 
129      ---------- 
130      bin_thresholds_ : list of ndarray 
131          For each feature, each array indicates how to map a feature into a 
132          binned feature. The semantic and size depends on the nature of the 
133          feature: 
134          - for real-valued features, the array corresponds to the real-valued 
135            bin thresholds (the upper bound of each bin). There are ``max_bins 
136            - 1`` thresholds, where ``max_bins = n_bins - 1`` is the number of 
137            bins used for non-missing values. 
138          - for categorical features, the array is a map from a binned category 
139            value to the raw category value. The size of the array is equal to 
140            ``min(max_bins, category_cardinality)`` where we ignore missing 
141            values in the cardinality. 
142      n_bins_non_missing_ : ndarray, dtype=np.uint32 
143          For each feature, gives the number of bins actually used for 
144          non-missing values. For features with a lot of unique values, this is 
145          equal to ``n_bins - 1``. 
146      is_categorical_ : ndarray of shape (n_features,), dtype=np.uint8 
147          Indicator for categorical features. 
148      missing_values_bin_idx_ : np.uint8 
149          The index of the bin where missing values are mapped. This is a 
150          constant across all features. This corresponds to the last bin, and 
151          it is always equal to ``n_bins - 1``. Note that if ``n_bins_non_missing_`` 
152          is less than ``n_bins - 1`` for a given feature, then there are 
153          empty (and unused) bins. 
154      """
155  
156      def __init__(
157          self,
158          n_bins=256,
159          subsample=int(2e5),
160          is_categorical=None,
161          known_categories=None,
162          random_state=None,
163          n_threads=None,
164      ):
165          self.n_bins = n_bins
166          self.subsample = subsample
167          self.is_categorical = is_categorical
168          self.known_categories = known_categories
169          self.random_state = random_state
170          self.n_threads = n_threads
171  
172      def fit(self, X, y=None):
173          """Fit data X by computing the binning thresholds. 
174   
175          The last bin is reserved for missing values, whether missing values 
176          are present in the data or not. 
177   
178          Parameters 
179          ---------- 
180          X : array-like of shape (n_samples, n_features) 
181              The data to bin. 
182          y: None 
183              Ignored. 
184   
185          Returns 
186          ------- 
187          self : object 
188          """
189          if not (3 <= self.n_bins <= 256):
190              # min is 3: at least 2 distinct bins and a missing values bin
191              raise ValueError(
192                  "n_bins={} should be no smaller than 3 and no larger than 256.".format(
193                      self.n_bins
194                  )
195              )
196  
197          X = check_array(X, dtype=[X_DTYPE], ensure_all_finite=False)
198          max_bins = self.n_bins - 1
199  
200          rng = check_random_state(self.random_state)
201          if self.subsample is not None and X.shape[0] > self.subsample:
202              subset = rng.choice(X.shape[0], self.subsample, replace=False)
203              X = X.take(subset, axis=0)
204  
205          if self.is_categorical is None:
206              self.is_categorical_ = np.zeros(X.shape[1], dtype=np.uint8)
207          else:
208              self.is_categorical_ = np.asarray(self.is_categorical, dtype=np.uint8)
209  
210          n_features = X.shape[1]
211          known_categories = self.known_categories
212          if known_categories is None:
213              known_categories = [None] * n_features
214  
215          # validate is_categorical and known_categories parameters
216          for f_idx in range(n_features):
217              is_categorical = self.is_categorical_[f_idx]
218              known_cats = known_categories[f_idx]
219              if is_categorical and known_cats is None:
220                  raise ValueError(
221                      f"Known categories for feature {f_idx} must be provided."
222                  )
223              if not is_categorical and known_cats is not None:
224                  raise ValueError(
225                      f"Feature {f_idx} isn't marked as a categorical feature, "
226                      "but categories were passed."
227                  )
228  
229          self.missing_values_bin_idx_ = self.n_bins - 1
230  
231          self.bin_thresholds_ = [None] * n_features
232          n_bins_non_missing = [None] * n_features
233  
234          non_cat_thresholds = Parallel(n_jobs=self.n_threads, backend="threading")(
235              delayed(_find_binning_thresholds)(X[:, f_idx], max_bins)
236              for f_idx in range(n_features)
237              if not self.is_categorical_[f_idx]
238          )
239  
240          non_cat_idx = 0
241          for f_idx in range(n_features):
242              if self.is_categorical_[f_idx]:
243                  # Since categories are assumed to be encoded in
244                  # [0, n_cats] and since n_cats <= max_bins,
245                  # the thresholds *are* the unique categorical values. This will
246                  # lead to the correct mapping in transform()
247                  thresholds = known_categories[f_idx]
248                  n_bins_non_missing[f_idx] = thresholds.shape[0]
249                  self.bin_thresholds_[f_idx] = thresholds
250              else:
251                  self.bin_thresholds_[f_idx] = non_cat_thresholds[non_cat_idx]
252                  n_bins_non_missing[f_idx] = self.bin_thresholds_[f_idx].shape[0] + 1
253                  non_cat_idx += 1
254  
255          self.n_bins_non_missing_ = np.array(n_bins_non_missing, dtype=np.uint32)
256          return self
257  
258      def transform(self, X):
259          """Bin data X. 
260   
261          Missing values will be mapped to the last bin. 
262   
263          For categorical features, the mapping will be incorrect for unknown 
264          categories. Since the BinMapper is given known_categories of the 
265          entire training data (i.e. before the call to train_test_split() in 
266          case of early-stopping), this never happens. 
267   
268          Parameters 
269          ---------- 
270          X : array-like of shape (n_samples, n_features) 
271              The data to bin. 
272   
273          Returns 
274          ------- 
275          X_binned : array-like of shape (n_samples, n_features) 
276              The binned data (fortran-aligned). 
277          """
278          X = check_array(X, dtype=[X_DTYPE], ensure_all_finite=False)
279          check_is_fitted(self)
280          if X.shape[1] != self.n_bins_non_missing_.shape[0]:
281              raise ValueError(
282                  "This estimator was fitted with {} features but {} got passed "
283                  "to transform()".format(self.n_bins_non_missing_.shape[0], X.shape[1])
284              )
285  
286          n_threads = _openmp_effective_n_threads(self.n_threads)
287          binned = np.zeros_like(X, dtype=X_BINNED_DTYPE, order="F")
288          _map_to_bins(
289              X,
290              self.bin_thresholds_,
291              self.is_categorical_,
292              self.missing_values_bin_idx_,
293              n_threads,
294              binned,
295          )
296          return binned

298      def make_known_categories_bitsets(self):
299          """Create bitsets of known categories. 
300   
301          Returns 
302          ------- 
303          - known_cat_bitsets : ndarray of shape (n_categorical_features, 8) 
304              Array of bitsets of known categories, for each categorical feature. 
305          - f_idx_map : ndarray of shape (n_features,) 
306              Map from original feature index to the corresponding index in the 
307              known_cat_bitsets array. 
308          """
309  
310          categorical_features_indices = np.flatnonzero(self.is_categorical_)
311  
312          n_features = self.is_categorical_.size
313          n_categorical_features = categorical_features_indices.size
314  
315          f_idx_map = np.zeros(n_features, dtype=np.uint32)
316          f_idx_map[categorical_features_indices] = np.arange(
317              n_categorical_features, dtype=np.uint32
318          )
319  
320          known_categories = self.bin_thresholds_
321  
322          known_cat_bitsets = np.zeros(
323              (n_categorical_features, 8), dtype=X_BITSET_INNER_DTYPE
324          )
325  
326          # TODO: complexity is O(n_categorical_features * 255). Maybe this is
327          # worth cythonizing
328          for mapped_f_idx, f_idx in enumerate(categorical_features_indices):
329              for raw_cat_val in known_categories[f_idx]:
330                  set_bitset_memoryview(known_cat_bitsets[mapped_f_idx], raw_cat_val)
331  
332          return known_cat_bitsets, f_idx_map
