36       def test_against_reference_values(self, v, x, r):
37           """The reference values are computed using mpmath as follows. 
38    
39           from mpmath import mp 
40           mp.dps = 100 
41    
42           def iv_ratio_mp(v, x): 
43               return mp.besseli(v, x) / mp.besseli(v - 1, x) 
44    
45           def _sample(n, *, v): 
46               '''Return n positive real numbers x such that iv_ratio(v, x) are 
47               roughly evenly spaced over (0, 1).  The formula is taken from [1]. 
48    
49               [1] Banerjee A., Dhillon, I. S., Ghosh, J., Sra, S. (2005). 
50                   "Clustering on the Unit Hypersphere using von Mises-Fisher 
51                   Distributions."  Journal of Machine Learning Research, 
52                   6(46):1345-1382. 
53               ''' 
54               r = np.arange(1, n+1) / (n+1) 
55               return r * (2*v-r*r) / (1-r*r) 
56    
57           for v in (0.5, 1, 2.34, 56.789): 
58               xs = _sample(5, v=v) 
59               for x in xs: 
60                   print(f"({v}, {x}, {float(iv_ratio_mp(v,x))}),") 
61           """
62           assert_allclose(iv_ratio(v, x), r, rtol=4e-16, atol=0)

82       @pytest.mark.parametrize('v', [0.5, 1, np.finfo(float).max, np.inf])
83       def test_zero_x(self, v):
84           """If x is +/-0.0, return x to ensure iv_ratio is an odd function."""
85           assert_equal(iv_ratio(v, 0.0), 0.0)
86           assert_equal(iv_ratio(v, -0.0), -0.0)

88       @pytest.mark.parametrize('v,x', [
89           (1, np.finfo(float).smallest_normal),
90           (1, np.finfo(float).smallest_subnormal),
91           (1, np.finfo(float).smallest_subnormal*2),
92           (1e20, 123),
93           (np.finfo(float).max, 1),
94           (np.finfo(float).max, np.sqrt(np.finfo(float).max)),
95       ])
96       def test_tiny_x(self, v, x):
97           """If x is much less than v, the bounds 
98    
99                       x                                 x 
100          --------------------------- <= R <= ----------------------- 
101          v-0.5+sqrt(x**2+(v+0.5)**2)         v-1+sqrt(x**2+(v+1)**2) 
102   
103          collapses to R ~= x/2v.  Test against this asymptotic expression. 
104          """
105          assert_equal(iv_ratio(v, x), (0.5*x)/v)
106  
107      @pytest.mark.parametrize('v,x', [
108          (1, 1e16),
109          (1e20, 1e40),
110          (np.sqrt(np.finfo(float).max), np.finfo(float).max),
111      ])
112      def test_huge_x(self, v, x):
113          """If x is much greater than v, the bounds 
114   
115                      x                                 x 
116          --------------------------- <= R <= --------------------------- 
117          v-0.5+sqrt(x**2+(v+0.5)**2)         v-0.5+sqrt(x**2+(v-0.5)**2) 
118   
119          collapses to R ~= 1.  Test against this asymptotic expression. 
120          """
121          assert_equal(iv_ratio(v, x), 1.0)
122  
123      @pytest.mark.parametrize('v,x', [
124          (np.finfo(float).max, np.finfo(float).max),
125          (np.finfo(float).max / 3, np.finfo(float).max),
126          (np.finfo(float).max, np.finfo(float).max / 3),
127      ])
128      def test_huge_v_x(self, v, x):
129          """If both x and v are very large, the bounds 
130   
131                      x                                 x 
132          --------------------------- <= R <= ----------------------- 
133          v-0.5+sqrt(x**2+(v+0.5)**2)         v-1+sqrt(x**2+(v+1)**2) 
134   
135          collapses to R ~= x/(v+sqrt(x**2+v**2).  Test against this asymptotic 
136          expression, and in particular that no numerical overflow occurs during 
137          intermediate calculations. 
138          """
139          t = x / v
140          expected = t / (1 + np.hypot(1, t))
141          assert_allclose(iv_ratio(v, x), expected, rtol=4e-16, atol=0)
