_scipy_spectral_test_shim.py 
1    """Helpers to utilize existing stft / istft tests for testing `ShortTimeFFT`. 
2     
3    This module provides the functions stft_compare() and istft_compare(), which, 
4    compares the output between the existing (i)stft() and the shortTimeFFT based 
5    _(i)stft_wrapper() implementations in this module. 
6     
7    For testing add the following imports to the file ``tests/test_spectral.py``:: 
8     
9        from ._scipy_spectral_test_shim import stft_compare as stft 
10       from ._scipy_spectral_test_shim import istft_compare as istft 
11    
12   and remove the existing imports of stft and istft. 
13    
14   The idea of these wrappers is not to provide a backward-compatible interface 
15   but to demonstrate that the ShortTimeFFT implementation is at least as capable 
16   as the existing one and delivers comparable results. Furthermore, the 
17   wrappers highlight the different philosophies of the implementations, 
18   especially in the border handling. 
19   """
20   import platform
21   from typing import cast, Literal

24   from numpy.testing import assert_allclose

27   from scipy.signal import csd, get_window, stft, istft

34   def _stft_wrapper(x, fs=1.0, window='hann', nperseg=256, noverlap=None,
35                     nfft=None, detrend=False, return_onesided=True,
36                     boundary='zeros', padded=True, axis=-1, scaling='spectrum'):
37       """Wrapper for the SciPy `stft()` function based on `ShortTimeFFT` for 
38       unit testing. 
39    
40       Handling the boundary and padding is where `ShortTimeFFT` and `stft()` 
41       differ in behavior. Parts of `_spectral_helper()` were copied to mimic 
42       the` stft()` behavior. 
43    
44       This function is meant to be solely used by `stft_compare()`. 
45       """
46       if scaling not in ('psd', 'spectrum'):  # same errors as in original stft:
47           raise ValueError(f"Parameter {scaling=} not in ['spectrum', 'psd']!")
48   
49       # The following lines are taken from the original _spectral_helper():
50       boundary_funcs = {'even': even_ext,
51                         'odd': odd_ext,
52                         'constant': const_ext,
53                         'zeros': zero_ext,
54                         None: None}
55   
56       if boundary not in boundary_funcs:
57           raise ValueError(f"Unknown boundary option '{boundary}', must be one" +
58                            f" of: {list(boundary_funcs.keys())}")
59       if x.size == 0:
60           return np.empty(x.shape), np.empty(x.shape), np.empty(x.shape)
61   
62       if nperseg is not None:  # if specified by user
63           nperseg = int(nperseg)
64           if nperseg < 1:
65               raise ValueError('nperseg must be a positive integer')
66   
67       # parse window; if array like, then set nperseg = win.shape
68       win, nperseg = _triage_segments(window, nperseg,
69                                       input_length=x.shape[axis])
70   
71       if nfft is None:
72           nfft = nperseg
73       elif nfft < nperseg:
74           raise ValueError('nfft must be greater than or equal to nperseg.')
75       else:
76           nfft = int(nfft)
77   
78       if noverlap is None:
79           noverlap = nperseg//2
80       else:
81           noverlap = int(noverlap)
82       if noverlap >= nperseg:
83           raise ValueError('noverlap must be less than nperseg.')
84       nstep = nperseg - noverlap
85       n = x.shape[axis]
86   
87       # Padding occurs after boundary extension, so that the extended signal ends
88       # in zeros, instead of introducing an impulse at the end.
89       # I.e. if x = [..., 3, 2]
90       # extend then pad -> [..., 3, 2, 2, 3, 0, 0, 0]
91       # pad then extend -> [..., 3, 2, 0, 0, 0, 2, 3]
92   
93       if boundary is not None:
94           ext_func = boundary_funcs[boundary]
95           # Extend by nperseg//2 in front and back:
96           x = ext_func(x, nperseg//2, axis=axis)
97   
98       if padded:
99           # Pad to integer number of windowed segments
100          # I.e make x.shape[-1] = nperseg + (nseg-1)*nstep, with integer nseg
101          x = np.moveaxis(x, axis, -1)
102  
103          # This is an edge case where shortTimeFFT returns one more time slice
104          # than the Scipy stft() shorten to remove last time slice:
105          if n % 2 == 1 and nperseg % 2 == 1 and noverlap % 2 == 1:
106              x = x[..., :axis - 1]
107  
108          nadd = (-(x.shape[-1]-nperseg) % nstep) % nperseg
109          zeros_shape = list(x.shape[:-1]) + [nadd]
110          x = np.concatenate((x, np.zeros(zeros_shape)), axis=-1)
111          x = np.moveaxis(x, -1, axis)
112  
113      #  ... end original _spectral_helper() code.
114      scale_to = {'spectrum': 'magnitude', 'psd': 'psd'}[scaling]
115  
116      if np.iscomplexobj(x) and return_onesided:
117          return_onesided = False
118      # using cast() to make mypy happy:
119      fft_mode = cast(FFT_MODE_TYPE, 'onesided' if return_onesided else 'twosided')
120  
121      ST = ShortTimeFFT(win, nstep, fs, fft_mode=fft_mode, mfft=nfft,
122                        scale_to=scale_to, phase_shift=None)
123  
124      k_off = nperseg // 2
125      p0 = 0  # ST.lower_border_end[1] + 1
126      nn = x.shape[axis] if padded else n+k_off+1
127      p1 = ST.upper_border_begin(nn)[1]  # ST.p_max(n) + 1
128  
129      # This is bad hack to pass the test test_roundtrip_boundary_extension():
130      if padded is True and nperseg - noverlap == 1:
131          p1 -= nperseg // 2 - 1  # the reasoning behind this is not clear to me
132  
133      detr = None if detrend is False else detrend
134      Sxx = ST.stft_detrend(x, detr, p0, p1, k_offset=k_off, axis=axis)
135      t = ST.t(nn, 0, p1 - p0, k_offset=0 if boundary is not None else k_off)
136      if x.dtype in (np.float32, np.complex64):
137          Sxx = Sxx.astype(np.complex64)
138  
139      # workaround for test_average_all_segments() - seems to be buggy behavior:
140      if boundary is None and padded is False:
141          t, Sxx = t[1:-1], Sxx[..., :-2]
142          t -= k_off / fs
143  
144      return ST.f, t, Sxx

147  def _istft_wrapper(Zxx, fs=1.0, window='hann', nperseg=None, noverlap=None,
148                     nfft=None, input_onesided=True, boundary=True, time_axis=-1,
149                     freq_axis=-2, scaling='spectrum') -> \
150          tuple[np.ndarray, np.ndarray, tuple[int, int]]:
151      """Wrapper for the SciPy `istft()` function based on `ShortTimeFFT` for 
152          unit testing. 
153   
154      Note that only option handling is implemented as far as to handle the unit 
155      tests. E.g., the case ``nperseg=None`` is not handled. 
156   
157      This function is meant to be solely used by `istft_compare()`. 
158      """
159      # *** Lines are taken from _spectral_py.istft() ***:
160      if Zxx.ndim < 2:
161          raise ValueError('Input stft must be at least 2d!')
162  
163      if freq_axis == time_axis:
164          raise ValueError('Must specify differing time and frequency axes!')
165  
166      nseg = Zxx.shape[time_axis]
167  
168      if input_onesided:
169          # Assume even segment length
170          n_default = 2*(Zxx.shape[freq_axis] - 1)
171      else:
172          n_default = Zxx.shape[freq_axis]
173  
174      # Check windowing parameters
175      if nperseg is None:
176          nperseg = n_default
177      else:
178          nperseg = int(nperseg)
179          if nperseg < 1:
180              raise ValueError('nperseg must be a positive integer')
181  
182      if nfft is None:
183          if input_onesided and (nperseg == n_default + 1):
184              # Odd nperseg, no FFT padding
185              nfft = nperseg
186          else:
187              nfft = n_default
188      elif nfft < nperseg:
189          raise ValueError('nfft must be greater than or equal to nperseg.')
190      else:
191          nfft = int(nfft)
192  
193      if noverlap is None:
194          noverlap = nperseg//2
195      else:
196          noverlap = int(noverlap)
197      if noverlap >= nperseg:
198          raise ValueError('noverlap must be less than nperseg.')
199      nstep = nperseg - noverlap
200  
201      # Get window as array
202      if isinstance(window, str) or type(window) is tuple:
203          win = get_window(window, nperseg)
204      else:
205          win = np.asarray(window)
206          if len(win.shape) != 1:
207              raise ValueError('window must be 1-D')
208          if win.shape[0] != nperseg:
209              raise ValueError(f'window must have length of {nperseg}')
210  
211      outputlength = nperseg + (nseg-1)*nstep
212      # *** End block of: Taken from _spectral_py.istft() ***
213  
214      # Using cast() to make mypy happy:
215      fft_mode = cast(FFT_MODE_TYPE, 'onesided' if input_onesided else 'twosided')
216      scale_to = cast(Literal['magnitude', 'psd'],
217                      {'spectrum': 'magnitude', 'psd': 'psd'}[scaling])
218  
219      ST = ShortTimeFFT(win, nstep, fs, fft_mode=fft_mode, mfft=nfft,
220                        scale_to=scale_to, phase_shift=None)
221  
222      if boundary:
223          j = nperseg if nperseg % 2 == 0 else nperseg - 1
224          k0 = ST.k_min + nperseg // 2
225          k1 = outputlength - j + k0
226      else:
227          raise NotImplementedError("boundary=False does not make sense with" +
228                                    "ShortTimeFFT.istft()!")
229  
230      x = ST.istft(Zxx, k0=k0, k1=k1, f_axis=freq_axis, t_axis=time_axis)
231      t = np.arange(k1 - k0) * ST.T
232      k_hi = ST.upper_border_begin(k1 - k0)[0]
233      # using cast() to make mypy happy:
234      return t, x, (ST.lower_border_end[0], k_hi)

237  def _csd_wrapper(x, y, fs=1.0, window='hann', nperseg=None, noverlap=None,
238                   nfft=None, detrend='constant', return_onesided=True,
239                   scaling='density', axis=-1, average='mean'):
240      """Wrapper for the `csd()` function based on `ShortTimeFFT` for 
241          unit testing. 
242      """
243      freqs, _, Pxy = _csd_test_shim(x, y, fs, window, nperseg, noverlap, nfft,
244                                     detrend, return_onesided, scaling, axis)
245  
246      # The following code is taken from csd():
247      if len(Pxy.shape) >= 2 and Pxy.size > 0:
248          if Pxy.shape[-1] > 1:
249              if average == 'median':
250                  # np.median must be passed real arrays for the desired result
251                  bias = _median_bias(Pxy.shape[-1])
252                  if np.iscomplexobj(Pxy):
253                      Pxy = (np.median(np.real(Pxy), axis=-1)
254                             + 1j * np.median(np.imag(Pxy), axis=-1))
255                  else:
256                      Pxy = np.median(Pxy, axis=-1)
257                  Pxy /= bias
258              elif average == 'mean':
259                  Pxy = Pxy.mean(axis=-1)
260              else:
261                  raise ValueError(f'average must be "median" or "mean", got {average}')
262          else:
263              Pxy = np.reshape(Pxy, Pxy.shape[:-1])
264  
265      return freqs, Pxy

268  def _csd_test_shim(x, y, fs=1.0, window='hann', nperseg=None, noverlap=None,
269                     nfft=None, detrend='constant', return_onesided=True,
270                     scaling='density', axis=-1):
271      """Compare output of  _spectral_helper() and ShortTimeFFT, more 
272      precisely _spect_helper_csd() for used in csd_wrapper(). 
273   
274     The motivation of this function is to test if the ShortTimeFFT-based 
275     wrapper `_spect_helper_csd()` returns the same values as `_spectral_helper`. 
276     This function should only be usd by csd() in (unit) testing. 
277     """
278      freqs, t, Pxy = _spectral_helper(x, y, fs, window, nperseg, noverlap, nfft,
279                                       detrend, return_onesided, scaling, axis,
280                                       mode='psd')
281      freqs1, Pxy1 = _spect_helper_csd(x, y, fs, window, nperseg, noverlap, nfft,
282                                       detrend, return_onesided, scaling, axis)
283  
284      np.testing.assert_allclose(freqs1, freqs)
285      amax_Pxy = max(np.abs(Pxy).max(), 1) if Pxy.size else 1
286      atol = np.finfo(Pxy.dtype).resolution * amax_Pxy  # needed for large Pxy
287      # for c_ in range(Pxy.shape[-1]):
288      #    np.testing.assert_allclose(Pxy1[:, c_], Pxy[:, c_], atol=atol)
289      np.testing.assert_allclose(Pxy1, Pxy, atol=atol)
290      return freqs, t, Pxy

293  def _spect_helper_csd(x, y, fs=1.0, window='hann', nperseg=None, noverlap=None,
294                        nfft=None, detrend='constant', return_onesided=True,
295                        scaling='density', axis=-1):
296      """Wrapper for replacing _spectral_helper() by using the ShortTimeFFT 
297        for use by csd(). 
298   
299      This function should be only used by _csd_test_shim() and is only useful 
300      for testing the ShortTimeFFT implementation. 
301      """
302  
303      # The following lines are taken from the original _spectral_helper():
304      same_data = y is x
305      axis = int(axis)
306  
307      # Ensure we have np.arrays, get outdtype
308      x = np.asarray(x)
309      if not same_data:
310          y = np.asarray(y)
311      #     outdtype = np.result_type(x, y, np.complex64)
312      # else:
313      #     outdtype = np.result_type(x, np.complex64)
314  
315      if not same_data:
316          # Check if we can broadcast the outer axes together
317          xouter = list(x.shape)
318          youter = list(y.shape)
319          xouter.pop(axis)
320          youter.pop(axis)
321          try:
322              outershape = np.broadcast(np.empty(xouter), np.empty(youter)).shape
323          except ValueError as e:
324              raise ValueError('x and y cannot be broadcast together.') from e
325  
326      if same_data:
327          if x.size == 0:
328              return np.empty(x.shape), np.empty(x.shape)
329      else:
330          if x.size == 0 or y.size == 0:
331              outshape = outershape + (min([x.shape[axis], y.shape[axis]]),)
332              emptyout = np.moveaxis(np.empty(outshape), -1, axis)
333              return emptyout, emptyout
334  
335      if nperseg is not None:  # if specified by user
336          nperseg = int(nperseg)
337          if nperseg < 1:
338              raise ValueError('nperseg must be a positive integer')
339  
340      # parse window; if array like, then set nperseg = win.shape
341      n = x.shape[axis] if same_data else max(x.shape[axis], y.shape[axis])
342      win, nperseg = _triage_segments(window, nperseg, input_length=n)
343  
344      if nfft is None:
345          nfft = nperseg
346      elif nfft < nperseg:
347          raise ValueError('nfft must be greater than or equal to nperseg.')
348      else:
349          nfft = int(nfft)
350  
351      if noverlap is None:
352          noverlap = nperseg // 2
353      else:
354          noverlap = int(noverlap)
355      if noverlap >= nperseg:
356          raise ValueError('noverlap must be less than nperseg.')
357      nstep = nperseg - noverlap
358  
359      if np.iscomplexobj(x) and return_onesided:
360          return_onesided = False
361  
362      # using cast() to make mypy happy:
363      fft_mode = cast(FFT_MODE_TYPE, 'onesided' if return_onesided
364                      else 'twosided')
365      scale = {'spectrum': 'magnitude', 'density': 'psd'}[scaling]
366      SFT = ShortTimeFFT(win, nstep, fs, fft_mode=fft_mode, mfft=nfft,
367                         scale_to=scale, phase_shift=None)
368  
369      # _spectral_helper() calculates X.conj()*Y instead of X*Y.conj():
370      Pxy = SFT.spectrogram(y, x, detr=None if detrend is False else detrend,
371                            p0=0, p1=(n-noverlap)//SFT.hop, k_offset=nperseg//2,
372                            axis=axis).conj()
373      # Note:
374      # 'onesided2X' scaling of ShortTimeFFT conflicts with the
375      # scaling='spectrum' parameter, since it doubles the squared magnitude,
376      # which in the view of the ShortTimeFFT implementation does not make sense.
377      # Hence, the doubling of the square is implemented here:
378      if return_onesided:
379          f_axis = Pxy.ndim - 1 + axis if axis < 0 else axis
380          Pxy = np.moveaxis(Pxy, f_axis, -1)
381          Pxy[..., 1:-1 if SFT.mfft % 2 == 0 else None] *= 2
382          Pxy = np.moveaxis(Pxy, -1, f_axis)
383  
384      return SFT.f, Pxy

387  def stft_compare(x, fs=1.0, window='hann', nperseg=256, noverlap=None,
388                   nfft=None, detrend=False, return_onesided=True,
389                   boundary='zeros', padded=True, axis=-1, scaling='spectrum'):
390      """Assert that the results from the existing `stft()` and `_stft_wrapper()` 
391      are close to each other. 
392   
393      For comparing the STFT values an absolute tolerance of the floating point 
394      resolution was added to circumvent problems with the following tests: 
395      * For float32 the tolerances are much higher in 
396        TestSTFT.test_roundtrip_float32()). 
397      * The TestSTFT.test_roundtrip_scaling() has a high relative deviation. 
398        Interestingly this did not appear in Scipy 1.9.1 but only in the current 
399        development version. 
400      """
401      kw = dict(x=x, fs=fs, window=window, nperseg=nperseg, noverlap=noverlap,
402                nfft=nfft, detrend=detrend, return_onesided=return_onesided,
403                boundary=boundary, padded=padded, axis=axis, scaling=scaling)
404      f, t, Zxx = stft(**kw)
405      f_wrapper, t_wrapper, Zxx_wrapper = _stft_wrapper(**kw)
406  
407      e_msg_part = " of `stft_wrapper()` differ from `stft()`."
408      assert_allclose(f_wrapper, f, err_msg=f"Frequencies {e_msg_part}")
409      assert_allclose(t_wrapper, t, err_msg=f"Time slices {e_msg_part}")
410  
411      # Adapted tolerances to account for:
412      atol = np.finfo(Zxx.dtype).resolution * 2
413      assert_allclose(Zxx_wrapper, Zxx, atol=atol,
414                      err_msg=f"STFT values {e_msg_part}")
415      return f, t, Zxx

418  def istft_compare(Zxx, fs=1.0, window='hann', nperseg=None, noverlap=None,
419                    nfft=None, input_onesided=True, boundary=True, time_axis=-1,
420                    freq_axis=-2, scaling='spectrum'):
421      """Assert that the results from the existing `istft()` and 
422      `_istft_wrapper()` are close to each other. 
423   
424      Quirks: 
425      * If ``boundary=False`` the comparison is skipped, since it does not 
426        make sense with ShortTimeFFT.istft(). Only used in test 
427        TestSTFT.test_roundtrip_boundary_extension(). 
428      * If ShortTimeFFT.istft() decides the STFT is not invertible, the 
429        comparison is skipped, since istft() only emits a warning and does not 
430        return a correct result. Only used in 
431        ShortTimeFFT.test_roundtrip_not_nola(). 
432      * For comparing the signals an absolute tolerance of the floating point 
433        resolution was added to account for the low accuracy of float32 (Occurs 
434        only in TestSTFT.test_roundtrip_float32()). 
435      """
436      kw = dict(Zxx=Zxx, fs=fs, window=window, nperseg=nperseg,
437                noverlap=noverlap, nfft=nfft, input_onesided=input_onesided,
438                boundary=boundary, time_axis=time_axis, freq_axis=freq_axis,
439                scaling=scaling)
440  
441      t, x = istft(**kw)
442      if not boundary:  # skip test_roundtrip_boundary_extension():
443          return t, x  # _istft_wrapper does() not implement this case
444      try:  # if inversion fails, istft() only emits a warning:
445          t_wrapper, x_wrapper, (k_lo, k_hi) = _istft_wrapper(**kw)
446      except ValueError as v:  # Do nothing if inversion fails:
447          if v.args[0] == "Short-time Fourier Transform not invertible!":
448              return t, x
449          raise v
450  
451      e_msg_part = " of `istft_wrapper()` differ from `istft()`"
452      assert_allclose(t, t_wrapper, err_msg=f"Sample times {e_msg_part}")
453  
454      # Adapted tolerances to account for resolution loss:
455      atol = np.finfo(x.dtype).resolution*2  # instead of default atol = 0
456      rtol = 1e-7  # default for np.allclose()
457  
458      # Relax atol on 32-Bit platforms a bit to pass CI tests.
459      #  - Not clear why there are discrepancies (in the FFT maybe?)
460      #  - Not sure what changed on 'i686' since earlier on those test passed
461      if x.dtype == np.float32 and platform.machine() == 'i686':
462          # float32 gets only used by TestSTFT.test_roundtrip_float32() so
463          # we are using the tolerances from there to circumvent CI problems
464          atol, rtol = 1e-4, 1e-5
465      elif platform.machine() in ('aarch64', 'i386', 'i686'):
466          atol = max(atol, 1e-12)  # 2e-15 seems too tight for 32-Bit platforms
467  
468      assert_allclose(x_wrapper[k_lo:k_hi], x[k_lo:k_hi], atol=atol, rtol=rtol,
469                      err_msg=f"Signal values {e_msg_part}")
470      return t, x

473  def csd_compare(x, y, fs=1.0, window='hann', nperseg=None, noverlap=None,
474                  nfft=None, detrend='constant', return_onesided=True,
475                  scaling='density', axis=-1, average='mean'):
476      """Assert that the results from the existing `csd()` and `_csd_wrapper()` 
477      are close to each other. """
478      kw = dict(x=x, y=y, fs=fs, window=window, nperseg=nperseg,
479                noverlap=noverlap, nfft=nfft, detrend=detrend,
480                return_onesided=return_onesided, scaling=scaling, axis=axis,
481                average=average)
482      freqs0, Pxy0 = csd(**kw)
483      freqs1, Pxy1 = _csd_wrapper(**kw)
484  
485      assert_allclose(freqs1, freqs0)
486      assert_allclose(Pxy1, Pxy0)
487      assert_allclose(freqs1, freqs0)
488      return freqs0, Pxy0
