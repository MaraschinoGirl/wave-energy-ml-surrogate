36   def make_new_dset(parent, shape=None, dtype=None, data=None, name=None,
37                     chunks=None, compression=None, shuffle=None,
38                     fletcher32=None, maxshape=None, compression_opts=None,
39                     fillvalue=None, scaleoffset=None, track_times=False,
40                     external=None, track_order=None, dcpl=None, dapl=None,
41                     efile_prefix=None, virtual_prefix=None, allow_unknown_filter=False,
42                     rdcc_nslots=None, rdcc_nbytes=None, rdcc_w0=None, *,
43                     fill_time=None):
44       """ Return a new low-level dataset identifier """
45   
46       # Convert data to a C-contiguous ndarray
47       if data is not None and not isinstance(data, Empty):
48           data = array_for_new_object(data, specified_dtype=dtype)
49   
50       # Validate shape
51       if shape is None:
52           if data is None:
53               if dtype is None:
54                   raise TypeError("One of data, shape or dtype must be specified")
55               data = Empty(dtype)
56           shape = data.shape
57       else:
58           shape = (shape,) if isinstance(shape, int) else tuple(shape)
59           if data is not None and (product(shape) != product(data.shape)):
60               raise ValueError("Shape tuple is incompatible with data")
61   
62       if isinstance(maxshape, int):
63           maxshape = (maxshape,)
64       tmp_shape = maxshape if maxshape is not None else shape
65   
66       # Validate chunk shape
67       if isinstance(chunks, int) and not isinstance(chunks, bool):
68           chunks = (chunks,)
69       if isinstance(chunks, tuple) and any(
70           chunk > dim for dim, chunk in zip(tmp_shape, chunks) if dim is not None
71       ):
72           errmsg = "Chunk shape must not be greater than data shape in any dimension. "\
73                    "{} is not compatible with {}".format(chunks, shape)
74           raise ValueError(errmsg)
75   
76       if isinstance(dtype, Datatype):
77           # Named types are used as-is
78           tid = dtype.id
79           dtype = tid.dtype  # Following code needs this
80       else:
81           # Validate dtype
82           if dtype is None and data is None:
83               dtype = numpy.dtype("=f4")
84           elif dtype is None and data is not None:
85               dtype = data.dtype
86           else:
87               dtype = numpy.dtype(dtype)
88           tid = h5t.py_create(dtype, logical=1)
89   
90       # Legacy
91       if any((compression, shuffle, fletcher32, maxshape, scaleoffset)) and chunks is False:
92           raise ValueError("Chunked format required for given storage options")
93   
94       # Legacy
95       if compression is True:
96           if compression_opts is None:
97               compression_opts = 4
98           compression = 'gzip'
99   
100      # Legacy
101      if compression in _LEGACY_GZIP_COMPRESSION_VALS:
102          if compression_opts is not None:
103              raise TypeError("Conflict in compression options")
104          compression_opts = compression
105          compression = 'gzip'
106      dcpl = filters.fill_dcpl(
107          dcpl or h5p.create(h5p.DATASET_CREATE), shape, dtype,
108          chunks, compression, compression_opts, shuffle, fletcher32,
109          maxshape, scaleoffset, external, allow_unknown_filter,
110          fill_time=fill_time)
111  
112      # Check that compression roundtrips correctly if it was specified
113      if compression is not None:
114          if isinstance(compression, filters.FilterRefBase):
115              compression = compression.filter_id
116          if isinstance(compression, int):
117              compression = filters.get_filter_name(compression)
118          if compression not in filters.get_filters(dcpl):
119              raise ValueError(f'compression {compression!r} not in filters {filters.get_filters(dcpl)!r}')
120  
121      if fillvalue is not None:
122          # prepare string-type dtypes for fillvalue
123          string_info = h5t.check_string_dtype(dtype)
124          if string_info is not None:
125              # fake vlen dtype for fixed len string fillvalue
126              # to not trigger unwanted encoding
127              dtype = h5t.string_dtype(string_info.encoding)
128              fillvalue = numpy.array(fillvalue, dtype=dtype)
129          else:
130              fillvalue = numpy.array(fillvalue)
131          dcpl.set_fill_value(fillvalue)
132  
133      if track_times is None:
134          # In case someone explicitly passes None for the default
135          track_times = False
136      if track_times in (True, False):
137          dcpl.set_obj_track_times(track_times)
138      else:
139          raise TypeError("track_times must be either True or False")
140      if track_order is True:
141          dcpl.set_attr_creation_order(
142              h5p.CRT_ORDER_TRACKED | h5p.CRT_ORDER_INDEXED)
143      elif track_order is False:
144          dcpl.set_attr_creation_order(0)
145      elif track_order is not None:
146          raise TypeError("track_order must be either True or False")
147  
148      if maxshape is not None:
149          maxshape = tuple(m if m is not None else h5s.UNLIMITED for m in maxshape)
150  
151      if any([efile_prefix, virtual_prefix, rdcc_nbytes, rdcc_nslots, rdcc_w0]):
152          dapl = dapl or h5p.create(h5p.DATASET_ACCESS)
153  
154      if efile_prefix is not None:
155          dapl.set_efile_prefix(efile_prefix)
156  
157      if virtual_prefix is not None:
158          dapl.set_virtual_prefix(virtual_prefix)
159  
160      if rdcc_nbytes or rdcc_nslots or rdcc_w0:
161          cache_settings = list(dapl.get_chunk_cache())
162          if rdcc_nslots is not None:
163              cache_settings[0] = rdcc_nslots
164          if rdcc_nbytes is not None:
165              cache_settings[1] = rdcc_nbytes
166          if rdcc_w0 is not None:
167              cache_settings[2] = rdcc_w0
168          dapl.set_chunk_cache(*cache_settings)
169  
170      if isinstance(data, Empty):
171          sid = h5s.create(h5s.NULL)
172      else:
173          sid = h5s.create_simple(shape, maxshape)
174  
175      dset_id = h5d.create(parent.id, name, tid, sid, dcpl=dcpl, dapl=dapl)
176  
177      if (data is not None) and (not isinstance(data, Empty)):
178          dset_id.write(h5s.ALL, h5s.ALL, data)
179  
180      return dset_id
181

183  def open_dset(parent, name, dapl=None, efile_prefix=None, virtual_prefix=None,
184                rdcc_nslots=None, rdcc_nbytes=None, rdcc_w0=None, **kwds):
185      """ Return an existing low-level dataset identifier """
186  
187      if any([efile_prefix, virtual_prefix, rdcc_nbytes, rdcc_nslots, rdcc_w0]):
188          dapl = dapl or h5p.create(h5p.DATASET_ACCESS)
189  
190      if efile_prefix is not None:
191          dapl.set_efile_prefix(efile_prefix)
192  
193      if virtual_prefix is not None:
194          dapl.set_virtual_prefix(virtual_prefix)
195  
196      if rdcc_nbytes or rdcc_nslots or rdcc_w0:
197          cache_settings = list(dapl.get_chunk_cache())
198          if rdcc_nslots is not None:
199              cache_settings[0] = rdcc_nslots
200          if rdcc_nbytes is not None:
201              cache_settings[1] = rdcc_nbytes
202          if rdcc_w0 is not None:
203              cache_settings[2] = rdcc_w0
204          dapl.set_chunk_cache(*cache_settings)
205  
206      dset_id = h5d.open(parent.id, name, dapl=dapl)
207  
208      return dset_id

212  class AbstractView(ABC):
213      _dset: "Dataset"
214  
215      def __init__(self, dset):
216          self._dset = dset
217  
218      def __len__(self):
219          return len(self._dset)
220  
221      @property
222      @abstractmethod
223      def dtype(self):
224          ...  # pragma: nocover
225  
226      @property
227      def ndim(self):
228          return self._dset.ndim
229  
230      @property
231      def shape(self):
232          return self._dset.shape
233  
234      @property
235      def size(self):
236          return self._dset.size
237  
238      @abstractmethod
239      def __getitem__(self, idx):
240          ...  # pragma: nocover
241  
242      def __array__(self, dtype=None, copy=None):
243          if copy is False:
244              raise ValueError(
245                  f"{self.__class__.__name__}.__array__ received {copy=} "
246                  "but memory allocation cannot be avoided on read"
247              )
248  
249          # If self.ndim == 0, convert np.generic back to np.ndarray
250          return numpy.asarray(self[()], dtype=dtype or self.dtype)

252  class AsTypeView(AbstractView):
253      """Wrapper to convert data on reading from a dataset. 
254      """
255      def __init__(self, dset, dtype):
256          super().__init__(dset)
257          self._dtype = numpy.dtype(dtype)
258  
259      @property
260      def dtype(self):
261          return self._dtype
262  
263      def __getitem__(self, idx):
264          return self._dset.__getitem__(idx, new_dtype=self._dtype)
265  
266      def __array__(self, dtype=None, copy=None):
267          return self._dset.__array__(dtype or self._dtype, copy)

270  class AsStrView(AbstractView):
271      """Wrapper to decode strings on reading the dataset"""
272      def __init__(self, dset, encoding, errors='strict'):
273          super().__init__(dset)
274          self.encoding = encoding
275          self.errors = errors
276  
277      @property
278      def dtype(self):
279          return numpy.dtype(object)
280  
281      def __getitem__(self, idx):
282          bytes_arr = self._dset[idx]
283          # numpy.char.decode() seems like the obvious thing to use. But it only
284          # accepts numpy string arrays, not object arrays of bytes (which we
285          # return from HDF5 variable-length strings). And the numpy
286          # implementation is not faster than doing it with a loop; in fact, by
287          # not converting the result to a numpy unicode array, the
288          # naive way can be faster! (Comparing with numpy 1.18.4, June 2020)
289          if numpy.isscalar(bytes_arr):
290              return bytes_arr.decode(self.encoding, self.errors)
291  
292          return numpy.array([
293              b.decode(self.encoding, self.errors) for b in bytes_arr.flat
294          ], dtype=object).reshape(bytes_arr.shape)

297  class FieldsView(AbstractView):
298      """Wrapper to extract named fields from a dataset with a struct dtype"""
299  
300      def __init__(self, dset, prior_dtype, names):
301          super().__init__(dset)
302          if isinstance(names, str):
303              self.extract_field = names
304              names = [names]
305          else:
306              self.extract_field = None
307          self.read_dtype = readtime_dtype(prior_dtype, names)
308  
309      @property
310      def dtype(self):
311          t = self.read_dtype
312          if self.extract_field is not None:
313              t = t[self.extract_field]
314          return t
315  
316      def __getitem__(self, idx):
317          data = self._dset.__getitem__(idx, new_dtype=self.read_dtype)
318          if self.extract_field is not None:
319              data = data[self.extract_field]
320          return data
321

323  def readtime_dtype(basetype, names):
324      """Make a NumPy compound dtype with a subset of available fields"""
325      if basetype.names is None:  # Names provided, but not compound
326          raise ValueError("Field names only allowed for compound types")
327  
328      for name in names:  # Check all names are legal
329          if name not in basetype.names:
330              raise ValueError("Field %s does not appear in this type." % name)
331  
332      return numpy.dtype([(name, basetype.fields[name][0]) for name in names])

354  class ChunkIterator:
355      """ 
356      Class to iterate through list of chunks of a given dataset 
357      """
358      def __init__(self, dset, source_sel=None):
359          self._shape = dset.shape
360          rank = len(dset.shape)
361  
362          if not dset.chunks:
363              # can only use with chunked datasets
364              raise TypeError("Chunked dataset required")
365  
366          self._layout = dset.chunks
367          if source_sel is None:
368              # select over entire dataset
369              self._sel = tuple(
370                  slice(0, self._shape[dim])
371                  for dim in range(rank)
372              )
373          else:
374              if isinstance(source_sel, slice):
375                  self._sel = (source_sel,)
376              else:
377                  self._sel = source_sel
378          if len(self._sel) != rank:
379              raise ValueError("Invalid selection - selection region must have same rank as dataset")
380          self._chunk_index = []
381          for dim in range(rank):
382              s = self._sel[dim]
383              if s.start < 0 or s.stop > self._shape[dim] or s.stop <= s.start:
384                  raise ValueError("Invalid selection - selection region must be within dataset space")
385              index = s.start // self._layout[dim]
386              self._chunk_index.append(index)
387  
388      def __iter__(self):
389          return self
390  
391      def __next__(self):
392          rank = len(self._shape)
393          slices = []
394          if rank == 0 or self._chunk_index[0] * self._layout[0] >= self._sel[0].stop:
395              # ran past the last chunk, end iteration
396              raise StopIteration()
397  
398          for dim in range(rank):
399              s = self._sel[dim]
400              start = self._chunk_index[dim] * self._layout[dim]
401              stop = (self._chunk_index[dim] + 1) * self._layout[dim]
402              # adjust the start if this is an edge chunk
403              if start < s.start:
404                  start = s.start
405              if stop > s.stop:
406                  stop = s.stop  # trim to end of the selection
407              s = slice(start, stop, 1)
408              slices.append(s)
409  
410          # bump up the last index and carry forward if we run outside the selection
411          dim = rank - 1
412          while dim >= 0:
413              s = self._sel[dim]
414              self._chunk_index[dim] += 1
415  
416              chunk_end = self._chunk_index[dim] * self._layout[dim]
417              if chunk_end < s.stop:
418                  # we still have room to extend along this dimensions
419                  return tuple(slices)
420  
421              if dim > 0:
422                  # reset to the start and continue iterating with higher dimension
423                  self._chunk_index[dim] = s.start // self._layout[dim]
424              dim -= 1
425          return tuple(slices)

428  class Dataset(HLObject):
429  
430      """ 
431          Represents an HDF5 dataset 
432      """
433  
434      def astype(self, dtype):
435          """ Get a wrapper allowing you to perform reads to a 
436          different destination type, e.g.: 
437   
438          >>> double_precision = dataset.astype('f8')[0:100:2] 
439          """
440          return AsTypeView(self, dtype)
441  
442      def asstr(self, encoding=None, errors='strict'):
443          """Get a wrapper to read string data as Python strings: 
444   
445          >>> str_array = dataset.asstr()[:] 
446   
447          The parameters have the same meaning as in ``bytes.decode()``. 
448          If ``encoding`` is unspecified, it will use the encoding in the HDF5 
449          datatype (either ascii or utf-8). 
450          """
451          string_info = h5t.check_string_dtype(self.dtype)
452          if string_info is None:
453              raise TypeError(
454                  "dset.asstr() can only be used on datasets with "
455                  "an HDF5 string datatype"
456              )
457          if encoding is None:
458              encoding = string_info.encoding
459          return AsStrView(self, encoding, errors=errors)
460  
461      def fields(self, names, *, _prior_dtype=None):
462          """Get a wrapper to read a subset of fields from a compound data type: 
463   
464          >>> 2d_coords = dataset.fields(['x', 'y'])[:] 
465   
466          If names is a string, a single field is extracted, and the resulting 
467          arrays will have that dtype. Otherwise, it should be an iterable, 
468          and the read data will have a compound dtype. 
469          """
470          if _prior_dtype is None:
471              _prior_dtype = self.dtype
472          return FieldsView(self, _prior_dtype, names)

494      @property
495      def shape(self):
496          """Numpy-style shape tuple giving dataset dimensions"""
497          if 'shape' in self._cache_props:
498              return self._cache_props['shape']
499  
500          with phil:
501              shape = self.id.shape
502  
503          # If the file is read-only, cache the shape to speed-up future uses.
504          # This cache is invalidated by .refresh() when using SWMR.
505          if self._readonly:
506              self._cache_props['shape'] = shape
507          return shape

515      @property
516      def size(self):
517          """Numpy-style attribute giving the total dataset size"""
518          if 'size' in self._cache_props:
519              return self._cache_props['size']
520  
521          if self._is_empty:
522              size = None
523          else:
524              size = product(self.shape)
525  
526          # If the file is read-only, cache the size to speed-up future uses.
527          # This cache is invalidated by .refresh() when using SWMR.
528          if self._readonly:
529              self._cache_props['size'] = size
530          return size
531  
532      @property
533      def nbytes(self):
534          """Numpy-style attribute giving the raw dataset size as the number of bytes"""
535          size = self.size
536          if size is None:  # if we are an empty 0-D array, then there are no bytes in the dataset
537              return 0
538          return self.dtype.itemsize * size

540      @property
541      def _selector(self):
542          """Internal object for optimised selection of data"""
543          if '_selector' in self._cache_props:
544              return self._cache_props['_selector']
545  
546          slr = _selector.Selector(self.id.get_space())
547  
548          # If the file is read-only, cache the reader to speed up future uses.
549          # This cache is invalidated by .refresh() when using SWMR.
550          if self._readonly:
551              self._cache_props['_selector'] = slr
552          return slr
553  
554      @property
555      def _fast_reader(self):
556          """Internal object for optimised reading of data"""
557          if '_fast_reader' in self._cache_props:
558              return self._cache_props['_fast_reader']
559  
560          rdr = _selector.Reader(self.id)
561  
562          # If the file is read-only, cache the reader to speed up future uses.
563          # This cache is invalidated by .refresh() when using SWMR.
564          if self._readonly:
565              self._cache_props['_fast_reader'] = rdr
566          return rdr

587          for x in ('gzip','lzf','szip'):
588              if x in self._filters:
589                  return x
590          return None

595          """ Compression setting.  Int(0-9) for gzip, 2-tuple for szip. """

598      @property
599      @with_phil

601          """Shuffle filter present (T/F)"""
602          return 'shuffle' in self._filters
603  
604      @property
605      @with_phil
606      def fletcher32(self):
607          """Fletcher32 filter is present (T/F)"""
608          return 'fletcher32' in self._filters
609  
610      @property
611      @with_phil
612      def scaleoffset(self):
613          """Scale/offset filter settings. For integer data types, this is 
614          the number of bits stored, or 0 for auto-detected. For floating 
615          point data types, this is the number of decimal places retained. 
616          If the scale/offset filter is not in use, this is None."""
617          try:
618              return self._filters['scaleoffset'][1]
619          except KeyError:
620              return None
621  
622      @property
623      @with_phil
624      def external(self):
625          """External file settings. Returns a list of tuples of 
626          (name, offset, size) for each external file entry, or returns None 
627          if no external files are used."""
628          count = self._dcpl.get_external_count()
629          if count<=0:
630              return None
631          ext_list = list()
632          for x in range(count):
633              (name, offset, size) = self._dcpl.get_external(x)
634              ext_list.append( (filename_decode(name), offset, size) )
635          return ext_list
636  
637      @property
638      @with_phil
639      def maxshape(self):
640          """Shape up to which this dataset can be resized.  Axes with value 
641          None have no resize limit. """
642          space = self.id.get_space()
643          dims = space.get_simple_extent_dims(True)
644          if dims is None:
645              return None
646  
647          return tuple(x if x != h5s.UNLIMITED else None for x in dims)
648  
649      @property
650      @with_phil
651      def fillvalue(self):
652          """Fill value for this dataset (0 by default)"""
653          arr = numpy.zeros((1,), dtype=self.dtype)
654          self._dcpl.get_fill_value(arr)
655          return arr[0]
656  
657      @cached_property
658      @with_phil
659      def _extent_type(self):
660          """Get extent type for this dataset - SIMPLE, SCALAR or NULL"""
661          return self.id.get_space().get_simple_extent_type()
662  
663      @cached_property
664      def _is_empty(self):
665          """Check if extent type is empty"""
666          return self._extent_type == h5s.NULL
667  
668      @cached_property
669      def _dcpl(self):
670          """ 
671          The dataset creation property list used when this dataset was created. 
672          """
673          return self.id.get_create_plist()

676      def _filters(self):
677          """ 
678          The active filters of the dataset. 
679          """

682      @with_phil
683      def __init__(self, bind, *, readonly=False):
684          """ Create a new Dataset object by binding to a low-level DatasetID. 
685          """
686          if not isinstance(bind, h5d.DatasetID):
687              raise ValueError("%s is not a DatasetID" % bind)
688          super().__init__(bind)

691          self._readonly = readonly

694      def resize(self, size, axis=None):
695          """ Resize the dataset, or the specified axis. 
696   
697          The dataset must be stored in chunked format; it can be resized up to 
698          the "maximum shape" (keyword maxshape) specified at creation time. 
699          The rank of the dataset cannot be changed. 
700   
701          "Size" should be a shape tuple, or if an axis is specified, an integer. 
702   
703          BEWARE: This functions differently than the NumPy resize() method! 
704          The data is not "reshuffled" to fit in the new shape; each axis is 
705          grown or shrunk independently.  The coordinates of existing data are 
706          fixed. 
707          """
708          with phil:
709              if self.chunks is None:
710                  raise TypeError("Only chunked datasets can be resized")
711  
712              if axis is not None:
713                  if not (axis >=0 and axis < self.id.rank):
714                      raise ValueError("Invalid axis (0 to %s allowed)" % (self.id.rank-1))
715                  try:
716                      newlen = int(size)
717                  except TypeError:
718                      raise TypeError("Argument must be a single int if axis is specified")
719                  size = list(self.shape)
720                  size[axis] = newlen
721  
722              size = tuple(size)
723              self.id.set_extent(size)
724              #h5f.flush(self.id)  # THG recommends

726      @with_phil
727      def __len__(self):
728          """ The size of the first axis.  TypeError if scalar. 
729   
730          Limited to 2**32 on 32-bit systems; Dataset.len() is preferred. 
731          """
732          size = self.len()
733          if size > sys.maxsize:
734              raise OverflowError("Value too big for Python's __len__; use Dataset.len() instead.")
735          return size

737      def len(self):
738          """ The size of the first axis.  TypeError if scalar. 
739   
740          Use of this method is preferred to len(dset), as Python's built-in 
741          len() cannot handle values greater then 2**32 on 32-bit systems. 
742          """
743          with phil:
744              shape = self.shape
745              if len(shape) == 0:
746                  raise TypeError("Attempt to take len() of scalar dataset")
747              return shape[0]

749      @with_phil
750      def __iter__(self):
751          """ Iterate over the first axis.  TypeError if scalar. 
752   
753          BEWARE: Modifications to the yielded data are *NOT* written to file. 
754          """
755          shape = self.shape
756          if len(shape) == 0:
757              raise TypeError("Can't iterate over a scalar dataset")
758          for i in range(shape[0]):
759              yield self[i]

786      @with_phil
787      def __getitem__(self, args, new_dtype=None):
788          """ Read a slice from the HDF5 dataset. 
789   
790          Takes slices and recarray-style field names (more than one is 
791          allowed!) in any order.  Obeys basic NumPy rules, including 
792          broadcasting. 
793   
794          Also supports: 
795   
796          * Boolean "mask" array indexing 
797          """
798          args = args if isinstance(args, tuple) else (args,)
799  
800          if self._fast_read_ok and (new_dtype is None):
801              try:
802                  return self._fast_reader.read(args)
803              except TypeError:
804                  pass  # Fall back to Python read pathway below
805  
806          if self._is_empty:
807              # Check 'is Ellipsis' to avoid equality comparison with an array:
808              # array equality returns an array, not a boolean.
809              if args == () or (len(args) == 1 and args[0] is Ellipsis):
810                  return Empty(self.dtype)
811              raise ValueError("Empty datasets cannot be sliced")
812  
813          # Sort field names from the rest of the args.
814          names = tuple(x for x in args if isinstance(x, str))
815  
816          if names:
817              # Read a subset of the fields in this structured dtype
818              if len(names) == 1:
819                  names = names[0]  # Read with simpler dtype of this field
820              args = tuple(x for x in args if not isinstance(x, str))
821              return self.fields(names, _prior_dtype=new_dtype)[args]
822  
823          if new_dtype is None:
824              new_dtype = self.dtype
825          mtype = h5t.py_create(new_dtype)
826  
827          # === Special-case region references ====
828  
829          if len(args) == 1 and isinstance(args[0], h5r.RegionReference):
830  
831              obj = h5r.dereference(args[0], self.id)
832              if obj != self.id:
833                  raise ValueError("Region reference must point to this dataset")
834  
835              sid = h5r.get_region(args[0], self.id)
836              mshape = sel.guess_shape(sid)
837              if mshape is None:
838                  # 0D with no data (NULL or deselected SCALAR)
839                  return Empty(new_dtype)
840              out = numpy.zeros(mshape, dtype=new_dtype)
841              if out.size == 0:
842                  return out
843  
844              sid_out = h5s.create_simple(mshape)
845              sid_out.select_all()
846              self.id.read(sid_out, sid, out, mtype)
847              return out
848  
849          # === Check for zero-sized datasets =====
850  
851          if self.size == 0:
852              # Check 'is Ellipsis' to avoid equality comparison with an array:
853              # array equality returns an array, not a boolean.
854              if args == () or (len(args) == 1 and args[0] is Ellipsis):
855                  return numpy.zeros(self.shape, dtype=new_dtype)
856  
857          # === Scalar dataspaces =================
858  
859          if self.shape == ():
860              fspace = self.id.get_space()
861              selection = sel2.select_read(fspace, args)
862              if selection.mshape is None:
863                  arr = numpy.zeros((), dtype=new_dtype)
864              else:
865                  arr = numpy.zeros(selection.mshape, dtype=new_dtype)
866              for mspace, fspace in selection:
867                  self.id.read(mspace, fspace, arr, mtype)
868              if selection.mshape is None:
869                  return arr[()]
870              return arr
871  
872          # === Everything else ===================
873  
874          # Perform the dataspace selection.
875          selection = sel.select(self.shape, args, dataset=self)
876  
877          if selection.nselect == 0:
878              return numpy.zeros(selection.array_shape, dtype=new_dtype)
879  
880          arr = numpy.zeros(selection.array_shape, new_dtype, order='C')
881  
882          # Perform the actual read
883          mspace = h5s.create_simple(selection.mshape)
884          fspace = selection.id
885          self.id.read(mspace, fspace, arr, mtype, dxpl=self._dxpl)
886  
887          # Patch up the output for NumPy
888          if arr.shape == ():
889              return arr[()]   # 0 dim array -> numpy scalar
890          return arr
891  
892      @with_phil
893      def __setitem__(self, args, val):
894          """ Write to the HDF5 dataset from a Numpy array. 
895   
896          NumPy's broadcasting rules are honored, for "simple" indexing 
897          (slices and integers).  For advanced indexing, the shapes must 
898          match. 
899          """
900          args = args if isinstance(args, tuple) else (args,)
901  
902          # Sort field indices from the slicing
903          names = tuple(x for x in args if isinstance(x, str))
904          args = tuple(x for x in args if not isinstance(x, str))
905  
906          # Generally we try to avoid converting the arrays on the Python
907          # side.  However, for compound literals this is unavoidable.
908          vlen = h5t.check_vlen_dtype(self.dtype)
909          if vlen is not None and vlen not in (bytes, str):
910              try:
911                  val = numpy.asarray(val, dtype=vlen)
912              except (ValueError, TypeError):
913                  try:
914                      val = numpy.array([numpy.array(x, dtype=vlen)
915                                         for x in val], dtype=self.dtype)
916                  except (ValueError, TypeError):
917                      pass
918              if vlen == val.dtype:
919                  if val.ndim > 1:
920                      tmp = numpy.empty(shape=val.shape[:-1], dtype=object)
921                      tmp.ravel()[:] = [i for i in val.reshape(
922                          (product(val.shape[:-1]), val.shape[-1])
923                      )]
924                  else:
925                      tmp = numpy.array([None], dtype=object)
926                      tmp[0] = val
927                  val = tmp
928          elif self.dtype.kind == "O" or \
929            (self.dtype.kind == 'V' and \
930            (not isinstance(val, numpy.ndarray) or val.dtype.kind != 'V') and \
931            (self.dtype.subdtype is None)):
932              if len(names) == 1 and self.dtype.fields is not None:
933                  # Single field selected for write, from a non-array source
934                  if not names[0] in self.dtype.fields:
935                      raise ValueError("No such field for indexing: %s" % names[0])
936                  dtype = self.dtype.fields[names[0]][0]
937                  cast_compound = True
938              else:
939                  dtype = self.dtype
940                  cast_compound = False
941  
942              val = numpy.asarray(val, dtype=dtype.base, order='C')
943              if cast_compound:
944                  val = val.view(numpy.dtype([(names[0], dtype)]))
945                  val = val.reshape(val.shape[:len(val.shape) - len(dtype.shape)])
946          elif (self.dtype.kind == 'S'
947                and (h5t.check_string_dtype(self.dtype).encoding == 'utf-8')
948                and (find_item_type(val) is str)
949          ):
950              # Writing str objects to a fixed-length UTF-8 string dataset.
951              # Numpy's normal conversion only handles ASCII characters, but
952              # when the destination is UTF-8, we want to allow any unicode.
953              # This *doesn't* handle numpy fixed-length unicode data ('U' dtype),
954              # as HDF5 has no equivalent, and converting fixed length UTF-32
955              # to variable length UTF-8 would obscure what's going on.
956              str_array = numpy.asarray(val, order='C', dtype=object)
957              val = numpy.array([
958                  s.encode('utf-8') for s in str_array.flat
959              ], dtype=self.dtype).reshape(str_array.shape)
960          else:
961              # If the input data is already an array, let HDF5 do the conversion.
962              # If it's a list or similar, don't make numpy guess a dtype for it.
963              dt = None if isinstance(val, numpy.ndarray) else self.dtype.base
964              val = numpy.asarray(val, order='C', dtype=dt)
965  
966          # Check for array dtype compatibility and convert
967          if self.dtype.subdtype is not None:
968              shp = self.dtype.subdtype[1]
969              valshp = val.shape[-len(shp):]
970              if valshp != shp:  # Last dimension has to match
971                  raise TypeError("When writing to array types, last N dimensions have to match (got %s, but should be %s)" % (valshp, shp,))
972              mtype = h5t.py_create(numpy.dtype((val.dtype, shp)))
973              mshape = val.shape[0:len(val.shape)-len(shp)]
974  
975          # Make a compound memory type if field-name slicing is required
976          elif len(names) != 0:
977  
978              mshape = val.shape
979  
980              # Catch common errors
981              if self.dtype.fields is None:
982                  raise TypeError("Illegal slicing argument (not a compound dataset)")
983              mismatch = [x for x in names if x not in self.dtype.fields]
984              if len(mismatch) != 0:
985                  mismatch = ", ".join('"%s"'%x for x in mismatch)
986                  raise ValueError("Illegal slicing argument (fields %s not in dataset type)" % mismatch)
987  
988              # Write non-compound source into a single dataset field
989              if len(names) == 1 and val.dtype.fields is None:
990                  subtype = h5t.py_create(val.dtype)
991                  mtype = h5t.create(h5t.COMPOUND, subtype.get_size())
992                  mtype.insert(self._e(names[0]), 0, subtype)
993  
994              # Make a new source type keeping only the requested fields
995              else:
996                  fieldnames = [x for x in val.dtype.names if x in names] # Keep source order
997                  mtype = h5t.create(h5t.COMPOUND, val.dtype.itemsize)
998                  for fieldname in fieldnames:
999                      subtype = h5t.py_create(val.dtype.fields[fieldname][0])
1000                     offset = val.dtype.fields[fieldname][1]
1001                     mtype.insert(self._e(fieldname), offset, subtype)
1002 
1003         # Use mtype derived from array (let DatasetID.write figure it out)
1004         else:
1005             mshape = val.shape
1006             mtype = None
1007 
1008         # Perform the dataspace selection
1009         selection = sel.select(self.shape, args, dataset=self)
1010 
1011         if selection.nselect == 0:
1012             return
1013 
1014         # Broadcast scalars if necessary.
1015         # In order to avoid slow broadcasting filling the destination by
1016         # the scalar value, we create an intermediate array of the same
1017         # size as the destination buffer provided that size is reasonable.
1018         # We assume as reasonable a size smaller or equal as the used dataset
1019         # chunk size if any.
1020         # In case of dealing with a non-chunked destination dataset or with
1021         # a selection whose size is larger than the dataset chunk size we fall
1022         # back to using an intermediate array of size equal to the last dimension
1023         # of the destination buffer.
1024         # The reasoning behind is that it makes sense to assume the creator of
1025         # the dataset used an appropriate chunk size according the available
1026         # memory. In any case, if we cannot afford to create an intermediate
1027         # array of the same size as the dataset chunk size, the user program has
1028         # little hope to go much further. Solves h5py issue #1067
1029         if mshape == () and selection.array_shape != ():
1030             if self.dtype.subdtype is not None:
1031                 raise TypeError("Scalar broadcasting is not supported for array dtypes")
1032             if self.chunks and (product(self.chunks) >= product(selection.array_shape)):
1033                 val2 = numpy.empty(selection.array_shape, dtype=val.dtype)
1034             else:
1035                 val2 = numpy.empty(selection.array_shape[-1], dtype=val.dtype)
1036             val2[...] = val
1037             val = val2
1038             mshape = val.shape
1039 
1040         # Perform the write, with broadcasting
1041         mspace = h5s.create_simple(selection.expand_shape(mshape))
1042         for fspace in selection.broadcast(mshape):
1043             self.id.write(mspace, fspace, val, mtype, dxpl=self._dxpl)

1045     def read_direct(self, dest, source_sel=None, dest_sel=None):
1046         """ Read data directly from HDF5 into an existing NumPy array. 
1047  
1048         The destination array must be C-contiguous and writable. 
1049         Selections must be the output of numpy.s_[<args>]. 
1050  
1051         Broadcasting is supported for simple indexing. 
1052         """
1053         with phil:
1054             if self._is_empty:
1055                 raise TypeError("Empty datasets have no numpy representation")
1056             if source_sel is None:
1057                 source_sel = sel.SimpleSelection(self.shape)
1058             else:
1059                 source_sel = sel.select(self.shape, source_sel, self)  # for numpy.s_
1060             fspace = source_sel.id
1061 
1062             if dest_sel is None:
1063                 dest_sel = sel.SimpleSelection(dest.shape)
1064             else:
1065                 dest_sel = sel.select(dest.shape, dest_sel)
1066 
1067             for mspace in dest_sel.broadcast(source_sel.array_shape):
1068                 self.id.read(mspace, fspace, dest, dxpl=self._dxpl)

1070     def write_direct(self, source, source_sel=None, dest_sel=None):
1071         """ Write data directly to HDF5 from a NumPy array. 
1072  
1073         The source array must be C-contiguous.  Selections must be 
1074         the output of numpy.s_[<args>]. 
1075  
1076         Broadcasting is supported for simple indexing. 
1077         """
1078         with phil:
1079             if self._is_empty:
1080                 raise TypeError("Empty datasets cannot be written to")
1081             if source_sel is None:
1082                 source_sel = sel.SimpleSelection(source.shape)
1083             else:
1084                 source_sel = sel.select(source.shape, source_sel)  # for numpy.s_
1085             mspace = source_sel.id
1086 
1087             if dest_sel is None:
1088                 dest_sel = sel.SimpleSelection(self.shape)
1089             else:
1090                 dest_sel = sel.select(self.shape, dest_sel, self)
1091 
1092             for fspace in dest_sel.broadcast(source_sel.array_shape):
1093                 self.id.write(mspace, fspace, source, dxpl=self._dxpl)
