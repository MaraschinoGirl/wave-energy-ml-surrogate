65   def _safe_divide(numerator, denominator):
66       """Prevents overflow and division by zero."""
67       # This is used for classifiers where the denominator might become zero exatly.
68       # For instance for log loss, HalfBinomialLoss, if proba=0 or proba=1 exactly, then
69       # denominator = hessian = 0, and we should set the node value in the line search to
70       # zero as there is no improvement of the loss possible.
71       # For numerical safety, we do this already for extremely tiny values.
72       if abs(denominator) < 1e-150:
73           return 0.0
74       else:
75           # Cast to Python float to trigger Python errors, e.g. ZeroDivisionError,
76           # without relying on `np.errstate` that is not supported by Pyodide.
77           result = float(numerator) / float(denominator)
78           # Cast to Python float to trigger a ZeroDivisionError without relying
79           # on `np.errstate` that is not supported by Pyodide.
80           result = float(numerator) / float(denominator)
81           if math.isinf(result):
82               warnings.warn("overflow encountered in _safe_divide", RuntimeWarning)
83           return result

86   def _init_raw_predictions(X, estimator, loss, use_predict_proba):
87       """Return the initial raw predictions. 
88    
89       Parameters 
90       ---------- 
91       X : ndarray of shape (n_samples, n_features) 
92           The data array. 
93       estimator : object 
94           The estimator to use to compute the predictions. 
95       loss : BaseLoss 
96           An instance of a loss function class. 
97       use_predict_proba : bool 
98           Whether estimator.predict_proba is used instead of estimator.predict. 
99    
100      Returns 
101      ------- 
102      raw_predictions : ndarray of shape (n_samples, K) 
103          The initial raw predictions. K is equal to 1 for binary 
104          classification and regression, and equal to the number of classes 
105          for multiclass classification. ``raw_predictions`` is casted 
106          into float64. 
107      """
108      # TODO: Use loss.fit_intercept_only where appropriate instead of
109      # DummyRegressor which is the default given by the `init` parameter,
110      # see also _init_state.
111      if use_predict_proba:
112          # Our parameter validation, set via _fit_context and _parameter_constraints
113          # already guarantees that estimator has a predict_proba method.
114          predictions = estimator.predict_proba(X)
115          if not loss.is_multiclass:
116              predictions = predictions[:, 1]  # probability of positive class
117          eps = np.finfo(np.float32).eps  # FIXME: This is quite large!
118          predictions = np.clip(predictions, eps, 1 - eps, dtype=np.float64)
119      else:
120          predictions = estimator.predict(X).astype(np.float64)
121  
122      if predictions.ndim == 1:
123          return loss.link.link(predictions).reshape(-1, 1)
124      else:
125          return loss.link.link(predictions)

128  def _update_terminal_regions(
129      loss,
130      tree,
131      X,
132      y,
133      neg_gradient,
134      raw_prediction,
135      sample_weight,
136      sample_mask,
137      learning_rate=0.1,
138      k=0,
139  ):
140      """Update the leaf values to be predicted by the tree and raw_prediction. 
141   
142      The current raw predictions of the model (of this stage) are updated. 
143   
144      Additionally, the terminal regions (=leaves) of the given tree are updated as well. 
145      This corresponds to the line search step in "Greedy Function Approximation" by 
146      Friedman, Algorithm 1 step 5. 
147   
148      Update equals: 
149          argmin_{x} loss(y_true, raw_prediction_old + x * tree.value) 
150   
151      For non-trivial cases like the Binomial loss, the update has no closed formula and 
152      is an approximation, again, see the Friedman paper. 
153   
154      Also note that the update formula for the SquaredError is the identity. Therefore, 
155      in this case, the leaf values don't need an update and only the raw_predictions are 
156      updated (with the learning rate included). 
157   
158      Parameters 
159      ---------- 
160      loss : BaseLoss 
161      tree : tree.Tree 
162          The tree object. 
163      X : ndarray of shape (n_samples, n_features) 
164          The data array. 
165      y : ndarray of shape (n_samples,) 
166          The target labels. 
167      neg_gradient : ndarray of shape (n_samples,) 
168          The negative gradient. 
169      raw_prediction : ndarray of shape (n_samples, n_trees_per_iteration) 
170          The raw predictions (i.e. values from the tree leaves) of the 
171          tree ensemble at iteration ``i - 1``. 
172      sample_weight : ndarray of shape (n_samples,) 
173          The weight of each sample. 
174      sample_mask : ndarray of shape (n_samples,) 
175          The sample mask to be used. 
176      learning_rate : float, default=0.1 
177          Learning rate shrinks the contribution of each tree by 
178           ``learning_rate``. 
179      k : int, default=0 
180          The index of the estimator being updated. 
181      """
182      # compute leaf for each sample in ``X``.
183      terminal_regions = tree.apply(X)
184  
185      if not isinstance(loss, HalfSquaredError):
186          # mask all which are not in sample mask.
187          masked_terminal_regions = terminal_regions.copy()
188          masked_terminal_regions[~sample_mask] = -1
189  
190          if isinstance(loss, HalfBinomialLoss):
191  
192              def compute_update(y_, indices, neg_gradient, raw_prediction, k):
193                  # Make a single Newton-Raphson step, see "Additive Logistic Regression:
194                  # A Statistical View of Boosting" FHT00 and note that we use a slightly
195                  # different version (factor 2) of "F" with proba=expit(raw_prediction).
196                  # Our node estimate is given by:
197                  #    sum(w * (y - prob)) / sum(w * prob * (1 - prob))
198                  # we take advantage that: y - prob = neg_gradient
199                  neg_g = neg_gradient.take(indices, axis=0)
200                  prob = y_ - neg_g
201                  # numerator = negative gradient = y - prob
202                  numerator = np.average(neg_g, weights=sw)
203                  # denominator = hessian = prob * (1 - prob)
204                  denominator = np.average(prob * (1 - prob), weights=sw)
205                  return _safe_divide(numerator, denominator)
206  
207          elif isinstance(loss, HalfMultinomialLoss):
208  
209              def compute_update(y_, indices, neg_gradient, raw_prediction, k):
210                  # we take advantage that: y - prob = neg_gradient
211                  neg_g = neg_gradient.take(indices, axis=0)
212                  prob = y_ - neg_g
213                  K = loss.n_classes
214                  # numerator = negative gradient * (k - 1) / k
215                  # Note: The factor (k - 1)/k appears in the original papers "Greedy
216                  # Function Approximation" by Friedman and "Additive Logistic
217                  # Regression" by Friedman, Hastie, Tibshirani. This factor is, however,
218                  # wrong or at least arbitrary as it directly multiplies the
219                  # learning_rate. We keep it for backward compatibility.
220                  numerator = np.average(neg_g, weights=sw)
221                  numerator *= (K - 1) / K
222                  # denominator = (diagonal) hessian = prob * (1 - prob)
223                  denominator = np.average(prob * (1 - prob), weights=sw)
224                  return _safe_divide(numerator, denominator)
225  
226          elif isinstance(loss, ExponentialLoss):
227  
228              def compute_update(y_, indices, neg_gradient, raw_prediction, k):
229                  neg_g = neg_gradient.take(indices, axis=0)
230                  # numerator = negative gradient = y * exp(-raw) - (1-y) * exp(raw)
231                  numerator = np.average(neg_g, weights=sw)
232                  # denominator = hessian = y * exp(-raw) + (1-y) * exp(raw)
233                  # if y=0: hessian = exp(raw) = -neg_g
234                  #    y=1: hessian = exp(-raw) = neg_g
235                  hessian = neg_g.copy()
236                  hessian[y_ == 0] *= -1
237                  denominator = np.average(hessian, weights=sw)
238                  return _safe_divide(numerator, denominator)
239  
240          else:
241  
242              def compute_update(y_, indices, neg_gradient, raw_prediction, k):
243                  return loss.fit_intercept_only(
244                      y_true=y_ - raw_prediction[indices, k],
245                      sample_weight=sw,
246                  )
247  
248          # update each leaf (= perform line search)
249          for leaf in np.nonzero(tree.children_left == TREE_LEAF)[0]:
250              indices = np.nonzero(masked_terminal_regions == leaf)[
251                  0
252              ]  # of terminal regions
253              y_ = y.take(indices, axis=0)
254              sw = None if sample_weight is None else sample_weight[indices]
255              update = compute_update(y_, indices, neg_gradient, raw_prediction, k)
256  
257              # TODO: Multiply here by learning rate instead of everywhere else.
258              tree.value[leaf, 0, 0] = update
259  
260      # update predictions (both in-bag and out-of-bag)
261      raw_prediction[:, k] += learning_rate * tree.value[:, 0, 0].take(
262          terminal_regions, axis=0
263      )

266  def set_huber_delta(loss, y_true, raw_prediction, sample_weight=None):
267      """Calculate and set self.closs.delta based on self.quantile."""
268      abserr = np.abs(y_true - raw_prediction.squeeze())
269      # sample_weight is always a ndarray, never None.
270      delta = _weighted_percentile(abserr, sample_weight, 100 * loss.quantile)
271      loss.closs.delta = float(delta)

274  class VerboseReporter:
275      """Reports verbose output to stdout. 
276   
277      Parameters 
278      ---------- 
279      verbose : int 
280          Verbosity level. If ``verbose==1`` output is printed once in a while 
281          (when iteration mod verbose_mod is zero).; if larger than 1 then output 
282          is printed for each update. 
283      """
284  
285      def __init__(self, verbose):
286          self.verbose = verbose
287  
288      def init(self, est, begin_at_stage=0):
289          """Initialize reporter 
290   
291          Parameters 
292          ---------- 
293          est : Estimator 
294              The estimator 
295   
296          begin_at_stage : int, default=0 
297              stage at which to begin reporting 
298          """
299          # header fields and line format str
300          header_fields = ["Iter", "Train Loss"]
301          verbose_fmt = ["{iter:>10d}", "{train_score:>16.4f}"]
302          # do oob?
303          if est.subsample < 1:
304              header_fields.append("OOB Improve")
305              verbose_fmt.append("{oob_impr:>16.4f}")
306          header_fields.append("Remaining Time")
307          verbose_fmt.append("{remaining_time:>16s}")
308  
309          # print the header line
310          print(("%10s " + "%16s " * (len(header_fields) - 1)) % tuple(header_fields))
311  
312          self.verbose_fmt = " ".join(verbose_fmt)
313          # plot verbose info each time i % verbose_mod == 0
314          self.verbose_mod = 1
315          self.start_time = time()
316          self.begin_at_stage = begin_at_stage
317  
318      def update(self, j, est):
319          """Update reporter with new iteration. 
320   
321          Parameters 
322          ---------- 
323          j : int 
324              The new iteration. 
325          est : Estimator 
326              The estimator. 
327          """
328          do_oob = est.subsample < 1
329          # we need to take into account if we fit additional estimators.
330          i = j - self.begin_at_stage  # iteration relative to the start iter
331          if (i + 1) % self.verbose_mod == 0:
332              oob_impr = est.oob_improvement_[j] if do_oob else 0
333              remaining_time = (
334                  (est.n_estimators - (j + 1)) * (time() - self.start_time) / float(i + 1)
335              )
336              if remaining_time > 60:
337                  remaining_time = "{0:.2f}m".format(remaining_time / 60.0)
338              else:
339                  remaining_time = "{0:.2f}s".format(remaining_time)
340              print(
341                  self.verbose_fmt.format(
342                      iter=j + 1,
343                      train_score=est.train_score_[j],
344                      oob_impr=oob_impr,
345                      remaining_time=remaining_time,
346                  )
347              )
348              if self.verbose == 1 and ((i + 1) // (self.verbose_mod * 10) > 0):
349                  # adjust verbose frequency (powers of 10)
350                  self.verbose_mod *= 10

427      def _fit_stage(
428          self,
429          i,
430          X,
431          y,
432          raw_predictions,
433          sample_weight,
434          sample_mask,
435          random_state,
436          X_csc=None,
437          X_csr=None,
438      ):
439          """Fit another stage of ``n_trees_per_iteration_`` trees."""
440          original_y = y
441  
442          if isinstance(self._loss, HuberLoss):
443              set_huber_delta(
444                  loss=self._loss,
445                  y_true=y,
446                  raw_prediction=raw_predictions,
447                  sample_weight=sample_weight,
448              )
449          # TODO: Without oob, i.e. with self.subsample = 1.0, we could call
450          # self._loss.loss_gradient and use it to set train_score_.
451          # But note that train_score_[i] is the score AFTER fitting the i-th tree.
452          # Note: We need the negative gradient!
453          neg_gradient = -self._loss.gradient(
454              y_true=y,
455              raw_prediction=raw_predictions,
456              sample_weight=None,  # We pass sample_weights to the tree directly.
457          )
458          # 2-d views of shape (n_samples, n_trees_per_iteration_) or (n_samples, 1)
459          # on neg_gradient to simplify the loop over n_trees_per_iteration_.
460          if neg_gradient.ndim == 1:
461              neg_g_view = neg_gradient.reshape((-1, 1))
462          else:
463              neg_g_view = neg_gradient
464  
465          for k in range(self.n_trees_per_iteration_):
466              if self._loss.is_multiclass:
467                  y = np.array(original_y == k, dtype=np.float64)
468  
469              # induce regression tree on the negative gradient
470              tree = DecisionTreeRegressor(
471                  criterion=self.criterion,
472                  splitter="best",
473                  max_depth=self.max_depth,
474                  min_samples_split=self.min_samples_split,
475                  min_samples_leaf=self.min_samples_leaf,
476                  min_weight_fraction_leaf=self.min_weight_fraction_leaf,
477                  min_impurity_decrease=self.min_impurity_decrease,
478                  max_features=self.max_features,
479                  max_leaf_nodes=self.max_leaf_nodes,
480                  random_state=random_state,
481                  ccp_alpha=self.ccp_alpha,
482              )
483  
484              if self.subsample < 1.0:
485                  # no inplace multiplication!
486                  sample_weight = sample_weight * sample_mask.astype(np.float64)
487  
488              X = X_csc if X_csc is not None else X
489              tree.fit(
490                  X, neg_g_view[:, k], sample_weight=sample_weight, check_input=False
491              )
492  
493              # update tree leaves
494              X_for_tree_update = X_csr if X_csr is not None else X
495              _update_terminal_regions(
496                  self._loss,
497                  tree.tree_,
498                  X_for_tree_update,
499                  y,
500                  neg_g_view[:, k],
501                  raw_predictions,
502                  sample_weight,
503                  sample_mask,
504                  learning_rate=self.learning_rate,
505                  k=k,
506              )
507  
508              # add tree to ensemble
509              self.estimators_[i, k] = tree
510  
511          return raw_predictions

513      def _set_max_features(self):
514          """Set self.max_features_."""
515          if isinstance(self.max_features, str):
516              if self.max_features == "auto":
517                  if is_classifier(self):
518                      max_features = max(1, int(np.sqrt(self.n_features_in_)))
519                  else:
520                      max_features = self.n_features_in_
521              elif self.max_features == "sqrt":
522                  max_features = max(1, int(np.sqrt(self.n_features_in_)))
523              else:  # self.max_features == "log2"
524                  max_features = max(1, int(np.log2(self.n_features_in_)))
525          elif self.max_features is None:
526              max_features = self.n_features_in_
527          elif isinstance(self.max_features, Integral):
528              max_features = self.max_features
529          else:  # float
530              max_features = max(1, int(self.max_features * self.n_features_in_))
531  
532          self.max_features_ = max_features

534      def _init_state(self):
535          """Initialize model state and allocate model state data structures."""
536  
537          self.init_ = self.init
538          if self.init_ is None:
539              if is_classifier(self):
540                  self.init_ = DummyClassifier(strategy="prior")
541              elif isinstance(self._loss, (AbsoluteError, HuberLoss)):
542                  self.init_ = DummyRegressor(strategy="quantile", quantile=0.5)
543              elif isinstance(self._loss, PinballLoss):
544                  self.init_ = DummyRegressor(strategy="quantile", quantile=self.alpha)
545              else:
546                  self.init_ = DummyRegressor(strategy="mean")
547  
548          self.estimators_ = np.empty(
549              (self.n_estimators, self.n_trees_per_iteration_), dtype=object
550          )
551          self.train_score_ = np.zeros((self.n_estimators,), dtype=np.float64)
552          # do oob?
553          if self.subsample < 1.0:
554              self.oob_improvement_ = np.zeros((self.n_estimators), dtype=np.float64)
555              self.oob_scores_ = np.zeros((self.n_estimators), dtype=np.float64)
556              self.oob_score_ = np.nan

558      def _clear_state(self):
559          """Clear the state of the gradient boosting model."""
560          if hasattr(self, "estimators_"):
561              self.estimators_ = np.empty((0, 0), dtype=object)
562          if hasattr(self, "train_score_"):
563              del self.train_score_
564          if hasattr(self, "oob_improvement_"):
565              del self.oob_improvement_
566          if hasattr(self, "oob_scores_"):
567              del self.oob_scores_
568          if hasattr(self, "oob_score_"):
569              del self.oob_score_
570          if hasattr(self, "init_"):
571              del self.init_
572          if hasattr(self, "_rng"):
573              del self._rng

575      def _resize_state(self):
576          """Add additional ``n_estimators`` entries to all attributes."""
577          # self.n_estimators is the number of additional est to fit
578          total_n_estimators = self.n_estimators
579          if total_n_estimators < self.estimators_.shape[0]:
580              raise ValueError(
581                  "resize with smaller n_estimators %d < %d"
582                  % (total_n_estimators, self.estimators_[0])
583              )
584  
585          self.estimators_ = np.resize(
586              self.estimators_, (total_n_estimators, self.n_trees_per_iteration_)
587          )
588          self.train_score_ = np.resize(self.train_score_, total_n_estimators)
589          if self.subsample < 1 or hasattr(self, "oob_improvement_"):
590              # if do oob resize arrays or create new if not available
591              if hasattr(self, "oob_improvement_"):
592                  self.oob_improvement_ = np.resize(
593                      self.oob_improvement_, total_n_estimators
594                  )
595                  self.oob_scores_ = np.resize(self.oob_scores_, total_n_estimators)
596                  self.oob_score_ = np.nan
597              else:
598                  self.oob_improvement_ = np.zeros(
599                      (total_n_estimators,), dtype=np.float64
600                  )
601                  self.oob_scores_ = np.zeros((total_n_estimators,), dtype=np.float64)
602                  self.oob_score_ = np.nan

615      def fit(self, X, y, sample_weight=None, monitor=None):
616          """Fit the gradient boosting model. 
617   
618          Parameters 
619          ---------- 
620          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
621              The input samples. Internally, it will be converted to 
622              ``dtype=np.float32`` and if a sparse matrix is provided 
623              to a sparse ``csr_matrix``. 
624   
625          y : array-like of shape (n_samples,) 
626              Target values (strings or integers in classification, real numbers 
627              in regression) 
628              For classification, labels must correspond to classes. 
629   
630          sample_weight : array-like of shape (n_samples,), default=None 
631              Sample weights. If None, then samples are equally weighted. Splits 
632              that would create child nodes with net zero or negative weight are 
633              ignored while searching for a split in each node. In the case of 
634              classification, splits are also ignored if they would result in any 
635              single class carrying a negative weight in either child node. 
636   
637          monitor : callable, default=None 
638              The monitor is called after each iteration with the current 
639              iteration, a reference to the estimator and the local variables of 
640              ``_fit_stages`` as keyword arguments ``callable(i, self, 
641              locals())``. If the callable returns ``True`` the fitting procedure 
642              is stopped. The monitor can be used for various things such as 
643              computing held-out estimates, early stopping, model introspect, and 
644              snapshotting. 
645   
646          Returns 
647          ------- 
648          self : object 
649              Fitted estimator. 
650          """
651          if not self.warm_start:
652              self._clear_state()
653  
654          # Check input
655          # Since check_array converts both X and y to the same dtype, but the
656          # trees use different types for X and y, checking them separately.
657  
658          X, y = validate_data(
659              self,
660              X,
661              y,
662              accept_sparse=["csr", "csc", "coo"],
663              dtype=DTYPE,
664              multi_output=True,
665          )
666          sample_weight_is_none = sample_weight is None
667          sample_weight = _check_sample_weight(sample_weight, X)
668          if sample_weight_is_none:
669              y = self._encode_y(y=y, sample_weight=None)
670          else:
671              y = self._encode_y(y=y, sample_weight=sample_weight)
672          y = column_or_1d(y, warn=True)  # TODO: Is this still required?
673  
674          self._set_max_features()
675  
676          # self.loss is guaranteed to be a string
677          self._loss = self._get_loss(sample_weight=sample_weight)
678  
679          if self.n_iter_no_change is not None:
680              stratify = y if is_classifier(self) else None
681              (
682                  X_train,
683                  X_val,
684                  y_train,
685                  y_val,
686                  sample_weight_train,
687                  sample_weight_val,
688              ) = train_test_split(
689                  X,
690                  y,
691                  sample_weight,
692                  random_state=self.random_state,
693                  test_size=self.validation_fraction,
694                  stratify=stratify,
695              )
696              if is_classifier(self):
697                  if self.n_classes_ != np.unique(y_train).shape[0]:
698                      # We choose to error here. The problem is that the init
699                      # estimator would be trained on y, which has some missing
700                      # classes now, so its predictions would not have the
701                      # correct shape.
702                      raise ValueError(
703                          "The training data after the early stopping split "
704                          "is missing some classes. Try using another random "
705                          "seed."
706                      )
707          else:
708              X_train, y_train, sample_weight_train = X, y, sample_weight
709              X_val = y_val = sample_weight_val = None
710  
711          n_samples = X_train.shape[0]
712  
713          # First time calling fit.
714          if not self._is_fitted():
715              # init state
716              self._init_state()
717  
718              # fit initial model and initialize raw predictions
719              if self.init_ == "zero":
720                  raw_predictions = np.zeros(
721                      shape=(n_samples, self.n_trees_per_iteration_),
722                      dtype=np.float64,
723                  )
724              else:
725                  # XXX clean this once we have a support_sample_weight tag
726                  if sample_weight_is_none:
727                      self.init_.fit(X_train, y_train)
728                  else:
729                      msg = (
730                          "The initial estimator {} does not support sample "
731                          "weights.".format(self.init_.__class__.__name__)
732                      )
733                      try:
734                          self.init_.fit(
735                              X_train, y_train, sample_weight=sample_weight_train
736                          )
737                      except TypeError as e:
738                          if "unexpected keyword argument 'sample_weight'" in str(e):
739                              # regular estimator without SW support
740                              raise ValueError(msg) from e
741                          else:  # regular estimator whose input checking failed
742                              raise
743                      except ValueError as e:
744                          if (
745                              "pass parameters to specific steps of "
746                              "your pipeline using the "
747                              "stepname__parameter" in str(e)
748                          ):  # pipeline
749                              raise ValueError(msg) from e
750                          else:  # regular estimator whose input checking failed
751                              raise
752  
753                  raw_predictions = _init_raw_predictions(
754                      X_train, self.init_, self._loss, is_classifier(self)
755                  )
756  
757              begin_at_stage = 0
758  
759              # The rng state must be preserved if warm_start is True
760              self._rng = check_random_state(self.random_state)
761  
762          # warm start: this is not the first time fit was called
763          else:
764              # add more estimators to fitted model
765              # invariant: warm_start = True
766              if self.n_estimators < self.estimators_.shape[0]:
767                  raise ValueError(
768                      "n_estimators=%d must be larger or equal to "
769                      "estimators_.shape[0]=%d when "
770                      "warm_start==True" % (self.n_estimators, self.estimators_.shape[0])
771                  )
772              begin_at_stage = self.estimators_.shape[0]
773              # The requirements of _raw_predict
774              # are more constrained than fit. It accepts only CSR
775              # matrices. Finite values have already been checked in _validate_data.
776              X_train = check_array(
777                  X_train,
778                  dtype=DTYPE,
779                  order="C",
780                  accept_sparse="csr",
781                  ensure_all_finite=False,
782              )
783              raw_predictions = self._raw_predict(X_train)
784              self._resize_state()
785  
786          # fit the boosting stages
787          n_stages = self._fit_stages(
788              X_train,
789              y_train,
790              raw_predictions,
791              sample_weight_train,
792              self._rng,
793              X_val,
794              y_val,
795              sample_weight_val,
796              begin_at_stage,
797              monitor,
798          )
799  
800          # change shape of arrays after fit (early-stopping or additional ests)
801          if n_stages != self.estimators_.shape[0]:
802              self.estimators_ = self.estimators_[:n_stages]
803              self.train_score_ = self.train_score_[:n_stages]
804              if hasattr(self, "oob_improvement_"):
805                  # OOB scores were computed
806                  self.oob_improvement_ = self.oob_improvement_[:n_stages]
807                  self.oob_scores_ = self.oob_scores_[:n_stages]
808                  self.oob_score_ = self.oob_scores_[-1]
809          self.n_estimators_ = n_stages
810          return self

812      def _fit_stages(
813          self,
814          X,
815          y,
816          raw_predictions,
817          sample_weight,
818          random_state,
819          X_val,
820          y_val,
821          sample_weight_val,
822          begin_at_stage=0,
823          monitor=None,
824      ):
825          """Iteratively fits the stages. 
826   
827          For each stage it computes the progress (OOB, train score) 
828          and delegates to ``_fit_stage``. 
829          Returns the number of stages fit; might differ from ``n_estimators`` 
830          due to early stopping. 
831          """
832          n_samples = X.shape[0]
833          do_oob = self.subsample < 1.0
834          sample_mask = np.ones((n_samples,), dtype=bool)
835          n_inbag = max(1, int(self.subsample * n_samples))
836  
837          if self.verbose:
838              verbose_reporter = VerboseReporter(verbose=self.verbose)
839              verbose_reporter.init(self, begin_at_stage)
840  
841          X_csc = csc_matrix(X) if issparse(X) else None
842          X_csr = csr_matrix(X) if issparse(X) else None
843  
844          if self.n_iter_no_change is not None:
845              loss_history = np.full(self.n_iter_no_change, np.inf)
846              # We create a generator to get the predictions for X_val after
847              # the addition of each successive stage
848              y_val_pred_iter = self._staged_raw_predict(X_val, check_input=False)
849  
850          # Older versions of GBT had its own loss functions. With the new common
851          # private loss function submodule _loss, we often are a factor of 2
852          # away from the old version. Here we keep backward compatibility for
853          # oob_scores_ and oob_improvement_, even if the old way is quite
854          # inconsistent (sometimes the gradient is half the gradient, sometimes
855          # not).
856          if isinstance(
857              self._loss,
858              (
859                  HalfSquaredError,
860                  HalfBinomialLoss,
861              ),
862          ):
863              factor = 2
864          else:
865              factor = 1
866  
867          # perform boosting iterations
868          i = begin_at_stage
869          for i in range(begin_at_stage, self.n_estimators):
870              # subsampling
871              if do_oob:
872                  sample_mask = _random_sample_mask(n_samples, n_inbag, random_state)
873                  y_oob_masked = y[~sample_mask]
874                  sample_weight_oob_masked = sample_weight[~sample_mask]
875                  if i == 0:  # store the initial loss to compute the OOB score
876                      initial_loss = factor * self._loss(
877                          y_true=y_oob_masked,
878                          raw_prediction=raw_predictions[~sample_mask],
879                          sample_weight=sample_weight_oob_masked,
880                      )
881  
882              # fit next stage of trees
883              raw_predictions = self._fit_stage(
884                  i,
885                  X,
886                  y,
887                  raw_predictions,
888                  sample_weight,
889                  sample_mask,
890                  random_state,
891                  X_csc=X_csc,
892                  X_csr=X_csr,
893              )
894  
895              # track loss
896              if do_oob:
897                  self.train_score_[i] = factor * self._loss(
898                      y_true=y[sample_mask],
899                      raw_prediction=raw_predictions[sample_mask],
900                      sample_weight=sample_weight[sample_mask],
901                  )
902                  self.oob_scores_[i] = factor * self._loss(
903                      y_true=y_oob_masked,
904                      raw_prediction=raw_predictions[~sample_mask],
905                      sample_weight=sample_weight_oob_masked,
906                  )
907                  previous_loss = initial_loss if i == 0 else self.oob_scores_[i - 1]
908                  self.oob_improvement_[i] = previous_loss - self.oob_scores_[i]
909                  self.oob_score_ = self.oob_scores_[-1]
910              else:
911                  # no need to fancy index w/ no subsampling
912                  self.train_score_[i] = factor * self._loss(
913                      y_true=y,
914                      raw_prediction=raw_predictions,
915                      sample_weight=sample_weight,
916                  )
917  
918              if self.verbose > 0:
919                  verbose_reporter.update(i, self)
920  
921              if monitor is not None:
922                  early_stopping = monitor(i, self, locals())
923                  if early_stopping:
924                      break
925  
926              # We also provide an early stopping based on the score from
927              # validation set (X_val, y_val), if n_iter_no_change is set
928              if self.n_iter_no_change is not None:
929                  # By calling next(y_val_pred_iter), we get the predictions
930                  # for X_val after the addition of the current stage
931                  validation_loss = factor * self._loss(
932                      y_val, next(y_val_pred_iter), sample_weight_val
933                  )
934  
935                  # Require validation_score to be better (less) than at least
936                  # one of the last n_iter_no_change evaluations
937                  if np.any(validation_loss + self.tol < loss_history):
938                      loss_history[i % len(loss_history)] = validation_loss
939                  else:
940                      break
941  
942          return i + 1
943

970          """Compute raw predictions of ``X`` for each iteration. 
971   
972          This method allows monitoring (i.e. determine error on testing set) 
973          after each stage. 
974   
975          Parameters 
976          ---------- 
977          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
978              The input samples. Internally, it will be converted to 
979              ``dtype=np.float32`` and if a sparse matrix is provided 
980              to a sparse ``csr_matrix``. 
981   
982          check_input : bool, default=True 
983              If False, the input arrays X will not be checked. 
984   
985          Returns 
986          ------- 
987          raw_predictions : generator of ndarray of shape (n_samples, k) 
988              The raw predictions of the input samples. The order of the 
989              classes corresponds to that in the attribute :term:`classes_`. 
990              Regression and binary classification are special cases with 
991              ``k == 1``, otherwise ``k==n_classes``.

995                  self, X, dtype=DTYPE, order="C", accept_sparse="csr", reset=False

997          raw_predictions = self._raw_predict_init(X)
998          for i in range(self.estimators_.shape[0]):
999              predict_stage(self.estimators_, i, X, self.learning_rate, raw_predictions)
1000             yield raw_predictions.copy()

1003     def feature_importances_(self):
1004         """The impurity-based feature importances. 
1005  
1006         The higher, the more important the feature. 
1007         The importance of a feature is computed as the (normalized) 
1008         total reduction of the criterion brought by that feature.  It is also 
1009         known as the Gini importance. 
1010  
1011         Warning: impurity-based feature importances can be misleading for 
1012         high cardinality features (many unique values). See 
1013         :func:`sklearn.inspection.permutation_importance` as an alternative. 
1014  
1015         Returns 
1016         ------- 
1017         feature_importances_ : ndarray of shape (n_features,) 
1018             The values of this array sum to 1, unless all trees are single node 
1019             trees consisting of only the root node, in which case it will be an 
1020             array of zeros. 
1021         """
1022         self._check_initialized()
1023 
1024         relevant_trees = [
1025             tree
1026             for stage in self.estimators_
1027             for tree in stage
1028             if tree.tree_.node_count > 1
1029         ]
1030         if not relevant_trees:
1031             # degenerate case where all trees have only one node
1032             return np.zeros(shape=self.n_features_in_, dtype=np.float64)
1033 
1034         relevant_feature_importances = [
1035             tree.tree_.compute_feature_importances(normalize=False)
1036             for tree in relevant_trees
1037         ]
1038         avg_feature_importances = np.mean(
1039             relevant_feature_importances, axis=0, dtype=np.float64
1040         )
1041         return avg_feature_importances / np.sum(avg_feature_importances)

1043     def _compute_partial_dependence_recursion(self, grid, target_features):
1044         """Fast partial dependence computation. 
1045  
1046         Parameters 
1047         ---------- 
1048         grid : ndarray of shape (n_samples, n_target_features), dtype=np.float32 
1049             The grid points on which the partial dependence should be 
1050             evaluated. 
1051         target_features : ndarray of shape (n_target_features,), dtype=np.intp 
1052             The set of target features for which the partial dependence 
1053             should be evaluated. 
1054  
1055         Returns 
1056         ------- 
1057         averaged_predictions : ndarray of shape \ 
1058                 (n_trees_per_iteration_, n_samples) 
1059             The value of the partial dependence function on each grid point. 
1060         """
1061         if self.init is not None:
1062             warnings.warn(
1063                 "Using recursion method with a non-constant init predictor "
1064                 "will lead to incorrect partial dependence values. "
1065                 "Got init=%s." % self.init,
1066                 UserWarning,
1067             )
1068         grid = np.asarray(grid, dtype=DTYPE, order="C")
1069         n_estimators, n_trees_per_stage = self.estimators_.shape
1070         averaged_predictions = np.zeros(
1071             (n_trees_per_stage, grid.shape[0]), dtype=np.float64, order="C"
1072         )
1073         target_features = np.asarray(target_features, dtype=np.intp, order="C")
1074 
1075         for stage in range(n_estimators):
1076             for k in range(n_trees_per_stage):
1077                 tree = self.estimators_[stage, k].tree_
1078                 tree.compute_partial_dependence(
1079                     grid, target_features, averaged_predictions[k]
1080                 )
1081         averaged_predictions *= self.learning_rate
1082 
1083         return averaged_predictions

1085     def apply(self, X):
1086         """Apply trees in the ensemble to X, return leaf indices. 
1087  
1088         .. versionadded:: 0.17 
1089  
1090         Parameters 
1091         ---------- 
1092         X : {array-like, sparse matrix} of shape (n_samples, n_features) 
1093             The input samples. Internally, its dtype will be converted to 
1094             ``dtype=np.float32``. If a sparse matrix is provided, it will 
1095             be converted to a sparse ``csr_matrix``. 
1096  
1097         Returns 
1098         ------- 
1099         X_leaves : array-like of shape (n_samples, n_estimators, n_classes) 
1100             For each datapoint x in X and for each tree in the ensemble, 
1101             return the index of the leaf x ends up in each estimator. 
1102             In the case of binary classification n_classes is 1. 
1103         """
1104 
1105         self._check_initialized()
1106         X = self.estimators_[0, 0]._validate_X_predict(X, check_input=True)
1107 
1108         # n_classes will be equal to 1 in the binary classification or the
1109         # regression case.
1110         n_estimators, n_classes = self.estimators_.shape
1111         leaves = np.zeros((X.shape[0], n_estimators, n_classes))
1112 
1113         for i in range(n_estimators):
1114             for j in range(n_classes):
1115                 estimator = self.estimators_[i, j]
1116                 leaves[:, i, j] = estimator.apply(X, check_input=False)
1117 
1118         return leaves

1126 class GradientBoostingClassifier(ClassifierMixin, BaseGradientBoosting):
1127     """Gradient Boosting for classification. 
1128  
1129     This algorithm builds an additive model in a forward stage-wise fashion; it 
1130     allows for the optimization of arbitrary differentiable loss functions. In 
1131     each stage ``n_classes_`` regression trees are fit on the negative gradient 
1132     of the loss function, e.g. binary or multiclass log loss. Binary 
1133     classification is a special case where only a single regression tree is 
1134     induced. 
1135  
1136     :class:`~sklearn.ensemble.HistGradientBoostingClassifier` is a much faster variant 
1137     of this algorithm for intermediate and large datasets (`n_samples >= 10_000`) and 
1138     supports monotonic constraints. 
1139  
1140     Read more in the :ref:`User Guide <gradient_boosting>`. 
1141  
1142     Parameters 
1143     ---------- 
1144     loss : {'log_loss', 'exponential'}, default='log_loss' 
1145         The loss function to be optimized. 'log_loss' refers to binomial and 
1146         multinomial deviance, the same as used in logistic regression. 
1147         It is a good choice for classification with probabilistic outputs. 
1148         For loss 'exponential', gradient boosting recovers the AdaBoost algorithm. 
1149  
1150     learning_rate : float, default=0.1 
1151         Learning rate shrinks the contribution of each tree by `learning_rate`. 
1152         There is a trade-off between learning_rate and n_estimators. 
1153         Values must be in the range `[0.0, inf)`. 
1154  
1155     n_estimators : int, default=100 
1156         The number of boosting stages to perform. Gradient boosting 
1157         is fairly robust to over-fitting so a large number usually 
1158         results in better performance. 
1159         Values must be in the range `[1, inf)`. 
1160  
1161     subsample : float, default=1.0 
1162         The fraction of samples to be used for fitting the individual base 
1163         learners. If smaller than 1.0 this results in Stochastic Gradient 
1164         Boosting. `subsample` interacts with the parameter `n_estimators`. 
1165         Choosing `subsample < 1.0` leads to a reduction of variance 
1166         and an increase in bias. 
1167         Values must be in the range `(0.0, 1.0]`. 
1168  
1169     criterion : {'friedman_mse', 'squared_error'}, default='friedman_mse' 
1170         The function to measure the quality of a split. Supported criteria are 
1171         'friedman_mse' for the mean squared error with improvement score by 
1172         Friedman, 'squared_error' for mean squared error. The default value of 
1173         'friedman_mse' is generally the best as it can provide a better 
1174         approximation in some cases. 
1175  
1176         .. versionadded:: 0.18 
1177  
1178     min_samples_split : int or float, default=2 
1179         The minimum number of samples required to split an internal node: 
1180  
1181         - If int, values must be in the range `[2, inf)`. 
1182         - If float, values must be in the range `(0.0, 1.0]` and `min_samples_split` 
1183           will be `ceil(min_samples_split * n_samples)`. 
1184  
1185         .. versionchanged:: 0.18 
1186            Added float values for fractions. 
1187  
1188     min_samples_leaf : int or float, default=1 
1189         The minimum number of samples required to be at a leaf node. 
1190         A split point at any depth will only be considered if it leaves at 
1191         least ``min_samples_leaf`` training samples in each of the left and 
1192         right branches.  This may have the effect of smoothing the model, 
1193         especially in regression. 
1194  
1195         - If int, values must be in the range `[1, inf)`. 
1196         - If float, values must be in the range `(0.0, 1.0)` and `min_samples_leaf` 
1197           will be `ceil(min_samples_leaf * n_samples)`. 
1198  
1199         .. versionchanged:: 0.18 
1200            Added float values for fractions. 
1201  
1202     min_weight_fraction_leaf : float, default=0.0 
1203         The minimum weighted fraction of the sum total of weights (of all 
1204         the input samples) required to be at a leaf node. Samples have 
1205         equal weight when sample_weight is not provided. 
1206         Values must be in the range `[0.0, 0.5]`. 
1207  
1208     max_depth : int or None, default=3 
1209         Maximum depth of the individual regression estimators. The maximum 
1210         depth limits the number of nodes in the tree. Tune this parameter 
1211         for best performance; the best value depends on the interaction 
1212         of the input variables. If None, then nodes are expanded until 
1213         all leaves are pure or until all leaves contain less than 
1214         min_samples_split samples. 
1215         If int, values must be in the range `[1, inf)`. 
1216  
1217     min_impurity_decrease : float, default=0.0 
1218         A node will be split if this split induces a decrease of the impurity 
1219         greater than or equal to this value. 
1220         Values must be in the range `[0.0, inf)`. 
1221  
1222         The weighted impurity decrease equation is the following:: 
1223  
1224             N_t / N * (impurity - N_t_R / N_t * right_impurity 
1225                                 - N_t_L / N_t * left_impurity) 
1226  
1227         where ``N`` is the total number of samples, ``N_t`` is the number of

1509     def _encode_y(self, y, sample_weight):
1510         # encode classes into 0 ... n_classes - 1 and sets attributes classes_
1511         # and n_trees_per_iteration_
1512         check_classification_targets(y)
1513 
1514         label_encoder = LabelEncoder()
1515         encoded_y_int = label_encoder.fit_transform(y)
1516         self.classes_ = label_encoder.classes_
1517         n_classes = self.classes_.shape[0]
1518         # only 1 tree for binary classification. For multiclass classification,
1519         # we build 1 tree per class.
1520         self.n_trees_per_iteration_ = 1 if n_classes <= 2 else n_classes
1521         encoded_y = encoded_y_int.astype(float, copy=False)
1522 
1523         # From here on, it is additional to the HGBT case.
1524         # expose n_classes_ attribute
1525         self.n_classes_ = n_classes
1526         if sample_weight is None:
1527             n_trim_classes = n_classes
1528         else:
1529             n_trim_classes = np.count_nonzero(np.bincount(encoded_y_int, sample_weight))
1530 
1531         if n_trim_classes < 2:
1532             raise ValueError(
1533                 "y contains %d class after sample_weight "
1534                 "trimmed classes with zero weights, while a "
1535                 "minimum of 2 classes are required." % n_trim_classes

1539     def _get_loss(self, sample_weight):
1540         if self.loss == "log_loss":
1541             if self.n_classes_ == 2:
1542                 return HalfBinomialLoss(sample_weight=sample_weight)
1543             else:
1544                 return HalfMultinomialLoss(
1545                     sample_weight=sample_weight, n_classes=self.n_classes_
1546                 )

1551                     f"problem, you have n_classes={self.n_classes_}. "
1552                     "Please use loss='log_loss' instead."
1553                 )
1554             else:

1561         ----------

1734 class GradientBoostingRegressor(RegressorMixin, BaseGradientBoosting):
1735     """Gradient Boosting for regression. 
1736  
1737     This estimator builds an additive model in a forward stage-wise fashion; it 
1738     allows for the optimization of arbitrary differentiable loss functions. In 
1739     each stage a regression tree is fit on the negative gradient of the given 
1740     loss function. 
1741  
1742     :class:`~sklearn.ensemble.HistGradientBoostingRegressor` is a much faster variant 
1743     of this algorithm for intermediate and large datasets (`n_samples >= 10_000`) and 
1744     supports monotonic constraints. 
1745  
1746     Read more in the :ref:`User Guide <gradient_boosting>`. 
1747  
1748     Parameters 
1749     ---------- 
1750     loss : {'squared_error', 'absolute_error', 'huber', 'quantile'}, \ 
1751             default='squared_error' 
1752         Loss function to be optimized. 'squared_error' refers to the squared 
1753         error for regression. 'absolute_error' refers to the absolute error of 
1754         regression and is a robust loss function. 'huber' is a 
1755         combination of the two. 'quantile' allows quantile regression (use 
1756         `alpha` to specify the quantile). 
1757         See 
1758         :ref:`sphx_glr_auto_examples_ensemble_plot_gradient_boosting_quantile.py` 
1759         for an example that demonstrates quantile regression for creating 
1760         prediction intervals with `loss='quantile'`. 
1761  
1762     learning_rate : float, default=0.1 
1763         Learning rate shrinks the contribution of each tree by `learning_rate`. 
1764         There is a trade-off between learning_rate and n_estimators. 
1765         Values must be in the range `[0.0, inf)`. 
1766  
1767     n_estimators : int, default=100 
1768         The number of boosting stages to perform. Gradient boosting 
1769         is fairly robust to over-fitting so a large number usually 
1770         results in better performance. 
1771         Values must be in the range `[1, inf)`. 
1772  
1773     subsample : float, default=1.0 
1774         The fraction of samples to be used for fitting the individual base 
1775         learners. If smaller than 1.0 this results in Stochastic Gradient 
1776         Boosting. `subsample` interacts with the parameter `n_estimators`. 
1777         Choosing `subsample < 1.0` leads to a reduction of variance 
1778         and an increase in bias. 
1779         Values must be in the range `(0.0, 1.0]`. 
1780  
1781     criterion : {'friedman_mse', 'squared_error'}, default='friedman_mse' 
1782         The function to measure the quality of a split. Supported criteria are 
1783         "friedman_mse" for the mean squared error with improvement score by 
1784         Friedman, "squared_error" for mean squared error. The default value of 
1785         "friedman_mse" is generally the best as it can provide a better 
1786         approximation in some cases. 
1787  
1788         .. versionadded:: 0.18 
1789  
1790     min_samples_split : int or float, default=2 
1791         The minimum number of samples required to split an internal node: 
1792  
1793         - If int, values must be in the range `[2, inf)`. 
1794         - If float, values must be in the range `(0.0, 1.0]` and `min_samples_split` 
1795           will be `ceil(min_samples_split * n_samples)`. 
1796  
1797         .. versionchanged:: 0.18 
1798            Added float values for fractions. 
1799  
1800     min_samples_leaf : int or float, default=1 
1801         The minimum number of samples required to be at a leaf node. 
1802         A split point at any depth will only be considered if it leaves at 
1803         least ``min_samples_leaf`` training samples in each of the left and 
1804         right branches.  This may have the effect of smoothing the model, 
1805         especially in regression. 
1806  
1807         - If int, values must be in the range `[1, inf)`. 
1808         - If float, values must be in the range `(0.0, 1.0)` and `min_samples_leaf` 
1809           will be `ceil(min_samples_leaf * n_samples)`. 
1810  
1811         .. versionchanged:: 0.18 
1812            Added float values for fractions. 
1813  
1814     min_weight_fraction_leaf : float, default=0.0 
1815         The minimum weighted fraction of the sum total of weights (of all 
1816         the input samples) required to be at a leaf node. Samples have 
1817         equal weight when sample_weight is not provided. 
1818         Values must be in the range `[0.0, 0.5]`. 
1819  
1820     max_depth : int or None, default=3 
1821         Maximum depth of the individual regression estimators. The maximum 
1822         depth limits the number of nodes in the tree. Tune this parameter 
1823         for best performance; the best value depends on the interaction 
1824         of the input variables. If None, then nodes are expanded until 
1825         all leaves are pure or until all leaves contain less than 
1826         min_samples_split samples. 
1827         If int, values must be in the range `[1, inf)`. 
1828  
1829     min_impurity_decrease : float, default=0.0 
1830         A node will be split if this split induces a decrease of the impurity 
1831         greater than or equal to this value. 
1832         Values must be in the range `[0.0, inf)`. 
1833  
1834         The weighted impurity decrease equation is the following:: 
1835  
1836             N_t / N * (impurity - N_t_R / N_t * right_impurity 
1837                                 - N_t_L / N_t * left_impurity) 
1838  
1839         where ``N`` is the total number of samples, ``N_t`` is the number of 
1840         samples at the current node, ``N_t_L`` is the number of samples in the 
1841         left child, and ``N_t_R`` is the number of samples in the right child. 
1842  
1843         ``N``, ``N_t``, ``N_t_R`` and ``N_t_L`` all refer to the weighted sum, 
1844         if ``sample_weight`` is passed. 
1845  
1846         .. versionadded:: 0.19 
1847  
1848     init : estimator or 'zero', default=None 
1849         An estimator object that is used to compute the initial predictions. 
1850         ``init`` has to provide :term:`fit` and :term:`predict`. If 'zero', the 
1851         initial raw predictions are set to zero. By default a 
1852         ``DummyEstimator`` is used, predicting either the average target value 
1853         (for loss='squared_error'), or a quantile for the other losses. 
1854  
1855     random_state : int, RandomState instance or None, default=None 
1856         Controls the random seed given to each Tree estimator at each 
1857         boosting iteration. 
1858         In addition, it controls the random permutation of the features at 
1859         each split (see Notes for more details). 
1860         It also controls the random splitting of the training data to obtain a 
1861         validation set if `n_iter_no_change` is not None. 
1862         Pass an int for reproducible output across multiple function calls. 
1863         See :term:`Glossary <random_state>`. 
1864  
1865     max_features : {'sqrt', 'log2'}, int or float, default=None 
1866         The number of features to consider when looking for the best split: 
1867  
1868         - If int, values must be in the range `[1, inf)`. 
1869         - If float, values must be in the range `(0.0, 1.0]` and the features 
1870           considered at each split will be `max(1, int(max_features * n_features_in_))`. 
1871         - If "sqrt", then `max_features=sqrt(n_features)`. 
1872         - If "log2", then `max_features=log2(n_features)`. 
1873         - If None, then `max_features=n_features`. 
1874  
1875         Choosing `max_features < n_features` leads to a reduction of variance 
1876         and an increase in bias. 
1877  
1878         Note: the search for a split does not stop until at least one 
1879         valid partition of the node samples is found, even if it requires to 
1880         effectively inspect more than ``max_features`` features. 
1881  
1882     alpha : float, default=0.9 
1883         The alpha-quantile of the huber loss function and the quantile 
1884         loss function. Only if ``loss='huber'`` or ``loss='quantile'``. 
1885         Values must be in the range `(0.0, 1.0)`. 
1886  
1887     verbose : int, default=0 
1888         Enable verbose output. If 1 then it prints progress and performance 
1889         once in a while (the more trees the lower the frequency). If greater 
1890         than 1 then it prints progress and performance for every tree. 
1891         Values must be in the range `[0, inf)`. 
1892  
1893     max_leaf_nodes : int, default=None 
1894         Grow trees with ``max_leaf_nodes`` in best-first fashion. 
1895         Best nodes are defined as relative reduction in impurity. 
1896         Values must be in the range `[2, inf)`. 
1897         If None, then unlimited number of leaf nodes. 
1898  
1899     warm_start : bool, default=False 
1900         When set to ``True``, reuse the solution of the previous call to fit 
1901         and add more estimators to the ensemble, otherwise, just erase the 
1902         previous solution. See :term:`the Glossary <warm_start>`. 
1903  
1904     validation_fraction : float, default=0.1 
1905         The proportion of training data to set aside as validation set for 
1906         early stopping. Values must be in the range `(0.0, 1.0)`. 
1907         Only used if ``n_iter_no_change`` is set to an integer. 
1908  
1909         .. versionadded:: 0.20 
1910  
1911     n_iter_no_change : int, default=None 
1912         ``n_iter_no_change`` is used to decide if early stopping will be used 
1913         to terminate training when validation score is not improving. By 
1914         default it is set to None to disable early stopping. If set to a 
1915         number, it will set aside ``validation_fraction`` size of the training 
1916         data as validation and terminate training when validation score is not 
1917         improving in all of the previous ``n_iter_no_change`` numbers of 
1918         iterations. 
1919         Values must be in the range `[1, inf)`. 
1920         See 
1921         :ref:`sphx_glr_auto_examples_ensemble_plot_gradient_boosting_early_stopping.py`. 
1922  
1923         .. versionadded:: 0.20 
1924  
1925     tol : float, default=1e-4 
1926         Tolerance for the early stopping. When the loss is not improving 
1927         by at least tol for ``n_iter_no_change`` iterations (if set to a 
1928         number), the training stops. 
1929         Values must be in the range `[0.0, inf)`. 
1930  
1931         .. versionadded:: 0.20 
1932  
1933     ccp_alpha : non-negative float, default=0.0 
1934         Complexity parameter used for Minimal Cost-Complexity Pruning. The 
1935         subtree with the largest cost complexity that is smaller than 
1936         ``ccp_alpha`` will be chosen. By default, no pruning is performed. 
1937         Values must be in the range `[0.0, inf)`. 
1938         See :ref:`minimal_cost_complexity_pruning` for details. See 
1939         :ref:`sphx_glr_auto_examples_tree_plot_cost_complexity_pruning.py` 
1940         for an example of such pruning.
