48   def _set_order(X, y, order="C"):
49       """Change the order of X and y if necessary. 
50    
51       Parameters 
52       ---------- 
53       X : {array-like, sparse matrix} of shape (n_samples, n_features) 
54           Training data. 
55    
56       y : ndarray of shape (n_samples,) 
57           Target values. 
58    
59       order : {None, 'C', 'F'} 
60           If 'C', dense arrays are returned as C-ordered, sparse matrices in csr 
61           format. If 'F', dense arrays are return as F-ordered, sparse matrices 
62           in csc format. 
63    
64       Returns 
65       ------- 
66       X : {array-like, sparse matrix} of shape (n_samples, n_features) 
67           Training data with guaranteed order. 
68    
69       y : ndarray of shape (n_samples,) 
70           Target values with guaranteed order. 
71       """
72       if order not in [None, "C", "F"]:
73           raise ValueError(
74               "Unknown value for order. Got {} instead of None, 'C' or 'F'.".format(order)
75           )
76       sparse_X = sparse.issparse(X)
77       sparse_y = sparse.issparse(y)
78       if order is not None:
79           sparse_format = "csc" if order == "F" else "csr"
80           if sparse_X:
81               X = X.asformat(sparse_format, copy=False)
82           else:
83               X = np.asarray(X, order=order)
84           if sparse_y:
85               y = y.asformat(sparse_format)
86           else:
87               y = np.asarray(y, order=order)
88       return X, y

95   def _alpha_grid(
96       X,
97       y,
98       Xy=None,
99       l1_ratio=1.0,
100      fit_intercept=True,
101      eps=1e-3,
102      n_alphas=100,
103      copy_X=True,
104      sample_weight=None,
105  ):
106      """Compute the grid of alpha values for elastic net parameter search 
107   
108      Parameters 
109      ---------- 
110      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
111          Training data. Pass directly as Fortran-contiguous data to avoid 
112          unnecessary memory duplication 
113   
114      y : ndarray of shape (n_samples,) or (n_samples, n_outputs) 
115          Target values 
116   
117      Xy : array-like of shape (n_features,) or (n_features, n_outputs),\ 
118           default=None 
119          Xy = np.dot(X.T, y) that can be precomputed. 
120   
121      l1_ratio : float, default=1.0 
122          The elastic net mixing parameter, with ``0 < l1_ratio <= 1``. 
123          For ``l1_ratio = 0`` the penalty is an L2 penalty. (currently not 
124          supported) ``For l1_ratio = 1`` it is an L1 penalty. For 
125          ``0 < l1_ratio <1``, the penalty is a combination of L1 and L2. 
126   
127      eps : float, default=1e-3 
128          Length of the path. ``eps=1e-3`` means that 
129          ``alpha_min / alpha_max = 1e-3`` 
130   
131      n_alphas : int, default=100 
132          Number of alphas along the regularization path 
133   
134      fit_intercept : bool, default=True 
135          Whether to fit an intercept or not 
136   
137      copy_X : bool, default=True 
138          If ``True``, X will be copied; else, it may be overwritten. 
139   
140      sample_weight : ndarray of shape (n_samples,), default=None 
141      """
142      if l1_ratio == 0:
143          raise ValueError(
144              "Automatic alpha grid generation is not supported for"
145              " l1_ratio=0. Please supply a grid by providing "
146              "your estimator with the appropriate `alphas=` "
147              "argument."
148          )
149      if Xy is not None:
150          Xyw = Xy
151      else:
152          X, y, X_offset, _, _ = _preprocess_data(
153              X,
154              y,
155              fit_intercept=fit_intercept,
156              copy=copy_X,
157              sample_weight=sample_weight,
158              check_input=False,
159          )
160          if sample_weight is not None:
161              if y.ndim > 1:
162                  yw = y * sample_weight.reshape(-1, 1)
163              else:
164                  yw = y * sample_weight
165          else:
166              yw = y
167          if sparse.issparse(X):
168              Xyw = safe_sparse_dot(X.T, yw, dense_output=True) - np.sum(yw) * X_offset
169          else:
170              Xyw = np.dot(X.T, yw)
171  
172      if Xyw.ndim == 1:
173          Xyw = Xyw[:, np.newaxis]
174      if sample_weight is not None:
175          n_samples = sample_weight.sum()
176      else:
177          n_samples = X.shape[0]
178      alpha_max = np.sqrt(np.sum(Xyw**2, axis=1)).max() / (n_samples * l1_ratio)
179  
180      if alpha_max <= np.finfo(np.float64).resolution:
181          return np.full(n_alphas, np.finfo(np.float64).resolution)
182  
183      return np.geomspace(alpha_max, alpha_max * eps, num=n_alphas)

203  def lasso_path(
204      X,
205      y,
206      *,
207      eps=1e-3,
208      n_alphas=100,
209      alphas=None,
210      precompute="auto",
211      Xy=None,
212      copy_X=True,
213      coef_init=None,
214      verbose=False,
215      return_n_iter=False,
216      positive=False,
217      **params,
218  ):
219      """Compute Lasso path with coordinate descent. 
220   
221      The Lasso optimization function varies for mono and multi-outputs. 
222   
223      For mono-output tasks it is:: 
224   
225          (1 / (2 * n_samples)) * ||y - Xw||^2_2 + alpha * ||w||_1 
226   
227      For multi-output tasks it is:: 
228   
229          (1 / (2 * n_samples)) * ||Y - XW||^2_Fro + alpha * ||W||_21 
230   
231      Where:: 
232   
233          ||W||_21 = \\sum_i \\sqrt{\\sum_j w_{ij}^2} 
234   
235      i.e. the sum of norm of each row. 
236   
237      Read more in the :ref:`User Guide <lasso>`. 
238   
239      Parameters 
240      ---------- 
241      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
242          Training data. Pass directly as Fortran-contiguous data to avoid 
243          unnecessary memory duplication. If ``y`` is mono-output then ``X`` 
244          can be sparse. 
245   
246      y : {array-like, sparse matrix} of shape (n_samples,) or \ 
247          (n_samples, n_targets) 
248          Target values. 
249   
250      eps : float, default=1e-3 
251          Length of the path. ``eps=1e-3`` means that 
252          ``alpha_min / alpha_max = 1e-3``. 
253   
254      n_alphas : int, default=100 
255          Number of alphas along the regularization path. 
256   
257      alphas : array-like, default=None 
258          List of alphas where to compute the models. 
259          If ``None`` alphas are set automatically. 
260   
261      precompute : 'auto', bool or array-like of shape \ 
262              (n_features, n_features), default='auto' 
263          Whether to use a precomputed Gram matrix to speed up 
264          calculations. If set to ``'auto'`` let us decide. The Gram 
265          matrix can also be passed as argument. 
266   
267      Xy : array-like of shape (n_features,) or (n_features, n_targets),\ 
268           default=None 
269          Xy = np.dot(X.T, y) that can be precomputed. It is useful 
270          only when the Gram matrix is precomputed. 
271   
272      copy_X : bool, default=True 
273          If ``True``, X will be copied; else, it may be overwritten. 
274   
275      coef_init : array-like of shape (n_features, ), default=None 
276          The initial values of the coefficients. 
277   
278      verbose : bool or int, default=False 
279          Amount of verbosity. 
280   
281      return_n_iter : bool, default=False 
282          Whether to return the number of iterations or not. 
283   
284      positive : bool, default=False 
285          If set to True, forces coefficients to be positive. 
286          (Only allowed when ``y.ndim == 1``). 
287   
288      **params : kwargs 
289          Keyword arguments passed to the coordinate descent solver. 
290   
291      Returns 
292      ------- 
293      alphas : ndarray of shape (n_alphas,) 
294          The alphas along the path where models are computed. 
295   
296      coefs : ndarray of shape (n_features, n_alphas) or \ 
297              (n_targets, n_features, n_alphas) 
298          Coefficients along the path. 
299   
300      dual_gaps : ndarray of shape (n_alphas,) 
301          The dual gaps at the end of the optimization for each alpha. 
302   
303      n_iters : list of int 
304          The number of iterations taken by the coordinate descent optimizer to 
305          reach the specified tolerance for each alpha. 
306   
307      See Also 
308      -------- 
309      lars_path : Compute Least Angle Regression or Lasso path using LARS 
310          algorithm. 
311      Lasso : The Lasso is a linear model that estimates sparse coefficients. 
312      LassoLars : Lasso model fit with Least Angle Regression a.k.a. Lars. 
313      LassoCV : Lasso linear model with iterative fitting along a regularization 
314          path. 
315      LassoLarsCV : Cross-validated Lasso using the LARS algorithm. 
316      sklearn.decomposition.sparse_encode : Estimator that can be used to 
317          transform signals into sparse linear combination of atoms from a fixed. 
318   
319      Notes 
320      ----- 
321      For an example, see 
322      :ref:`examples/linear_model/plot_lasso_lasso_lars_elasticnet_path.py 
323      <sphx_glr_auto_examples_linear_model_plot_lasso_lasso_lars_elasticnet_path.py>`. 
324   
325      To avoid unnecessary memory duplication the X argument of the fit method 
326      should be directly passed as a Fortran-contiguous numpy array. 
327   
328      Note that in certain cases, the Lars solver may be significantly 
329      faster to implement this functionality. In particular, linear 
330      interpolation can be used to retrieve model coefficients between the 
331      values output by lars_path 
332   
333      Examples 
334      -------- 
335   
336      Comparing lasso_path and lars_path with interpolation: 
337   
338      >>> import numpy as np 
339      >>> from sklearn.linear_model import lasso_path 
340      >>> X = np.array([[1, 2, 3.1], [2.3, 5.4, 4.3]]).T 
341      >>> y = np.array([1, 2, 3.1]) 
342      >>> # Use lasso_path to compute a coefficient path 
343      >>> _, coef_path, _ = lasso_path(X, y, alphas=[5., 1., .5]) 
344      >>> print(coef_path) 
345      [[0.         0.         0.46874778] 
346       [0.2159048  0.4425765  0.23689075]] 
347   
348      >>> # Now use lars_path and 1D linear interpolation to compute the 
349      >>> # same path 
350      >>> from sklearn.linear_model import lars_path 
351      >>> alphas, active, coef_path_lars = lars_path(X, y, method='lasso') 
352      >>> from scipy import interpolate 
353      >>> coef_path_continuous = interpolate.interp1d(alphas[::-1], 
354      ...                                             coef_path_lars[:, ::-1]) 
355      >>> print(coef_path_continuous([5., 1., .5])) 
356      [[0.         0.         0.46915237] 
357       [0.2159048  0.4425765  0.23668876]] 
358      """
359      return enet_path(
360          X,
361          y,
362          l1_ratio=1.0,
363          eps=eps,
364          n_alphas=n_alphas,
365          alphas=alphas,
366          precompute=precompute,
367          Xy=Xy,
368          copy_X=copy_X,
369          coef_init=coef_init,
370          verbose=verbose,
371          positive=positive,
372          return_n_iter=return_n_iter,
373          **params,
374      )

396  def enet_path(
397      X,
398      y,
399      *,
400      l1_ratio=0.5,
401      eps=1e-3,
402      n_alphas=100,
403      alphas=None,
404      precompute="auto",
405      Xy=None,
406      copy_X=True,
407      coef_init=None,
408      verbose=False,
409      return_n_iter=False,
410      positive=False,
411      check_input=True,
412      **params,
413  ):
414      """Compute elastic net path with coordinate descent. 
415   
416      The elastic net optimization function varies for mono and multi-outputs. 
417   
418      For mono-output tasks it is:: 
419   
420          1 / (2 * n_samples) * ||y - Xw||^2_2 
421          + alpha * l1_ratio * ||w||_1 
422          + 0.5 * alpha * (1 - l1_ratio) * ||w||^2_2 
423   
424      For multi-output tasks it is:: 
425   
426          (1 / (2 * n_samples)) * ||Y - XW||_Fro^2 
427          + alpha * l1_ratio * ||W||_21 
428          + 0.5 * alpha * (1 - l1_ratio) * ||W||_Fro^2 
429   
430      Where:: 
431   
432          ||W||_21 = \\sum_i \\sqrt{\\sum_j w_{ij}^2} 
433   
434      i.e. the sum of norm of each row. 
435   
436      Read more in the :ref:`User Guide <elastic_net>`. 
437   
438      Parameters 
439      ---------- 
440      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
441          Training data. Pass directly as Fortran-contiguous data to avoid 
442          unnecessary memory duplication. If ``y`` is mono-output then ``X`` 
443          can be sparse. 
444   
445      y : {array-like, sparse matrix} of shape (n_samples,) or \ 
446          (n_samples, n_targets) 
447          Target values. 
448   
449      l1_ratio : float, default=0.5 
450          Number between 0 and 1 passed to elastic net (scaling between 
451          l1 and l2 penalties). ``l1_ratio=1`` corresponds to the Lasso. 
452   
453      eps : float, default=1e-3 
454          Length of the path. ``eps=1e-3`` means that 
455          ``alpha_min / alpha_max = 1e-3``. 
456   
457      n_alphas : int, default=100 
458          Number of alphas along the regularization path. 
459   
460      alphas : array-like, default=None 
461          List of alphas where to compute the models. 
462          If None alphas are set automatically. 
463   
464      precompute : 'auto', bool or array-like of shape \ 
465              (n_features, n_features), default='auto' 
466          Whether to use a precomputed Gram matrix to speed up 
467          calculations. If set to ``'auto'`` let us decide. The Gram 
468          matrix can also be passed as argument. 
469   
470      Xy : array-like of shape (n_features,) or (n_features, n_targets),\ 
471           default=None 
472          Xy = np.dot(X.T, y) that can be precomputed. It is useful 
473          only when the Gram matrix is precomputed. 
474   
475      copy_X : bool, default=True 
476          If ``True``, X will be copied; else, it may be overwritten. 
477   
478      coef_init : array-like of shape (n_features, ), default=None 
479          The initial values of the coefficients. 
480   
481      verbose : bool or int, default=False 
482          Amount of verbosity. 
483   
484      return_n_iter : bool, default=False 
485          Whether to return the number of iterations or not. 
486   
487      positive : bool, default=False 
488          If set to True, forces coefficients to be positive. 
489          (Only allowed when ``y.ndim == 1``). 
490   
491      check_input : bool, default=True 
492          If set to False, the input validation checks are skipped (including the 
493          Gram matrix when provided). It is assumed that they are handled 
494          by the caller. 
495   
496      **params : kwargs 
497          Keyword arguments passed to the coordinate descent solver. 
498   
499      Returns 
500      ------- 
501      alphas : ndarray of shape (n_alphas,) 
502          The alphas along the path where models are computed. 
503   
504      coefs : ndarray of shape (n_features, n_alphas) or \ 
505              (n_targets, n_features, n_alphas) 
506          Coefficients along the path. 
507   
508      dual_gaps : ndarray of shape (n_alphas,) 
509          The dual gaps at the end of the optimization for each alpha. 
510   
511      n_iters : list of int 
512          The number of iterations taken by the coordinate descent optimizer to 
513          reach the specified tolerance for each alpha. 
514          (Is returned when ``return_n_iter`` is set to True). 
515   
516      See Also 
517      -------- 
518      MultiTaskElasticNet : Multi-task ElasticNet model trained with L1/L2 mixed-norm \ 
519      as regularizer. 
520      MultiTaskElasticNetCV : Multi-task L1/L2 ElasticNet with built-in cross-validation. 
521      ElasticNet : Linear regression with combined L1 and L2 priors as regularizer. 
522      ElasticNetCV : Elastic Net model with iterative fitting along a regularization path. 
523   
524      Notes 
525      ----- 
526      For an example, see 
527      :ref:`examples/linear_model/plot_lasso_lasso_lars_elasticnet_path.py 
528      <sphx_glr_auto_examples_linear_model_plot_lasso_lasso_lars_elasticnet_path.py>`. 
529   
530      Examples 
531      -------- 
532      >>> from sklearn.linear_model import enet_path 
533      >>> from sklearn.datasets import make_regression 
534      >>> X, y, true_coef = make_regression( 
535      ...    n_samples=100, n_features=5, n_informative=2, coef=True, random_state=0 
536      ... ) 
537      >>> true_coef 
538      array([ 0.        ,  0.        ,  0.        , 97.9..., 45.7...]) 
539      >>> alphas, estimated_coef, _ = enet_path(X, y, n_alphas=3) 
540      >>> alphas.shape 
541      (3,) 
542      >>> estimated_coef 
543       array([[ 0.        ,  0.78...,  0.56...], 
544              [ 0.        ,  1.12...,  0.61...], 
545              [-0.        , -2.12..., -1.12...], 
546              [ 0.        , 23.04..., 88.93...], 
547              [ 0.        , 10.63..., 41.56...]]) 
548      """
549      X_offset_param = params.pop("X_offset", None)
550      X_scale_param = params.pop("X_scale", None)
551      sample_weight = params.pop("sample_weight", None)
552      tol = params.pop("tol", 1e-4)
553      max_iter = params.pop("max_iter", 1000)
554      random_state = params.pop("random_state", None)
555      selection = params.pop("selection", "cyclic")
556  
557      if len(params) > 0:
558          raise ValueError("Unexpected parameters in params", params.keys())
559  
560      # We expect X and y to be already Fortran ordered when bypassing
561      # checks
562      if check_input:
563          X = check_array(
564              X,
565              accept_sparse="csc",
566              dtype=[np.float64, np.float32],
567              order="F",
568              copy=copy_X,
569          )
570          y = check_array(
571              y,
572              accept_sparse="csc",
573              dtype=X.dtype.type,
574              order="F",
575              copy=False,
576              ensure_2d=False,
577          )
578          if Xy is not None:
579              # Xy should be a 1d contiguous array or a 2D C ordered array
580              Xy = check_array(
581                  Xy, dtype=X.dtype.type, order="C", copy=False, ensure_2d=False
582              )
583  
584      n_samples, n_features = X.shape
585  
586      multi_output = False
587      if y.ndim != 1:
588          multi_output = True
589          n_targets = y.shape[1]
590  
591      if multi_output and positive:
592          raise ValueError("positive=True is not allowed for multi-output (y.ndim != 1)")
593  
594      # MultiTaskElasticNet does not support sparse matrices
595      if not multi_output and sparse.issparse(X):
596          if X_offset_param is not None:
597              # As sparse matrices are not actually centered we need this to be passed to
598              # the CD solver.
599              X_sparse_scaling = X_offset_param / X_scale_param
600              X_sparse_scaling = np.asarray(X_sparse_scaling, dtype=X.dtype)
601          else:
602              X_sparse_scaling = np.zeros(n_features, dtype=X.dtype)
603  
604      # X should have been passed through _pre_fit already if function is called
605      # from ElasticNet.fit
606      if check_input:
607          X, y, _, _, _, precompute, Xy = _pre_fit(
608              X,
609              y,
610              Xy,
611              precompute,
612              fit_intercept=False,
613              copy=False,
614              check_input=check_input,
615          )
616      if alphas is None:
617          # No need to normalize of fit_intercept: it has been done
618          # above
619          alphas = _alpha_grid(
620              X,
621              y,
622              Xy=Xy,
623              l1_ratio=l1_ratio,
624              fit_intercept=False,
625              eps=eps,
626              n_alphas=n_alphas,
627              copy_X=False,
628          )
629      elif len(alphas) > 1:
630          alphas = np.sort(alphas)[::-1]  # make sure alphas are properly ordered
631  
632      n_alphas = len(alphas)
633      dual_gaps = np.empty(n_alphas)
634      n_iters = []
635  
636      rng = check_random_state(random_state)
637      if selection not in ["random", "cyclic"]:
638          raise ValueError("selection should be either random or cyclic.")
639      random = selection == "random"
640  
641      if not multi_output:
642          coefs = np.empty((n_features, n_alphas), dtype=X.dtype)
643      else:
644          coefs = np.empty((n_targets, n_features, n_alphas), dtype=X.dtype)
645  
646      if coef_init is None:
647          coef_ = np.zeros(coefs.shape[:-1], dtype=X.dtype, order="F")
648      else:
649          coef_ = np.asfortranarray(coef_init, dtype=X.dtype)
650  
651      for i, alpha in enumerate(alphas):
652          # account for n_samples scaling in objectives between here and cd_fast
653          l1_reg = alpha * l1_ratio * n_samples
654          l2_reg = alpha * (1.0 - l1_ratio) * n_samples
655          if not multi_output and sparse.issparse(X):
656              model = cd_fast.sparse_enet_coordinate_descent(
657                  w=coef_,
658                  alpha=l1_reg,
659                  beta=l2_reg,
660                  X_data=X.data,
661                  X_indices=X.indices,
662                  X_indptr=X.indptr,
663                  y=y,
664                  sample_weight=sample_weight,
665                  X_mean=X_sparse_scaling,
666                  max_iter=max_iter,
667                  tol=tol,
668                  rng=rng,
669                  random=random,
670                  positive=positive,
671              )
672          elif multi_output:
673              model = cd_fast.enet_coordinate_descent_multi_task(
674                  coef_, l1_reg, l2_reg, X, y, max_iter, tol, rng, random
675              )
676          elif isinstance(precompute, np.ndarray):
677              # We expect precompute to be already Fortran ordered when bypassing
678              # checks
679              if check_input:
680                  precompute = check_array(precompute, dtype=X.dtype.type, order="C")
681              model = cd_fast.enet_coordinate_descent_gram(
682                  coef_,
683                  l1_reg,
684                  l2_reg,
685                  precompute,
686                  Xy,
687                  y,
688                  max_iter,
689                  tol,
690                  rng,
691                  random,
692                  positive,
693              )
694          elif precompute is False:
695              model = cd_fast.enet_coordinate_descent(
696                  coef_, l1_reg, l2_reg, X, y, max_iter, tol, rng, random, positive
697              )
698          else:
699              raise ValueError(
700                  "Precompute should be one of True, False, 'auto' or array-like. Got %r"
701                  % precompute
702              )
703          coef_, dual_gap_, eps_, n_iter_ = model
704          coefs[..., i] = coef_
705          # we correct the scale of the returned dual gap, as the objective
706          # in cd_fast is n_samples * the objective in this docstring.
707          dual_gaps[i] = dual_gap_ / n_samples
708          n_iters.append(n_iter_)
709  
710          if verbose:
711              if verbose > 2:
712                  print(model)
713              elif verbose > 1:
714                  print("Path: %03i out of %03i" % (i, n_alphas))
715              else:
716                  sys.stderr.write(".")
717  
718      if return_n_iter:
719          return alphas, coefs, dual_gaps, n_iters
720      return alphas, coefs, dual_gaps

727  class ElasticNet(MultiOutputMixin, RegressorMixin, LinearModel):
728      """Linear regression with combined L1 and L2 priors as regularizer. 
729   
730      Minimizes the objective function:: 
731   
732              1 / (2 * n_samples) * ||y - Xw||^2_2 
733              + alpha * l1_ratio * ||w||_1 
734              + 0.5 * alpha * (1 - l1_ratio) * ||w||^2_2 
735   
736      If you are interested in controlling the L1 and L2 penalty 
737      separately, keep in mind that this is equivalent to:: 
738   
739              a * ||w||_1 + 0.5 * b * ||w||_2^2 
740   
741      where:: 
742   
743              alpha = a + b and l1_ratio = a / (a + b) 
744   
745      The parameter l1_ratio corresponds to alpha in the glmnet R package while 
746      alpha corresponds to the lambda parameter in glmnet. Specifically, l1_ratio 
747      = 1 is the lasso penalty. Currently, l1_ratio <= 0.01 is not reliable, 
748      unless you supply your own sequence of alpha. 
749   
750      Read more in the :ref:`User Guide <elastic_net>`. 
751   
752      Parameters 
753      ---------- 
754      alpha : float, default=1.0 
755          Constant that multiplies the penalty terms. Defaults to 1.0. 
756          See the notes for the exact mathematical meaning of this 
757          parameter. ``alpha = 0`` is equivalent to an ordinary least square, 
758          solved by the :class:`LinearRegression` object. For numerical 
759          reasons, using ``alpha = 0`` with the ``Lasso`` object is not advised. 
760          Given this, you should use the :class:`LinearRegression` object. 
761   
762      l1_ratio : float, default=0.5 
763          The ElasticNet mixing parameter, with ``0 <= l1_ratio <= 1``. For 
764          ``l1_ratio = 0`` the penalty is an L2 penalty. ``For l1_ratio = 1`` it 
765          is an L1 penalty.  For ``0 < l1_ratio < 1``, the penalty is a 
766          combination of L1 and L2. 
767   
768      fit_intercept : bool, default=True 
769          Whether the intercept should be estimated or not. If ``False``, the 
770          data is assumed to be already centered. 
771   
772      precompute : bool or array-like of shape (n_features, n_features),\ 
773                   default=False 
774          Whether to use a precomputed Gram matrix to speed up 
775          calculations. The Gram matrix can also be passed as argument. 
776          For sparse input this option is always ``False`` to preserve sparsity. 
777          Check :ref:`an example on how to use a precomputed Gram Matrix in ElasticNet 
778          <sphx_glr_auto_examples_linear_model_plot_elastic_net_precomputed_gram_matrix_with_weighted_samples.py>` 
779          for details. 
780   
781      max_iter : int, default=1000 
782          The maximum number of iterations. 
783   
784      copy_X : bool, default=True 
785          If ``True``, X will be copied; else, it may be overwritten. 
786   
787      tol : float, default=1e-4 
788          The tolerance for the optimization: if the updates are 
789          smaller than ``tol``, the optimization code checks the 
790          dual gap for optimality and continues until it is smaller 
791          than ``tol``, see Notes below. 
792   
793      warm_start : bool, default=False 
794          When set to ``True``, reuse the solution of the previous call to fit as 
795          initialization, otherwise, just erase the previous solution. 
796          See :term:`the Glossary <warm_start>`. 
797   
798      positive : bool, default=False 
799          When set to ``True``, forces the coefficients to be positive. 
800   
801      random_state : int, RandomState instance, default=None 
802          The seed of the pseudo random number generator that selects a random 
803          feature to update. Used when ``selection`` == 'random'. 
804          Pass an int for reproducible output across multiple function calls. 
805          See :term:`Glossary <random_state>`. 
806   
807      selection : {'cyclic', 'random'}, default='cyclic' 
808          If set to 'random', a random coefficient is updated every iteration 
809          rather than looping over features sequentially by default. This 
810          (setting to 'random') often leads to significantly faster convergence 
811          especially when tol is higher than 1e-4. 
812   
813      Attributes 
814      ---------- 
815      coef_ : ndarray of shape (n_features,) or (n_targets, n_features) 
816          Parameter vector (w in the cost function formula). 
817   
818      sparse_coef_ : sparse matrix of shape (n_features,) or \ 
819              (n_targets, n_features) 
820          Sparse representation of the `coef_`. 
821   
822      intercept_ : float or ndarray of shape (n_targets,) 
823          Independent term in decision function. 
824   
825      n_iter_ : list of int 
826          Number of iterations run by the coordinate descent solver to reach 
827          the specified tolerance. 
828   
829      dual_gap_ : float or ndarray of shape (n_targets,) 
830          Given param alpha, the dual gaps at the end of the optimization, 
831          same shape as each observation of y. 
832   
833      n_features_in_ : int 
834          Number of features seen during :term:`fit`. 
835   
836          .. versionadded:: 0.24 
837   
838      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
839          Names of features seen during :term:`fit`. Defined only when `X` 
840          has feature names that are all strings. 
841   
842          .. versionadded:: 1.0 
843   
844      See Also 
845      -------- 
846      ElasticNetCV : Elastic net model with best model selection by 
847          cross-validation. 
848      SGDRegressor : Implements elastic net regression with incremental training. 
849      SGDClassifier : Implements logistic regression with elastic net penalty 
850          (``SGDClassifier(loss="log_loss", penalty="elasticnet")``). 
851   
852      Notes 
853      ----- 
854      To avoid unnecessary memory duplication the X argument of the fit method 
855      should be directly passed as a Fortran-contiguous numpy array. 
856   
857      The precise stopping criteria based on `tol` are the following: First, check that 
858      that maximum coordinate update, i.e. :math:`\\max_j |w_j^{new} - w_j^{old}|` 
859      is smaller than `tol` times the maximum absolute coefficient, :math:`\\max_j |w_j|`. 
860      If so, then additionally check whether the dual gap is smaller than `tol` times 
861      :math:`||y||_2^2 / n_{\text{samples}}`. 
862   
863      Examples 
864      -------- 
865      >>> from sklearn.linear_model import ElasticNet

867

870      >>> regr.fit(X, y) 
871      ElasticNet(random_state=0) 
872      >>> print(regr.coef_) 
873      [18.83816048 64.55968825] 
874      >>> print(regr.intercept_) 
875      1.451... 
876      >>> print(regr.predict([[0, 0]])) 
877      [1.451...]

927      @_fit_context(prefer_skip_nested_validation=True)
928      def fit(self, X, y, sample_weight=None, check_input=True):
929          """Fit model with coordinate descent. 
930   
931          Parameters 
932          ---------- 
933          X : {ndarray, sparse matrix, sparse array} of (n_samples, n_features) 
934              Data. 
935   
936              Note that large sparse matrices and arrays requiring `int64` 
937              indices are not accepted. 
938   
939          y : ndarray of shape (n_samples,) or (n_samples, n_targets) 
940              Target. Will be cast to X's dtype if necessary. 
941   
942          sample_weight : float or array-like of shape (n_samples,), default=None 
943              Sample weights. Internally, the `sample_weight` vector will be 
944              rescaled to sum to `n_samples`. 
945   
946              .. versionadded:: 0.23 
947   
948          check_input : bool, default=True 
949              Allow to bypass several input checking. 
950              Don't use this parameter unless you know what you do. 
951   
952          Returns 
953          ------- 
954          self : object 
955              Fitted estimator. 
956   
957          Notes 
958          ----- 
959          Coordinate descent is an algorithm that considers each column of 
960          data at a time hence it will automatically convert the X input 
961          as a Fortran-contiguous numpy array if necessary. 
962   
963          To avoid memory re-allocation it is advised to allocate the 
964          initial data in memory directly using that format. 
965          """
966          if self.alpha == 0:
967              warnings.warn(
968                  (
969                      "With alpha=0, this algorithm does not converge "
970                      "well. You are advised to use the LinearRegression "
971                      "estimator"
972                  ),
973                  stacklevel=2,
974              )
975  
976          # Remember if X is copied
977          X_copied = False
978          # We expect X and y to be float64 or float32 Fortran ordered arrays
979          # when bypassing checks
980          if check_input:
981              X_copied = self.copy_X and self.fit_intercept
982              X, y = validate_data(
983                  self,
984                  X,
985                  y,
986                  accept_sparse="csc",
987                  order="F",
988                  dtype=[np.float64, np.float32],
989                  force_writeable=True,
990                  accept_large_sparse=False,
991                  copy=X_copied,
992                  multi_output=True,
993                  y_numeric=True,
994              )
995              y = check_array(
996                  y, order="F", copy=False, dtype=X.dtype.type, ensure_2d=False
997              )
998  
999          n_samples, n_features = X.shape
1000         alpha = self.alpha
1001 
1002         if isinstance(sample_weight, numbers.Number):
1003             sample_weight = None
1004         if sample_weight is not None:
1005             if check_input:
1006                 sample_weight = _check_sample_weight(sample_weight, X, dtype=X.dtype)
1007             # TLDR: Rescale sw to sum up to n_samples.
1008             # Long: The objective function of Enet
1009             #
1010             #    1/2 * np.average(squared error, weights=sw)
1011             #    + alpha * penalty                                             (1)
1012             #
1013             # is invariant under rescaling of sw.
1014             # But enet_path coordinate descent minimizes
1015             #
1016             #     1/2 * sum(squared error) + alpha' * penalty                  (2)
1017             #
1018             # and therefore sets
1019             #
1020             #     alpha' = n_samples * alpha                                   (3)
1021             #
1022             # inside its function body, which results in objective (2) being
1023             # equivalent to (1) in case of no sw.
1024             # With sw, however, enet_path should set
1025             #
1026             #     alpha' = sum(sw) * alpha                                     (4)
1027             #
1028             # Therefore, we use the freedom of Eq. (1) to rescale sw before
1029             # calling enet_path, i.e.
1030             #
1031             #     sw *= n_samples / sum(sw)
1032             #
1033             # such that sum(sw) = n_samples. This way, (3) and (4) are the same.
1034             sample_weight = sample_weight * (n_samples / np.sum(sample_weight))
1035             # Note: Alternatively, we could also have rescaled alpha instead
1036             # of sample_weight:
1037             #
1038             #     alpha *= np.sum(sample_weight) / n_samples
1039 
1040         # Ensure copying happens only once, don't do it again if done above.
1041         # X and y will be rescaled if sample_weight is not None, order='F'
1042         # ensures that the returned X and y are still F-contiguous.
1043         should_copy = self.copy_X and not X_copied
1044         X, y, X_offset, y_offset, X_scale, precompute, Xy = _pre_fit(
1045             X,
1046             y,
1047             None,
1048             self.precompute,
1049             fit_intercept=self.fit_intercept,
1050             copy=should_copy,
1051             check_input=check_input,
1052             sample_weight=sample_weight,
1053         )
1054         # coordinate descent needs F-ordered arrays and _pre_fit might have
1055         # called _rescale_data
1056         if check_input or sample_weight is not None:
1057             X, y = _set_order(X, y, order="F")
1058         if y.ndim == 1:
1059             y = y[:, np.newaxis]
1060         if Xy is not None and Xy.ndim == 1:
1061             Xy = Xy[:, np.newaxis]
1062 
1063         n_targets = y.shape[1]
1064 
1065         if not self.warm_start or not hasattr(self, "coef_"):
1066             coef_ = np.zeros((n_targets, n_features), dtype=X.dtype, order="F")
1067         else:
1068             coef_ = self.coef_
1069             if coef_.ndim == 1:
1070                 coef_ = coef_[np.newaxis, :]
1071 
1072         dual_gaps_ = np.zeros(n_targets, dtype=X.dtype)
1073         self.n_iter_ = []
1074 
1075         for k in range(n_targets):
1076             if Xy is not None:
1077                 this_Xy = Xy[:, k]
1078             else:
1079                 this_Xy = None
1080             _, this_coef, this_dual_gap, this_iter = self.path(
1081                 X,
1082                 y[:, k],
1083                 l1_ratio=self.l1_ratio,
1084                 eps=None,
1085                 n_alphas=None,
1086                 alphas=[alpha],
1087                 precompute=precompute,
1088                 Xy=this_Xy,
1089                 copy_X=True,
1090                 coef_init=coef_[k],
1091                 verbose=False,
1092                 return_n_iter=True,
1093                 positive=self.positive,
1094                 check_input=False,
1095                 # from here on **params
1096                 tol=self.tol,
1097                 X_offset=X_offset,
1098                 X_scale=X_scale,
1099                 max_iter=self.max_iter,
1100                 random_state=self.random_state,
1101                 selection=self.selection,
1102                 sample_weight=sample_weight,
1103             )
1104             coef_[k] = this_coef[:, 0]
1105             dual_gaps_[k] = this_dual_gap[0]
1106             self.n_iter_.append(this_iter[0])
1107 
1108         if n_targets == 1:
1109             self.n_iter_ = self.n_iter_[0]
1110             self.coef_ = coef_[0]
1111             self.dual_gap_ = dual_gaps_[0]
1112         else:
1113             self.coef_ = coef_
1114             self.dual_gap_ = dual_gaps_
1115 
1116         self._set_intercept(X_offset, y_offset, X_scale)
1117 
1118         # check for finiteness of coefficients
1119         if not all(np.isfinite(w).all() for w in [self.coef_, self.intercept_]):
1120             raise ValueError(
1121                 "Coordinate descent iterations resulted in non-finite parameter"
1122                 " values. The input data may contain large values and need to"
1123                 " be preprocessed."
1124             )
1125 
1126         # return self for chaining fit and predict calls
1127         return self

1282  
1283     The precise stopping criteria based on `tol` are the following: First, check that 
1284     that maximum coordinate update, i.e. :math:`\\max_j |w_j^{new} - w_j^{old}|` 
1285     is smaller than `tol` times the maximum absolute coefficient, :math:`\\max_j |w_j|`. 
1286     If so, then additionally check whether the dual gap is smaller than `tol` times 
1287     :math:`||y||_2^2 / n_{\\text{samples}}`. 
1288  
1289     The target can be a 2-dimensional array, resulting in the optimization of the 
1290     following objective:: 
1291  
1292         (1 / (2 * n_samples)) * ||Y - XW||^2_F + alpha * ||W||_11 
1293  
1294     where :math:`||W||_{1,1}` is the sum of the magnitude of the matrix coefficients. 
1295     It should not be confused with :class:`~sklearn.linear_model.MultiTaskLasso` which 
1296     instead penalizes the :math:`L_{2,1}` norm of the coefficients, yielding row-wise 
1297     sparsity in the coefficients. 
1298  
1299     Examples 
1300     -------- 
1301     >>> from sklearn import linear_model 
1302     >>> clf = linear_model.Lasso(alpha=0.1) 
1303     >>> clf.fit([[0,0], [1, 1], [2, 2]], [0, 1, 2]) 
1304     Lasso(alpha=0.1) 
1305     >>> print(clf.coef_) 
1306     [0.85 0.  ] 
1307     >>> print(clf.intercept_) 
1308     0.15... 
1309     """
1310 
1311     _parameter_constraints: dict = {
1312         **ElasticNet._parameter_constraints,
1313     }
1314     _parameter_constraints.pop("l1_ratio")

1367     Parameters 
1368     ---------- 
1369     X : {array-like, sparse matrix} of shape (n_samples, n_features) 
1370         Training data. 
1371

1374  
1375     sample_weight : None or array-like of shape (n_samples,)

1380  
1381     test : list of indices 
1382         The indices of the test set. 
1383

1385         Function returning a list of models on the path. See 
1386         enet_path for an example of signature.

1388     path_params : dictionary 
1389         Parameters passed to the path function. 
1390

1392         Array of float that is used for cross-validation. If not 
1393         provided, computed using 'path'. 
1394  
1395     l1_ratio : float, default=1 
1396         float between 0 and 1 passed to ElasticNet (scaling between 
1397         l1 and l2 penalties). For ``l1_ratio = 0`` the penalty is an 
1398         L2 penalty. For ``l1_ratio = 1`` it is an L1 penalty. For ``0 
1399         < l1_ratio < 1``, the penalty is a combination of L1 and L2. 
1400  
1401     X_order : {'F', 'C'}, default=None 
1402         The order of the arrays expected by the path function to 
1403         avoid memory copies. 
1404  
1405     dtype : a numpy dtype, default=None 
1406         The dtype of the arrays expected by the path function to 
1407         avoid memory copies.

1409     X_train = X[train]
1410     y_train = y[train]
1411     X_test = X[test]
1412     y_test = y[test]
1413     if sample_weight is None:
1414         sw_train, sw_test = None, None
1415     else:
1416         sw_train = sample_weight[train]
1417         sw_test = sample_weight[test]
1418         n_samples = X_train.shape[0]
1419         # TLDR: Rescale sw_train to sum up to n_samples on the training set.
1420         # See TLDR and long comment inside ElasticNet.fit.
1421         sw_train *= n_samples / np.sum(sw_train)
1422         # Note: Alternatively, we could also have rescaled alpha instead
1423         # of sample_weight:
1424         #
1425         #     alpha *= np.sum(sample_weight) / n_samples
1426 
1427     if not sparse.issparse(X):
1428         for array, array_input in (
1429             (X_train, X),
1430             (y_train, y),
1431             (X_test, X),
1432             (y_test, y),
1433         ):
1434             if array.base is not array_input and not array.flags["WRITEABLE"]:
1435                 # fancy indexing should create a writable copy but it doesn't
1436                 # for read-only memmaps (cf. numpy#14132).
1437                 array.setflags(write=True)
1438 
1439     if y.ndim == 1:
1440         precompute = path_params["precompute"]
1441     else:
1442         # No Gram variant of multi-task exists right now.
1443         # Fall back to default enet_multitask
1444         precompute = False
1445 
1446     X_train, y_train, X_offset, y_offset, X_scale, precompute, Xy = _pre_fit(
1447         X_train,
1448         y_train,
1449         None,
1450         precompute,
1451         fit_intercept=fit_intercept,
1452         copy=False,
1453         sample_weight=sw_train,
1454     )
1455 
1456     path_params = path_params.copy()
1457     path_params["Xy"] = Xy
1458     path_params["X_offset"] = X_offset
1459     path_params["X_scale"] = X_scale
1460     path_params["precompute"] = precompute
1461     path_params["copy_X"] = False
1462     path_params["alphas"] = alphas
1463     # needed for sparse cd solver
1464     path_params["sample_weight"] = sw_train
1465 
1466     if "l1_ratio" in path_params:
1467         path_params["l1_ratio"] = l1_ratio
1468 
1469     # Do the ordering and type casting here, as if it is done in the path,
1470     # X is copied and a reference is kept here
1471     X_train = check_array(X_train, accept_sparse="csc", dtype=dtype, order=X_order)
1472     alphas, coefs, _ = path(X_train, y_train, **path_params)
1473     del X_train, y_train
1474 
1475     if y.ndim == 1:
1476         # Doing this so that it becomes coherent with multioutput.
1477         coefs = coefs[np.newaxis, :, :]
1478         y_offset = np.atleast_1d(y_offset)
1479         y_test = y_test[:, np.newaxis]
1480 
1481     intercepts = y_offset[:, np.newaxis] - np.dot(X_offset, coefs)
1482     X_test_coefs = safe_sparse_dot(X_test, coefs)
1483     residues = X_test_coefs - y_test[:, :, np.newaxis]
1484     residues += intercepts
1485     if sample_weight is None:
1486         this_mse = (residues**2).mean(axis=0)
1487     else:
1488         this_mse = np.average(residues**2, weights=sw_test, axis=0)
1489 
1490     return this_mse.mean(axis=0)

1560     def fit(self, X, y, sample_weight=None, **params):
1561         """Fit linear model with coordinate descent. 
1562  
1563         Fit is on grid of alphas and best alpha estimated by cross-validation. 
1564  
1565         Parameters 
1566         ---------- 
1567         X : {array-like, sparse matrix} of shape (n_samples, n_features) 
1568             Training data. Pass directly as Fortran-contiguous data 
1569             to avoid unnecessary memory duplication. If y is mono-output, 
1570             X can be sparse. Note that large sparse matrices and arrays 
1571             requiring `int64` indices are not accepted. 
1572  
1573         y : array-like of shape (n_samples,) or (n_samples, n_targets) 
1574             Target values. 
1575  
1576         sample_weight : float or array-like of shape (n_samples,), \ 
1577                 default=None 
1578             Sample weights used for fitting and evaluation of the weighted 
1579             mean squared error of each cv-fold. Note that the cross validated 
1580             MSE that is finally used to find the best model is the unweighted 
1581             mean over the (weighted) MSEs of each test fold. 
1582  
1583         **params : dict, default=None 
1584             Parameters to be passed to the CV splitter. 
1585  
1586             .. versionadded:: 1.4 
1587                 Only available if `enable_metadata_routing=True`, 
1588                 which can be set by using 
1589                 ``sklearn.set_config(enable_metadata_routing=True)``. 
1590                 See :ref:`Metadata Routing User Guide <metadata_routing>` for 
1591                 more details. 
1592  
1593         Returns 
1594         ------- 
1595         self : object 
1596             Returns an instance of fitted model. 
1597         """
1598         _raise_for_params(params, self, "fit")
1599 
1600         # This makes sure that there is no duplication in memory.
1601         # Dealing right with copy_X is important in the following:
1602         # Multiple functions touch X and subsamples of X and can induce a
1603         # lot of duplication of memory
1604         copy_X = self.copy_X and self.fit_intercept
1605 
1606         check_y_params = dict(
1607             copy=False, dtype=[np.float64, np.float32], ensure_2d=False
1608         )
1609         if isinstance(X, np.ndarray) or sparse.issparse(X):
1610             # Keep a reference to X
1611             reference_to_old_X = X
1612             # Let us not impose fortran ordering so far: it is
1613             # not useful for the cross-validation loop and will be done
1614             # by the model fitting itself
1615 
1616             # Need to validate separately here.
1617             # We can't pass multi_output=True because that would allow y to be
1618             # csr. We also want to allow y to be 64 or 32 but check_X_y only
1619             # allows to convert for 64.
1620             check_X_params = dict(
1621                 accept_sparse="csc",
1622                 dtype=[np.float64, np.float32],
1623                 force_writeable=True,
1624                 copy=False,
1625                 accept_large_sparse=False,
1626             )
1627             X, y = validate_data(
1628                 self, X, y, validate_separately=(check_X_params, check_y_params)
1629             )
1630             if sparse.issparse(X):
1631                 if hasattr(reference_to_old_X, "data") and not np.may_share_memory(
1632                     reference_to_old_X.data, X.data
1633                 ):
1634                     # X is a sparse matrix and has been copied
1635                     copy_X = False
1636             elif not np.may_share_memory(reference_to_old_X, X):
1637                 # X has been copied
1638                 copy_X = False
1639             del reference_to_old_X
1640         else:
1641             # Need to validate separately here.
1642             # We can't pass multi_output=True because that would allow y to be
1643             # csr. We also want to allow y to be 64 or 32 but check_X_y only
1644             # allows to convert for 64.
1645             check_X_params = dict(
1646                 accept_sparse="csc",
1647                 dtype=[np.float64, np.float32],
1648                 order="F",
1649                 force_writeable=True,
1650                 copy=copy_X,
1651             )
1652             X, y = validate_data(
1653                 self, X, y, validate_separately=(check_X_params, check_y_params)
1654             )
1655             copy_X = False
1656 
1657         check_consistent_length(X, y)
1658 
1659         if not self._is_multitask():
1660             if y.ndim > 1 and y.shape[1] > 1:
1661                 raise ValueError(
1662                     "For multi-task outputs, use MultiTask%s" % self.__class__.__name__
1663                 )
1664             y = column_or_1d(y, warn=True)
1665         else:
1666             if sparse.issparse(X):
1667                 raise TypeError("X should be dense but a sparse matrix waspassed")
1668             elif y.ndim == 1:
1669                 raise ValueError(
1670                     "For mono-task outputs, use %sCV" % self.__class__.__name__[9:]
1671                 )
1672 
1673         if isinstance(sample_weight, numbers.Number):
1674             sample_weight = None
1675         if sample_weight is not None:
1676             sample_weight = _check_sample_weight(sample_weight, X, dtype=X.dtype)
1677 
1678         model = self._get_estimator()
1679 
1680         # All LinearModelCV parameters except 'cv' are acceptable
1681         path_params = self.get_params()
1682 
1683         # Pop `intercept` that is not parameter of the path function
1684         path_params.pop("fit_intercept", None)
1685 
1686         if "l1_ratio" in path_params:
1687             l1_ratios = np.atleast_1d(path_params["l1_ratio"])
1688             # For the first path, we need to set l1_ratio
1689             path_params["l1_ratio"] = l1_ratios[0]
1690         else:
1691             l1_ratios = [
1692                 1,
1693             ]
1694         path_params.pop("cv", None)
1695         path_params.pop("n_jobs", None)
1696 
1697         alphas = self.alphas
1698         n_l1_ratio = len(l1_ratios)
1699

1702             target_type=Real,
1703             min_val=0.0,
1704             include_boundaries="left",

1711                     y,
1712                     l1_ratio=l1_ratio,

1715                     n_alphas=self.n_alphas,
1716                     copy_X=self.copy_X,
1717                     sample_weight=sample_weight,
1718                 )
1719                 for l1_ratio in l1_ratios
1720             ]
1721         else:
1722             # Making sure alphas entries are scalars.
1723             for index, alpha in enumerate(alphas):
1724                 check_scalar_alpha(alpha, f"alphas[{index}]")
1725             # Making sure alphas is properly ordered.
1726             alphas = np.tile(np.sort(alphas)[::-1], (n_l1_ratio, 1))
1727 
1728         # We want n_alphas to be the number of alphas used for each l1_ratio.
1729         n_alphas = len(alphas[0])
1730         path_params.update({"n_alphas": n_alphas})
1731 
1732         path_params["copy_X"] = copy_X
1733         # We are not computing in parallel, we can modify X
1734         # inplace in the folds
1735         if effective_n_jobs(self.n_jobs) > 1:
1736             path_params["copy_X"] = False
1737 
1738         # init cross-validation generator
1739         cv = check_cv(self.cv)
1740 
1741         if _routing_enabled():
1742             splitter_supports_sample_weight = get_routing_for_object(cv).consumes(
1743                 method="split", params=["sample_weight"]
1744             )
1745             if (
1746                 sample_weight is not None
1747                 and not splitter_supports_sample_weight
1748                 and not has_fit_parameter(self, "sample_weight")
1749             ):
1750                 raise ValueError(
1751                     "The CV splitter and underlying estimator do not support"
1752                     " sample weights."
1753                 )
1754 
1755             if splitter_supports_sample_weight:
1756                 params["sample_weight"] = sample_weight
1757 
1758             routed_params = process_routing(self, "fit", **params)
1759 
1760             if sample_weight is not None and not has_fit_parameter(
1761                 self, "sample_weight"
1762             ):
1763                 # MultiTaskElasticNetCV does not (yet) support sample_weight
1764                 sample_weight = None
1765         else:
1766             routed_params = Bunch()
1767             routed_params.splitter = Bunch(split=Bunch())
1768 
1769         # Compute path for all folds and compute MSE to get the best alpha
1770         folds = list(cv.split(X, y, **routed_params.splitter.split))
1771         best_mse = np.inf
1772 
1773         # We do a double for loop folded in one, in order to be able to
1774         # iterate in parallel on l1_ratio and folds
1775         jobs = (
1776             delayed(_path_residuals)(
1777                 X,
1778                 y,
1779                 sample_weight,
1780                 train,
1781                 test,
1782                 self.fit_intercept,
1783                 self.path,
1784                 path_params,
1785                 alphas=this_alphas,
1786                 l1_ratio=this_l1_ratio,
1787                 X_order="F",
1788                 dtype=X.dtype.type,
1789             )
1790             for this_l1_ratio, this_alphas in zip(l1_ratios, alphas)
1791             for train, test in folds
1792         )
1793         mse_paths = Parallel(
1794             n_jobs=self.n_jobs,
1795             verbose=self.verbose,
1796             prefer="threads",
1797         )(jobs)
1798         mse_paths = np.reshape(mse_paths, (n_l1_ratio, len(folds), -1))
1799         # The mean is computed over folds.
1800         mean_mse = np.mean(mse_paths, axis=1)
1801         self.mse_path_ = np.squeeze(np.moveaxis(mse_paths, 2, 1))
1802         for l1_ratio, l1_alphas, mse_alphas in zip(l1_ratios, alphas, mean_mse):
1803             i_best_alpha = np.argmin(mse_alphas)
1804             this_best_mse = mse_alphas[i_best_alpha]
1805             if this_best_mse < best_mse:
1806                 best_alpha = l1_alphas[i_best_alpha]
1807                 best_l1_ratio = l1_ratio
1808                 best_mse = this_best_mse
1809 
1810         self.l1_ratio_ = best_l1_ratio
1811         self.alpha_ = best_alpha
1812         if self.alphas is None:
1813             self.alphas_ = np.asarray(alphas)
1814             if n_l1_ratio == 1:
1815                 self.alphas_ = self.alphas_[0]
1816         # Remove duplicate alphas in case alphas is provided.
1817         else:
1818             self.alphas_ = np.asarray(alphas[0])
1819 
1820         # Refit the model with the parameters selected
1821         common_params = {
1822             name: value
1823             for name, value in self.get_params().items()
1824             if name in model.get_params()
1825         }
1826         model.set_params(**common_params)
1827         model.alpha = best_alpha
1828         model.l1_ratio = best_l1_ratio
1829         model.copy_X = copy_X
1830         precompute = getattr(self, "precompute", None)
1831         if isinstance(precompute, str) and precompute == "auto":
1832             model.precompute = False
1833 
1834         if sample_weight is None:
1835             # MultiTaskElasticNetCV does not (yet) support sample_weight, even
1836             # not sample_weight=None.
1837             model.fit(X, y)
1838         else:
1839             model.fit(X, y, sample_weight=sample_weight)
1840         if not hasattr(self, "l1_ratio"):
1841             del self.l1_ratio_
1842         self.coef_ = model.coef_
1843         self.intercept_ = model.intercept_
1844         self.dual_gap_ = model.dual_gap_
1845         self.n_iter_ = model.n_iter_
1846         return self

1880 class LassoCV(RegressorMixin, LinearModelCV):
1881     """Lasso linear model with iterative fitting along a regularization path. 
1882  
1883     See glossary entry for :term:`cross-validation estimator`. 
1884  
1885     The best model is selected by cross-validation. 
1886  
1887     The optimization objective for Lasso is:: 
1888  
1889         (1 / (2 * n_samples)) * ||y - Xw||^2_2 + alpha * ||w||_1 
1890  
1891     Read more in the :ref:`User Guide <lasso>`. 
1892  
1893     Parameters 
1894     ---------- 
1895     eps : float, default=1e-3 
1896         Length of the path. ``eps=1e-3`` means that 
1897         ``alpha_min / alpha_max = 1e-3``. 
1898  
1899     n_alphas : int, default=100 
1900         Number of alphas along the regularization path. 
1901  
1902     alphas : array-like, default=None 
1903         List of alphas where to compute the models. 
1904         If ``None`` alphas are set automatically.
