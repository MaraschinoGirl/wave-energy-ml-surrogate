1    import pytest
2    from unittest.mock import patch
3    from bs4.builder import DetectsXMLParsedAsHTML
4    
5    
6    class TestDetectsXMLParsedAsHTML:
7        @pytest.mark.parametrize(
8            "markup,looks_like_xml",
9            [
10               ("No xml declaration", False),
11               ("<html>obviously HTML</html", False),
12               ("<?xml ><html>Actually XHTML</html>", False),
13               ("<?xml>            <    html>Tricky XHTML</html>", False),
14               ("<?xml ><no-html-tag>", True),
15           ],
16       )
17       def test_warn_if_markup_looks_like_xml(self, markup, looks_like_xml):
18           # Test of our ability to guess at whether markup looks XML-ish
19           # _and_ not HTML-ish.
20           with patch("bs4.builder.DetectsXMLParsedAsHTML._warn") as mock:
21               for data in markup, markup.encode("utf8"):
22                   result = DetectsXMLParsedAsHTML.warn_if_markup_looks_like_xml(data)
23                   assert result == looks_like_xml
24                   if looks_like_xml:
25                       assert mock.called
26                   else:
27                       assert not mock.called
28                   mock.reset_mock()
