1    """ 
2    Our own implementation of the Newton algorithm 
3     
4    Unlike the scipy.optimize version, this version of the Newton conjugate 
5    gradient solver uses only one function call to retrieve the 
6    func value, the gradient value and a callable for the Hessian matvec 
7    product. If the function call is very expensive (e.g. for logistic 
8    regression with large design matrix), this approach gives very 
9    significant speedups. 
10   """

31   def _line_search_wolfe12(
32       f, fprime, xk, pk, gfk, old_fval, old_old_fval, verbose=0, **kwargs
33   ):
34       """ 
35       Same as line_search_wolfe1, but fall back to line_search_wolfe2 if 
36       suitable step length is not found, and raise an exception if a 
37       suitable step length is not found. 
38    
39       Raises 
40       ------ 
41       _LineSearchError 
42           If no suitable step size is found. 
43    
44       """
45       is_verbose = verbose >= 2
46       eps = 16 * np.finfo(np.asarray(old_fval).dtype).eps
47       if is_verbose:
48           print("  Line Search")
49           print(f"    eps=16 * finfo.eps={eps}")
50           print("    try line search wolfe1")
51   
52       ret = line_search_wolfe1(f, fprime, xk, pk, gfk, old_fval, old_old_fval, **kwargs)
53   
54       if is_verbose:
55           _not_ = "not " if ret[0] is None else ""
56           print("    wolfe1 line search was " + _not_ + "successful")
57   
58       if ret[0] is None:
59           # Have a look at the line_search method of our NewtonSolver class. We borrow
60           # the logic from there
61           # Deal with relative loss differences around machine precision.
62           args = kwargs.get("args", tuple())
63           fval = f(xk + pk, *args)
64           tiny_loss = np.abs(old_fval * eps)
65           loss_improvement = fval - old_fval
66           check = np.abs(loss_improvement) <= tiny_loss
67           if is_verbose:
68               print(
69                   "    check loss |improvement| <= eps * |loss_old|:"
70                   f" {np.abs(loss_improvement)} <= {tiny_loss} {check}"
71               )
72           if check:
73               # 2.1 Check sum of absolute gradients as alternative condition.
74               sum_abs_grad_old = scipy.linalg.norm(gfk, ord=1)
75               grad = fprime(xk + pk, *args)
76               sum_abs_grad = scipy.linalg.norm(grad, ord=1)
77               check = sum_abs_grad < sum_abs_grad_old
78               if is_verbose:
79                   print(
80                       "    check sum(|gradient|) < sum(|gradient_old|): "
81                       f"{sum_abs_grad} < {sum_abs_grad_old} {check}"
82                   )
83               if check:
84                   ret = (
85                       1.0,  # step size
86                       ret[1] + 1,  # number of function evaluations
87                       ret[2] + 1,  # number of gradient evaluations
88                       fval,
89                       old_fval,
90                       grad,
91                   )
92   
93       if ret[0] is None:
94           # line search failed: try different one.
95           # TODO: It seems that the new check for the sum of absolute gradients above
96           # catches all cases that, earlier, ended up here. In fact, our tests never
97           # trigger this "if branch" here and we can consider to remove it.
98           if is_verbose:
99               print("    last resort: try line search wolfe2")
100          ret = line_search_wolfe2(
101              f, fprime, xk, pk, gfk, old_fval, old_old_fval, **kwargs
102          )
103          if is_verbose:
104              _not_ = "not " if ret[0] is None else ""
105              print("    wolfe2 line search was " + _not_ + "successful")
106  
107      if ret[0] is None:
108          raise _LineSearchError()
109  
110      return ret

113  def _cg(fhess_p, fgrad, maxiter, tol, verbose=0):
114      """ 
115      Solve iteratively the linear system 'fhess_p . xsupi = fgrad' 
116      with a conjugate gradient descent. 
117   
118      Parameters 
119      ---------- 
120      fhess_p : callable 
121          Function that takes the gradient as a parameter and returns the 
122          matrix product of the Hessian and gradient. 
123   
124      fgrad : ndarray of shape (n_features,) or (n_features + 1,) 
125          Gradient vector. 
126   
127      maxiter : int 
128          Number of CG iterations. 
129   
130      tol : float 
131          Stopping criterion. 
132   
133      Returns 
134      ------- 
135      xsupi : ndarray of shape (n_features,) or (n_features + 1,) 
136          Estimated solution. 
137      """
138      eps = 16 * np.finfo(np.float64).eps
139      xsupi = np.zeros(len(fgrad), dtype=fgrad.dtype)
140      ri = np.copy(fgrad)  # residual = fgrad - fhess_p @ xsupi
141      psupi = -ri
142      i = 0
143      dri0 = np.dot(ri, ri)
144      # We also keep track of |p_i|^2.
145      psupi_norm2 = dri0
146      is_verbose = verbose >= 2
147  
148      while i <= maxiter:
149          if np.sum(np.abs(ri)) <= tol:
150              if is_verbose:
151                  print(
152                      f"  Inner CG solver iteration {i} stopped with\n"
153                      f"    sum(|residuals|) <= tol: {np.sum(np.abs(ri))} <= {tol}"
154                  )
155              break
156  
157          Ap = fhess_p(psupi)
158          # check curvature
159          curv = np.dot(psupi, Ap)
160          if 0 <= curv <= eps * psupi_norm2:
161              # See https://arxiv.org/abs/1803.02924, Algo 1 Capped Conjugate Gradient.
162              if is_verbose:
163                  print(
164                      f"  Inner CG solver iteration {i} stopped with\n"
165                      f"    tiny_|p| = eps * ||p||^2, eps = {eps}, "
166                      f"squred L2 norm ||p||^2 = {psupi_norm2}\n"
167                      f"    curvature <= tiny_|p|: {curv} <= {eps * psupi_norm2}"
168                  )
169              break
170          elif curv < 0:
171              if i > 0:
172                  if is_verbose:
173                      print(
174                          f"  Inner CG solver iteration {i} stopped with negative "
175                          f"curvature, curvature = {curv}"
176                      )
177                  break
178              else:
179                  # fall back to steepest descent direction
180                  xsupi += dri0 / curv * psupi
181                  if is_verbose:
182                      print("  Inner CG solver iteration 0 fell back to steepest descent")
183                  break
184          alphai = dri0 / curv
185          xsupi += alphai * psupi
186          ri += alphai * Ap
187          dri1 = np.dot(ri, ri)
188          betai = dri1 / dri0
189          psupi = -ri + betai * psupi
190          # We use  |p_i|^2 = |r_i|^2 + beta_i^2 |p_{i-1}|^2
191          psupi_norm2 = dri1 + betai**2 * psupi_norm2
192          i = i + 1
193          dri0 = dri1  # update np.dot(ri,ri) for next time.
194      if is_verbose and i > maxiter:
195          print(
196              f"  Inner CG solver stopped reaching maxiter={i - 1} with "
197              f"sum(|residuals|) = {np.sum(np.abs(ri))}"
198          )
199      return xsupi

202  def _newton_cg(
203      grad_hess,
204      func,
205      grad,
206      x0,
207      args=(),
208      tol=1e-4,
209      maxiter=100,
210      maxinner=200,
211      line_search=True,
212      warn=True,
213      verbose=0,
214  ):
215      """ 
216      Minimization of scalar function of one or more variables using the 
217      Newton-CG algorithm. 
218   
219      Parameters 
220      ---------- 
221      grad_hess : callable 
222          Should return the gradient and a callable returning the matvec product 
223          of the Hessian. 
224   
225      func : callable 
226          Should return the value of the function. 
227   
228      grad : callable 
229          Should return the function value and the gradient. This is used 
230          by the linesearch functions. 
231   
232      x0 : array of float 
233          Initial guess. 
234   
235      args : tuple, default=() 
236          Arguments passed to func_grad_hess, func and grad. 
237   
238      tol : float, default=1e-4 
239          Stopping criterion. The iteration will stop when 
240          ``max{|g_i | i = 1, ..., n} <= tol`` 
241          where ``g_i`` is the i-th component of the gradient. 
242   
243      maxiter : int, default=100 
244          Number of Newton iterations. 
245   
246      maxinner : int, default=200 
247          Number of CG iterations. 
248   
249      line_search : bool, default=True 
250          Whether to use a line search or not. 
251   
252      warn : bool, default=True 
253          Whether to warn when didn't converge. 
254   
255      Returns 
256      ------- 
257      xk : ndarray of float 
258          Estimated minimum. 
259      """
260      x0 = np.asarray(x0).flatten()
261      xk = np.copy(x0)
262      k = 0
263  
264      if line_search:
265          old_fval = func(x0, *args)
266          old_old_fval = None
267      else:
268          old_fval = 0
269  
270      is_verbose = verbose > 0
271  
272      # Outer loop: our Newton iteration
273      while k < maxiter:
274          # Compute a search direction pk by applying the CG method to
275          #  del2 f(xk) p = - fgrad f(xk) starting from 0.
276          fgrad, fhess_p = grad_hess(xk, *args)
277  
278          absgrad = np.abs(fgrad)
279          max_absgrad = np.max(absgrad)
280          check = max_absgrad <= tol
281          if is_verbose:
282              print(f"Newton-CG iter = {k}")
283              print("  Check Convergence")
284              print(f"    max |gradient| <= tol: {max_absgrad} <= {tol} {check}")
285          if check:
286              break
287  
288          maggrad = np.sum(absgrad)
289          eta = min([0.5, np.sqrt(maggrad)])
290          termcond = eta * maggrad
291  
292          # Inner loop: solve the Newton update by conjugate gradient, to
293          # avoid inverting the Hessian
294          xsupi = _cg(fhess_p, fgrad, maxiter=maxinner, tol=termcond, verbose=verbose)
295  
296          alphak = 1.0
297  
298          if line_search:
299              try:
300                  alphak, fc, gc, old_fval, old_old_fval, gfkp1 = _line_search_wolfe12(
301                      func,
302                      grad,
303                      xk,
304                      xsupi,
305                      fgrad,
306                      old_fval,
307                      old_old_fval,
308                      verbose=verbose,
309                      args=args,
310                  )
311              except _LineSearchError:
312                  warnings.warn("Line Search failed")
313                  break
314  
315          xk += alphak * xsupi  # upcast if necessary
316          k += 1
317  
318      if warn and k >= maxiter:
319          warnings.warn(
320              (
321                  f"newton-cg failed to converge at loss = {old_fval}. Increase the"
322                  " number of iterations."
323              ),
324              ConvergenceWarning,
325          )
326      elif is_verbose:
327          print(f"  Solver did converge at loss = {old_fval}.")
328      return xk, k

331  def _check_optimize_result(solver, result, max_iter=None, extra_warning_msg=None):
332      """Check the OptimizeResult for successful convergence 
333   
334      Parameters 
335      ---------- 
336      solver : str 
337         Solver name. Currently only `lbfgs` is supported. 
338   
339      result : OptimizeResult 
340         Result of the scipy.optimize.minimize function. 
341   
342      max_iter : int, default=None 
343         Expected maximum number of iterations. 
344   
345      extra_warning_msg : str, default=None 
346          Extra warning message. 
347   
348      Returns 
349      ------- 
350      n_iter : int 
351         Number of iterations. 
352      """
353      # handle both scipy and scikit-learn solver names
354      if solver == "lbfgs":
355          if result.status != 0:
356              result_message = result.message
357  
358              warning_msg = (
359                  "{} failed to converge (status={}):\n{}.\n\n"
360                  "Increase the number of iterations (max_iter) "
361                  "or scale the data as shown in:\n"
362                  "    https://scikit-learn.org/stable/modules/"
363                  "preprocessing.html"
364              ).format(solver, result.status, result_message)
365              if extra_warning_msg is not None:
366                  warning_msg += "\n" + extra_warning_msg
367              warnings.warn(warning_msg, ConvergenceWarning, stacklevel=2)
368          if max_iter is not None:
369              # In scipy <= 1.0.0, nit may exceed maxiter for lbfgs.
370              # See https://github.com/scipy/scipy/issues/7854
371              n_iter_i = min(result.nit, max_iter)
372          else:
373              n_iter_i = result.nit
374      else:
375          raise NotImplementedError
376  
377      return n_iter_i
