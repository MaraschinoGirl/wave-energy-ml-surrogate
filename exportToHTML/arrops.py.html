4    def arange_multi(starts, stops=None, lengths=None):
5        """ 
6        Create concatenated ranges of integers for multiple start/length. 
7     
8        Parameters 
9        ---------- 
10       starts : numpy.ndarray 
11           Starts for each range 
12       stops : numpy.ndarray 
13           Stops for each range 
14       lengths : numpy.ndarray 
15           Lengths for each range. Either stops or lengths must be provided. 
16    
17       Returns 
18       ------- 
19       concat_ranges : numpy.ndarray 
20           Concatenated ranges. 
21    
22       Notes 
23       ----- 
24       See the following illustrative example: 
25    
26       starts = np.array([1, 3, 4, 6]) 
27       stops = np.array([1, 5, 7, 6]) 
28    
29       print arange_multi(starts, lengths) 
30       >>> [3 4 4 5 6] 
31    
32       From: https://codereview.stackexchange.com/questions/83018/vectorized-numpy-version-of-arange-with-multiple-start-stop 
33    
34       """
35   
36       if (stops is None) == (lengths is None):
37           raise ValueError("Either stops or lengths must be provided!")
38   
39       if lengths is None:
40           lengths = stops - starts
41   
42       if np.isscalar(starts):
43           starts = np.full(len(stops), starts)
44   
45       # Repeat start position index length times and concatenate
46       cat_start = np.repeat(starts, lengths)
47   
48       # Create group counter that resets for each start/length
49       cat_counter = np.arange(lengths.sum()) - np.repeat(
50           lengths.cumsum() - lengths, lengths
51       )
52   
53       # Add group counter to group specific starts
54       cat_range = cat_start + cat_counter
55   
56       return cat_range

58   def overlap_intervals(starts1, ends1, starts2, ends2, closed=False, sort=False):
59       """ 
60       Take two sets of intervals and return the indices of pairs of overlapping intervals. 
61    
62       Parameters 
63       ---------- 
64       starts1, ends1, starts2, ends2 : numpy.ndarray 
65           Interval coordinates. Warning: if provided as pandas.Series, indices 
66           will be ignored. 
67    
68       closed : bool 
69           If True, then treat intervals as closed and report single-point overlaps. 
70       Returns 
71       ------- 
72       overlap_ids : numpy.ndarray 
73           An Nx2 array containing the indices of pairs of overlapping intervals. 
74           The 1st column contains ids from the 1st set, the 2nd column has ids 
75           from the 2nd set. 
76    
77       """
78   
79       # Convert to numpy arrays
80       starts1 = np.asarray(starts1)
81       ends1 = np.asarray(ends1)
82       starts2 = np.asarray(starts2)
83       ends2 = np.asarray(ends2)
84   
85       # Concatenate intervals lists
86       n1 = len(starts1)
87       n2 = len(starts2)
88       ids1 = np.arange(0, n1)
89       ids2 = np.arange(0, n2)
90   
91       # Sort all intervals together
92       order1 = np.lexsort([ends1, starts1])
93       order2 = np.lexsort([ends2, starts2])
94       starts1, ends1, ids1 = starts1[order1], ends1[order1], ids1[order1]
95       starts2, ends2, ids2 = starts2[order2], ends2[order2], ids2[order2]
96   
97       # Find interval overlaps
98       match_2in1_starts = np.searchsorted(starts2, starts1, "left")
99       match_2in1_ends = np.searchsorted(starts2, ends1, "right" if closed else "left")
100      # "right" is intentional here to avoid duplication
101      match_1in2_starts = np.searchsorted(starts1, starts2, "right")
102      match_1in2_ends = np.searchsorted(starts1, ends2, "right" if closed else "left")
103  
104      # Ignore self-overlaps
105      match_2in1_mask = match_2in1_ends > match_2in1_starts
106      match_1in2_mask = match_1in2_ends > match_1in2_starts
107      match_2in1_starts, match_2in1_ends = (
108          match_2in1_starts[match_2in1_mask],
109          match_2in1_ends[match_2in1_mask],
110      )
111      match_1in2_starts, match_1in2_ends = (
112          match_1in2_starts[match_1in2_mask],
113          match_1in2_ends[match_1in2_mask],
114      )
115  
116      # Generate IDs of pairs of overlapping intervals
117      overlap_ids = np.block(
118          [
119              [
120                  np.repeat(ids1[match_2in1_mask], match_2in1_ends - match_2in1_starts)[
121                      :, None
122                  ],
123                  ids2[arange_multi(match_2in1_starts, match_2in1_ends)][:, None],
124              ],
125              [
126                  ids1[arange_multi(match_1in2_starts, match_1in2_ends)][:, None],
127                  np.repeat(ids2[match_1in2_mask], match_1in2_ends - match_1in2_starts)[
128                      :, None
129                  ],
130              ],
131          ]
132      )
133  
134      if sort:
135          # Sort overlaps according to the 1st
136          overlap_ids = overlap_ids[np.lexsort([overlap_ids[:, 1], overlap_ids[:, 0]])]
137  
138      return overlap_ids
