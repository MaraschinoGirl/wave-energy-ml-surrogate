34   class MiscFutureTest(AsyncTestCase):
35       def test_future_set_result_unless_cancelled(self):
36           fut = Future()  # type: Future[int]
37           future_set_result_unless_cancelled(fut, 42)
38           self.assertEqual(fut.result(), 42)
39           self.assertFalse(fut.cancelled())
40   
41           fut = Future()
42           fut.cancel()
43           is_cancelled = fut.cancelled()
44           future_set_result_unless_cancelled(fut, 42)
45           self.assertEqual(fut.cancelled(), is_cancelled)
46           if not is_cancelled:
47               self.assertEqual(fut.result(), 42)

54   class CapServer(TCPServer):
55       @gen.coroutine
56       def handle_stream(self, stream, address):
57           data = yield stream.read_until(b"\n")
58           data = to_unicode(data)
59           if data == data.upper():
60               stream.write(b"error\talready capitalized\n")
61           else:
62               # data already has \n
63               stream.write(utf8("ok\t%s" % data.upper()))
64           stream.close()
65   
66   
67   class CapError(Exception):
68       pass
69   
70   
71   class BaseCapClient(object):
72       def __init__(self, port):
73           self.port = port
74   
75       def process_response(self, data):
76           m = re.match("(.*)\t(.*)\n", to_unicode(data))
77           if m is None:
78               raise Exception("did not match")
79           status, message = m.groups()
80           if status == "ok":
81               return message
82           else:
83               raise CapError(message)
84   
85   
86   class GeneratorCapClient(BaseCapClient):
87       @gen.coroutine
88       def capitalize(self, request_data):
89           logging.debug("capitalize")
90           stream = IOStream(socket.socket())
91           logging.debug("connecting")
92           yield stream.connect(("127.0.0.1", self.port))
93           stream.write(utf8(request_data + "\n"))
94           logging.debug("reading")
95           data = yield stream.read_until(b"\n")
96           logging.debug("returning")
97           stream.close()
98           raise gen.Return(self.process_response(data))

101  class ClientTestMixin(object):
102      client_class = None  # type: typing.Callable
103  
104      def setUp(self):
105          super().setUp()  # type: ignore
106          self.server = CapServer()
107          sock, port = bind_unused_port()
108          self.server.add_sockets([sock])
109          self.client = self.client_class(port=port)
110  
111      def tearDown(self):
112          self.server.stop()
113          super().tearDown()  # type: ignore
114  
115      def test_future(self: typing.Any):
116          future = self.client.capitalize("hello")
117          self.io_loop.add_future(future, self.stop)
118          self.wait()
119          self.assertEqual(future.result(), "HELLO")
120  
121      def test_future_error(self: typing.Any):
122          future = self.client.capitalize("HELLO")
123          self.io_loop.add_future(future, self.stop)
124          self.wait()
125          self.assertRaisesRegex(CapError, "already capitalized", future.result)  # type: ignore
126  
127      def test_generator(self: typing.Any):
128          @gen.coroutine
129          def f():
130              result = yield self.client.capitalize("hello")
131              self.assertEqual(result, "HELLO")
132  
133          self.io_loop.run_sync(f)
134  
135      def test_generator_error(self: typing.Any):
136          @gen.coroutine
137          def f():
138              with self.assertRaisesRegex(CapError, "already capitalized"):
139                  yield self.client.capitalize("HELLO")
140  
141          self.io_loop.run_sync(f)

148  class RunOnExecutorTest(AsyncTestCase):
149      @gen_test
150      def test_no_calling(self):
151          class Object(object):
152              def __init__(self):
153                  self.executor = futures.thread.ThreadPoolExecutor(1)
154  
155              @run_on_executor
156              def f(self):
157                  return 42
158  
159          o = Object()
160          answer = yield o.f()
161          self.assertEqual(answer, 42)
162  
163      @gen_test
164      def test_call_with_no_args(self):
165          class Object(object):
166              def __init__(self):
167                  self.executor = futures.thread.ThreadPoolExecutor(1)
168  
169              @run_on_executor()
170              def f(self):
171                  return 42
172  
173          o = Object()
174          answer = yield o.f()
175          self.assertEqual(answer, 42)
176  
177      @gen_test
178      def test_call_with_executor(self):
179          class Object(object):
180              def __init__(self):
181                  self.__executor = futures.thread.ThreadPoolExecutor(1)
182  
183              @run_on_executor(executor="_Object__executor")
184              def f(self):
185                  return 42
186  
187          o = Object()
188          answer = yield o.f()
189          self.assertEqual(answer, 42)
190  
191      @gen_test
192      def test_async_await(self):
193          class Object(object):
194              def __init__(self):
195                  self.executor = futures.thread.ThreadPoolExecutor(1)
196  
197              @run_on_executor()
198              def f(self):
199                  return 42
200  
201          o = Object()
202  
203          async def f():
204              answer = await o.f()
205              return answer
206  
207          result = yield f()
208          self.assertEqual(result, 42)
