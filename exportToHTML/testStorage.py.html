1    import unittest

9    class TestEnum(win32com.test.util.TestCase):
10       def testit(self):
11           fname, tmp = win32api.GetTempFileName(win32api.GetTempPath(), "stg")
12           m = storagecon.STGM_READWRITE | storagecon.STGM_SHARE_EXCLUSIVE
13           ##  file, mode, format, attrs (always 0), IID (IStorage or IPropertySetStorage, storage options(only used with STGFMT_DOCFILE)
14           pss = pythoncom.StgOpenStorageEx(
15               fname, m, storagecon.STGFMT_FILE, 0, pythoncom.IID_IPropertySetStorage
16           )
17           ###                               {"Version":2,"reserved":0,"SectorSize":512,"TemplateFile":'somefilename'})
18   
19           ## FMTID_SummaryInformation FMTID_DocSummaryInformation FMTID_UserDefinedProperties
20           psuser = pss.Create(
21               pythoncom.FMTID_UserDefinedProperties,
22               pythoncom.IID_IPropertySetStorage,
23               storagecon.PROPSETFLAG_DEFAULT,
24               storagecon.STGM_READWRITE
25               | storagecon.STGM_CREATE
26               | storagecon.STGM_SHARE_EXCLUSIVE,
27           )  ## it's very picky about flag combinations!
28           psuser.WriteMultiple((3, 4), ("hey", "bubba"))
29           psuser.WritePropertyNames((3, 4), ("property3", "property4"))
30           expected_summaries = []
31           expected_summaries.append(("property3", 3, pythoncom.VT_BSTR))
32           expected_summaries.append(("property4", 4, pythoncom.VT_BSTR))
33           psuser = None
34   
35           pssum = pss.Create(
36               pythoncom.FMTID_SummaryInformation,
37               pythoncom.IID_IPropertySetStorage,
38               storagecon.PROPSETFLAG_DEFAULT,
39               storagecon.STGM_READWRITE
40               | storagecon.STGM_CREATE
41               | storagecon.STGM_SHARE_EXCLUSIVE,
42           )
43           pssum.WriteMultiple(
44               (storagecon.PIDSI_AUTHOR, storagecon.PIDSI_COMMENTS), ("me", "comment")
45           )
46   
47           pssum = None
48           pss = None  ## doesn't seem to be a close or release method, and you can't even reopen it from the same process until previous object is gone
49   
50           pssread = pythoncom.StgOpenStorageEx(
51               fname,
52               storagecon.STGM_READ | storagecon.STGM_SHARE_EXCLUSIVE,
53               storagecon.STGFMT_FILE,
54               0,
55               pythoncom.IID_IPropertySetStorage,
56           )
57           found_summaries = []
58           for psstat in pssread:
59               ps = pssread.Open(
60                   psstat[0], storagecon.STGM_READ | storagecon.STGM_SHARE_EXCLUSIVE
61               )
62               for p in ps:
63                   p_val = ps.ReadMultiple((p[1],))[0]
64                   if (p[1] == storagecon.PIDSI_AUTHOR and p_val == "me") or (
65                       p[1] == storagecon.PIDSI_COMMENTS and p_val == "comment"
66                   ):
67                       pass
68                   else:
69                       self.fail(f"Uxexpected property {p}/{p_val}")
70               ps = None
71               ## FMTID_UserDefinedProperties can't exist without FMTID_DocSummaryInformation, and isn't returned independently from Enum
72               ## also can't be open at same time
73               if psstat[0] == pythoncom.FMTID_DocSummaryInformation:
74                   ps = pssread.Open(
75                       pythoncom.FMTID_UserDefinedProperties,
76                       storagecon.STGM_READ | storagecon.STGM_SHARE_EXCLUSIVE,
77                   )
78                   for p in ps:
79                       found_summaries.append(p)
80                   ps = None
81           psread = None
82           expected_summaries.sort()
83           found_summaries.sort()
84           self.assertEqual(expected_summaries, found_summaries)
