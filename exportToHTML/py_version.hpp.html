44   typedef const char* (*GetVersionFunc) ();

46   static PythonVersion GetPythonVersion(void *module) {
47       GetVersionFunc versionFunc;
48       *(void**)(&versionFunc) = dlsym(module, "Py_GetVersion");
49       if(versionFunc == nullptr) {
50           return PythonVersion_Unknown;
51       }
52       const char* version = versionFunc();
53   
54   #endif //_WIN32
55   
56       if (version != nullptr && strlen(version) >= 3 && version[1] == '.') {
57           if (version[0] == '2') {
58               switch (version[2]) {
59               case '5': return PythonVersion_25;
60               case '6': return PythonVersion_26;
61               case '7': return PythonVersion_27;
62               }
63           }
64           else if (version[0] == '3') {
65               switch (version[2]) {
66               case '0': return PythonVersion_30;
67               case '1':
68                   if(strlen(version) >= 4){
69                       if(version[3] == '0'){
70                           return PythonVersion_310;
71                       }
72                       if(version[3] == '1'){
73                           return PythonVersion_311;
74                       }
75                       if(version[3] == '2'){
76                           return PythonVersion_312;
77                       }
78                       if(version[3] == '3'){
79                           return PythonVersion_313;
80                       }
81                   }
82                   return PythonVersion_Unknown; // we don't care about 3.1 anymore...
83               
84               case '2': return PythonVersion_32;
85               case '3': return PythonVersion_33;
86               case '4': return PythonVersion_34;
87               case '5': return PythonVersion_35;
88               case '6': return PythonVersion_36;
89               case '7': return PythonVersion_37;
90               case '8': return PythonVersion_38;
91               case '9': return PythonVersion_39;
92               }
93           }
94       }
95       return PythonVersion_Unknown;
96   }
