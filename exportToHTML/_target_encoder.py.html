21   class TargetEncoder(OneToOneFeatureMixin, _BaseEncoder):
22       """Target Encoder for regression and classification targets. 
23    
24       Each category is encoded based on a shrunk estimate of the average target 
25       values for observations belonging to the category. The encoding scheme mixes 
26       the global target mean with the target mean conditioned on the value of the 
27       category (see [MIC]_). 
28    
29       When the target type is "multiclass", encodings are based 
30       on the conditional probability estimate for each class. The target is first 
31       binarized using the "one-vs-all" scheme via 
32       :class:`~sklearn.preprocessing.LabelBinarizer`, then the average target 
33       value for each class and each category is used for encoding, resulting in 
34       `n_features` * `n_classes` encoded output features. 
35    
36       :class:`TargetEncoder` considers missing values, such as `np.nan` or `None`, 
37       as another category and encodes them like any other category. Categories 
38       that are not seen during :meth:`fit` are encoded with the target mean, i.e. 
39       `target_mean_`. 
40    
41       For a demo on the importance of the `TargetEncoder` internal cross-fitting, 
42       see 
43       :ref:`sphx_glr_auto_examples_preprocessing_plot_target_encoder_cross_val.py`. 
44       For a comparison of different encoders, refer to 
45       :ref:`sphx_glr_auto_examples_preprocessing_plot_target_encoder.py`. Read 
46       more in the :ref:`User Guide <target_encoder>`. 
47    
48       .. note:: 
49           `fit(X, y).transform(X)` does not equal `fit_transform(X, y)` because a 
50           :term:`cross fitting` scheme is used in `fit_transform` for encoding. 
51           See the :ref:`User Guide <target_encoder>` for details. 
52    
53       .. versionadded:: 1.3 
54    
55       Parameters 
56       ---------- 
57       categories : "auto" or list of shape (n_features,) of array-like, default="auto" 
58           Categories (unique values) per feature: 
59    
60           - `"auto"` : Determine categories automatically from the training data. 
61           - list : `categories[i]` holds the categories expected in the i-th column. The 
62             passed categories should not mix strings and numeric values within a single 
63             feature, and should be sorted in case of numeric values. 
64    
65           The used categories are stored in the `categories_` fitted attribute. 
66    
67       target_type : {"auto", "continuous", "binary", "multiclass"}, default="auto" 
68           Type of target. 
69    
70           - `"auto"` : Type of target is inferred with 
71             :func:`~sklearn.utils.multiclass.type_of_target`. 
72           - `"continuous"` : Continuous target 
73           - `"binary"` : Binary target 
74           - `"multiclass"` : Multiclass target 
75    
76           .. note:: 
77               The type of target inferred with `"auto"` may not be the desired target 
78               type used for modeling. For example, if the target consisted of integers 
79               between 0 and 100, then :func:`~sklearn.utils.multiclass.type_of_target` 
80               will infer the target as `"multiclass"`. In this case, setting 
81               `target_type="continuous"` will specify the target as a regression 
82               problem. The `target_type_` attribute gives the target type used by the 
83               encoder. 
84    
85           .. versionchanged:: 1.4 
86              Added the option 'multiclass'. 
87    
88       smooth : "auto" or float, default="auto" 
89           The amount of mixing of the target mean conditioned on the value of the 
90           category with the global target mean. A larger `smooth` value will put 
91           more weight on the global target mean. 
92           If `"auto"`, then `smooth` is set to an empirical Bayes estimate. 
93    
94       cv : int, default=5 
95           Determines the number of folds in the :term:`cross fitting` strategy used in 
96           :meth:`fit_transform`. For classification targets, `StratifiedKFold` is used 
97           and for continuous targets, `KFold` is used. 
98    
99       shuffle : bool, default=True 
100          Whether to shuffle the data in :meth:`fit_transform` before splitting into 
101          folds. Note that the samples within each split will not be shuffled. 
102   
103      random_state : int, RandomState instance or None, default=None 
104          When `shuffle` is True, `random_state` affects the ordering of the 
105          indices, which controls the randomness of each fold. Otherwise, this 
106          parameter has no effect. 
107          Pass an int for reproducible output across multiple function calls. 
108          See :term:`Glossary <random_state>`. 
109   
110      Attributes 
111      ---------- 
112      encodings_ : list of shape (n_features,) or (n_features * n_classes) of \ 
113                      ndarray 
114          Encodings learnt on all of `X`. 
115          For feature `i`, `encodings_[i]` are the encodings matching the 
116          categories listed in `categories_[i]`. When `target_type_` is 
117          "multiclass", the encoding for feature `i` and class `j` is stored in 
118          `encodings_[j + (i * len(classes_))]`. E.g., for 2 features (f) and 
119          3 classes (c), encodings are ordered: 
120          f0_c0, f0_c1, f0_c2, f1_c0, f1_c1, f1_c2, 
121   
122      categories_ : list of shape (n_features,) of ndarray 
123          The categories of each input feature determined during fitting or 
124          specified in `categories` 
125          (in order of the features in `X` and corresponding with the output 
126          of :meth:`transform`). 
127   
128      target_type_ : str 
129          Type of target. 
130   
131      target_mean_ : float 
132          The overall mean of the target. This value is only used in :meth:`transform` 
133          to encode categories. 
134   
135      n_features_in_ : int 
136          Number of features seen during :term:`fit`. 
137   
138      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
139          Names of features seen during :term:`fit`. Defined only when `X` 
140          has feature names that are all strings. 
141   
142      classes_ : ndarray or None 
143          If `target_type_` is 'binary' or 'multiclass', holds the label for each class, 
144          otherwise `None`. 
145   
146      See Also 
147      -------- 
148      OrdinalEncoder : Performs an ordinal (integer) encoding of the categorical features. 
149          Contrary to TargetEncoder, this encoding is not supervised. Treating the 
150          resulting encoding as a numerical features therefore lead arbitrarily 
151          ordered values and therefore typically lead to lower predictive performance 
152          when used as preprocessing for a classifier or regressor. 
153      OneHotEncoder : Performs a one-hot encoding of categorical features. This 
154          unsupervised encoding is better suited for low cardinality categorical 
155          variables as it generate one new feature per unique category. 
156   
157      References 
158      ---------- 
159      .. [MIC] :doi:`Micci-Barreca, Daniele. "A preprocessing scheme for high-cardinality 
160         categorical attributes in classification and prediction problems" 
161         SIGKDD Explor. Newsl. 3, 1 (July 2001), 27â€“32. <10.1145/507533.507538>` 
162   
163      Examples 
164      -------- 
165      With `smooth="auto"`, the smoothing parameter is set to an empirical Bayes estimate: 
166   
167      >>> import numpy as np 
168      >>> from sklearn.preprocessing import TargetEncoder 
169      >>> X = np.array([["dog"] * 20 + ["cat"] * 30 + ["snake"] * 38], dtype=object).T 
170      >>> y = [90.3] * 5 + [80.1] * 15 + [20.4] * 5 + [20.1] * 25 + [21.2] * 8 + [49] * 30 
171      >>> enc_auto = TargetEncoder(smooth="auto") 
172      >>> X_trans = enc_auto.fit_transform(X, y) 
173   
174      >>> # A high `smooth` parameter puts more weight on global mean on the categorical 
175      >>> # encodings: 
176      >>> enc_high_smooth = TargetEncoder(smooth=5000.0).fit(X, y) 
177      >>> enc_high_smooth.target_mean_ 
178      np.float64(44...) 
179      >>> enc_high_smooth.encodings_ 
180      [array([44..., 44..., 44...])] 
181   
182      >>> # On the other hand, a low `smooth` parameter puts more weight on target 
183      >>> # conditioned on the value of the categorical: 
184      >>> enc_low_smooth = TargetEncoder(smooth=1.0).fit(X, y) 
185      >>> enc_low_smooth.encodings_ 
186      [array([20..., 80..., 43...])] 
187      """
188

193          "cv": [Interval(Integral, 2, None, closed="left")],

198      def __init__(
199          self,
200          categories="auto",
201          target_type="auto",
202          smooth="auto",
203          cv=5,
204          shuffle=True,
205          random_state=None,
206      ):
207          self.categories = categories
208          self.smooth = smooth
209          self.target_type = target_type
210          self.cv = cv
211          self.shuffle = shuffle
212          self.random_state = random_state

214      @_fit_context(prefer_skip_nested_validation=True)
215      def fit(self, X, y):
216          """Fit the :class:`TargetEncoder` to X and y. 
217   
218          Parameters 
219          ---------- 
220          X : array-like of shape (n_samples, n_features) 
221              The data to determine the categories of each feature. 
222   
223          y : array-like of shape (n_samples,) 
224              The target data used to encode the categories. 
225   
226          Returns 
227          ------- 
228          self : object 
229              Fitted encoder. 
230          """
231          self._fit_encodings_all(X, y)
232          return self
233  
234      @_fit_context(prefer_skip_nested_validation=True)
235      def fit_transform(self, X, y):
236          """Fit :class:`TargetEncoder` and transform X with the target encoding. 
237   
238          .. note:: 
239              `fit(X, y).transform(X)` does not equal `fit_transform(X, y)` because a 
240              :term:`cross fitting` scheme is used in `fit_transform` for encoding. 
241              See the :ref:`User Guide <target_encoder>`. for details. 
242   
243          Parameters 
244          ---------- 
245          X : array-like of shape (n_samples, n_features) 
246              The data to determine the categories of each feature. 
247   
248          y : array-like of shape (n_samples,) 
249              The target data used to encode the categories. 
250   
251          Returns 
252          ------- 
253          X_trans : ndarray of shape (n_samples, n_features) or \ 
254                      (n_samples, (n_features * n_classes)) 
255              Transformed input. 
256          """
257          from ..model_selection import KFold, StratifiedKFold  # avoid circular import
258  
259          X_ordinal, X_known_mask, y_encoded, n_categories = self._fit_encodings_all(X, y)
260  
261          # The cv splitter is voluntarily restricted to *KFold to enforce non
262          # overlapping validation folds, otherwise the fit_transform output will
263          # not be well-specified.
264          if self.target_type_ == "continuous":
265              cv = KFold(self.cv, shuffle=self.shuffle, random_state=self.random_state)
266          else:
267              cv = StratifiedKFold(
268                  self.cv, shuffle=self.shuffle, random_state=self.random_state
269              )
270  
271          # If 'multiclass' multiply axis=1 by num classes else keep shape the same
272          if self.target_type_ == "multiclass":
273              X_out = np.empty(
274                  (X_ordinal.shape[0], X_ordinal.shape[1] * len(self.classes_)),
275                  dtype=np.float64,
276              )
277          else:
278              X_out = np.empty_like(X_ordinal, dtype=np.float64)
279  
280          for train_idx, test_idx in cv.split(X, y):
281              X_train, y_train = X_ordinal[train_idx, :], y_encoded[train_idx]
282              y_train_mean = np.mean(y_train, axis=0)
283  
284              if self.target_type_ == "multiclass":
285                  encodings = self._fit_encoding_multiclass(
286                      X_train,
287                      y_train,
288                      n_categories,
289                      y_train_mean,
290                  )
291              else:
292                  encodings = self._fit_encoding_binary_or_continuous(
293                      X_train,
294                      y_train,
295                      n_categories,
296                      y_train_mean,
297                  )
298              self._transform_X_ordinal(
299                  X_out,
300                  X_ordinal,
301                  ~X_known_mask,
302                  test_idx,
303                  encodings,
304                  y_train_mean,
305              )
306          return X_out
307  
308      def transform(self, X):
309          """Transform X with the target encoding. 
310   
311          .. note:: 
312              `fit(X, y).transform(X)` does not equal `fit_transform(X, y)` because a 
313              :term:`cross fitting` scheme is used in `fit_transform` for encoding. 
314              See the :ref:`User Guide <target_encoder>`. for details. 
315   
316          Parameters 
317          ---------- 
318          X : array-like of shape (n_samples, n_features) 
319              The data to determine the categories of each feature. 
320   
321          Returns 
322          ------- 
323          X_trans : ndarray of shape (n_samples, n_features) or \ 
324                      (n_samples, (n_features * n_classes)) 
325              Transformed input. 
326          """
327          X_ordinal, X_known_mask = self._transform(
328              X, handle_unknown="ignore", ensure_all_finite="allow-nan"
329          )
330  
331          # If 'multiclass' multiply axis=1 by num of classes else keep shape the same
332          if self.target_type_ == "multiclass":
333              X_out = np.empty(
334                  (X_ordinal.shape[0], X_ordinal.shape[1] * len(self.classes_)),
335                  dtype=np.float64,
336              )
337          else:
338              X_out = np.empty_like(X_ordinal, dtype=np.float64)
339  
340          self._transform_X_ordinal(
341              X_out,
342              X_ordinal,
343              ~X_known_mask,
344              slice(None),
345              self.encodings_,
346              self.target_mean_,
347          )
348          return X_out
349  
350      def _fit_encodings_all(self, X, y):
351          """Fit a target encoding with all the data."""
352          # avoid circular import
353          from ..preprocessing import (
354              LabelBinarizer,
355              LabelEncoder,
356          )
357  
358          check_consistent_length(X, y)
359          self._fit(X, handle_unknown="ignore", ensure_all_finite="allow-nan")
360  
361          if self.target_type == "auto":
362              accepted_target_types = ("binary", "multiclass", "continuous")
363              inferred_type_of_target = type_of_target(y, input_name="y")
364              if inferred_type_of_target not in accepted_target_types:
365                  raise ValueError(
366                      "Unknown label type: Target type was inferred to be "
367                      f"{inferred_type_of_target!r}. Only {accepted_target_types} are "
368                      "supported."
369                  )
370              self.target_type_ = inferred_type_of_target
371          else:
372              self.target_type_ = self.target_type
373  
374          self.classes_ = None
375          if self.target_type_ == "binary":
376              label_encoder = LabelEncoder()
377              y = label_encoder.fit_transform(y)
378              self.classes_ = label_encoder.classes_
379          elif self.target_type_ == "multiclass":
380              label_binarizer = LabelBinarizer()
381              y = label_binarizer.fit_transform(y)
382              self.classes_ = label_binarizer.classes_
383          else:  # continuous
384              y = _check_y(y, y_numeric=True, estimator=self)
385  
386          self.target_mean_ = np.mean(y, axis=0)
387  
388          X_ordinal, X_known_mask = self._transform(
389              X, handle_unknown="ignore", ensure_all_finite="allow-nan"
390          )
391          n_categories = np.fromiter(
392              (len(category_for_feature) for category_for_feature in self.categories_),
393              dtype=np.int64,
394              count=len(self.categories_),
395          )
396          if self.target_type_ == "multiclass":
397              encodings = self._fit_encoding_multiclass(
398                  X_ordinal,
399                  y,
400                  n_categories,
401                  self.target_mean_,
402              )
403          else:
404              encodings = self._fit_encoding_binary_or_continuous(
405                  X_ordinal,
406                  y,
407                  n_categories,
408                  self.target_mean_,
409              )
410          self.encodings_ = encodings
411  
412          return X_ordinal, X_known_mask, y, n_categories
413  
414      def _fit_encoding_binary_or_continuous(
415          self, X_ordinal, y, n_categories, target_mean
416      ):
417          """Learn target encodings."""
418          if self.smooth == "auto":
419              y_variance = np.var(y)
420              encodings = _fit_encoding_fast_auto_smooth(
421                  X_ordinal,
422                  y,
423                  n_categories,
424                  target_mean,
425                  y_variance,
426              )
427          else:
428              encodings = _fit_encoding_fast(
429                  X_ordinal,
430                  y,
431                  n_categories,
432                  self.smooth,
433                  target_mean,
434              )
435          return encodings
436  
437      def _fit_encoding_multiclass(self, X_ordinal, y, n_categories, target_mean):
438          """Learn multiclass encodings. 
439   
440          Learn encodings for each class (c) then reorder encodings such that 
441          the same features (f) are grouped together. `reorder_index` enables 
442          converting from: 
443          f0_c0, f1_c0, f0_c1, f1_c1, f0_c2, f1_c2 
444          to: 
445          f0_c0, f0_c1, f0_c2, f1_c0, f1_c1, f1_c2 
446          """
447          n_features = self.n_features_in_
448          n_classes = len(self.classes_)
449  
450          encodings = []
451          for i in range(n_classes):
452              y_class = y[:, i]
453              encoding = self._fit_encoding_binary_or_continuous(
454                  X_ordinal,
455                  y_class,
456                  n_categories,
457                  target_mean[i],
458              )
459              encodings.extend(encoding)
460  
461          reorder_index = (
462              idx
463              for start in range(n_features)
464              for idx in range(start, (n_classes * n_features), n_features)
465          )
466          return [encodings[idx] for idx in reorder_index]
467  
468      def _transform_X_ordinal(
469          self,
470          X_out,
471          X_ordinal,
472          X_unknown_mask,
473          row_indices,
474          encodings,
475          target_mean,
476      ):
477          """Transform X_ordinal using encodings. 
478   
479          In the multiclass case, `X_ordinal` and `X_unknown_mask` have column 
480          (axis=1) size `n_features`, while `encodings` has length of size 
481          `n_features * n_classes`. `feat_idx` deals with this by repeating 
482          feature indices by `n_classes` E.g., for 3 features, 2 classes: 
483          0,0,1,1,2,2 
484   
485          Additionally, `target_mean` is of shape (`n_classes`,) so `mean_idx` 
486          cycles through 0 to `n_classes` - 1, `n_features` times. 
487          """
488          if self.target_type_ == "multiclass":
489              n_classes = len(self.classes_)
490              for e_idx, encoding in enumerate(encodings):
491                  # Repeat feature indices by n_classes
492                  feat_idx = e_idx // n_classes
493                  # Cycle through each class
494                  mean_idx = e_idx % n_classes
495                  X_out[row_indices, e_idx] = encoding[X_ordinal[row_indices, feat_idx]]
496                  X_out[X_unknown_mask[:, feat_idx], e_idx] = target_mean[mean_idx]
497          else:
498              for e_idx, encoding in enumerate(encodings):
499                  X_out[row_indices, e_idx] = encoding[X_ordinal[row_indices, e_idx]]
500                  X_out[X_unknown_mask[:, e_idx], e_idx] = target_mean
501  
502      def get_feature_names_out(self, input_features=None):
503          """Get output feature names for transformation. 
504   
505          Parameters 
506          ---------- 
507          input_features : array-like of str or None, default=None 
508              Not used, present here for API consistency by convention. 
509   
510          Returns 
511          ------- 
512          feature_names_out : ndarray of str objects 
513              Transformed feature names. `feature_names_in_` is used unless it is 
514              not defined, in which case the following input feature names are 
515              generated: `["x0", "x1", ..., "x(n_features_in_ - 1)"]`. 
516              When `type_of_target_` is "multiclass" the names are of the format 
517              '<feature_name>_<class_name>'. 
518          """
519          check_is_fitted(self, "n_features_in_")
520          feature_names = _check_feature_names_in(self, input_features)
521          if self.target_type_ == "multiclass":
522              feature_names = [
523                  f"{feature_name}_{class_name}"
524                  for feature_name in feature_names
525                  for class_name in self.classes_
526              ]
527              return np.asarray(feature_names, dtype=object)
528          else:
529              return feature_names
