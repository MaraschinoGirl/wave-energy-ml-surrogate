1    from typing import Any, TypeAlias
2    
3    import numpy as np
4    
5    # NOTE: `_StrLike_co` and `_BytesLike_co` are pointless, as `np.str_` and
6    # `np.bytes_` are already subclasses of their builtin counterpart
7    
8    _CharLike_co: TypeAlias = str | bytes
9    
10   # The 6 `<X>Like_co` type-aliases below represent all scalars that can be
11   # coerced into `<X>` (with the casting rule `same_kind`)
12   _BoolLike_co: TypeAlias = bool | np.bool
13   _UIntLike_co: TypeAlias = np.unsignedinteger[Any] | _BoolLike_co
14   _IntLike_co: TypeAlias = int | np.integer[Any] | _BoolLike_co
15   _FloatLike_co: TypeAlias = float | np.floating[Any] | _IntLike_co
16   _ComplexLike_co: TypeAlias = (
17       complex
18       | np.complexfloating[Any, Any]
19       | _FloatLike_co
20   )
21   _TD64Like_co: TypeAlias = np.timedelta64 | _IntLike_co
22   
23   _NumberLike_co: TypeAlias = int | float | complex | np.number[Any] | np.bool
24   _ScalarLike_co: TypeAlias = int | float | complex | str | bytes | np.generic
25   
26   # `_VoidLike_co` is technically not a scalar, but it's close enough
27   _VoidLike_co: TypeAlias = tuple[Any, ...] | np.void
