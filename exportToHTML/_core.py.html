39   class WebSocket:
40       """ 
41       Low level WebSocket interface. 
42    
43       This class is based on the WebSocket protocol `draft-hixie-thewebsocketprotocol-76 <http://tools.ietf.org/html/draft-hixie-thewebsocketprotocol-76>`_ 
44    
45       We can connect to the websocket server and send/receive data. 
46       The following example is an echo client. 
47    
48       >>> import websocket 
49       >>> ws = websocket.WebSocket() 
50       >>> ws.connect("ws://echo.websocket.events") 
51       >>> ws.recv() 
52       'echo.websocket.events sponsored by Lob.com' 
53       >>> ws.send("Hello, Server") 
54       19 
55       >>> ws.recv() 
56       'Hello, Server' 
57       >>> ws.close() 
58    
59       Parameters 
60       ---------- 
61       get_mask_key: func 
62           A callable function to get new mask keys, see the 
63           WebSocket.set_mask_key's docstring for more information. 
64       sockopt: tuple 
65           Values for socket.setsockopt. 
66           sockopt must be tuple and each element is argument of sock.setsockopt. 
67       sslopt: dict 
68           Optional dict object for ssl socket options. See FAQ for details. 
69       fire_cont_frame: bool 
70           Fire recv event for each cont frame. Default is False. 
71       enable_multithread: bool 
72           If set to True, lock send method. 
73       skip_utf8_validation: bool 
74           Skip utf8 validation. 
75       """
76   
77       def __init__(
78           self,
79           get_mask_key=None,
80           sockopt=None,
81           sslopt=None,
82           fire_cont_frame: bool = False,
83           enable_multithread: bool = True,
84           skip_utf8_validation: bool = False,
85           **_,
86       ):
87           """ 
88           Initialize WebSocket object. 
89    
90           Parameters 
91           ---------- 
92           sslopt: dict 
93               Optional dict object for ssl socket options. See FAQ for details. 
94           """
95           self.sock_opt = sock_opt(sockopt, sslopt)
96           self.handshake_response = None
97           self.sock: Optional[socket.socket] = None
98   
99           self.connected = False
100          self.get_mask_key = get_mask_key
101          # These buffer over the build-up of a single frame.
102          self.frame_buffer = frame_buffer(self._recv, skip_utf8_validation)
103          self.cont_frame = continuous_frame(fire_cont_frame, skip_utf8_validation)
104  
105          if enable_multithread:
106              self.lock = threading.Lock()
107              self.readlock = threading.Lock()
108          else:
109              self.lock = NoLock()
110              self.readlock = NoLock()

112      def __iter__(self):
113          """ 
114          Allow iteration over websocket, implying sequential `recv` executions. 
115          """
116          while True:
117              yield self.recv()

208      def connect(self, url, **options):
209          """ 
210          Connect to url. url is websocket url scheme. 
211          ie. ws://host:port/resource 
212          You can customize using 'options'. 
213          If you set "header" list object, you can set your own custom header. 
214   
215          >>> ws = WebSocket() 
216          >>> ws.connect("ws://echo.websocket.events", 
217                  ...     header=["User-Agent: MyProgram", 
218                  ...             "x-custom: header"]) 
219   
220          Parameters 
221          ---------- 
222          header: list or dict 
223              Custom http header list or dict. 
224          cookie: str 
225              Cookie value. 
226          origin: str 
227              Custom origin url. 
228          connection: str 
229              Custom connection header value. 
230              Default value "Upgrade" set in _handshake.py 
231          suppress_origin: bool 
232              Suppress outputting origin header. 
233          host: str 
234              Custom host header string. 
235          timeout: int or float 
236              Socket timeout time. This value is an integer or float. 
237              If you set None for this value, it means "use default_timeout value" 
238          http_proxy_host: str 
239              HTTP proxy host name. 
240          http_proxy_port: str or int 
241              HTTP proxy port. Default is 80. 
242          http_no_proxy: list 
243              Whitelisted host names that don't use the proxy. 
244          http_proxy_auth: tuple 
245              HTTP proxy auth information. Tuple of username and password. Default is None. 
246          http_proxy_timeout: int or float 
247              HTTP proxy timeout, default is 60 sec as per python-socks. 
248          redirect_limit: int 
249              Number of redirects to follow. 
250          subprotocols: list 
251              List of available subprotocols. Default is None. 
252          socket: socket 
253              Pre-initialized stream socket. 
254          """
255          self.sock_opt.timeout = options.get("timeout", self.sock_opt.timeout)
256          self.sock, addrs = connect(
257              url, self.sock_opt, proxy_info(**options), options.pop("socket", None)
258          )
259  
260          try:
261              self.handshake_response = handshake(self.sock, url, *addrs, **options)
262              for _ in range(options.pop("redirect_limit", 3)):
263                  if self.handshake_response.status in SUPPORTED_REDIRECT_STATUSES:
264                      url = self.handshake_response.headers["location"]
265                      self.sock.close()
266                      self.sock, addrs = connect(
267                          url,
268                          self.sock_opt,
269                          proxy_info(**options),
270                          options.pop("socket", None),
271                      )
272                      self.handshake_response = handshake(
273                          self.sock, url, *addrs, **options
274                      )
275              self.connected = True
276          except:
277              if self.sock:
278                  self.sock.close()
279                  self.sock = None
280              raise

282      def send(self, payload: Union[bytes, str], opcode: int = ABNF.OPCODE_TEXT) -> int:
283          """ 
284          Send the data as string. 
285   
286          Parameters 
287          ---------- 
288          payload: str 
289              Payload must be utf-8 string or unicode, 
290              If the opcode is OPCODE_TEXT. 
291              Otherwise, it must be string(byte array). 
292          opcode: int 
293              Operation code (opcode) to send. 
294          """
295  
296          frame = ABNF.create_frame(payload, opcode)
297          return self.send_frame(frame)

311      def send_frame(self, frame) -> int:
312          """ 
313          Send the data frame. 
314   
315          >>> ws = create_connection("ws://echo.websocket.events") 
316          >>> frame = ABNF.create_frame("Hello", ABNF.OPCODE_TEXT) 
317          >>> ws.send_frame(frame) 
318          >>> cont_frame = ABNF.create_frame("My name is ", ABNF.OPCODE_CONT, 0) 
319          >>> ws.send_frame(frame) 
320          >>> cont_frame = ABNF.create_frame("Foo Bar", ABNF.OPCODE_CONT, 1) 
321          >>> ws.send_frame(frame) 
322   
323          Parameters 
324          ---------- 
325          frame: ABNF frame 
326              frame data created by ABNF.create_frame 
327          """
328          if self.get_mask_key:
329              frame.get_mask_key = self.get_mask_key
330          data = frame.format()
331          length = len(data)
332          if isEnabledForTrace():
333              trace(f"++Sent raw: {repr(data)}")
334              trace(f"++Sent decoded: {frame.__str__()}")
335          with self.lock:
336              while data:
337                  l = self._send(data)
338                  data = data[l:]
339  
340          return length

379      def recv(self) -> Union[str, bytes]:
380          """ 
381          Receive string data(byte array) from the server. 
382   
383          Returns 
384          ---------- 
385          data: string (byte array) value. 
386          """
387          with self.readlock:
388              opcode, data = self.recv_data()
389          if opcode == ABNF.OPCODE_TEXT:
390              data_received: Union[bytes, str] = data
391              if isinstance(data_received, bytes):
392                  return data_received.decode("utf-8")
393              elif isinstance(data_received, str):
394                  return data_received
395          elif opcode == ABNF.OPCODE_BINARY:
396              data_binary: bytes = data
397              return data_binary
398          else:
399              return ""

401      def recv_data(self, control_frame: bool = False) -> tuple:
402          """ 
403          Receive data with operation code. 
404   
405          Parameters 
406          ---------- 
407          control_frame: bool 
408              a boolean flag indicating whether to return control frame 
409              data, defaults to False 
410   
411          Returns 
412          ------- 
413          opcode, frame.data: tuple 
414              tuple of operation code and string(byte array) value. 
415          """
416          opcode, frame = self.recv_data_frame(control_frame)
417          return opcode, frame.data

419      def recv_data_frame(self, control_frame: bool = False) -> tuple:
420          """ 
421          Receive data with operation code. 
422   
423          If a valid ping message is received, a pong response is sent. 
424   
425          Parameters 
426          ---------- 
427          control_frame: bool 
428              a boolean flag indicating whether to return control frame 
429              data, defaults to False 
430   
431          Returns 
432          ------- 
433          frame.opcode, frame: tuple 
434              tuple of operation code and string(byte array) value. 
435          """
436          while True:
437              frame = self.recv_frame()
438              if isEnabledForTrace():
439                  trace(f"++Rcv raw: {repr(frame.format())}")
440                  trace(f"++Rcv decoded: {frame.__str__()}")
441              if not frame:
442                  # handle error:
443                  # 'NoneType' object has no attribute 'opcode'
444                  raise WebSocketProtocolException(f"Not a valid frame {frame}")
445              elif frame.opcode in (
446                  ABNF.OPCODE_TEXT,
447                  ABNF.OPCODE_BINARY,
448                  ABNF.OPCODE_CONT,
449              ):
450                  self.cont_frame.validate(frame)
451                  self.cont_frame.add(frame)
452  
453                  if self.cont_frame.is_fire(frame):
454                      return self.cont_frame.extract(frame)
455  
456              elif frame.opcode == ABNF.OPCODE_CLOSE:
457                  self.send_close()
458                  return frame.opcode, frame
459              elif frame.opcode == ABNF.OPCODE_PING:
460                  if len(frame.data) < 126:
461                      self.pong(frame.data)
462                  else:
463                      raise WebSocketProtocolException("Ping message is too long")
464                  if control_frame:
465                      return frame.opcode, frame
466              elif frame.opcode == ABNF.OPCODE_PONG:
467                  if control_frame:
468                      return frame.opcode, frame

480      def send_close(self, status: int = STATUS_NORMAL, reason: bytes = b""):
481          """ 
482          Send close data to the server. 
483   
484          Parameters 
485          ---------- 
486          status: int 
487              Status code to send. See STATUS_XXX. 
488          reason: str or bytes 
489              The reason to close. This must be string or UTF-8 bytes. 
490          """
491          if status < 0 or status >= ABNF.LENGTH_16:
492              raise ValueError("code is invalid range")
493          self.connected = False
494          self.send(struct.pack("!H", status) + reason, ABNF.OPCODE_CLOSE)
495  
496      def close(self, status: int = STATUS_NORMAL, reason: bytes = b"", timeout: int = 3):
497          """ 
498          Close Websocket object 
499   
500          Parameters 
501          ---------- 
502          status: int 
503              Status code to send. See VALID_CLOSE_STATUS in ABNF. 
504          reason: bytes 
505              The reason to close in UTF-8. 
506          timeout: int or float 
507              Timeout until receive a close frame. 
508              If None, it will wait forever until receive a close frame. 
509          """
510          if not self.connected:
511              return
512          if status < 0 or status >= ABNF.LENGTH_16:
513              raise ValueError("code is invalid range")
514  
515          try:
516              self.connected = False
517              self.send(struct.pack("!H", status) + reason, ABNF.OPCODE_CLOSE)
518              sock_timeout = self.sock.gettimeout()
519              self.sock.settimeout(timeout)
520              start_time = time.time()
521              while timeout is None or time.time() - start_time < timeout:
522                  try:
523                      frame = self.recv_frame()
524                      if frame.opcode != ABNF.OPCODE_CLOSE:
525                          continue
526                      if isEnabledForError():
527                          recv_status = struct.unpack("!H", frame.data[0:2])[0]
528                          if recv_status >= 3000 and recv_status <= 4999:
529                              debug(f"close status: {repr(recv_status)}")
530                          elif recv_status != STATUS_NORMAL:
531                              error(f"close status: {repr(recv_status)}")
532                      break
533                  except:
534                      break
535              self.sock.settimeout(sock_timeout)
536              self.sock.shutdown(socket.SHUT_RDWR)
537          except:
538              pass
539  
540          self.shutdown()
541  
542      def abort(self):
543          """ 
544          Low-level asynchronous abort, wakes up other threads that are waiting in recv_* 
545          """
546          if self.connected:
547              self.sock.shutdown(socket.SHUT_RDWR)
548  
549      def shutdown(self):
550          """ 
551          close socket, immediately. 
552          """
553          if self.sock:
554              self.sock.close()
555              self.sock = None
556              self.connected = False
557  
558      def _send(self, data: Union[str, bytes]):
559          return send(self.sock, data)
560  
561      def _recv(self, bufsize):
562          try:
563              return recv(self.sock, bufsize)
564          except WebSocketConnectionClosedException:
565              if self.sock:
566                  self.sock.close()
567              self.sock = None
568              self.connected = False
569              raise

572  def create_connection(url: str, timeout=None, class_=WebSocket, **options):
573      """ 
574      Connect to url and return websocket object. 
575   
576      Connect to url and return the WebSocket object. 
577      Passing optional timeout parameter will set the timeout on the socket. 
578      If no timeout is supplied, 
579      the global default timeout setting returned by getdefaulttimeout() is used. 
580      You can customize using 'options'. 
581      If you set "header" list object, you can set your own custom header. 
582   
583      >>> conn = create_connection("ws://echo.websocket.events", 
584           ...     header=["User-Agent: MyProgram", 
585           ...             "x-custom: header"]) 
586   
587      Parameters 
588      ---------- 
589      class_: class 
590          class to instantiate when creating the connection. It has to implement 
591          settimeout and connect. It's __init__ should be compatible with 
592          WebSocket.__init__, i.e. accept all of it's kwargs. 
593      header: list or dict 
594          custom http header list or dict. 
595      cookie: str 
596          Cookie value. 
597      origin: str 
598          custom origin url. 
599      suppress_origin: bool 
600          suppress outputting origin header. 
601      host: str 
602          custom host header string. 
603      timeout: int or float 
604          socket timeout time. This value could be either float/integer. 
605          If set to None, it uses the default_timeout value. 
606      http_proxy_host: str 
607          HTTP proxy host name. 
608      http_proxy_port: str or int 
609          HTTP proxy port. If not set, set to 80. 
610      http_no_proxy: list 
611          Whitelisted host names that don't use the proxy. 
612      http_proxy_auth: tuple 
613          HTTP proxy auth information. tuple of username and password. Default is None. 
614      http_proxy_timeout: int or float 
615          HTTP proxy timeout, default is 60 sec as per python-socks. 
616      enable_multithread: bool 
617          Enable lock for multithread. 
618      redirect_limit: int 
619          Number of redirects to follow. 
620      sockopt: tuple 
621          Values for socket.setsockopt. 
622          sockopt must be a tuple and each element is an argument of sock.setsockopt. 
623      sslopt: dict 
624          Optional dict object for ssl socket options. See FAQ for details. 
625      subprotocols: list 
626          List of available subprotocols. Default is None. 
627      skip_utf8_validation: bool 
628          Skip utf8 validation. 
629      socket: socket 
630          Pre-initialized stream socket. 
631      """
632      sockopt = options.pop("sockopt", [])
633      sslopt = options.pop("sslopt", {})
634      fire_cont_frame = options.pop("fire_cont_frame", False)
635      enable_multithread = options.pop("enable_multithread", True)
636      skip_utf8_validation = options.pop("skip_utf8_validation", False)
637      websock = class_(
638          sockopt=sockopt,
639          sslopt=sslopt,
640          fire_cont_frame=fire_cont_frame,
641          enable_multithread=enable_multithread,
642          skip_utf8_validation=skip_utf8_validation,
643          **options,
644      )
645      websock.settimeout(timeout if timeout is not None else getdefaulttimeout())
646      websock.connect(url, **options)
647      return websock
