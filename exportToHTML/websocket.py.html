21   def _cast_unicode(s: str | bytes) -> str:
22       if isinstance(s, bytes):
23           return s.decode("utf-8")
24       return s
25   
26   
27   class TermSocket(tornado.websocket.WebSocketHandler):
28       """Handler for a terminal websocket"""
29   
30       def initialize(self, term_manager: TermManagerBase) -> None:
31           """Initialize the handler."""
32           self.term_manager = term_manager
33           self.term_name = ""
34           self.size = (None, None)
35           self.terminal: PtyWithClients | None = None
36           self._blocking_io_executor = term_manager.blocking_io_executor
37   
38           self._logger = logging.getLogger(__name__)
39           self._user_command = ""
40   
41           # Enable if the environment variable LOG_TERMINAL_OUTPUT is "true"
42           self._enable_output_logging = str.lower(os.getenv("LOG_TERMINAL_OUTPUT", "false")) == "true"
43   
44       def origin_check(self, origin: str | None = None) -> bool:
45           """Deprecated: backward-compat for terminado <= 0.5."""
46           origin = origin or self.request.headers.get("Origin", "")
47           assert origin is not None
48           return self.check_origin(origin)
49   
50       def open(self, url_component: Any = None) -> None:  # type:ignore[override]
51           """Websocket connection opened. 
52    
53           Call our terminal manager to get a terminal, and connect to it as a 
54           client. 
55           """
56           # Jupyter has a mixin to ping websockets and keep connections through
57           # proxies alive. Call super() to allow that to set up:
58           super().open(url_component)
59   
60           self._logger.info("TermSocket.open: %s", url_component)
61   
62           url_component = _cast_unicode(url_component)
63           self.term_name = url_component or "tty"
64           self.terminal = self.term_manager.get_terminal(url_component)
65           self.terminal.clients.append(self)
66           self.send_json_message(["setup", {}])
67           self._logger.info("TermSocket.open: Opened %s", self.term_name)
68           # Now drain the preopen buffer, if reconnect.
69           buffered = ""
70           preopen_buffer = self.terminal.read_buffer.copy()
71           while True:
72               if not preopen_buffer:
73                   break
74               s = preopen_buffer.popleft()
75               buffered += s
76           if buffered:
77               self.on_pty_read(buffered)
78   
79       def on_pty_read(self, text: str) -> None:
80           """Data read from pty; send to frontend"""
81           self.send_json_message(["stdout", text])
82   
83       def send_json_message(self, content: Any) -> None:
84           """Send a json message on the socket."""
85           json_msg = json.dumps(content)
86           self.write_message(json_msg)
87   
88           if self._enable_output_logging and content[0] == "stdout" and isinstance(content[1], str):
89               self.log_terminal_output(f"STDOUT: {content[1]}")
90   
91       @gen.coroutine
92       def on_message(self, message: str) -> None:  # type:ignore[misc]
93           """Handle incoming websocket message 
94    
95           We send JSON arrays, where the first element is a string indicating 
96           what kind of message this is. Data associated with the message follows. 
97           """
98           # logging.info("TermSocket.on_message: %s - (%s) %s", self.term_name, type(message), len(message) if isinstance(message, bytes) else message[:250])
99           command = json.loads(message)
100          msg_type = command[0]
101          assert self.terminal is not None
102          if msg_type == "stdin":
103              yield self.stdin_to_ptyproc(command[1])
104              if self._enable_output_logging:
105                  if command[1] == "\r":
106                      self.log_terminal_output(f"STDIN: {self._user_command}")
107                      self._user_command = ""
108                  else:
109                      self._user_command += command[1]
110          elif msg_type == "set_size":
111              self.size = command[1:3]
112              self.terminal.resize_to_smallest()
113  
114      def on_close(self) -> None:
115          """Handle websocket closing. 
116   
117          Disconnect from our terminal, and tell the terminal manager we're 
118          disconnecting. 
119          """
120          self._logger.info("Websocket closed")
121          if self.terminal:
122              self.terminal.clients.remove(self)
123              self.terminal.resize_to_smallest()
124          self.term_manager.client_disconnected(self)
125  
126      def on_pty_died(self) -> None:
127          """Terminal closed: tell the frontend, and close the socket."""
128          self.send_json_message(["disconnect", 1])
129          self.close()
130          self.terminal = None
131  
132      def log_terminal_output(self, log: str = "") -> None:
133          """ 
134          Logs the terminal input/output 
135          :param log: log line to write 
136          :return: 
137          """
138          self._logger.debug(log)
139  
140      @run_on_executor(executor="_blocking_io_executor")
141      def stdin_to_ptyproc(self, text: str) -> None:
142          """Handles stdin messages sent on the websocket. 
143   
144          This is a blocking call that should NOT be performed inside the 
145          server primary event loop thread. Messages must be handled 
146          asynchronously to prevent blocking on the PTY buffer. 
147          """
148          if self.terminal is not None:
149              self.terminal.ptyproc.write(text)
