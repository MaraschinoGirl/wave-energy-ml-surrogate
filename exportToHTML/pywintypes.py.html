1    # Magic utility that "redirects" to pywintypesXX.dll
2    import importlib.machinery
3    import importlib.util
4    import os
5    import sys
6    
7    
8    def __import_pywin32_system_module__(modname, globs):
9        # This has been through a number of iterations.  The problem: how to
10       # locate pywintypesXX.dll when it may be in a number of places, and how
11       # to avoid ever loading it twice.  This problem is compounded by the
12       # fact that the "right" way to do this requires win32api, but this
13       # itself requires pywintypesXX.
14       # And the killer problem is that someone may have done 'import win32api'
15       # before this code is called.  In that case Windows will have already
16       # loaded pywintypesXX as part of loading win32api - but by the time
17       # we get here, we may locate a different one.  This appears to work, but
18       # then starts raising bizarre TypeErrors complaining that something
19       # is not a pywintypes type when it clearly is!
20   
21       # So in what we hope is the last major iteration of this, we now
22       # rely on a _win32sysloader module, implemented in C but not relying
23       # on pywintypesXX.dll.  It then can check if the DLL we are looking for
24       # lib is already loaded.
25       # See if this is a debug build.
26       suffix = "_d" if "_d.pyd" in importlib.machinery.EXTENSION_SUFFIXES else ""
27       filename = "%s%d%d%s.dll" % (
28           modname,
29           sys.version_info.major,
30           sys.version_info.minor,
31           suffix,
32       )
33       if hasattr(sys, "frozen"):
34           # If we are running from a frozen program (py2exe, McMillan, freeze, PyInstaller)
35           # then we try and load the DLL from our sys.path
36           # XXX - This path may also benefit from _win32sysloader?  However,
37           # MarkH has never seen the DLL load problem with py2exe programs...
38           for look in sys.path:
39               # If the sys.path entry is a (presumably) .zip file, use the
40               # directory
41               if os.path.isfile(look):
42                   look = os.path.dirname(look)
43               found = os.path.join(look, filename)
44               if os.path.isfile(found):
45                   break
46           else:
47               raise ImportError(f"Module '{modname}' isn't in frozen sys.path {sys.path}")
48       else:
49           # First see if it already in our process - if so, we must use that.
50           import _win32sysloader
51   
52           found = _win32sysloader.GetModuleFilename(filename)
53           if found is None:
54               # We ask Windows to load it next.  This is in an attempt to
55               # get the exact same module loaded should pywintypes be imported
56               # first (which is how we are here) or if, eg, win32api was imported
57               # first thereby implicitly loading the DLL.
58   
59               # Sadly though, it doesn't quite work - if pywintypesXX.dll
60               # is in system32 *and* the executable's directory, on XP SP2, an
61               # import of win32api will cause Windows to load pywintypes
62               # from system32, where LoadLibrary for that name will
63               # load the one in the exe's dir.
64               # That shouldn't really matter though, so long as we only ever
65               # get one loaded.
66               found = _win32sysloader.LoadModule(filename)
67           if found is None:
68               # Windows can't find it - which although isn't relevent here,
69               # means that we *must* be the first win32 import, as an attempt
70               # to import win32api etc would fail when Windows attempts to
71               # locate the DLL.
72               # This is most likely to happen for "non-admin" installs, where
73               # we can't put the files anywhere else on the global path.
74   
75               # If there is a version in our Python directory, use that
76               if os.path.isfile(os.path.join(sys.prefix, filename)):
77                   found = os.path.join(sys.prefix, filename)
78           if found is None:
79               # Not in the Python directory?  Maybe we were installed via
80               # easy_install...
81               if os.path.isfile(os.path.join(os.path.dirname(__file__), filename)):
82                   found = os.path.join(os.path.dirname(__file__), filename)
83   
84           # There are 2 site-packages directories - one "global" and one "user".
85           # We could be in either, or both (but with different versions!). Factors include
86           # virtualenvs, post-install script being run or not, `pip install` flags, etc.
87   
88           # In a worst-case, it means, say 'python -c "import win32api"'
89           # will not work but 'python -c "import pywintypes, win32api"' will,
90           # but it's better than nothing.
91   
92           # We use the same logic as pywin32_bootstrap to find potential location for the dll
93           # Simply import pywin32_system32 and look in the paths in pywin32_system32.__path__
94   
95           if found is None:
96               import pywin32_system32
97   
98               for path in pywin32_system32.__path__:
99                   maybe = os.path.join(path, filename)
100                  if os.path.isfile(maybe):
101                      found = maybe
102                      break
103  
104          if found is None:
105              # give up in disgust.
106              raise ImportError(f"No system module '{modname}' ({filename})")
107      # After importing the module, sys.modules is updated to the DLL we just
108      # loaded - which isn't what we want. So we update sys.modules to refer to
109      # this module, and update our globals from it.
110      old_mod = sys.modules[modname]
111      # Load the DLL.
112      loader = importlib.machinery.ExtensionFileLoader(modname, found)
113      spec = importlib.machinery.ModuleSpec(name=modname, loader=loader, origin=found)
114      mod = importlib.util.module_from_spec(spec)
115      spec.loader.exec_module(mod)
116  
117      # Check the sys.modules[] behaviour we describe above is true...
118      assert sys.modules[modname] is mod
119      # as above - re-reset to the *old* module object then update globs.
120      sys.modules[modname] = old_mod
121      globs.update(mod.__dict__)
122  
123  
124  __import_pywin32_system_module__("pywintypes", globals())
