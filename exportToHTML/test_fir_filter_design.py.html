43       def check_response(self, h, expected_response, tol=.05):
44           N = len(h)
45           alpha = 0.5 * (N-1)
46           m = np.arange(0,N) - alpha   # time indices of taps
47           for freq, expected in expected_response:
48               actual = abs(np.sum(h*np.exp(-1.j*np.pi*m*freq)))
49               mse = abs(actual-expected)**2
50               assert mse < tol, f'response not as expected, mse={mse:g} > {tol:g}'

84       def mse(self, h, bands):
85           """Compute mean squared error versus ideal response across frequency 
86           band. 
87             h -- coefficients 
88             bands -- list of (left, right) tuples relative to 1==Nyquist of 
89               passbands 
90           """
91           w, H = freqz(h, worN=1024)
92           f = w/np.pi
93           passIndicator = np.zeros(len(w), bool)
94           for left, right in bands:
95               passIndicator |= (f >= left) & (f < right)
96           Hideal = np.where(passIndicator, 1, 0)
97           mse = np.mean(abs(abs(H)-Hideal)**2)
98           return mse

100      def test_scaling(self):
101          """ 
102          For one lowpass, bandpass, and highpass example filter, this test 
103          checks two things: 
104            - the mean squared error over the frequency domain of the unscaled 
105              filter is smaller than the scaled filter (true for rectangular 
106              window) 
107            - the response of the scaled filter is exactly unity at the center 
108              of the first passband 
109          """
110          N = 11
111          cases = [
112              ([.5], True, (0, 1)),
113              ([0.2, .6], False, (.4, 1)),
114              ([.5], False, (1, 1)),
115          ]
116          for cutoff, pass_zero, expected_response in cases:
117              h = firwin(N, cutoff, scale=False, pass_zero=pass_zero, window='ones')
118              hs = firwin(N, cutoff, scale=True, pass_zero=pass_zero, window='ones')
119              if len(cutoff) == 1:
120                  if pass_zero:
121                      cutoff = [0] + cutoff
122                  else:
123                      cutoff = cutoff + [1]
124              msg = 'least squares violation'
125              assert self.mse(h, [cutoff]) < self.mse(hs, [cutoff]), msg
126              self.check_response(hs, [expected_response], 1e-12)

242          """Test that invalid cutoff argument raises ValueError."""
243          # cutoff values must be greater than 0 and less than 1.

287          # `freq` and `gain` have different lengths.
288          with assert_raises(ValueError, match='must be of same length'):

436      def test_hilbert(self):
437          N = 11  # number of taps in the filter
438          a = 0.1  # width of the transition band
439  
440          # design an unity gain hilbert bandpass filter from w to 0.5-w
441          h = remez(11, [a, 0.5-a], [1], type='hilbert')
442  
443          # make sure the filter has correct # of taps
444          assert len(h) == N, "Number of Taps"
445  
446          # make sure it is type III (anti-symmetric tap coefficients)
447          assert_array_almost_equal(h[:(N-1)//2], -h[:-(N-1)//2-1:-1])
448  
449          # Since the requested response is symmetric, all even coefficients
450          # should be zero (or in this case really small)
451          assert (abs(h[1::2]) < 1e-15).all(), "Even Coefficients Equal Zero"
452  
453          # now check the frequency response
454          w, H = freqz(h, 1)
455          f = w/2/np.pi
456          Hmag = abs(H)
457  
458          # should have a zero at 0 and pi (in this case close to zero)
459          assert (Hmag[[0, -1]] < 0.02).all(), "Zero at zero and pi"
460  
461          # check that the pass band is close to unity
462          idx = np.logical_and(f > a, f < 0.5-a)
463          assert (abs(Hmag[idx] - 1) < 0.015).all(), "Pass Band Close To Unity"

508      def test_firls(self):
509          N = 11  # number of taps in the filter
510          a = 0.1  # width of the transition band
511  
512          # design a halfband symmetric low-pass filter
513          h = firls(11, [0, a, 0.5-a, 0.5], [1, 1, 0, 0], fs=1.0)
514  
515          # make sure the filter has correct # of taps
516          assert h.shape[0] == N
517  
518          # make sure it is symmetric
519          midx = (N-1) // 2
520          assert_array_almost_equal(h[:midx], h[:-midx-1:-1])
521  
522          # make sure the center tap is 0.5
523          assert_almost_equal(h[midx], 0.5)
524  
525          # For halfband symmetric, odd coefficients (except the center)
526          # should be zero (really small)
527          hodd = np.hstack((h[1:midx:2], h[-midx+1::2]))
528          assert_array_almost_equal(hodd, np.zeros_like(hodd))
529  
530          # now check the frequency response
531          w, H = freqz(h, 1)
532          f = w/2/np.pi
533          Hmag = np.abs(H)
534  
535          # check that the pass band is close to unity
536          idx = np.logical_and(f > 0, f < a)
537          assert_array_almost_equal(Hmag[idx], np.ones_like(Hmag[idx]), decimal=3)
538  
539          # check that the stop band is close to zero
540          idx = np.logical_and(f > 0.5-a, f < 0.5)
541          assert_array_almost_equal(Hmag[idx], np.zeros_like(Hmag[idx]), decimal=3)

613      def test_homomorphic(self):
614          # check that it can recover frequency responses of arbitrary
615          # linear-phase filters
616  
617          # for some cases we can get the actual filter back
618          h = [1, -1]
619          h_new = minimum_phase(np.convolve(h, h[::-1]))
620          xp_assert_close(h_new, np.asarray(h, dtype=np.float64), rtol=0.05)
621  
622          # but in general we only guarantee we get the magnitude back
623          rng = np.random.RandomState(0)
624          for n in (2, 3, 10, 11, 15, 16, 17, 20, 21, 100, 101):
625              h = rng.randn(n)
626              h_linear = np.convolve(h, h[::-1])
627              h_new = minimum_phase(h_linear)
628              xp_assert_close(np.abs(fft(h_new)), np.abs(fft(h)), rtol=1e-4)
629              h_new = minimum_phase(h_linear, half=False)
630              assert len(h_linear) == len(h_new)
631              xp_assert_close(np.abs(fft(h_new)), np.abs(fft(h_linear)), rtol=1e-4)
