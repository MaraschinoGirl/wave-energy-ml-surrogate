12   @keras_export("keras.datasets.reuters.load_data")
13   def load_data(
14       path="reuters.npz",
15       num_words=None,
16       skip_top=0,
17       maxlen=None,
18       test_split=0.2,
19       seed=113,
20       start_char=1,
21       oov_char=2,
22       index_from=3,
23   ):
24       """Loads the Reuters newswire classification dataset. 
25    
26       This is a dataset of 11,228 newswires from Reuters, labeled over 46 topics. 
27    
28       This was originally generated by parsing and preprocessing the classic 
29       Reuters-21578 dataset, but the preprocessing code is no longer packaged 
30       with Keras. See this 
31       [GitHub discussion](https://github.com/keras-team/keras/issues/12072) 
32       for more info. 
33    
34       Each newswire is encoded as a list of word indexes (integers). 
35       For convenience, words are indexed by overall frequency in the dataset, 
36       so that for instance the integer "3" encodes the 3rd most frequent word in 
37       the data. This allows for quick filtering operations such as: 
38       "only consider the top 10,000 most 
39       common words, but eliminate the top 20 most common words". 
40    
41       As a convention, "0" does not stand for a specific word, but instead is used 
42       to encode any unknown word. 
43    
44       Args: 
45           path: where to cache the data (relative to `~/.keras/dataset`). 
46           num_words: integer or None. Words are 
47               ranked by how often they occur (in the training set) and only 
48               the `num_words` most frequent words are kept. Any less frequent word 
49               will appear as `oov_char` value in the sequence data. If None, 
50               all words are kept. Defaults to `None`. 
51           skip_top: skip the top N most frequently occurring words 
52               (which may not be informative). These words will appear as 
53               `oov_char` value in the dataset. 0 means no words are 
54               skipped. Defaults to `0`. 
55           maxlen: int or None. Maximum sequence length. 
56               Any longer sequence will be truncated. None means no truncation. 
57               Defaults to `None`. 
58           test_split: Float between `0.` and `1.`. Fraction of the dataset to be 
59               used as test data. `0.2` means that 20% of the dataset is used as 
60               test data. Defaults to `0.2`. 
61           seed: int. Seed for reproducible data shuffling. 
62           start_char: int. The start of a sequence will be marked with this 
63               character. 0 is usually the padding character. Defaults to `1`. 
64           oov_char: int. The out-of-vocabulary character. 
65               Words that were cut out because of the `num_words` or 
66               `skip_top` limits will be replaced with this character. 
67           index_from: int. Index actual words with this index and higher. 
68    
69       Returns: 
70           Tuple of Numpy arrays: `(x_train, y_train), (x_test, y_test)`. 
71    
72       **`x_train`, `x_test`**: lists of sequences, which are lists of indexes 
73         (integers). If the num_words argument was specific, the maximum 
74         possible index value is `num_words - 1`. If the `maxlen` argument was 
75         specified, the largest possible sequence length is `maxlen`. 
76    
77       **`y_train`, `y_test`**: lists of integer labels (1 or 0). 
78    
79       **Note**: The 'out of vocabulary' character is only used for 
80       words that were present in the training set but are not included 
81       because they're not making the `num_words` cut here. 
82       Words that were not seen in the training set but are in the test set 
83       have simply been skipped. 
84       """
85       origin_folder = (
86           "https://storage.googleapis.com/tensorflow/tf-keras-datasets/"
87       )
88       path = get_file(
89           fname=path,
90           origin=origin_folder + "reuters.npz",
91           file_hash=(  # noqa: E501
92               "d6586e694ee56d7a4e65172e12b3e987c03096cb01eab99753921ef915959916"
93           ),
94       )
95       with np.load(path, allow_pickle=True) as f:
96           xs, labels = f["x"], f["y"]
97   
98       rng = np.random.RandomState(seed)
99       indices = np.arange(len(xs))
100      rng.shuffle(indices)
101      xs = xs[indices]
102      labels = labels[indices]
103  
104      if start_char is not None:
105          xs = [[start_char] + [w + index_from for w in x] for x in xs]
106      elif index_from:
107          xs = [[w + index_from for w in x] for x in xs]
108  
109      if maxlen:
110          xs, labels = remove_long_seq(maxlen, xs, labels)
111  
112      if not num_words:
113          num_words = max(max(x) for x in xs)
114  
115      # by convention, use 2 as OOV word
116      # reserve 'index_from' (=3 by default) characters:
117      # 0 (padding), 1 (start), 2 (OOV)
118      if oov_char is not None:
119          xs = [
120              [w if skip_top <= w < num_words else oov_char for w in x]
121              for x in xs
122          ]
123      else:
124          xs = [[w for w in x if skip_top <= w < num_words] for x in xs]
125  
126      idx = int(len(xs) * (1 - test_split))
127      x_train, y_train = (
128          np.array(xs[:idx], dtype="object"),
129          np.array(labels[:idx]),
130      )
131      x_test, y_test = np.array(xs[idx:], dtype="object"), np.array(labels[idx:])
132  
133      return (x_train, y_train), (x_test, y_test)

136  @keras_export("keras.datasets.reuters.get_word_index")
137  def get_word_index(path="reuters_word_index.json"):
138      """Retrieves a dict mapping words to their index in the Reuters dataset. 
139   
140      Actual word indices starts from 3, with 3 indices reserved for: 
141      0 (padding), 1 (start), 2 (oov). 
142   
143      E.g. word index of 'the' is 1, but the in the actual training data, the 
144      index of 'the' will be 1 + 3 = 4. Vice versa, to translate word indices in 
145      training data back to words using this mapping, indices need to subtract 3. 
146   
147      Args: 
148          path: where to cache the data (relative to `~/.keras/dataset`). 
149   
150      Returns: 
151          The word index dictionary. Keys are word strings, values are their 
152          index. 
153      """
154      origin_folder = (
155          "https://storage.googleapis.com/tensorflow/tf-keras-datasets/"
156      )
157      path = get_file(
158          path,
159          origin=origin_folder + "reuters_word_index.json",
160          file_hash="4d44cc38712099c9e383dc6e5f11a921",
161      )
162      with open(path) as f:
163          return json.load(f)
