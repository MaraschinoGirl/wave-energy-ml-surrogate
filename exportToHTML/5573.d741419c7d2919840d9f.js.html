43    * Check if a notebook is trusted
44    * @param notebook The notebook to check
45    * @returns true if the notebook is trusted, false otherwise
46    */
47   const isTrusted = (notebook) => {
48       const model = notebook.model;
49       if (!model) {
50           return false;
51       }
52       const cells = Array.from(model.cells);
53       let total = 0;
54       let trusted = 0;
55       for (const currentCell of cells) {
56           if (currentCell.type !== 'code') {
57               continue;
58           }
59           total++;
60           if (currentCell.trusted) {
61               trusted++;
62           }
63       }
64       return trusted === total;
65   };
