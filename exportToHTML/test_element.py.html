17   class TestNamedspacedAttribute:
18       def test_name_may_be_none_or_missing(self):
19           a = NamespacedAttribute("xmlns", None)
20           assert a == "xmlns"
21   
22           a = NamespacedAttribute("xmlns", "")
23           assert a == "xmlns"
24   
25           a = NamespacedAttribute("xmlns")
26           assert a == "xmlns"
27   
28       def test_namespace_may_be_none_or_missing(self):
29           a = NamespacedAttribute(None, "tag")
30           assert a == "tag"
31   
32           a = NamespacedAttribute("", "tag")
33           assert a == "tag"
34   
35       def test_attribute_is_equivalent_to_colon_separated_string(self):
36           a = NamespacedAttribute("a", "b")
37           assert "a:b" == a
38   
39       def test_attributes_are_equivalent_if_prefix_and_name_identical(self):
40           a = NamespacedAttribute("a", "b", "c")
41           b = NamespacedAttribute("a", "b", "c")
42           assert a == b
43   
44           # The actual namespace is not considered.
45           c = NamespacedAttribute("a", "b", None)
46           assert a == c
47   
48           # But name and prefix are important.
49           d = NamespacedAttribute("a", "z", "c")
50           assert a != d
51   
52           e = NamespacedAttribute("z", "b", "c")
53           assert a != e

56   class TestAttributeValueWithCharsetSubstitution:
57       """Certain attributes are designed to have the charset of the 
58       final document substituted into their value. 
59       """
60   
61       def test_charset_meta_attribute_value(self):
62           # The value of a CharsetMetaAttributeValue is whatever
63           # encoding the string is in.
64           value = CharsetMetaAttributeValue("euc-jp")
65           assert "euc-jp" == value
66           assert "euc-jp" == value.original_value
67           assert "utf8" == value.substitute_encoding("utf8")
68           assert "ascii" == value.substitute_encoding("ascii")
69   
70           # If the target encoding is a Python internal encoding,
71           # no encoding will be mentioned in the output HTML.
72           assert "" == value.substitute_encoding("palmos")
73   
74       def test_content_meta_attribute_value(self):
75           value = ContentMetaAttributeValue("text/html; charset=euc-jp")
76           assert "text/html; charset=euc-jp" == value
77           assert "text/html; charset=euc-jp" == value.original_value
78           assert "text/html; charset=utf8" == value.substitute_encoding("utf8")
79           assert "text/html; charset=ascii" == value.substitute_encoding("ascii")
80   
81           # If the target encoding is a Python internal encoding, the
82           # charset argument will be omitted altogether.
83           assert "text/html" == value.substitute_encoding("palmos")

86   class TestAttributeDicts:
87       def test_xml_attribute_value_handling(self):
88           # Verify that attribute values are processed according to the
89           # XML spec's rules.
90           d = XMLAttributeDict()
91           d["v"] = 100
92           assert d["v"] == "100"
93           d["v"] = 100.123
94           assert d["v"] == "100.123"
95   
96           # This preserves Beautiful Soup's old behavior in the absence of
97           # guidance from the spec.
98           d["v"] = False
99           assert d["v"] is False
100  
101          d["v"] = True
102          assert d["v"] is True
103  
104          d["v"] = None
105          assert d["v"] == ""
106  
107      def test_html_attribute_value_handling(self):
108          # Verify that attribute values are processed according to the
109          # HTML spec's rules.
110          d = HTMLAttributeDict()
111          d["v"] = 100
112          assert d["v"] == "100"
113          d["v"] = 100.123
114          assert d["v"] == "100.123"
115  
116          d["v"] = False
117          assert "v" not in d
118  
119          d["v"] = None
120          assert "v" not in d
121  
122          d["v"] = True
123          assert d["v"] == "v"
124  
125          attribute = NamespacedAttribute("prefix", "name", "namespace")
126          d[attribute] = True
127          assert d[attribute] == "name"
