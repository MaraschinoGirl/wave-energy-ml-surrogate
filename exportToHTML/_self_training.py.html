32   class SelfTrainingClassifier(ClassifierMixin, MetaEstimatorMixin, BaseEstimator):
33       """Self-training classifier. 
34    
35       This :term:`metaestimator` allows a given supervised classifier to function as a 
36       semi-supervised classifier, allowing it to learn from unlabeled data. It 
37       does this by iteratively predicting pseudo-labels for the unlabeled data 
38       and adding them to the training set. 
39    
40       The classifier will continue iterating until either max_iter is reached, or 
41       no pseudo-labels were added to the training set in the previous iteration. 
42    
43       Read more in the :ref:`User Guide <self_training>`. 
44    
45       Parameters 
46       ---------- 
47       estimator : estimator object 
48           An estimator object implementing `fit` and `predict_proba`. 
49           Invoking the `fit` method will fit a clone of the passed estimator, 
50           which will be stored in the `estimator_` attribute. 
51    
52           .. versionadded:: 1.6 
53               `estimator` was added to replace `base_estimator`. 
54    
55       base_estimator : estimator object 
56           An estimator object implementing `fit` and `predict_proba`. 
57           Invoking the `fit` method will fit a clone of the passed estimator, 
58           which will be stored in the `estimator_` attribute. 
59    
60           .. deprecated:: 1.6 
61               `base_estimator` was deprecated in 1.6 and will be removed in 1.8. 
62               Use `estimator` instead. 
63    
64       threshold : float, default=0.75 
65           The decision threshold for use with `criterion='threshold'`. 
66           Should be in [0, 1). When using the `'threshold'` criterion, a 
67           :ref:`well calibrated classifier <calibration>` should be used. 
68    
69       criterion : {'threshold', 'k_best'}, default='threshold' 
70           The selection criterion used to select which labels to add to the 
71           training set. If `'threshold'`, pseudo-labels with prediction 
72           probabilities above `threshold` are added to the dataset. If `'k_best'`, 
73           the `k_best` pseudo-labels with highest prediction probabilities are 
74           added to the dataset. When using the 'threshold' criterion, a 
75           :ref:`well calibrated classifier <calibration>` should be used. 
76    
77       k_best : int, default=10 
78           The amount of samples to add in each iteration. Only used when 
79           `criterion='k_best'`. 
80    
81       max_iter : int or None, default=10 
82           Maximum number of iterations allowed. Should be greater than or equal 
83           to 0. If it is `None`, the classifier will continue to predict labels 
84           until no new pseudo-labels are added, or all unlabeled samples have 
85           been labeled. 
86    
87       verbose : bool, default=False 
88           Enable verbose output. 
89    
90       Attributes 
91       ---------- 
92       estimator_ : estimator object 
93           The fitted estimator. 
94    
95       classes_ : ndarray or list of ndarray of shape (n_classes,) 
96           Class labels for each output. (Taken from the trained 
97           `estimator_`). 
98    
99       transduction_ : ndarray of shape (n_samples,) 
100          The labels used for the final fit of the classifier, including 
101          pseudo-labels added during fit. 
102   
103      labeled_iter_ : ndarray of shape (n_samples,) 
104          The iteration in which each sample was labeled. When a sample has 
105          iteration 0, the sample was already labeled in the original dataset. 
106          When a sample has iteration -1, the sample was not labeled in any 
107          iteration. 
108   
109      n_features_in_ : int 
110          Number of features seen during :term:`fit`. 
111   
112          .. versionadded:: 0.24 
113   
114      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
115          Names of features seen during :term:`fit`. Defined only when `X` 
116          has feature names that are all strings. 
117   
118          .. versionadded:: 1.0 
119   
120      n_iter_ : int 
121          The number of rounds of self-training, that is the number of times the 
122          base estimator is fitted on relabeled variants of the training set. 
123   
124      termination_condition_ : {'max_iter', 'no_change', 'all_labeled'} 
125          The reason that fitting was stopped. 
126   
127          - `'max_iter'`: `n_iter_` reached `max_iter`. 
128          - `'no_change'`: no new labels were predicted. 
129          - `'all_labeled'`: all unlabeled samples were labeled before `max_iter` 
130            was reached. 
131   
132      See Also 
133      -------- 
134      LabelPropagation : Label propagation classifier. 
135      LabelSpreading : Label spreading model for semi-supervised learning. 
136   
137      References 
138      ---------- 
139      :doi:`David Yarowsky. 1995. Unsupervised word sense disambiguation rivaling 
140      supervised methods. In Proceedings of the 33rd annual meeting on 
141      Association for Computational Linguistics (ACL '95). Association for 
142      Computational Linguistics, Stroudsburg, PA, USA, 189-196. 
143      <10.3115/981658.981684>` 
144   
145      Examples 
146      -------- 
147      >>> import numpy as np 
148      >>> from sklearn import datasets 
149      >>> from sklearn.semi_supervised import SelfTrainingClassifier 
150      >>> from sklearn.svm import SVC 
151      >>> rng = np.random.RandomState(42) 
152      >>> iris = datasets.load_iris() 
153      >>> random_unlabeled_points = rng.rand(iris.target.shape[0]) < 0.3 
154      >>> iris.target[random_unlabeled_points] = -1 
155      >>> svc = SVC(probability=True, gamma="auto") 
156      >>> self_training_model = SelfTrainingClassifier(svc) 
157      >>> self_training_model.fit(iris.data, iris.target) 
158      SelfTrainingClassifier(...) 
159      """

198      def _get_estimator(self):
199          """Get the estimator. 
200   
201          Returns 
202          ------- 
203          estimator_ : estimator object 
204              The cloned estimator object. 
205          """
206          # TODO(1.8): remove and only keep clone(self.estimator)
207          if self.estimator is None and self.base_estimator != "deprecated":
208              estimator_ = clone(self.base_estimator)
209  
210              warn(
211                  (
212                      "`base_estimator` has been deprecated in 1.6 and will be removed"
213                      " in 1.8. Please use `estimator` instead."
214                  ),
215                  FutureWarning,
216              )
217          # TODO(1.8) remove
218          elif self.estimator is None and self.base_estimator == "deprecated":
219              raise ValueError(
220                  "You must pass an estimator to SelfTrainingClassifier."
221                  " Use `estimator`."
222              )
223          elif self.estimator is not None and self.base_estimator != "deprecated":
224              raise ValueError(
225                  "You must pass only one estimator to SelfTrainingClassifier."
226                  " Use `estimator`."
227              )
228          else:
229              estimator_ = clone(self.estimator)
230          return estimator_

232      @_fit_context(
233          # SelfTrainingClassifier.estimator is not validated yet
234          prefer_skip_nested_validation=False
235      )
236      def fit(self, X, y, **params):
237          """ 
238          Fit self-training classifier using `X`, `y` as training data. 
239   
240          Parameters 
241          ---------- 
242          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
243              Array representing the data. 
244   
245          y : {array-like, sparse matrix} of shape (n_samples,) 
246              Array representing the labels. Unlabeled samples should have the 
247              label -1. 
248   
249          **params : dict 
250              Parameters to pass to the underlying estimators. 
251   
252              .. versionadded:: 1.6 
253                  Only available if `enable_metadata_routing=True`, 
254                  which can be set by using 
255                  ``sklearn.set_config(enable_metadata_routing=True)``. 
256                  See :ref:`Metadata Routing User Guide <metadata_routing>` for 
257                  more details. 
258   
259          Returns 
260          ------- 
261          self : object 
262              Fitted estimator. 
263          """
264          _raise_for_params(params, self, "fit")
265  
266          self.estimator_ = self._get_estimator()
267  
268          # we need row slicing support for sparse matrices, but costly finiteness check
269          # can be delegated to the base estimator.
270          X, y = validate_data(
271              self,
272              X,
273              y,
274              accept_sparse=["csr", "csc", "lil", "dok"],
275              ensure_all_finite=False,
276          )
277  
278          if y.dtype.kind in ["U", "S"]:
279              raise ValueError(
280                  "y has dtype string. If you wish to predict on "
281                  "string targets, use dtype object, and use -1"
282                  " as the label for unlabeled samples."
283              )
284  
285          has_label = y != -1
286  
287          if np.all(has_label):
288              warnings.warn("y contains no unlabeled samples", UserWarning)
289  
290          if self.criterion == "k_best" and (
291              self.k_best > X.shape[0] - np.sum(has_label)
292          ):
293              warnings.warn(
294                  (
295                      "k_best is larger than the amount of unlabeled "
296                      "samples. All unlabeled samples will be labeled in "
297                      "the first iteration"
298                  ),
299                  UserWarning,
300              )
301  
302          if _routing_enabled():
303              routed_params = process_routing(self, "fit", **params)
304          else:
305              routed_params = Bunch(estimator=Bunch(fit={}))
306  
307          self.transduction_ = np.copy(y)
308          self.labeled_iter_ = np.full_like(y, -1)
309          self.labeled_iter_[has_label] = 0
310  
311          self.n_iter_ = 0
312  
313          while not np.all(has_label) and (
314              self.max_iter is None or self.n_iter_ < self.max_iter
315          ):
316              self.n_iter_ += 1
317              self.estimator_.fit(
318                  X[safe_mask(X, has_label)],
319                  self.transduction_[has_label],
320                  **routed_params.estimator.fit,
321              )
322  
323              # Predict on the unlabeled samples
324              prob = self.estimator_.predict_proba(X[safe_mask(X, ~has_label)])
325              pred = self.estimator_.classes_[np.argmax(prob, axis=1)]
326              max_proba = np.max(prob, axis=1)
327  
328              # Select new labeled samples
329              if self.criterion == "threshold":
330                  selected = max_proba > self.threshold
331              else:
332                  n_to_select = min(self.k_best, max_proba.shape[0])
333                  if n_to_select == max_proba.shape[0]:
334                      selected = np.ones_like(max_proba, dtype=bool)
335                  else:
336                      # NB these are indices, not a mask
337                      selected = np.argpartition(-max_proba, n_to_select)[:n_to_select]
338  
339              # Map selected indices into original array
340              selected_full = np.nonzero(~has_label)[0][selected]
341  
342              # Add newly labeled confident predictions to the dataset
343              self.transduction_[selected_full] = pred[selected]
344              has_label[selected_full] = True
345              self.labeled_iter_[selected_full] = self.n_iter_
346  
347              if selected_full.shape[0] == 0:
348                  # no changed labels
349                  self.termination_condition_ = "no_change"
350                  break
351  
352              if self.verbose:
353                  print(
354                      f"End of iteration {self.n_iter_},"
355                      f" added {selected_full.shape[0]} new labels."
356                  )
357  
358          if self.n_iter_ == self.max_iter:
359              self.termination_condition_ = "max_iter"
360          if np.all(has_label):
361              self.termination_condition_ = "all_labeled"
362  
363          self.estimator_.fit(
364              X[safe_mask(X, has_label)],
365              self.transduction_[has_label],
366              **routed_params.estimator.fit,
367          )
368          self.classes_ = self.estimator_.classes_
369          return self

371      @available_if(_estimator_has("predict"))
372      def predict(self, X, **params):
373          """Predict the classes of `X`. 
374   
375          Parameters 
376          ---------- 
377          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
378              Array representing the data. 
379   
380          **params : dict of str -> object 
381              Parameters to pass to the underlying estimator's ``predict`` method. 
382   
383              .. versionadded:: 1.6 
384                  Only available if `enable_metadata_routing=True`, 
385                  which can be set by using 
386                  ``sklearn.set_config(enable_metadata_routing=True)``. 
387                  See :ref:`Metadata Routing User Guide <metadata_routing>` for 
388                  more details. 
389   
390          Returns 
391          ------- 
392          y : ndarray of shape (n_samples,) 
393              Array with predicted labels. 
394          """
395          check_is_fitted(self)
396          _raise_for_params(params, self, "predict")
397  
398          if _routing_enabled():
399              # metadata routing is enabled.
400              routed_params = process_routing(self, "predict", **params)
401          else:
402              routed_params = Bunch(estimator=Bunch(predict={}))
403  
404          X = validate_data(
405              self,
406              X,
407              accept_sparse=True,
408              ensure_all_finite=False,
409              reset=False,
410          )
411          return self.estimator_.predict(X, **routed_params.estimator.predict)

591      def get_metadata_routing(self):
592          """Get metadata routing of this object. 
593   
594          Please check :ref:`User Guide <metadata_routing>` on how the routing 
595          mechanism works. 
596   
597          .. versionadded:: 1.6 
598   
599          Returns 
600          ------- 
601          routing : MetadataRouter 
602              A :class:`~sklearn.utils.metadata_routing.MetadataRouter` encapsulating 
603              routing information. 
604          """
605          router = MetadataRouter(owner=self.__class__.__name__)
606          router.add(
607              estimator=self.estimator,
608              method_mapping=(
609                  MethodMapping()
610                  .add(callee="fit", caller="fit")
611                  .add(callee="score", caller="fit")
612                  .add(callee="predict", caller="predict")
613                  .add(callee="predict_proba", caller="predict_proba")
614                  .add(callee="decision_function", caller="decision_function")
615                  .add(callee="predict_log_proba", caller="predict_log_proba")
616                  .add(callee="score", caller="score")
617              ),
618          )
619          return router
