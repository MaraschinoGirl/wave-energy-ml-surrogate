set_file_audit.py 
1    import os
2    
3    import ntsecuritycon
4    import win32api
5    import win32con
6    import win32security
7    from win32security import (
8        ACL_REVISION_DS,
9        CONTAINER_INHERIT_ACE,
10       DACL_SECURITY_INFORMATION,
11       GROUP_SECURITY_INFORMATION,
12       OBJECT_INHERIT_ACE,
13       OWNER_SECURITY_INFORMATION,
14       PROTECTED_DACL_SECURITY_INFORMATION,
15       SACL_SECURITY_INFORMATION,
16       SE_FILE_OBJECT,
17   )
18   
19   ## SE_SECURITY_NAME needed to access SACL, SE_RESTORE_NAME needed to change owner to someone other than yourself
20   new_privs = (
21       (
22           win32security.LookupPrivilegeValue("", ntsecuritycon.SE_SECURITY_NAME),
23           win32con.SE_PRIVILEGE_ENABLED,
24       ),
25       (
26           win32security.LookupPrivilegeValue("", ntsecuritycon.SE_RESTORE_NAME),
27           win32con.SE_PRIVILEGE_ENABLED,
28       ),
29   )
30   ph = win32api.GetCurrentProcess()
31   th = win32security.OpenProcessToken(
32       ph, win32security.TOKEN_ALL_ACCESS | win32con.TOKEN_ADJUST_PRIVILEGES
33   )
34   modified_privs = win32security.AdjustTokenPrivileges(th, 0, new_privs)
35   
36   ## look up a few sids that should be available on most systems
37   my_sid = win32security.GetTokenInformation(th, ntsecuritycon.TokenUser)[0]
38   pwr_sid = win32security.LookupAccountName("", "Power Users")[0]
39   admin_sid = win32security.LookupAccountName("", "Administrators")[0]
40   everyone_sid = win32security.LookupAccountName("", "EveryOne")[0]
41   
42   ## create a dir and set security so Everyone has read permissions, and all files and subdirs inherit its ACLs
43   temp_dir = win32api.GetTempPath()
44   dir_name = win32api.GetTempFileName(temp_dir, "sfa")[0]
45   os.remove(dir_name)
46   os.mkdir(dir_name)
47   dir_dacl = win32security.ACL()
48   dir_dacl.AddAccessAllowedAceEx(
49       ACL_REVISION_DS,
50       CONTAINER_INHERIT_ACE | OBJECT_INHERIT_ACE,
51       win32con.GENERIC_READ,
52       everyone_sid,
53   )
54   ## make sure current user has permissions on dir
55   dir_dacl.AddAccessAllowedAceEx(
56       ACL_REVISION_DS,
57       CONTAINER_INHERIT_ACE | OBJECT_INHERIT_ACE,
58       win32con.GENERIC_ALL,
59       my_sid,
60   )
61   ## keep dir from inheriting any permissions so it only has ACEs explicitly set here
62   win32security.SetNamedSecurityInfo(
63       dir_name,
64       SE_FILE_OBJECT,
65       OWNER_SECURITY_INFORMATION
66       | GROUP_SECURITY_INFORMATION
67       | DACL_SECURITY_INFORMATION
68       | PROTECTED_DACL_SECURITY_INFORMATION,
69       pwr_sid,
70       pwr_sid,
71       dir_dacl,
72       None,
73   )
74   
75   ## Create a file in the dir and add some specific permissions to it
76   fname = win32api.GetTempFileName(dir_name, "sfa")[0]
77   print(fname)
78   file_sd = win32security.GetNamedSecurityInfo(
79       fname, SE_FILE_OBJECT, DACL_SECURITY_INFORMATION | SACL_SECURITY_INFORMATION
80   )
81   file_dacl = file_sd.GetSecurityDescriptorDacl()
82   file_sacl = file_sd.GetSecurityDescriptorSacl()
83   
84   if file_dacl is None:
85       file_dacl = win32security.ACL()
86   if file_sacl is None:
87       file_sacl = win32security.ACL()
88   
89   file_dacl.AddAccessDeniedAce(file_dacl.GetAclRevision(), win32con.DELETE, admin_sid)
90   file_dacl.AddAccessDeniedAce(file_dacl.GetAclRevision(), win32con.DELETE, my_sid)
91   file_dacl.AddAccessAllowedAce(file_dacl.GetAclRevision(), win32con.GENERIC_ALL, pwr_sid)
92   file_sacl.AddAuditAccessAce(
93       file_dacl.GetAclRevision(), win32con.GENERIC_ALL, my_sid, True, True
94   )
95   
96   win32security.SetNamedSecurityInfo(
97       fname,
98       SE_FILE_OBJECT,
99       DACL_SECURITY_INFORMATION | SACL_SECURITY_INFORMATION,
100      None,
101      None,
102      file_dacl,
103      file_sacl,
104  )
105  
106  win32security.AdjustTokenPrivileges(th, 0, modified_privs)
