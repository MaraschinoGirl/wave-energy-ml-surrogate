8    # Note: EnumMeta actually subclasses type directly, not ABCMeta.
9    # This is a temporary workaround to allow multiple creation of enums with builtins
10   # such as str as mixins, which due to the handling of ABCs of builtin types, cause
11   # spurious inconsistent metaclass structure. See #1595.
12   # Structurally: Iterable[T], Reversible[T], Container[T] where T is the enum itself
13   class EnumMeta(ABCMeta):
14       def __iter__(self: Type[_T]) -> Iterator[_T]: ...
15       def __reversed__(self: Type[_T]) -> Iterator[_T]: ...
16       def __contains__(self: Type[_T], member: object) -> bool: ...
17       def __getitem__(self: Type[_T], name: str) -> _T: ...
18       @property
19       def __members__(self: Type[_T]) -> Mapping[str, _T]: ...
20       def __len__(self) -> int: ...
21   
22   class Enum(metaclass=EnumMeta):
23       name: str
24       value: Any
25       _name_: str
26       _value_: Any
27       _member_names_: List[str]  # undocumented
28       _member_map_: Dict[str, Enum]  # undocumented
29       _value2member_map_: Dict[int, Enum]  # undocumented
30       if sys.version_info >= (3, 7):
31           _ignore_: Union[str, List[str]]
32       _order_: str
33       __order__: str
34       @classmethod
35       def _missing_(cls, value: object) -> Any: ...
36       @staticmethod
37       def _generate_next_value_(name: str, start: int, count: int, last_values: List[Any]) -> Any: ...
38       def __new__(cls: Type[_T], value: object) -> _T: ...
39       def __repr__(self) -> str: ...
40       def __str__(self) -> str: ...
41       def __dir__(self) -> List[str]: ...
42       def __format__(self, format_spec: str) -> str: ...
43       def __hash__(self) -> Any: ...
44       def __reduce_ex__(self, proto: object) -> Any: ...
