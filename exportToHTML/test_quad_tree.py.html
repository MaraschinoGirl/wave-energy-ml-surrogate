30       # Introduce a point into a quad tree where a similar point already exists.
31       # Test will hang if it doesn't complete.
32       Xs = []
33   
34       # check the case where points are actually different
35       Xs.append(np.array([[1, 2], [3, 4]], dtype=np.float32))
36       # check the case where points are the same on X axis
37       Xs.append(np.array([[1.0, 2.0], [1.0, 3.0]], dtype=np.float32))
38       # check the case where points are arbitrarily close on X axis
39       Xs.append(np.array([[1.00001, 2.0], [1.00002, 3.0]], dtype=np.float32))
40       # check the case where points are the same on Y axis
41       Xs.append(np.array([[1.0, 2.0], [3.0, 2.0]], dtype=np.float32))
42       # check the case where points are arbitrarily close on Y axis
43       Xs.append(np.array([[1.0, 2.00001], [3.0, 2.00002]], dtype=np.float32))
44       # check the case where points are arbitrarily close on both axes
45       Xs.append(np.array([[1.00001, 2.00001], [1.00002, 2.00002]], dtype=np.float32))
46   
47       # check the case where points are arbitrarily close on both axes
48       # close to machine epsilon - x axis
49       Xs.append(np.array([[1, 0.0003817754041], [2, 0.0003817753750]], dtype=np.float32))
50

52       # close to machine epsilon - y axis

63   @pytest.mark.parametrize("n_dimensions", (2, 3))
64   @pytest.mark.parametrize("protocol", (0, 1, 2))
65   def test_quad_tree_pickle(n_dimensions, protocol):
66       rng = check_random_state(0)
67   
68       X = rng.random_sample((10, n_dimensions))
69   
70       tree = _QuadTree(n_dimensions=n_dimensions, verbose=0)
71       tree.build_tree(X)
72   
73       s = pickle.dumps(tree, protocol=protocol)
74       bt2 = pickle.loads(s)
75   
76       for x in X:
77           cell_x_tree = tree.get_cell(x)
78           cell_x_bt2 = bt2.get_cell(x)
79           assert cell_x_tree == cell_x_bt2

82   @pytest.mark.parametrize("n_dimensions", (2, 3))
83   def test_qt_insert_duplicate(n_dimensions):
84       rng = check_random_state(0)
85   
86       X = rng.random_sample((10, n_dimensions))
87       Xd = np.r_[X, X[:5]]
88       tree = _QuadTree(n_dimensions=n_dimensions, verbose=0)
89       tree.build_tree(Xd)
90   
91       cumulative_size = tree.cumulative_size
92       leafs = tree.leafs
93   
94       # Assert that the first 5 are indeed duplicated and that the next
95       # ones are single point leaf
96       for i, x in enumerate(X):
97           cell_id = tree.get_cell(x)
98           assert leafs[cell_id]
99           assert cumulative_size[cell_id] == 1 + (i < 5)

102  def test_summarize():
103      # Simple check for quad tree's summarize
104  
105      angle = 0.9
106      X = np.array(
107          [[-10.0, -10.0], [9.0, 10.0], [10.0, 9.0], [10.0, 10.0]], dtype=np.float32
108      )
109      query_pt = X[0, :]
110      n_dimensions = X.shape[1]
111      offset = n_dimensions + 2
112  
113      qt = _QuadTree(n_dimensions, verbose=0)
114      qt.build_tree(X)
115  
116      idx, summary = qt._py_summarize(query_pt, X, angle)
117  
118      node_dist = summary[n_dimensions]
119      node_size = summary[n_dimensions + 1]
120  
121      # Summary should contain only 1 node with size 3 and distance to
122      # X[1:] barycenter
123      barycenter = X[1:].mean(axis=0)
124      ds2c = ((X[0] - barycenter) ** 2).sum()
125  
126      assert idx == offset
127      assert node_size == 3, "summary size = {}".format(node_size)
128      assert np.isclose(node_dist, ds2c)
129  
130      # Summary should contain all 3 node with size 1 and distance to
131      # each point in X[1:] for ``angle=0``
132      idx, summary = qt._py_summarize(query_pt, X, 0.0)
133      barycenter = X[1:].mean(axis=0)
134      ds2c = ((X[0] - barycenter) ** 2).sum()
135  
136      assert idx == 3 * (offset)
137      for i in range(3):
138          node_dist = summary[i * offset + n_dimensions]
139          node_size = summary[i * offset + n_dimensions + 1]
140  
141          ds2c = ((X[0] - X[i + 1]) ** 2).sum()
142  
143          assert node_size == 1, "summary size = {}".format(node_size)
144          assert np.isclose(node_dist, ds2c)
