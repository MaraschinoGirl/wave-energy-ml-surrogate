19       def test_rejected_input(self):
20           # Python's html.parser will occasionally reject markup,
21           # especially when there is a problem with the initial DOCTYPE
22           # declaration. Different versions of Python sound the alarm in
23           # different ways, but Beautiful Soup consistently raises
24           # errors as ParserRejectedMarkup exceptions.
25           bad_markup = [
26               # https://bugs.chromium.org/p/oss-fuzz/issues/detail?id=28873
27               # https://github.com/guidovranken/python-library-fuzzers/blob/master/corp-html/519e5b4269a01185a0d5e76295251921da2f0700
28               # https://github.com/python/cpython/issues/81928
29               b"\n<![\xff\xfe\xfe\xcd\x00",
30               # https://github.com/guidovranken/python-library-fuzzers/blob/master/corp-html/de32aa55785be29bbc72a1a8e06b00611fb3d9f8
31               # https://github.com/python/cpython/issues/78661
32               #
33               b"<![n\x00",
34               b"<![UNKNOWN[]]>",
35           ]
36           for markup in bad_markup:
37               with pytest.raises(ParserRejectedMarkup):
38                   self.soup(markup)

48       def test_builder_is_pickled(self):
49           """Unlike most tree builders, HTMLParserTreeBuilder and will 
50           be restored after pickling. 
51           """
52           tree = self.soup("<a><b>foo</a>")
53           dumped = pickle.dumps(tree, 2)
54           loaded = pickle.loads(dumped)
55           assert isinstance(loaded.builder, type(tree.builder))

66       def test_tracking_line_numbers(self):
67           # The html.parser TreeBuilder keeps track of line number and
68           # position of each element.
69           markup = "\n   <p>\n\n<sourceline>\n<b>text</b></sourceline><sourcepos></p>"
70           soup = self.soup(markup)
71           assert 2 == soup.p.sourceline
72           assert 3 == soup.p.sourcepos
73           assert "sourceline" == soup.p.find("sourceline").name
74   
75           # You can deactivate this behavior.
76           soup = self.soup(markup, store_line_numbers=False)
77           assert None is soup.p.sourceline
78           assert None is soup.p.sourcepos

80       def test_on_duplicate_attribute(self):
81           # The html.parser tree builder has a variety of ways of
82           # handling a tag that contains the same attribute multiple times.
83   
84           markup = '<a class="cls" href="url1" href="url2" href="url3" id="id">'
85   
86           # If you don't provide any particular value for
87           # on_duplicate_attribute, later values replace earlier values.
88           soup = self.soup(markup)
89           assert "url3" == soup.a["href"]
90           assert ["cls"] == soup.a["class"]
91           assert "id" == soup.a["id"]
92   
93           # You can also get this behavior explicitly.
94           def assert_attribute(
95               on_duplicate_attribute: _DuplicateAttributeHandler, expected: Any
96           ) -> None:
97               soup = self.soup(markup, on_duplicate_attribute=on_duplicate_attribute)
98               assert soup.a is not None
99               assert expected == soup.a["href"]
100  
101              # Verify that non-duplicate attributes are treated normally.
102              assert ["cls"] == soup.a["class"]
103              assert "id" == soup.a["id"]
104  
105          assert_attribute(None, "url3")
106          assert_attribute(BeautifulSoupHTMLParser.REPLACE, "url3")
107  
108          # You can ignore subsequent values in favor of the first.
109          assert_attribute(BeautifulSoupHTMLParser.IGNORE, "url1")
110  
111          # And you can pass in a callable that does whatever you want.
112          def accumulate(attrs, key, value):
113              if not isinstance(attrs[key], list):
114                  attrs[key] = [attrs[key]]
115              attrs[key].append(value)
116  
117          assert_attribute(accumulate, ["url1", "url2", "url3"])

119      def test_html5_attributes(self):
120          # The html.parser TreeBuilder can convert any entity named in
121          # the HTML5 spec to a sequence of Unicode characters, and
122          # convert those Unicode characters to a (potentially
123          # different) named entity on the way out.
124          for input_element, output_unicode, output_element in (
125              ("&RightArrowLeftArrow;", "\u21c4", b"&rlarr;"),
126              ("&models;", "\u22a7", b"&models;"),
127              ("&Nfr;", "\U0001d511", b"&Nfr;"),
128              ("&ngeqq;", "\u2267\u0338", b"&ngeqq;"),
129              ("&not;", "\xac", b"&not;"),
130              ("&Not;", "\u2aec", b"&Not;"),
131              ("&quot;", '"', b'"'),
132              ("&there4;", "\u2234", b"&there4;"),
133              ("&Therefore;", "\u2234", b"&there4;"),
134              ("&therefore;", "\u2234", b"&there4;"),
135              ("&fjlig;", "fj", b"fj"),
136              ("&sqcup;", "\u2294", b"&sqcup;"),
137              ("&sqcups;", "\u2294\ufe00", b"&sqcups;"),
138              ("&apos;", "'", b"'"),
139              ("&verbar;", "|", b"|"),
140          ):
141              markup = "<div>%s</div>" % input_element
142              div = self.soup(markup).div
143              without_element = div.encode()
144              expect = b"<div>%s</div>" % output_unicode.encode("utf8")
145              assert without_element == expect
146  
147              with_element = div.encode(formatter="html")
148              expect = b"<div>%s</div>" % output_element
149              assert with_element == expect

151      def test_invalid_html_entity(self):
152          # The html.parser treebuilder can't distinguish between an invalid
153          # HTML entity with a semicolon and an invalid HTML entity with no
154          # semicolon.
155          markup = "<p>a &nosuchentity b</p>"
156          soup = self.soup(markup)
157          assert "<p>a &amp;nosuchentity b</p>" == soup.p.decode()
158  
159          markup = "<p>a &nosuchentity; b</p>"
160          soup = self.soup(markup)
161          assert "<p>a &amp;nosuchentity b</p>" == soup.p.decode()
