11   class WilcoxonDistribution:
12   
13       def __init__(self, n):
14           n = np.asarray(n).astype(int, copy=False)
15           self.n = n
16           self._dists = {ni: _get_wilcoxon_distr(ni) for ni in np.unique(n)}
17   
18       def _cdf1(self, k, n):
19           pmfs = self._dists[n]
20           return pmfs[:k + 1].sum()
21   
22       def _cdf(self, k, n):
23           return np.vectorize(self._cdf1, otypes=[float])(k, n)
24   
25       def _sf1(self, k, n):
26           pmfs = self._dists[n]
27           return pmfs[k:].sum()
28   
29       def _sf(self, k, n):
30           return np.vectorize(self._sf1, otypes=[float])(k, n)
31   
32       def mean(self):
33           return self.n * (self.n + 1) / 4
34   
35       def _prep(self, k):
36           k = np.asarray(k).astype(int, copy=False)
37           mn = self.mean()
38           out = np.empty(k.shape, dtype=np.float64)
39           return k, mn, out
40   
41       def cdf(self, k):
42           k, mn, out = self._prep(k)
43           return _lazywhere(k <= mn, (k, self.n), self._cdf,
44                             f2=lambda k, n: 1 - self._sf(k+1, n))[()]
45   
46       def sf(self, k):
47           k, mn, out = self._prep(k)
48           return _lazywhere(k <= mn, (k, self.n), self._sf,
49                             f2=lambda k, n: 1 - self._cdf(k-1, n))[()]

52   def _wilcoxon_iv(x, y, zero_method, correction, alternative, method, axis):
53   
54       axis = np.asarray(axis)[()]
55       message = "`axis` must be an integer."
56       if not np.issubdtype(axis.dtype, np.integer) or axis.ndim != 0:
57           raise ValueError(message)
58   
59       message = '`axis` must be compatible with the shape(s) of `x` (and `y`)'
60       try:
61           if y is None:
62               x = np.asarray(x)
63               d = x
64           else:
65               x, y = _broadcast_arrays((x, y), axis=axis)
66               d = x - y
67           d = np.moveaxis(d, axis, -1)
68       except np.AxisError as e:
69           raise ValueError(message) from e
70   
71       message = "`x` and `y` must have the same length along `axis`."
72       if y is not None and x.shape[axis] != y.shape[axis]:
73           raise ValueError(message)
74   
75       message = "`x` (and `y`, if provided) must be an array of real numbers."
76       if np.issubdtype(d.dtype, np.integer):
77           d = d.astype(np.float64)
78       if not np.issubdtype(d.dtype, np.floating):
79           raise ValueError(message)
80   
81       zero_method = str(zero_method).lower()
82       zero_methods = {"wilcox", "pratt", "zsplit"}
83       message = f"`zero_method` must be one of {zero_methods}."
84       if zero_method not in zero_methods:
85           raise ValueError(message)
86   
87       corrections = {True, False}
88       message = f"`correction` must be one of {corrections}."
89       if correction not in corrections:
90           raise ValueError(message)
91   
92       alternative = str(alternative).lower()
93       alternatives = {"two-sided", "less", "greater"}
94       message = f"`alternative` must be one of {alternatives}."
95       if alternative not in alternatives:
96           raise ValueError(message)
97   
98       if not isinstance(method, stats.PermutationMethod):
99           methods = {"auto", "asymptotic", "exact"}
100          message = (f"`method` must be one of {methods} or "
101                     "an instance of `stats.PermutationMethod`.")
102          if method not in methods:
103              raise ValueError(message)
104      output_z = True if method == 'asymptotic' else False
105  
106      # For small samples, we decide later whether to perform an exact test or a
107      # permutation test. The reason is that the presence of ties is not
108      # known at the input validation stage.
109      n_zero = np.sum(d == 0)
110      if method == "auto" and d.shape[-1] > 50:
111          method = "asymptotic"
112  
113      return d, zero_method, correction, alternative, method, axis, output_z, n_zero

116  def _wilcoxon_statistic(d, method, zero_method='wilcox'):
117  
118      i_zeros = (d == 0)
119  
120      if zero_method == 'wilcox':
121          # Wilcoxon's method for treating zeros was to remove them from
122          # the calculation. We do this by replacing 0s with NaNs, which
123          # are ignored anyway.
124          if not d.flags['WRITEABLE']:
125              d = d.copy()
126          d[i_zeros] = np.nan
127  
128      i_nan = np.isnan(d)
129      n_nan = np.sum(i_nan, axis=-1)
130      count = d.shape[-1] - n_nan
131  
132      r, t = _rankdata(abs(d), 'average', return_ties=True)
133  
134      r_plus = np.sum((d > 0) * r, axis=-1)
135      r_minus = np.sum((d < 0) * r, axis=-1)
136  
137      has_ties = (t == 0).any()
138  
139      if zero_method == "zsplit":
140          # The "zero-split" method for treating zeros is to add half their contribution
141          # to r_plus and half to r_minus.
142          # See gh-2263 for the origin of this method.
143          r_zero_2 = np.sum(i_zeros * r, axis=-1) / 2
144          r_plus += r_zero_2
145          r_minus += r_zero_2
146  
147      mn = count * (count + 1.) * 0.25
148      se = count * (count + 1.) * (2. * count + 1.)
149  
150      if zero_method == "pratt":
151          # Pratt's method for treating zeros was just to modify the z-statistic.
152  
153          # normal approximation needs to be adjusted, see Cureton (1967)
154          n_zero = i_zeros.sum(axis=-1)
155          mn -= n_zero * (n_zero + 1.) * 0.25
156          se -= n_zero * (n_zero + 1.) * (2. * n_zero + 1.)
157  
158          # zeros are not to be included in tie-correction.
159          # any tie counts corresponding with zeros are in the 0th column
160          t[i_zeros.any(axis=-1), 0] = 0
161  
162      tie_correct = (t**3 - t).sum(axis=-1)
163      se -= tie_correct/2
164      se = np.sqrt(se / 24)
165  
166      # se = 0 means that no non-zero values are left in d. we only need z
167      # if method is asymptotic. however, if method="auto", the switch to
168      # asymptotic might only happen after the statistic is calculated, so z
169      # needs to be computed. in all other cases, avoid division by zero warning
170      # (z is not needed anyways)
171      if method in ["asymptotic", "auto"]:
172          z = (r_plus - mn) / se
173      else:
174          z = np.nan
175  
176      return r_plus, r_minus, se, z, count, has_ties

179  def _correction_sign(z, alternative):
180      if alternative == 'greater':
181          return 1
182      elif alternative == 'less':
183          return -1
184      else:
185          return np.sign(z)

188  def _wilcoxon_nd(x, y=None, zero_method='wilcox', correction=True,
189                   alternative='two-sided', method='auto', axis=0):
190  
191      temp = _wilcoxon_iv(x, y, zero_method, correction, alternative, method, axis)
192      d, zero_method, correction, alternative, method, axis, output_z, n_zero = temp
193  
194      if d.size == 0:
195          NaN = _get_nan(d)
196          res = _morestats.WilcoxonResult(statistic=NaN, pvalue=NaN)
197          if method == 'asymptotic':
198              res.zstatistic = NaN
199          return res
200  
201      r_plus, r_minus, se, z, count, has_ties = _wilcoxon_statistic(
202          d, method, zero_method
203      )
204  
205      # we only know if there are ties after computing the statistic and not
206      # at the input validation stage. if the original method was auto and
207      # the decision was to use an exact test, we override this to
208      # a permutation test now (since method='exact' is not exact in the
209      # presence of ties)
210      if method == "auto":
211          if not (has_ties or n_zero > 0):
212              method = "exact"
213          elif d.shape[-1] <= 13:
214              # the possible outcomes to be simulated by the permutation test
215              # are 2**n, where n is the sample size.
216              # if n <= 13, the p-value is deterministic since 2**13 is less
217              # than 9999, the default number of n_resamples
218              method = stats.PermutationMethod()
219          else:
220              # if there are ties and the sample size is too large to
221              # run a deterministic permutation test, fall back to asymptotic
222              method = "asymptotic"
223  
224      if method == 'asymptotic':
225          if correction:
226              sign = _correction_sign(z, alternative)
227              z -= sign * 0.5 / se
228          p = _get_pvalue(z, _SimpleNormal(), alternative, xp=np)
229      elif method == 'exact':
230          dist = WilcoxonDistribution(count)
231          # The null distribution in `dist` is exact only if there are no ties
232          # or zeros. If there are ties or zeros, the statistic can be non-
233          # integral, but the null distribution is only defined for integral
234          # values of the statistic. Therefore, we're conservative: round
235          # non-integral statistic up before computing CDF and down before
236          # computing SF. This preserves symmetry w.r.t. alternatives and
237          # order of the input arguments. See gh-19872.
238          if alternative == 'less':
239              p = dist.cdf(np.ceil(r_plus))
240          elif alternative == 'greater':
241              p = dist.sf(np.floor(r_plus))
242          else:
243              p = 2 * np.minimum(dist.sf(np.floor(r_plus)),
244                                 dist.cdf(np.ceil(r_plus)))
245              p = np.clip(p, 0, 1)
246      else:  # `PermutationMethod` instance (already validated)
247          p = stats.permutation_test(
248              (d,), lambda d: _wilcoxon_statistic(d, method, zero_method)[0],
249              permutation_type='samples', **method._asdict(),
250              alternative=alternative, axis=-1).pvalue
251  
252      # for backward compatibility...
253      statistic = np.minimum(r_plus, r_minus) if alternative=='two-sided' else r_plus
254      z = -np.abs(z) if (alternative == 'two-sided' and method == 'asymptotic') else z
255  
256      res = _morestats.WilcoxonResult(statistic=statistic, pvalue=p[()])
257      if output_z:
258          res.zstatistic = z[()]
259      return res
