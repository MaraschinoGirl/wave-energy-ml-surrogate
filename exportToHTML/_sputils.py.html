25   def upcast(*args):
26       """Returns the nearest supported sparse dtype for the 
27       combination of one or more types. 
28    
29       upcast(t0, t1, ..., tn) -> T  where T is a supported dtype 
30    
31       Examples 
32       -------- 
33       >>> from scipy.sparse._sputils import upcast 
34       >>> upcast('int32') 
35       <class 'numpy.int32'> 
36       >>> upcast('bool') 
37       <class 'numpy.bool'> 
38       >>> upcast('int32','float32') 
39       <class 'numpy.float64'> 
40       >>> upcast('bool',complex,float) 
41       <class 'numpy.complex128'> 
42    
43       """
44   
45       t = _upcast_memo.get(hash(args))
46       if t is not None:
47           return t
48   
49       upcast = np.result_type(*args)
50   
51       for t in supported_dtypes:
52           if np.can_cast(upcast, t):
53               _upcast_memo[hash(args)] = t
54               return t
55   
56       raise TypeError(f'no supported conversion for types: {args!r}')
57   
58   
59   def upcast_char(*args):
60       """Same as `upcast` but taking dtype.char as input (faster)."""
61       t = _upcast_memo.get(args)
62       if t is not None:
63           return t
64       t = upcast(*map(np.dtype, args))
65       _upcast_memo[args] = t
66       return t

69   def upcast_scalar(dtype, scalar):
70       """Determine data type for binary operation between an array of 
71       type `dtype` and a scalar. 
72       """
73       return (np.array([0], dtype=dtype) * scalar).dtype

76   def downcast_intp_index(arr):
77       """ 
78       Down-cast index array to np.intp dtype if it is of a larger dtype. 
79    
80       Raise an error if the array contains a value that is too large for 
81       intp. 
82       """
83       if arr.dtype.itemsize > np.dtype(np.intp).itemsize:
84           if arr.size == 0:
85               return arr.astype(np.intp)
86           maxval = arr.max()
87           minval = arr.min()
88           if maxval > np.iinfo(np.intp).max or minval < np.iinfo(np.intp).min:
89               raise ValueError("Cannot deal with arrays with indices larger "
90                                "than the machine maximum address size "
91                                "(e.g. 64-bit indices on 32-bit machine).")
92           return arr.astype(np.intp)
93       return arr

96   def to_native(A):
97       """ 
98       Ensure that the data type of the NumPy array `A` has native byte order. 
99    
100      `A` must be a NumPy array.  If the data type of `A` does not have native 
101      byte order, a copy of `A` with a native byte order is returned. Otherwise 
102      `A` is returned. 
103      """
104      dt = A.dtype
105      if dt.isnative:
106          # Don't call `asarray()` if A is already native, to avoid unnecessarily
107          # creating a view of the input array.
108          return A
109      return np.asarray(A, dtype=dt.newbyteorder('native'))

112  def getdtype(dtype, a=None, default=None):
113      """Form a supported numpy dtype based on input arguments. 
114   
115      Returns a valid ``numpy.dtype`` from `dtype` if not None, 
116      or else ``a.dtype`` if possible, or else the given `default` 
117      if not None, or else raise a ``TypeError``. 
118   
119      The resulting ``dtype`` must be in ``supported_dtypes``: 
120          bool_, int8, uint8, int16, uint16, int32, uint32, 
121          int64, uint64, longlong, ulonglong, float32, float64, 
122          longdouble, complex64, complex128, clongdouble 
123      """
124      if dtype is None:
125          try:
126              newdtype = a.dtype
127          except AttributeError as e:
128              if default is not None:
129                  newdtype = np.dtype(default)
130              else:
131                  raise TypeError("could not interpret data type") from e
132      else:
133          newdtype = np.dtype(dtype)
134  
135      if newdtype not in supported_dtypes:
136          supported_dtypes_fmt = ", ".join(t.__name__ for t in supported_dtypes)
137          raise ValueError(f"scipy.sparse does not support dtype {newdtype.name}. "
138                           f"The only supported types are: {supported_dtypes_fmt}.")
139      return newdtype

142  def getdata(obj, dtype=None, copy=False) -> np.ndarray:
143      """ 
144      This is a wrapper of `np.array(obj, dtype=dtype, copy=copy)` 
145      that will generate a warning if the result is an object array. 
146      """
147      data = np.array(obj, dtype=dtype, copy=copy)
148      # Defer to getdtype for checking that the dtype is OK.
149      # This is called for the validation only; we don't need the return value.
150      getdtype(data.dtype)
151      return data

154  def safely_cast_index_arrays(A, idx_dtype=np.int32, msg=""):
155      """Safely cast sparse array indices to `idx_dtype`. 
156   
157      Check the shape of `A` to determine if it is safe to cast its index 
158      arrays to dtype `idx_dtype`. If any dimension in shape is larger than 
159      fits in the dtype, casting is unsafe so raise ``ValueError``. 
160      If safe, cast the index arrays to `idx_dtype` and return the result 
161      without changing the input `A`. The caller can assign results to `A` 
162      attributes if desired or use the recast index arrays directly. 
163   
164      Unless downcasting is needed, the original index arrays are returned. 
165      You can test e.g. ``A.indptr is new_indptr`` to see if downcasting occurred. 
166   
167      .. versionadded:: 1.15.0 
168   
169      Parameters 
170      ---------- 
171      A : sparse array or matrix 
172          The array for which index arrays should be downcast. 
173      idx_dtype : dtype 
174          Desired dtype. Should be an integer dtype (default: ``np.int32``). 
175          Most of scipy.sparse uses either int64 or int32. 
176      msg : string, optional 
177          A string to be added to the end of the ValueError message 
178          if the array shape is too big to fit in `idx_dtype`. 
179          The error message is ``f"<index> values too large for {msg}"`` 
180          It should indicate why the downcasting is needed, e.g. "SuperLU", 
181          and defaults to f"dtype {idx_dtype}". 
182   
183      Returns 
184      ------- 
185      idx_arrays : ndarray or tuple of ndarrays 
186          Based on ``A.format``, index arrays are returned after casting to `idx_dtype`. 
187          For CSC/CSR, returns ``(indices, indptr)``. 
188          For COO, returns ``coords``. 
189          For DIA, returns ``offsets``. 
190          For BSR, returns ``(indices, indptr)``. 
191   
192      Raises 
193      ------ 
194      ValueError 
195          If the array has shape that would not fit in the new dtype, or if 
196          the sparse format does not use index arrays. 
197   
198      Examples 
199      -------- 
200      >>> import numpy as np 
201      >>> from scipy import sparse 
202      >>> data = [3] 
203      >>> coords = (np.array([3]), np.array([1]))  # Note: int64 arrays 
204      >>> A = sparse.coo_array((data, coords)) 
205      >>> A.coords[0].dtype 
206      dtype('int64') 
207   
208      >>> # rescast after construction, raising exception if shape too big 
209      >>> coords = sparse.safely_cast_index_arrays(A, np.int32) 
210      >>> A.coords[0] is coords[0]  # False if casting is needed 
211      False 
212      >>> A.coords = coords  # set the index dtype of A 
213      >>> A.coords[0].dtype 
214      dtype('int32') 
215      """
216      if not msg:
217          msg = f"dtype {idx_dtype}"
218      # check for safe downcasting
219      max_value = np.iinfo(idx_dtype).max
220  
221      if A.format in ("csc", "csr"):
222          # indptr[-1] is max b/c indptr always sorted
223          if A.indptr[-1] > max_value:
224              raise ValueError(f"indptr values too large for {msg}")
225  
226          # check shape vs dtype
227          if max(*A.shape) > max_value:
228              if (A.indices > max_value).any():
229                  raise ValueError(f"indices values too large for {msg}")
230  
231          indices = A.indices.astype(idx_dtype, copy=False)
232          indptr = A.indptr.astype(idx_dtype, copy=False)
233          return indices, indptr
234  
235      elif A.format == "coo":
236          if max(*A.shape) > max_value:
237              if any((co > max_value).any() for co in A.coords):
238                  raise ValueError(f"coords values too large for {msg}")
239          return tuple(co.astype(idx_dtype, copy=False) for co in A.coords)
240  
241      elif A.format == "dia":
242          if max(*A.shape) > max_value:
243              if (A.offsets > max_value).any():
244                  raise ValueError(f"offsets values too large for {msg}")
245          offsets = A.offsets.astype(idx_dtype, copy=False)
246          return offsets
247  
248      elif A.format == 'bsr':
249          R, C = A.blocksize
250          if A.indptr[-1] * R > max_value:
251              raise ValueError("indptr values too large for {msg}")
252          if max(*A.shape) > max_value:
253              if (A.indices * C > max_value).any():
254                  raise ValueError(f"indices values too large for {msg}")
255          indices = A.indices.astype(idx_dtype, copy=False)
256          indptr = A.indptr.astype(idx_dtype, copy=False)
257          return indices, indptr
258  
259      else:
260          raise TypeError(f'Format {A.format} is not associated with index arrays. '
261                          'DOK and LIL have dict and list, not array.')

264  def get_index_dtype(arrays=(), maxval=None, check_contents=False):
265      """ 
266      Based on input (integer) arrays `a`, determine a suitable index data 
267      type that can hold the data in the arrays. 
268   
269      Parameters 
270      ---------- 
271      arrays : tuple of array_like 
272          Input arrays whose types/contents to check 
273      maxval : float, optional 
274          Maximum value needed 
275      check_contents : bool, optional 
276          Whether to check the values in the arrays and not just their types. 
277          Default: False (check only the types) 
278   
279      Returns 
280      ------- 
281      dtype : dtype 
282          Suitable index data type (int32 or int64) 
283   
284      Examples 
285      -------- 
286      >>> import numpy as np 
287      >>> from scipy import sparse 
288      >>> # select index dtype based on shape 
289      >>> shape = (3, 3) 
290      >>> idx_dtype = sparse.get_index_dtype(maxval=max(shape)) 
291      >>> data = [1.1, 3.0, 1.5] 
292      >>> indices = np.array([0, 1, 0], dtype=idx_dtype) 
293      >>> indptr = np.array([0, 2, 3, 3], dtype=idx_dtype) 
294      >>> A = sparse.csr_array((data, indices, indptr), shape=shape) 
295      >>> A.indptr.dtype 
296      dtype('int32') 
297   
298      >>> # select based on larger of existing arrays and shape 
299      >>> shape = (3, 3) 
300      >>> idx_dtype = sparse.get_index_dtype(A.indptr, maxval=max(shape)) 
301      >>> idx_dtype 
302      <class 'numpy.int32'> 
303      """
304      # not using intc directly due to misinteractions with pythran
305      if np.intc().itemsize != 4:
306          return np.int64
307  
308      int32min = np.int32(np.iinfo(np.int32).min)
309      int32max = np.int32(np.iinfo(np.int32).max)
310  
311      if maxval is not None:
312          maxval = np.int64(maxval)
313          if maxval > int32max:
314              return np.int64
315  
316      if isinstance(arrays, np.ndarray):
317          arrays = (arrays,)
318  
319      for arr in arrays:
320          arr = np.asarray(arr)
321          if not np.can_cast(arr.dtype, np.int32):
322              if check_contents:
323                  if arr.size == 0:
324                      # a bigger type not needed
325                      continue
326                  elif np.issubdtype(arr.dtype, np.integer):
327                      maxval = arr.max()
328                      minval = arr.min()
329                      if minval >= int32min and maxval <= int32max:
330                          # a bigger type not needed
331                          continue
332              return np.int64
333      return np.int32

336  def get_sum_dtype(dtype: np.dtype) -> np.dtype:
337      """Mimic numpy's casting for np.sum"""
338      if dtype.kind == 'u' and np.can_cast(dtype, np.uint):
339          return np.uint
340      if np.can_cast(dtype, np.int_):
341          return np.int_
342      return dtype

346      """Is x either a scalar, an array scalar, or a 0-dim array?"""
347      return np.isscalar(x) or (isdense(x) and x.ndim == 0)

350  def isintlike(x) -> bool:
351      """Is x appropriate as an index into a sparse matrix? Returns True 
352      if it can be cast safely to a machine int. 
353      """
354      # Fast-path check to eliminate non-scalar values. operator.index would
355      # catch this case too, but the exception catching is slow.
356      if np.ndim(x) != 0:
357          return False
358      try:
359          operator.index(x)
360      except (TypeError, ValueError):
361          try:
362              loose_int = bool(int(x) == x)
363          except (TypeError, ValueError):
364              return False
365          if loose_int:
366              msg = "Inexact indices into sparse matrices are not allowed"
367              raise ValueError(msg)
368          return loose_int
369      return True

372  def isshape(x, nonneg=False, *, allow_nd=(2,)) -> bool:
373      """Is x a valid tuple of dimensions? 
374   
375      If nonneg, also checks that the dimensions are non-negative. 
376      Shapes of length in the tuple allow_nd are allowed. 
377      """
378      ndim = len(x)
379      if ndim not in allow_nd:
380          return False
381  
382      for d in x:
383          if not isintlike(d):
384              return False
385          if nonneg and d < 0:
386              return False
387      return True

392              (len(t) == 0 or np.isscalar(t[0]))) or
393              (isinstance(t, np.ndarray) and (t.ndim == 1)))

406  def validateaxis(axis) -> None:
407      if axis is None:
408          return
409      axis_type = type(axis)
410  
411      # In NumPy, you can pass in tuples for 'axis', but they are
412      # not very useful for sparse matrices given their limited
413      # dimensions, so let's make it explicit that they are not
414      # allowed to be passed in
415      if isinstance(axis, tuple):
416          raise TypeError("Tuples are not accepted for the 'axis' parameter. "
417                          "Please pass in one of the following: "
418                          "{-2, -1, 0, 1, None}.")
419  
420      # If not a tuple, check that the provided axis is actually
421      # an integer and raise a TypeError similar to NumPy's
422      if not np.issubdtype(np.dtype(axis_type), np.integer):
423          raise TypeError(f"axis must be an integer, not {axis_type.__name__}")
424  
425      if not (-2 <= axis <= 1):
426          raise ValueError("axis out of range")

429  def check_shape(args, current_shape=None, *, allow_nd=(2,)) -> tuple[int, ...]:
430      """Imitate numpy.matrix handling of shape arguments 
431   
432      Parameters 
433      ---------- 
434      args : array_like 
435          Data structures providing information about the shape of the sparse array. 
436      current_shape : tuple, optional 
437          The current shape of the sparse array or matrix. 
438          If None (default), the current shape will be inferred from args. 
439      allow_nd : tuple of ints, optional default: (2,) 
440          If shape does not have a length in the tuple allow_nd an error is raised. 
441   
442      Returns 
443      ------- 
444      new_shape: tuple 
445          The new shape after validation. 
446      """
447      if len(args) == 0:
448          raise TypeError("function missing 1 required positional argument: 'shape'")
449      if len(args) == 1:
450          try:
451              shape_iter = iter(args[0])
452          except TypeError:
453              new_shape = (operator.index(args[0]), )
454          else:
455              new_shape = tuple(operator.index(arg) for arg in shape_iter)
456      else:
457          new_shape = tuple(operator.index(arg) for arg in args)
458  
459      if current_shape is None:
460          if len(new_shape) not in allow_nd:
461              raise ValueError(f'shape must have length in {allow_nd}. Got {new_shape=}')
462          if any(d < 0 for d in new_shape):
463              raise ValueError("'shape' elements cannot be negative")
464      else:
465          # Check the current size only if needed
466          current_size = prod(current_shape)
467  
468          # Check for negatives
469          negative_indexes = [i for i, x in enumerate(new_shape) if x < 0]
470          if not negative_indexes:
471              new_size = prod(new_shape)
472              if new_size != current_size:
473                  raise ValueError(f'cannot reshape array of size {current_size}'
474                                   f' into shape {new_shape}')
475          elif len(negative_indexes) == 1:
476              skip = negative_indexes[0]
477              specified = prod(new_shape[:skip] + new_shape[skip+1:])
478              unspecified, remainder = divmod(current_size, specified)
479              if remainder != 0:
480                  err_shape = tuple('newshape' if x < 0 else x for x in new_shape)
481                  raise ValueError(f'cannot reshape array of size {current_size}'
482                                   f' into shape {err_shape}')
483              new_shape = new_shape[:skip] + (unspecified,) + new_shape[skip+1:]
484          else:
485              raise ValueError('can only specify one unknown dimension')
486  
487      if len(new_shape) not in allow_nd:
488          raise ValueError(f'shape must have length in {allow_nd}. Got {new_shape=}')
489  
490      return new_shape

493  def broadcast_shapes(*shapes):
494      """Check if shapes can be broadcast and return resulting shape 
495   
496      This is similar to the NumPy ``broadcast_shapes`` function but 
497      does not check memory consequences of the resulting dense matrix. 
498   
499      Parameters 
500      ---------- 
501      *shapes : tuple of shape tuples 
502          The tuple of shapes to be considered for broadcasting. 
503          Shapes should be tuples of non-negative integers. 
504   
505      Returns 
506      ------- 
507      new_shape : tuple of integers 
508          The shape that results from broadcasting th input shapes. 
509      """
510      if not shapes:
511          return ()
512      shapes = [shp if isinstance(shp, (tuple, list)) else (shp,) for shp in shapes]
513      big_shp = max(shapes, key=len)
514      out = list(big_shp)
515      for shp in shapes:
516          if shp is big_shp:
517              continue
518          for i, x in enumerate(shp, start=-len(shp)):
519              if x != 1 and x != out[i]:
520                  if out[i] != 1:
521                      raise ValueError("shapes cannot be broadcast to a single shape.")
522                  out[i] = x
523      return (*out,)

526  def check_reshape_kwargs(kwargs):
527      """Unpack keyword arguments for reshape function. 
528   
529      This is useful because keyword arguments after star arguments are not 
530      allowed in Python 2, but star keyword arguments are. This function unpacks 
531      'order' and 'copy' from the star keyword arguments (with defaults) and 
532      throws an error for any remaining. 
533      """
534  
535      order = kwargs.pop('order', 'C')
536      copy = kwargs.pop('copy', False)
537      if kwargs:  # Some unused kwargs remain
538          raise TypeError("reshape() got unexpected keywords arguments: "
539                          f"{', '.join(kwargs.keys())}")
540      return order, copy

543  def is_pydata_spmatrix(m) -> bool:
544      """ 
545      Check whether object is pydata/sparse matrix, avoiding importing the module. 
546      """
547      base_cls = getattr(sys.modules.get('sparse'), 'SparseArray', None)
548      return base_cls is not None and isinstance(m, base_cls)

551  def convert_pydata_sparse_to_scipy(
552      arg: Any,
553      target_format: None | Literal["csc", "csr"] = None,
554      accept_fv: Any = None,
555  ) -> Union[Any, "sp.spmatrix"]:
556      """ 
557      Convert a pydata/sparse array to scipy sparse matrix, 
558      pass through anything else. 
559      """
560      if is_pydata_spmatrix(arg):
561          # The `accept_fv` keyword is new in PyData Sparse 0.15.4 (May 2024),
562          # remove the `except` once the minimum supported version is >=0.15.4
563          try:
564              arg = arg.to_scipy_sparse(accept_fv=accept_fv)
565          except TypeError:
566              arg = arg.to_scipy_sparse()
567          if target_format is not None:
568              arg = arg.asformat(target_format)
569          elif arg.format not in ("csc", "csr"):
570              arg = arg.tocsc()
571      return arg

592  def _todata(s) -> np.ndarray:
593      """Access nonzero values, possibly after summing duplicates. 
594   
595      Parameters 
596      ---------- 
597      s : sparse array 
598          Input sparse array. 
599   
600      Returns 
601      ------- 
602      data: ndarray 
603        Nonzero values of the array, with shape (s.nnz,) 
604   
605      """
606      if isinstance(s, sp._data._data_matrix):
607          return s._deduped_data()
608  
609      if isinstance(s, sp.dok_array):
610          return np.fromiter(s.values(), dtype=s.dtype, count=s.nnz)
611  
612      if isinstance(s, sp.lil_array):
613          data = np.empty(s.nnz, dtype=s.dtype)
614          sp._csparsetools.lil_flatten_to_array(s.data, data)
615          return data
616  
617      return s.tocoo()._deduped_data()
