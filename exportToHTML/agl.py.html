5110 def toUnicode(glyph, isZapfDingbats=False):
5111     """Convert glyph names to Unicode, such as ``'longs_t.oldstyle'`` --> ``u'≈øt'`` 
5112  
5113     If ``isZapfDingbats`` is ``True``, the implementation recognizes additional 
5114     glyph names (as required by the AGL specification). 
5115     """
5116     # https://github.com/adobe-type-tools/agl-specification#2-the-mapping
5117     #
5118     # 1. Drop all the characters from the glyph name starting with
5119     #    the first occurrence of a period (U+002E; FULL STOP), if any.
5120     glyph = glyph.split(".", 1)[0]
5121 
5122     # 2. Split the remaining string into a sequence of components,
5123     #    using underscore (U+005F; LOW LINE) as the delimiter.
5124     components = glyph.split("_")
5125 
5126     # 3. Map each component to a character string according to the
5127     #    procedure below, and concatenate those strings; the result
5128     #     is the character string to which the glyph name is mapped.
5129     result = [_glyphComponentToUnicode(c, isZapfDingbats) for c in components]
5130     return "".join(result)
5131

5133 def _glyphComponentToUnicode(component, isZapfDingbats):
5134     # If the font is Zapf Dingbats (PostScript FontName: ZapfDingbats),
5135     # and the component is in the ITC Zapf Dingbats Glyph List, then
5136     # map it to the corresponding character in that list.
5137     dingbat = _zapfDingbatsToUnicode(component) if isZapfDingbats else None
5138     if dingbat:
5139         return dingbat
5140 
5141     # Otherwise, if the component is in AGL, then map it
5142     # to the corresponding character in that list.
5143     uchars = LEGACY_AGL2UV.get(component)
5144     if uchars:
5145         return "".join(map(chr, uchars))
5146 
5147     # Otherwise, if the component is of the form "uni" (U+0075,
5148     # U+006E, and U+0069) followed by a sequence of uppercase
5149     # hexadecimal digits (0‚Äì9 and A‚ÄìF, meaning U+0030 through
5150     # U+0039 and U+0041 through U+0046), if the length of that
5151     # sequence is a multiple of four, and if each group of four
5152     # digits represents a value in the ranges 0000 through D7FF
5153     # or E000 through FFFF, then interpret each as a Unicode scalar
5154     # value and map the component to the string made of those
5155     # scalar values. Note that the range and digit-length
5156     # restrictions mean that the "uni" glyph name prefix can be
5157     # used only with UVs in the Basic Multilingual Plane (BMP).
5158     uni = _uniToUnicode(component)
5159     if uni:
5160         return uni
5161 
5162     # Otherwise, if the component is of the form "u" (U+0075)
5163     # followed by a sequence of four to six uppercase hexadecimal
5164     # digits (0‚Äì9 and A‚ÄìF, meaning U+0030 through U+0039 and
5165     # U+0041 through U+0046), and those digits represents a value
5166     # in the ranges 0000 through D7FF or E000 through 10FFFF, then
