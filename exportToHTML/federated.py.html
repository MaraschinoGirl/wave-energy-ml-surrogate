27   def run_server(port: int, world_size: int, with_ssl: bool) -> None:
28       """Run federated server for test."""
29       if with_ssl:
30           xgboost.federated.run_federated_server(
31               world_size,
32               port,
33               server_key_path=SERVER_KEY,
34               server_cert_path=SERVER_CERT,
35               client_cert_path=CLIENT_CERT,
36           )
37       else:
38           xgboost.federated.run_federated_server(world_size, port)

41   def run_worker(
42       port: int, world_size: int, rank: int, with_ssl: bool, device: str
43   ) -> None:
44       """Run federated client worker for test."""
45       comm_env: CollArgs = {
46           "dmlc_communicator": "federated",
47           "federated_server_address": f"localhost:{port}",
48           "federated_world_size": world_size,
49           "federated_rank": rank,
50       }
51       if with_ssl:
52           comm_env["federated_server_cert_path"] = SERVER_CERT
53           comm_env["federated_client_key_path"] = CLIENT_KEY
54           comm_env["federated_client_cert_path"] = CLIENT_CERT
55   
56       cpu_count = os.cpu_count()
57       assert cpu_count is not None
58       n_threads = cpu_count // world_size
59   
60       # Always call this before using distributed module
61       with xgb.collective.CommunicatorContext(**comm_env):
62           # Load file, file will not be sharded in federated mode.
63           X, y = load_svmlight_file(f"agaricus.txt-{rank}.train")
64           dtrain = xgb.DMatrix(X, y)
65           X, y = load_svmlight_file(f"agaricus.txt-{rank}.test")
66           dtest = xgb.DMatrix(X, y)
67   
68           # Specify parameters via map, definition are same as c++ version
69           param = {
70               "max_depth": 2,
71               "eta": 1,
72               "objective": "binary:logistic",
73               "nthread": n_threads,
74               "tree_method": "hist",
75               "device": device,
76           }
77   
78           # Specify validations set to watch performance
79           watchlist = [(dtest, "eval"), (dtrain, "train")]
80           num_round = 20
81   
82           # Run training, all the features in training API is available.
83           results: TrainingCallback.EvalsLog = {}
84           bst = xgb.train(
85               param,
86               dtrain,
87               num_round,
88               evals=watchlist,
89               early_stopping_rounds=2,
90               evals_result=results,
91           )
92           assert tm.non_increasing(cast(List[float], results["train"]["logloss"]))
93           assert tm.non_increasing(cast(List[float], results["eval"]["logloss"]))
94   
95           # save the model, only ask process 0 to save the model.
96           if xgb.collective.get_rank() == 0:
97               with tempfile.TemporaryDirectory() as tmpdir:
98                   bst.save_model(os.path.join(tmpdir, "model.json"))
99               xgb.collective.communicator_print("Finished training\n")

102  def run_federated(world_size: int, with_ssl: bool, use_gpu: bool) -> None:
103      """Launcher for clients and the server."""
104      port = 9091
105  
106      server = multiprocessing.Process(
107          target=run_server, args=(port, world_size, with_ssl)
108      )
109      server.start()
110      time.sleep(1)
111      if not server.is_alive():
112          raise ValueError("Error starting Federated Learning server")
113  
114      workers = []
115      for rank in range(world_size):
116          device = f"cuda:{rank}" if use_gpu else "cpu"
117          worker = multiprocessing.Process(
118              target=run_worker, args=(port, world_size, rank, with_ssl, device)
119          )
120          workers.append(worker)
121          worker.start()
122      for worker in workers:
123          worker.join()
124      server.terminate()

127  def run_federated_learning(with_ssl: bool, use_gpu: bool, test_path: str) -> None:
128      """Run federated learning tests."""
129      n_workers = 2
130  
131      if with_ssl:
132          command = "openssl req -x509 -newkey rsa:2048 -days 7 -nodes -keyout {part}-key.pem -out {part}-cert.pem -subj /C=US/CN=localhost"  # pylint: disable=line-too-long
133          server_key = command.format(part="server").split()
134          subprocess.check_call(server_key)
135          client_key = command.format(part="client").split()
136          subprocess.check_call(client_key)
137  
138      train_path = os.path.join(tm.data_dir(test_path), "agaricus.txt.train")
139      test_path = os.path.join(tm.data_dir(test_path), "agaricus.txt.test")
140  
141      X_train, y_train = load_svmlight_file(train_path)
142      X_test, y_test = load_svmlight_file(test_path)
143  
144      X0, X1, y0, y1 = train_test_split(X_train, y_train, test_size=0.5)
145      X0_valid, X1_valid, y0_valid, y1_valid = train_test_split(
146          X_test, y_test, test_size=0.5
147      )
148  
149      dump_svmlight_file(X0, y0, "agaricus.txt-0.train")
150      dump_svmlight_file(X0_valid, y0_valid, "agaricus.txt-0.test")
151  
152      dump_svmlight_file(X1, y1, "agaricus.txt-1.train")
153      dump_svmlight_file(X1_valid, y1_valid, "agaricus.txt-1.test")
154  
155      run_federated(world_size=n_workers, with_ssl=with_ssl, use_gpu=use_gpu)
