4    """Button class. 
5     
6    Represents a button in the frontend using a widget.  Allows user to listen for 
7    click events on the button and trigger backend code when the clicks are fired. 
8    """

11   from .domwidget import DOMWidget

15   from .trait_types import Color, InstanceDict
16   
17   from traitlets import Unicode, Bool, CaselessStrEnum, Instance, validate, default

20   @register
21   class ButtonStyle(Style, CoreWidget):
22       """Button style widget."""
23       _model_name = Unicode('ButtonStyleModel').tag(sync=True)
24       button_color = Color(None, allow_none=True, help="Color of the button").tag(sync=True)
25       font_family = Unicode(None, allow_none=True, help="Button text font family.").tag(sync=True)
26       font_size = Unicode(None, allow_none=True, help="Button text font size.").tag(sync=True)
27       font_style = Unicode(None, allow_none=True, help="Button text font style.").tag(sync=True)
28       font_variant = Unicode(None, allow_none=True, help="Button text font variant.").tag(sync=True)
29       font_weight = Unicode(None, allow_none=True, help="Button text font weight.").tag(sync=True)
30       text_color = Unicode(None, allow_none=True, help="Button text color.").tag(sync=True)
31       text_decoration = Unicode(None, allow_none=True, help="Button text decoration.").tag(sync=True)

34   @register
35   class Button(DOMWidget, CoreWidget):
36       """Button widget. 
37    
38       This widget has an `on_click` method that allows you to listen for the 
39       user clicking on the button.  The click event itself is stateless. 
40    
41       Parameters 
42       ---------- 
43       description: str 
44          description displayed on the button 
45       icon: str 
46          font-awesome icon names, without the 'fa-' prefix 
47       disabled: bool 
48          whether user interaction is enabled 
49       """
50       _view_name = Unicode('ButtonView').tag(sync=True)
51       _model_name = Unicode('ButtonModel').tag(sync=True)
52   
53       description = Unicode(help="Button label.").tag(sync=True)
54       disabled = Bool(False, help="Enable or disable user changes.").tag(sync=True)
55       icon = Unicode('', help="Font-awesome icon names, without the 'fa-' prefix.").tag(sync=True)
56   
57       button_style = CaselessStrEnum(
58           values=['primary', 'success', 'info', 'warning', 'danger', ''], default_value='',
59           help="""Use a predefined styling for the button.""").tag(sync=True)
60   
61       style = InstanceDict(ButtonStyle).tag(sync=True, **widget_serialization)
62   
63       def __init__(self, **kwargs):
64           super().__init__(**kwargs)
65           self._click_handlers = CallbackDispatcher()
66           self.on_msg(self._handle_button_msg)
67   
68       @validate('icon')
69       def _validate_icon(self, proposal):
70           """Strip 'fa-' if necessary'"""
71           value = proposal['value']
72           if 'fa-' in value:
73               deprecation("icons names no longer need 'fa-', "
74               "just use the class names themselves (for example, 'gear spin' instead of 'fa-gear fa-spin')",
75               internal=['ipywidgets/widgets/', 'traitlets/traitlets.py', '/contextlib.py'])
76               value = value.replace('fa-', '')
77           return value
78   
79       def on_click(self, callback, remove=False):
80           """Register a callback to execute when the button is clicked. 
81    
82           The callback will be called with one argument, the clicked button 
83           widget instance. 
84    
85           Parameters 
86           ---------- 
87           remove: bool (optional) 
88               Set to true to remove the callback from the list of callbacks. 
89           """
90           self._click_handlers.register_callback(callback, remove=remove)
91   
92       def click(self):
93           """Programmatically trigger a click event. 
94    
95           This will call the callbacks registered to the clicked button 
96           widget instance. 
97           """
98           self._click_handlers(self)
99   
100      def _handle_button_msg(self, _, content, buffers):
101          """Handle a msg from the front-end. 
102   
103          Parameters 
104          ---------- 
105          content: dict 
106              Content of the msg. 
107          """
108          if content.get('event', '') == 'click':
109              self.click()
