32   class TestFileOpen(TestCase):
33   
34       """ 
35           Feature: Opening files with Python-style modes. 
36       """
37   
38       def test_default(self):
39           """ Default semantics in the presence or absence of a file """
40           fname = self.mktemp()
41   
42           # No existing file; error
43           with pytest.raises(FileNotFoundError):
44               with File(fname):
45                   pass
46   
47           # Existing readonly file; open read-only
48           with File(fname, 'w'):
49               pass
50           os.chmod(fname, stat.S_IREAD)
51           try:
52               with File(fname) as f:
53                   self.assertTrue(f)
54                   self.assertEqual(f.mode, 'r')
55           finally:
56               os.chmod(fname, stat.S_IWRITE)
57   
58           # File exists but is not HDF5; raise OSError
59           with open(fname, 'wb') as f:
60               f.write(b'\x00')
61           with self.assertRaises(OSError):
62               File(fname)
63   
64       def test_create(self):
65           """ Mode 'w' opens file in overwrite mode """
66           fname = self.mktemp()
67           fid = File(fname, 'w')
68           self.assertTrue(fid)
69           fid.create_group('foo')
70           fid.close()
71           fid = File(fname, 'w')
72           self.assertNotIn('foo', fid)
73           fid.close()
74   
75       def test_create_exclusive(self):
76           """ Mode 'w-' opens file in exclusive mode """
77           fname = self.mktemp()
78           fid = File(fname, 'w-')
79           self.assertTrue(fid)
80           fid.close()
81           with self.assertRaises(FileExistsError):
82               File(fname, 'w-')
83   
84       def test_append(self):
85           """ Mode 'a' opens file in append/readwrite mode, creating if necessary """
86           fname = self.mktemp()
87           fid = File(fname, 'a')
88           try:
89               self.assertTrue(fid)
90               fid.create_group('foo')
91               assert 'foo' in fid
92           finally:
93               fid.close()
94           fid = File(fname, 'a')
95           try:
96               assert 'foo' in fid
97               fid.create_group('bar')
98               assert 'bar' in fid
99           finally:
100              fid.close()

159  class TestSpaceStrategy(TestCase):
160  
161      """ 
162          Feature: Create file with specified file space strategy 
163      """
164  
165      def test_create_with_space_strategy(self):
166          """ Create file with file space strategy """
167          fname = self.mktemp()
168          fid = File(fname, 'w', fs_strategy="page",
169                     fs_persist=True, fs_threshold=100)
170          self.assertTrue(fid)
171          # Unable to set file space strategy of an existing file
172          with self.assertRaises(ValueError):
173              File(fname, 'a', fs_strategy="page")
174          # Invalid file space strategy type
175          with self.assertRaises(ValueError):
176              File(self.mktemp(), 'w', fs_strategy="invalid")
177  
178          dset = fid.create_dataset('foo', (100,), dtype='uint8')
179          dset[...] = 1
180          dset = fid.create_dataset('bar', (100,), dtype='uint8')
181          dset[...] = 1
182          del fid['foo']
183          fid.close()
184  
185          fid = File(fname, 'a')
186          plist = fid.id.get_create_plist()
187          fs_strat = plist.get_file_space_strategy()
188          assert(fs_strat[0] == 1)
189          assert(fs_strat[1] == True)
190          assert(fs_strat[2] == 100)
191  
192          dset = fid.create_dataset('foo2', (100,), dtype='uint8')
193          dset[...] = 1
194          fid.close()

203      def test_only_with_page_strategy(self):
204          """Allow page buffering only with fs_strategy="page". 
205          """
206          fname = self.mktemp()
207          with File(fname, mode='w', fs_strategy='page', page_buf_size=16*1024):
208              pass
209          with self.assertRaises(OSError):
210              File(fname, mode='w', page_buf_size=16*1024)
211          with self.assertRaises(OSError):
212              File(fname, mode='w', fs_strategy='fsm', page_buf_size=16*1024)
213          with self.assertRaises(OSError):
214              File(fname, mode='w', fs_strategy='aggregate', page_buf_size=16*1024)

216      def test_check_page_buf_size(self):
217          """Verify set page buffer size, and minimum meta and raw eviction criteria."""
218          fname = self.mktemp()
219          pbs = 16 * 1024
220          mm = 19
221          mr = 67
222          with File(fname, mode='w', fs_strategy='page',
223                    page_buf_size=pbs, min_meta_keep=mm, min_raw_keep=mr) as f:
224              fapl = f.id.get_access_plist()
225              self.assertEqual(fapl.get_page_buffer_size(), (pbs, mm, mr))

355          # 2022/02/26: hnmaarrfk

360          # I've learned a few things:

545  
546      """ 
547          Feature: Files can be create with user blocks 
548      """
549  
550      def test_create_blocksize(self):
551          """ User blocks created with w, w-, x and properties work correctly """
552          f = File(self.mktemp(), 'w-', userblock_size=512)
553          try:
554              self.assertEqual(f.userblock_size, 512)
555          finally:
556              f.close()
557  
558          f = File(self.mktemp(), 'x', userblock_size=512)
559          try:
560              self.assertEqual(f.userblock_size, 512)
561          finally:
562              f.close()
563  
564          f = File(self.mktemp(), 'w', userblock_size=512)
565          try:
566              self.assertEqual(f.userblock_size, 512)
567          finally:
568              f.close()
569          # User block size must be an integer
570          with self.assertRaises(ValueError):
571              File(self.mktemp(), 'w', userblock_size='non')
572  
573      def test_write_only(self):
574          """ User block only allowed for write """
575          name = self.mktemp()
576          f = File(name, 'w')
577          f.close()
578  
579          with self.assertRaises(ValueError):
580              f = h5py.File(name, 'r', userblock_size=512)
581  
582          with self.assertRaises(ValueError):
583              f = h5py.File(name, 'r+', userblock_size=512)
584  
585      def test_match_existing(self):
586          """ User block size must match that of file when opening for append """
587          name = self.mktemp()
588          f = File(name, 'w', userblock_size=512)
589          f.close()
590  
591          with self.assertRaises(ValueError):
592              f = File(name, 'a', userblock_size=1024)
593  
594          f = File(name, 'a', userblock_size=512)
595          try:
596              self.assertEqual(f.userblock_size, 512)
597          finally:
598              f.close()
599  
600      def test_power_of_two(self):
601          """ User block size must be a power of 2 and at least 512 """
602          name = self.mktemp()
603  
604          with self.assertRaises(ValueError):
605              f = File(name, 'w', userblock_size=128)
606  
607          with self.assertRaises(ValueError):
608              f = File(name, 'w', userblock_size=513)

610          with self.assertRaises(ValueError):
611              f = File(name, 'w', userblock_size=1023)
612  
613      def test_write_block(self):
614          """ Test that writing to a user block does not destroy the file """
615          name = self.mktemp()
616  
617          f = File(name, 'w', userblock_size=512)
618          f.create_group("Foobar")
619          f.close()
620  
621          pyfile = open(name, 'r+b')
622          try:
623              pyfile.write(b'X' * 512)
624          finally:
625              pyfile.close()
626  
627          f = h5py.File(name, 'r')
628          try:
629              assert "Foobar" in f
630          finally:
631              f.close()
632  
633          pyfile = open(name, 'rb')
634          try:
635              self.assertEqual(pyfile.read(512), b'X' * 512)
636          finally:
637              pyfile.close()
