10   @keras_export("keras.layers.Bidirectional")
11   class Bidirectional(Layer):
12       """Bidirectional wrapper for RNNs. 
13    
14       Args: 
15           layer: `keras.layers.RNN` instance, such as 
16               `keras.layers.LSTM` or `keras.layers.GRU`. 
17               It could also be a `keras.layers.Layer` instance 
18               that meets the following criteria: 
19               1. Be a sequence-processing layer (accepts 3D+ inputs). 
20               2. Have a `go_backwards`, `return_sequences` and `return_state` 
21               attribute (with the same semantics as for the `RNN` class). 
22               3. Have an `input_spec` attribute. 
23               4. Implement serialization via `get_config()` and `from_config()`. 
24               Note that the recommended way to create new RNN layers is to write a 
25               custom RNN cell and use it with `keras.layers.RNN`, instead of 
26               subclassing `keras.layers.Layer` directly. 
27               When `return_sequences` is `True`, the output of the masked 
28               timestep will be zero regardless of the layer's original 
29               `zero_output_for_mask` value. 
30           merge_mode: Mode by which outputs of the forward and backward RNNs 
31               will be combined. One of `{"sum", "mul", "concat", "ave", None}`. 
32               If `None`, the outputs will not be combined, 
33               they will be returned as a list. Defaults to `"concat"`. 
34           backward_layer: Optional `keras.layers.RNN`, 
35               or `keras.layers.Layer` instance to be used to handle 
36               backwards input processing. 
37               If `backward_layer` is not provided, the layer instance passed 
38               as the `layer` argument will be used to generate the backward layer 
39               automatically. 
40               Note that the provided `backward_layer` layer should have properties 
41               matching those of the `layer` argument, in particular 
42               it should have the same values for `stateful`, `return_states`, 
43               `return_sequences`, etc. In addition, `backward_layer` 
44               and `layer` should have different `go_backwards` argument values. 
45               A `ValueError` will be raised if these requirements are not met. 
46    
47       Call arguments: 
48           The call arguments for this layer are the same as those of the 
49           wrapped RNN layer. Beware that when passing the `initial_state` 
50           argument during the call of this layer, the first half in the 
51           list of elements in the `initial_state` list will be passed to 
52           the forward RNN call and the last half in the list of elements 
53           will be passed to the backward RNN call. 
54    
55       Note: instantiating a `Bidirectional` layer from an existing RNN layer 
56       instance will not reuse the weights state of the RNN layer instance -- the 
57       `Bidirectional` layer will have freshly initialized weights. 
58    
59       Examples: 
60    
61       ```python 
62       model = Sequential([ 
63           Input(shape=(5, 10)), 
64           Bidirectional(LSTM(10, return_sequences=True), 
65           Bidirectional(LSTM(10)), 
66           Dense(5, activation="softmax"), 
67       ]) 
68       model.compile(loss='categorical_crossentropy', optimizer='rmsprop') 
69    
70       # With custom backward layer 
71       forward_layer = LSTM(10, return_sequences=True) 
72       backward_layer = LSTM(10, activation='relu', return_sequences=True, 
73                             go_backwards=True) 
74       model = Sequential([ 
75           Input(shape=(5, 10)), 
76           Bidirectional(forward_layer, backward_layer=backward_layer), 
77           Dense(5, activation="softmax"), 
78       ]) 
79       model.compile(loss='categorical_crossentropy', optimizer='rmsprop') 
80       ``` 
81       """

83       def __init__(
84           self,
85           layer,
86           merge_mode="concat",
87           weights=None,
88           backward_layer=None,
89           **kwargs,
90       ):
91           if not isinstance(layer, Layer):
92               raise ValueError(
93                   "Please initialize `Bidirectional` layer with a "
94                   f"`keras.layers.Layer` instance. Received: {layer}"
95               )
96           if backward_layer is not None and not isinstance(backward_layer, Layer):
97               raise ValueError(
98                   "`backward_layer` need to be a `keras.layers.Layer` "
99                   f"instance. Received: {backward_layer}"
100              )
101          if merge_mode not in ["sum", "mul", "ave", "concat", None]:
102              raise ValueError(
103                  f"Invalid merge mode. Received: {merge_mode}. "
104                  "Merge mode should be one of "
105                  '{"sum", "mul", "ave", "concat", None}'
106              )
107          super().__init__(**kwargs)
108  
109          # Recreate the forward layer from the original layer config, so that it
110          # will not carry over any state from the layer.
111          config = serialization_lib.serialize_keras_object(layer)
112          config["config"]["name"] = "forward_" + utils.removeprefix(
113              layer.name, "forward_"
114          )
115          self.forward_layer = serialization_lib.deserialize_keras_object(config)
116  
117          if backward_layer is None:
118              config = serialization_lib.serialize_keras_object(layer)
119              config["config"]["go_backwards"] = True
120              config["config"]["name"] = "backward_" + utils.removeprefix(
121                  layer.name, "backward_"
122              )
123              self.backward_layer = serialization_lib.deserialize_keras_object(
124                  config
125              )
126          else:
127              self.backward_layer = backward_layer
128          self._verify_layer_config()
129  
130          def force_zero_output_for_mask(layer):
131              # Force the zero_output_for_mask to be True if returning sequences.
132              if getattr(layer, "zero_output_for_mask", None) is not None:
133                  layer.zero_output_for_mask = layer.return_sequences
134  
135          force_zero_output_for_mask(self.forward_layer)
136          force_zero_output_for_mask(self.backward_layer)
137  
138          self.merge_mode = merge_mode
139          if weights:
140              nw = len(weights)
141              self.forward_layer.initial_weights = weights[: nw // 2]
142              self.backward_layer.initial_weights = weights[nw // 2 :]
143          self.stateful = layer.stateful
144          self.return_sequences = layer.return_sequences
145          self.return_state = layer.return_state
146          self.supports_masking = True
147          self.input_spec = layer.input_spec
148  
149      def _verify_layer_config(self):
150          """Ensure the forward and backward layers have valid common property."""
151          if self.forward_layer.go_backwards == self.backward_layer.go_backwards:
152              raise ValueError(
153                  "Forward layer and backward layer should have different "
154                  "`go_backwards` value. Received: "
155                  "forward_layer.go_backwards "
156                  f"{self.forward_layer.go_backwards}, "
157                  "backward_layer.go_backwards="
158                  f"{self.backward_layer.go_backwards}"
159              )
160  
161          common_attributes = ("stateful", "return_sequences", "return_state")
162          for a in common_attributes:
163              forward_value = getattr(self.forward_layer, a)
164              backward_value = getattr(self.backward_layer, a)
165              if forward_value != backward_value:
166                  raise ValueError(
167                      "Forward layer and backward layer are expected to have "
168                      f'the same value for attribute "{a}", got '
169                      f'"{forward_value}" for forward layer and '
170                      f'"{backward_value}" for backward layer'
171                  )
172  
173      def compute_output_shape(self, sequences_shape, initial_state_shape=None):
174          output_shape = self.forward_layer.compute_output_shape(sequences_shape)
175  
176          if self.return_state:
177              output_shape, state_shape = output_shape[0], output_shape[1:]
178  
179          if self.merge_mode == "concat":
180              output_shape = list(output_shape)
181              output_shape[-1] *= 2
182              output_shape = tuple(output_shape)
183          elif self.merge_mode is None:
184              output_shape = [output_shape, output_shape]
185  
186          if self.return_state:
187              if self.merge_mode is None:
188                  return tuple(output_shape) + state_shape + state_shape
189              return tuple([output_shape]) + (state_shape) + (state_shape)
190          return tuple(output_shape)
191  
192      def call(
193          self,
194          sequences,
195          initial_state=None,
196          mask=None,
197          training=None,
198      ):
199          kwargs = {}
200          if self.forward_layer._call_has_training_arg:
201              kwargs["training"] = training
202          if self.forward_layer._call_has_mask_arg:
203              kwargs["mask"] = mask
204  
205          if initial_state is not None:
206              # initial_states are not keras tensors, eg eager tensor from np
207              # array.  They are only passed in from kwarg initial_state, and
208              # should be passed to forward/backward layer via kwarg
209              # initial_state as well.
210              forward_inputs, backward_inputs = sequences, sequences
211              half = len(initial_state) // 2
212              forward_state = initial_state[:half]
213              backward_state = initial_state[half:]
214          else:
215              forward_inputs, backward_inputs = sequences, sequences
216              forward_state, backward_state = None, None
217  
218          y = self.forward_layer(
219              forward_inputs, initial_state=forward_state, **kwargs
220          )
221          y_rev = self.backward_layer(
222              backward_inputs, initial_state=backward_state, **kwargs
223          )
224  
225          if self.return_state:
226              states = tuple(y[1:] + y_rev[1:])
227              y = y[0]
228              y_rev = y_rev[0]
229  
230          y = ops.cast(y, self.compute_dtype)
231          y_rev = ops.cast(y_rev, self.compute_dtype)
232  
233          if self.return_sequences:
234              y_rev = ops.flip(y_rev, axis=1)
235          if self.merge_mode == "concat":
236              output = ops.concatenate([y, y_rev], axis=-1)
237          elif self.merge_mode == "sum":
238              output = y + y_rev
239          elif self.merge_mode == "ave":
240              output = (y + y_rev) / 2
241          elif self.merge_mode == "mul":
242              output = y * y_rev
243          elif self.merge_mode is None:
244              output = (y, y_rev)
245          else:
246              raise ValueError(
247                  "Unrecognized value for `merge_mode`. "
248                  f"Received: {self.merge_mode}"
249                  'Expected one of {"concat", "sum", "ave", "mul"}.'
250              )
251          if self.return_state:
252              if self.merge_mode is None:
253                  return output + states
254              return (output,) + states
255          return output
256  
257      def reset_states(self):
258          # Compatibility alias.
259          self.reset_state()
260  
261      def reset_state(self):
262          if not self.stateful:
263              raise AttributeError("Layer must be stateful.")
264          self.forward_layer.reset_state()
265          self.backward_layer.reset_state()
266  
267      @property
268      def states(self):
269          if self.forward_layer.states and self.backward_layer.states:
270              return tuple(self.forward_layer.states + self.backward_layer.states)
271          return None
272  
273      def build(self, sequences_shape, initial_state_shape=None):
274          if not self.forward_layer.built:
275              self.forward_layer.build(sequences_shape)
276          if not self.backward_layer.built:
277              self.backward_layer.build(sequences_shape)
278          self.built = True
279  
280      def compute_mask(self, _, mask):
281          if isinstance(mask, list):
282              mask = mask[0]
283          if self.return_sequences:
284              if not self.merge_mode:
285                  output_mask = (mask, mask)
286              else:
287                  output_mask = mask
288          else:
289              output_mask = (None, None) if not self.merge_mode else None
290  
291          if self.return_state and self.states is not None:
292              state_mask = (None for _ in self.states)
293              if isinstance(output_mask, list):
294                  return output_mask + state_mask * 2
295              return (output_mask,) + state_mask * 2
296          return output_mask
297  
298      def get_config(self):
299          config = {"merge_mode": self.merge_mode}
300          config["layer"] = serialization_lib.serialize_keras_object(
301              self.forward_layer
302          )
303          config["backward_layer"] = serialization_lib.serialize_keras_object(
304              self.backward_layer
305          )
306          base_config = super().get_config()
307          return {**base_config, **config}
308  
309      @classmethod
310      def from_config(cls, config, custom_objects=None):
311          # Instead of updating the input, create a copy and use that.
312          config = copy.deepcopy(config)
313  
314          config["layer"] = serialization_lib.deserialize_keras_object(
315              config["layer"], custom_objects=custom_objects
316          )
317          # Handle (optional) backward layer instantiation.
318          backward_layer_config = config.pop("backward_layer", None)
319          if backward_layer_config is not None:
320              backward_layer = serialization_lib.deserialize_keras_object(
321                  backward_layer_config, custom_objects=custom_objects
322              )
323              config["backward_layer"] = backward_layer
324          # Instantiate the wrapper, adjust it and return it.
325          layer = cls(**config)
326          return layer
