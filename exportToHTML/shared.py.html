17   def validate_leaf_output(leaf: np.ndarray, num_parallel_tree: int) -> None:
18       """Validate output for predict leaf tests."""
19       for i in range(leaf.shape[0]):  # n_samples
20           for j in range(leaf.shape[1]):  # n_rounds
21               for k in range(leaf.shape[2]):  # n_classes
22                   tree_group = leaf[i, j, k, :]
23                   assert tree_group.shape[0] == num_parallel_tree
24                   # No sampling, all trees within forest are the same
25                   assert np.all(tree_group == tree_group[0])

28   def validate_data_initialization(
29       dmatrix: Type, model: Type[xgb.XGBModel], X: ArrayLike, y: ArrayLike
30   ) -> None:
31       """Assert that we don't create duplicated DMatrix."""
32   
33       old_init = dmatrix.__init__
34       count = [0]
35   
36       def new_init(self: Any, **kwargs: Any) -> Callable:
37           count[0] += 1
38           return old_init(self, **kwargs)
39   
40       dmatrix.__init__ = new_init
41       model(n_estimators=1).fit(X, y, eval_set=[(X, y)])
42   
43       assert count[0] == 1
44       count[0] = 0  # only 1 DMatrix is created.
45   
46       y_copy = y.copy()
47       model(n_estimators=1).fit(X, y, eval_set=[(X, y_copy)])
48       assert count[0] == 2  # a different Python object is considered different
49   
50       dmatrix.__init__ = old_init

54   def get_feature_weights(
55       *,
56       X: ArrayLike,
57       y: ArrayLike,
58       fw: np.ndarray,
59       parser_path: str,
60       tree_method: str,
61       model: Type[xgb.XGBModel] = xgb.XGBRegressor,
62   ) -> np.ndarray:
63       """Get feature weights using the demo parser."""
64       with tempfile.TemporaryDirectory() as tmpdir:
65           colsample_bynode = 0.5
66           reg = model(
67               tree_method=tree_method,
68               colsample_bynode=colsample_bynode,
69               feature_weights=fw,
70           )
71   
72           reg.fit(X, y)
73           model_path = os.path.join(tmpdir, "model.json")
74           reg.save_model(model_path)
75           with open(model_path, "r", encoding="utf-8") as fd:
76               model = json.load(fd)
77   
78           spec = importlib.util.spec_from_file_location("JsonParser", parser_path)
79           assert spec is not None
80           jsonm = importlib.util.module_from_spec(spec)
81           assert spec.loader is not None
82           spec.loader.exec_module(jsonm)
83           model = jsonm.Model(model)
84           splits: Dict[int, int] = {}
85           total_nodes = 0
86           for tree in model.trees:
87               n_nodes = len(tree.nodes)
88               total_nodes += n_nodes
89               for n in range(n_nodes):
90                   if tree.is_leaf(n):
91                       continue
92                   if splits.get(tree.split_index(n), None) is None:
93                       splits[tree.split_index(n)] = 1
94                   else:
95                       splits[tree.split_index(n)] += 1
96   
97           od = collections.OrderedDict(sorted(splits.items()))
98           tuples = list(od.items())
99           k, v = list(zip(*tuples))
100          w = np.polyfit(k, v, deg=1)
101          return w
