84   def _cached_transform(
85       sub_pipeline, *, cache, param_name, param_value, transform_params
86   ):
87       """Transform a parameter value using a sub-pipeline and cache the result. 
88    
89       Parameters 
90       ---------- 
91       sub_pipeline : Pipeline 
92           The sub-pipeline to be used for transformation. 
93       cache : dict 
94           The cache dictionary to store the transformed values. 
95       param_name : str 
96           The name of the parameter to be transformed. 
97       param_value : object 
98           The value of the parameter to be transformed. 
99       transform_params : dict 
100          The metadata to be used for transformation. This passed to the 
101          `transform` method of the sub-pipeline. 
102   
103      Returns 
104      ------- 
105      transformed_value : object 
106          The transformed value of the parameter. 
107      """
108      if param_name not in cache:
109          # If the parameter is a tuple, transform each element of the
110          # tuple. This is needed to support the pattern present in
111          # `lightgbm` and `xgboost` where users can pass multiple
112          # validation sets.
113          if isinstance(param_value, tuple):
114              cache[param_name] = tuple(
115                  sub_pipeline.transform(element, **transform_params)
116                  for element in param_value
117              )
118          else:
119              cache[param_name] = sub_pipeline.transform(param_value, **transform_params)
120  
121      return cache[param_name]

124  class Pipeline(_BaseComposition):
125      """ 
126      A sequence of data transformers with an optional final predictor. 
127   
128      `Pipeline` allows you to sequentially apply a list of transformers to 
129      preprocess the data and, if desired, conclude the sequence with a final 
130      :term:`predictor` for predictive modeling. 
131   
132      Intermediate steps of the pipeline must be transformers, that is, they 
133      must implement `fit` and `transform` methods. 
134      The final :term:`estimator` only needs to implement `fit`. 
135      The transformers in the pipeline can be cached using ``memory`` argument. 
136   
137      The purpose of the pipeline is to assemble several steps that can be 
138      cross-validated together while setting different parameters. For this, it 
139      enables setting parameters of the various steps using their names and the 
140      parameter name separated by a `'__'`, as in the example below. A step's 
141      estimator may be replaced entirely by setting the parameter with its name 
142      to another estimator, or a transformer removed by setting it to 
143      `'passthrough'` or `None`. 
144   
145      For an example use case of `Pipeline` combined with 
146      :class:`~sklearn.model_selection.GridSearchCV`, refer to 
147      :ref:`sphx_glr_auto_examples_compose_plot_compare_reduction.py`. The 
148      example :ref:`sphx_glr_auto_examples_compose_plot_digits_pipe.py` shows how 
149      to grid search on a pipeline using `'__'` as a separator in the parameter names. 
150   
151      Read more in the :ref:`User Guide <pipeline>`. 
152   
153      .. versionadded:: 0.5 
154   
155      Parameters 
156      ---------- 
157      steps : list of tuples 
158          List of (name of step, estimator) tuples that are to be chained in 
159          sequential order. To be compatible with the scikit-learn API, all steps 
160          must define `fit`. All non-last steps must also define `transform`. See 
161          :ref:`Combining Estimators <combining_estimators>` for more details. 
162   
163      transform_input : list of str, default=None 
164          The names of the :term:`metadata` parameters that should be transformed by the 
165          pipeline before passing it to the step consuming it. 
166   
167          This enables transforming some input arguments to ``fit`` (other than ``X``) 
168          to be transformed by the steps of the pipeline up to the step which requires 
169          them. Requirement is defined via :ref:`metadata routing <metadata_routing>`. 
170          For instance, this can be used to pass a validation set through the pipeline. 
171   
172          You can only set this if metadata routing is enabled, which you 
173          can enable using ``sklearn.set_config(enable_metadata_routing=True)``. 
174   
175          .. versionadded:: 1.6 
176   
177      memory : str or object with the joblib.Memory interface, default=None 
178          Used to cache the fitted transformers of the pipeline. The last step 
179          will never be cached, even if it is a transformer. By default, no 
180          caching is performed. If a string is given, it is the path to the 
181          caching directory. Enabling caching triggers a clone of the transformers 
182          before fitting. Therefore, the transformer instance given to the 
183          pipeline cannot be inspected directly. Use the attribute ``named_steps`` 
184          or ``steps`` to inspect estimators within the pipeline. Caching the 
185          transformers is advantageous when fitting is time consuming. See 
186          :ref:`sphx_glr_auto_examples_neighbors_plot_caching_nearest_neighbors.py` 
187          for an example on how to enable caching. 
188   
189      verbose : bool, default=False 
190          If True, the time elapsed while fitting each step will be printed as it 
191          is completed. 
192   
193      Attributes 
194      ---------- 
195      named_steps : :class:`~sklearn.utils.Bunch` 
196          Dictionary-like object, with the following attributes. 
197          Read-only attribute to access any step parameter by user given name. 
198          Keys are step names and values are steps parameters. 
199   
200      classes_ : ndarray of shape (n_classes,) 
201          The classes labels. Only exist if the last step of the pipeline is a 
202          classifier. 
203   
204      n_features_in_ : int 
205          Number of features seen during :term:`fit`. Only defined if the 
206          underlying first estimator in `steps` exposes such an attribute 
207          when fit. 
208   
209          .. versionadded:: 0.24 
210   
211      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
212          Names of features seen during :term:`fit`. Only defined if the 
213          underlying estimator exposes such an attribute when fit. 
214   
215          .. versionadded:: 1.0 
216   
217      See Also 
218      -------- 
219      make_pipeline : Convenience function for simplified pipeline construction. 
220   
221      Examples 
222      -------- 
223      >>> from sklearn.svm import SVC 
224      >>> from sklearn.preprocessing import StandardScaler 
225      >>> from sklearn.datasets import make_classification 
226      >>> from sklearn.model_selection import train_test_split 
227      >>> from sklearn.pipeline import Pipeline 
228      >>> X, y = make_classification(random_state=0) 
229      >>> X_train, X_test, y_train, y_test = train_test_split(X, y, 
230      ...                                                     random_state=0) 
231      >>> pipe = Pipeline([('scaler', StandardScaler()), ('svc', SVC())]) 
232      >>> # The pipeline can be used as any other estimator 
233      >>> # and avoids leaking the test set into the train set 
234      >>> pipe.fit(X_train, y_train).score(X_test, y_test) 
235      0.88 
236      >>> # An estimator's parameter can be set using '__' syntax 
237      >>> pipe.set_params(svc__C=10).fit(X_train, y_train).score(X_test, y_test) 
238      0.76 
239      """

323      def _validate_steps(self):
324          names, estimators = zip(*self.steps)
325  
326          # validate names
327          self._validate_names(names)
328  
329          # validate estimators
330          transformers = estimators[:-1]
331          estimator = estimators[-1]
332  
333          for t in transformers:
334              if t is None or t == "passthrough":
335                  continue
336              if not (hasattr(t, "fit") or hasattr(t, "fit_transform")) or not hasattr(
337                  t, "transform"
338              ):
339                  raise TypeError(
340                      "All intermediate steps should be "
341                      "transformers and implement fit and transform "
342                      "or be the string 'passthrough' "
343                      "'%s' (type %s) doesn't" % (t, type(t))
344                  )
345  
346          # We allow last estimator to be None as an identity transformation
347          if (
348              estimator is not None
349              and estimator != "passthrough"
350              and not hasattr(estimator, "fit")
351          ):
352              raise TypeError(
353                  "Last step of Pipeline should implement fit "
354                  "or be the string 'passthrough'. "
355                  "'%s' (type %s) doesn't" % (estimator, type(estimator))
356              )

358      def _iter(self, with_final=True, filter_passthrough=True):
359          """ 
360          Generate (idx, (name, trans)) tuples from self.steps 
361   
362          When filter_passthrough is True, 'passthrough' and None transformers 
363          are filtered out. 
364          """
365          stop = len(self.steps)
366          if not with_final:
367              stop -= 1
368  
369          for idx, (name, trans) in enumerate(islice(self.steps, 0, stop)):
370              if not filter_passthrough:
371                  yield idx, name, trans
372              elif trans is not None and trans != "passthrough":
373                  yield idx, name, trans

381      def __getitem__(self, ind):
382          """Returns a sub-pipeline or a single estimator in the pipeline 
383   
384          Indexing with an integer will return an estimator; using a slice 
385          returns another Pipeline instance which copies a slice of this 
386          Pipeline. This copy is shallow: modifying (or fitting) estimators in 
387          the sub-pipeline will affect the larger pipeline and vice-versa. 
388          However, replacing a value in `step` will not affect a copy. 
389   
390          See 
391          :ref:`sphx_glr_auto_examples_feature_selection_plot_feature_selection_pipeline.py` 
392          for an example of how to use slicing to inspect part of a pipeline. 
393          """
394          if isinstance(ind, slice):
395              if ind.step not in (1, None):
396                  raise ValueError("Pipeline slicing only supports a step of 1")
397              return self.__class__(
398                  self.steps[ind], memory=self.memory, verbose=self.verbose
399              )
400          try:
401              name, est = self.steps[ind]
402          except TypeError:
403              # Not an int, try get step by name
404              return self.named_steps[ind]
405          return est

445      def _check_method_params(self, method, props, **kwargs):
446          if _routing_enabled():
447              routed_params = process_routing(self, method, **props, **kwargs)
448              return routed_params
449          else:
450              fit_params_steps = Bunch(
451                  **{
452                      name: Bunch(**{method: {} for method in METHODS})
453                      for name, step in self.steps
454                      if step is not None
455                  }
456              )
457              for pname, pval in props.items():
458                  if "__" not in pname:
459                      raise ValueError(
460                          "Pipeline.fit does not accept the {} parameter. "
461                          "You can pass parameters to specific steps of your "
462                          "pipeline using the stepname__parameter format, e.g. "
463                          "`Pipeline.fit(X, y, logisticregression__sample_weight"
464                          "=sample_weight)`.".format(pname)
465                      )
466                  step, param = pname.split("__", 1)
467                  fit_params_steps[step]["fit"][param] = pval
468                  # without metadata routing, fit_transform and fit_predict
469                  # get all the same params and pass it to the last fit.
470                  fit_params_steps[step]["fit_transform"][param] = pval
471                  fit_params_steps[step]["fit_predict"][param] = pval
472              return fit_params_steps

474      def _get_metadata_for_step(self, *, step_idx, step_params, all_params):
475          """Get params (metadata) for step `name`. 
476   
477          This transforms the metadata up to this step if required, which is 
478          indicated by the `transform_input` parameter. 
479   
480          If a param in `step_params` is included in the `transform_input` list, 
481          it will be transformed. 
482   
483          Parameters 
484          ---------- 
485          step_idx : int 
486              Index of the step in the pipeline. 
487   
488          step_params : dict 
489              Parameters specific to the step. These are routed parameters, e.g. 
490              `routed_params[name]`. If a parameter name here is included in the 
491              `pipeline.transform_input`, then it will be transformed. Note that 
492              these parameters are *after* routing, so the aliases are already 
493              resolved. 
494   
495          all_params : dict 
496              All parameters passed by the user. Here this is used to call 
497              `transform` on the slice of the pipeline itself. 
498   
499          Returns 
500          ------- 
501          dict 
502              Parameters to be passed to the step. The ones which should be 
503              transformed are transformed. 
504          """
505          if (
506              self.transform_input is None
507              or not all_params
508              or not step_params
509              or step_idx == 0
510          ):
511              # we only need to process step_params if transform_input is set
512              # and metadata is given by the user.
513              return step_params
514  
515          sub_pipeline = self[:step_idx]
516          sub_metadata_routing = get_routing_for_object(sub_pipeline)
517          # here we get the metadata required by sub_pipeline.transform
518          transform_params = {
519              key: value
520              for key, value in all_params.items()
521              if key
522              in sub_metadata_routing.consumes(
523                  method="transform", params=all_params.keys()
524              )
525          }
526          transformed_params = dict()  # this is to be returned
527          transformed_cache = dict()  # used to transform each param once
528          # `step_params` is the output of `process_routing`, so it has a dict for each
529          # method (e.g. fit, transform, predict), which are the args to be passed to
530          # those methods. We need to transform the parameters which are in the
531          # `transform_input`, before returning these dicts.
532          for method, method_params in step_params.items():
533              transformed_params[method] = Bunch()
534              for param_name, param_value in method_params.items():
535                  # An example of `(param_name, param_value)` is
536                  # `('sample_weight', array([0.5, 0.5, ...]))`
537                  if param_name in self.transform_input:
538                      # This parameter now needs to be transformed by the sub_pipeline, to
539                      # this step. We cache these computations to avoid repeating them.
540                      transformed_params[method][param_name] = _cached_transform(
541                          sub_pipeline,
542                          cache=transformed_cache,
543                          param_name=param_name,
544                          param_value=param_value,
545                          transform_params=transform_params,
546                      )
547                  else:
548                      transformed_params[method][param_name] = param_value
549          return transformed_params

553      def _fit(self, X, y=None, routed_params=None, raw_params=None):
554          """Fit the pipeline except the last step. 
555   
556          routed_params is the output of `process_routing` 
557          raw_params is the parameters passed by the user, used when `transform_input` 
558              is set by the user, to transform metadata using a sub-pipeline. 
559          """
560          # shallow copy of steps - this should really be steps_
561          self.steps = list(self.steps)
562          self._validate_steps()
563          # Setup the memory
564          memory = check_memory(self.memory)
565  
566          fit_transform_one_cached = memory.cache(_fit_transform_one)
567  
568          for step_idx, name, transformer in self._iter(
569              with_final=False, filter_passthrough=False
570          ):
571              if transformer is None or transformer == "passthrough":
572                  with _print_elapsed_time("Pipeline", self._log_message(step_idx)):
573                      continue
574  
575              if hasattr(memory, "location") and memory.location is None:
576                  # we do not clone when caching is disabled to
577                  # preserve backward compatibility
578                  cloned_transformer = transformer
579              else:
580                  cloned_transformer = clone(transformer)
581              # Fit or load from cache the current transformer
582              step_params = self._get_metadata_for_step(
583                  step_idx=step_idx,
584                  step_params=routed_params[name],
585                  all_params=raw_params,
586              )
587  
588              X, fitted_transformer = fit_transform_one_cached(
589                  cloned_transformer,
590                  X,
591                  y,
592                  weight=None,
593                  message_clsname="Pipeline",
594                  message=self._log_message(step_idx),
595                  params=step_params,
596              )
597              # Replace the transformer of the step with the fitted
598              # transformer. This is necessary when loading the transformer
599              # from the cache.
600              self.steps[step_idx] = (name, fitted_transformer)
601          return X

603      @_fit_context(
604          # estimators in Pipeline.steps are not validated yet
605          prefer_skip_nested_validation=False
606      )
607      def fit(self, X, y=None, **params):
608          """Fit the model. 
609   
610          Fit all the transformers one after the other and sequentially transform the 
611          data. Finally, fit the transformed data using the final estimator. 
612   
613          Parameters 
614          ---------- 
615          X : iterable 
616              Training data. Must fulfill input requirements of first step of the 
617              pipeline. 
618   
619          y : iterable, default=None 
620              Training targets. Must fulfill label requirements for all steps of 
621              the pipeline. 
622   
623          **params : dict of str -> object 
624              - If `enable_metadata_routing=False` (default): Parameters passed to the 
625                ``fit`` method of each step, where each parameter name is prefixed such 
626                that parameter ``p`` for step ``s`` has key ``s__p``. 
627   
628              - If `enable_metadata_routing=True`: Parameters requested and accepted by 
629                steps. Each step must have requested certain metadata for these parameters 
630                to be forwarded to them. 
631   
632              .. versionchanged:: 1.4 
633                  Parameters are now passed to the ``transform`` method of the 
634                  intermediate steps as well, if requested, and if 
635                  `enable_metadata_routing=True` is set via 
636                  :func:`~sklearn.set_config`. 
637   
638              See :ref:`Metadata Routing User Guide <metadata_routing>` for more 
639              details. 
640   
641          Returns 
642          ------- 
643          self : object 
644              Pipeline with fitted steps. 
645          """
646          if not _routing_enabled() and self.transform_input is not None:
647              raise ValueError(
648                  "The `transform_input` parameter can only be set if metadata "
649                  "routing is enabled. You can enable metadata routing using "
650                  "`sklearn.set_config(enable_metadata_routing=True)`."
651              )
652  
653          routed_params = self._check_method_params(method="fit", props=params)
654          Xt = self._fit(X, y, routed_params, raw_params=params)
655          with _print_elapsed_time("Pipeline", self._log_message(len(self.steps) - 1)):
656              if self._final_estimator != "passthrough":
657                  last_step_params = self._get_metadata_for_step(
658                      step_idx=len(self) - 1,
659                      step_params=routed_params[self.steps[-1][0]],
660                      all_params=params,
661                  )
662                  self._final_estimator.fit(Xt, y, **last_step_params["fit"])
663  
664          return self

679          """Fit the model and transform with the final estimator. 
680

682          the data. Only valid if the final estimator either implements 
683          `fit_transform` or `fit` and `transform`. 
684   
685          Parameters 
686          ---------- 
687          X : iterable 
688              Training data. Must fulfill input requirements of first step of the 
689              pipeline. 
690   
691          y : iterable, default=None 
692              Training targets. Must fulfill label requirements for all steps of 
693              the pipeline. 
694   
695          **params : dict of str -> object

697                ``fit`` method of each step, where each parameter name is prefixed such 
698                that parameter ``p`` for step ``s`` has key ``s__p``.

701                steps. Each step must have requested certain metadata for these parameters

710              details.

712          Returns

719

723                  return Xt

731                      Xt, y, **last_step_params["fit_transform"]

1301  
1302         An empty pipeline is considered fitted. 
1303         """
1304 
1305         # First find the last step that is not 'passthrough'
1306         last_step = None
1307         for _, estimator in reversed(self.steps):
1308             if estimator != "passthrough":
1309                 last_step = estimator
1310                 break
1311 
1312         if last_step is None:
1313             # All steps are 'passthrough', so the pipeline is considered fitted
1314             return True
1315 
1316         try:
1317             # check if the last step of the pipeline is fitted
1318             # we only check the last step since if the last step is fit, it
1319             # means the previous steps should also be fit. This is faster than
1320             # checking if every step of the pipeline is fit.
1321             check_is_fitted(last_step)
1322             return True
1323         except NotFittedError:
1324             return False

1422     """Generate names for estimators."""
1423 
1424     names = [
1425         estimator if isinstance(estimator, str) else type(estimator).__name__.lower()
1426         for estimator in estimators

1431

1445 def make_pipeline(*steps, memory=None, transform_input=None, verbose=False):
1446     """Construct a :class:`Pipeline` from the given estimators. 
1447  
1448     This is a shorthand for the :class:`Pipeline` constructor; it does not 
1449     require, and does not permit, naming the estimators. Instead, their names 
1450     will be set to the lowercase of their types automatically. 
1451  
1452     Parameters 
1453     ---------- 
1454     *steps : list of Estimator objects 
1455         List of the scikit-learn estimators that are chained together. 
1456  
1457     memory : str or object with the joblib.Memory interface, default=None 
1458         Used to cache the fitted transformers of the pipeline. The last step 
1459         will never be cached, even if it is a transformer. By default, no 
1460         caching is performed. If a string is given, it is the path to the 
1461         caching directory. Enabling caching triggers a clone of the transformers 
1462         before fitting. Therefore, the transformer instance given to the 
1463         pipeline cannot be inspected directly. Use the attribute ``named_steps`` 
1464         or ``steps`` to inspect estimators within the pipeline. Caching the 
1465         transformers is advantageous when fitting is time consuming. 
1466  
1467     transform_input : list of str, default=None 
1468         This enables transforming some input arguments to ``fit`` (other than ``X``) 
1469         to be transformed by the steps of the pipeline up to the step which requires 
1470         them. Requirement is defined via :ref:`metadata routing <metadata_routing>`. 
1471         This can be used to pass a validation set through the pipeline for instance. 
1472  
1473         You can only set this if metadata routing is enabled, which you 
1474         can enable using ``sklearn.set_config(enable_metadata_routing=True)``. 
1475  
1476         .. versionadded:: 1.6 
1477  
1478     verbose : bool, default=False 
1479         If True, the time elapsed while fitting each step will be printed as it 
1480         is completed. 
1481  
1482     Returns 
1483     ------- 
1484     p : Pipeline 
1485         Returns a scikit-learn :class:`Pipeline` object. 
1486  
1487     See Also 
1488     -------- 
1489     Pipeline : Class for creating a pipeline of transforms with a final 
1490         estimator. 
1491  
1492     Examples 
1493     -------- 
1494     >>> from sklearn.naive_bayes import GaussianNB 
1495     >>> from sklearn.preprocessing import StandardScaler 
1496     >>> from sklearn.pipeline import make_pipeline 
1497     >>> make_pipeline(StandardScaler(), GaussianNB(priors=None)) 
1498     Pipeline(steps=[('standardscaler', StandardScaler()), 
1499                     ('gaussiannb', GaussianNB())]) 
1500     """
1501     return Pipeline(
1502         _name_estimators(steps),
1503         transform_input=transform_input,
1504         memory=memory,
1505         verbose=verbose,
1506     )

1947               sub-transformers. See :ref:`Metadata Routing User Guide 
1948               <metadata_routing>` for more details. 
1949  
1950             .. versionchanged:: 1.5 
1951                 `**params` can now be routed via metadata routing API. 
1952  
1953         Returns 
1954         ------- 
1955         X_t : array-like or sparse matrix of \ 
1956                 shape (n_samples, sum_n_components) 
1957             The `hstack` of results of transformers. `sum_n_components` is the 
1958             sum of `n_components` (output dimension) over transformers. 
1959         """
1960         if _routing_enabled():
1961             routed_params = process_routing(self, "fit_transform", **params)
1962         else:
1963             # TODO(SLEP6): remove when metadata routing cannot be disabled.
1964             routed_params = Bunch()
1965             for name, obj in self.transformer_list:
1966                 if hasattr(obj, "fit_transform"):
1967                     routed_params[name] = Bunch(fit_transform={})
1968                     routed_params[name].fit_transform = params
1969                 else:
1970                     routed_params[name] = Bunch(fit={})
1971                     routed_params[name] = Bunch(transform={})
1972                     routed_params[name].fit = params
1973 
1974         results = self._parallel_func(X, y, _fit_transform_one, routed_params)
1975         if not results:
1976             # All transformers are None
1977             return np.zeros((X.shape[0], 0))
1978 
1979         Xs, transformers = zip(*results)
1980         self._update_transformer_list(transformers)
1981 
1982         return self._hstack(Xs)

1989     def _parallel_func(self, X, y, func, routed_params):

1991         self.transformer_list = list(self.transformer_list)
1992         self._validate_transformers()
1993         self._validate_transformer_weights()

2014         X : iterable or array-like, depending on transformers 
2015             Input data to be transformed. 
2016  
2017         **params : dict, default=None 
2018  
2019             Parameters routed to the `transform` method of the sub-transformers via the 
2020             metadata routing API. See :ref:`Metadata Routing User Guide 
2021             <metadata_routing>` for more details.

2023             .. versionadded:: 1.5 
2024  
2025         Returns 
2026         ------- 
2027         X_t : array-like or sparse matrix of shape (n_samples, sum_n_components) 
2028             The `hstack` of results of transformers. `sum_n_components` is the 
2029             sum of `n_components` (output dimension) over transformers.

2031         _raise_for_params(params, self, "transform")

2033         if _routing_enabled():
2034             routed_params = process_routing(self, "transform", **params)

2051     def _hstack(self, Xs):
2052         adapter = _get_container_adapter("transform", self)
2053         if adapter and all(adapter.is_supported_container(X) for X in Xs):
2054             return adapter.hstack(Xs)
2055 
2056         if any(sparse.issparse(f) for f in Xs):
2057             Xs = sparse.hstack(Xs).tocsr()
2058         else:
2059             Xs = np.hstack(Xs)
2060         return Xs
