6    # This does the basic command line handling.
7    
8    # Keep this as short as possible, cos error output is only redirected if
9    # this runs OK.  Errors in imported modules are much better - the messages go somewhere (not any more :-)
10   
11   import os
12   import sys
13   
14   import win32api
15   import win32ui
16   
17   if not sys.argv:
18       # Initialize sys.argv from commandline. When sys.argv is empty list (
19       # different from [''] meaning "no cmd line arguments" ), then C
20       # bootstrapping or another method of invocation failed to initialize
21       # sys.argv and it will be done here. ( This was a workaround for a bug in
22       # win32ui but is retained for other situations. )
23       argv = win32api.CommandLineToArgv(win32api.GetCommandLine())
24       sys.argv = argv[1:]
25       if os.getcwd() not in sys.path and "." not in sys.path:
26           sys.path.insert(0, os.getcwd())
27   
28   # You may wish to redirect error output somewhere useful if you have startup errors.
29   # eg, 'import win32traceutil' will do this for you.
30   # import win32traceutil # Just uncomment this line to see error output!
31

34   #   softspace=1

36   #       win32ui.OutputDebug(message)

39   # To fix a problem with Pythonwin when started from the Pythonwin directory,
40   # we update the pywin path to ensure it is absolute.
41   # If it is indeed relative, it will be relative to our current directory.
42   # If it's already absolute, then this will have no affect.
43   import pywin
44   import pywin.framework
45   
46   # Ensure we're working on __path__ as list, not Iterable
47   pywin.__path__ = list(pywin.__path__)
48   pywin.framework.__path__ = list(pywin.framework.__path__)
49   
50   pywin.__path__[0] = win32ui.FullPath(pywin.__path__[0])
51   pywin.framework.__path__[0] = win32ui.FullPath(pywin.framework.__path__[0])
52   
53   # make a few weird sys values.  This is so later we can clobber sys.argv to trick
54   # scripts when running under a GUI environment.
55   
56   moduleName = "pywin.framework.intpyapp"
57   sys.appargvoffset = 0
58   sys.appargv = sys.argv[:]
59   # Must check for /app param here.
60   if len(sys.argv) >= 2 and sys.argv[0].lower() in ("/app", "-app"):
61       from . import cmdline
62   
63       moduleName = cmdline.FixArgFileName(sys.argv[1])
64       sys.appargvoffset = 2
65       newargv = sys.argv[sys.appargvoffset :]
66       #   newargv.insert(0, sys.argv[0])
67       sys.argv = newargv
68   
69   # Import the application module.
70   __import__(moduleName)
