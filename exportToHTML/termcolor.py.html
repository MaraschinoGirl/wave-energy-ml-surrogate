91   def _can_do_colour(
92       *, no_color: bool | None = None, force_color: bool | None = None
93   ) -> bool:
94       """Check env vars and for tty/dumb terminal"""
95       # First check overrides:
96       # "User-level configuration files and per-instance command-line arguments should
97       # override $NO_COLOR. A user should be able to export $NO_COLOR in their shell
98       # configuration file as a default, but configure a specific program in its
99       # configuration file to specifically enable color."
100      # https://no-color.org
101      if no_color is not None and no_color:
102          return False
103      if force_color is not None and force_color:
104          return True
105  
106      # Then check env vars:
107      if os.environ.get("ANSI_COLORS_DISABLED"):
108          return False
109      if os.environ.get("NO_COLOR"):
110          return False
111      if os.environ.get("FORCE_COLOR"):
112          return True
113  
114      # Then check system:
115      if os.environ.get("TERM") == "dumb":
116          return False
117      if not hasattr(sys.stdout, "fileno"):
118          return False
119  
120      try:
121          return os.isatty(sys.stdout.fileno())
122      except io.UnsupportedOperation:
123          return sys.stdout.isatty()

126  def colored(
127      text: object,
128      color: str | None = None,
129      on_color: str | None = None,
130      attrs: Iterable[str] | None = None,
131      *,
132      no_color: bool | None = None,
133      force_color: bool | None = None,
134  ) -> str:
135      """Colorize text. 
136   
137      Available text colors: 
138          black, red, green, yellow, blue, magenta, cyan, white, 
139          light_grey, dark_grey, light_red, light_green, light_yellow, light_blue, 
140          light_magenta, light_cyan. 
141   
142      Available text highlights: 
143          on_black, on_red, on_green, on_yellow, on_blue, on_magenta, on_cyan, on_white, 
144          on_light_grey, on_dark_grey, on_light_red, on_light_green, on_light_yellow, 
145          on_light_blue, on_light_magenta, on_light_cyan. 
146   
147      Available attributes: 
148          bold, dark, underline, blink, reverse, concealed. 
149   
150      Example: 
151          colored('Hello, World!', 'red', 'on_black', ['bold', 'blink']) 
152          colored('Hello, World!', 'green') 
153      """
154      result = str(text)
155      if not _can_do_colour(no_color=no_color, force_color=force_color):
156          return result
157  
158      fmt_str = "\033[%dm%s"
159      if color is not None:
160          result = fmt_str % (COLORS[color], result)
161  
162      if on_color is not None:
163          result = fmt_str % (HIGHLIGHTS[on_color], result)
164  
165      if attrs is not None:
166          for attr in attrs:
167              result = fmt_str % (ATTRIBUTES[attr], result)
168  
169      result += RESET
170  
171      return result

174  def cprint(
175      text: object,
176      color: str | None = None,
177      on_color: str | None = None,
178      attrs: Iterable[str] | None = None,
179      *,
180      no_color: bool | None = None,
181      force_color: bool | None = None,
182      **kwargs: Any,
183  ) -> None:
184      """Print colorized text. 
185   
186      It accepts arguments of print function. 
187      """
188  
189      print(
190          (
191              colored(
192                  text,
193                  color,
194                  on_color,
195                  attrs,
196                  no_color=no_color,
197                  force_color=force_color,
198              )
199          ),
200          **kwargs,
201      )
