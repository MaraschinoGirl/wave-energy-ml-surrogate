13   def _check_rows_and_columns(a, b):
14       """Unpacks the row and column arrays and checks their shape."""
15       check_consistent_length(*a)
16       check_consistent_length(*b)
17       checks = lambda x: check_array(x, ensure_2d=False)
18       a_rows, a_cols = map(checks, a)
19       b_rows, b_cols = map(checks, b)
20       return a_rows, a_cols, b_rows, b_cols

23   def _jaccard(a_rows, a_cols, b_rows, b_cols):
24       """Jaccard coefficient on the elements of the two biclusters."""
25       intersection = (a_rows * b_rows).sum() * (a_cols * b_cols).sum()
26   
27       a_size = a_rows.sum() * a_cols.sum()
28       b_size = b_rows.sum() * b_cols.sum()
29   
30       return intersection / (a_size + b_size - intersection)
31   
32   
33   def _pairwise_similarity(a, b, similarity):
34       """Computes pairwise similarity matrix. 
35    
36       result[i, j] is the Jaccard coefficient of a's bicluster i and b's 
37       bicluster j. 
38    
39       """
40       a_rows, a_cols, b_rows, b_cols = _check_rows_and_columns(a, b)
41       n_a = a_rows.shape[0]
42       n_b = b_rows.shape[0]
43       result = np.array(
44           [
45               [similarity(a_rows[i], a_cols[i], b_rows[j], b_cols[j]) for j in range(n_b)]
46               for i in range(n_a)
47           ]
48       )
49       return result

52   @validate_params(
53       {
54           "a": [tuple],
55           "b": [tuple],
56           "similarity": [callable, StrOptions({"jaccard"})],
57       },
58       prefer_skip_nested_validation=True,
59   )
60   def consensus_score(a, b, *, similarity="jaccard"):
61       """The similarity of two sets of biclusters. 
62    
63       Similarity between individual biclusters is computed. Then the best 
64       matching between sets is found by solving a linear sum assignment problem, 
65       using a modified Jonker-Volgenant algorithm. 
66       The final score is the sum of similarities divided by the size of 
67       the larger set. 
68    
69       Read more in the :ref:`User Guide <biclustering>`. 
70    
71       Parameters 
72       ---------- 
73       a : tuple (rows, columns) 
74           Tuple of row and column indicators for a set of biclusters. 
75    
76       b : tuple (rows, columns) 
77           Another set of biclusters like ``a``. 
78    
79       similarity : 'jaccard' or callable, default='jaccard' 
80           May be the string "jaccard" to use the Jaccard coefficient, or 
81           any function that takes four arguments, each of which is a 1d 
82           indicator vector: (a_rows, a_columns, b_rows, b_columns). 
83    
84       Returns 
85       ------- 
86       consensus_score : float 
87          Consensus score, a non-negative value, sum of similarities 
88          divided by size of larger set. 
89    
90       See Also 
91       -------- 
92       scipy.optimize.linear_sum_assignment : Solve the linear sum assignment problem. 
93    
94       References 
95       ---------- 
96       * Hochreiter, Bodenhofer, et. al., 2010. `FABIA: factor analysis 
97         for bicluster acquisition 
98         <https://www.ncbi.nlm.nih.gov/pmc/articles/PMC2881408/>`__. 
99    
100      Examples 
101      -------- 
102      >>> from sklearn.metrics import consensus_score 
103      >>> a = ([[True, False], [False, True]], [[False, True], [True, False]]) 
104      >>> b = ([[False, True], [True, False]], [[True, False], [False, True]]) 
105      >>> consensus_score(a, b, similarity='jaccard') 
106      np.float64(1.0) 
107      """
108      if similarity == "jaccard":
109          similarity = _jaccard
110      matrix = _pairwise_similarity(a, b, similarity)
111      row_indices, col_indices = linear_sum_assignment(1.0 - matrix)
112      n_a = len(a[0])
113      n_b = len(b[0])
114      return matrix[row_indices, col_indices].sum() / max(n_a, n_b)
