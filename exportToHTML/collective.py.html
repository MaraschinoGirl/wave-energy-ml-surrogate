23   @dataclass
24   class Config:
25       """User configuration for the communicator context. This is used for easier 
26       integration with distributed frameworks. Users of the collective module can pass the 
27       parameters directly into tracker and the communicator. 
28    
29       .. versionadded:: 3.0 
30    
31       Attributes 
32       ---------- 
33       retry : See `dmlc_retry` in :py:meth:`init`. 
34    
35       timeout : 
36           See `dmlc_timeout` in :py:meth:`init`. This is only used for communicators, not 
37           the tracker. They are different parameters since the timeout for tracker limits 
38           only the time for starting and finalizing the communication group, whereas the 
39           timeout for communicators limits the time used for collective operations. 
40    
41       tracker_host_ip : See :py:class:`~xgboost.tracker.RabitTracker`. 
42    
43       tracker_port : See :py:class:`~xgboost.tracker.RabitTracker`. 
44    
45       tracker_timeout : See :py:class:`~xgboost.tracker.RabitTracker`. 
46    
47       """
48   
49       retry: Optional[int] = None
50       timeout: Optional[int] = None
51   
52       tracker_host_ip: Optional[str] = None
53       tracker_port: Optional[int] = None
54       tracker_timeout: Optional[int] = None
55   
56       def get_comm_config(self, args: _Args) -> _Args:
57           """Update the arguments for the communicator."""
58           if self.retry is not None:
59               args["dmlc_retry"] = self.retry
60           if self.timeout is not None:
61               args["dmlc_timeout"] = self.timeout
62           return args

65   def init(**args: _ArgVals) -> None:
66       """Initialize the collective library with arguments. 
67    
68       Parameters 
69       ---------- 
70       args : 
71           Keyword arguments representing the parameters and their values. 
72    
73           Accepted parameters: 
74             - dmlc_communicator: The type of the communicator. 
75               * rabit: Use Rabit. This is the default if the type is unspecified. 
76               * federated: Use the gRPC interface for Federated Learning. 
77    
78           Only applicable to the Rabit communicator: 
79             - dmlc_tracker_uri: Hostname of the tracker. 
80             - dmlc_tracker_port: Port number of the tracker. 
81             - dmlc_task_id: ID of the current task, can be used to obtain deterministic 
82             - dmlc_retry: The number of retry when handling network errors. 
83             - dmlc_timeout: Timeout in seconds. 
84             - dmlc_nccl_path: Path to load (dlopen) nccl for GPU-based communication. 
85    
86           Only applicable to the Federated communicator: 
87             - federated_server_address: Address of the federated server. 
88             - federated_world_size: Number of federated workers. 
89             - federated_rank: Rank of the current worker. 
90             - federated_server_cert: Server certificate file path. Only needed for the SSL 
91               mode. 
92             - federated_client_key: Client key file path. Only needed for the SSL mode. 
93             - federated_client_cert: Client certificate file path. Only needed for the SSL 
94               mode. 
95    
96           Use upper case for environment variables, use lower case for runtime configuration. 
97    
98       """
99       _check_call(_LIB.XGCommunicatorInit(make_jcargs(**args)))

137  def communicator_print(msg: Any) -> None:
138      """Print message to the communicator. 
139   
140      This function can be used to communicate the information of 
141      the progress to the communicator. 
142   
143      Parameters 
144      ---------- 
145      msg : str 
146          The message to be printed to the communicator. 
147      """
148      if not isinstance(msg, str):
149          msg = str(msg)
150      is_dist = _LIB.XGCommunicatorIsDistributed()
151      if is_dist != 0:
152          _check_call(_LIB.XGCommunicatorPrint(c_str(msg.strip())))
153      else:
154          print(msg.strip(), flush=True)

171  def broadcast(data: _T, root: int) -> _T:
172      """Broadcast object from one node to all other nodes. 
173   
174      Parameters 
175      ---------- 
176      data : any type that can be pickled 
177          Input data, if current rank does not equal root, this can be None 
178      root : int 
179          Rank of the node to broadcast data from. 
180   
181      Returns 
182      ------- 
183      object : int 
184          the result of broadcast. 
185      """
186      rank = get_rank()
187      length = ctypes.c_ulong()
188      if root == rank:
189          assert data is not None, "need to pass in data when broadcasting"
190          s = pickle.dumps(data, protocol=pickle.HIGHEST_PROTOCOL)
191          length.value = len(s)
192      # Run first broadcast
193      _check_call(
194          _LIB.XGCommunicatorBroadcast(
195              ctypes.byref(length), ctypes.sizeof(ctypes.c_ulong), root
196          )
197      )
198      if root != rank:
199          dptr = (ctypes.c_char * length.value)()
200          # run second
201          _check_call(
202              _LIB.XGCommunicatorBroadcast(
203                  ctypes.cast(dptr, ctypes.c_void_p), length.value, root
204              )
205          )
206          data = pickle.loads(dptr.raw)
207          del dptr
208      else:
209          _check_call(
210              _LIB.XGCommunicatorBroadcast(
211                  ctypes.cast(ctypes.c_char_p(s), ctypes.c_void_p), length.value, root
212              )
213          )
214          del s
215      return data

218  # enumeration of dtypes
219  def _map_dtype(dtype: np.dtype) -> int:
220      dtype_map = {
221          np.dtype("float16"): 0,
222          np.dtype("float32"): 1,
223          np.dtype("float64"): 2,
224          np.dtype("int8"): 4,
225          np.dtype("int16"): 5,
226          np.dtype("int32"): 6,
227          np.dtype("int64"): 7,
228          np.dtype("uint8"): 8,
229          np.dtype("uint16"): 9,
230          np.dtype("uint32"): 10,
231          np.dtype("uint64"): 11,
232      }
233      try:
234          dtype_map.update({np.dtype("float128"): 3})
235      except TypeError:  # float128 doesn't exist on the system
236          pass
237  
238      if dtype not in dtype_map:
239          raise TypeError(f"data type {dtype} is not supported on the current platform.")
240  
241      return dtype_map[dtype]

256  def allreduce(data: np.ndarray, op: Op) -> np.ndarray:  # pylint:disable=invalid-name
257      """Perform allreduce, return the result. 
258   
259      Parameters 
260      ---------- 
261      data : 
262          Input data. 
263      op : 
264          Reduction operator. 
265   
266      Returns 
267      ------- 
268      result : 
269          The result of allreduce, have same shape as data 
270   
271      Notes 
272      ----- 
273      This function is not thread-safe. 
274      """
275      if not isinstance(data, np.ndarray):
276          raise TypeError("allreduce only takes in numpy.ndarray")
277      buf = data.ravel().copy()
278      _check_call(
279          _LIB.XGCommunicatorAllreduce(
280              buf.ctypes.data_as(ctypes.c_void_p),
281              buf.size,
282              _map_dtype(buf.dtype),
283              int(op),
284          )
285      )
286      return buf

294  class CommunicatorContext:
295      """A context controlling collective communicator initialization and finalization."""
296  
297      def __init__(self, **args: _ArgVals) -> None:
298          self.args = args
299          key = "dmlc_nccl_path"
300          if args.get(key, None) is not None:
301              return
302  
303          binfo = build_info()
304          if not binfo["USE_DLOPEN_NCCL"]:
305              return
306  
307          try:
308              # PyPI package of NCCL.
309              from nvidia.nccl import lib
310  
311              # There are two versions of nvidia-nccl, one is from PyPI, another one from
312              # nvidia-pyindex. We support only the first one as the second one is too old
313              # (2.9.8 as of writing).
314              if lib.__file__ is not None:
315                  dirname: Optional[str] = os.path.dirname(lib.__file__)
316              else:
317                  dirname = None
318  
319              if dirname:
320                  path = os.path.join(dirname, "libnccl.so.2")
321                  self.args[key] = path
322          except ImportError:
323              pass
324  
325      def __enter__(self) -> _Args:
326          init(**self.args)
327          assert is_distributed()
328          LOGGER.debug("-------------- communicator say hello ------------------")
329          return self.args
330  
331      def __exit__(self, *args: Any) -> None:
332          finalize()
333          LOGGER.debug("--------------- communicator say bye ------------------")
