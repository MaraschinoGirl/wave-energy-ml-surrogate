17   @register
18   class DatetimePicker(DescriptionWidget, ValueWidget, CoreWidget):
19       """ 
20       Display a widget for picking datetimes. 
21    
22       Parameters 
23       ---------- 
24    
25       value: datetime.datetime 
26           The current value of the widget. 
27    
28       disabled: bool 
29           Whether to disable user changes. 
30    
31       min: datetime.datetime 
32           The lower allowed datetime bound 
33    
34       max: datetime.datetime 
35           The upper allowed datetime bound 
36    
37       Examples 
38       -------- 
39    
40       >>> import datetime 
41       >>> import ipydatetime 
42       >>> datetime_pick = ipydatetime.DatetimePicker() 
43       >>> datetime_pick.value = datetime.datetime(2018, 09, 5, 12, 34, 3) 
44       """
45   
46       _view_name = Unicode("DatetimeView").tag(sync=True)
47       _model_name = Unicode("DatetimeModel").tag(sync=True)
48   
49       value = Datetime(None, allow_none=True).tag(sync=True, **datetime_serialization)
50       disabled = Bool(False, help="Enable or disable user changes.").tag(sync=True)
51   
52       min = Datetime(None, allow_none=True).tag(sync=True, **datetime_serialization)
53       max = Datetime(None, allow_none=True).tag(sync=True, **datetime_serialization)
54   
55       def _validate_tz(self, value):
56           if value.tzinfo is None:
57               raise TraitError('%s values needs to be timezone aware' % (self.__class__.__name__,))
58           return value
59   
60       @validate("value")
61       def _validate_value(self, proposal):
62           """Cap and floor value"""
63           value = proposal["value"]
64           if value is None:
65               return value
66           value = self._validate_tz(value)
67           if self.min and self.min > value:
68               value = max(value, self.min)
69           if self.max and self.max < value:
70               value = min(value, self.max)
71           return value
72   
73       @validate("min")
74       def _validate_min(self, proposal):
75           """Enforce min <= value <= max"""
76           min = proposal["value"]
77           if min is None:
78               return min
79           min = self._validate_tz(min)
80           if self.max and min > self.max:
81               raise TraitError("Setting min > max")
82           if self.value and min > self.value:
83               self.value = min
84           return min
85   
86       @validate("max")
87       def _validate_max(self, proposal):
88           """Enforce min <= value <= max"""
89           max = proposal["value"]
90           if max is None:
91               return max
92           max = self._validate_tz(max)
93           if self.min and max < self.min:
94               raise TraitError("setting max < min")
95           if self.value and max < self.value:
96               self.value = max
97           return max
98

100  @register
101  class NaiveDatetimePicker(DatetimePicker):
102      """ 
103      Display a widget for picking naive datetimes (i.e. timezone unaware). 
104   
105      Parameters 
106      ---------- 
107   
108      value: datetime.datetime 
109          The current value of the widget. 
110   
111      disabled: bool 
112          Whether to disable user changes. 
113   
114      min: datetime.datetime 
115          The lower allowed datetime bound 
116   
117      max: datetime.datetime 
118          The upper allowed datetime bound 
119   
120      Examples 
121      -------- 
122   
123      >>> import datetime 
124      >>> import ipydatetime 
125      >>> datetime_pick = ipydatetime.NaiveDatetimePicker() 
126      >>> datetime_pick.value = datetime.datetime(2018, 09, 5, 12, 34, 3) 
127      """
128  
129      # Replace the serializers and model names:
130  
131      _model_name = Unicode("NaiveDatetimeModel").tag(sync=True)
132  
133      value = Datetime(None, allow_none=True).tag(sync=True, **naive_serialization)
134  
135      min = Datetime(None, allow_none=True).tag(sync=True, **naive_serialization)
136      max = Datetime(None, allow_none=True).tag(sync=True, **naive_serialization)
137  
138      def _validate_tz(self, value):
139          if value.tzinfo is not None:
140              raise TraitError('%s values needs to be timezone unaware' % (self.__class__.__name__,))
141          return value
