12   class TestFormatter(SoupTest):
13       def test_default_attributes(self):
14           # Test the default behavior of Formatter.attributes().
15           formatter = Formatter()
16           tag = Tag(name="tag")
17           tag["b"] = "1"
18           tag["a"] = "2"
19   
20           # Attributes come out sorted by name. In Python 3, attributes
21           # normally come out of a dictionary in the order they were
22           # added.
23           assert [("a", "2"), ("b", "1")] == formatter.attributes(tag)
24   
25           # This works even if Tag.attrs is None, though this shouldn't
26           # normally happen.
27           tag.attrs = None
28           assert [] == formatter.attributes(tag)
29   
30           assert " " == formatter.indent

32       def test_sort_attributes(self):
33           # Test the ability to override Formatter.attributes() to,
34           # e.g., disable the normal sorting of attributes.
35           class UnsortedFormatter(Formatter):
36               def attributes(self, tag):
37                   self.called_with = tag
38                   for k, v in sorted(tag.attrs.items()):
39                       if k == "ignore":
40                           continue
41                       yield k, v
42   
43           soup = self.soup('<p cval="1" aval="2" ignore="ignored"></p>')
44           formatter = UnsortedFormatter()
45           decoded = soup.decode(formatter=formatter)
46   
47           # attributes() was called on the <p> tag. It filtered out one
48           # attribute and sorted the other two.
49           assert formatter.called_with == soup.p
50           assert '<p aval="2" cval="1"></p>' == decoded

52       def test_empty_attributes_are_booleans(self):
53           # Test the behavior of empty_attributes_are_booleans as well
54           # as which Formatters have it enabled.
55   
56           for name in ("html", "minimal", None):
57               formatter = HTMLFormatter.REGISTRY[name]
58               assert False is formatter.empty_attributes_are_booleans
59   
60           formatter = XMLFormatter.REGISTRY[None]
61           assert False is formatter.empty_attributes_are_booleans
62   
63           formatter = HTMLFormatter.REGISTRY["html5"]
64           assert True is formatter.empty_attributes_are_booleans
65   
66           # Verify that the constructor sets the value.
67           formatter = Formatter(empty_attributes_are_booleans=True)
68           assert True is formatter.empty_attributes_are_booleans
69   
70           # Now demonstrate what it does to markup.
71           for markup in ("<option selected></option>", '<option selected=""></option>'):
72               soup = self.soup(markup)
73               for formatter in ("html", "minimal", "xml", None):
74                   assert b'<option selected=""></option>' == soup.option.encode(
75                       formatter="html"
76                   )
77                   assert b"<option selected></option>" == soup.option.encode(
78                       formatter="html5"
79                   )

81       @pytest.mark.parametrize(
82           "indent,expect",
83           [
84               (None, "<a>\n<b>\ntext\n</b>\n</a>\n"),
85               (-1, "<a>\n<b>\ntext\n</b>\n</a>\n"),
86               (0, "<a>\n<b>\ntext\n</b>\n</a>\n"),
87               ("", "<a>\n<b>\ntext\n</b>\n</a>\n"),
88               (1, "<a>\n <b>\n  text\n </b>\n</a>\n"),
89               (2, "<a>\n  <b>\n    text\n  </b>\n</a>\n"),
90               ("\t", "<a>\n\t<b>\n\t\ttext\n\t</b>\n</a>\n"),
91               ("abc", "<a>\nabc<b>\nabcabctext\nabc</b>\n</a>\n"),
92               # Some invalid inputs -- the default behavior is used.
93               (object(), "<a>\n <b>\n  text\n </b>\n</a>\n"),
94               (b"bytes", "<a>\n <b>\n  text\n </b>\n</a>\n"),
95           ],
96       )
97       def test_indent(self, indent, expect):
98           # Pretty-print a tree with a Formatter set to
99           # indent in a certain way and verify the results.
100          soup = self.soup("<a><b>text</b></a>")
101          formatter = Formatter(indent=indent)
102          assert soup.prettify(formatter=formatter) == expect
103  
104          # Pretty-printing only happens with prettify(), not
105          # encode().
106          assert soup.encode(formatter=formatter) != expect

123      @pytest.mark.parametrize(
124          "s,expect_html,expect_html5",
125          [
126              # The html5 formatter is much less aggressive about escaping ampersands
127              # than the html formatter.
128              ("foo & bar", "foo &amp; bar", "foo & bar"),
129              ("foo&", "foo&amp;", "foo&"),
130              ("foo&&& bar", "foo&amp;&amp;&amp; bar", "foo&&& bar"),
131              ("x=1&y=2", "x=1&amp;y=2", "x=1&y=2"),
132              ("&123", "&amp;123", "&123"),
133              ("&abc", "&amp;abc", "&abc"),
134              ("foo &0 bar", "foo &amp;0 bar", "foo &0 bar"),
135              ("foo &lolwat bar", "foo &amp;lolwat bar", "foo &lolwat bar"),
136              # But both formatters escape what the HTML5 spec considers ambiguous ampersands.
137              ("&nosuchentity;", "&amp;nosuchentity;", "&amp;nosuchentity;"),
138          ],

140      def test_entity_substitution(self, s, expect_html, expect_html5):
141          assert HTMLFormatter.REGISTRY["html"].substitute(s) == expect_html
142          assert HTMLFormatter.REGISTRY["html5"].substitute(s) == expect_html5
143          assert HTMLFormatter.REGISTRY["html5-4.12"].substitute(s) == expect_html

145      def test_entity_round_trip(self):
146          # This is more an explanatory test and a way to avoid regressions than a test of functionality.
147  
148          markup = "<p>Some division signs: ÷ &divide; &#247; &#xf7;. These are made with: ÷ &amp;divide; &amp;#247;</p>"
149          soup = self.soup(markup)
150          assert (
151              "Some division signs: ÷ ÷ ÷ ÷. These are made with: ÷ &divide; &#247;"
152              == soup.p.string
153          )
154  
155          # Oops, I forgot to mention the entity.
156          soup.p.string = soup.p.string + " &#xf7;"
157  
158          assert (
159              "Some division signs: ÷ ÷ ÷ ÷. These are made with: ÷ &divide; &#247; &#xf7;"
160              == soup.p.string
161          )
162  
163          expect = "<p>Some division signs: &divide; &divide; &divide; &divide;. These are made with: &divide; &amp;divide; &amp;#247; &amp;#xf7;</p>"
164          assert expect == soup.p.decode(formatter="html")
165          assert expect == soup.p.decode(formatter="html5")
166  
167          markup = "<p>a & b</p>"
168          soup = self.soup(markup)
169          assert "<p>a &amp; b</p>" == soup.p.decode(formatter="html")
170          assert "<p>a & b</p>" == soup.p.decode(formatter="html5")
