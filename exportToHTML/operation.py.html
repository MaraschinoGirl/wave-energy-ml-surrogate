15   @keras_export("keras.Operation")
16   class Operation:
17       def __init__(self, dtype=None, name=None):
18           if name is None:
19               name = auto_name(self.__class__.__name__)
20           if not isinstance(name, str) or "/" in name:
21               raise ValueError(
22                   "Argument `name` must be a string and "
23                   "cannot contain character `/`. "
24                   f"Received: name={name} (of type {type(name)})"
25               )
26           self._dtype_policy = dtype_policies.get(dtype)
27           self.name = name
28           self._inbound_nodes = []
29           self._outbound_nodes = []

31       @traceback_utils.filter_traceback
32       def __call__(self, *args, **kwargs):
33           if traceback_utils.is_traceback_filtering_enabled():
34               # Wrap self.call to provide helpful info in case of exception
35               if any_symbolic_tensors(args, kwargs):
36                   call_fn = self.symbolic_call
37               else:
38                   if getattr(self, "_remat_mode", None) is not None:
39                       if getattr(self, "quantization_mode", None) is not None:
40                           call_fn = self.rematerialized_call(
41                               self.quantized_call,
42                               *args,
43                               **kwargs,
44                           )
45                       else:
46                           call_fn = self.rematerialized_call(
47                               self.call, *args, **kwargs
48                           )
49                   else:
50                       if getattr(self, "quantization_mode", None) is not None:
51                           call_fn = self.quantized_call
52                       else:
53                           call_fn = self.call
54               call_fn = traceback_utils.inject_argument_info_in_traceback(
55                   call_fn,
56                   object_name=(f"{self.__class__.__name__}.call()"),
57               )
58               return call_fn(*args, **kwargs)
59   
60           # Plain flow.
61           if any_symbolic_tensors(args, kwargs):
62               return self.symbolic_call(*args, **kwargs)
63           elif getattr(self, "_remat_mode", None) is not None:
64               if getattr(self, "quantization_mode", None) is not None:
65                   return self.rematerialized_call(
66                       self.quantized_call, *args, **kwargs
67                   )(*args, **kwargs)
68               else:
69                   return self.rematerialized_call(self.call, *args, **kwargs)(
70                       *args, **kwargs
71                   )
72           else:
73               if getattr(self, "quantization_mode", None) is not None:
74                   return self.quantized_call(*args, **kwargs)
75               else:
76                   return self.call(*args, **kwargs)

78       def symbolic_call(self, *args, **kwargs):
79           # Perform shape/dtype inference.
80           outputs = self.compute_output_spec(*args, **kwargs)
81           # Record a new node in the operations graph.
82           # The Node wires itself to inbound and outbound ops.  The
83           # Node constructor updates this op's self._inbound_nodes,
84           # sets _keras_history on the outputs, and adds itself to the
85           # `_outbound_nodes` of the ops that produced the inputs to this
86           # call.
87           Node(
88               operation=self, call_args=args, call_kwargs=kwargs, outputs=outputs
89           )
90           return outputs

98       def compute_output_spec(self, *args, **kwargs):
99           try:
100              return backend.compute_output_spec(self.call, *args, **kwargs)
101          except Exception as e:
102              new_e = e.__class__(
103                  "Could not automatically infer the output shape / dtype of "
104                  f"'{self.name}' (of type {self.__class__.__name__}). "
105                  f"Either the `{self.__class__.__name__}.call()` method "
106                  f"is incorrect, or you need to implement the "
107                  f"`{self.__class__.__name__}.compute_output_spec() / "
108                  "compute_output_shape()` method. "
109                  f"Error encountered:\n\n{e}"
110              )
111              raise new_e.with_traceback(e.__traceback__) from None

113      def __new__(cls, *args, **kwargs):
114          """We override __new__ to saving serializable constructor arguments. 
115   
116          These arguments are used to auto-generate an object serialization 
117          config, which enables user-created subclasses to be serializable 
118          out of the box in most cases without forcing the user 
119          to manually implement `get_config()`. 
120          """
121          instance = super(Operation, cls).__new__(cls)
122  
123          # Generate a config to be returned by default by `get_config()`.
124          arg_names = inspect.getfullargspec(cls.__init__).args
125          kwargs.update(dict(zip(arg_names[1 : len(args) + 1], args)))
126  
127          # Explicitly serialize `dtype` to support auto_config
128          dtype = kwargs.get("dtype", None)
129          if dtype is not None and isinstance(dtype, dtype_policies.DTypePolicy):
130              # For backward compatibility, we use a str (`name`) for
131              # `DTypePolicy`
132              if dtype.quantization_mode is None:
133                  kwargs["dtype"] = dtype.name
134              # Otherwise, use `dtype_policies.serialize`
135              else:
136                  kwargs["dtype"] = dtype_policies.serialize(dtype)
137  
138          # For safety, we only rely on auto-configs for a small set of
139          # serializable types.
140          supported_types = (str, int, float, bool, type(None))
141          try:
142              flat_arg_values = tree.flatten(kwargs)
143              auto_config = True
144              for value in flat_arg_values:
145                  if not isinstance(value, supported_types):
146                      auto_config = False
147                      break
148          except TypeError:
149              auto_config = False
150          try:
151              instance._lock = False
152              if auto_config:
153                  from keras.src.saving import serialization_lib
154  
155                  instance._auto_config = serialization_lib.SerializableDict(
156                      **kwargs
157                  )
158              else:
159                  instance._auto_config = None
160              instance._lock = True
161          except RecursionError:
162              # Setting an instance attribute in __new__ has the potential
163              # to trigger an infinite recursion if a subclass overrides
164              # setattr in an unsafe way.
165              pass
166          return instance

168      @python_utils.default
169      def get_config(self):
170          """Returns the config of the object. 
171   
172          An object config is a Python dictionary (serializable) 
173          containing the information needed to re-instantiate it. 
174          """
175          config = {
176              "name": self.name,
177          }
178  
179          if not python_utils.is_default(self.get_config):
180              # In this case the subclass implements get_config()
181              return config
182  
183          # In this case the subclass doesn't implement get_config():
184          # Let's see if we can autogenerate it.
185          if getattr(self, "_auto_config", None) is not None:
186              xtra_args = set(config.keys())
187              config.update(self._auto_config.config)
188              # Remove args non explicitly supported
189              argspec = inspect.getfullargspec(self.__init__)
190              if argspec.varkw != "kwargs":
191                  for key in xtra_args - xtra_args.intersection(argspec.args[1:]):
192                      config.pop(key, None)
193              return config
194          else:
195              raise NotImplementedError(
196                  textwrap.dedent(
197                      f"""
198          Object {self.__class__.__name__} was created by passing
199          non-serializable argument values in `__init__()`,
200          and therefore the object must override `get_config()` in
201          order to be serializable. Please implement `get_config()`.
202  
203          Example:
204  
205          class CustomLayer(keras.layers.Layer):
206              def __init__(self, arg1, arg2, **kwargs):
207                  super().__init__(**kwargs)
208                  self.arg1 = arg1
209                  self.arg2 = arg2
210  
211              def get_config(self):
212                  config = super().get_config()
213                  config.update({{
214                      "arg1": self.arg1,
215                      "arg2": self.arg2,
216                  }})
217                  return config"""
218                  )
219              )

221      @classmethod
222      def from_config(cls, config):
223          """Creates an operation from its config. 
224   
225          This method is the reverse of `get_config`, capable of instantiating the 
226          same operation from the config dictionary. 
227   
228          Note: If you override this method, you might receive a serialized dtype 
229          config, which is a `dict`. You can deserialize it as follows: 
230   
231          ```python 
232          if "dtype" in config and isinstance(config["dtype"], dict): 
233              policy = dtype_policies.deserialize(config["dtype"]) 
234          ``` 
235   
236          Args: 
237              config: A Python dictionary, typically the output of `get_config`. 
238   
239          Returns: 
240              An operation instance. 
241          """
242          # Explicitly deserialize dtype config if needed. This enables users to
243          # directly interact with the instance of `DTypePolicy`.
244          if "dtype" in config and isinstance(config["dtype"], dict):
245              config = config.copy()
246              policy = dtype_policies.deserialize(config["dtype"])
247              if (
248                  not isinstance(policy, dtype_policies.DTypePolicyMap)
249                  and policy.quantization_mode is None
250              ):
251                  # For backward compatibility, we use a str (`name`) for
252                  # `DTypePolicy`
253                  policy = policy.name
254              config["dtype"] = policy
255          try:
256              return cls(**config)
257          except Exception as e:
258              raise TypeError(
259                  f"Error when deserializing class '{cls.__name__}' using "
260                  f"config={config}.\n\nException encountered: {e}"
261              )

290      def _get_node_attribute_at_index(self, node_index, attr, attr_name):
291          """Private utility to retrieves an attribute (e.g. inputs) from a node. 
292   
293          This is used to implement the properties: 
294          - output 
295          - input 
296   
297          Args: 
298              node_index: Integer index of the node from which 
299                  to retrieve the attribute. 
300              attr: Exact node attribute name. 
301              attr_name: Human-readable attribute name, for error messages. 
302   
303          Returns: 
304              The operation's attribute `attr` at the node of index `node_index`. 
305          """
306          if not self._inbound_nodes:
307              raise AttributeError(
308                  f"The layer {self.name} has never been called "
309                  f"and thus has no defined {attr_name}."
310              )
311          if not len(self._inbound_nodes) > node_index:
312              raise ValueError(
313                  f"Asked to get {attr_name} at node "
314                  f"{node_index}, but the operation has only "
315                  f"{len(self._inbound_nodes)} inbound nodes."
316              )
317          values = getattr(self._inbound_nodes[node_index], attr)
318          if isinstance(values, list) and len(values) == 1:
319              return values[0]
320          else:
321              return values
