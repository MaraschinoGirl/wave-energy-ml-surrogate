1    """CIFAR10 small images classification dataset."""
2    
3    import os
4    
5    import numpy as np
6    
7    from keras.src import backend
8    from keras.src.api_export import keras_export
9    from keras.src.datasets.cifar import load_batch
10   from keras.src.utils.file_utils import get_file
11   
12   
13   @keras_export("keras.datasets.cifar10.load_data")
14   def load_data():
15       """Loads the CIFAR10 dataset. 
16    
17       This is a dataset of 50,000 32x32 color training images and 10,000 test 
18       images, labeled over 10 categories. See more info at the 
19       [CIFAR homepage](https://www.cs.toronto.edu/~kriz/cifar.html). 
20    
21       The classes are: 
22    
23       | Label | Description | 
24       |:-----:|-------------| 
25       |   0   | airplane    | 
26       |   1   | automobile  | 
27       |   2   | bird        | 
28       |   3   | cat         | 
29       |   4   | deer        | 
30       |   5   | dog         | 
31       |   6   | frog        | 
32       |   7   | horse       | 
33       |   8   | ship        | 
34       |   9   | truck       | 
35    
36       Returns: 
37           Tuple of NumPy arrays: `(x_train, y_train), (x_test, y_test)`. 
38    
39       **`x_train`**: `uint8` NumPy array of grayscale image data with shapes 
40         `(50000, 32, 32, 3)`, containing the training data. Pixel values range 
41         from 0 to 255. 
42    
43       **`y_train`**: `uint8` NumPy array of labels (integers in range 0-9) 
44         with shape `(50000, 1)` for the training data. 
45    
46       **`x_test`**: `uint8` NumPy array of grayscale image data with shapes 
47         `(10000, 32, 32, 3)`, containing the test data. Pixel values range 
48         from 0 to 255. 
49    
50       **`y_test`**: `uint8` NumPy array of labels (integers in range 0-9) 
51         with shape `(10000, 1)` for the test data. 
52    
53       Example: 
54    
55       ```python 
56       (x_train, y_train), (x_test, y_test) = keras.datasets.cifar10.load_data() 
57       assert x_train.shape == (50000, 32, 32, 3) 
58       assert x_test.shape == (10000, 32, 32, 3) 
59       assert y_train.shape == (50000, 1) 
60       assert y_test.shape == (10000, 1) 
61       ``` 
62       """
63       dirname = "cifar-10-batches-py-target"
64       origin = "https://www.cs.toronto.edu/~kriz/cifar-10-python.tar.gz"
65       path = get_file(
66           fname=dirname,
67           origin=origin,
68           extract=True,
69           file_hash=(  # noqa: E501
70               "6d958be074577803d12ecdefd02955f39262c83c16fe9348329d7fe0b5c001ce"
71           ),
72       )
73   
74       num_train_samples = 50000
75   
76       x_train = np.empty((num_train_samples, 3, 32, 32), dtype="uint8")
77       y_train = np.empty((num_train_samples,), dtype="uint8")
78   
79       # batches are within an inner folder
80       path = os.path.join(path, "cifar-10-batches-py")
81       for i in range(1, 6):
82           fpath = os.path.join(path, "data_batch_" + str(i))
83           (
84               x_train[(i - 1) * 10000 : i * 10000, :, :, :],
85               y_train[(i - 1) * 10000 : i * 10000],
86           ) = load_batch(fpath)
87   
88       fpath = os.path.join(path, "test_batch")
89       x_test, y_test = load_batch(fpath)
90   
91       y_train = np.reshape(y_train, (len(y_train), 1))
92       y_test = np.reshape(y_test, (len(y_test), 1))
93   
94       if backend.image_data_format() == "channels_last":
95           x_train = x_train.transpose(0, 2, 3, 1)
96           x_test = x_test.transpose(0, 2, 3, 1)
97   
98       x_test = x_test.astype(x_train.dtype)
99       y_test = y_test.astype(y_train.dtype)
100  
101      return (x_train, y_train), (x_test, y_test)
