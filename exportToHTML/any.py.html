8    """Contains the Any helper APIs."""
9    
10   from typing import Optional
11   
12   from google.protobuf import descriptor
13   from google.protobuf.message import Message
14   
15   from google.protobuf.any_pb2 import Any
16   
17   
18   def pack(
19       msg: Message,
20       type_url_prefix: Optional[str] = 'type.googleapis.com/',
21       deterministic: Optional[bool] = None,
22   ) -> Any:
23     any_msg = Any()
24     any_msg.Pack(
25         msg=msg, type_url_prefix=type_url_prefix, deterministic=deterministic
26     )
27     return any_msg
28   
29   
30   def unpack(any_msg: Any, msg: Message) -> bool:
31     return any_msg.Unpack(msg=msg)
32   
33   
34   def type_name(any_msg: Any) -> str:
35     return any_msg.TypeName()
36   
37   
38   def is_type(any_msg: Any, des: descriptor.Descriptor) -> bool:
39     return any_msg.Is(des)
