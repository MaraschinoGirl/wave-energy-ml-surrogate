3    import numpy as np
4    from numpy.testing import assert_allclose, assert_array_equal

9    def test_int_float_dict():
10       rng = np.random.RandomState(0)
11       keys = np.unique(rng.randint(100, size=10).astype(np.intp))
12       values = rng.rand(len(keys))
13   
14       d = IntFloatDict(keys, values)
15       for key, value in zip(keys, values):
16           assert d[key] == value
17       assert len(d) == len(keys)
18   
19       d.append(120, 3.0)
20       assert d[120] == 3.0
21       assert len(d) == len(keys) + 1
22       for i in range(2000):
23           d.append(i + 1000, 4.0)
24       assert d[1100] == 4.0

27   def test_int_float_dict_argmin():
28       # Test the argmin implementation on the IntFloatDict
29       keys = np.arange(100, dtype=np.intp)
30       values = np.arange(100, dtype=np.float64)
31       d = IntFloatDict(keys, values)
32       assert argmin(d) == (0, 0)

35   def test_to_arrays():
36       # Test that an IntFloatDict is converted into arrays
37       # of keys and values correctly
38       keys_in = np.array([1, 2, 3], dtype=np.intp)
39       values_in = np.array([4, 5, 6], dtype=np.float64)
40   
41       d = IntFloatDict(keys_in, values_in)
42       keys_out, values_out = d.to_arrays()
43   
44       assert keys_out.dtype == keys_in.dtype
45       assert values_in.dtype == values_out.dtype
46       assert_array_equal(keys_out, keys_in)
47       assert_allclose(values_out, values_in)
