44   // holder to ensure we release the GIL even in error conditions
45   class GilHolder {
46       PyGILState_STATE _gilState;
47       PyGILState_Release* _release;
48   public:
49       GilHolder(PyGILState_Ensure* acquire, PyGILState_Release* release) {
50           _gilState = acquire();
51           _release = release;
52       }
53   
54       ~GilHolder() {
55           _release(_gilState);
56       }
57   };
