1    """Fashion-MNIST dataset."""
2    
3    import gzip
4    import os
5    
6    import numpy as np
7    
8    from keras.src.api_export import keras_export
9    from keras.src.utils.file_utils import get_file
10   
11   
12   @keras_export("keras.datasets.fashion_mnist.load_data")
13   def load_data():
14       """Loads the Fashion-MNIST dataset. 
15    
16       This is a dataset of 60,000 28x28 grayscale images of 10 fashion categories, 
17       along with a test set of 10,000 images. This dataset can be used as 
18       a drop-in replacement for MNIST. 
19    
20       The classes are: 
21    
22       | Label | Description | 
23       |:-----:|-------------| 
24       |   0   | T-shirt/top | 
25       |   1   | Trouser     | 
26       |   2   | Pullover    | 
27       |   3   | Dress       | 
28       |   4   | Coat        | 
29       |   5   | Sandal      | 
30       |   6   | Shirt       | 
31       |   7   | Sneaker     | 
32       |   8   | Bag         | 
33       |   9   | Ankle boot  | 
34    
35       Returns: 
36    
37       Tuple of NumPy arrays: `(x_train, y_train), (x_test, y_test)`. 
38    
39       **`x_train`**: `uint8` NumPy array of grayscale image data with shapes 
40         `(60000, 28, 28)`, containing the training data. 
41    
42       **`y_train`**: `uint8` NumPy array of labels (integers in range 0-9) 
43         with shape `(60000,)` for the training data. 
44    
45       **`x_test`**: `uint8` NumPy array of grayscale image data with shapes 
46         (10000, 28, 28), containing the test data. 
47    
48       **`y_test`**: `uint8` NumPy array of labels (integers in range 0-9) 
49         with shape `(10000,)` for the test data. 
50    
51       Example: 
52    
53       ```python 
54       (x_train, y_train), (x_test, y_test) = fashion_mnist.load_data() 
55       assert x_train.shape == (60000, 28, 28) 
56       assert x_test.shape == (10000, 28, 28) 
57       assert y_train.shape == (60000,) 
58       assert y_test.shape == (10000,) 
59       ``` 
60    
61       License: 
62    
63       The copyright for Fashion-MNIST is held by Zalando SE. 
64       Fashion-MNIST is licensed under the [MIT license]( 
65           https://github.com/zalandoresearch/fashion-mnist/blob/master/LICENSE). 
66       """
67       dirname = os.path.join("datasets", "fashion-mnist")
68       base = "https://storage.googleapis.com/tensorflow/tf-keras-datasets/"
69       files = [
70           "train-labels-idx1-ubyte.gz",
71           "train-images-idx3-ubyte.gz",
72           "t10k-labels-idx1-ubyte.gz",
73           "t10k-images-idx3-ubyte.gz",
74       ]
75   
76       paths = []
77       for fname in files:
78           paths.append(get_file(fname, origin=base + fname, cache_subdir=dirname))
79   
80       with gzip.open(paths[0], "rb") as lbpath:
81           y_train = np.frombuffer(lbpath.read(), np.uint8, offset=8)
82   
83       with gzip.open(paths[1], "rb") as imgpath:
84           x_train = np.frombuffer(imgpath.read(), np.uint8, offset=16).reshape(
85               len(y_train), 28, 28
86           )
87   
88       with gzip.open(paths[2], "rb") as lbpath:
89           y_test = np.frombuffer(lbpath.read(), np.uint8, offset=8)
90   
91       with gzip.open(paths[3], "rb") as imgpath:
92           x_test = np.frombuffer(imgpath.read(), np.uint8, offset=16).reshape(
93               len(y_test), 28, 28
94           )
95   
96       return (x_train, y_train), (x_test, y_test)
