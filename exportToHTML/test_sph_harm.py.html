50   def test_first_harmonics():
51       # Test against explicit representations of the first four
52       # spherical harmonics which use `theta` as the azimuthal angle,
53       # `phi` as the polar angle, and include the Condon-Shortley
54       # phase.
55   
56       # sph_harm is deprecated and is implemented as a shim around sph_harm_y.
57       # This test is maintained to verify the correctness of the shim.
58   
59       # Notation is Ymn
60       def Y00(theta, phi):
61           return 0.5*np.sqrt(1/np.pi)
62   
63       def Yn11(theta, phi):
64           return 0.5*np.sqrt(3/(2*np.pi))*np.exp(-1j*theta)*np.sin(phi)
65   
66       def Y01(theta, phi):
67           return 0.5*np.sqrt(3/np.pi)*np.cos(phi)
68   
69       def Y11(theta, phi):
70           return -0.5*np.sqrt(3/(2*np.pi))*np.exp(1j*theta)*np.sin(phi)
71   
72       harms = [Y00, Yn11, Y01, Y11]
73       m = [0, -1, 0, 1]
74       n = [0, 1, 1, 1]
75   
76       theta = np.linspace(0, 2*np.pi)
77       phi = np.linspace(0, np.pi)
78       theta, phi = np.meshgrid(theta, phi)
79   
80       for harm, m, n in zip(harms, m, n):
81           with suppress_warnings() as sup:
82               sup.filter(category=DeprecationWarning)
83               assert_allclose(sc.sph_harm(m, n, theta, phi),
84                               harm(theta, phi),
85                               rtol=1e-15, atol=1e-15,
86                               err_msg=f"Y^{m}_{n} incorrect")
