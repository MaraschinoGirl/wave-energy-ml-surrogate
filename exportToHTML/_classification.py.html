31   def _adjusted_metric(metric, metric_kwargs, p=None):
32       metric_kwargs = metric_kwargs or {}
33       if metric == "minkowski":
34           metric_kwargs["p"] = p
35           if p == 2:
36               metric = "euclidean"
37       return metric, metric_kwargs

40   class KNeighborsClassifier(KNeighborsMixin, ClassifierMixin, NeighborsBase):
41       """Classifier implementing the k-nearest neighbors vote. 
42    
43       Read more in the :ref:`User Guide <classification>`. 
44    
45       Parameters 
46       ---------- 
47       n_neighbors : int, default=5 
48           Number of neighbors to use by default for :meth:`kneighbors` queries. 
49    
50       weights : {'uniform', 'distance'}, callable or None, default='uniform' 
51           Weight function used in prediction.  Possible values: 
52    
53           - 'uniform' : uniform weights.  All points in each neighborhood 
54             are weighted equally. 
55           - 'distance' : weight points by the inverse of their distance. 
56             in this case, closer neighbors of a query point will have a 
57             greater influence than neighbors which are further away. 
58           - [callable] : a user-defined function which accepts an 
59             array of distances, and returns an array of the same shape 
60             containing the weights. 
61    
62           Refer to the example entitled 
63           :ref:`sphx_glr_auto_examples_neighbors_plot_classification.py` 
64           showing the impact of the `weights` parameter on the decision 
65           boundary. 
66    
67       algorithm : {'auto', 'ball_tree', 'kd_tree', 'brute'}, default='auto' 
68           Algorithm used to compute the nearest neighbors: 
69    
70           - 'ball_tree' will use :class:`BallTree` 
71           - 'kd_tree' will use :class:`KDTree` 
72           - 'brute' will use a brute-force search. 
73           - 'auto' will attempt to decide the most appropriate algorithm 
74             based on the values passed to :meth:`fit` method. 
75    
76           Note: fitting on sparse input will override the setting of 
77           this parameter, using brute force. 
78    
79       leaf_size : int, default=30 
80           Leaf size passed to BallTree or KDTree.  This can affect the 
81           speed of the construction and query, as well as the memory 
82           required to store the tree.  The optimal value depends on the 
83           nature of the problem. 
84    
85       p : float, default=2 
86           Power parameter for the Minkowski metric. When p = 1, this is equivalent 
87           to using manhattan_distance (l1), and euclidean_distance (l2) for p = 2. 
88           For arbitrary p, minkowski_distance (l_p) is used. This parameter is expected 
89           to be positive. 
90    
91       metric : str or callable, default='minkowski' 
92           Metric to use for distance computation. Default is "minkowski", which 
93           results in the standard Euclidean distance when p = 2. See the 
94           documentation of `scipy.spatial.distance 
95           <https://docs.scipy.org/doc/scipy/reference/spatial.distance.html>`_ and 
96           the metrics listed in 
97           :class:`~sklearn.metrics.pairwise.distance_metrics` for valid metric 
98           values. 
99    
100          If metric is "precomputed", X is assumed to be a distance matrix and 
101          must be square during fit. X may be a :term:`sparse graph`, in which 
102          case only "nonzero" elements may be considered neighbors. 
103   
104          If metric is a callable function, it takes two arrays representing 1D 
105          vectors as inputs and must return one value indicating the distance 
106          between those vectors. This works for Scipy's metrics, but is less 
107          efficient than passing the metric name as a string. 
108   
109      metric_params : dict, default=None 
110          Additional keyword arguments for the metric function. 
111   
112      n_jobs : int, default=None 
113          The number of parallel jobs to run for neighbors search. 
114          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
115          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
116          for more details. 
117          Doesn't affect :meth:`fit` method. 
118   
119      Attributes 
120      ---------- 
121      classes_ : array of shape (n_classes,) 
122          Class labels known to the classifier 
123   
124      effective_metric_ : str or callble 
125          The distance metric used. It will be same as the `metric` parameter 
126          or a synonym of it, e.g. 'euclidean' if the `metric` parameter set to 
127          'minkowski' and `p` parameter set to 2. 
128   
129      effective_metric_params_ : dict 
130          Additional keyword arguments for the metric function. For most metrics 
131          will be same with `metric_params` parameter, but may also contain the 
132          `p` parameter value if the `effective_metric_` attribute is set to 
133          'minkowski'. 
134   
135      n_features_in_ : int 
136          Number of features seen during :term:`fit`. 
137   
138          .. versionadded:: 0.24 
139   
140      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
141          Names of features seen during :term:`fit`. Defined only when `X` 
142          has feature names that are all strings. 
143   
144          .. versionadded:: 1.0 
145   
146      n_samples_fit_ : int 
147          Number of samples in the fitted data. 
148   
149      outputs_2d_ : bool 
150          False when `y`'s shape is (n_samples, ) or (n_samples, 1) during fit 
151          otherwise True. 
152   
153      See Also 
154      -------- 
155      RadiusNeighborsClassifier: Classifier based on neighbors within a fixed radius. 
156      KNeighborsRegressor: Regression based on k-nearest neighbors. 
157      RadiusNeighborsRegressor: Regression based on neighbors within a fixed radius. 
158      NearestNeighbors: Unsupervised learner for implementing neighbor searches. 
159   
160      Notes 
161      ----- 
162      See :ref:`Nearest Neighbors <neighbors>` in the online documentation 
163      for a discussion of the choice of ``algorithm`` and ``leaf_size``. 
164   
165      .. warning:: 
166   
167         Regarding the Nearest Neighbors algorithms, if it is found that two 
168         neighbors, neighbor `k+1` and `k`, have identical distances 
169         but different labels, the results will depend on the ordering of the 
170         training data. 
171   
172      https://en.wikipedia.org/wiki/K-nearest_neighbor_algorithm 
173   
174      Examples 
175      -------- 
176      >>> X = [[0], [1], [2], [3]] 
177      >>> y = [0, 0, 1, 1] 
178      >>> from sklearn.neighbors import KNeighborsClassifier 
179      >>> neigh = KNeighborsClassifier(n_neighbors=3) 
180      >>> neigh.fit(X, y) 
181      KNeighborsClassifier(...) 
182      >>> print(neigh.predict([[1.1]])) 
183      [0] 
184      >>> print(neigh.predict_proba([[0.9]])) 
185      [[0.666... 0.333...]] 
186      """

222          """Fit the k-nearest neighbors classifier from the training dataset. 
223   
224          Parameters 
225          ---------- 
226          X : {array-like, sparse matrix} of shape (n_samples, n_features) or \ 
227                  (n_samples, n_samples) if metric='precomputed' 
228              Training data. 
229   
230          y : {array-like, sparse matrix} of shape (n_samples,) or \ 
231                  (n_samples, n_outputs) 
232              Target values. 
233   
234          Returns 
235          ------- 
236          self : KNeighborsClassifier 
237              The fitted k-nearest neighbors classifier. 
238          """
239          return self._fit(X, y)

241      def predict(self, X):
242          """Predict the class labels for the provided data. 
243   
244          Parameters 
245          ---------- 
246          X : {array-like, sparse matrix} of shape (n_queries, n_features), \ 
247                  or (n_queries, n_indexed) if metric == 'precomputed', or None 
248              Test samples. If `None`, predictions for all indexed points are 
249              returned; in this case, points are not considered their own 
250              neighbors. 
251   
252          Returns 
253          ------- 
254          y : ndarray of shape (n_queries,) or (n_queries, n_outputs) 
255              Class labels for each data sample. 
256          """
257          check_is_fitted(self, "_fit_method")
258          if self.weights == "uniform":
259              if self._fit_method == "brute" and ArgKminClassMode.is_usable_for(
260                  X, self._fit_X, self.metric
261              ):
262                  probabilities = self.predict_proba(X)
263                  if self.outputs_2d_:
264                      return np.stack(
265                          [
266                              self.classes_[idx][np.argmax(probas, axis=1)]
267                              for idx, probas in enumerate(probabilities)
268                          ],
269                          axis=1,
270                      )
271                  return self.classes_[np.argmax(probabilities, axis=1)]
272              # In that case, we do not need the distances to perform
273              # the weighting so we do not compute them.
274              neigh_ind = self.kneighbors(X, return_distance=False)
275              neigh_dist = None
276          else:
277              neigh_dist, neigh_ind = self.kneighbors(X)
278  
279          classes_ = self.classes_
280          _y = self._y
281          if not self.outputs_2d_:
282              _y = self._y.reshape((-1, 1))
283              classes_ = [self.classes_]
284  
285          n_outputs = len(classes_)
286          n_queries = _num_samples(self._fit_X if X is None else X)
287          weights = _get_weights(neigh_dist, self.weights)
288          if weights is not None and _all_with_any_reduction_axis_1(weights, value=0):
289              raise ValueError(
290                  "All neighbors of some sample is getting zero weights. "
291                  "Please modify 'weights' to avoid this case if you are "
292                  "using a user-defined function."
293              )
294  
295          y_pred = np.empty((n_queries, n_outputs), dtype=classes_[0].dtype)
296          for k, classes_k in enumerate(classes_):
297              if weights is None:
298                  mode, _ = _mode(_y[neigh_ind, k], axis=1)
299              else:
300                  mode, _ = weighted_mode(_y[neigh_ind, k], weights, axis=1)
301  
302              mode = np.asarray(mode.ravel(), dtype=np.intp)
303              y_pred[:, k] = classes_k.take(mode)
304  
305          if not self.outputs_2d_:
306              y_pred = y_pred.ravel()
307  
308          return y_pred
309  
310      def predict_proba(self, X):
311          """Return probability estimates for the test data X. 
312   
313          Parameters 
314          ---------- 
315          X : {array-like, sparse matrix} of shape (n_queries, n_features), \ 
316                  or (n_queries, n_indexed) if metric == 'precomputed', or None 
317              Test samples. If `None`, predictions for all indexed points are 
318              returned; in this case, points are not considered their own 
319              neighbors. 
320   
321          Returns 
322          ------- 
323          p : ndarray of shape (n_queries, n_classes), or a list of n_outputs \ 
324                  of such arrays if n_outputs > 1. 
325              The class probabilities of the input samples. Classes are ordered 
326              by lexicographic order. 
327          """
328          check_is_fitted(self, "_fit_method")
329          if self.weights == "uniform":
330              # TODO: systematize this mapping of metric for
331              # PairwiseDistancesReductions.
332              metric, metric_kwargs = _adjusted_metric(
333                  metric=self.metric, metric_kwargs=self.metric_params, p=self.p
334              )
335              if (
336                  self._fit_method == "brute"
337                  and ArgKminClassMode.is_usable_for(X, self._fit_X, metric)
338                  # TODO: Implement efficient multi-output solution
339                  and not self.outputs_2d_
340              ):
341                  if self.metric == "precomputed":
342                      X = _check_precomputed(X)
343                  else:
344                      X = validate_data(
345                          self, X, accept_sparse="csr", reset=False, order="C"
346                      )
347  
348                  probabilities = ArgKminClassMode.compute(
349                      X,
350                      self._fit_X,
351                      k=self.n_neighbors,
352                      weights=self.weights,
353                      Y_labels=self._y,
354                      unique_Y_labels=self.classes_,
355                      metric=metric,
356                      metric_kwargs=metric_kwargs,
357                      # `strategy="parallel_on_X"` has in practice be shown
358                      # to be more efficient than `strategy="parallel_on_Y``
359                      # on many combination of datasets.
360                      # Hence, we choose to enforce it here.
361                      # For more information, see:
362                      # https://github.com/scikit-learn/scikit-learn/pull/24076#issuecomment-1445258342  # noqa
363                      # TODO: adapt the heuristic for `strategy="auto"` for
364                      # `ArgKminClassMode` and use `strategy="auto"`.
365                      strategy="parallel_on_X",
366                  )
367                  return probabilities
368  
369              # In that case, we do not need the distances to perform
370              # the weighting so we do not compute them.
371              neigh_ind = self.kneighbors(X, return_distance=False)
372              neigh_dist = None
373          else:
374              neigh_dist, neigh_ind = self.kneighbors(X)
375  
376          classes_ = self.classes_
377          _y = self._y
378          if not self.outputs_2d_:
379              _y = self._y.reshape((-1, 1))
380              classes_ = [self.classes_]
381  
382          n_queries = _num_samples(self._fit_X if X is None else X)
383  
384          weights = _get_weights(neigh_dist, self.weights)
385          if weights is None:
386              weights = np.ones_like(neigh_ind)
387          elif _all_with_any_reduction_axis_1(weights, value=0):
388              raise ValueError(
389                  "All neighbors of some sample is getting zero weights. "
390                  "Please modify 'weights' to avoid this case if you are "
391                  "using a user-defined function."
392              )
393  
394          all_rows = np.arange(n_queries)
395          probabilities = []
396          for k, classes_k in enumerate(classes_):
397              pred_labels = _y[:, k][neigh_ind]
398              proba_k = np.zeros((n_queries, classes_k.size))
399  
400              # a simple ':' index doesn't work right
401              for i, idx in enumerate(pred_labels.T):  # loop is O(n_neighbors)
402                  proba_k[all_rows, idx] += weights[:, i]
403  
404              # normalize 'votes' into real [0,1] probabilities
405              normalizer = proba_k.sum(axis=1)[:, np.newaxis]
406              proba_k /= normalizer
407  
408              probabilities.append(proba_k)
409  
410          if not self.outputs_2d_:
411              probabilities = probabilities[0]
412  
413          return probabilities

415      # This function is defined here only to modify the parent docstring
416      # and add information about X=None
417      def score(self, X, y, sample_weight=None):
418          """ 
419          Return the mean accuracy on the given test data and labels. 
420   
421          In multi-label classification, this is the subset accuracy 
422          which is a harsh metric since you require for each sample that 
423          each label set be correctly predicted. 
424   
425          Parameters 
426          ---------- 
427          X : array-like of shape (n_samples, n_features), or None 
428              Test samples. If `None`, predictions for all indexed points are 
429              used; in this case, points are not considered their own 
430              neighbors. This means that `knn.fit(X, y).score(None, y)` 
431              implicitly performs a leave-one-out cross-validation procedure 
432              and is equivalent to `cross_val_score(knn, X, y, cv=LeaveOneOut())` 
433              but typically much faster. 
434   
435          y : array-like of shape (n_samples,) or (n_samples, n_outputs) 
436              True labels for `X`. 
437   
438          sample_weight : array-like of shape (n_samples,), default=None 
439              Sample weights. 
440   
441          Returns 
442          ------- 
443          score : float 
444              Mean accuracy of ``self.predict(X)`` w.r.t. `y`. 
445          """
446          return super().score(X, y, sample_weight)

506          results in the standard Euclidean distance when p = 2. See the 
507          documentation of `scipy.spatial.distance 
508          <https://docs.scipy.org/doc/scipy/reference/spatial.distance.html>`_ and

652          ---------- 
653          X : {array-like, sparse matrix} of shape (n_samples, n_features) or \ 
654                  (n_samples, n_samples) if metric='precomputed' 
655              Training data. 
656   
657          y : {array-like, sparse matrix} of shape (n_samples,) or \ 
658                  (n_samples, n_outputs) 
659              Target values. 
660   
661          Returns 
662          ------- 
663          self : RadiusNeighborsClassifier 
664              The fitted radius neighbors classifier. 
665          """
666          self._fit(X, y)
667  
668          classes_ = self.classes_
669          _y = self._y
670          if not self.outputs_2d_:
671              _y = self._y.reshape((-1, 1))
672              classes_ = [self.classes_]
673  
674          if self.outlier_label is None:
675              outlier_label_ = None
676  
677          elif self.outlier_label == "most_frequent":
678              outlier_label_ = []
679              # iterate over multi-output, get the most frequent label for each
680              # output.
681              for k, classes_k in enumerate(classes_):
682                  label_count = np.bincount(_y[:, k])
683                  outlier_label_.append(classes_k[label_count.argmax()])
684  
685          else:
686              if _is_arraylike(self.outlier_label) and not isinstance(
687                  self.outlier_label, str
688              ):
689                  if len(self.outlier_label) != len(classes_):
690                      raise ValueError(
691                          "The length of outlier_label: {} is "
692                          "inconsistent with the output "
693                          "length: {}".format(self.outlier_label, len(classes_))
694                      )
695                  outlier_label_ = self.outlier_label
696              else:
697                  outlier_label_ = [self.outlier_label] * len(classes_)
698  
699              for classes, label in zip(classes_, outlier_label_):
700                  if _is_arraylike(label) and not isinstance(label, str):
701                      # ensure the outlier label for each output is a scalar.
702                      raise TypeError(
703                          "The outlier_label of classes {} is "
704                          "supposed to be a scalar, got "
705                          "{}.".format(classes, label)
706                      )
707                  if np.append(classes, label).dtype != classes.dtype:
708                      # ensure the dtype of outlier label is consistent with y.
709                      raise TypeError(
710                          "The dtype of outlier_label {} is "
711                          "inconsistent with classes {} in "
712                          "y.".format(label, classes)
713                      )
714  
715          self.outlier_label_ = outlier_label_
716  
717          return self
718  
719      def predict(self, X):
720          """Predict the class labels for the provided data. 
721   
722          Parameters 
723          ---------- 
724          X : {array-like, sparse matrix} of shape (n_queries, n_features), \ 
725                  or (n_queries, n_indexed) if metric == 'precomputed', or None 
726              Test samples. If `None`, predictions for all indexed points are 
727              returned; in this case, points are not considered their own 
728              neighbors. 
729   
730          Returns 
731          ------- 
732          y : ndarray of shape (n_queries,) or (n_queries, n_outputs) 
733              Class labels for each data sample. 
734          """
735  
736          probs = self.predict_proba(X)
737          classes_ = self.classes_
738  
739          if not self.outputs_2d_:
740              probs = [probs]
741              classes_ = [self.classes_]
742  
743          n_outputs = len(classes_)
744          n_queries = probs[0].shape[0]
745          y_pred = np.empty((n_queries, n_outputs), dtype=classes_[0].dtype)
746  
747          for k, prob in enumerate(probs):
748              # iterate over multi-output, assign labels based on probabilities
749              # of each output.
750              max_prob_index = prob.argmax(axis=1)
751              y_pred[:, k] = classes_[k].take(max_prob_index)
752  
753              outlier_zero_probs = (prob == 0).all(axis=1)
754              if outlier_zero_probs.any():
755                  zero_prob_index = np.flatnonzero(outlier_zero_probs)
756                  y_pred[zero_prob_index, k] = self.outlier_label_[k]
757  
758          if not self.outputs_2d_:
759              y_pred = y_pred.ravel()
760  
761          return y_pred
762  
763      def predict_proba(self, X):
764          """Return probability estimates for the test data X. 
765   
766          Parameters 
767          ---------- 
768          X : {array-like, sparse matrix} of shape (n_queries, n_features), \ 
769                  or (n_queries, n_indexed) if metric == 'precomputed', or None 
770              Test samples. If `None`, predictions for all indexed points are 
771              returned; in this case, points are not considered their own 
772              neighbors. 
773   
774          Returns 
775          ------- 
776          p : ndarray of shape (n_queries, n_classes), or a list of \ 
777                  n_outputs of such arrays if n_outputs > 1. 
778              The class probabilities of the input samples. Classes are ordered 
779              by lexicographic order. 
780          """
781          check_is_fitted(self, "_fit_method")
782          n_queries = _num_samples(self._fit_X if X is None else X)
783  
784          metric, metric_kwargs = _adjusted_metric(
785              metric=self.metric, metric_kwargs=self.metric_params, p=self.p
786          )
787  
788          if (
789              self.weights == "uniform"
790              and self._fit_method == "brute"
791              and not self.outputs_2d_
792              and RadiusNeighborsClassMode.is_usable_for(X, self._fit_X, metric)
793          ):
794              probabilities = RadiusNeighborsClassMode.compute(
795                  X=X,
796                  Y=self._fit_X,
797                  radius=self.radius,
798                  weights=self.weights,
799                  Y_labels=self._y,
800                  unique_Y_labels=self.classes_,
801                  outlier_label=self.outlier_label,
802                  metric=metric,
803                  metric_kwargs=metric_kwargs,
804                  strategy="parallel_on_X",
805                  # `strategy="parallel_on_X"` has in practice be shown
806                  # to be more efficient than `strategy="parallel_on_Y``
807                  # on many combination of datasets.
808                  # Hence, we choose to enforce it here.
809                  # For more information, see:
810                  # https://github.com/scikit-learn/scikit-learn/pull/26828/files#r1282398471  # noqa
811              )
812              return probabilities
813  
814          neigh_dist, neigh_ind = self.radius_neighbors(X)
815          outlier_mask = np.zeros(n_queries, dtype=bool)
816          outlier_mask[:] = [len(nind) == 0 for nind in neigh_ind]
817          outliers = np.flatnonzero(outlier_mask)
818          inliers = np.flatnonzero(~outlier_mask)
819  
820          classes_ = self.classes_
821          _y = self._y
822          if not self.outputs_2d_:
823              _y = self._y.reshape((-1, 1))
824              classes_ = [self.classes_]
825  
826          if self.outlier_label_ is None and outliers.size > 0:
827              raise ValueError(
828                  "No neighbors found for test samples %r, "
829                  "you can try using larger radius, "
830                  "giving a label for outliers, "
831                  "or considering removing them from your dataset." % outliers
832              )
833  
834          weights = _get_weights(neigh_dist, self.weights)
835          if weights is not None:
836              weights = weights[inliers]
837  
838          probabilities = []
839          # iterate over multi-output, measure probabilities of the k-th output.
840          for k, classes_k in enumerate(classes_):
841              pred_labels = np.zeros(len(neigh_ind), dtype=object)
842              pred_labels[:] = [_y[ind, k] for ind in neigh_ind]
843  
844              proba_k = np.zeros((n_queries, classes_k.size))
845              proba_inl = np.zeros((len(inliers), classes_k.size))
846  
847              # samples have different size of neighbors within the same radius
848              if weights is None:
849                  for i, idx in enumerate(pred_labels[inliers]):
850                      proba_inl[i, :] = np.bincount(idx, minlength=classes_k.size)
851              else:
852                  for i, idx in enumerate(pred_labels[inliers]):
853                      proba_inl[i, :] = np.bincount(
854                          idx, weights[i], minlength=classes_k.size
855                      )
856              proba_k[inliers, :] = proba_inl
857  
858              if outliers.size > 0:
859                  _outlier_label = self.outlier_label_[k]
860                  label_index = np.flatnonzero(classes_k == _outlier_label)
861                  if label_index.size == 1:
862                      proba_k[outliers, label_index[0]] = 1.0
863                  else:
864                      warnings.warn(
865                          "Outlier label {} is not in training "
866                          "classes. All class probabilities of "
867                          "outliers will be assigned with 0."
868                          "".format(self.outlier_label_[k])
869                      )
870  
871              # normalize 'votes' into real [0,1] probabilities
872              normalizer = proba_k.sum(axis=1)[:, np.newaxis]
873              normalizer[normalizer == 0.0] = 1.0
874              proba_k /= normalizer
875  
876              probabilities.append(proba_k)
877  
878          if not self.outputs_2d_:
879              probabilities = probabilities[0]
880  
881          return probabilities
882  
883      # This function is defined here only to modify the parent docstring
884      # and add information about X=None
885      def score(self, X, y, sample_weight=None):
886          """ 
887          Return the mean accuracy on the given test data and labels. 
888   
889          In multi-label classification, this is the subset accuracy 
890          which is a harsh metric since you require for each sample that 
891          each label set be correctly predicted. 
892   
893          Parameters 
894          ---------- 
895          X : array-like of shape (n_samples, n_features), or None 
896              Test samples. If `None`, predictions for all indexed points are 
897              used; in this case, points are not considered their own 
898              neighbors. This means that `knn.fit(X, y).score(None, y)` 
899              implicitly performs a leave-one-out cross-validation procedure 
900              and is equivalent to `cross_val_score(knn, X, y, cv=LeaveOneOut())` 
901              but typically much faster. 
902   
903          y : array-like of shape (n_samples,) or (n_samples, n_outputs) 
904              True labels for `X`. 
905   
906          sample_weight : array-like of shape (n_samples,), default=None 
907              Sample weights. 
908   
909          Returns 
910          ------- 
911          score : float 
912              Mean accuracy of ``self.predict(X)`` w.r.t. `y`. 
913          """
914          return super().score(X, y, sample_weight)
