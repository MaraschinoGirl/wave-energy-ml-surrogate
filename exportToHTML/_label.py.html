30   class LabelEncoder(TransformerMixin, BaseEstimator, auto_wrap_output_keys=None):
31       """Encode target labels with value between 0 and n_classes-1. 
32    
33       This transformer should be used to encode target values, *i.e.* `y`, and 
34       not the input `X`. 
35    
36       Read more in the :ref:`User Guide <preprocessing_targets>`. 
37    
38       .. versionadded:: 0.12 
39    
40       Attributes 
41       ---------- 
42       classes_ : ndarray of shape (n_classes,) 
43           Holds the label for each class. 
44    
45       See Also 
46       -------- 
47       OrdinalEncoder : Encode categorical features using an ordinal encoding 
48           scheme. 
49       OneHotEncoder : Encode categorical features as a one-hot numeric array. 
50    
51       Examples 
52       -------- 
53       `LabelEncoder` can be used to normalize labels. 
54    
55       >>> from sklearn.preprocessing import LabelEncoder 
56       >>> le = LabelEncoder() 
57       >>> le.fit([1, 2, 2, 6]) 
58       LabelEncoder() 
59       >>> le.classes_ 
60       array([1, 2, 6]) 
61       >>> le.transform([1, 1, 2, 6]) 
62       array([0, 0, 1, 2]...) 
63       >>> le.inverse_transform([0, 0, 1, 2]) 
64       array([1, 1, 2, 6]) 
65    
66       It can also be used to transform non-numerical labels (as long as they are 
67       hashable and comparable) to numerical labels. 
68    
69       >>> le = LabelEncoder() 
70       >>> le.fit(["paris", "paris", "tokyo", "amsterdam"]) 
71       LabelEncoder() 
72       >>> list(le.classes_) 
73       [np.str_('amsterdam'), np.str_('paris'), np.str_('tokyo')] 
74       >>> le.transform(["tokyo", "tokyo", "paris"]) 
75       array([2, 2, 1]...) 
76       >>> list(le.inverse_transform([2, 2, 1])) 
77       [np.str_('tokyo'), np.str_('tokyo'), np.str_('paris')] 
78       """
79   
80       def fit(self, y):
81           """Fit label encoder. 
82    
83           Parameters 
84           ---------- 
85           y : array-like of shape (n_samples,) 
86               Target values. 
87    
88           Returns 
89           ------- 
90           self : returns an instance of self. 
91               Fitted label encoder. 
92           """
93           y = column_or_1d(y, warn=True)
94           self.classes_ = _unique(y)
95           return self
96   
97       def fit_transform(self, y):
98           """Fit label encoder and return encoded labels. 
99    
100          Parameters 
101          ---------- 
102          y : array-like of shape (n_samples,) 
103              Target values. 
104   
105          Returns 
106          ------- 
107          y : array-like of shape (n_samples,) 
108              Encoded labels. 
109          """
110          y = column_or_1d(y, warn=True)
111          self.classes_, y = _unique(y, return_inverse=True)
112          return y
113  
114      def transform(self, y):
115          """Transform labels to normalized encoding. 
116   
117          Parameters 
118          ---------- 
119          y : array-like of shape (n_samples,) 
120              Target values. 
121   
122          Returns 
123          ------- 
124          y : array-like of shape (n_samples,) 
125              Labels as normalized encodings. 
126          """
127          check_is_fitted(self)
128          xp, _ = get_namespace(y)
129          y = column_or_1d(y, dtype=self.classes_.dtype, warn=True)
130          # transform of empty array is empty array
131          if _num_samples(y) == 0:
132              return xp.asarray([])
133  
134          return _encode(y, uniques=self.classes_)
135  
136      def inverse_transform(self, y):
137          """Transform labels back to original encoding. 
138   
139          Parameters 
140          ---------- 
141          y : array-like of shape (n_samples,) 
142              Target values. 
143   
144          Returns 
145          ------- 
146          y : ndarray of shape (n_samples,) 
147              Original encoding. 
148          """
149          check_is_fitted(self)
150          xp, _ = get_namespace(y)
151          y = column_or_1d(y, warn=True)
152          # inverse transform of empty array is empty array
153          if _num_samples(y) == 0:
154              return xp.asarray([])
155  
156          diff = _setdiff1d(
157              ar1=y,
158              ar2=xp.arange(self.classes_.shape[0], device=device(y)),
159              xp=xp,
160          )
161          if diff.shape[0]:
162              raise ValueError("y contains previously unseen labels: %s" % str(diff))
163          y = xp.asarray(y)
164          return xp.take(self.classes_, y, axis=0)
165  
166      def __sklearn_tags__(self):
167          tags = super().__sklearn_tags__()
168          tags.array_api_support = True
169          tags.input_tags.two_d_array = False
170          tags.target_tags.one_d_labels = True
171          return tags

174  class LabelBinarizer(TransformerMixin, BaseEstimator, auto_wrap_output_keys=None):
175      """Binarize labels in a one-vs-all fashion. 
176   
177      Several regression and binary classification algorithms are 
178      available in scikit-learn. A simple way to extend these algorithms 
179      to the multi-class classification case is to use the so-called 
180      one-vs-all scheme. 
181   
182      At learning time, this simply consists in learning one regressor 
183      or binary classifier per class. In doing so, one needs to convert 
184      multi-class labels to binary labels (belong or does not belong 
185      to the class). `LabelBinarizer` makes this process easy with the 
186      transform method. 
187   
188      At prediction time, one assigns the class for which the corresponding 
189      model gave the greatest confidence. `LabelBinarizer` makes this easy 
190      with the :meth:`inverse_transform` method. 
191   
192      Read more in the :ref:`User Guide <preprocessing_targets>`. 
193   
194      Parameters 
195      ---------- 
196      neg_label : int, default=0 
197          Value with which negative labels must be encoded. 
198   
199      pos_label : int, default=1 
200          Value with which positive labels must be encoded. 
201   
202      sparse_output : bool, default=False 
203          True if the returned array from transform is desired to be in sparse 
204          CSR format. 
205   
206      Attributes 
207      ---------- 
208      classes_ : ndarray of shape (n_classes,) 
209          Holds the label for each class. 
210   
211      y_type_ : str 
212          Represents the type of the target data as evaluated by 
213          :func:`~sklearn.utils.multiclass.type_of_target`. Possible type are 
214          'continuous', 'continuous-multioutput', 'binary', 'multiclass', 
215          'multiclass-multioutput', 'multilabel-indicator', and 'unknown'. 
216   
217      sparse_input_ : bool 
218          `True` if the input data to transform is given as a sparse matrix, 
219           `False` otherwise. 
220   
221      See Also 
222      -------- 
223      label_binarize : Function to perform the transform operation of 
224          LabelBinarizer with fixed classes. 
225      OneHotEncoder : Encode categorical features using a one-hot aka one-of-K 
226          scheme. 
227   
228      Examples 
229      -------- 
230      >>> from sklearn.preprocessing import LabelBinarizer 
231      >>> lb = LabelBinarizer() 
232      >>> lb.fit([1, 2, 6, 4, 2]) 
233      LabelBinarizer() 
234      >>> lb.classes_ 
235      array([1, 2, 4, 6]) 
236      >>> lb.transform([1, 6]) 
237      array([[1, 0, 0, 0], 
238             [0, 0, 0, 1]]) 
239   
240      Binary targets transform to a column vector 
241   
242      >>> lb = LabelBinarizer() 
243      >>> lb.fit_transform(['yes', 'no', 'no', 'yes']) 
244      array([[1], 
245             [0], 
246             [0], 
247             [1]]) 
248   
249      Passing a 2D matrix for multilabel classification 
250   
251      >>> import numpy as np 
252      >>> lb.fit(np.array([[0, 1, 1], [1, 0, 0]])) 
253      LabelBinarizer() 
254      >>> lb.classes_ 
255      array([0, 1, 2]) 
256      >>> lb.transform([0, 1, 2, 1]) 
257      array([[1, 0, 0], 
258             [0, 1, 0], 
259             [0, 0, 1], 
260             [0, 1, 0]]) 
261      """
262  
263      _parameter_constraints: dict = {
264          "neg_label": [Integral],
265          "pos_label": [Integral],
266          "sparse_output": ["boolean"],
267      }
268  
269      def __init__(self, *, neg_label=0, pos_label=1, sparse_output=False):
270          self.neg_label = neg_label
271          self.pos_label = pos_label
272          self.sparse_output = sparse_output
273  
274      @_fit_context(prefer_skip_nested_validation=True)
275      def fit(self, y):
276          """Fit label binarizer. 
277   
278          Parameters 
279          ---------- 
280          y : ndarray of shape (n_samples,) or (n_samples, n_classes) 
281              Target values. The 2-d matrix should only contain 0 and 1, 
282              represents multilabel classification. 
283   
284          Returns 
285          ------- 
286          self : object 
287              Returns the instance itself. 
288          """
289          if self.neg_label >= self.pos_label:
290              raise ValueError(
291                  f"neg_label={self.neg_label} must be strictly less than "
292                  f"pos_label={self.pos_label}."
293              )
294  
295          if self.sparse_output and (self.pos_label == 0 or self.neg_label != 0):
296              raise ValueError(
297                  "Sparse binarization is only supported with non "
298                  "zero pos_label and zero neg_label, got "
299                  f"pos_label={self.pos_label} and neg_label={self.neg_label}"
300              )
301  
302          self.y_type_ = type_of_target(y, input_name="y")
303  
304          if "multioutput" in self.y_type_:
305              raise ValueError(
306                  "Multioutput target data is not supported with label binarization"
307              )
308          if _num_samples(y) == 0:
309              raise ValueError("y has 0 samples: %r" % y)
310  
311          self.sparse_input_ = sp.issparse(y)
312          self.classes_ = unique_labels(y)
313          return self
314  
315      def fit_transform(self, y):
316          """Fit label binarizer/transform multi-class labels to binary labels. 
317   
318          The output of transform is sometimes referred to as 
319          the 1-of-K coding scheme. 
320   
321          Parameters 
322          ---------- 
323          y : {ndarray, sparse matrix} of shape (n_samples,) or \ 
324                  (n_samples, n_classes) 
325              Target values. The 2-d matrix should only contain 0 and 1, 
326              represents multilabel classification. Sparse matrix can be 
327              CSR, CSC, COO, DOK, or LIL. 
328   
329          Returns 
330          ------- 
331          Y : {ndarray, sparse matrix} of shape (n_samples, n_classes) 
332              Shape will be (n_samples, 1) for binary problems. Sparse matrix 
333              will be of CSR format. 
334          """
335          return self.fit(y).transform(y)
336  
337      def transform(self, y):
338          """Transform multi-class labels to binary labels. 
339   
340          The output of transform is sometimes referred to by some authors as 
341          the 1-of-K coding scheme. 
342   
343          Parameters 
344          ---------- 
345          y : {array, sparse matrix} of shape (n_samples,) or \ 
346                  (n_samples, n_classes) 
347              Target values. The 2-d matrix should only contain 0 and 1, 
348              represents multilabel classification. Sparse matrix can be 
349              CSR, CSC, COO, DOK, or LIL. 
350   
351          Returns 
352          ------- 
353          Y : {ndarray, sparse matrix} of shape (n_samples, n_classes) 
354              Shape will be (n_samples, 1) for binary problems. Sparse matrix 
355              will be of CSR format. 
356          """
357          check_is_fitted(self)
358  
359          y_is_multilabel = type_of_target(y).startswith("multilabel")
360          if y_is_multilabel and not self.y_type_.startswith("multilabel"):
361              raise ValueError("The object was not fitted with multilabel input.")
362  
363          return label_binarize(
364              y,
365              classes=self.classes_,
366              pos_label=self.pos_label,
367              neg_label=self.neg_label,
368              sparse_output=self.sparse_output,
369          )
370  
371      def inverse_transform(self, Y, threshold=None):
372          """Transform binary labels back to multi-class labels. 
373   
374          Parameters 
375          ---------- 
376          Y : {ndarray, sparse matrix} of shape (n_samples, n_classes) 
377              Target values. All sparse matrices are converted to CSR before 
378              inverse transformation. 
379   
380          threshold : float, default=None 
381              Threshold used in the binary and multi-label cases. 
382   
383              Use 0 when ``Y`` contains the output of :term:`decision_function` 
384              (classifier). 
385              Use 0.5 when ``Y`` contains the output of :term:`predict_proba`. 
386   
387              If None, the threshold is assumed to be half way between 
388              neg_label and pos_label. 
389   
390          Returns 
391          ------- 
392          y : {ndarray, sparse matrix} of shape (n_samples,) 
393              Target values. Sparse matrix will be of CSR format. 
394   
395          Notes 
396          ----- 
397          In the case when the binary labels are fractional 
398          (probabilistic), :meth:`inverse_transform` chooses the class with the 
399          greatest value. Typically, this allows to use the output of a 
400          linear model's :term:`decision_function` method directly as the input 
401          of :meth:`inverse_transform`. 
402          """
403          check_is_fitted(self)
404  
405          if threshold is None:
406              threshold = (self.pos_label + self.neg_label) / 2.0
407  
408          if self.y_type_ == "multiclass":
409              y_inv = _inverse_binarize_multiclass(Y, self.classes_)
410          else:
411              y_inv = _inverse_binarize_thresholding(
412                  Y, self.y_type_, self.classes_, threshold
413              )
414  
415          if self.sparse_input_:
416              y_inv = sp.csr_matrix(y_inv)
417          elif sp.issparse(y_inv):
418              y_inv = y_inv.toarray()
419  
420          return y_inv
421

423          tags = super().__sklearn_tags__()
424          tags.input_tags.two_d_array = False
425          tags.target_tags.one_d_labels = True
426          return tags

432          "classes": ["array-like"],
433          "neg_label": [Interval(Integral, None, None, closed="neither")],
434          "pos_label": [Interval(Integral, None, None, closed="neither")],
435          "sparse_output": ["boolean"],

439  def label_binarize(y, *, classes, neg_label=0, pos_label=1, sparse_output=False):
440      """Binarize labels in a one-vs-all fashion. 
441   
442      Several regression and binary classification algorithms are 
443      available in scikit-learn. A simple way to extend these algorithms 
444      to the multi-class classification case is to use the so-called 
445      one-vs-all scheme. 
446   
447      This function makes it possible to compute this transformation for a 
448      fixed set of class labels known ahead of time. 
449   
450      Parameters 
451      ---------- 
452      y : array-like or sparse matrix 
453          Sequence of integer labels or multilabel data to encode. 
454   
455      classes : array-like of shape (n_classes,) 
456          Uniquely holds the label for each class. 
457   
458      neg_label : int, default=0 
459          Value with which negative labels must be encoded. 
460   
461      pos_label : int, default=1 
462          Value with which positive labels must be encoded. 
463   
464      sparse_output : bool, default=False, 
465          Set to true if output binary array is desired in CSR sparse format. 
466   
467      Returns 
468      ------- 
469      Y : {ndarray, sparse matrix} of shape (n_samples, n_classes) 
470          Shape will be (n_samples, 1) for binary problems. Sparse matrix will 
471          be of CSR format. 
472   
473      See Also 
474      -------- 
475      LabelBinarizer : Class used to wrap the functionality of label_binarize and 
476          allow for fitting to classes independently of the transform operation. 
477   
478      Examples 
479      -------- 
480      >>> from sklearn.preprocessing import label_binarize 
481      >>> label_binarize([1, 6], classes=[1, 2, 4, 6]) 
482      array([[1, 0, 0, 0], 
483             [0, 0, 0, 1]]) 
484   
485      The class ordering is preserved: 
486   
487      >>> label_binarize([1, 6], classes=[1, 6, 4, 2]) 
488      array([[1, 0, 0, 0], 
489             [0, 1, 0, 0]]) 
490   
491      Binary targets transform to a column vector 
492   
493      >>> label_binarize(['yes', 'no', 'no', 'yes'], classes=['no', 'yes']) 
494      array([[1], 
495             [0], 
496             [0], 
497             [1]]) 
498      """
499      if not isinstance(y, list):
500          # XXX Workaround that will be removed when list of list format is
501          # dropped
502          y = check_array(
503              y, input_name="y", accept_sparse="csr", ensure_2d=False, dtype=None
504          )
505      else:
506          if _num_samples(y) == 0:
507              raise ValueError("y has 0 samples: %r" % y)
508      if neg_label >= pos_label:
509          raise ValueError(
510              "neg_label={0} must be strictly less than pos_label={1}.".format(
511                  neg_label, pos_label
512              )
513          )
514  
515      if sparse_output and (pos_label == 0 or neg_label != 0):
516          raise ValueError(
517              "Sparse binarization is only supported with non "
518              "zero pos_label and zero neg_label, got "
519              "pos_label={0} and neg_label={1}"
520              "".format(pos_label, neg_label)
521          )
522  
523      # To account for pos_label == 0 in the dense case
524      pos_switch = pos_label == 0
525      if pos_switch:
526          pos_label = -neg_label
527  
528      y_type = type_of_target(y)
529      if "multioutput" in y_type:
530          raise ValueError(
531              "Multioutput target data is not supported with label binarization"
532          )
533      if y_type == "unknown":
534          raise ValueError("The type of target data is not known")
535  
536      n_samples = y.shape[0] if sp.issparse(y) else len(y)
537      n_classes = len(classes)
538      classes = np.asarray(classes)
539  
540      if y_type == "binary":
541          if n_classes == 1:
542              if sparse_output:
543                  return sp.csr_matrix((n_samples, 1), dtype=int)
544              else:
545                  Y = np.zeros((len(y), 1), dtype=int)
546                  Y += neg_label
547                  return Y
548          elif len(classes) >= 3:
549              y_type = "multiclass"
550  
551      sorted_class = np.sort(classes)
552      if y_type == "multilabel-indicator":
553          y_n_classes = y.shape[1] if hasattr(y, "shape") else len(y[0])
554          if classes.size != y_n_classes:
555              raise ValueError(
556                  "classes {0} mismatch with the labels {1} found in the data".format(
557                      classes, unique_labels(y)
558                  )
559              )
560  
561      if y_type in ("binary", "multiclass"):
562          y = column_or_1d(y)
563  
564          # pick out the known labels from y
565          y_in_classes = np.isin(y, classes)
566          y_seen = y[y_in_classes]
567          indices = np.searchsorted(sorted_class, y_seen)
568          indptr = np.hstack((0, np.cumsum(y_in_classes)))
569  
570          data = np.empty_like(indices)
571          data.fill(pos_label)
572          Y = sp.csr_matrix((data, indices, indptr), shape=(n_samples, n_classes))
573      elif y_type == "multilabel-indicator":
574          Y = sp.csr_matrix(y)
575          if pos_label != 1:
576              data = np.empty_like(Y.data)
577              data.fill(pos_label)
578              Y.data = data
579      else:
580          raise ValueError(
581              "%s target data is not supported with label binarization" % y_type
582          )
583  
584      if not sparse_output:
585          Y = Y.toarray()
586          Y = Y.astype(int, copy=False)
587  
588          if neg_label != 0:
589              Y[Y == 0] = neg_label
590  
591          if pos_switch:
592              Y[Y == pos_label] = 0
593      else:
594          Y.data = Y.data.astype(int, copy=False)
595  
596      # preserve label ordering
597      if np.any(classes != sorted_class):
598          indices = np.searchsorted(sorted_class, classes)
599          Y = Y[:, indices]
600  
601      if y_type == "binary":
602          if sparse_output:
603              Y = Y.getcol(-1)
604          else:
605              Y = Y[:, -1].reshape((-1, 1))
606  
607      return Y
608

610  def _inverse_binarize_multiclass(y, classes):
611      """Inverse label binarization transformation for multiclass. 
612   
613      Multiclass uses the maximal score instead of a threshold. 
614      """
615      classes = np.asarray(classes)
616  
617      if sp.issparse(y):
618          # Find the argmax for each row in y where y is a CSR matrix
619  
620          y = y.tocsr()
621          n_samples, n_outputs = y.shape
622          outputs = np.arange(n_outputs)
623          row_max = min_max_axis(y, 1)[1]
624          row_nnz = np.diff(y.indptr)
625  
626          y_data_repeated_max = np.repeat(row_max, row_nnz)
627          # picks out all indices obtaining the maximum per row
628          y_i_all_argmax = np.flatnonzero(y_data_repeated_max == y.data)
629  
630          # For corner case where last row has a max of 0
631          if row_max[-1] == 0:
632              y_i_all_argmax = np.append(y_i_all_argmax, [len(y.data)])
633  
634          # Gets the index of the first argmax in each row from y_i_all_argmax
635          index_first_argmax = np.searchsorted(y_i_all_argmax, y.indptr[:-1])
636          # first argmax of each row
637          y_ind_ext = np.append(y.indices, [0])
638          y_i_argmax = y_ind_ext[y_i_all_argmax[index_first_argmax]]
639          # Handle rows of all 0
640          y_i_argmax[np.where(row_nnz == 0)[0]] = 0
641  
642          # Handles rows with max of 0 that contain negative numbers
643          samples = np.arange(n_samples)[(row_nnz > 0) & (row_max.ravel() == 0)]
644          for i in samples:
645              ind = y.indices[y.indptr[i] : y.indptr[i + 1]]
646              y_i_argmax[i] = classes[np.setdiff1d(outputs, ind)][0]
647  
648          return classes[y_i_argmax]
649      else:
650          return classes.take(y.argmax(axis=1), mode="clip")
651

653  def _inverse_binarize_thresholding(y, output_type, classes, threshold):
654      """Inverse label binarization transformation using thresholding."""
655  
656      if output_type == "binary" and y.ndim == 2 and y.shape[1] > 2:
657          raise ValueError("output_type='binary', but y.shape = {0}".format(y.shape))
658  
659      if output_type != "binary" and y.shape[1] != len(classes):
660          raise ValueError(
661              "The number of class is not equal to the number of dimension of y."
662          )
663  
664      classes = np.asarray(classes)
665  
666      # Perform thresholding
667      if sp.issparse(y):
668          if threshold > 0:
669              if y.format not in ("csr", "csc"):
670                  y = y.tocsr()
671              y.data = np.array(y.data > threshold, dtype=int)
672              y.eliminate_zeros()
673          else:
674              y = np.array(y.toarray() > threshold, dtype=int)
675      else:
676          y = np.array(y > threshold, dtype=int)
677  
678      # Inverse transform data
679      if output_type == "binary":
680          if sp.issparse(y):
681              y = y.toarray()
682          if y.ndim == 2 and y.shape[1] == 2:
683              return classes[y[:, 1]]
684          else:
685              if len(classes) == 1:
686                  return np.repeat(classes[0], len(y))
687              else:
688                  return classes[y.ravel()]
689  
690      elif output_type == "multilabel-indicator":
691          return y
692  
693      else:
694          raise ValueError("{0} format is not supported".format(output_type))

697  class MultiLabelBinarizer(TransformerMixin, BaseEstimator, auto_wrap_output_keys=None):
698      """Transform between iterable of iterables and a multilabel format. 
699   
700      Although a list of sets or tuples is a very intuitive format for multilabel 
701      data, it is unwieldy to process. This transformer converts between this 
702      intuitive format and the supported multilabel format: a (samples x classes) 
703      binary matrix indicating the presence of a class label. 
704   
705      Parameters 
706      ---------- 
707      classes : array-like of shape (n_classes,), default=None 
708          Indicates an ordering for the class labels. 
709          All entries should be unique (cannot contain duplicate classes). 
710   
711      sparse_output : bool, default=False 
712          Set to True if output binary array is desired in CSR sparse format. 
713   
714      Attributes 
715      ---------- 
716      classes_ : ndarray of shape (n_classes,) 
717          A copy of the `classes` parameter when provided. 
718          Otherwise it corresponds to the sorted set of classes found 
719          when fitting. 
720   
721      See Also 
722      -------- 
723      OneHotEncoder : Encode categorical features using a one-hot aka one-of-K 
724          scheme. 
725   
726      Examples 
727      -------- 
728      >>> from sklearn.preprocessing import MultiLabelBinarizer 
729      >>> mlb = MultiLabelBinarizer() 
730      >>> mlb.fit_transform([(1, 2), (3,)]) 
731      array([[1, 1, 0], 
732             [0, 0, 1]]) 
733      >>> mlb.classes_ 
734      array([1, 2, 3]) 
735   
736      >>> mlb.fit_transform([{'sci-fi', 'thriller'}, {'comedy'}]) 
737      array([[0, 1, 1], 
738             [1, 0, 0]]) 
739      >>> list(mlb.classes_) 
740      ['comedy', 'sci-fi', 'thriller'] 
741   
742      A common mistake is to pass in a list, which leads to the following issue: 
743   
744      >>> mlb = MultiLabelBinarizer() 
745      >>> mlb.fit(['sci-fi', 'thriller', 'comedy']) 
746      MultiLabelBinarizer() 
747      >>> mlb.classes_ 
748      array(['-', 'c', 'd', 'e', 'f', 'h', 'i', 'l', 'm', 'o', 'r', 's', 't', 
749          'y'], dtype=object) 
750   
751      To correct this, the list of labels should be passed in as: 
752   
753      >>> mlb = MultiLabelBinarizer() 
754      >>> mlb.fit([['sci-fi', 'thriller', 'comedy']]) 
755      MultiLabelBinarizer() 
756      >>> mlb.classes_ 
757      array(['comedy', 'sci-fi', 'thriller'], dtype=object) 
758      """
759  
760      _parameter_constraints: dict = {
761          "classes": ["array-like", None],
762          "sparse_output": ["boolean"],
763      }
764  
765      def __init__(self, *, classes=None, sparse_output=False):
766          self.classes = classes
767          self.sparse_output = sparse_output
768  
769      @_fit_context(prefer_skip_nested_validation=True)
770      def fit(self, y):
771          """Fit the label sets binarizer, storing :term:`classes_`. 
772   
773          Parameters 
774          ---------- 
775          y : iterable of iterables 
776              A set of labels (any orderable and hashable object) for each 
777              sample. If the `classes` parameter is set, `y` will not be 
778              iterated. 
779   
780          Returns 
781          ------- 
782          self : object 
783              Fitted estimator. 
784          """
785          self._cached_dict = None
786  
787          if self.classes is None:
788              classes = sorted(set(itertools.chain.from_iterable(y)))
789          elif len(set(self.classes)) < len(self.classes):
790              raise ValueError(
791                  "The classes argument contains duplicate "
792                  "classes. Remove these duplicates before passing "
793                  "them to MultiLabelBinarizer."
794              )
795          else:
796              classes = self.classes
797          dtype = int if all(isinstance(c, int) for c in classes) else object
798          self.classes_ = np.empty(len(classes), dtype=dtype)
799          self.classes_[:] = classes
800          return self
801  
802      @_fit_context(prefer_skip_nested_validation=True)
803      def fit_transform(self, y):
804          """Fit the label sets binarizer and transform the given label sets. 
805   
806          Parameters 
807          ---------- 
808          y : iterable of iterables 
809              A set of labels (any orderable and hashable object) for each 
810              sample. If the `classes` parameter is set, `y` will not be 
811              iterated. 
812   
813          Returns 
814          ------- 
815          y_indicator : {ndarray, sparse matrix} of shape (n_samples, n_classes) 
816              A matrix such that `y_indicator[i, j] = 1` iff `classes_[j]` 
817              is in `y[i]`, and 0 otherwise. Sparse matrix will be of CSR 
818              format. 
819          """
820          if self.classes is not None:
821              return self.fit(y).transform(y)
822  
823          self._cached_dict = None
824  
825          # Automatically increment on new class
826          class_mapping = defaultdict(int)
827          class_mapping.default_factory = class_mapping.__len__
828          yt = self._transform(y, class_mapping)
829  
830          # sort classes and reorder columns
831          tmp = sorted(class_mapping, key=class_mapping.get)
832  
833          # (make safe for tuples)
834          dtype = int if all(isinstance(c, int) for c in tmp) else object
835          class_mapping = np.empty(len(tmp), dtype=dtype)
836          class_mapping[:] = tmp
837          self.classes_, inverse = np.unique(class_mapping, return_inverse=True)
838          # ensure yt.indices keeps its current dtype
839          yt.indices = np.asarray(inverse[yt.indices], dtype=yt.indices.dtype)
840  
841          if not self.sparse_output:
842              yt = yt.toarray()
843  
844          return yt
845  
846      def transform(self, y):
847          """Transform the given label sets. 
848   
849          Parameters 
850          ---------- 
851          y : iterable of iterables 
852              A set of labels (any orderable and hashable object) for each 
853              sample. If the `classes` parameter is set, `y` will not be 
854              iterated. 
855   
856          Returns 
857          ------- 
858          y_indicator : array or CSR matrix, shape (n_samples, n_classes) 
859              A matrix such that `y_indicator[i, j] = 1` iff `classes_[j]` is in 
860              `y[i]`, and 0 otherwise. 
861          """
862          check_is_fitted(self)
863  
864          class_to_index = self._build_cache()
865          yt = self._transform(y, class_to_index)
866  
867          if not self.sparse_output:
868              yt = yt.toarray()
869  
870          return yt
871  
872      def _build_cache(self):
873          if self._cached_dict is None:
874              self._cached_dict = dict(zip(self.classes_, range(len(self.classes_))))
875  
876          return self._cached_dict
877  
878      def _transform(self, y, class_mapping):
879          """Transforms the label sets with a given mapping. 
880   
881          Parameters 
882          ---------- 
883          y : iterable of iterables 
884              A set of labels (any orderable and hashable object) for each 
885              sample. If the `classes` parameter is set, `y` will not be 
886              iterated. 
887   
888          class_mapping : Mapping 
889              Maps from label to column index in label indicator matrix. 
890   
891          Returns 
892          ------- 
893          y_indicator : sparse matrix of shape (n_samples, n_classes) 
894              Label indicator matrix. Will be of CSR format. 
895          """
896          indices = array.array("i")
897          indptr = array.array("i", [0])
898          unknown = set()
899          for labels in y:
900              index = set()
901              for label in labels:
902                  try:
903                      index.add(class_mapping[label])
904                  except KeyError:
905                      unknown.add(label)
906              indices.extend(index)
907              indptr.append(len(indices))
908          if unknown:
909              warnings.warn(
910                  "unknown class(es) {0} will be ignored".format(sorted(unknown, key=str))
911              )
912          data = np.ones(len(indices), dtype=int)
913  
914          return sp.csr_matrix(
915              (data, indices, indptr), shape=(len(indptr) - 1, len(class_mapping))
916          )
917  
918      def inverse_transform(self, yt):
919          """Transform the given indicator matrix into label sets. 
920   
921          Parameters 
922          ---------- 
923          yt : {ndarray, sparse matrix} of shape (n_samples, n_classes) 
924              A matrix containing only 1s ands 0s. 
925   
926          Returns 
927          ------- 
928          y : list of tuples 
929              The set of labels for each sample such that `y[i]` consists of 
930              `classes_[j]` for each `yt[i, j] == 1`. 
931          """
932          check_is_fitted(self)
933  
934          if yt.shape[1] != len(self.classes_):
935              raise ValueError(
936                  "Expected indicator for {0} classes, but got {1}".format(
937                      len(self.classes_), yt.shape[1]
938                  )
939              )
940  
941          if sp.issparse(yt):
942              yt = yt.tocsr()
943              if len(yt.data) != 0 and len(np.setdiff1d(yt.data, [0, 1])) > 0:
944                  raise ValueError("Expected only 0s and 1s in label indicator.")
945              return [
946                  tuple(self.classes_.take(yt.indices[start:end]))
947                  for start, end in zip(yt.indptr[:-1], yt.indptr[1:])
948              ]
949          else:
950              unexpected = np.setdiff1d(yt, [0, 1])
951              if len(unexpected) > 0:
952                  raise ValueError(
953                      "Expected only 0s and 1s in label indicator. Also got {0}".format(
954                          unexpected
955                      )
956                  )
957              return [tuple(self.classes_.compress(indicators)) for indicators in yt]
