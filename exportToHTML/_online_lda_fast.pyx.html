10   def mean_change(const floating[:] arr_1, const floating[:] arr_2):
11       """Calculate the mean difference between two arrays. 
12    
13       Equivalent to np.abs(arr_1 - arr2).mean(). 
14       """
15   
16       cdef float64_t total, diff
17       cdef intp_t i, size
18   
19       size = arr_1.shape[0]
20       total = 0.0
21       for i in range(size):
22           diff = fabs(arr_1[i] - arr_2[i])
23           total += diff
24   
25       return total / size

28   def _dirichlet_expectation_1d(
29       floating[:] doc_topic,
30       floating doc_topic_prior,
31       floating[:] out
32   ):
33       """Dirichlet expectation for a single sample: 
34           exp(E[log(theta)]) for theta ~ Dir(doc_topic) 
35       after adding doc_topic_prior to doc_topic, in-place. 
36    
37       Equivalent to 
38           doc_topic += doc_topic_prior 
39           out[:] = np.exp(psi(doc_topic) - psi(np.sum(doc_topic))) 
40       """
41   
42       cdef floating dt, psi_total, total
43       cdef intp_t i, size
44   
45       size = doc_topic.shape[0]
46   
47       total = 0.0
48       for i in range(size):
49           dt = doc_topic[i] + doc_topic_prior
50           doc_topic[i] = dt
51           total += dt
52       psi_total = psi(total)
53   
54       for i in range(size):
55           out[i] = exp(psi(doc_topic[i]) - psi_total)

58   def _dirichlet_expectation_2d(const floating[:, :] arr):
59       """Dirichlet expectation for multiple samples: 
60       E[log(theta)] for theta ~ Dir(arr). 
61    
62       Equivalent to psi(arr) - psi(np.sum(arr, axis=1))[:, np.newaxis]. 
63    
64       Note that unlike _dirichlet_expectation_1d, this function doesn't compute 
65       the exp and doesn't add in the prior. 
66       """
67       cdef floating row_total, psi_row_total
68       cdef floating[:, :] d_exp
69       cdef intp_t i, j, n_rows, n_cols
70   
71       n_rows = arr.shape[0]
72       n_cols = arr.shape[1]
73   
74       d_exp = np.empty_like(arr)
75       for i in range(n_rows):
76           row_total = 0
77           for j in range(n_cols):
78               row_total += arr[i, j]
79           psi_row_total = psi(row_total)
80   
81           for j in range(n_cols):
82               d_exp[i, j] = psi(arr[i, j]) - psi_row_total
83   
84       return d_exp.base

87   # Psi function for positive arguments. Optimized for speed, not accuracy.
88   #
89   # After: J. Bernardo (1976). Algorithm AS 103: Psi (Digamma) Function.
90   # https://www.uv.es/~bernardo/1976AppStatist.pdf
91   cdef floating psi(floating x) noexcept nogil:
92       cdef double EULER = 0.577215664901532860606512090082402431
93       if x <= 1e-6:
94           # psi(x) = -EULER - 1/x + O(x)
95           return -EULER - 1. / x
96   
97       cdef floating r, result = 0
98   
99       # psi(x + 1) = psi(x) + 1/x
100      while x < 6:
101          result -= 1. / x
102          x += 1
103  
104      # psi(x) = log(x) - 1/(2x) - 1/(12x**2) + 1/(120x**4) - 1/(252x**6)
105      #          + O(1/x**8)
106      r = 1. / x
107      result += log(x) - .5 * r
108      r = r * r
109      result -= r * ((1./12.) - r * ((1./120.) - r * (1./252.)))
110      return result
