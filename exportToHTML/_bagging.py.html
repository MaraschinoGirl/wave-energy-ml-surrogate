56   def _generate_indices(random_state, bootstrap, n_population, n_samples):
57       """Draw randomly sampled indices."""
58       # Draw sample indices
59       if bootstrap:
60           indices = random_state.randint(0, n_population, n_samples)
61       else:
62           indices = sample_without_replacement(
63               n_population, n_samples, random_state=random_state
64           )
65   
66       return indices

69   def _generate_bagging_indices(
70       random_state,
71       bootstrap_features,
72       bootstrap_samples,
73       n_features,
74       n_samples,
75       max_features,
76       max_samples,
77   ):
78       """Randomly draw feature and sample indices."""
79       # Get valid random state
80       random_state = check_random_state(random_state)
81   
82       # Draw indices
83       feature_indices = _generate_indices(
84           random_state, bootstrap_features, n_features, max_features
85       )
86       sample_indices = _generate_indices(
87           random_state, bootstrap_samples, n_samples, max_samples
88       )
89   
90       return feature_indices, sample_indices

93   def _parallel_build_estimators(
94       n_estimators,
95       ensemble,
96       X,
97       y,
98       seeds,
99       total_n_estimators,
100      verbose,
101      check_input,
102      fit_params,
103  ):
104      """Private function used to build a batch of estimators within a job."""
105      # Retrieve settings
106      n_samples, n_features = X.shape
107      max_features = ensemble._max_features
108      max_samples = ensemble._max_samples
109      bootstrap = ensemble.bootstrap
110      bootstrap_features = ensemble.bootstrap_features
111      has_check_input = has_fit_parameter(ensemble.estimator_, "check_input")
112      requires_feature_indexing = bootstrap_features or max_features != n_features
113  
114      # Build estimators
115      estimators = []
116      estimators_features = []
117  
118      # TODO: (slep6) remove if condition for unrouted sample_weight when metadata
119      # routing can't be disabled.
120      support_sample_weight = has_fit_parameter(ensemble.estimator_, "sample_weight")
121      if not _routing_enabled() and (
122          not support_sample_weight and fit_params.get("sample_weight") is not None
123      ):
124          raise ValueError(
125              "The base estimator doesn't support sample weight, but sample_weight is "
126              "passed to the fit method."
127          )
128  
129      for i in range(n_estimators):
130          if verbose > 1:
131              print(
132                  "Building estimator %d of %d for this parallel run (total %d)..."
133                  % (i + 1, n_estimators, total_n_estimators)
134              )
135  
136          random_state = seeds[i]
137          estimator = ensemble._make_estimator(append=False, random_state=random_state)
138  
139          if has_check_input:
140              estimator_fit = partial(estimator.fit, check_input=check_input)
141          else:
142              estimator_fit = estimator.fit
143  
144          # Draw random feature, sample indices
145          features, indices = _generate_bagging_indices(
146              random_state,
147              bootstrap_features,
148              bootstrap,
149              n_features,
150              n_samples,
151              max_features,
152              max_samples,
153          )
154  
155          fit_params_ = fit_params.copy()
156  
157          # TODO(SLEP6): remove if condition for unrouted sample_weight when metadata
158          # routing can't be disabled.
159          # 1. If routing is enabled, we will check if the routing supports sample
160          # weight and use it if it does.
161          # 2. If routing is not enabled, we will check if the base
162          # estimator supports sample_weight and use it if it does.
163  
164          # Note: Row sampling can be achieved either through setting sample_weight or
165          # by indexing. The former is more efficient. Therefore, use this method
166          # if possible, otherwise use indexing.
167          if _routing_enabled():
168              request_or_router = get_routing_for_object(ensemble.estimator_)
169              consumes_sample_weight = request_or_router.consumes(
170                  "fit", ("sample_weight",)
171              )
172          else:
173              consumes_sample_weight = support_sample_weight
174          if consumes_sample_weight:
175              # Draw sub samples, using sample weights, and then fit
176              curr_sample_weight = _check_sample_weight(
177                  fit_params_.pop("sample_weight", None), X
178              ).copy()
179  
180              if bootstrap:
181                  sample_counts = np.bincount(indices, minlength=n_samples)
182                  curr_sample_weight *= sample_counts
183              else:
184                  not_indices_mask = ~indices_to_mask(indices, n_samples)
185                  curr_sample_weight[not_indices_mask] = 0
186  
187              fit_params_["sample_weight"] = curr_sample_weight
188              X_ = X[:, features] if requires_feature_indexing else X
189              estimator_fit(X_, y, **fit_params_)
190          else:
191              # cannot use sample_weight, so use indexing
192              y_ = _safe_indexing(y, indices)
193              X_ = _safe_indexing(X, indices)
194              fit_params_ = _check_method_params(X, params=fit_params_, indices=indices)
195              if requires_feature_indexing:
196                  X_ = X_[:, features]
197              estimator_fit(X_, y_, **fit_params_)
198  
199          estimators.append(estimator)
200          estimators_features.append(features)
201  
202      return estimators, estimators_features

205  def _parallel_predict_proba(estimators, estimators_features, X, n_classes):
206      """Private function used to compute (proba-)predictions within a job."""
207      n_samples = X.shape[0]
208      proba = np.zeros((n_samples, n_classes))
209  
210      for estimator, features in zip(estimators, estimators_features):
211          if hasattr(estimator, "predict_proba"):
212              proba_estimator = estimator.predict_proba(X[:, features])
213  
214              if n_classes == len(estimator.classes_):
215                  proba += proba_estimator
216  
217              else:
218                  proba[:, estimator.classes_] += proba_estimator[
219                      :, range(len(estimator.classes_))
220                  ]
221  
222          else:
223              # Resort to voting
224              predictions = estimator.predict(X[:, features])
225  
226              for i in range(n_samples):
227                  proba[i, predictions[i]] += 1
228  
229      return proba

232  def _parallel_predict_log_proba(estimators, estimators_features, X, n_classes):
233      """Private function used to compute log probabilities within a job."""
234      n_samples = X.shape[0]
235      log_proba = np.empty((n_samples, n_classes))
236      log_proba.fill(-np.inf)
237      all_classes = np.arange(n_classes, dtype=int)
238  
239      for estimator, features in zip(estimators, estimators_features):
240          log_proba_estimator = estimator.predict_log_proba(X[:, features])
241  
242          if n_classes == len(estimator.classes_):
243              log_proba = np.logaddexp(log_proba, log_proba_estimator)
244  
245          else:
246              log_proba[:, estimator.classes_] = np.logaddexp(
247                  log_proba[:, estimator.classes_],
248                  log_proba_estimator[:, range(len(estimator.classes_))],
249              )
250  
251              missing = np.setdiff1d(all_classes, estimator.classes_)
252              log_proba[:, missing] = np.logaddexp(log_proba[:, missing], -np.inf)
253  
254      return log_proba

257  def _parallel_decision_function(estimators, estimators_features, X):
258      """Private function used to compute decisions within a job."""
259      return sum(
260          estimator.decision_function(X[:, features])
261          for estimator, features in zip(estimators, estimators_features)
262      )

265  def _parallel_predict_regression(estimators, estimators_features, X):
266      """Private function used to compute predictions within a job."""
267      return sum(
268          estimator.predict(X[:, features])
269          for estimator, features in zip(estimators, estimators_features)

273  class BaseBagging(BaseEnsemble, metaclass=ABCMeta):
274      """Base class for Bagging meta-estimator. 
275   
276      Warning: This class should not be used directly. Use derived classes 
277      instead. 
278      """
279  
280      _parameter_constraints: dict = {
281          "estimator": [HasMethods(["fit", "predict"]), None],
282          "n_estimators": [Interval(Integral, 1, None, closed="left")],
283          "max_samples": [
284              Interval(Integral, 1, None, closed="left"),
285              Interval(RealNotInt, 0, 1, closed="right"),
286          ],
287          "max_features": [
288              Interval(Integral, 1, None, closed="left"),
289              Interval(RealNotInt, 0, 1, closed="right"),
290          ],
291          "bootstrap": ["boolean"],
292          "bootstrap_features": ["boolean"],
293          "oob_score": ["boolean"],
294          "warm_start": ["boolean"],
295          "n_jobs": [None, Integral],
296          "random_state": ["random_state"],
297          "verbose": ["verbose"],
298      }

300      @abstractmethod
301      def __init__(
302          self,
303          estimator=None,
304          n_estimators=10,
305          *,
306          max_samples=1.0,
307          max_features=1.0,
308          bootstrap=True,
309          bootstrap_features=False,
310          oob_score=False,
311          warm_start=False,
312          n_jobs=None,
313          random_state=None,
314          verbose=0,
315      ):
316          super().__init__(
317              estimator=estimator,
318              n_estimators=n_estimators,
319          )
320          self.max_samples = max_samples
321          self.max_features = max_features
322          self.bootstrap = bootstrap
323          self.bootstrap_features = bootstrap_features
324          self.oob_score = oob_score
325          self.warm_start = warm_start
326          self.n_jobs = n_jobs
327          self.random_state = random_state
328          self.verbose = verbose
329  
330      # TODO(1.7): remove `sample_weight` from the signature after deprecation
331      # cycle; pop it from `fit_params` before the `_raise_for_params` check and
332      # reinsert later, for backwards compatibility
333      @_deprecate_positional_args(version="1.7")
334      @_fit_context(
335          # BaseBagging.estimator is not validated yet
336          prefer_skip_nested_validation=False
337      )
338      def fit(self, X, y, *, sample_weight=None, **fit_params):
339          """Build a Bagging ensemble of estimators from the training set (X, y). 
340   
341          Parameters 
342          ---------- 
343          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
344              The training input samples. Sparse matrices are accepted only if 
345              they are supported by the base estimator. 
346   
347          y : array-like of shape (n_samples,) 
348              The target values (class labels in classification, real numbers in 
349              regression). 
350   
351          sample_weight : array-like of shape (n_samples,), default=None 
352              Sample weights. If None, then samples are equally weighted. 
353              Note that this is supported only if the base estimator supports 
354              sample weighting. 
355   
356          **fit_params : dict 
357              Parameters to pass to the underlying estimators. 
358   
359              .. versionadded:: 1.5 
360   
361                  Only available if `enable_metadata_routing=True`, 
362                  which can be set by using 
363                  ``sklearn.set_config(enable_metadata_routing=True)``. 
364                  See :ref:`Metadata Routing User Guide <metadata_routing>` for 
365                  more details. 
366   
367          Returns 
368          ------- 
369          self : object 
370              Fitted estimator. 
371          """
372          _raise_for_params(fit_params, self, "fit")
373  
374          # Convert data (X is required to be 2d and indexable)
375          X, y = validate_data(
376              self,
377              X,
378              y,
379              accept_sparse=["csr", "csc"],
380              dtype=None,
381              ensure_all_finite=False,
382              multi_output=True,
383          )
384  
385          if sample_weight is not None:
386              sample_weight = _check_sample_weight(sample_weight, X, dtype=None)
387              fit_params["sample_weight"] = sample_weight
388  
389          return self._fit(X, y, max_samples=self.max_samples, **fit_params)

392          return {}

394      def _fit(
395          self,
396          X,
397          y,
398          max_samples=None,
399          max_depth=None,
400          check_input=True,
401          **fit_params,
402      ):
403          """Build a Bagging ensemble of estimators from the training 
404             set (X, y). 
405   
406          Parameters 
407          ---------- 
408          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
409              The training input samples. Sparse matrices are accepted only if 
410              they are supported by the base estimator. 
411   
412          y : array-like of shape (n_samples,) 
413              The target values (class labels in classification, real numbers in 
414              regression). 
415   
416          max_samples : int or float, default=None 
417              Argument to use instead of self.max_samples. 
418   
419          max_depth : int, default=None 
420              Override value used when constructing base estimator. Only 
421              supported if the base estimator has a max_depth parameter. 
422   
423          check_input : bool, default=True 
424              Override value used when fitting base estimator. Only supported 
425              if the base estimator has a check_input parameter for fit function. 
426              If the meta-estimator already checks the input, set this value to 
427              False to prevent redundant input validation. 
428   
429          **fit_params : dict, default=None 
430              Parameters to pass to the :term:`fit` method of the underlying 
431              estimator. 
432   
433          Returns 
434          ------- 
435          self : object 
436              Fitted estimator. 
437          """
438          random_state = check_random_state(self.random_state)
439  
440          # Remap output
441          n_samples = X.shape[0]
442          self._n_samples = n_samples
443          y = self._validate_y(y)
444  
445          # Check parameters
446          self._validate_estimator(self._get_estimator())
447  
448          if _routing_enabled():
449              routed_params = process_routing(self, "fit", **fit_params)
450          else:
451              routed_params = Bunch()
452              routed_params.estimator = Bunch(fit=fit_params)
453              if "sample_weight" in fit_params:
454                  routed_params.estimator.fit["sample_weight"] = fit_params[
455                      "sample_weight"
456                  ]
457  
458          if max_depth is not None:
459              self.estimator_.max_depth = max_depth
460  
461          # Validate max_samples
462          if max_samples is None:
463              max_samples = self.max_samples
464          elif not isinstance(max_samples, numbers.Integral):
465              max_samples = int(max_samples * X.shape[0])
466  
467          if max_samples > X.shape[0]:
468              raise ValueError("max_samples must be <= n_samples")
469  
470          # Store validated integer row sampling value
471          self._max_samples = max_samples
472  
473          # Validate max_features
474          if isinstance(self.max_features, numbers.Integral):
475              max_features = self.max_features
476          elif isinstance(self.max_features, float):
477              max_features = int(self.max_features * self.n_features_in_)
478  
479          if max_features > self.n_features_in_:
480              raise ValueError("max_features must be <= n_features")
481  
482          max_features = max(1, int(max_features))
483  
484          # Store validated integer feature sampling value
485          self._max_features = max_features
486  
487          # Other checks
488          if not self.bootstrap and self.oob_score:
489              raise ValueError("Out of bag estimation only available if bootstrap=True")
490  
491          if self.warm_start and self.oob_score:
492              raise ValueError("Out of bag estimate only available if warm_start=False")
493  
494          if hasattr(self, "oob_score_") and self.warm_start:
495              del self.oob_score_
496  
497          if not self.warm_start or not hasattr(self, "estimators_"):
498              # Free allocated memory, if any
499              self.estimators_ = []
500              self.estimators_features_ = []
501  
502          n_more_estimators = self.n_estimators - len(self.estimators_)
503  
504          if n_more_estimators < 0:
505              raise ValueError(
506                  "n_estimators=%d must be larger or equal to "
507                  "len(estimators_)=%d when warm_start==True"
508                  % (self.n_estimators, len(self.estimators_))
509              )
510  
511          elif n_more_estimators == 0:
512              warn(
513                  "Warm-start fitting without increasing n_estimators does not "
514                  "fit new trees."
515              )
516              return self
517  
518          # Parallel loop
519          n_jobs, n_estimators, starts = _partition_estimators(
520              n_more_estimators, self.n_jobs
521          )
522          total_n_estimators = sum(n_estimators)
523  
524          # Advance random state to state after training
525          # the first n_estimators
526          if self.warm_start and len(self.estimators_) > 0:
527              random_state.randint(MAX_INT, size=len(self.estimators_))
528  
529          seeds = random_state.randint(MAX_INT, size=n_more_estimators)
530          self._seeds = seeds
531  
532          all_results = Parallel(
533              n_jobs=n_jobs, verbose=self.verbose, **self._parallel_args()
534          )(
535              delayed(_parallel_build_estimators)(
536                  n_estimators[i],
537                  self,
538                  X,
539                  y,
540                  seeds[starts[i] : starts[i + 1]],
541                  total_n_estimators,
542                  verbose=self.verbose,
543                  check_input=check_input,
544                  fit_params=routed_params.estimator.fit,
545              )
546              for i in range(n_jobs)
547          )
548  
549          # Reduce
550          self.estimators_ += list(
551              itertools.chain.from_iterable(t[0] for t in all_results)
552          )
553          self.estimators_features_ += list(
554              itertools.chain.from_iterable(t[1] for t in all_results)
555          )
556  
557          if self.oob_score:
558              self._set_oob_score(X, y)
559  
560          return self
561

571      def _get_estimators_indices(self):
572          # Get drawn indices along both sample and feature axes
573          for seed in self._seeds:
574              # Operations accessing random_state must be performed identically
575              # to those in `_parallel_build_estimators()`
576              feature_indices, sample_indices = _generate_bagging_indices(
577                  seed,
578                  self.bootstrap_features,
579                  self.bootstrap,
580                  self.n_features_in_,
581                  self._n_samples,
582                  self._max_features,
583                  self._max_samples,
584              )
585  
586              yield feature_indices, sample_indices

588      @property
589      def estimators_samples_(self):
590          """ 
591          The subset of drawn samples for each base estimator. 
592   
593          Returns a dynamically generated list of indices identifying 
594          the samples used for fitting each member of the ensemble, i.e., 
595          the in-bag samples. 
596   
597          Note: the list is re-created at each call to the property in order 
598          to reduce the object memory footprint by not storing the sampling 
599          data. Thus fetching the property may be slower than expected. 
600          """
601          return [sample_indices for _, sample_indices in self._get_estimators_indices()]

635  class BaggingClassifier(ClassifierMixin, BaseBagging):
636      """A Bagging classifier. 
637   
638      A Bagging classifier is an ensemble meta-estimator that fits base 
639      classifiers each on random subsets of the original dataset and then 
640      aggregate their individual predictions (either by voting or by averaging) 
641      to form a final prediction. Such a meta-estimator can typically be used as 
642      a way to reduce the variance of a black-box estimator (e.g., a decision 
643      tree), by introducing randomization into its construction procedure and 
644      then making an ensemble out of it. 
645   
646      This algorithm encompasses several works from the literature. When random 
647      subsets of the dataset are drawn as random subsets of the samples, then 
648      this algorithm is known as Pasting [1]_. If samples are drawn with 
649      replacement, then the method is known as Bagging [2]_. When random subsets 
650      of the dataset are drawn as random subsets of the features, then the method 
651      is known as Random Subspaces [3]_. Finally, when base estimators are built 
652      on subsets of both samples and features, then the method is known as 
653      Random Patches [4]_. 
654   
655      Read more in the :ref:`User Guide <bagging>`. 
656   
657      .. versionadded:: 0.15 
658   
659      Parameters 
660      ---------- 
661      estimator : object, default=None 
662          The base estimator to fit on random subsets of the dataset. 
663          If None, then the base estimator is a 
664          :class:`~sklearn.tree.DecisionTreeClassifier`. 
665   
666          .. versionadded:: 1.2 
667             `base_estimator` was renamed to `estimator`. 
668   
669      n_estimators : int, default=10 
670          The number of base estimators in the ensemble. 
671   
672      max_samples : int or float, default=1.0 
673          The number of samples to draw from X to train each base estimator (with 
674          replacement by default, see `bootstrap` for more details). 
675   
676          - If int, then draw `max_samples` samples. 
677          - If float, then draw `max_samples * X.shape[0]` samples. 
678   
679      max_features : int or float, default=1.0 
680          The number of features to draw from X to train each base estimator ( 
681          without replacement by default, see `bootstrap_features` for more 
682          details). 
683   
684          - If int, then draw `max_features` features. 
685          - If float, then draw `max(1, int(max_features * n_features_in_))` features. 
686   
687      bootstrap : bool, default=True 
688          Whether samples are drawn with replacement. If False, sampling 
689          without replacement is performed. 
690   
691      bootstrap_features : bool, default=False 
692          Whether features are drawn with replacement. 
693   
694      oob_score : bool, default=False 
695          Whether to use out-of-bag samples to estimate 
696          the generalization error. Only available if bootstrap=True. 
697

838      def _set_oob_score(self, X, y):
839          n_samples = y.shape[0]
840          n_classes_ = self.n_classes_
841  
842          predictions = np.zeros((n_samples, n_classes_))
843  
844          for estimator, samples, features in zip(
845              self.estimators_, self.estimators_samples_, self.estimators_features_
846          ):
847              # Create mask for OOB samples
848              mask = ~indices_to_mask(samples, n_samples)
849  
850              if hasattr(estimator, "predict_proba"):
851                  predictions[mask, :] += estimator.predict_proba(
852                      (X[mask, :])[:, features]
853                  )
854  
855              else:
856                  p = estimator.predict((X[mask, :])[:, features])
857                  j = 0
858  
859                  for i in range(n_samples):
860                      if mask[i]:
861                          predictions[i, p[j]] += 1
862                          j += 1
863  
864          if (predictions.sum(axis=1) == 0).any():
865              warn(
866                  "Some inputs do not have OOB scores. "
867                  "This probably means too few estimators were used "
868                  "to compute any reliable oob estimates."
869              )
870  
871          oob_decision_function = predictions / predictions.sum(axis=1)[:, np.newaxis]
872          oob_score = accuracy_score(y, np.argmax(predictions, axis=1))
873  
874          self.oob_decision_function_ = oob_decision_function
875          self.oob_score_ = oob_score

885      def predict(self, X):
886          """Predict class for X. 
887   
888          The predicted class of an input sample is computed as the class with 
889          the highest mean predicted probability. If base estimators do not 
890          implement a ``predict_proba`` method, then it resorts to voting. 
891   
892          Parameters 
893          ---------- 
894          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
895              The training input samples. Sparse matrices are accepted only if 
896              they are supported by the base estimator. 
897   
898          Returns 
899          ------- 
900          y : ndarray of shape (n_samples,) 
901              The predicted classes. 
902          """
903          predicted_probabilitiy = self.predict_proba(X)
904          return self.classes_.take((np.argmax(predicted_probabilitiy, axis=1)), axis=0)

906      def predict_proba(self, X):
907          """Predict class probabilities for X. 
908   
909          The predicted class probabilities of an input sample is computed as 
910          the mean predicted class probabilities of the base estimators in the 
911          ensemble. If base estimators do not implement a ``predict_proba`` 
912          method, then it resorts to voting and the predicted class probabilities 
913          of an input sample represents the proportion of estimators predicting 
914          each class. 
915   
916          Parameters 
917          ---------- 
918          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
919              The training input samples. Sparse matrices are accepted only if 
920              they are supported by the base estimator. 
921   
922          Returns 
923          ------- 
924          p : ndarray of shape (n_samples, n_classes) 
925              The class probabilities of the input samples. The order of the 
926              classes corresponds to that in the attribute :term:`classes_`. 
927          """
928          check_is_fitted(self)
929          # Check data
930          X = validate_data(
931              self,
932              X,
933              accept_sparse=["csr", "csc"],
934              dtype=None,
935              ensure_all_finite=False,
936              reset=False,
937          )
938  
939          # Parallel loop
940          n_jobs, _, starts = _partition_estimators(self.n_estimators, self.n_jobs)
941  
942          all_proba = Parallel(
943              n_jobs=n_jobs, verbose=self.verbose, **self._parallel_args()
944          )(
945              delayed(_parallel_predict_proba)(
946                  self.estimators_[starts[i] : starts[i + 1]],
947                  self.estimators_features_[starts[i] : starts[i + 1]],
948                  X,
949                  self.n_classes_,
950              )
951              for i in range(n_jobs)
952          )
953  
954          # Reduce
955          proba = sum(all_proba) / self.n_estimators
956  
957          return proba

959      def predict_log_proba(self, X):
960          """Predict class log-probabilities for X. 
961   
962          The predicted class log-probabilities of an input sample is computed as 
963          the log of the mean predicted class probabilities of the base 
964          estimators in the ensemble. 
965   
966          Parameters 
967          ---------- 
968          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
969              The training input samples. Sparse matrices are accepted only if 
970              they are supported by the base estimator. 
971   
972          Returns 
973          ------- 
974          p : ndarray of shape (n_samples, n_classes) 
975              The class log-probabilities of the input samples. The order of the 
976              classes corresponds to that in the attribute :term:`classes_`. 
977          """
978          check_is_fitted(self)
979          if hasattr(self.estimator_, "predict_log_proba"):
980              # Check data
981              X = validate_data(
982                  self,
983                  X,
984                  accept_sparse=["csr", "csc"],
985                  dtype=None,
986                  ensure_all_finite=False,
987                  reset=False,
988              )
989  
990              # Parallel loop
991              n_jobs, _, starts = _partition_estimators(self.n_estimators, self.n_jobs)
992  
993              all_log_proba = Parallel(n_jobs=n_jobs, verbose=self.verbose)(
994                  delayed(_parallel_predict_log_proba)(
995                      self.estimators_[starts[i] : starts[i + 1]],
996                      self.estimators_features_[starts[i] : starts[i + 1]],
997                      X,
998                      self.n_classes_,
999                  )
1000                 for i in range(n_jobs)
1001             )
1002 
1003             # Reduce
1004             log_proba = all_log_proba[0]
1005 
1006             for j in range(1, len(all_log_proba)):
1007                 log_proba = np.logaddexp(log_proba, all_log_proba[j])
1008 
1009             log_proba -= np.log(self.n_estimators)
1010 
1011         else:
1012             log_proba = np.log(self.predict_proba(X))
1013 
1014         return log_proba

1066 class BaggingRegressor(RegressorMixin, BaseBagging):
1067     """A Bagging regressor. 
1068  
1069     A Bagging regressor is an ensemble meta-estimator that fits base 
1070     regressors each on random subsets of the original dataset and then 
1071     aggregate their individual predictions (either by voting or by averaging) 
1072     to form a final prediction. Such a meta-estimator can typically be used as 
1073     a way to reduce the variance of a black-box estimator (e.g., a decision 
1074     tree), by introducing randomization into its construction procedure and 
1075     then making an ensemble out of it. 
1076  
1077     This algorithm encompasses several works from the literature. When random 
1078     subsets of the dataset are drawn as random subsets of the samples, then 
1079     this algorithm is known as Pasting [1]_. If samples are drawn with 
1080     replacement, then the method is known as Bagging [2]_. When random subsets 
1081     of the dataset are drawn as random subsets of the features, then the method 
1082     is known as Random Subspaces [3]_. Finally, when base estimators are built 
1083     on subsets of both samples and features, then the method is known as 
1084     Random Patches [4]_. 
1085  
1086     Read more in the :ref:`User Guide <bagging>`. 
1087

1089  
1090     Parameters 
1091     ----------

1300         n_samples = y.shape[0]
1301

1303         n_predictions = np.zeros((n_samples,))
