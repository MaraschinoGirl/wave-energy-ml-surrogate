27   @pytest.mark.filterwarnings("ignore:Passing a BlockManager:DeprecationWarning")
28   def test_multi_thread_string_io_read_csv(all_parsers, request):
29       # see gh-11786
30       parser = all_parsers
31       if parser.engine == "pyarrow":
32           pa = pytest.importorskip("pyarrow")
33           if Version(pa.__version__) < Version("16.0"):
34               request.applymarker(
35                   pytest.mark.xfail(reason="# ValueError: Found non-unique column index")
36               )
37       max_row_range = 100
38       num_files = 10
39   
40       bytes_to_df = (
41           "\n".join([f"{i:d},{i:d},{i:d}" for i in range(max_row_range)]).encode()
42           for _ in range(num_files)
43       )
44   
45       # Read all files in many threads.
46       with ExitStack() as stack:
47           files = [stack.enter_context(BytesIO(b)) for b in bytes_to_df]
48   
49           pool = stack.enter_context(ThreadPool(8))
50   
51           results = pool.map(parser.read_csv, files)
52           first_result = results[0]
53   
54           for result in results:
55               tm.assert_frame_equal(first_result, result)

58   def _generate_multi_thread_dataframe(parser, path, num_rows, num_tasks):
59       """ 
60       Generate a DataFrame via multi-thread. 
61    
62       Parameters 
63       ---------- 
64       parser : BaseParser 
65           The parser object to use for reading the data. 
66       path : str 
67           The location of the CSV file to read. 
68       num_rows : int 
69           The number of rows to read per task. 
70       num_tasks : int 
71           The number of tasks to use for reading this DataFrame. 
72    
73       Returns 
74       ------- 
75       df : DataFrame 
76       """
77   
78       def reader(arg):
79           """ 
80           Create a reader for part of the CSV. 
81    
82           Parameters 
83           ---------- 
84           arg : tuple 
85               A tuple of the following: 
86    
87               * start : int 
88                   The starting row to start for parsing CSV 
89               * nrows : int 
90                   The number of rows to read. 
91    
92           Returns 
93           ------- 
94           df : DataFrame 
95           """
96           start, nrows = arg
97   
98           if not start:
99               return parser.read_csv(
100                  path, index_col=0, header=0, nrows=nrows, parse_dates=["date"]
101              )
102  
103          return parser.read_csv(
104              path,
105              index_col=0,
106              header=None,
107              skiprows=int(start) + 1,
108              nrows=nrows,
109              parse_dates=[9],
110          )
111  
112      tasks = [
113          (num_rows * i // num_tasks, num_rows // num_tasks) for i in range(num_tasks)
114      ]
115  
116      with ThreadPool(processes=num_tasks) as pool:
117          results = pool.map(reader, tasks)
118  
119      header = results[0].columns
120  
121      for r in results[1:]:
122          r.columns = header
123  
124      final_dataframe = pd.concat(results)
125      return final_dataframe

128  @xfail_pyarrow  # ValueError: The 'nrows' option is not supported
129  def test_multi_thread_path_multipart_read_csv(all_parsers):
130      # see gh-11786
131      num_tasks = 4
132      num_rows = 48
133  
134      parser = all_parsers
135      file_name = "__thread_pool_reader__.csv"
136      df = DataFrame(
137          {
138              "a": np.random.default_rng(2).random(num_rows),
139              "b": np.random.default_rng(2).random(num_rows),
140              "c": np.random.default_rng(2).random(num_rows),
141              "d": np.random.default_rng(2).random(num_rows),
142              "e": np.random.default_rng(2).random(num_rows),
143              "foo": ["foo"] * num_rows,
144              "bar": ["bar"] * num_rows,
145              "baz": ["baz"] * num_rows,
146              "date": pd.date_range("20000101 09:00:00", periods=num_rows, freq="s"),
147              "int": np.arange(num_rows, dtype="int64"),
148          }
149      )
150  
151      with tm.ensure_clean(file_name) as path:
152          df.to_csv(path)
153  
154          final_dataframe = _generate_multi_thread_dataframe(
155              parser, path, num_rows, num_tasks
156          )
157          tm.assert_frame_equal(df, final_dataframe)
