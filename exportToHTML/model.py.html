34   @keras_export(["keras.Model", "keras.models.Model"])
35   class Model(Trainer, base_trainer.Trainer, Layer):
36       """A model grouping layers into an object with training/inference features. 
37    
38       There are three ways to instantiate a `Model`: 
39    
40       ## With the "Functional API" 
41    
42       You start from `Input`, 
43       you chain layer calls to specify the model's forward pass, 
44       and finally, you create your model from inputs and outputs: 
45    
46       ```python 
47       inputs = keras.Input(shape=(37,)) 
48       x = keras.layers.Dense(32, activation="relu")(inputs) 
49       outputs = keras.layers.Dense(5, activation="softmax")(x) 
50       model = keras.Model(inputs=inputs, outputs=outputs) 
51       ``` 
52    
53       Note: Only dicts, lists, and tuples of input tensors are supported. Nested 
54       inputs are not supported (e.g. lists of list or dicts of dict). 
55    
56       A new Functional API model can also be created by using the 
57       intermediate tensors. This enables you to quickly extract sub-components 
58       of the model. 
59    
60       Example: 
61    
62       ```python 
63       inputs = keras.Input(shape=(None, None, 3)) 
64       processed = keras.layers.RandomCrop(width=128, height=128)(inputs) 
65       conv = keras.layers.Conv2D(filters=32, kernel_size=3)(processed) 
66       pooling = keras.layers.GlobalAveragePooling2D()(conv) 
67       feature = keras.layers.Dense(10)(pooling) 
68    
69       full_model = keras.Model(inputs, feature) 
70       backbone = keras.Model(processed, conv) 
71       activations = keras.Model(conv, feature) 
72       ``` 
73    
74       Note that the `backbone` and `activations` models are not 
75       created with `keras.Input` objects, but with the tensors that originate 
76       from `keras.Input` objects. Under the hood, the layers and weights will 
77       be shared across these models, so that user can train the `full_model`, and 
78       use `backbone` or `activations` to do feature extraction. 
79       The inputs and outputs of the model can be nested structures of tensors as 
80       well, and the created models are standard Functional API models that support 
81       all the existing APIs. 
82    
83       ## By subclassing the `Model` class 
84    
85       In that case, you should define your 
86       layers in `__init__()` and you should implement the model's forward pass 
87       in `call()`. 
88    
89       ```python 
90       class MyModel(keras.Model): 
91           def __init__(self): 
92               super().__init__() 
93               self.dense1 = keras.layers.Dense(32, activation="relu") 
94               self.dense2 = keras.layers.Dense(5, activation="softmax") 
95    
96           def call(self, inputs): 
97               x = self.dense1(inputs) 
98               return self.dense2(x) 
99    
100      model = MyModel() 
101      ``` 
102   
103      If you subclass `Model`, you can optionally have 
104      a `training` argument (boolean) in `call()`, which you can use to specify 
105      a different behavior in training and inference: 
106   
107      ```python 
108      class MyModel(keras.Model): 
109          def __init__(self): 
110              super().__init__() 
111              self.dense1 = keras.layers.Dense(32, activation="relu") 
112              self.dense2 = keras.layers.Dense(5, activation="softmax") 
113              self.dropout = keras.layers.Dropout(0.5) 
114   
115          def call(self, inputs, training=False): 
116              x = self.dense1(inputs) 
117              x = self.dropout(x, training=training) 
118              return self.dense2(x) 
119   
120      model = MyModel() 
121      ``` 
122   
123      Once the model is created, you can config the model with losses and metrics 
124      with `model.compile()`, train the model with `model.fit()`, or use the model 
125      to do prediction with `model.predict()`. 
126   
127      ## With the `Sequential` class 
128   
129      In addition, `keras.Sequential` is a special case of model where 
130      the model is purely a stack of single-input, single-output layers. 
131   
132      ```python 
133      model = keras.Sequential([ 
134          keras.Input(shape=(None, None, 3)), 
135          keras.layers.Conv2D(filters=32, kernel_size=3), 
136      ]) 
137      ``` 
138      """

177      @traceback_utils.filter_traceback
178      def get_layer(self, name=None, index=None):
179          """Retrieves a layer based on either its name (unique) or index. 
180   
181          If `name` and `index` are both provided, `index` will take precedence. 
182          Indices are based on order of horizontal graph traversal (bottom-up). 
183   
184          Args: 
185              name: String, name of layer. 
186              index: Integer, index of layer. 
187   
188          Returns: 
189              A layer instance. 
190          """
191          if index is not None and name is not None:
192              raise ValueError(
193                  "Provide only a layer name or a layer index. Received: "
194                  f"index={index}, name={name}."
195              )
196          if index is not None:
197              if len(self.layers) <= index:
198                  raise ValueError(
199                      f"Was asked to retrieve layer at index {index}"
200                      f" but model only has {len(self.layers)}"
201                      " layers."
202                  )
203              else:
204                  return self.layers[index]
205  
206          if name is not None:
207              for layer in self.layers:
208                  if layer.name == name:
209                      return layer
210              raise ValueError(
211                  f"No such layer: {name}. Existing layers are: "
212                  f"{list(layer.name for layer in self.layers)}."
213              )
214          raise ValueError(
215              "Provide either a layer name or layer index at `get_layer`."
216          )

269      @traceback_utils.filter_traceback
270      def save(self, filepath, overwrite=True, zipped=None, **kwargs):
271          """Saves a model as a `.keras` file. 
272   
273          Args: 
274              filepath: `str` or `pathlib.Path` object. 
275                  The path where to save the model. Must end in `.keras` 
276                  (unless saving the model as an unzipped directory 
277                  via `zipped=False`). 
278              overwrite: Whether we should overwrite any existing model at 
279                  the target location, or instead ask the user via 
280                  an interactive prompt. 
281              zipped: Whether to save the model as a zipped `.keras` 
282                  archive (default when saving locally), or as an 
283                  unzipped directory (default when saving on the 
284                  Hugging Face Hub). 
285   
286          Example: 
287   
288          ```python 
289          model = keras.Sequential( 
290              [ 
291                  keras.layers.Dense(5, input_shape=(3,)), 
292                  keras.layers.Softmax(), 
293              ], 
294          ) 
295          model.save("model.keras") 
296          loaded_model = keras.saving.load_model("model.keras") 
297          x = keras.random.uniform((10, 3)) 
298          assert np.allclose(model.predict(x), loaded_model.predict(x)) 
299          ``` 
300   
301          Note that `model.save()` is an alias for `keras.saving.save_model()`. 
302   
303          The saved `.keras` file contains: 
304   
305          - The model's configuration (architecture) 
306          - The model's weights 
307          - The model's optimizer's state (if any) 
308   
309          Thus models can be reinstantiated in the exact same state. 
310          """
311          return saving_api.save_model(
312              self, filepath, overwrite=overwrite, zipped=zipped, **kwargs
313          )

328      @traceback_utils.filter_traceback
329      def load_weights(self, filepath, skip_mismatch=False, **kwargs):
330          """Load weights from a file saved via `save_weights()`. 
331   
332          Weights are loaded based on the network's 
333          topology. This means the architecture should be the same as when the 
334          weights were saved. Note that layers that don't have weights are not 
335          taken into account in the topological ordering, so adding or removing 
336          layers is fine as long as they don't have weights. 
337   
338          **Partial weight loading** 
339   
340          If you have modified your model, for instance by adding a new layer 
341          (with weights) or by changing the shape of the weights of a layer, 
342          you can choose to ignore errors and continue loading 
343          by setting `skip_mismatch=True`. In this case any layer with 
344          mismatching weights will be skipped. A warning will be displayed 
345          for each skipped layer. 
346   
347          Args: 
348              filepath: String, path to the weights file to load. 
349                  It can either be a `.weights.h5` file 
350                  or a legacy `.h5` weights file. 
351              skip_mismatch: Boolean, whether to skip loading of layers where 
352                  there is a mismatch in the number of weights, or a mismatch in 
353                  the shape of the weights. 
354          """
355          saving_api.load_weights(
356              self, filepath, skip_mismatch=skip_mismatch, **kwargs
357          )

359      def quantize(self, mode, **kwargs):
360          """Quantize the weights of the model. 
361   
362          Note that the model must be built first before calling this method. 
363          `quantize` will recursively call `quantize(mode)` in all layers and 
364          will be skipped if the layer doesn't implement the function. 
365   
366          Args: 
367              mode: The mode of the quantization. Only 'int8' is supported at this 
368                  time. 
369          """
370          from keras.src.dtype_policies import QUANTIZATION_MODES
371  
372          type_check = kwargs.pop("type_check", True)
373          if kwargs:
374              raise ValueError(
375                  "Unrecognized keyword arguments "
376                  f"passed to {self.__class__.__name__}: {kwargs}"
377              )
378          if mode not in QUANTIZATION_MODES:
379              raise ValueError(
380                  "Invalid quantization mode. "
381                  f"Expected one of {QUANTIZATION_MODES}. Received: mode={mode}"
382              )
383          mode_changed = False
384          for layer in self._flatten_layers():
385              list_of_sublayers = list(layer._flatten_layers())
386              if len(list_of_sublayers) == 1:  # leaves of the model
387                  try:
388                      layer.quantize(mode, type_check=type_check)
389                      mode_changed = True
390                  except NotImplementedError as e:
391                      warnings.warn(str(e))
392          # We need to set these functions to `None` to remake them for changed
393          # call function
394          if mode_changed:
395              self.train_function = None
396              self.test_function = None
397              self.predict_function = None

399      def build_from_config(self, config):
400          if not config:
401              return
402          status = False
403          if "input_shape" in config:
404              # Case: all inputs are in the first arg (possibly nested).
405              if utils.is_default(self.build):
406                  status = self._build_by_run_for_single_pos_arg(
407                      config["input_shape"]
408                  )
409              else:
410                  try:
411                      self.build(config["input_shape"])
412                      status = True
413                  except:
414                      pass
415              self._build_shapes_dict = config
416  
417          elif "shapes_dict" in config:
418              # Case: inputs were recorded as multiple keyword arguments.
419              if utils.is_default(self.build):
420                  status = self._build_by_run_for_kwargs(config["shapes_dict"])
421              else:
422                  try:
423                      self.build(**config["shapes_dict"])
424                      status = True
425                  except:
426                      pass
427              self._build_shapes_dict = config["shapes_dict"]
428  
429          if not status:
430              warnings.warn(
431                  f"Model '{self.name}' had a build config, but the model "
432                  "cannot be built automatically in "
433                  "`build_from_config(config)`. "
434                  "You should implement "
435                  "`def build_from_config(self, config)`, "
436                  "and you might also want to implement the method "
437                  " that generates the config at saving time, "
438                  "`def get_build_config(self)`. "
439                  "The method `build_from_config()` is meant to "
440                  "create the state of the model (i.e. its variables) "
441                  "upon deserialization.",
442                  stacklevel=2,
443              )

464          self,
465          filepath,
466          format="tf_saved_model",
467          verbose=None,
468          input_signature=None,
469          **kwargs,
470      ):
471          """Export the model as an artifact for inference. 
472   
473          Args: 
474              filepath: `str` or `pathlib.Path` object. The path to save the 
475                  artifact. 
476              format: `str`. The export format. Supported values: 
477                  `"tf_saved_model"` and `"onnx"`.  Defaults to 
478                  `"tf_saved_model"`. 
479              verbose: `bool`. Whether to print a message during export. Defaults 
480                  to `None`, which uses the default value set by different

482              input_signature: Optional. Specifies the shape and dtype of the 
483                  model inputs. Can be a structure of `keras.InputSpec`, 
484                  `tf.TensorSpec`, `backend.KerasTensor`, or backend tensor. If 
485                  not provided, it will be automatically computed. Defaults to 
486                  `None`. 
487              **kwargs: Additional keyword arguments: 
488                  - Specific to the JAX backend and `format="tf_saved_model"`: 
489                      - `is_static`: Optional `bool`. Indicates whether `fn` is 
490                          static. Set to `False` if `fn` involves state updates 
491                          (e.g., RNG seeds and counters). 
492                      - `jax2tf_kwargs`: Optional `dict`. Arguments for 
493                          `jax2tf.convert`. See the documentation for 
494                          [`jax2tf.convert`]( 
495                              https://github.com/google/jax/blob/main/jax/experimental/jax2tf/README.md). 
496                          If `native_serialization` and `polymorphic_shapes` are 
497                          not provided, they will be automatically computed. 
498

500          and Torch backends. 
501   
502          **Note:** Be aware that the exported artifact may contain information 
503          from the local file system when using `format="onnx"`, `verbose=True` 
504          and Torch backend. 
505   
506          Examples: 
507   
508          Here's how to export a TensorFlow SavedModel for inference. 
509   
510          ```python 
511          # Export the model as a TensorFlow SavedModel artifact 
512          model.export("path/to/location", format="tf_saved_model")

514          # Load the artifact in a different process/environment

518   
519          Here's how to export an ONNX for inference. 
520   
521          ```python 
522          # Export the model as a ONNX artifact 
523          model.export("path/to/location", format="onnx") 
524   
525          # Load the artifact in a different process/environment 
526          ort_session = onnxruntime.InferenceSession("path/to/location") 
527          ort_inputs = { 
528              k.name: v for k, v in zip(ort_session.get_inputs(), input_data) 
529          } 
530          predictions = ort_session.run(None, ort_inputs) 
531          ``` 
532          """

534          from keras.src.export import export_saved_model

536          available_formats = ("tf_saved_model", "onnx")

540                  f"{list(available_formats)}."

543          if format == "tf_saved_model":

545                  self,
546                  filepath,
547                  verbose,
548                  input_signature=input_signature,
549                  **kwargs,

551          elif format == "onnx":
552              export_onnx(
553                  self,
554                  filepath,
555                  verbose,
556                  input_signature=input_signature,
557                  **kwargs,
558              )

560      @classmethod
561      def from_config(cls, config, custom_objects=None):
562          from keras.src.models.functional import Functional
563  
564          functional_config_keys = [
565              "name",
566              "layers",
567              "input_layers",
568              "output_layers",
569          ]
570          is_functional_config = all(
571              key in config for key in functional_config_keys
572          )
573          argspec = inspect.getfullargspec(cls.__init__)
574          functional_init_args = inspect.getfullargspec(Functional.__init__).args[
575              1:
576          ]
577          revivable_as_functional = (
578              cls in {Functional, Model}
579              or argspec.args[1:] == functional_init_args
580              or (argspec.varargs == "args" and argspec.varkw == "kwargs")
581          )
582          if is_functional_config and revivable_as_functional:
583              # Revive Functional model
584              # (but not Functional subclasses with a custom __init__)
585              from keras.src.models.functional import functional_from_config
586  
587              return functional_from_config(
588                  cls, config, custom_objects=custom_objects
589              )
590  
591          # Either the model has a custom __init__, or the config
592          # does not contain all the information necessary to
593          # revive a Functional model. This happens when the user creates
594          # subclassed models where `get_config()` is returning
595          # insufficient information to be considered a Functional model.
596          # In this case, we fall back to provide all config into the
597          # constructor of the class.
598          try:
599              return cls(**config)
600          except TypeError as e:
601              raise TypeError(
602                  "Unable to revive model from config. When overriding "
603                  "the `get_config()` method, make sure that the "
604                  "returned config contains all items used as arguments "
605                  f"in the  constructor to {cls}, "
606                  "which is the default behavior. "
607                  "You can override this default behavior by defining a "
608                  "`from_config(cls, config)` class method to specify "
609                  "how to create an "
610                  f"instance of {cls.__name__} from its config.\n\n"
611                  f"Received config={config}\n\n"
612                  f"Error encountered during deserialization: {e}"
613              )

620      def get_state_tree(self, value_format="backend_tensor"):
621          """Retrieves tree-like structure of model variables. 
622   
623          This method allows retrieval of different model variables (trainable, 
624          non-trainable, optimizer, and metrics). The variables are returned in a 
625          nested dictionary format, where the keys correspond to the variable 
626          names and the values are the nested representations of the variables. 
627   
628          Returns: 
629              dict: A dictionary containing the nested representations of the 
630                  requested variables. The keys are the variable names, and the 
631                  values are the corresponding nested dictionaries. 
632              value_format: One of `"backend_tensor"`, `"numpy_array"`. 
633                  The kind of array to return as the leaves of the nested 
634                      state tree. 
635   
636          Example: 
637   
638          ```python 
639          model = keras.Sequential([ 
640              keras.Input(shape=(1,), name="my_input"), 
641              keras.layers.Dense(1, activation="sigmoid", name="my_dense"), 
642          ], name="my_sequential") 
643          model.compile(optimizer="adam", loss="mse", metrics=["mae"]) 
644          model.fit(np.array([[1.0]]), np.array([[1.0]])) 
645          state_tree = model.get_state_tree() 
646          ``` 
647   
648          The `state_tree` dictionary returned looks like: 
649   
650          ``` 
651          { 
652              'metrics_variables': { 
653                  'loss': { 
654                      'count': ..., 
655                      'total': ..., 
656                  }, 
657                  'mean_absolute_error': { 
658                      'count': ..., 
659                      'total': ..., 
660                  } 
661              }, 
662              'trainable_variables': { 
663                  'my_sequential': { 
664                      'my_dense': { 
665                          'bias': ..., 
666                          'kernel': ..., 
667                      } 
668                  } 
669              }, 
670              'non_trainable_variables': {}, 
671              'optimizer_variables': { 
672                  'adam': { 
673                          'iteration': ..., 
674                          'learning_rate': ..., 
675                          'my_sequential_my_dense_bias_momentum': ..., 
676                          'my_sequential_my_dense_bias_velocity': ..., 
677                          'my_sequential_my_dense_kernel_momentum': ..., 
678                          'my_sequential_my_dense_kernel_velocity': ..., 
679                      } 
680                  } 
681              } 
682          } 
683          ``` 
684          """
685          variables = {}
686          variables["trainable_variables"] = self._create_nested_dict(
687              self.trainable_variables, value_format
688          )
689          variables["non_trainable_variables"] = self._create_nested_dict(
690              self.non_trainable_variables, value_format
691          )
692          variables["optimizer_variables"] = self._create_nested_dict(
693              self.optimizer.variables, value_format
694          )
695          variables["metrics_variables"] = self._create_nested_dict(
696              self.metrics_variables, value_format
697          )
698          return variables
699  
700      def _create_nested_dict(self, variables, value_format):
701          flat_dict = {}
702          for v in variables:
703              if v.path in flat_dict:
704                  raise ValueError(
705                      "The following variable path is found twice in the model: "
706                      f"'{v.path}'. `get_state_tree()` can only be called when "
707                      "all variable paths are unique. Make sure to give unique "
708                      "names to your layers (and other objects)."
709                  )
710              if value_format == "backend_tensor":
711                  flat_dict[v.path] = v.value
712              elif value_format == "numpy_array":
713                  flat_dict[v.path] = v.numpy()
714              else:
715                  raise ValueError(
716                      "Invalid `value_format` argument. Expected one of "
717                      "{'numpy_array', 'backend_tensor'}. Received: "
718                      f"value_format={value_format}"
719                  )
720  
721          nested_dict = {}
722          for path, value in flat_dict.items():
723              parts = path.split("/")
724              current_dict = nested_dict
725              for part in parts[:-1]:
726                  if part not in current_dict:
727                      current_dict[part] = {}
728                  current_dict = current_dict[part]
729              current_dict[parts[-1]] = value
730  
731          return nested_dict
732  
733      def set_state_tree(self, state_tree):
734          """Assigns values to variables of the model. 
735   
736          This method takes a dictionary of nested variable values, which 
737          represents the state tree of the model, and assigns them to the 
738          corresponding variables of the model. The dictionary keys represent the 
739          variable names (e.g., `'trainable_variables'`, `'optimizer_variables'`), 
740          and the values are nested dictionaries containing the variable 
741          paths and their corresponding values. 
742   
743          Args: 
744              state_tree: A dictionary representing the state tree of the model. 
745                  The keys are the variable names, and the values are nested 
746                  dictionaries representing the variable paths and their values. 
747          """
748          for k, v in state_tree.items():
749              path_value_dict = self._flatten_nested_dict(v)
750              if k == "trainable_variables":
751                  self._assign_variable_values(
752                      self.trainable_variables, path_value_dict
753                  )
754              elif k == "non_trainable_variables":
755                  self._assign_variable_values(
756                      self.non_trainable_variables, path_value_dict
757                  )
758              elif k == "optimizer_variables":
759                  self._assign_variable_values(
760                      self.optimizer.variables, path_value_dict
761                  )
762              elif k == "metrics_variables":
763                  self._assign_variable_values(
764                      self.metrics_variables, path_value_dict
765                  )
766              else:
767                  raise ValueError(f"Unknown variable name: {k}")
768  
769      def _assign_variable_values(self, variables, path_value_dict):
770          for path, value in path_value_dict.items():
771              for variable in variables:
772                  if variable.path == path:
773                      variable.assign(value)
774  
775      def _flatten_nested_dict(self, nested_dict):
776          flat_dict = {}
777  
778          def _flatten(current_dict, prefix=""):
779              for key, value in current_dict.items():
780                  if isinstance(value, dict):
781                      _flatten(value, prefix + key + "/")
782                  else:
783                      flat_dict[prefix + key] = value
784  
785          _flatten(nested_dict)
786          return flat_dict

827      """Inject `Functional` into the hierarchy of this class if needed."""
828      from keras.src.models import functional
829  
830      if cls is Model:

832      # In case there is any multiple inheritance, we stop injecting the
833      # class if keras model is not in its class hierarchy.
834      if cls is object:
835          return object

838          inject_functional_model_class(base) for base in cls.__bases__
839      )
840      # Trigger any `__new__` class swapping that needed to happen on `Functional`
841      # but did not because functional was not in the class hierarchy.
842      cls.__new__(cls)
