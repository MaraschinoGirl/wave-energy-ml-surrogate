14   class Normal(ContinuousDistribution):
15       r"""Normal distribution with prescribed mean and standard deviation. 
16    
17       The probability density function of the normal distribution is: 
18    
19       .. math:: 
20    
21           f(x) = \frac{1}{\sigma \sqrt{2 \pi}} \exp { 
22               \left( -\frac{1}{2}\left( \frac{x - \mu}{\sigma} \right)^2 \right)} 
23    
24       """
25       # `ShiftedScaledDistribution` allows this to be generated automatically from
26       # an instance of `StandardNormal`, but the normal distribution is so frequently
27       # used that it's worth a bit of code duplication to get better performance.
28       _mu_domain = _RealDomain(endpoints=(-inf, inf))
29       _sigma_domain = _RealDomain(endpoints=(0, inf))
30       _x_support = _RealDomain(endpoints=(-inf, inf))
31   
32       _mu_param = _RealParameter('mu',  symbol=r'\mu', domain=_mu_domain,
33                                  typical=(-1, 1))
34       _sigma_param = _RealParameter('sigma', symbol=r'\sigma', domain=_sigma_domain,
35                                     typical=(0.5, 1.5))
36       _x_param = _RealParameter('x', domain=_x_support, typical=(-1, 1))
37   
38       _parameterizations = [_Parameterization(_mu_param, _sigma_param)]
39   
40       _variable = _x_param
41       _normalization = 1/np.sqrt(2*np.pi)
42       _log_normalization = np.log(2*np.pi)/2
43   
44       def __new__(cls, mu=None, sigma=None, **kwargs):
45           if mu is None and sigma is None:
46               return super().__new__(StandardNormal)
47           return super().__new__(cls)
48   
49       def __init__(self, *, mu=0., sigma=1., **kwargs):
50           super().__init__(mu=mu, sigma=sigma, **kwargs)
51   
52       def _logpdf_formula(self, x, *, mu, sigma, **kwargs):
53           return StandardNormal._logpdf_formula(self, (x - mu)/sigma) - np.log(sigma)
54   
55       def _pdf_formula(self, x, *, mu, sigma, **kwargs):
56           return StandardNormal._pdf_formula(self, (x - mu)/sigma) / sigma
57   
58       def _logcdf_formula(self, x, *, mu, sigma, **kwargs):
59           return StandardNormal._logcdf_formula(self, (x - mu)/sigma)
60   
61       def _cdf_formula(self, x, *, mu, sigma, **kwargs):
62           return StandardNormal._cdf_formula(self, (x - mu)/sigma)
63   
64       def _logccdf_formula(self, x, *, mu, sigma, **kwargs):
65           return StandardNormal._logccdf_formula(self, (x - mu)/sigma)
66   
67       def _ccdf_formula(self, x, *, mu, sigma, **kwargs):
68           return StandardNormal._ccdf_formula(self, (x - mu)/sigma)
69   
70       def _icdf_formula(self, x, *, mu, sigma, **kwargs):
71           return StandardNormal._icdf_formula(self, x) * sigma + mu
72   
73       def _ilogcdf_formula(self, x, *, mu, sigma, **kwargs):
74           return StandardNormal._ilogcdf_formula(self, x) * sigma + mu
75   
76       def _iccdf_formula(self, x, *, mu, sigma, **kwargs):
77           return StandardNormal._iccdf_formula(self, x) * sigma + mu
78   
79       def _ilogccdf_formula(self, x, *, mu, sigma, **kwargs):
80           return StandardNormal._ilogccdf_formula(self, x) * sigma + mu
81   
82       def _entropy_formula(self, *, mu, sigma, **kwargs):
83           return StandardNormal._entropy_formula(self) + np.log(abs(sigma))
84   
85       def _logentropy_formula(self, *, mu, sigma, **kwargs):
86           lH0 = StandardNormal._logentropy_formula(self)
87           with np.errstate(divide='ignore'):
88               # sigma = 1 -> log(sigma) = 0 -> log(log(sigma)) = -inf
89               # Silence the unnecessary runtime warning
90               lls = np.log(np.log(abs(sigma))+0j)
91           return special.logsumexp(np.broadcast_arrays(lH0, lls), axis=0)
92   
93       def _median_formula(self, *, mu, sigma, **kwargs):
94           return mu
95   
96       def _mode_formula(self, *, mu, sigma, **kwargs):
97           return mu
98   
99       def _moment_raw_formula(self, order, *, mu, sigma, **kwargs):
100          if order == 0:
101              return np.ones_like(mu)
102          elif order == 1:
103              return mu
104          else:
105              return None
106      _moment_raw_formula.orders = [0, 1]  # type: ignore[attr-defined]
107  
108      def _moment_central_formula(self, order, *, mu, sigma, **kwargs):
109          if order == 0:
110              return np.ones_like(mu)
111          elif order % 2:
112              return np.zeros_like(mu)
113          else:
114              # exact is faster (and obviously more accurate) for reasonable orders
115              return sigma**order * special.factorial2(int(order) - 1, exact=True)
116  
117      def _sample_formula(self, sample_shape, full_shape, rng, *, mu, sigma, **kwargs):
118          return rng.normal(loc=mu, scale=sigma, size=full_shape)[()]

121  def _log_diff(log_p, log_q):
122      return special.logsumexp([log_p, log_q+np.pi*1j], axis=0)

125  class StandardNormal(Normal):
126      r"""Standard normal distribution. 
127   
128      The probability density function of the standard normal distribution is: 
129   
130      .. math:: 
131   
132          f(x) = \frac{1}{\sqrt{2 \pi}} \exp \left( -\frac{1}{2} x^2 \right) 
133   
134      """
135      _x_support = _RealDomain(endpoints=(-inf, inf))
136      _x_param = _RealParameter('x', domain=_x_support, typical=(-5, 5))
137      _variable = _x_param
138      _parameterizations = []
139      _normalization = 1/np.sqrt(2*np.pi)
140      _log_normalization = np.log(2*np.pi)/2
141      mu = np.float64(0.)
142      sigma = np.float64(1.)
143  
144      def __init__(self, **kwargs):
145          ContinuousDistribution.__init__(self, **kwargs)
146  
147      def _logpdf_formula(self, x, **kwargs):
148          return -(self._log_normalization + x**2/2)
149  
150      def _pdf_formula(self, x, **kwargs):
151          return self._normalization * np.exp(-x**2/2)
152  
153      def _logcdf_formula(self, x, **kwargs):
154          return special.log_ndtr(x)
155  
156      def _cdf_formula(self, x, **kwargs):
157          return special.ndtr(x)
158  
159      def _logccdf_formula(self, x, **kwargs):
160          return special.log_ndtr(-x)
161  
162      def _ccdf_formula(self, x, **kwargs):
163          return special.ndtr(-x)
164  
165      def _icdf_formula(self, x, **kwargs):
166          return special.ndtri(x)
167  
168      def _ilogcdf_formula(self, x, **kwargs):
169          return special.ndtri_exp(x)
170  
171      def _iccdf_formula(self, x, **kwargs):
172          return -special.ndtri(x)
173  
174      def _ilogccdf_formula(self, x, **kwargs):
175          return -special.ndtri_exp(x)
176  
177      def _entropy_formula(self, **kwargs):
178          return (1 + np.log(2*np.pi))/2
179  
180      def _logentropy_formula(self, **kwargs):
181          return np.log1p(np.log(2*np.pi)) - np.log(2)
182  
183      def _median_formula(self, **kwargs):
184          return 0
185  
186      def _mode_formula(self, **kwargs):
187          return 0
188  
189      def _moment_raw_formula(self, order, **kwargs):
190          raw_moments = {0: 1, 1: 0, 2: 1, 3: 0, 4: 3, 5: 0}
191          return raw_moments.get(order, None)
192  
193      def _moment_central_formula(self, order, **kwargs):
194          return self._moment_raw_formula(order, **kwargs)
195  
196      def _moment_standardized_formula(self, order, **kwargs):
197          return self._moment_raw_formula(order, **kwargs)
198  
199      def _sample_formula(self, sample_shape, full_shape, rng, **kwargs):
200          return rng.normal(size=full_shape)[()]

204  class _LogUniform(ContinuousDistribution):
205      r"""Log-uniform distribution. 
206   
207      The probability density function of the log-uniform distribution is: 
208   
209      .. math:: 
210   
211          f(x; a, b) = \frac{1} 
212                            {x (\log(b) - \log(a))} 
213   
214      If :math:`\log(X)` is a random variable that follows a uniform distribution 
215      between :math:`\log(a)` and :math:`\log(b)`, then :math:`X` is log-uniformly 
216      distributed with shape parameters :math:`a` and :math:`b`. 
217   
218      """
219  
220      _a_domain = _RealDomain(endpoints=(0, inf))
221      _b_domain = _RealDomain(endpoints=('a', inf))
222      _log_a_domain = _RealDomain(endpoints=(-inf, inf))
223      _log_b_domain = _RealDomain(endpoints=('log_a', inf))
224      _x_support = _RealDomain(endpoints=('a', 'b'), inclusive=(True, True))
225  
226      _a_param = _RealParameter('a', domain=_a_domain, typical=(1e-3, 0.9))
227      _b_param = _RealParameter('b', domain=_b_domain, typical=(1.1, 1e3))
228      _log_a_param = _RealParameter('log_a', symbol=r'\log(a)',
229                                    domain=_log_a_domain, typical=(-3, -0.1))
230      _log_b_param = _RealParameter('log_b', symbol=r'\log(b)',
231                                    domain=_log_b_domain, typical=(0.1, 3))
232      _x_param = _RealParameter('x', domain=_x_support, typical=('a', 'b'))
233  
234      _b_domain.define_parameters(_a_param)
235      _log_b_domain.define_parameters(_log_a_param)
236      _x_support.define_parameters(_a_param, _b_param)
237  
238      _parameterizations = [_Parameterization(_log_a_param, _log_b_param),
239                            _Parameterization(_a_param, _b_param)]
240      _variable = _x_param
241  
242      def __init__(self, *, a=None, b=None, log_a=None, log_b=None, **kwargs):
243          super().__init__(a=a, b=b, log_a=log_a, log_b=log_b, **kwargs)
244  
245      def _process_parameters(self, a=None, b=None, log_a=None, log_b=None, **kwargs):
246          a = np.exp(log_a) if a is None else a
247          b = np.exp(log_b) if b is None else b
248          log_a = np.log(a) if log_a is None else log_a
249          log_b = np.log(b) if log_b is None else log_b
250          kwargs.update(dict(a=a, b=b, log_a=log_a, log_b=log_b))
251          return kwargs
252  
253      # def _logpdf_formula(self, x, *, log_a, log_b, **kwargs):
254      #     return -np.log(x) - np.log(log_b - log_a)
255  
256      def _pdf_formula(self, x, *, log_a, log_b, **kwargs):
257          return ((log_b - log_a)*x)**-1
258  
259      # def _cdf_formula(self, x, *, log_a, log_b, **kwargs):
260      #     return (np.log(x) - log_a)/(log_b - log_a)
261  
262      def _moment_raw_formula(self, order, log_a, log_b, **kwargs):
263          if order == 0:
264              return self._one
265          t1 = self._one / (log_b - log_a) / order
266          t2 = np.real(np.exp(_log_diff(order * log_b, order * log_a)))
267          return t1 * t2

270  class Uniform(ContinuousDistribution):
271      r"""Uniform distribution. 
272   
273      The probability density function of the uniform distribution is: 
274   
275      .. math:: 
276   
277          f(x; a, b) = \frac{1} 
278                            {b - a} 
279   
280      """
281  
282      _a_domain = _RealDomain(endpoints=(-inf, inf))
283      _b_domain = _RealDomain(endpoints=('a', inf))
284      _x_support = _RealDomain(endpoints=('a', 'b'), inclusive=(True, True))
285  
286      _a_param = _RealParameter('a', domain=_a_domain, typical=(1e-3, 0.9))
287      _b_param = _RealParameter('b', domain=_b_domain, typical=(1.1, 1e3))
288      _x_param = _RealParameter('x', domain=_x_support, typical=('a', 'b'))
289  
290      _b_domain.define_parameters(_a_param)
291      _x_support.define_parameters(_a_param, _b_param)
292  
293      _parameterizations = [_Parameterization(_a_param, _b_param)]
294      _variable = _x_param
295  
296      def __init__(self, *, a=None, b=None, **kwargs):
297          super().__init__(a=a, b=b, **kwargs)
298  
299      def _process_parameters(self, a=None, b=None, ab=None, **kwargs):
300          ab = b - a
301          kwargs.update(dict(a=a, b=b, ab=ab))
302          return kwargs
303  
304      def _logpdf_formula(self, x, *, ab, **kwargs):
305          return np.where(np.isnan(x), np.nan, -np.log(ab))
306  
307      def _pdf_formula(self, x, *, ab, **kwargs):
308          return np.where(np.isnan(x), np.nan, 1/ab)
309  
310      def _logcdf_formula(self, x, *, a, ab, **kwargs):
311          with np.errstate(divide='ignore'):
312              return np.log(x - a) - np.log(ab)
313  
314      def _cdf_formula(self, x, *, a, ab, **kwargs):
315          return (x - a) / ab
316  
317      def _logccdf_formula(self, x, *, b, ab, **kwargs):
318          with np.errstate(divide='ignore'):
319              return np.log(b - x) - np.log(ab)
320  
321      def _ccdf_formula(self, x, *, b, ab, **kwargs):
322          return (b - x) / ab
323  
324      def _icdf_formula(self, p, *, a, ab, **kwargs):
325          return a + ab*p
326  
327      def _iccdf_formula(self, p, *, b, ab, **kwargs):
328          return b - ab*p
329  
330      def _entropy_formula(self, *, ab, **kwargs):
331          return np.log(ab)
332  
333      def _mode_formula(self, *, a, b, ab, **kwargs):
334          return a + 0.5*ab
335  
336      def _median_formula(self, *, a, b, ab, **kwargs):
337          return a + 0.5*ab
338  
339      def _moment_raw_formula(self, order, a, b, ab, **kwargs):
340          np1 = order + 1
341          return (b**np1 - a**np1) / (np1 * ab)
342  
343      def _moment_central_formula(self, order, ab, **kwargs):
344          return ab**2/12 if order == 2 else None
345  
346      _moment_central_formula.orders = [2]  # type: ignore[attr-defined]
347  
348      def _sample_formula(self, sample_shape, full_shape, rng, a, b, ab, **kwargs):
349          try:
350              return rng.uniform(a, b, size=full_shape)[()]
351          except OverflowError:  # happens when there are NaNs
352              return rng.uniform(0, 1, size=full_shape)*ab + a

355  class _Gamma(ContinuousDistribution):
356      # Gamma distribution for testing only
357      _a_domain = _RealDomain(endpoints=(0, inf))
358      _x_support = _RealDomain(endpoints=(0, inf), inclusive=(False, False))
359  
360      _a_param = _RealParameter('a', domain=_a_domain, typical=(0.1, 10))
361      _x_param = _RealParameter('x', domain=_x_support, typical=(0.1, 10))
362  
363      _parameterizations = [_Parameterization(_a_param)]
364      _variable = _x_param
365  
366      def _pdf_formula(self, x, *, a, **kwargs):
367          return x ** (a - 1) * np.exp(-x) / special.gamma(a)
