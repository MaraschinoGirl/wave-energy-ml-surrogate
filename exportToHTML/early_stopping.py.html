3    from keras.src import ops
4    from keras.src.api_export import keras_export
5    from keras.src.callbacks.callback import Callback
6    from keras.src.trainers import compile_utils
7    from keras.src.utils import io_utils

10   @keras_export("keras.callbacks.EarlyStopping")
11   class EarlyStopping(Callback):
12       """Stop training when a monitored metric has stopped improving. 
13    
14       Assuming the goal of a training is to minimize the loss. With this, the 
15       metric to be monitored would be `'loss'`, and mode would be `'min'`. A 
16       `model.fit()` training loop will check at end of every epoch whether 
17       the loss is no longer decreasing, considering the `min_delta` and 
18       `patience` if applicable. Once it's found no longer decreasing, 
19       `model.stop_training` is marked True and the training terminates. 
20    
21       The quantity to be monitored needs to be available in `logs` dict. 
22       To make it so, pass the loss or metrics at `model.compile()`. 
23    
24       Args: 
25           monitor: Quantity to be monitored. Defaults to `"val_loss"`. 
26           min_delta: Minimum change in the monitored quantity to qualify as an 
27               improvement, i.e. an absolute change of less than min_delta, will 
28               count as no improvement. Defaults to `0`. 
29           patience: Number of epochs with no improvement after which training will 
30               be stopped. Defaults to `0`. 
31           verbose: Verbosity mode, 0 or 1. Mode 0 is silent, and mode 1 displays 
32               messages when the callback takes an action. Defaults to `0`. 
33           mode: One of `{"auto", "min", "max"}`. In `min` mode, training will stop 
34               when the quantity monitored has stopped decreasing; in `"max"` mode 
35               it will stop when the quantity monitored has stopped increasing; in 
36               `"auto"` mode, the direction is automatically inferred from the name 
37               of the monitored quantity. Defaults to `"auto"`. 
38           baseline: Baseline value for the monitored quantity. If not `None`, 
39               training will stop if the model doesn't show improvement over the 
40               baseline. Defaults to `None`. 
41           restore_best_weights: Whether to restore model weights from the epoch 
42               with the best value of the monitored quantity. If `False`, the model 
43               weights obtained at the last step of training are used. An epoch 
44               will be restored regardless of the performance relative to the 
45               `baseline`. If no epoch improves on `baseline`, training will run 
46               for `patience` epochs and restore weights from the best epoch in 
47               that set. Defaults to `False`. 
48           start_from_epoch: Number of epochs to wait before starting to monitor 
49               improvement. This allows for a warm-up period in which no 
50               improvement is expected and thus training will not be stopped. 
51               Defaults to `0`. 
52    
53       Example: 
54    
55       >>> callback = keras.callbacks.EarlyStopping(monitor='loss', 
56       ...                                               patience=3) 
57       >>> # This callback will stop the training when there is no improvement in 
58       >>> # the loss for three consecutive epochs. 
59       >>> model = keras.models.Sequential([keras.layers.Dense(10)]) 
60       >>> model.compile(keras.optimizers.SGD(), loss='mse') 
61       >>> history = model.fit(np.arange(100).reshape(5, 20), np.zeros(5), 
62       ...                     epochs=10, batch_size=1, callbacks=[callback], 
63       ...                     verbose=0) 
64       >>> len(history.history['loss'])  # Only 4 epochs are run. 
65       4 
66       """
67   
68       def __init__(
69           self,
70           monitor="val_loss",
71           min_delta=0,
72           patience=0,
73           verbose=0,
74           mode="auto",
75           baseline=None,
76           restore_best_weights=False,
77           start_from_epoch=0,
78       ):
79           super().__init__()
80   
81           self.monitor = monitor
82           self.patience = patience
83           self.verbose = verbose
84           self.baseline = baseline
85           self.min_delta = abs(min_delta)
86           self.wait = 0
87           self.stopped_epoch = 0
88           self.restore_best_weights = restore_best_weights
89           self.best_weights = None
90           self.start_from_epoch = start_from_epoch
91   
92           if mode not in ["auto", "min", "max"]:
93               warnings.warn(
94                   f"EarlyStopping mode {mode} is unknown, fallback to auto mode.",
95                   stacklevel=2,
96               )
97               mode = "auto"
98           self.mode = mode
99           self.monitor_op = None
100  
101      def _set_monitor_op(self):
102          if self.mode == "min":
103              self.monitor_op = ops.less
104          elif self.mode == "max":
105              self.monitor_op = ops.greater
106          else:
107              metric_name = self.monitor.removeprefix("val_")
108              if metric_name == "loss":
109                  self.monitor_op = ops.less
110              if hasattr(self.model, "metrics"):
111                  all_metrics = []
112                  for m in self.model.metrics:
113                      if isinstance(
114                          m,
115                          (
116                              compile_utils.CompileMetrics,
117                              compile_utils.MetricsList,
118                          ),
119                      ):
120                          all_metrics.extend(m.metrics)
121                  for m in all_metrics:
122                      if m.name == metric_name:
123                          if hasattr(m, "_direction"):
124                              if m._direction == "up":
125                                  self.monitor_op = ops.greater
126                              else:
127                                  self.monitor_op = ops.less
128          if self.monitor_op is None:
129              raise ValueError(
130                  f"EarlyStopping callback received monitor={self.monitor} "
131                  "but Keras isn't able to automatically determine whether "
132                  "that metric should be maximized or minimized. "
133                  "Pass `mode='max'` in order to do early stopping based "
134                  "on the highest metric value, or pass `mode='min'` "
135                  "in order to use the lowest value."
136              )
137          if self.monitor_op == ops.less:
138              self.min_delta *= -1
139  
140      def on_train_begin(self, logs=None):
141          # Allow instances to be re-used
142          self.wait = 0
143          self.stopped_epoch = 0
144          self.best = None
145          self.best_weights = None
146          self.best_epoch = 0
147  
148      def on_epoch_end(self, epoch, logs=None):
149          if self.monitor_op is None:
150              # Delay setup until the model's metrics are all built
151              self._set_monitor_op()
152  
153          current = self.get_monitor_value(logs)
154          if current is None or epoch < self.start_from_epoch:
155              # If no monitor value exists or still in initial warm-up stage.
156              return
157          if self.restore_best_weights and self.best_weights is None:
158              # If best weights were never set,
159              # then the current weights are the best.
160              self.best_weights = self.model.get_weights()
161              self.best_epoch = epoch
162  
163          self.wait += 1
164          if self._is_improvement(current, self.best):
165              self.best = current
166              self.best_epoch = epoch
167              if self.restore_best_weights:
168                  self.best_weights = self.model.get_weights()
169              # Only restart wait if we beat both the baseline and our previous
170              # best.
171              if self.baseline is None or self._is_improvement(
172                  current, self.baseline
173              ):
174                  self.wait = 0
175              return
176  
177          if self.wait >= self.patience and epoch > 0:
178              # Patience has been exceeded: stop training
179              self.stopped_epoch = epoch
180              self.model.stop_training = True
181  
182      def on_train_end(self, logs=None):
183          if self.stopped_epoch > 0 and self.verbose > 0:
184              io_utils.print_msg(
185                  f"Epoch {self.stopped_epoch + 1}: early stopping"
186              )
187          if self.restore_best_weights and self.best_weights is not None:
188              if self.verbose > 0:
189                  io_utils.print_msg(
190                      "Restoring model weights from "
191                      "the end of the best epoch: "
192                      f"{self.best_epoch + 1}."
193                  )
194              self.model.set_weights(self.best_weights)
195  
196      def get_monitor_value(self, logs):
197          logs = logs or {}
198          monitor_value = logs.get(self.monitor)
199          if monitor_value is None:
200              warnings.warn(
201                  (
202                      f"Early stopping conditioned on metric `{self.monitor}` "
203                      "which is not available. "
204                      f"Available metrics are: {','.join(list(logs.keys()))}"
205                  ),
206                  stacklevel=2,
207              )
208          return monitor_value
209  
210      def _is_improvement(self, monitor_value, reference_value):
211          if reference_value is None:
212              return True
213          return self.monitor_op(monitor_value - self.min_delta, reference_value)
