21   def _check_params(X, metric, p, metric_params):
22       """Check the validity of the input parameters"""
23       params = zip(["metric", "p", "metric_params"], [metric, p, metric_params])
24       est_params = X.get_params()
25       for param_name, func_param in params:
26           if func_param != est_params[param_name]:
27               raise ValueError(
28                   "Got %s for %s, while the estimator has %s for the same parameter."
29                   % (func_param, param_name, est_params[param_name])
30               )

33   def _query_include_self(X, include_self, mode):
34       """Return the query based on include_self param"""
35       if include_self == "auto":
36           include_self = mode == "connectivity"
37   
38       # it does not include each sample as its own neighbors
39       if not include_self:
40           X = None
41   
42       return X

45   @validate_params(
46       {
47           "X": ["array-like", "sparse matrix", KNeighborsMixin],
48           "n_neighbors": [Interval(Integral, 1, None, closed="left")],
49           "mode": [StrOptions({"connectivity", "distance"})],
50           "metric": [StrOptions(set(itertools.chain(*VALID_METRICS.values()))), callable],
51           "p": [Interval(Real, 0, None, closed="right"), None],
52           "metric_params": [dict, None],
53           "include_self": ["boolean", StrOptions({"auto"})],
54           "n_jobs": [Integral, None],
55       },
56       prefer_skip_nested_validation=False,  # metric is not validated yet
57   )
58   def kneighbors_graph(
59       X,
60       n_neighbors,
61       *,
62       mode="connectivity",
63       metric="minkowski",
64       p=2,
65       metric_params=None,
66       include_self=False,
67       n_jobs=None,
68   ):
69       """Compute the (weighted) graph of k-Neighbors for points in X. 
70    
71       Read more in the :ref:`User Guide <unsupervised_neighbors>`. 
72    
73       Parameters 
74       ---------- 
75       X : {array-like, sparse matrix} of shape (n_samples, n_features) 
76           Sample data. 
77    
78       n_neighbors : int 
79           Number of neighbors for each sample. 
80    
81       mode : {'connectivity', 'distance'}, default='connectivity' 
82           Type of returned matrix: 'connectivity' will return the connectivity 
83           matrix with ones and zeros, and 'distance' will return the distances 
84           between neighbors according to the given metric. 
85    
86       metric : str, default='minkowski' 
87           Metric to use for distance computation. Default is "minkowski", which 
88           results in the standard Euclidean distance when p = 2. See the 
89           documentation of `scipy.spatial.distance 
90           <https://docs.scipy.org/doc/scipy/reference/spatial.distance.html>`_ and 
91           the metrics listed in 
92           :class:`~sklearn.metrics.pairwise.distance_metrics` for valid metric 
93           values. 
94    
95       p : float, default=2 
96           Power parameter for the Minkowski metric. When p = 1, this is equivalent 
97           to using manhattan_distance (l1), and euclidean_distance (l2) for p = 2. 
98           For arbitrary p, minkowski_distance (l_p) is used. This parameter is expected 
99           to be positive. 
100   
101      metric_params : dict, default=None 
102          Additional keyword arguments for the metric function. 
103   
104      include_self : bool or 'auto', default=False 
105          Whether or not to mark each sample as the first nearest neighbor to 
106          itself. If 'auto', then True is used for mode='connectivity' and False 
107          for mode='distance'. 
108   
109      n_jobs : int, default=None 
110          The number of parallel jobs to run for neighbors search. 
111          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
112          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
113          for more details. 
114   
115      Returns 
116      ------- 
117      A : sparse matrix of shape (n_samples, n_samples) 
118          Graph where A[i, j] is assigned the weight of edge that 
119          connects i to j. The matrix is of CSR format. 
120   
121      See Also 
122      -------- 
123      radius_neighbors_graph: Compute the (weighted) graph of Neighbors for points in X. 
124   
125      Examples 
126      -------- 
127      >>> X = [[0], [3], [1]] 
128      >>> from sklearn.neighbors import kneighbors_graph 
129      >>> A = kneighbors_graph(X, 2, mode='connectivity', include_self=True) 
130      >>> A.toarray() 
131      array([[1., 0., 1.], 
132             [0., 1., 1.], 
133             [1., 0., 1.]]) 
134      """
135      if not isinstance(X, KNeighborsMixin):
136          X = NearestNeighbors(
137              n_neighbors=n_neighbors,
138              metric=metric,
139              p=p,
140              metric_params=metric_params,
141              n_jobs=n_jobs,
142          ).fit(X)
143      else:
144          _check_params(X, metric, p, metric_params)
145  
146      query = _query_include_self(X._fit_X, include_self, mode)
147      return X.kneighbors_graph(X=query, n_neighbors=n_neighbors, mode=mode)

167      mode="connectivity",
168      metric="minkowski",

170      metric_params=None,
171      include_self=False,

174      """Compute the (weighted) graph of Neighbors for points in X. 
175   
176      Neighborhoods are restricted the points at a distance lower than 
177      radius. 
178   
179      Read more in the :ref:`User Guide <unsupervised_neighbors>`. 
180   
181      Parameters 
182      ---------- 
183      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
184          Sample data. 
185   
186      radius : float 
187          Radius of neighborhoods. 
188   
189      mode : {'connectivity', 'distance'}, default='connectivity' 
190          Type of returned matrix: 'connectivity' will return the connectivity 
191          matrix with ones and zeros, and 'distance' will return the distances 
192          between neighbors according to the given metric. 
193   
194      metric : str, default='minkowski' 
195          Metric to use for distance computation. Default is "minkowski", which 
196          results in the standard Euclidean distance when p = 2. See the 
197          documentation of `scipy.spatial.distance 
198          <https://docs.scipy.org/doc/scipy/reference/spatial.distance.html>`_ and 
199          the metrics listed in 
200          :class:`~sklearn.metrics.pairwise.distance_metrics` for valid metric 
201          values. 
202   
203      p : float, default=2 
204          Power parameter for the Minkowski metric. When p = 1, this is 
205          equivalent to using manhattan_distance (l1), and euclidean_distance 
206          (l2) for p = 2. For arbitrary p, minkowski_distance (l_p) is used. 
207   
208      metric_params : dict, default=None 
209          Additional keyword arguments for the metric function. 
210   
211      include_self : bool or 'auto', default=False 
212          Whether or not to mark each sample as the first nearest neighbor to 
213          itself. If 'auto', then True is used for mode='connectivity' and False 
214          for mode='distance'. 
215   
216      n_jobs : int, default=None 
217          The number of parallel jobs to run for neighbors search. 
218          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
219          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
220          for more details. 
221   
222      Returns 
223      ------- 
224      A : sparse matrix of shape (n_samples, n_samples) 
225          Graph where A[i, j] is assigned the weight of edge that connects 
226          i to j. The matrix is of CSR format. 
227   
228      See Also 
229      -------- 
230      kneighbors_graph: Compute the weighted graph of k-neighbors for points in X. 
231   
232      Examples 
233      -------- 
234      >>> X = [[0], [3], [1]] 
235      >>> from sklearn.neighbors import radius_neighbors_graph 
236      >>> A = radius_neighbors_graph(X, 1.5, mode='connectivity', 
237      ...                            include_self=True) 
238      >>> A.toarray() 
239      array([[1., 0., 1.], 
240             [0., 1., 0.], 
241             [1., 0., 1.]]) 
242      """
243      if not isinstance(X, RadiusNeighborsMixin):
244          X = NearestNeighbors(
245              radius=radius,
246              metric=metric,
247              p=p,
248              metric_params=metric_params,
249              n_jobs=n_jobs,
250          ).fit(X)
251      else:
252          _check_params(X, metric, p, metric_params)
253  
254      query = _query_include_self(X._fit_X, include_self, mode)
255      return X.radius_neighbors_graph(query, radius, mode)

258  class KNeighborsTransformer(
259      ClassNamePrefixFeaturesOutMixin, KNeighborsMixin, TransformerMixin, NeighborsBase
260  ):
261      """Transform X into a (weighted) graph of k nearest neighbors. 
262   
263      The transformed data is a sparse graph as returned by kneighbors_graph. 
264   
265      Read more in the :ref:`User Guide <neighbors_transformer>`. 
266   
267      .. versionadded:: 0.22 
268   
269      Parameters 
270      ---------- 
271      mode : {'distance', 'connectivity'}, default='distance' 
272          Type of returned matrix: 'connectivity' will return the connectivity 
273          matrix with ones and zeros, and 'distance' will return the distances 
274          between neighbors according to the given metric. 
275   
276      n_neighbors : int, default=5 
277          Number of neighbors for each sample in the transformed sparse graph. 
278          For compatibility reasons, as each sample is considered as its own 
279          neighbor, one extra neighbor will be computed when mode == 'distance'. 
280          In this case, the sparse graph contains (n_neighbors + 1) neighbors. 
281   
282      algorithm : {'auto', 'ball_tree', 'kd_tree', 'brute'}, default='auto' 
283          Algorithm used to compute the nearest neighbors: 
284   
285          - 'ball_tree' will use :class:`BallTree` 
286          - 'kd_tree' will use :class:`KDTree` 
287          - 'brute' will use a brute-force search. 
288          - 'auto' will attempt to decide the most appropriate algorithm 
289            based on the values passed to :meth:`fit` method. 
290   
291          Note: fitting on sparse input will override the setting of 
292          this parameter, using brute force. 
293   
294      leaf_size : int, default=30 
295          Leaf size passed to BallTree or KDTree.  This can affect the 
296          speed of the construction and query, as well as the memory 
297          required to store the tree.  The optimal value depends on the 
298          nature of the problem. 
299   
300      metric : str or callable, default='minkowski' 
301          Metric to use for distance computation. Default is "minkowski", which 
302          results in the standard Euclidean distance when p = 2. See the 
303          documentation of `scipy.spatial.distance 
304          <https://docs.scipy.org/doc/scipy/reference/spatial.distance.html>`_ and 
305          the metrics listed in 
306          :class:`~sklearn.metrics.pairwise.distance_metrics` for valid metric 
307          values. 
308   
309          If metric is a callable function, it takes two arrays representing 1D 
310          vectors as inputs and must return one value indicating the distance 
311          between those vectors. This works for Scipy's metrics, but is less 
312          efficient than passing the metric name as a string. 
313   
314          Distance matrices are not supported. 
315   
316      p : float, default=2 
317          Parameter for the Minkowski metric from 
318          sklearn.metrics.pairwise.pairwise_distances. When p = 1, this is 
319          equivalent to using manhattan_distance (l1), and euclidean_distance 
320          (l2) for p = 2. For arbitrary p, minkowski_distance (l_p) is used. 
321          This parameter is expected to be positive. 
322   
323      metric_params : dict, default=None 
324          Additional keyword arguments for the metric function. 
325   
326      n_jobs : int, default=None 
327          The number of parallel jobs to run for neighbors search. 
328          If ``-1``, then the number of jobs is set to the number of CPU cores. 
329   
330      Attributes 
331      ---------- 
332      effective_metric_ : str or callable 
333          The distance metric used. It will be same as the `metric` parameter 
334          or a synonym of it, e.g. 'euclidean' if the `metric` parameter set to 
335          'minkowski' and `p` parameter set to 2. 
336   
337      effective_metric_params_ : dict 
338          Additional keyword arguments for the metric function. For most metrics 
339          will be same with `metric_params` parameter, but may also contain the 
340          `p` parameter value if the `effective_metric_` attribute is set to 
341          'minkowski'. 
342   
343      n_features_in_ : int 
344          Number of features seen during :term:`fit`. 
345   
346          .. versionadded:: 0.24 
347   
348      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
349          Names of features seen during :term:`fit`. Defined only when `X` 
350          has feature names that are all strings. 
351   
352          .. versionadded:: 1.0 
353   
354      n_samples_fit_ : int 
355          Number of samples in the fitted data. 
356   
357      See Also 
358      -------- 
359      kneighbors_graph : Compute the weighted graph of k-neighbors for 
360          points in X. 
361      RadiusNeighborsTransformer : Transform X into a weighted graph of 
362          neighbors nearer than a radius. 
363   
364      Notes 
365      ----- 
366      For an example of using :class:`~sklearn.neighbors.KNeighborsTransformer` 
367      in combination with :class:`~sklearn.manifold.TSNE` see 
368      :ref:`sphx_glr_auto_examples_neighbors_approximate_nearest_neighbors.py`. 
369   
370      Examples 
371      -------- 
372      >>> from sklearn.datasets import load_wine 
373      >>> from sklearn.neighbors import KNeighborsTransformer 
374      >>> X, _ = load_wine(return_X_y=True) 
375      >>> X.shape 
376      (178, 13) 
377      >>> transformer = KNeighborsTransformer(n_neighbors=5, mode='distance') 
378      >>> X_dist_graph = transformer.fit_transform(X) 
379      >>> X_dist_graph.shape 
380      (178, 178) 
381      """
382  
383      _parameter_constraints: dict = {
384          **NeighborsBase._parameter_constraints,
385          "mode": [StrOptions({"distance", "connectivity"})],
386      }
387      _parameter_constraints.pop("radius")
388  
389      def __init__(
390          self,
391          *,
392          mode="distance",
393          n_neighbors=5,
394          algorithm="auto",
395          leaf_size=30,
396          metric="minkowski",
397          p=2,
398          metric_params=None,
399          n_jobs=None,
400      ):
401          super(KNeighborsTransformer, self).__init__(
402              n_neighbors=n_neighbors,
403              radius=None,
404              algorithm=algorithm,
405              leaf_size=leaf_size,
406              metric=metric,
407              p=p,
408              metric_params=metric_params,
409              n_jobs=n_jobs,
410          )
411          self.mode = mode
412  
413      @_fit_context(
414          # KNeighborsTransformer.metric is not validated yet
415          prefer_skip_nested_validation=False
416      )
417      def fit(self, X, y=None):
418          """Fit the k-nearest neighbors transformer from the training dataset. 
419   
420          Parameters 
421          ---------- 
422          X : {array-like, sparse matrix} of shape (n_samples, n_features) or \ 
423                  (n_samples, n_samples) if metric='precomputed' 
424              Training data. 
425          y : Ignored 
426              Not used, present for API consistency by convention. 
427   
428          Returns 
429          ------- 
430          self : KNeighborsTransformer 
431              The fitted k-nearest neighbors transformer. 
432          """
433          self._fit(X)
434          self._n_features_out = self.n_samples_fit_
435          return self
436  
437      def transform(self, X):
438          """Compute the (weighted) graph of Neighbors for points in X. 
439   
440          Parameters 
441          ---------- 
442          X : array-like of shape (n_samples_transform, n_features) 
443              Sample data. 
444   
445          Returns 
446          ------- 
447          Xt : sparse matrix of shape (n_samples_transform, n_samples_fit) 
448              Xt[i, j] is assigned the weight of edge that connects i to j. 
449              Only the neighbors have an explicit value. 
450              The diagonal is always explicit. 
451              The matrix is of CSR format. 
452          """
453          check_is_fitted(self)
454          add_one = self.mode == "distance"
455          return self.kneighbors_graph(
456              X, mode=self.mode, n_neighbors=self.n_neighbors + add_one
457          )
458  
459      def fit_transform(self, X, y=None):
460          """Fit to data, then transform it. 
461   
462          Fits transformer to X and y with optional parameters fit_params 
463          and returns a transformed version of X. 
464   
465          Parameters 
466          ---------- 
467          X : array-like of shape (n_samples, n_features) 
468              Training set. 
469   
470          y : Ignored 
471              Not used, present for API consistency by convention. 
472   
473          Returns 
474          ------- 
475          Xt : sparse matrix of shape (n_samples, n_samples) 
476              Xt[i, j] is assigned the weight of edge that connects i to j. 
477              Only the neighbors have an explicit value. 
478              The diagonal is always explicit. 
479              The matrix is of CSR format. 
480          """
481          return self.fit(X).transform(X)

680          return self.radius_neighbors_graph(X, mode=self.mode, sort_results=True)
681  
682      def fit_transform(self, X, y=None):
683          """Fit to data, then transform it. 
684   
685          Fits transformer to X and y with optional parameters fit_params 
686          and returns a transformed version of X. 
687   
688          Parameters 
689          ---------- 
690          X : array-like of shape (n_samples, n_features) 
691              Training set. 
692   
693          y : Ignored 
694              Not used, present for API consistency by convention. 
695   
696          Returns 
697          ------- 
698          Xt : sparse matrix of shape (n_samples, n_samples) 
699              Xt[i, j] is assigned the weight of edge that connects i to j. 
700              Only the neighbors have an explicit value. 
701              The diagonal is always explicit. 
702              The matrix is of CSR format. 
703          """
704          return self.fit(X).transform(X)
