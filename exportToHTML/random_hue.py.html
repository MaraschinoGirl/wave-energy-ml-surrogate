7    @keras_export("keras.layers.RandomHue")
8    class RandomHue(BaseImagePreprocessingLayer):
9        """Randomly adjusts the hue on given images. 
10    
11       This layer will randomly increase/reduce the hue for the input RGB 
12       images. 
13    
14       The image hue is adjusted by converting the image(s) to HSV and rotating the 
15       hue channel (H) by delta. The image is then converted back to RGB. 
16    
17       Args: 
18           factor: A single float or a tuple of two floats. 
19               `factor` controls the extent to which the 
20               image hue is impacted. `factor=0.0` makes this layer perform a 
21               no-op operation, while a value of `1.0` performs the most aggressive 
22               contrast adjustment available. If a tuple is used, a `factor` is 
23               sampled between the two values for every image augmented. If a 
24               single float is used, a value between `0.0` and the passed float is 
25               sampled. In order to ensure the value is always the same, please 
26               pass a tuple with two identical floats: `(0.5, 0.5)`. 
27           value_range: the range of values the incoming images will have. 
28               Represented as a two-number tuple written `[low, high]`. This is 
29               typically either `[0, 1]` or `[0, 255]` depending on how your 
30               preprocessing pipeline is set up. 
31           seed: Integer. Used to create a random seed. 
32    
33       Example: 
34    
35       ```python 
36       (images, labels), _ = keras.datasets.cifar10.load_data() 
37       random_hue = keras.layers.RandomHue(factor=0.5, value_range=[0, 1]) 
38       images = keras.ops.cast(images, "float32") 
39       augmented_images_batch = random_hue(images[:8]) 
40       ``` 
41       """
42   
43       _USE_BASE_FACTOR = True
44       _FACTOR_BOUNDS = (0, 1)
45   
46       def __init__(
47           self,
48           factor,
49           value_range=(0, 255),
50           data_format=None,
51           seed=None,
52           **kwargs,
53       ):
54           super().__init__(data_format=data_format, **kwargs)
55           self._set_factor(factor)
56           self.value_range = value_range
57           self.seed = seed
58           self.generator = self.backend.random.SeedGenerator(seed)
59   
60       def get_random_transformation(self, data, training=True, seed=None):
61           if isinstance(data, dict):
62               images = data["images"]
63           else:
64               images = data
65           images_shape = self.backend.shape(images)
66           rank = len(images_shape)
67           if rank == 3:
68               batch_size = 1
69           elif rank == 4:
70               batch_size = images_shape[0]
71           else:
72               raise ValueError(
73                   "Expected the input image to be rank 3 or 4. Received "
74                   f"inputs.shape={images_shape}"
75               )
76   
77           if seed is None:
78               seed = self._get_seed_generator(self.backend._backend)
79           invert = self.backend.random.uniform((batch_size,), seed=seed)
80   
81           invert = self.backend.numpy.where(
82               invert > 0.5,
83               -self.backend.numpy.ones_like(invert),
84               self.backend.numpy.ones_like(invert),
85           )
86           factor = self.backend.random.uniform(
87               (batch_size,),
88               minval=self.factor[0],
89               maxval=self.factor[1],
90               seed=seed,
91           )
92           return {"factor": invert * factor * 0.5}
93   
94       def transform_images(self, images, transformation=None, training=True):
95           def _apply_random_hue(images, transformation):
96               images = self.backend.cast(images, self.compute_dtype)
97               images = self._transform_value_range(
98                   images, self.value_range, (0, 1)
99               )
100              adjust_factors = transformation["factor"]
101              adjust_factors = self.backend.cast(adjust_factors, images.dtype)
102              adjust_factors = self.backend.numpy.expand_dims(adjust_factors, -1)
103              adjust_factors = self.backend.numpy.expand_dims(adjust_factors, -1)
104              images = self.backend.image.rgb_to_hsv(
105                  images, data_format=self.data_format
106              )
107              if self.data_format == "channels_first":
108                  h_channel = images[:, 0, :, :] + adjust_factors
109                  h_channel = self.backend.numpy.where(
110                      h_channel > 1.0, h_channel - 1.0, h_channel
111                  )
112                  h_channel = self.backend.numpy.where(
113                      h_channel < 0.0, h_channel + 1.0, h_channel
114                  )
115                  images = self.backend.numpy.stack(
116                      [h_channel, images[:, 1, :, :], images[:, 2, :, :]], axis=1
117                  )
118              else:
119                  h_channel = images[..., 0] + adjust_factors
120                  h_channel = self.backend.numpy.where(
121                      h_channel > 1.0, h_channel - 1.0, h_channel
122                  )
123                  h_channel = self.backend.numpy.where(
124                      h_channel < 0.0, h_channel + 1.0, h_channel
125                  )
126                  images = self.backend.numpy.stack(
127                      [h_channel, images[..., 1], images[..., 2]], axis=-1
128                  )
129              images = self.backend.image.hsv_to_rgb(
130                  images, data_format=self.data_format
131              )
132              images = self.backend.numpy.clip(images, 0, 1)
133              images = self._transform_value_range(
134                  images, (0, 1), self.value_range
135              )
136              images = self.backend.cast(images, self.compute_dtype)
137              return images
138  
139          if training:
140              images = _apply_random_hue(images, transformation)
141          return images
