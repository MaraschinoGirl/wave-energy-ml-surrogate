17   def logsumexp(a, axis=None, b=None, keepdims=False, return_sign=False):
18       """Compute the log of the sum of exponentials of input elements. 
19    
20       Parameters 
21       ---------- 
22       a : array_like 
23           Input array. 
24       axis : None or int or tuple of ints, optional 
25           Axis or axes over which the sum is taken. By default `axis` is None, 
26           and all elements are summed. 
27    
28           .. versionadded:: 0.11.0 
29       b : array-like, optional 
30           Scaling factor for exp(`a`) must be of the same shape as `a` or 
31           broadcastable to `a`. These values may be negative in order to 
32           implement subtraction. 
33    
34           .. versionadded:: 0.12.0 
35       keepdims : bool, optional 
36           If this is set to True, the axes which are reduced are left in the 
37           result as dimensions with size one. With this option, the result 
38           will broadcast correctly against the original array. 
39    
40           .. versionadded:: 0.15.0 
41       return_sign : bool, optional 
42           If this is set to True, the result will be a pair containing sign 
43           information; if False, results that are negative will be returned 
44           as NaN. Default is False (no sign information). 
45    
46           .. versionadded:: 0.16.0 
47    
48       Returns 
49       ------- 
50       res : ndarray 
51           The result, ``np.log(np.sum(np.exp(a)))`` calculated in a numerically 
52           more stable way. If `b` is given then ``np.log(np.sum(b*np.exp(a)))`` 
53           is returned. If ``return_sign`` is True, ``res`` contains the log of 
54           the absolute value of the argument. 
55       sgn : ndarray 
56           If ``return_sign`` is True, this will be an array of floating-point 
57           numbers matching res containing +1, 0, -1 (for real-valued inputs) 
58           or a complex phase (for complex inputs). This gives the sign of the 
59           argument of the logarithm in ``res``. 
60           If ``return_sign`` is False, only one result is returned. 
61    
62       See Also 
63       -------- 
64       numpy.logaddexp, numpy.logaddexp2 
65    
66       Notes 
67       ----- 
68       NumPy has a logaddexp function which is very similar to `logsumexp`, but 
69       only handles two arguments. `logaddexp.reduce` is similar to this 
70       function, but may be less stable. 
71    
72       The logarithm is a multivalued function: for each :math:`x` there is an 
73       infinite number of :math:`z` such that :math:`exp(z) = x`. The convention 
74       is to return the :math:`z` whose imaginary part lies in :math:`(-pi, pi]`. 
75    
76       Examples 
77       -------- 
78       >>> import numpy as np 
79       >>> from scipy.special import logsumexp 
80       >>> a = np.arange(10) 
81       >>> logsumexp(a) 
82       9.4586297444267107 
83       >>> np.log(np.sum(np.exp(a))) 
84       9.4586297444267107 
85    
86       With weights 
87    
88       >>> a = np.arange(10) 
89       >>> b = np.arange(10, 0, -1) 
90       >>> logsumexp(a, b=b) 
91       9.9170178533034665 
92       >>> np.log(np.sum(b*np.exp(a))) 
93       9.9170178533034647 
94    
95       Returning a sign flag 
96    
97       >>> logsumexp([1,2],b=[1,-1],return_sign=True) 
98       (1.5413248546129181, -1.0) 
99    
100      Notice that `logsumexp` does not directly support masked arrays. To use it 
101      on a masked array, convert the mask into zero weights: 
102   
103      >>> a = np.ma.array([np.log(2), 2, np.log(3)], 
104      ...                  mask=[False, True, False]) 
105      >>> b = (~a.mask).astype(int) 
106      >>> logsumexp(a.data, b=b), np.log(5) 
107      1.6094379124341005, 1.6094379124341005 
108   
109      """
110      xp = array_namespace(a, b)
111      a, b = xp_broadcast_promote(a, b, ensure_writeable=True, force_floating=True, xp=xp)
112      a = xpx.atleast_nd(a, ndim=1, xp=xp)
113      b = xpx.atleast_nd(b, ndim=1, xp=xp) if b is not None else b
114      axis = tuple(range(a.ndim)) if axis is None else axis
115  
116      if xp_size(a) != 0:
117          with np.errstate(divide='ignore', invalid='ignore'):  # log of zero is OK
118              out, sgn = _logsumexp(a, b, axis=axis, return_sign=return_sign, xp=xp)
119      else:
120          shape = np.asarray(a.shape)  # NumPy is convenient for shape manipulation
121          shape[axis] = 1
122          out = xp.full(tuple(shape), -xp.inf, dtype=a.dtype)
123          sgn = xp.sign(out)
124  
125      if xp.isdtype(out.dtype, 'complex floating'):
126          if return_sign:
127              real = xp.real(sgn)
128              imag = xp_float_to_complex(_wrap_radians(xp.imag(sgn), xp))
129              sgn = real + imag*1j
130          else:
131              real = xp.real(out)
132              imag = xp_float_to_complex(_wrap_radians(xp.imag(out), xp))
133              out = real + imag*1j
134  
135      # Deal with shape details - reducing dimensions and convert 0-D to scalar for NumPy
136      out = xp.squeeze(out, axis=axis) if not keepdims else out
137      sgn = xp.squeeze(sgn, axis=axis) if (sgn is not None and not keepdims) else sgn
138      out = out[()] if out.ndim == 0 else out
139      sgn = sgn[()] if (sgn is not None and sgn.ndim == 0) else sgn
140  
141      return (out, sgn) if return_sign else out
142

144  def _wrap_radians(x, xp=None):
145      xp = array_namespace(x) if xp is None else xp
146      # Wrap radians to (-pi, pi] interval
147      out = -((-x + math.pi) % (2 * math.pi) - math.pi)
148      # preserve relative precision
149      no_wrap = xp.abs(x) < xp.pi
150      out[no_wrap] = x[no_wrap]
151      return out

154  def _elements_and_indices_with_max_real(a, axis=-1, xp=None):
155      # This is an array-API compatible `max` function that works something
156      # like `np.max` for complex input. The important part is that it finds
157      # the element with maximum real part. When there are multiple complex values
158      # with this real part, it doesn't matter which we choose.
159      # We could use `argmax` on real component, but array API doesn't yet have
160      # `take_along_axis`, and even if it did, we would have problems with axis tuples.
161      # Feel free to rewrite! It's ugly, but it's not the purpose of the PR, and
162      # it gets the job done.
163      xp = array_namespace(a) if xp is None else xp
164  
165      if xp.isdtype(a.dtype, "complex floating"):
166          # select all elements with max real part.
167          real_a = xp.real(a)
168          max = xp.max(real_a, axis=axis, keepdims=True)
169          mask = real_a == max
170  
171          # Of those, choose one arbitrarily. This is a reasonably
172          # simple, array-API compatible way of doing so that doesn't
173          # have a problem with `axis` being a tuple or None.
174          i = xp.reshape(xp.arange(xp_size(a)), a.shape)
175          i[~mask] = -1
176          max_i = xp.max(i, axis=axis, keepdims=True)
177          mask = i == max_i
178          a = xp_copy(a)
179          a[~mask] = 0
180          max = xp.sum(a, axis=axis, dtype=a.dtype, keepdims=True)
181      else:
182          max = xp.max(a, axis=axis, keepdims=True)
183          mask = a == max
184  
185      return xp.asarray(max), xp.asarray(mask)

188  def _sign(x, xp):
189      return x / xp.where(x == 0, xp.asarray(1, dtype=x.dtype), xp.abs(x))

192  def _logsumexp(a, b, axis, return_sign, xp):
193  
194      # This has been around for about a decade, so let's consider it a feature:
195      # Even if element of `a` is infinite or NaN, it adds nothing to the sum if
196      # the corresponding weight is zero.
197      if b is not None:
198          a[b == 0] = -xp.inf
199  
200      # Find element with maximum real part, since this is what affects the magnitude
201      # of the exponential. Possible enhancement: include log of `b` magnitude in `a`.
202      a_max, i_max = _elements_and_indices_with_max_real(a, axis=axis, xp=xp)
203  
204      # for precision, these terms are separated out of the main sum.
205      a[i_max] = -xp.inf
206      i_max_dt = xp.astype(i_max, a.dtype)
207      # This is an inefficient way of getting `m` because it is the sum of a sparse
208      # array; however, this is the simplest way I can think of to get the right shape.
209      m = (xp.sum(i_max_dt, axis=axis, keepdims=True, dtype=a.dtype) if b is None
210           else xp.sum(b * i_max_dt, axis=axis, keepdims=True, dtype=a.dtype))
211  
212      # Arithmetic between infinities will introduce NaNs.
213      # `+ a_max` at the end naturally corrects for removing them here.
214      shift = xp.where(xp.isfinite(a_max), a_max, xp.asarray(0, dtype=a_max.dtype))
215  
216      # Shift, exponentiate, scale, and sum
217      exp = b * xp.exp(a - shift) if b is not None else xp.exp(a - shift)
218      s = xp.sum(exp, axis=axis, keepdims=True, dtype=exp.dtype)
219      s = xp.where(s == 0, s, s/m)
220  
221      # Separate sign/magnitude information
222      sgn = None
223      if return_sign:
224          # Use the numpy>=2.0 convention for sign.
225          # When all array libraries agree, this can become sng = xp.sign(s).
226          sgn = _sign(s + 1, xp=xp) * _sign(m, xp=xp)
227  
228          if xp.isdtype(s.dtype, "real floating"):
229              # The log functions need positive arguments
230              s = xp.where(s < -1, -s - 2, s)
231              m = xp.abs(m)
232          else:
233              # `a_max` can have a sign component for complex input
234              j = xp.asarray(1j, dtype=a_max.dtype)
235              sgn = sgn * xp.exp(xp.imag(a_max) * j)
236  
237      # Take log and undo shift
238      out = xp.log1p(s) + xp.log(m) + a_max
239  
240      out = xp_real(out) if return_sign else out
241  
242      return out, sgn

245  def softmax(x, axis=None):
246      r"""Compute the softmax function. 
247   
248      The softmax function transforms each element of a collection by 
249      computing the exponential of each element divided by the sum of the 
250      exponentials of all the elements. That is, if `x` is a one-dimensional 
251      numpy array:: 
252   
253          softmax(x) = np.exp(x)/sum(np.exp(x)) 
254   
255      Parameters 
256      ---------- 
257      x : array_like 
258          Input array. 
259      axis : int or tuple of ints, optional 
260          Axis to compute values along. Default is None and softmax will be 
261          computed over the entire array `x`. 
262   
263      Returns 
264      ------- 
265      s : ndarray 
266          An array the same shape as `x`. The result will sum to 1 along the 
267          specified axis. 
268   
269      Notes 
270      ----- 
271      The formula for the softmax function :math:`\sigma(x)` for a vector 
272      :math:`x = \{x_0, x_1, ..., x_{n-1}\}` is 
273   
274      .. math:: \sigma(x)_j = \frac{e^{x_j}}{\sum_k e^{x_k}} 
275   
276      The `softmax` function is the gradient of `logsumexp`. 
277   
278      The implementation uses shifting to avoid overflow. See [1]_ for more 
279      details. 
280   
281      .. versionadded:: 1.2.0 
282   
283      References 
284      ---------- 
285      .. [1] P. Blanchard, D.J. Higham, N.J. Higham, "Accurately computing the 
286         log-sum-exp and softmax functions", IMA Journal of Numerical Analysis, 
287         Vol.41(4), :doi:`10.1093/imanum/draa038`. 
288   
289      Examples 
290      -------- 
291      >>> import numpy as np 
292      >>> from scipy.special import softmax 
293      >>> np.set_printoptions(precision=5) 
294   
295      >>> x = np.array([[1, 0.5, 0.2, 3], 
296      ...               [1,  -1,   7, 3], 
297      ...               [2,  12,  13, 3]]) 
298      ... 
299   
300      Compute the softmax transformation over the entire array. 
301   
302      >>> m = softmax(x) 
303      >>> m 
304      array([[  4.48309e-06,   2.71913e-06,   2.01438e-06,   3.31258e-05], 
305             [  4.48309e-06,   6.06720e-07,   1.80861e-03,   3.31258e-05], 
306             [  1.21863e-05,   2.68421e-01,   7.29644e-01,   3.31258e-05]]) 
307   
308      >>> m.sum() 
309      1.0 
310   
311      Compute the softmax transformation along the first axis (i.e., the 
312      columns). 
313   
314      >>> m = softmax(x, axis=0) 
315   
316      >>> m 
317      array([[  2.11942e-01,   1.01300e-05,   2.75394e-06,   3.33333e-01], 
318             [  2.11942e-01,   2.26030e-06,   2.47262e-03,   3.33333e-01], 
319             [  5.76117e-01,   9.99988e-01,   9.97525e-01,   3.33333e-01]]) 
320   
321      >>> m.sum(axis=0) 
322      array([ 1.,  1.,  1.,  1.]) 
323   
324      Compute the softmax transformation along the second axis (i.e., the rows). 
325   
326      >>> m = softmax(x, axis=1) 
327      >>> m 
328      array([[  1.05877e-01,   6.42177e-02,   4.75736e-02,   7.82332e-01], 
329             [  2.42746e-03,   3.28521e-04,   9.79307e-01,   1.79366e-02], 
330             [  1.22094e-05,   2.68929e-01,   7.31025e-01,   3.31885e-05]]) 
331   
332      >>> m.sum(axis=1) 
333      array([ 1.,  1.,  1.]) 
334   
335      """
336      x = _asarray_validated(x, check_finite=False)
337      x_max = np.amax(x, axis=axis, keepdims=True)
338      exp_x_shifted = np.exp(x - x_max)
339      return exp_x_shifted / np.sum(exp_x_shifted, axis=axis, keepdims=True)

342  def log_softmax(x, axis=None):
343      r"""Compute the logarithm of the softmax function. 
344   
345      In principle:: 
346   
347          log_softmax(x) = log(softmax(x)) 
348   
349      but using a more accurate implementation. 
350   
351      Parameters 
352      ---------- 
353      x : array_like 
354          Input array. 
355      axis : int or tuple of ints, optional 
356          Axis to compute values along. Default is None and softmax will be 
357          computed over the entire array `x`. 
358   
359      Returns 
360      ------- 
361      s : ndarray or scalar 
362          An array with the same shape as `x`. Exponential of the result will 
363          sum to 1 along the specified axis. If `x` is a scalar, a scalar is 
364          returned. 
365   
366      Notes 
367      ----- 
368      `log_softmax` is more accurate than ``np.log(softmax(x))`` with inputs that 
369      make `softmax` saturate (see examples below). 
370   
371      .. versionadded:: 1.5.0 
372   
373      Examples 
374      -------- 
375      >>> import numpy as np 
376      >>> from scipy.special import log_softmax 
377      >>> from scipy.special import softmax 
378      >>> np.set_printoptions(precision=5) 
379   
380      >>> x = np.array([1000.0, 1.0]) 
381   
382      >>> y = log_softmax(x) 
383      >>> y 
384      array([   0., -999.]) 
385   
386      >>> with np.errstate(divide='ignore'): 
387      ...   y = np.log(softmax(x)) 
388      ... 
389      >>> y 
390      array([  0., -inf]) 
391   
392      """
393  
394      x = _asarray_validated(x, check_finite=False)
395  
396      x_max = np.amax(x, axis=axis, keepdims=True)
397  
398      if x_max.ndim > 0:
399          x_max[~np.isfinite(x_max)] = 0
400      elif not np.isfinite(x_max):
401          x_max = 0
402  
403      tmp = x - x_max
404      exp_tmp = np.exp(tmp)
405  
406      # suppress warnings about log of zero
407      with np.errstate(divide='ignore'):
408          s = np.sum(exp_tmp, axis=axis, keepdims=True)
409          out = np.log(s)
410  
411      out = tmp - out
412      return out
