8    @keras_export(["keras.Regularizer", "keras.regularizers.Regularizer"])
9    class Regularizer:
10       """Regularizer base class. 
11    
12       Regularizers allow you to apply penalties on layer parameters or layer 
13       activity during optimization. These penalties are summed into the loss 
14       function that the network optimizes. 
15    
16       Regularization penalties are applied on a per-layer basis. The exact API 
17       will depend on the layer, but many layers (e.g. `Dense`, `Conv1D`, `Conv2D` 
18       and `Conv3D`) have a unified API. 
19    
20       These layers expose 3 keyword arguments: 
21    
22       - `kernel_regularizer`: Regularizer to apply a penalty on the layer's kernel 
23       - `bias_regularizer`: Regularizer to apply a penalty on the layer's bias 
24       - `activity_regularizer`: Regularizer to apply a penalty on the layer's 
25           output 
26    
27       All layers (including custom layers) expose `activity_regularizer` as a 
28       settable property, whether or not it is in the constructor arguments. 
29    
30       The value returned by the `activity_regularizer` is divided by the input 
31       batch size so that the relative weighting between the weight regularizers 
32       and the activity regularizers does not change with the batch size. 
33    
34       You can access a layer's regularization penalties by calling `layer.losses` 
35       after calling the layer on inputs. 
36    
37       ## Example 
38    
39       >>> layer = Dense( 
40       ...     5, input_dim=5, 
41       ...     kernel_initializer='ones', 
42       ...     kernel_regularizer=L1(0.01), 
43       ...     activity_regularizer=L2(0.01)) 
44       >>> tensor = ops.ones(shape=(5, 5)) * 2.0 
45       >>> out = layer(tensor) 
46    
47       >>> # The kernel regularization term is 0.25 
48       >>> # The activity regularization term (after dividing by the batch size) 
49       >>> # is 5 
50       >>> ops.sum(layer.losses) 
51       5.25 
52    
53       ## Available penalties 
54    
55       ```python 
56       L1(0.3)  # L1 Regularization Penalty 
57       L2(0.1)  # L2 Regularization Penalty 
58       L1L2(l1=0.01, l2=0.01)  # L1 + L2 penalties 
59       ``` 
60    
61       ## Directly calling a regularizer 
62    
63       Compute a regularization loss on a tensor by directly calling a regularizer 
64       as if it is a one-argument function. 
65    
66       E.g. 
67    
68       >>> regularizer = L2(2.) 
69       >>> tensor = ops.ones(shape=(5, 5)) 
70       >>> regularizer(tensor) 
71       50.0 
72    
73       ## Developing new regularizers 
74    
75       Any function that takes in a weight matrix and returns a scalar 
76       tensor can be used as a regularizer, e.g.: 
77    
78       >>> def l1_reg(weight_matrix): 
79       ...    return 0.01 * ops.sum(ops.absolute(weight_matrix)) 
80       ... 
81       >>> layer = Dense(5, input_dim=5, 
82       ...     kernel_initializer='ones', kernel_regularizer=l1_reg) 
83       >>> tensor = ops.ones(shape=(5, 5)) 
84       >>> out = layer(tensor) 
85       >>> layer.losses 
86       0.25 
87    
88       Alternatively, you can write your custom regularizers in an 
89       object-oriented way by extending this regularizer base class, e.g.: 
90    
91       >>> class L2Regularizer(Regularizer): 
92       ...   def __init__(self, l2=0.): 
93       ...     self.l2 = l2 
94       ... 
95       ...   def __call__(self, x): 
96       ...     return self.l2 * ops.sum(ops.square(x)) 
97       ... 
98       ...   def get_config(self): 
99       ...     return {'l2': float(self.l2)} 
100      ... 
101      >>> layer = Dense( 
102      ...   5, input_dim=5, kernel_initializer='ones', 
103      ...   kernel_regularizer=L2Regularizer(l2=0.5)) 
104   
105      >>> tensor = ops.ones(shape=(5, 5)) 
106      >>> out = layer(tensor) 
107      >>> layer.losses 
108      12.5 
109   
110      ### A note on serialization and deserialization: 
111   
112      Registering the regularizers as serializable is optional if you are just 
113      training and executing models, exporting to and from SavedModels, or saving 
114      and loading weight checkpoints. 
115   
116      Registration is required for saving and 
117      loading models to HDF5 format, Keras model cloning, some visualization 
118      utilities, and exporting models to and from JSON. If using this 
119      functionality, you must make sure any python process running your model has 
120      also defined and registered your custom regularizer. 
121      """
122  
123      def __call__(self, x):
124          """Compute a regularization penalty from an input tensor."""
125          return 0.0

138

141

168  @keras_export(["keras.regularizers.L1L2", "keras.regularizers.l1_l2"])
169  class L1L2(Regularizer):
170      """A regularizer that applies both L1 and L2 regularization penalties. 
171   
172      The L1 regularization penalty is computed as: 
173      `loss = l1 * reduce_sum(abs(x))` 
174   
175      The L2 regularization penalty is computed as 
176      `loss = l2 * reduce_sum(square(x))` 
177   
178      L1L2 may be passed to a layer as a string identifier: 
179   
180      >>> dense = Dense(3, kernel_regularizer='l1_l2') 
181   
182      In this case, the default values used are `l1=0.01` and `l2=0.01`. 
183   
184      Arguments: 
185          l1: float, L1 regularization factor. 
186          l2: float, L2 regularization factor. 
187      """
188  
189      def __init__(self, l1=0.0, l2=0.0):
190          # The default value for l1 and l2 are different from the value in l1_l2
191          # for backward compatibility reason. Eg, L1L2(l2=0.1) will only have l2
192          # and no l1 penalty.
193          l1 = 0.0 if l1 is None else l1
194          l2 = 0.0 if l2 is None else l2
195          validate_float_arg(l1, name="l1")
196          validate_float_arg(l2, name="l2")
197  
198          self.l1 = l1
199          self.l2 = l2
200  
201      def __call__(self, x):
202          regularization = ops.convert_to_tensor(0.0, dtype=x.dtype)
203          if self.l1:
204              regularization += self.l1 * ops.sum(ops.absolute(x))
205          if self.l2:
206              regularization += self.l2 * ops.sum(ops.square(x))
207          return regularization
208  
209      def get_config(self):
210          return {"l1": float(self.l1), "l2": float(self.l2)}

271  @keras_export(
272      [
273          "keras.regularizers.OrthogonalRegularizer",
274          "keras.regularizers.orthogonal_regularizer",
275      ]
276  )
277  class OrthogonalRegularizer(Regularizer):
278      """Regularizer that encourages input vectors to be orthogonal to each other. 
279   
280      It can be applied to either the rows of a matrix (`mode="rows"`) or its 
281      columns (`mode="columns"`). When applied to a `Dense` kernel of shape 
282      `(input_dim, units)`, rows mode will seek to make the feature vectors 
283      (i.e. the basis of the output space) orthogonal to each other. 
284   
285      Arguments: 
286          factor: Float. The regularization factor. The regularization penalty 
287              will be proportional to `factor` times the mean of the dot products 
288              between the L2-normalized rows (if `mode="rows"`, or columns if 
289              `mode="columns"`) of the inputs, excluding the product of each 
290              row/column with itself.  Defaults to `0.01`. 
291          mode: String, one of `{"rows", "columns"}`. Defaults to `"rows"`. In 
292              rows mode, the regularization effect seeks to make the rows of the 
293              input orthogonal to each other. In columns mode, it seeks to make 
294              the columns of the input orthogonal to each other. 
295   
296      Example: 
297   
298      >>> regularizer = OrthogonalRegularizer(factor=0.01) 
299      >>> layer = Dense(units=4, kernel_regularizer=regularizer) 
300      """
301  
302      def __init__(self, factor=0.01, mode="rows"):
303          validate_float_arg(factor, name="factor")
304          self.factor = ops.convert_to_tensor(factor)
305          if mode not in {"rows", "columns"}:
306              raise ValueError(
307                  "Invalid value for argument `mode`. Expected one of "
308                  f'{{"rows", "columns"}}. Received: mode={mode}'
309              )
310          self.mode = mode
311  
312      def __call__(self, inputs):
313          if len(inputs.shape) != 2:
314              raise ValueError(
315                  "Inputs to OrthogonalRegularizer must have rank 2. Received: "
316                  f"inputs.shape={inputs.shape}"
317              )
318          if self.mode == "rows":
319              inputs = normalize(inputs, axis=1)
320              product = ops.matmul(inputs, ops.transpose(inputs))
321              size = inputs.shape[0]
322          else:
323              inputs = normalize(inputs, axis=0)
324              product = ops.matmul(ops.transpose(inputs), inputs)
325              size = inputs.shape[1]
326          product_no_diagonal = product * (
327              1.0 - ops.eye(size, dtype=inputs.dtype)
328          )
329          num_pairs = size * (size - 1.0) / 2.0
330          return (
331              self.factor
332              * 0.5
333              * ops.sum(ops.absolute(product_no_diagonal))
334              / num_pairs
335          )
336  
337      def get_config(self):
338          return {"factor": float(self.factor), "mode": self.mode}

341  def validate_float_arg(value, name):
342      """check penalty number availability, raise ValueError if failed."""
343      if (
344          not isinstance(value, (float, int))
345          or (math.isinf(value) or math.isnan(value))
346          or value < 0
347      ):
348          raise ValueError(
349              f"Invalid value for argument {name}: expected a non-negative float."
350              f"Received: {name}={value}"
351          )
352      return float(value)
