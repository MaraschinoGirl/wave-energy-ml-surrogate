4    import functools
5    import warnings
6    from inspect import signature
7    
8    __all__ = ["deprecated"]
9    
10   
11   class deprecated:
12       """Decorator to mark a function or class as deprecated. 
13    
14       Issue a warning when the function is called/the class is instantiated and 
15       adds a warning to the docstring. 
16    
17       The optional extra argument will be appended to the deprecation message 
18       and the docstring. Note: to use this with the default value for extra, put 
19       in an empty of parentheses: 
20    
21       Examples 
22       -------- 
23       >>> from sklearn.utils import deprecated 
24       >>> deprecated() 
25       <sklearn.utils.deprecation.deprecated object at ...> 
26       >>> @deprecated() 
27       ... def some_function(): pass 
28    
29       Parameters 
30       ---------- 
31       extra : str, default='' 
32             To be added to the deprecation messages. 
33       """
34   
35       # Adapted from https://wiki.python.org/moin/PythonDecoratorLibrary,
36       # but with many changes.
37   
38       def __init__(self, extra=""):
39           self.extra = extra
40   
41       def __call__(self, obj):
42           """Call method 
43    
44           Parameters 
45           ---------- 
46           obj : object 
47           """
48           if isinstance(obj, type):
49               return self._decorate_class(obj)
50           elif isinstance(obj, property):
51               # Note that this is only triggered properly if the `deprecated`
52               # decorator is placed before the `property` decorator, like so:
53               #
54               # @deprecated(msg)
55               # @property
56               # def deprecated_attribute_(self):
57               #     ...
58               return self._decorate_property(obj)
59           else:
60               return self._decorate_fun(obj)
61   
62       def _decorate_class(self, cls):
63           msg = "Class %s is deprecated" % cls.__name__
64           if self.extra:
65               msg += "; %s" % self.extra
66   
67           new = cls.__new__
68           sig = signature(cls)
69   
70           def wrapped(cls, *args, **kwargs):
71               warnings.warn(msg, category=FutureWarning)
72               if new is object.__new__:
73                   return object.__new__(cls)
74   
75               return new(cls, *args, **kwargs)
76   
77           cls.__new__ = wrapped
78   
79           wrapped.__name__ = "__new__"
80           wrapped.deprecated_original = new
81           # Restore the original signature, see PEP 362.
82           cls.__signature__ = sig
83   
84           return cls
85   
86       def _decorate_fun(self, fun):
87           """Decorate function fun"""
88   
89           msg = "Function %s is deprecated" % fun.__name__
90           if self.extra:
91               msg += "; %s" % self.extra
92   
93           @functools.wraps(fun)
94           def wrapped(*args, **kwargs):
95               warnings.warn(msg, category=FutureWarning)
96               return fun(*args, **kwargs)
97   
98           # Add a reference to the wrapped function so that we can introspect
99           # on function arguments in Python 2 (already works in Python 3)
100          wrapped.__wrapped__ = fun
101  
102          return wrapped
103  
104      def _decorate_property(self, prop):
105          msg = self.extra
106  
107          @property
108          @functools.wraps(prop.fget)
109          def wrapped(*args, **kwargs):
110              warnings.warn(msg, category=FutureWarning)
111              return prop.fget(*args, **kwargs)
112  
113          return wrapped
114  
115  
116  def _is_deprecated(func):
117      """Helper to check if func is wrapped by our deprecated decorator"""
118      closures = getattr(func, "__closure__", [])
119      if closures is None:
120          closures = []
121      is_deprecated = "deprecated" in "".join(
122          [c.cell_contents for c in closures if isinstance(c.cell_contents, str)]
123      )
124      return is_deprecated

127  # TODO: remove in 1.7
128  def _deprecate_Xt_in_inverse_transform(X, Xt):
129      """Helper to deprecate the `Xt` argument in favor of `X` in inverse_transform."""
130      if X is not None and Xt is not None:
131          raise TypeError("Cannot use both X and Xt. Use X only.")
132  
133      if X is None and Xt is None:
134          raise TypeError("Missing required positional argument: X.")
135  
136      if Xt is not None:
137          warnings.warn(
138              "Xt was renamed X in version 1.5 and will be removed in 1.7.",
139              FutureWarning,
140          )
141          return Xt
142  
143      return X

146  # TODO(1.8): remove force_all_finite and change the default value of ensure_all_finite
147  # to True (remove None without deprecation).
148  def _deprecate_force_all_finite(force_all_finite, ensure_all_finite):
149      """Helper to deprecate force_all_finite in favor of ensure_all_finite."""
150      if force_all_finite != "deprecated":
151          warnings.warn(
152              "'force_all_finite' was renamed to 'ensure_all_finite' in 1.6 and will be "
153              "removed in 1.8.",
154              FutureWarning,
155          )
156  
157          if ensure_all_finite is not None:
158              raise ValueError(
159                  "'force_all_finite' and 'ensure_all_finite' cannot be used together. "
160                  "Pass `ensure_all_finite` only."
161              )
162  
163          return force_all_finite
164  
165      if ensure_all_finite is None:
166          return True
167  
168      return ensure_all_finite
