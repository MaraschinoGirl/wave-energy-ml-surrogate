65   class SimplePipeline(BaseEstimator):
66       """A very simple pipeline, assuming the last step is always a predictor. 
67    
68       Parameters 
69       ---------- 
70       steps : iterable of objects 
71           An iterable of transformers with the last step being a predictor. 
72       """
73   
74       def __init__(self, steps):
75           self.steps = steps
76   
77       def fit(self, X, y, **fit_params):
78           self.steps_ = []
79           params = process_routing(self, "fit", **fit_params)
80           X_transformed = X
81           for i, step in enumerate(self.steps[:-1]):
82               transformer = clone(step).fit(
83                   X_transformed, y, **params.get(f"step_{i}").fit
84               )
85               self.steps_.append(transformer)
86               X_transformed = transformer.transform(
87                   X_transformed, **params.get(f"step_{i}").transform
88               )
89   
90           self.steps_.append(
91               clone(self.steps[-1]).fit(X_transformed, y, **params.predictor.fit)
92           )
93           return self
94   
95       def predict(self, X, **predict_params):
96           check_is_fitted(self)
97           X_transformed = X
98           params = process_routing(self, "predict", **predict_params)
99           for i, step in enumerate(self.steps_[:-1]):
100              X_transformed = step.transform(X, **params.get(f"step_{i}").transform)
101  
102          return self.steps_[-1].predict(X_transformed, **params.predictor.predict)
103  
104      def get_metadata_routing(self):
105          router = MetadataRouter(owner=self.__class__.__name__)
106          for i, step in enumerate(self.steps[:-1]):
107              router.add(
108                  **{f"step_{i}": step},
109                  method_mapping=MethodMapping()
110                  .add(caller="fit", callee="fit")
111                  .add(caller="fit", callee="transform")
112                  .add(caller="predict", callee="transform"),
113              )
114          router.add(
115              predictor=self.steps[-1],
116              method_mapping=MethodMapping()
117              .add(caller="fit", callee="fit")
118              .add(caller="predict", callee="predict"),
119          )
120          return router

299  @config_context(enable_metadata_routing=True)
300  def test_simple_metadata_routing():
301      # Tests that metadata is properly routed
302  
303      # The underlying estimator doesn't accept or request metadata
304      clf = WeightedMetaClassifier(estimator=NonConsumingClassifier())
305      clf.fit(X, y)
306  
307      # Meta-estimator consumes sample_weight, but doesn't forward it to the underlying
308      # estimator
309      clf = WeightedMetaClassifier(estimator=NonConsumingClassifier())
310      clf.fit(X, y, sample_weight=my_weights)
311  
312      # If the estimator accepts the metadata but doesn't explicitly say it doesn't
313      # need it, there's an error
314      clf = WeightedMetaClassifier(estimator=ConsumingClassifier())
315      err_message = (
316          "[sample_weight] are passed but are not explicitly set as requested or"
317          " not requested for ConsumingClassifier.fit"
318      )
319      with pytest.raises(ValueError, match=re.escape(err_message)):
320          clf.fit(X, y, sample_weight=my_weights)
321  
322      # Explicitly saying the estimator doesn't need it, makes the error go away,
323      # because in this case `WeightedMetaClassifier` consumes `sample_weight`. If
324      # there was no consumer of sample_weight, passing it would result in an
325      # error.
326      clf = WeightedMetaClassifier(
327          estimator=ConsumingClassifier().set_fit_request(sample_weight=False)
328      )
329      # this doesn't raise since WeightedMetaClassifier itself is a consumer,
330      # and passing metadata to the consumer directly is fine regardless of its
331      # metadata_request values.
332      clf.fit(X, y, sample_weight=my_weights)
333      check_recorded_metadata(clf.estimator_, method="fit", parent="fit")
334  
335      # Requesting a metadata will make the meta-estimator forward it correctly
336      clf = WeightedMetaClassifier(
337          estimator=ConsumingClassifier().set_fit_request(sample_weight=True)
338      )
339      clf.fit(X, y, sample_weight=my_weights)
340      check_recorded_metadata(
341          clf.estimator_, method="fit", parent="fit", sample_weight=my_weights
342      )
343  
344      # And requesting it with an alias
345      clf = WeightedMetaClassifier(
346          estimator=ConsumingClassifier().set_fit_request(
347              sample_weight="alternative_weight"
348          )
349      )
350      clf.fit(X, y, alternative_weight=my_weights)
351      check_recorded_metadata(
352          clf.estimator_, method="fit", parent="fit", sample_weight=my_weights
353      )

357  def test_nested_routing():
358      # check if metadata is routed in a nested routing situation.
359      pipeline = SimplePipeline(
360          [
361              MetaTransformer(
362                  transformer=ConsumingTransformer()
363                  .set_fit_request(metadata=True, sample_weight=False)
364                  .set_transform_request(sample_weight=True, metadata=False)
365              ),
366              WeightedMetaRegressor(
367                  estimator=ConsumingRegressor()
368                  .set_fit_request(sample_weight="inner_weights", metadata=False)
369                  .set_predict_request(sample_weight=False)
370              ).set_fit_request(sample_weight="outer_weights"),
371          ]
372      )
373      w1, w2, w3 = [1], [2], [3]
374      pipeline.fit(
375          X, y, metadata=my_groups, sample_weight=w1, outer_weights=w2, inner_weights=w3
376      )
377      check_recorded_metadata(
378          pipeline.steps_[0].transformer_,
379          method="fit",
380          parent="fit",
381          metadata=my_groups,
382      )
383      check_recorded_metadata(
384          pipeline.steps_[0].transformer_,
385          method="transform",
386          parent="fit",
387          sample_weight=w1,
388      )
389      check_recorded_metadata(
390          pipeline.steps_[1], method="fit", parent="fit", sample_weight=w2
391      )
392      check_recorded_metadata(
393          pipeline.steps_[1].estimator_, method="fit", parent="fit", sample_weight=w3
394      )
395  
396      pipeline.predict(X, sample_weight=w3)
397      check_recorded_metadata(
398          pipeline.steps_[0].transformer_,
399          method="transform",
400          parent="fit",
401          sample_weight=w3,
