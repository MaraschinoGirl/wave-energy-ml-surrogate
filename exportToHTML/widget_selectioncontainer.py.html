4    """SelectionContainer class. 
5     
6    Represents a multipage container that can be used to group other widgets into 
7    pages. 
8    """
9    
10   from .widget_box import Box
11   from .widget import register
12   from .widget_core import CoreWidget
13   from traitlets import Unicode, Dict, CInt, TraitError, validate, observe
14   from .trait_types import TypedTuple
15   from itertools import chain, repeat, islice
16   
17   # Inspired by an itertools recipe: https://docs.python.org/3/library/itertools.html#itertools-recipes
18   def pad(iterable, padding=None, length=None):
19       """Returns the sequence elements and then returns None up to the given size (or indefinitely if size is None)."""
20       return islice(chain(iterable, repeat(padding)), length)
21   
22   class _SelectionContainer(Box, CoreWidget):
23       """Base class used to display multiple child widgets."""
24       titles = TypedTuple(trait=Unicode(), help="Titles of the pages").tag(sync=True)
25       selected_index = CInt(
26           help="""The index of the selected page. This is either an integer selecting a particular sub-widget, or None to have no widgets selected.""",
27           allow_none=True,
28           default_value=None
29       ).tag(sync=True)
30   
31       @validate('selected_index')
32       def _validated_index(self, proposal):
33           if proposal.value is None or 0 <= proposal.value < len(self.children):
34               return proposal.value
35           else:
36               raise TraitError('Invalid selection: index out of bounds')
37   
38       @validate('titles')
39       def _validate_titles(self, proposal):
40           return tuple(pad(proposal.value, '', len(self.children)))
41   
42       @observe('children')
43       def _observe_children(self, change):
44           self._reset_selected_index()
45           self._reset_titles()
46   
47       def _reset_selected_index(self):
48           if self.selected_index is not None and len(self.children) < self.selected_index:
49               self.selected_index = None
50   
51       def _reset_titles(self):
52           if len(self.titles) != len(self.children):
53               # Run validation function
54               self.titles = tuple(self.titles)
55   
56       def set_title(self, index, title):
57           """Sets the title of a container page. 
58           Parameters 
59           ---------- 
60           index : int 
61               Index of the container page 
62           title : unicode 
63               New title 
64           """
65           titles = list(self.titles)
66           # for backwards compatibility with ipywidgets 7.x
67           if title is None:
68               title = ''
69           titles[index]=title
70           self.titles = tuple(titles)
71   
72       def get_title(self, index):
73           """Gets the title of a container page. 
74           Parameters 
75           ---------- 
76           index : int 
77               Index of the container page 
78           """
79           return self.titles[index]

81   @register
82   class Accordion(_SelectionContainer):
83       """Displays children each on a separate accordion page."""
84       _view_name = Unicode('AccordionView').tag(sync=True)
85       _model_name = Unicode('AccordionModel').tag(sync=True)

88   @register
89   class Tab(_SelectionContainer):
90       """Displays children each on a separate accordion tab."""
91       _view_name = Unicode('TabView').tag(sync=True)
92       _model_name = Unicode('TabModel').tag(sync=True)
93   
94       def __init__(self, children=(), **kwargs):
95           if len(children) > 0 and 'selected_index' not in kwargs:
96               kwargs['selected_index'] = 0
97           super().__init__(children=children, **kwargs)
98   
99       def _reset_selected_index(self):
100          # if there are no tabs, then none should be selected
101          num_children = len(self.children)
102          if num_children == 0:
103              self.selected_index = None
104  
105          # if there are tabs, but none is selected, select the first one
106          elif self.selected_index == None:
107              self.selected_index = 0
108  
109          # if there are tabs and a selection, but the selection is no longer
110          # valid, select the last tab.
111          elif num_children < self.selected_index:
112              self.selected_index = num_children - 1
