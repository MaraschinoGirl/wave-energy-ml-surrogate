104  def diric(x, n):
105      """Periodic sinc function, also called the Dirichlet function. 
106   
107      The Dirichlet function is defined as:: 
108   
109          diric(x, n) = sin(x * n/2) / (n * sin(x / 2)), 
110   
111      where `n` is a positive integer. 
112   
113      Parameters 
114      ---------- 
115      x : array_like 
116          Input data 
117      n : int 
118          Integer defining the periodicity. 
119   
120      Returns 
121      ------- 
122      diric : ndarray 
123   
124      Examples 
125      -------- 
126      >>> import numpy as np 
127      >>> from scipy import special 
128      >>> import matplotlib.pyplot as plt 
129   
130      >>> x = np.linspace(-8*np.pi, 8*np.pi, num=201) 
131      >>> plt.figure(figsize=(8, 8)); 
132      >>> for idx, n in enumerate([2, 3, 4, 9]): 
133      ...     plt.subplot(2, 2, idx+1) 
134      ...     plt.plot(x, special.diric(x, n)) 
135      ...     plt.title('diric, n={}'.format(n)) 
136      >>> plt.show() 
137   
138      The following example demonstrates that `diric` gives the magnitudes 
139      (modulo the sign and scaling) of the Fourier coefficients of a 
140      rectangular pulse. 
141   
142      Suppress output of values that are effectively 0: 
143   
144      >>> np.set_printoptions(suppress=True) 
145   
146      Create a signal `x` of length `m` with `k` ones: 
147   
148      >>> m = 8 
149      >>> k = 3 
150      >>> x = np.zeros(m) 
151      >>> x[:k] = 1 
152   
153      Use the FFT to compute the Fourier transform of `x`, and 
154      inspect the magnitudes of the coefficients: 
155   
156      >>> np.abs(np.fft.fft(x)) 
157      array([ 3.        ,  2.41421356,  1.        ,  0.41421356,  1.        , 
158              0.41421356,  1.        ,  2.41421356]) 
159   
160      Now find the same values (up to sign) using `diric`. We multiply 
161      by `k` to account for the different scaling conventions of 
162      `numpy.fft.fft` and `diric`: 
163   
164      >>> theta = np.linspace(0, 2*np.pi, m, endpoint=False) 
165      >>> k * special.diric(theta, k) 
166      array([ 3.        ,  2.41421356,  1.        , -0.41421356, -1.        , 
167             -0.41421356,  1.        ,  2.41421356]) 
168      """
169      x, n = asarray(x), asarray(n)
170      n = asarray(n + (x-x))
171      x = asarray(x + (n-n))
172      if issubdtype(x.dtype, inexact):
173          ytype = x.dtype
174      else:
175          ytype = float
176      y = zeros(x.shape, ytype)
177  
178      # empirical minval for 32, 64 or 128 bit float computations
179      # where sin(x/2) < minval, result is fixed at +1 or -1
180      if np.finfo(ytype).eps < 1e-18:
181          minval = 1e-11
182      elif np.finfo(ytype).eps < 1e-15:
183          minval = 1e-7
184      else:
185          minval = 1e-3
186  
187      mask1 = (n <= 0) | (n != floor(n))
188      place(y, mask1, nan)
189  
190      x = x / 2
191      denom = sin(x)
192      mask2 = (1-mask1) & (abs(denom) < minval)
193      xsub = extract(mask2, x)
194      nsub = extract(mask2, n)
195      zsub = xsub / pi
196      place(y, mask2, pow(-1, np.round(zsub)*(nsub-1)))
197  
198      mask = (1-mask1) & (1-mask2)
199      xsub = extract(mask, x)
200      nsub = extract(mask, n)
201      dsub = extract(mask, denom)
202      place(y, mask, sin(nsub*xsub)/(nsub*dsub))
203      return y

206  def jnjnp_zeros(nt):
207      """Compute zeros of integer-order Bessel functions Jn and Jn'. 
208   
209      Results are arranged in order of the magnitudes of the zeros. 
210   
211      Parameters 
212      ---------- 
213      nt : int 
214          Number (<=1200) of zeros to compute 
215   
216      Returns 
217      ------- 
218      zo[l-1] : ndarray 
219          Value of the lth zero of Jn(x) and Jn'(x). Of length `nt`. 
220      n[l-1] : ndarray 
221          Order of the Jn(x) or Jn'(x) associated with lth zero. Of length `nt`. 
222      m[l-1] : ndarray 
223          Serial number of the zeros of Jn(x) or Jn'(x) associated 
224          with lth zero. Of length `nt`. 
225      t[l-1] : ndarray 
226          0 if lth zero in zo is zero of Jn(x), 1 if it is a zero of Jn'(x). Of 
227          length `nt`. 
228   
229      See Also 
230      -------- 
231      jn_zeros, jnp_zeros : to get separated arrays of zeros. 
232   
233      References 
234      ---------- 
235      .. [1] Zhang, Shanjie and Jin, Jianming. "Computation of Special 
236             Functions", John Wiley and Sons, 1996, chapter 5. 
237             https://people.sc.fsu.edu/~jburkardt/f77_src/special_functions/special_functions.html 
238   
239      """
240      if not isscalar(nt) or (floor(nt) != nt) or (nt > 1200):
241          raise ValueError("Number must be integer <= 1200.")
242      nt = int(nt)
243      n, m, t, zo = _specfun.jdzo(nt)
244      return zo[1:nt+1], n[:nt], m[:nt], t[:nt]

248      """Compute nt zeros of Bessel functions Jn(x), Jn'(x), Yn(x), and Yn'(x). 
249   
250      Returns 4 arrays of length `nt`, corresponding to the first `nt`

255      ---------- 
256      n : int 
257          Order of the Bessel functions 
258      nt : int 
259          Number (<=1200) of zeros to compute 
260   
261      Returns 
262      ------- 
263      Jn : ndarray 
264          First `nt` zeros of Jn 
265      Jnp : ndarray 
266          First `nt` zeros of Jn' 
267      Yn : ndarray 
268          First `nt` zeros of Yn 
269      Ynp : ndarray 
270          First `nt` zeros of Yn' 
271   
272      See Also 
273      -------- 
274      jn_zeros, jnp_zeros, yn_zeros, ynp_zeros 
275   
276      References 
277      ---------- 
278      .. [1] Zhang, Shanjie and Jin, Jianming. "Computation of Special 
279             Functions", John Wiley and Sons, 1996, chapter 5. 
280             https://people.sc.fsu.edu/~jburkardt/f77_src/special_functions/special_functions.html 
281   
282      Examples 
283      --------

285      :math:`Y_1` and :math:`Y_1'`. 
286   
287      >>> from scipy.special import jnyn_zeros 
288      >>> jn_roots, jnp_roots, yn_roots, ynp_roots = jnyn_zeros(1, 3) 
289      >>> jn_roots, yn_roots 
290      (array([ 3.83170597,  7.01558667, 10.17346814]), 
291       array([2.19714133, 5.42968104, 8.59600587])) 
292   
293      Plot :math:`J_1`, :math:`J_1'`, :math:`Y_1`, :math:`Y_1'` and their roots. 
294   
295      >>> import numpy as np 
296      >>> import matplotlib.pyplot as plt 
297      >>> from scipy.special import jnyn_zeros, jvp, jn, yvp, yn 
298      >>> jn_roots, jnp_roots, yn_roots, ynp_roots = jnyn_zeros(1, 3) 
299      >>> fig, ax = plt.subplots() 
300      >>> xmax= 11 
301      >>> x = np.linspace(0, xmax) 
302      >>> x[0] += 1e-15

304      >>> ax.plot(x, jvp(1, x, 1), label=r"$J_1'$", c='b') 
305      >>> ax.plot(x, yn(1, x), label=r"$Y_1$", c='y')

309      ...            label=r"$J_1$ roots") 
310      >>> ax.scatter(jnp_roots, zeros, s=30, c='b', zorder=5, 
311      ...            label=r"$J_1'$ roots") 
312      >>> ax.scatter(yn_roots, zeros, s=30, c='y', zorder=5, 
313      ...            label=r"$Y_1$ roots") 
314      >>> ax.scatter(ynp_roots, zeros, s=30, c='c', zorder=5, 
315      ...            label=r"$Y_1'$ roots") 
316      >>> ax.hlines(0, 0, xmax, color='k') 
317      >>> ax.set_ylim(-0.6, 0.6) 
318      >>> ax.set_xlim(0, xmax) 
319      >>> ax.legend(ncol=2, bbox_to_anchor=(1., 0.75)) 
320      >>> plt.tight_layout() 
321      >>> plt.show() 
322      """
323      if not (isscalar(nt) and isscalar(n)):
324          raise ValueError("Arguments must be scalars.")
325      if (floor(n) != n) or (floor(nt) != nt):
326          raise ValueError("Arguments must be integers.")
327      if (nt <= 0):
328          raise ValueError("nt > 0")
329      return _specfun.jyzo(abs(n), nt)

583  def y0_zeros(nt, complex=False):
584      """Compute nt zeros of Bessel function Y0(z), and derivative at each zero. 
585   
586      The derivatives are given by Y0'(z0) = -Y1(z0) at each zero z0. 
587   
588      Parameters 
589      ---------- 
590      nt : int 
591          Number of zeros to return 
592      complex : bool, default False 
593          Set to False to return only the real zeros; set to True to return only 
594          the complex zeros with negative real part and positive imaginary part. 
595          Note that the complex conjugates of the latter are also zeros of the 
596          function, but are not returned by this routine. 
597   
598      Returns 
599      ------- 
600      z0n : ndarray 
601          Location of nth zero of Y0(z) 
602      y0pz0n : ndarray 
603          Value of derivative Y0'(z0) for nth zero 
604   
605      References 
606      ---------- 
607      .. [1] Zhang, Shanjie and Jin, Jianming. "Computation of Special 
608             Functions", John Wiley and Sons, 1996, chapter 5. 
609             https://people.sc.fsu.edu/~jburkardt/f77_src/special_functions/special_functions.html 
610   
611      Examples 
612      -------- 
613      Compute the first 4 real roots and the derivatives at the roots of 
614      :math:`Y_0`: 
615   
616      >>> import numpy as np 
617      >>> from scipy.special import y0_zeros 
618      >>> zeros, grads = y0_zeros(4) 
619      >>> with np.printoptions(precision=5):

814  def _bessel_diff_formula(v, z, n, L, phase):
815      # from AMS55.
816      # L(v, z) = J(v, z), Y(v, z), H1(v, z), H2(v, z), phase = -1
817      # L(v, z) = I(v, z) or exp(v*pi*i)K(v, z), phase = 1
818      # For K, you can pull out the exp((v-k)*pi*i) into the caller
819      v = asarray(v)
820      p = 1.0
821      s = L(v-n, z)
822      for i in range(1, n+1):
823          p = phase * (p * (n-i+1)) / i   # = choose(k, i)
824          s += p*L(v-n + i*2, z)
825      return s / (2.**n)

2799             raise ValueError("`N` and `k` must scalar integers be with `exact=True`.")

2802         non_integral = not (floor_N == N and floor_k == k)

2828 def _range_prod(lo, hi, k=1):
2829     """ 
2830     Product of a range of numbers spaced k apart (from hi). 
2831  
2832     For k=1, this returns the product of 
2833     lo * (lo+1) * (lo+2) * ... * (hi-2) * (hi-1) * hi 
2834     = hi! / (lo-1)! 
2835  
2836     For k>1, it correspond to taking only every k'th number when 
2837     counting down from hi - e.g. 18!!!! = _range_prod(1, 18, 4). 
2838  
2839     Breaks into smaller products first for speed: 
2840     _range_prod(2, 9) = ((2*3)*(4*5))*((6*7)*(8*9)) 
2841     """
2842     if lo == 1 and k == 1:
2843         return math.factorial(hi)
2844 
2845     if lo + k < hi:
2846         mid = (hi + lo) // 2
2847         if k > 1:
2848             # make sure mid is a multiple of k away from hi
2849             mid = mid - ((mid - hi) % k)
2850         return _range_prod(lo, mid, k) * _range_prod(mid + k, hi, k)
2851     elif lo + k == hi:
2852         return lo * hi
2853     else:
2854         return hi

2857 def _factorialx_array_exact(n, k=1):
2858     """ 
2859     Exact computation of factorial for an array. 
2860  
2861     The factorials are computed in incremental fashion, by taking 
2862     the sorted unique values of n and multiplying the intervening 
2863     numbers between the different unique values. 
2864  
2865     In other words, the factorial for the largest input is only 
2866     computed once, with each other result computed in the process. 
2867  
2868     k > 1 corresponds to the multifactorial. 
2869     """
2870     un = np.unique(n)
2871     # numpy changed nan-sorting behaviour with 1.21, see numpy/numpy#18070;
2872     # to unify the behaviour, we remove the nan's here; the respective
2873     # values will be set separately at the end
2874     un = un[~np.isnan(un)]
2875 
2876     # Convert to object array if np.int64 can't handle size
2877     if np.isnan(n).any():
2878         dt = float
2879     elif k in _FACTORIALK_LIMITS_64BITS.keys():
2880         if un[-1] > _FACTORIALK_LIMITS_64BITS[k]:
2881             # e.g. k=1: 21! > np.iinfo(np.int64).max
2882             dt = object
2883         elif un[-1] > _FACTORIALK_LIMITS_32BITS[k]:
2884             # e.g. k=3: 26!!! > np.iinfo(np.int32).max
2885             dt = np.int64
2886         else:
2887             dt = np.dtype("long")
2888     else:
2889         # for k >= 10, we always use object
2890         dt = object
2891 
2892     out = np.empty_like(n, dtype=dt)
2893 
2894     # Handle invalid/trivial values
2895     un = un[un > 1]
2896     out[n < 2] = 1
2897     out[n < 0] = 0
2898 
2899     # Calculate products of each range of numbers
2900     # we can only multiply incrementally if the values are k apart;
2901     # therefore we partition `un` into "lanes", i.e. its residues modulo k
2902     for lane in range(0, k):
2903         ul = un[(un % k) == lane] if k > 1 else un
2904         if ul.size:
2905             # after np.unique, un resp. ul are sorted, ul[0] is the smallest;
2906             # cast to python ints to avoid overflow with np.int-types
2907             val = _range_prod(1, int(ul[0]), k=k)
2908             out[n == ul[0]] = val
2909             for i in range(len(ul) - 1):
2910                 # by the filtering above, we have ensured that prev & current
2911                 # are a multiple of k apart
2912                 prev = ul[i]
2913                 current = ul[i + 1]
2914                 # we already multiplied all factors until prev; continue
2915                 # building the full factorial from the following (`prev + 1`);
2916                 # use int() for the same reason as above
2917                 val *= _range_prod(int(prev + 1), int(current), k=k)
2918                 out[n == current] = val
2919 
2920     if np.isnan(n).any():
2921         out = out.astype(np.float64)
2922         out[np.isnan(n)] = np.nan
2923     return out

2946 def _gamma1p(vals):

2948     returns gamma(n+1), though with NaN at -1 instead of inf, c.f. #21827

2961 def _factorialx_approx_core(n, k, extend):
2962     """ 
2963     Core approximation to multifactorial for array n and integer k. 
2964     """
2965     if k == 1:
2966         # shortcut for k=1; same for both extensions, because we assume the
2967         # handling of extend == 'zero' happens in _factorialx_array_approx
2968         result = _gamma1p(n)
2969         if isinstance(n, np.ndarray):
2970             # gamma does not maintain 0-dim arrays; fix it
2971             result = np.array(result)
2972         return result
2973 
2974     if extend == "complex":
2975         # see https://numpy.org/doc/stable/reference/generated/numpy.power.html
2976         p_dtype = complex if (_is_subdtype(type(k), "c") or k < 0) else None
2977         with warnings.catch_warnings():
2978             # do not warn about 0 * inf, nan / nan etc.; the results are correct
2979             warnings.simplefilter("ignore", RuntimeWarning)
2980             # don't use `(n-1)/k` in np.power; underflows if 0 is of a uintX type
2981             result = np.power(k, n / k, dtype=p_dtype) * _gamma1p(n / k)
2982             result *= rgamma(1 / k + 1) / np.power(k, 1 / k, dtype=p_dtype)
2983         if isinstance(n, np.ndarray):
2984             # ensure we keep array-ness for 0-dim inputs; already n/k above loses it
2985             result = np.array(result)
2986         return result
2987 
2988     # at this point we are guaranteed that extend='zero' and that k>0 is an integer
2989     n_mod_k = n % k
2990     # scalar case separately, unified handling would be inefficient for arrays;
2991     # don't use isscalar due to numpy/numpy#23574; 0-dim arrays treated below
2992     if not isinstance(n, np.ndarray):
2993         return (
2994             np.power(k, (n - n_mod_k) / k)
2995             * gamma(n / k + 1) / gamma(n_mod_k / k + 1)
2996             * max(n_mod_k, 1)
2997         )
2998 
2999     # factor that's independent of the residue class (see factorialk docstring)
3000     result = np.power(k, n / k) * gamma(n / k + 1)
3001     # factor dependent on residue r (for `r=0` it's 1, so we skip `r=0`
3002     # below and thus also avoid evaluating `max(r, 1)`)
3003     def corr(k, r): return np.power(k, -r / k) / gamma(r / k + 1) * r
3004     for r in np.unique(n_mod_k):
3005         if r == 0:
3006             continue
3007         # cast to int because uint types break on `-r`
3008         result[n_mod_k == r] *= corr(k, int(r))
3009     return result

3012 def _is_subdtype(dtype, dtypes):
3013     """ 
3014     Shorthand for calculating whether dtype is subtype of some dtypes. 
3015  
3016     Also allows specifying a list instead of just a single dtype. 
3017  
3018     Additionaly, the most important supertypes from 
3019         https://numpy.org/doc/stable/reference/arrays.scalars.html 
3020     can optionally be specified using abbreviations as follows: 
3021         "i": np.integer 
3022         "f": np.floating 
3023         "c": np.complexfloating 
3024         "n": np.number (contains the other three) 
3025     """
3026     dtypes = dtypes if isinstance(dtypes, list) else [dtypes]
3027     # map single character abbreviations, if they are in dtypes
3028     mapping = {
3029         "i": np.integer,
3030         "f": np.floating,
3031         "c": np.complexfloating,
3032         "n": np.number
3033     }
3034     dtypes = [mapping.get(x, x) for x in dtypes]
3035     return any(np.issubdtype(dtype, dt) for dt in dtypes)

3249 def factorialk(n, k, exact=False, extend="zero"):
3250     """Multifactorial of n of order k, n(!!...!). 
3251  
3252     This is the multifactorial of n skipping k values.  For example, 
3253  
3254       factorialk(17, 4) = 17!!!! = 17 * 13 * 9 * 5 * 1 
3255  
3256     In particular, for any integer ``n``, we have 
3257  
3258       factorialk(n, 1) = factorial(n) 
3259  
3260       factorialk(n, 2) = factorial2(n) 
3261  
3262     Parameters 
3263     ---------- 
3264     n : int or float or complex (or array_like thereof) 
3265         Input values for multifactorial. Non-integer values require 
3266         ``extend='complex'``. By default, the return value for ``n < 0`` is 0. 
3267     n : int or float or complex (or array_like thereof) 
3268         Order of multifactorial. Non-integer values require ``extend='complex'``. 
3269     exact : bool, optional 
3270         If ``exact`` is set to True, calculate the answer exactly using 
3271         integer arithmetic, otherwise use an approximation (faster, 
3272         but yields floats instead of integers) 
3273         Default is False. 
3274     extend : string, optional 
3275         One of ``'zero'`` or ``'complex'``; this determines how values ``n<0`` are 
3276         handled - by default they are 0, but it is possible to opt into the complex 
3277         extension of the multifactorial. This enables passing complex values, 
3278         not only to ``n`` but also to ``k``. 
3279  
3280         .. warning:: 
3281  
3282            Using the ``'complex'`` extension also changes the values of the 
3283            multifactorial at integers ``n != 1 (mod k)`` by a factor depending 
3284            on both ``k`` and ``n % k``, see below or [1]. 
3285  
3286     Returns 
3287     ------- 
3288     nf : int or float or complex or ndarray 
3289         Multifactorial (order ``k``) of ``n``, as integer, float or complex (depending 
3290         on ``exact`` and ``extend``). Array inputs are returned as arrays. 
3291  
3292     Examples 
3293     -------- 
3294     >>> from scipy.special import factorialk 
3295     >>> factorialk(5, k=1, exact=True) 
3296     120 
3297     >>> factorialk(5, k=3, exact=True) 
3298     10 
3299     >>> factorialk([5, 7, 9], k=3, exact=True) 
3300     array([ 10,  28, 162]) 
3301     >>> factorialk([5, 7, 9], k=3, exact=False) 
3302     array([ 10.,  28., 162.]) 
3303  
3304     Notes 
3305     ----- 
3306     While less straight-forward than for the double-factorial, it's possible to 
3307     calculate a general approximation formula of n!(k) by studying ``n`` for a given 
3308     remainder ``r < k`` (thus ``n = m * k + r``, resp. ``r = n % k``), which can be 
3309     put together into something valid for all integer values ``n >= 0`` & ``k > 0``:: 
3310  
3311       n!(k) = k ** ((n - r)/k) * gamma(n/k + 1) / gamma(r/k + 1) * max(r, 1) 
3312  
3313     This is the basis of the approximation when ``exact=False``. 
3314  
3315     In principle, any fixed choice of ``r`` (ignoring its relation ``r = n%k`` 
3316     to ``n``) would provide a suitable analytic continuation from integer ``n`` 
3317     to complex ``z`` (not only satisfying the functional equation but also 
3318     being logarithmically convex, c.f. Bohr-Mollerup theorem) -- in fact, the 
3319     choice of ``r`` above only changes the function by a constant factor. The 
3320     final constraint that determines the canonical continuation is ``f(1) = 1``, 
3321     which forces ``r = 1`` (see also [1]).:: 
3322  
3323       z!(k) = k ** ((z - 1)/k) * gamma(z/k + 1) / gamma(1/k + 1) 
3324  
3325     References 
3326     ---------- 
3327     .. [1] Complex extension to multifactorial 
3328             https://en.wikipedia.org/wiki/Double_factorial#Alternative_extension_of_the_multifactorial 
3329     """
3330     return _factorialx_wrapper("factorialk", n, k=k, exact=exact, extend=extend)

3333 def stirling2(N, K, *, exact=False):
3334     r"""Generate Stirling number(s) of the second kind. 
3335  
3336     Stirling numbers of the second kind count the number of ways to 
3337     partition a set with N elements into K non-empty subsets. 
3338  
3339     The values this function returns are calculated using a dynamic 
3340     program which avoids redundant computation across the subproblems 
3341     in the solution. For array-like input, this implementation also 
3342     avoids redundant computation across the different Stirling number 
3343     calculations. 
3344  
3345     The numbers are sometimes denoted 
3346  
3347     .. math:: 
3348  
3349         {N \brace{K}} 
3350  
3351     see [1]_ for details. This is often expressed-verbally-as 
3352     "N subset K". 
3353  
3354     Parameters 
3355     ---------- 
3356     N : int, ndarray 
3357         Number of things. 
3358     K : int, ndarray 
3359         Number of non-empty subsets taken. 
3360     exact : bool, optional 
3361         Uses dynamic programming (DP) with floating point 
3362         numbers for smaller arrays and uses a second order approximation due to 
3363         Temme for larger entries  of `N` and `K` that allows trading speed for 
3364         accuracy. See [2]_ for a description. Temme approximation is used for 
3365         values ``n>50``. The max error from the DP has max relative error 
3366         ``4.5*10^-16`` for ``n<=50`` and the max error from the Temme approximation 
3367         has max relative error ``5*10^-5`` for ``51 <= n < 70`` and 
3368         ``9*10^-6`` for ``70 <= n < 101``. Note that these max relative errors will 
3369         decrease further as `n` increases. 
3370  
3371     Returns 
3372     ------- 
3373     val : int, float, ndarray 
3374         The number of partitions. 
3375  
3376     See Also 
3377     -------- 
3378     comb : The number of combinations of N things taken k at a time. 
3379  
3380     Notes 
3381     ----- 
3382     - If N < 0, or K < 0, then 0 is returned. 
3383     - If K > N, then 0 is returned. 
3384  
3385     The output type will always be `int` or ndarray of `object`. 
3386     The input must contain either numpy or python integers otherwise a 
3387     TypeError is raised. 
3388  
3389     References 
3390     ---------- 
3391     .. [1] R. L. Graham, D. E. Knuth and O. Patashnik, "Concrete 
3392         Mathematics: A Foundation for Computer Science," Addison-Wesley 
3393         Publishing Company, Boston, 1989. Chapter 6, page 258. 
3394  
3395     .. [2] Temme, Nico M. "Asymptotic estimates of Stirling numbers." 
3396         Studies in Applied Mathematics 89.3 (1993): 233-243. 
3397  
3398     Examples 
3399     -------- 
3400     >>> import numpy as np 
3401     >>> from scipy.special import stirling2 
3402     >>> k = np.array([3, -1, 3]) 
3403     >>> n = np.array([10, 10, 9]) 
3404     >>> stirling2(n, k) 
3405     array([9330.0, 0.0, 3025.0]) 
3406  
3407     """
3408     output_is_scalar = np.isscalar(N) and np.isscalar(K)
3409     # make a min-heap of unique (n,k) pairs
3410     N, K = asarray(N), asarray(K)
3411     if not np.issubdtype(N.dtype, np.integer):
3412         raise TypeError("Argument `N` must contain only integers")
3413     if not np.issubdtype(K.dtype, np.integer):
3414         raise TypeError("Argument `K` must contain only integers")
3415     if not exact:
3416         # NOTE: here we allow np.uint via casting to double types prior to
3417         # passing to private ufunc dispatcher. All dispatched functions
3418         # take double type for (n,k) arguments and return double.
3419         return _stirling2_inexact(N.astype(float), K.astype(float))
3420     nk_pairs = list(
3421         set([(n.take(0), k.take(0))
3422              for n, k in np.nditer([N, K], ['refs_ok'])])
3423     )
3424     heapify(nk_pairs)
3425     # base mapping for small values
3426     snsk_vals = defaultdict(int)
3427     for pair in [(0, 0), (1, 1), (2, 1), (2, 2)]:
3428         snsk_vals[pair] = 1
3429     # for each pair in the min-heap, calculate the value, store for later
3430     n_old, n_row = 2, [0, 1, 1]
3431     while nk_pairs:
3432         n, k = heappop(nk_pairs)
3433         if n < 2 or k > n or k <= 0:
3434             continue
3435         elif k == n or k == 1:
3436             snsk_vals[(n, k)] = 1
3437             continue
3438         elif n != n_old:
3439             num_iters = n - n_old
3440             while num_iters > 0:
3441                 n_row.append(1)
3442                 # traverse from back to remove second row
3443                 for j in range(len(n_row)-2, 1, -1):
3444                     n_row[j] = n_row[j]*j + n_row[j-1]
3445                 num_iters -= 1
3446             snsk_vals[(n, k)] = n_row[k]
3447         else:
3448             snsk_vals[(n, k)] = n_row[k]
3449         n_old, n_row = n, n_row
3450     out_types = [object, object, object] if exact else [float, float, float]
3451     # for each pair in the map, fetch the value, and populate the array
3452     it = np.nditer(
3453         [N, K, None],
3454         ['buffered', 'refs_ok'],
3455         [['readonly'], ['readonly'], ['writeonly', 'allocate']],
3456         op_dtypes=out_types,
3457     )
3458     with it:
3459         while not it.finished:
3460             it[2] = snsk_vals[(int(it[0]), int(it[1]))]
3461             it.iternext()
3462         output = it.operands[2]
3463         # If N and K were both scalars, convert output to scalar.
3464         if output_is_scalar:
3465             output = output.take(0)
3466     return output
