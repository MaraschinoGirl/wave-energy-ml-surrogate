12   class WSGIAppMixin:
13       # TODO: Now that WSGIAdapter is gone, this is a pretty weak test.
14       def get_executor(self):
15           raise NotImplementedError()
16   
17       def get_app(self):
18           executor = self.get_executor()
19           # The barrier test in DummyExecutorTest will always wait the full
20           # value of this timeout, so we don't want it to be too high.
21           self.barrier = threading.Barrier(2, timeout=0.3)
22   
23           def make_container(app):
24               return WSGIContainer(validator(app), executor=executor)
25   
26           return RuleRouter(
27               [
28                   ("/simple", make_container(self.simple_wsgi_app)),
29                   ("/barrier", make_container(self.barrier_wsgi_app)),
30                   ("/streaming_barrier", make_container(self.streaming_barrier_wsgi_app)),
31               ]
32           )
33   
34       def respond_plain(self, start_response):
35           status = "200 OK"
36           response_headers = [("Content-Type", "text/plain")]
37           start_response(status, response_headers)
38   
39       def simple_wsgi_app(self, environ, start_response):
40           self.respond_plain(start_response)
41           return [b"Hello world!"]
42   
43       def barrier_wsgi_app(self, environ, start_response):
44           self.respond_plain(start_response)
45           try:
46               n = self.barrier.wait()
47           except threading.BrokenBarrierError:
48               return [b"broken barrier"]
49           else:
50               return [b"ok %d" % n]
51   
52       def streaming_barrier_wsgi_app(self, environ, start_response):
53           self.respond_plain(start_response)
54           yield b"ok "
55           try:
56               n = self.barrier.wait()
57           except threading.BrokenBarrierError:
58               yield b"broken barrier"
59           else:
60               yield b"%d" % n
