47   def upfirdn_naive(x, h, up=1, down=1):
48       """Naive upfirdn processing in Python. 
49    
50       Note: arg order (x, h) differs to facilitate apply_along_axis use. 
51       """
52       h = np.asarray(h)
53       out = np.zeros(len(x) * up, x.dtype)
54       out[::up] = x
55       out = np.convolve(h, out)[::down][:_output_len(len(h), len(x), up, down)]
56       return out

59   class UpFIRDnCase:
60       """Test _UpFIRDn object"""
61       def __init__(self, up, down, h, x_dtype):
62           self.up = up
63           self.down = down
64           self.h = np.atleast_1d(h)
65           self.x_dtype = x_dtype
66           self.rng = np.random.RandomState(17)
67   
68       def __call__(self):
69           # tiny signal
70           self.scrub(np.ones(1, self.x_dtype))
71           # ones
72           self.scrub(np.ones(10, self.x_dtype))  # ones
73           # randn
74           x = self.rng.randn(10).astype(self.x_dtype)
75           if self.x_dtype in (np.complex64, np.complex128):
76               x += 1j * self.rng.randn(10)
77           self.scrub(x)
78           # ramp
79           self.scrub(np.arange(10).astype(self.x_dtype))
80           # 3D, random
81           size = (2, 3, 5)
82           x = self.rng.randn(*size).astype(self.x_dtype)
83           if self.x_dtype in (np.complex64, np.complex128):
84               x += 1j * self.rng.randn(*size)
85           for axis in range(len(size)):
86               self.scrub(x, axis=axis)
87           x = x[:, ::2, 1::3].T
88           for axis in range(len(size)):
89               self.scrub(x, axis=axis)
90   
91       def scrub(self, x, axis=-1):
92           yr = np.apply_along_axis(upfirdn_naive, axis, x,
93                                    self.h, self.up, self.down)
94           want_len = _output_len(len(self.h), x.shape[axis], self.up, self.down)
95           assert yr.shape[axis] == want_len
96           y = upfirdn(self.h, x, self.up, self.down, axis=axis)
97           assert y.shape[axis] == want_len
98           assert y.shape == yr.shape
99           dtypes = (self.h.dtype, x.dtype)
100          if all(d == np.complex64 for d in dtypes):
101              assert y.dtype == np.complex64
102          elif np.complex64 in dtypes and np.float32 in dtypes:
103              assert y.dtype == np.complex64
104          elif all(d == np.float32 for d in dtypes):
105              assert y.dtype == np.float32
106          elif np.complex128 in dtypes or np.complex64 in dtypes:
107              assert y.dtype == np.complex128
108          else:
109              assert y.dtype == np.float64
110          xp_assert_close(yr.astype(y.dtype), y)

118      def test_valid_input(self):
119          assert_raises(ValueError, upfirdn, [1], [1], 1, 0)  # up or down < 1
120          assert_raises(ValueError, upfirdn, [], [1], 1, 1)  # h.ndim != 1
121          assert_raises(ValueError, upfirdn, [[1]], [1], 1, 1)

123      @pytest.mark.parametrize('len_h', [1, 2, 3, 4, 5])
124      @pytest.mark.parametrize('len_x', [1, 2, 3, 4, 5])
125      def test_singleton(self, len_h, len_x):
126          # gh-9844: lengths producing expected outputs
127          h = np.zeros(len_h)
128          h[len_h // 2] = 1.  # make h a delta
129          x = np.ones(len_x)
130          y = upfirdn(h, x, 1, 1)
131          want = np.pad(x, (len_h // 2, (len_h - 1) // 2), 'constant')
132          xp_assert_close(y, want)

134      def test_shift_x(self):
135          # gh-9844: shifted x can change values?
136          y = upfirdn([1, 1], [1.], 1, 1)
137          xp_assert_close(y, np.asarray([1.0, 1.0]))  # was [0, 1] in the issue
138          y = upfirdn([1, 1], [0., 1.], 1, 1)
139          xp_assert_close(y, np.asarray([0.0, 1.0, 1.0]))

144      @pytest.mark.parametrize('len_h, len_x, up, down, expected', [

156          y = upfirdn(h, x, up, down)
157          expected = np.asarray(expected, dtype=np.float64)

160      @pytest.mark.parametrize('down, want_len', [  # lengths from MATLAB
161          (2, 5015),
162          (11, 912),
163          (79, 127),
164      ])
165      def test_vs_convolve(self, down, want_len):
166          # Check that up=1.0 gives same answer as convolve + slicing
167          random_state = np.random.RandomState(17)
168          try_types = (int, np.float32, np.complex64, float, complex)
169          size = 10000
170  
171          for dtype in try_types:
172              x = random_state.randn(size).astype(dtype)
173              if dtype in (np.complex64, np.complex128):
174                  x += 1j * random_state.randn(size)
175  
176              h = firwin(31, 1. / down, window='hamming')
177              yl = upfirdn_naive(x, h, 1, down)
178              y = upfirdn(h, x, up=1, down=down)
179              assert y.shape == (want_len,)
180              assert yl.shape[0] == y.shape[0]
181              xp_assert_close(yl, y, atol=1e-7, rtol=1e-7)

189      @pytest.mark.parametrize('x_dtype', _UPFIRDN_TYPES)
190      @pytest.mark.parametrize('h_dtype', _UPFIRDN_TYPES)
191      @pytest.mark.parametrize('p_max, q_max',
192                               list(product((10, 100), (10, 100))))
193      def test_vs_naive(self, x_dtype, h_dtype, p_max, q_max):
194          tests = self._random_factors(p_max, q_max, h_dtype, x_dtype)
195          for test in tests:
196              test()

198      def _random_factors(self, p_max, q_max, h_dtype, x_dtype):
199          n_rep = 3
200          longest_h = 25
201          random_state = np.random.RandomState(17)
202          tests = []
203  
204          for _ in range(n_rep):
205              # Randomize the up/down factors somewhat
206              p_add = q_max if p_max > q_max else 1
207              q_add = p_max if q_max > p_max else 1
208              p = random_state.randint(p_max) + p_add
209              q = random_state.randint(q_max) + q_add
210  
211              # Generate random FIR coefficients
212              len_h = random_state.randint(longest_h) + 1
213              h = np.atleast_1d(random_state.randint(len_h))
214              h = h.astype(h_dtype)
215              if h_dtype is complex:
216                  h += 1j * random_state.randint(len_h)
217  
218              tests.append(UpFIRDnCase(p, q, h, x_dtype))
219  
220          return tests

222      @pytest.mark.parametrize('mode', _upfirdn_modes)
223      def test_extensions(self, mode):
224          """Test vs. manually computed results for modes not in numpy's pad."""
225          x = np.array([1, 2, 3, 1], dtype=float)
226          npre, npost = 6, 6
227          y = _pad_test(x, npre=npre, npost=npost, mode=mode)
228          if mode == 'antisymmetric':
229              y_expected = np.asarray(
230                  [3.0, 1, -1, -3, -2, -1, 1, 2, 3, 1, -1, -3, -2, -1, 1, 2])
231          elif mode == 'antireflect':
232              y_expected = np.asarray(
233                  [1.0, 2, 3, 1, -1, 0, 1, 2, 3, 1, -1, 0, 1, 2, 3, 1])
234          elif mode == 'smooth':
235              y_expected = np.asarray(
236                  [-5.0, -4, -3, -2, -1, 0, 1, 2, 3, 1, -1, -3, -5, -7, -9, -11])
237          elif mode == "line":
238              lin_slope = (x[-1] - x[0]) / (len(x) - 1)
239              left = x[0] + np.arange(-npre, 0, 1) * lin_slope
240              right = x[-1] + np.arange(1, npost + 1) * lin_slope
241              y_expected = np.concatenate((left, x, right))
242          else:
243              y_expected = np.pad(x, (npre, npost), mode=mode)
244          xp_assert_close(y, y_expected)
245  
246      @pytest.mark.parametrize(
247          'size, h_len, mode, dtype',
248          product(
249              [8],
250              [4, 5, 26],  # include cases with h_len > 2*size
251              _upfirdn_modes,
252              [np.float32, np.float64, np.complex64, np.complex128],
253          )
254      )
255      def test_modes(self, size, h_len, mode, dtype):
256          random_state = np.random.RandomState(5)
257          x = random_state.randn(size).astype(dtype)
258          if dtype in (np.complex64, np.complex128):
259              x += 1j * random_state.randn(size)
260          h = np.arange(1, 1 + h_len, dtype=x.real.dtype)
261  
262          y = upfirdn(h, x, up=1, down=1, mode=mode)
263          # expected result: pad the input, filter with zero padding, then crop
264          npad = h_len - 1
265          if mode in ['antisymmetric', 'antireflect', 'smooth', 'line']:
266              # use _pad_test test function for modes not supported by np.pad.
267              xpad = _pad_test(x, npre=npad, npost=npad, mode=mode)
268          else:
269              xpad = np.pad(x, npad, mode=mode)
270          ypad = upfirdn(h, xpad, up=1, down=1, mode='constant')
271          y_expected = ypad[npad:-npad]
272  
273          atol = rtol = np.finfo(dtype).eps * 1e2
274          xp_assert_close(y, y_expected, atol=atol, rtol=rtol)

277  def test_output_len_long_input():
278      # Regression test for gh-17375.  On Windows, a large enough input
279      # that should have been well within the capabilities of 64 bit integers
280      # would result in a 32 bit overflow because of a bug in Cython 0.29.32.
281      len_h = 1001
282      in_len = 10**8
283      up = 320
284      down = 441
285      out_len = _output_len(len_h, in_len, up, down)
286      # The expected value was computed "by hand" from the formula
287      #   (((in_len - 1) * up + len_h) - 1) // down + 1
288      assert out_len == 72562360
