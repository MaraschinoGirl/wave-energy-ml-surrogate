10   class TFDataLayer(Layer):
11       """Layer that can safely used in a tf.data pipeline. 
12    
13       The `call()` method must solely rely on `self.backend` ops. 
14    
15       Only supports a single input tensor argument. 
16       """
17   
18       def __init__(self, **kwargs):
19           super().__init__(**kwargs)
20           self.backend = backend_utils.DynamicBackend()
21           self._allow_non_tensor_positional_args = True
22   
23       def __call__(self, inputs, **kwargs):
24           sample_input = tree.flatten(inputs)[0]
25           if (
26               not isinstance(sample_input, keras.KerasTensor)
27               and backend_utils.in_tf_graph()
28               and not jax_utils.is_in_jax_tracing_scope(sample_input)
29           ):
30               # We're in a TF graph, e.g. a tf.data pipeline.
31               self.backend.set_backend("tensorflow")
32               inputs = tree.map_structure(
33                   lambda x: self.backend.convert_to_tensor(
34                       x, dtype=self.compute_dtype
35                   ),
36                   inputs,
37               )
38               switch_convert_input_args = False
39               if self._convert_input_args:
40                   self._convert_input_args = False
41                   switch_convert_input_args = True
42               try:
43                   outputs = super().__call__(inputs, **kwargs)
44               finally:
45                   self.backend.reset()
46                   if switch_convert_input_args:
47                       self._convert_input_args = True
48               return outputs
49           return super().__call__(inputs, **kwargs)

51       @tracking.no_automatic_dependency_tracking
52       def _get_seed_generator(self, backend=None):
53           if backend is None or backend == keras.backend.backend():
54               return self.generator
55           if not hasattr(self, "_backend_generators"):
56               self._backend_generators = {}
57           if backend in self._backend_generators:
58               return self._backend_generators[backend]
59           seed_generator = SeedGenerator(self.seed, backend=self.backend)
60           self._backend_generators[backend] = seed_generator
61           return seed_generator
62   
63       def convert_weight(self, weight):
64           """Convert the weight if it is from the a different backend."""
65           if self.backend.name == keras.backend.backend():
66               return weight
67           else:
68               weight = keras.ops.convert_to_numpy(weight)
69               return self.backend.convert_to_tensor(weight)
