50   def _retry_with_clean_cache(
51       openml_path: str,
52       data_home: Optional[str],
53       no_retry_exception: Optional[Exception] = None,
54   ) -> Callable:
55       """If the first call to the decorated function fails, the local cached 
56       file is removed, and the function is called again. If ``data_home`` is 
57       ``None``, then the function is called once. We can provide a specific 
58       exception to not retry on using `no_retry_exception` parameter. 
59       """
60   
61       def decorator(f):
62           @wraps(f)
63           def wrapper(*args, **kw):
64               if data_home is None:
65                   return f(*args, **kw)
66               try:
67                   return f(*args, **kw)
68               except URLError:
69                   raise
70               except Exception as exc:
71                   if no_retry_exception is not None and isinstance(
72                       exc, no_retry_exception
73                   ):
74                       raise
75                   warn("Invalid cache, redownloading file", RuntimeWarning)
76                   local_path = _get_local_path(openml_path, data_home)
77                   if os.path.exists(local_path):
78                       os.unlink(local_path)
79                   return f(*args, **kw)
80   
81           return wrapper
82   
83       return decorator

86   def _retry_on_network_error(
87       n_retries: int = 3, delay: float = 1.0, url: str = ""
88   ) -> Callable:
89       """If the function call results in a network error, call the function again 
90       up to ``n_retries`` times with a ``delay`` between each call. If the error 
91       has a 412 status code, don't call the function again as this is a specific 
92       OpenML error. 
93       The url parameter is used to give more information to the user about the 
94       error. 
95       """
96   
97       def decorator(f):
98           @wraps(f)
99           def wrapper(*args, **kwargs):
100              retry_counter = n_retries
101              while True:
102                  try:
103                      return f(*args, **kwargs)
104                  except (URLError, TimeoutError) as e:
105                      # 412 is a specific OpenML error code.
106                      if isinstance(e, HTTPError) and e.code == 412:
107                          raise
108                      if retry_counter == 0:
109                          raise
110                      warn(
111                          f"A network error occurred while downloading {url}. Retrying..."
112                      )
113                      retry_counter -= 1
114                      time.sleep(delay)
115  
116          return wrapper
117  
118      return decorator

121  def _open_openml_url(
122      openml_path: str, data_home: Optional[str], n_retries: int = 3, delay: float = 1.0
123  ):
124      """ 
125      Returns a resource from OpenML.org. Caches it to data_home if required. 
126   
127      Parameters 
128      ---------- 
129      openml_path : str 
130          OpenML URL that will be accessed. This will be prefixes with 
131          _OPENML_PREFIX. 
132   
133      data_home : str 
134          Directory to which the files will be cached. If None, no caching will 
135          be applied. 
136   
137      n_retries : int, default=3 
138          Number of retries when HTTP errors are encountered. Error with status 
139          code 412 won't be retried as they represent OpenML generic errors. 
140   
141      delay : float, default=1.0 
142          Number of seconds between retries. 
143   
144      Returns 
145      ------- 
146      result : stream 
147          A stream to the OpenML resource. 
148      """
149  
150      def is_gzip_encoded(_fsrc):
151          return _fsrc.info().get("Content-Encoding", "") == "gzip"
152  
153      req = Request(_OPENML_PREFIX + openml_path)
154      req.add_header("Accept-encoding", "gzip")
155  
156      if data_home is None:
157          fsrc = _retry_on_network_error(n_retries, delay, req.full_url)(urlopen)(req)
158          if is_gzip_encoded(fsrc):
159              return gzip.GzipFile(fileobj=fsrc, mode="rb")
160          return fsrc
161  
162      local_path = _get_local_path(openml_path, data_home)
163      dir_name, file_name = os.path.split(local_path)
164      if not os.path.exists(local_path):
165          os.makedirs(dir_name, exist_ok=True)
166          try:
167              # Create a tmpdir as a subfolder of dir_name where the final file will
168              # be moved to if the download is successful. This guarantees that the
169              # renaming operation to the final location is atomic to ensure the
170              # concurrence safety of the dataset caching mechanism.
171              with TemporaryDirectory(dir=dir_name) as tmpdir:
172                  with closing(
173                      _retry_on_network_error(n_retries, delay, req.full_url)(urlopen)(
174                          req
175                      )
176                  ) as fsrc:
177                      opener: Callable
178                      if is_gzip_encoded(fsrc):
179                          opener = open
180                      else:
181                          opener = gzip.GzipFile
182                      with opener(os.path.join(tmpdir, file_name), "wb") as fdst:
183                          shutil.copyfileobj(fsrc, fdst)
184                  shutil.move(fdst.name, local_path)
185          except Exception:
186              if os.path.exists(local_path):
187                  os.unlink(local_path)
188              raise
189  
190      # XXX: First time, decompression will not be necessary (by using fsrc), but
191      # it will happen nonetheless
192      return gzip.GzipFile(local_path, "rb")

201  def _get_json_content_from_openml_api(
202      url: str,
203      error_message: Optional[str],
204      data_home: Optional[str],
205      n_retries: int = 3,
206      delay: float = 1.0,
207  ) -> Dict:
208      """ 
209      Loads json data from the openml api. 
210   
211      Parameters 
212      ---------- 
213      url : str 
214          The URL to load from. Should be an official OpenML endpoint. 
215   
216      error_message : str or None 
217          The error message to raise if an acceptable OpenML error is thrown 
218          (acceptable error is, e.g., data id not found. Other errors, like 404's 
219          will throw the native error message). 
220   
221      data_home : str or None 
222          Location to cache the response. None if no cache is required. 
223   
224      n_retries : int, default=3 
225          Number of retries when HTTP errors are encountered. Error with status 
226          code 412 won't be retried as they represent OpenML generic errors. 
227   
228      delay : float, default=1.0 
229          Number of seconds between retries. 
230   
231      Returns 
232      ------- 
233      json_data : json 
234          the json result from the OpenML server if the call was successful. 
235          An exception otherwise. 
236      """
237  
238      @_retry_with_clean_cache(url, data_home=data_home)
239      def _load_json():
240          with closing(
241              _open_openml_url(url, data_home, n_retries=n_retries, delay=delay)
242          ) as response:
243              return json.loads(response.read().decode("utf-8"))
244  
245      try:
246          return _load_json()
247      except HTTPError as error:
248          # 412 is an OpenML specific error code, indicating a generic error
249          # (e.g., data not found)
250          if error.code != 412:
251              raise error
252  
253      # 412 error, not in except for nicer traceback
254      raise OpenMLError(error_message)

257  def _get_data_info_by_name(
258      name: str,
259      version: Union[int, str],
260      data_home: Optional[str],
261      n_retries: int = 3,
262      delay: float = 1.0,
263  ):
264      """ 
265      Utilizes the openml dataset listing api to find a dataset by 
266      name/version 
267      OpenML api function: 
268      https://www.openml.org/api_docs#!/data/get_data_list_data_name_data_name 
269   
270      Parameters 
271      ---------- 
272      name : str 
273          name of the dataset 
274   
275      version : int or str 
276          If version is an integer, the exact name/version will be obtained from 
277          OpenML. If version is a string (value: "active") it will take the first 
278          version from OpenML that is annotated as active. Any other string 
279          values except "active" are treated as integer. 
280   
281      data_home : str or None 
282          Location to cache the response. None if no cache is required. 
283   
284      n_retries : int, default=3 
285          Number of retries when HTTP errors are encountered. Error with status 
286          code 412 won't be retried as they represent OpenML generic errors. 
287   
288      delay : float, default=1.0 
289          Number of seconds between retries. 
290   
291      Returns 
292      ------- 
293      first_dataset : json 
294          json representation of the first dataset object that adhired to the 
295          search criteria 
296   
297      """
298      if version == "active":
299          # situation in which we return the oldest active version
300          url = _SEARCH_NAME.format(name) + "/status/active/"
301          error_msg = "No active dataset {} found.".format(name)
302          json_data = _get_json_content_from_openml_api(
303              url,
304              error_msg,
305              data_home=data_home,
306              n_retries=n_retries,
307              delay=delay,
308          )
309          res = json_data["data"]["dataset"]
310          if len(res) > 1:
311              first_version = version = res[0]["version"]
312              warning_msg = (
313                  "Multiple active versions of the dataset matching the name"
314                  f" {name} exist. Versions may be fundamentally different, "
315                  f"returning version {first_version}. "
316                  "Available versions:\n"
317              )
318              for r in res:
319                  warning_msg += f"- version {r['version']}, status: {r['status']}\n"
320                  warning_msg += (
321                      f"  url: https://www.openml.org/search?type=data&id={r['did']}\n"
322                  )
323              warn(warning_msg)
324          return res[0]
325  
326      # an integer version has been provided
327      url = (_SEARCH_NAME + "/data_version/{}").format(name, version)
328      try:
329          json_data = _get_json_content_from_openml_api(
330              url,
331              error_message=None,
332              data_home=data_home,
333              n_retries=n_retries,
334              delay=delay,
335          )
336      except OpenMLError:
337          # we can do this in 1 function call if OpenML does not require the
338          # specification of the dataset status (i.e., return datasets with a
339          # given name / version regardless of active, deactivated, etc. )
340          # TODO: feature request OpenML.
341          url += "/status/deactivated"
342          error_msg = "Dataset {} with version {} not found.".format(name, version)
343          json_data = _get_json_content_from_openml_api(
344              url,
345              error_msg,
346              data_home=data_home,
347              n_retries=n_retries,
348              delay=delay,
349          )
350  
351      return json_data["data"]["dataset"][0]

365          error_message,
366          data_home=data_home,
367          n_retries=n_retries,
368          delay=delay,
369      )

415  def _get_num_samples(data_qualities: OpenmlQualitiesType) -> int:
416      """Get the number of samples from data qualities. 
417   
418      Parameters 
419      ---------- 
420      data_qualities : list of dict 
421          Used to retrieve the number of instances (samples) in the dataset. 
422   
423      Returns 
424      ------- 
425      n_samples : int 
426          The number of samples in the dataset or -1 if data qualities are 
427          unavailable. 
428      """
429      # If the data qualities are unavailable, we return -1
430      default_n_samples = -1
431  
432      qualities = {d["name"]: d["value"] for d in data_qualities}
433      return int(float(qualities.get("NumberOfInstances", default_n_samples)))

436  def _load_arff_response(
437      url: str,
438      data_home: Optional[str],
439      parser: str,
440      output_type: str,
441      openml_columns_info: dict,
442      feature_names_to_select: List[str],
443      target_names_to_select: List[str],
444      shape: Optional[Tuple[int, int]],
445      md5_checksum: str,
446      n_retries: int = 3,
447      delay: float = 1.0,
448      read_csv_kwargs: Optional[Dict] = None,
449  ):
450      """Load the ARFF data associated with the OpenML URL. 
451   
452      In addition of loading the data, this function will also check the 
453      integrity of the downloaded file from OpenML using MD5 checksum. 
454   
455      Parameters 
456      ---------- 
457      url : str 
458          The URL of the ARFF file on OpenML. 
459   
460      data_home : str 
461          The location where to cache the data. 
462   
463      parser : {"liac-arff", "pandas"} 
464          The parser used to parse the ARFF file. 
465   
466      output_type : {"numpy", "pandas", "sparse"} 
467          The type of the arrays that will be returned. The possibilities are: 
468   
469          - `"numpy"`: both `X` and `y` will be NumPy arrays; 
470          - `"sparse"`: `X` will be sparse matrix and `y` will be a NumPy array; 
471          - `"pandas"`: `X` will be a pandas DataFrame and `y` will be either a 
472            pandas Series or DataFrame. 
473   
474      openml_columns_info : dict 
475          The information provided by OpenML regarding the columns of the ARFF 
476          file. 
477   
478      feature_names_to_select : list of str 
479          The list of the features to be selected. 
480   
481      target_names_to_select : list of str 
482          The list of the target variables to be selected. 
483   
484      shape : tuple or None 
485          With `parser="liac-arff"`, when using a generator to load the data, 
486          one needs to provide the shape of the data beforehand. 
487   
488      md5_checksum : str 
489          The MD5 checksum provided by OpenML to check the data integrity. 
490   
491      n_retries : int, default=3 
492          The number of times to retry downloading the data if it fails. 
493   
494      delay : float, default=1.0 
495          The delay between two consecutive downloads in seconds. 
496   
497      read_csv_kwargs : dict, default=None 
498          Keyword arguments to pass to `pandas.read_csv` when using the pandas parser. 
499          It allows to overwrite the default options. 
500   
501          .. versionadded:: 1.3 
502   
503      Returns 
504      ------- 
505      X : {ndarray, sparse matrix, dataframe} 
506          The data matrix. 
507   
508      y : {ndarray, dataframe, series} 
509          The target. 
510   
511      frame : dataframe or None 
512          A dataframe containing both `X` and `y`. `None` if 
513          `output_array_type != "pandas"`. 
514   
515      categories : list of str or None 
516          The names of the features that are categorical. `None` if 
517          `output_array_type == "pandas"`. 
518      """
519      gzip_file = _open_openml_url(url, data_home, n_retries=n_retries, delay=delay)
520      with closing(gzip_file):
521          md5 = hashlib.md5()
522          for chunk in iter(lambda: gzip_file.read(4096), b""):
523              md5.update(chunk)
524          actual_md5_checksum = md5.hexdigest()
525  
526      if actual_md5_checksum != md5_checksum:
527          raise ValueError(
528              f"md5 checksum of local file for {url} does not match description: "
529              f"expected: {md5_checksum} but got {actual_md5_checksum}. "
530              "Downloaded file could have been modified / corrupted, clean cache "
531              "and retry..."
532          )
533  
534      def _open_url_and_load_gzip_file(url, data_home, n_retries, delay, arff_params):
535          gzip_file = _open_openml_url(url, data_home, n_retries=n_retries, delay=delay)
536          with closing(gzip_file):
537              return load_arff_from_gzip_file(gzip_file, **arff_params)
538  
539      arff_params: Dict = dict(
540          parser=parser,
541          output_type=output_type,
542          openml_columns_info=openml_columns_info,
543          feature_names_to_select=feature_names_to_select,
544          target_names_to_select=target_names_to_select,
545          shape=shape,
546          read_csv_kwargs=read_csv_kwargs or {},
547      )
548      try:
549          X, y, frame, categories = _open_url_and_load_gzip_file(
550              url, data_home, n_retries, delay, arff_params
551          )
552      except Exception as exc:
553          if parser != "pandas":
554              raise
555  
556          from pandas.errors import ParserError
557  
558          if not isinstance(exc, ParserError):
559              raise
560  
561          # A parsing error could come from providing the wrong quotechar
562          # to pandas. By default, we use a double quote. Thus, we retry
563          # with a single quote before to raise the error.
564          arff_params["read_csv_kwargs"].update(quotechar="'")
565          X, y, frame, categories = _open_url_and_load_gzip_file(
566              url, data_home, n_retries, delay, arff_params
567          )
568  
569      return X, y, frame, categories

572  def _download_data_to_bunch(
573      url: str,
574      sparse: bool,
575      data_home: Optional[str],
576      *,
577      as_frame: bool,
578      openml_columns_info: List[dict],
579      data_columns: List[str],
580      target_columns: List[str],
581      shape: Optional[Tuple[int, int]],
582      md5_checksum: str,
583      n_retries: int = 3,
584      delay: float = 1.0,
585      parser: str,
586      read_csv_kwargs: Optional[Dict] = None,
587  ):
588      """Download ARFF data, load it to a specific container and create to Bunch. 
589   
590      This function has a mechanism to retry/cache/clean the data. 
591   
592      Parameters 
593      ---------- 
594      url : str 
595          The URL of the ARFF file on OpenML. 
596   
597      sparse : bool 
598          Whether the dataset is expected to use the sparse ARFF format. 
599   
600      data_home : str 
601          The location where to cache the data. 
602   
603      as_frame : bool 
604          Whether or not to return the data into a pandas DataFrame. 
605   
606      openml_columns_info : list of dict 
607          The information regarding the columns provided by OpenML for the 
608          ARFF dataset. The information is stored as a list of dictionaries. 
609   
610      data_columns : list of str 
611          The list of the features to be selected. 
612   
613      target_columns : list of str 
614          The list of the target variables to be selected. 
615   
616      shape : tuple or None 
617          With `parser="liac-arff"`, when using a generator to load the data, 
618          one needs to provide the shape of the data beforehand. 
619   
620      md5_checksum : str 
621          The MD5 checksum provided by OpenML to check the data integrity. 
622   
623      n_retries : int, default=3 
624          Number of retries when HTTP errors are encountered. Error with status 
625          code 412 won't be retried as they represent OpenML generic errors. 
626   
627      delay : float, default=1.0 
628          Number of seconds between retries. 
629   
630      parser : {"liac-arff", "pandas"} 
631          The parser used to parse the ARFF file. 
632   
633      read_csv_kwargs : dict, default=None 
634          Keyword arguments to pass to `pandas.read_csv` when using the pandas parser. 
635          It allows to overwrite the default options. 
636   
637          .. versionadded:: 1.3 
638   
639      Returns 
640      ------- 
641      data : :class:`~sklearn.utils.Bunch` 
642          Dictionary-like object, with the following attributes. 
643   
644          X : {ndarray, sparse matrix, dataframe} 
645              The data matrix. 
646          y : {ndarray, dataframe, series} 
647              The target. 
648          frame : dataframe or None 
649              A dataframe containing both `X` and `y`. `None` if 
650              `output_array_type != "pandas"`. 
651          categories : list of str or None 
652              The names of the features that are categorical. `None` if 
653              `output_array_type == "pandas"`. 
654      """
655      # Prepare which columns and data types should be returned for the X and y
656      features_dict = {feature["name"]: feature for feature in openml_columns_info}
657  
658      if sparse:
659          output_type = "sparse"
660      elif as_frame:
661          output_type = "pandas"
662      else:
663          output_type = "numpy"
664  
665      # XXX: target columns should all be categorical or all numeric
666      _verify_target_data_type(features_dict, target_columns)
667      for name in target_columns:
668          column_info = features_dict[name]
669          n_missing_values = int(column_info["number_of_missing_values"])
670          if n_missing_values > 0:
671              raise ValueError(
672                  f"Target column '{column_info['name']}' has {n_missing_values} missing "
673                  "values. Missing values are not supported for target columns."
674              )
675  
676      no_retry_exception = None
677      if parser == "pandas":
678          # If we get a ParserError with pandas, then we don't want to retry and we raise
679          # early.
680          from pandas.errors import ParserError
681  
682          no_retry_exception = ParserError
683  
684      X, y, frame, categories = _retry_with_clean_cache(
685          url, data_home, no_retry_exception
686      )(_load_arff_response)(
687          url,
688          data_home,
689          parser=parser,
690          output_type=output_type,
691          openml_columns_info=features_dict,
692          feature_names_to_select=data_columns,
693          target_names_to_select=target_columns,
694          shape=shape,
695          md5_checksum=md5_checksum,
696          n_retries=n_retries,
697          delay=delay,
698          read_csv_kwargs=read_csv_kwargs,
699      )
700  
701      return Bunch(
702          data=X,
703          target=y,
704          frame=frame,
705          categories=categories,
706          feature_names=data_columns,
707          target_names=target_columns,
708      )

711  def _verify_target_data_type(features_dict, target_columns):
712      # verifies the data type of the y array in case there are multiple targets
713      # (throws an error if these targets do not comply with sklearn support)
714      if not isinstance(target_columns, list):
715          raise ValueError("target_column should be list, got: %s" % type(target_columns))
716      found_types = set()
717      for target_column in target_columns:
718          if target_column not in features_dict:
719              raise KeyError(f"Could not find target_column='{target_column}'")
720          if features_dict[target_column]["data_type"] == "numeric":
721              found_types.add(np.float64)
722          else:
723              found_types.add(object)
724  
725          # note: we compare to a string, not boolean
726          if features_dict[target_column]["is_ignore"] == "true":
727              warn(f"target_column='{target_column}' has flag is_ignore.")
728          if features_dict[target_column]["is_row_identifier"] == "true":
729              warn(f"target_column='{target_column}' has flag is_row_identifier.")
730      if len(found_types) > 1:
731          raise ValueError(
732              "Can only handle homogeneous multi-target datasets, "
733              "i.e., all targets are either numeric or "
734              "categorical."
735          )

738  def _valid_data_column_names(features_list, target_columns):
739      # logic for determining on which columns can be learned. Note that from the
740      # OpenML guide follows that columns that have the `is_row_identifier` or
741      # `is_ignore` flag, these can not be learned on. Also target columns are
742      # excluded.
743      valid_data_column_names = []
744      for feature in features_list:
745          if (
746              feature["name"] not in target_columns
747              and feature["is_ignore"] != "true"
748              and feature["is_row_identifier"] != "true"
749          ):
750              valid_data_column_names.append(feature["name"])
751      return valid_data_column_names

773  def fetch_openml(
774      name: Optional[str] = None,
775      *,
776      version: Union[str, int] = "active",
777      data_id: Optional[int] = None,
778      data_home: Optional[Union[str, os.PathLike]] = None,
779      target_column: Optional[Union[str, List]] = "default-target",
780      cache: bool = True,
781      return_X_y: bool = False,
782      as_frame: Union[str, bool] = "auto",
783      n_retries: int = 3,
784      delay: float = 1.0,
785      parser: str = "auto",
786      read_csv_kwargs: Optional[Dict] = None,
787  ):
788      """Fetch dataset from openml by name or dataset id. 
789   
790      Datasets are uniquely identified by either an integer ID or by a 
791      combination of name and version (i.e. there might be multiple 
792      versions of the 'iris' dataset). Please give either name or data_id 
793      (not both). In case a name is given, a version can also be 
794      provided. 
795   
796      Read more in the :ref:`User Guide <openml>`. 
797   
798      .. versionadded:: 0.20 
799   
800      .. note:: EXPERIMENTAL 
801   
802          The API is experimental (particularly the return value structure), 
803          and might have small backward-incompatible changes without notice 
804          or warning in future releases. 
805   
806      Parameters 
807      ---------- 
808      name : str, default=None 
809          String identifier of the dataset. Note that OpenML can have multiple 
810          datasets with the same name. 
811   
812      version : int or 'active', default='active' 
813          Version of the dataset. Can only be provided if also ``name`` is given. 
814          If 'active' the oldest version that's still active is used. Since 
815          there may be more than one active version of a dataset, and those 
816          versions may fundamentally be different from one another, setting an 
817          exact version is highly recommended. 
818   
819      data_id : int, default=None 
820          OpenML ID of the dataset. The most specific way of retrieving a 
821          dataset. If data_id is not given, name (and potential version) are 
822          used to obtain a dataset. 
823   
824      data_home : str or path-like, default=None 
825          Specify another download and cache folder for the data sets. By default 
826          all scikit-learn data is stored in '~/scikit_learn_data' subfolders. 
827   
828      target_column : str, list or None, default='default-target' 
829          Specify the column name in the data to use as target. If 
830          'default-target', the standard target column a stored on the server 
831          is used. If ``None``, all columns are returned as data and the 
832          target is ``None``. If list (of strings), all columns with these names 
833          are returned as multi-target (Note: not all scikit-learn classifiers 
834          can handle all types of multi-output combinations). 
835   
836      cache : bool, default=True 
837          Whether to cache the downloaded datasets into `data_home`. 
838   
839      return_X_y : bool, default=False 
840          If True, returns ``(data, target)`` instead of a Bunch object. See 
841          below for more information about the `data` and `target` objects. 
842   
843      as_frame : bool or 'auto', default='auto' 
844          If True, the data is a pandas DataFrame including columns with 
845          appropriate dtypes (numeric, string or categorical). The target is 
846          a pandas DataFrame or Series depending on the number of target_columns. 
847          The Bunch will contain a ``frame`` attribute with the target and the 
848          data. If ``return_X_y`` is True, then ``(data, target)`` will be pandas 
849          DataFrames or Series as describe above. 
850   
851          If `as_frame` is 'auto', the data and target will be converted to 
852          DataFrame or Series as if `as_frame` is set to True, unless the dataset 
853          is stored in sparse format. 
854   
855          If `as_frame` is False, the data and target will be NumPy arrays and 
856          the `data` will only contain numerical values when `parser="liac-arff"` 
857          where the categories are provided in the attribute `categories` of the 
858          `Bunch` instance. When `parser="pandas"`, no ordinal encoding is made. 
859   
860          .. versionchanged:: 0.24 
861             The default value of `as_frame` changed from `False` to `'auto'` 
862             in 0.24. 
863   
864      n_retries : int, default=3 
865          Number of retries when HTTP errors or network timeouts are encountered. 
866          Error with status code 412 won't be retried as they represent OpenML 
867          generic errors. 
868   
869      delay : float, default=1.0 
870          Number of seconds between retries. 
871   
872      parser : {"auto", "pandas", "liac-arff"}, default="auto" 
873          Parser used to load the ARFF file. Two parsers are implemented: 
874   
875          - `"pandas"`: this is the most efficient parser. However, it requires 
876            pandas to be installed and can only open dense datasets. 
877          - `"liac-arff"`: this is a pure Python ARFF parser that is much less 
878            memory- and CPU-efficient. It deals with sparse ARFF datasets. 
879   
880          If `"auto"`, the parser is chosen automatically such that `"liac-arff"` 
881          is selected for sparse ARFF datasets, otherwise `"pandas"` is selected. 
882   
883          .. versionadded:: 1.2 
884          .. versionchanged:: 1.4 
885             The default value of `parser` changes from `"liac-arff"` to 
886             `"auto"`. 
887   
888      read_csv_kwargs : dict, default=None 
889          Keyword arguments passed to :func:`pandas.read_csv` when loading the data 
890          from a ARFF file and using the pandas parser. It can allow to 
891          overwrite some default parameters. 
892   
893          .. versionadded:: 1.3 
894   
895      Returns 
896      ------- 
897      data : :class:`~sklearn.utils.Bunch` 
898          Dictionary-like object, with the following attributes. 
899   
900          data : np.array, scipy.sparse.csr_matrix of floats, or pandas DataFrame 
901              The feature matrix. Categorical features are encoded as ordinals. 
902          target : np.array, pandas Series or DataFrame 
903              The regression target or classification labels, if applicable. 
904              Dtype is float if numeric, and object if categorical. If 
905              ``as_frame`` is True, ``target`` is a pandas object. 
906          DESCR : str 
907              The full description of the dataset. 
908          feature_names : list 
909              The names of the dataset columns. 
910          target_names: list 
911              The names of the target columns. 
912   
913          .. versionadded:: 0.22 
914   
915          categories : dict or None 
916              Maps each categorical feature name to a list of values, such 
917              that the value encoded as i is ith in the list. If ``as_frame`` 
918              is True, this is None. 
919          details : dict 
920              More metadata from OpenML. 
921          frame : pandas DataFrame 
922              Only present when `as_frame=True`. DataFrame with ``data`` and 
923              ``target``. 
924   
925      (data, target) : tuple if ``return_X_y`` is True 
926   
927          .. note:: EXPERIMENTAL 
928   
929              This interface is **experimental** and subsequent releases may 
930              change attributes without notice (although there should only be 
931              minor changes to ``data`` and ``target``). 
932   
933          Missing values in the 'data' are represented as NaN's. Missing values 
934          in 'target' are represented as NaN's (numerical target) or None 
935          (categorical target). 
936   
937      Notes 
938      ----- 
939      The `"pandas"` and `"liac-arff"` parsers can lead to different data types 
940      in the output. The notable differences are the following: 
941   
942      - The `"liac-arff"` parser always encodes categorical features as `str` objects. 
943        To the contrary, the `"pandas"` parser instead infers the type while 
944        reading and numerical categories will be casted into integers whenever 
945        possible. 
946      - The `"liac-arff"` parser uses float64 to encode numerical features 
947        tagged as 'REAL' and 'NUMERICAL' in the metadata. The `"pandas"` 
948        parser instead infers if these numerical features corresponds 
949        to integers and uses panda's Integer extension dtype. 
950      - In particular, classification datasets with integer categories are 
951        typically loaded as such `(0, 1, ...)` with the `"pandas"` parser while 
952        `"liac-arff"` will force the use of string encoded class labels such as 
953        `"0"`, `"1"` and so on. 
954      - The `"pandas"` parser will not strip single quotes - i.e. `'` - from 
955        string columns. For instance, a string `'my string'` will be kept as is 
956        while the `"liac-arff"` parser will strip the single quotes. For 
957        categorical columns, the single quotes are stripped from the values. 
958   
959      In addition, when `as_frame=False` is used, the `"liac-arff"` parser 
960      returns ordinally encoded data where the categories are provided in the 
961      attribute `categories` of the `Bunch` instance. Instead, `"pandas"` returns 
962      a NumPy array were the categories are not encoded. 
963   
964      Examples 
965      -------- 
966      >>> from sklearn.datasets import fetch_openml 
967      >>> adult = fetch_openml("adult", version=2)  # doctest: +SKIP 
968      >>> adult.frame.info()  # doctest: +SKIP 
969      <class 'pandas.core.frame.DataFrame'> 
970      RangeIndex: 48842 entries, 0 to 48841 
971      Data columns (total 15 columns): 
972       #   Column          Non-Null Count  Dtype 
973      ---  ------          --------------  ----- 
974       0   age             48842 non-null  int64 
975       1   workclass       46043 non-null  category 
976       2   fnlwgt          48842 non-null  int64 
977       3   education       48842 non-null  category 
978       4   education-num   48842 non-null  int64 
979       5   marital-status  48842 non-null  category 
980       6   occupation      46033 non-null  category 
981       7   relationship    48842 non-null  category 
982       8   race            48842 non-null  category 
983       9   sex             48842 non-null  category 
984       10  capital-gain    48842 non-null  int64 
985       11  capital-loss    48842 non-null  int64 
986       12  hours-per-week  48842 non-null  int64 
987       13  native-country  47985 non-null  category 
988       14  class           48842 non-null  category 
989      dtypes: category(9), int64(6) 
990      memory usage: 2.7 MB 
991      """
992      if cache is False:
993          # no caching will be applied
994          data_home = None
995      else:
996          data_home = get_data_home(data_home=data_home)
997          data_home = join(str(data_home), "openml")
998  
999      # check valid function arguments. data_id XOR (name, version) should be
1000     # provided
1001     if name is not None:
1002         # OpenML is case-insensitive, but the caching mechanism is not
1003         # convert all data names (str) to lower case
1004         name = name.lower()
1005         if data_id is not None:
1006             raise ValueError(
1007                 "Dataset data_id={} and name={} passed, but you can only "
1008                 "specify a numeric data_id or a name, not "
1009                 "both.".format(data_id, name)
1010             )
1011         data_info = _get_data_info_by_name(
1012             name, version, data_home, n_retries=n_retries, delay=delay
1013         )
1014         data_id = data_info["did"]
1015     elif data_id is not None:
1016         # from the previous if statement, it is given that name is None
1017         if version != "active":
1018             raise ValueError(
1019                 "Dataset data_id={} and version={} passed, but you can only "
1020                 "specify a numeric data_id or a version, not "
1021                 "both.".format(data_id, version)
1022             )
1023     else:
1024         raise ValueError(
1025             "Neither name nor data_id are provided. Please provide name or data_id."
1026         )
1027 
1028     data_description = _get_data_description_by_id(data_id, data_home)
1029     if data_description["status"] != "active":
1030         warn(
1031             "Version {} of dataset {} is inactive, meaning that issues have "
1032             "been found in the dataset. Try using a newer version from "
1033             "this URL: {}".format(
1034                 data_description["version"],
1035                 data_description["name"],
1036                 data_description["url"],
1037             )
1038         )
1039     if "error" in data_description:
1040         warn(
1041             "OpenML registered a problem with the dataset. It might be "
1042             "unusable. Error: {}".format(data_description["error"])
1043         )
1044     if "warning" in data_description:
1045         warn(
1046             "OpenML raised a warning on the dataset. It might be "
1047             "unusable. Warning: {}".format(data_description["warning"])
1048         )
1049 
1050     return_sparse = data_description["format"].lower() == "sparse_arff"
1051     as_frame = not return_sparse if as_frame == "auto" else as_frame
1052     if parser == "auto":
1053         parser_ = "liac-arff" if return_sparse else "pandas"
1054     else:
1055         parser_ = parser
1056 
1057     if parser_ == "pandas":
1058         try:
1059             check_pandas_support("`fetch_openml`")
1060         except ImportError as exc:
1061             if as_frame:
1062                 err_msg = (
1063                     "Returning pandas objects requires pandas to be installed. "
1064                     "Alternatively, explicitly set `as_frame=False` and "
1065                     "`parser='liac-arff'`."
1066                 )
1067             else:
1068                 err_msg = (
1069                     f"Using `parser={parser!r}` with dense data requires pandas to be "
1070                     "installed. Alternatively, explicitly set `parser='liac-arff'`."
1071                 )
1072             raise ImportError(err_msg) from exc
1073 
1074     if return_sparse:
1075         if as_frame:
1076             raise ValueError(
1077                 "Sparse ARFF datasets cannot be loaded with as_frame=True. "
1078                 "Use as_frame=False or as_frame='auto' instead."
1079             )
1080         if parser_ == "pandas":
1081             raise ValueError(
1082                 f"Sparse ARFF datasets cannot be loaded with parser={parser!r}. "
1083                 "Use parser='liac-arff' or parser='auto' instead."
1084             )
1085 
1086     # download data features, meta-info about column types
1087     features_list = _get_data_features(data_id, data_home)
1088 
1089     if not as_frame:
1090         for feature in features_list:
1091             if "true" in (feature["is_ignore"], feature["is_row_identifier"]):
1092                 continue
1093             if feature["data_type"] == "string":
1094                 raise ValueError(
1095                     "STRING attributes are not supported for "
1096                     "array representation. Try as_frame=True"
1097                 )
1098 
1099     if target_column == "default-target":
1100         # determines the default target based on the data feature results
1101         # (which is currently more reliable than the data description;
1102         # see issue: https://github.com/openml/OpenML/issues/768)
1103         target_columns = [
1104             feature["name"]
1105             for feature in features_list
1106             if feature["is_target"] == "true"
1107         ]
1108     elif isinstance(target_column, str):
1109         # for code-simplicity, make target_column by default a list
1110         target_columns = [target_column]
1111     elif target_column is None:
1112         target_columns = []
1113     else:
1114         # target_column already is of type list
1115         target_columns = target_column
1116     data_columns = _valid_data_column_names(features_list, target_columns)
1117 
1118     shape: Optional[Tuple[int, int]]
1119     # determine arff encoding to return
1120     if not return_sparse:
1121         # The shape must include the ignored features to keep the right indexes
1122         # during the arff data conversion.
1123         data_qualities = _get_data_qualities(data_id, data_home)
1124         shape = _get_num_samples(data_qualities), len(features_list)
1125     else:
1126         shape = None
1127 
1128     # obtain the data
1129     url = _DATA_FILE.format(data_description["file_id"])
1130     bunch = _download_data_to_bunch(
1131         url,
1132         return_sparse,
1133         data_home,
1134         as_frame=bool(as_frame),
1135         openml_columns_info=features_list,
1136         shape=shape,
1137         target_columns=target_columns,
1138         data_columns=data_columns,
1139         md5_checksum=data_description["md5_checksum"],
1140         n_retries=n_retries,
1141         delay=delay,
1142         parser=parser_,
1143         read_csv_kwargs=read_csv_kwargs,
1144     )
1145 
1146     if return_X_y:
1147         return bunch.data, bunch.target
1148 
1149     description = "{}\n\nDownloaded from openml.org.".format(
1150         data_description.pop("description")
1151     )
1152 
1153     bunch.update(
1154         DESCR=description,
1155         details=data_description,
1156         url="https://www.openml.org/d/{}".format(data_id),
1157     )
1158 
1159     return bunch
