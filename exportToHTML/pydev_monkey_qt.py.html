8    def set_trace_in_qt():
9        from _pydevd_bundle.pydevd_comm import get_global_debugger
10   
11       py_db = get_global_debugger()
12       if py_db is not None:
13           threading.current_thread()  # Create the dummy thread for qt.
14           py_db.enable_tracing()

20   def patch_qt(qt_support_mode):
21       """ 
22       This method patches qt (PySide2, PySide, PyQt4, PyQt5) so that we have hooks to set the tracing for QThread. 
23       """
24       if not qt_support_mode:
25           return
26   
27       if qt_support_mode is True or qt_support_mode == "True":
28           # do not break backward compatibility
29           qt_support_mode = "auto"
30   
31       if qt_support_mode == "auto":
32           qt_support_mode = os.getenv("PYDEVD_PYQT_MODE", "auto")
33   
34       # Avoid patching more than once
35       global _patched_qt
36       if _patched_qt:
37           return
38   
39       pydev_log.debug("Qt support mode: %s", qt_support_mode)
40   
41       _patched_qt = True
42   
43       if qt_support_mode == "auto":
44           patch_qt_on_import = None
45           try:
46               import PySide2  # @UnresolvedImport @UnusedImport
47   
48               qt_support_mode = "pyside2"
49           except:
50               try:
51                   import Pyside  # @UnresolvedImport @UnusedImport
52   
53                   qt_support_mode = "pyside"
54               except:
55                   try:
56                       import PyQt5  # @UnresolvedImport @UnusedImport
57   
58                       qt_support_mode = "pyqt5"
59                   except:
60                       try:
61                           import PyQt4  # @UnresolvedImport @UnusedImport
62   
63                           qt_support_mode = "pyqt4"
64                       except:
65                           return
66   
67       if qt_support_mode == "pyside2":
68           try:
69               import PySide2.QtCore  # @UnresolvedImport
70   
71               _internal_patch_qt(PySide2.QtCore, qt_support_mode)
72           except:
73               return
74   
75       elif qt_support_mode == "pyside":
76           try:
77               import PySide.QtCore  # @UnresolvedImport
78   
79               _internal_patch_qt(PySide.QtCore, qt_support_mode)
80           except:
81               return
82   
83       elif qt_support_mode == "pyqt5":
84           try:
85               import PyQt5.QtCore  # @UnresolvedImport
86   
87               _internal_patch_qt(PyQt5.QtCore)
88           except:
89               return
90   
91       elif qt_support_mode == "pyqt4":
92           # Ok, we have an issue here:
93           # PyDev-452: Selecting PyQT API version using sip.setapi fails in debug mode
94           # http://pyqt.sourceforge.net/Docs/PyQt4/incompatible_apis.html
95           # Mostly, if the user uses a different API version (i.e.: v2 instead of v1),
96           # that has to be done before importing PyQt4 modules (PySide/PyQt5 don't have this issue
97           # as they only implements v2).
98           patch_qt_on_import = "PyQt4"
99   
100          def get_qt_core_module():
101              import PyQt4.QtCore  # @UnresolvedImport
102  
103              return PyQt4.QtCore
104  
105          _patch_import_to_patch_pyqt_on_import(patch_qt_on_import, get_qt_core_module)
106  
107      else:
108          raise ValueError("Unexpected qt support mode: %s" % (qt_support_mode,))

111  def _patch_import_to_patch_pyqt_on_import(patch_qt_on_import, get_qt_core_module):
112      # I don't like this approach very much as we have to patch __import__, but I like even less
113      # asking the user to configure something in the client side...
114      # So, our approach is to patch PyQt4 right before the user tries to import it (at which
115      # point he should've set the sip api version properly already anyways).
116  
117      pydev_log.debug("Setting up Qt post-import monkeypatch.")
118  
119      dotted = patch_qt_on_import + "."
120      original_import = __import__
121  
122      from _pydev_bundle._pydev_sys_patch import patch_sys_module, patch_reload, cancel_patches_in_sys_module
123  
124      patch_sys_module()
125      patch_reload()
126  
127      def patched_import(name, *args, **kwargs):
128          if patch_qt_on_import == name or name.startswith(dotted):
129              builtins.__import__ = original_import
130              cancel_patches_in_sys_module()
131              _internal_patch_qt(get_qt_core_module())  # Patch it only when the user would import the qt module
132          return original_import(name, *args, **kwargs)
133  
134      import builtins  # Py3
135  
136      builtins.__import__ = patched_import

139  def _internal_patch_qt(QtCore, qt_support_mode="auto"):
140      pydev_log.debug("Patching Qt: %s", QtCore)
141  
142      _original_thread_init = QtCore.QThread.__init__
143      _original_runnable_init = QtCore.QRunnable.__init__
144      _original_QThread = QtCore.QThread
145  
146      class FuncWrapper:
147          def __init__(self, original):
148              self._original = original
149  
150          def __call__(self, *args, **kwargs):
151              set_trace_in_qt()
152              return self._original(*args, **kwargs)
153  
154      class StartedSignalWrapper(QtCore.QObject):  # Wrapper for the QThread.started signal
155          try:
156              _signal = QtCore.Signal()  # @UndefinedVariable
157          except:
158              _signal = QtCore.pyqtSignal()  # @UndefinedVariable
159  
160          def __init__(self, thread, original_started):
161              QtCore.QObject.__init__(self)
162              self.thread = thread
163              self.original_started = original_started
164              if qt_support_mode in ("pyside", "pyside2"):
165                  self._signal = original_started
166              else:
167                  self._signal.connect(self._on_call)
168                  self.original_started.connect(self._signal)
169  
170          def connect(self, func, *args, **kwargs):
171              if qt_support_mode in ("pyside", "pyside2"):
172                  return self._signal.connect(FuncWrapper(func), *args, **kwargs)
173              else:
174                  return self._signal.connect(func, *args, **kwargs)
175  
176          def disconnect(self, *args, **kwargs):
177              return self._signal.disconnect(*args, **kwargs)
178  
179          def emit(self, *args, **kwargs):
180              return self._signal.emit(*args, **kwargs)
181  
182          def _on_call(self, *args, **kwargs):
183              set_trace_in_qt()
184  
185      class ThreadWrapper(QtCore.QThread):  # Wrapper for QThread
186          def __init__(self, *args, **kwargs):
187              _original_thread_init(self, *args, **kwargs)
188  
189              # In PyQt5 the program hangs when we try to call original run method of QThread class.
190              # So we need to distinguish instances of QThread class and instances of QThread inheritors.
191              if self.__class__.run == _original_QThread.run:
192                  self.run = self._exec_run
193              else:
194                  self._original_run = self.run
195                  self.run = self._new_run
196              self._original_started = self.started
197              self.started = StartedSignalWrapper(self, self.started)
198  
199          def _exec_run(self):
200              set_trace_in_qt()
201              self.exec_()
202              return None
203  
204          def _new_run(self):
205              set_trace_in_qt()
206              return self._original_run()
207  
208      class RunnableWrapper(QtCore.QRunnable):  # Wrapper for QRunnable
209          def __init__(self, *args, **kwargs):
210              _original_runnable_init(self, *args, **kwargs)
211  
212              self._original_run = self.run
213              self.run = self._new_run
214  
215          def _new_run(self):
216              set_trace_in_qt()
217              return self._original_run()
218  
219      QtCore.QThread = ThreadWrapper
220      QtCore.QRunnable = RunnableWrapper
