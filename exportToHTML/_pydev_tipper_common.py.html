7    
8        if inspect.ismodule(mod):

10   
11       lines = linecache.getlines(f)

13       if inspect.isclass(mod):
14           name = mod.__name__
15           pat = re.compile(r"^\s*class\s*" + name + r"\b")
16           for i in range(len(lines)):
17               if pat.match(lines[i]):
18                   return f, i, 0

22       if inspect.ismethod(mod):

33   
34       if inspect.isframe(mod):

36   
37       if inspect.iscode(mod):
38           if not hasattr(mod, "co_filename"):
39               return None, 0, 0
40   
41           if not hasattr(mod, "co_firstlineno"):
42               return mod.co_filename, 0, 0
43   
44           lnum = mod.co_firstlineno
45           pat = re.compile(r"^(\s*def\s)|(.*(?<!\w)lambda(:|\s))|^(\s*@)")
46           while lnum > 0:
47               if pat.match(lines[lnum]):
48                   break
49               lnum -= 1
50   
51           return f, lnum, 0
