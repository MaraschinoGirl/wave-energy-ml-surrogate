60   def find_gui_and_backend():
61       """Return the gui and mpl backend."""
62       matplotlib = sys.modules["matplotlib"]
63       # WARNING: this assumes matplotlib 1.1 or newer!!
64       backend = matplotlib.rcParams["backend"]
65   
66       # Translate to the real case as in 3.9 the case was forced to lowercase
67       # but our internal mapping is in the original case.
68       realcase_backend = lowercase_convert.get(backend, backend)
69   
70       # In this case, we need to find what the appropriate gui selection call
71       # should be for IPython, so we can activate inputhook accordingly
72       gui = backend2gui.get(realcase_backend, None)
73       return gui, backend

84   def is_interactive_backend(backend):
85       """Check if backend is interactive"""
86       matplotlib = sys.modules["matplotlib"]
87       new_api_version = (3, 9)
88       installed_version = (
89           _get_major_version(matplotlib),
90           _get_minor_version(matplotlib)
91       )
92   
93       if installed_version >= new_api_version:
94           interactive_bk = matplotlib.backends.backend_registry.list_builtin(
95               matplotlib.backends.BackendFilter.INTERACTIVE)
96           non_interactive_bk = matplotlib.backends.backend_registry.list_builtin(
97               matplotlib.backends.BackendFilter.NON_INTERACTIVE)
98       else:
99           from matplotlib.rcsetup import interactive_bk, non_interactive_bk  # @UnresolvedImport
100  
101      if backend in interactive_bk:
102          return True
103      elif backend in non_interactive_bk:
104          return False
105      else:
106          return matplotlib.is_interactive()

110      """Patch matplotlib function 'use'"""
111      matplotlib = sys.modules["matplotlib"]
112  
113      def patched_use(*args, **kwargs):

116          enable_gui_function(gui)

118      matplotlib.real_use = matplotlib.use
119      matplotlib.use = patched_use

133  def activate_matplotlib(enable_gui_function):
134      """Set interactive to True for interactive backends. 
135      enable_gui_function - Function which enables gui, should be run in the main thread. 
136      """
137      matplotlib = sys.modules["matplotlib"]
138      gui, backend = find_gui_and_backend()
139      is_interactive = is_interactive_backend(backend)
140      if is_interactive:
141          enable_gui_function(gui)
142          if not matplotlib.is_interactive():
143              sys.stdout.write("Backend %s is interactive backend. Turning interactive mode on.\n" % backend)
144          matplotlib.interactive(True)
145      else:
146          if matplotlib.is_interactive():
147              sys.stdout.write("Backend %s is non-interactive backend. Turning interactive mode off.\n" % backend)
148          matplotlib.interactive(False)
149      patch_use(enable_gui_function)
150      patch_is_interactive()

153  def flag_calls(func):
154      """Wrap a function to detect and flag when it gets called. 
155   
156      This is a decorator which takes a function and wraps it in a function with 
157      a 'called' attribute. wrapper.called is initialized to False. 
158   
159      The wrapper.called attribute is set to False right before each call to the 
160      wrapped function, so if the call fails it remains False.  After the call 
161      completes, wrapper.called is set to True and the output is returned. 
162   
163      Testing for truth in wrapper.called allows you to determine if a call to 
164      func() was attempted and succeeded."""
165  
166      # don't wrap twice
167      if hasattr(func, "called"):
168          return func
169  
170      def wrapper(*args, **kw):
171          wrapper.called = False
172          out = func(*args, **kw)
173          wrapper.called = True
174          return out
175  
176      wrapper.called = False
177      wrapper.__doc__ = func.__doc__
178      return wrapper

182      pylab = sys.modules["pylab"]
183      pylab.show._needmain = False
184      # We need to detect at runtime whether show() is called by the user.
185      # For this, we wrap it into a decorator which adds a 'called' flag.
186      pylab.draw_if_interactive = flag_calls(pylab.draw_if_interactive)
