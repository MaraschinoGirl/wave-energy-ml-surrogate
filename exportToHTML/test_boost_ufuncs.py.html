32   @pytest.mark.parametrize('func, args, expected', test_data)
33   def test_stats_boost_ufunc(func, args, expected):
34       type_sigs = func.types
35       type_chars = [sig.split('->')[-1] for sig in type_sigs]
36       for type_char in type_chars:
37           typ, rtol = type_char_to_type_tol[type_char]
38           args = [typ(arg) for arg in args]
39           # Harmless overflow warnings are a "feature" of some wrappers on some
40           # platforms. This test is about dtype and accuracy, so let's avoid false
41           # test failures cause by these warnings. See gh-17432.
42           with np.errstate(over='ignore'):
43               value = func(*args)
44           assert isinstance(value, typ)
45           assert_allclose(value, expected, rtol=rtol)

48   def test_landau():
49       # Test that Landau distribution ufuncs are wrapped as expected;
50       # accuracy is tested by Boost.
51       x = np.linspace(-3, 10, 10)
52       args = (0, 1)
53       res = tanhsinh(lambda x: scu._landau_pdf(x, *args), -np.inf, x)
54       cdf = scu._landau_cdf(x, *args)
55       assert_allclose(res.integral, cdf)
56       sf = scu._landau_sf(x, *args)
57       assert_allclose(sf, 1-cdf)
58       ppf = scu._landau_ppf(cdf, *args)
59       assert_allclose(ppf, x)
60       isf = scu._landau_isf(sf, *args)
61       assert_allclose(isf, x, rtol=1e-6)
