19   def _unique_multiclass(y, xp=None):
20       xp, is_array_api_compliant = get_namespace(y, xp=xp)
21       if hasattr(y, "__array__") or is_array_api_compliant:
22           return cached_unique(xp.asarray(y), xp=xp)
23       else:
24           return set(y)

27   def _unique_indicator(y, xp=None):
28       xp, _ = get_namespace(y, xp=xp)
29       return xp.arange(
30           check_array(y, input_name="y", accept_sparse=["csr", "csc", "coo"]).shape[1]
31       )

36       "multiclass": _unique_multiclass,
37       "multilabel-indicator": _unique_indicator,

41   def unique_labels(*ys):
42       """Extract an ordered array of unique labels. 
43    
44       We don't allow: 
45           - mix of multilabel and multiclass (single label) targets 
46           - mix of label indicator matrix and anything else, 
47             because there are no explicit labels) 
48           - mix of label indicator matrices of different sizes 
49           - mix of string and integer labels 
50    
51       At the moment, we also don't allow "multiclass-multioutput" input type. 
52    
53       Parameters 
54       ---------- 
55       *ys : array-likes 
56           Label values. 
57    
58       Returns 
59       ------- 
60       out : ndarray of shape (n_unique_labels,) 
61           An ordered array of unique labels. 
62    
63       Examples 
64       -------- 
65       >>> from sklearn.utils.multiclass import unique_labels 
66       >>> unique_labels([3, 5, 5, 5, 7, 7]) 
67       array([3, 5, 7]) 
68       >>> unique_labels([1, 2, 3, 4], [2, 2, 3, 4]) 
69       array([1, 2, 3, 4]) 
70       >>> unique_labels([1, 2, 10], [5, 11]) 
71       array([ 1,  2,  5, 10, 11]) 
72       """
73       ys = attach_unique(*ys, return_tuple=True)
74       xp, is_array_api_compliant = get_namespace(*ys)
75       if len(ys) == 0:
76           raise ValueError("No argument has been passed.")
77       # Check that we don't mix label format
78   
79       ys_types = set(type_of_target(x) for x in ys)
80       if ys_types == {"binary", "multiclass"}:
81           ys_types = {"multiclass"}
82   
83       if len(ys_types) > 1:
84           raise ValueError("Mix type of y not allowed, got types %s" % ys_types)
85   
86       label_type = ys_types.pop()
87   
88       # Check consistency for the indicator format
89       if (
90           label_type == "multilabel-indicator"
91           and len(
92               set(
93                   check_array(y, accept_sparse=["csr", "csc", "coo"]).shape[1] for y in ys
94               )
95           )
96           > 1
97       ):
98           raise ValueError(
99               "Multi-label binary indicator input with different numbers of labels"
100          )
101  
102      # Get the unique set of labels
103      _unique_labels = _FN_UNIQUE_LABELS.get(label_type, None)
104      if not _unique_labels:
105          raise ValueError("Unknown label type: %s" % repr(ys))
106  
107      if is_array_api_compliant:
108          # array_api does not allow for mixed dtypes
109          unique_ys = xp.concat([_unique_labels(y, xp=xp) for y in ys])
110          return xp.unique_values(unique_ys)
111  
112      ys_labels = set(
113          chain.from_iterable((i for i in _unique_labels(y, xp=xp)) for y in ys)
114      )
115      # Check that we don't mix string type with number type
116      if len(set(isinstance(label, str) for label in ys_labels)) > 1:
117          raise ValueError("Mix of label input types (string and number)")
118  
119      return xp.asarray(sorted(ys_labels))

122  def _is_integral_float(y):
123      xp, is_array_api_compliant = get_namespace(y)
124      return xp.isdtype(y.dtype, "real floating") and bool(
125          xp.all(xp.astype((xp.astype(y, xp.int64)), y.dtype) == y)
126      )

129  def is_multilabel(y):
130      """Check if ``y`` is in a multilabel format. 
131   
132      Parameters 
133      ---------- 
134      y : ndarray of shape (n_samples,) 
135          Target values. 
136   
137      Returns 
138      ------- 
139      out : bool 
140          Return ``True``, if ``y`` is in a multilabel format, else ```False``. 
141   
142      Examples 
143      -------- 
144      >>> import numpy as np 
145      >>> from sklearn.utils.multiclass import is_multilabel 
146      >>> is_multilabel([0, 1, 0, 1]) 
147      False 
148      >>> is_multilabel([[1], [0, 2], []]) 
149      False 
150      >>> is_multilabel(np.array([[1, 0], [0, 0]])) 
151      True 
152      >>> is_multilabel(np.array([[1], [0], [0]])) 
153      False 
154      >>> is_multilabel(np.array([[1, 0, 0]])) 
155      True 
156      """
157      xp, is_array_api_compliant = get_namespace(y)
158      if hasattr(y, "__array__") or isinstance(y, Sequence) or is_array_api_compliant:
159          # DeprecationWarning will be replaced by ValueError, see NEP 34
160          # https://numpy.org/neps/nep-0034-infer-dtype-is-object.html
161          check_y_kwargs = dict(
162              accept_sparse=True,
163              allow_nd=True,
164              ensure_all_finite=False,
165              ensure_2d=False,
166              ensure_min_samples=0,
167              ensure_min_features=0,
168          )
169          with warnings.catch_warnings():
170              warnings.simplefilter("error", VisibleDeprecationWarning)
171              try:
172                  y = check_array(y, dtype=None, **check_y_kwargs)
173              except (VisibleDeprecationWarning, ValueError) as e:
174                  if str(e).startswith("Complex data not supported"):
175                      raise
176  
177                  # dtype=object should be provided explicitly for ragged arrays,
178                  # see NEP 34
179                  y = check_array(y, dtype=object, **check_y_kwargs)
180  
181      if not (hasattr(y, "shape") and y.ndim == 2 and y.shape[1] > 1):
182          return False
183  
184      if issparse(y):
185          if y.format in ("dok", "lil"):
186              y = y.tocsr()
187          labels = xp.unique_values(y.data)
188          return (
189              len(y.data) == 0
190              or (labels.size == 1 or (labels.size == 2) and (0 in labels))
191              and (y.dtype.kind in "biu" or _is_integral_float(labels))  # bool, int, uint
192          )
193      else:
194          labels = cached_unique(y, xp=xp)
195  
196          return labels.shape[0] < 3 and (
197              xp.isdtype(y.dtype, ("bool", "signed integer", "unsigned integer"))
198              or _is_integral_float(labels)
199          )

202  def check_classification_targets(y):
203      """Ensure that target y is of a non-regression type. 
204   
205      Only the following target types (as defined in type_of_target) are allowed: 
206          'binary', 'multiclass', 'multiclass-multioutput', 
207          'multilabel-indicator', 'multilabel-sequences' 
208   
209      Parameters 
210      ---------- 
211      y : array-like 
212          Target values. 
213      """
214      y_type = type_of_target(y, input_name="y")
215      if y_type not in [
216          "binary",
217          "multiclass",
218          "multiclass-multioutput",
219          "multilabel-indicator",
220          "multilabel-sequences",
221      ]:
222          raise ValueError(
223              f"Unknown label type: {y_type}. Maybe you are trying to fit a "
224              "classifier, which expects discrete classes on a "
225              "regression target with continuous values."
226          )

229  def type_of_target(y, input_name="", raise_unknown=False):
230      """Determine the type of data indicated by the target. 
231   
232      Note that this type is the most specific type that can be inferred. 
233      For example: 
234   
235      * ``binary`` is more specific but compatible with ``multiclass``. 
236      * ``multiclass`` of integers is more specific but compatible with ``continuous``. 
237      * ``multilabel-indicator`` is more specific but compatible with 
238        ``multiclass-multioutput``. 
239   
240      Parameters 
241      ---------- 
242      y : {array-like, sparse matrix} 
243          Target values. If a sparse matrix, `y` is expected to be a 
244          CSR/CSC matrix. 
245   
246      input_name : str, default="" 
247          The data name used to construct the error message. 
248   
249          .. versionadded:: 1.1.0 
250   
251      raise_unknown : bool, default=False 
252          If `True`, raise an error when the type of target returned by 
253          :func:`~sklearn.utils.multiclass.type_of_target` is `"unknown"`. 
254   
255          .. versionadded:: 1.6 
256   
257      Returns 
258      ------- 
259      target_type : str 
260          One of: 
261   
262          * 'continuous': `y` is an array-like of floats that are not all 
263            integers, and is 1d or a column vector. 
264          * 'continuous-multioutput': `y` is a 2d array of floats that are 
265            not all integers, and both dimensions are of size > 1. 
266          * 'binary': `y` contains <= 2 discrete values and is 1d or a column 
267            vector. 
268          * 'multiclass': `y` contains more than two discrete values, is not a 
269            sequence of sequences, and is 1d or a column vector. 
270          * 'multiclass-multioutput': `y` is a 2d array that contains more 
271            than two discrete values, is not a sequence of sequences, and both 
272            dimensions are of size > 1. 
273          * 'multilabel-indicator': `y` is a label indicator matrix, an array 
274            of two dimensions with at least two columns, and at most 2 unique 
275            values. 
276          * 'unknown': `y` is array-like but none of the above, such as a 3d 
277            array, sequence of sequences, or an array of non-sequence objects. 
278   
279      Examples 
280      -------- 
281      >>> from sklearn.utils.multiclass import type_of_target 
282      >>> import numpy as np 
283      >>> type_of_target([0.1, 0.6]) 
284      'continuous' 
285      >>> type_of_target([1, -1, -1, 1]) 
286      'binary' 
287      >>> type_of_target(['a', 'b', 'a']) 
288      'binary' 
289      >>> type_of_target([1.0, 2.0]) 
290      'binary' 
291      >>> type_of_target([1, 0, 2]) 
292      'multiclass' 
293      >>> type_of_target([1.0, 0.0, 3.0]) 
294      'multiclass' 
295      >>> type_of_target(['a', 'b', 'c']) 
296      'multiclass' 
297      >>> type_of_target(np.array([[1, 2], [3, 1]])) 
298      'multiclass-multioutput' 
299      >>> type_of_target([[1, 2]]) 
300      'multilabel-indicator' 
301      >>> type_of_target(np.array([[1.5, 2.0], [3.0, 1.6]])) 
302      'continuous-multioutput' 
303      >>> type_of_target(np.array([[0, 1], [1, 1]])) 
304      'multilabel-indicator' 
305      """
306      xp, is_array_api_compliant = get_namespace(y)
307  
308      def _raise_or_return():
309          """Depending on the value of raise_unknown, either raise an error or return 
310          'unknown'. 
311          """
312          if raise_unknown:
313              input = input_name if input_name else "data"
314              raise ValueError(f"Unknown label type for {input}: {y!r}")
315          else:
316              return "unknown"
317  
318      valid = (
319          (isinstance(y, Sequence) or issparse(y) or hasattr(y, "__array__"))
320          and not isinstance(y, str)
321          or is_array_api_compliant
322      )
323  
324      if not valid:
325          raise ValueError(
326              "Expected array-like (array or non-string sequence), got %r" % y
327          )
328  
329      sparse_pandas = y.__class__.__name__ in ["SparseSeries", "SparseArray"]
330      if sparse_pandas:
331          raise ValueError("y cannot be class 'SparseSeries' or 'SparseArray'")
332  
333      if is_multilabel(y):
334          return "multilabel-indicator"
335  
336      # DeprecationWarning will be replaced by ValueError, see NEP 34
337      # https://numpy.org/neps/nep-0034-infer-dtype-is-object.html
338      # We therefore catch both deprecation (NumPy < 1.24) warning and
339      # value error (NumPy >= 1.24).
340      check_y_kwargs = dict(
341          accept_sparse=True,
342          allow_nd=True,
343          ensure_all_finite=False,
344          ensure_2d=False,
345          ensure_min_samples=0,
346          ensure_min_features=0,
347      )
348  
349      with warnings.catch_warnings():
350          warnings.simplefilter("error", VisibleDeprecationWarning)
351          if not issparse(y):
352              try:
353                  y = check_array(y, dtype=None, **check_y_kwargs)
354              except (VisibleDeprecationWarning, ValueError) as e:
355                  if str(e).startswith("Complex data not supported"):
356                      raise
357  
358                  # dtype=object should be provided explicitly for ragged arrays,
359                  # see NEP 34
360                  y = check_array(y, dtype=object, **check_y_kwargs)
361  
362      try:
363          # TODO(1.7): Change to ValueError when byte labels is deprecated.
364          # labels in bytes format
365          first_row_or_val = y[[0], :] if issparse(y) else y[0]
366          if isinstance(first_row_or_val, bytes):
367              warnings.warn(
368                  (
369                      "Support for labels represented as bytes is deprecated in v1.5 and"
370                      " will error in v1.7. Convert the labels to a string or integer"
371                      " format."
372                  ),
373                  FutureWarning,
374              )
375          # The old sequence of sequences format
376          if (
377              not hasattr(first_row_or_val, "__array__")
378              and isinstance(first_row_or_val, Sequence)
379              and not isinstance(first_row_or_val, str)
380          ):
381              raise ValueError(
382                  "You appear to be using a legacy multi-label data"
383                  " representation. Sequence of sequences are no"
384                  " longer supported; use a binary array or sparse"
385                  " matrix instead - the MultiLabelBinarizer"
386                  " transformer can convert to this format."
387              )
388      except IndexError:
389          pass
390  
391      # Invalid inputs
392      if y.ndim not in (1, 2):
393          # Number of dimension greater than 2: [[[1, 2]]]
394          return _raise_or_return()
395      if not min(y.shape):
396          # Empty ndarray: []/[[]]
397          if y.ndim == 1:
398              # 1-D empty array: []
399              return "binary"  # []
400          # 2-D empty array: [[]]
401          return _raise_or_return()
402      if not issparse(y) and y.dtype == object and not isinstance(y.flat[0], str):
403          # [obj_1] and not ["label_1"]
404          return _raise_or_return()
405  
406      # Check if multioutput
407      if y.ndim == 2 and y.shape[1] > 1:
408          suffix = "-multioutput"  # [[1, 2], [1, 2]]
409      else:
410          suffix = ""  # [1, 2, 3] or [[1], [2], [3]]
411  
412      # Check float and contains non-integer float values
413      if xp.isdtype(y.dtype, "real floating"):
414          # [.1, .2, 3] or [[.1, .2, 3]] or [[1., .2]] and not [1., 2., 3.]
415          data = y.data if issparse(y) else y
416          if xp.any(data != xp.astype(data, int)):
417              _assert_all_finite(data, input_name=input_name)
418              return "continuous" + suffix
419  
420      # Check multiclass
421      if issparse(first_row_or_val):
422          first_row_or_val = first_row_or_val.data
423      if cached_unique(y).shape[0] > 2 or (y.ndim == 2 and len(first_row_or_val) > 1):
424          # [1, 2, 3] or [[1., 2., 3]] or [[1, 2]]
425          return "multiclass" + suffix
426      else:
427          return "binary"  # [1, 2] or [["a"], ["b"]]

430  def _check_partial_fit_first_call(clf, classes=None):
431      """Private helper function for factorizing common classes param logic. 
432   
433      Estimators that implement the ``partial_fit`` API need to be provided with 
434      the list of possible classes at the first call to partial_fit. 
435   
436      Subsequent calls to partial_fit should check that ``classes`` is still 
437      consistent with a previous value of ``clf.classes_`` when provided. 
438   
439      This function returns True if it detects that this was the first call to 
440      ``partial_fit`` on ``clf``. In that case the ``classes_`` attribute is also 
441      set on ``clf``. 
442   
443      """
444      if getattr(clf, "classes_", None) is None and classes is None:
445          raise ValueError("classes must be passed on the first call to partial_fit.")
446  
447      elif classes is not None:
448          if getattr(clf, "classes_", None) is not None:
449              if not np.array_equal(clf.classes_, unique_labels(classes)):
450                  raise ValueError(
451                      "`classes=%r` is not the same as on last call "
452                      "to partial_fit, was: %r" % (classes, clf.classes_)
453                  )
454  
455          else:
456              # This is the first call to partial_fit
457              clf.classes_ = unique_labels(classes)
458              return True
459  
460      # classes is None and clf.classes_ has already previously been set:
461      # nothing to do
462      return False

465  def class_distribution(y, sample_weight=None):
466      """Compute class priors from multioutput-multiclass target data. 
467   
468      Parameters 
469      ---------- 
470      y : {array-like, sparse matrix} of size (n_samples, n_outputs) 
471          The labels for each example. 
472   
473      sample_weight : array-like of shape (n_samples,), default=None 
474          Sample weights. 
475   
476      Returns 
477      ------- 
478      classes : list of size n_outputs of ndarray of size (n_classes,) 
479          List of classes for each column. 
480   
481      n_classes : list of int of size n_outputs 
482          Number of classes in each column. 
483   
484      class_prior : list of size n_outputs of ndarray of size (n_classes,) 
485          Class distribution of each column. 
486      """
487      classes = []
488      n_classes = []
489      class_prior = []
490  
491      n_samples, n_outputs = y.shape
492      if sample_weight is not None:
493          sample_weight = np.asarray(sample_weight)
494  
495      if issparse(y):
496          y = y.tocsc()
497          y_nnz = np.diff(y.indptr)
498  
499          for k in range(n_outputs):
500              col_nonzero = y.indices[y.indptr[k] : y.indptr[k + 1]]
501              # separate sample weights for zero and non-zero elements
502              if sample_weight is not None:
503                  nz_samp_weight = sample_weight[col_nonzero]
504                  zeros_samp_weight_sum = np.sum(sample_weight) - np.sum(nz_samp_weight)
505              else:
506                  nz_samp_weight = None
507                  zeros_samp_weight_sum = y.shape[0] - y_nnz[k]
508  
509              classes_k, y_k = np.unique(
510                  y.data[y.indptr[k] : y.indptr[k + 1]], return_inverse=True
511              )
512              class_prior_k = np.bincount(y_k, weights=nz_samp_weight)
513  
514              # An explicit zero was found, combine its weight with the weight
515              # of the implicit zeros
516              if 0 in classes_k:
517                  class_prior_k[classes_k == 0] += zeros_samp_weight_sum
518  
519              # If an there is an implicit zero and it is not in classes and
520              # class_prior, make an entry for it
521              if 0 not in classes_k and y_nnz[k] < y.shape[0]:
522                  classes_k = np.insert(classes_k, 0, 0)
523                  class_prior_k = np.insert(class_prior_k, 0, zeros_samp_weight_sum)
524  
525              classes.append(classes_k)
526              n_classes.append(classes_k.shape[0])
527              class_prior.append(class_prior_k / class_prior_k.sum())
528      else:
529          for k in range(n_outputs):
530              classes_k, y_k = np.unique(y[:, k], return_inverse=True)
531              classes.append(classes_k)
532              n_classes.append(classes_k.shape[0])
533              class_prior_k = np.bincount(y_k, weights=sample_weight)
534              class_prior.append(class_prior_k / class_prior_k.sum())
535  
536      return (classes, n_classes, class_prior)

539  def _ovr_decision_function(predictions, confidences, n_classes):
540      """Compute a continuous, tie-breaking OvR decision function from OvO. 
541   
542      It is important to include a continuous value, not only votes, 
543      to make computing AUC or calibration meaningful. 
544   
545      Parameters 
546      ---------- 
547      predictions : array-like of shape (n_samples, n_classifiers) 
548          Predicted classes for each binary classifier. 
549   
550      confidences : array-like of shape (n_samples, n_classifiers) 
551          Decision functions or predicted probabilities for positive class 
552          for each binary classifier. 
553   
554      n_classes : int 
555          Number of classes. n_classifiers must be 
556          ``n_classes * (n_classes - 1 ) / 2``. 
557      """
558      n_samples = predictions.shape[0]
559      votes = np.zeros((n_samples, n_classes))
560      sum_of_confidences = np.zeros((n_samples, n_classes))
561  
562      k = 0
563      for i in range(n_classes):
564          for j in range(i + 1, n_classes):
565              sum_of_confidences[:, i] -= confidences[:, k]
566              sum_of_confidences[:, j] += confidences[:, k]
567              votes[predictions[:, k] == 0, i] += 1
568              votes[predictions[:, k] == 1, j] += 1
569              k += 1
570  
571      # Monotonically transform the sum_of_confidences to (-1/3, 1/3)
572      # and add it with votes. The monotonic transformation  is
573      # f: x -> x / (3 * (|x| + 1)), it uses 1/3 instead of 1/2
574      # to ensure that we won't reach the limits and change vote order.
575      # The motivation is to use confidence levels as a way to break ties in
576      # the votes without switching any decision made based on a difference
577      # of 1 vote.
578      transformed_confidences = sum_of_confidences / (
579          3 * (np.abs(sum_of_confidences) + 1)
580      )
581      return votes + transformed_confidences
