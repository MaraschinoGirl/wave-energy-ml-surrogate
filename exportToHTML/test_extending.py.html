58   def test_cython(tmp_path):
59       import glob
60       # build the examples in a temporary directory
61       srcdir = os.path.join(os.path.dirname(__file__), '..')
62       shutil.copytree(srcdir, tmp_path / 'random')
63       build_dir = tmp_path / 'random' / '_examples' / 'cython'
64       target_dir = build_dir / "build"
65       os.makedirs(target_dir, exist_ok=True)
66       # Ensure we use the correct Python interpreter even when `meson` is
67       # installed in a different Python environment (see gh-24956)
68       native_file = str(build_dir / 'interpreter-native-file.ini')
69       with open(native_file, 'w') as f:
70           f.write("[binaries]\n")
71           f.write(f"python = '{sys.executable}'\n")
72           f.write(f"python3 = '{sys.executable}'")
73       if sys.platform == "win32":
74           subprocess.check_call(["meson", "setup",
75                                  "--buildtype=release",
76                                  "--vsenv", "--native-file", native_file,
77                                  str(build_dir)],
78                                 cwd=target_dir,
79                                 )
80       else:
81           subprocess.check_call(["meson", "setup",
82                                  "--native-file", native_file, str(build_dir)],
83                                 cwd=target_dir
84                                 )
85       subprocess.check_call(["meson", "compile", "-vv"], cwd=target_dir)
86   
87       # gh-16162: make sure numpy's __init__.pxd was used for cython
88       # not really part of this test, but it is a convenient place to check
89   
90       g = glob.glob(str(target_dir / "*" / "extending.pyx.c"))
91       with open(g[0]) as fid:
92           txt_to_find = 'NumPy API declarations from "numpy/__init__'
93           for line in fid:
94               if txt_to_find in line:
95                   break
96           else:
97               assert False, ("Could not find '{}' in C file, "
98                              "wrong pxd used".format(txt_to_find))
99       # import without adding the directory to sys.path
100      suffix = sysconfig.get_config_var('EXT_SUFFIX')
101  
102      def load(modname):
103          so = (target_dir / modname).with_suffix(suffix)
104          spec = spec_from_file_location(modname, so)
105          mod = module_from_spec(spec)
106          spec.loader.exec_module(mod)
107          return mod
108  
109      # test that the module can be imported
110      load("extending")
111      load("extending_cpp")
112      # actually test the cython c-extension
113      extending_distributions = load("extending_distributions")
114      from numpy.random import PCG64
115      values = extending_distributions.uniforms_ex(PCG64(0), 10, 'd')
116      assert values.shape == (10,)
117      assert values.dtype == np.float64
