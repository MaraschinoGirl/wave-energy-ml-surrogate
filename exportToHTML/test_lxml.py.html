65       def test_tracking_line_numbers(self):
66           # The lxml TreeBuilder cannot keep track of line numbers from
67           # the original markup. Even if you ask for line numbers, we
68           # don't have 'em.
69           #
70           # However, for consistency with other parsers, Tag.sourceline
71           # and Tag.sourcepos are always set to None, rather than being
72           # available as an alias for find().
73           soup = self.soup(
74               "\n   <p>\n\n<sourceline>\n<b>text</b></sourceline><sourcepos></p>",
75               store_line_numbers=True,
76           )
77           assert None is soup.p.sourceline
78           assert None is soup.p.sourcepos

92       def test_namespace_indexing(self):
93           soup = self.soup(
94               '<?xml version="1.1"?>\n'
95               "<root>"
96               '<tag xmlns="http://unprefixed-namespace.com">content</tag>'
97               '<prefix:tag2 xmlns:prefix="http://prefixed-namespace.com">content</prefix:tag2>'
98               '<prefix2:tag3 xmlns:prefix2="http://another-namespace.com">'
99               '<subtag xmlns="http://another-unprefixed-namespace.com">'
100              '<subsubtag xmlns="http://yet-another-unprefixed-namespace.com">'
101              "</prefix2:tag3>"
102              "</root>"
103          )
104  
105          # The BeautifulSoup object includes every namespace prefix
106          # defined in the entire document. This is the default set of
107          # namespaces used by soupsieve.
108          #
109          # Un-prefixed namespaces are not included, and if a given
110          # prefix is defined twice, only the first prefix encountered
111          # in the document shows up here.
112          assert soup._namespaces == {
113              "xml": "http://www.w3.org/XML/1998/namespace",
114              "prefix": "http://prefixed-namespace.com",
115              "prefix2": "http://another-namespace.com",
116          }
117  
118          # A Tag object includes only the namespace prefixes
119          # that were in scope when it was parsed.
120  
121          # We do not track un-prefixed namespaces as we can only hold
122          # one (the first one), and it will be recognized as the
123          # default namespace by soupsieve, even when operating from a
124          # tag with a different un-prefixed namespace.
125          assert soup.tag._namespaces == {
126              "xml": "http://www.w3.org/XML/1998/namespace",
127          }
128  
129          assert soup.tag2._namespaces == {
130              "prefix": "http://prefixed-namespace.com",
131              "xml": "http://www.w3.org/XML/1998/namespace",
132          }
133  
134          assert soup.subtag._namespaces == {
135              "prefix2": "http://another-namespace.com",
136              "xml": "http://www.w3.org/XML/1998/namespace",
137          }
138  
139          assert soup.subsubtag._namespaces == {
140              "prefix2": "http://another-namespace.com",
141              "xml": "http://www.w3.org/XML/1998/namespace",
142          }

144      @pytest.mark.skipif(not SOUP_SIEVE_PRESENT, reason="Soup Sieve not installed")
145      def test_namespace_interaction_with_select_and_find(self):
146          # Demonstrate how namespaces interact with select* and
147          # find* methods.
148  
149          soup = self.soup(
150              '<?xml version="1.1"?>\n'
151              "<root>"
152              '<tag xmlns="http://unprefixed-namespace.com">content</tag>'
153              '<prefix:tag2 xmlns:prefix="http://prefixed-namespace.com">content</tag>'
154              '<subtag xmlns:prefix="http://another-namespace-same-prefix.com">'
155              "<prefix:tag3>"
156              "</subtag>"
157              "</root>"
158          )
159  
160          # soupselect uses namespace URIs.
161          assert soup.select_one("tag").name == "tag"
162          assert soup.select_one("prefix|tag2").name == "tag2"
163  
164          # If a prefix is declared more than once, only the first usage
165          # is registered with the BeautifulSoup object.
166          assert soup.select_one("prefix|tag3") is None
167  
168          # But you can always explicitly specify a namespace dictionary.
169          assert (
170              soup.select_one("prefix|tag3", namespaces=soup.subtag._namespaces).name
171              == "tag3"
172          )
173  
174          # And a Tag (as opposed to the BeautifulSoup object) will
175          # have a set of default namespaces scoped to that Tag.
176          assert soup.subtag.select_one("prefix|tag3").name == "tag3"
177  
178          # the find() methods aren't fully namespace-aware; they just
179          # look at prefixes.
180          assert soup.find("tag").name == "tag"
181          assert soup.find("prefix:tag2").name == "tag2"
182          assert soup.find("prefix:tag3").name == "tag3"
183          assert soup.subtag.find("prefix:tag3").name == "tag3"

185      def test_pickle_restores_builder(self):
186          # The lxml TreeBuilder is not picklable, so when unpickling
187          # a document created with it, a new TreeBuilder of the
188          # appropriate class is created.
189          soup = self.soup("<a>some markup</a>")
190          assert isinstance(soup.builder, self.default_builder)
191          pickled = pickle.dumps(soup)
192          unpickled = pickle.loads(pickled)
193  
194          assert "some markup" == unpickled.a.string
195          assert unpickled.builder != soup.builder
196          assert isinstance(unpickled.builder, self.default_builder)
