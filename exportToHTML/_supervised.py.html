25   def check_clusterings(labels_true, labels_pred):
26       """Check that the labels arrays are 1D and of same dimension. 
27    
28       Parameters 
29       ---------- 
30       labels_true : array-like of shape (n_samples,) 
31           The true labels. 
32    
33       labels_pred : array-like of shape (n_samples,) 
34           The predicted labels. 
35       """
36       labels_true = check_array(
37           labels_true,
38           ensure_2d=False,
39           ensure_min_samples=0,
40           dtype=None,
41       )
42   
43       labels_pred = check_array(
44           labels_pred,
45           ensure_2d=False,
46           ensure_min_samples=0,
47           dtype=None,
48       )
49   
50       type_label = type_of_target(labels_true)
51       type_pred = type_of_target(labels_pred)
52   
53       if "continuous" in (type_pred, type_label):
54           msg = (
55               "Clustering metrics expects discrete values but received"
56               f" {type_label} values for label, and {type_pred} values "
57               "for target"
58           )
59           warnings.warn(msg, UserWarning)
60   
61       # input checks
62       if labels_true.ndim != 1:
63           raise ValueError("labels_true must be 1D: shape is %r" % (labels_true.shape,))
64       if labels_pred.ndim != 1:
65           raise ValueError("labels_pred must be 1D: shape is %r" % (labels_pred.shape,))
66       check_consistent_length(labels_true, labels_pred)
67   
68       return labels_true, labels_pred

71   def _generalized_average(U, V, average_method):
72       """Return a particular mean of two numbers."""
73       if average_method == "min":
74           return min(U, V)
75       elif average_method == "geometric":
76           return np.sqrt(U * V)
77       elif average_method == "arithmetic":
78           return np.mean([U, V])
79       elif average_method == "max":
80           return max(U, V)
81       else:
82           raise ValueError(
83               "'average_method' must be 'min', 'geometric', 'arithmetic', or 'max'"
84           )

97   def contingency_matrix(
98       labels_true, labels_pred, *, eps=None, sparse=False, dtype=np.int64
99   ):
100      """Build a contingency matrix describing the relationship between labels. 
101   
102      Parameters 
103      ---------- 
104      labels_true : array-like of shape (n_samples,) 
105          Ground truth class labels to be used as a reference. 
106   
107      labels_pred : array-like of shape (n_samples,) 
108          Cluster labels to evaluate. 
109   
110      eps : float, default=None 
111          If a float, that value is added to all values in the contingency 
112          matrix. This helps to stop NaN propagation. 
113          If ``None``, nothing is adjusted. 
114   
115      sparse : bool, default=False 
116          If `True`, return a sparse CSR continency matrix. If `eps` is not 
117          `None` and `sparse` is `True` will raise ValueError. 
118   
119          .. versionadded:: 0.18 
120   
121      dtype : numeric type, default=np.int64 
122          Output dtype. Ignored if `eps` is not `None`. 
123   
124          .. versionadded:: 0.24 
125   
126      Returns 
127      ------- 
128      contingency : {array-like, sparse}, shape=[n_classes_true, n_classes_pred] 
129          Matrix :math:`C` such that :math:`C_{i, j}` is the number of samples in 
130          true class :math:`i` and in predicted class :math:`j`. If 
131          ``eps is None``, the dtype of this array will be integer unless set 
132          otherwise with the ``dtype`` argument. If ``eps`` is given, the dtype 
133          will be float. 
134          Will be a ``sklearn.sparse.csr_matrix`` if ``sparse=True``. 
135   
136      Examples 
137      -------- 
138      >>> from sklearn.metrics.cluster import contingency_matrix 
139      >>> labels_true = [0, 0, 1, 1, 2, 2] 
140      >>> labels_pred = [1, 0, 2, 1, 0, 2] 
141      >>> contingency_matrix(labels_true, labels_pred) 
142      array([[1, 1, 0], 
143             [0, 1, 1], 
144             [1, 0, 1]]) 
145      """
146  
147      if eps is not None and sparse:
148          raise ValueError("Cannot set 'eps' when sparse=True")
149  
150      classes, class_idx = np.unique(labels_true, return_inverse=True)
151      clusters, cluster_idx = np.unique(labels_pred, return_inverse=True)
152      n_classes = classes.shape[0]
153      n_clusters = clusters.shape[0]
154      # Using coo_matrix to accelerate simple histogram calculation,
155      # i.e. bins are consecutive integers
156      # Currently, coo_matrix is faster than histogram2d for simple cases
157      contingency = sp.coo_matrix(
158          (np.ones(class_idx.shape[0]), (class_idx, cluster_idx)),
159          shape=(n_classes, n_clusters),
160          dtype=dtype,
161      )
162      if sparse:
163          contingency = contingency.tocsr()
164          contingency.sum_duplicates()
165      else:
166          contingency = contingency.toarray()
167          if eps is not None:
168              # don't use += as contingency is integer
169              contingency = contingency + eps
170      return contingency

183  def pair_confusion_matrix(labels_true, labels_pred):
184      """Pair confusion matrix arising from two clusterings. 
185   
186      The pair confusion matrix :math:`C` computes a 2 by 2 similarity matrix 
187      between two clusterings by considering all pairs of samples and counting 
188      pairs that are assigned into the same or into different clusters under 
189      the true and predicted clusterings [1]_. 
190   
191      Considering a pair of samples that is clustered together a positive pair, 
192      then as in binary classification the count of true negatives is 
193      :math:`C_{00}`, false negatives is :math:`C_{10}`, true positives is 
194      :math:`C_{11}` and false positives is :math:`C_{01}`. 
195   
196      Read more in the :ref:`User Guide <pair_confusion_matrix>`. 
197   
198      Parameters 
199      ---------- 
200      labels_true : array-like of shape (n_samples,), dtype=integral 
201          Ground truth class labels to be used as a reference. 
202   
203      labels_pred : array-like of shape (n_samples,), dtype=integral 
204          Cluster labels to evaluate. 
205   
206      Returns 
207      ------- 
208      C : ndarray of shape (2, 2), dtype=np.int64 
209          The contingency matrix. 
210   
211      See Also 
212      -------- 
213      sklearn.metrics.rand_score : Rand Score. 
214      sklearn.metrics.adjusted_rand_score : Adjusted Rand Score. 
215      sklearn.metrics.adjusted_mutual_info_score : Adjusted Mutual Information. 
216   
217      References 
218      ---------- 
219      .. [1] :doi:`Hubert, L., Arabie, P. "Comparing partitions." 
220             Journal of Classification 2, 193–218 (1985). 
221             <10.1007/BF01908075>` 
222   
223      Examples 
224      -------- 
225      Perfectly matching labelings have all non-zero entries on the 
226      diagonal regardless of actual label values: 
227   
228        >>> from sklearn.metrics.cluster import pair_confusion_matrix 
229        >>> pair_confusion_matrix([0, 0, 1, 1], [1, 1, 0, 0]) 
230        array([[8, 0], 
231               [0, 4]]... 
232   
233      Labelings that assign all classes members to the same clusters 
234      are complete but may be not always pure, hence penalized, and 
235      have some off-diagonal non-zero entries: 
236   
237        >>> pair_confusion_matrix([0, 0, 1, 2], [0, 0, 1, 1]) 
238        array([[8, 2], 
239               [0, 2]]... 
240   
241      Note that the matrix is not symmetric. 
242      """
243      labels_true, labels_pred = check_clusterings(labels_true, labels_pred)
244      n_samples = np.int64(labels_true.shape[0])
245  
246      # Computation using the contingency data
247      contingency = contingency_matrix(
248          labels_true, labels_pred, sparse=True, dtype=np.int64
249      )
250      n_c = np.ravel(contingency.sum(axis=1))
251      n_k = np.ravel(contingency.sum(axis=0))
252      sum_squares = (contingency.data**2).sum()
253      C = np.empty((2, 2), dtype=np.int64)
254      C[1, 1] = sum_squares - n_samples
255      C[0, 1] = contingency.dot(n_k).sum() - sum_squares
256      C[1, 0] = contingency.transpose().dot(n_c).sum() - sum_squares
257      C[0, 0] = n_samples**2 - C[0, 1] - C[1, 0] - sum_squares
258      return C

268  def rand_score(labels_true, labels_pred):
269      """Rand index. 
270   
271      The Rand Index computes a similarity measure between two clusterings 
272      by considering all pairs of samples and counting pairs that are 
273      assigned in the same or different clusters in the predicted and 
274      true clusterings [1]_ [2]_. 
275   
276      The raw RI score [3]_ is: 
277   
278      .. code-block:: text 
279   
280          RI = (number of agreeing pairs) / (number of pairs) 
281   
282      Read more in the :ref:`User Guide <rand_score>`. 
283   
284      Parameters 
285      ---------- 
286      labels_true : array-like of shape (n_samples,), dtype=integral 
287          Ground truth class labels to be used as a reference. 
288   
289      labels_pred : array-like of shape (n_samples,), dtype=integral 
290          Cluster labels to evaluate. 
291   
292      Returns 
293      ------- 
294      RI : float 
295         Similarity score between 0.0 and 1.0, inclusive, 1.0 stands for 
296         perfect match. 
297   
298      See Also 
299      -------- 
300      adjusted_rand_score: Adjusted Rand Score. 
301      adjusted_mutual_info_score: Adjusted Mutual Information. 
302   
303      References 
304      ---------- 
305      .. [1] :doi:`Hubert, L., Arabie, P. "Comparing partitions." 
306         Journal of Classification 2, 193–218 (1985). 
307         <10.1007/BF01908075>`. 
308   
309      .. [2] `Wikipedia: Simple Matching Coefficient 
310          <https://en.wikipedia.org/wiki/Simple_matching_coefficient>`_ 
311   
312      .. [3] `Wikipedia: Rand Index <https://en.wikipedia.org/wiki/Rand_index>`_ 
313   
314      Examples 
315      -------- 
316      Perfectly matching labelings have a score of 1 even 
317   
318        >>> from sklearn.metrics.cluster import rand_score 
319        >>> rand_score([0, 0, 1, 1], [1, 1, 0, 0]) 
320        1.0 
321   
322      Labelings that assign all classes members to the same clusters 
323      are complete but may not always be pure, hence penalized: 
324   
325        >>> rand_score([0, 0, 1, 2], [0, 0, 1, 1]) 
326        np.float64(0.83...) 
327      """
328      contingency = pair_confusion_matrix(labels_true, labels_pred)
329      numerator = contingency.diagonal().sum()
330      denominator = contingency.sum()
331  
332      if numerator == denominator or denominator == 0:
333          # Special limit cases: no clustering since the data is not split;
334          # or trivial clustering where each document is assigned a unique
335          # cluster. These are perfect matches hence return 1.0.
336          return 1.0
337  
338      return numerator / denominator

348  def adjusted_rand_score(labels_true, labels_pred):
349      """Rand index adjusted for chance. 
350   
351      The Rand Index computes a similarity measure between two clusterings 
352      by considering all pairs of samples and counting pairs that are 
353      assigned in the same or different clusters in the predicted and 
354      true clusterings. 
355   
356      The raw RI score is then "adjusted for chance" into the ARI score 
357      using the following scheme:: 
358   
359          ARI = (RI - Expected_RI) / (max(RI) - Expected_RI) 
360   
361      The adjusted Rand index is thus ensured to have a value close to 
362      0.0 for random labeling independently of the number of clusters and 
363      samples and exactly 1.0 when the clusterings are identical (up to 
364      a permutation). The adjusted Rand index is bounded below by -0.5 for 
365      especially discordant clusterings. 
366   
367      ARI is a symmetric measure:: 
368   
369          adjusted_rand_score(a, b) == adjusted_rand_score(b, a) 
370   
371      Read more in the :ref:`User Guide <adjusted_rand_score>`. 
372   
373      Parameters 
374      ---------- 
375      labels_true : array-like of shape (n_samples,), dtype=int 
376          Ground truth class labels to be used as a reference. 
377   
378      labels_pred : array-like of shape (n_samples,), dtype=int 
379          Cluster labels to evaluate. 
380   
381      Returns 
382      ------- 
383      ARI : float 
384         Similarity score between -0.5 and 1.0. Random labelings have an ARI 
385         close to 0.0. 1.0 stands for perfect match. 
386   
387      See Also 
388      -------- 
389      adjusted_mutual_info_score : Adjusted Mutual Information. 
390   
391      References 
392      ---------- 
393      .. [Hubert1985] L. Hubert and P. Arabie, Comparing Partitions, 
394        Journal of Classification 1985 
395        https://link.springer.com/article/10.1007%2FBF01908075 
396   
397      .. [Steinley2004] D. Steinley, Properties of the Hubert-Arabie 
398        adjusted Rand index, Psychological Methods 2004 
399   
400      .. [wk] https://en.wikipedia.org/wiki/Rand_index#Adjusted_Rand_index 
401   
402      .. [Chacon] :doi:`Minimum adjusted Rand index for two clusterings of a given size, 
403        2022, J. E. Chacón and A. I. Rastrojo <10.1007/s11634-022-00491-w>` 
404   
405      Examples 
406      -------- 
407      Perfectly matching labelings have a score of 1 even 
408   
409        >>> from sklearn.metrics.cluster import adjusted_rand_score 
410        >>> adjusted_rand_score([0, 0, 1, 1], [0, 0, 1, 1]) 
411        1.0 
412        >>> adjusted_rand_score([0, 0, 1, 1], [1, 1, 0, 0]) 
413        1.0 
414   
415      Labelings that assign all classes members to the same clusters 
416      are complete but may not always be pure, hence penalized:: 
417   
418        >>> adjusted_rand_score([0, 0, 1, 2], [0, 0, 1, 1]) 
419        0.57... 
420   
421      ARI is symmetric, so labelings that have pure clusters with members 
422      coming from the same classes but unnecessary splits are penalized:: 
423   
424        >>> adjusted_rand_score([0, 0, 1, 1], [0, 0, 1, 2]) 
425        0.57... 
426   
427      If classes members are completely split across different clusters, the 
428      assignment is totally incomplete, hence the ARI is very low:: 
429   
430        >>> adjusted_rand_score([0, 0, 0, 0], [0, 1, 2, 3]) 
431        0.0 
432   
433      ARI may take a negative value for especially discordant labelings that 
434      are a worse choice than the expected value of random labels:: 
435   
436        >>> adjusted_rand_score([0, 0, 1, 1], [0, 1, 0, 1]) 
437        -0.5 
438   
439      See :ref:`sphx_glr_auto_examples_cluster_plot_adjusted_for_chance_measures.py` 
440      for a more detailed example. 
441      """
442      (tn, fp), (fn, tp) = pair_confusion_matrix(labels_true, labels_pred)
443      # convert to Python integer types, to avoid overflow or underflow
444      tn, fp, fn, tp = int(tn), int(fp), int(fn), int(tp)
445  
446      # Special cases: empty data or full agreement
447      if fn == 0 and fp == 0:
448          return 1.0
449  
450      return 2.0 * (tp * tn - fn * fp) / ((tp + fn) * (fn + tn) + (tp + fp) * (fp + tn))

461  def homogeneity_completeness_v_measure(labels_true, labels_pred, *, beta=1.0):
462      """Compute the homogeneity and completeness and V-Measure scores at once. 
463   
464      Those metrics are based on normalized conditional entropy measures of 
465      the clustering labeling to evaluate given the knowledge of a Ground 
466      Truth class labels of the same samples. 
467   
468      A clustering result satisfies homogeneity if all of its clusters 
469      contain only data points which are members of a single class. 
470   
471      A clustering result satisfies completeness if all the data points 
472      that are members of a given class are elements of the same cluster. 
473   
474      Both scores have positive values between 0.0 and 1.0, larger values 
475      being desirable. 
476   
477      Those 3 metrics are independent of the absolute values of the labels: 
478      a permutation of the class or cluster label values won't change the 
479      score values in any way. 
480   
481      V-Measure is furthermore symmetric: swapping ``labels_true`` and 
482      ``label_pred`` will give the same score. This does not hold for 
483      homogeneity and completeness. V-Measure is identical to 
484      :func:`normalized_mutual_info_score` with the arithmetic averaging 
485      method. 
486   
487      Read more in the :ref:`User Guide <homogeneity_completeness>`. 
488   
489      Parameters 
490      ---------- 
491      labels_true : array-like of shape (n_samples,) 
492          Ground truth class labels to be used as a reference. 
493   
494      labels_pred : array-like of shape (n_samples,) 
495          Cluster labels to evaluate. 
496   
497      beta : float, default=1.0 
498          Ratio of weight attributed to ``homogeneity`` vs ``completeness``. 
499          If ``beta`` is greater than 1, ``completeness`` is weighted more 
500          strongly in the calculation. If ``beta`` is less than 1, 
501          ``homogeneity`` is weighted more strongly. 
502   
503      Returns 
504      ------- 
505      homogeneity : float 
506          Score between 0.0 and 1.0. 1.0 stands for perfectly homogeneous labeling. 
507   
508      completeness : float 
509          Score between 0.0 and 1.0. 1.0 stands for perfectly complete labeling. 
510   
511      v_measure : float 
512          Harmonic mean of the first two. 
513   
514      See Also 
515      -------- 
516      homogeneity_score : Homogeneity metric of cluster labeling. 
517      completeness_score : Completeness metric of cluster labeling. 
518      v_measure_score : V-Measure (NMI with arithmetic mean option). 
519   
520      Examples 
521      -------- 
522      >>> from sklearn.metrics import homogeneity_completeness_v_measure 
523      >>> y_true, y_pred = [0, 0, 1, 1, 2, 2], [0, 0, 1, 2, 2, 2] 
524      >>> homogeneity_completeness_v_measure(y_true, y_pred) 
525      (np.float64(0.71...), np.float64(0.77...), np.float64(0.73...)) 
526      """
527      labels_true, labels_pred = check_clusterings(labels_true, labels_pred)
528  
529      if len(labels_true) == 0:
530          return 1.0, 1.0, 1.0
531  
532      entropy_C = entropy(labels_true)
533      entropy_K = entropy(labels_pred)
534  
535      contingency = contingency_matrix(labels_true, labels_pred, sparse=True)
536      MI = mutual_info_score(None, None, contingency=contingency)
537  
538      homogeneity = MI / (entropy_C) if entropy_C else 1.0
539      completeness = MI / (entropy_K) if entropy_K else 1.0
540  
541      if homogeneity + completeness == 0.0:
542          v_measure_score = 0.0
543      else:
544          v_measure_score = (
545              (1 + beta)
546              * homogeneity
547              * completeness
548              / (beta * homogeneity + completeness)
549          )
550  
551      return homogeneity, completeness, v_measure_score

561  def homogeneity_score(labels_true, labels_pred):
562      """Homogeneity metric of a cluster labeling given a ground truth. 
563   
564      A clustering result satisfies homogeneity if all of its clusters 
565      contain only data points which are members of a single class. 
566   
567      This metric is independent of the absolute values of the labels: 
568      a permutation of the class or cluster label values won't change the 
569      score value in any way. 
570   
571      This metric is not symmetric: switching ``label_true`` with ``label_pred`` 
572      will return the :func:`completeness_score` which will be different in 
573      general. 
574   
575      Read more in the :ref:`User Guide <homogeneity_completeness>`. 
576   
577      Parameters 
578      ---------- 
579      labels_true : array-like of shape (n_samples,) 
580          Ground truth class labels to be used as a reference. 
581   
582      labels_pred : array-like of shape (n_samples,) 
583          Cluster labels to evaluate. 
584   
585      Returns 
586      ------- 
587      homogeneity : float 
588         Score between 0.0 and 1.0. 1.0 stands for perfectly homogeneous labeling. 
589   
590      See Also 
591      -------- 
592      completeness_score : Completeness metric of cluster labeling. 
593      v_measure_score : V-Measure (NMI with arithmetic mean option). 
594   
595      References 
596      ---------- 
597   
598      .. [1] `Andrew Rosenberg and Julia Hirschberg, 2007. V-Measure: A 
599         conditional entropy-based external cluster evaluation measure 
600         <https://aclweb.org/anthology/D/D07/D07-1043.pdf>`_ 
601   
602      Examples 
603      -------- 
604   
605      Perfect labelings are homogeneous:: 
606   
607        >>> from sklearn.metrics.cluster import homogeneity_score 
608        >>> homogeneity_score([0, 0, 1, 1], [1, 1, 0, 0]) 
609        np.float64(1.0) 
610   
611      Non-perfect labelings that further split classes into more clusters can be 
612      perfectly homogeneous:: 
613   
614        >>> print("%.6f" % homogeneity_score([0, 0, 1, 1], [0, 0, 1, 2])) 
615        1.000000 
616        >>> print("%.6f" % homogeneity_score([0, 0, 1, 1], [0, 1, 2, 3])) 
617        1.000000 
618   
619      Clusters that include samples from different classes do not make for an 
620      homogeneous labeling:: 
621   
622        >>> print("%.6f" % homogeneity_score([0, 0, 1, 1], [0, 1, 0, 1])) 
623        0.0... 
624        >>> print("%.6f" % homogeneity_score([0, 0, 1, 1], [0, 0, 0, 0])) 
625        0.0... 
626      """
627      return homogeneity_completeness_v_measure(labels_true, labels_pred)[0]

637  def completeness_score(labels_true, labels_pred):
638      """Compute completeness metric of a cluster labeling given a ground truth. 
639   
640      A clustering result satisfies completeness if all the data points 
641      that are members of a given class are elements of the same cluster. 
642   
643      This metric is independent of the absolute values of the labels: 
644      a permutation of the class or cluster label values won't change the 
645      score value in any way. 
646   
647      This metric is not symmetric: switching ``label_true`` with ``label_pred`` 
648      will return the :func:`homogeneity_score` which will be different in 
649      general. 
650   
651      Read more in the :ref:`User Guide <homogeneity_completeness>`. 
652   
653      Parameters 
654      ---------- 
655      labels_true : array-like of shape (n_samples,) 
656          Ground truth class labels to be used as a reference. 
657   
658      labels_pred : array-like of shape (n_samples,) 
659          Cluster labels to evaluate. 
660   
661      Returns 
662      ------- 
663      completeness : float 
664         Score between 0.0 and 1.0. 1.0 stands for perfectly complete labeling. 
665   
666      See Also 
667      -------- 
668      homogeneity_score : Homogeneity metric of cluster labeling. 
669      v_measure_score : V-Measure (NMI with arithmetic mean option). 
670   
671      References 
672      ---------- 
673   
674      .. [1] `Andrew Rosenberg and Julia Hirschberg, 2007. V-Measure: A 
675         conditional entropy-based external cluster evaluation measure 
676         <https://aclweb.org/anthology/D/D07/D07-1043.pdf>`_ 
677   
678      Examples 
679      -------- 
680   
681      Perfect labelings are complete:: 
682   
683        >>> from sklearn.metrics.cluster import completeness_score 
684        >>> completeness_score([0, 0, 1, 1], [1, 1, 0, 0])

714  def v_measure_score(labels_true, labels_pred, *, beta=1.0):
715      """V-measure cluster labeling given a ground truth. 
716   
717      This score is identical to :func:`normalized_mutual_info_score` with 
718      the ``'arithmetic'`` option for averaging. 
719   
720      The V-measure is the harmonic mean between homogeneity and completeness:: 
721   
722          v = (1 + beta) * homogeneity * completeness 
723               / (beta * homogeneity + completeness) 
724   
725      This metric is independent of the absolute values of the labels: 
726      a permutation of the class or cluster label values won't change the 
727      score value in any way. 
728   
729      This metric is furthermore symmetric: switching ``label_true`` with 
730      ``label_pred`` will return the same score value. This can be useful to 
731      measure the agreement of two independent label assignments strategies 
732      on the same dataset when the real ground truth is not known. 
733   
734      Read more in the :ref:`User Guide <homogeneity_completeness>`. 
735   
736      Parameters 
737      ---------- 
738      labels_true : array-like of shape (n_samples,) 
739          Ground truth class labels to be used as a reference. 
740   
741      labels_pred : array-like of shape (n_samples,) 
742          Cluster labels to evaluate. 
743   
744      beta : float, default=1.0 
745          Ratio of weight attributed to ``homogeneity`` vs ``completeness``. 
746          If ``beta`` is greater than 1, ``completeness`` is weighted more 
747          strongly in the calculation. If ``beta`` is less than 1, 
748          ``homogeneity`` is weighted more strongly. 
749   
750      Returns 
751      ------- 
752      v_measure : float 
753         Score between 0.0 and 1.0. 1.0 stands for perfectly complete labeling. 
754   
755      See Also 
756      -------- 
757      homogeneity_score : Homogeneity metric of cluster labeling. 
758      completeness_score : Completeness metric of cluster labeling. 
759      normalized_mutual_info_score : Normalized Mutual Information. 
760   
761      References 
762      ---------- 
763   
764      .. [1] `Andrew Rosenberg and Julia Hirschberg, 2007. V-Measure: A 
765         conditional entropy-based external cluster evaluation measure 
766         <https://aclweb.org/anthology/D/D07/D07-1043.pdf>`_ 
767   
768      Examples 
769      -------- 
770      Perfect labelings are both homogeneous and complete, hence have score 1.0:: 
771   
772        >>> from sklearn.metrics.cluster import v_measure_score 
773        >>> v_measure_score([0, 0, 1, 1], [0, 0, 1, 1]) 
774        np.float64(1.0) 
775        >>> v_measure_score([0, 0, 1, 1], [1, 1, 0, 0]) 
776        np.float64(1.0) 
777   
778      Labelings that assign all classes members to the same clusters 
779      are complete but not homogeneous, hence penalized:: 
780   
781        >>> print("%.6f" % v_measure_score([0, 0, 1, 2], [0, 0, 1, 1])) 
782        0.8... 
783        >>> print("%.6f" % v_measure_score([0, 1, 2, 3], [0, 0, 1, 1])) 
784        0.66... 
785   
786      Labelings that have pure clusters with members coming from the same 
787      classes are homogeneous but un-necessary splits harm completeness 
788      and thus penalize V-measure as well:: 
789   
790        >>> print("%.6f" % v_measure_score([0, 0, 1, 1], [0, 0, 1, 2])) 
791        0.8... 
792        >>> print("%.6f" % v_measure_score([0, 0, 1, 1], [0, 1, 2, 3])) 
793        0.66... 
794   
795      If classes members are completely split across different clusters, 
796      the assignment is totally incomplete, hence the V-Measure is null:: 
797   
798        >>> print("%.6f" % v_measure_score([0, 0, 0, 0], [0, 1, 2, 3])) 
799        0.0... 
800   
801      Clusters that include samples from totally different classes totally 
802      destroy the homogeneity of the labeling, hence:: 
803   
804        >>> print("%.6f" % v_measure_score([0, 0, 1, 1], [0, 0, 0, 0])) 
805        0.0... 
806      """
807      return homogeneity_completeness_v_measure(labels_true, labels_pred, beta=beta)[2]

818  def mutual_info_score(labels_true, labels_pred, *, contingency=None):
819      """Mutual Information between two clusterings. 
820   
821      The Mutual Information is a measure of the similarity between two labels 
822      of the same data. Where :math:`|U_i|` is the number of the samples 
823      in cluster :math:`U_i` and :math:`|V_j|` is the number of the 
824      samples in cluster :math:`V_j`, the Mutual Information 
825      between clusterings :math:`U` and :math:`V` is given as: 
826   
827      .. math:: 
828   
829          MI(U,V)=\\sum_{i=1}^{|U|} \\sum_{j=1}^{|V|} \\frac{|U_i\\cap V_j|}{N} 
830          \\log\\frac{N|U_i \\cap V_j|}{|U_i||V_j|} 
831   
832      This metric is independent of the absolute values of the labels: 
833      a permutation of the class or cluster label values won't change the 
834      score value in any way. 
835   
836      This metric is furthermore symmetric: switching :math:`U` (i.e 
837      ``label_true``) with :math:`V` (i.e. ``label_pred``) will return the 
838      same score value. This can be useful to measure the agreement of two 
839      independent label assignments strategies on the same dataset when the 
840      real ground truth is not known. 
841   
842      Read more in the :ref:`User Guide <mutual_info_score>`. 
843   
844      Parameters 
845      ---------- 
846      labels_true : array-like of shape (n_samples,), dtype=integral 
847          A clustering of the data into disjoint subsets, called :math:`U` in 
848          the above formula. 
849   
850      labels_pred : array-like of shape (n_samples,), dtype=integral 
851          A clustering of the data into disjoint subsets, called :math:`V` in 
852          the above formula. 
853   
854      contingency : {array-like, sparse matrix} of shape \ 
855              (n_classes_true, n_classes_pred), default=None 
856          A contingency matrix given by the 
857          :func:`~sklearn.metrics.cluster.contingency_matrix` function. If value 
858          is ``None``, it will be computed, otherwise the given value is used, 
859          with ``labels_true`` and ``labels_pred`` ignored. 
860   
861      Returns 
862      ------- 
863      mi : float 
864         Mutual information, a non-negative value, measured in nats using the 
865         natural logarithm. 
866   
867      See Also 
868      -------- 
869      adjusted_mutual_info_score : Adjusted against chance Mutual Information. 
870      normalized_mutual_info_score : Normalized Mutual Information. 
871   
872      Notes 
873      ----- 
874      The logarithm used is the natural logarithm (base-e). 
875   
876      Examples 
877      -------- 
878      >>> from sklearn.metrics import mutual_info_score 
879      >>> labels_true = [0, 1, 1, 0, 1, 0] 
880      >>> labels_pred = [0, 1, 0, 0, 1, 1] 
881      >>> mutual_info_score(labels_true, labels_pred) 
882      np.float64(0.056...) 
883      """
884      if contingency is None:
885          labels_true, labels_pred = check_clusterings(labels_true, labels_pred)
886          contingency = contingency_matrix(labels_true, labels_pred, sparse=True)
887      else:
888          contingency = check_array(
889              contingency,
890              accept_sparse=["csr", "csc", "coo"],
891              dtype=[int, np.int32, np.int64],
892          )
893  
894      if isinstance(contingency, np.ndarray):
895          # For an array
896          nzx, nzy = np.nonzero(contingency)
897          nz_val = contingency[nzx, nzy]
898      else:
899          # For a sparse matrix
900          nzx, nzy, nz_val = sp.find(contingency)
901  
902      contingency_sum = contingency.sum()
903      pi = np.ravel(contingency.sum(axis=1))
904      pj = np.ravel(contingency.sum(axis=0))
905  
906      # Since MI <= min(H(X), H(Y)), any labelling with zero entropy, i.e. containing a
907      # single cluster, implies MI = 0
908      if pi.size == 1 or pj.size == 1:
909          return 0.0
910  
911      log_contingency_nm = np.log(nz_val)
912      contingency_nm = nz_val / contingency_sum
913      # Don't need to calculate the full outer product, just for non-zeroes
914      outer = pi.take(nzx).astype(np.int64, copy=False) * pj.take(nzy).astype(
915          np.int64, copy=False
916      )
917      log_outer = -np.log(outer) + log(pi.sum()) + log(pj.sum())
918      mi = (
919          contingency_nm * (log_contingency_nm - log(contingency_sum))
920          + contingency_nm * log_outer
921      )
922      mi = np.where(np.abs(mi) < np.finfo(mi.dtype).eps, 0.0, mi)
923      return np.clip(mi.sum(), 0.0, None)
924

935      labels_true, labels_pred, *, average_method="arithmetic"
936  ):
937      """Adjusted Mutual Information between two clusterings. 
938   
939      Adjusted Mutual Information (AMI) is an adjustment of the Mutual 
940      Information (MI) score to account for chance. It accounts for the fact that 
941      the MI is generally higher for two clusterings with a larger number of 
942      clusters, regardless of whether there is actually more information shared. 
943      For two clusterings :math:`U` and :math:`V`, the AMI is given as:: 
944   
945          AMI(U, V) = [MI(U, V) - E(MI(U, V))] / [avg(H(U), H(V)) - E(MI(U, V))] 
946   
947      This metric is independent of the absolute values of the labels: 
948      a permutation of the class or cluster label values won't change the 
949      score value in any way. 
950   
951      This metric is furthermore symmetric: switching :math:`U` (``label_true``) 
952      with :math:`V` (``labels_pred``) will return the same score value. This can 
953      be useful to measure the agreement of two independent label assignments 
954      strategies on the same dataset when the real ground truth is not known. 
955   
956      Be mindful that this function is an order of magnitude slower than other 
957      metrics, such as the Adjusted Rand Index. 
958   
959      Read more in the :ref:`User Guide <mutual_info_score>`. 
960   
961      Parameters 
962      ---------- 
963      labels_true : int array-like of shape (n_samples,) 
964          A clustering of the data into disjoint subsets, called :math:`U` in 
965          the above formula. 
966   
967      labels_pred : int array-like of shape (n_samples,) 
968          A clustering of the data into disjoint subsets, called :math:`V` in 
969          the above formula. 
970   
971      average_method : {'min', 'geometric', 'arithmetic', 'max'}, default='arithmetic' 
972          How to compute the normalizer in the denominator. 
973   
974          .. versionadded:: 0.20 
975   
976          .. versionchanged:: 0.22 
977             The default value of ``average_method`` changed from 'max' to 
978             'arithmetic'. 
979   
980      Returns 
981      ------- 
982      ami: float (upperlimited by 1.0) 
983         The AMI returns a value of 1 when the two partitions are identical 
984         (ie perfectly matched). Random partitions (independent labellings) have 
985         an expected AMI around 0 on average hence can be negative. The value is 
986         in adjusted nats (based on the natural logarithm). 
987   
988      See Also 
989      -------- 
990      adjusted_rand_score : Adjusted Rand Index. 
991      mutual_info_score : Mutual Information (not adjusted for chance). 
992   
993      References 
994      ---------- 
995      .. [1] `Vinh, Epps, and Bailey, (2010). Information Theoretic Measures for 
996         Clusterings Comparison: Variants, Properties, Normalization and 
997         Correction for Chance, JMLR 
998         <http://jmlr.csail.mit.edu/papers/volume11/vinh10a/vinh10a.pdf>`_ 
999   
1000     .. [2] `Wikipedia entry for the Adjusted Mutual Information 
1001        <https://en.wikipedia.org/wiki/Adjusted_Mutual_Information>`_ 
1002  
1003     Examples 
1004     -------- 
1005  
1006     Perfect labelings are both homogeneous and complete, hence have 
1007     score 1.0:: 
1008  
1009       >>> from sklearn.metrics.cluster import adjusted_mutual_info_score 
1010       >>> adjusted_mutual_info_score([0, 0, 1, 1], [0, 0, 1, 1]) 
1011       ... # doctest: +SKIP 
1012       1.0 
1013       >>> adjusted_mutual_info_score([0, 0, 1, 1], [1, 1, 0, 0]) 
1014       ... # doctest: +SKIP 
1015       1.0 
1016  
1017     If classes members are completely split across different clusters, 
1018     the assignment is totally in-complete, hence the AMI is null:: 
1019  
1020       >>> adjusted_mutual_info_score([0, 0, 0, 0], [0, 1, 2, 3]) 
1021       ... # doctest: +SKIP 
1022       0.0 
1023     """
1024     labels_true, labels_pred = check_clusterings(labels_true, labels_pred)
1025     n_samples = labels_true.shape[0]
1026     classes = np.unique(labels_true)
1027     clusters = np.unique(labels_pred)
1028 
1029     # Special limit cases: no clustering since the data is not split.
1030     # It corresponds to both labellings having zero entropy.
1031     # This is a perfect match hence return 1.0.
1032     if (
1033         classes.shape[0] == clusters.shape[0] == 1
1034         or classes.shape[0] == clusters.shape[0] == 0
1035     ):
1036         return 1.0
1037 
1038     contingency = contingency_matrix(labels_true, labels_pred, sparse=True)
1039     # Calculate the MI for the two clusterings
1040     mi = mutual_info_score(labels_true, labels_pred, contingency=contingency)
1041     # Calculate the expected value for the mutual information
1042     emi = expected_mutual_information(contingency, n_samples)
1043     # Calculate entropy for each labeling
1044     h_true, h_pred = entropy(labels_true), entropy(labels_pred)
1045     normalizer = _generalized_average(h_true, h_pred, average_method)
1046     denominator = normalizer - emi
1047     # Avoid 0.0 / 0.0 when expectation equals maximum, i.e. a perfect match.
1048     # normalizer should always be >= emi, but because of floating-point
1049     # representation, sometimes emi is slightly larger. Correct this
1050     # by preserving the sign.
1051     if denominator < 0:
1052         denominator = min(denominator, -np.finfo("float64").eps)
1053     else:
1054         denominator = max(denominator, np.finfo("float64").eps)
1055     ami = (mi - emi) / denominator
1056     return ami

1182 def fowlkes_mallows_score(labels_true, labels_pred, *, sparse=False):
1183     """Measure the similarity of two clusterings of a set of points. 
1184  
1185     .. versionadded:: 0.18 
1186  
1187     The Fowlkes-Mallows index (FMI) is defined as the geometric mean between of 
1188     the precision and recall:: 
1189  
1190         FMI = TP / sqrt((TP + FP) * (TP + FN)) 
1191  
1192     Where ``TP`` is the number of **True Positive** (i.e. the number of pairs of 
1193     points that belong to the same cluster in both ``labels_true`` and 
1194     ``labels_pred``), ``FP`` is the number of **False Positive** (i.e. the 
1195     number of pairs of points that belong to the same cluster in 
1196     ``labels_pred`` but not in ``labels_true``) and ``FN`` is the number of 
1197     **False Negative** (i.e. the number of pairs of points that belong to the 
1198     same cluster in ``labels_true`` but not in ``labels_pred``). 
1199  
1200     The score ranges from 0 to 1. A high value indicates a good similarity 
1201     between two clusters. 
1202  
1203     Read more in the :ref:`User Guide <fowlkes_mallows_scores>`. 
1204  
1205     Parameters 
1206     ---------- 
1207     labels_true : array-like of shape (n_samples,), dtype=int 
1208         A clustering of the data into disjoint subsets. 
1209  
1210     labels_pred : array-like of shape (n_samples,), dtype=int 
1211         A clustering of the data into disjoint subsets. 
1212  
1213     sparse : bool, default=False 
1214         Compute contingency matrix internally with sparse matrix. 
1215  
1216     Returns 
1217     ------- 
1218     score : float 
1219        The resulting Fowlkes-Mallows score. 
1220  
1221     References 
1222     ---------- 
1223     .. [1] `E. B. Fowkles and C. L. Mallows, 1983. "A method for comparing two 
1224        hierarchical clusterings". Journal of the American Statistical 
1225        Association 
1226        <https://www.tandfonline.com/doi/abs/10.1080/01621459.1983.10478008>`_ 
1227  
1228     .. [2] `Wikipedia entry for the Fowlkes-Mallows Index 
1229            <https://en.wikipedia.org/wiki/Fowlkes-Mallows_index>`_ 
1230  
1231     Examples 
1232     -------- 
1233  
1234     Perfect labelings are both homogeneous and complete, hence have 
1235     score 1.0:: 
1236  
1237       >>> from sklearn.metrics.cluster import fowlkes_mallows_score 
1238       >>> fowlkes_mallows_score([0, 0, 1, 1], [0, 0, 1, 1]) 
1239       np.float64(1.0) 
1240       >>> fowlkes_mallows_score([0, 0, 1, 1], [1, 1, 0, 0]) 
1241       np.float64(1.0) 
1242  
1243     If classes members are completely split across different clusters, 
1244     the assignment is totally random, hence the FMI is null:: 
1245  
1246       >>> fowlkes_mallows_score([0, 0, 0, 0], [0, 1, 2, 3]) 
1247       0.0 
1248     """
1249     labels_true, labels_pred = check_clusterings(labels_true, labels_pred)
1250     (n_samples,) = labels_true.shape
1251 
1252     c = contingency_matrix(labels_true, labels_pred, sparse=True)
1253     c = c.astype(np.int64, copy=False)
1254     tk = np.dot(c.data, c.data) - n_samples
1255     pk = np.sum(np.asarray(c.sum(axis=0)).ravel() ** 2) - n_samples
1256     qk = np.sum(np.asarray(c.sum(axis=1)).ravel() ** 2) - n_samples
1257     return np.sqrt(tk / pk) * np.sqrt(tk / qk) if tk != 0.0 else 0.0
