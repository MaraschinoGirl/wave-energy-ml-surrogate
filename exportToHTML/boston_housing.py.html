4    from keras.src.utils.file_utils import get_file

7    @keras_export("keras.datasets.boston_housing.load_data")
8    def load_data(path="boston_housing.npz", test_split=0.2, seed=113):
9        """Loads the Boston Housing dataset. 
10    
11       This is a dataset taken from the StatLib library which is maintained at 
12       Carnegie Mellon University. 
13    
14       **WARNING:** This dataset has an ethical problem: the authors of this 
15       dataset included a variable, "B", that may appear to assume that racial 
16       self-segregation influences house prices. As such, we strongly discourage 
17       the use of this dataset, unless in the context of illustrating ethical 
18       issues in data science and machine learning. 
19    
20       Samples contain 13 attributes of houses at different locations around the 
21       Boston suburbs in the late 1970s. Targets are the median values of 
22       the houses at a location (in k$). 
23    
24       The attributes themselves are defined in the 
25       [StatLib website](http://lib.stat.cmu.edu/datasets/boston). 
26    
27       Args: 
28           path: path where to cache the dataset locally 
29               (relative to `~/.keras/datasets`). 
30           test_split: fraction of the data to reserve as test set. 
31           seed: Random seed for shuffling the data 
32               before computing the test split. 
33    
34       Returns: 
35           Tuple of NumPy arrays: `(x_train, y_train), (x_test, y_test)`. 
36    
37       **x_train, x_test**: NumPy arrays with shape `(num_samples, 13)` 
38           containing either the training samples (for x_train), 
39           or test samples (for y_train). 
40    
41       **y_train, y_test**: NumPy arrays of shape `(num_samples,)` containing the 
42           target scalars. The targets are float scalars typically between 10 and 
43           50 that represent the home prices in k$. 
44       """
45       assert 0 <= test_split < 1
46       origin_folder = (
47           "https://storage.googleapis.com/tensorflow/tf-keras-datasets/"
48       )
49       path = get_file(
50           path,
51           origin=origin_folder + "boston_housing.npz",
52           file_hash=(  # noqa: E501
53               "f553886a1f8d56431e820c5b82552d9d95cfcb96d1e678153f8839538947dff5"
54           ),
55       )
56       with np.load(path, allow_pickle=True) as f:
57           x = f["x"]
58           y = f["y"]
59   
60       rng = np.random.RandomState(seed)
61       indices = np.arange(len(x))
62       rng.shuffle(indices)
63       x = x[indices]
64       y = y[indices]
65   
66       x_train = np.array(x[: int(len(x) * (1 - test_split))])
67       y_train = np.array(y[: int(len(x) * (1 - test_split))])
68       x_test = np.array(x[int(len(x) * (1 - test_split)) :])
69       y_test = np.array(y[int(len(x) * (1 - test_split)) :])
70       return (x_train, y_train), (x_test, y_test)
