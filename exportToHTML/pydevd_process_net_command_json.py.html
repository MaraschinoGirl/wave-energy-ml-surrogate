64   def _convert_rules_to_exclude_filters(rules, on_error):
65       exclude_filters = []
66       if not isinstance(rules, list):
67           on_error('Invalid "rules" (expected list of dicts). Found: %s' % (rules,))
68   
69       else:
70           directory_exclude_filters = []
71           module_exclude_filters = []
72           glob_exclude_filters = []
73   
74           for rule in rules:
75               if not isinstance(rule, dict):
76                   on_error('Invalid "rules" (expected list of dicts). Found: %s' % (rules,))
77                   continue
78   
79               include = rule.get("include")
80               if include is None:
81                   on_error('Invalid "rule" (expected dict with "include"). Found: %s' % (rule,))
82                   continue
83   
84               path = rule.get("path")
85               module = rule.get("module")
86               if path is None and module is None:
87                   on_error('Invalid "rule" (expected dict with "path" or "module"). Found: %s' % (rule,))
88                   continue
89   
90               if path is not None:
91                   glob_pattern = path
92                   if "*" not in path and "?" not in path:
93                       if os.path.isdir(glob_pattern):
94                           # If a directory was specified, add a '/**'
95                           # to be consistent with the glob pattern required
96                           # by pydevd.
97                           if not glob_pattern.endswith("/") and not glob_pattern.endswith("\\"):
98                               glob_pattern += "/"
99                           glob_pattern += "**"
100                      directory_exclude_filters.append(ExcludeFilter(glob_pattern, not include, True))
101                  else:
102                      glob_exclude_filters.append(ExcludeFilter(glob_pattern, not include, True))
103  
104              elif module is not None:
105                  module_exclude_filters.append(ExcludeFilter(module, not include, False))
106  
107              else:
108                  on_error("Internal error: expected path or module to be specified.")
109  
110          # Note that we have to sort the directory/module exclude filters so that the biggest
111          # paths match first.
112          # i.e.: if we have:
113          # /sub1/sub2/sub3
114          # a rule with /sub1/sub2 would match before a rule only with /sub1.
115          directory_exclude_filters = sorted(directory_exclude_filters, key=lambda exclude_filter: -len(exclude_filter.name))
116          module_exclude_filters = sorted(module_exclude_filters, key=lambda exclude_filter: -len(exclude_filter.name))
117          exclude_filters = directory_exclude_filters + glob_exclude_filters + module_exclude_filters
118  
119      return exclude_filters

122  class IDMap(object):
123      def __init__(self):
124          self._value_to_key = {}
125          self._key_to_value = {}
126          self._next_id = partial(next, itertools.count(0))
127  
128      def obtain_value(self, key):
129          return self._key_to_value[key]
130  
131      def obtain_key(self, value):
132          try:
133              key = self._value_to_key[value]
134          except KeyError:
135              key = self._next_id()
136              self._key_to_value[key] = value
137              self._value_to_key[value] = key
138          return key

150      def process_net_command_json(self, py_db, json_contents, send_response=True):
151          """ 
152          Processes a debug adapter protocol json command. 
153          """
154  
155          DEBUG = False
156  
157          try:
158              if isinstance(json_contents, bytes):
159                  json_contents = json_contents.decode("utf-8")
160  
161              request = self.from_json(json_contents, update_ids_from_dap=True)
162          except Exception as e:
163              try:
164                  loaded_json = json.loads(json_contents)
165                  request = Request(loaded_json.get("command", "<unknown>"), loaded_json["seq"])
166              except:
167                  # There's not much we can do in this case...
168                  pydev_log.exception("Error loading json: %s", json_contents)
169                  return
170  
171              error_msg = str(e)
172              if error_msg.startswith("'") and error_msg.endswith("'"):
173                  error_msg = error_msg[1:-1]
174  
175              # This means a failure processing the request (but we were able to load the seq,
176              # so, answer with a failure response).
177              def on_request(py_db, request):
178                  error_response = {
179                      "type": "response",
180                      "request_seq": request.seq,
181                      "success": False,
182                      "command": request.command,
183                      "message": error_msg,
184                  }
185                  return NetCommand(CMD_RETURN, 0, error_response, is_json=True)
186  
187          else:
188              if DebugInfoHolder.DEBUG_TRACE_LEVEL >= 1:
189                  pydev_log.info(
190                      "Process %s: %s\n"
191                      % (
192                          request.__class__.__name__,
193                          json.dumps(request.to_dict(update_ids_to_dap=True), indent=4, sort_keys=True),
194                      )
195                  )
196  
197              assert request.type == "request"
198              method_name = "on_%s_request" % (request.command.lower(),)
199              on_request = getattr(self, method_name, None)
200              if on_request is None:
201                  print("Unhandled: %s not available in PyDevJsonCommandProcessor.\n" % (method_name,))
202                  return
203  
204              if DEBUG:
205                  print("Handled in pydevd: %s (in PyDevJsonCommandProcessor).\n" % (method_name,))
206  
207          with py_db._main_lock:
208              if request.__class__ == PydevdAuthorizeRequest:
209                  authorize_request = request  # : :type authorize_request: PydevdAuthorizeRequest
210                  access_token = authorize_request.arguments.debugServerAccessToken
211                  py_db.authentication.login(access_token)
212  
213              if not py_db.authentication.is_authenticated():
214                  response = Response(request.seq, success=False, command=request.command, message="Client not authenticated.", body={})
215                  cmd = NetCommand(CMD_RETURN, 0, response, is_json=True)
216                  py_db.writer.add_command(cmd)
217                  return
218  
219              cmd = on_request(py_db, request)
220              if cmd is not None and send_response:
221                  py_db.writer.add_command(cmd)

313          """ 
314          :param CompletionsRequest request: 
315          """
316          arguments = request.arguments  # : :type arguments: CompletionsArguments
317          seq = request.seq
318          text = arguments.text
319          frame_id = arguments.frameId
320          thread_id = py_db.suspended_frames_manager.get_thread_id_for_variable_reference(frame_id)

323              body = CompletionsResponseBody([])

327              return NetCommand(CMD_RETURN, 0, variables_response, is_json=True)

329          # Note: line and column are 1-based (convert to 0-based for pydevd).
330          column = arguments.column - 1
331

333              # line is optional
334              line = -1
335          else:
336              line = arguments.line - 1
337  
338          self.api.request_completions(py_db, seq, thread_id, frame_id, text, line=line, column=column)

340      def _resolve_remote_root(self, local_root, remote_root):
341          if remote_root == ".":
342              cwd = os.getcwd()
343              append_pathsep = local_root.endswith("\\") or local_root.endswith("/")
344              return cwd + (os.path.sep if append_pathsep else "")
345          return remote_root

347      def _set_debug_options(self, py_db, args, start_reason):
348          rules = args.get("rules")
349          stepping_resumes_all_threads = args.get("steppingResumesAllThreads", True)
350          self.api.set_stepping_resumes_all_threads(py_db, stepping_resumes_all_threads)
351  
352          terminate_child_processes = args.get("terminateChildProcesses", True)
353          self.api.set_terminate_child_processes(py_db, terminate_child_processes)
354  
355          terminate_keyboard_interrupt = args.get("onTerminate", "kill") == "KeyboardInterrupt"
356          self.api.set_terminate_keyboard_interrupt(py_db, terminate_keyboard_interrupt)
357  
358          variable_presentation = args.get("variablePresentation", None)
359          if isinstance(variable_presentation, dict):
360  
361              def get_variable_presentation(setting, default):
362                  value = variable_presentation.get(setting, default)
363                  if value not in ("group", "inline", "hide"):
364                      pydev_log.info(
365                          'The value set for "%s" (%s) in the variablePresentation is not valid. Valid values are: "group", "inline", "hide"'
366                          % (
367                              setting,
368                              value,
369                          )
370                      )
371                      value = default
372  
373                  return value
374  
375              default = get_variable_presentation("all", "group")
376  
377              special_presentation = get_variable_presentation("special", default)
378              function_presentation = get_variable_presentation("function", default)
379              class_presentation = get_variable_presentation("class", default)
380              protected_presentation = get_variable_presentation("protected", default)
381  
382              self.api.set_variable_presentation(
383                  py_db,
384                  self.api.VariablePresentation(special_presentation, function_presentation, class_presentation, protected_presentation),
385              )
386  
387          exclude_filters = []
388  
389          if rules is not None:
390              exclude_filters = _convert_rules_to_exclude_filters(rules, lambda msg: self.api.send_error_message(py_db, msg))
391  
392          self.api.set_exclude_filters(py_db, exclude_filters)
393  
394          debug_options = _extract_debug_options(
395              args.get("options"),
396              args.get("debugOptions"),
397          )
398          self._options.update_fom_debug_options(debug_options)
399          self._options.update_from_args(args)
400  
401          self.api.set_use_libraries_filter(py_db, self._options.just_my_code)
402  
403          if self._options.client_os:
404              self.api.set_ide_os(self._options.client_os)
405  
406          path_mappings = []
407          for pathMapping in args.get("pathMappings", []):
408              localRoot = pathMapping.get("localRoot", "")
409              remoteRoot = pathMapping.get("remoteRoot", "")
410              remoteRoot = self._resolve_remote_root(localRoot, remoteRoot)
411              if (localRoot != "") and (remoteRoot != ""):
412                  path_mappings.append((localRoot, remoteRoot))
413  
414          if bool(path_mappings):
415              pydevd_file_utils.setup_client_server_paths(path_mappings)
416  
417          resolve_symlinks = args.get("resolveSymlinks", None)
418          if resolve_symlinks is not None:
419              pydevd_file_utils.set_resolve_symlinks(resolve_symlinks)
420  
421          redirecting = args.get("isOutputRedirected")
422          if self._options.redirect_output:
423              py_db.enable_output_redirection(True, True)
424              redirecting = True
425          else:
426              py_db.enable_output_redirection(False, False)
427  
428          py_db.is_output_redirected = redirecting
429  
430          self.api.set_show_return_values(py_db, self._options.show_return_value)
431  
432          if not self._options.break_system_exit_zero:
433              ignore_system_exit_codes = [0, None]
434              if self._options.django_debug or self._options.flask_debug:
435                  ignore_system_exit_codes += [3]
436  
437              self.api.set_ignore_system_exit_codes(py_db, ignore_system_exit_codes)
438  
439          auto_reload = args.get("autoReload", {})
440          if not isinstance(auto_reload, dict):
441              pydev_log.info("Expected autoReload to be a dict. Received: %s" % (auto_reload,))
442              auto_reload = {}
443  
444          enable_auto_reload = auto_reload.get("enable", False)
445          watch_dirs = auto_reload.get("watchDirectories")
446          if not watch_dirs:
447              watch_dirs = []
448              # Note: by default this is no longer done because on some cases there are entries in the PYTHONPATH
449              # such as the home directory or /python/x64, where the site packages are in /python/x64/libs, so,
450              # we only watch the current working directory as well as executed script.
451              # check = getattr(sys, 'path', [])[:]
452              # # By default only watch directories that are in the project roots /
453              # # program dir (if available), sys.argv[0], as well as the current dir (we don't want to
454              # # listen to the whole site-packages by default as it can be huge).
455              # watch_dirs = [pydevd_file_utils.absolute_path(w) for w in check]
456              # watch_dirs = [w for w in watch_dirs if py_db.in_project_roots_filename_uncached(w) and os.path.isdir(w)]
457  
458              program = args.get("program")
459              if program:
460                  if os.path.isdir(program):
461                      watch_dirs.append(program)
462                  else:
463                      watch_dirs.append(os.path.dirname(program))
464              watch_dirs.append(os.path.abspath("."))
465  
466              argv = getattr(sys, "argv", [])
467              if argv:
468                  f = argv[0]
469                  if f:  # argv[0] could be None (https://github.com/microsoft/debugpy/issues/987)
470                      if os.path.isdir(f):
471                          watch_dirs.append(f)
472                      else:
473                          watch_dirs.append(os.path.dirname(f))
474  
475          if not isinstance(watch_dirs, (list, set, tuple)):
476              watch_dirs = (watch_dirs,)
477          new_watch_dirs = set()
478          for w in watch_dirs:
479              try:
480                  new_watch_dirs.add(pydevd_file_utils.get_path_with_real_case(pydevd_file_utils.absolute_path(w)))
481              except Exception:
482                  pydev_log.exception("Error adding watch dir: %s", w)
483          watch_dirs = new_watch_dirs
484  
485          poll_target_time = auto_reload.get("pollingInterval", 1)
486          exclude_patterns = auto_reload.get(
487              "exclude", ("**/.git/**", "**/__pycache__/**", "**/node_modules/**", "**/.metadata/**", "**/site-packages/**")
488          )
489          include_patterns = auto_reload.get("include", ("**/*.py", "**/*.pyw"))
490          self.api.setup_auto_reload_watcher(py_db, enable_auto_reload, watch_dirs, poll_target_time, exclude_patterns, include_patterns)
491  
492          if self._options.stop_on_entry and start_reason == "launch":
493              self.api.stop_on_entry()
494  
495          self.api.set_gui_event_loop(py_db, self._options.gui_event_loop)

497      def _send_process_event(self, py_db, start_method):
498          argv = getattr(sys, "argv", [])
499          if len(argv) > 0:
500              name = argv[0]
501          else:
502              name = ""
503  
504          if isinstance(name, bytes):
505              name = name.decode(file_system_encoding, "replace")
506              name = name.encode("utf-8")
507  
508          body = ProcessEventBody(
509              name=name,
510              systemProcessId=os.getpid(),
511              isLocalProcess=True,
512              startMethod=start_method,
513          )
514          event = ProcessEvent(body)
515          py_db.writer.add_command(NetCommand(CMD_PROCESS_EVENT, 0, event, is_json=True))

552      def on_continue_request(self, py_db, request):
553          """ 
554          :param ContinueRequest request: 
555          """
556          arguments = request.arguments  # : :type arguments: ContinueArguments
557          thread_id = arguments.threadId
558  
559          def on_resumed():
560              body = {"allThreadsContinued": thread_id == "*"}
561              response = pydevd_base_schema.build_response(request, kwargs={"body": body})
562              cmd = NetCommand(CMD_RETURN, 0, response, is_json=True)
563              py_db.writer.add_command(cmd)
564  
565          if py_db.multi_threads_single_notification:
566              # Only send resumed notification when it has actually resumed!
567              # (otherwise the user could send a continue, receive the notification and then
568              # request a new pause which would be paused without sending any notification as
569              # it didn't really run in the first place).
570              py_db.threads_suspended_single_notification.add_on_resumed_callback(on_resumed)
571              self.api.request_resume_thread(thread_id)
572          else:
573              # Only send resumed notification when it has actually resumed!
574              # (otherwise the user could send a continue, receive the notification and then
575              # request a new pause which would be paused without sending any notification as
576              # it didn't really run in the first place).
577              self.api.request_resume_thread(thread_id)
578              on_resumed()

580      def on_next_request(self, py_db, request):
581          """ 
582          :param NextRequest request:

584          arguments = request.arguments  # : :type arguments: NextArguments

694      def _get_hit_condition_expression(self, hit_condition):
695          """Following hit condition values are supported 
696   
697          * x or == x when breakpoint is hit x times 
698          * >= x when breakpoint is hit more than or equal to x times 
699          * % x when breakpoint is hit multiple of x times 
700   
701          Returns '@HIT@ == x' where @HIT@ will be replaced by number of hits 
702          """
703          if not hit_condition:
704              return None
705  
706          expr = hit_condition.strip()
707          try:
708              int(expr)
709              return "@HIT@ == {}".format(expr)
710          except ValueError:
711              pass
712  
713          if expr.startswith("%"):
714              return "@HIT@ {} == 0".format(expr)
715  
716          if expr.startswith("==") or expr.startswith(">") or expr.startswith("<"):
717              return "@HIT@ {}".format(expr)
718  
719          return hit_condition

864                  error_msg = "Breakpoint in file that does not exist."

866              elif error_code == self.api.ADD_BREAKPOINT_FILE_EXCLUDED_BY_FILTERS:

882                  error_msg = "Breakpoint not validated (reason unknown -- please report as bug)."

885                  verified=False, id=breakpoint_id, line=result.translated_line, message=error_msg, source=source

890      def on_setexceptionbreakpoints_request(self, py_db, request):
891          """ 
892          :param SetExceptionBreakpointsRequest request: 
893          """
894          # : :type arguments: SetExceptionBreakpointsArguments
895          arguments = request.arguments
896          filters = arguments.filters
897          exception_options = arguments.exceptionOptions
898          self.api.remove_all_exception_breakpoints(py_db)
899  
900          # Can't set these in the DAP.
901          condition = None
902          expression = None
903          notify_on_first_raise_only = False
904  
905          ignore_libraries = 1 if py_db.get_use_libraries_filter() else 0
906  
907          if exception_options:
908              break_raised = False
909              break_uncaught = False
910  
911              for option in exception_options:
912                  option = ExceptionOptions(**option)
913                  if not option.path:
914                      continue
915  
916                  # never: never breaks
917                  #
918                  # always: always breaks
919                  #
920                  # unhandled: breaks when exception unhandled
921                  #
922                  # userUnhandled: breaks if the exception is not handled by user code
923  
924                  notify_on_handled_exceptions = 1 if option.breakMode == "always" else 0
925                  notify_on_unhandled_exceptions = 1 if option.breakMode == "unhandled" else 0
926                  notify_on_user_unhandled_exceptions = 1 if option.breakMode == "userUnhandled" else 0
927                  exception_paths = option.path
928                  break_raised |= notify_on_handled_exceptions
929                  break_uncaught |= notify_on_unhandled_exceptions
930  
931                  exception_names = []
932                  if len(exception_paths) == 0:
933                      continue
934  
935                  elif len(exception_paths) == 1:
936                      if "Python Exceptions" in exception_paths[0]["names"]:
937                          exception_names = ["BaseException"]
938  
939                  else:
940                      path_iterator = iter(exception_paths)
941                      if "Python Exceptions" in next(path_iterator)["names"]:
942                          for path in path_iterator:
943                              for ex_name in path["names"]:
944                                  exception_names.append(ex_name)
945  
946                  for exception_name in exception_names:
947                      self.api.add_python_exception_breakpoint(
948                          py_db,
949                          exception_name,
950                          condition,
951                          expression,
952                          notify_on_handled_exceptions,
953                          notify_on_unhandled_exceptions,
954                          notify_on_user_unhandled_exceptions,
955                          notify_on_first_raise_only,
956                          ignore_libraries,
957                      )
958  
959          else:
960              break_raised = "raised" in filters
961              break_uncaught = "uncaught" in filters
962              break_user = "userUnhandled" in filters
963              if break_raised or break_uncaught or break_user:
964                  notify_on_handled_exceptions = 1 if break_raised else 0
965                  notify_on_unhandled_exceptions = 1 if break_uncaught else 0
966                  notify_on_user_unhandled_exceptions = 1 if break_user else 0
967                  exception = "BaseException"
968  
969                  self.api.add_python_exception_breakpoint(
970                      py_db,
971                      exception,
972                      condition,
973                      expression,
974                      notify_on_handled_exceptions,
975                      notify_on_unhandled_exceptions,
976                      notify_on_user_unhandled_exceptions,
977                      notify_on_first_raise_only,
978                      ignore_libraries,
979                  )
980  
981          if break_raised:
982              btype = None
983              if self._options.django_debug:
984                  btype = "django"
985              elif self._options.flask_debug:
986                  btype = "jinja2"
987  
988              if btype:
989                  self.api.add_plugins_exception_breakpoint(py_db, btype, "BaseException")  # Note: Exception name could be anything here.
990  
991          # Note: no body required on success.
992          set_breakpoints_response = pydevd_base_schema.build_response(request)
993          return NetCommand(CMD_RETURN, 0, set_breakpoints_response, is_json=True)

1182                     # File might not exist at all, or we might not have a permission to read it,
1183                     # but it might also be inside a zipfile, or an IPython cell. In this case,
1184                     # linecache might still be able to retrieve the source.
1185                     lines = (linecache.getline(server_filename, i) for i in itertools.count(1))
1186                     lines = itertools.takewhile(bool, lines)  # empty lines are '\n', EOF is ''
1187 
1188                     # If we didn't get at least one line back, reset it to None so that it's
1189                     # reported as error below, and not as an empty file.
1190                     content = "".join(lines) or None
1191 
1192             if content is None:
1193                 frame_id = pydevd_file_utils.get_frame_id_from_source_reference(source_reference)
1194                 pydev_log.debug("Found frame id: %s for source reference: %s", frame_id, source_reference)
1195                 if frame_id is not None:
1196                     try:
1197                         content = self.api.get_decompiled_source_from_frame_id(py_db, frame_id)
1198                     except Exception:
1199                         pydev_log.exception("Error getting source for frame id: %s", frame_id)
1200                         content = None
1201 
1202         body = SourceResponseBody(content or "")
1203         response_args = {"body": body}
1204 
1205         if content is None:
1206             if source_reference == 0:
1207                 message = "Source unavailable"
1208             elif server_filename:
1209                 message = "Unable to retrieve source for %s" % (server_filename,)
1210             else:
1211                 message = "Invalid sourceReference %d" % (source_reference,)
1212             response_args.update({"success": False, "message": message})
1213 
1214         response = pydevd_base_schema.build_response(request, kwargs=response_args)
1215         return NetCommand(CMD_RETURN, 0, response, is_json=True)
