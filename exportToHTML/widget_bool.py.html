37   class _Bool(DescriptionWidget, ValueWidget, CoreWidget):
38       """A base class for creating widgets that represent booleans."""
39       value = Bool(False, help="Bool value").tag(sync=True)
40       disabled = Bool(False, help="Enable or disable user changes.").tag(sync=True)
41   
42       def __init__(self, value=None, **kwargs):
43           if value is not None:
44               kwargs['value'] = value
45           super().__init__(**kwargs)
46   
47       _model_name = Unicode('BoolModel').tag(sync=True)

50   @register
51   class Checkbox(_Bool):
52       """Displays a boolean `value` in the form of a checkbox. 
53    
54       Parameters 
55       ---------- 
56       value : {True,False} 
57           value of the checkbox: True-checked, False-unchecked 
58       description : str 
59           description displayed next to the checkbox 
60       indent : {True,False} 
61           indent the control to align with other controls with a description. The style.description_width attribute controls this width for consistence with other controls. 
62       """
63       _view_name = Unicode('CheckboxView').tag(sync=True)
64       _model_name = Unicode('CheckboxModel').tag(sync=True)
65       indent = Bool(True, help="Indent the control to align with other controls with a description.").tag(sync=True)
66       style = InstanceDict(CheckboxStyle, help="Styling customizations").tag(sync=True, **widget_serialization)

70   @register
71   class ToggleButton(_Bool):
72       """Displays a boolean `value` in the form of a toggle button. 
73    
74       Parameters 
75       ---------- 
76       value : {True,False} 
77           value of the toggle button: True-pressed, False-unpressed 
78       description : str 
79           description displayed on the button 
80       icon: str 
81           font-awesome icon name 
82       style: instance of DescriptionStyle 
83           styling customizations 
84       button_style: enum 
85           button predefined styling 
86       """
87       _view_name = Unicode('ToggleButtonView').tag(sync=True)
88       _model_name = Unicode('ToggleButtonModel').tag(sync=True)
89   
90       icon = Unicode('', help= "Font-awesome icon.").tag(sync=True)
91   
92       button_style = CaselessStrEnum(
93           values=['primary', 'success', 'info', 'warning', 'danger', ''], default_value='',
94           help="""Use a predefined styling for the button.""").tag(sync=True)
95       style = InstanceDict(ToggleButtonStyle, help="Styling customizations").tag(sync=True, **widget_serialization)

98   @register
99   class Valid(_Bool):
100      """Displays a boolean `value` in the form of a green check (True / valid) 
101      or a red cross (False / invalid). 
102   
103      Parameters 
104      ---------- 
105      value: {True,False} 
106          value of the Valid widget 
107      """
108      readout = Unicode('Invalid', help="Message displayed when the value is False").tag(sync=True)
109      _view_name = Unicode('ValidView').tag(sync=True)
110      _model_name = Unicode('ValidModel').tag(sync=True)
