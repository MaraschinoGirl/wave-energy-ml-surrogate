11   def test_smacof():
12       # test metric smacof using the data of "Modern Multidimensional Scaling",
13       # Borg & Groenen, p 154
14       sim = np.array([[0, 5, 3, 4], [5, 0, 2, 2], [3, 2, 0, 1], [4, 2, 1, 0]])
15       Z = np.array([[-0.266, -0.539], [0.451, 0.252], [0.016, -0.238], [-0.200, 0.524]])
16       X, _ = mds.smacof(sim, init=Z, n_components=2, max_iter=1, n_init=1)
17       X_true = np.array(
18           [[-1.415, -2.471], [1.633, 1.107], [0.249, -0.067], [-0.468, 1.431]]
19       )
20       assert_array_almost_equal(X, X_true, decimal=3)

50   @pytest.mark.parametrize("k", [0.5, 1.5, 2])
51   def test_normed_stress(k):
52       """Test that non-metric MDS normalized stress is scale-invariant."""
53       sim = np.array([[0, 5, 3, 4], [5, 0, 2, 2], [3, 2, 0, 1], [4, 2, 1, 0]])
54   
55       X1, stress1 = mds.smacof(sim, metric=False, max_iter=5, random_state=0)
56       X2, stress2 = mds.smacof(k * sim, metric=False, max_iter=5, random_state=0)
57   
58       assert_allclose(stress1, stress2, rtol=1e-5)
59       assert_allclose(X1, X2, rtol=1e-5)

73   @pytest.mark.parametrize("metric", [True, False])
74   def test_normalized_stress_auto(metric, monkeypatch):
75       rng = np.random.RandomState(0)
76       X = rng.randn(4, 3)
77       dist = euclidean_distances(X)
78   
79       mock = Mock(side_effect=mds._smacof_single)
80       monkeypatch.setattr("sklearn.manifold._mds._smacof_single", mock)
81   
82       est = mds.MDS(metric=metric, normalized_stress="auto", random_state=rng)
83       est.fit_transform(X)
84       assert mock.call_args[1]["normalized_stress"] != metric
85   
86       mds.smacof(dist, metric=metric, normalized_stress="auto", random_state=rng)
87       assert mock.call_args[1]["normalized_stress"] != metric
