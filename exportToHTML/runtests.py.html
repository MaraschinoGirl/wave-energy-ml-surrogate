62   def test_runner_factory(stderr):
63       class TornadoTextTestRunner(unittest.TextTestRunner):
64           def __init__(self, *args, **kwargs):
65               kwargs["stream"] = stderr
66               super().__init__(*args, **kwargs)
67   
68           def run(self, test):
69               result = super().run(test)
70               if result.skipped:
71                   skip_reasons = set(reason for (test, reason) in result.skipped)
72                   self.stream.write(  # type: ignore
73                       textwrap.fill(
74                           "Some tests were skipped because: %s"
75                           % ", ".join(sorted(skip_reasons))
76                       )
77                   )
78                   self.stream.write("\n")  # type: ignore
79               return result
80   
81       return TornadoTextTestRunner

84   class LogCounter(logging.Filter):
85       """Counts the number of WARNING or higher log records."""
86   
87       def __init__(self, *args, **kwargs):
88           super().__init__(*args, **kwargs)
89           self.info_count = self.warning_count = self.error_count = 0
90   
91       def filter(self, record):
92           if record.levelno >= logging.ERROR:
93               self.error_count += 1
94           elif record.levelno >= logging.WARNING:
95               self.warning_count += 1
96           elif record.levelno >= logging.INFO:
97               self.info_count += 1
98           return True

101  class CountingStderr(io.IOBase):
102      def __init__(self, real):
103          self.real = real
104          self.byte_count = 0
105  
106      def write(self, data):
107          self.byte_count += len(data)
108          return self.real.write(data)
109  
110      def flush(self):
111          return self.real.flush()

114  def main():
115      # Be strict about most warnings (This is set in our test running
116      # scripts to catch import-time warnings, but set it again here to
117      # be sure). This also turns on warnings that are ignored by
118      # default, including DeprecationWarnings and python 3.2's
119      # ResourceWarnings.
120      warnings.filterwarnings("error")
121      # setuptools sometimes gives ImportWarnings about things that are on
122      # sys.path even if they're not being used.
123      warnings.filterwarnings("ignore", category=ImportWarning)
124      # Tornado generally shouldn't use anything deprecated, but some of
125      # our dependencies do (last match wins).
126      warnings.filterwarnings("ignore", category=DeprecationWarning)
127      warnings.filterwarnings("error", category=DeprecationWarning, module=r"tornado\..*")
128      warnings.filterwarnings("ignore", category=PendingDeprecationWarning)
129      warnings.filterwarnings(
130          "error", category=PendingDeprecationWarning, module=r"tornado\..*"
131      )
132  
133      logging.getLogger("tornado.access").setLevel(logging.CRITICAL)
134  
135      define(
136          "httpclient",
137          type=str,
138          default=None,
139          callback=lambda s: AsyncHTTPClient.configure(
140              s, defaults=dict(allow_ipv6=False)
141          ),
142      )
143      define("httpserver", type=str, default=None, callback=HTTPServer.configure)
144      define("resolver", type=str, default=None, callback=Resolver.configure)
145      define(
146          "debug_gc",
147          type=str,
148          multiple=True,
149          help="A comma-separated list of gc module debug constants, "
150          "e.g. DEBUG_STATS or DEBUG_COLLECTABLE,DEBUG_OBJECTS",
151          callback=lambda values: gc.set_debug(
152              reduce(operator.or_, (getattr(gc, v) for v in values))
153          ),
154      )
155      define(
156          "fail-if-logs",
157          default=True,
158          help="If true, fail the tests if any log output is produced (unless captured by ExpectLog)",
159      )
160  
161      def set_locale(x):
162          locale.setlocale(locale.LC_ALL, x)
163  
164      define("locale", type=str, default=None, callback=set_locale)
165  
166      log_counter = LogCounter()
167      add_parse_callback(lambda: logging.getLogger().handlers[0].addFilter(log_counter))
168  
169      # Certain errors (especially "unclosed resource" errors raised in
170      # destructors) go directly to stderr instead of logging. Count
171      # anything written by anything but the test runner as an error.
172      orig_stderr = sys.stderr
173      counting_stderr = CountingStderr(orig_stderr)
174      sys.stderr = counting_stderr  # type: ignore
175  
176      import tornado.testing
177  
178      kwargs = {}
179  
180      # HACK:  unittest.main will make its own changes to the warning
181      # configuration, which may conflict with the settings above
182      # or command-line flags like -bb.  Passing warnings=False
183      # suppresses this behavior, although this looks like an implementation
184      # detail.  http://bugs.python.org/issue15626
185      kwargs["warnings"] = False
186  
187      kwargs["testRunner"] = test_runner_factory(orig_stderr)
188      try:
189          tornado.testing.main(**kwargs)
190      finally:
191          # The tests should run clean; consider it a failure if they
192          # logged anything at info level or above.
193          if (
194              log_counter.info_count > 0
195              or log_counter.warning_count > 0
196              or log_counter.error_count > 0
197              or counting_stderr.byte_count > 0
198          ):
199              logging.error(
200                  "logged %d infos, %d warnings, %d errors, and %d bytes to stderr",
201                  log_counter.info_count,
202                  log_counter.warning_count,
203                  log_counter.error_count,
204                  counting_stderr.byte_count,
205              )
206              if options.fail_if_logs:
207                  sys.exit(1)
208  
209  
210  if __name__ == "__main__":
211      main()
