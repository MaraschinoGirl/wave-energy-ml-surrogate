15   """A tool to parse and pretty-print JSON5. 
16    
17   Usage: 
18    
19       $ echo '{foo:"bar"}' | python -m json5 
20       { 
21           foo: 'bar', 
22       } 
23       $ echo '{foo:"bar"}' | python -m json5 --as-json 
24       { 
25           "foo": "bar" 
26       } 
27   """
28   
29   import argparse
30   import sys
31   
32   import json5
33   from json5.host import Host
34   from json5.version import __version__
35   
36   QUOTE_STYLES = {q.value: q for q in json5.QuoteStyle}
37   
38   
39   def main(argv=None, host=None):
40       host = host or Host()
41   
42       args = _parse_args(host, argv)
43   
44       if args.version:
45           host.print(__version__)
46           return 0
47   
48       if args.cmd:
49           inp = args.cmd
50       elif args.file == '-':
51           inp = host.stdin.read()
52       else:
53           inp = host.read_text_file(args.file)
54   
55       if args.indent == 'None':
56           args.indent = None
57       else:
58           try:
59               args.indent = int(args.indent)
60           except ValueError:
61               pass
62   
63       if args.as_json:
64           args.quote_keys = True
65           args.trailing_commas = False
66           args.quote_style = json5.QuoteStyle.ALWAYS_DOUBLE.value
67   
68       obj = json5.loads(inp, strict=args.strict)
69       s = json5.dumps(
70           obj,
71           indent=args.indent,
72           quote_keys=args.quote_keys,
73           trailing_commas=args.trailing_commas,
74           quote_style=QUOTE_STYLES[args.quote_style],
75       )
76       host.print(s)
77       return 0
78   
79   
80   class _HostedArgumentParser(argparse.ArgumentParser):
81       """An argument parser that plays nicely w/ host objects."""
82   
83       def __init__(self, host, **kwargs):
84           self.host = host
85           super().__init__(**kwargs)
86   
87       def exit(self, status=0, message=None):
88           if message:
89               self._print_message(message, self.host.stderr)
90           sys.exit(status)
91   
92       def error(self, message):
93           self.host.print(f'usage: {self.usage}', end='', file=self.host.stderr)
94           self.host.print('    -h/--help for help\n', file=self.host.stderr)
95           self.exit(2, f'error: {message}\n')
96   
97       def print_help(self, file=None):
98           self.host.print(self.format_help(), file=file)
99   
100  
101  def _parse_args(host, argv):
102      usage = 'json5 [options] [FILE]\n'
103  
104      parser = _HostedArgumentParser(
105          host,
106          prog='json5',
107          usage=usage,
108          description=__doc__,
109          formatter_class=argparse.RawDescriptionHelpFormatter,
110      )
111      parser.add_argument(
112          '-V',
113          '--version',
114          action='store_true',
115          help=f'show JSON5 library version ({__version__})',
116      )
117      parser.add_argument(
118          '-c',
119          metavar='STR',
120          dest='cmd',
121          help='inline json5 string to read instead of reading from a file',
122      )
123      parser.add_argument(
124          '--as-json',
125          dest='as_json',
126          action='store_const',
127          const=True,
128          default=False,
129          help='output as JSON (same as --quote-keys --no-trailing-commas)',
130      )
131      parser.add_argument(
132          '--indent',
133          dest='indent',
134          default=4,
135          help='amount to indent each line (default is 4 spaces)',
136      )
137      parser.add_argument(
138          '--quote-keys',
139          action='store_true',
140          default=False,
141          help='quote all object keys',
142      )
143      parser.add_argument(
144          '--no-quote-keys',
145          action='store_false',
146          dest='quote_keys',
147          help="don't quote object keys that are identifiers"
148          ' (this is the default)',
149      )
150      parser.add_argument(
151          '--trailing-commas',
152          action='store_true',
153          default=True,
154          help='add commas after the last item in multi-line '
155          'objects and arrays (this is the default)',
156      )
157      parser.add_argument(
158          '--no-trailing-commas',
159          dest='trailing_commas',
160          action='store_false',
161          help='do not add commas after the last item in '
162          'multi-line lists and objects',
163      )
164      parser.add_argument(
165          '--strict',
166          action='store_true',
167          default=True,
168          help='Do not allow control characters (\\x00-\\x1f) in strings '
169          '(default)',
170      )
171      parser.add_argument(
172          '--no-strict',
173          dest='strict',
174          action='store_false',
175          help='Allow control characters (\\x00-\\x1f) in strings',
176      )
177      parser.add_argument(
178          '--quote-style',
179          action='store',
180          default='always_double',
181          choices=QUOTE_STYLES.keys(),
182          help='Controls how strings are encoded. By default they are always '
183          'double-quoted ("always_double")',
184      )
185      parser.add_argument(
186          'file',
187          metavar='FILE',
188          nargs='?',
189          default='-',
190          help='optional file to read JSON5 document from; if '
191          'not specified or "-", will read from stdin '
192          'instead',
193      )
194      return parser.parse_args(argv)
195  
196  
197  if __name__ == '__main__':  # pragma: no cover
198      sys.exit(main())
