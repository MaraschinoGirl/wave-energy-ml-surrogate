17   class Quantizer:
18       def __init__(self, output_dtype="int8"):
19           self.output_dtype = output_dtype
20   
21       def __call__(self, x):
22           """Compute a quantized output from an input tensor."""
23           return x

27           """Creates a quantizer from its config. 
28    
29           This method is the reverse of `get_config`, 
30           capable of instantiating the same quantizer from the config 
31           dictionary. 
32    
33           This method is used by Keras `model_to_estimator`, saving and 
34           loading models to HDF5 formats, Keras model cloning, some visualization 
35           utilities, and exporting models to and from JSON. 
36    
37           Args: 
38               config: A Python dictionary, typically the output of get_config. 
39    
40           Returns: 
41               A quantizer instance. 
42           """
43           return cls(**config)

45       def get_config(self):
46           """Returns the config of the quantizer. 
47    
48           A quantizer config is a Python dictionary (serializable) 
49           containing all configuration parameters of the quantizer. 
50           The same quantizer can be reinstantiated later 
51           (without any saved state) from this configuration. 
52

63           raise NotImplementedError(f"{self} does not implement get_config()")

66   @keras_export("keras.quantizers.abs_max_quantize")
67   def abs_max_quantize(
68       inputs,
69       axis,
70       value_range=(-127, 127),
71       dtype="int8",
72       epsilon=backend.epsilon(),
73       to_numpy=False,
74   ):
75       if to_numpy:
76           # Save memory on the device using numpy
77           original_dtype = backend.standardize_dtype(inputs.dtype)
78           inputs = ops.convert_to_numpy(inputs)
79           axis = standardize_axis_for_numpy(axis)
80           scale = np.divide(
81               value_range[1],
82               np.add(np.max(np.abs(inputs), axis=axis, keepdims=True), epsilon),
83           )
84           outputs = np.multiply(inputs, scale)
85           outputs = np.clip(np.round(outputs), value_range[0], value_range[1])
86           outputs = outputs.astype(dtype)
87           return ops.convert_to_tensor(outputs), ops.convert_to_tensor(
88               scale, dtype=original_dtype
89           )
90   
91       inputs = ops.convert_to_tensor(inputs)
92       scale = ops.divide(
93           value_range[1],
94           ops.add(ops.max(ops.abs(inputs), axis=axis, keepdims=True), epsilon),
95       )
96       scale = ops.cast(scale, backend.standardize_dtype(inputs.dtype))
97       outputs = ops.multiply(inputs, scale)
98       outputs = ops.clip(ops.round(outputs), value_range[0], value_range[1])
99       outputs = ops.cast(outputs, dtype)
100      return outputs, scale

104  class AbsMaxQuantizer(Quantizer):
105      def __init__(
106          self,
107          axis,
108          value_range=(-127, 127),
109          epsilon=backend.epsilon(),
110          output_dtype="int8",
111      ):
112          Quantizer.__init__(self, output_dtype=output_dtype)
113          if isinstance(axis, int):
114              axis = (axis,)
115          self.axis = tuple(axis)
116          self.value_range = value_range
117          self.epsilon = epsilon
118  
119      def __call__(self, x):
120          quantized_x, scale = abs_max_quantize(
121              x, self.axis, self.value_range, self.output_dtype, self.epsilon
122          )
123          return quantized_x, scale
124  
125      def get_config(self):
126          return {
127              "axis": self.axis,
128              "value_range": self.value_range,
129              "epsilon": self.epsilon,
130              "output_dtype": self.output_dtype,
131          }

134  def adjust_and_nudge(min_range, max_range, num_bits, narrow_range):
135      """Adjusts and nudges the quantization range for better accuracy."""
136      # Use higher precision for the computation.
137      compute_dtype = backend.result_type(min_range.dtype, "float32")
138      min_range = ops.cast(min_range, compute_dtype)
139      max_range = ops.cast(max_range, compute_dtype)
140  
141      quant_max = (1 << num_bits) - 1
142      quant_min = 0 if not narrow_range else 1
143      diff_range = ops.subtract(max_range, min_range)
144  
145      # Calculate the scale and ensure it's positive
146      scale = ops.divide(diff_range, quant_max - quant_min)
147  
148      # Re-calculate the inverse to avoid loss of precision
149      inv_scale = ops.divide(quant_max - quant_min, diff_range)
150  
151      # Calculate the zero point from the min range
152      zero_point_from_min = quant_min - ops.divide(min_range, scale)
153  
154      # Ensure zero point is within valid range [0, quant_max]
155      zero_point = ops.clip(zero_point_from_min, quant_min, quant_max)
156  
157      # Nudge zero point if it's very close to an integer
158      nudged_zero_point = ops.round(zero_point)
159  
160      # Calculate nudged limits
161      nudged_min = ops.multiply(ops.subtract(quant_min, nudged_zero_point), scale)
162      nudged_max = ops.multiply(ops.subtract(quant_max, nudged_zero_point), scale)
163  
164      return nudged_min, nudged_max, scale, inv_scale

171          self.narrow_range = narrow_range
172          self.axis = axis

188  @keras_export("keras.quantizers.fake_quant_with_min_max_vars")
189  def fake_quant_with_min_max_vars(
190      inputs,
191      min_vals,
192      max_vals,
193      num_bits=8,
194      narrow_range=False,
195      axis=None,
196  ):
197      """Perform per-tensor or per-channel fake quantization. 
198   
199      `[min_vals, max_vals]` define the clamping range for the `inputs`. 
200   
201      The `inputs` are quantized into the quantization range: 
202      - `[0, 2^num_bits - 1]` when `narrow_range=False` 
203      - `[1, 2^num_bits - 1]` when `narrow_range=True` 
204   
205      After quantization, the values are dequantized and output as floats within 
206      the `[min_vals, max_vals]` interval. 
207   
208      This operation supports gradient computation, allowing `min_vals` and 
209      `max_vals` to be trained. 
210   
211      Args: 
212          inputs: Input Keras tensor of float dtype. 
213          min_vals: A global minimum scalar or a per-channel minimum tensor. 
214          max_vals: A global maximum scalar or a per-channel maximum tensor. 
215          num_bits: Quantization bit width (e.g., `8` for int8). Defaults to `8`. 
216          narrow_range: Whether to use narrow quantization range. Defaults to 
217              `False`. 
218          axis: Axis along which to perform per-channel quantization. If `None`, 
219                per-tensor quantization is performed. Defaults to `None`. 
220   
221   
222      Returns: 
223          Tensor: A Keras tensor with fake quantization applied. 
224      """
225      if any_symbolic_tensors((inputs,)):
226          return FakeQuantWithMinMaxVars().symbolic_call(
227              inputs, min_vals, max_vals
228          )
229  
230      inputs = ops.convert_to_tensor(inputs)
231      min_vals = ops.convert_to_tensor(min_vals)
232      max_vals = ops.convert_to_tensor(max_vals)
233      num_bits = int(num_bits)
234  
235      if axis is not None:
236          axis = canonicalize_axis(axis, inputs.ndim)
237  
238      # Shortcut for TensorFlow backend by using `tf.quantization.fake_quant_*`
239      # apis. This is necessary to be recognizable for the TFLite converter.
240      if backend.backend() == "tensorflow":
241          import tensorflow as tf
242  
243          # `tf.quantization.fake_quant_*` only supports float32.
244          dtype = backend.standardize_dtype(inputs.dtype)
245          if axis is None:
246              outputs = tf.quantization.fake_quant_with_min_max_vars(
247                  ops.cast(inputs, "float32"),
248                  ops.cast(ops.reshape(min_vals, ()), "float32"),
249                  ops.cast(ops.reshape(max_vals, ()), "float32"),
250                  num_bits=num_bits,
251                  narrow_range=narrow_range,
252              )
253              return ops.cast(outputs, dtype=dtype)
254          else:
255              # `tf.quantization.fake_quant_with_min_max_vars_per_channel` only
256              # supports the last channel for the per-channel quantization. We
257              # use `ops.swapaxes` for the pre- and post-processing.
258              last_axis = inputs.ndim - 1
259              inputs = ops.swapaxes(inputs, axis, last_axis)
260              outputs = tf.quantization.fake_quant_with_min_max_vars_per_channel(
261                  ops.cast(inputs, "float32"),
262                  ops.cast(min_vals, "float32"),
263                  ops.cast(max_vals, "float32"),
264                  num_bits=num_bits,
265                  narrow_range=narrow_range,
266              )
267              outputs = ops.cast(outputs, dtype=dtype)
268              return ops.swapaxes(outputs, last_axis, axis)
269  
270      @ops.custom_gradient
271      def _fake_quant_with_min_max_vars_per_channel(x, min_val, max_val):
272          dtype = backend.standardize_dtype(x.dtype)
273  
274          # Calculate quantization parameters for all channels at once
275          nudged_min, nudged_max, scale, inv_scale = adjust_and_nudge(
276              min_val, max_val, num_bits, narrow_range
277          )
278  
279          quant_zero = ops.floor(
280              ops.add(ops.multiply(-nudged_min, inv_scale), 0.5)
281          )
282          x_clamped = ops.clip(
283              x, ops.cast(nudged_min, x.dtype), ops.cast(nudged_max, x.dtype)
284          )
285          x_clamped_shifted = ops.subtract(x_clamped, nudged_min)
286          result = ops.multiply(
287              ops.floor(
288                  ops.add(
289                      ops.subtract(
290                          ops.multiply(x_clamped_shifted, inv_scale), quant_zero
291                      ),
292                      0.5,
293                  )
294              ),
295              scale,
296          )
297          result = ops.cast(result, dtype=dtype)
298  
299          # Create gradient mask for all channels
300          masks = ops.logical_and(
301              ops.greater_equal(x, nudged_min), ops.less_equal(x, nudged_max)
302          )
303  
304          def grad(*args, upstream=None):
305              if upstream is None:
306                  (upstream,) = args
307  
308              # Gradient for x
309              dx = ops.where(masks, upstream, 0.0)
310              axes = [i for i in range(len(dx.shape)) if i != axis]
311  
312              # Gradient for min_val
313              # When x is clipped to min, the gradient flows to min_val
314              min_mask = ops.less_equal(x, nudged_min)
315              grad_min = ops.where(min_mask, upstream, 0.0)
316              if axis is not None:
317                  grad_min = ops.sum(grad_min, axis=axes)
318              else:
319                  grad_min = ops.sum(grad_min)
320  
321              # Gradient for max_val
322              # When x is clipped to max, the gradient flows to max_val
323              max_mask = ops.greater_equal(x, nudged_max)
324              grad_max = ops.where(max_mask, upstream, 0.0)
325              if axis is not None:
326                  grad_max = ops.sum(grad_max, axis=axes)
327              else:
328                  grad_max = ops.sum(grad_max)
329  
330              return dx, grad_min, grad_max
331  
332          return result, grad
333  
334      return _fake_quant_with_min_max_vars_per_channel(inputs, min_vals, max_vals)

340  @keras_export("keras.quantizers.compute_float8_scale")
341  def compute_float8_scale(amax, scale, dtype_max, margin=0):
342      # The algorithm for computing the new scale is sourced from
343      # https://docs.nvidia.com/deeplearning/transformer-engine/user-guide/api/jax.html#transformer_engine.jax.update_fp8_metas
344      # wherein the `original_scale` corresponds to the reciprocal of the
345      # `scale` passed in this function.
346      scale = ops.reciprocal(scale)
347      sf = ops.divide(ops.divide(dtype_max, amax), 2**margin)
348      sf = ops.where(amax > 0.0, sf, scale)
349      sf = ops.where(ops.isfinite(amax), sf, scale)
350      return ops.reciprocal(sf)

353  @keras_export("keras.quantizers.compute_float8_amax_history")
354  def compute_float8_amax_history(x, amax_history):
355      amax_update = ops.cast(ops.max(ops.abs(x)), amax_history.dtype)
356      new_amax_history = ops.scatter_update(
357          ops.roll(amax_history, shift=-1),
358          [[0]],
359          ops.reshape(amax_update, [1]),
360      )
361      return new_amax_history

364  @keras_export("keras.quantizers.quantize_and_dequantize")
365  def quantize_and_dequantize(inputs, scale, quantized_dtype, compute_dtype):
366      # Quantize
367      quantized_dtype_max = ops.cast(
368          float(ml_dtypes.finfo(quantized_dtype).max), compute_dtype
369      )
370      x = ops.divide(inputs, ops.cast(scale, compute_dtype))
371      x = ops.clip(x, -quantized_dtype_max, quantized_dtype_max)
372      x = ops.cast(x, quantized_dtype)
373  
374      # Dequantize
375      x = ops.multiply(ops.cast(x, compute_dtype), ops.cast(scale, compute_dtype))
376      return x
