23   class _SubsampleMetaSplitter:
24       """Splitter that subsamples a given fraction of the dataset"""
25   
26       def __init__(self, *, base_cv, fraction, subsample_test, random_state):
27           self.base_cv = base_cv
28           self.fraction = fraction
29           self.subsample_test = subsample_test
30           self.random_state = random_state
31   
32       def split(self, X, y, **kwargs):
33           for train_idx, test_idx in self.base_cv.split(X, y, **kwargs):
34               train_idx = resample(
35                   train_idx,
36                   replace=False,
37                   random_state=self.random_state,
38                   n_samples=int(self.fraction * len(train_idx)),
39               )
40               if self.subsample_test:
41                   test_idx = resample(
42                       test_idx,
43                       replace=False,
44                       random_state=self.random_state,
45                       n_samples=int(self.fraction * len(test_idx)),
46                   )
47               yield train_idx, test_idx
48

50   def _top_k(results, k, itr):
51       # Return the best candidates of a given iteration
52       iteration, mean_test_score, params = (
53           np.asarray(a)
54           for a in (results["iter"], results["mean_test_score"], results["params"])
55       )
56       iter_indices = np.flatnonzero(iteration == itr)
57       scores = mean_test_score[iter_indices]
58       # argsort() places NaNs at the end of the array so we move NaNs to the
59       # front of the array so the last `k` items are the those with the
60       # highest scores.
61       sorted_indices = np.roll(np.argsort(scores), np.count_nonzero(np.isnan(scores)))
62       return np.array(params[iter_indices][sorted_indices[-k:]])

65   class BaseSuccessiveHalving(BaseSearchCV):
66       """Implements successive halving. 
67    
68       Ref: 
69       Almost optimal exploration in multi-armed bandits, ICML 13 
70       Zohar Karnin, Tomer Koren, Oren Somekh 
71       """
72   
73       _parameter_constraints: dict = {
74           **BaseSearchCV._parameter_constraints,
75           # overwrite `scoring` since multi-metrics are not supported
76           "scoring": [StrOptions(set(get_scorer_names())), callable, None],
77           "random_state": ["random_state"],
78           "max_resources": [
79               Interval(Integral, 0, None, closed="neither"),
80               StrOptions({"auto"}),
81           ],
82           "min_resources": [
83               Interval(Integral, 0, None, closed="neither"),
84               StrOptions({"exhaust", "smallest"}),
85           ],
86           "resource": [str],
87           "factor": [Interval(Real, 0, None, closed="neither")],
88           "aggressive_elimination": ["boolean"],
89       }
90       _parameter_constraints.pop("pre_dispatch")  # not used in this class
91   
92       def __init__(
93           self,
94           estimator,
95           *,
96           scoring=None,
97           n_jobs=None,
98           refit=True,
99           cv=5,
100          verbose=0,
101          random_state=None,
102          error_score=np.nan,
103          return_train_score=True,
104          max_resources="auto",
105          min_resources="exhaust",
106          resource="n_samples",
107          factor=3,
108          aggressive_elimination=False,
109      ):
110          super().__init__(
111              estimator,
112              scoring=scoring,
113              n_jobs=n_jobs,
114              refit=refit,
115              cv=cv,
116              verbose=verbose,
117              error_score=error_score,
118              return_train_score=return_train_score,
119          )
120  
121          self.random_state = random_state
122          self.max_resources = max_resources
123          self.resource = resource
124          self.factor = factor
125          self.min_resources = min_resources
126          self.aggressive_elimination = aggressive_elimination

128      def _check_input_parameters(self, X, y, split_params):
129          # We need to enforce that successive calls to cv.split() yield the same
130          # splits: see https://github.com/scikit-learn/scikit-learn/issues/15149
131          if not _yields_constant_splits(self._checked_cv_orig):
132              raise ValueError(
133                  "The cv parameter must yield consistent folds across "
134                  "calls to split(). Set its random_state to an int, or set "
135                  "shuffle=False."
136              )
137  
138          if (
139              self.resource != "n_samples"
140              and self.resource not in self.estimator.get_params()
141          ):
142              raise ValueError(
143                  f"Cannot use resource={self.resource} which is not supported "
144                  f"by estimator {self.estimator.__class__.__name__}"
145              )
146  
147          if isinstance(self, HalvingRandomSearchCV):
148              if self.min_resources == self.n_candidates == "exhaust":
149                  # for n_candidates=exhaust to work, we need to know what
150                  # min_resources is. Similarly min_resources=exhaust needs to
151                  # know the actual number of candidates.
152                  raise ValueError(
153                      "n_candidates and min_resources cannot be both set to 'exhaust'."
154                  )
155  
156          self.min_resources_ = self.min_resources
157          if self.min_resources_ in ("smallest", "exhaust"):
158              if self.resource == "n_samples":
159                  n_splits = self._checked_cv_orig.get_n_splits(X, y, **split_params)
160                  # please see https://gph.is/1KjihQe for a justification
161                  magic_factor = 2
162                  self.min_resources_ = n_splits * magic_factor
163                  if is_classifier(self.estimator):
164                      y = validate_data(self, X="no_validation", y=y)
165                      check_classification_targets(y)
166                      n_classes = np.unique(y).shape[0]
167                      self.min_resources_ *= n_classes
168              else:
169                  self.min_resources_ = 1
170              # if 'exhaust', min_resources_ might be set to a higher value later
171              # in _run_search
172  
173          self.max_resources_ = self.max_resources
174          if self.max_resources_ == "auto":
175              if not self.resource == "n_samples":
176                  raise ValueError(
177                      "resource can only be 'n_samples' when max_resources='auto'"
178                  )
179              self.max_resources_ = _num_samples(X)
180  
181          if self.min_resources_ > self.max_resources_:
182              raise ValueError(
183                  f"min_resources_={self.min_resources_} is greater "
184                  f"than max_resources_={self.max_resources_}."
185              )
186  
187          if self.min_resources_ == 0:
188              raise ValueError(
189                  f"min_resources_={self.min_resources_}: you might have passed "
190                  "an empty dataset X."
191              )

193      @staticmethod
194      def _select_best_index(refit, refit_metric, results):
195          """Custom refit callable to return the index of the best candidate. 
196   
197          We want the best candidate out of the last iteration. By default 
198          BaseSearchCV would return the best candidate out of all iterations. 
199   
200          Currently, we only support for a single metric thus `refit` and 
201          `refit_metric` are not required. 
202          """
203          last_iter = np.max(results["iter"])
204          last_iter_indices = np.flatnonzero(results["iter"] == last_iter)
205  
206          test_scores = results["mean_test_score"][last_iter_indices]
207          # If all scores are NaNs there is no way to pick between them,
208          # so we (arbitrarily) declare the zero'th entry the best one
209          if np.isnan(test_scores).all():
210              best_idx = 0
211          else:
212              best_idx = np.nanargmax(test_scores)
213  
214          return last_iter_indices[best_idx]

216      @_fit_context(
217          # Halving*SearchCV.estimator is not validated yet
218          prefer_skip_nested_validation=False
219      )
220      def fit(self, X, y=None, **params):
221          """Run fit with all sets of parameters. 
222   
223          Parameters 
224          ---------- 
225   
226          X : array-like, shape (n_samples, n_features) 
227              Training vector, where `n_samples` is the number of samples and 
228              `n_features` is the number of features. 
229   
230          y : array-like, shape (n_samples,) or (n_samples, n_output), optional 
231              Target relative to X for classification or regression; 
232              None for unsupervised learning. 
233   
234          **params : dict of string -> object 
235              Parameters passed to the ``fit`` method of the estimator. 
236   
237          Returns 
238          ------- 
239          self : object 
240              Instance of fitted estimator. 
241          """
242          self._checked_cv_orig = check_cv(
243              self.cv, y, classifier=is_classifier(self.estimator)
244          )
245  
246          routed_params = self._get_routed_params_for_fit(params)
247          self._check_input_parameters(
248              X=X, y=y, split_params=routed_params.splitter.split
249          )
250  
251          self._n_samples_orig = _num_samples(X)
252  
253          super().fit(X, y=y, **params)
254  
255          # Set best_score_: BaseSearchCV does not set it, as refit is a callable
256          self.best_score_ = self.cv_results_["mean_test_score"][self.best_index_]
257  
258          return self

260      def _run_search(self, evaluate_candidates):
261          candidate_params = self._generate_candidate_params()
262  
263          if self.resource != "n_samples" and any(
264              self.resource in candidate for candidate in candidate_params
265          ):
266              # Can only check this now since we need the candidates list
267              raise ValueError(
268                  f"Cannot use parameter {self.resource} as the resource since "
269                  "it is part of the searched parameters."
270              )
271  
272          # n_required_iterations is the number of iterations needed so that the
273          # last iterations evaluates less than `factor` candidates.
274          n_required_iterations = 1 + floor(log(len(candidate_params), self.factor))
275  
276          if self.min_resources == "exhaust":
277              # To exhaust the resources, we want to start with the biggest
278              # min_resources possible so that the last (required) iteration
279              # uses as many resources as possible
280              last_iteration = n_required_iterations - 1
281              self.min_resources_ = max(
282                  self.min_resources_,
283                  self.max_resources_ // self.factor**last_iteration,
284              )
285  
286          # n_possible_iterations is the number of iterations that we can
287          # actually do starting from min_resources and without exceeding
288          # max_resources. Depending on max_resources and the number of
289          # candidates, this may be higher or smaller than
290          # n_required_iterations.
291          n_possible_iterations = 1 + floor(
292              log(self.max_resources_ // self.min_resources_, self.factor)
293          )
294  
295          if self.aggressive_elimination:
296              n_iterations = n_required_iterations
297          else:
298              n_iterations = min(n_possible_iterations, n_required_iterations)
299  
300          if self.verbose:
301              print(f"n_iterations: {n_iterations}")
302              print(f"n_required_iterations: {n_required_iterations}")
303              print(f"n_possible_iterations: {n_possible_iterations}")
304              print(f"min_resources_: {self.min_resources_}")
305              print(f"max_resources_: {self.max_resources_}")
306              print(f"aggressive_elimination: {self.aggressive_elimination}")
307              print(f"factor: {self.factor}")
308  
309          self.n_resources_ = []
310          self.n_candidates_ = []
311  
312          for itr in range(n_iterations):
313              power = itr  # default
314              if self.aggressive_elimination:
315                  # this will set n_resources to the initial value (i.e. the
316                  # value of n_resources at the first iteration) for as many
317                  # iterations as needed (while candidates are being
318                  # eliminated), and then go on as usual.
319                  power = max(0, itr - n_required_iterations + n_possible_iterations)
320  
321              n_resources = int(self.factor**power * self.min_resources_)
322              # guard, probably not needed
323              n_resources = min(n_resources, self.max_resources_)
324              self.n_resources_.append(n_resources)
325  
326              n_candidates = len(candidate_params)
327              self.n_candidates_.append(n_candidates)
328  
329              if self.verbose:
330                  print("-" * 10)
331                  print(f"iter: {itr}")
332                  print(f"n_candidates: {n_candidates}")
333                  print(f"n_resources: {n_resources}")
334  
335              if self.resource == "n_samples":
336                  # subsampling will be done in cv.split()
337                  cv = _SubsampleMetaSplitter(
338                      base_cv=self._checked_cv_orig,
339                      fraction=n_resources / self._n_samples_orig,
340                      subsample_test=True,
341                      random_state=self.random_state,
342                  )
343  
344              else:
345                  # Need copy so that the n_resources of next iteration does
346                  # not overwrite
347                  candidate_params = [c.copy() for c in candidate_params]
348                  for candidate in candidate_params:
349                      candidate[self.resource] = n_resources
350                  cv = self._checked_cv_orig
351  
352              more_results = {
353                  "iter": [itr] * n_candidates,
354                  "n_resources": [n_resources] * n_candidates,
355              }
356  
357              results = evaluate_candidates(
358                  candidate_params, cv, more_results=more_results
359              )
360  
361              n_candidates_to_keep = ceil(n_candidates / self.factor)
362              candidate_params = _top_k(results, n_candidates_to_keep, itr)
363  
364          self.n_remaining_candidates_ = len(candidate_params)
365          self.n_required_iterations_ = n_required_iterations
366          self.n_possible_iterations_ = n_possible_iterations
367          self.n_iterations_ = n_iterations

374  class HalvingGridSearchCV(BaseSuccessiveHalving):
375      """Search over specified parameter values with successive halving. 
376   
377      The search strategy starts evaluating all the candidates with a small 
378      amount of resources and iteratively selects the best candidates, using 
379      more and more resources. 
380   
381      Read more in the :ref:`User guide <successive_halving_user_guide>`. 
382   
383      .. note:: 
384   
385        This estimator is still **experimental** for now: the predictions 
386        and the API might change without any deprecation cycle. To use it, 
387        you need to explicitly import ``enable_halving_search_cv``:: 
388   
389          >>> # explicitly require this experimental feature 
390          >>> from sklearn.experimental import enable_halving_search_cv # noqa 
391          >>> # now you can import normally from model_selection 
392          >>> from sklearn.model_selection import HalvingGridSearchCV 
393   
394      Parameters 
395      ---------- 
396      estimator : estimator object 
397          This is assumed to implement the scikit-learn estimator interface. 
398          Either estimator needs to provide a ``score`` function, 
399          or ``scoring`` must be passed. 
400   
401      param_grid : dict or list of dictionaries 
402          Dictionary with parameters names (string) as keys and lists of 
403          parameter settings to try as values, or a list of such 
404          dictionaries, in which case the grids spanned by each dictionary 
405          in the list are explored. This enables searching over any sequence 
406          of parameter settings. 
407   
408      factor : int or float, default=3 
409          The 'halving' parameter, which determines the proportion of candidates 
410          that are selected for each subsequent iteration. For example, 
411          ``factor=3`` means that only one third of the candidates are selected. 
412   
413      resource : ``'n_samples'`` or str, default='n_samples' 
414          Defines the resource that increases with each iteration. By default, 
415          the resource is the number of samples. It can also be set to any 
416          parameter of the base estimator that accepts positive integer 
417          values, e.g. 'n_iterations' or 'n_estimators' for a gradient 
418          boosting estimator. In this case ``max_resources`` cannot be 'auto' 
419          and must be set explicitly. 
420   
421      max_resources : int, default='auto' 
422          The maximum amount of resource that any candidate is allowed to use 
423          for a given iteration. By default, this is set to ``n_samples`` when 
424          ``resource='n_samples'`` (default), else an error is raised. 
425   
426      min_resources : {'exhaust', 'smallest'} or int, default='exhaust' 
427          The minimum amount of resource that any candidate is allowed to use 
428          for a given iteration. Equivalently, this defines the amount of 
429          resources `r0` that are allocated for each candidate at the first 
430          iteration. 
431   
432          - 'smallest' is a heuristic that sets `r0` to a small value: 
433   
434            - ``n_splits * 2`` when ``resource='n_samples'`` for a regression problem 
435            - ``n_classes * n_splits * 2`` when ``resource='n_samples'`` for a 
436              classification problem 
437            - ``1`` when ``resource != 'n_samples'`` 
438   
439          - 'exhaust' will set `r0` such that the **last** iteration uses as 
440            much resources as possible. Namely, the last iteration will use the 
441            highest value smaller than ``max_resources`` that is a multiple of 
442            both ``min_resources`` and ``factor``. In general, using 'exhaust' 
443            leads to a more accurate estimator, but is slightly more time 
444            consuming. 
445   
446          Note that the amount of resources used at each iteration is always a 
447          multiple of ``min_resources``. 
448   
449      aggressive_elimination : bool, default=False 
450          This is only relevant in cases where there isn't enough resources to 
451          reduce the remaining candidates to at most `factor` after the last 
452          iteration. If ``True``, then the search process will 'replay' the 
453          first iteration for as long as needed until the number of candidates 
454          is small enough. This is ``False`` by default, which means that the 
455          last iteration may evaluate more than ``factor`` candidates. See 
456          :ref:`aggressive_elimination` for more details. 
457   
458      cv : int, cross-validation generator or iterable, default=5 
459          Determines the cross-validation splitting strategy. 
460          Possible inputs for cv are: 
461   
462          - integer, to specify the number of folds in a `(Stratified)KFold`, 
463          - :term:`CV splitter`, 
464          - An iterable yielding (train, test) splits as arrays of indices. 
465   
466          For integer/None inputs, if the estimator is a classifier and ``y`` is 
467          either binary or multiclass, :class:`StratifiedKFold` is used. In all 
468          other cases, :class:`KFold` is used. These splitters are instantiated 
469          with `shuffle=False` so the splits will be the same across calls. 
470   
471          Refer :ref:`User Guide <cross_validation>` for the various 
472          cross-validation strategies that can be used here. 
473   
474          .. note:: 
475              Due to implementation details, the folds produced by `cv` must be 
476              the same across multiple calls to `cv.split()`. For 
477              built-in `scikit-learn` iterators, this can be achieved by 
478              deactivating shuffling (`shuffle=False`), or by setting the 
479              `cv`'s `random_state` parameter to an integer. 
480   
481      scoring : str, callable, or None, default=None 
482          A single string (see :ref:`scoring_parameter`) or a callable 
483          (see :ref:`scoring_callable`) to evaluate the predictions on the test set. 
484          If None, the estimator's score method is used. 
485   
486      refit : bool, default=True 
487          If True, refit an estimator using the best found parameters on the 
488          whole dataset. 
489   
490          The refitted estimator is made available at the ``best_estimator_`` 
491          attribute and permits using ``predict`` directly on this 
492          ``HalvingGridSearchCV`` instance. 
493   
494      error_score : 'raise' or numeric 
495          Value to assign to the score if an error occurs in estimator fitting. 
496          If set to 'raise', the error is raised. If a numeric value is given, 
497          FitFailedWarning is raised. This parameter does not affect the refit 
498          step, which will always raise the error. Default is ``np.nan``. 
499   
500      return_train_score : bool, default=False 
501          If ``False``, the ``cv_results_`` attribute will not include training 
502          scores. 
503          Computing training scores is used to get insights on how different 
504          parameter settings impact the overfitting/underfitting trade-off. 
505          However computing the scores on the training set can be computationally 
506          expensive and is not strictly required to select the parameters that 
507          yield the best generalization performance. 
508   
509      random_state : int, RandomState instance or None, default=None 
510          Pseudo random number generator state used for subsampling the dataset 
511          when `resources != 'n_samples'`. Ignored otherwise. 
512          Pass an int for reproducible output across multiple function calls. 
513          See :term:`Glossary <random_state>`. 
514   
515      n_jobs : int or None, default=None 
516          Number of jobs to run in parallel. 
517          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
518          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
519          for more details. 
520   
521      verbose : int 
522          Controls the verbosity: the higher, the more messages. 
523   
524      Attributes 
525      ---------- 
526      n_resources_ : list of int 
527          The amount of resources used at each iteration. 
528   
529      n_candidates_ : list of int 
530          The number of candidate parameters that were evaluated at each 
531          iteration. 
532   
533      n_remaining_candidates_ : int 
534          The number of candidate parameters that are left after the last 
535          iteration. It corresponds to `ceil(n_candidates[-1] / factor)` 
536   
537      max_resources_ : int 
538          The maximum number of resources that any candidate is allowed to use 
539          for a given iteration. Note that since the number of resources used 
540          at each iteration must be a multiple of ``min_resources_``, the 
541          actual number of resources used at the last iteration may be smaller 
542          than ``max_resources_``. 
543   
544      min_resources_ : int 
545          The amount of resources that are allocated for each candidate at the 
546          first iteration. 
547   
548      n_iterations_ : int 
549          The actual number of iterations that were run. This is equal to 
550          ``n_required_iterations_`` if ``aggressive_elimination`` is ``True``. 
551          Else, this is equal to ``min(n_possible_iterations_, 
552          n_required_iterations_)``. 
553   
554      n_possible_iterations_ : int 
555          The number of iterations that are possible starting with 
556          ``min_resources_`` resources and without exceeding 
557          ``max_resources_``. 
558   
559      n_required_iterations_ : int 
560          The number of iterations that are required to end up with less than 
561          ``factor`` candidates at the last iteration, starting with 
562          ``min_resources_`` resources. This will be smaller than 
563          ``n_possible_iterations_`` when there isn't enough resources. 
564   
565      cv_results_ : dict of numpy (masked) ndarrays 
566          A dict with keys as column headers and values as columns, that can be 
567          imported into a pandas ``DataFrame``. It contains lots of information 
568          for analysing the results of a search. 
569          Please refer to the :ref:`User guide<successive_halving_cv_results>` 
570          for details. 
571   
572      best_estimator_ : estimator or dict 
573          Estimator that was chosen by the search, i.e. estimator 
574          which gave highest score (or smallest loss if specified) 
575          on the left out data. Not available if ``refit=False``. 
576   
577      best_score_ : float 
578          Mean cross-validated score of the best_estimator. 
579   
580      best_params_ : dict 
581          Parameter setting that gave the best results on the hold out data. 
582   
583      best_index_ : int 
584          The index (of the ``cv_results_`` arrays) which corresponds to the best 
585          candidate parameter setting. 
586   
587          The dict at ``search.cv_results_['params'][search.best_index_]`` gives 
588          the parameter setting for the best model, that gives the highest 
589          mean score (``search.best_score_``). 
590   
591      scorer_ : function or a dict 
592          Scorer function used on the held out data to choose the best 
593          parameters for the model. 
594   
595      n_splits_ : int 
596          The number of cross-validation splits (folds/iterations). 
597   
598      refit_time_ : float 
599          Seconds used for refitting the best model on the whole dataset. 
600   
601          This is present only if ``refit`` is not False. 
602   
603      multimetric_ : bool 
604          Whether or not the scorers compute several metrics. 
605   
606      classes_ : ndarray of shape (n_classes,) 
607          The classes labels. This is present only if ``refit`` is specified and 
608          the underlying estimator is a classifier. 
609   
610      n_features_in_ : int 
611          Number of features seen during :term:`fit`. Only defined if 
612          `best_estimator_` is defined (see the documentation for the `refit` 
613          parameter for more details) and that `best_estimator_` exposes 
614          `n_features_in_` when fit. 
615   
616          .. versionadded:: 0.24 
617   
618      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
619          Names of features seen during :term:`fit`. Only defined if 
620          `best_estimator_` is defined (see the documentation for the `refit` 
621          parameter for more details) and that `best_estimator_` exposes 
622          `feature_names_in_` when fit. 
623   
624          .. versionadded:: 1.0 
625

704  class HalvingRandomSearchCV(BaseSuccessiveHalving):
705      """Randomized search on hyper parameters. 
706   
707      The search strategy starts evaluating all the candidates with a small 
708      amount of resources and iteratively selects the best candidates, using more 
709      and more resources. 
710   
711      The candidates are sampled at random from the parameter space and the 
712      number of sampled candidates is determined by ``n_candidates``. 
713   
714      Read more in the :ref:`User guide<successive_halving_user_guide>`. 
715   
716      .. note:: 
717   
718        This estimator is still **experimental** for now: the predictions 
719        and the API might change without any deprecation cycle. To use it, 
720        you need to explicitly import ``enable_halving_search_cv``:: 
721   
722          >>> # explicitly require this experimental feature 
723          >>> from sklearn.experimental import enable_halving_search_cv # noqa 
724          >>> # now you can import normally from model_selection 
725          >>> from sklearn.model_selection import HalvingRandomSearchCV 
726   
727      Parameters 
728      ---------- 
729      estimator : estimator object 
730          This is assumed to implement the scikit-learn estimator interface. 
731          Either estimator needs to provide a ``score`` function, 
732          or ``scoring`` must be passed. 
733   
734      param_distributions : dict or list of dicts 
735          Dictionary with parameters names (`str`) as keys and distributions 
736          or lists of parameters to try. Distributions must provide a ``rvs`` 
737          method for sampling (such as those from scipy.stats.distributions). 
738          If a list is given, it is sampled uniformly. 
739          If a list of dicts is given, first a dict is sampled uniformly, and 
740          then a parameter is sampled using that dict as above. 
741   
742      n_candidates : "exhaust" or int, default="exhaust" 
743          The number of candidate parameters to sample, at the first 
744          iteration. Using 'exhaust' will sample enough candidates so that the 
745          last iteration uses as many resources as possible, based on 
746          `min_resources`, `max_resources` and `factor`. In this case, 
747          `min_resources` cannot be 'exhaust'. 
748   
749      factor : int or float, default=3 
750          The 'halving' parameter, which determines the proportion of candidates 
751          that are selected for each subsequent iteration. For example, 
752          ``factor=3`` means that only one third of the candidates are selected. 
753   
754      resource : ``'n_samples'`` or str, default='n_samples' 
755          Defines the resource that increases with each iteration. By default, 
756          the resource is the number of samples. It can also be set to any 
757          parameter of the base estimator that accepts positive integer 
758          values, e.g. 'n_iterations' or 'n_estimators' for a gradient 
759          boosting estimator. In this case ``max_resources`` cannot be 'auto' 
760          and must be set explicitly. 
761   
762      max_resources : int, default='auto' 
763          The maximum number of resources that any candidate is allowed to use 
764          for a given iteration. By default, this is set ``n_samples`` when 
765          ``resource='n_samples'`` (default), else an error is raised. 
766   
767      min_resources : {'exhaust', 'smallest'} or int, default='smallest' 
768          The minimum amount of resource that any candidate is allowed to use 
769          for a given iteration. Equivalently, this defines the amount of 
770          resources `r0` that are allocated for each candidate at the first 
771          iteration. 
772   
773          - 'smallest' is a heuristic that sets `r0` to a small value: 
774   
775            - ``n_splits * 2`` when ``resource='n_samples'`` for a regression problem 
776            - ``n_classes * n_splits * 2`` when ``resource='n_samples'`` for a 
777              classification problem 
778            - ``1`` when ``resource != 'n_samples'`` 
779   
780          - 'exhaust' will set `r0` such that the **last** iteration uses as 
781            much resources as possible. Namely, the last iteration will use the 
782            highest value smaller than ``max_resources`` that is a multiple of 
783            both ``min_resources`` and ``factor``. In general, using 'exhaust' 
784            leads to a more accurate estimator, but is slightly more time 
785            consuming. 'exhaust' isn't available when `n_candidates='exhaust'`. 
786   
787          Note that the amount of resources used at each iteration is always a 
788          multiple of ``min_resources``. 
789   
790      aggressive_elimination : bool, default=False 
791          This is only relevant in cases where there isn't enough resources to 
792          reduce the remaining candidates to at most `factor` after the last 
793          iteration. If ``True``, then the search process will 'replay' the 
794          first iteration for as long as needed until the number of candidates 
795          is small enough. This is ``False`` by default, which means that the 
796          last iteration may evaluate more than ``factor`` candidates. See 
797          :ref:`aggressive_elimination` for more details. 
798   
799      cv : int, cross-validation generator or an iterable, default=5 
800          Determines the cross-validation splitting strategy. 
801          Possible inputs for cv are: 
802   
803          - integer, to specify the number of folds in a `(Stratified)KFold`, 
804          - :term:`CV splitter`, 
805          - An iterable yielding (train, test) splits as arrays of indices. 
806   
807          For integer/None inputs, if the estimator is a classifier and ``y`` is 
808          either binary or multiclass, :class:`StratifiedKFold` is used. In all 
809          other cases, :class:`KFold` is used. These splitters are instantiated 
810          with `shuffle=False` so the splits will be the same across calls. 
811   
812          Refer :ref:`User Guide <cross_validation>` for the various 
813          cross-validation strategies that can be used here. 
814   
815          .. note:: 
816              Due to implementation details, the folds produced by `cv` must be 
817              the same across multiple calls to `cv.split()`. For 
818              built-in `scikit-learn` iterators, this can be achieved by 
819              deactivating shuffling (`shuffle=False`), or by setting the 
820              `cv`'s `random_state` parameter to an integer. 
821   
822      scoring : str, callable, or None, default=None 
823          A single string (see :ref:`scoring_parameter`) or a callable 
824          (see :ref:`scoring_callable`) to evaluate the predictions on the test set. 
825          If None, the estimator's score method is used. 
826   
827      refit : bool, default=True 
828          If True, refit an estimator using the best found parameters on the 
829          whole dataset. 
830   
831          The refitted estimator is made available at the ``best_estimator_`` 
832          attribute and permits using ``predict`` directly on this 
833          ``HalvingRandomSearchCV`` instance. 
834   
835      error_score : 'raise' or numeric 
836          Value to assign to the score if an error occurs in estimator fitting. 
837          If set to 'raise', the error is raised. If a numeric value is given, 
838          FitFailedWarning is raised. This parameter does not affect the refit 
839          step, which will always raise the error. Default is ``np.nan``. 
840   
841      return_train_score : bool, default=False 
842          If ``False``, the ``cv_results_`` attribute will not include training 
843          scores. 
844          Computing training scores is used to get insights on how different 
845          parameter settings impact the overfitting/underfitting trade-off. 
846          However computing the scores on the training set can be computationally 
847          expensive and is not strictly required to select the parameters that 
848          yield the best generalization performance. 
849   
850      random_state : int, RandomState instance or None, default=None 
851          Pseudo random number generator state used for subsampling the dataset 
852          when `resources != 'n_samples'`. Also used for random uniform 
853          sampling from lists of possible values instead of scipy.stats 
854          distributions. 
855          Pass an int for reproducible output across multiple function calls. 
856          See :term:`Glossary <random_state>`. 
857   
858      n_jobs : int or None, default=None 
859          Number of jobs to run in parallel. 
860          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
861          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
862          for more details. 
863   
864      verbose : int 
865          Controls the verbosity: the higher, the more messages. 
866   
867      Attributes 
868      ---------- 
869      n_resources_ : list of int 
870          The amount of resources used at each iteration. 
871   
872      n_candidates_ : list of int 
873          The number of candidate parameters that were evaluated at each 
874          iteration. 
875   
876      n_remaining_candidates_ : int 
877          The number of candidate parameters that are left after the last 
878          iteration. It corresponds to `ceil(n_candidates[-1] / factor)` 
879   
880      max_resources_ : int

882          for a given iteration. Note that since the number of resources used at 
883          each iteration must be a multiple of ``min_resources_``, the actual 
884          number of resources used at the last iteration may be smaller than 
885          ``max_resources_``. 
886   
887      min_resources_ : int

889          first iteration. 
890   
891      n_iterations_ : int

895          n_required_iterations_)``.

897      n_possible_iterations_ : int

899          ``min_resources_`` resources and without exceeding

906          ``n_possible_iterations_`` when there isn't enough resources. 
907   
908      cv_results_ : dict of numpy (masked) ndarrays

913          for details. 
914   
915      best_estimator_ : estimator or dict 
916          Estimator that was chosen by the search, i.e. estimator 
917          which gave highest score (or smallest loss if specified)

919   
920      best_score_ : float 
921          Mean cross-validated score of the best_estimator. 
922   
923      best_params_ : dict 
924          Parameter setting that gave the best results on the hold out data. 
925   
926      best_index_ : int 
927          The index (of the ``cv_results_`` arrays) which corresponds to the best 
928          candidate parameter setting. 
929   
930          The dict at ``search.cv_results_['params'][search.best_index_]`` gives 
931          the parameter setting for the best model, that gives the highest 
932          mean score (``search.best_score_``). 
933   
934      scorer_ : function or a dict 
935          Scorer function used on the held out data to choose the best 
936          parameters for the model. 
937   
938      n_splits_ : int 
939          The number of cross-validation splits (folds/iterations). 
940   
941      refit_time_ : float 
942          Seconds used for refitting the best model on the whole dataset. 
943   
944          This is present only if ``refit`` is not False. 
945   
946      multimetric_ : bool 
947          Whether or not the scorers compute several metrics. 
948   
949      classes_ : ndarray of shape (n_classes,) 
950          The classes labels. This is present only if ``refit`` is specified and 
951          the underlying estimator is a classifier. 
952   
953      n_features_in_ : int 
954          Number of features seen during :term:`fit`. Only defined if 
955          `best_estimator_` is defined (see the documentation for the `refit`

957          `n_features_in_` when fit.
