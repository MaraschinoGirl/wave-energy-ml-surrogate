12   # A very simple demo - note that this does no more than you can do with
13   # builtin Python file objects, so for something as simple as this, you
14   # generally *should* use builtin Python objects.  Only use win32file etc
15   # when you need win32 specific features not available in Python.
16   def SimpleFileDemo():
17       testName = os.path.join(win32api.GetTempPath(), "win32file_demo_test_file")
18       if os.path.exists(testName):
19           os.unlink(testName)
20       # Open the file for writing.
21       handle = win32file.CreateFile(
22           testName, win32file.GENERIC_WRITE, 0, None, win32con.CREATE_NEW, 0, None
23       )
24       test_data = b"Hello\0there"
25       win32file.WriteFile(handle, test_data)
26       handle.Close()
27       # Open it for reading.
28       handle = win32file.CreateFile(
29           testName, win32file.GENERIC_READ, 0, None, win32con.OPEN_EXISTING, 0, None
30       )
31       rc, data = win32file.ReadFile(handle, 1024)
32       handle.Close()
33       if data == test_data:
34           print("Successfully wrote and read a file")
35       else:
36           raise Exception("Got different data back???")
37       os.unlink(testName)
38   
39   
40   if __name__ == "__main__":
41       SimpleFileDemo()
