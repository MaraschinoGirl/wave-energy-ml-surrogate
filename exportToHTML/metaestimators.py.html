20   class _BaseComposition(BaseEstimator, metaclass=ABCMeta):
21       """Handles parameter management for classifiers composed of named estimators."""
22   
23       steps: List[Any]
24   
25       @abstractmethod
26       def __init__(self):
27           pass
28   
29       def _get_params(self, attr, deep=True):
30           out = super().get_params(deep=deep)
31           if not deep:
32               return out
33   
34           estimators = getattr(self, attr)
35           try:
36               out.update(estimators)
37           except (TypeError, ValueError):
38               # Ignore TypeError for cases where estimators is not a list of
39               # (name, estimator) and ignore ValueError when the list is not
40               # formatted correctly. This is to prevent errors when calling
41               # `set_params`. `BaseEstimator.set_params` calls `get_params` which
42               # can error for invalid values for `estimators`.
43               return out
44   
45           for name, estimator in estimators:
46               if hasattr(estimator, "get_params"):
47                   for key, value in estimator.get_params(deep=True).items():
48                       out["%s__%s" % (name, key)] = value
49           return out
50   
51       def _set_params(self, attr, **params):
52           # Ensure strict ordering of parameter setting:
53           # 1. All steps
54           if attr in params:
55               setattr(self, attr, params.pop(attr))
56           # 2. Replace items with estimators in params
57           items = getattr(self, attr)
58           if isinstance(items, list) and items:
59               # Get item names used to identify valid names in params
60               # `zip` raises a TypeError when `items` does not contains
61               # elements of length 2
62               with suppress(TypeError):
63                   item_names, _ = zip(*items)
64                   for name in list(params.keys()):
65                       if "__" not in name and name in item_names:
66                           self._replace_estimator(attr, name, params.pop(name))
67   
68           # 3. Step parameters and other initialisation arguments
69           super().set_params(**params)
70           return self
71   
72       def _replace_estimator(self, attr, name, new_val):
73           # assumes `name` is a valid estimator name
74           new_estimators = list(getattr(self, attr))
75           for i, (estimator_name, _) in enumerate(new_estimators):
76               if estimator_name == name:
77                   new_estimators[i] = (name, new_val)
78                   break
79           setattr(self, attr, new_estimators)
80   
81       def _validate_names(self, names):
82           if len(set(names)) != len(names):
83               raise ValueError("Names provided are not unique: {0!r}".format(list(names)))
84           invalid_names = set(names).intersection(self.get_params(deep=False))
85           if invalid_names:
86               raise ValueError(
87                   "Estimator names conflict with constructor arguments: {0!r}".format(
88                       sorted(invalid_names)
89                   )
90               )
91           invalid_names = [name for name in names if "__" in name]
92           if invalid_names:
93               raise ValueError(
94                   "Estimator names must not contain __: got {0!r}".format(invalid_names)
95               )

98   def _safe_split(estimator, X, y, indices, train_indices=None):
99       """Create subset of dataset and properly handle kernels. 
100   
101      Slice X, y according to indices for cross-validation, but take care of 
102      precomputed kernel-matrices or pairwise affinities / distances. 
103   
104      If ``estimator._pairwise is True``, X needs to be square and 
105      we slice rows and columns. If ``train_indices`` is not None, 
106      we slice rows using ``indices`` (assumed the test set) and columns 
107      using ``train_indices``, indicating the training set. 
108   
109      Labels y will always be indexed only along the first axis. 
110   
111      Parameters 
112      ---------- 
113      estimator : object 
114          Estimator to determine whether we should slice only rows or rows and 
115          columns. 
116   
117      X : array-like, sparse matrix or iterable 
118          Data to be indexed. If ``estimator._pairwise is True``, 
119          this needs to be a square array-like or sparse matrix. 
120   
121      y : array-like, sparse matrix or iterable 
122          Targets to be indexed. 
123   
124      indices : array of int 
125          Rows to select from X and y. 
126          If ``estimator._pairwise is True`` and ``train_indices is None`` 
127          then ``indices`` will also be used to slice columns. 
128   
129      train_indices : array of int or None, default=None 
130          If ``estimator._pairwise is True`` and ``train_indices is not None``, 
131          then ``train_indices`` will be use to slice the columns of X. 
132   
133      Returns 
134      ------- 
135      X_subset : array-like, sparse matrix or list 
136          Indexed data. 
137   
138      y_subset : array-like, sparse matrix or list 
139          Indexed targets. 
140   
141      """
142      if get_tags(estimator).input_tags.pairwise:
143          if not hasattr(X, "shape"):
144              raise ValueError(
145                  "Precomputed kernels or affinity matrices have "
146                  "to be passed as arrays or sparse matrices."
147              )
148          # X is a precomputed square kernel matrix
149          if X.shape[0] != X.shape[1]:
150              raise ValueError("X should be a square kernel matrix")
151          if train_indices is None:
152              X_subset = X[np.ix_(indices, indices)]
153          else:
154              X_subset = X[np.ix_(indices, train_indices)]
155      else:
156          X_subset = _safe_indexing(X, indices)
157  
158      if y is not None:
159          y_subset = _safe_indexing(y, indices)
160      else:
161          y_subset = None
162  
163      return X_subset, y_subset
