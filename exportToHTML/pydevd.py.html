186  def install_breakpointhook(pydevd_breakpointhook=None):
187      if pydevd_breakpointhook is None:
188  
189          def pydevd_breakpointhook(*args, **kwargs):
190              hookname = os.getenv("PYTHONBREAKPOINT")
191              if (
192                  hookname is not None
193                  and len(hookname) > 0
194                  and hasattr(sys, "__breakpointhook__")
195                  and sys.__breakpointhook__ != pydevd_breakpointhook
196              ):
197                  sys.__breakpointhook__(*args, **kwargs)
198              else:
199                  settrace(*args, **kwargs)
200  
201      if sys.version_info[0:2] >= (3, 7):
202          # There are some choices on how to provide the breakpoint hook. Namely, we can provide a
203          # PYTHONBREAKPOINT which provides the import path for a method to be executed or we
204          # can override sys.breakpointhook.
205          # pydevd overrides sys.breakpointhook instead of providing an environment variable because
206          # it's possible that the debugger starts the user program but is not available in the
207          # PYTHONPATH (and would thus fail to be imported if PYTHONBREAKPOINT was set to pydevd.settrace).
208          # Note that the implementation still takes PYTHONBREAKPOINT in account (so, if it was provided
209          # by someone else, it'd still work).
210          sys.breakpointhook = pydevd_breakpointhook
211      else:
212          if sys.version_info[0] >= 3:
213              import builtins as __builtin__  # Py3 noqa
214          else:
215              import __builtin__  # noqa
216  
217          # In older versions, breakpoint() isn't really available, so, install the hook directly
218          # in the builtins.
219          __builtin__.breakpoint = pydevd_breakpointhook
220          sys.__breakpointhook__ = pydevd_breakpointhook
221  
222  
223  # Install the breakpoint hook at import time.
224  install_breakpointhook()

326                  if can_exit():

329                  py_db.check_output_redirect()

332                  pydev_log.debug("No threads alive, finishing debug session")
333                  py_db.dispose_and_kill_all_pydevd_threads()
334          except:
335              pydev_log.exception()

350  class AbstractSingleNotificationBehavior(object):
351      """ 
352      The basic usage should be: 
353   
354      # Increment the request time for the suspend. 
355      single_notification_behavior.increment_suspend_time() 
356   
357      # Notify that this is a pause request (when a pause, not a breakpoint). 
358      single_notification_behavior.on_pause() 
359   
360      # Mark threads to be suspended. 
361      set_suspend(...) 
362   
363      # On do_wait_suspend, use notify_thread_suspended: 
364      def do_wait_suspend(...): 
365          with single_notification_behavior.notify_thread_suspended(thread_id, thread, reason): 
366              ... 
367      """
368  
369      __slots__ = [
370          "_last_resume_notification_time",
371          "_last_suspend_notification_time",
372          "_lock",
373          "_next_request_time",
374          "_suspend_time_request",
375          "_suspended_thread_id_to_thread",
376          "_pause_requested",
377          "_py_db",
378      ]
379  
380      NOTIFY_OF_PAUSE_TIMEOUT = 0.5
381  
382      def __init__(self, py_db):
383          self._py_db = weakref.ref(py_db)
384          self._next_request_time = partial(next, itertools.count())
385          self._last_suspend_notification_time = -1
386          self._last_resume_notification_time = -1
387          self._suspend_time_request = self._next_request_time()
388          self._lock = thread.allocate_lock()
389          self._suspended_thread_id_to_thread = {}
390          self._pause_requested = False
391  
392      def send_suspend_notification(self, thread_id, thread, stop_reason):
393          raise AssertionError("abstract: subclasses must override.")
394  
395      def send_resume_notification(self, thread_id):
396          raise AssertionError("abstract: subclasses must override.")
397  
398      def increment_suspend_time(self):
399          with self._lock:
400              self._suspend_time_request = self._next_request_time()
401  
402      def on_pause(self):
403          # Upon a pause, we should force sending new suspend notifications
404          # if no notification is sent after some time and there's some thread already stopped.
405          with self._lock:
406              self._pause_requested = True
407              global_suspend_time = self._suspend_time_request
408          py_db = self._py_db()
409          if py_db is not None:
410              py_db.timeout_tracker.call_on_timeout(
411                  self.NOTIFY_OF_PAUSE_TIMEOUT, self._notify_after_timeout, kwargs={"global_suspend_time": global_suspend_time}
412              )
413  
414      def _notify_after_timeout(self, global_suspend_time):
415          with self._lock:
416              if self._suspended_thread_id_to_thread:
417                  if global_suspend_time > self._last_suspend_notification_time:
418                      self._last_suspend_notification_time = global_suspend_time
419                      # Notify about any thread which is currently suspended.
420                      pydev_log.info("Sending suspend notification after timeout.")
421                      thread_id, thread = next(iter(self._suspended_thread_id_to_thread.items()))
422                      self.send_suspend_notification(thread_id, thread, CMD_THREAD_SUSPEND)
423  
424      def on_thread_suspend(self, thread_id, thread, stop_reason):
425          with self._lock:
426              pause_requested = self._pause_requested
427              if pause_requested:
428                  # When a suspend notification is sent, reset the pause flag.
429                  self._pause_requested = False
430  
431              self._suspended_thread_id_to_thread[thread_id] = thread
432  
433              # CMD_THREAD_SUSPEND should always be a side-effect of a break, so, only
434              # issue for a CMD_THREAD_SUSPEND if a pause is pending.
435              if stop_reason != CMD_THREAD_SUSPEND or pause_requested:
436                  if self._suspend_time_request > self._last_suspend_notification_time:
437                      pydev_log.info("Sending suspend notification.")
438                      self._last_suspend_notification_time = self._suspend_time_request
439                      self.send_suspend_notification(thread_id, thread, stop_reason)
440                  else:
441                      pydev_log.info(
442                          "Suspend not sent (it was already sent). Last suspend % <= Last resume %s",
443                          self._last_suspend_notification_time,
444                          self._last_resume_notification_time,
445                      )
446              else:
447                  pydev_log.info(
448                      "Suspend not sent because stop reason is thread suspend and pause was not requested.",
449                  )
450  
451      def on_thread_resume(self, thread_id, thread):
452          # on resume (step, continue all):
453          with self._lock:
454              self._suspended_thread_id_to_thread.pop(thread_id)
455              if self._last_resume_notification_time < self._last_suspend_notification_time:
456                  pydev_log.info("Sending resume notification.")
457                  self._last_resume_notification_time = self._last_suspend_notification_time
458                  self.send_resume_notification(thread_id)
459              else:
460                  pydev_log.info(
461                      "Resume not sent (it was already sent). Last resume %s >= Last suspend %s",
462                      self._last_resume_notification_time,
463                      self._last_suspend_notification_time,
464                  )
465  
466      @contextmanager
467      def notify_thread_suspended(self, thread_id, thread, stop_reason):
468          self.on_thread_suspend(thread_id, thread, stop_reason)
469          try:
470              yield  # At this point the thread must be actually suspended.
471          finally:
472              self.on_thread_resume(thread_id, thread)

475  class ThreadsSuspendedSingleNotification(AbstractSingleNotificationBehavior):
476      __slots__ = AbstractSingleNotificationBehavior.__slots__ + ["multi_threads_single_notification", "_callbacks", "_callbacks_lock"]
477  
478      def __init__(self, py_db):
479          AbstractSingleNotificationBehavior.__init__(self, py_db)
480          # If True, pydevd will send a single notification when all threads are suspended/resumed.
481          self.multi_threads_single_notification = False
482          self._callbacks_lock = threading.Lock()
483          self._callbacks = []
484  
485      def add_on_resumed_callback(self, callback):
486          with self._callbacks_lock:
487              self._callbacks.append(callback)
488  
489      @overrides(AbstractSingleNotificationBehavior.send_resume_notification)
490      def send_resume_notification(self, thread_id):
491          py_db = self._py_db()
492          if py_db is not None:
493              py_db.writer.add_command(py_db.cmd_factory.make_thread_resume_single_notification(thread_id))
494  
495              with self._callbacks_lock:
496                  callbacks = self._callbacks
497                  self._callbacks = []
498  
499              for callback in callbacks:
500                  callback()
501  
502      @overrides(AbstractSingleNotificationBehavior.send_suspend_notification)
503      def send_suspend_notification(self, thread_id, thread, stop_reason):
504          py_db = self._py_db()
505          if py_db is not None:
506              py_db.writer.add_command(py_db.cmd_factory.make_thread_suspend_single_notification(py_db, thread_id, thread, stop_reason))
507  
508      @overrides(AbstractSingleNotificationBehavior.notify_thread_suspended)
509      @contextmanager
510      def notify_thread_suspended(self, thread_id, thread, stop_reason):
511          if self.multi_threads_single_notification:
512              pydev_log.info("Thread suspend mode: single notification")
513              with AbstractSingleNotificationBehavior.notify_thread_suspended(self, thread_id, thread, stop_reason):

520  class _Authentication(object):
521      __slots__ = ["access_token", "client_access_token", "_authenticated", "_wrong_attempts"]
522  
523      def __init__(self):
524          # A token to be send in the command line or through the settrace api -- when such token
525          # is given, the first message sent to the IDE must pass the same token to authenticate.
526          # Note that if a disconnect is sent, the same message must be resent to authenticate.
527          self.access_token = None
528  
529          # This token is the one that the client requires to accept a connection from pydevd
530          # (it's stored here and just passed back when required, it's not used internally
531          # for anything else).
532          self.client_access_token = None
533  
534          self._authenticated = None
535  
536          self._wrong_attempts = 0
537  
538      def is_authenticated(self):
539          if self._authenticated is None:
540              return self.access_token is None
541          return self._authenticated
542  
543      def login(self, access_token):
544          if self._wrong_attempts >= 10:  # A user can fail to authenticate at most 10 times.
545              return
546  
547          self._authenticated = access_token == self.access_token
548          if not self._authenticated:
549              self._wrong_attempts += 1
550          else:
551              self._wrong_attempts = 0
552  
553      def logout(self):
554          self._authenticated = None
555          self._wrong_attempts = 0

558  class PyDB(object):
559      """Main debugging class 
560      Lots of stuff going on here: 
561   
562      PyDB starts two threads on startup that connect to remote debugger (RDB) 
563      The threads continuously read & write commands to RDB. 
564      PyDB communicates with these threads through command queues. 
565         Every RDB command is processed by calling process_net_command. 
566         Every PyDB net command is sent to the net by posting NetCommand to WriterThread queue 
567   
568         Some commands need to be executed on the right thread (suspend/resume & friends) 
569         These are placed on the internal command queue. 
570      """
571

814      def setup_auto_reload_watcher(self, enable_auto_reload, watch_dirs, poll_target_time, exclude_patterns, include_patterns):
815          try:
816              with self._lock_create_fs_notify:
817                  # When setting up, dispose of the previous one (if any).
818                  if self._fsnotify_thread is not None:
819                      self._fsnotify_thread.do_kill_pydev_thread()
820                      self._fsnotify_thread = None
821  
822                  if not enable_auto_reload:
823                      return
824  
825                  exclude_patterns = tuple(exclude_patterns)
826                  include_patterns = tuple(include_patterns)
827  
828                  def accept_directory(absolute_filename, cache={}):
829                      try:
830                          return cache[absolute_filename]
831                      except:
832                          if absolute_filename and absolute_filename[-1] not in ("/", "\\"):
833                              # I.e.: for directories we always end with '/' or '\\' so that
834                              # we match exclusions such as "**/node_modules/**"
835                              absolute_filename += os.path.sep
836  
837                          # First include what we want
838                          for include_pattern in include_patterns:
839                              if glob_matches_path(absolute_filename, include_pattern):
840                                  cache[absolute_filename] = True
841                                  return True
842  
843                          # Then exclude what we don't want
844                          for exclude_pattern in exclude_patterns:
845                              if glob_matches_path(absolute_filename, exclude_pattern):
846                                  cache[absolute_filename] = False
847                                  return False
848  
849                          # By default track all directories not excluded.
850                          cache[absolute_filename] = True
851                          return True
852  
853                  def accept_file(absolute_filename, cache={}):
854                      try:
855                          return cache[absolute_filename]
856                      except:
857                          # First include what we want
858                          for include_pattern in include_patterns:
859                              if glob_matches_path(absolute_filename, include_pattern):
860                                  cache[absolute_filename] = True
861                                  return True
862  
863                          # Then exclude what we don't want
864                          for exclude_pattern in exclude_patterns:
865                              if glob_matches_path(absolute_filename, exclude_pattern):
866                                  cache[absolute_filename] = False
867                                  return False
868  
869                          # By default don't track files not included.
870                          cache[absolute_filename] = False
871                          return False
872  
873                  self._fsnotify_thread = FSNotifyThread(self, PyDevdAPI(), watch_dirs)
874                  watcher = self._fsnotify_thread.watcher
875                  watcher.accept_directory = accept_directory
876                  watcher.accept_file = accept_file
877  
878                  watcher.target_time_for_single_scan = poll_target_time
879                  watcher.target_time_for_notification = poll_target_time
880                  self._fsnotify_thread.start()
881          except:
882              pydev_log.exception("Error setting up auto-reload.")

950      def handle_breakpoint_condition(self, info, pybreakpoint, new_frame):
951          condition = pybreakpoint.condition
952          try:
953              if pybreakpoint.handle_hit_condition(new_frame):
954                  return True
955  
956              if not condition:
957                  return False
958  
959              return eval(condition, new_frame.f_globals, new_frame.f_locals)
960          except Exception as e:
961              if not isinstance(e, self.skip_print_breakpoint_exception):
962                  stack_trace = io.StringIO()
963                  etype, value, tb = sys.exc_info()
964                  traceback.print_exception(etype, value, tb.tb_next, file=stack_trace)
965  
966                  msg = "Error while evaluating expression in conditional breakpoint: %s\n%s" % (condition, stack_trace.getvalue())
967                  api = PyDevdAPI()
968                  api.send_error_message(self, msg)
969  
970              if not isinstance(e, self.skip_suspend_on_breakpoint_exception):
971                  try:
972                      # add exception_type and stacktrace into thread additional info
973                      etype, value, tb = sys.exc_info()
974                      error = "".join(traceback.format_exception_only(etype, value))
975                      stack = traceback.extract_stack(f=tb.tb_frame.f_back)
976  
977                      # On self.set_suspend(thread, CMD_SET_BREAK) this info will be
978                      # sent to the client.
979                      info.conditional_breakpoint_exception = ("Condition:\n" + condition + "\n\nError:\n" + error, stack)
980                  except:
981                      pydev_log.exception()
982                  return True
983  
984              return False
985  
986          finally:
987              etype, value, tb = None, None, None

1042     def get_file_type(self, frame, abs_real_path_and_basename=None, _cache_file_type=_CACHE_FILE_TYPE):
1043         """ 
1044         :param abs_real_path_and_basename: 
1045             The result from get_abs_path_real_path_and_base_from_file or 
1046             get_abs_path_real_path_and_base_from_frame. 
1047  
1048         :return 
1049             _pydevd_bundle.pydevd_dont_trace_files.PYDEV_FILE: 
1050                 If it's a file internal to the debugger which shouldn't be 
1051                 traced nor shown to the user. 
1052  
1053             _pydevd_bundle.pydevd_dont_trace_files.LIB_FILE: 
1054                 If it's a file in a library which shouldn't be traced. 
1055  
1056             None: 
1057                 If it's a regular user file which should be traced. 
1058         """
1059         if abs_real_path_and_basename is None:
1060             try:
1061                 # Make fast path faster!
1062                 abs_real_path_and_basename = NORM_PATHS_AND_BASE_CONTAINER[frame.f_code.co_filename]
1063             except:
1064                 abs_real_path_and_basename = get_abs_path_real_path_and_base_from_frame(frame)
1065 
1066         # Note 1: we have to take into account that we may have files as '<string>', and that in
1067         # this case the cache key can't rely only on the filename. With the current cache, there's
1068         # still a potential miss if 2 functions which have exactly the same content are compiled
1069         # with '<string>', but in practice as we only separate the one from python -c from the rest
1070         # this shouldn't be a problem in practice.
1071 
1072         # Note 2: firstlineno added to make misses faster in the first comparison.
1073 
1074         # Note 3: this cache key is repeated in pydevd_frame_evaluator.pyx:get_func_code_info (for
1075         # speedups).
1076         cache_key = (frame.f_code.co_firstlineno, abs_real_path_and_basename[0], frame.f_code)
1077         try:
1078             return _cache_file_type[cache_key]
1079         except:
1080             if abs_real_path_and_basename[0] == "<string>":
1081                 f = frame.f_back
1082                 while f is not None:
1083                     if self.get_file_type(f) != self.PYDEV_FILE and pydevd_file_utils.basename(f.f_code.co_filename) not in (
1084                         "runpy.py",
1085                         "<string>",
1086                     ):
1087                         # We found some back frame that's not internal, which means we must consider
1088                         # this a library file.
1089                         # This is done because we only want to trace files as <string> if they don't
1090                         # have any back frame (which is the case for python -c ...), for all other
1091                         # cases we don't want to trace them because we can't show the source to the
1092                         # user (at least for now...).
1093 
1094                         # Note that we return as a LIB_FILE and not PYDEV_FILE because we still want
1095                         # to show it in the stack.
1096                         _cache_file_type[cache_key] = LIB_FILE
1097                         return LIB_FILE
1098 
1099                     f = f.f_back
1100                 else:
1101                     # This is a top-level file (used in python -c), so, trace it as usual... we
1102                     # still won't be able to show the sources, but some tests require this to work.
1103                     _cache_file_type[cache_key] = None
1104                     return None
1105 
1106             file_type = self._internal_get_file_type(abs_real_path_and_basename)
1107             if file_type is None:
1108                 if self.dont_trace_external_files(abs_real_path_and_basename[0]):
1109                     file_type = PYDEV_FILE
1110 
1111             _cache_file_type[cache_key] = file_type
1112             return file_type
