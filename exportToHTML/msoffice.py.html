14   class OleClientItem(object.CmdTarget):
15       def __init__(self, doc):
16           object.CmdTarget.__init__(self, win32uiole.CreateOleClientItem(doc))
17   
18       def OnGetItemPosition(self):
19           # For now return a hard-coded rect.
20           return (10, 10, 210, 210)
21   
22       def OnActivate(self):
23           # Allow only one inplace activate item per frame
24           view = self.GetActiveView()
25           item = self.GetDocument().GetInPlaceActiveItem(view)
26           if item is not None and item._obj_ != self._obj_:
27               item.Close()
28           self._obj_.OnActivate()
29   
30       def OnChange(self, oleNotification, dwParam):
31           self._obj_.OnChange(oleNotification, dwParam)
32           self.GetDocument().UpdateAllViews(None)
33   
34       def OnChangeItemPosition(self, rect):
35           # During in-place activation CEmbed_ExcelCntrItem::OnChangeItemPosition
36           #  is called by the server to change the position of the in-place
37           #  window.  Usually, this is a result of the data in the server
38           #  document changing such that the extent has changed or as a result
39           #  of in-place resizing.
40           #
41           # The default here is to call the base class, which will call
42           #  COleClientItem::SetItemRects to move the item
43           #  to the new position.
44           if not self._obj_.OnChangeItemPosition(self, rect):
45               return 0
46   
47           # TODO: update any cache you may have of the item's rectangle/extent
48           return 1

57   class ExcelView(docview.ScrollView):
58       def OnInitialUpdate(self):
59           self.HookMessage(self.OnSetFocus, win32con.WM_SETFOCUS)
60           self.HookMessage(self.OnSize, win32con.WM_SIZE)
61   
62           self.SetScrollSizes(win32con.MM_TEXT, (100, 100))
63           rc = self._obj_.OnInitialUpdate()
64           self.EmbedExcel()
65           return rc
66   
67       def EmbedExcel(self):
68           doc = self.GetDocument()
69           self.clientItem = OleClientItem(doc)
70           self.clientItem.CreateNewItem("Excel.Sheet")
71           self.clientItem.DoVerb(-1, self)
72           doc.UpdateAllViews(None)
73   
74       def OnDraw(self, dc):
75           doc = self.GetDocument()
76           pos = doc.GetStartPosition()
77           clientItem, pos = doc.GetNextItem(pos)
78           clientItem.Draw(dc, (10, 10, 210, 210))
79   
80       # Special handling of OnSetFocus and OnSize are required for a container
81       #  when an object is being edited in-place.
82       def OnSetFocus(self, msg):
83           item = self.GetDocument().GetInPlaceActiveItem(self)
84           if (
85               item is not None
86               and item.GetItemState() == win32uiole.COleClientItem_activeUIState
87           ):
88               wnd = item.GetInPlaceWindow()
89               if wnd is not None:
90                   wnd.SetFocus()
91               return 0  # Don't get the base version called.
92           return 1  # Call the base version.
93   
94       def OnSize(self, params):
95           item = self.GetDocument().GetInPlaceActiveItem(self)
96           if item is not None:
97               item.SetItemRects()
98           return 1  # do call the base!

114  class WordFrame(window.MDIChildWnd):
115      def __init__(self, doc=None):
116          self._obj_ = win32ui.CreateMDIChild()
117          self._obj_.AttachObject(self)
118          # Don't call base class doc/view version...
119  
120      def Create(self, title, rect=None, parent=None):
121          WordModule = gencache.EnsureModule(
122              "{00020905-0000-0000-C000-000000000046}", 1033, 8, 0
123          )
124          if WordModule is None:
125              raise ImportError(
126                  "Microsoft Word version 8 does not appear to be installed."
127              )
128  
129          # WordModule.Word doesn't exist in WordModule, WordModule.Words does, but CreateControl still fails
130          class MyWordControl(activex.Control, WordModule.Word): ...
131  
132          style = win32con.WS_CHILD | win32con.WS_VISIBLE | win32con.WS_OVERLAPPEDWINDOW
133          self._obj_.CreateWindow(None, title, style, rect, parent)
134  
135          rect = self.GetClientRect()
136          rect = (0, 0, rect[2] - rect[0], rect[3] - rect[1])
137          self.ocx = MyWordControl()
138          self.ocx.CreateControl(
139              "Microsoft Word", win32con.WS_VISIBLE | win32con.WS_CHILD, rect, self, 20000
140          )
