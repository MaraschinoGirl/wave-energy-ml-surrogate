1    import numpy as np

3    from scipy._lib._util import _get_nan
4    from scipy._lib._array_api import array_namespace, xp_copysign

6    from ._axis_nan_policy import _axis_nan_policy_factory

9    @_axis_nan_policy_factory(
10       lambda x: x, n_outputs=1, result_to_tuple=lambda x: (x,)
11   )
12   def variation(a, axis=0, nan_policy='propagate', ddof=0, *, keepdims=False):
13       """ 
14       Compute the coefficient of variation. 
15    
16       The coefficient of variation is the standard deviation divided by the 
17       mean.  This function is equivalent to:: 
18    
19           np.std(x, axis=axis, ddof=ddof) / np.mean(x) 
20    
21       The default for ``ddof`` is 0, but many definitions of the coefficient 
22       of variation use the square root of the unbiased sample variance 
23       for the sample standard deviation, which corresponds to ``ddof=1``. 
24    
25       The function does not take the absolute value of the mean of the data, 
26       so the return value is negative if the mean is negative. 
27    
28       Parameters 
29       ---------- 
30       a : array_like 
31           Input array. 
32       axis : int or None, optional 
33           Axis along which to calculate the coefficient of variation. 
34           Default is 0. If None, compute over the whole array `a`. 
35       nan_policy : {'propagate', 'raise', 'omit'}, optional 
36           Defines how to handle when input contains ``nan``. 
37           The following options are available: 
38    
39             * 'propagate': return ``nan`` 
40             * 'raise': raise an exception 
41             * 'omit': perform the calculation with ``nan`` values omitted 
42    
43           The default is 'propagate'. 
44       ddof : int, optional 
45           Gives the "Delta Degrees Of Freedom" used when computing the 
46           standard deviation.  The divisor used in the calculation of the 
47           standard deviation is ``N - ddof``, where ``N`` is the number of 
48           elements.  `ddof` must be less than ``N``; if it isn't, the result 
49           will be ``nan`` or ``inf``, depending on ``N`` and the values in 
50           the array.  By default `ddof` is zero for backwards compatibility, 
51           but it is recommended to use ``ddof=1`` to ensure that the sample 
52           standard deviation is computed as the square root of the unbiased 
53           sample variance. 
54    
55       Returns 
56       ------- 
57       variation : ndarray 
58           The calculated variation along the requested axis. 
59    
60       Notes 
61       ----- 
62       There are several edge cases that are handled without generating a 
63       warning: 
64    
65       * If both the mean and the standard deviation are zero, ``nan`` 
66         is returned. 
67       * If the mean is zero and the standard deviation is nonzero, ``inf`` 
68         is returned. 
69       * If the input has length zero (either because the array has zero 
70         length, or all the input values are ``nan`` and ``nan_policy`` is 
71         ``'omit'``), ``nan`` is returned. 
72       * If the input contains ``inf``, ``nan`` is returned. 
73    
74       References 
75       ---------- 
76       .. [1] Zwillinger, D. and Kokoska, S. (2000). CRC Standard 
77          Probability and Statistics Tables and Formulae. Chapman & Hall: New 
78          York. 2000. 
79    
80       Examples 
81       -------- 
82       >>> import numpy as np 
83       >>> from scipy.stats import variation 
84       >>> variation([1, 2, 3, 4, 5], ddof=1) 
85       0.5270462766947299 
86    
87       Compute the variation along a given dimension of an array that contains 
88       a few ``nan`` values: 
89    
90       >>> x = np.array([[  10.0, np.nan, 11.0, 19.0, 23.0, 29.0, 98.0], 
91       ...               [  29.0,   30.0, 32.0, 33.0, 35.0, 56.0, 57.0], 
92       ...               [np.nan, np.nan, 12.0, 13.0, 16.0, 16.0, 17.0]]) 
93       >>> variation(x, axis=1, ddof=1, nan_policy='omit') 
94       array([1.05109361, 0.31428986, 0.146483  ]) 
95    
96       """
97       xp = array_namespace(a)
98       a = xp.asarray(a)
99       # `nan_policy` and `keepdims` are handled by `_axis_nan_policy`
100      # `axis=None` is only handled for NumPy backend
101      if axis is None:
102          a = xp.reshape(a, (-1,))
103          axis = 0
104  
105      n = a.shape[axis]
106      NaN = _get_nan(a)
107  
108      if a.size == 0 or ddof > n:
109          # Handle as a special case to avoid spurious warnings.
110          # The return values, if any, are all nan.
111          shp = list(a.shape)
112          shp.pop(axis)
113          result = xp.full(shp, fill_value=NaN)
114          return result[()] if result.ndim == 0 else result
115  
116      mean_a = xp.mean(a, axis=axis)
117  
118      if ddof == n:
119          # Another special case.  Result is either inf or nan.
120          std_a = xp.std(a, axis=axis, correction=0)
121          result = xp.where(std_a > 0, xp_copysign(xp.asarray(xp.inf), mean_a), NaN)
122          return result[()] if result.ndim == 0 else result
123  
124      with np.errstate(divide='ignore', invalid='ignore'):
125          std_a = xp.std(a, axis=axis, correction=ddof)
126          result = std_a / mean_a
127  
128      return result[()] if result.ndim == 0 else result
