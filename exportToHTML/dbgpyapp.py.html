3    import sys
4    
5    import win32con
6    import win32ui
7    from pywin.framework import intpyapp
8    
9    version = "0.3.0"
10   
11   
12   class DebuggerPythonApp(intpyapp.InteractivePythonApp):
13       def LoadMainFrame(self):
14           "Create the main applications frame"
15           self.frame = self.CreateMainFrame()
16           self.SetMainFrame(self.frame)
17           self.frame.LoadFrame(win32ui.IDR_DEBUGGER, win32con.WS_OVERLAPPEDWINDOW)
18           self.frame.DragAcceptFiles()  # we can accept these.
19           self.frame.ShowWindow(win32con.SW_HIDE)
20           self.frame.UpdateWindow()
21   
22           # but we do rehook, hooking the new code objects.
23           self.HookCommands()
24   
25       def InitInstance(self):
26           # Use a registry path of "Python\Pythonwin Debugger
27           win32ui.SetAppName(win32ui.LoadString(win32ui.IDR_DEBUGGER))
28           win32ui.SetRegistryKey(f"Python {sys.winver}")
29           # We _need_ the Scintilla color editor.
30           # (and we _always_ get it now :-)
31   
32           numMRU = win32ui.GetProfileVal("Settings", "Recent File List Size", 10)
33           win32ui.LoadStdProfileSettings(numMRU)
34   
35           self.LoadMainFrame()
36   
37           # Display the interactive window if the user wants it.
38           from pywin.framework import interact
39   
40           interact.CreateInteractiveWindowUserPreference()
41   
42           # Load the modules we use internally.
43           self.LoadSystemModules()
44           # Load additional module the user may want.
45           self.LoadUserModules()
46   
47           #       win32ui.CreateDebuggerThread()
48           win32ui.EnableControlContainer()
