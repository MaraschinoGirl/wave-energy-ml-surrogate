183  @keras_export("keras.metrics.RootMeanSquaredError")
184  class RootMeanSquaredError(reduction_metrics.Mean):
185      """Computes root mean squared error metric between `y_true` and `y_pred`. 
186   
187      Formula: 
188   
189      ```python 
190      loss = sqrt(mean((y_pred - y_true) ** 2)) 
191      ``` 
192   
193      Args: 
194          name: (Optional) string name of the metric instance. 
195          dtype: (Optional) data type of the metric result. 
196   
197      Examples: 
198   
199      >>> m = keras.metrics.RootMeanSquaredError() 
200      >>> m.update_state([[0, 1], [0, 0]], [[1, 1], [0, 0]]) 
201      >>> m.result() 
202      0.5 
203   
204      >>> m.reset_state() 
205      >>> m.update_state([[0, 1], [0, 0]], [[1, 1], [0, 0]], 
206      ...                sample_weight=[1, 0]) 
207      >>> m.result() 
208      0.70710677 
209   
210      Usage with `compile()` API: 
211   
212      ```python 
213      model.compile( 
214          optimizer='sgd', 
215          loss='mse', 
216          metrics=[keras.metrics.RootMeanSquaredError()]) 
217      ``` 
218      """
219  
220      def __init__(self, name="root_mean_squared_error", dtype=None):
221          super().__init__(name, dtype=dtype)
222          # Metric should be minimized during optimization.
223          self._direction = "down"
224  
225      def update_state(self, y_true, y_pred, sample_weight=None):
226          """Accumulates root mean squared error statistics. 
227   
228          Args: 
229              y_true: The ground truth values. 
230              y_pred: The predicted values. 
231              sample_weight: Optional weighting of each example. Can 
232                  be a `Tensor` whose rank is either 0, or the same rank as 
233                  `y_true`, and must be broadcastable to `y_true`. 
234                  Defaults to `1`. 
235   
236          Returns: 
237              Update op. 
238          """
239          y_true = ops.convert_to_tensor(y_true, self._dtype)
240          y_pred = ops.convert_to_tensor(y_pred, self._dtype)
241          y_true, y_pred = squeeze_or_expand_to_same_rank(y_true, y_pred)
242          error_sq = ops.square(y_pred - y_true)
243          return super().update_state(error_sq, sample_weight=sample_weight)
244  
245      def result(self):
246          return ops.sqrt(super().result())

250  class CosineSimilarity(reduction_metrics.MeanMetricWrapper):
251      """Computes the cosine similarity between the labels and predictions. 
252   
253      Formula: 
254   
255      ```python 
256      loss = sum(l2_norm(y_true) * l2_norm(y_pred)) 
257      ``` 
258      See: [Cosine Similarity](https://en.wikipedia.org/wiki/Cosine_similarity). 
259      This metric keeps the average cosine similarity between `predictions` and 
260      `labels` over a stream of data. 
261   
262      Args: 
263          name: (Optional) string name of the metric instance. 
264          dtype: (Optional) data type of the metric result. 
265          axis: (Optional) Defaults to `-1`. The dimension along which the cosine 
266              similarity is computed. 
267   
268      Examples: 
269   
270      >>> # l2_norm(y_true) = [[0., 1.], [1./1.414, 1./1.414]] 
271      >>> # l2_norm(y_pred) = [[1., 0.], [1./1.414, 1./1.414]] 
272      >>> # l2_norm(y_true) . l2_norm(y_pred) = [[0., 0.], [0.5, 0.5]] 
273      >>> # result = mean(sum(l2_norm(y_true) . l2_norm(y_pred), axis=1)) 
274      >>> #        = ((0. + 0.) +  (0.5 + 0.5)) / 2 
275      >>> m = keras.metrics.CosineSimilarity(axis=1) 
276      >>> m.update_state([[0., 1.], [1., 1.]], [[1., 0.], [1., 1.]]) 
277      >>> m.result() 
278      0.49999997 
279   
280      >>> m.reset_state() 
281      >>> m.update_state([[0., 1.], [1., 1.]], [[1., 0.], [1., 1.]], 
282      ...                sample_weight=[0.3, 0.7]) 
283      >>> m.result() 
284      0.6999999 
285   
286      Usage with `compile()` API: 
287   
288      ```python 
289      model.compile( 
290          optimizer='sgd', 
291          loss='mse', 
292          metrics=[keras.metrics.CosineSimilarity(axis=1)]) 
293      ``` 
294      """
295  
296      def __init__(self, name="cosine_similarity", dtype=None, axis=-1):
297          super().__init__(cosine_similarity, name, dtype=dtype, axis=axis)
298          # Metric should be maximized during optimization.
299          self._direction = "up"
300  
301      def get_config(self):
302          return {"name": self.name, "dtype": self.dtype}

352  @keras_export("keras.metrics.R2Score")
353  class R2Score(reduction_metrics.Metric):
354      """Computes R2 score. 
355   
356      Formula: 
357   
358      ```python 
359      sum_squares_residuals = sum((y_true - y_pred) ** 2) 
360      sum_squares = sum((y_true - mean(y_true)) ** 2) 
361      R2 = 1 - sum_squares_residuals / sum_squares 
362      ``` 
363   
364      This is also called the 
365      [coefficient of determination]( 
366      https://en.wikipedia.org/wiki/Coefficient_of_determination). 
367   
368      It indicates how close the fitted regression line 
369      is to ground-truth data. 
370   
371      - The highest score possible is 1.0. It indicates that the predictors 
372          perfectly accounts for variation in the target. 
373      - A score of 0.0 indicates that the predictors do not 
374          account for variation in the target. 
375      - It can also be negative if the model is worse than random. 
376   
377      This metric can also compute the "Adjusted R2" score. 
378   
379      Args: 
380          class_aggregation: Specifies how to aggregate scores corresponding to 
381              different output classes (or target dimensions), 
382              i.e. different dimensions on the last axis of the predictions. 
383              Equivalent to `multioutput` argument in Scikit-Learn. 
384              Should be one of 
385              `None` (no aggregation), `"uniform_average"`, 
386              `"variance_weighted_average"`. 
387          num_regressors: Number of independent regressors used 
388              ("Adjusted R2" score). 0 is the standard R2 score. 
389              Defaults to `0`. 
390          name: Optional. string name of the metric instance. 
391          dtype: Optional. data type of the metric result. 
392   
393      Example: 
394   
395      >>> y_true = np.array([[1], [4], [3]], dtype=np.float32) 
396      >>> y_pred = np.array([[2], [4], [4]], dtype=np.float32) 
397      >>> metric = keras.metrics.R2Score() 
398      >>> metric.update_state(y_true, y_pred) 
399      >>> result = metric.result() 
400      >>> result 
401      0.57142854 
402      """
403  
404      def __init__(
405          self,
406          class_aggregation="uniform_average",
407          num_regressors=0,
408          name="r2_score",
409          dtype=None,
410      ):
411          super().__init__(name=name, dtype=dtype)
412          # Metric should be maximized during optimization.
413          self._direction = "up"
414  
415          valid_class_aggregation_values = (
416              None,
417              "uniform_average",
418              "variance_weighted_average",
419          )
420          if class_aggregation not in valid_class_aggregation_values:
421              raise ValueError(
422                  "Invalid value for argument `class_aggregation`. Expected "
423                  f"one of {valid_class_aggregation_values}. "
424                  f"Received: class_aggregation={class_aggregation}"
425              )
426          if num_regressors < 0:
427              raise ValueError(
428                  "Invalid value for argument `num_regressors`. "
429                  "Expected a value >= 0. "
430                  f"Received: num_regressors={num_regressors}"
431              )
432          self.class_aggregation = class_aggregation
433          self.num_regressors = num_regressors
434          self.num_samples = self.add_variable(
435              shape=(),
436              initializer=initializers.Zeros(),
437              name="num_samples",
438          )
439          self._built = False
440  
441      def _build(self, y_true_shape, y_pred_shape):
442          if len(y_pred_shape) != 2 or len(y_true_shape) != 2:
443              raise ValueError(
444                  "R2Score expects 2D inputs with shape "
445                  "(batch_size, output_dim). Received input "
446                  f"shapes: y_pred.shape={y_pred_shape} and "
447                  f"y_true.shape={y_true_shape}."
448              )
449          if y_pred_shape[-1] is None or y_true_shape[-1] is None:
450              raise ValueError(
451                  "R2Score expects 2D inputs with shape "
452                  "(batch_size, output_dim), with output_dim fully "
453                  "defined (not None). Received input "
454                  f"shapes: y_pred.shape={y_pred_shape} and "
455                  f"y_true.shape={y_true_shape}."
456              )
457          num_classes = y_pred_shape[-1]
458          self.squared_sum = self.add_variable(
459              name="squared_sum",
460              shape=[num_classes],
461              initializer=initializers.Zeros(),
462          )
463          self.sum = self.add_variable(
464              name="sum",
465              shape=[num_classes],
466              initializer=initializers.Zeros(),
467          )
468          self.total_mse = self.add_variable(
469              name="residual",
470              shape=[num_classes],
471              initializer=initializers.Zeros(),
472          )
473          self.count = self.add_variable(
474              name="count",
475              shape=[num_classes],
476              initializer=initializers.Zeros(),
477          )
478          self._built = True
479  
480      def update_state(self, y_true, y_pred, sample_weight=None):
481          """Accumulates root mean squared error statistics. 
482   
483          Args: 
484              y_true: The ground truth values. 
485              y_pred: The predicted values. 
486              sample_weight: Optional weighting of each example. Can 
487                  be a `Tensor` whose rank is either 0, or the same rank as 
488                  `y_true`, and must be broadcastable to `y_true`. 
489                  Defaults to `1`. 
490   
491          Returns: 
492              Update op. 
493          """
494          y_true = ops.convert_to_tensor(y_true, dtype=self._dtype)
495          y_pred = ops.convert_to_tensor(y_pred, dtype=self._dtype)
496          y_true, y_pred = squeeze_or_expand_to_same_rank(y_true, y_pred)
497          if not self._built:
498              self._build(y_true.shape, y_pred.shape)
499  
500          if sample_weight is None:
501              sample_weight = 1
502  
503          sample_weight = ops.convert_to_tensor(sample_weight, dtype=self.dtype)
504  
505          if len(sample_weight.shape) == 1:
506              # Make sure there's a features dimension
507              sample_weight = ops.expand_dims(sample_weight, axis=1)
508  
509          sample_weight = ops.broadcast_to(sample_weight, ops.shape(y_true))
510  
511          weighted_y_true = y_true * ops.cast(sample_weight, y_true.dtype)
512          self.sum.assign(self.sum + ops.sum(weighted_y_true, axis=0))
513          self.squared_sum.assign(
514              self.squared_sum + ops.sum(y_true * weighted_y_true, axis=0)
515          )
516          self.total_mse.assign(
517              self.total_mse
518              + ops.sum(
519                  (y_true - y_pred) ** 2 * ops.cast(sample_weight, y_true.dtype),
520                  axis=0,
521              )
522          )
523          self.count.assign(self.count + ops.sum(sample_weight, axis=0))
524          self.num_samples.assign(self.num_samples + ops.size(y_true))
525  
526      def result(self):
527          mean = self.sum / self.count
528          total = self.squared_sum - self.sum * mean
529          raw_scores = 1 - (self.total_mse / total)
530          raw_scores = ops.where(ops.isinf(raw_scores), 0.0, raw_scores)
531  
532          if self.class_aggregation == "uniform_average":
533              r2_score = ops.mean(raw_scores)
534          elif self.class_aggregation == "variance_weighted_average":
535              weighted_sum = ops.sum(total * raw_scores)
536              sum_of_weights = ops.sum(total)
537              r2_score = weighted_sum / sum_of_weights
538          else:
539              r2_score = raw_scores
540  
541          if self.num_regressors != 0:
542              if self.num_regressors > self.num_samples - 1:
543                  warnings.warn(
544                      "More independent predictors than datapoints "
545                      "in adjusted R2 score. Falling back to standard R2 score.",
546                      stacklevel=2,
547                  )
548              elif self.num_regressors == self.num_samples - 1:
549                  warnings.warn(
550                      "Division by zero in Adjusted R2 score. "
551                      "Falling back to standard R2 score.",
552                      stacklevel=2,
553                  )
554              else:
555                  n = ops.convert_to_tensor(self.num_samples, dtype="float32")
556                  p = ops.convert_to_tensor(self.num_regressors, dtype="float32")
557                  num = ops.multiply(
558                      ops.subtract(1.0, r2_score), ops.subtract(n, 1.0)
559                  )
560                  den = ops.subtract(ops.subtract(n, p), 1.0)
561                  r2_score = ops.subtract(1.0, ops.divide(num, den))
562          return r2_score
563  
564      def reset_state(self):
565          for v in self.variables:
566              v.assign(ops.zeros(v.shape, dtype=v.dtype))
567  
568      def get_config(self):
569          config = {
570              "name": self.name,
571              "dtype": self.dtype,
572              "class_aggregation": self.class_aggregation,
573              "num_regressors": self.num_regressors,
574          }
575          base_config = super().get_config()
576          return {**base_config, **config}

579  def cosine_similarity(y_true, y_pred, axis=-1):
580      """Computes the cosine similarity between labels and predictions. 
581   
582      Formula: 
583   
584      ```python 
585      loss = sum(l2_norm(y_true) * l2_norm(y_pred)) 
586      ``` 
587   
588      Args: 
589          y_true: Tensor of true targets. 
590          y_pred: Tensor of predicted targets. 
591          axis: Axis along which to determine similarity. Defaults to `-1`. 
592   
593      Returns: 
594          Cosine similarity tensor. 
595   
596      Example: 
597   
598      >>> y_true = [[0., 1.], [1., 1.], [1., 1.]] 
599      >>> y_pred = [[1., 0.], [1., 1.], [-1., -1.]] 
600      >>> loss = keras.losses.cosine_similarity(y_true, y_pred, axis=-1) 
601      [0., 0.99999994, -0.99999994] 
602      """
603      y_pred = ops.convert_to_tensor(y_pred)
604      y_true = ops.convert_to_tensor(y_true, dtype=y_pred.dtype)
605      y_true, y_pred = squeeze_or_expand_to_same_rank(y_true, y_pred)
606      y_pred = normalize(y_pred, axis=axis)
607      y_true = normalize(y_true, axis=axis)
608      return ops.sum(y_true * y_pred, axis=axis)
609
