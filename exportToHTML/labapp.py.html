144  class LabBuildApp(JupyterApp, DebugLogFileMixin):
145      version = version
146      description = """ 
147      Build the JupyterLab application 
148   
149      The application is built in the JupyterLab app directory in `/staging`. 
150      When the build is complete it is put in the JupyterLab app `/static` 
151      directory, where it is used to serve the application. 
152      """
153      aliases = build_aliases
154      flags = build_flags
155  
156      # Not configurable!
157      core_config = Instance(CoreConfig, allow_none=True)
158  
159      app_dir = Unicode("", config=True, help="The app directory to build in")
160  
161      name = Unicode("JupyterLab", config=True, help="The name of the built application")
162  
163      version = Unicode("", config=True, help="The version of the built application")
164  
165      dev_build = Bool(
166          None,
167          allow_none=True,
168          config=True,
169          help="Whether to build in dev mode. Defaults to True (dev mode) if there are any locally linked extensions, else defaults to False (production mode).",
170      )
171  
172      minimize = Bool(
173          True,
174          config=True,
175          help="Whether to minimize a production build (defaults to True).",
176      )
177  
178      pre_clean = Bool(
179          False, config=True, help="Whether to clean before building (defaults to False)"
180      )
181  
182      splice_source = Bool(False, config=True, help="Splice source packages into app directory.")
183  
184      def start(self):
185          app_dir = self.app_dir or get_app_dir()
186          app_options = AppOptions(
187              app_dir=app_dir,
188              logger=self.log,
189              core_config=self.core_config,
190              splice_source=self.splice_source,
191          )
192          self.log.info(f"JupyterLab {version}")
193          with self.debug_logging():
194              if self.pre_clean:
195                  self.log.info(f"Cleaning {app_dir}")
196                  clean(app_options=app_options)
197              self.log.info(f"Building in {app_dir}")
198              try:
199                  production = None if self.dev_build is None else not self.dev_build
200                  build(
201                      name=self.name,
202                      version=self.version,
203                      app_options=app_options,
204                      production=production,
205                      minimize=self.minimize,
206                  )
207              except Exception as e:
208                  self.log.error(build_failure_msg)
209                  raise e

684          if self.dev_mode or self.app_dir.startswith(DEV_DIR + os.sep):
685              self.dev_mode = True
686              self.log.info("Running JupyterLab in dev mode")
687  
688          if self.watch and self.core_mode:
689              self.log.warning("Cannot watch in core mode, did you mean --dev-mode?")
690              self.watch = False
691  
692          if self.core_mode and self.dev_mode:
693              self.log.warning("Conflicting modes, choosing dev_mode over core_mode")
694              self.core_mode = False
695  
696          # Set the paths based on JupyterLab's mode.
697          if self.dev_mode:
698              dev_static_dir = ujoin(DEV_DIR, "static")
699              self.static_paths = [dev_static_dir]
700              self.template_paths = [dev_static_dir]
701              if not self.extensions_in_dev_mode:
702                  # Add an exception for @jupyterlab/galata-extension
703                  galata_extension = pjoin(HERE, "galata")
704                  self.labextensions_path = (
705                      [galata_extension]
706                      if galata_extension in map(os.path.abspath, self.labextensions_path)
707                      else []
708                  )
709                  self.extra_labextensions_path = (
710                      [galata_extension]
711                      if galata_extension in map(os.path.abspath, self.extra_labextensions_path)
712                      else []
713                  )
714          elif self.core_mode:
715              dev_static_dir = ujoin(HERE, "static")
716              self.static_paths = [dev_static_dir]
717              self.template_paths = [dev_static_dir]
718              self.labextensions_path = []
719              self.extra_labextensions_path = []
720          else:
721              self.static_paths = [self.static_dir]
722              self.template_paths = [self.templates_dir]
723  
724      def _prepare_templates(self):
725          super()._prepare_templates()
726          self.jinja2_env.globals.update(custom_css=self.custom_css)

728      def initialize_handlers(self):  # noqa
729          handlers = []
730  
731          # Set config for Jupyterlab
732          page_config = self.serverapp.web_app.settings.setdefault("page_config_data", {})
733          page_config.update(get_static_page_config(logger=self.log, level="all"))
734  
735          page_config.setdefault("buildAvailable", not self.core_mode and not self.dev_mode)
736          page_config.setdefault("buildCheck", not self.core_mode and not self.dev_mode)
737          page_config["devMode"] = self.dev_mode
738          page_config["token"] = self.serverapp.identity_provider.token
739          page_config["exposeAppInBrowser"] = self.expose_app_in_browser
740          page_config["quitButton"] = self.serverapp.quit_button
741          page_config["allow_hidden_files"] = self.serverapp.contents_manager.allow_hidden
742          if hasattr(self.serverapp.contents_manager, "delete_to_trash"):
743              page_config["delete_to_trash"] = self.serverapp.contents_manager.delete_to_trash
744  
745          # Client-side code assumes notebookVersion is a JSON-encoded string
746          page_config["notebookVersion"] = json.dumps(jpserver_version_info)
747  
748          self.log.info(f"JupyterLab extension loaded from {HERE!s}")
749          self.log.info(f"JupyterLab application directory is {self.app_dir!s}")
750  
751          if self.custom_css:
752              handlers.append(
753                  (
754                      r"/custom/(.*)(?<!\.js)$",
755                      self.serverapp.web_app.settings["static_handler_class"],
756                      {
757                          "path": self.serverapp.web_app.settings["static_custom_path"],
758                          "no_cache_paths": ["/"],  # don't cache anything in custom
759                      },
760                  )
761              )
762  
763          app_options = AppOptions(
764              logger=self.log,
765              app_dir=self.app_dir,
766              labextensions_path=self.extra_labextensions_path + self.labextensions_path,
767              splice_source=self.splice_source,
768          )
769          builder = Builder(self.core_mode, app_options=app_options)
770          build_handler = (build_path, BuildHandler, {"builder": builder})
771          handlers.append(build_handler)
772  
773          errored = False
774  
775          if self.core_mode:
776              self.log.info(CORE_NOTE.strip())
777              ensure_core(self.log)
778          elif self.dev_mode:
779              if not (self.watch or self.skip_dev_build):
780                  ensure_dev(self.log)
781                  self.log.info(DEV_NOTE)
782          else:
783              if self.splice_source:
784                  ensure_dev(self.log)
785              msgs = ensure_app(self.app_dir)
786              if msgs:
787                  [self.log.error(msg) for msg in msgs]
788                  handler = (self.app_url, ErrorHandler, {"messages": msgs})
789                  handlers.append(handler)
790                  errored = True
791  
792          if self.watch:
793              self.log.info("Starting JupyterLab watch mode...")
794              if self.dev_mode:
795                  watch_dev(self.log)
796              else:
797                  watch(app_options=app_options)
798                  page_config["buildAvailable"] = False
799              self.cache_files = False
800  
801          if not self.core_mode and not errored:
802              # Add extension management handlers
803              provider = self.extension_manager
804              entry_point = EXT_MANAGERS.get(provider)
805              if entry_point is None:
806                  self.log.error(f"Extension Manager: No manager defined for provider '{provider}'.")
807                  raise NotImplementedError()
808              else:
809                  self.log.info(f"Extension Manager is '{provider}'.")
810              manager_factory = entry_point.load()
811              config = self.settings.get("config", {}).get("LabServerApp", {})
812  
813              blocked_extensions_uris = config.get("blocked_extensions_uris", "")
814              allowed_extensions_uris = config.get("allowed_extensions_uris", "")
815  
816              if (blocked_extensions_uris) and (allowed_extensions_uris):
817                  self.log.error(
818                      "Simultaneous LabServerApp.blocked_extensions_uris and LabServerApp.allowed_extensions_uris is not supported. Please define only one of those."
819                  )
820                  import sys
821  
822                  sys.exit(-1)
823  
824              listings_config = {
825                  "blocked_extensions_uris": set(
826                      filter(lambda uri: len(uri) > 0, blocked_extensions_uris.split(","))
827                  ),
828                  "allowed_extensions_uris": set(
829                      filter(lambda uri: len(uri) > 0, allowed_extensions_uris.split(","))
830                  ),
831                  "listings_refresh_seconds": config.get("listings_refresh_seconds", 60 * 60),
832                  "listings_tornado_options": config.get("listings_tornado_options", {}),
833              }
834              if len(listings_config["blocked_extensions_uris"]) or len(
835                  listings_config["allowed_extensions_uris"]
836              ):
837                  self.log.debug(f"Extension manager will be constrained by {listings_config}")
838  
839              try:
840                  ext_manager = manager_factory(app_options, listings_config, self)
841                  metadata = dataclasses.asdict(ext_manager.metadata)
842              except Exception as err:
843                  self.log.warning(
844                      f"Failed to instantiate the extension manager {provider}. Falling back to read-only manager.",
845                      exc_info=err,
846                  )
847                  ext_manager = ReadOnlyExtensionManager(app_options, listings_config, self)
848                  metadata = dataclasses.asdict(ext_manager.metadata)
849  
850              page_config["extensionManager"] = metadata
851              ext_handler = (
852                  extensions_handler_path,
853                  ExtensionHandler,
854                  {"manager": ext_manager},
855              )
856              handlers.append(ext_handler)
857  
858              # Add plugin manager handlers
859              lock_rules = frozenset(
860                  {rule for rule, value in page_config.get("lockedExtensions", {}).items() if value}
861              )
862              handlers.append(
863                  (
864                      plugins_handler_path,
865                      PluginHandler,
866                      {
867                          "manager": PluginManager(
868                              app_options=app_options,
869                              ext_options={
870                                  "lock_rules": lock_rules,
871                                  "all_locked": self.lock_all_plugins,
872                              },
873                              parent=self,
874                          )
875                      },
876                  )
877              )
878  
879              # Add announcement handlers
880              page_config["news"] = {"disabled": self.news_url is None}
881              handlers.extend(
882                  [
883                      (
884                          check_update_handler_path,
885                          CheckForUpdateHandler,
886                          {
887                              "update_checker": self.check_for_updates_class(__version__),
888                          },
889                      ),
890                      (
891                          news_handler_path,
892                          NewsHandler,
893                          {
894                              "news_url": self.news_url,
895                          },
896                      ),
897                  ]
898              )
899  
900          # If running under JupyterHub, add more metadata.
901          if "hub_prefix" in self.serverapp.tornado_settings:
902              tornado_settings = self.serverapp.tornado_settings
903              hub_prefix = tornado_settings["hub_prefix"]
904              page_config["hubPrefix"] = hub_prefix
905              page_config["hubHost"] = tornado_settings["hub_host"]
906              page_config["hubUser"] = tornado_settings["user"]
907              page_config["shareUrl"] = ujoin(hub_prefix, "user-redirect")
908              # Assume the server_name property indicates running JupyterHub 1.0.
909              if hasattr(self.serverapp, "server_name"):
910                  page_config["hubServerName"] = self.serverapp.server_name
911              # avoid setting API token in page config
912              # $JUPYTERHUB_API_TOKEN identifies the server, not the client
913              # but at least make sure we don't use the token
914              # if the serverapp set one
915              page_config["token"] = ""
916  
917          # Update Jupyter Server's webapp settings with jupyterlab settings.
918          self.serverapp.web_app.settings["page_config_data"] = page_config
919  
920          # Extend Server handlers with jupyterlab handlers.
921          self.handlers.extend(handlers)
922          super().initialize_handlers()
923
