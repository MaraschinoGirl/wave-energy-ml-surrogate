15   class PydevPlugin(Plugin):
16       def __init__(self, configuration):
17           self.configuration = configuration
18           Plugin.__init__(self)
19   
20       def begin(self):
21           # Called before any test is run (it's always called, with multiprocess or not)
22           self.start_time = time.time()
23           self.coverage_files, self.coverage = start_coverage_support(self.configuration)
24   
25       def finalize(self, result):
26           # Called after all tests are run (it's always called, with multiprocess or not)
27           self.coverage.stop()
28           self.coverage.save()
29   
30           pydev_runfiles_xml_rpc.notifyTestRunFinished("Finished in: %.2f secs." % (time.time() - self.start_time,))
31   
32       # ===================================================================================================================
33       # Methods below are not called with multiprocess (so, we monkey-patch MultiProcessTestRunner.consolidate
34       # so that they're called, but unfortunately we loose some info -- i.e.: the time for each test in this
35       # process).
36       # ===================================================================================================================
37   
38       class Sentinel(object):
39           pass
40   
41       @contextmanager
42       def _without_user_address(self, test):
43           # #PyDev-1095: Conflict between address in test and test.address() in PydevPlugin().report_cond()
44           user_test_instance = test.test
45           user_address = self.Sentinel
46           user_class_address = self.Sentinel
47           try:
48               if "address" in user_test_instance.__dict__:
49                   user_address = user_test_instance.__dict__.pop("address")
50           except:
51               # Just ignore anything here.
52               pass
53           try:
54               user_class_address = user_test_instance.__class__.address
55               del user_test_instance.__class__.address
56           except:
57               # Just ignore anything here.
58               pass
59   
60           try:
61               yield
62           finally:
63               if user_address is not self.Sentinel:
64                   user_test_instance.__dict__["address"] = user_address
65   
66               if user_class_address is not self.Sentinel:
67                   user_test_instance.__class__.address = user_class_address
68   
69       def _get_test_address(self, test):
70           try:
71               if hasattr(test, "address"):
72                   with self._without_user_address(test):
73                       address = test.address()
74   
75                   # test.address() is something as:
76                   # ('D:\\workspaces\\temp\\test_workspace\\pytesting1\\src\\mod1\\hello.py', 'mod1.hello', 'TestCase.testMet1')
77                   #
78                   # and we must pass: location, test
79                   #    E.g.: ['D:\\src\\mod1\\hello.py', 'TestCase.testMet1']
80                   address = address[0], address[2]
81               else:
82                   # multiprocess
83                   try:
84                       address = test[0], test[1]
85                   except TypeError:
86                       # It may be an error at setup, in which case it's not really a test, but a Context object.
87                       f = test.context.__file__
88                       if f.endswith(".pyc"):
89                           f = f[:-1]
90                       elif f.endswith("$py.class"):
91                           f = f[: -len("$py.class")] + ".py"
92                       address = f, "?"
93           except:
94               sys.stderr.write("PyDev: Internal pydev error getting test address. Please report at the pydev bug tracker\n")
95               traceback.print_exc()
96               sys.stderr.write("\n\n\n")
97               address = "?", "?"
98           return address
99   
100      def report_cond(self, cond, test, captured_output, error=""):
101          """ 
102          @param cond: fail, error, ok 
103          """
104  
105          address = self._get_test_address(test)
106  
107          error_contents = self.get_io_from_error(error)
108          try:
109              time_str = "%.2f" % (time.time() - test._pydev_start_time)
110          except:
111              time_str = "?"
112  
113          pydev_runfiles_xml_rpc.notifyTest(cond, captured_output, error_contents, address[0], address[1], time_str)
114  
115      def startTest(self, test):
116          test._pydev_start_time = time.time()
117          file, test = self._get_test_address(test)
118          pydev_runfiles_xml_rpc.notifyStartTest(file, test)
119  
120      def get_io_from_error(self, err):
121          if type(err) == type(()):
122              if len(err) != 3:
123                  if len(err) == 2:
124                      return err[1]  # multiprocess
125              s = StringIO()
126              etype, value, tb = err
127              if isinstance(value, str):
128                  return value
129              traceback.print_exception(etype, value, tb, file=s)
130              return s.getvalue()
131          return err
132  
133      def get_captured_output(self, test):
134          if hasattr(test, "capturedOutput") and test.capturedOutput:
135              return test.capturedOutput
136          return ""
137  
138      def addError(self, test, err):
139          self.report_cond(
140              "error",
141              test,
142              self.get_captured_output(test),
143              err,
144          )
145  
146      def addFailure(self, test, err):
147          self.report_cond(
148              "fail",
149              test,
150              self.get_captured_output(test),
151              err,
152          )
153  
154      def addSuccess(self, test):
155          self.report_cond(
156              "ok",
157              test,
158              self.get_captured_output(test),
159              "",
160          )

176  # new_consolidate
177  # =======================================================================================================================
178  def new_consolidate(self, result, batch_result):
179      """ 
180      Used so that it can work with the multiprocess plugin. 
181      Monkeypatched because nose seems a bit unsupported at this time (ideally 
182      the plugin would have this support by default). 
183      """
184      ret = original(self, result, batch_result)
185  
186      parent_frame = sys._getframe().f_back
187      # addr is something as D:\pytesting1\src\mod1\hello.py:TestCase.testMet4
188      # so, convert it to what report_cond expects
189      addr = parent_frame.f_locals["addr"]
190      i = addr.rindex(":")
191      addr = [addr[:i], addr[i + 1 :]]
192  
193      output, testsRun, failures, errors, errorClasses = batch_result
194      if failures or errors:
195          for failure in failures:
196              PYDEV_NOSE_PLUGIN_SINGLETON.report_cond("fail", addr, output, failure)
197  
198          for error in errors:
199              PYDEV_NOSE_PLUGIN_SINGLETON.report_cond("error", addr, output, error)
200      else:
201          PYDEV_NOSE_PLUGIN_SINGLETON.report_cond("ok", addr, output)
202  
203      return ret
204  
205  
206  MultiProcessTestRunner.consolidate = new_consolidate
