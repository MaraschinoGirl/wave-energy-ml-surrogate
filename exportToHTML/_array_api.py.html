22   def yield_namespaces(include_numpy_namespaces=True):
23       """Yield supported namespace. 
24    
25       This is meant to be used for testing purposes only. 
26    
27       Parameters 
28       ---------- 
29       include_numpy_namespaces : bool, default=True 
30           If True, also yield numpy namespaces. 
31    
32       Returns 
33       ------- 
34       array_namespace : str 
35           The name of the Array API namespace. 
36       """
37       for array_namespace in [
38           # The following is used to test the array_api_compat wrapper when
39           # array_api_dispatch is enabled: in particular, the arrays used in the
40           # tests are regular numpy arrays without any "device" attribute.
41           "numpy",
42           # Stricter NumPy-based Array API implementation. The
43           # array_api_strict.Array instances always have a dummy "device" attribute.
44           "array_api_strict",
45           "cupy",
46           "torch",
47       ]:
48           if not include_numpy_namespaces and array_namespace in _NUMPY_NAMESPACE_NAMES:
49               continue
50           yield array_namespace

53   def yield_namespace_device_dtype_combinations(include_numpy_namespaces=True):
54       """Yield supported namespace, device, dtype tuples for testing. 
55    
56       Use this to test that an estimator works with all combinations. 
57    
58       Parameters 
59       ---------- 
60       include_numpy_namespaces : bool, default=True 
61           If True, also yield numpy namespaces. 
62    
63       Returns 
64       ------- 
65       array_namespace : str 
66           The name of the Array API namespace. 
67    
68       device : str 
69           The name of the device on which to allocate the arrays. Can be None to 
70           indicate that the default value should be used. 
71    
72       dtype_name : str 
73           The name of the data type to use for arrays. Can be None to indicate 
74           that the default value should be used. 
75       """
76       for array_namespace in yield_namespaces(
77           include_numpy_namespaces=include_numpy_namespaces
78       ):
79           if array_namespace == "torch":
80               for device, dtype in itertools.product(
81                   ("cpu", "cuda"), ("float64", "float32")
82               ):
83                   yield array_namespace, device, dtype
84               yield array_namespace, "mps", "float32"
85           else:
86               yield array_namespace, None, None

89   def _check_array_api_dispatch(array_api_dispatch):
90       """Check that array_api_compat is installed and NumPy version is compatible. 
91    
92       array_api_compat follows NEP29, which has a higher minimum NumPy version than 
93       scikit-learn. 
94       """
95       if array_api_dispatch:
96           try:
97               import array_api_compat  # noqa
98           except ImportError:
99               raise ImportError(
100                  "array_api_compat is required to dispatch arrays using the API"
101                  " specification"
102              )
103  
104          numpy_version = parse_version(numpy.__version__)
105          min_numpy_version = "1.21"
106          if numpy_version < parse_version(min_numpy_version):
107              raise ImportError(
108                  f"NumPy must be {min_numpy_version} or newer (found"
109                  f" {numpy.__version__}) to dispatch array using"
110                  " the array API specification"
111              )
112  
113          scipy_version = parse_version(scipy.__version__)
114          min_scipy_version = "1.14.0"
115          if scipy_version < parse_version(min_scipy_version):
116              raise ImportError(
117                  f"SciPy must be {min_scipy_version} or newer"
118                  " (found {scipy.__version__}) to dispatch array using"
119                  " the array API specification"
120              )
121  
122          if os.environ.get("SCIPY_ARRAY_API") != "1":
123              raise RuntimeError(
124                  "Scikit-learn array API support was enabled but scipy's own support is "
125                  "not enabled. Please set the SCIPY_ARRAY_API=1 environment variable "
126                  "before importing sklearn or scipy. More details at: "
127                  "https://docs.scipy.org/doc/scipy/dev/api-dev/array_api.html"
128              )

131  def _single_array_device(array):
132      """Hardware device where the array data resides on."""
133      if (
134          isinstance(array, (numpy.ndarray, numpy.generic))
135          or not hasattr(array, "device")
136          # When array API dispatch is disabled, we expect the scikit-learn code
137          # to use np.asarray so that the resulting NumPy array will implicitly use the
138          # CPU. In this case, scikit-learn should stay as device neutral as possible,
139          # hence the use of `device=None` which is accepted by all libraries, before
140          # and after the expected conversion to NumPy via np.asarray.
141          or not get_config()["array_api_dispatch"]
142      ):
143          return None
144      else:
145          return array.device

148  def device(*array_list, remove_none=True, remove_types=(str,)):
149      """Hardware device where the array data resides on. 
150   
151      If the hardware device is not the same for all arrays, an error is raised. 
152   
153      Parameters 
154      ---------- 
155      *array_list : arrays 
156          List of array instances from NumPy or an array API compatible library. 
157   
158      remove_none : bool, default=True 
159          Whether to ignore None objects passed in array_list. 
160   
161      remove_types : tuple or list, default=(str,) 
162          Types to ignore in array_list. 
163   
164      Returns 
165      ------- 
166      out : device 
167          `device` object (see the "Device Support" section of the array API spec). 
168      """
169      array_list = _remove_non_arrays(
170          *array_list, remove_none=remove_none, remove_types=remove_types
171      )
172  
173      if not array_list:
174          return None
175  
176      device_ = _single_array_device(array_list[0])
177  
178      # Note: here we cannot simply use a Python `set` as it requires
179      # hashable members which is not guaranteed for Array API device
180      # objects. In particular, CuPy devices are not hashable at the
181      # time of writing.
182      for array in array_list[1:]:
183          device_other = _single_array_device(array)
184          if device_ != device_other:
185              raise ValueError(
186                  f"Input arrays use different devices: {str(device_)}, "
187                  f"{str(device_other)}"
188              )
189  
190      return device_

225  def isdtype(dtype, kind, *, xp):
226      """Returns a boolean indicating whether a provided dtype is of type "kind". 
227   
228      Included in the v2022.12 of the Array API spec. 
229      https://data-apis.org/array-api/latest/API_specification/generated/array_api.isdtype.html 
230      """
231      if isinstance(kind, tuple):
232          return any(_isdtype_single(dtype, k, xp=xp) for k in kind)
233      else:
234          return _isdtype_single(dtype, kind, xp=xp)
235  
236  
237  def _isdtype_single(dtype, kind, *, xp):
238      if isinstance(kind, str):
239          if kind == "bool":
240              return dtype == xp.bool
241          elif kind == "signed integer":
242              return dtype in {xp.int8, xp.int16, xp.int32, xp.int64}
243          elif kind == "unsigned integer":
244              return dtype in {xp.uint8, xp.uint16, xp.uint32, xp.uint64}
245          elif kind == "integral":
246              return any(
247                  _isdtype_single(dtype, k, xp=xp)
248                  for k in ("signed integer", "unsigned integer")
249              )
250          elif kind == "real floating":
251              return dtype in supported_float_dtypes(xp)
252          elif kind == "complex floating":
253              # Some name spaces might not have support for complex dtypes.
254              complex_dtypes = set()
255              if hasattr(xp, "complex64"):
256                  complex_dtypes.add(xp.complex64)
257              if hasattr(xp, "complex128"):
258                  complex_dtypes.add(xp.complex128)
259              return dtype in complex_dtypes
260          elif kind == "numeric":
261              return any(
262                  _isdtype_single(dtype, k, xp=xp)
263                  for k in ("integral", "real floating", "complex floating")
264              )
265          else:
266              raise ValueError(f"Unrecognized data type kind: {kind!r}")
267      else:
268          return dtype == kind

271  def supported_float_dtypes(xp):
272      """Supported floating point types for the namespace. 
273   
274      Note: float16 is not officially part of the Array API spec at the 
275      time of writing but scikit-learn estimators and functions can choose 
276      to accept it when xp.float16 is defined. 
277   
278      https://data-apis.org/array-api/latest/API_specification/data_types.html 
279      """
280      if hasattr(xp, "float16"):
281          return (xp.float64, xp.float32, xp.float16)
282      else:
283          return (xp.float64, xp.float32)

286  def ensure_common_namespace_device(reference, *arrays):
287      """Ensure that all arrays use the same namespace and device as reference. 
288   
289      If necessary the arrays are moved to the same namespace and device as 
290      the reference array. 
291   
292      Parameters 
293      ---------- 
294      reference : array 
295          Reference array. 
296   
297      *arrays : array 
298          Arrays to check. 
299   
300      Returns 
301      ------- 
302      arrays : list 
303          Arrays with the same namespace and device as reference. 
304      """
305      xp, is_array_api = get_namespace(reference)
306  
307      if is_array_api:
308          device_ = device(reference)
309          # Move arrays to the same namespace and device as the reference array.
310          return [xp.asarray(a, device=device_) for a in arrays]
311      else:
312          return arrays

315  def _check_device_cpu(device):  # noqa
316      if device not in {"cpu", None}:
317          raise ValueError(f"Unsupported device for NumPy: {device!r}")
318  
319  
320  def _accept_device_cpu(func):
321      @wraps(func)
322      def wrapped_func(*args, **kwargs):
323          _check_device_cpu(kwargs.pop("device", None))
324          return func(*args, **kwargs)
325  
326      return wrapped_func

329  class _NumPyAPIWrapper:
330      """Array API compat wrapper for any numpy version 
331   
332      NumPy < 2 does not implement the namespace. NumPy 2 and later should 
333      progressively implement more an more of the latest Array API spec but this 
334      is still work in progress at this time. 
335   
336      This wrapper makes it possible to write code that uses the standard Array 
337      API while working with any version of NumPy supported by scikit-learn. 
338   
339      See the `get_namespace()` public function for more details. 
340      """
341  
342      # TODO: once scikit-learn drops support for NumPy < 2, this class can be
343      # removed, assuming Array API compliance of NumPy 2 is actually sufficient
344      # for scikit-learn's needs.
345  
346      # Creation functions in spec:
347      # https://data-apis.org/array-api/latest/API_specification/creation_functions.html
348      _CREATION_FUNCS = {
349          "arange",
350          "empty",
351          "empty_like",
352          "eye",
353          "full",
354          "full_like",
355          "linspace",
356          "ones",
357          "ones_like",
358          "zeros",
359          "zeros_like",
360      }
361      # Data types in spec
362      # https://data-apis.org/array-api/latest/API_specification/data_types.html
363      _DTYPES = {
364          "int8",
365          "int16",
366          "int32",
367          "int64",
368          "uint8",
369          "uint16",
370          "uint32",
371          "uint64",
372          # XXX: float16 is not part of the Array API spec but exposed by
373          # some namespaces.
374          "float16",
375          "float32",
376          "float64",
377          "complex64",
378          "complex128",
379      }
380  
381      def __getattr__(self, name):
382          attr = getattr(numpy, name)
383  
384          # Support device kwargs and make sure they are on the CPU
385          if name in self._CREATION_FUNCS:
386              return _accept_device_cpu(attr)
387  
388          # Convert to dtype objects
389          if name in self._DTYPES:
390              return numpy.dtype(attr)
391          return attr
392  
393      @property
394      def bool(self):
395          return numpy.bool_
396  
397      def astype(self, x, dtype, *, copy=True, casting="unsafe"):
398          # astype is not defined in the top level NumPy namespace
399          return x.astype(dtype, copy=copy, casting=casting)
400  
401      def asarray(self, x, *, dtype=None, device=None, copy=None):  # noqa
402          _check_device_cpu(device)
403          # Support copy in NumPy namespace
404          if copy is True:
405              return numpy.array(x, copy=True, dtype=dtype)
406          else:
407              return numpy.asarray(x, dtype=dtype)
408  
409      def unique_inverse(self, x):
410          return numpy.unique(x, return_inverse=True)
411  
412      def unique_counts(self, x):
413          return numpy.unique(x, return_counts=True)
414  
415      def unique_values(self, x):
416          return numpy.unique(x)
417  
418      def unique_all(self, x):
419          return numpy.unique(
420              x, return_index=True, return_inverse=True, return_counts=True
421          )
422  
423      def concat(self, arrays, *, axis=None):
424          return numpy.concatenate(arrays, axis=axis)
425  
426      def reshape(self, x, shape, *, copy=None):
427          """Gives a new shape to an array without changing its data. 
428   
429          The Array API specification requires shape to be a tuple. 
430          https://data-apis.org/array-api/latest/API_specification/generated/array_api.reshape.html 
431          """
432          if not isinstance(shape, tuple):
433              raise TypeError(
434                  f"shape must be a tuple, got {shape!r} of type {type(shape)}"
435              )
436  
437          if copy is True:
438              x = x.copy()
439          return numpy.reshape(x, shape)
440  
441      def isdtype(self, dtype, kind):
442          try:
443              return isdtype(dtype, kind, xp=self)
444          except TypeError:
445              # In older versions of numpy, data types that arise from outside
446              # numpy like from a Polars Series raise a TypeError.
447              # e.g. TypeError: Cannot interpret 'Int64' as a data type.
448              # Therefore, we return False.
449              # TODO: Remove when minimum supported version of numpy is >= 1.21.
450              return False
451  
452      def pow(self, x1, x2):
453          return numpy.power(x1, x2)

459  def _remove_non_arrays(*arrays, remove_none=True, remove_types=(str,)):
460      """Filter arrays to exclude None and/or specific types. 
461   
462      Raise ValueError if no arrays are left after filtering. 
463   
464      Sparse arrays are always filtered out. 
465   
466      Parameters 
467      ---------- 
468      *arrays : array objects 
469          Array objects. 
470   
471      remove_none : bool, default=True 
472          Whether to ignore None objects passed in arrays. 
473   
474      remove_types : tuple or list, default=(str,) 
475          Types to ignore in the arrays. 
476   
477      Returns 
478      ------- 
479      filtered_arrays : list 
480          List of arrays filtered as requested. An empty list is returned if no input 
481          passes the filters. 
482      """
483      filtered_arrays = []
484      remove_types = tuple(remove_types)
485      for array in arrays:
486          if remove_none and array is None:
487              continue
488          if isinstance(array, remove_types):
489              continue
490          if sp.issparse(array):
491              continue
492          filtered_arrays.append(array)
493  
494      return filtered_arrays

497  def get_namespace(*arrays, remove_none=True, remove_types=(str,), xp=None):
498      """Get namespace of arrays. 
499   
500      Introspect `arrays` arguments and return their common Array API compatible 
501      namespace object, if any. 
502   
503      Note that sparse arrays are filtered by default. 
504   
505      See: https://numpy.org/neps/nep-0047-array-api-standard.html 
506   
507      If `arrays` are regular numpy arrays, an instance of the `_NumPyAPIWrapper` 
508      compatibility wrapper is returned instead. 
509   
510      Namespace support is not enabled by default. To enabled it call: 
511   
512        sklearn.set_config(array_api_dispatch=True) 
513   
514      or: 
515   
516        with sklearn.config_context(array_api_dispatch=True): 
517            # your code here 
518   
519      Otherwise an instance of the `_NumPyAPIWrapper` compatibility wrapper is 
520      always returned irrespective of the fact that arrays implement the 
521      `__array_namespace__` protocol or not. 
522   
523      Note that if no arrays pass the set filters, ``_NUMPY_API_WRAPPER_INSTANCE, False`` 
524      is returned. 
525   
526      Parameters 
527      ---------- 
528      *arrays : array objects 
529          Array objects. 
530   
531      remove_none : bool, default=True 
532          Whether to ignore None objects passed in arrays. 
533   
534      remove_types : tuple or list, default=(str,) 
535          Types to ignore in the arrays. 
536   
537      xp : module, default=None 
538          Precomputed array namespace module. When passed, typically from a caller 
539          that has already performed inspection of its own inputs, skips array 
540          namespace inspection. 
541   
542      Returns 
543      ------- 
544      namespace : module 
545          Namespace shared by array objects. If any of the `arrays` are not arrays, 
546          the namespace defaults to NumPy. 
547   
548      is_array_api_compliant : bool 
549          True if the arrays are containers that implement the Array API spec. 
550          Always False when array_api_dispatch=False. 
551      """
552      array_api_dispatch = get_config()["array_api_dispatch"]
553      if not array_api_dispatch:
554          if xp is not None:
555              return xp, False
556          else:
557              return _NUMPY_API_WRAPPER_INSTANCE, False
558  
559      if xp is not None:
560          return xp, True
561  
562      arrays = _remove_non_arrays(
563          *arrays,
564          remove_none=remove_none,
565          remove_types=remove_types,
566      )
567  
568      if not arrays:
569          return _NUMPY_API_WRAPPER_INSTANCE, False
570  
571      _check_array_api_dispatch(array_api_dispatch)
572  
573      # array-api-compat is a required dependency of scikit-learn only when
574      # configuring `array_api_dispatch=True`. Its import should therefore be
575      # protected by _check_array_api_dispatch to display an informative error
576      # message in case it is missing.
577      import array_api_compat
578  
579      namespace, is_array_api_compliant = array_api_compat.get_namespace(*arrays), True
580  
581      if namespace.__name__ == "array_api_strict" and hasattr(
582          namespace, "set_array_api_strict_flags"
583      ):
584          namespace.set_array_api_strict_flags(api_version="2023.12")
585  
586      return namespace, is_array_api_compliant
587

589  def get_namespace_and_device(*array_list, remove_none=True, remove_types=(str,)):
590      """Combination into one single function of `get_namespace` and `device`. 
591   
592      Parameters 
593      ---------- 
594      *array_list : array objects 
595          Array objects. 
596      remove_none : bool, default=True 
597          Whether to ignore None objects passed in arrays. 
598      remove_types : tuple or list, default=(str,) 
599          Types to ignore in the arrays. 
600   
601      Returns 
602      ------- 
603      namespace : module 
604          Namespace shared by array objects. If any of the `arrays` are not arrays, 
605          the namespace defaults to NumPy. 
606      is_array_api_compliant : bool 
607          True if the arrays are containers that implement the Array API spec. 
608          Always False when array_api_dispatch=False. 
609      device : device 
610          `device` object (see the "Device Support" section of the array API spec). 
611      """
612      array_list = _remove_non_arrays(
613          *array_list,
614          remove_none=remove_none,
615          remove_types=remove_types,
616      )
617  
618      skip_remove_kwargs = dict(remove_none=False, remove_types=[])
619  
620      xp, is_array_api = get_namespace(*array_list, **skip_remove_kwargs)
621      arrays_device = device(*array_list, **skip_remove_kwargs)
622      if is_array_api:
623          return xp, is_array_api, arrays_device
624      else:
625          return xp, False, arrays_device

628  def _expit(X, xp=None):
629      xp, _ = get_namespace(X, xp=xp)
630      if _is_numpy_namespace(xp):
631          return xp.asarray(special.expit(numpy.asarray(X)))
632  
633      return 1.0 / (1.0 + xp.exp(-X))

636  def _fill_or_add_to_diagonal(array, value, xp, add_value=True, wrap=False):
637      """Implementation to facilitate adding or assigning specified values to the 
638      diagonal of a 2-d array. 
639   
640      If ``add_value`` is `True` then the values will be added to the diagonal 
641      elements otherwise the values will be assigned to the diagonal elements. 
642      By default, ``add_value`` is set to `True. This is currently only 
643      supported for 2-d arrays. 
644   
645      The implementation is taken from the `numpy.fill_diagonal` function: 
646      https://github.com/numpy/numpy/blob/v2.0.0/numpy/lib/_index_tricks_impl.py#L799-L929 
647      """
648      if array.ndim != 2:
649          raise ValueError(
650              f"array should be 2-d. Got array with shape {tuple(array.shape)}"
651          )
652  
653      value = xp.asarray(value, dtype=array.dtype, device=device(array))
654      end = None
655      # Explicit, fast formula for the common case.  For 2-d arrays, we
656      # accept rectangular ones.
657      step = array.shape[1] + 1
658      if not wrap:
659          end = array.shape[1] * array.shape[1]
660  
661      array_flat = xp.reshape(array, (-1,))
662      if add_value:
663          array_flat[:end:step] += value
664      else:
665          array_flat[:end:step] = value

668  def _max_precision_float_dtype(xp, device):
669      """Return the float dtype with the highest precision supported by the device."""
670      # TODO: Update to use `__array_namespace__info__()` from array-api v2023.12
671      # when/if that becomes more widespread.
672      xp_name = xp.__name__
673      if xp_name in {"array_api_compat.torch", "torch"} and (
674          str(device).startswith("mps")
675      ):  # pragma: no cover
676          return xp.float32
677      return xp.float64

680  def _find_matching_floating_dtype(*arrays, xp):
681      """Find a suitable floating point dtype when computing with arrays. 
682   
683      If any of the arrays are floating point, return the dtype with the highest 
684      precision by following official type promotion rules: 
685   
686      https://data-apis.org/array-api/latest/API_specification/type_promotion.html 
687   
688      If there are no floating point input arrays (all integral inputs for 
689      instance), return the default floating point dtype for the namespace. 
690      """
691      dtyped_arrays = [a for a in arrays if hasattr(a, "dtype")]
692      floating_dtypes = [
693          a.dtype for a in dtyped_arrays if xp.isdtype(a.dtype, "real floating")
694      ]
695      if floating_dtypes:
696          # Return the floating dtype with the highest precision:
697          return xp.result_type(*floating_dtypes)
698  
699      # If none of the input arrays have a floating point dtype, they must be all
700      # integer arrays or containers of Python scalars: return the default
701      # floating point dtype for the namespace (implementation specific).
702      return xp.asarray(0.0).dtype

705  def _average(a, axis=None, weights=None, normalize=True, xp=None):
706      """Partial port of np.average to support the Array API. 
707   
708      It does a best effort at mimicking the return dtype rule described at 
709      https://numpy.org/doc/stable/reference/generated/numpy.average.html but 
710      only for the common cases needed in scikit-learn. 
711      """
712      xp, _, device_ = get_namespace_and_device(a, weights)
713  
714      if _is_numpy_namespace(xp):
715          if normalize:
716              return xp.asarray(numpy.average(a, axis=axis, weights=weights))
717          elif axis is None and weights is not None:
718              return xp.asarray(numpy.dot(a, weights))
719  
720      a = xp.asarray(a, device=device_)
721      if weights is not None:
722          weights = xp.asarray(weights, device=device_)
723  
724      if weights is not None and a.shape != weights.shape:
725          if axis is None:
726              raise TypeError(
727                  f"Axis must be specified when the shape of a {tuple(a.shape)} and "
728                  f"weights {tuple(weights.shape)} differ."
729              )
730  
731          if tuple(weights.shape) != (a.shape[axis],):
732              raise ValueError(
733                  f"Shape of weights weights.shape={tuple(weights.shape)} must be "
734                  f"consistent with a.shape={tuple(a.shape)} and {axis=}."
735              )
736  
737          # If weights are 1D, add singleton dimensions for broadcasting
738          shape = [1] * a.ndim
739          shape[axis] = a.shape[axis]
740          weights = xp.reshape(weights, shape)
741  
742      if xp.isdtype(a.dtype, "complex floating"):
743          raise NotImplementedError(
744              "Complex floating point values are not supported by average."
745          )
746      if weights is not None and xp.isdtype(weights.dtype, "complex floating"):
747          raise NotImplementedError(
748              "Complex floating point values are not supported by average."
749          )
750  
751      output_dtype = _find_matching_floating_dtype(a, weights, xp=xp)
752      a = xp.astype(a, output_dtype)
753  
754      if weights is None:
755          return (xp.mean if normalize else xp.sum)(a, axis=axis)
756  
757      weights = xp.astype(weights, output_dtype)
758  
759      sum_ = xp.sum(xp.multiply(a, weights), axis=axis)
760  
761      if not normalize:
762          return sum_
763  
764      scale = xp.sum(weights, axis=axis)
765      if xp.any(scale == 0.0):
766          raise ZeroDivisionError("Weights sum to zero, can't be normalized")
767  
768      return sum_ / scale

772      # TODO: refactor once nan-aware reductions are standardized:
773      # https://github.com/data-apis/array-api/issues/621
774      xp, _ = get_namespace(X, xp=xp)
775      if _is_numpy_namespace(xp):
776          return xp.asarray(numpy.nanmin(X, axis=axis))

778      else:

782          mask = xp.all(mask, axis=axis)
783          if xp.any(mask):
784              X = xp.where(mask, xp.asarray(xp.nan), X)

818  def _asarray_with_order(
819      array, dtype=None, order=None, copy=None, *, xp=None, device=None
820  ):
821      """Helper to support the order kwarg only for NumPy-backed arrays 
822   
823      Memory layout parameter `order` is not exposed in the Array API standard, 
824      however some input validation code in scikit-learn needs to work both 
825      for classes and functions that will leverage Array API only operations 
826      and for code that inherently relies on NumPy backed data containers with 
827      specific memory layout constraints (e.g. our own Cython code). The 
828      purpose of this helper is to make it possible to share code for data 
829      container validation without memory copies for both downstream use cases: 
830      the `order` parameter is only enforced if the input array implementation 
831      is NumPy based, otherwise `order` is just silently ignored. 
832      """
833      xp, _ = get_namespace(array, xp=xp)
834      if _is_numpy_namespace(xp):
835          # Use NumPy API to support order
836          if copy is True:
837              array = numpy.array(array, order=order, dtype=dtype)
838          else:
839              array = numpy.asarray(array, order=order, dtype=dtype)
840  
841          # At this point array is a NumPy ndarray. We convert it to an array
842          # container that is consistent with the input's namespace.
843          return xp.asarray(array)
844      else:
845          return xp.asarray(array, dtype=dtype, copy=copy, device=device)

863      """Convert X into a NumPy ndarray on the CPU."""
864      xp_name = xp.__name__

874  def _estimator_with_converted_arrays(estimator, converter):
875      """Create new estimator which converting all attributes that are arrays. 
876   
877      The converter is called on all NumPy arrays and arrays that support the 
878      `DLPack interface <https://dmlc.github.io/dlpack/latest/>`__. 
879   
880      Parameters 
881      ---------- 
882      estimator : Estimator 
883          Estimator to convert 
884   
885      converter : callable 
886          Callable that takes an array attribute and returns the converted array. 
887   
888      Returns 
889      ------- 
890      new_estimator : Estimator 
891          Convert estimator 
892      """
893      from sklearn.base import clone
894  
895      new_estimator = clone(estimator)
896      for key, attribute in vars(estimator).items():
897          if hasattr(attribute, "__dlpack__") or isinstance(attribute, numpy.ndarray):
898              attribute = converter(attribute)
899          setattr(new_estimator, key, attribute)
900      return new_estimator

913  def indexing_dtype(xp):
914      """Return a platform-specific integer dtype suitable for indexing. 
915   
916      On 32-bit platforms, this will typically return int32 and int64 otherwise. 
917   
918      Note: using dtype is recommended for indexing transient array 
919      datastructures. For long-lived arrays, such as the fitted attributes of 
920      estimators, it is instead recommended to use platform-independent int32 if 
921      we do not expect to index more 2B elements. Using fixed dtypes simplifies 
922      the handling of serialized models, e.g. to deploy a model fit on a 64-bit 
923      platform to a target 32-bit platform such as WASM/pyodide. 
924      """
925      # Currently this is implemented with simple hack that assumes that
926      # following "may be" statements in the Array API spec always hold:
927      # > The default integer data type should be the same across platforms, but
928      # > the default may vary depending on whether Python is 32-bit or 64-bit.
929      # > The default array index data type may be int32 on 32-bit platforms, but
930      # > the default should be int64 otherwise.
931      # https://data-apis.org/array-api/latest/API_specification/data_types.html#default-data-types
932      # TODO: once sufficiently adopted, we might want to instead rely on the
933      # newer inspection API: https://github.com/data-apis/array-api/issues/640
934      return xp.asarray(0).dtype

937  def _searchsorted(a, v, *, side="left", sorter=None, xp=None):
938      # Temporary workaround needed as long as searchsorted is not widely
939      # adopted by implementers of the Array API spec. This is a quite
940      # recent addition to the spec:
941      # https://data-apis.org/array-api/latest/API_specification/generated/array_api.searchsorted.html # noqa
942      xp, _ = get_namespace(a, v, xp=xp)
943      if hasattr(xp, "searchsorted"):
944          return xp.searchsorted(a, v, side=side, sorter=sorter)

946      a_np = _convert_to_numpy(a, xp=xp)
947      v_np = _convert_to_numpy(v, xp=xp)
948      indices = numpy.searchsorted(a_np, v_np, side=side, sorter=sorter)
949      return xp.asarray(indices, device=device(a))

1006 # Note: This is a helper for the functions `_isin` and
1007 # `_setdiff1d`. It is not meant to be called directly.
1008 def _in1d(ar1, ar2, xp, assume_unique=False, invert=False):
1009     """Checks whether each element of an array is also present in a 
1010     second array. 
1011  
1012     Returns a boolean array the same length as `ar1` that is True 
1013     where an element of `ar1` is in `ar2` and False otherwise. 
1014  
1015     This function has been adapted using the original implementation 
1016     present in numpy: 
1017     https://github.com/numpy/numpy/blob/v1.26.0/numpy/lib/arraysetops.py#L524-L758 
1018     """
1019     xp, _ = get_namespace(ar1, ar2, xp=xp)
1020 
1021     # This code is run to make the code significantly faster
1022     if ar2.shape[0] < 10 * ar1.shape[0] ** 0.145:
1023         if invert:
1024             mask = xp.ones(ar1.shape[0], dtype=xp.bool, device=device(ar1))
1025             for a in ar2:
1026                 mask &= ar1 != a
1027         else:
1028             mask = xp.zeros(ar1.shape[0], dtype=xp.bool, device=device(ar1))
1029             for a in ar2:
1030                 mask |= ar1 == a
1031         return mask
1032 
1033     if not assume_unique:
1034         ar1, rev_idx = xp.unique_inverse(ar1)
1035         ar2 = xp.unique_values(ar2)
1036 
1037     ar = xp.concat((ar1, ar2))
1038     device_ = device(ar)
1039     # We need this to be a stable sort.
1040     order = xp.argsort(ar, stable=True)
1041     reverse_order = xp.argsort(order, stable=True)
1042     sar = xp.take(ar, order, axis=0)
1043     if invert:
1044         bool_ar = sar[1:] != sar[:-1]
1045     else:
1046         bool_ar = sar[1:] == sar[:-1]
1047     flag = xp.concat((bool_ar, xp.asarray([invert], device=device_)))
1048     ret = xp.take(flag, reverse_order, axis=0)
1049 
1050     if assume_unique:
1051         return ret[: ar1.shape[0]]
1052     else:
1053         return xp.take(ret, rev_idx, axis=0)

1056 def _count_nonzero(X, axis=None, sample_weight=None, xp=None, device=None):
1057     """A variant of `sklearn.utils.sparsefuncs.count_nonzero` for the Array API. 
1058  
1059     If the array `X` is sparse, and we are using the numpy namespace then we 
1060     simply call the original function. This function only supports 2D arrays. 
1061     """
1062     from .sparsefuncs import count_nonzero
1063 
1064     xp, _ = get_namespace(X, sample_weight, xp=xp)
1065     if _is_numpy_namespace(xp) and sp.issparse(X):
1066         return count_nonzero(X, axis=axis, sample_weight=sample_weight)
1067 
1068     assert X.ndim == 2
1069 
1070     weights = xp.ones_like(X, device=device)
1071     if sample_weight is not None:
1072         sample_weight = xp.asarray(sample_weight, device=device)
1073         sample_weight = xp.reshape(sample_weight, (sample_weight.shape[0], 1))
1074         weights = xp.astype(weights, sample_weight.dtype) * sample_weight

1077     return xp.sum(xp.where(X != 0, weights, zero_scalar), axis=axis)
