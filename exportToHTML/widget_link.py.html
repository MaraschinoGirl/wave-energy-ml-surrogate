9    from .widget import Widget, register, widget_serialization
10   from .widget_core import CoreWidget
11   
12   from traitlets import Unicode, Tuple, Instance, TraitError
13   
14   
15   class WidgetTraitTuple(Tuple):
16       """Traitlet for validating a single (Widget, 'trait_name') pair"""
17   
18       info_text = "A (Widget, 'trait_name') pair"
19   
20       def __init__(self, **kwargs):
21           super().__init__(Instance(Widget), Unicode(), **kwargs)
22           if "default_value" not in kwargs and not kwargs.get("allow_none", False):
23               # This is to keep consistent behavior for spec generation between traitlets 4 and 5
24               # Having a default empty container is explicitly not allowed in traitlets 5 when
25               # there are traits specified (as the default value will be invalid), but we do it
26               # anyway as there is no empty "default" that makes sense.
27               self.default_args = ()
28   
29       def validate_elements(self, obj, value):
30           value = super().validate_elements(obj, value)
31           widget, trait_name = value
32           trait = widget.traits().get(trait_name)
33           trait_repr = "{}.{}".format(widget.__class__.__name__, trait_name)
34           # Can't raise TraitError because the parent will swallow the message
35           # and throw it away in a new, less informative TraitError
36           if trait is None:
37               raise TypeError("No such trait: %s" % trait_repr)
38           elif not trait.metadata.get('sync'):
39               raise TypeError("%s cannot be synced" % trait_repr)
40           return value

43   @register
44   class Link(CoreWidget):
45       """Link Widget 
46    
47       source: a (Widget, 'trait_name') tuple for the source trait 
48       target: a (Widget, 'trait_name') tuple that should be updated 
49       """
50   
51       _model_name = Unicode('LinkModel').tag(sync=True)
52       target = WidgetTraitTuple(help="The target (widget, 'trait_name') pair").tag(sync=True, **widget_serialization)
53       source = WidgetTraitTuple(help="The source (widget, 'trait_name') pair").tag(sync=True, **widget_serialization)
54   
55       def __init__(self, source, target, **kwargs):
56           kwargs['source'] = source
57           kwargs['target'] = target
58           super().__init__(**kwargs)
59   
60       # for compatibility with traitlet links
61       def unlink(self):
62           self.close()
63   
64   
65   def jslink(attr1, attr2):
66       """Link two widget attributes on the frontend so they remain in sync. 
67    
68       The link is created in the front-end and does not rely on a roundtrip 
69       to the backend. 
70    
71       Parameters 
72       ---------- 
73       source : a (Widget, 'trait_name') tuple for the first trait 
74       target : a (Widget, 'trait_name') tuple for the second trait 
75    
76       Examples 
77       -------- 
78    
79       >>> c = link((widget1, 'value'), (widget2, 'value')) 
80       """
81       return Link(attr1, attr2)
82   
83   
84   @register
85   class DirectionalLink(Link):
86       """A directional link 
87    
88       source: a (Widget, 'trait_name') tuple for the source trait 
89       target: a (Widget, 'trait_name') tuple that should be updated 
90       when the source trait changes. 
91       """
92       _model_name = Unicode('DirectionalLinkModel').tag(sync=True)
93   
94   
95   def jsdlink(source, target):
96       """Link a source widget attribute with a target widget attribute. 
97    
98       The link is created in the front-end and does not rely on a roundtrip 
99       to the backend. 
100   
101      Parameters 
102      ---------- 
103      source : a (Widget, 'trait_name') tuple for the source trait 
104      target : a (Widget, 'trait_name') tuple for the target trait 
105   
106      Examples 
107      -------- 
108   
109      >>> c = dlink((src_widget, 'value'), (tgt_widget, 'value')) 
110      """
111      return DirectionalLink(source, target)
