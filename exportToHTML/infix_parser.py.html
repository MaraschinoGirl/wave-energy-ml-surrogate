5    # This file implements a simple "shunting yard algorithm" parser for infix
6    # languages with parentheses. It is used as the core of our parser for
7    # formulas, but is generic enough to be used for other purposes as well
8    # (e.g. parsing linear constraints). It just builds a parse tree; semantics
9    # are somebody else's problem.
10   #
11   # Plus it spends energy on tracking where each item in the parse tree comes
12   # from, to allow high-quality error reporting.
13   #
14   # You are expected to provide an collection of Operators, a collection of
15   # atomic types, and an iterator that provides Tokens. Each Operator should
16   # have a unique token_type (which is an arbitrary Python object), and each
17   # Token should have a matching token_type, or one of the special types
18   # Token.LPAREN, Token.RPAREN. Each Token is required to have a valid Origin
19   # attached, for error reporting.
20   
21   # XX: still seriously consider putting the magic intercept handling into the
22   # tokenizer. we'd still need separate term-sets that get pasted together by ~
23   # to create the modeldesc, though... heck maybe we should just have a
24   # modeldesc be 1-or-more termsets, with the convention that if it's 1, then
25   # it's a rhs, and if it's 2, it's (lhs, rhs), and otherwise you're on your
26   # own. Test: would this be useful for multiple-group log-linear models,
27   # maybe? Answer: Perhaps. outcome ~ x1 + x2 ~ group. But lots of other
28   # plausible, maybe better ways to write this -- (outcome | group) ~ x1 + x2?
29   # "outcome ~ x1 + x2", group="group"? etc.

43   class _UniqueValue:
44       def __init__(self, print_as):
45           self._print_as = print_as
46   
47       def __repr__(self):
48           return "%s(%r)" % (self.__class__.__name__, self._print_as)

53   class Token:
54       """A token with possible payload. 
55    
56       .. attribute:: type 
57    
58          An arbitrary object indicating the type of this token. Should be 
59         :term:`hashable`, but otherwise it can be whatever you like. 
60       """
61   
62       LPAREN = _UniqueValue("LPAREN")
63       RPAREN = _UniqueValue("RPAREN")
64   
65       def __init__(self, type, origin, extra=None):
66           self.type = type
67           self.origin = origin
68           self.extra = extra
69   
70       __repr__ = repr_pretty_delegate
71   
72       def _repr_pretty_(self, p, cycle):
73           assert not cycle
74           kwargs = []
75           if self.extra is not None:
76               kwargs = [("extra", self.extra)]
77           return repr_pretty_impl(p, self, [self.type, self.origin], kwargs)
78   
79       __getstate__ = no_pickling

82   class ParseNode(object):
83       def __init__(self, type, token, args, origin):
84           self.type = type
85           self.token = token
86           self.args = args
87           self.origin = origin
88   
89       __repr__ = repr_pretty_delegate
90   
91       def _repr_pretty_(self, p, cycle):
92           return repr_pretty_impl(p, self, [self.type, self.token, self.args])
93   
94       __getstate__ = no_pickling

97   class Operator(object):
98       def __init__(self, token_type, arity, precedence):
99           self.token_type = token_type
100          self.arity = arity
101          self.precedence = precedence
102  
103      def __repr__(self):
104          return "%s(%r, %r, %r)" % (
105              self.__class__.__name__,
106              self.token_type,
107              self.arity,
108              self.precedence,
109          )
110  
111      __getstate__ = no_pickling

114  class _StackOperator(object):
115      def __init__(self, op, token):
116          self.op = op
117          self.token = token
118  
119      __getstate__ = no_pickling

122  _open_paren = Operator(Token.LPAREN, -1, -9999999)

125  class _ParseContext(object):
126      def __init__(self, unary_ops, binary_ops, atomic_types, trace):
127          self.op_stack = []
128          self.noun_stack = []
129          self.unary_ops = unary_ops
130          self.binary_ops = binary_ops
131          self.atomic_types = atomic_types
132          self.trace = trace
133  
134      __getstate__ = no_pickling
135

137  def _read_noun_context(token, c):
138      if token.type == Token.LPAREN:
139          if c.trace:
140              print("Pushing open-paren")
141          c.op_stack.append(_StackOperator(_open_paren, token))
142          return True
143      elif token.type in c.unary_ops:
144          if c.trace:
145              print("Pushing unary op %r" % (token.type,))
146          c.op_stack.append(_StackOperator(c.unary_ops[token.type], token))
147          return True
148      elif token.type in c.atomic_types:
149          if c.trace:
150              print("Pushing noun %r (%r)" % (token.type, token.extra))
151          c.noun_stack.append(ParseNode(token.type, token, [], token.origin))
152          return False
153      else:
154          raise PatsyError(
155              "expected a noun, not '%s'" % (token.origin.relevant_code(),), token
156          )
157  
158  
159  def _run_op(c):
160      assert c.op_stack
161      stackop = c.op_stack.pop()
162      args = []
163      for i in range(stackop.op.arity):
164          args.append(c.noun_stack.pop())
165      args.reverse()
166      if c.trace:
167          print("Reducing %r (%r)" % (stackop.op.token_type, args))
168      node = ParseNode(
169          stackop.op.token_type,
170          stackop.token,
171          args,
172          Origin.combine([stackop.token] + args),
173      )
174      c.noun_stack.append(node)
175  
176  
177  def _read_op_context(token, c):
178      if token.type == Token.RPAREN:
179          if c.trace:
180              print("Found close-paren")
181          while c.op_stack and c.op_stack[-1].op.token_type != Token.LPAREN:
182              _run_op(c)
183          if not c.op_stack:
184              raise PatsyError("missing '(' or extra ')'", token)
185          assert c.op_stack[-1].op.token_type == Token.LPAREN
186          # Expand the origin of the item on top of the noun stack to include
187          # the open and close parens:
188          combined = Origin.combine([c.op_stack[-1].token, c.noun_stack[-1].token, token])
189          c.noun_stack[-1].origin = combined
190          # Pop the open-paren
191          c.op_stack.pop()
192          return False
193      elif token.type in c.binary_ops:
194          if c.trace:
195              print("Found binary operator %r" % (token.type))
196          stackop = _StackOperator(c.binary_ops[token.type], token)
197          while c.op_stack and stackop.op.precedence <= c.op_stack[-1].op.precedence:
198              _run_op(c)
199          if c.trace:
200              print("Pushing binary operator %r" % (token.type))
201          c.op_stack.append(stackop)
202          return True
203      else:
204          raise PatsyError(
205              "expected an operator, not '%s'" % (token.origin.relevant_code(),), token
206          )

209  def infix_parse(tokens, operators, atomic_types, trace=False):
210      token_source = iter(tokens)
211  
212      unary_ops = {}
213      binary_ops = {}
214      for op in operators:
215          assert op.precedence > _open_paren.precedence
216          if op.arity == 1:
217              unary_ops[op.token_type] = op
218          elif op.arity == 2:
219              binary_ops[op.token_type] = op
220          else:
221              raise ValueError("operators must be unary or binary")
222  
223      c = _ParseContext(unary_ops, binary_ops, atomic_types, trace)
224  
225      # This is an implementation of Dijkstra's shunting yard algorithm:
226      #   http://en.wikipedia.org/wiki/Shunting_yard_algorithm
227      #   http://www.engr.mun.ca/~theo/Misc/exp_parsing.htm
228  
229      want_noun = True
230      for token in token_source:
231          if c.trace:
232              print("Reading next token (want_noun=%r)" % (want_noun,))
233          if want_noun:
234              want_noun = _read_noun_context(token, c)
235          else:
236              want_noun = _read_op_context(token, c)
237      if c.trace:
238          print("End of token stream")
239  
240      if want_noun:
241          raise PatsyError(
242              "expected a noun, but instead the expression ended",
243              c.op_stack[-1].token.origin,
244          )
245  
246      while c.op_stack:
247          if c.op_stack[-1].op.token_type == Token.LPAREN:
248              raise PatsyError("Unmatched '('", c.op_stack[-1].token)
249          _run_op(c)
250  
251      assert len(c.noun_stack) == 1
252      return c.noun_stack.pop()

256  def test_infix_parse():
257      ops = [Operator("+", 2, 10), Operator("*", 2, 20), Operator("-", 1, 30)]
258      atomic = ["ATOM1", "ATOM2"]
259      # a + -b * (c + d)
260      mock_origin = Origin("asdf", 2, 3)
261      tokens = [
262          Token("ATOM1", mock_origin, "a"),
263          Token("+", mock_origin, "+"),
264          Token("-", mock_origin, "-"),
265          Token("ATOM2", mock_origin, "b"),
266          Token("*", mock_origin, "*"),
267          Token(Token.LPAREN, mock_origin, "("),
268          Token("ATOM1", mock_origin, "c"),
269          Token("+", mock_origin, "+"),
270          Token("ATOM2", mock_origin, "d"),
271          Token(Token.RPAREN, mock_origin, ")"),
272      ]
273      tree = infix_parse(tokens, ops, atomic)
274  
275      def te(tree, type, extra):
276          assert tree.type == type
277          assert tree.token.extra == extra
278  
279      te(tree, "+", "+")
280      te(tree.args[0], "ATOM1", "a")
281      assert tree.args[0].args == []
282      te(tree.args[1], "*", "*")
283      te(tree.args[1].args[0], "-", "-")
284      assert len(tree.args[1].args[0].args) == 1
285      te(tree.args[1].args[0].args[0], "ATOM2", "b")
286      te(tree.args[1].args[1], "+", "+")
287      te(tree.args[1].args[1].args[0], "ATOM1", "c")
288      te(tree.args[1].args[1].args[1], "ATOM2", "d")
289  
290      import pytest
291  
292      # No ternary ops
293      pytest.raises(ValueError, infix_parse, [], [Operator("+", 3, 10)], ["ATOMIC"])
294  
295      # smoke test just to make sure there are no egregious bugs in 'trace'
296      infix_parse(tokens, ops, atomic, trace=True)
