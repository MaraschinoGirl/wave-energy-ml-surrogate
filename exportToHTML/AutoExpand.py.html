34       def expand_word_event(self, event):
35           curinsert = self.text.index("insert")
36           curline = self.text.get("insert linestart", "insert lineend")
37           if not self.state:
38               words = self.getwords()
39               index = 0
40           else:
41               words, index, insert, line = self.state
42               if insert != curinsert or line != curline:
43                   words = self.getwords()
44                   index = 0
45           if not words:
46               self.text.bell()
47               return "break"
48           word = self.getprevword()
49           self.text.delete("insert - %d chars" % len(word), "insert")
50           newword = words[index]

52           if index == 0:
53               self.text.bell()  # Warn we cycled around
54           self.text.insert("insert", newword)
55           curinsert = self.text.index("insert")
56           curline = self.text.get("insert linestart", "insert lineend")

60       def getwords(self):
61           word = self.getprevword()
62           if not word:
63               return []
64           before = self.text.get("1.0", "insert wordstart")
65           wbefore = re.findall(r"\b" + word + r"\w+\b", before)
66           del before
67           after = self.text.get("insert wordend", "end")
68           wafter = re.findall(r"\b" + word + r"\w+\b", after)
69           del after
70           if not wbefore and not wafter:
71               return []
72           words = []
73           dict = {}
74           # search backwards through words before
75           wbefore.reverse()
76           for w in wbefore:
77               if dict.get(w):
78                   continue
79               words.append(w)
80               dict[w] = w
81           # search onwards through words after
82           for w in wafter:
83               if dict.get(w):
84                   continue
85               words.append(w)
86               dict[w] = w
87           words.append(word)
88           return words

90       def getprevword(self):
91           line = self.text.get("insert linestart", "insert")
92           i = len(line)
93           while i > 0 and line[i - 1] in self.wordchars:
94               i -= 1
95           return line[i:]
