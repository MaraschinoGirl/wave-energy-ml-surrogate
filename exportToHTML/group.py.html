35           """ Create a new Group object by binding to a low-level GroupID. 
36           """
37           with phil:

39                   raise ValueError("%s is not a GroupID" % bind)
40               super().__init__(bind)

44           h5p.CRT_ORDER_TRACKED | h5p.CRT_ORDER_INDEXED)
45       _gcpl_crt_order.set_attr_creation_order(
46           h5p.CRT_ORDER_TRACKED | h5p.CRT_ORDER_INDEXED)

48       def create_group(self, name, track_order=None):
49           """ Create and return a new subgroup. 
50    
51           Name may be absolute or relative.  Fails if the target name already 
52           exists. 
53    
54           track_order 
55               Track dataset/group/attribute creation order under this group 
56               if True. If None use global default h5.get_config().track_order. 
57           """
58           if track_order is None:
59               track_order = h5.get_config().track_order
60   
61           with phil:
62               name, lcpl = self._e(name, lcpl=True)
63               gcpl = Group._gcpl_crt_order if track_order else None
64               gid = h5g.create(self.id, name, lcpl=lcpl, gcpl=gcpl)
65               return Group(gid)

67       def create_dataset(self, name, shape=None, dtype=None, data=None, **kwds):
68           """ Create a new HDF5 dataset 
69    
70           name 
71               Name of the dataset (absolute or relative).  Provide None to make 
72               an anonymous dataset. 
73           shape 
74               Dataset shape.  Use "()" for scalar datasets.  Required if "data" 
75               isn't provided. 
76           dtype 
77               Numpy dtype or string.  If omitted, dtype('f') will be used. 
78               Required if "data" isn't provided; otherwise, overrides data 
79               array's dtype. 
80           data 
81               Provide data to initialize the dataset.  If used, you can omit 
82               shape and dtype arguments. 
83    
84           Keyword-only arguments: 
85    
86           chunks 
87               (Tuple or int) Chunk shape, or True to enable auto-chunking. Integers can 
88               be used for 1D shape. 
89    
90           maxshape 
91               (Tuple or int) Make the dataset resizable up to this shape. Use None for 
92               axes within the tuple you want to be unlimited. Integers can be used for 1D shape. 
93               For 1D datasets with unlimited maxshape, a shape tuple of length 1 must be 
94               provided, ``(None,)``. Passing ``None`` sets ``maxshape` to `shape`, making the 
95               dataset un-resizable, which is the default. 
96           compression 
97               (String or int) Compression strategy.  Legal values are 'gzip', 
98               'szip', 'lzf'.  If an integer in range(10), this indicates gzip 
99               compression level. Otherwise, an integer indicates the number of a 
100              dynamically loaded compression filter. 
101          compression_opts 
102              Compression settings.  This is an integer for gzip, 2-tuple for 
103              szip, etc. If specifying a dynamically loaded compression filter 
104              number, this must be a tuple of values. 
105          scaleoffset 
106              (Integer) Enable scale/offset filter for (usually) lossy 
107              compression of integer or floating-point data. For integer 
108              data, the value of scaleoffset is the number of bits to 
109              retain (pass 0 to let HDF5 determine the minimum number of 
110              bits necessary for lossless compression). For floating point 
111              data, scaleoffset is the number of digits after the decimal 
112              place to retain; stored values thus have absolute error 
113              less than 0.5*10**(-scaleoffset). 
114          shuffle 
115              (T/F) Enable shuffle filter. 
116          fletcher32 
117              (T/F) Enable fletcher32 error detection. Not permitted in 
118              conjunction with the scale/offset filter. 
119          fillvalue 
120              (Scalar) Use this value for uninitialized parts of the dataset. 
121          track_times 
122              (T/F) Enable dataset creation timestamps. 
123          track_order 
124              (T/F) Track attribute creation order if True. If omitted use 
125              global default h5.get_config().track_order. 
126          external 
127              (Iterable of tuples) Sets the external storage property, thus 
128              designating that the dataset will be stored in one or more 
129              non-HDF5 files external to the HDF5 file.  Adds each tuple 
130              of (name, offset, size) to the dataset's list of external files. 
131              Each name must be a str, bytes, or os.PathLike; each offset and 
132              size, an integer.  If only a name is given instead of an iterable 
133              of tuples, it is equivalent to [(name, 0, h5py.h5f.UNLIMITED)]. 
134          efile_prefix 
135              (String) External dataset file prefix for dataset access property 
136              list. Does not persist in the file. 
137          virtual_prefix 
138              (String) Virtual dataset file prefix for dataset access property 
139              list. Does not persist in the file. 
140          allow_unknown_filter 
141              (T/F) Do not check that the requested filter is available for use. 
142              This should only be used with ``write_direct_chunk``, where the caller 
143              compresses the data before handing it to h5py. 
144          rdcc_nbytes 
145              Total size of the dataset's chunk cache in bytes. The default size 
146              is 1024**2 (1 MiB). 
147          rdcc_w0 
148              The chunk preemption policy for this dataset.  This must be 
149              between 0 and 1 inclusive and indicates the weighting according to 
150              which chunks which have been fully read or written are penalized 
151              when determining which chunks to flush from cache.  A value of 0 
152              means fully read or written chunks are treated no differently than 
153              other chunks (the preemption is strictly LRU) while a value of 1 
154              means fully read or written chunks are always preempted before 
155              other chunks.  If your application only reads or writes data once, 
156              this can be safely set to 1.  Otherwise, this should be set lower 
157              depending on how often you re-read or re-write the same data.  The 
158              default value is 0.75. 
159          rdcc_nslots 
160              The number of chunk slots in the dataset's chunk cache. Increasing 
161              this value reduces the number of cache collisions, but slightly 
162              increases the memory used. Due to the hashing strategy, this value 
163              should ideally be a prime number. As a rule of thumb, this value 
164              should be at least 10 times the number of chunks that can fit in 
165              rdcc_nbytes bytes. For maximum performance, this value should be set 
166              approximately 100 times that number of chunks. The default value is 
167              521. 
168          """
169          if 'track_order' not in kwds:
170              kwds['track_order'] = h5.get_config().track_order
171  
172          if 'efile_prefix' in kwds:
173              kwds['efile_prefix'] = self._e(kwds['efile_prefix'])
174  
175          if 'virtual_prefix' in kwds:
176              kwds['virtual_prefix'] = self._e(kwds['virtual_prefix'])
177  
178          with phil:
179              group = self
180              if name:
181                  name = self._e(name)
182                  if b'/' in name.lstrip(b'/'):
183                      parent_path, name = name.rsplit(b'/', 1)
184                      group = self.require_group(parent_path)
185  
186              dsid = dataset.make_new_dset(group, shape, dtype, data, name, **kwds)
187              dset = dataset.Dataset(dsid)
188              return dset

223          def build_virtual_dataset(
224                  self, name, shape, dtype, maxshape=None, fillvalue=None
225          ):
226              """Assemble a virtual dataset in this group. 
227   
228              This is used as a context manager:: 
229   
230                  with f.build_virtual_dataset('virt', (10, 1000), np.uint32) as layout: 
231                      layout[0] = h5py.VirtualSource('foo.h5', 'data', (1000,)) 
232   
233              name 
234                  (str) Name of the new dataset 
235              shape 
236                  (tuple) Shape of the dataset 
237              dtype 
238                  A numpy dtype for data read from the virtual dataset 
239              maxshape 
240                  (tuple, optional) Maximum dimensions if the dataset can grow. 
241                  Use None for unlimited dimensions. 
242              fillvalue 
243                  The value used where no data is available. 
244              """
245              from .vds import VirtualLayout
246              layout = VirtualLayout(shape, dtype, maxshape, self.file.filename)
247              yield layout

249              self.create_virtual_dataset(name, layout, fillvalue)

251      def require_dataset(self, name, shape, dtype, exact=False, **kwds):
252          """ Open a dataset, creating it if it doesn't exist. 
253   
254          If keyword "exact" is False (default), an existing dataset must have 
255          the same shape and a conversion-compatible dtype to be returned.  If 
256          True, the shape and dtype must match exactly. 
257   
258          If keyword "maxshape" is given, the maxshape and dtype must match 
259          instead. 
260   
261          If any of the keywords "rdcc_nslots", "rdcc_nbytes", or "rdcc_w0" are 
262          given, they will be used to configure the dataset's chunk cache. 
263   
264          Other dataset keywords (see create_dataset) may be provided, but are 
265          only used if a new dataset is to be created. 
266   
267          Raises TypeError if an incompatible object already exists, or if the 
268          shape, maxshape or dtype don't match according to the above rules. 
269          """
270          if 'efile_prefix' in kwds:
271              kwds['efile_prefix'] = self._e(kwds['efile_prefix'])
272  
273          if 'virtual_prefix' in kwds:
274              kwds['virtual_prefix'] = self._e(kwds['virtual_prefix'])
275  
276          with phil:
277              if name not in self:
278                  return self.create_dataset(name, *(shape, dtype), **kwds)
279  
280              if isinstance(shape, int):
281                  shape = (shape,)
282  
283              try:
284                  dsid = dataset.open_dset(self, self._e(name), **kwds)
285                  dset = dataset.Dataset(dsid)
286              except KeyError:
287                  dset = self[name]
288                  raise TypeError("Incompatible object (%s) already exists" % dset.__class__.__name__)
289  
290              if shape != dset.shape:
291                  if "maxshape" not in kwds:
292                      raise TypeError("Shapes do not match (existing %s vs new %s)" % (dset.shape, shape))
293                  elif kwds["maxshape"] != dset.maxshape:
294                      raise TypeError("Max shapes do not match (existing %s vs new %s)" % (dset.maxshape, kwds["maxshape"]))
295  
296              if exact:
297                  if dtype != dset.dtype:
298                      raise TypeError("Datatypes do not exactly match (existing %s vs new %s)" % (dset.dtype, dtype))
299              elif not numpy.can_cast(dtype, dset.dtype):
300                  raise TypeError("Datatypes cannot be safely cast (existing %s vs new %s)" % (dset.dtype, dtype))
301  
302              return dset

304      def create_dataset_like(self, name, other, **kwupdate):
305          """ Create a dataset similar to `other`. 
306   
307          name 
308              Name of the dataset (absolute or relative).  Provide None to make 
309              an anonymous dataset. 
310          other 
311              The dataset which the new dataset should mimic. All properties, such 
312              as shape, dtype, chunking, ... will be taken from it, but no data 
313              or attributes are being copied. 
314   
315          Any dataset keywords (see create_dataset) may be provided, including 
316          shape and dtype, in which case the provided values take precedence over 
317          those from `other`. 
318          """
319          for k in ('shape', 'dtype', 'chunks', 'compression',
320                    'compression_opts', 'scaleoffset', 'shuffle', 'fletcher32',
321                    'fillvalue'):
322              kwupdate.setdefault(k, getattr(other, k))
323          # TODO: more elegant way to pass these (dcpl to create_dataset?)
324          dcpl = other.id.get_create_plist()
325          kwupdate.setdefault('track_times', dcpl.get_obj_track_times())
326          kwupdate.setdefault('track_order', dcpl.get_attr_creation_order() > 0)
327  
328          # Special case: the maxshape property always exists, but if we pass it
329          # to create_dataset, the new dataset will automatically get chunked
330          # layout. So we copy it only if it is different from shape.
331          if other.maxshape != other.shape:
332              kwupdate.setdefault('maxshape', other.maxshape)
333  
334          return self.create_dataset(name, **kwupdate)

336      def require_group(self, name):
337          # TODO: support kwargs like require_dataset
338          """Return a group, creating it if it doesn't exist. 
339   
340          TypeError is raised if something with that name already exists that 
341          isn't a group. 
342          """
343          with phil:
344              if name not in self:
345                  return self.create_group(name)
346              grp = self[name]
347              if not isinstance(grp, Group):
348                  raise TypeError("Incompatible object (%s) already exists" % grp.__class__.__name__)
349              return grp

351      @with_phil
352      def __getitem__(self, name):
353          """ Open an object in the file """
354  
355          if isinstance(name, h5r.Reference):
356              oid = h5r.dereference(name, self.id)
357              if oid is None:
358                  raise ValueError("Invalid HDF5 object reference")
359          elif isinstance(name, (bytes, str)):
360              oid = h5o.open(self.id, self._e(name), lapl=self._lapl)
361          else:
362              raise TypeError("Accessing a group is done with bytes or str, "
363                              "not {}".format(type(name)))
364  
365          otype = h5i.get_type(oid)
366          if otype == h5i.GROUP:
367              return Group(oid)
368          elif otype == h5i.DATASET:
369              return dataset.Dataset(oid, readonly=(self.file.mode == 'r'))
370          elif otype == h5i.DATATYPE:
371              return datatype.Datatype(oid)
372          else:
373              raise TypeError("Unknown object type")

375      def get(self, name, default=None, getclass=False, getlink=False):
376          """ Retrieve an item or other information. 
377   
378          "name" given only: 
379              Return the item, or "default" if it doesn't exist 
380   
381          "getclass" is True: 
382              Return the class of object (Group, Dataset, etc.), or "default" 
383              if nothing with that name exists 
384   
385          "getlink" is True: 
386              Return HardLink, SoftLink or ExternalLink instances.  Return 
387              "default" if nothing with that name exists. 
388   
389          "getlink" and "getclass" are True: 
390              Return HardLink, SoftLink and ExternalLink classes.  Return 
391              "default" if nothing with that name exists. 
392   
393          Example: 
394   
395          >>> cls = group.get('foo', getclass=True) 
396          >>> if cls == SoftLink: 
397          """
398          # pylint: disable=arguments-differ
399  
400          with phil:
401              if not (getclass or getlink):
402                  try:
403                      return self[name]
404                  except KeyError:
405                      return default
406  
407              if name not in self:
408                  return default
409  
410              elif getclass and not getlink:
411                  typecode = h5o.get_info(self.id, self._e(name), lapl=self._lapl).type
412  
413                  try:
414                      return {h5o.TYPE_GROUP: Group,
415                              h5o.TYPE_DATASET: dataset.Dataset,
416                              h5o.TYPE_NAMED_DATATYPE: datatype.Datatype}[typecode]
417                  except KeyError:
418                      raise TypeError("Unknown object type")
419  
420              elif getlink:
421                  typecode = self.id.links.get_info(self._e(name), lapl=self._lapl).type
422  
423                  if typecode == h5l.TYPE_SOFT:
424                      if getclass:
425                          return SoftLink
426                      linkbytes = self.id.links.get_val(self._e(name), lapl=self._lapl)
427                      return SoftLink(self._d(linkbytes))
428  
429                  elif typecode == h5l.TYPE_EXTERNAL:
430                      if getclass:
431                          return ExternalLink
432                      filebytes, linkbytes = self.id.links.get_val(self._e(name), lapl=self._lapl)
433                      return ExternalLink(
434                          filename_decode(filebytes), self._d(linkbytes)
435                      )
436  
437                  elif typecode == h5l.TYPE_HARD:
438                      return HardLink if getclass else HardLink()
439  
440                  else:
441                      raise TypeError("Unknown link type")

443      def __setitem__(self, name, obj):
444          """ Add an object to the group.  The name must not already be in use. 
445   
446          The action taken depends on the type of object assigned: 
447   
448          Named HDF5 object (Dataset, Group, Datatype) 
449              A hard link is created at "name" which points to the 
450              given object. 
451   
452          SoftLink or ExternalLink 
453              Create the corresponding link. 
454   
455          Numpy ndarray 
456              The array is converted to a dataset object, with default 
457              settings (contiguous storage, etc.). 
458   
459          Numpy dtype 
460              Commit a copy of the datatype as a named datatype in the file. 
461   
462          Anything else 
463              Attempt to convert it to an ndarray and store it.  Scalar 
464              values are stored as scalar datasets. Raise ValueError if we 
465              can't understand the resulting array dtype. 
466          """
467          with phil:
468              name, lcpl = self._e(name, lcpl=True)
469  
470              if isinstance(obj, HLObject):
471                  h5o.link(obj.id, self.id, name, lcpl=lcpl, lapl=self._lapl)
472  
473              elif isinstance(obj, SoftLink):
474                  self.id.links.create_soft(name, self._e(obj.path), lcpl=lcpl, lapl=self._lapl)
475  
476              elif isinstance(obj, ExternalLink):
477                  fn = filename_encode(obj.filename)
478                  self.id.links.create_external(name, fn, self._e(obj.path),
479                                                lcpl=lcpl, lapl=self._lapl)
480  
481              elif isinstance(obj, numpy.dtype):
482                  htype = h5t.py_create(obj, logical=True)
483                  htype.commit(self.id, name, lcpl=lcpl)
484  
485              else:
486                  ds = self.create_dataset(None, data=obj)
487                  h5o.link(ds.id, self.id, name, lcpl=lcpl)

511      @with_phil
512      def __contains__(self, name):
513          """ Test if a member name exists """
514          if hasattr(h5g, "_path_valid"):
515              if not self.id:
516                  return False
517              return h5g._path_valid(self.id, self._e(name), self._lapl)

520      def copy(self, source, dest, name=None,
521               shallow=False, expand_soft=False, expand_external=False,
522               expand_refs=False, without_attrs=False):
523          """Copy an object or group. 
524   
525          The source can be a path, Group, Dataset, or Datatype object.  The 
526          destination can be either a path or a Group object.  The source and 
527          destinations need not be in the same file. 
528   
529          If the source is a Group object, all objects contained in that group 
530          will be copied recursively. 
531   
532          When the destination is a Group object, by default the target will 
533          be created in that group with its current name (basename of obj.name). 
534          You can override that by setting "name" to a string. 
535   
536          There are various options which all default to "False": 
537   
538           - shallow: copy only immediate members of a group. 
539   
540           - expand_soft: expand soft links into new objects. 
541   
542           - expand_external: expand external links into new objects. 
543   
544           - expand_refs: copy objects that are pointed to by references. 
545   
546           - without_attrs: copy object without copying attributes. 
547   
548         Example: 
549   
550          >>> f = File('myfile.hdf5', 'w') 
551          >>> f.create_group("MyGroup") 
552          >>> list(f.keys()) 
553          ['MyGroup'] 
554          >>> f.copy('MyGroup', 'MyCopy') 
555          >>> list(f.keys()) 
556          ['MyGroup', 'MyCopy'] 
557   
558          """
559          with phil:
560              if isinstance(source, HLObject):
561                  source_path = '.'
562              else:
563                  # Interpret source as a path relative to this group
564                  source_path = source
565                  source = self
566  
567              if isinstance(dest, Group):
568                  if name is not None:
569                      dest_path = name
570                  elif source_path == '.':
571                      dest_path = pp.basename(h5i.get_name(source.id))
572                  else:
573                      # copy source into dest group: dest_name/source_name
574                      dest_path = pp.basename(h5i.get_name(source[source_path].id))
575  
576              elif isinstance(dest, HLObject):
577                  raise TypeError("Destination must be path or Group object")
578              else:
579                  # Interpret destination as a path relative to this group
580                  dest_path = dest
581                  dest = self
582  
583              flags = 0
584              if shallow:
585                  flags |= h5o.COPY_SHALLOW_HIERARCHY_FLAG
586              if expand_soft:
587                  flags |= h5o.COPY_EXPAND_SOFT_LINK_FLAG
588              if expand_external:
589                  flags |= h5o.COPY_EXPAND_EXT_LINK_FLAG
590              if expand_refs:
591                  flags |= h5o.COPY_EXPAND_REFERENCE_FLAG
592              if without_attrs:
593                  flags |= h5o.COPY_WITHOUT_ATTR_FLAG
594              if flags:
595                  copypl = h5p.create(h5p.OBJECT_COPY)
596                  copypl.set_copy_object(flags)
597              else:
598                  copypl = None
599  
600              h5o.copy(source.id, self._e(source_path), dest.id, self._e(dest_path),
601                       copypl, base.dlcpl)

603      def move(self, source, dest):
604          """ Move a link to a new location in the file. 
605   
606          If "source" is a hard link, this effectively renames the object.  If 
607          "source" is a soft or external link, the link itself is moved, with its 
608          value unmodified. 
609          """
610          with phil:
611              if source == dest:
612                  return
613              self.id.links.move(self._e(source), self.id, self._e(dest),
614                                 lapl=self._lapl, lcpl=self._lcpl)

616      def visit(self, func):
617          """ Recursively visit all names in this group and subgroups. 
618   
619          Note: visit ignores soft and external links. To visit those, use 
620          visit_links. 
621   
622          You supply a callable (function, method or callable object); it 
623          will be called exactly once for each link in this group and every 
624          group below it. Your callable must conform to the signature: 
625   
626              func(<member name>) => <None or return value> 
627   
628          Returning None continues iteration, returning anything else stops 
629          and immediately returns that value from the visit method.  No 
630          particular order of iteration within groups is guaranteed. 
631   
632          Example: 
633   
634          >>> # List the entire contents of the file 
635          >>> f = File("foo.hdf5") 
636          >>> list_of_names = [] 
637          >>> f.visit(list_of_names.append) 
638          """
639          with phil:
640              def proxy(name):
641                  """ Call the function with the text name, not bytes """
642                  return func(self._d(name))
643              return h5o.visit(self.id, proxy)

645      def visititems(self, func):
646          """ Recursively visit names and objects in this group. 
647   
648          Note: visititems ignores soft and external links. To visit those, use 
649          visititems_links. 
650   
651          You supply a callable (function, method or callable object); it 
652          will be called exactly once for each link in this group and every 
653          group below it. Your callable must conform to the signature: 
654   
655              func(<member name>, <object>) => <None or return value> 
656   
657          Returning None continues iteration, returning anything else stops 
658          and immediately returns that value from the visit method.  No 
659          particular order of iteration within groups is guaranteed. 
660   
661          Example: 
662   
663          # Get a list of all datasets in the file 
664          >>> mylist = [] 
665          >>> def func(name, obj): 
666          ...     if isinstance(obj, Dataset): 
667          ...         mylist.append(name) 
668          ... 
669          >>> f = File('foo.hdf5') 
670          >>> f.visititems(func) 
671          """
672          with phil:
673              def proxy(name):
674                  """ Use the text name of the object, not bytes """
675                  name = self._d(name)
676                  return func(name, self[name])
677              return h5o.visit(self.id, proxy)
678  
679      def visit_links(self, func):
680          """ Recursively visit all names in this group and subgroups. 
681          Each link will be visited exactly once, regardless of its target. 
682   
683          You supply a callable (function, method or callable object); it 
684          will be called exactly once for each link in this group and every 
685          group below it. Your callable must conform to the signature: 
686   
687              func(<member name>) => <None or return value> 
688   
689          Returning None continues iteration, returning anything else stops 
690          and immediately returns that value from the visit method.  No 
691          particular order of iteration within groups is guaranteed. 
692   
693          Example: 
694   
695          >>> # List the entire contents of the file 
696          >>> f = File("foo.hdf5") 
697          >>> list_of_names = [] 
698          >>> f.visit_links(list_of_names.append) 
699          """
700          with phil:
701              def proxy(name):
702                  """ Call the function with the text name, not bytes """
703                  return func(self._d(name))
704              return self.id.links.visit(proxy)
705  
706      def visititems_links(self, func):
707          """ Recursively visit links in this group. 
708          Each link will be visited exactly once, regardless of its target. 
709   
710          You supply a callable (function, method or callable object); it 
711          will be called exactly once for each link in this group and every 
712          group below it. Your callable must conform to the signature: 
713   
714              func(<member name>, <link>) => <None or return value> 
715   
716          Returning None continues iteration, returning anything else stops 
717          and immediately returns that value from the visit method.  No 
718          particular order of iteration within groups is guaranteed. 
719   
720          Example: 
721   
722          # Get a list of all softlinks in the file 
723          >>> mylist = [] 
724          >>> def func(name, link): 
725          ...     if isinstance(link, SoftLink): 
726          ...         mylist.append(name) 
727          ... 
728          >>> f = File('foo.hdf5') 
729          >>> f.visititems_links(func) 
730          """
731          with phil:
732              def proxy(name):
733                  """ Use the text name of the object, not bytes """
734                  name = self._d(name)
735                  return func(name, self.get(name, getlink=True))
736              return self.id.links.visit(proxy)
737  
738      @with_phil
739      def __repr__(self):
740          if not self:
741              r = u"<Closed HDF5 group>"
742          else:
743              namestr = (
744                  '"%s"' % self.name
745              ) if self.name is not None else u"(anonymous)"
746              r = '<HDF5 group %s (%d members)>' % (namestr, len(self))
747  
748          return r
