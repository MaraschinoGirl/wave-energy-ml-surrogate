2    # framework.
3    # We demonstrate:
4    # * Reloading your Python module without shutting down IIS (eg, when your
5    #   .py implementation file changes.)
6    # * Custom command-line handling - both additional options and commands.

26   # Notes on reloading
27   # If your HttpFilterProc or HttpExtensionProc functions raises
28   # 'isapi.InternalReloadException', the framework will not treat it
29   # as an error but instead will terminate your extension, reload your
30   # extension module, re-initialize the instance, and re-issue the request.
31   # The Initialize functions are called with None as their param.  The
32   # return code from the terminate function is ignored.
33   #
34   # This is all the framework does to help you.  It is up to your code
35   # when you raise this exception.  This sample uses a Win32 "find
36   # notification".  Whenever windows tells us one of the files in the
37   # directory has changed, we check if the time of our source-file has
38   # changed, and set a flag.  Next imcoming request, we check the flag and
39   # raise the special exception if set.
40   #
41   # The end result is that the module is automatically reloaded whenever
42   # the source-file changes - you need take no further action to see your
43   # changes reflected in the running server.
44   
45   # The framework only reloads your module - if you have libraries you
46   # depend on and also want reloaded, you must arrange for this yourself.
47   # One way of doing this would be to special case the import of these
48   # modules.  Eg:
49   # --
50   # try:
51   #    my_module = reload(my_module) # module already imported - reload it
52   # except NameError:
53   #    import my_module # first time around - import it.
54   # --
55   # When your module is imported for the first time, the NameError will
56   # be raised, and the module imported.  When the ISAPI framework reloads
57   # your module, the existing module will avoid the NameError, and allow
58   # you to reload that module.

75   # A watcher thread that checks for __file__ changing.
76   # When it detects it, it simply sets "change_detected" to true.
77   class ReloadWatcherThread(threading.Thread):
78       def __init__(self):
79           self.change_detected = False
80           self.filename = __file__
81           if self.filename.endswith("c") or self.filename.endswith("o"):
82               self.filename = self.filename[:-1]
83           self.handle = win32file.FindFirstChangeNotification(
84               os.path.dirname(self.filename),
85               False,  # watch tree?
86               win32con.FILE_NOTIFY_CHANGE_LAST_WRITE,
87           )
88           threading.Thread.__init__(self)
89   
90       def run(self):
91           last_time = os.stat(self.filename)[stat.ST_MTIME]
92           while 1:
93               try:
94                   rc = win32event.WaitForSingleObject(self.handle, win32event.INFINITE)
95                   win32file.FindNextChangeNotification(self.handle)
96               except win32event.error as details:
97                   # handle closed - thread should terminate.
98                   if details.winerror != winerror.ERROR_INVALID_HANDLE:
99                       raise
100                  break
101              this_time = os.stat(self.filename)[stat.ST_MTIME]
102              if this_time != last_time:
103                  print("Detected file change - flagging for reload.")
104                  self.change_detected = True
105                  last_time = this_time
106  
107      def stop(self):
108          win32file.FindCloseChangeNotification(self.handle)

111  # The ISAPI extension - handles requests in our virtual dir, and sends the
112  # response to the client.
113  class Extension(SimpleExtension):
114      "Python advanced sample Extension"
115  
116      def __init__(self):
117          self.reload_watcher = ReloadWatcherThread()
118          self.reload_watcher.start()
119  
120      def HttpExtensionProc(self, ecb):
121          # NOTE: If you use a ThreadPoolExtension, you must still perform
122          # this check in HttpExtensionProc - raising the exception from
123          # The "Dispatch" method will just cause the exception to be
124          # rendered to the browser.
125          if self.reload_watcher.change_detected:
126              print("Doing reload")
127              raise InternalReloadException
128  
129          url = ecb.GetServerVariable("UNICODE_URL")
130          if url.endswith("ReportUnhealthy"):
131              ecb.ReportUnhealthy("I'm a little sick")
132  
133          ecb.SendResponseHeaders("200 OK", "Content-Type: text/html\r\n\r\n", 0)
134          print("<HTML><BODY>", file=ecb)
135  
136          qs = ecb.GetServerVariable("QUERY_STRING")
137          if qs:
138              queries = qs.split("&")
139              print("<PRE>", file=ecb)
140              for q in queries:
141                  val = ecb.GetServerVariable(q, "&lt;no such variable&gt;")
142                  print(f"{q}={val!r}", file=ecb)
143              print("</PRE><P/>", file=ecb)
144  
145          print("This module has been imported", file=ecb)
146          print("%d times" % (reload_counter,), file=ecb)
147          print("</BODY></HTML>", file=ecb)
148          ecb.close()
149          return isapicon.HSE_STATUS_SUCCESS
150  
151      def TerminateExtension(self, status):
152          self.reload_watcher.stop()
