11   def pjoin(*args):
12       """Join paths to create a real path."""
13       return osp.abspath(osp.join(*args))
14   
15   
16   def _get_default_core_data():
17       """Get the data for the app template."""
18       with open(pjoin(HERE, "staging", "package.json")) as fid:
19           return json.load(fid)

22   def _is_lab_package(name):
23       """Whether a package name is in the lab namespace"""
24       return name.startswith("@jupyterlab/")
25   
26   
27   def _only_nonlab(collection):
28       """Filter a dict/sequence to remove all lab packages 
29    
30       This is useful to take the default values of e.g. singletons and filter 
31       away the '@jupyterlab/' namespace packages, but leave any others (e.g. 
32       lumino and react). 
33       """
34       if isinstance(collection, dict):
35           return {k: v for (k, v) in collection.items() if not _is_lab_package(k)}
36       elif isinstance(collection, (list, tuple)):
37           return list(filterfalse(_is_lab_package, collection))
38       msg = "collection arg should be either dict or list/tuple"
39       raise TypeError(msg)

42   class CoreConfig:
43       """An object representing a core config. 
44    
45       This enables custom lab application to override some parts of the core 
46       configuration of the build system. 
47       """
48   
49       def __init__(self):
50           self._data = _get_default_core_data()
51   
52       def add(self, name, semver, extension=False, mime_extension=False):
53           """Remove an extension/singleton. 
54    
55           If neither extension or mimeExtension is True (the default) 
56           the package is added as a singleton dependency. 
57    
58           name: string 
59               The npm package name 
60           semver: string 
61               The semver range for the package 
62           extension: bool 
63               Whether the package is an extension 
64           mime_extension: bool 
65               Whether the package is a MIME extension 
66           """
67           data = self._data
68           if not name:
69               msg = "Missing package name"
70               raise ValueError(msg)
71           if not semver:
72               msg = "Missing package semver"
73               raise ValueError(msg)
74           if name in data["resolutions"]:
75               msg = f"Package already present: {name!r}"
76               raise ValueError(msg)
77           data["resolutions"][name] = semver
78   
79           # If both mimeExtension and extensions are True, treat
80           # as mime extension
81           if mime_extension:
82               data["jupyterlab"]["mimeExtensions"][name] = ""
83               data["dependencies"][name] = semver
84           elif extension:
85               data["jupyterlab"]["extensions"][name] = ""
86               data["dependencies"][name] = semver
87           else:
88               data["jupyterlab"]["singletonPackages"].append(name)
89   
90       def remove(self, name):
91           """Remove a package/extension. 
92    
93           name: string 
94               The npm package name 
95           """
96           data = self._data
97           maps = (
98               data["dependencies"],
99               data["resolutions"],
100              data["jupyterlab"]["extensions"],
101              data["jupyterlab"]["mimeExtensions"],
102          )
103          for m in maps:
104              try:
105                  del m[name]
106              except KeyError:
107                  pass
108  
109          data["jupyterlab"]["singletonPackages"].remove(name)
110  
111      def clear_packages(self, lab_only=True):
112          """Clear the packages/extensions."""
113          data = self._data
114          # Clear all dependencies
115          if lab_only:
116              # Clear all "@jupyterlab/" dependencies
117              data["dependencies"] = _only_nonlab(data["dependencies"])
118              data["resolutions"] = _only_nonlab(data["resolutions"])
119              data["jupyterlab"]["extensions"] = _only_nonlab(data["jupyterlab"]["extensions"])
120              data["jupyterlab"]["mimeExtensions"] = _only_nonlab(
121                  data["jupyterlab"]["mimeExtensions"]
122              )
123              data["jupyterlab"]["singletonPackages"] = _only_nonlab(
124                  data["jupyterlab"]["singletonPackages"]
125              )
126          else:
127              data["dependencies"] = {}
128              data["resolutions"] = {}
129              data["jupyterlab"]["extensions"] = {}
130              data["jupyterlab"]["mimeExtensions"] = {}
131              data["jupyterlab"]["singletonPackages"] = []
132  
133      @property
134      def extensions(self):
135          """A dict mapping all extension names to their semver"""
136          data = self._data
137          return {k: data["resolutions"][k] for k in data["jupyterlab"]["extensions"]}
138  
139      @property
140      def mime_extensions(self):
141          """A dict mapping all MIME extension names to their semver"""
142          data = self._data
143          return {k: data["resolutions"][k] for k in data["jupyterlab"]["mimeExtensions"]}
144  
145      @property
146      def singletons(self):
147          """A dict mapping all singleton names to their semver"""
148          data = self._data
149          return {
150              k: data["resolutions"].get(k, None) for k in data["jupyterlab"]["singletonPackages"]
151          }
152  
153      @property
154      def static_dir(self):
155          return self._data["jupyterlab"]["staticDir"]
156  
157      @static_dir.setter
158      def static_dir(self, static_dir):
159          self._data["jupyterlab"]["staticDir"] = static_dir
