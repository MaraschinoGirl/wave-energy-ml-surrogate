13   class _BinaryClassifierCurveDisplayMixin:
14       """Mixin class to be used in Displays requiring a binary classifier. 
15    
16       The aim of this class is to centralize some validations regarding the estimator and 
17       the target and gather the response of the estimator. 
18       """
19   
20       def _validate_plot_params(self, *, ax=None, name=None):
21           check_matplotlib_support(f"{self.__class__.__name__}.plot")
22           import matplotlib.pyplot as plt
23   
24           if ax is None:
25               _, ax = plt.subplots()
26   
27           name = self.estimator_name if name is None else name
28           return ax, ax.figure, name
29   
30       @classmethod
31       def _validate_and_get_response_values(
32           cls, estimator, X, y, *, response_method="auto", pos_label=None, name=None
33       ):
34           check_matplotlib_support(f"{cls.__name__}.from_estimator")
35   
36           name = estimator.__class__.__name__ if name is None else name
37   
38           y_pred, pos_label = _get_response_values_binary(
39               estimator,
40               X,
41               response_method=response_method,
42               pos_label=pos_label,
43           )
44   
45           return y_pred, pos_label, name
46   
47       @classmethod
48       def _validate_from_predictions_params(
49           cls, y_true, y_pred, *, sample_weight=None, pos_label=None, name=None
50       ):
51           check_matplotlib_support(f"{cls.__name__}.from_predictions")
52   
53           if type_of_target(y_true) != "binary":
54               raise ValueError(
55                   f"The target y is not binary. Got {type_of_target(y_true)} type of"
56                   " target."
57               )
58   
59           check_consistent_length(y_true, y_pred, sample_weight)
60           pos_label = _check_pos_label_consistency(pos_label, y_true)
61   
62           name = name if name is not None else "Classifier"
63   
64           return pos_label, name

67   def _validate_score_name(score_name, scoring, negate_score):
68       """Validate the `score_name` parameter. 
69    
70       If `score_name` is provided, we just return it as-is. 
71       If `score_name` is `None`, we use `Score` if `negate_score` is `False` and 
72       `Negative score` otherwise. 
73       If `score_name` is a string or a callable, we infer the name. We replace `_` by 
74       spaces and capitalize the first letter. We remove `neg_` and replace it by 
75       `"Negative"` if `negate_score` is `False` or just remove it otherwise. 
76       """
77       if score_name is not None:
78           return score_name
79       elif scoring is None:
80           return "Negative score" if negate_score else "Score"
81       else:
82           score_name = scoring.__name__ if callable(scoring) else scoring
83           if negate_score:
84               if score_name.startswith("neg_"):
85                   score_name = score_name[4:]
86               else:
87                   score_name = f"Negative {score_name}"
88           elif score_name.startswith("neg_"):
89               score_name = f"Negative {score_name[4:]}"
90           score_name = score_name.replace("_", " ")
91           return score_name.capitalize()

94   def _interval_max_min_ratio(data):
95       """Compute the ratio between the largest and smallest inter-point distances. 
96    
97       A value larger than 5 typically indicates that the parameter range would 
98       better be displayed with a log scale while a linear scale would be more 
99       suitable otherwise. 
100      """
101      diff = np.diff(np.sort(data))
102      return diff.max() / diff.min()

105  def _validate_style_kwargs(default_style_kwargs, user_style_kwargs):
106      """Create valid style kwargs by avoiding Matplotlib alias errors. 
107   
108      Matplotlib raises an error when, for example, 'color' and 'c', or 'linestyle' and 
109      'ls', are specified together. To avoid this, we automatically keep only the one 
110      specified by the user and raise an error if the user specifies both. 
111   
112      Parameters 
113      ---------- 
114      default_style_kwargs : dict 
115          The Matplotlib style kwargs used by default in the scikit-learn display. 
116      user_style_kwargs : dict 
117          The user-defined Matplotlib style kwargs. 
118   
119      Returns 
120      ------- 
121      valid_style_kwargs : dict 
122          The validated style kwargs taking into account both default and user-defined 
123          Matplotlib style kwargs. 
124      """
125  
126      invalid_to_valid_kw = {
127          "ls": "linestyle",
128          "c": "color",
129          "ec": "edgecolor",
130          "fc": "facecolor",
131          "lw": "linewidth",
132          "mec": "markeredgecolor",
133          "mfcalt": "markerfacecoloralt",
134          "ms": "markersize",
135          "mew": "markeredgewidth",
136          "mfc": "markerfacecolor",
137          "aa": "antialiased",
138          "ds": "drawstyle",
139          "font": "fontproperties",
140          "family": "fontfamily",
141          "name": "fontname",
142          "size": "fontsize",
143          "stretch": "fontstretch",
144          "style": "fontstyle",
145          "variant": "fontvariant",
146          "weight": "fontweight",
147          "ha": "horizontalalignment",
148          "va": "verticalalignment",
149          "ma": "multialignment",
150      }
151      for invalid_key, valid_key in invalid_to_valid_kw.items():
152          if invalid_key in user_style_kwargs and valid_key in user_style_kwargs:
153              raise TypeError(
154                  f"Got both {invalid_key} and {valid_key}, which are aliases of one "
155                  "another"
156              )
157      valid_style_kwargs = default_style_kwargs.copy()
158  
159      for key in user_style_kwargs.keys():
160          if key in invalid_to_valid_kw:
161              valid_style_kwargs[invalid_to_valid_kw[key]] = user_style_kwargs[key]
162          else:
163              valid_style_kwargs[key] = user_style_kwargs[key]
164  
165      return valid_style_kwargs

168  def _despine(ax):
169      """Remove the top and right spines of the plot. 
170   
171      Parameters 
172      ---------- 
173      ax : matplotlib.axes.Axes 
174          The axes of the plot to despine. 
175      """
176      for s in ["top", "right"]:
177          ax.spines[s].set_visible(False)
178      for s in ["bottom", "left"]:
179          ax.spines[s].set_bounds(0, 1)
