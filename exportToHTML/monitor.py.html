16   class _MonitorMessage(TypedDict):
17       event: int
18       value: int
19       endpoint: bytes
20   
21   
22   def parse_monitor_message(msg: list[bytes]) -> _MonitorMessage:
23       """decode zmq_monitor event messages. 
24    
25       Parameters 
26       ---------- 
27       msg : list(bytes) 
28           zmq multipart message that has arrived on a monitor PAIR socket. 
29    
30           First frame is:: 
31    
32               16 bit event id 
33               32 bit event value 
34               no padding 
35    
36           Second frame is the endpoint as a bytestring 
37    
38       Returns 
39       ------- 
40       event : dict 
41           event description as dict with the keys `event`, `value`, and `endpoint`. 
42       """
43       if len(msg) != 2 or len(msg[0]) != 6:
44           raise RuntimeError(f"Invalid event message format: {msg}")
45       event_id, value = struct.unpack("=hi", msg[0])
46       event: _MonitorMessage = {
47           'event': zmq.Event(event_id),
48           'value': zmq.Event(value),
49           'endpoint': msg[1],
50       }
51       return event
52   
53   
54   async def _parse_monitor_msg_async(
55       awaitable_msg: Awaitable[list[bytes]],
56   ) -> _MonitorMessage:
57       """Like parse_monitor_msg, but awaitable 
58    
59       Given awaitable message, return awaitable for the parsed monitor message. 
60       """
61   
62       msg = await awaitable_msg
63       # 4.0-style event API
64       return parse_monitor_message(msg)

81   def recv_monitor_message(
82       socket: zmq.Socket,
83       flags: int = 0,
84   ) -> _MonitorMessage | Awaitable[_MonitorMessage]:
85       """Receive and decode the given raw message from the monitoring socket and return a dict. 
86    
87       Requires libzmq ≥ 4.0 
88    
89       The returned dict will have the following entries: 
90         event : int 
91           the event id as described in `libzmq.zmq_socket_monitor` 
92         value : int 
93           the event value associated with the event, see `libzmq.zmq_socket_monitor` 
94         endpoint : str 
95           the affected endpoint 
96    
97       .. versionchanged:: 23.1 
98           Support for async sockets added. 
99           When called with a async socket, 
100          returns an awaitable for the monitor message. 
101   
102      Parameters 
103      ---------- 
104      socket : zmq.Socket 
105          The PAIR socket (created by other.get_monitor_socket()) on which to recv the message 
106      flags : int 
107          standard zmq recv flags 
108   
109      Returns 
110      ------- 
111      event : dict 
112          event description as dict with the keys `event`, `value`, and `endpoint`. 
113      """
114  
115      _check_version((4, 0), 'libzmq event API')
116      # will always return a list
117      msg = socket.recv_multipart(flags)
118  
119      # transparently handle asyncio socket,
120      # returns a Future instead of a dict
121      if isinstance(msg, Awaitable):
122          return _parse_monitor_msg_async(msg)
123  
124      # 4.0-style event API
125      return parse_monitor_message(msg)
