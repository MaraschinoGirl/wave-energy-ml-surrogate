11   this_dir = os.path.abspath(os.path.dirname(__file__))
12   g_rcname = os.path.abspath(
13       os.path.join(this_dir, "..", "test", "win32rcparser", "test.rc")
14   )
15   
16   if not os.path.isfile(g_rcname):
17       raise RuntimeError(f"Can't locate test.rc (should be at '{g_rcname}')")

20   class DemoWindow:
21       def __init__(self, dlg_template):
22           self.dlg_template = dlg_template
23   
24       def CreateWindow(self):
25           self._DoCreate(win32gui.CreateDialogIndirect)
26   
27       def DoModal(self):
28           return self._DoCreate(win32gui.DialogBoxIndirect)
29   
30       def _DoCreate(self, fn):
31           message_map = {
32               win32con.WM_INITDIALOG: self.OnInitDialog,
33               win32con.WM_CLOSE: self.OnClose,
34               win32con.WM_DESTROY: self.OnDestroy,
35               win32con.WM_COMMAND: self.OnCommand,
36           }
37           return fn(0, self.dlg_template, 0, message_map)
38   
39       def OnInitDialog(self, hwnd, msg, wparam, lparam):
40           self.hwnd = hwnd
41           # centre the dialog
42           desktop = win32gui.GetDesktopWindow()
43           l, t, r, b = win32gui.GetWindowRect(self.hwnd)
44           dt_l, dt_t, dt_r, dt_b = win32gui.GetWindowRect(desktop)
45           centre_x, centre_y = win32gui.ClientToScreen(
46               desktop, ((dt_r - dt_l) // 2, (dt_b - dt_t) // 2)
47           )
48           win32gui.MoveWindow(
49               hwnd, centre_x - (r // 2), centre_y - (b // 2), r - l, b - t, 0
50           )
51   
52       def OnCommand(self, hwnd, msg, wparam, lparam):
53           # Needed to make OK/Cancel work - no other controls are handled.
54           id = win32api.LOWORD(wparam)
55           if id in [win32con.IDOK, win32con.IDCANCEL]:
56               win32gui.EndDialog(hwnd, id)
57   
58       def OnClose(self, hwnd, msg, wparam, lparam):
59           win32gui.EndDialog(hwnd, 0)
60   
61       def OnDestroy(self, hwnd, msg, wparam, lparam):
62           pass
63   
64   
65   def DemoModal():
66       # Load the .rc file.
67       resources = win32rcparser.Parse(g_rcname)
68       for id, ddef in resources.dialogs.items():
69           print("Displaying dialog", id)
70           w = DemoWindow(ddef)
71           w.DoModal()
