73   # def is_unhandled_exception(container_obj, py_db, frame, int last_raise_line, set raise_lines):
74   # ELSE
75   def is_unhandled_exception(container_obj, py_db, frame, last_raise_line, raise_lines):
76       # ENDIF
77       if frame.f_lineno in raise_lines:
78           return True
79   
80       else:
81           try_except_infos = container_obj.try_except_infos
82           if try_except_infos is None:
83               container_obj.try_except_infos = try_except_infos = py_db.collect_try_except_info(frame.f_code)
84   
85           if not try_except_infos:
86               # Consider the last exception as unhandled because there's no try..except in it.
87               return True
88           else:
89               # Now, consider only the try..except for the raise
90               valid_try_except_infos = []
91               for try_except_info in try_except_infos:
92                   if try_except_info.is_line_in_try_block(last_raise_line):
93                       valid_try_except_infos.append(try_except_info)
94   
95               if not valid_try_except_infos:
96                   return True
97   
98               else:
99                   # Note: check all, not only the "valid" ones to cover the case
100                  # in "tests_python.test_tracing_on_top_level.raise_unhandled10"
101                  # where one try..except is inside the other with only a raise
102                  # and it's gotten in the except line.
103                  for try_except_info in try_except_infos:
104                      if try_except_info.is_line_in_except_block(frame.f_lineno):
105                          if frame.f_lineno == try_except_info.except_line or frame.f_lineno in try_except_info.raise_lines_in_except:
106                              # In a raise inside a try..except block or some except which doesn't
107                              # match the raised exception.
108                              return True
109      return False

180          # ENDIF
181          if event == "exception":

311      # IFDEF CYTHON
312      # cpdef trace_dispatch(self, frame, str event, arg):
313      #     cdef tuple abs_path_canonical_path_and_base;
314      #     cdef bint is_exception_event;
315      #     cdef bint has_exception_breakpoints;
316      #     cdef bint can_skip;
317      #     cdef bint stop;
318      #     cdef bint stop_on_plugin_breakpoint;
319      #     cdef PyDBAdditionalThreadInfo info;
320      #     cdef int step_cmd;
321      #     cdef int line;
322      #     cdef bint is_line;
323      #     cdef bint is_call;
324      #     cdef bint is_return;
325      #     cdef bint should_stop;
326      #     cdef dict breakpoints_for_file;
327      #     cdef dict stop_info;
328      #     cdef str curr_func_name;
329      #     cdef dict frame_skips_cache;
330      #     cdef object frame_cache_key;
331      #     cdef tuple line_cache_key;
332      #     cdef int breakpoints_in_line_cache;
333      #     cdef int breakpoints_in_frame_cache;
334      #     cdef bint has_breakpoint_in_frame;
335      #     cdef bint is_coroutine_or_generator;
336      #     cdef int bp_line;
337      #     cdef object bp;
338      #     cdef int pydev_smart_parent_offset
339      #     cdef int pydev_smart_child_offset
340      #     cdef tuple pydev_smart_step_into_variants
341      # ELSE
342      def trace_dispatch(self, frame, event, arg):
343          # ENDIF
344          # Note: this is a big function because most of the logic related to hitting a breakpoint and
345          # stepping is contained in it. Ideally this could be split among multiple functions, but the
346          # problem in this case is that in pure-python function calls are expensive and even more so
347          # when tracing is on (because each function call will get an additional tracing call). We
348          # try to address this by using the info.is_tracing for the fastest possible return, but the
349          # cost is still high (maybe we could use code-generation in the future and make the code
350          # generation be better split among what each part does).
351  
352          try:
353              # DEBUG = '_debugger_case_yield_from.py' in frame.f_code.co_filename
354              py_db, abs_path_canonical_path_and_base, info, thread, frame_skips_cache, frame_cache_key = self._args
355              # if DEBUG: print('frame trace_dispatch %s %s %s %s %s %s, stop: %s' % (frame.f_lineno, frame.f_code.co_name, frame.f_code.co_filename, event, constant_to_str(info.pydev_step_cmd), arg, info.pydev_step_stop))
356              info.is_tracing += 1
357  
358              # TODO: This shouldn't be needed. The fact that frame.f_lineno
359              # is None seems like a bug in Python 3.11.
360              # Reported in: https://github.com/python/cpython/issues/94485
361              line = frame.f_lineno or 0  # Workaround or case where frame.f_lineno is None
362              line_cache_key = (frame_cache_key, line)
363  
364              if py_db.pydb_disposed:
365                  return None if event == "call" else NO_FTRACE
366  
367              plugin_manager = py_db.plugin
368              has_exception_breakpoints = (
369                  py_db.break_on_caught_exceptions or py_db.break_on_user_uncaught_exceptions or py_db.has_plugin_exception_breaks
370              )
371  
372              stop_frame = info.pydev_step_stop
373              step_cmd = info.pydev_step_cmd
374              function_breakpoint_on_call_event = None
375  
376              if frame.f_code.co_flags & 0xA0:  # 0xa0 ==  CO_GENERATOR = 0x20 | CO_COROUTINE = 0x80
377                  # Dealing with coroutines and generators:
378                  # When in a coroutine we change the perceived event to the debugger because
379                  # a call, StopIteration exception and return are usually just pausing/unpausing it.
380                  if event == "line":
381                      is_line = True
382                      is_call = False
383                      is_return = False
384                      is_exception_event = False
385  
386                  elif event == "return":
387                      is_line = False
388                      is_call = False
389                      is_return = True
390                      is_exception_event = False
391  
392                      returns_cache_key = (frame_cache_key, "returns")
393                      return_lines = frame_skips_cache.get(returns_cache_key)
394                      if return_lines is None:
395                          # Note: we're collecting the return lines by inspecting the bytecode as
396                          # there are multiple returns and multiple stop iterations when awaiting and
397                          # it doesn't give any clear indication when a coroutine or generator is
398                          # finishing or just pausing.
399                          return_lines = set()
400                          for x in py_db.collect_return_info(frame.f_code):
401                              # Note: cython does not support closures in cpdefs (so we can't use
402                              # a list comprehension).
403                              return_lines.add(x.return_line)
404  
405                          frame_skips_cache[returns_cache_key] = return_lines
406  
407                      if line not in return_lines:
408                          # Not really a return (coroutine/generator paused).
409                          return self.trace_dispatch
410                      else:
411                          if self.exc_info:
412                              self.handle_user_exception(frame)
413                              return self.trace_dispatch
414  
415                          # Tricky handling: usually when we're on a frame which is about to exit
416                          # we set the step mode to step into, but in this case we'd end up in the
417                          # asyncio internal machinery, which is not what we want, so, we just
418                          # ask the stop frame to be a level up.
419                          #
420                          # Note that there's an issue here which we may want to fix in the future: if
421                          # the back frame is a frame which is filtered, we won't stop properly.
422                          # Solving this may not be trivial as we'd need to put a scope in the step
423                          # in, but we may have to do it anyways to have a step in which doesn't end
424                          # up in asyncio).
425                          #
426                          # Note2: we don't revert to a step in if we're doing scoped stepping
427                          # (because on scoped stepping we're always receiving a call/line/return
428                          # event for each line in ipython, so, we can't revert to step in on return
429                          # as the return shouldn't mean that we've actually completed executing a
430                          # frame in this case).
431                          if stop_frame is frame and not info.pydev_use_scoped_step_frame:
432                              if step_cmd in (CMD_STEP_OVER, CMD_STEP_OVER_MY_CODE, CMD_STEP_INTO, CMD_STEP_INTO_MY_CODE):
433                                  f = self._get_unfiltered_back_frame(py_db, frame)
434                                  if f is not None:
435                                      info.pydev_step_cmd = CMD_STEP_INTO_COROUTINE
436                                      info.pydev_step_stop = f
437                                  else:
438                                      if step_cmd == CMD_STEP_OVER:
439                                          info.pydev_step_cmd = CMD_STEP_INTO
440                                          info.pydev_step_stop = None
441  
442                                      elif step_cmd == CMD_STEP_OVER_MY_CODE:
443                                          info.pydev_step_cmd = CMD_STEP_INTO_MY_CODE
444                                          info.pydev_step_stop = None
445  
446                              elif step_cmd == CMD_STEP_INTO_COROUTINE:
447                                  # We're exiting this one, so, mark the new coroutine context.
448                                  f = self._get_unfiltered_back_frame(py_db, frame)
449                                  if f is not None:
450                                      info.pydev_step_stop = f
451                                  else:
452                                      info.pydev_step_cmd = CMD_STEP_INTO
453                                      info.pydev_step_stop = None
454  
455                  elif event == "exception":
456                      breakpoints_for_file = None
457                      if has_exception_breakpoints:
458                          should_stop, frame, exc_info = should_stop_on_exception(
459                              self._args[0], self._args[2], frame, self._args[3], arg, self.exc_info
460                          )
461                          self.exc_info = exc_info
462                          if should_stop:
463                              if handle_exception(self._args[0], self._args[3], frame, arg, EXCEPTION_TYPE_HANDLED):
464                                  return self.trace_dispatch
465  
466                      return self.trace_dispatch
467                  else:
468                      # event == 'call' or event == 'c_XXX'
469                      return self.trace_dispatch
470  
471              else:  # Not coroutine nor generator
472                  if event == "line":
473                      is_line = True
474                      is_call = False
475                      is_return = False
476                      is_exception_event = False
477  
478                  elif event == "return":
479                      is_line = False
480                      is_return = True
481                      is_call = False
482                      is_exception_event = False
483  
484                      # If we are in single step mode and something causes us to exit the current frame, we need to make sure we break
485                      # eventually.  Force the step mode to step into and the step stop frame to None.
486                      # I.e.: F6 in the end of a function should stop in the next possible position (instead of forcing the user
487                      # to make a step in or step over at that location).
488                      # Note: this is especially troublesome when we're skipping code with the
489                      # @DontTrace comment.
490                      if (
491                          stop_frame is frame
492                          and not info.pydev_use_scoped_step_frame
493                          and is_return
494                          and step_cmd
495                          in (CMD_STEP_OVER, CMD_STEP_RETURN, CMD_STEP_OVER_MY_CODE, CMD_STEP_RETURN_MY_CODE, CMD_SMART_STEP_INTO)
496                      ):
497                          if step_cmd in (CMD_STEP_OVER, CMD_STEP_RETURN, CMD_SMART_STEP_INTO):
498                              info.pydev_step_cmd = CMD_STEP_INTO
499                          else:
500                              info.pydev_step_cmd = CMD_STEP_INTO_MY_CODE
501                          info.pydev_step_stop = None
502  
503                      if self.exc_info:
504                          if self.handle_user_exception(frame):
505                              return self.trace_dispatch
506  
507                  elif event == "call":
508                      is_line = False
509                      is_call = True
510                      is_return = False
511                      is_exception_event = False
512                      if frame.f_code.co_firstlineno == frame.f_lineno:  # Check line to deal with async/await.
513                          function_breakpoint_on_call_event = py_db.function_breakpoint_name_to_breakpoint.get(frame.f_code.co_name)
514  
515                  elif event == "exception":
516                      is_exception_event = True
517                      breakpoints_for_file = None
518                      if has_exception_breakpoints:
519                          should_stop, frame, exc_info = should_stop_on_exception(
520                              self._args[0], self._args[2], frame, self._args[3], arg, self.exc_info
521                          )
522                          self.exc_info = exc_info
523                          if should_stop:
524                              if handle_exception(self._args[0], self._args[3], frame, arg, EXCEPTION_TYPE_HANDLED):
525                                  return self.trace_dispatch
526                      is_line = False
527                      is_return = False
528                      is_call = False
529  
530                  else:
531                      # Unexpected: just keep the same trace func (i.e.: event == 'c_XXX').
532                      return self.trace_dispatch
533  
534              if not is_exception_event:
535                  breakpoints_for_file = py_db.breakpoints.get(abs_path_canonical_path_and_base[1])
536  
537                  can_skip = False
538  
539                  if info.pydev_state == 1:  # STATE_RUN = 1
540                      # we can skip if:
541                      # - we have no stop marked
542                      # - we should make a step return/step over and we're not in the current frame
543                      # - we're stepping into a coroutine context and we're not in that context
544                      if step_cmd == -1:
545                          can_skip = True
546  
547                      elif step_cmd in (
548                          CMD_STEP_OVER,
549                          CMD_STEP_RETURN,
550                          CMD_STEP_OVER_MY_CODE,
551                          CMD_STEP_RETURN_MY_CODE,
552                      ) and not self._is_same_frame(stop_frame, frame):
553                          can_skip = True
554  
555                      elif step_cmd == CMD_SMART_STEP_INTO and (
556                          stop_frame is not None
557                          and stop_frame is not frame
558                          and stop_frame is not frame.f_back
559                          and (frame.f_back is None or stop_frame is not frame.f_back.f_back)
560                      ):
561                          can_skip = True
562  
563                      elif step_cmd == CMD_STEP_INTO_MY_CODE:
564                          if py_db.apply_files_filter(frame, frame.f_code.co_filename, True) and (
565                              frame.f_back is None or py_db.apply_files_filter(frame.f_back, frame.f_back.f_code.co_filename, True)
566                          ):

568  
569                      elif step_cmd == CMD_STEP_INTO_COROUTINE:
570                          f = frame
571                          while f is not None:
572                              if self._is_same_frame(stop_frame, f):
573                                  break
574                              f = f.f_back
575                          else:
576                              can_skip = True
577  
578                      if can_skip:
579                          if plugin_manager is not None and (py_db.has_plugin_line_breaks or py_db.has_plugin_exception_breaks):
580                              can_skip = plugin_manager.can_skip(py_db, frame)
581  
582                          if (
583                              can_skip
584                              and py_db.show_return_values
585                              and info.pydev_step_cmd in (CMD_STEP_OVER, CMD_STEP_OVER_MY_CODE)
586                              and self._is_same_frame(stop_frame, frame.f_back)
587                          ):
588                              # trace function for showing return values after step over
589                              can_skip = False
590  
591                  # Let's check to see if we are in a function that has a breakpoint. If we don't have a breakpoint,
592                  # we will return nothing for the next trace
593                  # also, after we hit a breakpoint and go to some other debugging state, we have to force the set trace anyway,
594                  # so, that's why the additional checks are there.
595  
596                  if function_breakpoint_on_call_event:
597                      pass  # Do nothing here (just keep on going as we can't skip it).
598  
599                  elif not breakpoints_for_file:
600                      if can_skip:
601                          if has_exception_breakpoints:
602                              return self.trace_exception
603                          else:
604                              return None if is_call else NO_FTRACE
605  
606                  else:
607                      # When cached, 0 means we don't have a breakpoint and 1 means we have.
608                      if can_skip:
609                          breakpoints_in_line_cache = frame_skips_cache.get(line_cache_key, -1)
610                          if breakpoints_in_line_cache == 0:
611                              return self.trace_dispatch

613                      breakpoints_in_frame_cache = frame_skips_cache.get(frame_cache_key, -1)
614                      if breakpoints_in_frame_cache != -1:
615                          # Gotten from cache.
616                          has_breakpoint_in_frame = breakpoints_in_frame_cache == 1

619                          has_breakpoint_in_frame = False

621                          try:
622                              func_lines = set()
623                              for offset_and_lineno in dis.findlinestarts(frame.f_code):
624                                  if offset_and_lineno[1] is not None:
625                                      func_lines.add(offset_and_lineno[1])
626                          except:
627                              # This is a fallback for implementations where we can't get the function
628                              # lines -- i.e.: jython (in this case clients need to provide the function
629                              # name to decide on the skip or we won't be able to skip the function
630                              # completely).
631  
632                              # Checks the breakpoint to see if there is a context match in some function.
633                              curr_func_name = frame.f_code.co_name
634  
635                              # global context is set with an empty name
636                              if curr_func_name in ("?", "<module>", "<lambda>"):
637                                  curr_func_name = ""
638  
639                              for bp in breakpoints_for_file.values():
640                                  # will match either global or some function
641                                  if bp.func_name in ("None", curr_func_name):
642                                      has_breakpoint_in_frame = True
643                                      break
644                          else:
645                              for bp_line in breakpoints_for_file:  # iterate on keys
646                                  if bp_line in func_lines:
647                                      has_breakpoint_in_frame = True
648                                      break
649  
650                          # Cache the value (1 or 0 or -1 for default because of cython).
651                          if has_breakpoint_in_frame:
652                              frame_skips_cache[frame_cache_key] = 1
653                          else:
654                              frame_skips_cache[frame_cache_key] = 0
655  
656                      if can_skip and not has_breakpoint_in_frame:
657                          if has_exception_breakpoints:
658                              return self.trace_exception
659                          else:
660                              return None if is_call else NO_FTRACE
661  
662              # We may have hit a breakpoint or we are already in step mode. Either way, let's check what we should do in this frame
663              # if DEBUG: print('NOT skipped: %s %s %s %s' % (frame.f_lineno, frame.f_code.co_name, event, frame.__class__.__name__))
664  
665              try:
666                  stop_on_plugin_breakpoint = False
667                  # return is not taken into account for breakpoint hit because we'd have a double-hit in this case
668                  # (one for the line and the other for the return).
669  
670                  stop_info = {}
671                  breakpoint = None
672                  stop = False
673                  stop_reason = CMD_SET_BREAK
674                  bp_type = None
675  
676                  if function_breakpoint_on_call_event:
677                      breakpoint = function_breakpoint_on_call_event
678                      stop = True
679                      new_frame = frame
680                      stop_reason = CMD_SET_FUNCTION_BREAK
681  
682                  elif is_line and info.pydev_state != STATE_SUSPEND and breakpoints_for_file is not None and line in breakpoints_for_file:
683                      breakpoint = breakpoints_for_file[line]
684                      new_frame = frame
685                      stop = True
686  
687                  elif plugin_manager is not None and py_db.has_plugin_line_breaks:
688                      result = plugin_manager.get_breakpoint(py_db, frame, event, self._args[2])
689                      if result:
690                          stop_on_plugin_breakpoint = True
691                          breakpoint, new_frame, bp_type = result
692  
693                  if breakpoint:
694                      # ok, hit breakpoint, now, we have to discover if it is a conditional breakpoint
695                      # lets do the conditional stuff here
696                      if breakpoint.expression is not None:
697                          py_db.handle_breakpoint_expression(breakpoint, info, new_frame)
698  
699                      if stop or stop_on_plugin_breakpoint:
700                          eval_result = False
701                          if breakpoint.has_condition:
702                              eval_result = py_db.handle_breakpoint_condition(info, breakpoint, new_frame)
703                              if not eval_result:
704                                  stop = False
705                                  stop_on_plugin_breakpoint = False
706  
707                      if is_call and (
708                          frame.f_code.co_name in ("<lambda>", "<module>") or (line == 1 and frame.f_code.co_name.startswith("<cell"))
709                      ):
710                          # If we find a call for a module, it means that the module is being imported/executed for the
711                          # first time. In this case we have to ignore this hit as it may later duplicated by a
712                          # line event at the same place (so, if there's a module with a print() in the first line
713                          # the user will hit that line twice, which is not what we want).
714                          #
715                          # For lambda, as it only has a single statement, it's not interesting to trace
716                          # its call and later its line event as they're usually in the same line.
717                          #
718                          # For ipython, <cell xxx> may be executed having each line compiled as a new
719                          # module, so it's the same case as <module>.

721                          return self.trace_dispatch
722  
723                      # Handle logpoint (on a logpoint we should never stop).
724                      if (stop or stop_on_plugin_breakpoint) and breakpoint.is_logpoint:
725                          stop = False
726                          stop_on_plugin_breakpoint = False
727  
728                          if info.pydev_message is not None and len(info.pydev_message) > 0:
729                              cmd = py_db.cmd_factory.make_io_message(info.pydev_message + os.linesep, "1")
730                              py_db.writer.add_command(cmd)
731  
732                  if py_db.show_return_values:
733                      if is_return and (
734                          (
735                              info.pydev_step_cmd in (CMD_STEP_OVER, CMD_STEP_OVER_MY_CODE, CMD_SMART_STEP_INTO)
736                              and (self._is_same_frame(stop_frame, frame.f_back))
737                          )
738                          or (info.pydev_step_cmd in (CMD_STEP_RETURN, CMD_STEP_RETURN_MY_CODE) and (self._is_same_frame(stop_frame, frame)))
739                          or (info.pydev_step_cmd in (CMD_STEP_INTO, CMD_STEP_INTO_COROUTINE))

741                              info.pydev_step_cmd == CMD_STEP_INTO_MY_CODE

743                              and not py_db.apply_files_filter(frame.f_back, frame.f_back.f_code.co_filename, True)
744                          )
745                      ):

748                  elif py_db.remove_return_values_flag:
749                      try:
750                          self._remove_return_values(py_db, frame)

752                          py_db.remove_return_values_flag = False

754                  if stop:

756                          thread,
757                          stop_reason,
758                          suspend_other_threads=breakpoint and breakpoint.suspend_policy == "ALL",

760  
761                  elif stop_on_plugin_breakpoint and plugin_manager is not None:

764                          frame = result

766                  # if thread has a suspend flag, we suspend with a busy wait

770                  else:

777                  # the tracing function, so, we can't do much about it (just let the user know).

783                          thread,

790                  raise

804                              should_skip = self.should_skip = 0

808                  plugin_stop = False

810                      stop = False
811  
812                  elif step_cmd in (CMD_STEP_INTO, CMD_STEP_INTO_MY_CODE, CMD_STEP_INTO_COROUTINE):

815                          if not info.pydev_use_scoped_step_frame:
816                              if force_check_project_scope or py_db.is_files_filter_enabled:

819                                  stop = True

826                              # We can only stop inside the ipython call.
827                              filename = frame.f_code.co_filename
828                              if filename.endswith(".pyc"):
829                                  filename = filename[:-1]
830  
831                              if not filename.endswith(PYDEVD_IPYTHON_CONTEXT[0]):

833                                  while f is not None:
834                                      if f.f_code.co_name == PYDEVD_IPYTHON_CONTEXT[1]:
835                                          f2 = f.f_back
836                                          if f2 is not None and f2.f_code.co_name == PYDEVD_IPYTHON_CONTEXT[2]:

842                                  del f
843  
844                              if not stop:
845                                  # In scoped mode if step in didn't work in this context it won't work
846                                  # afterwards anyways.
847                                  return None if is_call else NO_FTRACE

851                              stop = False
852                          else:
853                              if force_check_project_scope or py_db.is_files_filter_enabled:

855                                      frame.f_back, frame.f_back.f_code.co_filename, force_check_project_scope
856                                  )
857                                  if stop:

859                                      # (i.e.: double-stop at the same place due to some filtering).

861                                          stop = False
862                              else:
863                                  stop = True
864                      else:
865                          stop = False
866  
867                      if stop:
868                          if step_cmd == CMD_STEP_INTO_COROUTINE:
869                              # i.e.: Check if we're stepping into the proper context.
870                              f = frame
871                              while f is not None:
872                                  if self._is_same_frame(stop_frame, f):
873                                      break
874                                  f = f.f_back
875                              else:
876                                  stop = False
877  
878                      if plugin_manager is not None:

882  
883                  elif step_cmd in (CMD_STEP_OVER, CMD_STEP_OVER_MY_CODE):

885                      # difference is that when we return from a frame in one we go to regular step

889                      # i.e.: don't stop in: (stop_frame is frame.f_back and is_return) as we'd stop twice in that line.

891                      if plugin_manager is not None:
892                          result = plugin_manager.cmd_step_over(py_db, frame, event, self._args[2], self._args[3], stop_info, stop)
893                          if result:
894                              stop, plugin_stop = result
895  
896                  elif step_cmd == CMD_SMART_STEP_INTO:
897                      stop = False
898                      back = frame.f_back
899                      if self._is_same_frame(stop_frame, frame) and is_return:
900                          # We're exiting the smart step into initial frame (so, we probably didn't find our target).
901                          stop = True
902  
903                      elif self._is_same_frame(stop_frame, back) and is_line:
904                          if info.pydev_smart_child_offset != -1:
905                              # i.e.: in this case, we're not interested in the pause in the parent, rather
906                              # we're interested in the pause in the child (when the parent is at the proper place).

908  
909                          else:
910                              pydev_smart_parent_offset = info.pydev_smart_parent_offset
911  
912                              pydev_smart_step_into_variants = info.pydev_smart_step_into_variants
913                              if pydev_smart_parent_offset >= 0 and pydev_smart_step_into_variants:
914                                  # Preferred mode (when the smart step into variants are available
915                                  # and the offset is set).
916                                  stop = get_smart_step_into_variant_from_frame_offset(
917                                      back.f_lasti, pydev_smart_step_into_variants
918                                  ) is get_smart_step_into_variant_from_frame_offset(
919                                      pydev_smart_parent_offset, pydev_smart_step_into_variants
920                                  )
921  
922                              else:
923                                  # Only the name/line is available, so, check that.
924                                  curr_func_name = frame.f_code.co_name
925  
926                                  # global context is set with an empty name
927                                  if curr_func_name in ("?", "<module>") or curr_func_name is None:
928                                      curr_func_name = ""
929                                  if curr_func_name == info.pydev_func_name and stop_frame.f_lineno == info.pydev_next_line:
930                                      stop = True
931  
932                          if not stop:
933                              # In smart step into, if we didn't hit it in this frame once, that'll
934                              # not be the case next time either, so, disable tracing for this frame.
935                              return None if is_call else NO_FTRACE
936  
937                      elif back is not None and self._is_same_frame(stop_frame, back.f_back) and is_line:
938                          # Ok, we have to track 2 stops at this point, the parent and the child offset.
939                          # This happens when handling a step into which targets a function inside a list comprehension
940                          # or generator (in which case an intermediary frame is created due to an internal function call).
941                          pydev_smart_parent_offset = info.pydev_smart_parent_offset
942                          pydev_smart_child_offset = info.pydev_smart_child_offset
943                          # print('matched back frame', pydev_smart_parent_offset, pydev_smart_child_offset)
944                          # print('parent f_lasti', back.f_back.f_lasti)
945                          # print('child f_lasti', back.f_lasti)
946                          stop = False
947                          if pydev_smart_child_offset >= 0 and pydev_smart_child_offset >= 0:
948                              pydev_smart_step_into_variants = info.pydev_smart_step_into_variants
949  
950                              if pydev_smart_parent_offset >= 0 and pydev_smart_step_into_variants:
951                                  # Note that we don't really check the parent offset, only the offset of
952                                  # the child (because this is a generator, the parent may have moved forward
953                                  # already -- and that's ok, so, we just check that the parent frame
954                                  # matches in this case).
955                                  smart_step_into_variant = get_smart_step_into_variant_from_frame_offset(
956                                      pydev_smart_parent_offset, pydev_smart_step_into_variants
957                                  )
958                                  # print('matched parent offset', pydev_smart_parent_offset)

962                                      get_smart_step_into_variant_from_frame_offset(back.f_lasti, children_variants)

964                                  )
965                                  # print('stop at child', stop)

967                          if not stop:

969                              # not be the case next time either, so, disable tracing for this frame.
970                              return None if is_call else NO_FTRACE
971  
972                  elif step_cmd in (CMD_STEP_RETURN, CMD_STEP_RETURN_MY_CODE):
973                      stop = is_return and self._is_same_frame(stop_frame, frame)
974  
975                  else:
976                      stop = False
977  
978                  if stop and step_cmd != -1 and is_return and hasattr(frame, "f_back"):

980                      if f_code is not None:
981                          if py_db.get_file_type(frame.f_back) == py_db.PYDEV_FILE:
982                              stop = False
983  
984                  if plugin_stop:
985                      plugin_manager.stop(py_db, frame, event, self._args[3], stop_info, arg, step_cmd)

988                          self.set_suspend(thread, step_cmd, original_step_cmd=info.pydev_original_step_cmd)
989                          self.do_wait_suspend(thread, frame, event, arg)
990                      elif is_return:  # return event
991                          back = frame.f_back
992                          if back is not None:
993                              # When we get to the pydevd run function, the debugging has actually finished for the main thread
994                              # (note that it can still go on for other threads, but for this one, we just make it finish)

998                                  back = None
999  
1000                             elif base == TRACE_PROPERTY:
1001                                 # We dont want to trace the return event of pydevd_traceproperty (custom property for debugging)
1002                                 # if we're in a return, we want it to appear to the user in the previous frame!
1003                                 return None if is_call else NO_FTRACE
1004 
1005                             elif pydevd_dont_trace.should_trace_hook is not None:

1008                                     # Also, we have to reset the tracing, because if the parent's parent (or some
1009                                     # other parent) has to be traced and it's not currently, we wouldn't stop where
1010                                     # we should anymore (so, a step in/over/return may not stop anywhere if no parent is traced).
1011                                     # Related test: _debugger_case17a.py
1012                                     py_db.set_trace_for_frame_and_parents(thread.ident, back)
1013                                     return None if is_call else NO_FTRACE
1014 
1015                         if back is not None:
1016                             # if we're in a return, we want it to appear to the user in the previous frame!
1017                             self.set_suspend(thread, step_cmd, original_step_cmd=info.pydev_original_step_cmd)
1018                             self.do_wait_suspend(thread, back, event, arg)
1019                         else:
1020                             # in jython we may not have a back frame
1021                             info.pydev_step_stop = None
1022                             info.pydev_original_step_cmd = -1
1023                             info.pydev_step_cmd = -1
1024                             info.pydev_state = STATE_RUN
1025                             info.update_stepping_info()
1026 
1027                 # if we are quitting, let's stop the tracing
1028                 if py_db.quitting:
1029                     return None if is_call else NO_FTRACE
1030 
1031                 return self.trace_dispatch
1032             except:
1033                 # Unfortunately Python itself stops the tracing when it originates from
1034                 # the tracing function, so, we can't do much about it (just let the user know).
1035                 exc = sys.exc_info()[0]
1036                 cmd = py_db.cmd_factory.make_console_message(
1037                     "%s raised from within the callback set in sys.settrace.\nDebugging will be disabled for this thread (%s).\n"
1038                     % (
1039                         exc,
1040                         thread,
1041                     )
1042                 )
1043                 py_db.writer.add_command(cmd)
1044                 if not issubclass(exc, (KeyboardInterrupt, SystemExit)):
1045                     pydev_log.exception()
1046                 raise
1047 
1048         finally:
1049             info.is_tracing -= 1
1050 
1051         # end trace_dispatch

1131                                 # up the stack is library code.

1153                                 exc_break.notify_on_first_raise_only
1154                                 and not py_db.skip_on_exceptions_thrown_in_same_context

1173                 add_exception_to_frame(frame, (exception, value, trace))
1174 
1175                 if exception_breakpoint is not None and exception_breakpoint.expression is not None:
1176                     py_db.handle_breakpoint_expression(exception_breakpoint, info, frame)
