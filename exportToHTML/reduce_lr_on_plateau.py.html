3    import numpy as np
4    
5    from keras.src import backend
6    from keras.src.api_export import keras_export
7    from keras.src.callbacks.callback import Callback
8    from keras.src.utils import io_utils
9

11   @keras_export("keras.callbacks.ReduceLROnPlateau")
12   class ReduceLROnPlateau(Callback):
13       """Reduce learning rate when a metric has stopped improving. 
14    
15       Models often benefit from reducing the learning rate by a factor 
16       of 2-10 once learning stagnates. This callback monitors a 
17       quantity and if no improvement is seen for a 'patience' number 
18       of epochs, the learning rate is reduced. 
19    
20       Example: 
21    
22       ```python 
23       reduce_lr = ReduceLROnPlateau(monitor='val_loss', factor=0.2, 
24                                     patience=5, min_lr=0.001) 
25       model.fit(x_train, y_train, callbacks=[reduce_lr]) 
26       ``` 
27    
28       Args: 
29           monitor: String. Quantity to be monitored. 
30           factor: Float. Factor by which the learning rate will be reduced. 
31               `new_lr = lr * factor`. 
32           patience: Integer. Number of epochs with no improvement after which 
33               learning rate will be reduced. 
34           verbose: Integer. 0: quiet, 1: update messages. 
35           mode: String. One of `{'auto', 'min', 'max'}`. In `'min'` mode, 
36               the learning rate will be reduced when the 
37               quantity monitored has stopped decreasing; in `'max'` mode it will 
38               be reduced when the quantity monitored has stopped increasing; in 
39               `'auto'` mode, the direction is automatically inferred from the name 
40               of the monitored quantity. 
41           min_delta: Float. Threshold for measuring the new optimum, to only focus 
42               on significant changes. 
43           cooldown: Integer. Number of epochs to wait before resuming normal 
44               operation after the learning rate has been reduced. 
45           min_lr: Float. Lower bound on the learning rate. 
46       """
47   
48       def __init__(
49           self,
50           monitor="val_loss",
51           factor=0.1,
52           patience=10,
53           verbose=0,
54           mode="auto",
55           min_delta=1e-4,
56           cooldown=0,
57           min_lr=0.0,
58           **kwargs,
59       ):
60           super().__init__()
61   
62           self.monitor = monitor
63           if factor >= 1.0:
64               raise ValueError(
65                   "ReduceLROnPlateau does not support a factor >= 1.0. "
66                   f"Received factor={factor}"
67               )
68   
69           self.factor = factor
70           self.min_lr = min_lr
71           self.min_delta = min_delta
72           self.patience = patience
73           self.verbose = verbose
74           self.cooldown = cooldown
75           self.cooldown_counter = 0  # Cooldown counter.
76           self.wait = 0
77           self.best = 0
78           self.mode = mode
79           self.monitor_op = None
80           self._reset()
81   
82       def _reset(self):
83           """Resets wait counter and cooldown counter."""
84           if self.mode not in {"auto", "min", "max"}:
85               warnings.warn(
86                   f"Learning rate reduction mode {self.mode} is unknown, "
87                   "fallback to auto mode.",
88                   stacklevel=2,
89               )
90               self.mode = "auto"
91           if self.mode == "min" or (
92               self.mode == "auto" and "acc" not in self.monitor
93           ):
94               self.monitor_op = lambda a, b: np.less(a, b - self.min_delta)
95               self.best = np.inf
96           else:
97               self.monitor_op = lambda a, b: np.greater(a, b + self.min_delta)
98               self.best = -np.inf
99           self.cooldown_counter = 0
100          self.wait = 0
101  
102      def on_train_begin(self, logs=None):
103          self._reset()
104  
105      def on_epoch_end(self, epoch, logs=None):
106          logs = logs or {}
107          logs["learning_rate"] = float(
108              backend.convert_to_numpy(self.model.optimizer.learning_rate)
109          )
110          current = logs.get(self.monitor)
111  
112          if current is None:
113              warnings.warn(
114                  "Learning rate reduction is conditioned on metric "
115                  f"`{self.monitor}` which is not available. Available metrics "
116                  f"are: {','.join(list(logs.keys()))}.",
117                  stacklevel=2,
118              )
119          else:
120              if self.in_cooldown():
121                  self.cooldown_counter -= 1
122                  self.wait = 0
123  
124              if self.monitor_op(current, self.best):
125                  self.best = current
126                  self.wait = 0
127              elif not self.in_cooldown():
128                  self.wait += 1
129                  if self.wait >= self.patience:
130                      old_lr = float(
131                          backend.convert_to_numpy(
132                              self.model.optimizer.learning_rate
133                          )
134                      )
135                      if old_lr > np.float32(self.min_lr):
136                          new_lr = old_lr * self.factor
137                          new_lr = max(new_lr, self.min_lr)
138                          self.model.optimizer.learning_rate = new_lr
139                          if self.verbose > 0:
140                              io_utils.print_msg(
141                                  f"\nEpoch {epoch + 1}: "
142                                  "ReduceLROnPlateau reducing "
143                                  f"learning rate to {new_lr}."
144                              )
145                          self.cooldown_counter = self.cooldown
146                          self.wait = 0
147  
148      def in_cooldown(self):
149          return self.cooldown_counter > 0
