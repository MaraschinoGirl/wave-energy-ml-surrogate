38   def assert_csc_almost_equal(r, l):
39       r = csc_array(r)
40       l = csc_array(l)
41       assert_equal(r.indptr, l.indptr)
42       assert_equal(r.indices, l.indices)
43       assert_array_almost_equal_nulp(r.data, l.data, 10000)

46   class TestHBReader:
47       def test_simple(self):
48           m = hb_read(StringIO(SIMPLE), spmatrix=False)
49           assert_csc_almost_equal(m, SIMPLE_MATRIX)
50           assert not isspmatrix(m)
51           m = hb_read(StringIO(SIMPLE), spmatrix=True)
52           assert isspmatrix(m)
53           m = hb_read(StringIO(SIMPLE))  # default
54           assert isspmatrix(m)

57   class TestHBReadWrite:
58   
59       def check_save_load(self, value):
60           with tempfile.NamedTemporaryFile(mode='w+t') as file:
61               hb_write(file, value)
62               file.file.seek(0)
63               value_loaded = hb_read(file, spmatrix=False)
64           assert_csc_almost_equal(value, value_loaded)
65   
66       def test_simple(self):
67           random_arr = random_array((10, 100), density=0.1)
68           for format in ('coo', 'csc', 'csr', 'bsr', 'dia', 'dok', 'lil'):
69               arr = random_arr.asformat(format, copy=False)
70               self.check_save_load(arr)
