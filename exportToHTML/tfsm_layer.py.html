8    @keras_export("keras.layers.TFSMLayer")
9    class TFSMLayer(layers.Layer):
10       """Reload a Keras model/layer that was saved via SavedModel / ExportArchive. 
11    
12       Arguments: 
13           filepath: `str` or `pathlib.Path` object. The path to the SavedModel. 
14           call_endpoint: Name of the endpoint to use as the `call()` method 
15               of the reloaded layer. If the SavedModel was created 
16               via `model.export()`, 
17               then the default endpoint name is `'serve'`. In other cases 
18               it may be named `'serving_default'`. 
19    
20       Example: 
21    
22       ```python 
23       model.export("path/to/artifact") 
24       reloaded_layer = TFSMLayer("path/to/artifact") 
25       outputs = reloaded_layer(inputs) 
26       ``` 
27    
28       The reloaded object can be used like a regular Keras layer, and supports 
29       training/fine-tuning of its trainable weights. Note that the reloaded 
30       object retains none of the internal structure or custom methods of the 
31       original object -- it's a brand new layer created around the saved 
32       function. 
33    
34       **Limitations:** 
35    
36       * Only call endpoints with a single `inputs` tensor argument 
37       (which may optionally be a dict/tuple/list of tensors) are supported. 
38       For endpoints with multiple separate input tensor arguments, consider 
39       subclassing `TFSMLayer` and implementing a `call()` method with a 
40       custom signature. 
41       * If you need training-time behavior to differ from inference-time behavior 
42       (i.e. if you need the reloaded object to support a `training=True` argument 
43       in `__call__()`), make sure that the training-time call function is 
44       saved as a standalone endpoint in the artifact, and provide its name 
45       to the `TFSMLayer` via the `call_training_endpoint` argument. 
46       """
47   
48       def __init__(
49           self,
50           filepath,
51           call_endpoint="serve",
52           call_training_endpoint=None,
53           trainable=True,
54           name=None,
55           dtype=None,
56       ):
57           if backend.backend() != "tensorflow":
58               raise NotImplementedError(
59                   "The TFSMLayer is only currently supported with the "
60                   "TensorFlow backend."
61               )
62   
63           # Initialize an empty layer, then add_weight() etc. as needed.
64           super().__init__(trainable=trainable, name=name, dtype=dtype)
65   
66           self._reloaded_obj = tf.saved_model.load(filepath)
67   
68           self.filepath = filepath
69           self.call_endpoint = call_endpoint
70           self.call_training_endpoint = call_training_endpoint
71   
72           # Resolve the call function.
73           if hasattr(self._reloaded_obj, call_endpoint):
74               # Case 1: it's set as an attribute.
75               self.call_endpoint_fn = getattr(self._reloaded_obj, call_endpoint)
76           elif call_endpoint in self._reloaded_obj.signatures:
77               # Case 2: it's listed in the `signatures` field.
78               self.call_endpoint_fn = self._reloaded_obj.signatures[call_endpoint]
79           else:
80               raise ValueError(
81                   f"The endpoint '{call_endpoint}' "
82                   "is neither an attribute of the reloaded SavedModel, "
83                   "nor an entry in the `signatures` field of "
84                   "the reloaded SavedModel. Select another endpoint via "
85                   "the `call_endpoint` argument. Available endpoints for "
86                   "this SavedModel: "
87                   f"{list(self._reloaded_obj.signatures.keys())}"
88               )
89   
90           # Resolving the training function.
91           if call_training_endpoint:
92               if hasattr(self._reloaded_obj, call_training_endpoint):
93                   self.call_training_endpoint_fn = getattr(
94                       self._reloaded_obj, call_training_endpoint
95                   )
96               elif call_training_endpoint in self._reloaded_obj.signatures:
97                   self.call_training_endpoint_fn = self._reloaded_obj.signatures[
98                       call_training_endpoint
99                   ]
100              else:
101                  raise ValueError(
102                      f"The endpoint '{call_training_endpoint}' "
103                      "is neither an attribute of the reloaded SavedModel, "
104                      "nor an entry in the `signatures` field of "
105                      "the reloaded SavedModel. Available endpoints for "
106                      "this SavedModel: "
107                      f"{list(self._reloaded_obj.signatures.keys())}"
108                  )
109  
110          # Add trainable and non-trainable weights from the call_endpoint_fn.
111          all_fns = [self.call_endpoint_fn]
112          if call_training_endpoint:
113              all_fns.append(self.call_training_endpoint_fn)
114          tvs, ntvs = _list_variables_used_by_fns(all_fns)
115          for v in tvs:
116              self._add_existing_weight(v)
117          for v in ntvs:
118              self._add_existing_weight(v)
119          self.built = True
120  
121      def _add_existing_weight(self, weight):
122          """Tracks an existing weight."""
123          self._track_variable(weight)
124  
125      def call(self, inputs, training=False, **kwargs):
126          if training:
127              if self.call_training_endpoint:
128                  return self.call_training_endpoint_fn(inputs, **kwargs)
129          return self.call_endpoint_fn(inputs, **kwargs)
130  
131      def get_config(self):
132          base_config = super().get_config()
133          config = {
134              # Note: this is not intended to be portable.
135              "filepath": self.filepath,
136              "call_endpoint": self.call_endpoint,
137              "call_training_endpoint": self.call_training_endpoint,
138          }
139          return {**base_config, **config}
