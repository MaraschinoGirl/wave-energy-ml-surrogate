44   def _random_hermitian_matrix(n, posdef=False, dtype=float):
45       "Generate random sym/hermitian array of the given size n"
46       if dtype in COMPLEX_DTYPES:
47           A = np.random.rand(n, n) + np.random.rand(n, n)*1.0j
48           A = (A + A.conj().T)/2
49       else:
50           A = np.random.rand(n, n)
51           A = (A + A.T)/2
52   
53       if posdef:
54           A += sqrt(2*n)*np.eye(n)
55   
56       return A.astype(dtype)

66   def symrand(dim_or_eigv, rng):
67       """Return a random symmetric (Hermitian) matrix. 
68    
69       If 'dim_or_eigv' is an integer N, return a NxN matrix, with eigenvalues 
70           uniformly distributed on (-1,1). 
71    
72       If 'dim_or_eigv' is  1-D real array 'a', return a matrix whose 
73                         eigenvalues are 'a'. 
74       """
75       if isinstance(dim_or_eigv, int):
76           dim = dim_or_eigv
77           d = rng.random(dim)*2 - 1
78       elif (isinstance(dim_or_eigv, ndarray) and
79             len(dim_or_eigv.shape) == 1):
80           dim = dim_or_eigv.shape[0]
81           d = dim_or_eigv
82       else:
83           raise TypeError("input type not supported.")
84   
85       v = ortho_group.rvs(dim)
86       h = v.T.conj() @ diag(d) @ v
87       # to avoid roundoff errors, symmetrize the matrix (again)
88       h = 0.5*(h.T+h)
89       return h

187      def _check_gen_eig(self, A, B, atol_homog=1e-13, rtol_homog=1e-13,
188                                     atol=1e-13, rtol=1e-13):
189          if B is not None:
190              A, B = asarray(A), asarray(B)
191              B0 = B
192          else:
193              A = asarray(A)
194              B0 = B
195              B = np.eye(*A.shape)
196          msg = f"\n{A!r}\n{B!r}"
197  
198          # Eigenvalues in homogeneous coordinates
199          w, vr = eig(A, B0, homogeneous_eigvals=True)
200          wt = eigvals(A, B0, homogeneous_eigvals=True)
201          val1 = A @ vr * w[1, :]
202          val2 = B @ vr * w[0, :]
203          for i in range(val1.shape[1]):
204              assert_allclose(val1[:, i], val2[:, i],
205                              rtol=rtol_homog, atol=atol_homog, err_msg=msg)
206  
207          if B0 is None:
208              assert_allclose(w[1, :], 1)
209              assert_allclose(wt[1, :], 1)
210  
211          perm = np.lexsort(w)
212          permt = np.lexsort(wt)
213          assert_allclose(w[:, perm], wt[:, permt], atol=1e-7, rtol=1e-7,
214                          err_msg=msg)
215  
216          length = np.empty(len(vr))
217  
218          for i in range(len(vr)):
219              length[i] = norm(vr[:, i])
220  
221          assert_allclose(length, np.ones(length.size), err_msg=msg,
222                          atol=1e-7, rtol=1e-7)
223  
224          # Convert homogeneous coordinates
225          beta_nonzero = (w[1, :] != 0)
226          wh = w[0, beta_nonzero] / w[1, beta_nonzero]
227  
228          # Eigenvalues in standard coordinates
229          w, vr = eig(A, B0)
230          wt = eigvals(A, B0)
231          val1 = A @ vr
232          val2 = B @ vr * w
233          res = val1 - val2
234          for i in range(res.shape[1]):
235              if np.all(isfinite(res[:, i])):
236                  assert_allclose(res[:, i], 0,
237                                  rtol=rtol, atol=atol, err_msg=msg)
238  
239          # try to consistently order eigenvalues, including complex conjugate pairs
240          w_fin = w[isfinite(w)]
241          wt_fin = wt[isfinite(wt)]
242  
243          # prune noise in the real parts
244          w_fin = -1j * np.real_if_close(1j*w_fin, tol=1e-10)
245          wt_fin = -1j * np.real_if_close(1j*wt_fin, tol=1e-10)
246  
247          perm = argsort(abs(w_fin) + w_fin.imag)
248          permt = argsort(abs(wt_fin) + wt_fin.imag)
249  
250          assert_allclose(w_fin[perm], wt_fin[permt],
251                          atol=1e-7, rtol=1e-7, err_msg=msg)
252  
253          length = np.empty(len(vr))
254          for i in range(len(vr)):
255              length[i] = norm(vr[:, i])
256          assert_allclose(length, np.ones(length.size), err_msg=msg)
257  
258          # Compare homogeneous and nonhomogeneous versions
259          assert_allclose(sort(wh), sort(w[np.isfinite(w)]))

413      def create_bandmat(self):
414          """Create the full matrix `self.fullmat` and 
415             the corresponding band matrix `self.bandmat`."""
416          N = 10
417          self.KL = 2   # number of subdiagonals (below the diagonal)
418          self.KU = 2   # number of superdiagonals (above the diagonal)
419  
420          # symmetric band matrix
421          self.sym_mat = (diag(full(N, 1.0))
422                          + diag(full(N-1, -1.0), -1) + diag(full(N-1, -1.0), 1)
423                          + diag(full(N-2, -2.0), -2) + diag(full(N-2, -2.0), 2))
424  
425          # hermitian band matrix
426          self.herm_mat = (diag(full(N, -1.0))
427                           + 1j*diag(full(N-1, 1.0), -1)
428                           - 1j*diag(full(N-1, 1.0), 1)
429                           + diag(full(N-2, -2.0), -2)
430                           + diag(full(N-2, -2.0), 2))
431  
432          # general real band matrix
433          self.real_mat = (diag(full(N, 1.0))
434                           + diag(full(N-1, -1.0), -1) + diag(full(N-1, -3.0), 1)
435                           + diag(full(N-2, 2.0), -2) + diag(full(N-2, -2.0), 2))
436  
437          # general complex band matrix
438          self.comp_mat = (1j*diag(full(N, 1.0))
439                           + diag(full(N-1, -1.0), -1)
440                           + 1j*diag(full(N-1, -3.0), 1)
441                           + diag(full(N-2, 2.0), -2)
442                           + diag(full(N-2, -2.0), 2))
443  
444          # Eigenvalues and -vectors from linalg.eig
445          ew, ev = linalg.eig(self.sym_mat)
446          ew = ew.real
447          args = argsort(ew)
448          self.w_sym_lin = ew[args]
449          self.evec_sym_lin = ev[:, args]
450  
451          ew, ev = linalg.eig(self.herm_mat)
452          ew = ew.real
453          args = argsort(ew)
454          self.w_herm_lin = ew[args]
455          self.evec_herm_lin = ev[:, args]
456  
457          # Extract upper bands from symmetric and hermitian band matrices
458          # (for use in dsbevd, dsbevx, zhbevd, zhbevx
459          #  and their single precision versions)
460          LDAB = self.KU + 1
461          self.bandmat_sym = zeros((LDAB, N), dtype=float)
462          self.bandmat_herm = zeros((LDAB, N), dtype=complex)
463          for i in range(LDAB):
464              self.bandmat_sym[LDAB-i-1, i:N] = diag(self.sym_mat, i)
465              self.bandmat_herm[LDAB-i-1, i:N] = diag(self.herm_mat, i)
466  
467          # Extract bands from general real and complex band matrix
468          # (for use in dgbtrf, dgbtrs and their single precision versions)
469          LDAB = 2*self.KL + self.KU + 1
470          self.bandmat_real = zeros((LDAB, N), dtype=float)
471          self.bandmat_real[2*self.KL, :] = diag(self.real_mat)  # diagonal
472          for i in range(self.KL):
473              # superdiagonals
474              self.bandmat_real[2*self.KL-1-i, i+1:N] = diag(self.real_mat, i+1)
475              # subdiagonals
476              self.bandmat_real[2*self.KL+1+i, 0:N-1-i] = diag(self.real_mat,
477                                                               -i-1)
478  
479          self.bandmat_comp = zeros((LDAB, N), dtype=complex)
480          self.bandmat_comp[2*self.KL, :] = diag(self.comp_mat)  # diagonal
481          for i in range(self.KL):
482              # superdiagonals
483              self.bandmat_comp[2*self.KL-1-i, i+1:N] = diag(self.comp_mat, i+1)
484              # subdiagonals
485              self.bandmat_comp[2*self.KL+1+i, 0:N-1-i] = diag(self.comp_mat,
486                                                               -i-1)
487  
488          # absolute value for linear equation system A*x = b
489          self.b = 1.0*arange(N)
490          self.bc = self.b * (1 + 1j)

2525     def check(self, A, B, sort, AA, BB, alpha, beta, Q, Z):
2526         Id = np.eye(*A.shape)
2527         # make sure Q and Z are orthogonal
2528         assert_array_almost_equal(Q @ Q.T.conj(), Id)
2529         assert_array_almost_equal(Z @ Z.T.conj(), Id)
2530         # check factorization
2531         assert_array_almost_equal(Q @ AA, A @ Z)
2532         assert_array_almost_equal(Q @ BB, B @ Z)
2533         # check shape of AA and BB
2534         assert_array_equal(np.tril(AA, -2), np.zeros(AA.shape))
2535         assert_array_equal(np.tril(BB, -1), np.zeros(BB.shape))
2536         # check eigenvalues
2537         for i in range(A.shape[0]):
2538             # does the current diagonal element belong to a 2-by-2 block
2539             # that was already checked?
2540             if i > 0 and A[i, i - 1] != 0:
2541                 continue
2542             # take care of 2-by-2 blocks
2543             if i < AA.shape[0] - 1 and AA[i + 1, i] != 0:
2544                 evals, _ = eig(AA[i:i + 2, i:i + 2], BB[i:i + 2, i:i + 2])
2545                 # make sure the pair of complex conjugate eigenvalues
2546                 # is ordered consistently (positive imaginary part first)
2547                 if evals[0].imag < 0:
2548                     evals = evals[[1, 0]]
2549                 tmp = alpha[i:i + 2]/beta[i:i + 2]
2550                 if tmp[0].imag < 0:
2551                     tmp = tmp[[1, 0]]
2552                 assert_array_almost_equal(evals, tmp)
2553             else:
2554                 if alpha[i] == 0 and beta[i] == 0:
2555                     assert_equal(AA[i, i], 0)
2556                     assert_equal(BB[i, i], 0)
2557                 elif beta[i] == 0:
2558                     assert_equal(BB[i, i], 0)
2559                 else:
2560                     assert_almost_equal(AA[i, i]/BB[i, i], alpha[i]/beta[i])
2561         sortfun = _select_function(sort)
2562         lastsort = True
2563         for i in range(A.shape[0]):
2564             cursort = sortfun(np.array([alpha[i]]), np.array([beta[i]]))
2565             # once the sorting criterion was not matched all subsequent
2566             # eigenvalues also shouldn't match
2567             if not lastsort:
