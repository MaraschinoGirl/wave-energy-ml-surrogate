8    def test_csc_getrow():
9        N = 10
10       np.random.seed(0)
11       X = np.random.random((N, N))
12       X[X > 0.7] = 0
13       Xcsc = csc_matrix(X)
14   
15       for i in range(N):
16           arr_row = X[i:i + 1, :]
17           csc_row = Xcsc.getrow(i)
18   
19           assert_array_almost_equal(arr_row, csc_row.toarray())
20           assert_(type(csc_row) is csr_matrix)

54   def test_csc_empty_slices(matrix_input, axis, expected_shape):
55       # see gh-11127 for related discussion
56       slice_1 = matrix_input.toarray().shape[0] - 1
57       slice_2 = slice_1
58       slice_3 = slice_2 - 1
59   
60       if axis == 0:
61           actual_shape_1 = matrix_input[slice_1:slice_2, :].toarray().shape
62           actual_shape_2 = matrix_input[slice_1:slice_3, :].toarray().shape
63       elif axis == 1:
64           actual_shape_1 = matrix_input[:, slice_1:slice_2].toarray().shape
65           actual_shape_2 = matrix_input[:, slice_1:slice_3].toarray().shape
66       elif axis == 'both':
67           actual_shape_1 = matrix_input[slice_1:slice_2, slice_1:slice_2].toarray().shape
68           actual_shape_2 = matrix_input[slice_1:slice_3, slice_1:slice_3].toarray().shape
69   
70       assert actual_shape_1 == expected_shape
71       assert actual_shape_1 == actual_shape_2

74   @pytest.mark.parametrize('ax', (-2, -1, 0, 1, None))
75   def test_argmax_overflow(ax):
76       # See gh-13646: Windows integer overflow for large sparse matrices.
77       dim = (100000, 100000)
78       A = lil_matrix(dim)
79       A[-2, -2] = 42
80       A[-3, -3] = 0.1234
81       A = csc_matrix(A)
82       idx = A.argmax(axis=ax)
83   
84       if ax is None:
85           # idx is a single flattened index
86           # that we need to convert to a 2d index pair;
87           # can't do this with np.unravel_index because
88           # the dimensions are too large
89           ii = idx % dim[0]
90           jj = idx // dim[0]
91       else:
92           # idx is an array of size of A.shape[ax];
93           # check the max index to make sure no overflows
94           # we encountered
95           assert np.count_nonzero(idx) == A.nnz
96           ii, jj = np.max(idx), np.argmax(idx)
97   
98       assert A[ii, jj] == A[-2, -2]
