16   class _IDCounter:
17       """Generate sequential ids with a prefix."""
18   
19       def __init__(self, prefix):
20           self.prefix = prefix
21           self.count = 0
22   
23       def get_id(self):
24           self.count += 1
25           return f"{self.prefix}-{self.count}"

37   class _VisualBlock:
38       """HTML Representation of Estimator 
39    
40       Parameters 
41       ---------- 
42       kind : {'serial', 'parallel', 'single'} 
43           kind of HTML block 
44    
45       estimators : list of estimators or `_VisualBlock`s or a single estimator 
46           If kind != 'single', then `estimators` is a list of 
47           estimators. 
48           If kind == 'single', then `estimators` is a single estimator. 
49    
50       names : list of str, default=None 
51           If kind != 'single', then `names` corresponds to estimators. 
52           If kind == 'single', then `names` is a single string corresponding to 
53           the single estimator. 
54    
55       name_details : list of str, str, or None, default=None 
56           If kind != 'single', then `name_details` corresponds to `names`. 
57           If kind == 'single', then `name_details` is a single string 
58           corresponding to the single estimator. 
59    
60       name_caption : str, default=None 
61           The caption below the name. `None` stands for no caption. 
62           Only active when kind == 'single'. 
63    
64       doc_link_label : str, default=None 
65           The label for the documentation link. If provided, the label would be 
66           "Documentation for {doc_link_label}". Otherwise it will look for `names`. 
67           Only active when kind == 'single'. 
68    
69       dash_wrapped : bool, default=True 
70           If true, wrapped HTML element will be wrapped with a dashed border. 
71           Only active when kind != 'single'. 
72       """
73   
74       def __init__(
75           self,
76           kind,
77           estimators,
78           *,
79           names=None,
80           name_details=None,
81           name_caption=None,
82           doc_link_label=None,
83           dash_wrapped=True,
84       ):
85           self.kind = kind
86           self.estimators = estimators
87           self.dash_wrapped = dash_wrapped
88           self.name_caption = name_caption
89           self.doc_link_label = doc_link_label
90   
91           if self.kind in ("parallel", "serial"):
92               if names is None:
93                   names = (None,) * len(estimators)
94               if name_details is None:
95                   name_details = (None,) * len(estimators)
96   
97           self.names = names
98           self.name_details = name_details
99   
100      def _sk_visual_block_(self):
101          return self

104  def _write_label_html(
105      out,
106      name,
107      name_details,
108      name_caption=None,
109      doc_link_label=None,
110      outer_class="sk-label-container",
111      inner_class="sk-label",
112      checked=False,
113      doc_link="",
114      is_fitted_css_class="",
115      is_fitted_icon="",
116  ):
117      """Write labeled html with or without a dropdown with named details. 
118   
119      Parameters 
120      ---------- 
121      out : file-like object 
122          The file to write the HTML representation to. 
123      name : str 
124          The label for the estimator. It corresponds either to the estimator class name 
125          for a simple estimator or in the case of a `Pipeline` and `ColumnTransformer`, 
126          it corresponds to the name of the step. 
127      name_details : str 
128          The details to show as content in the dropdown part of the toggleable label. It 
129          can contain information such as non-default parameters or column information for 
130          `ColumnTransformer`. 
131      name_caption : str, default=None 
132          The caption below the name. If `None`, no caption will be created. 
133      doc_link_label : str, default=None 
134          The label for the documentation link. If provided, the label would be 
135          "Documentation for {doc_link_label}". Otherwise it will look for `name`. 
136      outer_class : {"sk-label-container", "sk-item"}, default="sk-label-container" 
137          The CSS class for the outer container. 
138      inner_class : {"sk-label", "sk-estimator"}, default="sk-label" 
139          The CSS class for the inner container. 
140      checked : bool, default=False 
141          Whether the dropdown is folded or not. With a single estimator, we intend to 
142          unfold the content. 
143      doc_link : str, default="" 
144          The link to the documentation for the estimator. If an empty string, no link is 
145          added to the diagram. This can be generated for an estimator if it uses the 
146          `_HTMLDocumentationLinkMixin`. 
147      is_fitted_css_class : {"", "fitted"} 
148          The CSS class to indicate whether or not the estimator is fitted. The 
149          empty string means that the estimator is not fitted and "fitted" means that the 
150          estimator is fitted. 
151      is_fitted_icon : str, default="" 
152          The HTML representation to show the fitted information in the diagram. An empty 
153          string means that no information is shown. 
154      """
155      out.write(
156          f'<div class="{outer_class}"><div'
157          f' class="{inner_class} {is_fitted_css_class} sk-toggleable">'
158      )
159      name = html.escape(name)
160  
161      if name_details is not None:
162          name_details = html.escape(str(name_details))
163          checked_str = "checked" if checked else ""
164          est_id = _ESTIMATOR_ID_COUNTER.get_id()
165  
166          if doc_link:
167              doc_label = "<span>Online documentation</span>"
168              if doc_link_label is not None:
169                  doc_label = f"<span>Documentation for {doc_link_label}</span>"
170              elif name is not None:
171                  doc_label = f"<span>Documentation for {name}</span>"
172              doc_link = (
173                  f'<a class="sk-estimator-doc-link {is_fitted_css_class}"'
174                  f' rel="noreferrer" target="_blank" href="{doc_link}">?{doc_label}</a>'
175              )
176  
177          name_caption_div = (
178              ""
179              if name_caption is None
180              else f'<div class="caption">{html.escape(name_caption)}</div>'
181          )
182          name_caption_div = f"<div><div>{name}</div>{name_caption_div}</div>"
183          links_div = (
184              f"<div>{doc_link}{is_fitted_icon}</div>"
185              if doc_link or is_fitted_icon
186              else ""
187          )
188  
189          label_html = (
190              f'<label for="{est_id}" class="sk-toggleable__label {is_fitted_css_class} '
191              f'sk-toggleable__label-arrow">{name_caption_div}{links_div}</label>'
192          )
193  
194          fmt_str = (
195              f'<input class="sk-toggleable__control sk-hidden--visually" id="{est_id}" '
196              f'type="checkbox" {checked_str}>{label_html}<div '
197              f'class="sk-toggleable__content {is_fitted_css_class}"><pre>{name_details}'
198              "</pre></div> "
199          )
200          out.write(fmt_str)
201      else:
202          out.write(f"<label>{name}</label>")
203      out.write("</div></div>")  # outer_class inner_class

206  def _get_visual_block(estimator):
207      """Generate information about how to display an estimator."""
208      if hasattr(estimator, "_sk_visual_block_"):
209          try:
210              return estimator._sk_visual_block_()
211          except Exception:
212              return _VisualBlock(
213                  "single",
214                  estimator,
215                  names=estimator.__class__.__name__,
216                  name_details=str(estimator),
217              )
218  
219      if isinstance(estimator, str):
220          return _VisualBlock(
221              "single", estimator, names=estimator, name_details=estimator
222          )
223      elif estimator is None:
224          return _VisualBlock("single", estimator, names="None", name_details="None")
225  
226      # check if estimator looks like a meta estimator (wraps estimators)
227      if hasattr(estimator, "get_params") and not isclass(estimator):
228          estimators = [
229              (key, est)
230              for key, est in estimator.get_params(deep=False).items()
231              if hasattr(est, "get_params") and hasattr(est, "fit") and not isclass(est)
232          ]
233          if estimators:
234              return _VisualBlock(
235                  "parallel",
236                  [est for _, est in estimators],
237                  names=[f"{key}: {est.__class__.__name__}" for key, est in estimators],
238                  name_details=[str(est) for _, est in estimators],
239              )
240  
241      return _VisualBlock(
242          "single",
243          estimator,
244          names=estimator.__class__.__name__,
245          name_details=str(estimator),
246      )

249  def _write_estimator_html(
250      out,
251      estimator,
252      estimator_label,
253      estimator_label_details,
254      is_fitted_css_class,
255      is_fitted_icon="",
256      first_call=False,
257  ):
258      """Write estimator to html in serial, parallel, or by itself (single). 
259   
260      For multiple estimators, this function is called recursively. 
261   
262      Parameters 
263      ---------- 
264      out : file-like object 
265          The file to write the HTML representation to. 
266      estimator : estimator object 
267          The estimator to visualize. 
268      estimator_label : str 
269          The label for the estimator. It corresponds either to the estimator class name 
270          for simple estimator or in the case of `Pipeline` and `ColumnTransformer`, it 
271          corresponds to the name of the step. 
272      estimator_label_details : str 
273          The details to show as content in the dropdown part of the toggleable label. 
274          It can contain information as non-default parameters or column information for 
275          `ColumnTransformer`. 
276      is_fitted_css_class : {"", "fitted"} 
277          The CSS class to indicate whether or not the estimator is fitted or not. The 
278          empty string means that the estimator is not fitted and "fitted" means that the 
279          estimator is fitted. 
280      is_fitted_icon : str, default="" 
281          The HTML representation to show the fitted information in the diagram. An empty 
282          string means that no information is shown. If the estimator to be shown is not 
283          the first estimator (i.e. `first_call=False`), `is_fitted_icon` is always an 
284          empty string. 
285      first_call : bool, default=False 
286          Whether this is the first time this function is called. 
287      """
288      if first_call:
289          est_block = _get_visual_block(estimator)
290      else:
291          is_fitted_icon = ""
292          with config_context(print_changed_only=True):
293              est_block = _get_visual_block(estimator)
294      # `estimator` can also be an instance of `_VisualBlock`
295      if hasattr(estimator, "_get_doc_link"):
296          doc_link = estimator._get_doc_link()
297      else:
298          doc_link = ""
299      if est_block.kind in ("serial", "parallel"):
300          dashed_wrapped = first_call or est_block.dash_wrapped
301          dash_cls = " sk-dashed-wrapped" if dashed_wrapped else ""
302          out.write(f'<div class="sk-item{dash_cls}">')
303  
304          if estimator_label:
305              _write_label_html(
306                  out,
307                  estimator_label,
308                  estimator_label_details,
309                  doc_link=doc_link,
310                  is_fitted_css_class=is_fitted_css_class,
311                  is_fitted_icon=is_fitted_icon,
312              )
313  
314          kind = est_block.kind
315          out.write(f'<div class="sk-{kind}">')
316          est_infos = zip(est_block.estimators, est_block.names, est_block.name_details)
317  
318          for est, name, name_details in est_infos:
319              if kind == "serial":
320                  _write_estimator_html(
321                      out,
322                      est,
323                      name,
324                      name_details,
325                      is_fitted_css_class=is_fitted_css_class,
326                  )
327              else:  # parallel
328                  out.write('<div class="sk-parallel-item">')
329                  # wrap element in a serial visualblock
330                  serial_block = _VisualBlock("serial", [est], dash_wrapped=False)
331                  _write_estimator_html(
332                      out,
333                      serial_block,
334                      name,
335                      name_details,
336                      is_fitted_css_class=is_fitted_css_class,
337                  )
338                  out.write("</div>")  # sk-parallel-item
339  
340          out.write("</div></div>")
341      elif est_block.kind == "single":
342          _write_label_html(
343              out,
344              est_block.names,
345              est_block.name_details,
346              est_block.name_caption,
347              est_block.doc_link_label,
348              outer_class="sk-item",
349              inner_class="sk-estimator",
350              checked=first_call,
351              doc_link=doc_link,
352              is_fitted_css_class=is_fitted_css_class,
353              is_fitted_icon=is_fitted_icon,
354          )

357  def estimator_html_repr(estimator):
358      """Build a HTML representation of an estimator. 
359   
360      Read more in the :ref:`User Guide <visualizing_composite_estimators>`. 
361   
362      Parameters 
363      ---------- 
364      estimator : estimator object 
365          The estimator to visualize. 
366   
367      Returns 
368      ------- 
369      html: str 
370          HTML representation of estimator. 
371   
372      Examples 
373      -------- 
374      >>> from sklearn.utils._estimator_html_repr import estimator_html_repr 
375      >>> from sklearn.linear_model import LogisticRegression 
376      >>> estimator_html_repr(LogisticRegression()) 
377      '<style>...</div>' 
378      """
379      from sklearn.exceptions import NotFittedError
380      from sklearn.utils.validation import check_is_fitted
381  
382      if not hasattr(estimator, "fit"):
383          status_label = "<span>Not fitted</span>"
384          is_fitted_css_class = ""
385      else:
386          try:
387              check_is_fitted(estimator)
388              status_label = "<span>Fitted</span>"
389              is_fitted_css_class = "fitted"
390          except NotFittedError:
391              status_label = "<span>Not fitted</span>"
392              is_fitted_css_class = ""
393  
394      is_fitted_icon = (
395          f'<span class="sk-estimator-doc-link {is_fitted_css_class}">'
396          f"i{status_label}</span>"
397      )
398      with closing(StringIO()) as out:
399          container_id = _CONTAINER_ID_COUNTER.get_id()
400          style_template = Template(_CSS_STYLE)
401          style_with_id = style_template.substitute(id=container_id)
402          estimator_str = str(estimator)
403  
404          # The fallback message is shown by default and loading the CSS sets
405          # div.sk-text-repr-fallback to display: none to hide the fallback message.
406          #
407          # If the notebook is trusted, the CSS is loaded which hides the fallback
408          # message. If the notebook is not trusted, then the CSS is not loaded and the
409          # fallback message is shown by default.
410          #
411          # The reverse logic applies to HTML repr div.sk-container.
412          # div.sk-container is hidden by default and the loading the CSS displays it.
413          fallback_msg = (
414              "In a Jupyter environment, please rerun this cell to show the HTML"
415              " representation or trust the notebook. <br />On GitHub, the"
416              " HTML representation is unable to render, please try loading this page"
417              " with nbviewer.org."
418          )
419          html_template = (
420              f"<style>{style_with_id}</style>"
421              f'<div id="{container_id}" class="sk-top-container">'
422              '<div class="sk-text-repr-fallback">'
423              f"<pre>{html.escape(estimator_str)}</pre><b>{fallback_msg}</b>"
424              "</div>"
425              '<div class="sk-container" hidden>'
426          )
427  
428          out.write(html_template)
429  
430          _write_estimator_html(
431              out,
432              estimator,
433              estimator.__class__.__name__,
434              estimator_str,
435              first_call=True,
436              is_fitted_css_class=is_fitted_css_class,
437              is_fitted_icon=is_fitted_icon,
438          )
439          out.write("</div></div>")
440  
441          html_output = out.getvalue()
442          return html_output

445  class _HTMLDocumentationLinkMixin:
446      """Mixin class allowing to generate a link to the API documentation. 
447   
448      This mixin relies on three attributes: 
449      - `_doc_link_module`: it corresponds to the root module (e.g. `sklearn`). Using this 
450        mixin, the default value is `sklearn`. 
451      - `_doc_link_template`: it corresponds to the template used to generate the 
452        link to the API documentation. Using this mixin, the default value is 
453        `"https://scikit-learn.org/{version_url}/modules/generated/ 
454        {estimator_module}.{estimator_name}.html"`. 
455      - `_doc_link_url_param_generator`: it corresponds to a function that generates the 
456        parameters to be used in the template when the estimator module and name are not 
457        sufficient. 
458   
459      The method :meth:`_get_doc_link` generates the link to the API documentation for a 
460      given estimator. 
461   
462      This useful provides all the necessary states for 
463      :func:`sklearn.utils.estimator_html_repr` to generate a link to the API 
464      documentation for the estimator HTML diagram. 
465   
466      Examples 
467      -------- 
468      If the default values for `_doc_link_module`, `_doc_link_template` are not suitable, 
469      then you can override them and provide a method to generate the URL parameters: 
470      >>> from sklearn.base import BaseEstimator 
471      >>> doc_link_template = "https://address.local/{single_param}.html" 
472      >>> def url_param_generator(estimator): 
473      ...     return {"single_param": estimator.__class__.__name__} 
474      >>> class MyEstimator(BaseEstimator): 
475      ...     # use "builtins" since it is the associated module when declaring 
476      ...     # the class in a docstring 
477      ...     _doc_link_module = "builtins" 
478      ...     _doc_link_template = doc_link_template 
479      ...     _doc_link_url_param_generator = url_param_generator 
480      >>> estimator = MyEstimator() 
481      >>> estimator._get_doc_link() 
482      'https://address.local/MyEstimator.html' 
483   
484      If instead of overriding the attributes inside the class definition, you want to 
485      override a class instance, you can use `types.MethodType` to bind the method to the 
486      instance: 
487      >>> import types 
488      >>> estimator = BaseEstimator() 
489      >>> estimator._doc_link_template = doc_link_template 
490      >>> estimator._doc_link_url_param_generator = types.MethodType( 
491      ...     url_param_generator, estimator) 
492      >>> estimator._get_doc_link() 
493      'https://address.local/BaseEstimator.html' 
494      """
495  
496      _doc_link_module = "sklearn"
497      _doc_link_url_param_generator = None
498  
499      @property
500      def _doc_link_template(self):
501          sklearn_version = parse_version(__version__)
502          if sklearn_version.dev is None:
503              version_url = f"{sklearn_version.major}.{sklearn_version.minor}"
504          else:
505              version_url = "dev"
506          return getattr(
507              self,
508              "__doc_link_template",
509              (
510                  f"https://scikit-learn.org/{version_url}/modules/generated/"
511                  "{estimator_module}.{estimator_name}.html"
512              ),
513          )
514  
515      @_doc_link_template.setter
516      def _doc_link_template(self, value):
517          setattr(self, "__doc_link_template", value)
518  
519      def _get_doc_link(self):
520          """Generates a link to the API documentation for a given estimator. 
521   
522          This method generates the link to the estimator's documentation page 
523          by using the template defined by the attribute `_doc_link_template`. 
524   
525          Returns 
526          ------- 
527          url : str 
528              The URL to the API documentation for this estimator. If the estimator does 
529              not belong to module `_doc_link_module`, the empty string (i.e. `""`) is 
530              returned. 
531          """
532          if self.__class__.__module__.split(".")[0] != self._doc_link_module:
533              return ""
534  
535          if self._doc_link_url_param_generator is None:
536              estimator_name = self.__class__.__name__
537              # Construct the estimator's module name, up to the first private submodule.
538              # This works because in scikit-learn all public estimators are exposed at
539              # that level, even if they actually live in a private sub-module.
540              estimator_module = ".".join(
541                  itertools.takewhile(
542                      lambda part: not part.startswith("_"),
543                      self.__class__.__module__.split("."),
544                  )
545              )
546              return self._doc_link_template.format(
547                  estimator_module=estimator_module, estimator_name=estimator_name
548              )
549          return self._doc_link_template.format(**self._doc_link_url_param_generator())
