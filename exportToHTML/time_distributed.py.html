1    """Wrapper layer to apply every temporal slice of an input."""
2    
3    from keras.src import backend
4    from keras.src import ops
5    from keras.src.api_export import keras_export
6    from keras.src.layers.core.wrapper import Wrapper
7    from keras.src.layers.layer import Layer
8    
9    
10   @keras_export("keras.layers.TimeDistributed")
11   class TimeDistributed(Wrapper):
12       """This wrapper allows to apply a layer to every temporal slice of an input. 
13    
14       Every input should be at least 3D, and the dimension of index one of the 
15       first input will be considered to be the temporal dimension. 
16    
17       Consider a batch of 32 video samples, where each sample is a 128x128 RGB 
18       image with `channels_last` data format, across 10 timesteps. 
19       The batch input shape is `(32, 10, 128, 128, 3)`. 
20    
21       You can then use `TimeDistributed` to apply the same `Conv2D` layer to each 
22       of the 10 timesteps, independently: 
23    
24       >>> inputs = layers.Input(shape=(10, 128, 128, 3), batch_size=32) 
25       >>> conv_2d_layer = layers.Conv2D(64, (3, 3)) 
26       >>> outputs = layers.TimeDistributed(conv_2d_layer)(inputs) 
27       >>> outputs.shape 
28       (32, 10, 126, 126, 64) 
29    
30       Because `TimeDistributed` applies the same instance of `Conv2D` to each of 
31       the timestamps, the same set of weights are used at each timestamp. 
32    
33       Args: 
34           layer: a `keras.layers.Layer` instance. 
35    
36       Call arguments: 
37           inputs: Input tensor of shape (batch, time, ...) or nested tensors, 
38               and each of which has shape (batch, time, ...). 
39           training: Python boolean indicating whether the layer should behave in 
40               training mode or in inference mode. This argument is passed to the 
41               wrapped layer (only if the layer supports this argument). 
42           mask: Binary tensor of shape `(samples, timesteps)` indicating whether 
43               a given timestep should be masked. This argument is passed to the 
44               wrapped layer (only if the layer supports this argument). 
45       """
46   
47       def __init__(self, layer, **kwargs):
48           if not isinstance(layer, Layer):
49               raise ValueError(
50                   "Please initialize `TimeDistributed` layer with a "
51                   f"`keras.layers.Layer` instance. Received: {layer}"
52               )
53           super().__init__(layer, **kwargs)
54           self.supports_masking = True
55   
56       def _get_child_input_shape(self, input_shape):
57           if not isinstance(input_shape, (tuple, list)) or len(input_shape) < 3:
58               raise ValueError(
59                   "`TimeDistributed` Layer should be passed an `input_shape` "
60                   f"with at least 3 dimensions, received: {input_shape}"
61               )
62           return (input_shape[0], *input_shape[2:])
63   
64       def compute_output_shape(self, input_shape):
65           child_input_shape = self._get_child_input_shape(input_shape)
66           child_output_shape = self.layer.compute_output_shape(child_input_shape)
67           return (child_output_shape[0], input_shape[1], *child_output_shape[1:])
68   
69       def build(self, input_shape):
70           child_input_shape = self._get_child_input_shape(input_shape)
71           super().build(child_input_shape)
72           self.built = True
73   
74       def call(self, inputs, training=None, mask=None):
75           input_shape = ops.shape(inputs)
76           mask_shape = None if mask is None else ops.shape(mask)
77           batch_size = input_shape[0]
78           timesteps = input_shape[1]
79   
80           # For TF backend with graph mode and `partial_batch_size`, skip
81           # evaluation of `batch_size` as it can be a `strided_slice` and
82           # not a constant.
83           if backend.backend() == "tensorflow":
84               from keras.src.utils.module_utils import tensorflow as tf
85   
86               if (
87                   not tf.executing_eagerly
88                   and mask_shape is not None
89                   and mask_shape[1:2] != (timesteps,)
90               ):
91                   raise ValueError(
92                       "`TimeDistributed` Layer should be passed a `mask` of "
93                       f"shape ({batch_size}, {timesteps}, ...), "
94                       f"received: mask.shape={mask_shape}"
95                   )
96           elif mask_shape is not None and mask_shape[:2] != (
97               batch_size,
98               timesteps,
99           ):
100              raise ValueError(
101                  "`TimeDistributed` Layer should be passed a `mask` of "
102                  f"shape ({batch_size}, {timesteps}, ...), "
103                  f"received: mask.shape={mask_shape}"
104              )
105  
106          def time_distributed_transpose(data):
107              """Swaps the timestep and batch dimensions of a tensor."""
108              axes = [1, 0, *range(2, len(data.shape))]
109              return ops.transpose(data, axes=axes)
110  
111          inputs = time_distributed_transpose(inputs)
112          if mask is not None:
113              mask = time_distributed_transpose(mask)
114  
115          def step_function(i):
116              kwargs = {}
117              if self.layer._call_has_mask_arg and mask is not None:
118                  kwargs["mask"] = mask[i]
119              if self.layer._call_has_training_arg:
120                  kwargs["training"] = training
121              return self.layer.call(inputs[i], **kwargs)
122  
123          # Implementation #1: is the time axis is static, use a Python for loop.
124  
125          if inputs.shape[0] is not None:
126              outputs = ops.stack(
127                  [step_function(i) for i in range(inputs.shape[0])]
128              )
129              return time_distributed_transpose(outputs)
130  
131          # Implementation #2: use backend.vectorized_map.
132  
133          outputs = backend.vectorized_map(step_function, ops.arange(timesteps))
134          return time_distributed_transpose(outputs)
