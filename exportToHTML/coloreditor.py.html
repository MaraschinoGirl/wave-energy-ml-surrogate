182      def DoConfigChange(self):
183          SyntEditViewParent.DoConfigChange(self)
184          tabSize = GetEditorOption("Tab Size", 4, 2)
185          indentSize = GetEditorOption("Indent Size", 4, 2)
186          bUseTabs = GetEditorOption("Use Tabs", 0)
187          bSmartTabs = GetEditorOption("Smart Tabs", 1)
188          ext = self.idle.IDLEExtension("AutoIndent")  # Required extension.
189  
190          self.SCISetViewWS(GetEditorOption("View Whitespace", 0))
191          self.SCISetViewEOL(GetEditorOption("View EOL", 0))
192          self.SCISetIndentationGuides(GetEditorOption("View Indentation Guides", 0))
193  
194          if GetEditorOption("Right Edge Enabled", 0):
195              mode = scintillacon.EDGE_BACKGROUND
196          else:
197              mode = scintillacon.EDGE_NONE
198          self.SCISetEdgeMode(mode)
199          self.SCISetEdgeColumn(GetEditorOption("Right Edge Column", 75))
200          self.SCISetEdgeColor(
201              GetEditorOption("Right Edge Color", win32api.RGB(0xEF, 0xEF, 0xEF))
202          )
203  
204          width = GetEditorOption("Marker Margin Width", 16)
205          self.SCISetMarginWidthN(1, width)
206          width = GetEditorOption("Fold Margin Width", 12)
207          self.SCISetMarginWidthN(2, width)
208          width = GetEditorOption("Line Number Margin Width", 0)
209          self.SCISetMarginWidthN(0, width)
210          self.bFolding = GetEditorOption("Enable Folding", 1)
211          fold_flags = 0
212          self.SendScintilla(
213              scintillacon.SCI_SETMODEVENTMASK, scintillacon.SC_MOD_CHANGEFOLD
214          )
215          if self.bFolding:
216              if GetEditorOption("Fold Lines", 1):
217                  fold_flags = 16
218  
219          self.SCISetProperty("fold", self.bFolding)
220          self.SCISetFoldFlags(fold_flags)
221  
222          tt_color = GetEditorOption("Tab Timmy Color", win32api.RGB(0xFF, 0, 0))
223          self.SendScintilla(scintillacon.SCI_INDICSETFORE, 1, tt_color)
224  
225          tt_use = GetEditorOption("Use Tab Timmy", 1)
226          if tt_use:
227              self.SCISetProperty("tab.timmy.whinge.level", "1")
228  
229          # Auto-indent has very complicated behaviour.  In a nutshell, the only
230          # way to get sensible behaviour from it is to ensure tabwidth != indentsize.
231          # Further, usetabs will only ever go from 1->0, never 0->1.
232          # This is _not_ the behaviour Pythonwin wants:
233          # * Tab width is arbitary, so should have no impact on smarts.
234          # * bUseTabs setting should reflect how new files are created, and
235          #   if Smart Tabs disabled, existing files are edited
236          # * If "Smart Tabs" is enabled, bUseTabs should have no bearing
237          #   for existing files (unless of course no context can be determined)
238          #
239          # So for smart tabs we configure the widget with completely dummy
240          # values (ensuring tabwidth != indentwidth), ask it to guess, then
241          # look at the values it has guessed, and re-configure
242          if bSmartTabs:
243              ext.config(usetabs=1, tabwidth=5, indentwidth=4)
244              ext.set_indentation_params(1)
245              if ext.indentwidth == 5:
246                  # Either 5 literal spaces, or a single tab character. Assume a tab
247                  usetabs = 1
248                  indentwidth = tabSize
249              else:
250                  # Either Indented with spaces, and indent size has been guessed or
251                  # an empty file (or no context found - tough!)
252                  if self.GetTextLength() == 0:  # emtpy
253                      usetabs = bUseTabs
254                      indentwidth = indentSize
255                  else:  # guessed.
256                      indentwidth = ext.indentwidth
257                      usetabs = 0
258              # Tab size can never be guessed - set at user preference.
259              ext.config(usetabs=usetabs, indentwidth=indentwidth, tabwidth=tabSize)
260          else:
261              # Don't want smart-tabs - just set the options!
262              ext.config(usetabs=bUseTabs, tabwidth=tabSize, indentwidth=indentSize)
263          self.SCISetIndent(indentSize)
264          self.SCISetTabWidth(tabSize)

286      def _EndUserStateChange(self, info):
287          scrollOff = info[1] - self.GetFirstVisibleLine()
288          if scrollOff:
289              self.LineScroll(scrollOff)
290          # Make sure we don't reset the cursor beyond the buffer.
291          max = self.GetTextLength()
292          newPos = min(info[0][0], max), min(info[0][1], max)
293          self.SetSel(newPos)

355      def OnCmdViewFold(self, cid, code):  # Handle the menu command
356          if cid == win32ui.ID_VIEW_FOLD_EXPAND_ALL:
357              self.FoldExpandAllEvent(None)
358          elif cid == win32ui.ID_VIEW_FOLD_EXPAND:
359              self.FoldExpandEvent(None)
360          elif cid == win32ui.ID_VIEW_FOLD_COLLAPSE_ALL:
361              self.FoldCollapseAllEvent(None)
362          elif cid == win32ui.ID_VIEW_FOLD_COLLAPSE:
363              self.FoldCollapseEvent(None)
364          else:
365              print("Unknown collapse/expand ID")

367      def OnUpdateViewFold(self, cmdui):  # Update the tick on the UI.
368          if not self.bFolding:
369              cmdui.Enable(0)
370              return
371          id = cmdui.m_nID
372          if id in (win32ui.ID_VIEW_FOLD_EXPAND_ALL, win32ui.ID_VIEW_FOLD_COLLAPSE_ALL):
373              cmdui.Enable()
374          else:
375              enable = 0
376              lineno = self.LineFromChar(self.GetSel()[0])
377              foldable = (
378                  self.SCIGetFoldLevel(lineno) & scintillacon.SC_FOLDLEVELHEADERFLAG
379              )
380              is_expanded = self.SCIGetFoldExpanded(lineno)
381              if id == win32ui.ID_VIEW_FOLD_EXPAND:
382                  if foldable and not is_expanded:
383                      enable = 1
384              elif id == win32ui.ID_VIEW_FOLD_COLLAPSE:
385                  if foldable and is_expanded:
386                      enable = 1
387              cmdui.Enable(enable)

395      def ToggleBookmarkEvent(self, event, pos=-1):
396          """Toggle a bookmark at the specified or current position"""
397          if pos == -1:
398              pos, end = self.GetSel()
399          startLine = self.LineFromChar(pos)
400          self.GetDocument().MarkerToggle(startLine + 1, MARKER_BOOKMARK)
401          return 0

403      def GotoNextBookmarkEvent(self, event, fromPos=-1):
404          """Move to the next bookmark"""
405          if fromPos == -1:
406              fromPos, end = self.GetSel()
407          startLine = self.LineFromChar(fromPos) + 1  # Zero based line to start
408          nextLine = self.GetDocument().MarkerGetNext(startLine + 1, MARKER_BOOKMARK) - 1
409          if nextLine < 0:
410              nextLine = self.GetDocument().MarkerGetNext(0, MARKER_BOOKMARK) - 1
411          if nextLine < 0 or nextLine == startLine - 1:
412              win32api.MessageBeep()
413          else:
414              self.SCIEnsureVisible(nextLine)
415              self.SCIGotoLine(nextLine)
416          return 0

418      def TabKeyEvent(self, event):
419          """Insert an indent.  If no selection, a single indent, otherwise a block indent"""
420          # Handle auto-complete first.
421          if self.SCIAutoCActive():

423              return 0

425          return self.bindings.fire("<<smart-indent>>", event)

441      def FoldTopLevelEvent(self, event=None):
442          if not self.bFolding:
443              return 1
444  
445          win32ui.DoWaitCursor(1)
446          try:
447              self.Colorize()
448              maxLine = self.GetLineCount()
449              # Find the first line, and check out its state.
450              for lineSeek in range(maxLine):
451                  if self.SCIGetFoldLevel(lineSeek) & scintillacon.SC_FOLDLEVELHEADERFLAG:
452                      expanding = not self.SCIGetFoldExpanded(lineSeek)
453                      break
454              else:
455                  # no folds here!
456                  return
457              for lineSeek in range(lineSeek, maxLine):
458                  level = self.SCIGetFoldLevel(lineSeek)
459                  level_no = (
460                      level
461                      & scintillacon.SC_FOLDLEVELNUMBERMASK
462                      - scintillacon.SC_FOLDLEVELBASE
463                  )
464                  is_header = level & scintillacon.SC_FOLDLEVELHEADERFLAG
465                  # print(lineSeek, level_no, is_header)
466                  if level_no == 0 and is_header:
467                      if (expanding and not self.SCIGetFoldExpanded(lineSeek)) or (
468                          not expanding and self.SCIGetFoldExpanded(lineSeek)
469                      ):
470                          self.SCIToggleFold(lineSeek)
471          finally:
472              win32ui.DoWaitCursor(-1)

474      def FoldExpandSecondLevelEvent(self, event):
475          if not self.bFolding:
476              return 1
477          win32ui.DoWaitCursor(1)
478          ## I think this is needed since Scintilla may not have
479          ## already formatted parts of file outside visible window.
480          self.Colorize()
481          levels = [scintillacon.SC_FOLDLEVELBASE]
482          ## Scintilla's level number is based on amount of whitespace indentation
483          for lineno in range(self.GetLineCount()):
484              level = self.SCIGetFoldLevel(lineno)
485              if not level & scintillacon.SC_FOLDLEVELHEADERFLAG:
486                  continue
487              curr_level = level & scintillacon.SC_FOLDLEVELNUMBERMASK
488              if curr_level > levels[-1]:
489                  levels.append(curr_level)
490              try:
491                  level_ind = levels.index(curr_level)
492              except ValueError:
493                  ## probably syntax error in source file, bail
494                  break
495              levels = levels[: level_ind + 1]
496              if level_ind == 1 and not self.SCIGetFoldExpanded(lineno):
497                  self.SCIToggleFold(lineno)
498          win32ui.DoWaitCursor(-1)

598          if self.bSetMenus:
599              return
600          self.bSetMenus = 1
601  
602          submenu = win32ui.CreatePopupMenu()
603          newitems = idle.GetMenuItems("edit")
604          flags = win32con.MF_STRING | win32con.MF_ENABLED
605          for text, event in newitems:
606              id = bindings.event_to_commands.get(event)
607              if id is not None:
608                  keyname = pywin.scintilla.view.configManager.get_key_binding(
609                      event, ["editor"]
610                  )
611                  if keyname is not None:
612                      text += "\t" + keyname
613                  submenu.AppendMenu(flags, id, text)
614  
615          mainMenu = self.GetSharedMenu()
616          editMenu = mainMenu.GetSubMenu(1)
617          editMenu.AppendMenu(win32con.MF_SEPARATOR, 0, "")
618          editMenu.AppendMenu(
619              win32con.MF_STRING | win32con.MF_POPUP | win32con.MF_ENABLED,
620              submenu.GetHandle(),
621              "&Source Code",
622          )
