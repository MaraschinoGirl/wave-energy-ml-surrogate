97       def legalize(self):
98           """Check that all the element of the list are valid and remove SetLineno."""
99           lineno_pos = []
100          set_lineno = None
101          current_lineno = self.first_lineno
102  
103          for pos, instr in enumerate(self):
104              if isinstance(instr, SetLineno):
105                  set_lineno = instr.lineno
106                  lineno_pos.append(pos)
107                  continue
108              # Filter out Labels
109              if not isinstance(instr, Instr):
110                  continue
111              if set_lineno is not None:
112                  instr.lineno = set_lineno
113              elif instr.lineno is None:
114                  instr.lineno = current_lineno
115              else:
116                  current_lineno = instr.lineno
117  
118          for i in reversed(lineno_pos):
119              del self[i]

131  class _InstrList(list):
132      def _flat(self):
133          instructions = []
134          labels = {}
135          jumps = []
136  
137          offset = 0
138          for index, instr in enumerate(self):
139              if isinstance(instr, Label):
140                  instructions.append("label_instr%s" % index)
141                  labels[instr] = offset
142              else:
143                  if isinstance(instr, Instr) and isinstance(instr.arg, Label):
144                      target_label = instr.arg
145                      instr = _bytecode.ConcreteInstr(instr.name, 0, lineno=instr.lineno)
146                      jumps.append((target_label, instr))
147                  instructions.append(instr)
148                  offset += 1
149  
150          for target_label, instr in jumps:
151              instr.arg = labels[target_label]
152  
153          return instructions
154  
155      def __eq__(self, other):
156          if not isinstance(other, _InstrList):
157              other = _InstrList(other)
158  
159          return self._flat() == other._flat()

177          if not isinstance(instr, (Label, SetLineno, Instr)):
178              raise ValueError(
179                  "Bytecode must only contain Label, " "SetLineno, and Instr objects, " "but %s was found" % type(instr).__name__
180              )

182      def _copy_attr_from(self, bytecode):
183          super()._copy_attr_from(bytecode)
184          if isinstance(bytecode, Bytecode):

194      def compute_stacksize(self, *, check_pre_and_post=True):
195          cfg = _bytecode.ControlFlowGraph.from_bytecode(self)

198      def to_code(self, compute_jumps_passes=None, stacksize=None, *, check_pre_and_post=True):
199          # Prevent reconverting the concrete bytecode to bytecode and cfg to do the
200          # calculation if we need to do it.
201          if stacksize is None:
202              stacksize = self.compute_stacksize(check_pre_and_post=check_pre_and_post)
203          bc = self.to_concrete_bytecode(compute_jumps_passes=compute_jumps_passes)
204          return bc.to_code(stacksize=stacksize)
