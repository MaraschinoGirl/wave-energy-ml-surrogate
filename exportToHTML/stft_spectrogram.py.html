12   @keras_export("keras.layers.STFTSpectrogram")
13   class STFTSpectrogram(layers.Layer):
14       """Layer to compute the Short-Time Fourier Transform (STFT) on a 1D signal. 
15    
16       A layer that computes Spectrograms of the input signal to produce 
17       a spectrogram. This layers utilizes Short-Time Fourier Transform (STFT) by 
18       The layer computes Spectrograms based on STFT by utilizing convolution 
19       kernels, which allows parallelization on GPUs and trainable kernels for 
20       fine-tuning support. This layer allows different modes of output 
21       (e.g., log-scaled magnitude, phase, power spectral density, etc.) and 
22       provides flexibility in windowing, padding, and scaling options for the 
23       STFT calculation. 
24    
25       Examples: 
26    
27       Apply it as a non-trainable preprocessing layer on 3 audio tracks of 
28       1 channel, 10 seconds and sampled at 16 kHz. 
29    
30       >>> layer = keras.layers.STFTSpectrogram( 
31       ...     mode='log', 
32       ...     frame_length=256, 
33       ...     frame_step=128,   # 50% overlap 
34       ...     fft_length=512, 
35       ...     window="hann", 
36       ...     padding="valid", 
37       ...     trainable=False,  # non-trainable, preprocessing only 
38       ... ) 
39       >>> layer(keras.random.uniform(shape=(3, 160000, 1))).shape 
40       (3, 1249, 257) 
41    
42       Apply it as a trainable processing layer on 3 stereo audio tracks of 
43       2 channels, 10 seconds and sampled at 16 kHz. This is initialized as the 
44       non-trainable layer, but then can be trained jointly within a model. 
45    
46       >>> layer = keras.layers.STFTSpectrogram( 
47       ...     mode='log', 
48       ...     frame_length=256, 
49       ...     frame_step=128,    # 50% overlap 
50       ...     fft_length=512, 
51       ...     window="hamming",  # hamming windowing function 
52       ...     padding="same",    # padding to preserve the time dimension 
53       ...     trainable=True,    # trainable, this is the default in keras 
54       ... ) 
55       >>> layer(keras.random.uniform(shape=(3, 160000, 2))).shape 
56       (3, 1250, 514) 
57    
58       Similar to the last example, but add an extra dimension so the output is 
59       an image to be used with image models. We apply this here on a signal of 
60       3 input channels to output an image tensor, hence is directly applicable 
61       with an image model. 
62    
63       >>> layer = keras.layers.STFTSpectrogram( 
64       ...     mode='log', 
65       ...     frame_length=256, 
66       ...     frame_step=128, 
67       ...     fft_length=512, 
68       ...     padding="same", 
69       ...     expand_dims=True,  # this adds the extra dimension 
70       ... ) 
71       >>> layer(keras.random.uniform(shape=(3, 160000, 3))).shape 
72       (3, 1250, 257, 3) 
73    
74       Args: 
75           mode: String, the output type of the spectrogram. Can be one of 
76               `"log"`, `"magnitude`", `"psd"`, `"real`", `"imag`", `"angle`", 
77               `"stft`". Defaults to `"log`". 
78           frame_length: Integer, The length of each frame (window) for STFT in 
79               samples. Defaults to 256. 
80           frame_step: Integer, the step size (hop length) between 
81               consecutive frames. If not provided, defaults to half the 
82               frame_length. Defaults to `frame_length // 2`. 
83           fft_length: Integer, the size of frequency bins used in the Fast-Fourier 
84               Transform (FFT) to apply to each frame. Should be greater than or 
85               equal to `frame_length`.  Recommended to be a power of two. Defaults 
86               to the smallest power of two that is greater than or equal 
87               to `frame_length`. 
88           window: (String or array_like), the windowing function to apply to each 
89               frame. Can be `"hann`" (default), `"hamming`", or a custom window 
90               provided as an array_like. 
91           periodic: Boolean, if True, the window function will be treated as 
92               periodic. Defaults to `False`. 
93           scaling: String, type of scaling applied to the window. Can be 
94               `"density`", `"spectrum`", or None. Default is `"density`". 
95           padding: String, padding strategy. Can be `"valid`" or `"same`". 
96               Defaults to `"valid"`. 
97           expand_dims: Boolean, if True, will expand the output into spectrograms 
98               into two dimensions to be compatible with image models. 
99               Defaults to `False`. 
100          data_format: String, either `"channels_last"` or `"channels_first"`. 
101              The ordering of the dimensions in the inputs. `"channels_last"` 
102              corresponds to inputs with shape `(batch, height, width, channels)` 
103              while `"channels_first"` corresponds to inputs with shape 
104              `(batch, channels, height, weight)`. Defaults to `"channels_last"`. 
105   
106      Raises: 
107          ValueError: If an invalid value is provided for `"mode`", `"scaling`", 
108              `"padding`", or other input arguments. 
109          TypeError: If the input data type is not one of `"float16`", 
110              `"float32`", or `"float64`". 
111   
112      Input shape: 
113          A 3D tensor of shape `(batch_size, time_length, input_channels)`, if 
114          `data_format=="channels_last"`, and of shape 
115          `(batch_size, input_channels, time_length)` if 
116          `data_format=="channels_first"`, where `time_length` is the length of 
117          the input signal, and `input_channels` is the number of input channels. 
118          The same kernels are applied to each channel independently. 
119   
120      Output shape: 
121          If `data_format=="channels_first" and not expand_dims`, a 3D tensor: 
122              `(batch_size, input_channels * freq_channels, new_time_length)` 
123          If `data_format=="channels_last" and not expand_dims`, a 3D tensor: 
124              `(batch_size, new_time_length, input_channels * freq_channels)` 
125          If `data_format=="channels_first" and expand_dims`, a 4D tensor: 
126              `(batch_size, input_channels, new_time_length, freq_channels)` 
127          If `data_format=="channels_last" and expand_dims`, a 4D tensor: 
128              `(batch_size, new_time_length, freq_channels, input_channels)` 
129   
130          where `new_time_length` depends on the padding, and `freq_channels` is 
131          the number of FFT bins `(fft_length // 2 + 1)`. 
132      """
133  
134      def __init__(
135          self,
136          mode="log",
137          frame_length=256,
138          frame_step=None,
139          fft_length=None,
140          window="hann",
141          periodic=False,
142          scaling="density",
143          padding="valid",
144          expand_dims=False,
145          data_format=None,
146          **kwargs,
147      ):
148          if frame_step is not None and (
149              frame_step > frame_length or frame_step < 1
150          ):
151              raise ValueError(
152                  "`frame_step` should be a positive integer not greater than "
153                  f"`frame_length`. Received frame_step={frame_step}, "
154                  f"frame_length={frame_length}"
155              )
156  
157          if fft_length is not None and fft_length < frame_length:
158              raise ValueError(
159                  "`fft_length` should be not less than `frame_length`. "
160                  f"Received fft_length={fft_length}, frame_length={frame_length}"
161              )
162  
163          if fft_length is not None and (fft_length & -fft_length) != fft_length:
164              warnings.warn(
165                  "`fft_length` is recommended to be a power of two. "
166                  f"Received fft_length={fft_length}"
167              )
168  
169          all_modes = ["log", "magnitude", "psd", "real", "imag", "angle", "stft"]
170  
171          if mode not in all_modes:
172              raise ValueError(
173                  "Output mode is invalid, it must be one of "
174                  f"{', '.join(all_modes)}. Received: mode={mode}"
175              )
176  
177          if scaling is not None and scaling not in ["density", "spectrum"]:
178              raise ValueError(
179                  "Scaling is invalid, it must be `None`, 'density' "
180                  f"or 'spectrum'. Received scaling={scaling}"
181              )
182  
183          if padding not in ["valid", "same"]:
184              raise ValueError(
185                  "Padding is invalid, it should be 'valid', 'same'. "
186                  f"Received: padding={padding}"
187              )
188  
189          if isinstance(window, str):
190              # throws an exception for invalid window function
191              scipy.signal.get_window(window, 1)
192  
193          super().__init__(**kwargs)
194  
195          self.mode = mode
196  
197          self.frame_length = frame_length
198          self.frame_step = frame_step
199          self._frame_step = frame_step or self.frame_length // 2
200          self.fft_length = fft_length
201          self._fft_length = fft_length or (
202              2 ** int(math.ceil(math.log2(frame_length)))
203          )
204  
205          self.window = window
206          self.periodic = periodic
207          self.scaling = scaling
208          self.padding = padding
209          self.expand_dims = expand_dims
210          self.data_format = backend.standardize_data_format(data_format)
211          self.input_spec = layers.input_spec.InputSpec(ndim=3)
212  
213      def build(self, input_shape):
214          shape = (self.frame_length, 1, self._fft_length // 2 + 1)
215  
216          if self.mode != "imag":
217              self.real_kernel = self.add_weight(
218                  name="real_kernel",
219                  shape=shape,
220                  initializer=initializers.STFT(
221                      "real", self.window, self.scaling, self.periodic
222                  ),
223              )
224          if self.mode != "real":
225              self.imag_kernel = self.add_weight(
226                  name="imag_kernel",
227                  shape=shape,
228                  initializer=initializers.STFT(
229                      "imag", self.window, self.scaling, self.periodic
230                  ),
231              )
232          self.built = True
233  
234      def _adjust_shapes(self, outputs):
235          _, channels, freq_channels, time_seq = ops.shape(outputs)
236          batch_size = -1
237          if self.data_format == "channels_last":
238              if self.expand_dims:
239                  outputs = ops.transpose(outputs, [0, 3, 2, 1])
240                  # [batch_size, time_seq, freq_channels, input_channels]
241              else:
242                  outputs = ops.reshape(
243                      outputs,
244                      [batch_size, channels * freq_channels, time_seq],
245                  )
246                  # [batch_size, input_channels * freq_channels, time_seq]
247                  outputs = ops.transpose(outputs, [0, 2, 1])
248          else:
249              if self.expand_dims:
250                  outputs = ops.transpose(outputs, [0, 1, 3, 2])
251                  # [batch_size, channels, time_seq, freq_channels]
252              else:
253                  outputs = ops.reshape(
254                      outputs,
255                      [batch_size, channels * freq_channels, time_seq],
256                  )
257          return outputs
258  
259      def _apply_conv(self, inputs, kernel):
260          if self.data_format == "channels_last":
261              _, time_seq, channels = ops.shape(inputs)
262              inputs = ops.transpose(inputs, [0, 2, 1])
263              inputs = ops.reshape(inputs, [-1, time_seq, 1])
264          else:
265              _, channels, time_seq = ops.shape(inputs)
266              inputs = ops.reshape(inputs, [-1, 1, time_seq])
267  
268          outputs = ops.conv(
269              inputs,
270              ops.cast(kernel, backend.standardize_dtype(inputs.dtype)),
271              padding=self.padding,
272              strides=self._frame_step,
273              data_format=self.data_format,
274          )
275          batch_size = -1
276          if self.data_format == "channels_last":
277              _, time_seq, freq_channels = ops.shape(outputs)
278              outputs = ops.transpose(outputs, [0, 2, 1])
279              outputs = ops.reshape(
280                  outputs,
281                  [batch_size, channels, freq_channels, time_seq],
282              )
283          else:
284              _, freq_channels, time_seq = ops.shape(outputs)
285              outputs = ops.reshape(
286                  outputs,
287                  [batch_size, channels, freq_channels, time_seq],
288              )
289          return outputs
290  
291      def call(self, inputs):
292          dtype = inputs.dtype
293          if backend.standardize_dtype(dtype) not in {
294              "float16",
295              "float32",
296              "float64",
297          }:
298              raise TypeError(
299                  "Invalid input type. Expected `float16`, `float32` or "
300                  f"`float64`. Received: input type={dtype}"
301              )
302  
303          real_signal = None
304          imag_signal = None
305          power = None
306  
307          if self.mode != "imag":
308              real_signal = self._apply_conv(inputs, self.real_kernel)
309          if self.mode != "real":
310              imag_signal = self._apply_conv(inputs, self.imag_kernel)
311  
312          if self.mode == "real":
313              return self._adjust_shapes(real_signal)
314          elif self.mode == "imag":
315              return self._adjust_shapes(imag_signal)
316          elif self.mode == "angle":
317              return self._adjust_shapes(ops.arctan2(imag_signal, real_signal))
318          elif self.mode == "stft":
319              return self._adjust_shapes(
320                  ops.concatenate([real_signal, imag_signal], axis=2)
321              )
322          else:
323              power = ops.square(real_signal) + ops.square(imag_signal)
324  
325          if self.mode == "psd":
326              return self._adjust_shapes(
327                  power
328                  + ops.pad(
329                      power[:, :, 1:-1, :], [[0, 0], [0, 0], [1, 1], [0, 0]]
330                  )
331              )
332          linear_stft = self._adjust_shapes(
333              ops.sqrt(ops.maximum(power, backend.epsilon()))
334          )
335  
336          if self.mode == "magnitude":
337              return linear_stft
338          else:
339              return ops.log(ops.maximum(linear_stft, backend.epsilon()))
340  
341      def compute_output_shape(self, input_shape):
342          if self.data_format == "channels_last":
343              channels = input_shape[-1]
344          else:
345              channels = input_shape[1]
346          freq_channels = self._fft_length // 2 + 1
347          if self.mode == "stft":
348              freq_channels *= 2
349          shape = ops.operation_utils.compute_conv_output_shape(
350              input_shape,
351              freq_channels * channels,
352              (self.frame_length,),
353              strides=self._frame_step,
354              padding=self.padding,
355              data_format=self.data_format,
356          )
357          if self.data_format == "channels_last":
358              batch_size, time_seq, _ = shape
359          else:
360              batch_size, _, time_seq = shape
361          if self.expand_dims:
362              if self.data_format == "channels_last":
363                  return (batch_size, time_seq, freq_channels, channels)
364              else:
365                  return (batch_size, channels, time_seq, freq_channels)
366          return shape

368      def get_config(self):
369          config = super().get_config()
370          config.update(
371              {
372                  "mode": self.mode,
373                  "frame_length": self.frame_length,
374                  "frame_step": self.frame_step,
375                  "fft_length": self.fft_length,
376                  "window": self.window,
377                  "periodic": self.periodic,
378                  "scaling": self.scaling,
379                  "padding": self.padding,
380                  "data_format": self.data_format,
381                  "expand_dims": self.expand_dims,
382              }
383          )
384          return config
