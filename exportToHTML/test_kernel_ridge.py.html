1    import numpy as np
2    import pytest
3    
4    from sklearn.datasets import make_regression
5    from sklearn.kernel_ridge import KernelRidge
6    from sklearn.linear_model import Ridge
7    from sklearn.metrics.pairwise import pairwise_kernels
8    from sklearn.utils._testing import assert_array_almost_equal, ignore_warnings
9    from sklearn.utils.fixes import CSC_CONTAINERS, CSR_CONTAINERS
10   
11   X, y = make_regression(n_features=10, random_state=0)
12   Y = np.array([y, y]).T
13   
14   
15   def test_kernel_ridge():
16       pred = Ridge(alpha=1, fit_intercept=False).fit(X, y).predict(X)
17       pred2 = KernelRidge(kernel="linear", alpha=1).fit(X, y).predict(X)
18       assert_array_almost_equal(pred, pred2)
19   
20   
21   @pytest.mark.parametrize("sparse_container", [*CSR_CONTAINERS, *CSC_CONTAINERS])
22   def test_kernel_ridge_sparse(sparse_container):
23       X_sparse = sparse_container(X)
24       pred = (
25           Ridge(alpha=1, fit_intercept=False, solver="cholesky")
26           .fit(X_sparse, y)
27           .predict(X_sparse)
28       )
29       pred2 = KernelRidge(kernel="linear", alpha=1).fit(X_sparse, y).predict(X_sparse)
30       assert_array_almost_equal(pred, pred2)
31   
32   
33   def test_kernel_ridge_singular_kernel():
34       # alpha=0 causes a LinAlgError in computing the dual coefficients,
35       # which causes a fallback to a lstsq solver. This is tested here.
36       pred = Ridge(alpha=0, fit_intercept=False).fit(X, y).predict(X)
37       kr = KernelRidge(kernel="linear", alpha=0)
38       ignore_warnings(kr.fit)(X, y)
39       pred2 = kr.predict(X)
40       assert_array_almost_equal(pred, pred2)
41   
42   
43   def test_kernel_ridge_precomputed():
44       for kernel in ["linear", "rbf", "poly", "cosine"]:
45           K = pairwise_kernels(X, X, metric=kernel)
46           pred = KernelRidge(kernel=kernel).fit(X, y).predict(X)
47           pred2 = KernelRidge(kernel="precomputed").fit(K, y).predict(K)
48           assert_array_almost_equal(pred, pred2)
49   
50   
51   def test_kernel_ridge_precomputed_kernel_unchanged():
52       K = np.dot(X, X.T)
53       K2 = K.copy()
54       KernelRidge(kernel="precomputed").fit(K, y)
55       assert_array_almost_equal(K, K2)
56   
57   
58   def test_kernel_ridge_sample_weights():
59       K = np.dot(X, X.T)  # precomputed kernel
60       sw = np.random.RandomState(0).rand(X.shape[0])
61   
62       pred = Ridge(alpha=1, fit_intercept=False).fit(X, y, sample_weight=sw).predict(X)
63       pred2 = KernelRidge(kernel="linear", alpha=1).fit(X, y, sample_weight=sw).predict(X)
64       pred3 = (
65           KernelRidge(kernel="precomputed", alpha=1)
66           .fit(K, y, sample_weight=sw)
67           .predict(K)
68       )
69       assert_array_almost_equal(pred, pred2)
70       assert_array_almost_equal(pred, pred3)
71

73   def test_kernel_ridge_multi_output():
74       pred = Ridge(alpha=1, fit_intercept=False).fit(X, Y).predict(X)
75       pred2 = KernelRidge(kernel="linear", alpha=1).fit(X, Y).predict(X)
76       assert_array_almost_equal(pred, pred2)
77   
78       pred3 = KernelRidge(kernel="linear", alpha=1).fit(X, y).predict(X)
79       pred3 = np.array([pred3, pred3]).T
80       assert_array_almost_equal(pred2, pred3)
