3    typedef enum {
4        SF_ERROR_OK = 0,    /* no error */
5        SF_ERROR_SINGULAR,  /* singularity encountered */
6        SF_ERROR_UNDERFLOW, /* floating point underflow */
7        SF_ERROR_OVERFLOW,  /* floating point overflow */
8        SF_ERROR_SLOW,      /* too many iterations required */
9        SF_ERROR_LOSS,      /* loss of precision */
10       SF_ERROR_NO_RESULT, /* no result obtained */
11       SF_ERROR_DOMAIN,    /* out of domain */
12       SF_ERROR_ARG,       /* invalid input parameter */
13       SF_ERROR_OTHER,     /* unclassified error */
14       SF_ERROR_MEMORY,    /* memory allocation failed */
15       SF_ERROR__LAST
16   } sf_error_t;

32   template <typename T>
33   XSF_HOST_DEVICE void set_error_and_nan(const char *name, sf_error_t code, T &value) {
34       if (code != SF_ERROR_OK) {
35           set_error(name, code, nullptr);
36   
37           if (code == SF_ERROR_DOMAIN || code == SF_ERROR_OVERFLOW || code == SF_ERROR_NO_RESULT) {
38               value = std::numeric_limits<T>::quiet_NaN();
39           }
40       }
41   }
42   
43   template <typename T>
44   XSF_HOST_DEVICE void set_error_and_nan(const char *name, sf_error_t code, std::complex<T> &value) {
45       if (code != SF_ERROR_OK) {
46           set_error(name, code, nullptr);
47   
48           if (code == SF_ERROR_DOMAIN || code == SF_ERROR_OVERFLOW || code == SF_ERROR_NO_RESULT) {
49               value.real(std::numeric_limits<T>::quiet_NaN());
50               value.imag(std::numeric_limits<T>::quiet_NaN());
51           }
52       }
53   }
