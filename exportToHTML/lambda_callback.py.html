2    from keras.src.callbacks.callback import Callback

5    @keras_export("keras.callbacks.LambdaCallback")
6    class LambdaCallback(Callback):
7        """Callback for creating simple, custom callbacks on-the-fly. 
8     
9        This callback is constructed with anonymous functions that will be called 
10       at the appropriate time (during `Model.{fit | evaluate | predict}`). 
11       Note that the callbacks expects positional arguments, as: 
12    
13       - `on_epoch_begin` and `on_epoch_end` expect two positional arguments: 
14         `epoch`, `logs` 
15       - `on_train_begin` and `on_train_end` expect one positional argument: 
16         `logs` 
17       - `on_train_batch_begin` and `on_train_batch_end` expect two positional 
18         arguments: `batch`, `logs` 
19       - See `Callback` class definition for the full list of functions and their 
20         expected arguments. 
21    
22       Args: 
23           on_epoch_begin: called at the beginning of every epoch. 
24           on_epoch_end: called at the end of every epoch. 
25           on_train_begin: called at the beginning of model training. 
26           on_train_end: called at the end of model training. 
27           on_train_batch_begin: called at the beginning of every train batch. 
28           on_train_batch_end: called at the end of every train batch. 
29           kwargs: Any function in `Callback` that you want to override by 
30               passing `function_name=function`. For example, 
31               `LambdaCallback(.., on_train_end=train_end_fn)`. The custom function 
32               needs to have same arguments as the ones defined in `Callback`. 
33    
34       Example: 
35    
36       ```python 
37       # Print the batch number at the beginning of every batch. 
38       batch_print_callback = LambdaCallback( 
39           on_train_batch_begin=lambda batch,logs: print(batch)) 
40    
41       # Stream the epoch loss to a file in JSON format. The file content 
42       # is not well-formed JSON but rather has a JSON object per line. 
43       import json 
44       json_log = open('loss_log.json', mode='wt', buffering=1) 
45       json_logging_callback = LambdaCallback( 
46           on_epoch_end=lambda epoch, logs: json_log.write( 
47               json.dumps({'epoch': epoch, 'loss': logs['loss']}) + '\n'), 
48           on_train_end=lambda logs: json_log.close() 
49       ) 
50    
51       # Terminate some processes after having finished model training. 
52       processes = ... 
53       cleanup_callback = LambdaCallback( 
54           on_train_end=lambda logs: [ 
55               p.terminate() for p in processes if p.is_alive()]) 
56    
57       model.fit(..., 
58                 callbacks=[batch_print_callback, 
59                            json_logging_callback, 
60                            cleanup_callback]) 
61       ``` 
62       """
63   
64       def __init__(
65           self,
66           on_epoch_begin=None,
67           on_epoch_end=None,
68           on_train_begin=None,
69           on_train_end=None,
70           on_train_batch_begin=None,
71           on_train_batch_end=None,
72           **kwargs,
73       ):
74           super().__init__()
75           self.__dict__.update(kwargs)
76           if on_epoch_begin is not None:
77               self.on_epoch_begin = on_epoch_begin
78           if on_epoch_end is not None:
79               self.on_epoch_end = on_epoch_end
80           if on_train_begin is not None:
81               self.on_train_begin = on_train_begin
82           if on_train_end is not None:
83               self.on_train_end = on_train_end
84           if on_train_batch_begin is not None:
85               self.on_train_batch_begin = on_train_batch_begin
86           if on_train_batch_end is not None:
87               self.on_train_batch_end = on_train_batch_end
