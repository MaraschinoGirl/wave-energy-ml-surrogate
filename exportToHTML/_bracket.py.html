60   def _bracket_root(func, xl0, xr0=None, *, xmin=None, xmax=None, factor=None,
61                     args=(), maxiter=1000):
62       """Bracket the root of a monotonic scalar function of one variable 
63    
64       This function works elementwise when `xl0`, `xr0`, `xmin`, `xmax`, `factor`, and 
65       the elements of `args` are broadcastable arrays. 
66    
67       Parameters 
68       ---------- 
69       func : callable 
70           The function for which the root is to be bracketed. 
71           The signature must be:: 
72    
73               func(x: ndarray, *args) -> ndarray 
74    
75           where each element of ``x`` is a finite real and ``args`` is a tuple, 
76           which may contain an arbitrary number of arrays that are broadcastable 
77           with `x`. ``func`` must be an elementwise function: each element 
78           ``func(x)[i]`` must equal ``func(x[i])`` for all indices ``i``. 
79       xl0, xr0: float array_like 
80           Starting guess of bracket, which need not contain a root. If `xr0` is 
81           not provided, ``xr0 = xl0 + 1``. Must be broadcastable with one another. 
82       xmin, xmax : float array_like, optional 
83           Minimum and maximum allowable endpoints of the bracket, inclusive. Must 
84           be broadcastable with `xl0` and `xr0`. 
85       factor : float array_like, default: 2 
86           The factor used to grow the bracket. See notes for details. 
87       args : tuple, optional 
88           Additional positional arguments to be passed to `func`.  Must be arrays 
89           broadcastable with `xl0`, `xr0`, `xmin`, and `xmax`. If the callable to be 
90           bracketed requires arguments that are not broadcastable with these 
91           arrays, wrap that callable with `func` such that `func` accepts 
92           only `x` and broadcastable arrays. 
93       maxiter : int, optional 
94           The maximum number of iterations of the algorithm to perform. 
95    
96       Returns 
97       ------- 
98       res : _RichResult 
99           An instance of `scipy._lib._util._RichResult` with the following 
100          attributes. The descriptions are written as though the values will be 
101          scalars; however, if `func` returns an array, the outputs will be 
102          arrays of the same shape. 
103   
104          xl, xr : float 
105              The lower and upper ends of the bracket, if the algorithm 
106              terminated successfully. 
107          fl, fr : float 
108              The function value at the lower and upper ends of the bracket. 
109          nfev : int 
110              The number of function evaluations required to find the bracket. 
111              This is distinct from the number of times `func` is *called* 
112              because the function may evaluated at multiple points in a single 
113              call. 
114          nit : int 
115              The number of iterations of the algorithm that were performed. 
116          status : int 
117              An integer representing the exit status of the algorithm. 
118   
119              - ``0`` : The algorithm produced a valid bracket. 
120              - ``-1`` : The bracket expanded to the allowable limits without finding a bracket. 
121              - ``-2`` : The maximum number of iterations was reached. 
122              - ``-3`` : A non-finite value was encountered. 
123              - ``-4`` : Iteration was terminated by `callback`. 
124              - ``-5``: The initial bracket does not satisfy `xmin <= xl0 < xr0 < xmax`. 
125              - ``1`` : The algorithm is proceeding normally (in `callback` only). 
126              - ``2`` : A bracket was found in the opposite search direction (in `callback` only). 
127   
128          success : bool 
129              ``True`` when the algorithm terminated successfully (status ``0``). 
130   
131      Notes 
132      ----- 
133      This function generalizes an algorithm found in pieces throughout 
134      `scipy.stats`. The strategy is to iteratively grow the bracket ``(l, r)`` 
135       until ``func(l) < 0 < func(r)``. The bracket grows to the left as follows. 
136   
137      - If `xmin` is not provided, the distance between `xl0` and `l` is iteratively 
138        increased by `factor`. 
139      - If `xmin` is provided, the distance between `xmin` and `l` is iteratively 
140        decreased by `factor`. Note that this also *increases* the bracket size. 
141   
142      Growth of the bracket to the right is analogous. 
143   
144      Growth of the bracket in one direction stops when the endpoint is no longer 
145      finite, the function value at the endpoint is no longer finite, or the 
146      endpoint reaches its limiting value (`xmin` or `xmax`). Iteration terminates 
147      when the bracket stops growing in both directions, the bracket surrounds 
148      the root, or a root is found (accidentally). 
149   
150      If two brackets are found - that is, a bracket is found on both sides in 
151      the same iteration, the smaller of the two is returned. 
152      If roots of the function are found, both `l` and `r` are set to the 
153      leftmost root. 
154

173      # The approach is to treat the left and right searches as though they were
174      # (almost) totally independent one-sided bracket searches. (The interaction
175      # is considered when checking for termination and preparing the result
176      # object.)
177      # `x` is the "moving" end of the bracket
178      x = xp.concat(xs)
179      f = xp.concat(fs)
180      invalid_bracket = xp.concat((invalid_bracket, invalid_bracket))
181      n = x.shape[0] // 2

183      # `x_last` is the previous location of the moving end of the bracket. If
184      # the signs of `f` and `f_last` are different, `x` and `x_last` form a
185      # bracket.
186      x_last = xp.concat((x[n:], x[:n]))
187      f_last = xp.concat((f[n:], f[:n]))
188      # `x0` is the "fixed" end of the bracket.
189      x0 = x_last
190      # We don't need to retain the corresponding function value, since the
191      # fixed end of the bracket is only needed to compute the new value of the
192      # moving end; it is never returned.
193      limit = xp.concat((xmin, xmax))

195      factor = xp_ravel(xp.broadcast_to(factor, shape), xp=xp)
196      factor = xp.astype(factor, dtype, copy=False)

200      args = [xp.concat((arg, arg)) for arg in args]

202      # This is needed due to inner workings of `eim._loop`.

207      # For searches without a limit, the distance between the fixed end of the
208      # bracket `x0` and the moving end `x` will grow by `factor` each iteration.
209      # For searches with a limit, the distance between the `limit` and moving
210      # end of the bracket `x` will shrink by `factor` each iteration.

212      ni = ~i
213      d = xp.zeros_like(x)
214      d[i] = x[i] - x0[i]
215      d[ni] = limit[ni] - x[ni]

222                         active=active, d=d, x_last=x_last, f_last=f_last,

226                        ('nit', 'nit'), ('nfev', 'nfev'), ('fl', 'fl'),

230      def pre_func_eval(work):
231          # Initialize moving end of bracket
232          x = xp.zeros_like(work.x)
233  
234          # Unlimited brackets grow by `factor` by increasing distance from fixed
235          # end to moving end.
236          i = xp.isinf(work.limit)  # indices of unlimited brackets
237          work.d[i] *= work.factor[i]
238          x[i] = work.x0[i] + work.d[i]
239  
240          # Limited brackets grow by decreasing the distance from the limit to
241          # the moving end.
242          ni = ~i  # indices of limited brackets
243          work.d[ni] /= work.factor[ni]
244          x[ni] = work.limit[ni] - work.d[ni]
245  
246          return x

248      def post_func_eval(x, f, work):
249          # Keep track of the previous location of the moving end so that we can
250          # return a narrower bracket. (The alternative is to remember the
251          # original fixed end, but then the bracket would be wider than needed.)
252          work.x_last = work.x
253          work.f_last = work.f
254          work.x = x
255          work.f = f

257      def check_termination(work):
258          # Condition 0: initial bracket is invalid
259          stop = (work.status == eim._EINPUTERR)
260  
261          # Condition 1: a valid bracket (or the root itself) has been found
262          sf = xp.sign(work.f)
263          sf_last = xp.sign(work.f_last)
264          i = ((sf_last == -sf) | (sf_last == 0) | (sf == 0)) & ~stop
265          work.status[i] = eim._ECONVERGED
266          stop[i] = True
267  
268          # Condition 2: the other side's search found a valid bracket.
269          # (If we just found a bracket with the rightward search, we can stop
270          #  the leftward search, and vice-versa.)
271          # To do this, we need to set the status of the other side's search;
272          # this is tricky because `work.status` contains only the *active*
273          # elements, so we don't immediately know the index of the element we
274          # need to set - or even if it's still there. (That search may have
275          # terminated already, e.g. by reaching its `limit`.)
276          # To facilitate this, `work.active` contains a unit integer index of
277          # each search. Index `k` (`k < n)` and `k + n` correspond with a
278          # leftward and rightward search, respectively. Elements are removed
279          # from `work.active` just as they are removed from `work.status`, so
280          # we use `work.active` to help find the right location in
281          # `work.status`.
282          # Get the integer indices of the elements that can also stop
283          also_stop = (work.active[i] + work.n) % (2*work.n)
284          # Check whether they are still active.
285          # To start, we need to find out where in `work.active` they would
286          # appear if they are indeed there.
287          j = xp.searchsorted(work.active, also_stop)
288          # If the location exceeds the length of the `work.active`, they are
289          # not there.
290          j = j[j < work.active.shape[0]]
291          # Check whether they are still there.
292          j = j[also_stop == work.active[j]]
293          # Now convert these to boolean indices to use with `work.status`.
294          i = xp.zeros_like(stop)
295          i[j] = True  # boolean indices of elements that can also stop
296          i = i & ~stop
297          work.status[i] = _ESTOPONESIDE
298          stop[i] = True
299  
300          # Condition 3: moving end of bracket reaches limit
301          i = (work.x == work.limit) & ~stop
302          work.status[i] = _ELIMITS
303          stop[i] = True
304  
305          # Condition 4: non-finite value encountered
306          i = ~(xp.isfinite(work.x) & xp.isfinite(work.f)) & ~stop
307          work.status[i] = eim._EVALUEERR
308          stop[i] = True
309  
310          return stop

315      def customize_result(res, shape):
316          n = res['x'].shape[0] // 2
317  
318          # To avoid ambiguity, below we refer to `xl0`, the initial left endpoint
319          # as `a` and `xr0`, the initial right endpoint, as `b`.
320          # Because we treat the two one-sided searches as though they were
321          # independent, what we keep track of in `work` and what we want to
322          # return in `res` look quite different. Combine the results from the
323          # two one-sided searches before reporting the results to the user.
324          # - "a" refers to the leftward search (the moving end started at `a`)
325          # - "b" refers to the rightward search (the moving end started at `b`)
326          # - "l" refers to the left end of the bracket (closer to -oo)
327          # - "r" refers to the right end of the bracket (closer to +oo)
328          xal = res['x'][:n]
329          xar = res['x_last'][:n]
330          xbl = res['x_last'][n:]
331          xbr = res['x'][n:]
332  
333          fal = res['f'][:n]
334          far = res['f_last'][:n]
335          fbl = res['f_last'][n:]
336          fbr = res['f'][n:]
337  
338          # Initialize the brackets and corresponding function values to return
339          # to the user. Brackets may not be valid (e.g. there is no root,
340          # there weren't enough iterations, NaN encountered), but we still need
341          # to return something. One option would be all NaNs, but what I've
342          # chosen here is the left- and right-most points at which the function
343          # has been evaluated. This gives the user some information about what
344          # interval of the real line has been searched and shows that there is
345          # no sign change between the two ends.
346          xl = xp.asarray(xal, copy=True)
347          fl = xp.asarray(fal, copy=True)
348          xr = xp.asarray(xbr, copy=True)
349          fr = xp.asarray(fbr, copy=True)
350  
351          # `status` indicates whether the bracket is valid or not. If so,
352          # we want to adjust the bracket we return to be the narrowest possible
353          # given the points at which we evaluated the function.
354          # For example if bracket "a" is valid and smaller than bracket "b" OR
355          # if bracket "a" is valid and bracket "b" is not valid, we want to
356          # return bracket "a" (and vice versa).
357          sa = res['status'][:n]
358          sb = res['status'][n:]
359  
360          da = xar - xal
361          db = xbr - xbl
362  
363          i1 = ((da <= db) & (sa == 0)) | ((sa == 0) & (sb != 0))
364          i2 = ((db <= da) & (sb == 0)) | ((sb == 0) & (sa != 0))
365  
366          xr[i1] = xar[i1]
367          fr[i1] = far[i1]
368          xl[i2] = xbl[i2]
369          fl[i2] = fbl[i2]
370  
371          # Finish assembling the result object
372          res['xl'] = xl
373          res['xr'] = xr
374          res['fl'] = fl
375          res['fr'] = fr
376  
377          res['nit'] = xp.maximum(res['nit'][:n], res['nit'][n:])
378          res['nfev'] = res['nfev'][:n] + res['nfev'][n:]
379          # If the status on one side is zero, the status is zero. In any case,
380          # report the status from one side only.
381          res['status'] = xp.where(sa == 0, sa, sb)
382          res['success'] = (res['status'] == 0)
383  
384          del res['x']
385          del res['f']
386          del res['x_last']
387          del res['f_last']
388  
389          return shape[:-1]
390  
391      return eim._loop(work, callback, shape, maxiter, func, args, dtype,
392                       pre_func_eval, post_func_eval, check_termination,
393                       post_termination_check, customize_result, res_work_pairs,
394                       xp)

470          raise ValueError(message)

475  def _bracket_minimum(func, xm0, *, xl0=None, xr0=None, xmin=None, xmax=None,
476                       factor=None, args=(), maxiter=1000):
477      """Bracket the minimum of a unimodal scalar function of one variable 
478   
479      This function works elementwise when `xm0`, `xl0`, `xr0`, `xmin`, `xmax`, 
480      and the elements of `args` are broadcastable arrays. 
481   
482      Parameters 
483      ---------- 
484      func : callable 
485          The function for which the minimum is to be bracketed. 
486          The signature must be:: 
487   
488              func(x: ndarray, *args) -> ndarray 
489   
490          where each element of ``x`` is a finite real and ``args`` is a tuple, 
491          which may contain an arbitrary number of arrays that are broadcastable 
492          with ``x``. `func` must be an elementwise function: each element 
493          ``func(x)[i]`` must equal ``func(x[i])`` for all indices `i`. 
494      xm0: float array_like 
495          Starting guess for middle point of bracket. 
496      xl0, xr0: float array_like, optional 
497          Starting guesses for left and right endpoints of the bracket. Must be 
498          broadcastable with one another and with `xm0`. 
499      xmin, xmax : float array_like, optional 
500          Minimum and maximum allowable endpoints of the bracket, inclusive. Must 
501          be broadcastable with `xl0`, `xm0`, and `xr0`. 
502      factor : float array_like, optional 
503          Controls expansion of bracket endpoint in downhill direction. Works 
504          differently in the cases where a limit is set in the downhill direction 
505          with `xmax` or `xmin`. See Notes. 
506      args : tuple, optional 
507          Additional positional arguments to be passed to `func`.  Must be arrays 
508          broadcastable with `xl0`, `xm0`, `xr0`, `xmin`, and `xmax`. If the 
509          callable to be bracketed requires arguments that are not broadcastable 
510          with these arrays, wrap that callable with `func` such that `func` 
511          accepts only ``x`` and broadcastable arrays. 
512      maxiter : int, optional 
513          The maximum number of iterations of the algorithm to perform. The number 
514          of function evaluations is three greater than the number of iterations. 
515   
516      Returns 
517      ------- 
518      res : _RichResult 
519          An instance of `scipy._lib._util._RichResult` with the following 
520          attributes. The descriptions are written as though the values will be 
521          scalars; however, if `func` returns an array, the outputs will be 
522          arrays of the same shape. 
523   
524          xl, xm, xr : float 
525              The left, middle, and right points of the bracket, if the algorithm 
526              terminated successfully. 
527          fl, fm, fr : float 
528              The function value at the left, middle, and right points of the bracket. 
529          nfev : int 
530              The number of function evaluations required to find the bracket. 
531          nit : int 
532              The number of iterations of the algorithm that were performed. 
533          status : int 
534              An integer representing the exit status of the algorithm. 
535   
536              - ``0`` : The algorithm produced a valid bracket. 
537              - ``-1`` : The bracket expanded to the allowable limits. Assuming 
538                         unimodality, this implies the endpoint at the limit is a 
539                         minimizer. 
540              - ``-2`` : The maximum number of iterations was reached. 
541              - ``-3`` : A non-finite value was encountered. 
542              - ``-4`` : ``None`` shall pass. 
543              - ``-5`` : The initial bracket does not satisfy 
544                         `xmin <= xl0 < xm0 < xr0 <= xmax`. 
545   
546          success : bool 
547              ``True`` when the algorithm terminated successfully (status ``0``). 
548   
549      Notes 
550      ----- 
551      Similar to `scipy.optimize.bracket`, this function seeks to find real 
552      points ``xl < xm < xr`` such that ``f(xl) >= f(xm)`` and ``f(xr) >= f(xm)``, 
553      where at least one of the inequalities is strict. Unlike `scipy.optimize.bracket`, 
554      this function can operate in a vectorized manner on array input, so long as 
555      the input arrays are broadcastable with each other. Also unlike 
556      `scipy.optimize.bracket`, users may specify minimum and maximum endpoints 
557      for the desired bracket. 
558   
559      Given an initial trio of points ``xl = xl0``, ``xm = xm0``, ``xr = xr0``, 
560      the algorithm checks if these points already give a valid bracket. If not, 
561      a new endpoint, ``w`` is chosen in the "downhill" direction, ``xm`` becomes the new 
562      opposite endpoint, and either `xl` or `xr` becomes the new middle point, 
563      depending on which direction is downhill. The algorithm repeats from here. 
564   
565      The new endpoint `w` is chosen differently depending on whether or not a 
566      boundary `xmin` or `xmax` has been set in the downhill direction. Without 
567      loss of generality, suppose the downhill direction is to the right, so that 
568      ``f(xl) > f(xm) > f(xr)``. If there is no boundary to the right, then `w` 
569      is chosen to be ``xr + factor * (xr - xm)`` where `factor` is controlled by 
570      the user (defaults to 2.0) so that step sizes increase in geometric proportion. 
571      If there is a boundary, `xmax` in this case, then `w` is chosen to be 
572      ``xmax - (xmax - xr)/factor``, with steps slowing to a stop at 
573      `xmax`. This cautious approach ensures that a minimum near but distinct from 
574      the boundary isn't missed while also detecting whether or not the `xmax` is 
575      a minimizer when `xmax` is reached after a finite number of steps. 
576      """  # noqa: E501
577      callback = None  # works; I just don't want to test it
578  
579      temp = _bracket_minimum_iv(func, xm0, xl0, xr0, xmin, xmax, factor, args, maxiter)
580      func, xm0, xl0, xr0, xmin, xmax, factor, args, maxiter, xp = temp
581  
582      xs = (xl0, xm0, xr0)
583      temp = eim._initialize(func, xs, args)
584      func, xs, fs, args, shape, dtype, xp = temp
585  
586      xl0, xm0, xr0 = xs
587      fl0, fm0, fr0 = fs
588      xmin = xp.astype(xp.broadcast_to(xmin, shape), dtype, copy=False)
589      xmin = xp_ravel(xmin, xp=xp)
590      xmax = xp.astype(xp.broadcast_to(xmax, shape), dtype, copy=False)
591      xmax = xp_ravel(xmax, xp=xp)
592      invalid_bracket = ~((xmin <= xl0) & (xl0 < xm0) & (xm0 < xr0) & (xr0 <= xmax))
593      # We will modify factor later on so make a copy. np.broadcast_to returns
594      # a read-only view.
595      factor = xp.astype(xp.broadcast_to(factor, shape), dtype, copy=True)
596      factor = xp_ravel(factor)
597  
598      # To simplify the logic, swap xl and xr if f(xl) < f(xr). We should always be
599      # marching downhill in the direction from xl to xr.
600      comp = fl0 < fr0
601      xl0[comp], xr0[comp] = xr0[comp], xl0[comp]
602      fl0[comp], fr0[comp] = fr0[comp], fl0[comp]
603      # We only need the boundary in the direction we're traveling.
604      limit = xp.where(comp, xmin, xmax)
605  
606      unlimited = xp.isinf(limit)
607      limited = ~unlimited
608      step = xp.empty_like(xl0)
609  
610      step[unlimited] = (xr0[unlimited] - xm0[unlimited])
611      step[limited] = (limit[limited] - xr0[limited])
612  
613      # Step size is divided by factor for case where there is a limit.
614      factor[limited] = 1 / factor[limited]
615  
616      status = xp.full_like(xl0, eim._EINPROGRESS, dtype=xp.int32)
617      status[invalid_bracket] = eim._EINPUTERR
618      nit, nfev = 0, 3
619  
620      work = _RichResult(xl=xl0, xm=xm0, xr=xr0, xr0=xr0, fl=fl0, fm=fm0, fr=fr0,
621                         step=step, limit=limit, limited=limited, factor=factor, nit=nit,
622                         nfev=nfev, status=status, args=args)
623  
624      res_work_pairs = [('status', 'status'), ('xl', 'xl'), ('xm', 'xm'), ('xr', 'xr'),
625                        ('nit', 'nit'), ('nfev', 'nfev'), ('fl', 'fl'), ('fm', 'fm'),
626                        ('fr', 'fr')]

628      def pre_func_eval(work):
629          work.step *= work.factor
630          x = xp.empty_like(work.xr)
631          x[~work.limited] = work.xr0[~work.limited] + work.step[~work.limited]
632          x[work.limited] = work.limit[work.limited] - work.step[work.limited]
633          # Since the new bracket endpoint is calculated from an offset with the
634          # limit, it may be the case that the new endpoint equals the old endpoint,
635          # when the old endpoint is sufficiently close to the limit. We use the
636          # limit itself as the new endpoint in these cases.
637          x[work.limited] = xp.where(
638              x[work.limited] == work.xr[work.limited],
639              work.limit[work.limited],
640              x[work.limited],
641          )
642          return x
643  
644      def post_func_eval(x, f, work):
645          work.xl, work.xm, work.xr = work.xm, work.xr, x
646          work.fl, work.fm, work.fr = work.fm, work.fr, f
647  
648      def check_termination(work):
649          # Condition 0: Initial bracket is invalid.
650          stop = (work.status == eim._EINPUTERR)
651  
652          # Condition 1: A valid bracket has been found.
653          i = (
654              (work.fl >= work.fm) & (work.fr > work.fm)
655              | (work.fl > work.fm) & (work.fr >= work.fm)
656          ) & ~stop
657          work.status[i] = eim._ECONVERGED
658          stop[i] = True
659  
660          # Condition 2: Moving end of bracket reaches limit.
661          i = (work.xr == work.limit) & ~stop
662          work.status[i] = _ELIMITS
663          stop[i] = True
664  
665          # Condition 3: non-finite value encountered
666          i = ~(xp.isfinite(work.xr) & xp.isfinite(work.fr)) & ~stop
667          work.status[i] = eim._EVALUEERR
668          stop[i] = True
669  
670          return stop
671  
672      def post_termination_check(work):
673          pass
674  
675      def customize_result(res, shape):
676          # Reorder entries of xl and xr if they were swapped due to f(xl0) < f(xr0).
677          comp = res['xl'] > res['xr']
678          res['xl'][comp], res['xr'][comp] = res['xr'][comp], res['xl'][comp]
679          res['fl'][comp], res['fr'][comp] = res['fr'][comp], res['fl'][comp]
680          return shape
681  
682      return eim._loop(work, callback, shape,
683                       maxiter, func, args, dtype,
684                       pre_func_eval, post_func_eval,
685                       check_termination, post_termination_check,
686                       customize_result, res_work_pairs, xp)
