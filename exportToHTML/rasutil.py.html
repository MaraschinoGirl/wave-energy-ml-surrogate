13   def Connect(rasEntryName, numRetries=5):
14       """Make a connection to the specified RAS entry. 
15    
16       Returns a tuple of (bool, handle) on success. 
17       - bool is 1 if a new connection was established, or 0 is a connection already existed. 
18       - handle is a RAS HANDLE that can be passed to Disconnect() to end the connection. 
19    
20       Raises a ConnectionError if the connection could not be established. 
21       """
22       assert numRetries > 0
23       for info in win32ras.EnumConnections():
24           if info[1].lower() == rasEntryName.lower():
25               print("Already connected to", rasEntryName)
26               return 0, info[0]
27   
28       dial_params, have_pw = win32ras.GetEntryDialParams(None, rasEntryName)
29       if not have_pw:
30           print("Error: The password is not saved for this connection")
31           print(
32               "Please connect manually selecting the 'save password' option and try again"
33           )
34           sys.exit(1)
35   
36       print("Connecting to", rasEntryName, "...")
37       retryCount = numRetries
38       while retryCount > 0:
39           rasHandle, errCode = win32ras.Dial(None, None, dial_params, None)
40           if win32ras.IsHandleValid(rasHandle):
41               bValid = 1
42               break
43           print("Retrying...")
44           win32api.Sleep(5000)
45           retryCount -= 1
46   
47       if errCode:
48           raise ConnectionError(errCode, win32ras.GetErrorString(errCode))
49       return 1, rasHandle

52   def Disconnect(handle):
53       if isinstance(handle, str):  # have they passed a connection name?
54           for info in win32ras.EnumConnections():
55               if info[1].lower() == handle.lower():
56                   handle = info[0]
57                   break
58           else:
59               raise ConnectionError(0, "Not connected to entry '%s'" % handle)
60   
61       win32ras.HangUp(handle)
