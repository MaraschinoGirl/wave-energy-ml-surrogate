10   class Weight(keras.layers.Layer):
11   
12       def __init__(self, **kwargs):
13           super(Weight, self).__init__(**kwargs)
14           self.kernel = None
15   
16       def build(self, input_shape):
17           self.kernel = self.add_weight(
18               shape=(1000, 100),
19               name='kernel',
20               initializer='glorot_uniform',
21           )
22   
23       def compute_output_shape(self, input_shape):
24           return K.int_shape(self.kernel)
25   
26       def call(self, inputs, mask=None, training=None):
27           return K.expand_dims(self.kernel, axis=0)

32       def test_weight_drop_shape(self):
33           model = keras.models.Sequential()
34           model.add(TargetedDropout(
35               layer=Weight(),
36               drop_rate=0.0,
37               target_rate=0.35,
38               drop_patterns=['kernel'],
39               mode=TargetedDropout.MODE_WEIGHT,
40               input_shape=(1,)
41           ))
42           model.compile('sgd', 'mse')
43   
44           model_path = os.path.join(tempfile.gettempdir(), 'keras_targeted_dropout_%f.h5' % np.random.random())
45           model.save(model_path)
46           model = keras.models.load_model(model_path, custom_objects={
47               'Weight': Weight,
48               'TargetedDropout': TargetedDropout,
49           })
50   
51           dropped = model.predict(np.ones((1, 1)))[0]
52           row, col = dropped.shape
53           for c in range(col):
54               rate = np.sum(np.equal(dropped[:, c], 0.0)) / row
55               self.assertGreater(rate, 0.34)
56               self.assertLess(rate, 0.36)
57           rate = np.mean(np.equal(dropped, 0.0))
58           self.assertGreater(rate, 0.34)
59           self.assertLess(rate, 0.36)
