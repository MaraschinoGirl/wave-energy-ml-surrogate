15   class table_T_S_I__5(DefaultTable.DefaultTable):
16       def decompile(self, data, ttFont):
17           numGlyphs = ttFont["maxp"].numGlyphs
18           assert len(data) == 2 * numGlyphs
19           a = array.array("H")
20           a.frombytes(data)
21           if sys.byteorder != "big":
22               a.byteswap()
23           self.glyphGrouping = {}
24           for i in range(numGlyphs):
25               self.glyphGrouping[ttFont.getGlyphName(i)] = a[i]
26   
27       def compile(self, ttFont):
28           glyphNames = ttFont.getGlyphOrder()
29           a = array.array("H")
30           for i in range(len(glyphNames)):
31               a.append(self.glyphGrouping.get(glyphNames[i], 0))
32           if sys.byteorder != "big":
33               a.byteswap()
34           return a.tobytes()
35   
36       def toXML(self, writer, ttFont):
37           names = sorted(self.glyphGrouping.keys())
38           for glyphName in names:
39               writer.simpletag(
40                   "glyphgroup", name=glyphName, value=self.glyphGrouping[glyphName]
41               )
42               writer.newline()
43   
44       def fromXML(self, name, attrs, content, ttFont):
45           if not hasattr(self, "glyphGrouping"):
46               self.glyphGrouping = {}
47           if name != "glyphgroup":
48               return
49           self.glyphGrouping[attrs["name"]] = safeEval(attrs["value"])
