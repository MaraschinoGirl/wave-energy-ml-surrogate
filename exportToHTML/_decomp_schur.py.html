17   def schur(a, output='real', lwork=None, overwrite_a=False, sort=None,
18             check_finite=True):
19       """ 
20       Compute Schur decomposition of a matrix. 
21    
22       The Schur decomposition is:: 
23    
24           A = Z T Z^H 
25    
26       where Z is unitary and T is either upper-triangular, or for real 
27       Schur decomposition (output='real'), quasi-upper triangular. In 
28       the quasi-triangular form, 2x2 blocks describing complex-valued 
29       eigenvalue pairs may extrude from the diagonal. 
30    
31       Parameters 
32       ---------- 
33       a : (M, M) array_like 
34           Matrix to decompose 
35       output : {'real', 'complex'}, optional 
36           When the dtype of `a` is real, this specifies whether to compute 
37           the real or complex Schur decomposition. 
38           When the dtype of `a` is complex, this argument is ignored, and the 
39           complex Schur decomposition is computed. 
40       lwork : int, optional 
41           Work array size. If None or -1, it is automatically computed. 
42       overwrite_a : bool, optional 
43           Whether to overwrite data in a (may improve performance). 
44       sort : {None, callable, 'lhp', 'rhp', 'iuc', 'ouc'}, optional 
45           Specifies whether the upper eigenvalues should be sorted. A callable 
46           may be passed that, given an eigenvalue, returns a boolean denoting 
47           whether the eigenvalue should be sorted to the top-left (True). 
48    
49           - If ``output='complex'`` OR the dtype of `a` is complex, the callable 
50             should have one argument: the eigenvalue expressed as a complex number. 
51           - If ``output='real'`` AND the dtype of `a` is real, the callable should have 
52             two arguments: the real and imaginary parts of the eigenvalue, respectively. 
53    
54           Alternatively, string parameters may be used:: 
55    
56               'lhp'   Left-hand plane (real(eigenvalue) < 0.0) 
57               'rhp'   Right-hand plane (real(eigenvalue) >= 0.0) 
58               'iuc'   Inside the unit circle (abs(eigenvalue) <= 1.0) 
59               'ouc'   Outside the unit circle (abs(eigenvalue) > 1.0) 
60    
61           Defaults to None (no sorting). 
62       check_finite : bool, optional 
63           Whether to check that the input matrix contains only finite numbers. 
64           Disabling may give a performance gain, but may result in problems 
65           (crashes, non-termination) if the inputs do contain infinities or NaNs. 
66    
67       Returns 
68       ------- 
69       T : (M, M) ndarray 
70           Schur form of A. It is real-valued for the real Schur decomposition. 
71       Z : (M, M) ndarray 
72           An unitary Schur transformation matrix for A. 
73           It is real-valued for the real Schur decomposition. 
74       sdim : int 
75           If and only if sorting was requested, a third return value will 
76           contain the number of eigenvalues satisfying the sort condition. 
77           Note that complex conjugate pairs for which the condition is true 
78           for either eigenvalue count as 2. 
79    
80       Raises 
81       ------ 
82       LinAlgError 
83           Error raised under three conditions: 
84    
85           1. The algorithm failed due to a failure of the QR algorithm to 
86              compute all eigenvalues. 
87           2. If eigenvalue sorting was requested, the eigenvalues could not be 
88              reordered due to a failure to separate eigenvalues, usually because 
89              of poor conditioning. 
90           3. If eigenvalue sorting was requested, roundoff errors caused the 
91              leading eigenvalues to no longer satisfy the sorting condition. 
92    
93       See Also 
94       -------- 
95       rsf2csf : Convert real Schur form to complex Schur form 
96    
97       Examples 
98       -------- 
99       >>> import numpy as np 
100      >>> from scipy.linalg import schur, eigvals 
101      >>> A = np.array([[0, 2, 2], [0, 1, 2], [1, 0, 1]]) 
102      >>> T, Z = schur(A) 
103      >>> T 
104      array([[ 2.65896708,  1.42440458, -1.92933439], 
105             [ 0.        , -0.32948354, -0.49063704], 
106             [ 0.        ,  1.31178921, -0.32948354]]) 
107      >>> Z 
108      array([[0.72711591, -0.60156188, 0.33079564], 
109             [0.52839428, 0.79801892, 0.28976765], 
110             [0.43829436, 0.03590414, -0.89811411]]) 
111   
112      >>> T2, Z2 = schur(A, output='complex') 
113      >>> T2 
114      array([[ 2.65896708, -1.22839825+1.32378589j,  0.42590089+1.51937378j], # may vary 
115             [ 0.        , -0.32948354+0.80225456j, -0.59877807+0.56192146j], 
116             [ 0.        ,  0.                    , -0.32948354-0.80225456j]]) 
117      >>> eigvals(T2) 
118      array([2.65896708, -0.32948354+0.80225456j, -0.32948354-0.80225456j])   # may vary 
119   
120      A custom eigenvalue-sorting condition that sorts by positive imaginary part 
121      is satisfied by only one eigenvalue. 
122   
123      >>> _, _, sdim = schur(A, output='complex', sort=lambda x: x.imag > 1e-15) 
124      >>> sdim 
125      1 
126   
127      When ``output='real'`` and the array `a` is real, the `sort` callable must accept 
128      the real and imaginary parts as separate arguments. Note that now the complex 
129      eigenvalues ``-0.32948354+0.80225456j`` and ``-0.32948354-0.80225456j`` will be 
130      treated as a complex conjugate pair, and according to the `sdim` documentation, 
131      complex conjugate pairs for which the condition is True for *either* eigenvalue 
132      increase `sdim` by *two*. 
133   
134      >>> _, _, sdim = schur(A, output='real', sort=lambda x, y: y > 1e-15) 
135      >>> sdim 
136      2 
137   
138      """
139      if output not in ['real', 'complex', 'r', 'c']:
140          raise ValueError("argument must be 'real', or 'complex'")
141      if check_finite:
142          a1 = asarray_chkfinite(a)
143      else:
144          a1 = asarray(a)
145      if np.issubdtype(a1.dtype, np.integer):
146          a1 = asarray(a, dtype=np.dtype("long"))
147      if len(a1.shape) != 2 or (a1.shape[0] != a1.shape[1]):
148          raise ValueError('expected square matrix')
149  
150      typ = a1.dtype.char
151      if output in ['complex', 'c'] and typ not in ['F', 'D']:
152          if typ in _double_precision:
153              a1 = a1.astype('D')
154          else:
155              a1 = a1.astype('F')
156  
157      # accommodate empty matrix
158      if a1.size == 0:
159          t0, z0 = schur(np.eye(2, dtype=a1.dtype))
160          if sort is None:
161              return (np.empty_like(a1, dtype=t0.dtype),
162                      np.empty_like(a1, dtype=z0.dtype))
163          else:
164              return (np.empty_like(a1, dtype=t0.dtype),
165                      np.empty_like(a1, dtype=z0.dtype), 0)
166  
167      overwrite_a = overwrite_a or (_datacopied(a1, a))
168      gees, = get_lapack_funcs(('gees',), (a1,))
169      if lwork is None or lwork == -1:
170          # get optimal work array
171          result = gees(lambda x: None, a1, lwork=-1)
172          lwork = result[-2][0].real.astype(np.int_)
173  
174      if sort is None:
175          sort_t = 0
176          def sfunction(x, y=None):
177              return None
178      else:
179          sort_t = 1
180          if callable(sort):
181              sfunction = sort
182          elif sort == 'lhp':
183              def sfunction(x, y=None):
184                  return x.real < 0.0
185          elif sort == 'rhp':
186              def sfunction(x, y=None):
187                  return x.real >= 0.0
188          elif sort == 'iuc':
189              def sfunction(x, y=None):
190                  z = x if y is None else x + y*1j
191                  return abs(z) <= 1.0
192          elif sort == 'ouc':
193              def sfunction(x, y=None):
194                  z = x if y is None else x + y*1j
195                  return abs(z) > 1.0
196          else:
197              raise ValueError("'sort' parameter must either be 'None', or a "
198                               "callable, or one of ('lhp','rhp','iuc','ouc')")
199  
200      result = gees(sfunction, a1, lwork=lwork, overwrite_a=overwrite_a,
201                    sort_t=sort_t)
202  
203      info = result[-1]
204      if info < 0:
205          raise ValueError(f'illegal value in {-info}-th argument of internal gees')
206      elif info == a1.shape[0] + 1:
207          raise LinAlgError('Eigenvalues could not be separated for reordering.')
208      elif info == a1.shape[0] + 2:
209          raise LinAlgError('Leading eigenvalues do not satisfy sort condition.')
210      elif info > 0:
211          raise LinAlgError("Schur form not found. Possibly ill-conditioned.")
212  
213      if sort is None:
214          return result[0], result[-3]
215      else:
216          return result[0], result[-3], result[1]

222  _array_kind = {'b': 0, 'h': 0, 'B': 0, 'i': 0, 'l': 0,
223                 'f': 0, 'd': 0, 'F': 1, 'D': 1}
224  _array_precision = {'i': 1, 'l': 1, 'f': 0, 'd': 1, 'F': 0, 'D': 1}
225  _array_type = [['f', 'd'], ['F', 'D']]

228  def _commonType(*arrays):
229      kind = 0
230      precision = 0
231      for a in arrays:
232          t = a.dtype.char
233          kind = max(kind, _array_kind[t])
234          precision = max(precision, _array_precision[t])
235      return _array_type[kind][precision]
236  
237  
238  def _castCopy(type, *arrays):
239      cast_arrays = ()
240      for a in arrays:
241          if a.dtype.char == type:
242              cast_arrays = cast_arrays + (a.copy(),)
243          else:
244              cast_arrays = cast_arrays + (a.astype(type),)
245      if len(cast_arrays) == 1:
246          return cast_arrays[0]
247      else:
248          return cast_arrays

251  def rsf2csf(T, Z, check_finite=True):
252      """ 
253      Convert real Schur form to complex Schur form. 
254   
255      Convert a quasi-diagonal real-valued Schur form to the upper-triangular 
256      complex-valued Schur form. 
257   
258      Parameters 
259      ---------- 
260      T : (M, M) array_like 
261          Real Schur form of the original array 
262      Z : (M, M) array_like 
263          Schur transformation matrix 
264      check_finite : bool, optional 
265          Whether to check that the input arrays contain only finite numbers. 
266          Disabling may give a performance gain, but may result in problems 
267          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
268   
269      Returns 
270      ------- 
271      T : (M, M) ndarray 
272          Complex Schur form of the original array 
273      Z : (M, M) ndarray 
274          Schur transformation matrix corresponding to the complex form 
275   
276      See Also 
277      -------- 
278      schur : Schur decomposition of an array 
279   
280      Examples 
281      -------- 
282      >>> import numpy as np 
283      >>> from scipy.linalg import schur, rsf2csf 
284      >>> A = np.array([[0, 2, 2], [0, 1, 2], [1, 0, 1]]) 
285      >>> T, Z = schur(A) 
286      >>> T 
287      array([[ 2.65896708,  1.42440458, -1.92933439], 
288             [ 0.        , -0.32948354, -0.49063704], 
289             [ 0.        ,  1.31178921, -0.32948354]]) 
290      >>> Z 
291      array([[0.72711591, -0.60156188, 0.33079564], 
292             [0.52839428, 0.79801892, 0.28976765], 
293             [0.43829436, 0.03590414, -0.89811411]]) 
294      >>> T2 , Z2 = rsf2csf(T, Z) 
295      >>> T2 
296      array([[2.65896708+0.j, -1.64592781+0.743164187j, -1.21516887+1.00660462j], 
297             [0.+0.j , -0.32948354+8.02254558e-01j, -0.82115218-2.77555756e-17j], 
298             [0.+0.j , 0.+0.j, -0.32948354-0.802254558j]]) 
299      >>> Z2 
300      array([[0.72711591+0.j,  0.28220393-0.31385693j,  0.51319638-0.17258824j], 
301             [0.52839428+0.j,  0.24720268+0.41635578j, -0.68079517-0.15118243j], 
302             [0.43829436+0.j, -0.76618703+0.01873251j, -0.03063006+0.46857912j]]) 
303   
304      """
305      if check_finite:
306          Z, T = map(asarray_chkfinite, (Z, T))
307      else:
308          Z, T = map(asarray, (Z, T))
309  
310      for ind, X in enumerate([Z, T]):
311          if X.ndim != 2 or X.shape[0] != X.shape[1]:
312              raise ValueError(f"Input '{'ZT'[ind]}' must be square.")
313  
314      if T.shape[0] != Z.shape[0]:
315          message = f"Input array shapes must match: Z: {Z.shape} vs. T: {T.shape}"
316          raise ValueError(message)
317      N = T.shape[0]
318      t = _commonType(Z, T, array([3.0], 'F'))
319      Z, T = _castCopy(t, Z, T)
320  
321      for m in range(N-1, 0, -1):
322          if abs(T[m, m-1]) > eps*(abs(T[m-1, m-1]) + abs(T[m, m])):
323              mu = eigvals(T[m-1:m+1, m-1:m+1]) - T[m, m]
324              r = norm([mu[0], T[m, m-1]])
325              c = mu[0] / r
326              s = T[m, m-1] / r
327              G = array([[c.conj(), s], [-s, c]], dtype=t)
328  
329              T[m-1:m+1, m-1:] = G.dot(T[m-1:m+1, m-1:])
330              T[:m+1, m-1:m+1] = T[:m+1, m-1:m+1].dot(G.conj().T)
331              Z[:, m-1:m+1] = Z[:, m-1:m+1].dot(G.conj().T)
332  
333          T[m, m-1] = 0.0
334      return T, Z
