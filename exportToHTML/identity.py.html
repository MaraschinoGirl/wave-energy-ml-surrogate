7    @keras_export("keras.layers.Identity")
8    class Identity(Layer):
9        """Identity layer. 
10    
11       This layer should be used as a placeholder when no operation is to be 
12       performed. The layer just returns its `inputs` argument as output. 
13       """
14   
15       def __init__(self, **kwargs):
16           super().__init__(**kwargs)
17           self.supports_masking = True
18           self.built = True
19   
20       def call(self, inputs):
21           return inputs
22   
23       def compute_output_shape(self, input_shape):
24           return input_shape
25   
26       def compute_output_spec(self, inputs):
27           return tree.map_structure(
28               lambda x: KerasTensor(x.shape, dtype=x.dtype, sparse=x.sparse),
29               inputs,
30           )
