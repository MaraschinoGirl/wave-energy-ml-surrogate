2    usage = """\ 
3    testDCOM.py - Simple DCOM test 
4    Usage: testDCOM.py serverName 
5     
6    Attempts to start the Python.Interpreter object on the named machine, 
7    and checks that the object is indeed running remotely. 
8     
9    Requires the named server be configured to run DCOM (using dcomcnfg.exe), 
10   and the Python.Interpreter object installed and registered on that machine. 
11    
12   The Python.Interpreter object must be installed on the local machine, 
13   but no special DCOM configuration should be necessary. 
14   """
15   import sys
16   
17   # NOTE: If you configured the object locally using dcomcnfg, you could
18   # simple use Dispatch rather than DispatchEx.
19   import pythoncom
20   import win32api
21   import win32com.client
22   
23   
24   def test(serverName):
25       if serverName.lower() == win32api.GetComputerName().lower():
26           print("You must specify a remote server name, not the local machine!")
27           return
28   
29       # Hack to overcome a DCOM limitation.  As the Python.Interpreter object
30       # is probably installed locally as an InProc object, DCOM seems to ignore
31       # all settings, and use the local object.
32       clsctx = pythoncom.CLSCTX_SERVER & ~pythoncom.CLSCTX_INPROC_SERVER
33       ob = win32com.client.DispatchEx("Python.Interpreter", serverName, clsctx=clsctx)
34       ob.Exec("import win32api")
35       actualName = ob.Eval("win32api.GetComputerName()")
36       if serverName.lower() != actualName.lower():
37           print(
38               "Error: The object created on server '{}' reported its name as '{}'".format(
39                   serverName, actualName
40               )
41           )
42       else:
43           print("Object created and tested OK on server '%s'" % serverName)
44   
45   
46   if __name__ == "__main__":
47       if len(sys.argv) == 2:
48           test(sys.argv[1])
49       else:
50           print(usage)
