46   # --- CONTEXT structures and constants -----------------------------------------
47   
48   # The following values specify the type of access in the first parameter
49   # of the exception record when the exception code specifies an access
50   # violation.
51   EXCEPTION_READ_FAULT = 0  # exception caused by a read
52   EXCEPTION_WRITE_FAULT = 1  # exception caused by a write
53   EXCEPTION_EXECUTE_FAULT = 8  # exception caused by an instruction fetch
54   
55   CONTEXT_i386 = 0x00010000  # this assumes that i386 and
56   CONTEXT_i486 = 0x00010000  # i486 have identical context records
57   
58   CONTEXT_CONTROL = CONTEXT_i386 | long(0x00000001)  # SS:SP, CS:IP, FLAGS, BP
59   CONTEXT_INTEGER = CONTEXT_i386 | long(0x00000002)  # AX, BX, CX, DX, SI, DI
60   CONTEXT_SEGMENTS = CONTEXT_i386 | long(0x00000004)  # DS, ES, FS, GS
61   CONTEXT_FLOATING_POINT = CONTEXT_i386 | long(0x00000008)  # 387 state
62   CONTEXT_DEBUG_REGISTERS = CONTEXT_i386 | long(0x00000010)  # DB 0-3,6,7
63   CONTEXT_EXTENDED_REGISTERS = CONTEXT_i386 | long(0x00000020)  # cpu specific extensions
64   
65   CONTEXT_FULL = CONTEXT_CONTROL | CONTEXT_INTEGER | CONTEXT_SEGMENTS
66   
67   CONTEXT_ALL = (
68       CONTEXT_CONTROL | CONTEXT_INTEGER | CONTEXT_SEGMENTS | CONTEXT_FLOATING_POINT | CONTEXT_DEBUG_REGISTERS | CONTEXT_EXTENDED_REGISTERS
69   )
70   
71   SIZE_OF_80387_REGISTERS = 80
72   MAXIMUM_SUPPORTED_EXTENSION = 512

75   # typedef struct _FLOATING_SAVE_AREA {
76   #     DWORD   ControlWord;
77   #     DWORD   StatusWord;
78   #     DWORD   TagWord;
79   #     DWORD   ErrorOffset;
80   #     DWORD   ErrorSelector;
81   #     DWORD   DataOffset;
82   #     DWORD   DataSelector;
83   #     BYTE    RegisterArea[SIZE_OF_80387_REGISTERS];
84   #     DWORD   Cr0NpxState;
85   # } FLOATING_SAVE_AREA;
86   class FLOATING_SAVE_AREA(Structure):
87       _pack_ = 1
88       _fields_ = [
89           ("ControlWord", DWORD),
90           ("StatusWord", DWORD),
91           ("TagWord", DWORD),
92           ("ErrorOffset", DWORD),
93           ("ErrorSelector", DWORD),
94           ("DataOffset", DWORD),
95           ("DataSelector", DWORD),
96           ("RegisterArea", BYTE * SIZE_OF_80387_REGISTERS),
97           ("Cr0NpxState", DWORD),
98       ]
99   
100      _integer_members = ("ControlWord", "StatusWord", "TagWord", "ErrorOffset", "ErrorSelector", "DataOffset", "DataSelector", "Cr0NpxState")
101  
102      @classmethod
103      def from_dict(cls, fsa):
104          "Instance a new structure from a Python dictionary."
105          fsa = dict(fsa)
106          s = cls()
107          for key in cls._integer_members:
108              setattr(s, key, fsa.get(key))
109          ra = fsa.get("RegisterArea", None)
110          if ra is not None:
111              for index in compat.xrange(0, SIZE_OF_80387_REGISTERS):
112                  s.RegisterArea[index] = ra[index]
113          return s
114  
115      def to_dict(self):
116          "Convert a structure into a Python dictionary."
117          fsa = dict()
118          for key in self._integer_members:
119              fsa[key] = getattr(self, key)
120          ra = [self.RegisterArea[index] for index in compat.xrange(0, SIZE_OF_80387_REGISTERS)]
121          ra = tuple(ra)
122          fsa["RegisterArea"] = ra
123          return fsa

130  # typedef struct _CONTEXT {
131  #     DWORD ContextFlags;
132  #     DWORD   Dr0;
133  #     DWORD   Dr1;
134  #     DWORD   Dr2;
135  #     DWORD   Dr3;
136  #     DWORD   Dr6;
137  #     DWORD   Dr7;
138  #     FLOATING_SAVE_AREA FloatSave;
139  #     DWORD   SegGs;
140  #     DWORD   SegFs;
141  #     DWORD   SegEs;
142  #     DWORD   SegDs;
143  #     DWORD   Edi;
144  #     DWORD   Esi;
145  #     DWORD   Ebx;
146  #     DWORD   Edx;
147  #     DWORD   Ecx;
148  #     DWORD   Eax;
149  #     DWORD   Ebp;
150  #     DWORD   Eip;
151  #     DWORD   SegCs;
152  #     DWORD   EFlags;
153  #     DWORD   Esp;
154  #     DWORD   SegSs;
155  #     BYTE    ExtendedRegisters[MAXIMUM_SUPPORTED_EXTENSION];
156  # } CONTEXT;
157  class CONTEXT(Structure):
158      arch = ARCH_I386
159  
160      _pack_ = 1
161  
162      # Context Frame
163      #
164      #  This frame has a several purposes: 1) it is used as an argument to
165      #  NtContinue, 2) is is used to constuct a call frame for APC delivery,
166      #  and 3) it is used in the user level thread creation routines.
167      #
168      #  The layout of the record conforms to a standard call frame.
169  
170      _fields_ = [
171          # The flags values within this flag control the contents of
172          # a CONTEXT record.
173          #
174          # If the context record is used as an input parameter, then
175          # for each portion of the context record controlled by a flag
176          # whose value is set, it is assumed that that portion of the
177          # context record contains valid context. If the context record
178          # is being used to modify a threads context, then only that
179          # portion of the threads context will be modified.
180          #
181          # If the context record is used as an IN OUT parameter to capture
182          # the context of a thread, then only those portions of the thread's
183          # context corresponding to set flags will be returned.
184          #
185          # The context record is never used as an OUT only parameter.
186          ("ContextFlags", DWORD),
187          # This section is specified/returned if CONTEXT_DEBUG_REGISTERS is
188          # set in ContextFlags.  Note that CONTEXT_DEBUG_REGISTERS is NOT
189          # included in CONTEXT_FULL.
190          ("Dr0", DWORD),
191          ("Dr1", DWORD),
192          ("Dr2", DWORD),
193          ("Dr3", DWORD),
194          ("Dr6", DWORD),
195          ("Dr7", DWORD),
196          # This section is specified/returned if the
197          # ContextFlags word contains the flag CONTEXT_FLOATING_POINT.
198          ("FloatSave", FLOATING_SAVE_AREA),
199          # This section is specified/returned if the
200          # ContextFlags word contains the flag CONTEXT_SEGMENTS.
201          ("SegGs", DWORD),
202          ("SegFs", DWORD),
203          ("SegEs", DWORD),
204          ("SegDs", DWORD),
205          # This section is specified/returned if the
206          # ContextFlags word contains the flag CONTEXT_INTEGER.
207          ("Edi", DWORD),
208          ("Esi", DWORD),
209          ("Ebx", DWORD),
210          ("Edx", DWORD),
211          ("Ecx", DWORD),
212          ("Eax", DWORD),
213          # This section is specified/returned if the
214          # ContextFlags word contains the flag CONTEXT_CONTROL.
215          ("Ebp", DWORD),
216          ("Eip", DWORD),
217          ("SegCs", DWORD),  # MUST BE SANITIZED
218          ("EFlags", DWORD),  # MUST BE SANITIZED
219          ("Esp", DWORD),
220          ("SegSs", DWORD),
221          # This section is specified/returned if the ContextFlags word
222          # contains the flag CONTEXT_EXTENDED_REGISTERS.
223          # The format and contexts are processor specific.
224          ("ExtendedRegisters", BYTE * MAXIMUM_SUPPORTED_EXTENSION),
225      ]
226  
227      _ctx_debug = ("Dr0", "Dr1", "Dr2", "Dr3", "Dr6", "Dr7")
228      _ctx_segs = (
229          "SegGs",
230          "SegFs",
231          "SegEs",
232          "SegDs",
233      )
234      _ctx_int = ("Edi", "Esi", "Ebx", "Edx", "Ecx", "Eax")
235      _ctx_ctrl = ("Ebp", "Eip", "SegCs", "EFlags", "Esp", "SegSs")
236  
237      @classmethod
238      def from_dict(cls, ctx):
239          "Instance a new structure from a Python dictionary."
240          ctx = Context(ctx)
241          s = cls()
242          ContextFlags = ctx["ContextFlags"]
243          setattr(s, "ContextFlags", ContextFlags)
244          if (ContextFlags & CONTEXT_DEBUG_REGISTERS) == CONTEXT_DEBUG_REGISTERS:
245              for key in s._ctx_debug:
246                  setattr(s, key, ctx[key])
247          if (ContextFlags & CONTEXT_FLOATING_POINT) == CONTEXT_FLOATING_POINT:
248              fsa = ctx["FloatSave"]
249              s.FloatSave = FLOATING_SAVE_AREA.from_dict(fsa)
250          if (ContextFlags & CONTEXT_SEGMENTS) == CONTEXT_SEGMENTS:
251              for key in s._ctx_segs:
252                  setattr(s, key, ctx[key])
253          if (ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER:
254              for key in s._ctx_int:
255                  setattr(s, key, ctx[key])
256          if (ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL:
257              for key in s._ctx_ctrl:
258                  setattr(s, key, ctx[key])
259          if (ContextFlags & CONTEXT_EXTENDED_REGISTERS) == CONTEXT_EXTENDED_REGISTERS:
260              er = ctx["ExtendedRegisters"]
261              for index in compat.xrange(0, MAXIMUM_SUPPORTED_EXTENSION):
262                  s.ExtendedRegisters[index] = er[index]
263          return s
264  
265      def to_dict(self):
266          "Convert a structure into a Python native type."
267          ctx = Context()
268          ContextFlags = self.ContextFlags
269          ctx["ContextFlags"] = ContextFlags
270          if (ContextFlags & CONTEXT_DEBUG_REGISTERS) == CONTEXT_DEBUG_REGISTERS:
271              for key in self._ctx_debug:
272                  ctx[key] = getattr(self, key)
273          if (ContextFlags & CONTEXT_FLOATING_POINT) == CONTEXT_FLOATING_POINT:
274              ctx["FloatSave"] = self.FloatSave.to_dict()
275          if (ContextFlags & CONTEXT_SEGMENTS) == CONTEXT_SEGMENTS:
276              for key in self._ctx_segs:
277                  ctx[key] = getattr(self, key)
278          if (ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER:
279              for key in self._ctx_int:
280                  ctx[key] = getattr(self, key)
281          if (ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL:
282              for key in self._ctx_ctrl:
283                  ctx[key] = getattr(self, key)
284          if (ContextFlags & CONTEXT_EXTENDED_REGISTERS) == CONTEXT_EXTENDED_REGISTERS:
285              er = [self.ExtendedRegisters[index] for index in compat.xrange(0, MAXIMUM_SUPPORTED_EXTENSION)]
286              er = tuple(er)
287              ctx["ExtendedRegisters"] = er
288          return ctx

421  # BOOL WINAPI GetThreadContext(
422  #   __in     HANDLE hThread,
423  #   __inout  LPCONTEXT lpContext
424  # );
425  def GetThreadContext(hThread, ContextFlags=None, raw=False):
426      _GetThreadContext = windll.kernel32.GetThreadContext
427      _GetThreadContext.argtypes = [HANDLE, LPCONTEXT]
428      _GetThreadContext.restype = bool
429      _GetThreadContext.errcheck = RaiseIfZero
430  
431      if ContextFlags is None:
432          ContextFlags = CONTEXT_ALL | CONTEXT_i386
433      Context = CONTEXT()
434      Context.ContextFlags = ContextFlags
435      _GetThreadContext(hThread, byref(Context))
436      if raw:
437          return Context
438      return Context.to_dict()

443  #   __in  const CONTEXT* lpContext

445  def SetThreadContext(hThread, lpContext):
446      _SetThreadContext = windll.kernel32.SetThreadContext
447      _SetThreadContext.argtypes = [HANDLE, LPCONTEXT]
448      _SetThreadContext.restype = bool
449      _SetThreadContext.errcheck = RaiseIfZero
450  
451      if isinstance(lpContext, dict):
452          lpContext = CONTEXT.from_dict(lpContext)
453      _SetThreadContext(hThread, byref(lpContext))
