35   def dense_block(x, blocks, name):
36       """A dense block. 
37    
38       Args: 
39           x: input tensor. 
40           blocks: integer, the number of building blocks. 
41           name: string, block label. 
42    
43       Returns: 
44           Output tensor for the block. 
45       """
46       for i in range(blocks):
47           x = conv_block(x, 32, name=name + "_block" + str(i + 1))
48       return x

51   def transition_block(x, reduction, name):
52       """A transition block. 
53    
54       Args: 
55           x: input tensor. 
56           reduction: float, compression rate at transition layers. 
57           name: string, block label. 
58    
59       Returns: 
60           Output tensor for the block. 
61       """
62       bn_axis = 3 if backend.image_data_format() == "channels_last" else 1
63       x = layers.BatchNormalization(
64           axis=bn_axis, epsilon=1.001e-5, name=name + "_bn"
65       )(x)
66       x = layers.Activation("relu", name=name + "_relu")(x)
67       x = layers.Conv2D(
68           int(x.shape[bn_axis] * reduction),
69           1,
70           use_bias=False,
71           name=name + "_conv",
72       )(x)
73       x = layers.AveragePooling2D(2, strides=2, name=name + "_pool")(x)
74       return x

77   def conv_block(x, growth_rate, name):
78       """A building block for a dense block. 
79    
80       Args: 
81           x: input tensor. 
82           growth_rate: float, growth rate at dense layers. 
83           name: string, block label. 
84    
85       Returns: 
86           Output tensor for the block. 
87       """
88       bn_axis = 3 if backend.image_data_format() == "channels_last" else 1
89       x1 = layers.BatchNormalization(
90           axis=bn_axis, epsilon=1.001e-5, name=name + "_0_bn"
91       )(x)
92       x1 = layers.Activation("relu", name=name + "_0_relu")(x1)
93       x1 = layers.Conv2D(
94           4 * growth_rate, 1, use_bias=False, name=name + "_1_conv"
95       )(x1)
96       x1 = layers.BatchNormalization(
97           axis=bn_axis, epsilon=1.001e-5, name=name + "_1_bn"
98       )(x1)
99       x1 = layers.Activation("relu", name=name + "_1_relu")(x1)
100      x1 = layers.Conv2D(
101          growth_rate, 3, padding="same", use_bias=False, name=name + "_2_conv"
102      )(x1)
103      x = layers.Concatenate(axis=bn_axis, name=name + "_concat")([x, x1])
104      return x

107  def DenseNet(
108      blocks,
109      include_top=True,
110      weights="imagenet",
111      input_tensor=None,
112      input_shape=None,
113      pooling=None,
114      classes=1000,
115      classifier_activation="softmax",
116      name="densenet",
117  ):
118      """Instantiates the DenseNet architecture. 
119   
120      Reference: 
121      - [Densely Connected Convolutional Networks]( 
122          https://arxiv.org/abs/1608.06993) (CVPR 2017) 
123   
124      This function returns a Keras image classification model, 
125      optionally loaded with weights pre-trained on ImageNet. 
126   
127      For image classification use cases, see 
128      [this page for detailed examples]( 
129        https://keras.io/api/applications/#usage-examples-for-image-classification-models). 
130   
131      For transfer learning use cases, make sure to read the 
132      [guide to transfer learning & fine-tuning]( 
133        https://keras.io/guides/transfer_learning/). 
134   
135      Note: each Keras Application expects a specific kind of input preprocessing. 
136      For DenseNet, call `keras.applications.densenet.preprocess_input` 
137      on your inputs before passing them to the model. 
138      `densenet.preprocess_input` will scale pixels between 0 and 1 and then 
139      will normalize each channel with respect to the ImageNet 
140      dataset statistics. 
141   
142      Args: 
143          blocks: numbers of building blocks for the four dense layers. 
144          include_top: whether to include the fully-connected 
145              layer at the top of the network. 
146          weights: one of `None` (random initialization), 
147              `"imagenet"` (pre-training on ImageNet), 
148              or the path to the weights file to be loaded. 
149          input_tensor: optional Keras tensor 
150              (i.e. output of `layers.Input()`) 
151              to use as image input for the model. 
152          input_shape: optional shape tuple, only to be specified 
153              if `include_top` is False (otherwise the input shape 
154              has to be `(224, 224, 3)` 
155              (with `'channels_last'` data format) 
156              or `(3, 224, 224)` (with `'channels_first'` data format). 
157              It should have exactly 3 inputs channels, 
158              and width and height should be no smaller than 32. 
159              E.g. `(200, 200, 3)` would be one valid value. 
160          pooling: optional pooling mode for feature extraction 
161              when `include_top` is `False`. 
162              - `None` means that the output of the model will be 
163                  the 4D tensor output of the 
164                  last convolutional block. 
165              - `avg` means that global average pooling 
166                  will be applied to the output of the 
167                  last convolutional block, and thus 
168                  the output of the model will be a 2D tensor. 
169              - `max` means that global max pooling will 
170                  be applied. 
171          classes: optional number of classes to classify images 
172              into, only to be specified if `include_top` is `True`, and 
173              if no `weights` argument is specified. Defaults to `1000`. 
174          classifier_activation: A `str` or callable. 
175              The activation function to use 
176              on the "top" layer. Ignored unless `include_top=True`. Set 
177              `classifier_activation=None` to return the logits of the "top" 
178              layer. When loading pretrained weights, `classifier_activation` 
179              can only be `None` or `"softmax"`. 
180          name: The name of the model (string). 
181   
182      Returns: 
183          A model instance. 
184      """
185      if backend.image_data_format() == "channels_first":
186          raise ValueError(
187              "DenseNet does not support the `channels_first` image data "
188              "format. Switch to `channels_last` by editing your local "
189              "config file at ~/.keras/keras.json"
190          )
191      if not (weights in {"imagenet", None} or file_utils.exists(weights)):
192          raise ValueError(
193              "The `weights` argument should be either "
194              "`None` (random initialization), `imagenet` "
195              "(pre-training on ImageNet), "
196              "or the path to the weights file to be loaded."
197          )
198  
199      if weights == "imagenet" and include_top and classes != 1000:
200          raise ValueError(
201              'If using `weights` as `"imagenet"` with `include_top`'
202              " as true, `classes` should be 1000"
203          )
204  
205      # Determine proper input shape
206      input_shape = imagenet_utils.obtain_input_shape(
207          input_shape,
208          default_size=224,
209          min_size=32,
210          data_format=backend.image_data_format(),
211          require_flatten=include_top,
212          weights=weights,
213      )
214  
215      if input_tensor is None:
216          img_input = layers.Input(shape=input_shape)
217      else:
218          if not backend.is_keras_tensor(input_tensor):
219              img_input = layers.Input(tensor=input_tensor, shape=input_shape)
220          else:
221              img_input = input_tensor
222  
223      bn_axis = 3 if backend.image_data_format() == "channels_last" else 1
224  
225      x = layers.ZeroPadding2D(padding=((3, 3), (3, 3)))(img_input)
226      x = layers.Conv2D(64, 7, strides=2, use_bias=False, name="conv1_conv")(x)
227      x = layers.BatchNormalization(
228          axis=bn_axis, epsilon=1.001e-5, name="conv1_bn"
229      )(x)
230      x = layers.Activation("relu", name="conv1_relu")(x)
231      x = layers.ZeroPadding2D(padding=((1, 1), (1, 1)))(x)
232      x = layers.MaxPooling2D(3, strides=2, name="pool1")(x)
233  
234      x = dense_block(x, blocks[0], name="conv2")
235      x = transition_block(x, 0.5, name="pool2")
236      x = dense_block(x, blocks[1], name="conv3")
237      x = transition_block(x, 0.5, name="pool3")
238      x = dense_block(x, blocks[2], name="conv4")
239      x = transition_block(x, 0.5, name="pool4")
240      x = dense_block(x, blocks[3], name="conv5")
241  
242      x = layers.BatchNormalization(axis=bn_axis, epsilon=1.001e-5, name="bn")(x)
243      x = layers.Activation("relu", name="relu")(x)
244  
245      if include_top:
246          x = layers.GlobalAveragePooling2D(name="avg_pool")(x)
247  
248          imagenet_utils.validate_activation(classifier_activation, weights)
249          x = layers.Dense(
250              classes, activation=classifier_activation, name="predictions"
251          )(x)
252      else:
253          if pooling == "avg":
254              x = layers.GlobalAveragePooling2D(name="avg_pool")(x)
255          elif pooling == "max":
256              x = layers.GlobalMaxPooling2D(name="max_pool")(x)
257  
258      # Ensure that the model takes into account
259      # any potential predecessors of `input_tensor`.
260      if input_tensor is not None:
261          inputs = operation_utils.get_source_inputs(input_tensor)
262      else:
263          inputs = img_input
264  
265      # Create model.
266      model = Functional(inputs, x, name=name)
267  
268      # Load weights.
269      if weights == "imagenet":
270          if include_top:
271              if blocks == [6, 12, 24, 16]:
272                  weights_path = file_utils.get_file(
273                      "densenet121_weights_tf_dim_ordering_tf_kernels.h5",
274                      DENSENET121_WEIGHT_PATH,
275                      cache_subdir="models",
276                      file_hash="9d60b8095a5708f2dcce2bca79d332c7",
277                  )
278              elif blocks == [6, 12, 32, 32]:
279                  weights_path = file_utils.get_file(
280                      "densenet169_weights_tf_dim_ordering_tf_kernels.h5",
281                      DENSENET169_WEIGHT_PATH,
282                      cache_subdir="models",
283                      file_hash="d699b8f76981ab1b30698df4c175e90b",
284                  )
285              elif blocks == [6, 12, 48, 32]:
286                  weights_path = file_utils.get_file(
287                      "densenet201_weights_tf_dim_ordering_tf_kernels.h5",
288                      DENSENET201_WEIGHT_PATH,
289                      cache_subdir="models",
290                      file_hash="1ceb130c1ea1b78c3bf6114dbdfd8807",
291                  )
292              else:
293                  raise ValueError("weights_path undefined")
294          else:
295              if blocks == [6, 12, 24, 16]:
296                  weights_path = file_utils.get_file(
297                      "densenet121_weights_tf_dim_ordering_tf_kernels_notop.h5",
298                      DENSENET121_WEIGHT_PATH_NO_TOP,
299                      cache_subdir="models",
300                      file_hash="30ee3e1110167f948a6b9946edeeb738",
301                  )
302              elif blocks == [6, 12, 32, 32]:
303                  weights_path = file_utils.get_file(
304                      "densenet169_weights_tf_dim_ordering_tf_kernels_notop.h5",
305                      DENSENET169_WEIGHT_PATH_NO_TOP,
306                      cache_subdir="models",
307                      file_hash="b8c4d4c20dd625c148057b9ff1c1176b",
308                  )
309              elif blocks == [6, 12, 48, 32]:
310                  weights_path = file_utils.get_file(
311                      "densenet201_weights_tf_dim_ordering_tf_kernels_notop.h5",
312                      DENSENET201_WEIGHT_PATH_NO_TOP,
313                      cache_subdir="models",
314                      file_hash="c13680b51ded0fb44dff2d8f86ac8bb1",
315                  )
316              else:
317                  raise ValueError("weights_path undefined")
318          model.load_weights(weights_path)
319      elif weights is not None:
320          model.load_weights(weights)
321  
322      return model
