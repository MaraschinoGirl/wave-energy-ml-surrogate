35   class RootResults(OptimizeResult):
36       """Represents the root finding result. 
37    
38       Attributes 
39       ---------- 
40       root : float 
41           Estimated root location. 
42       iterations : int 
43           Number of iterations needed to find the root. 
44       function_calls : int 
45           Number of times the function was called. 
46       converged : bool 
47           True if the routine converged. 
48       flag : str 
49           Description of the cause of termination. 
50       method : str 
51           Root finding method used. 
52    
53       """
54   
55       def __init__(self, root, iterations, function_calls, flag, method):
56           self.root = root
57           self.iterations = iterations
58           self.function_calls = function_calls
59           self.converged = flag == _ECONVERGED
60           if flag in flag_map:
61               self.flag = flag_map[flag]
62           else:
63               self.flag = flag
64           self.method = method
65   
66   
67   def results_c(full_output, r, method):
68       if full_output:
69           x, funcalls, iterations, flag = r
70           results = RootResults(root=x,
71                                 iterations=iterations,
72                                 function_calls=funcalls,
73                                 flag=flag, method=method)
74           return x, results
75       else:
76           return r

79   def _results_select(full_output, r, method):
80       """Select from a tuple of (root, funccalls, iterations, flag)"""
81       x, funcalls, iterations, flag = r
82       if full_output:
83           results = RootResults(root=x,
84                                 iterations=iterations,
85                                 function_calls=funcalls,
86                                 flag=flag, method=method)
87           return x, results
88       return x

91   def _wrap_nan_raise(f):
92   
93       def f_raise(x, *args):
94           fx = f(x, *args)
95           f_raise._function_calls += 1
96           if np.isnan(fx):
97               msg = (f'The function value at x={x} is NaN; '
98                      'solver cannot continue.')
99               err = ValueError(msg)
100              err._x = x
101              err._function_calls = f_raise._function_calls
102              raise err
103          return fx
104  
105      f_raise._function_calls = 0
106      return f_raise

109  def newton(func, x0, fprime=None, args=(), tol=1.48e-8, maxiter=50,
110             fprime2=None, x1=None, rtol=0.0,
111             full_output=False, disp=True):
112      """ 
113      Find a root of a real or complex function using the Newton-Raphson 
114      (or secant or Halley's) method. 
115   
116      Find a root of the scalar-valued function `func` given a nearby scalar 
117      starting point `x0`. 
118      The Newton-Raphson method is used if the derivative `fprime` of `func` 
119      is provided, otherwise the secant method is used. If the second order 
120      derivative `fprime2` of `func` is also provided, then Halley's method is 
121      used. 
122   
123      If `x0` is a sequence with more than one item, `newton` returns an array: 
124      the roots of the function from each (scalar) starting point in `x0`. 
125      In this case, `func` must be vectorized to return a sequence or array of 
126      the same shape as its first argument. If `fprime` (`fprime2`) is given, 
127      then its return must also have the same shape: each element is the first 
128      (second) derivative of `func` with respect to its only variable evaluated 
129      at each element of its first argument. 
130   
131      `newton` is for finding roots of a scalar-valued functions of a single 
132      variable. For problems involving several variables, see `root`. 
133   
134      Parameters 
135      ---------- 
136      func : callable 
137          The function whose root is wanted. It must be a function of a 
138          single variable of the form ``f(x,a,b,c...)``, where ``a,b,c...`` 
139          are extra arguments that can be passed in the `args` parameter. 
140      x0 : float, sequence, or ndarray 
141          An initial estimate of the root that should be somewhere near the 
142          actual root. If not scalar, then `func` must be vectorized and return 
143          a sequence or array of the same shape as its first argument. 
144      fprime : callable, optional 
145          The derivative of the function when available and convenient. If it 
146          is None (default), then the secant method is used. 
147      args : tuple, optional 
148          Extra arguments to be used in the function call. 
149      tol : float, optional 
150          The allowable error of the root's value. If `func` is complex-valued, 
151          a larger `tol` is recommended as both the real and imaginary parts 
152          of `x` contribute to ``|x - x0|``. 
153      maxiter : int, optional 
154          Maximum number of iterations. 
155      fprime2 : callable, optional 
156          The second order derivative of the function when available and 
157          convenient. If it is None (default), then the normal Newton-Raphson 
158          or the secant method is used. If it is not None, then Halley's method 
159          is used. 
160      x1 : float, optional 
161          Another estimate of the root that should be somewhere near the 
162          actual root. Used if `fprime` is not provided. 
163      rtol : float, optional 
164          Tolerance (relative) for termination. 
165      full_output : bool, optional 
166          If `full_output` is False (default), the root is returned. 
167          If True and `x0` is scalar, the return value is ``(x, r)``, where ``x`` 
168          is the root and ``r`` is a `RootResults` object. 
169          If True and `x0` is non-scalar, the return value is ``(x, converged, 
170          zero_der)`` (see Returns section for details). 
171      disp : bool, optional 
172          If True, raise a RuntimeError if the algorithm didn't converge, with 
173          the error message containing the number of iterations and current 
174          function value. Otherwise, the convergence status is recorded in a 
175          `RootResults` return object. 
176          Ignored if `x0` is not scalar. 
177          *Note: this has little to do with displaying, however, 
178          the `disp` keyword cannot be renamed for backwards compatibility.* 
179   
180      Returns 
181      ------- 
182      root : float, sequence, or ndarray 
183          Estimated location where function is zero. 
184      r : `RootResults`, optional 
185          Present if ``full_output=True`` and `x0` is scalar. 
186          Object containing information about the convergence. In particular, 
187          ``r.converged`` is True if the routine converged. 
188      converged : ndarray of bool, optional 
189          Present if ``full_output=True`` and `x0` is non-scalar. 
190          For vector functions, indicates which elements converged successfully. 
191      zero_der : ndarray of bool, optional 
192          Present if ``full_output=True`` and `x0` is non-scalar. 
193          For vector functions, indicates which elements had a zero derivative. 
194   
195      See Also 
196      -------- 
197      root_scalar : interface to root solvers for scalar functions 
198      root : interface to root solvers for multi-input, multi-output functions 
199   
200      Notes 
201      ----- 
202      The convergence rate of the Newton-Raphson method is quadratic, 
203      the Halley method is cubic, and the secant method is 
204      sub-quadratic. This means that if the function is well-behaved 
205      the actual error in the estimated root after the nth iteration 
206      is approximately the square (cube for Halley) of the error 
207      after the (n-1)th step. However, the stopping criterion used 
208      here is the step size and there is no guarantee that a root 
209      has been found. Consequently, the result should be verified. 
210      Safer algorithms are brentq, brenth, ridder, and bisect, 
211      but they all require that the root first be bracketed in an 
212      interval where the function changes sign. The brentq algorithm 
213      is recommended for general use in one dimensional problems 
214      when such an interval has been found. 
215   
216      When `newton` is used with arrays, it is best suited for the following 
217      types of problems: 
218   
219      * The initial guesses, `x0`, are all relatively the same distance from 
220        the roots. 
221      * Some or all of the extra arguments, `args`, are also arrays so that a 
222        class of similar problems can be solved together. 
223      * The size of the initial guesses, `x0`, is larger than O(100) elements. 
224        Otherwise, a naive loop may perform as well or better than a vector. 
225   
226      Examples 
227      -------- 
228      >>> import numpy as np 
229      >>> import matplotlib.pyplot as plt 
230      >>> from scipy import optimize 
231   
232      >>> def f(x): 
233      ...     return (x**3 - 1)  # only one real root at x = 1 
234   
235      ``fprime`` is not provided, use the secant method: 
236   
237      >>> root = optimize.newton(f, 1.5) 
238      >>> root 
239      1.0000000000000016 
240      >>> root = optimize.newton(f, 1.5, fprime2=lambda x: 6 * x) 
241      >>> root 
242      1.0000000000000016 
243   
244      Only ``fprime`` is provided, use the Newton-Raphson method: 
245   
246      >>> root = optimize.newton(f, 1.5, fprime=lambda x: 3 * x**2) 
247      >>> root 
248      1.0 
249   
250      Both ``fprime2`` and ``fprime`` are provided, use Halley's method: 
251   
252      >>> root = optimize.newton(f, 1.5, fprime=lambda x: 3 * x**2, 
253      ...                        fprime2=lambda x: 6 * x) 
254      >>> root 
255      1.0 
256   
257      When we want to find roots for a set of related starting values and/or 
258      function parameters, we can provide both of those as an array of inputs: 
259   
260      >>> f = lambda x, a: x**3 - a 
261      >>> fder = lambda x, a: 3 * x**2 
262      >>> rng = np.random.default_rng() 
263      >>> x = rng.standard_normal(100) 
264      >>> a = np.arange(-50, 50) 
265      >>> vec_res = optimize.newton(f, x, fprime=fder, args=(a, ), maxiter=200) 
266   
267      The above is the equivalent of solving for each value in ``(x, a)`` 
268      separately in a for-loop, just faster: 
269   
270      >>> loop_res = [optimize.newton(f, x0, fprime=fder, args=(a0,), 
271      ...                             maxiter=200) 
272      ...             for x0, a0 in zip(x, a)] 
273      >>> np.allclose(vec_res, loop_res) 
274      True 
275   
276      Plot the results found for all values of ``a``: 
277   
278      >>> analytical_result = np.sign(a) * np.abs(a)**(1/3) 
279      >>> fig, ax = plt.subplots() 
280      >>> ax.plot(a, analytical_result, 'o') 
281      >>> ax.plot(a, vec_res, '.') 
282      >>> ax.set_xlabel('$a$') 
283      >>> ax.set_ylabel('$x$ where $f(x, a)=0$') 
284      >>> plt.show() 
285   
286      """
287      if tol <= 0:
288          raise ValueError(f"tol too small ({tol:g} <= 0)")
289      maxiter = operator.index(maxiter)
290      if maxiter < 1:
291          raise ValueError("maxiter must be greater than 0")
292      if np.size(x0) > 1:
293          return _array_newton(func, x0, fprime, args, tol, maxiter, fprime2,
294                               full_output)
295  
296      # Convert to float (don't use float(x0); this works also for complex x0)
297      # Use np.asarray because we want x0 to be a numpy object, not a Python
298      # object. e.g. np.complex(1+1j) > 0 is possible, but (1 + 1j) > 0 raises
299      # a TypeError
300      x0 = np.asarray(x0)[()] * 1.0
301      p0 = x0
302      funcalls = 0
303      if fprime is not None:
304          # Newton-Raphson method
305          method = "newton"
306          for itr in range(maxiter):
307              # first evaluate fval
308              fval = func(p0, *args)
309              funcalls += 1
310              # If fval is 0, a root has been found, then terminate
311              if fval == 0:
312                  return _results_select(
313                      full_output, (p0, funcalls, itr, _ECONVERGED), method)
314              fder = fprime(p0, *args)
315              funcalls += 1
316              if fder == 0:
317                  msg = "Derivative was zero."
318                  if disp:
319                      msg += (
320                          " Failed to converge after %d iterations, value is %s."
321                          % (itr + 1, p0))
322                      raise RuntimeError(msg)
323                  warnings.warn(msg, RuntimeWarning, stacklevel=2)
324                  return _results_select(
325                      full_output, (p0, funcalls, itr + 1, _ECONVERR), method)
326              newton_step = fval / fder
327              if fprime2:
328                  fder2 = fprime2(p0, *args)
329                  funcalls += 1
330                  method = "halley"
331                  # Halley's method:
332                  #   newton_step /= (1.0 - 0.5 * newton_step * fder2 / fder)
333                  # Only do it if denominator stays close enough to 1
334                  # Rationale: If 1-adj < 0, then Halley sends x in the
335                  # opposite direction to Newton. Doesn't happen if x is close
336                  # enough to root.
337                  adj = newton_step * fder2 / fder / 2
338                  if np.abs(adj) < 1:
339                      newton_step /= 1.0 - adj
340              p = p0 - newton_step
341              if np.isclose(p, p0, rtol=rtol, atol=tol):
342                  return _results_select(
343                      full_output, (p, funcalls, itr + 1, _ECONVERGED), method)
344              p0 = p
345      else:
346          # Secant method
347          method = "secant"
348          if x1 is not None:
349              if x1 == x0:
350                  raise ValueError("x1 and x0 must be different")
351              p1 = x1
352          else:
353              eps = 1e-4
354              p1 = x0 * (1 + eps)
355              p1 += (eps if p1 >= 0 else -eps)
356          q0 = func(p0, *args)
357          funcalls += 1
358          q1 = func(p1, *args)
359          funcalls += 1
360          if abs(q1) < abs(q0):
361              p0, p1, q0, q1 = p1, p0, q1, q0
362          for itr in range(maxiter):
363              if q1 == q0:
364                  if p1 != p0:
365                      msg = f"Tolerance of {p1 - p0} reached."
366                      if disp:
367                          msg += (
368                              " Failed to converge after %d iterations, value is %s."
369                              % (itr + 1, p1))
370                          raise RuntimeError(msg)
371                      warnings.warn(msg, RuntimeWarning, stacklevel=2)
372                  p = (p1 + p0) / 2.0
373                  return _results_select(
374                      full_output, (p, funcalls, itr + 1, _ECONVERR), method)
375              else:
376                  if abs(q1) > abs(q0):
377                      p = (-q0 / q1 * p1 + p0) / (1 - q0 / q1)
378                  else:
379                      p = (-q1 / q0 * p0 + p1) / (1 - q1 / q0)
380              if np.isclose(p, p1, rtol=rtol, atol=tol):
381                  return _results_select(
382                      full_output, (p, funcalls, itr + 1, _ECONVERGED), method)
383              p0, q0 = p1, q1
384              p1 = p
385              q1 = func(p1, *args)
386              funcalls += 1
387  
388      if disp:
389          msg = ("Failed to converge after %d iterations, value is %s."
390                 % (itr + 1, p))
391          raise RuntimeError(msg)
392  
393      return _results_select(full_output, (p, funcalls, itr + 1, _ECONVERR), method)

396  def _array_newton(func, x0, fprime, args, tol, maxiter, fprime2, full_output):
397      """ 
398      A vectorized version of Newton, Halley, and secant methods for arrays. 
399   
400      Do not use this method directly. This method is called from `newton` 
401      when ``np.size(x0) > 1`` is ``True``. For docstring, see `newton`. 
402      """
403      # Explicitly copy `x0` as `p` will be modified inplace, but the
404      # user's array should not be altered.
405      p = np.array(x0, copy=True)
406  
407      failures = np.ones_like(p, dtype=bool)
408      nz_der = np.ones_like(failures)
409      if fprime is not None:
410          # Newton-Raphson method
411          for iteration in range(maxiter):
412              # first evaluate fval
413              fval = np.asarray(func(p, *args))
414              # If all fval are 0, all roots have been found, then terminate
415              if not fval.any():
416                  failures = fval.astype(bool)
417                  break
418              fder = np.asarray(fprime(p, *args))
419              nz_der = (fder != 0)
420              # stop iterating if all derivatives are zero
421              if not nz_der.any():
422                  break
423              # Newton step
424              dp = fval[nz_der] / fder[nz_der]
425              if fprime2 is not None:
426                  fder2 = np.asarray(fprime2(p, *args))
427                  dp = dp / (1.0 - 0.5 * dp * fder2[nz_der] / fder[nz_der])
428              # only update nonzero derivatives
429              p = np.asarray(p, dtype=np.result_type(p, dp, np.float64))
430              p[nz_der] -= dp
431              failures[nz_der] = np.abs(dp) >= tol  # items not yet converged
432              # stop iterating if there aren't any failures, not incl zero der
433              if not failures[nz_der].any():
434                  break
435      else:
436          # Secant method
437          dx = np.finfo(float).eps**0.33
438          p1 = p * (1 + dx) + np.where(p >= 0, dx, -dx)
439          q0 = np.asarray(func(p, *args))
440          q1 = np.asarray(func(p1, *args))
441          active = np.ones_like(p, dtype=bool)
442          for iteration in range(maxiter):
443              nz_der = (q1 != q0)
444              # stop iterating if all derivatives are zero
445              if not nz_der.any():
446                  p = (p1 + p) / 2.0
447                  break
448              # Secant Step
449              dp = (q1 * (p1 - p))[nz_der] / (q1 - q0)[nz_der]
450              # only update nonzero derivatives
451              p = np.asarray(p, dtype=np.result_type(p, p1, dp, np.float64))
452              p[nz_der] = p1[nz_der] - dp
453              active_zero_der = ~nz_der & active
454              p[active_zero_der] = (p1 + p)[active_zero_der] / 2.0
455              active &= nz_der  # don't assign zero derivatives again
456              failures[nz_der] = np.abs(dp) >= tol  # not yet converged
457              # stop iterating if there aren't any failures, not incl zero der
458              if not failures[nz_der].any():
459                  break
460              p1, p = p, p1
461              q0 = q1
462              q1 = np.asarray(func(p1, *args))
463  
464      zero_der = ~nz_der & failures  # don't include converged with zero-ders
465      if zero_der.any():
466          # Secant warnings
467          if fprime is None:
468              nonzero_dp = (p1 != p)
469              # non-zero dp, but infinite newton step
470              zero_der_nz_dp = (zero_der & nonzero_dp)
471              if zero_der_nz_dp.any():
472                  rms = np.sqrt(
473                      sum((p1[zero_der_nz_dp] - p[zero_der_nz_dp]) ** 2)
474                  )
475                  warnings.warn(f'RMS of {rms:g} reached', RuntimeWarning, stacklevel=3)
476          # Newton or Halley warnings
477          else:
478              all_or_some = 'all' if zero_der.all() else 'some'
479              msg = f'{all_or_some:s} derivatives were zero'
480              warnings.warn(msg, RuntimeWarning, stacklevel=3)
481      elif failures.any():
482          all_or_some = 'all' if failures.all() else 'some'
483          msg = f'{all_or_some:s} failed to converge after {maxiter:d} iterations'
484          if failures.all():
485              raise RuntimeError(msg)
486          warnings.warn(msg, RuntimeWarning, stacklevel=3)
487  
488      if full_output:
489          result = namedtuple('result', ('root', 'converged', 'zero_der'))
490          p = result(p, ~failures, zero_der)
491  
492      return p

495  def bisect(f, a, b, args=(),
496             xtol=_xtol, rtol=_rtol, maxiter=_iter,
497             full_output=False, disp=True):
498      """ 
499      Find root of a function within an interval using bisection. 
500   
501      Basic bisection routine to find a root of the function `f` between the 
502      arguments `a` and `b`. `f(a)` and `f(b)` cannot have the same signs. 
503      Slow but sure. 
504   
505      Parameters 
506      ---------- 
507      f : function 
508          Python function returning a number.  `f` must be continuous, and 
509          f(a) and f(b) must have opposite signs. 
510      a : scalar 
511          One end of the bracketing interval [a,b]. 
512      b : scalar 
513          The other end of the bracketing interval [a,b]. 
514      xtol : number, optional 
515          The computed root ``x0`` will satisfy ``np.allclose(x, x0, 
516          atol=xtol, rtol=rtol)``, where ``x`` is the exact root. The 
517          parameter must be positive. 
518      rtol : number, optional 
519          The computed root ``x0`` will satisfy ``np.allclose(x, x0, 
520          atol=xtol, rtol=rtol)``, where ``x`` is the exact root. The 
521          parameter cannot be smaller than its default value of 
522          ``4*np.finfo(float).eps``. 
523      maxiter : int, optional 
524          If convergence is not achieved in `maxiter` iterations, an error is 
525          raised. Must be >= 0. 
526      args : tuple, optional 
527          Containing extra arguments for the function `f`. 
528          `f` is called by ``apply(f, (x)+args)``. 
529      full_output : bool, optional 
530          If `full_output` is False, the root is returned. If `full_output` is 
531          True, the return value is ``(x, r)``, where x is the root, and r is 
532          a `RootResults` object. 
533      disp : bool, optional 
534          If True, raise RuntimeError if the algorithm didn't converge. 
535          Otherwise, the convergence status is recorded in a `RootResults` 
536          return object. 
537   
538      Returns 
539      ------- 
540      root : float 
541          Root of `f` between `a` and `b`. 
542      r : `RootResults` (present if ``full_output = True``) 
543          Object containing information about the convergence. In particular, 
544          ``r.converged`` is True if the routine converged. 
545   
546      Examples 
547      -------- 
548   
549      >>> def f(x): 
550      ...     return (x**2 - 1) 
551   
552      >>> from scipy import optimize 
553   
554      >>> root = optimize.bisect(f, 0, 2) 
555      >>> root 
556      1.0 
557   
558      >>> root = optimize.bisect(f, -2, 0) 
559      >>> root 
560      -1.0 
561   
562      See Also 
563      -------- 
564      brentq, brenth, bisect, newton 
565      fixed_point : scalar fixed-point finder 
566      fsolve : n-dimensional root-finding 
567   
568      """
569      if not isinstance(args, tuple):
570          args = (args,)
571      maxiter = operator.index(maxiter)
572      if xtol <= 0:
573          raise ValueError(f"xtol too small ({xtol:g} <= 0)")
574      if rtol < _rtol:
575          raise ValueError(f"rtol too small ({rtol:g} < {_rtol:g})")
576      f = _wrap_nan_raise(f)
577      r = _zeros._bisect(f, a, b, xtol, rtol, maxiter, args, full_output, disp)
578      return results_c(full_output, r, "bisect")

581  def ridder(f, a, b, args=(),
582             xtol=_xtol, rtol=_rtol, maxiter=_iter,
583             full_output=False, disp=True):
584      """ 
585      Find a root of a function in an interval using Ridder's method. 
586   
587      Parameters 
588      ---------- 
589      f : function 
590          Python function returning a number. f must be continuous, and f(a) and 
591          f(b) must have opposite signs. 
592      a : scalar 
593          One end of the bracketing interval [a,b]. 
594      b : scalar 
595          The other end of the bracketing interval [a,b]. 
596      xtol : number, optional 
597          The computed root ``x0`` will satisfy ``np.allclose(x, x0, 
598          atol=xtol, rtol=rtol)``, where ``x`` is the exact root. The 
599          parameter must be positive. 
600      rtol : number, optional

679  def brentq(f, a, b, args=(),

687      rootfinding routines here. It is a safe version of the secant method that 
688      uses inverse quadratic extrapolation. Brent's method combines root 
689      bracketing, interval bisection, and inverse quadratic interpolation. It is 
690      sometimes known as the van Wijngaarden-Dekker-Brent method. Brent (1973) 
691      claims convergence is guaranteed for functions computable within [a,b]. 
692   
693      [Brent1973]_ provides the classic description of the algorithm. Another 
694      description can be found in a recent edition of Numerical Recipes, including 
695      [PressEtal1992]_. A third description is at

699      extrapolation step.

716          and ``rtol/2``. [Brent1973]_

718          The computed root ``x0`` will satisfy ``np.allclose(x, x0, 
719          atol=xtol, rtol=rtol)``, where ``x`` is the exact root. The

721          ``4*np.finfo(float).eps``. For nice functions, Brent's

921      notclosefvals = (

928  def _secant(xvals, fvals):
929      """Perform a secant step, taking a little care"""
930      # Secant has many "mathematically" equivalent formulations
931      # x2 = x0 - (x1 - x0)/(f1 - f0) * f0
932      #    = x1 - (x1 - x0)/(f1 - f0) * f1
933      #    = (-x1 * f0 + x0 * f1) / (f1 - f0)
934      #    = (-f0 / f1 * x1 + x0) / (1 - f0 / f1)
935      #    = (-f1 / f0 * x0 + x1) / (1 - f1 / f0)
936      x0, x1 = xvals[:2]
937      f0, f1 = fvals[:2]
938      if f0 == f1:
939          return np.nan
940      if np.abs(f1) > np.abs(f0):
941          x2 = (-f0 / f1 * x1 + x0) / (1 - f0 / f1)
942      else:
943          x2 = (-f1 / f0 * x0 + x1) / (1 - f1 / f0)
944      return x2

947  def _update_bracket(ab, fab, c, fc):
948      """Update a bracket given (c, fc), return the discarded endpoints."""
949      fa, fb = fab
950      idx = (0 if np.sign(fa) * np.sign(fc) > 0 else 1)
951      rx, rfx = ab[idx], fab[idx]
952      fab[idx] = fc
953      ab[idx] = c
954      return rx, rfx

957  def _compute_divided_differences(xvals, fvals, N=None, full=True,
958                                   forward=True):
959      """Return a matrix of divided differences for the xvals, fvals pairs 
960   
961      DD[i, j] = f[x_{i-j}, ..., x_i] for 0 <= j <= i 
962   
963      If full is False, just return the main diagonal(or last row): 
964        f[a], f[a, b] and f[a, b, c]. 
965      If forward is False, return f[c], f[b, c], f[a, b, c]."""
966      if full:
967          if forward:
968              xvals = np.asarray(xvals)
969          else:
970              xvals = np.array(xvals)[::-1]
971          M = len(xvals)
972          N = M if N is None else min(N, M)
973          DD = np.zeros([M, N])
974          DD[:, 0] = fvals[:]
975          for i in range(1, N):
976              DD[i:, i] = (np.diff(DD[i - 1:, i - 1]) /
977                           (xvals[i:] - xvals[:M - i]))
978          return DD
979  
980      xvals = np.asarray(xvals)
981      dd = np.array(fvals)
982      row = np.array(fvals)
983      idx2Use = (0 if forward else -1)
984      dd[0] = fvals[idx2Use]
985      for i in range(1, len(xvals)):
986          denom = xvals[i:i + len(row) - 1] - xvals[:len(row) - 1]
987          row = np.diff(row)[:] / denom
988          dd[i] = row[idx2Use]
989      return dd

992  def _interpolated_poly(xvals, fvals, x):
993      """Compute p(x) for the polynomial passing through the specified locations. 
994   
995      Use Neville's algorithm to compute p(x) where p is the minimal degree 
996      polynomial passing through the points xvals, fvals"""
997      xvals = np.asarray(xvals)
998      N = len(xvals)
999      Q = np.zeros([N, N])
1000     D = np.zeros([N, N])
1001     Q[:, 0] = fvals[:]
1002     D[:, 0] = fvals[:]
1003     for k in range(1, N):
1004         alpha = D[k:, k - 1] - Q[k - 1:N - 1, k - 1]
1005         diffik = xvals[0:N - k] - xvals[k:N]
1006         Q[k:, k] = (xvals[k:] - x) / diffik * alpha
1007         D[k:, k] = (xvals[:N - k] - x) / diffik * alpha
1008     # Expect Q[-1, 1:] to be small relative to Q[-1, 0] as x approaches a root
1009     return np.sum(Q[-1, 1:]) + Q[-1, 0]

1022 def _newton_quadratic(ab, fab, d, fd, k):
1023     """Apply Newton-Raphson like steps, using divided differences to approximate f' 
1024  
1025     ab is a real interval [a, b] containing a root, 
1026     fab holds the real values of f(a), f(b) 
1027     d is a real number outside [ab, b] 
1028     k is the number of steps to apply 
1029     """
1030     a, b = ab
1031     fa, fb = fab
1032     _, B, A = _compute_divided_differences([a, b, d], [fa, fb, fd],
1033                                            forward=True, full=False)
1034 
1035     # _P  is the quadratic polynomial through the 3 points
1036     def _P(x):
1037         # Horner evaluation of fa + B * (x - a) + A * (x - a) * (x - b)
1038         return (A * (x - b) + B) * (x - a) + fa
1039 
1040     if A == 0:
1041         r = a - fa / B
1042     else:
1043         r = (a if np.sign(A) * np.sign(fa) > 0 else b)
1044         # Apply k Newton-Raphson steps to _P(x), starting from x=r
1045         for i in range(k):
1046             r1 = r - _P(r) / (B + A * (2 * r - a - b))
1047             if not (ab[0] < r1 < ab[1]):
1048                 if (ab[0] < r < ab[1]):
1049                     return r
1050                 r = sum(ab) / 2.0
1051                 break
1052             r = r1
1053 
1054     return r

1057 class TOMS748Solver:
1058     """Solve f(x, *args) == 0 using Algorithm748 of Alefeld, Potro & Shi. 
1059     """
1060     _MU = 0.5
1061     _K_MIN = 1
1062     _K_MAX = 100  # A very high value for real usage. Expect 1, 2, maybe 3.
1063 
1064     def __init__(self):
1065         self.f = None
1066         self.args = None
1067         self.function_calls = 0
1068         self.iterations = 0
1069         self.k = 2
1070         # ab=[a,b] is a global interval containing a root
1071         self.ab = [np.nan, np.nan]
1072         # fab is function values at a, b
1073         self.fab = [np.nan, np.nan]
1074         self.d = None
1075         self.fd = None
1076         self.e = None
1077         self.fe = None
1078         self.disp = False
1079         self.xtol = _xtol
1080         self.rtol = _rtol
1081         self.maxiter = _iter
1082 
1083     def configure(self, xtol, rtol, maxiter, disp, k):
1084         self.disp = disp
1085         self.xtol = xtol
1086         self.rtol = rtol
1087         self.maxiter = maxiter
1088         # Silently replace a low value of k with 1
1089         self.k = max(k, self._K_MIN)
1090         # Noisily replace a high value of k with self._K_MAX
1091         if self.k > self._K_MAX:
1092             msg = "toms748: Overriding k: ->%d" % self._K_MAX
1093             warnings.warn(msg, RuntimeWarning, stacklevel=3)
1094             self.k = self._K_MAX
1095 
1096     def _callf(self, x, error=True):
1097         """Call the user-supplied function, update book-keeping"""
1098         fx = self.f(x, *self.args)
1099         self.function_calls += 1
1100         if not np.isfinite(fx) and error:
1101             raise ValueError(f"Invalid function value: f({x:f}) -> {fx} ")
1102         return fx
1103 
1104     def get_result(self, x, flag=_ECONVERGED):
1105         r"""Package the result and statistics into a tuple."""
1106         return (x, self.function_calls, self.iterations, flag)
1107 
1108     def _update_bracket(self, c, fc):
1109         return _update_bracket(self.ab, self.fab, c, fc)

1111     def start(self, f, a, b, args=()):
1112         r"""Prepare for the iterations."""
1113         self.function_calls = 0
1114         self.iterations = 0
1115 
1116         self.f = f
1117         self.args = args
1118         self.ab[:] = [a, b]
1119         if not np.isfinite(a) or np.imag(a) != 0:
1120             raise ValueError(f"Invalid x value: {a} ")
1121         if not np.isfinite(b) or np.imag(b) != 0:
1122             raise ValueError(f"Invalid x value: {b} ")
1123 
1124         fa = self._callf(a)
1125         if not np.isfinite(fa) or np.imag(fa) != 0:
1126             raise ValueError(f"Invalid function value: f({a:f}) -> {fa} ")
1127         if fa == 0:
1128             return _ECONVERGED, a
1129         fb = self._callf(b)
1130         if not np.isfinite(fb) or np.imag(fb) != 0:
1131             raise ValueError(f"Invalid function value: f({b:f}) -> {fb} ")
1132         if fb == 0:
1133             return _ECONVERGED, b
1134 
1135         if np.sign(fb) * np.sign(fa) > 0:
1136             raise ValueError("f(a) and f(b) must have different signs, but "
1137                              f"f({a:e})={fa:e}, f({b:e})={fb:e} ")
1138         self.fab[:] = [fa, fb]
1139 
1140         return _EINPROGRESS, sum(self.ab) / 2.0
1141 
1142     def get_status(self):
1143         """Determine the current status."""
1144         a, b = self.ab[:2]
1145         if np.isclose(a, b, rtol=self.rtol, atol=self.xtol):
1146             return _ECONVERGED, sum(self.ab) / 2.0
1147         if self.iterations >= self.maxiter:
1148             return _ECONVERR, sum(self.ab) / 2.0
1149         return _EINPROGRESS, sum(self.ab) / 2.0

1151     def iterate(self):
1152         """Perform one step in the algorithm. 
1153  
1154         Implements Algorithm 4.1(k=1) or 4.2(k=2) in [APS1995] 
1155         """
1156         self.iterations += 1
1157         eps = np.finfo(float).eps
1158         d, fd, e, fe = self.d, self.fd, self.e, self.fe
1159         ab_width = self.ab[1] - self.ab[0]  # Need the start width below
1160         c = None
1161 
1162         for nsteps in range(2, self.k+2):
1163             # If the f-values are sufficiently separated, perform an inverse
1164             # polynomial interpolation step. Otherwise, nsteps repeats of
1165             # an approximate Newton-Raphson step.
1166             if _notclose(self.fab + [fd, fe], rtol=0, atol=32*eps):
1167                 c0 = _inverse_poly_zero(self.ab[0], self.ab[1], d, e,
1168                                         self.fab[0], self.fab[1], fd, fe)
1169                 if self.ab[0] < c0 < self.ab[1]:
1170                     c = c0
1171             if c is None:
1172                 c = _newton_quadratic(self.ab, self.fab, d, fd, nsteps)
1173 
1174             fc = self._callf(c)
1175             if fc == 0:
1176                 return _ECONVERGED, c
1177 
1178             # re-bracket
1179             e, fe = d, fd
1180             d, fd = self._update_bracket(c, fc)
1181 
1182         # u is the endpoint with the smallest f-value
1183         uix = (0 if np.abs(self.fab[0]) < np.abs(self.fab[1]) else 1)
1184         u, fu = self.ab[uix], self.fab[uix]
1185 
1186         _, A = _compute_divided_differences(self.ab, self.fab,
1187                                             forward=(uix == 0), full=False)
1188         c = u - 2 * fu / A
1189         if np.abs(c - u) > 0.5 * (self.ab[1] - self.ab[0]):
1190             c = sum(self.ab) / 2.0
1191         else:
1192             if np.isclose(c, u, rtol=eps, atol=0):
1193                 # c didn't change (much).
1194                 # Either because the f-values at the endpoints have vastly
1195                 # differing magnitudes, or because the root is very close to
1196                 # that endpoint
1197                 frs = np.frexp(self.fab)[1]
1198                 if frs[uix] < frs[1 - uix] - 50:  # Differ by more than 2**50
1199                     c = (31 * self.ab[uix] + self.ab[1 - uix]) / 32
1200                 else:
1201                     # Make a bigger adjustment, about the
1202                     # size of the requested tolerance.
1203                     mm = (1 if uix == 0 else -1)
1204                     adj = mm * np.abs(c) * self.rtol + mm * self.xtol
1205                     c = u + adj
1206                 if not self.ab[0] < c < self.ab[1]:
1207                     c = sum(self.ab) / 2.0
1208 
1209         fc = self._callf(c)
1210         if fc == 0:
1211             return _ECONVERGED, c
1212 
1213         e, fe = d, fd
1214         d, fd = self._update_bracket(c, fc)
1215 
1216         # If the width of the new interval did not decrease enough, bisect
1217         if self.ab[1] - self.ab[0] > self._MU * ab_width:
1218             e, fe = d, fd
1219             z = sum(self.ab) / 2.0
1220             fz = self._callf(z)
1221             if fz == 0:
1222                 return _ECONVERGED, z
1223             d, fd = self._update_bracket(z, fz)
1224 
1225         # Record d and e for next iteration
1226         self.d, self.fd = d, fd
1227         self.e, self.fe = e, fe
1228 
1229         status, xn = self.get_status()
1230         return status, xn
1231 
1232     def solve(self, f, a, b, args=(),
1233               xtol=_xtol, rtol=_rtol, k=2, maxiter=_iter, disp=True):
1234         r"""Solve f(x) = 0 given an interval containing a root."""
1235         self.configure(xtol=xtol, rtol=rtol, maxiter=maxiter, disp=disp, k=k)
1236         status, xn = self.start(f, a, b, args)
1237         if status == _ECONVERGED:
1238             return self.get_result(xn)
1239 
1240         # The first step only has two x-values.
1241         c = _secant(self.ab, self.fab)
1242         if not self.ab[0] < c < self.ab[1]:
1243             c = sum(self.ab) / 2.0
1244         fc = self._callf(c)
1245         if fc == 0:
1246             return self.get_result(c)
1247 
1248         self.d, self.fd = self._update_bracket(c, fc)
1249         self.e, self.fe = None, None
1250         self.iterations += 1
1251 
1252         while True:
1253             status, xn = self.iterate()
1254             if status == _ECONVERGED:
1255                 return self.get_result(xn)
1256             if status == _ECONVERR:
1257                 fmt = "Failed to converge after %d iterations, bracket is %s"
1258                 if disp:
1259                     msg = fmt % (self.iterations + 1, self.ab)
1260                     raise RuntimeError(msg)
1261                 return self.get_result(xn, _ECONVERR)

1264 def toms748(f, a, b, args=(), k=1,
1265             xtol=_xtol, rtol=_rtol, maxiter=_iter,
1266             full_output=False, disp=True):
1267     """ 
1268     Find a root using TOMS Algorithm 748 method. 
1269  
1270     Implements the Algorithm 748 method of Alefeld, Potro and Shi to find a 
1271     root of the function `f` on the interval ``[a , b]``, where ``f(a)`` and 
1272     `f(b)` must have opposite signs. 
1273  
1274     It uses a mixture of inverse cubic interpolation and 
1275     "Newton-quadratic" steps. [APS1995]. 
1276  
1277     Parameters 
1278     ---------- 
1279     f : function 
1280         Python function returning a scalar. The function :math:`f` 
1281         must be continuous, and :math:`f(a)` and :math:`f(b)` 
1282         have opposite signs. 
1283     a : scalar, 
1284         lower boundary of the search interval 
1285     b : scalar, 
1286         upper boundary of the search interval 
1287     args : tuple, optional 
1288         containing extra arguments for the function `f`. 
1289         `f` is called by ``f(x, *args)``. 
1290     k : int, optional 
1291         The number of Newton quadratic steps to perform each 
1292         iteration. ``k>=1``. 
1293     xtol : scalar, optional 
1294         The computed root ``x0`` will satisfy ``np.allclose(x, x0, 
1295         atol=xtol, rtol=rtol)``, where ``x`` is the exact root. The 
1296         parameter must be positive. 
1297     rtol : scalar, optional 
1298         The computed root ``x0`` will satisfy ``np.allclose(x, x0, 
1299         atol=xtol, rtol=rtol)``, where ``x`` is the exact root. 
1300     maxiter : int, optional 
1301         If convergence is not achieved in `maxiter` iterations, an error is 
1302         raised. Must be >= 0. 
1303     full_output : bool, optional 
1304         If `full_output` is False, the root is returned. If `full_output` is 
1305         True, the return value is ``(x, r)``, where `x` is the root, and `r` is 
1306         a `RootResults` object. 
1307     disp : bool, optional 
1308         If True, raise RuntimeError if the algorithm didn't converge. 
1309         Otherwise, the convergence status is recorded in the `RootResults` 
1310         return object. 
1311  
1312     Returns 
1313     ------- 
1314     root : float 
1315         Approximate root of `f` 
1316     r : `RootResults` (present if ``full_output = True``) 
1317         Object containing information about the convergence. In particular, 
1318         ``r.converged`` is True if the routine converged. 
1319  
1320     See Also 
1321     -------- 
1322     brentq, brenth, ridder, bisect, newton 
1323     fsolve : find roots in N dimensions. 
1324  
1325     Notes 
1326     ----- 
1327     `f` must be continuous. 
1328     Algorithm 748 with ``k=2`` is asymptotically the most efficient 
1329     algorithm known for finding roots of a four times continuously 
1330     differentiable function. 
1331     In contrast with Brent's algorithm, which may only decrease the length of 
1332     the enclosing bracket on the last step, Algorithm 748 decreases it each 
1333     iteration with the same asymptotic efficiency as it finds the root. 
1334  
1335     For easy statement of efficiency indices, assume that `f` has 4 
1336     continuous deriviatives. 
1337     For ``k=1``, the convergence order is at least 2.7, and with about 
1338     asymptotically 2 function evaluations per iteration, the efficiency 
1339     index is approximately 1.65. 
1340     For ``k=2``, the order is about 4.6 with asymptotically 3 function 
1341     evaluations per iteration, and the efficiency index 1.66. 
1342     For higher values of `k`, the efficiency index approaches 
1343     the kth root of ``(3k-2)``, hence ``k=1`` or ``k=2`` are 
1344     usually appropriate. 
1345  
1346     References 
1347     ---------- 
1348     .. [APS1995] 
1349        Alefeld, G. E. and Potra, F. A. and Shi, Yixun, 
1350        *Algorithm 748: Enclosing Zeros of Continuous Functions*, 
1351        ACM Trans. Math. Softw. Volume 221(1995) 
1352        doi = {10.1145/210089.210111} 
1353  
1354     Examples 
1355     -------- 
1356     >>> def f(x): 
1357     ...     return (x**3 - 1)  # only one real root at x = 1 
1358  
1359     >>> from scipy import optimize 
1360     >>> root, results = optimize.toms748(f, 0, 2, full_output=True) 
1361     >>> root 
1362     1.0 
1363     >>> results 
1364           converged: True 
1365                flag: converged 
1366      function_calls: 11 
1367          iterations: 5 
1368                root: 1.0 
1369              method: toms748 
1370     """
1371     if xtol <= 0:
1372         raise ValueError(f"xtol too small ({xtol:g} <= 0)")
1373     if rtol < _rtol / 4:
1374         raise ValueError(f"rtol too small ({rtol:g} < {_rtol/4:g})")
1375     maxiter = operator.index(maxiter)
1376     if maxiter < 1:
1377         raise ValueError("maxiter must be greater than 0")
1378     if not np.isfinite(a):
1379         raise ValueError(f"a is not finite {a}")
1380     if not np.isfinite(b):
1381         raise ValueError(f"b is not finite {b}")
1382     if a >= b:
1383         raise ValueError(f"a and b are not an interval [{a}, {b}]")
1384     if not k >= 1:
1385         raise ValueError(f"k too small ({k} < 1)")
1386 
1387     if not isinstance(args, tuple):
1388         args = (args,)
1389     f = _wrap_nan_raise(f)
1390     solver = TOMS748Solver()
1391     result = solver.solve(f, a, b, args=args, k=k, xtol=xtol, rtol=rtol,
1392                           maxiter=maxiter, disp=disp)
1393     x, function_calls, iterations, flag = result
1394     return _results_select(full_output, (x, function_calls, iterations, flag),
1395                            "toms748")
