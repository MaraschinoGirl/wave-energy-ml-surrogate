22   # This class allows running Tcl code. Tkinter uses it internally a lot, and
23   # it's often handy to drop a piece of Tcl code into a tkinter program. Example:
24   #
25   #    >>> import tkinter, _tkinter
26   #    >>> tkapp = tkinter.Tk().tk
27   #    >>> isinstance(tkapp, _tkinter.TkappType)
28   #    True
29   #    >>> tkapp.call('set', 'foo', (1,2,3))
30   #    (1, 2, 3)
31   #    >>> tkapp.eval('return $foo')
32   #    '1 2 3'
33   #    >>>
34   #
