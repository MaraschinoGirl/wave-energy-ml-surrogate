26   // NOTE: BUFSIZE must be the same from add_code_to_python_process.py
27   #define BUFSIZE 2048
28   
29   // Helper to free data when we leave the scope.
30   class DataToFree {
31   
32   public:
33   
34       HANDLE hMapFile;
35       void* mapViewOfFile;
36       char* codeToRun;
37   
38       DataToFree() {
39           this->hMapFile = nullptr;
40           this->mapViewOfFile = nullptr;
41           this->codeToRun = nullptr;
42       }
43   
44       ~DataToFree() {
45           if (this->hMapFile != nullptr) {
46               CloseHandle(this->hMapFile);
47               this->hMapFile = nullptr;
48           }
49           if (this->mapViewOfFile != nullptr) {
50               UnmapViewOfFile(this->mapViewOfFile);
51               this->mapViewOfFile = nullptr;
52           }
53           if (this->codeToRun != nullptr) {
54               delete this->codeToRun;
55               this->codeToRun = nullptr;
56           }
57       }
58   };
59   
60   
61   
62   extern "C"
63   {
64       /** 
65        * This method will read the code to be executed from the named shared memory 
66        * and execute it. 
67        */
68       DECLDIR int RunCodeInMemoryInAttachedDll() {
69           // PRINT("Attempting to run Python code from named shared memory.")
70           //get the code to be run (based on https://docs.microsoft.com/en-us/windows/win32/memory/creating-named-shared-memory).
71           HANDLE hMapFile;
72           char* mapViewOfFile;
73   
74           DataToFree dataToFree;
75   
76           std::string namedSharedMemoryName("__pydevd_pid_code_to_run__");
77           namedSharedMemoryName += std::to_string(GetCurrentProcessId());
78   
79           hMapFile = OpenFileMappingA(
80               FILE_MAP_ALL_ACCESS, // read/write access
81               FALSE, // do not inherit the name
82               namedSharedMemoryName.c_str()); // name of mapping object
83   
84           if (hMapFile == nullptr) {
85               std::cout << "Error opening named shared memory (OpenFileMapping): " << GetLastError() + " name: " << namedSharedMemoryName << std::endl;
86               return 1;
87           } else {
88               // PRINT("Opened named shared memory.")
89           }
90   
91           dataToFree.hMapFile = hMapFile;
92   
93           mapViewOfFile = reinterpret_cast < char* > (MapViewOfFile(hMapFile, // handle to map object
94               FILE_MAP_ALL_ACCESS, // read/write permission
95               0,
96               0,
97               BUFSIZE));
98   
99           if (mapViewOfFile == nullptr) {
100              std::cout << "Error mapping view of named shared memory (MapViewOfFile): " << GetLastError() << std::endl;
101              return 1;
102          } else {
103              // PRINT("Mapped view of file.")
104          }
105          dataToFree.mapViewOfFile = mapViewOfFile;
106          // std::cout << "Will run contents: " << mapViewOfFile << std::endl;
107  
108          dataToFree.codeToRun = new char[BUFSIZE];
109          memmove(dataToFree.codeToRun, mapViewOfFile, BUFSIZE);
110  
111          int attachInfo = 0;
112          return AttachAndRunPythonCode(dataToFree.codeToRun, &attachInfo);
113      }
