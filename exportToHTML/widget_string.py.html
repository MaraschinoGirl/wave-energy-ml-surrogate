53   class _String(DescriptionWidget, ValueWidget, CoreWidget):
54       """Base class used to create widgets that represent a string."""
55   
56       value = Unicode(help="String value").tag(sync=True)
57   
58       # We set a zero-width space as a default placeholder to make sure the baseline matches
59       # the text, not the bottom margin. See the last paragraph of
60       # https://www.w3.org/TR/CSS2/visudet.html#leading
61       placeholder = Unicode('\u200b', help="Placeholder text to display when nothing has been typed").tag(sync=True)
62       style = InstanceDict(_StringStyle).tag(sync=True, **widget_serialization)
63   
64       def __init__(self, value=None, **kwargs):
65           if value is not None:
66               kwargs['value'] = value
67           super().__init__(**kwargs)
68   
69       _model_name = Unicode('StringModel').tag(sync=True)

108  @register
109  class Text(_String):
110      """Single line textbox widget."""
111      _view_name = Unicode('TextView').tag(sync=True)
112      _model_name = Unicode('TextModel').tag(sync=True)
113      disabled = Bool(False, help="Enable or disable user changes").tag(sync=True)
114      continuous_update = Bool(True, help="Update the value as the user types. If False, update on submission, e.g., pressing Enter or navigating away.").tag(sync=True)
115      style = InstanceDict(TextStyle).tag(sync=True, **widget_serialization)
116  
117      def __init__(self, *args, **kwargs):
118          super().__init__(*args, **kwargs)
119          self._submission_callbacks = CallbackDispatcher()
120          self.on_msg(self._handle_string_msg)
121  
122      def _handle_string_msg(self, _, content, buffers):
123          """Handle a msg from the front-end. 
124   
125          Parameters 
126          ---------- 
127          content: dict 
128              Content of the msg. 
129          """
130          if content.get('event', '') == 'submit':
131              self._submission_callbacks(self)
132  
133      def on_submit(self, callback, remove=False):
134          """(Un)Register a callback to handle text submission. 
135   
136          Triggered when the user clicks enter. 
137   
138          Parameters 
139          ---------- 
140          callback: callable 
141              Will be called with exactly one argument: the Widget instance 
142          remove: bool (optional) 
143              Whether to unregister the callback 
144          """
145          deprecation("on_submit is deprecated. Instead, set the .continuous_update attribute to False and observe the value changing with: mywidget.observe(callback, 'value').")
146          self._submission_callbacks.register_callback(callback, remove=remove)

149  @register
150  class Password(Text):
151      """Single line textbox widget."""
152      _view_name = Unicode('PasswordView').tag(sync=True)
153      _model_name = Unicode('PasswordModel').tag(sync=True)
154      disabled = Bool(False, help="Enable or disable user changes").tag(sync=True)
155  
156      def _repr_keys(self):
157          # Don't include password value in repr!
158          super_keys = super()._repr_keys()
159          for key in super_keys:
160              if key != 'value':
161                  yield key
