14   def flatten_test_suite(test_suite, ret):
15       if isinstance(test_suite, unittest.TestSuite):
16           for t in test_suite._tests:
17               flatten_test_suite(t, ret)
18   
19       elif isinstance(test_suite, unittest.TestCase):
20           ret.append(test_suite)

24   # execute_tests_in_parallel
25   # =======================================================================================================================
26   def execute_tests_in_parallel(tests, jobs, split, verbosity, coverage_files, coverage_include):
27       """ 
28       @param tests: list(PydevTestSuite) 
29           A list with the suites to be run 
30    
31       @param split: str 
32           Either 'module' or the number of tests that should be run in each batch 
33    
34       @param coverage_files: list(file) 
35           A list with the files that should be used for giving coverage information (if empty, coverage information 
36           should not be gathered). 
37    
38       @param coverage_include: str 
39           The pattern that should be included in the coverage. 
40    
41       @return: bool 
42           Returns True if the tests were actually executed in parallel. If the tests were not executed because only 1 
43           should be used (e.g.: 2 jobs were requested for running 1 test), False will be returned and no tests will be 
44           run. 
45    
46           It may also return False if in debug mode (in which case, multi-processes are not accepted) 
47       """
48       try:
49           from _pydevd_bundle.pydevd_comm import get_global_debugger
50   
51           if get_global_debugger() is not None:
52               return False
53       except:
54           pass  # Ignore any error here.
55   
56       # This queue will receive the tests to be run. Each entry in a queue is a list with the tests to be run together When
57       # split == 'tests', each list will have a single element, when split == 'module', each list will have all the tests
58       # from a given module.
59       tests_queue = []
60   
61       queue_elements = []
62       if split == "module":
63           module_to_tests = {}
64           for test in tests:
65               lst = []
66               flatten_test_suite(test, lst)
67               for test in lst:
68                   key = (test.__pydev_pyfile__, test.__pydev_module_name__)
69                   module_to_tests.setdefault(key, []).append(test)
70   
71           for key, tests in module_to_tests.items():
72               queue_elements.append(tests)
73   
74           if len(queue_elements) < jobs:
75               # Don't create jobs we will never use.
76               jobs = len(queue_elements)
77   
78       elif split == "tests":
79           for test in tests:
80               lst = []
81               flatten_test_suite(test, lst)
82               for test in lst:
83                   queue_elements.append([test])
84   
85           if len(queue_elements) < jobs:
86               # Don't create jobs we will never use.
87               jobs = len(queue_elements)
88   
89       else:
90           raise AssertionError("Do not know how to handle: %s" % (split,))
91   
92       for test_cases in queue_elements:
93           test_queue_elements = []
94           for test_case in test_cases:
95               try:
96                   test_name = test_case.__class__.__name__ + "." + test_case._testMethodName
97               except AttributeError:
98                   # Support for jython 2.1 (__testMethodName is pseudo-private in the test case)
99                   test_name = test_case.__class__.__name__ + "." + test_case._TestCase__testMethodName
100  
101              test_queue_elements.append(test_case.__pydev_pyfile__ + "|" + test_name)
102  
103          tests_queue.append(test_queue_elements)
104  
105      if jobs < 2:
106          return False
107  
108      sys.stdout.write("Running tests in parallel with: %s jobs.\n" % (jobs,))
109  
110      queue = Queue.Queue()
111      for item in tests_queue:
112          queue.put(item, block=False)
113  
114      providers = []
115      clients = []
116      for i in range(jobs):
117          test_cases_provider = CommunicationThread(queue)
118          providers.append(test_cases_provider)
119  
120          test_cases_provider.start()
121          port = test_cases_provider.port
122  
123          if coverage_files:
124              clients.append(ClientThread(i, port, verbosity, coverage_files.pop(0), coverage_include))
125          else:
126              clients.append(ClientThread(i, port, verbosity))
127  
128      for client in clients:
129          client.start()
130  
131      client_alive = True
132      while client_alive:
133          client_alive = False
134          for client in clients:
135              # Wait for all the clients to exit.
136              if not client.finished:
137                  client_alive = True
138                  time.sleep(0.2)
139                  break
140  
141      for provider in providers:
142          provider.shutdown()
143  
144      return True

150  class CommunicationThread(threading.Thread):
151      def __init__(self, tests_queue):
152          threading.Thread.__init__(self)
153          self.daemon = True
154          self.queue = tests_queue
155          self.finished = False
156          from _pydev_bundle.pydev_imports import SimpleXMLRPCServer
157          from _pydev_bundle import pydev_localhost
158  
159          # Create server
160          server = SimpleXMLRPCServer((pydev_localhost.get_localhost(), 0), logRequests=False)
161          server.register_function(self.GetTestsToRun)
162          server.register_function(self.notifyStartTest)
163          server.register_function(self.notifyTest)
164          server.register_function(self.notifyCommands)
165          self.port = server.socket.getsockname()[1]
166          self.server = server
167  
168      def GetTestsToRun(self, job_id):
169          """ 
170          @param job_id: 
171   
172          @return: list(str) 
173              Each entry is a string in the format: filename|Test.testName 
174          """
175          try:
176              ret = self.queue.get(block=False)
177              return ret
178          except:  # Any exception getting from the queue (empty or not) means we finished our work on providing the tests.
179              self.finished = True
180              return []
181  
182      def notifyCommands(self, job_id, commands):
183          # Batch notification.
184          for command in commands:
185              getattr(self, command[0])(job_id, *command[1], **command[2])
186  
187          return True
188  
189      def notifyStartTest(self, job_id, *args, **kwargs):
190          pydev_runfiles_xml_rpc.notifyStartTest(*args, **kwargs)
191          return True
192  
193      def notifyTest(self, job_id, *args, **kwargs):
194          pydev_runfiles_xml_rpc.notifyTest(*args, **kwargs)
195          return True
196  
197      def shutdown(self):
198          if hasattr(self.server, "shutdown"):
199              self.server.shutdown()
200          else:
201              self._shutdown = True
202  
203      def run(self):
204          if hasattr(self.server, "shutdown"):
205              self.server.serve_forever()
206          else:
207              self._shutdown = False
208              while not self._shutdown:
209                  self.server.handle_request()

215  class ClientThread(threading.Thread):
216      def __init__(self, job_id, port, verbosity, coverage_output_file=None, coverage_include=None):
217          threading.Thread.__init__(self)
218          self.daemon = True
219          self.port = port
220          self.job_id = job_id
221          self.verbosity = verbosity
222          self.finished = False
223          self.coverage_output_file = coverage_output_file
224          self.coverage_include = coverage_include
225  
226      def _reader_thread(self, pipe, target):
227          while True:
228              target.write(pipe.read(1))
229  
230      def run(self):
231          try:
232              from _pydev_runfiles import pydev_runfiles_parallel_client
233              # TODO: Support Jython:
234              #
235              # For jython, instead of using sys.executable, we should use:
236              # r'D:\bin\jdk_1_5_09\bin\java.exe',
237              # '-classpath',
238              # 'D:/bin/jython-2.2.1/jython.jar',
239              # 'org.python.util.jython',
240  
241              args = [
242                  sys.executable,
243                  pydev_runfiles_parallel_client.__file__,
244                  str(self.job_id),
245                  str(self.port),
246                  str(self.verbosity),
247              ]
248  
249              if self.coverage_output_file and self.coverage_include:
250                  args.append(self.coverage_output_file)
251                  args.append(self.coverage_include)
252  
253              import subprocess
254  
255              if False:
256                  proc = subprocess.Popen(args, env=os.environ, shell=False, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
257  
258                  thread.start_new_thread(self._reader_thread, (proc.stdout, sys.stdout))
259  
260                  thread.start_new_thread(target=self._reader_thread, args=(proc.stderr, sys.stderr))
261              else:
262                  proc = subprocess.Popen(args, env=os.environ, shell=False)
263                  proc.wait()
264  
265          finally:
266              self.finished = True
