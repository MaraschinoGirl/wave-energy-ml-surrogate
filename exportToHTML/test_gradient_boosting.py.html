252  def test_gamma():
253      # For a Gamma distributed target, we expect an HGBT trained with the Gamma deviance
254      # (loss) to give better results than an HGBT with any other loss function, measured
255      # in out-of-sample Gamma deviance as metric/score.
256      # Note that squared error could potentially predict negative values which is
257      # invalid (np.inf) for the Gamma deviance. A Poisson HGBT (having a log link)
258      # does not have that defect.
259      # Important note: It seems that a Poisson HGBT almost always has better
260      # out-of-sample performance than the Gamma HGBT, measured in Gamma deviance.
261      # LightGBM shows the same behaviour. Hence, we only compare to a squared error
262      # HGBT, but not to a Poisson deviance HGBT.
263      rng = np.random.RandomState(42)
264      n_train, n_test, n_features = 500, 100, 20
265      X = make_low_rank_matrix(
266          n_samples=n_train + n_test,
267          n_features=n_features,
268          random_state=rng,
269      )
270      # We create a log-linear Gamma model. This gives y.min ~ 1e-2, y.max ~ 1e2
271      coef = rng.uniform(low=-10, high=20, size=n_features)
272      # Numpy parametrizes gamma(shape=k, scale=theta) with mean = k * theta and
273      # variance = k * theta^2. We parametrize it instead with mean = exp(X @ coef)
274      # and variance = dispersion * mean^2 by setting k = 1 / dispersion,
275      # theta =  dispersion * mean.
276      dispersion = 0.5
277      y = rng.gamma(shape=1 / dispersion, scale=dispersion * np.exp(X @ coef))
278      X_train, X_test, y_train, y_test = train_test_split(
279          X, y, test_size=n_test, random_state=rng
280      )
281      gbdt_gamma = HistGradientBoostingRegressor(loss="gamma", random_state=123)
282      gbdt_mse = HistGradientBoostingRegressor(loss="squared_error", random_state=123)
283      dummy = DummyRegressor(strategy="mean")
284      for model in (gbdt_gamma, gbdt_mse, dummy):
285          model.fit(X_train, y_train)
286  
287      for X, y in [(X_train, y_train), (X_test, y_test)]:
288          loss_gbdt_gamma = mean_gamma_deviance(y, gbdt_gamma.predict(X))
289          # We restrict the squared error HGBT to predict at least the minimum seen y at
290          # train time to make it strictly positive.
291          loss_gbdt_mse = mean_gamma_deviance(
292              y, np.maximum(np.min(y_train), gbdt_mse.predict(X))
293          )
294          loss_dummy = mean_gamma_deviance(y, dummy.predict(X))
295          assert loss_gbdt_gamma < loss_dummy
296          assert loss_gbdt_gamma < loss_gbdt_mse

299  @pytest.mark.parametrize("quantile", [0.2, 0.5, 0.8])
300  def test_quantile_asymmetric_error(quantile):
301      """Test quantile regression for asymmetric distributed targets."""
302      n_samples = 10_000
303      rng = np.random.RandomState(42)
304      # take care that X @ coef + intercept > 0
305      X = np.concatenate(
306          (
307              np.abs(rng.randn(n_samples)[:, None]),
308              -rng.randint(2, size=(n_samples, 1)),
309          ),
310          axis=1,
311      )
312      intercept = 1.23
313      coef = np.array([0.5, -2])
314      # For an exponential distribution with rate lambda, e.g. exp(-lambda * x),
315      # the quantile at level q is:
316      #   quantile(q) = - log(1 - q) / lambda
317      #   scale = 1/lambda = -quantile(q) / log(1-q)
318      y = rng.exponential(
319          scale=-(X @ coef + intercept) / np.log(1 - quantile), size=n_samples
320      )
321      model = HistGradientBoostingRegressor(
322          loss="quantile",
323          quantile=quantile,
324          max_iter=25,
325          random_state=0,
326          max_leaf_nodes=10,
327      ).fit(X, y)
328      assert_allclose(np.mean(model.predict(X) > y), quantile, rtol=1e-2)
329  
330      pinball_loss = PinballLoss(quantile=quantile)
331      loss_true_quantile = pinball_loss(y, X @ coef + intercept)
332      loss_pred_quantile = pinball_loss(y, model.predict(X))
333      # we are overfitting
334      assert loss_pred_quantile <= loss_true_quantile

346  def test_poisson():
347      # For Poisson distributed target, Poisson loss should give better results
348      # than least squares measured in Poisson deviance as metric.
349      rng = np.random.RandomState(42)
350      n_train, n_test, n_features = 500, 100, 100
351      X = make_low_rank_matrix(
352          n_samples=n_train + n_test, n_features=n_features, random_state=rng
353      )
354      # We create a log-linear Poisson model and downscale coef as it will get
355      # exponentiated.
356      coef = rng.uniform(low=-2, high=2, size=n_features) / np.max(X, axis=0)
357      y = rng.poisson(lam=np.exp(X @ coef))
358      X_train, X_test, y_train, y_test = train_test_split(
359          X, y, test_size=n_test, random_state=rng
360      )
361      gbdt_pois = HistGradientBoostingRegressor(loss="poisson", random_state=rng)
362      gbdt_ls = HistGradientBoostingRegressor(loss="squared_error", random_state=rng)
363      gbdt_pois.fit(X_train, y_train)
364      gbdt_ls.fit(X_train, y_train)
365      dummy = DummyRegressor(strategy="mean").fit(X_train, y_train)
366  
367      for X, y in [(X_train, y_train), (X_test, y_test)]:
368          metric_pois = mean_poisson_deviance(y, gbdt_pois.predict(X))
369          # squared_error might produce non-positive predictions => clip
370          metric_ls = mean_poisson_deviance(y, np.clip(gbdt_ls.predict(X), 1e-15, None))
371          metric_dummy = mean_poisson_deviance(y, dummy.predict(X))
372          assert metric_pois < metric_ls
373          assert metric_pois < metric_dummy

524  def test_missing_values_minmax_imputation():
525      # Compare the buit-in missing value handling of Histogram GBC with an
526      # a-priori missing value imputation strategy that should yield the same
527      # results in terms of decision function.
528      #
529      # Each feature (containing NaNs) is replaced by 2 features:
530      # - one where the nans are replaced by min(feature) - 1
531      # - one where the nans are replaced by max(feature) + 1
532      # A split where nans go to the left has an equivalent split in the
533      # first (min) feature, and a split where nans go to the right has an
534      # equivalent split in the second (max) feature.
535      #
536      # Assuming the data is such that there is never a tie to select the best
537      # feature to split on during training, the learned decision trees should be
538      # strictly equivalent (learn a sequence of splits that encode the same
539      # decision function).
540      #
541      # The MinMaxImputer transformer is meant to be a toy implementation of the
542      # "Missing In Attributes" (MIA) missing value handling for decision trees
543      # https://www.sciencedirect.com/science/article/abs/pii/S0167865508000305
544      # The implementation of MIA as an imputation transformer was suggested by
545      # "Remark 3" in :arxiv:'<1902.06931>`
546  
547      class MinMaxImputer(TransformerMixin, BaseEstimator):
548          def fit(self, X, y=None):
549              mm = MinMaxScaler().fit(X)
550              self.data_min_ = mm.data_min_
551              self.data_max_ = mm.data_max_
552              return self

554          def transform(self, X):
555              X_min, X_max = X.copy(), X.copy()
556  
557              for feature_idx in range(X.shape[1]):
558                  nan_mask = np.isnan(X[:, feature_idx])
559                  X_min[nan_mask, feature_idx] = self.data_min_[feature_idx] - 1
560                  X_max[nan_mask, feature_idx] = self.data_max_[feature_idx] + 1
561

564      def make_missing_value_data(n_samples=int(1e4), seed=0):

568          # Pre-bin the data to ensure a deterministic handling by the 2
569          # strategies and also make it easier to insert np.nan in a structured
570          # way:
571          X = KBinsDiscretizer(n_bins=42, encode="ordinal").fit_transform(X)
572  
573          # First feature has missing values completely at random:
574          rnd_mask = rng.rand(X.shape[0]) > 0.9
575          X[rnd_mask, 0] = np.nan
576  
577          # Second and third features have missing values for extreme values
578          # (censoring missingness):
579          low_mask = X[:, 1] == 0
580          X[low_mask, 1] = np.nan
581  
582          high_mask = X[:, 2] == X[:, 2].max()
583          X[high_mask, 2] = np.nan
584  
585          # Make the last feature nan pattern very informative:
586          y_max = np.percentile(y, 70)
587          y_max_mask = y >= y_max
588          y[y_max_mask] = y_max
589          X[y_max_mask, 3] = np.nan
590  
591          # Check that there is at least one missing value in each feature:
592          for feature_idx in range(X.shape[1]):
593              assert any(np.isnan(X[:, feature_idx]))
594  
595          # Let's use a test set to check that the learned decision function is
596          # the same as evaluated on unseen data. Otherwise it could just be the
597          # case that we find two independent ways to overfit the training set.
598          return train_test_split(X, y, random_state=rng)
599  
600      # n_samples need to be large enough to minimize the likelihood of having
601      # several candidate splits with the same gain value in a given tree.
602      X_train, X_test, y_train, y_test = make_missing_value_data(
603          n_samples=int(1e4), seed=0

606      # Use a small number of leaf nodes and iterations so as to keep
607      # under-fitting models to minimize the likelihood of ties when training the
608      # model.

618      assert gbm1.score(X_test, y_test) == pytest.approx(gbm2.score(X_test, y_test))

621      # decision function check.

1303 def test_interaction_cst_numerically():
1304     """Check that interaction constraints have no forbidden interactions."""
1305     rng = np.random.RandomState(42)
1306     n_samples = 1000
1307     X = rng.uniform(size=(n_samples, 2))
1308     # Construct y with a strong interaction term
1309     # y = x0 + x1 + 5 * x0 * x1
1310     y = np.hstack((X, 5 * X[:, [0]] * X[:, [1]])).sum(axis=1)
1311 
1312     est = HistGradientBoostingRegressor(random_state=42)
1313     est.fit(X, y)
1314     est_no_interactions = HistGradientBoostingRegressor(
1315         interaction_cst=[{0}, {1}], random_state=42
1316     )
1317     est_no_interactions.fit(X, y)
1318 
1319     delta = 0.25
1320     # Make sure we do not extrapolate out of the training set as tree-based estimators
1321     # are very bad in doing so.
1322     X_test = X[(X[:, 0] < 1 - delta) & (X[:, 1] < 1 - delta)]
1323     X_delta_d_0 = X_test + [delta, 0]
1324     X_delta_0_d = X_test + [0, delta]
1325     X_delta_d_d = X_test + [delta, delta]
1326 
1327     # Note: For the y from above as a function of x0 and x1, we have
1328     # y(x0+d, x1+d) = y(x0, x1) + 5 * d * (2/5 + x0 + x1) + 5 * d**2
1329     # y(x0+d, x1)   = y(x0, x1) + 5 * d * (1/5 + x1)
1330     # y(x0,   x1+d) = y(x0, x1) + 5 * d * (1/5 + x0)
1331     # Without interaction constraints, we would expect a result of 5 * d**2 for the
1332     # following expression, but zero with constraints in place.
1333     assert_allclose(
1334         est_no_interactions.predict(X_delta_d_d)
1335         + est_no_interactions.predict(X_test)
1336         - est_no_interactions.predict(X_delta_d_0)
1337         - est_no_interactions.predict(X_delta_0_d),
1338         0,
1339         atol=1e-12,
1340     )
1341 
1342     # Correct result of the expressions is 5 * delta**2. But this is hard to achieve by
1343     # a fitted tree-based model. However, with 100 iterations the expression should
1344     # at least be positive!
1345     assert np.all(
1346         est.predict(X_delta_d_d)
1347         + est.predict(X_test)
1348         - est.predict(X_delta_d_0)
1349         - est.predict(X_delta_0_d)
1350         > 0.01
1351     )
