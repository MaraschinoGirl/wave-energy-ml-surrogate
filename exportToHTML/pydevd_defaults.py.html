pydevd_defaults.py 
1    """ 
2    This module holds the customization settings for the debugger. 
3    """
4    
5    from _pydevd_bundle.pydevd_constants import QUOTED_LINE_PROTOCOL
6    from _pydev_bundle import pydev_log
7    import sys
8    
9    
10   class PydevdCustomization(object):
11       DEFAULT_PROTOCOL: str = QUOTED_LINE_PROTOCOL
12   
13       # Debug mode may be set to 'debugpy-dap'.
14       #
15       # In 'debugpy-dap' mode the following settings are done to PyDB:
16       #
17       # py_db.skip_suspend_on_breakpoint_exception = (BaseException,)
18       # py_db.skip_print_breakpoint_exception = (NameError,)
19       # py_db.multi_threads_single_notification = True
20       DEBUG_MODE: str = ""
21   
22       # This may be a <sys_path_entry>;<module_name> to be pre-imported
23       # Something as: 'c:/temp/foo;my_module.bar'
24       #
25       # What's done in this case is something as:
26       #
27       # sys.path.insert(0, <sys_path_entry>)
28       # try:
29       #     import <module_name>
30       # finally:
31       #     del sys.path[0]
32       #
33       # If the pre-import fails an output message is
34       # sent (but apart from that debugger execution
35       # should continue).
36       PREIMPORT: str = ""
37   
38   
39   def on_pydb_init(py_db):
40       if PydevdCustomization.DEBUG_MODE == "debugpy-dap":
41           pydev_log.debug("Apply debug mode: debugpy-dap")
42           py_db.skip_suspend_on_breakpoint_exception = (BaseException,)
43           py_db.skip_print_breakpoint_exception = (NameError,)
44           py_db.multi_threads_single_notification = True
45       elif not PydevdCustomization.DEBUG_MODE:
46           pydev_log.debug("Apply debug mode: default")
47       else:
48           pydev_log.debug("WARNING: unknown debug mode: %s", PydevdCustomization.DEBUG_MODE)
49   
50       if PydevdCustomization.PREIMPORT:
51           pydev_log.debug("Preimport: %s", PydevdCustomization.PREIMPORT)
52           try:
53               sys_path_entry, module_name = PydevdCustomization.PREIMPORT.rsplit(";", maxsplit=1)
54           except Exception:
55               pydev_log.exception("Expected ';' in %s" % (PydevdCustomization.PREIMPORT,))
56           else:
57               try:
58                   sys.path.insert(0, sys_path_entry)
59                   try:
60                       __import__(module_name)
61                   finally:
62                       sys.path.remove(sys_path_entry)
63               except Exception:
64                   pydev_log.exception("Error importing %s (with sys.path entry: %s)" % (module_name, sys_path_entry))
