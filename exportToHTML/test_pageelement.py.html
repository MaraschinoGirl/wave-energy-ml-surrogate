54       def test_encode_deeply_nested_document(self):
55           # This test verifies that encoding a string doesn't involve
56           # any recursive function calls. If it did, this test would
57           # overflow the Python interpreter stack.
58           limit = sys.getrecursionlimit() + 1
59           markup = "<span>" * limit
60           soup = self.soup(markup)
61           encoded = soup.encode()
62           assert limit == encoded.count(b"<span>")

64       def test_deprecated_renderContents(self):
65           html = "<b>\N{SNOWMAN}</b>"
66           soup = self.soup(html)
67           with warnings.catch_warnings(record=True) as w:
68               soup.renderContents()
69               assert "\N{SNOWMAN}".encode("utf8") == soup.b.renderContents()
70           msgs = [str(warning.message) for warning in w]
71           assert all(
72               x
73               == "Call to deprecated method renderContents. (Replaced by encode_contents) -- Deprecated since version 4.0.0."
74               for x in msgs
75           )

190      def test_prettify_handles_nested_string_literal_tags(self):
191          # Most of this markup is inside a <pre> tag, so prettify()
192          # only does three things to it:
193          # 1. Add a newline and a space between the <div> and the <pre>
194          # 2. Add a newline after the </pre>
195          # 3. Add a newline at the end.
196          #
197          # The contents of the <pre> tag are left completely alone.  In
198          # particular, we don't start adding whitespace again once we
199          # encounter the first </pre> tag, because we know it's not
200          # the one that put us into string literal mode.
201          markup = """<div><pre><code>some 
202  <script><pre>code</pre></script> for you  
203  </code></pre></div>"""
204  
205          expect = """<div> 
206   <pre><code>some 
207  <script><pre>code</pre></script> for you  
208  </code></pre> 
209  </div> 
210  """
211          soup = self.soup(markup)
212          assert expect == soup.div.prettify()

234          # Here's the <meta> tag saying that a document is
235          # encoded in Shift-JIS.
236          meta_tag = (
237              '<meta content="text/html; charset=x-sjis" ' 'http-equiv="Content-type"/>'
238          )
239          soup = self.soup(meta_tag)
240  
241          # Parse the document, and the charset apprears unchanged.
242          assert soup.meta["content"] == "text/html; charset=x-sjis"
243  
244          # Encode the document into some encoding, and the encoding is
245          # substituted into the meta tag.
246          utf_8 = soup.encode("utf-8")
247          assert b"charset=utf-8" in utf_8
248  
249          euc_jp = soup.encode("euc_jp")
250          assert b"charset=euc_jp" in euc_jp
251  
252          shift_jis = soup.encode("shift-jis")
253          assert b"charset=shift-jis" in shift_jis
254  
255          utf_16_u = soup.encode("utf-16").decode("utf-16")
256          assert "charset=utf-16" in utf_16_u

378      def test_copy_tag_copies_contents(self):
379          html = "<div class='a b c'><b>Foo<a></a></b><b>Bar</b></div>end"
380          soup = self.soup(html)
381          div = soup.div
382          div_copy = copy.copy(div)
383  
384          # The two tags look the same, and evaluate to equal.
385          assert str(div) == str(div_copy)
386          assert div == div_copy
387  
388          # But they're not the same object.
389          assert div is not div_copy
390  
391          # And they don't have the same relation to the parse tree. The
392          # copy is not associated with a parse tree at all.
393          assert None is div_copy.parent
394          assert None is div_copy.previous_element
395          assert None is div_copy.find(string="Bar").next_element
396          assert None is not div.find(string="Bar").next_element
397  
398          # Modifying one of the tag's multi-valued attributes
399          # doesn't modify the other.
400          assert div["class"] is not div_copy["class"]
401          div["class"].append("d")
402          assert "a b c d".split() == div["class"]
403          assert "a b c".split() == div_copy["class"]
404          assert isinstance(div_copy["class"], AttributeValueList)

409      def test_comparison(self):
410          soup = self.soup("<a>string</a> <a>string</a>")
411          first_a, second_a = soup.find_all('a')
412          first_string, second_string = soup.find_all(string='string')
413  
414          # Tags with the same markup are equal.
415          assert first_a == second_a
416  
417          # NavigableStrings with the same content are equal, and also
418          # equal to a Python string with the same content...
419          assert first_string == second_string == "string"
420  
421          # ...but not equivalent to a bytestring with the same content.
422          assert first_string != b"string"
423  
424      def test_hash(self):
425          soup = self.soup("<a>string</a> <a>string</a>")
426          first_a, second_a = soup.find_all('a')
427          first_string, second_string = soup.find_all(string='string')
428  
429          # Tags with the same markup hash to the same value.
430          assert hash(first_a) == hash(second_a)
431  
432          # But they're not the same object.
433          assert id(first_a) != id(second_a)
434  
435          # NavigableStrings with the same contents hash to the value of
436          # the contents.
437          assert hash(first_string) == hash(second_string) == hash("string")
