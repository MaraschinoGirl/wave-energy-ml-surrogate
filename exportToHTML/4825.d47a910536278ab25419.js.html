71   function tokenize(stream, state) {
72     // Finally advance the stream
73     var ch = stream.next();
74   
75     // BLOCKCOMMENT
76     if (ch === '/' && stream.eat('*')) {
77       state.continueComment = true;
78       return "comment";
79     } else if (state.continueComment === true) { // in comment block
80       //comment ends at the beginning of the line
81       if (ch === '*' && stream.peek() === '/') {
82         stream.next();
83         state.continueComment = false;
84       } else if (stream.skipTo('*')) { //comment is potentially later in line
85         stream.skipTo('*');
86         stream.next();
87         if (stream.eat('/'))
88           state.continueComment = false;
89       } else {
90         stream.skipToEnd();
91       }
92       return "comment";
93     }
94   
95     if (ch == "*" && stream.column() == stream.indentation()) {
96       stream.skipToEnd()
97       return "comment"
98     }
99   
100    // DoubleOperator match
101    var doubleOperator = ch + stream.peek();
102  
103    if ((ch === '"' || ch === "'") && !state.continueString) {
104      state.continueString = ch
105      return "string"
106    } else if (state.continueString) {
107      if (state.continueString == ch) {
108        state.continueString = null;
109      } else if (stream.skipTo(state.continueString)) {
110        // quote found on this line
111        stream.next();
112        state.continueString = null;
113      } else {
114        stream.skipToEnd();
115      }
116      return "string";
117    } else if (state.continueString !== null && stream.eol()) {
118      stream.skipTo(state.continueString) || stream.skipToEnd();
119      return "string";
120    } else if (/[\d\.]/.test(ch)) { //find numbers

129      stream.next();
130      return "operator";

132      stream.next();
