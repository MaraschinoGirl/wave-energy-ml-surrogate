95   def _find_widget_refs_by_state(widget, state):
96       """Find references to other widgets in a widget's state"""
97       # Copy keys to allow changes to state during iteration:
98       keys = tuple(state.keys())
99       for key in keys:
100          value = getattr(widget, key)
101          # Trivial case: Direct references to other widgets:
102          if isinstance(value, Widget):
103              yield value
104          # Also check for buried references in known, JSON-able structures
105          # Note: This might miss references buried in more esoteric structures
106          elif isinstance(value, (list, tuple)):
107              for item in value:
108                  if isinstance(item, Widget):
109                      yield item
110          elif isinstance(value, dict):
111              for item in value.values():
112                  if isinstance(item, Widget):
113                      yield item

116  def _get_recursive_state(widget, store=None, drop_defaults=False):
117      """Gets the embed state of a widget, and all other widgets it refers to as well"""
118      if store is None:
119          store = dict()
120      state = widget._get_embed_state(drop_defaults=drop_defaults)
121      store[widget.model_id] = state
122  
123      # Loop over all values included in state (i.e. don't consider excluded values):
124      for ref in _find_widget_refs_by_state(widget, state['state']):
125          if ref.model_id not in store:
126              _get_recursive_state(ref, store, drop_defaults=drop_defaults)
127      return store

130  def add_resolved_links(store, drop_defaults):
131      """Adds the state of any link models between two models in store"""
132      for widget_id, widget in widget_module._instances.items(): # go over all widgets
133          if isinstance(widget, Link) and widget_id not in store:
134              if widget.source[0].model_id in store and widget.target[0].model_id in store:
135                  store[widget.model_id] = widget._get_embed_state(drop_defaults=drop_defaults)

138  def dependency_state(widgets, drop_defaults=True):
139      """Get the state of all widgets specified, and their dependencies. 
140   
141      This uses a simple dependency finder, including: 
142       - any widget directly referenced in the state of an included widget 
143       - any widget in a list/tuple attribute in the state of an included widget 
144       - any widget in a dict attribute in the state of an included widget 
145       - any jslink/jsdlink between two included widgets 
146      What this alogorithm does not do: 
147       - Find widget references in nested list/dict structures 
148       - Find widget references in other types of attributes 
149   
150      Note that this searches the state of the widgets for references, so if 
151      a widget reference is not included in the serialized state, it won't 
152      be considered as a dependency. 
153   
154      Parameters 
155      ---------- 
156      widgets: single widget or list of widgets. 
157         This function will return the state of every widget mentioned 
158         and of all their dependencies. 
159      drop_defaults: boolean 
160          Whether to drop default values from the widget states. 
161   
162      Returns 
163      ------- 
164      A dictionary with the state of the widgets and any widget they 
165      depend on. 
166      """
167      # collect the state of all relevant widgets
168      if widgets is None:
169          # Get state of all widgets, no smart resolution needed.
170          state = Widget.get_manager_state(drop_defaults=drop_defaults, widgets=None)['state']
171      else:
172          try:
173              widgets[0]
174          except (IndexError, TypeError):
175              widgets = [widgets]
176          state = {}
177          for widget in widgets:
178              _get_recursive_state(widget, state, drop_defaults)
179          # Add any links between included widgets:
180          add_resolved_links(state, drop_defaults)
181      return state

184  @doc_subst(_doc_snippets)
185  def embed_data(views, drop_defaults=True, state=None):
186      """Gets data for embedding. 
187   
188      Use this to get the raw data for embedding if you have special 
189      formatting needs. 
190   
191      Parameters 
192      ---------- 
193      {views_attribute} 
194      drop_defaults: boolean 
195          Whether to drop default values from the widget states. 
196      state: dict or None (default) 
197          The state to include. When set to None, the state of all widgets 
198          know to the widget manager is included. Otherwise it uses the 
199          passed state directly. This allows for end users to include a 
200          smaller state, under the responsibility that this state is 
201          sufficient to reconstruct the embedded views. 
202   
203      Returns 
204      ------- 
205      A dictionary with the following entries: 
206          manager_state: dict of the widget manager state data 
207          view_specs: a list of widget view specs 
208      """
209      if views is None:
210          views = [w for w in widget_module._instances.values() if isinstance(w, DOMWidget)]
211      else:
212          try:
213              views[0]
214          except (IndexError, TypeError):
215              views = [views]
216  
217      if state is None:
218          # Get state of all known widgets
219          state = Widget.get_manager_state(drop_defaults=drop_defaults, widgets=None)['state']
220  
221      # Rely on ipywidget to get the default values
222      json_data = Widget.get_manager_state(widgets=[])
223      # but plug in our own state
224      json_data['state'] = state
225  
226      view_specs = [w.get_view_spec() for w in views]
227  
228      return dict(manager_state=json_data, view_specs=view_specs)

230  script_escape_re = re.compile(r'<(script|/script|!--)', re.IGNORECASE)
231  def escape_script(s):
232      """Escape a string that will be the content of an HTML script tag. 
233   
234      We replace the opening bracket of <script, </script, and <!-- with the unicode 
235      equivalent. This is inspired by the documentation for the script tag at 
236      https://html.spec.whatwg.org/multipage/scripting.html#restrictions-for-contents-of-script-elements 
237   
238      We only replace these three cases so that most html or other content 
239      involving `<` is readable. 
240      """
241      return script_escape_re.sub(r'\\u003c\1', s)

243  @doc_subst(_doc_snippets)
244  def embed_snippet(views,
245                    drop_defaults=True,
246                    state=None,
247                    indent=2,
248                    embed_url=None,
249                    requirejs=True,
250                    cors=True
251                   ):
252      """Return a snippet that can be embedded in an HTML file. 
253   
254      Parameters 
255      ---------- 
256      {views_attribute} 
257      {embed_kwargs} 
258   
259      Returns 
260      ------- 
261      A unicode string with an HTML snippet containing several `<script>` tags. 
262      """
263  
264      data = embed_data(views, drop_defaults=drop_defaults, state=state)
265  
266      widget_views = '\n'.join(
267          widget_view_template.format(view_spec=escape_script(json.dumps(view_spec)))
268          for view_spec in data['view_specs']
269      )
270  
271      if embed_url is None:
272          embed_url = DEFAULT_EMBED_REQUIREJS_URL if requirejs else DEFAULT_EMBED_SCRIPT_URL
273  
274      load = load_requirejs_template if requirejs else load_template
275  
276      use_cors = ' crossorigin="anonymous"' if cors else ''
277      values = {
278          'load': load.format(embed_url=embed_url, use_cors=use_cors),
279          'json_data': escape_script(json.dumps(data['manager_state'], indent=indent)),
280          'widget_views': widget_views,
281      }
282  
283      return snippet_template.format(**values)

286  @doc_subst(_doc_snippets)
287  def embed_minimal_html(fp, views, title='IPyWidget export', template=None, **kwargs):
288      """Write a minimal HTML file with widget views embedded. 
289   
290      Parameters 
291      ---------- 
292      fp: filename or file-like object 
293          The file to write the HTML output to. 
294      {views_attribute} 
295      title: title of the html page. 
296      template: Template in which to embed the widget state. 
297          This should be a Python string with placeholders 
298          `{{title}}` and `{{snippet}}`. The `{{snippet}}` placeholder 
299          will be replaced by all the widgets. 
300      {embed_kwargs} 
301      """
302      snippet = embed_snippet(views, **kwargs)
303  
304      values = {
305          'title': title,
306          'snippet': snippet,
307      }
308      if template is None:
309          template = html_template
310  
311      html_code = template.format(**values)
312  
313      # Check if fp is writable:
314      if hasattr(fp, 'write'):
315          fp.write(html_code)
316      else:
317          # Assume fp is a filename:
318          with open(fp, "w") as f:
319              f.write(html_code)
