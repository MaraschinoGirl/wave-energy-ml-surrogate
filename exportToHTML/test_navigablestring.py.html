19   class TestNavigableString(SoupTest):
20       def test_text_acquisition_methods(self):
21           # These methods are intended for use against Tag, but they
22           # work on NavigableString as well,
23   
24           s = NavigableString("fee ")
25           cdata = CData("fie ")
26           comment = Comment("foe ")
27   
28           assert "fee " == s.get_text()
29           assert "fee " == s.string
30           assert "fee" == s.get_text(strip=True)
31           assert ["fee "] == list(s.strings)
32           assert ["fee"] == list(s.stripped_strings)
33           assert ["fee "] == list(s._all_strings())
34   
35           assert "fie " == cdata.get_text()
36           assert "fie " == cdata.string
37           assert "fie" == cdata.get_text(strip=True)
38           assert ["fie "] == list(cdata.strings)
39           assert ["fie"] == list(cdata.stripped_strings)
40           assert ["fie "] == list(cdata._all_strings())
41   
42           # Since a Comment isn't normally considered 'text',
43           # these methods generally do nothing.
44           assert "" == comment.get_text()
45           assert [] == list(comment.strings)
46           assert [] == list(comment.stripped_strings)
47           assert [] == list(comment._all_strings())
48   
49           # Unless you specifically say that comments are okay.
50           assert "foe" == comment.get_text(strip=True, types=Comment)
51           assert "foe " == comment.get_text(types=(Comment, NavigableString))

61   class TestNavigableStringSubclasses(SoupTest):
62       def test_cdata(self):
63           # None of the current builders turn CDATA sections into CData
64           # objects, but you can create them manually.
65           soup = self.soup("")
66           cdata = CData("foo")
67           soup.insert(1, cdata)
68           assert str(soup) == "<![CDATA[foo]]>"
69           assert soup.find(string="foo") == "foo"
70           assert soup.contents[0] == "foo"
71   
72       def test_cdata_is_never_formatted(self):
73           """Text inside a CData object is passed into the formatter. 
74    
75           But the return value is ignored. 
76           """
77   
78           self.count = 0
79   
80           def increment(*args):
81               self.count += 1
82               return "BITTER FAILURE"
83   
84           soup = self.soup("")
85           cdata = CData("<><><>")
86           soup.insert(1, cdata)
87           assert b"<![CDATA[<><><>]]>" == soup.encode(formatter=increment)
88           assert 1 == self.count

102      def test_default_string_containers(self):
103          # In some cases, we use different NavigableString subclasses for
104          # the same text in different tags.
105          soup = self.soup("<div>text</div><script>text</script><style>text</style>")
106          assert [NavigableString, Script, Stylesheet] == [
107              x.__class__ for x in soup.find_all(string=True)
108          ]
109  
110          # The TemplateString is a little unusual because it's generally found
111          # _inside_ children of a <template> element, not a direct child of the
112          # <template> element.
113          soup = self.soup(
114              "<template>Some text<p>In a tag</p></template>Some text outside"
115          )
116          assert all(
117              isinstance(x, TemplateString)
118              for x in soup.template._all_strings(types=None)
119          )
120  
121          # Once the <template> tag closed, we went back to using
122          # NavigableString.
123          outside = soup.template.next_sibling
124          assert isinstance(outside, NavigableString)
125          assert not isinstance(outside, TemplateString)
126  
127          # The TemplateString is also unusual because it can contain
128          # NavigableString subclasses of _other_ types, such as
129          # Comment.
130          markup = b"<template>Some text<p>In a tag</p><!--with a comment--></template>"
131          soup = self.soup(markup)
132          assert markup == soup.template.encode("utf8")

134      def test_ruby_strings(self):
135          markup = "<ruby>漢 <rp>(</rp><rt>kan</rt><rp>)</rp> 字 <rp>(</rp><rt>ji</rt><rp>)</rp></ruby>"
136          soup = self.soup(markup)
137          assert isinstance(soup.rp.string, RubyParenthesisString)
138          assert isinstance(soup.rt.string, RubyTextString)
139  
140          # Just as a demo, here's what this means for get_text usage.
141          assert "漢字" == soup.get_text(strip=True)
142          assert "漢(kan)字(ji)" == soup.get_text(
143              strip=True, types=(NavigableString, RubyTextString, RubyParenthesisString)
