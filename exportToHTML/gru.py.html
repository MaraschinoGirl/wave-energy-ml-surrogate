15   @keras_export("keras.layers.GRUCell")
16   class GRUCell(Layer, DropoutRNNCell):
17       """Cell class for the GRU layer. 
18    
19       This class processes one step within the whole time sequence input, whereas 
20       `keras.layer.GRU` processes the whole sequence. 
21    
22       Args: 
23           units: Positive integer, dimensionality of the output space. 
24           activation: Activation function to use. Default: hyperbolic tangent 
25               (`tanh`). If you pass None, no activation is applied 
26               (ie. "linear" activation: `a(x) = x`). 
27           recurrent_activation: Activation function to use for the recurrent step. 
28               Default: sigmoid (`sigmoid`). If you pass `None`, no activation is 
29               applied (ie. "linear" activation: `a(x) = x`). 
30           use_bias: Boolean, (default `True`), whether the layer 
31               should use a bias vector. 
32           kernel_initializer: Initializer for the `kernel` weights matrix, 
33               used for the linear transformation of the inputs. Default: 
34               `"glorot_uniform"`. 
35           recurrent_initializer: Initializer for the `recurrent_kernel` 
36               weights matrix, used for the linear transformation 
37               of the recurrent state. Default: `"orthogonal"`. 
38           bias_initializer: Initializer for the bias vector. Default: `"zeros"`. 
39           kernel_regularizer: Regularizer function applied to the `kernel` weights 
40               matrix. Default: `None`. 
41           recurrent_regularizer: Regularizer function applied to the 
42               `recurrent_kernel` weights matrix. Default: `None`. 
43           bias_regularizer: Regularizer function applied to the bias vector. 
44               Default: `None`. 
45           kernel_constraint: Constraint function applied to the `kernel` weights 
46               matrix. Default: `None`. 
47           recurrent_constraint: Constraint function applied to the 
48               `recurrent_kernel` weights matrix. Default: `None`. 
49           bias_constraint: Constraint function applied to the bias vector. 
50               Default: `None`. 
51           dropout: Float between 0 and 1. Fraction of the units to drop for the 
52               linear transformation of the inputs. Default: 0. 
53           recurrent_dropout: Float between 0 and 1. Fraction of the units to drop 
54               for the linear transformation of the recurrent state. Default: 0. 
55           reset_after: GRU convention (whether to apply reset gate after or 
56               before matrix multiplication). False = "before", 
57               True = "after" (default and cuDNN compatible). 
58           seed: Random seed for dropout. 
59    
60       Call arguments: 
61           inputs: A 2D tensor, with shape `(batch, features)`. 
62           states: A 2D tensor with shape `(batch, units)`, which is the state 
63               from the previous time step. 
64           training: Python boolean indicating whether the layer should behave in 
65               training mode or in inference mode. Only relevant when `dropout` or 
66               `recurrent_dropout` is used. 
67    
68       Example: 
69    
70       >>> inputs = np.random.random((32, 10, 8)) 
71       >>> rnn = keras.layers.RNN(keras.layers.GRUCell(4)) 
72       >>> output = rnn(inputs) 
73       >>> output.shape 
74       (32, 4) 
75       >>> rnn = keras.layers.RNN( 
76       ...    keras.layers.GRUCell(4), 
77       ...    return_sequences=True, 
78       ...    return_state=True) 
79       >>> whole_sequence_output, final_state = rnn(inputs) 
80       >>> whole_sequence_output.shape 
81       (32, 10, 4) 
82       >>> final_state.shape 
83       (32, 4) 
84       """

86       def __init__(
87           self,
88           units,
89           activation="tanh",
90           recurrent_activation="sigmoid",
91           use_bias=True,
92           kernel_initializer="glorot_uniform",
93           recurrent_initializer="orthogonal",
94           bias_initializer="zeros",
95           kernel_regularizer=None,
96           recurrent_regularizer=None,
97           bias_regularizer=None,
98           kernel_constraint=None,
99           recurrent_constraint=None,
100          bias_constraint=None,
101          dropout=0.0,
102          recurrent_dropout=0.0,
103          reset_after=True,
104          seed=None,
105          **kwargs,
106      ):
107          if units <= 0:
108              raise ValueError(
109                  "Received an invalid value for argument `units`, "
110                  f"expected a positive integer, got {units}."
111              )
112          implementation = kwargs.pop("implementation", 2)
113          super().__init__(**kwargs)
114          self.implementation = implementation
115          self.units = units
116          self.activation = activations.get(activation)
117          self.recurrent_activation = activations.get(recurrent_activation)
118          self.use_bias = use_bias
119  
120          self.kernel_initializer = initializers.get(kernel_initializer)
121          self.recurrent_initializer = initializers.get(recurrent_initializer)
122          self.bias_initializer = initializers.get(bias_initializer)
123  
124          self.kernel_regularizer = regularizers.get(kernel_regularizer)
125          self.recurrent_regularizer = regularizers.get(recurrent_regularizer)
126          self.bias_regularizer = regularizers.get(bias_regularizer)
127  
128          self.kernel_constraint = constraints.get(kernel_constraint)
129          self.recurrent_constraint = constraints.get(recurrent_constraint)
130          self.bias_constraint = constraints.get(bias_constraint)
131  
132          self.dropout = min(1.0, max(0.0, dropout))
133          self.recurrent_dropout = min(1.0, max(0.0, recurrent_dropout))
134          if self.recurrent_dropout != 0.0:
135              self.implementation = 1
136          if self.implementation == 1:
137              self.dropout_mask_count = 3
138          self.seed = seed
139          self.seed_generator = backend.random.SeedGenerator(seed=seed)
140  
141          self.reset_after = reset_after
142          self.state_size = self.units
143          self.output_size = self.units

145      def build(self, input_shape):
146          super().build(input_shape)
147          input_dim = input_shape[-1]
148          self.kernel = self.add_weight(
149              shape=(input_dim, self.units * 3),
150              name="kernel",
151              initializer=self.kernel_initializer,
152              regularizer=self.kernel_regularizer,
153              constraint=self.kernel_constraint,
154          )
155          self.recurrent_kernel = self.add_weight(
156              shape=(self.units, self.units * 3),
157              name="recurrent_kernel",
158              initializer=self.recurrent_initializer,
159              regularizer=self.recurrent_regularizer,
160              constraint=self.recurrent_constraint,
161          )
162  
163          if self.use_bias:
164              if not self.reset_after:
165                  bias_shape = (3 * self.units,)
166              else:
167                  # separate biases for input and recurrent kernels
168                  # Note: the shape is intentionally different from CuDNNGRU
169                  # biases `(2 * 3 * self.units,)`, so that we can distinguish the
170                  # classes when loading and converting saved weights.
171                  bias_shape = (2, 3 * self.units)
172              self.bias = self.add_weight(
173                  shape=bias_shape,
174                  name="bias",
175                  initializer=self.bias_initializer,
176                  regularizer=self.bias_regularizer,
177                  constraint=self.bias_constraint,
178              )
179          else:
180              self.bias = None
181          self.built = True
182  
183      def call(self, inputs, states, training=False):
184          h_tm1 = (
185              states[0] if tree.is_nested(states) else states
186          )  # previous state
187  
188          if self.use_bias:
189              if not self.reset_after:
190                  input_bias, recurrent_bias = self.bias, None
191              else:
192                  input_bias, recurrent_bias = (
193                      ops.squeeze(e, axis=0)
194                      for e in ops.split(self.bias, self.bias.shape[0], axis=0)
195                  )
196  
197          if self.implementation == 1:
198              if training and 0.0 < self.dropout < 1.0:
199                  dp_mask = self.get_dropout_mask(inputs)
200                  inputs_z = inputs * dp_mask[0]
201                  inputs_r = inputs * dp_mask[1]
202                  inputs_h = inputs * dp_mask[2]
203              else:
204                  inputs_z = inputs
205                  inputs_r = inputs
206                  inputs_h = inputs
207  
208              x_z = ops.matmul(inputs_z, self.kernel[:, : self.units])
209              x_r = ops.matmul(
210                  inputs_r, self.kernel[:, self.units : self.units * 2]
211              )
212              x_h = ops.matmul(inputs_h, self.kernel[:, self.units * 2 :])
213  
214              if self.use_bias:
215                  x_z += input_bias[: self.units]
216                  x_r += input_bias[self.units : self.units * 2]
217                  x_h += input_bias[self.units * 2 :]
218  
219              if training and 0.0 < self.recurrent_dropout < 1.0:
220                  rec_dp_mask = self.get_recurrent_dropout_mask(h_tm1)
221                  h_tm1_z = h_tm1 * rec_dp_mask[0]
222                  h_tm1_r = h_tm1 * rec_dp_mask[1]
223                  h_tm1_h = h_tm1 * rec_dp_mask[2]
224              else:
225                  h_tm1_z = h_tm1
226                  h_tm1_r = h_tm1
227                  h_tm1_h = h_tm1
228  
229              recurrent_z = ops.matmul(
230                  h_tm1_z, self.recurrent_kernel[:, : self.units]
231              )
232              recurrent_r = ops.matmul(
233                  h_tm1_r, self.recurrent_kernel[:, self.units : self.units * 2]
234              )
235              if self.reset_after and self.use_bias:
236                  recurrent_z += recurrent_bias[: self.units]
237                  recurrent_r += recurrent_bias[self.units : self.units * 2]
238  
239              z = self.recurrent_activation(x_z + recurrent_z)
240              r = self.recurrent_activation(x_r + recurrent_r)
241  
242              # reset gate applied after/before matrix multiplication
243              if self.reset_after:
244                  recurrent_h = ops.matmul(
245                      h_tm1_h, self.recurrent_kernel[:, self.units * 2 :]
246                  )
247                  if self.use_bias:
248                      recurrent_h += recurrent_bias[self.units * 2 :]
249                  recurrent_h = r * recurrent_h
250              else:
251                  recurrent_h = ops.matmul(
252                      r * h_tm1_h, self.recurrent_kernel[:, self.units * 2 :]
253                  )
254  
255              hh = self.activation(x_h + recurrent_h)
256          else:
257              if training and 0.0 < self.dropout < 1.0:
258                  dp_mask = self.get_dropout_mask(inputs)
259                  inputs = inputs * dp_mask
260  
261              # inputs projected by all gate matrices at once
262              matrix_x = ops.matmul(inputs, self.kernel)
263              if self.use_bias:
264                  # biases: bias_z_i, bias_r_i, bias_h_i
265                  matrix_x += input_bias
266  
267              x_z, x_r, x_h = ops.split(matrix_x, 3, axis=-1)
268  
269              if self.reset_after:
270                  # hidden state projected by all gate matrices at once
271                  matrix_inner = ops.matmul(h_tm1, self.recurrent_kernel)
272                  if self.use_bias:
273                      matrix_inner += recurrent_bias
274              else:
275                  # hidden state projected separately for update/reset and new
276                  matrix_inner = ops.matmul(
277                      h_tm1, self.recurrent_kernel[:, : 2 * self.units]
278                  )
279  
280              recurrent_z = matrix_inner[:, : self.units]
281              recurrent_r = matrix_inner[:, self.units : self.units * 2]
282              recurrent_h = matrix_inner[:, self.units * 2 :]
283  
284              z = self.recurrent_activation(x_z + recurrent_z)
285              r = self.recurrent_activation(x_r + recurrent_r)
286  
287              if self.reset_after:
288                  recurrent_h = r * recurrent_h
289              else:
290                  recurrent_h = ops.matmul(
291                      r * h_tm1, self.recurrent_kernel[:, 2 * self.units :]
292                  )
293  
294              hh = self.activation(x_h + recurrent_h)
295  
296          # previous and candidate state mixed by update gate
297          h = z * h_tm1 + (1 - z) * hh
298          new_state = [h] if tree.is_nested(states) else h
299          return h, new_state

342  @keras_export("keras.layers.GRU")
343  class GRU(RNN):
344      """Gated Recurrent Unit - Cho et al. 2014. 
345   
346      Based on available runtime hardware and constraints, this layer 
347      will choose different implementations (cuDNN-based or backend-native) 
348      to maximize the performance. If a GPU is available and all 
349      the arguments to the layer meet the requirement of the cuDNN kernel 
350      (see below for details), the layer will use a fast cuDNN implementation 
351      when using the TensorFlow backend. 
352   
353      The requirements to use the cuDNN implementation are: 
354   
355      1. `activation` == `tanh` 
356      2. `recurrent_activation` == `sigmoid` 
357      3. `recurrent_dropout` == 0 
358      4. `unroll` is `False` 
359      5. `use_bias` is `True` 
360      6. `reset_after` is `True` 
361      7. Inputs, if use masking, are strictly right-padded. 
362      8. Eager execution is enabled in the outermost context. 
363   
364      There are two variants of the GRU implementation. The default one is based 
365      on [v3](https://arxiv.org/abs/1406.1078v3) and has reset gate applied to 
366      hidden state before matrix multiplication. The other one is based on 
367      [original](https://arxiv.org/abs/1406.1078v1) and has the order reversed. 
368   
369      The second variant is compatible with CuDNNGRU (GPU-only) and allows 
370      inference on CPU. Thus it has separate biases for `kernel` and 
371      `recurrent_kernel`. To use this variant, set `reset_after=True` and 
372      `recurrent_activation='sigmoid'`. 
373   
374      For example: 
375   
376      >>> inputs = np.random.random((32, 10, 8)) 
377      >>> gru = keras.layers.GRU(4) 
378      >>> output = gru(inputs) 
379      >>> output.shape 
380      (32, 4) 
381      >>> gru = keras.layers.GRU(4, return_sequences=True, return_state=True) 
382      >>> whole_sequence_output, final_state = gru(inputs) 
383      >>> whole_sequence_output.shape 
384      (32, 10, 4) 
385      >>> final_state.shape 
386      (32, 4) 
387   
388      Args: 
389          units: Positive integer, dimensionality of the output space. 
390          activation: Activation function to use. 
391              Default: hyperbolic tangent (`tanh`). 
392              If you pass `None`, no activation is applied 
393              (ie. "linear" activation: `a(x) = x`). 
394          recurrent_activation: Activation function to use 
395              for the recurrent step. 
396              Default: sigmoid (`sigmoid`). 
397              If you pass `None`, no activation is applied 
398              (ie. "linear" activation: `a(x) = x`). 
399          use_bias: Boolean, (default `True`), whether the layer 
400              should use a bias vector. 
401          kernel_initializer: Initializer for the `kernel` weights matrix, 
402              used for the linear transformation of the inputs. Default: 
403              `"glorot_uniform"`. 
404          recurrent_initializer: Initializer for the `recurrent_kernel` 
405              weights matrix, used for the linear transformation of the recurrent 
406              state. Default: `"orthogonal"`. 
407          bias_initializer: Initializer for the bias vector. Default: `"zeros"`. 
408          kernel_regularizer: Regularizer function applied to the `kernel` weights 
409              matrix. Default: `None`. 
410          recurrent_regularizer: Regularizer function applied to the 
411              `recurrent_kernel` weights matrix. Default: `None`. 
412          bias_regularizer: Regularizer function applied to the bias vector. 
413              Default: `None`. 
414          activity_regularizer: Regularizer function applied to the output of the 
415              layer (its "activation"). Default: `None`. 
416          kernel_constraint: Constraint function applied to the `kernel` weights 
417              matrix. Default: `None`. 
418          recurrent_constraint: Constraint function applied to the 
419              `recurrent_kernel` weights matrix. Default: `None`. 
420          bias_constraint: Constraint function applied to the bias vector. 
421              Default: `None`. 
422          dropout: Float between 0 and 1. Fraction of the units to drop for the 
423              linear transformation of the inputs. Default: 0. 
424          recurrent_dropout: Float between 0 and 1. Fraction of the units to drop 
425              for the linear transformation of the recurrent state. Default: 0. 
426          seed: Random seed for dropout. 
427          return_sequences: Boolean. Whether to return the last output 
428              in the output sequence, or the full sequence. Default: `False`. 
429          return_state: Boolean. Whether to return the last state in addition 
430              to the output. Default: `False`. 
431          go_backwards: Boolean (default `False`). 
432              If `True`, process the input sequence backwards and return the 
433              reversed sequence. 
434          stateful: Boolean (default: `False`). If `True`, the last state 
435              for each sample at index i in a batch will be used as initial 
436              state for the sample of index i in the following batch. 
437          unroll: Boolean (default: `False`). 
438              If `True`, the network will be unrolled, 
439              else a symbolic loop will be used. 
440              Unrolling can speed-up a RNN, 
441              although it tends to be more memory-intensive. 
442              Unrolling is only suitable for short sequences. 
443          reset_after: GRU convention (whether to apply reset gate after or 
444              before matrix multiplication). `False` is `"before"`, 
445              `True` is `"after"` (default and cuDNN compatible). 
446          use_cudnn: Whether to use a cuDNN-backed implementation. `"auto"` will 
447              attempt to use cuDNN when feasible, and will fallback to the 
448              default implementation if not. 
449   
450      Call arguments: 
451          inputs: A 3D tensor, with shape `(batch, timesteps, feature)`. 
452          mask: Binary tensor of shape `(samples, timesteps)` indicating whether 
453              a given timestep should be masked  (optional). 
454              An individual `True` entry indicates that the corresponding timestep 
455              should be utilized, while a `False` entry indicates that the 
456              corresponding timestep should be ignored. Defaults to `None`. 
457          training: Python boolean indicating whether the layer should behave in 
458              training mode or in inference mode. This argument is passed to the 
459              cell when calling it. This is only relevant if `dropout` or 
460              `recurrent_dropout` is used  (optional). Defaults to `None`. 
461          initial_state: List of initial state tensors to be passed to the first 
462              call of the cell (optional, `None` causes creation 
463              of zero-filled initial state tensors). Defaults to `None`. 
464      """
465  
466      def __init__(
467          self,
468          units,
469          activation="tanh",
470          recurrent_activation="sigmoid",
471          use_bias=True,
472          kernel_initializer="glorot_uniform",
473          recurrent_initializer="orthogonal",
474          bias_initializer="zeros",
475          kernel_regularizer=None,
476          recurrent_regularizer=None,
477          bias_regularizer=None,
478          activity_regularizer=None,
479          kernel_constraint=None,
480          recurrent_constraint=None,
481          bias_constraint=None,
482          dropout=0.0,
483          recurrent_dropout=0.0,
484          seed=None,
485          return_sequences=False,
486          return_state=False,
487          go_backwards=False,
488          stateful=False,
489          unroll=False,
490          reset_after=True,
491          use_cudnn="auto",
492          **kwargs,
493      ):
494          cell = GRUCell(
495              units,
496              activation=activation,
497              recurrent_activation=recurrent_activation,
498              use_bias=use_bias,
499              kernel_initializer=kernel_initializer,
500              recurrent_initializer=recurrent_initializer,
501              bias_initializer=bias_initializer,
502              kernel_regularizer=kernel_regularizer,
503              recurrent_regularizer=recurrent_regularizer,
504              bias_regularizer=bias_regularizer,
505              kernel_constraint=kernel_constraint,
506              recurrent_constraint=recurrent_constraint,
507              bias_constraint=bias_constraint,
508              dropout=dropout,
509              recurrent_dropout=recurrent_dropout,
510              reset_after=reset_after,
511              dtype=kwargs.get("dtype", None),
512              trainable=kwargs.get("trainable", True),
513              name="gru_cell",
514              seed=seed,
515              implementation=kwargs.pop("implementation", 2),
516          )
517          super().__init__(
518              cell,
519              return_sequences=return_sequences,
520              return_state=return_state,
521              go_backwards=go_backwards,
522              stateful=stateful,
523              unroll=unroll,
524              activity_regularizer=activity_regularizer,
525              **kwargs,
526          )
527          self.input_spec = InputSpec(ndim=3)
528          if use_cudnn not in ("auto", True, False):
529              raise ValueError(
530                  "Invalid valid received for argument `use_cudnn`. "
531                  "Expected one of {'auto', True, False}. "
532                  f"Received: use_cudnn={use_cudnn}"
533              )
534          self.use_cudnn = use_cudnn
535          if (
536              backend.backend() == "tensorflow"
537              and backend.cudnn_ok(
538                  cell.activation,
539                  cell.recurrent_activation,
540                  self.unroll,
541                  cell.use_bias,
542                  reset_after=reset_after,
543              )
544              and use_cudnn in (True, "auto")
545          ):
546              self.supports_jit = False

548      def inner_loop(self, sequences, initial_state, mask, training=False):
549          if tree.is_nested(initial_state):
550              initial_state = initial_state[0]
551          if tree.is_nested(mask):
552              mask = mask[0]
553          if self.use_cudnn in ("auto", True):
554              if not self.recurrent_dropout:
555                  try:
556                      if training and self.dropout:
557                          dp_mask = self.cell.get_dropout_mask(sequences[:, 0, :])
558                          dp_mask = ops.expand_dims(dp_mask, axis=1)
559                          dp_mask = ops.broadcast_to(
560                              dp_mask, ops.shape(sequences)
561                          )
562                          dp_sequences = sequences * dp_mask
563                      else:
564                          dp_sequences = sequences
565                      # Backends are allowed to specify (optionally) optimized
566                      # implementation of the inner GRU loop. In the case of
567                      # TF for instance, it will leverage cuDNN when feasible, and
568                      # it will raise NotImplementedError otherwise.
569                      out = backend.gru(
570                          dp_sequences,
571                          initial_state,
572                          mask,
573                          kernel=self.cell.kernel,
574                          recurrent_kernel=self.cell.recurrent_kernel,
575                          bias=self.cell.bias,
576                          activation=self.cell.activation,
577                          recurrent_activation=self.cell.recurrent_activation,
578                          return_sequences=self.return_sequences,
579                          go_backwards=self.go_backwards,
580                          unroll=self.unroll,
581                          reset_after=self.cell.reset_after,
582                      )
583                      # We disable jit_compile for the model in this case,
584                      # since cuDNN ops aren't XLA compatible.
585                      if backend.backend() == "tensorflow":
586                          self.supports_jit = False
587                      return out
588                  except NotImplementedError:
589                      pass
590          if self.use_cudnn is True:
591              raise ValueError(
592                  "use_cudnn=True was specified, "
593                  "but cuDNN is not supported for this layer configuration "
594                  "with this backend. Pass use_cudnn='auto' to fallback "
595                  "to a non-cuDNN implementation."
596              )
597          return super().inner_loop(
598              sequences, initial_state, mask=mask, training=training
599          )
600  
601      def call(self, sequences, initial_state=None, mask=None, training=False):
602          return super().call(
603              sequences, mask=mask, training=training, initial_state=initial_state
604          )
