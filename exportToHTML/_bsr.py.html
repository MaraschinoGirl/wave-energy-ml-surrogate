27       def __init__(self, arg1, shape=None, dtype=None, copy=False,
28                    blocksize=None, *, maxprint=None):
29           _data_matrix.__init__(self, arg1, maxprint=maxprint)
30   
31           if issparse(arg1):
32               if arg1.format == self.format and copy:
33                   arg1 = arg1.copy()
34               else:
35                   arg1 = arg1.tobsr(blocksize=blocksize)
36               self.indptr, self.indices, self.data, self._shape = (
37                   arg1.indptr, arg1.indices, arg1.data, arg1._shape
38               )
39   
40           elif isinstance(arg1,tuple):
41               if isshape(arg1):
42                   # it's a tuple of matrix dimensions (M,N)
43                   self._shape = check_shape(arg1)
44                   M,N = self.shape
45                   # process blocksize
46                   if blocksize is None:
47                       blocksize = (1,1)
48                   else:
49                       if not isshape(blocksize):
50                           raise ValueError(f'invalid blocksize={blocksize}')
51                       blocksize = tuple(blocksize)
52                   self.data = np.zeros((0,) + blocksize, getdtype(dtype, default=float))
53   
54                   R,C = blocksize
55                   if (M % R) != 0 or (N % C) != 0:
56                       raise ValueError('shape must be multiple of blocksize')
57   
58                   # Select index dtype large enough to pass array and
59                   # scalar parameters to sparsetools
60                   idx_dtype = self._get_index_dtype(maxval=max(M//R, N//C, R, C))
61                   self.indices = np.zeros(0, dtype=idx_dtype)
62                   self.indptr = np.zeros(M//R + 1, dtype=idx_dtype)
63   
64               elif len(arg1) == 2:
65                   # (data,(row,col)) format
66                   coo = self._coo_container(arg1, dtype=dtype, shape=shape)
67                   bsr = coo.tobsr(blocksize=blocksize)
68                   self.indptr, self.indices, self.data, self._shape = (
69                       bsr.indptr, bsr.indices, bsr.data, bsr._shape
70                   )
71   
72               elif len(arg1) == 3:
73                   # (data,indices,indptr) format
74                   (data, indices, indptr) = arg1
75   
76                   # Select index dtype large enough to pass array and
77                   # scalar parameters to sparsetools
78                   maxval = 1
79                   if shape is not None:
80                       maxval = max(shape)
81                   if blocksize is not None:
82                       maxval = max(maxval, max(blocksize))
83                   idx_dtype = self._get_index_dtype((indices, indptr), maxval=maxval,
84                                                     check_contents=True)
85                   if not copy:
86                       copy = copy_if_needed
87                   self.indices = np.array(indices, copy=copy, dtype=idx_dtype)
88                   self.indptr = np.array(indptr, copy=copy, dtype=idx_dtype)
89                   self.data = getdata(data, copy=copy, dtype=dtype)
90                   if self.data.ndim != 3:
91                       raise ValueError(
92                           f'BSR data must be 3-dimensional, got shape={self.data.shape}'
93                       )
94                   if blocksize is not None:
95                       if not isshape(blocksize):
96                           raise ValueError(f'invalid blocksize={blocksize}')
97                       if tuple(blocksize) != self.data.shape[1:]:
98                           raise ValueError(
99                               f'mismatching blocksize={blocksize}'
100                              f' vs {self.data.shape[1:]}'
101                          )
102              else:
103                  raise ValueError('unrecognized bsr_array constructor usage')
104          else:
105              # must be dense
106              try:
107                  arg1 = np.asarray(arg1)
108              except Exception as e:
109                  raise ValueError("unrecognized form for "
110                                   f"{self.format}_matrix constructor") from e
111              if isinstance(self, sparray) and arg1.ndim != 2:
112                  raise ValueError(f"BSR arrays don't support {arg1.ndim}D input. Use 2D")
113              arg1 = self._coo_container(arg1, dtype=dtype).tobsr(blocksize=blocksize)
114              self.indptr, self.indices, self.data, self._shape = (
115                  arg1.indptr, arg1.indices, arg1.data, arg1._shape
116              )
117  
118          if shape is not None:
119              self._shape = check_shape(shape)
120          else:
121              if self.shape is None:
122                  # shape not already set, try to infer dimensions
123                  try:
124                      M = len(self.indptr) - 1
125                      N = self.indices.max() + 1
126                  except Exception as e:
127                      raise ValueError('unable to infer matrix dimensions') from e
128                  else:
129                      R,C = self.blocksize
130                      self._shape = check_shape((M*R,N*C))
131  
132          if self.shape is None:
133              if shape is None:
134                  # TODO infer shape here
135                  raise ValueError('need to infer shape')
136              else:
137                  self._shape = check_shape(shape)
138  
139          if dtype is not None:
140              self.data = self.data.astype(getdtype(dtype, self.data), copy=False)
141  
142          self.check_format(full_check=False)

144      def check_format(self, full_check=True):
145          """Check whether the array/matrix respects the BSR format. 
146   
147          Parameters 
148          ---------- 
149          full_check : bool, optional 
150              If `True`, run rigorous check, scanning arrays for valid values. 
151              Note that activating those check might copy arrays for casting, 
152              modifying indices and index pointers' inplace. 
153              If `False`, run basic checks on attributes. O(1) operations. 
154              Default is `True`. 
155          """
156          M,N = self.shape
157          R,C = self.blocksize
158  
159          # index arrays should have integer data types
160          if self.indptr.dtype.kind != 'i':
161              warn(f"indptr array has non-integer dtype ({self.indptr.dtype.name})",
162                   stacklevel=2)
163          if self.indices.dtype.kind != 'i':
164              warn(f"indices array has non-integer dtype ({self.indices.dtype.name})",
165                   stacklevel=2)
166  
167          # check array shapes
168          if self.indices.ndim != 1 or self.indptr.ndim != 1:
169              raise ValueError("indices, and indptr should be 1-D")
170          if self.data.ndim != 3:
171              raise ValueError("data should be 3-D")
172  
173          # check index pointer
174          if (len(self.indptr) != M//R + 1):
175              raise ValueError("index pointer size (%d) should be (%d)" %
176                                  (len(self.indptr), M//R + 1))
177          if (self.indptr[0] != 0):
178              raise ValueError("index pointer should start with 0")
179  
180          # check index and data arrays
181          if (len(self.indices) != len(self.data)):
182              raise ValueError("indices and data should have the same size")
183          if (self.indptr[-1] > len(self.indices)):
184              raise ValueError("Last value of index pointer should be less than "
185                                  "the size of index and data arrays")
186  
187          self.prune()
188  
189          if full_check:
190              # check format validity (more expensive)
191              if self.nnz > 0:
192                  if self.indices.max() >= N//C:
193                      raise ValueError("column index values must be < %d (now max %d)"
194                                       % (N//C, self.indices.max()))
195                  if self.indices.min() < 0:
196                      raise ValueError("column index values must be >= 0")
197                  if np.diff(self.indptr).min() < 0:
198                      raise ValueError("index pointer values must form a "
199                                          "non-decreasing sequence")
200  
201              idx_dtype = self._get_index_dtype((self.indices, self.indptr))
202              self.indptr = np.asarray(self.indptr, dtype=idx_dtype)
203              self.indices = np.asarray(self.indices, dtype=idx_dtype)
204              self.data = to_native(self.data)
205          # if not self.has_sorted_indices():
206          #    warn('Indices were not in sorted order. Sorting indices.')
207          #    self.sort_indices(check_first=False)
208

232      def __repr__(self):
233          _, fmt = _formats[self.format]
234          sparse_cls = 'array' if isinstance(self, sparray) else 'matrix'
235          b = 'x'.join(str(x) for x in self.blocksize)
236          return (

241      def diagonal(self, k=0):
242          rows, cols = self.shape
243          if k <= -rows or k >= cols:
244              return np.empty(0, dtype=self.data.dtype)
245          R, C = self.blocksize
246          y = np.zeros(min(rows + min(k, 0), cols - max(k, 0)),
247                       dtype=upcast(self.dtype))
248          _sparsetools.bsr_diagonal(k, rows // R, cols // C, R, C,
249                                    self.indptr, self.indices,
250                                    np.ravel(self.data), y)
251          return y

297      def _matmul_sparse(self, other):
298          M, K1 = self.shape
299          K2, N = other.shape
300  
301          R,n = self.blocksize
302  
303          # convert to this format
304          if other.format == "bsr":
305              C = other.blocksize[1]
306          else:
307              C = 1
308  
309          if other.format == "csr" and n == 1:
310              other = other.tobsr(blocksize=(n,C), copy=False)  # lightweight conversion
311          else:
312              other = other.tobsr(blocksize=(n,C))
313  
314          idx_dtype = self._get_index_dtype((self.indptr, self.indices,
315                                             other.indptr, other.indices))
316  
317          bnnz = csr_matmat_maxnnz(M//R, N//C,
318                                   self.indptr.astype(idx_dtype),
319                                   self.indices.astype(idx_dtype),
320                                   other.indptr.astype(idx_dtype),
321                                   other.indices.astype(idx_dtype))
322  
323          idx_dtype = self._get_index_dtype((self.indptr, self.indices,
324                                             other.indptr, other.indices),
325                                            maxval=bnnz)
326          indptr = np.empty(self.indptr.shape, dtype=idx_dtype)
327          indices = np.empty(bnnz, dtype=idx_dtype)
328          data = np.empty(R*C*bnnz, dtype=upcast(self.dtype,other.dtype))
329  
330          bsr_matmat(bnnz, M//R, N//C, R, C, n,
331                     self.indptr.astype(idx_dtype),
332                     self.indices.astype(idx_dtype),
333                     np.ravel(self.data),
334                     other.indptr.astype(idx_dtype),
335                     other.indices.astype(idx_dtype),
336                     np.ravel(other.data),
337                     indptr,
338                     indices,
339                     data)
340  
341          data = data.reshape(-1,R,C)
342  
343          # TODO eliminate zeros
344  
345          return self._bsr_container(
346              (data, indices, indptr), shape=(M, N), blocksize=(R, C)
347          )

353      def tobsr(self, blocksize=None, copy=False):
354          """Convert this array/matrix into Block Sparse Row Format. 
355   
356          With copy=False, the data/indices may be shared between this 
357          array/matrix and the resultant bsr_array/bsr_matrix. 
358   
359          If blocksize=(R, C) is provided, it will be used for determining 
360          block size of the bsr_array/bsr_matrix. 
361          """
362          if blocksize not in [None, self.blocksize]:
363              return self.tocsr().tobsr(blocksize=blocksize)
364          if copy:
365              return self.copy()
366          else:
367              return self
368  
369      def tocsr(self, copy=False):
370          M, N = self.shape
371          R, C = self.blocksize
372          nnz = self.nnz
373          idx_dtype = self._get_index_dtype((self.indptr, self.indices),
374                                            maxval=max(nnz, N))
375          indptr = np.empty(M + 1, dtype=idx_dtype)
376          indices = np.empty(nnz, dtype=idx_dtype)
377          data = np.empty(nnz, dtype=upcast(self.dtype))
378  
379          bsr_tocsr(M // R,  # n_brow
380                    N // C,  # n_bcol
381                    R, C,
382                    self.indptr.astype(idx_dtype, copy=False),
383                    self.indices.astype(idx_dtype, copy=False),
384                    self.data,
385                    indptr,
386                    indices,
387                    data)
388          return self._csr_container((data, indices, indptr), shape=self.shape)

397      def tocoo(self, copy=True):
398          """Convert this array/matrix to COOrdinate format. 
399   
400          When copy=False the data array will be shared between 
401          this array/matrix and the resultant coo_array/coo_matrix. 
402          """
403  
404          M,N = self.shape
405          R,C = self.blocksize
406  
407          indptr_diff = np.diff(self.indptr)
408          if indptr_diff.dtype.itemsize > np.dtype(np.intp).itemsize:
409              # Check for potential overflow
410              indptr_diff_limited = indptr_diff.astype(np.intp)
411              if np.any(indptr_diff_limited != indptr_diff):
412                  raise ValueError("Matrix too big to convert")
413              indptr_diff = indptr_diff_limited
414  
415          idx_dtype = self._get_index_dtype(maxval=max(M, N))
416          row = (R * np.arange(M//R, dtype=idx_dtype)).repeat(indptr_diff)
417          row = row.repeat(R*C).reshape(-1,R,C)
418          row += np.tile(np.arange(R, dtype=idx_dtype).reshape(-1,1), (1,C))
419          row = row.reshape(-1)
420  
421          col = ((C * self.indices).astype(idx_dtype, copy=False)
422                 .repeat(R*C).reshape(-1,R,C))
423          col += np.tile(np.arange(C, dtype=idx_dtype), (R,1))
424          col = col.reshape(-1)
425  
426          data = self.data.reshape(-1)
427  
428          if copy:
429              data = data.copy()
430  
431          return self._coo_container(
432              (data, (row, col)), shape=self.shape
433          )

440      def transpose(self, axes=None, copy=False):
441          if axes is not None and axes != (1, 0):
442              raise ValueError("Sparse matrices do not support "
443                                "an 'axes' parameter because swapping "
444                                "dimensions is the only logical permutation.")
445  
446          R, C = self.blocksize
447          M, N = self.shape
448          NBLK = self.nnz//(R*C)
449  
450          if self.nnz == 0:
451              return self._bsr_container((N, M), blocksize=(C, R),
452                                         dtype=self.dtype, copy=copy)
453  
454          indptr = np.empty(N//C + 1, dtype=self.indptr.dtype)
455          indices = np.empty(NBLK, dtype=self.indices.dtype)
456          data = np.empty((NBLK, C, R), dtype=self.data.dtype)
457  
458          bsr_transpose(M//R, N//C, R, C,
459                        self.indptr, self.indices, self.data.ravel(),
460                        indptr, indices, data.ravel())
461  
462          return self._bsr_container((data, indices, indptr),
463                                     shape=(N, M), copy=copy)

471      def eliminate_zeros(self):
472          """Remove zero elements in-place."""
473  
474          if not self.nnz:
475              return  # nothing to do
476  
477          R,C = self.blocksize
478          M,N = self.shape
479  
480          mask = (self.data != 0).reshape(-1,R*C).sum(axis=1)  # nonzero blocks
481  
482          nonzero_blocks = mask.nonzero()[0]
483  
484          self.data[:len(nonzero_blocks)] = self.data[nonzero_blocks]
485  
486          # modifies self.indptr and self.indices *in place*
487          _sparsetools.csr_eliminate_zeros(M//R, N//C, self.indptr,
488                                           self.indices, mask)
489          self.prune()
490  
491      def sum_duplicates(self):
492          """Eliminate duplicate array/matrix entries by adding them together 
493   
494          The is an *in place* operation 
495          """
496          if self.has_canonical_format:
497              return
498          self.sort_indices()
499          R, C = self.blocksize
500          M, N = self.shape
501  
502          # port of _sparsetools.csr_sum_duplicates
503          n_row = M // R
504          nnz = 0
505          row_end = 0
506          for i in range(n_row):
507              jj = row_end
508              row_end = self.indptr[i+1]
509              while jj < row_end:
510                  j = self.indices[jj]
511                  x = self.data[jj]
512                  jj += 1
513                  while jj < row_end and self.indices[jj] == j:
514                      x += self.data[jj]
515                      jj += 1
516                  self.indices[nnz] = j
517                  self.data[nnz] = x
518                  nnz += 1
519              self.indptr[i+1] = nnz
520  
521          self.prune()  # nnz may have changed
522          self.has_canonical_format = True
523  
524      def sort_indices(self):
525          """Sort the indices of this array/matrix *in place* 
526          """
527          if self.has_sorted_indices:
528              return
529  
530          R,C = self.blocksize
531          M,N = self.shape
532  
533          bsr_sort_indices(M//R, N//C, R, C, self.indptr, self.indices, self.data.ravel())
534  
535          self.has_sorted_indices = True
536  
537      def prune(self):
538          """Remove empty space after all non-zero elements. 
539          """
540  
541          R,C = self.blocksize
542          M,N = self.shape
543  
544          if len(self.indptr) != M//R + 1:
545              raise ValueError("index pointer has invalid length")
546  
547          bnnz = self.indptr[-1]
548  
549          if len(self.indices) < bnnz:
550              raise ValueError("indices array has too few elements")
551          if len(self.data) < bnnz:
552              raise ValueError("data array has too few elements")
553  
554          self.data = self.data[:bnnz]
555          self.indices = self.indices[:bnnz]

557      # utility functions
558      def _binopt(self, other, op, in_shape=None, out_shape=None):
559          """Apply the binary operation fn to two sparse matrices."""
560  
561          # Ideally we'd take the GCDs of the blocksize dimensions
562          # and explode self and other to match.
563          other = self.__class__(other, blocksize=self.blocksize)
564  
565          # e.g. bsr_plus_bsr, etc.
566          fn = getattr(_sparsetools, self.format + op + self.format)
567  
568          R,C = self.blocksize
569  
570          max_bnnz = len(self.data) + len(other.data)
571          idx_dtype = self._get_index_dtype((self.indptr, self.indices,
572                                             other.indptr, other.indices),
573                                            maxval=max_bnnz)
574          indptr = np.empty(self.indptr.shape, dtype=idx_dtype)
575          indices = np.empty(max_bnnz, dtype=idx_dtype)
576  
577          bool_ops = ['_ne_', '_lt_', '_gt_', '_le_', '_ge_']
578          if op in bool_ops:
579              data = np.empty(R*C*max_bnnz, dtype=np.bool_)
580          else:
581              data = np.empty(R*C*max_bnnz, dtype=upcast(self.dtype,other.dtype))
582  
583          fn(self.shape[0]//R, self.shape[1]//C, R, C,
584             self.indptr.astype(idx_dtype),
585             self.indices.astype(idx_dtype),
586             self.data,
587             other.indptr.astype(idx_dtype),
588             other.indices.astype(idx_dtype),
589             np.ravel(other.data),
590             indptr,
591             indices,
592             data)
593  
594          actual_bnnz = indptr[-1]
595          indices = indices[:actual_bnnz]
596          data = data[:R*C*actual_bnnz]
597  
598          if actual_bnnz < max_bnnz/2:
599              indices = indices.copy()
600              data = data.copy()
601  
602          data = data.reshape(-1,R,C)
603  
604          return self.__class__((data, indices, indptr), shape=self.shape)

606      # needed by _data_matrix
607      def _with_data(self,data,copy=True):
608          """Returns a matrix with the same sparsity structure as self, 
609          but with different data.  By default the structure arrays 
610          (i.e. .indptr and .indices) are copied. 
611          """
612          if copy:
613              return self.__class__((data,self.indices.copy(),self.indptr.copy()),
614                                     shape=self.shape,dtype=data.dtype)
615          else:
616              return self.__class__((data,self.indices,self.indptr),
617                                     shape=self.shape,dtype=data.dtype)

657  class bsr_array(_bsr_base, sparray):
658      """ 
659      Block Sparse Row format sparse array. 
660   
661      This can be instantiated in several ways: 
662          bsr_array(D, [blocksize=(R,C)]) 
663              where D is a 2-D ndarray. 
664   
665          bsr_array(S, [blocksize=(R,C)]) 
666              with another sparse array or matrix S (equivalent to S.tobsr()) 
667   
668          bsr_array((M, N), [blocksize=(R,C), dtype]) 
669              to construct an empty sparse array with shape (M, N) 
670              dtype is optional, defaulting to dtype='d'. 
671   
672          bsr_array((data, ij), [blocksize=(R,C), shape=(M, N)]) 
673              where ``data`` and ``ij`` satisfy ``a[ij[0, k], ij[1, k]] = data[k]`` 
674   
675          bsr_array((data, indices, indptr), [shape=(M, N)]) 
676              is the standard BSR representation where the block column 
677              indices for row i are stored in ``indices[indptr[i]:indptr[i+1]]`` 
678              and their corresponding block values are stored in 
679              ``data[ indptr[i]: indptr[i+1] ]``. If the shape parameter is not 
680              supplied, the array dimensions are inferred from the index arrays. 
681   
682      Attributes 
683      ---------- 
684      dtype : dtype 
685          Data type of the array 
686      shape : 2-tuple 
687          Shape of the array 
688      ndim : int 
689          Number of dimensions (this is always 2) 
690      nnz 
691      size 
692      data 
693          BSR format data array of the array 
694      indices 
695          BSR format index array of the array 
696      indptr 
697          BSR format index pointer array of the array 
698      blocksize 
699          Block size 
700      has_sorted_indices : bool 
701          Whether indices are sorted 
702      has_canonical_format : bool 
703      T 
704   
705      Notes 
706      ----- 
707      Sparse arrays can be used in arithmetic operations: they support 
708      addition, subtraction, multiplication, division, and matrix power. 
709   
710      **Summary of BSR format** 
711   
712      The Block Sparse Row (BSR) format is very similar to the Compressed 
713      Sparse Row (CSR) format. BSR is appropriate for sparse matrices with dense 
714      sub matrices like the last example below. Such sparse block matrices often 
715      arise in vector-valued finite element discretizations. In such cases, BSR is 
716      considerably more efficient than CSR and CSC for many sparse arithmetic 
717      operations. 
718   
719      **Blocksize** 
720   
721      The blocksize (R,C) must evenly divide the shape of the sparse array (M,N). 
722      That is, R and C must satisfy the relationship ``M % R = 0`` and 
723      ``N % C = 0``. 
724   
725      If no blocksize is specified, a simple heuristic is applied to determine 
726      an appropriate blocksize. 
727   
728      **Canonical Format** 
729   
730      In canonical format, there are no duplicate blocks and indices are sorted 
731      per row. 
732   
733      **Limitations** 
734   
735      Block Sparse Row format sparse arrays do not support slicing. 
736   
737      Examples 
738      -------- 
739      >>> import numpy as np 
740      >>> from scipy.sparse import bsr_array 
741      >>> bsr_array((3, 4), dtype=np.int8).toarray() 
742      array([[0, 0, 0, 0], 
743             [0, 0, 0, 0], 
744             [0, 0, 0, 0]], dtype=int8) 
745   
746      >>> row = np.array([0, 0, 1, 2, 2, 2]) 
747      >>> col = np.array([0, 2, 2, 0, 1, 2]) 
748      >>> data = np.array([1, 2, 3 ,4, 5, 6]) 
749      >>> bsr_array((data, (row, col)), shape=(3, 3)).toarray() 
750      array([[1, 0, 2], 
751             [0, 0, 3], 
752             [4, 5, 6]]) 
753   
754      >>> indptr = np.array([0, 2, 3, 6]) 
755      >>> indices = np.array([0, 2, 2, 0, 1, 2]) 
756      >>> data = np.array([1, 2, 3, 4, 5, 6]).repeat(4).reshape(6, 2, 2) 
757      >>> bsr_array((data,indices,indptr), shape=(6, 6)).toarray() 
758      array([[1, 1, 0, 0, 2, 2], 
759             [1, 1, 0, 0, 2, 2], 
760             [0, 0, 0, 0, 3, 3], 
761             [0, 0, 0, 0, 3, 3], 
762             [4, 4, 5, 5, 6, 6], 
763             [4, 4, 5, 5, 6, 6]])
