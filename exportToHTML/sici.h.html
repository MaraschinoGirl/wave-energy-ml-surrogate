5    /*                                                     sici.c 
6     * 
7     *     Sine and cosine integrals 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double x, Ci, Si, sici(); 
14    * 
15    * sici( x, &Si, &Ci ); 
16    * 
17    * 
18    * DESCRIPTION: 
19    * 
20    * Evaluates the integrals 
21    * 
22    *                          x 
23    *                          - 
24    *                         |  cos t - 1 
25    *   Ci(x) = eul + ln x +  |  --------- dt, 
26    *                         |      t 
27    *                        - 
28    *                         0 
29    *             x 
30    *             - 
31    *            |  sin t 
32    *   Si(x) =  |  ----- dt 
33    *            |    t 
34    *           - 
35    *            0 
36    * 
37    * where eul = 0.57721566490153286061 is Euler's constant. 
38    * The integrals are approximated by rational functions. 
39    * For x > 8 auxiliary functions f(x) and g(x) are employed 
40    * such that 
41    * 
42    * Ci(x) = f(x) sin(x) - g(x) cos(x) 
43    * Si(x) = pi/2 - f(x) cos(x) - g(x) sin(x) 
44    * 
45    * 
46    * ACCURACY: 
47    *    Test interval = [0,50]. 
48    * Absolute error, except relative when > 1: 
49    * arithmetic   function   # trials      peak         rms 
50    *    IEEE        Si        30000       4.4e-16     7.3e-17 
51    *    IEEE        Ci        30000       6.9e-16     5.1e-17 
52    */

141      XSF_HOST_DEVICE inline int sici(double x, double *si, double *ci) {
142          double z, c, s, f, g;
143          short sign;
144  
145          if (x < 0.0) {
146              sign = -1;
147              x = -x;
148          } else {
149              sign = 0;
150          }
151  
152          if (x == 0.0) {
153              *si = 0.0;
154              *ci = -std::numeric_limits<double>::infinity();
155              return (0);
156          }
157  
158          if (x > 1.0e9) {
159              if (std::isinf(x)) {
160                  if (sign == -1) {
161                      *si = -M_PI_2;
162                      *ci = std::numeric_limits<double>::quiet_NaN();
163                  } else {
164                      *si = M_PI_2;
165                      *ci = 0;
166                  }
167                  return 0;
168              }
169              *si = M_PI_2 - std::cos(x) / x;
170              *ci = std::sin(x) / x;
171          }
172  
173          if (x > 4.0) {
174              goto asympt;
175          }
176  
177          z = x * x;
178          s = x * polevl(z, detail::sici_SN, 5) / polevl(z, detail::sici_SD, 5);
179          c = z * polevl(z, detail::sici_CN, 5) / polevl(z, detail::sici_CD, 5);
180  
181          if (sign) {
182              s = -s;
183          }
184          *si = s;
185          *ci = detail::SCIPY_EULER + std::log(x) + c; /* real part if x < 0 */
186          return (0);
187  
188          /* The auxiliary functions are: 
189           * 
190           * 
191           * *si = *si - M_PI_2; 
192           * c = cos(x); 
193           * s = sin(x); 
194           * 
195           * t = *ci * s - *si * c; 
196           * a = *ci * c + *si * s; 
197           * 
198           * *si = t; 
199           * *ci = -a; 
200           */
201  
202      asympt:
203  
204          s = std::sin(x);
205          c = std::cos(x);
206          z = 1.0 / (x * x);
207          if (x < 8.0) {
208              f = polevl(z, detail::sici_FN4, 6) / (x * p1evl(z, detail::sici_FD4, 7));
209              g = z * polevl(z, detail::sici_GN4, 7) / p1evl(z, detail::sici_GD4, 7);
210          } else {
211              f = polevl(z, detail::sici_FN8, 8) / (x * p1evl(z, detail::sici_FD8, 8));
212              g = z * polevl(z, detail::sici_GN8, 8) / p1evl(z, detail::sici_GD8, 9);
213          }
214          *si = M_PI_2 - f * c - g * s;
215          if (sign) {
216              *si = -(*si);
217          }
218          *ci = f * s - g * c;
219  
220          return (0);
221      }
