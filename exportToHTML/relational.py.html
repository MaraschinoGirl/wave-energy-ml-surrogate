198  class _LinePlotter(_RelationalPlotter):
199  
200      _legend_attributes = ["color", "linewidth", "marker", "dashes"]
201  
202      def __init__(
203          self, *,
204          data=None, variables={},
205          estimator=None, n_boot=None, seed=None, errorbar=None,
206          sort=True, orient="x", err_style=None, err_kws=None, legend=None
207      ):
208  
209          # TODO this is messy, we want the mapping to be agnostic about
210          # the kind of plot to draw, but for the time being we need to set
211          # this information so the SizeMapping can use it
212          self._default_size_range = (
213              np.r_[.5, 2] * mpl.rcParams["lines.linewidth"]
214          )
215  
216          super().__init__(data=data, variables=variables)
217  
218          self.estimator = estimator
219          self.errorbar = errorbar
220          self.n_boot = n_boot
221          self.seed = seed
222          self.sort = sort
223          self.orient = orient
224          self.err_style = err_style
225          self.err_kws = {} if err_kws is None else err_kws
226  
227          self.legend = legend
228  
229      def plot(self, ax, kws):
230          """Draw the plot onto an axes, passing matplotlib kwargs."""
231  
232          # Draw a test plot, using the passed in kwargs. The goal here is to
233          # honor both (a) the current state of the plot cycler and (b) the
234          # specified kwargs on all the lines we will draw, overriding when
235          # relevant with the data semantics. Note that we won't cycle
236          # internally; in other words, if `hue` is not used, all elements will
237          # have the same color, but they will have the color that you would have
238          # gotten from the corresponding matplotlib function, and calling the
239          # function will advance the axes property cycle.
240  
241          kws = normalize_kwargs(kws, mpl.lines.Line2D)
242          kws.setdefault("markeredgewidth", 0.75)
243          kws.setdefault("markeredgecolor", "w")
244  
245          # Set default error kwargs
246          err_kws = self.err_kws.copy()
247          if self.err_style == "band":
248              err_kws.setdefault("alpha", .2)
249          elif self.err_style == "bars":
250              pass
251          elif self.err_style is not None:
252              err = "`err_style` must be 'band' or 'bars', not {}"
253              raise ValueError(err.format(self.err_style))
254  
255          # Initialize the aggregation object
256          weighted = "weight" in self.plot_data
257          agg = (WeightedAggregator if weighted else EstimateAggregator)(
258              self.estimator, self.errorbar, n_boot=self.n_boot, seed=self.seed,
259          )
260  
261          # TODO abstract variable to aggregate over here-ish. Better name?
262          orient = self.orient
263          if orient not in {"x", "y"}:
264              err = f"`orient` must be either 'x' or 'y', not {orient!r}."
265              raise ValueError(err)
266          other = {"x": "y", "y": "x"}[orient]
267  
268          # TODO How to handle NA? We don't want NA to propagate through to the
269          # estimate/CI when some values are present, but we would also like
270          # matplotlib to show "gaps" in the line when all values are missing.
271          # This is straightforward absent aggregation, but complicated with it.
272          # If we want to use nas, we need to conditionalize dropna in iter_data.
273  
274          # Loop over the semantic subsets and add to the plot
275          grouping_vars = "hue", "size", "style"
276          for sub_vars, sub_data in self.iter_data(grouping_vars, from_comp_data=True):
277  
278              if self.sort:
279                  sort_vars = ["units", orient, other]
280                  sort_cols = [var for var in sort_vars if var in self.variables]
281                  sub_data = sub_data.sort_values(sort_cols)
282  
283              if (
284                  self.estimator is not None
285                  and sub_data[orient].value_counts().max() > 1
286              ):
287                  if "units" in self.variables:
288                      # TODO eventually relax this constraint
289                      err = "estimator must be None when specifying units"
290                      raise ValueError(err)
291                  grouped = sub_data.groupby(orient, sort=self.sort)
292                  # Could pass as_index=False instead of reset_index,
293                  # but that fails on a corner case with older pandas.
294                  sub_data = (
295                      grouped
296                      .apply(agg, other, **groupby_apply_include_groups(False))
297                      .reset_index()
298                  )
299              else:
300                  sub_data[f"{other}min"] = np.nan
301                  sub_data[f"{other}max"] = np.nan
302  
303              # Apply inverse axis scaling
304              for var in "xy":
305                  _, inv = _get_transform_functions(ax, var)
306                  for col in sub_data.filter(regex=f"^{var}"):
307                      sub_data[col] = inv(sub_data[col])
308  
309              # --- Draw the main line(s)
310  
311              if "units" in self.variables:   # XXX why not add to grouping variables?
312                  lines = []
313                  for _, unit_data in sub_data.groupby("units"):
314                      lines.extend(ax.plot(unit_data["x"], unit_data["y"], **kws))
315              else:
316                  lines = ax.plot(sub_data["x"], sub_data["y"], **kws)
317  
318              for line in lines:
319  
320                  if "hue" in sub_vars:
321                      line.set_color(self._hue_map(sub_vars["hue"]))
322  
323                  if "size" in sub_vars:
324                      line.set_linewidth(self._size_map(sub_vars["size"]))
325  
326                  if "style" in sub_vars:
327                      attributes = self._style_map(sub_vars["style"])
328                      if "dashes" in attributes:
329                          line.set_dashes(attributes["dashes"])
330                      if "marker" in attributes:
331                          line.set_marker(attributes["marker"])
332  
333              line_color = line.get_color()
334              line_alpha = line.get_alpha()
335              line_capstyle = line.get_solid_capstyle()
336  
337              # --- Draw the confidence intervals
338  
339              if self.estimator is not None and self.errorbar is not None:
340  
341                  # TODO handling of orientation will need to happen here
342  
343                  if self.err_style == "band":
344  
345                      func = {"x": ax.fill_between, "y": ax.fill_betweenx}[orient]
346                      func(
347                          sub_data[orient],
348                          sub_data[f"{other}min"], sub_data[f"{other}max"],
349                          color=line_color, **err_kws
350                      )
351  
352                  elif self.err_style == "bars":
353  
354                      error_param = {
355                          f"{other}err": (
356                              sub_data[other] - sub_data[f"{other}min"],
357                              sub_data[f"{other}max"] - sub_data[other],
358                          )
359                      }
360                      ebars = ax.errorbar(
361                          sub_data["x"], sub_data["y"], **error_param,
362                          linestyle="", color=line_color, alpha=line_alpha,
363                          **err_kws
364                      )
365  
366                      # Set the capstyle properly on the error bars
367                      for obj in ebars.get_children():
368                          if isinstance(obj, mpl.collections.LineCollection):
369                              obj.set_capstyle(line_capstyle)
370  
371          # Finalize the axes details
372          self._add_axis_labels(ax)
373          if self.legend:
374              legend_artist = partial(mpl.lines.Line2D, xdata=[], ydata=[])
375              attrs = {"hue": "color", "size": "linewidth", "style": None}
376              self.add_legend_data(ax, legend_artist, kws, attrs)
377              handles, _ = ax.get_legend_handles_labels()
378              if handles:
379                  legend = ax.legend(title=self.legend_title)
380                  adjust_legend_subtitles(legend)

383  class _ScatterPlotter(_RelationalPlotter):
384  
385      _legend_attributes = ["color", "s", "marker"]
386  
387      def __init__(self, *, data=None, variables={}, legend=None):
388  
389          # TODO this is messy, we want the mapping to be agnostic about
390          # the kind of plot to draw, but for the time being we need to set
391          # this information so the SizeMapping can use it
392          self._default_size_range = (
393              np.r_[.5, 2] * np.square(mpl.rcParams["lines.markersize"])
394          )
395  
396          super().__init__(data=data, variables=variables)
397  
398          self.legend = legend
399  
400      def plot(self, ax, kws):
401  
402          # --- Determine the visual attributes of the plot
403  
404          data = self.comp_data.dropna()
405          if data.empty:
406              return
407  
408          kws = normalize_kwargs(kws, mpl.collections.PathCollection)
409  
410          # Define the vectors of x and y positions
411          empty = np.full(len(data), np.nan)
412          x = data.get("x", empty)
413          y = data.get("y", empty)
414  
415          # Apply inverse scaling to the coordinate variables
416          _, inv_x = _get_transform_functions(ax, "x")
417          _, inv_y = _get_transform_functions(ax, "y")
418          x, y = inv_x(x), inv_y(y)
419  
420          if "style" in self.variables:
421              # Use a representative marker so scatter sets the edgecolor
422              # properly for line art markers. We currently enforce either
423              # all or none line art so this works.
424              example_level = self._style_map.levels[0]
425              example_marker = self._style_map(example_level, "marker")
426              kws.setdefault("marker", example_marker)
427  
428          # Conditionally set the marker edgecolor based on whether the marker is "filled"
429          # See https://github.com/matplotlib/matplotlib/issues/17849 for context
430          m = kws.get("marker", mpl.rcParams.get("marker", "o"))
431          if not isinstance(m, mpl.markers.MarkerStyle):
432              # TODO in more recent matplotlib (which?) can pass a MarkerStyle here
433              m = mpl.markers.MarkerStyle(m)
434          if m.is_filled():
435              kws.setdefault("edgecolor", "w")
436  
437          # Draw the scatter plot
438          points = ax.scatter(x=x, y=y, **kws)
439  
440          # Apply the mapping from semantic variables to artist attributes
441  
442          if "hue" in self.variables:
443              points.set_facecolors(self._hue_map(data["hue"]))
444  
445          if "size" in self.variables:
446              points.set_sizes(self._size_map(data["size"]))
447  
448          if "style" in self.variables:
449              p = [self._style_map(val, "path") for val in data["style"]]
450              points.set_paths(p)
451  
452          # Apply dependent default attributes
453  
454          if "linewidth" not in kws:
455              sizes = points.get_sizes()
456              linewidth = .08 * np.sqrt(np.percentile(sizes, 10))
457              points.set_linewidths(linewidth)
458              kws["linewidth"] = linewidth
459  
460          # Finalize the axes details
461          self._add_axis_labels(ax)
462          if self.legend:
463              attrs = {"hue": "color", "size": "s", "style": None}
464              self.add_legend_data(ax, _scatter_legend_artist, kws, attrs)
465              handles, _ = ax.get_legend_handles_labels()
466              if handles:
467                  legend = ax.legend(title=self.legend_title)
468                  adjust_legend_subtitles(legend)

471  def lineplot(
472      data=None, *,
473      x=None, y=None, hue=None, size=None, style=None, units=None, weights=None,
474      palette=None, hue_order=None, hue_norm=None,
475      sizes=None, size_order=None, size_norm=None,
476      dashes=True, markers=None, style_order=None,
477      estimator="mean", errorbar=("ci", 95), n_boot=1000, seed=None,
478      orient="x", sort=True, err_style="band", err_kws=None,
479      legend="auto", ci="deprecated", ax=None, **kwargs
480  ):
481  
482      # Handle deprecation of ci parameter
483      errorbar = _deprecate_ci(errorbar, ci)
484  
485      p = _LinePlotter(
486          data=data,
487          variables=dict(
488              x=x, y=y, hue=hue, size=size, style=style, units=units, weight=weights
489          ),
490          estimator=estimator, n_boot=n_boot, seed=seed, errorbar=errorbar,
491          sort=sort, orient=orient, err_style=err_style, err_kws=err_kws,
492          legend=legend,
493      )
494  
495      p.map_hue(palette=palette, order=hue_order, norm=hue_norm)
496      p.map_size(sizes=sizes, order=size_order, norm=size_norm)
497      p.map_style(markers=markers, dashes=dashes, order=style_order)
498  
499      if ax is None:
500          ax = plt.gca()
501  
502      if "style" not in p.variables and not {"ls", "linestyle"} & set(kwargs):  # XXX
503          kwargs["dashes"] = "" if dashes is None or isinstance(dashes, bool) else dashes
504  
505      if not p.has_xy_data:
506          return ax
507  
508      p._attach(ax)
509  
510      # Other functions have color as an explicit param,
511      # and we should probably do that here too
512      color = kwargs.pop("color", kwargs.pop("c", None))
513      kwargs["color"] = _default_color(ax.plot, hue, color, kwargs)
514  
515      p.plot(ax, kwargs)
516      return ax

606  def scatterplot(
607      data=None, *,
608      x=None, y=None, hue=None, size=None, style=None,
609      palette=None, hue_order=None, hue_norm=None,
610      sizes=None, size_order=None, size_norm=None,
611      markers=True, style_order=None, legend="auto", ax=None,
612      **kwargs
613  ):
614  
615      p = _ScatterPlotter(
616          data=data,
617          variables=dict(x=x, y=y, hue=hue, size=size, style=style),
618          legend=legend
619      )
620  
621      p.map_hue(palette=palette, order=hue_order, norm=hue_norm)
622      p.map_size(sizes=sizes, order=size_order, norm=size_norm)
623      p.map_style(markers=markers, order=style_order)
624  
625      if ax is None:
626          ax = plt.gca()
627  
628      if not p.has_xy_data:
629          return ax
630  
631      p._attach(ax)
632  
633      color = kwargs.pop("color", None)
634      kwargs["color"] = _default_color(ax.scatter, hue, color, kwargs)
635  
636      p.plot(ax, kwargs)

700  def relplot(
701      data=None, *,
702      x=None, y=None, hue=None, size=None, style=None, units=None, weights=None,
703      row=None, col=None, col_wrap=None, row_order=None, col_order=None,
704      palette=None, hue_order=None, hue_norm=None,
705      sizes=None, size_order=None, size_norm=None,
706      markers=None, dashes=None, style_order=None,
707      legend="auto", kind="scatter", height=5, aspect=1, facet_kws=None,
708      **kwargs
709  ):
710  
711      if kind == "scatter":
712  
713          Plotter = _ScatterPlotter
714          func = scatterplot
715          markers = True if markers is None else markers
716  
717      elif kind == "line":
718  
719          Plotter = _LinePlotter
720          func = lineplot
721          dashes = True if dashes is None else dashes
722  
723      else:
724          err = f"Plot kind {kind} not recognized"
725          raise ValueError(err)
726  
727      # Check for attempt to plot onto specific axes and warn
728      if "ax" in kwargs:
729          msg = (
730              "relplot is a figure-level function and does not accept "
731              "the `ax` parameter. You may wish to try {}".format(kind + "plot")
732          )
733          warnings.warn(msg, UserWarning)
734          kwargs.pop("ax")
735  
736      # Use the full dataset to map the semantics
737      variables = dict(x=x, y=y, hue=hue, size=size, style=style)
738      if kind == "line":
739          variables["units"] = units
740          variables["weight"] = weights
741      else:
742          if units is not None:
743              msg = "The `units` parameter has no effect with kind='scatter'."
744              warnings.warn(msg, stacklevel=2)
745          if weights is not None:
746              msg = "The `weights` parameter has no effect with kind='scatter'."
747              warnings.warn(msg, stacklevel=2)
748      p = Plotter(
749          data=data,
750          variables=variables,
751          legend=legend,
752      )
753      p.map_hue(palette=palette, order=hue_order, norm=hue_norm)
754      p.map_size(sizes=sizes, order=size_order, norm=size_norm)
755      p.map_style(markers=markers, dashes=dashes, order=style_order)
756  
757      # Extract the semantic mappings
758      if "hue" in p.variables:
759          palette = p._hue_map.lookup_table
760          hue_order = p._hue_map.levels
761          hue_norm = p._hue_map.norm
762      else:
763          palette = hue_order = hue_norm = None
764  
765      if "size" in p.variables:
766          sizes = p._size_map.lookup_table
767          size_order = p._size_map.levels
768          size_norm = p._size_map.norm
769  
770      if "style" in p.variables:
771          style_order = p._style_map.levels
772          if markers:
773              markers = {k: p._style_map(k, "marker") for k in style_order}
774          else:
775              markers = None
776          if dashes:
777              dashes = {k: p._style_map(k, "dashes") for k in style_order}
778          else:
779              dashes = None
780      else:
781          markers = dashes = style_order = None
782  
783      # Now extract the data that would be used to draw a single plot
784      variables = p.variables
785      plot_data = p.plot_data
786  
787      # Define the common plotting parameters
788      plot_kws = dict(
789          palette=palette, hue_order=hue_order, hue_norm=hue_norm,
790          sizes=sizes, size_order=size_order, size_norm=size_norm,
791          markers=markers, dashes=dashes, style_order=style_order,
792          legend=False,
793      )
794      plot_kws.update(kwargs)
795      if kind == "scatter":
796          plot_kws.pop("dashes")
797  
798      # Add the grid semantics onto the plotter
799      grid_variables = dict(
800          x=x, y=y, row=row, col=col, hue=hue, size=size, style=style,
801      )
802      if kind == "line":
803          grid_variables.update(units=units, weights=weights)
804      p.assign_variables(data, grid_variables)
805  
806      # Define the named variables for plotting on each facet
807      # Rename the variables with a leading underscore to avoid
808      # collisions with faceting variable names
809      plot_variables = {v: f"_{v}" for v in variables}
810      if "weight" in plot_variables:
811          plot_variables["weights"] = plot_variables.pop("weight")
812      plot_kws.update(plot_variables)
813  
814      # Pass the row/col variables to FacetGrid with their original
815      # names so that the axes titles render correctly
816      for var in ["row", "col"]:
817          # Handle faceting variables that lack name information
818          if var in p.variables and p.variables[var] is None:
819              p.variables[var] = f"_{var}_"
820      grid_kws = {v: p.variables.get(v) for v in ["row", "col"]}
821  
822      # Rename the columns of the plot_data structure appropriately
823      new_cols = plot_variables.copy()
824      new_cols.update(grid_kws)
825      full_data = p.plot_data.rename(columns=new_cols)
826  
827      # Set up the FacetGrid object
828      facet_kws = {} if facet_kws is None else facet_kws.copy()
829      g = FacetGrid(
830          data=full_data.dropna(axis=1, how="all"),
831          **grid_kws,
832          col_wrap=col_wrap, row_order=row_order, col_order=col_order,
833          height=height, aspect=aspect, dropna=False,
834          **facet_kws
835      )
836  
837      # Draw the plot
838      g.map_dataframe(func, **plot_kws)
839  
840      # Label the axes, using the original variables
841      # Pass "" when the variable name is None to overwrite internal variables
842      g.set_axis_labels(variables.get("x") or "", variables.get("y") or "")
843  
844      if legend:
845          # Replace the original plot data so the legend uses numeric data with
846          # the correct type, since we force a categorical mapping above.
847          p.plot_data = plot_data
848  
849          # Handle the additional non-semantic keyword arguments out here.
850          # We're selective because some kwargs may be seaborn function specific
851          # and not relevant to the matplotlib artists going into the legend.
852          # Ideally, we will have a better solution where we don't need to re-make
853          # the legend out here and will have parity with the axes-level functions.
854          keys = ["c", "color", "alpha", "m", "marker"]
855          if kind == "scatter":
856              legend_artist = _scatter_legend_artist
857              keys += ["s", "facecolor", "fc", "edgecolor", "ec", "linewidth", "lw"]
858          else:
859              legend_artist = partial(mpl.lines.Line2D, xdata=[], ydata=[])
860              keys += [
861                  "markersize", "ms",
862                  "markeredgewidth", "mew",
863                  "markeredgecolor", "mec",
864                  "linestyle", "ls",
865                  "linewidth", "lw",
866              ]
867  
868          common_kws = {k: v for k, v in kwargs.items() if k in keys}
869          attrs = {"hue": "color", "style": None}
870          if kind == "scatter":
871              attrs["size"] = "s"
872          elif kind == "line":
873              attrs["size"] = "linewidth"
874          p.add_legend_data(g.axes.flat[0], legend_artist, common_kws, attrs)
875          if p.legend_data:
876              g.add_legend(legend_data=p.legend_data,
877                           label_order=p.legend_order,
878                           title=p.legend_title,
879                           adjust_subtitles=True)
880  
881      # Rename the columns of the FacetGrid's `data` attribute
882      # to match the original column names
883      orig_cols = {
884          f"_{k}": f"_{k}_" if v is None else v for k, v in variables.items()
885      }
886      grid_data = g.data.rename(columns=orig_cols)
887      if data is not None and (x is not None or y is not None):
888          if not isinstance(data, pd.DataFrame):
889              data = pd.DataFrame(data)
890          g.data = pd.merge(
891              data,
892              grid_data[grid_data.columns.difference(data.columns)],
893              left_index=True,
894              right_index=True,
895          )
896      else:
897          g.data = grid_data
898  
899      return g
