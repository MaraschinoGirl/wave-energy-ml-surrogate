8    class ReportLabPen(BasePen):
9        """A pen for drawing onto a ``reportlab.graphics.shapes.Path`` object."""
10   
11       def __init__(self, glyphSet, path=None):
12           BasePen.__init__(self, glyphSet)
13           if path is None:
14               path = Path()
15           self.path = path
16   
17       def _moveTo(self, p):
18           (x, y) = p
19           self.path.moveTo(x, y)
20   
21       def _lineTo(self, p):
22           (x, y) = p
23           self.path.lineTo(x, y)
24   
25       def _curveToOne(self, p1, p2, p3):
26           (x1, y1) = p1
27           (x2, y2) = p2
28           (x3, y3) = p3
29           self.path.curveTo(x1, y1, x2, y2, x3, y3)
30   
31       def _closePath(self):
32           self.path.closePath()
