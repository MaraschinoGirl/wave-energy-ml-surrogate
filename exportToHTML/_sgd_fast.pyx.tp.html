403      cdef Py_ssize_t n_features = weights.shape[0]

405      cdef WeightVector{{name_suffix}} w = WeightVector{{name_suffix}}(weights, average_weights)
406      cdef {{c_type}} *x_data_ptr = NULL
407      cdef int *x_ind_ptr = NULL
408

410      cdef int no_improvement_count = 0
411      cdef bint infinity = False
412      cdef int xnnz
413      cdef double eta = 0.0
414      cdef double p = 0.0
415      cdef double update = 0.0
416      cdef double intercept_update = 0.0
417      cdef double sumloss = 0.0
418      cdef double score = 0.0
419      cdef double best_loss = INFINITY
420      cdef double best_score = -INFINITY
421      cdef {{c_type}} y = 0.0
422      cdef {{c_type}} sample_weight
423      cdef {{c_type}} class_weight = 1.0
424      cdef unsigned int count = 0
425      cdef unsigned int train_count = n_samples - np.sum(validation_mask)
426      cdef unsigned int epoch = 0
427      cdef unsigned int i = 0
428      cdef int is_hinge = isinstance(loss, Hinge)
429      cdef double optimal_init = 0.0
430      cdef double dloss = 0.0
431      cdef double MAX_DLOSS = 1e12
432  
433      cdef long long sample_index
434  
435      # q vector is only used for L1 regularization
436      cdef {{c_type}}[::1] q = None
437      cdef {{c_type}} * q_data_ptr = NULL
438      if penalty_type == L1 or penalty_type == ELASTICNET:
439          q = np.zeros((n_features,), dtype={{np_type}}, order="c")
440          q_data_ptr = &q[0]
441      cdef double u = 0.0
442  
443      if penalty_type == L2:
444          l1_ratio = 0.0
445      elif penalty_type == L1:
446          l1_ratio = 1.0
447  
448      eta = eta0
449  
450      if learning_rate == OPTIMAL:
451          typw = np.sqrt(1.0 / np.sqrt(alpha))

454          # initialize t such that eta at first sample equals eta0
455          optimal_init = 1.0 / (initial_eta0 * alpha)
456  
457      t_start = time()
458      with nogil:
459          for epoch in range(max_iter):
460              sumloss = 0
461              if verbose > 0:
462                  with gil:
463                      print("-- Epoch %d" % (epoch + 1))
464              if shuffle:
465                  dataset.shuffle(seed)
466              for i in range(n_samples):
467                  dataset.next(&x_data_ptr, &x_ind_ptr, &xnnz,
468                               &y, &sample_weight)
469  
470                  sample_index = dataset.index_data_ptr[dataset.current_index]
471                  if validation_mask[sample_index]:
472                      # do not learn on the validation set
473                      continue
474  
475                  p = w.dot(x_data_ptr, x_ind_ptr, xnnz) + intercept
476                  if learning_rate == OPTIMAL:
477                      eta = 1.0 / (alpha * (optimal_init + t - 1))
478                  elif learning_rate == INVSCALING:
479                      eta = eta0 / pow(t, power_t)
480  
481                  if verbose or not early_stopping:
482                      sumloss += loss.cy_loss(y, p)
483  
484                  if y > 0.0:
485                      class_weight = weight_pos
486                  else:
487                      class_weight = weight_neg
488  
489                  if learning_rate == PA1:
490                      update = sqnorm(x_data_ptr, x_ind_ptr, xnnz)
491                      if update == 0:
492                          continue
493                      update = min(C, loss.cy_loss(y, p) / update)
494                  elif learning_rate == PA2:
495                      update = sqnorm(x_data_ptr, x_ind_ptr, xnnz)
496                      update = loss.cy_loss(y, p) / (update + 0.5 / C)
497                  else:
498                      dloss = loss.cy_gradient(y, p)
499                      # clip dloss with large values to avoid numerical
500                      # instabilities
501                      if dloss < -MAX_DLOSS:
502                          dloss = -MAX_DLOSS
503                      elif dloss > MAX_DLOSS:
504                          dloss = MAX_DLOSS
505                      update = -eta * dloss
506  
507                  if learning_rate >= PA1:
508                      if is_hinge:
509                          # classification
510                          update *= y
511                      elif y - p < 0:
512                          # regression
513                          update *= -1
514  
515                  update *= class_weight * sample_weight
516  
517                  if penalty_type >= L2:
518                      # do not scale to negative values when eta or alpha are too
519                      # big: instead set the weights to zero
520                      w.scale(max(0, 1.0 - ((1.0 - l1_ratio) * eta * alpha)))
521  
522                  if update != 0.0:
523                      w.add(x_data_ptr, x_ind_ptr, xnnz, update)
524                  if fit_intercept == 1:
525                      intercept_update = update
526                      if one_class:  # specific for One-Class SVM
527                          intercept_update -= 2. * eta * alpha
528                      if intercept_update != 0:
529                          intercept += intercept_update * intercept_decay
530  
531                  if 0 < average <= t:
532                      # compute the average for the intercept and update the
533                      # average weights, this is done regardless as to whether
534                      # the update is 0
535  
536                      w.add_average(x_data_ptr, x_ind_ptr, xnnz,
537                                    update, (t - average + 1))
538                      average_intercept += ((intercept - average_intercept) /
539                                            (t - average + 1))
540  
541                  if penalty_type == L1 or penalty_type == ELASTICNET:
542                      u += (l1_ratio * eta * alpha)
543                      l1penalty{{name_suffix}}(w, q_data_ptr, x_ind_ptr, xnnz, u)
544  
545                  t += 1
546                  count += 1
547  
548              # report epoch information
549              if verbose > 0:
550                  with gil:
551                      print("Norm: %.2f, NNZs: %d, Bias: %.6f, T: %d, "
552                            "Avg. loss: %f"
553                            % (w.norm(), np.nonzero(weights)[0].shape[0],
554                               intercept, count, sumloss / train_count))
555                      print("Total training time: %.2f seconds."
556                            % (time() - t_start))
557  
558              # floating-point under-/overflow check.
559              if (not isfinite(intercept) or any_nonfinite(weights)):
560                  infinity = True
561                  break
562  
563              #evaluate the score on the validation set
564              if early_stopping:
565                  with gil:
566                      score = validation_score_cb(weights.base, intercept)
567                  if tol > -INFINITY and score < best_score + tol:
568                      no_improvement_count += 1
569                  else:
570                      no_improvement_count = 0
571                  if score > best_score:
572                      best_score = score
573              # or evaluate the loss on the training set
574              else:
575                  if tol > -INFINITY and sumloss > best_loss - tol * train_count:
576                      no_improvement_count += 1
577                  else:
578                      no_improvement_count = 0
579                  if sumloss < best_loss:
580                      best_loss = sumloss
581  
582              # if there is no improvement several times in a row
583              if no_improvement_count >= n_iter_no_change:
584                  if learning_rate == ADAPTIVE and eta > 1e-6:
585                      eta = eta / 5
586                      no_improvement_count = 0
587                  else:
588                      if verbose:
589                          with gil:
590                              print("Convergence after %d epochs took %.2f "
591                                    "seconds" % (epoch + 1, time() - t_start))
592                      break
593  
594      if infinity:
595          raise ValueError(("Floating-point under-/overflow occurred at epoch"
596                            " #%d. Scaling input data with StandardScaler or"
597                            " MinMaxScaler might help.") % (epoch + 1))
598  
599      w.reset_wscale()
600  
601      return (
602          weights.base,
603          intercept,
604          None if average_weights is None else average_weights.base,
605          average_intercept,
606          epoch + 1
607      )

635  cdef void l1penalty{{name_suffix}}(
636      WeightVector{{name_suffix}} w,
637      {{c_type}} * q_data_ptr,
638      int *x_ind_ptr,
639      int xnnz,
640      double u,
641  ) noexcept nogil:
642      """Apply the L1 penalty to each updated feature
643  
644      This implements the truncated gradient approach by
645      [Tsuruoka, Y., Tsujii, J., and Ananiadou, S., 2009].
646      """
647      cdef double z = 0.0
648      cdef int j = 0
649      cdef int idx = 0
650      cdef double wscale = w.wscale
651      cdef {{c_type}} *w_data_ptr = w.w_data_ptr
652      for j in range(xnnz):
653          idx = x_ind_ptr[j]
654          z = w_data_ptr[idx]
655          if wscale * z > 0.0:
656              w_data_ptr[idx] = max(
657                  0.0, w_data_ptr[idx] - ((u + q_data_ptr[idx]) / wscale))
658  
659          elif wscale * z < 0.0:
660              w_data_ptr[idx] = min(
661                  0.0, w_data_ptr[idx] + ((u - q_data_ptr[idx]) / wscale))
662  
663          q_data_ptr[idx] += wscale * (w_data_ptr[idx] - z)
