1    import os

3    import ntsecuritycon
4    import win32api
5    import win32con
6    import win32security

9    temp_dir = win32api.GetTempPath()
10   fname = win32api.GetTempFileName(temp_dir, "rsk")[0]
11   print(fname)
12   ## file can't exist
13   os.remove(fname)
14   
15   ## enable backup and restore privs
16   required_privs = (
17       (
18           win32security.LookupPrivilegeValue("", ntsecuritycon.SE_BACKUP_NAME),
19           win32con.SE_PRIVILEGE_ENABLED,
20       ),
21       (
22           win32security.LookupPrivilegeValue("", ntsecuritycon.SE_RESTORE_NAME),
23           win32con.SE_PRIVILEGE_ENABLED,
24       ),
25   )
26   ph = win32api.GetCurrentProcess()
27   th = win32security.OpenProcessToken(
28       ph, win32con.TOKEN_READ | win32con.TOKEN_ADJUST_PRIVILEGES
29   )
30   adjusted_privs = win32security.AdjustTokenPrivileges(th, 0, required_privs)
31   
32   try:
33       sa = win32security.SECURITY_ATTRIBUTES()
34       my_sid = win32security.GetTokenInformation(th, ntsecuritycon.TokenUser)[0]
35       sa.SECURITY_DESCRIPTOR.SetSecurityDescriptorOwner(my_sid, 0)
36   
37       k, disp = win32api.RegCreateKeyEx(
38           win32con.HKEY_CURRENT_USER,
39           "Python test key",
40           SecurityAttributes=sa,
41           samDesired=win32con.KEY_ALL_ACCESS,
42           Class="some class",
43           Options=0,
44       )
45       win32api.RegSetValue(k, None, win32con.REG_SZ, "Default value for python test key")
46   
47       subk, disp = win32api.RegCreateKeyEx(
48           k,
49           "python test subkey",
50           SecurityAttributes=sa,
51           samDesired=win32con.KEY_ALL_ACCESS,
52           Class="some other class",
53           Options=0,
54       )
55       win32api.RegSetValue(subk, None, win32con.REG_SZ, "Default value for subkey")
56   
57       win32api.RegSaveKeyEx(
58           k, fname, Flags=winnt.REG_STANDARD_FORMAT, SecurityAttributes=sa
59       )
60   
61       restored_key, disp = win32api.RegCreateKeyEx(
62           win32con.HKEY_CURRENT_USER,
63           "Python test key(restored)",
64           SecurityAttributes=sa,
65           samDesired=win32con.KEY_ALL_ACCESS,
66           Class="restored class",
67           Options=0,
68       )
69       win32api.RegRestoreKey(restored_key, fname)
70   finally:
71       win32security.AdjustTokenPrivileges(th, 0, adjusted_privs)
