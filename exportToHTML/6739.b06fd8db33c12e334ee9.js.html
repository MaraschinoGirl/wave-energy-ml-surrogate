77   function tokenBase(stream, state) {
78     firstWordMatch = stream.string.match(/(^[\w-]+\s*=\s*$)|(^\s*[\w-]+\s*=\s*[\w-])|(^\s*(\.|#|@|\$|\&|\[|\d|\+|::?|\{|\>|~|\/)?\s*[\w-]*([a-z0-9-]|\*|\/\*)(\(|,)?)/);
79     state.context.line.firstWord = firstWordMatch ? firstWordMatch[0].replace(/^\s*/, "") : "";
80     state.context.line.indent = stream.indentation();
81     ch = stream.peek();
82   
83     // Line comment
84     if (stream.match("//")) {
85       stream.skipToEnd();
86       return ["comment", "comment"];
87     }
88     // Block comment
89     if (stream.match("/*")) {
90       state.tokenize = tokenCComment;
91       return tokenCComment(stream, state);
92     }
93     // String
94     if (ch == "\"" || ch == "'") {
95       stream.next();
96       state.tokenize = tokenString(ch);
97       return state.tokenize(stream, state);
98     }
99     // Def
100    if (ch == "@") {
101      stream.next();
102      stream.eatWhile(/[\w\\-]/);
103      return ["def", stream.current()];
104    }
105    // ID selector or Hex color
106    if (ch == "#") {
107      stream.next();
108      // Hex color
109      if (stream.match(/^[0-9a-f]{3}([0-9a-f]([0-9a-f]{2}){0,2})?\b(?!-)/i)) {
110        return ["atom", "atom"];
111      }
112      // ID selector
113      if (stream.match(/^[a-z][\w-]*/i)) {
114        return ["builtin", "hash"];
115      }
116    }
117    // Vendor prefixes
118    if (stream.match(vendorPrefixesRegexp)) {
119      return ["meta", "vendor-prefixes"];
120    }
121    // Numbers
122    if (stream.match(/^-?[0-9]?\.?[0-9]/)) {
123      stream.eatWhile(/[a-z%]/i);
124      return ["number", "unit"];
125    }
126    // !important|optional
127    if (ch == "!") {
128      stream.next();
129      return [stream.match(/^(important|optional)/i) ? "keyword": "operator", "important"];
130    }
131    // Class
132    if (ch == "." && stream.match(/^\.[a-z][\w-]*/i)) {
133      return ["qualifier", "qualifier"];
134    }
135    // url url-prefix domain regexp
136    if (stream.match(documentTypesRegexp)) {
137      if (stream.peek() == "(") state.tokenize = tokenParenthesized;
138      return ["property", "word"];
139    }
140    // Mixins / Functions
141    if (stream.match(/^[a-z][\w-]*\(/i)) {
142      stream.backUp(1);
143      return ["keyword", "mixin"];
144    }
145    // Block mixins
146    if (stream.match(/^(\+|-)[a-z][\w-]*\(/i)) {
147      stream.backUp(1);
148      return ["keyword", "block-mixin"];
149    }
150    // Parent Reference BEM naming
151    if (stream.string.match(/^\s*&/) && stream.match(/^[-_]+[a-z][\w-]*/)) {
152      return ["qualifier", "qualifier"];
153    }
154    // / Root Reference & Parent Reference
155    if (stream.match(/^(\/|&)(-|_|:|\.|#|[a-z])/)) {
156      stream.backUp(1);
157      return ["variableName.special", "reference"];
158    }
159    if (stream.match(/^&{1}\s*$/)) {
160      return ["variableName.special", "reference"];
161    }
162    // Word operator
163    if (stream.match(wordOperatorKeywordsRegexp)) {
164      return ["operator", "operator"];
165    }
166    // Word
167    if (stream.match(/^\$?[-_]*[a-z0-9]+[\w-]*/i)) {
168      // Variable
169      if (stream.match(/^(\.|\[)[\w-\'\"\]]+/i, false)) {
170        if (!wordIsTag(stream.current())) {
171          stream.match('.');
172          return ["variable", "variable-name"];
173        }
174      }
175      return ["variable", "word"];
176    }
177    // Operators
178    if (stream.match(operatorsRegexp)) {
179      return ["operator", stream.current()];
180    }
181    // Delimiters
182    if (/[:;,{}\[\]\(\)]/.test(ch)) {
183      stream.next();
184      return [null, ch];
185    }
186    // Non-detected items
187    stream.next();
188    return [null, null];
189  }

194  function tokenCComment(stream, state) {
195    var maybeEnd = false, ch;
196    while ((ch = stream.next()) != null) {
197      if (maybeEnd && ch == "/") {
198        state.tokenize = null;
199        break;
200      }
201      maybeEnd = (ch == "*");
202    }
203    return ["comment", "comment"];

209  function tokenString(quote) {
210    return function(stream, state) {
211      var escaped = false, ch;
212      while ((ch = stream.next()) != null) {
213        if (ch == quote && !escaped) {
214          if (quote == ")") stream.backUp(1);
215          break;
216        }
217        escaped = !escaped && ch == "\\";
218      }
219      if (ch == quote || !escaped && quote != ")") state.tokenize = null;
220      return ["string", "string"];
221    };
222  }

287  function wordIsVendorPrefix(word) {
288    return word.toLowerCase().match(vendorPrefixesRegexp);
289  }

291  function wordAsValue(word) {
292    var wordLC = word.toLowerCase();
293    var override = "variable";
294    if (wordIsTag(word)) override = "tag";
295    else if (wordIsBlock(word)) override = "block-keyword";
296    else if (wordIsProperty(word)) override = "property";
297    else if (wordLC in valueKeywords || wordLC in commonAtoms) override = "atom";
298    else if (wordLC == "return" || wordLC in colorKeywords) override = "keyword";
299  
300    // Font family
301    else if (word.match(/^[A-Z]/)) override = "string";
302    return override;
303  }

305  function typeIsBlock(type, stream) {
306    return ((endOfLine(stream) && (type == "{" || type == "]" || type == "hash" || type == "qualifier")) || type == "block-mixin");
307  }

310    return type == "{" && stream.match(/^\s*\$?[\w-]+/i, false);

313  function typeIsPseudo(type, stream) {
314    return type == ":" && stream.match(/^[a-z-]+/, false);

317  function startOfLine(stream) {
318    return stream.sol() || stream.string.match(new RegExp("^\\s*" + escapeRegExp(stream.current())));
319  }

321  function endOfLine(stream) {
322    return stream.eol() || stream.match(/^\s*$/, false);

325  function firstWordOfLine(line) {
326    var re = /^\s*[-_]*[a-z0-9]+[\w-]*/i;
327    var result = typeof line == "string" ? line.match(re) : line.string.match(re);
328    return result ? result[0].replace(/^\s*/, "") : "";
329  }

332  /**
333   * Block
334   */
335  states.block = function(type, stream, state) {
336    if ((type == "comment" && startOfLine(stream)) ||
337        (type == "," && endOfLine(stream)) ||
338        type == "mixin") {
339      return pushContext(state, stream, "block", 0);
340    }
341    if (typeIsInterpolation(type, stream)) {
342      return pushContext(state, stream, "interpolation");
343    }
344    if (endOfLine(stream) && type == "]") {
345      if (!/^\s*(\.|#|:|\[|\*|&)/.test(stream.string) && !wordIsTag(firstWordOfLine(stream))) {
346        return pushContext(state, stream, "block", 0);
347      }
348    }
349    if (typeIsBlock(type, stream)) {
350      return pushContext(state, stream, "block");
351    }
352    if (type == "}" && endOfLine(stream)) {
353      return pushContext(state, stream, "block", 0);
354    }
355    if (type == "variable-name") {
356      if (stream.string.match(/^\s?\$[\w-\.\[\]\'\"]+$/) || wordIsBlock(firstWordOfLine(stream))) {
357        return pushContext(state, stream, "variableName");
358      }
359      else {
360        return pushContext(state, stream, "variableName", 0);
361      }
362    }
363    if (type == "=") {
364      if (!endOfLine(stream) && !wordIsBlock(firstWordOfLine(stream))) {
365        return pushContext(state, stream, "block", 0);
366      }
367      return pushContext(state, stream, "block");
368    }
369    if (type == "*") {
370      if (endOfLine(stream) || stream.match(/\s*(,|\.|#|\[|:|{)/,false)) {
371        override = "tag";
372        return pushContext(state, stream, "block");
373      }
374    }
375    if (typeIsPseudo(type, stream)) {
376      return pushContext(state, stream, "pseudo");
377    }
378    if (/@(font-face|media|supports|(-moz-)?document)/.test(type)) {
379      return pushContext(state, stream, endOfLine(stream) ? "block" : "atBlock");
380    }
381    if (/@(-(moz|ms|o|webkit)-)?keyframes$/.test(type)) {
382      return pushContext(state, stream, "keyframes");
383    }
384    if (/@extends?/.test(type)) {
385      return pushContext(state, stream, "extend", 0);
386    }
387    if (type && type.charAt(0) == "@") {
388  
389      // Property Lookup
390      if (stream.indentation() > 0 && wordIsProperty(stream.current().slice(1))) {
391        override = "variable";
392        return "block";
393      }
394      if (/(@import|@require|@charset)/.test(type)) {
395        return pushContext(state, stream, "block", 0);
396      }
397      return pushContext(state, stream, "block");
398    }
399    if (type == "reference" && endOfLine(stream)) {
400      return pushContext(state, stream, "block");
401    }
402    if (type == "(") {
403      return pushContext(state, stream, "parens");
404    }
405  
406    if (type == "vendor-prefixes") {
407      return pushContext(state, stream, "vendorPrefixes");
408    }
409    if (type == "word") {
410      var word = stream.current();
411      override = wordAsValue(word);
412  
413      if (override == "property") {
414        if (startOfLine(stream)) {
415          return pushContext(state, stream, "block", 0);
416        } else {
417          override = "atom";
418          return "block";
419        }
420      }
421  
422      if (override == "tag") {
423  
424        // tag is a css value
425        if (/embed|menu|pre|progress|sub|table/.test(word)) {
426          if (wordIsProperty(firstWordOfLine(stream))) {
427            override = "atom";
428            return "block";
429          }
430        }
431  
432        // tag is an attribute
433        if (stream.string.match(new RegExp("\\[\\s*" + word + "|" + word +"\\s*\\]"))) {
434          override = "atom";
435          return "block";
436        }
437  
438        // tag is a variable
439        if (tagVariablesRegexp.test(word)) {
440          if ((startOfLine(stream) && stream.string.match(/=/)) ||
441              (!startOfLine(stream) &&
442               !stream.string.match(/^(\s*\.|#|\&|\[|\/|>|\*)/) &&
443               !wordIsTag(firstWordOfLine(stream)))) {
444            override = "variable";
445            if (wordIsBlock(firstWordOfLine(stream)))  return "block";
446            return pushContext(state, stream, "block", 0);
447          }
448        }
449  
450        if (endOfLine(stream)) return pushContext(state, stream, "block");
451      }
452      if (override == "block-keyword") {
453        override = "keyword";
454  
455        // Postfix conditionals
456        if (stream.current(/(if|unless)/) && !startOfLine(stream)) {
457          return "block";
458        }
459        return pushContext(state, stream, "block");
460      }
461      if (word == "return") return pushContext(state, stream, "block", 0);
462  
463      // Placeholder selector
464      if (override == "variable" && stream.string.match(/^\s?\$[\w-\.\[\]\'\"]+$/)) {
465        return pushContext(state, stream, "block");
466      }
467    }
468    return state.context.type;
469  };

472  /**
473   * Parens
474   */
475  states.parens = function(type, stream, state) {
476    if (type == "(") return pushContext(state, stream, "parens");
477    if (type == ")") {
478      if (state.context.prev.type == "parens") {
479        return popContext(state, stream);
480      }
481      if ((stream.string.match(/^[a-z][\w-]*\(/i) && endOfLine(stream)) ||
482          wordIsBlock(firstWordOfLine(stream)) ||
483          /(\.|#|:|\[|\*|&|>|~|\+|\/)/.test(firstWordOfLine(stream)) ||
484          (!stream.string.match(/^-?[a-z][\w-\.\[\]\'\"]*\s*=/) &&
485           wordIsTag(firstWordOfLine(stream)))) {
486        return pushContext(state, stream, "block");
487      }
488      if (stream.string.match(/^[\$-]?[a-z][\w-\.\[\]\'\"]*\s*=/) ||
489          stream.string.match(/^\s*(\(|\)|[0-9])/) ||
490          stream.string.match(/^\s+[a-z][\w-]*\(/i) ||
491          stream.string.match(/^\s+[\$-]?[a-z]/i)) {
492        return pushContext(state, stream, "block", 0);
493      }
494      if (endOfLine(stream)) return pushContext(state, stream, "block");
495      else return pushContext(state, stream, "block", 0);
496    }
497    if (type && type.charAt(0) == "@" && wordIsProperty(stream.current().slice(1))) {
498      override = "variable";
499    }
500    if (type == "word") {
501      var word = stream.current();
502      override = wordAsValue(word);
503      if (override == "tag" && tagVariablesRegexp.test(word)) {
504        override = "variable";
505      }
506      if (override == "property" || word == "to") override = "atom";
507    }
508    if (type == "variable-name") {
509      return pushContext(state, stream, "variableName");
510    }
511    if (typeIsPseudo(type, stream)) {
512      return pushContext(state, stream, "pseudo");
513    }
514    return state.context.type;
515  };

544  /**
545   * atBlock
546   */
547  states.atBlock = function(type, stream, state) {
548    if (type == "(") return pushContext(state, stream, "atBlock_parens");
549    if (typeIsBlock(type, stream)) {
550      return pushContext(state, stream, "block");
551    }
552    if (typeIsInterpolation(type, stream)) {
553      return pushContext(state, stream, "interpolation");
554    }
555    if (type == "word") {
556      var word = stream.current().toLowerCase();
557      if (/^(only|not|and|or)$/.test(word))
558        override = "keyword";
559      else if (documentTypes.hasOwnProperty(word))
560        override = "tag";
561      else if (mediaTypes.hasOwnProperty(word))
562        override = "attribute";
563      else if (mediaFeatures.hasOwnProperty(word))
564        override = "property";
565      else if (nonStandardPropertyKeywords.hasOwnProperty(word))
566        override = "string.special";
567      else override = wordAsValue(stream.current());
568      if (override == "tag" && endOfLine(stream)) {
569        return pushContext(state, stream, "block");
570      }
571    }
572    if (type == "operator" && /^(not|and|or)$/.test(stream.current())) {
573      override = "keyword";
574    }
575    return state.context.type;
576  };

578  states.atBlock_parens = function(type, stream, state) {
579    if (type == "{" || type == "}") return state.context.type;
580    if (type == ")") {
581      if (endOfLine(stream)) return pushContext(state, stream, "block");
582      else return pushContext(state, stream, "atBlock");
583    }
584    if (type == "word") {
585      var word = stream.current().toLowerCase();
586      override = wordAsValue(word);
587      if (/^(max|min)/.test(word)) override = "property";
588      if (override == "tag") {
589        tagVariablesRegexp.test(word) ? override = "variable" : override = "atom";
590      }
591      return state.context.type;
592    }
593    return states.atBlock(type, stream, state);
594  };

600  states.keyframes = function(type, stream, state) {
601    if (stream.indentation() == "0" && ((type == "}" && startOfLine(stream)) || type == "]" || type == "hash"
602                                        || type == "qualifier" || wordIsTag(stream.current()))) {
603      return popAndPass(type, stream, state);
604    }
605    if (type == "{") return pushContext(state, stream, "keyframes");
606    if (type == "}") {
607      if (startOfLine(stream)) return popContext(state, stream, true);
608      else return pushContext(state, stream, "keyframes");
609    }
610    if (type == "unit" && /^[0-9]+\%$/.test(stream.current())) {
611      return pushContext(state, stream, "keyframes");
612    }
613    if (type == "word") {
614      override = wordAsValue(stream.current());
615      if (override == "block-keyword") {
616        override = "keyword";
617        return pushContext(state, stream, "keyframes");
618      }
619    }
620    if (/@(font-face|media|supports|(-moz-)?document)/.test(type)) {
621      return pushContext(state, stream, endOfLine(stream) ? "block" : "atBlock");
622    }
623    if (type == "mixin") {
624      return pushContext(state, stream, "block", 0);
625    }
626    return state.context.type;
627  };

632   */
633  states.interpolation = function(type, stream, state) {
634    if (type == "{") popContext(state, stream) && pushContext(state, stream, "block");
635    if (type == "}") {
636      if (stream.string.match(/^\s*(\.|#|:|\[|\*|&|>|~|\+|\/)/i) ||
637          (stream.string.match(/^\s*[a-z]/i) && wordIsTag(firstWordOfLine(stream)))) {
638        return pushContext(state, stream, "block");
639      }
640      if (!stream.string.match(/^(\{|\s*\&)/) ||
641          stream.match(/\s*[\w-]/,false)) {
642        return pushContext(state, stream, "block", 0);
643      }
644      return pushContext(state, stream, "block");
645    }
646    if (type == "variable-name") {
647      return pushContext(state, stream, "variableName", 0);
648    }
649    if (type == "word") {
650      override = wordAsValue(stream.current());
651      if (override == "tag") override = "atom";
652    }
653    return state.context.type;
654  };

682  const stylus = {
683    name: "stylus",
684    startState: function() {
685      return {
686        tokenize: null,
687        state: "block",
688        context: new Context("block", 0, null)
689      };
690    },
691    token: function(stream, state) {
692      if (!state.tokenize && stream.eatSpace()) return null;
693      style = (state.tokenize || tokenBase)(stream, state);
694      if (style && typeof style == "object") {
695        type = style[1];
696        style = style[0];
697      }
698      override = style;
699      state.state = states[state.state](type, stream, state);
700      return override;
701    },
702    indent: function(state, textAfter, iCx) {
703      var cx = state.context,
704          ch = textAfter && textAfter.charAt(0),
705          indent = cx.indent,
706          lineFirstWord = firstWordOfLine(textAfter),
707          lineIndent = cx.line.indent,
708          prevLineFirstWord = state.context.prev ? state.context.prev.line.firstWord : "",
709          prevLineIndent = state.context.prev ? state.context.prev.line.indent : lineIndent;
710  
711      if (cx.prev &&
712          (ch == "}" && (cx.type == "block" || cx.type == "atBlock" || cx.type == "keyframes") ||
713           ch == ")" && (cx.type == "parens" || cx.type == "atBlock_parens") ||
714           ch == "{" && (cx.type == "at"))) {
715        indent = cx.indent - iCx.unit;
716      } else if (!(/(\})/.test(ch))) {
717        if (/@|\$|\d/.test(ch) ||
718            /^\{/.test(textAfter) ||
719  /^\s*\/(\/|\*)/.test(textAfter) ||
720            /^\s*\/\*/.test(prevLineFirstWord) ||
721            /^\s*[\w-\.\[\]\'\"]+\s*(\?|:|\+)?=/i.test(textAfter) ||
722            /^(\+|-)?[a-z][\w-]*\(/i.test(textAfter) ||
723            /^return/.test(textAfter) ||
724            wordIsBlock(lineFirstWord)) {
725          indent = lineIndent;
726        } else if (/(\.|#|:|\[|\*|&|>|~|\+|\/)/.test(ch) || wordIsTag(lineFirstWord)) {
727          if (/\,\s*$/.test(prevLineFirstWord)) {
728            indent = prevLineIndent;
729          } else if (!state.sol() && (/(\.|#|:|\[|\*|&|>|~|\+|\/)/.test(prevLineFirstWord) || wordIsTag(prevLineFirstWord))) {
730            indent = lineIndent <= prevLineIndent ? prevLineIndent : prevLineIndent + iCx.unit;
731          } else {
732            indent = lineIndent;
733          }
734        } else if (!/,\s*$/.test(textAfter) && (wordIsVendorPrefix(lineFirstWord) || wordIsProperty(lineFirstWord))) {
735          if (wordIsBlock(prevLineFirstWord)) {
736            indent = lineIndent <= prevLineIndent ? prevLineIndent : prevLineIndent + iCx.unit;
737          } else if (/^\{/.test(prevLineFirstWord)) {
738            indent = lineIndent <= prevLineIndent ? lineIndent : prevLineIndent + iCx.unit;
739          } else if (wordIsVendorPrefix(prevLineFirstWord) || wordIsProperty(prevLineFirstWord)) {
740            indent = lineIndent >= prevLineIndent ? prevLineIndent : lineIndent;
741          } else if (/^(\.|#|:|\[|\*|&|@|\+|\-|>|~|\/)/.test(prevLineFirstWord) ||
742                     /=\s*$/.test(prevLineFirstWord) ||
743                     wordIsTag(prevLineFirstWord) ||
744                     /^\$[\w-\.\[\]\'\"]/.test(prevLineFirstWord)) {
745            indent = prevLineIndent + iCx.unit;
746          } else {
747            indent = lineIndent;
748          }
749        }
750      }
751      return indent;
752    },
753    languageData: {
754      indentOnInput: /^\s*\}$/,
755      commentTokens: {line: "//", block: {open: "/*", close: "*/"}},
756      autocomplete: hintWords
757    }
758  };
