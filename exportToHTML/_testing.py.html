76   def ignore_warnings(obj=None, category=Warning):
77       """Context manager and decorator to ignore warnings. 
78    
79       Note: Using this (in both variants) will clear all warnings 
80       from all python modules loaded. In case you need to test 
81       cross-module-warning-logging, this is not your tool of choice. 
82    
83       Parameters 
84       ---------- 
85       obj : callable, default=None 
86           callable where you want to ignore the warnings. 
87       category : warning class, default=Warning 
88           The category to filter. If Warning, all categories will be muted. 
89    
90       Examples 
91       -------- 
92       >>> import warnings 
93       >>> from sklearn.utils._testing import ignore_warnings 
94       >>> with ignore_warnings(): 
95       ...     warnings.warn('buhuhuhu') 
96    
97       >>> def nasty_warn(): 
98       ...     warnings.warn('buhuhuhu') 
99       ...     print(42) 
100   
101      >>> ignore_warnings(nasty_warn)() 
102      42 
103      """
104      if isinstance(obj, type) and issubclass(obj, Warning):
105          # Avoid common pitfall of passing category as the first positional
106          # argument which result in the test not being run
107          warning_name = obj.__name__
108          raise ValueError(
109              "'obj' should be a callable where you want to ignore warnings. "
110              "You passed a warning class instead: 'obj={warning_name}'. "
111              "If you want to pass a warning class to ignore_warnings, "
112              "you should use 'category={warning_name}'".format(warning_name=warning_name)
113          )
114      elif callable(obj):
115          return _IgnoreWarnings(category=category)(obj)
116      else:
117          return _IgnoreWarnings(category=category)

120  class _IgnoreWarnings:
121      """Improved and simplified Python warnings context manager and decorator. 
122   
123      This class allows the user to ignore the warnings raised by a function. 
124      Copied from Python 2.7.5 and modified as required. 
125   
126      Parameters 
127      ---------- 
128      category : tuple of warning class, default=Warning 
129          The category to filter. By default, all the categories will be muted. 
130   
131      """
132  
133      def __init__(self, category):
134          self._record = True
135          self._module = sys.modules["warnings"]
136          self._entered = False
137          self.log = []
138          self.category = category
139  
140      def __call__(self, fn):
141          """Decorator to catch and hide warnings without visual nesting."""
142  
143          @wraps(fn)
144          def wrapper(*args, **kwargs):
145              with warnings.catch_warnings():
146                  warnings.simplefilter("ignore", self.category)
147                  return fn(*args, **kwargs)
148  
149          return wrapper
150  
151      def __repr__(self):
152          args = []
153          if self._record:
154              args.append("record=True")
155          if self._module is not sys.modules["warnings"]:
156              args.append("module=%r" % self._module)
157          name = type(self).__name__
158          return "%s(%s)" % (name, ", ".join(args))
159  
160      def __enter__(self):
161          if self._entered:
162              raise RuntimeError("Cannot enter %r twice" % self)
163          self._entered = True
164          self._filters = self._module.filters
165          self._module.filters = self._filters[:]
166          self._showwarning = self._module.showwarning
167          warnings.simplefilter("ignore", self.category)
168  
169      def __exit__(self, *exc_info):
170          if not self._entered:
171              raise RuntimeError("Cannot exit %r without entering first" % self)
172          self._module.filters = self._filters
173          self._module.showwarning = self._showwarning
174          self.log[:] = []

177  def assert_allclose(
178      actual, desired, rtol=None, atol=0.0, equal_nan=True, err_msg="", verbose=True
179  ):
180      """dtype-aware variant of numpy.testing.assert_allclose 
181   
182      This variant introspects the least precise floating point dtype 
183      in the input argument and automatically sets the relative tolerance 
184      parameter to 1e-4 float32 and use 1e-7 otherwise (typically float64 
185      in scikit-learn). 
186   
187      `atol` is always left to 0. by default. It should be adjusted manually 
188      to an assertion-specific value in case there are null values expected 
189      in `desired`. 
190   
191      The aggregate tolerance is `atol + rtol * abs(desired)`. 
192   
193      Parameters 
194      ---------- 
195      actual : array_like 
196          Array obtained. 
197      desired : array_like 
198          Array desired. 
199      rtol : float, optional, default=None 
200          Relative tolerance. 
201          If None, it is set based on the provided arrays' dtypes. 
202      atol : float, optional, default=0. 
203          Absolute tolerance. 
204      equal_nan : bool, optional, default=True 
205          If True, NaNs will compare equal. 
206      err_msg : str, optional, default='' 
207          The error message to be printed in case of failure. 
208      verbose : bool, optional, default=True 
209          If True, the conflicting values are appended to the error message. 
210   
211      Raises 
212      ------ 
213      AssertionError 
214          If actual and desired are not equal up to specified precision. 
215   
216      See Also 
217      -------- 
218      numpy.testing.assert_allclose 
219   
220      Examples 
221      -------- 
222      >>> import numpy as np 
223      >>> from sklearn.utils._testing import assert_allclose 
224      >>> x = [1e-5, 1e-3, 1e-1] 
225      >>> y = np.arccos(np.cos(x)) 
226      >>> assert_allclose(x, y, rtol=1e-5, atol=0) 
227      >>> a = np.full(shape=10, fill_value=1e-5, dtype=np.float32) 
228      >>> assert_allclose(a, 1e-5) 
229      """
230      dtypes = []
231  
232      actual, desired = np.asanyarray(actual), np.asanyarray(desired)
233      dtypes = [actual.dtype, desired.dtype]
234  
235      if rtol is None:
236          rtols = [1e-4 if dtype == np.float32 else 1e-7 for dtype in dtypes]
237          rtol = max(rtols)
238  
239      np_assert_allclose(
240          actual,
241          desired,
242          rtol=rtol,
243          atol=atol,
244          equal_nan=equal_nan,
245          err_msg=err_msg,
246          verbose=verbose,
247      )

250  def assert_allclose_dense_sparse(x, y, rtol=1e-07, atol=1e-9, err_msg=""):
251      """Assert allclose for sparse and dense data. 
252   
253      Both x and y need to be either sparse or dense, they 
254      can't be mixed. 
255   
256      Parameters 
257      ---------- 
258      x : {array-like, sparse matrix} 
259          First array to compare. 
260   
261      y : {array-like, sparse matrix} 
262          Second array to compare. 
263   
264      rtol : float, default=1e-07 
265          relative tolerance; see numpy.allclose. 
266   
267      atol : float, default=1e-9 
268          absolute tolerance; see numpy.allclose. Note that the default here is 
269          more tolerant than the default for numpy.testing.assert_allclose, where 
270          atol=0. 
271   
272      err_msg : str, default='' 
273          Error message to raise. 
274      """
275      if sp.sparse.issparse(x) and sp.sparse.issparse(y):
276          x = x.tocsr()
277          y = y.tocsr()
278          x.sum_duplicates()
279          y.sum_duplicates()
280          assert_array_equal(x.indices, y.indices, err_msg=err_msg)
281          assert_array_equal(x.indptr, y.indptr, err_msg=err_msg)
282          assert_allclose(x.data, y.data, rtol=rtol, atol=atol, err_msg=err_msg)
283      elif not sp.sparse.issparse(x) and not sp.sparse.issparse(y):
284          # both dense
285          assert_allclose(x, y, rtol=rtol, atol=atol, err_msg=err_msg)
286      else:
287          raise ValueError(
288              "Can only compare two sparse matrices, not a sparse matrix and an array."
289          )

411  def create_memmap_backed_data(data, mmap_mode="r", return_folder=False):
412      """ 
413      Parameters 
414      ---------- 
415      data 
416      mmap_mode : str, default='r' 
417      return_folder :  bool, default=False 
418      """
419      temp_folder = tempfile.mkdtemp(prefix="sklearn_testing_")
420      atexit.register(functools.partial(_delete_folder, temp_folder, warn=True))
421      filename = op.join(temp_folder, "data.pkl")
422      joblib.dump(data, filename)
423      memmap_backed_data = joblib.load(filename, mmap_mode=mmap_mode)
424      result = (
425          memmap_backed_data if not return_folder else (memmap_backed_data, temp_folder)
426      )
427      return result

433  def _get_args(function, varargs=False):
434      """Helper to get function arguments."""
435  
436      try:
437          params = signature(function).parameters
438      except ValueError:
439          # Error on builtin C function
440          return []
441      args = [
442          key
443          for key, param in params.items()
444          if param.kind not in (param.VAR_POSITIONAL, param.VAR_KEYWORD)
445      ]
446      if varargs:
447          varargs = [
448              param.name
449              for param in params.values()
450              if param.kind == param.VAR_POSITIONAL
451          ]
452          if len(varargs) == 0:
453              varargs = None
454          return args, varargs
455      else:
456          return args

459  def _get_func_name(func):
460      """Get function full name. 
461   
462      Parameters 
463      ---------- 
464      func : callable 
465          The function object. 
466   
467      Returns 
468      ------- 
469      name : str 
470          The function name. 
471      """
472      parts = []
473      module = inspect.getmodule(func)
474      if module:
475          parts.append(module.__name__)
476  
477      qualname = func.__qualname__
478      if qualname != func.__name__:
479          parts.append(qualname[: qualname.find(".")])
480  
481      parts.append(func.__name__)
482      return ".".join(parts)

485  def check_docstring_parameters(func, doc=None, ignore=None):
486      """Helper to check docstring. 
487   
488      Parameters 
489      ---------- 
490      func : callable 
491          The function object to test. 
492      doc : str, default=None 
493          Docstring if it is passed manually to the test. 
494      ignore : list, default=None 
495          Parameters to ignore. 
496   
497      Returns 
498      ------- 
499      incorrect : list 
500          A list of string describing the incorrect results. 
501      """
502      from numpydoc import docscrape
503  
504      incorrect = []
505      ignore = [] if ignore is None else ignore
506  
507      func_name = _get_func_name(func)
508      if not func_name.startswith("sklearn.") or func_name.startswith(
509          "sklearn.externals"
510      ):
511          return incorrect
512      # Don't check docstring for property-functions
513      if inspect.isdatadescriptor(func):
514          return incorrect
515      # Don't check docstring for setup / teardown pytest functions
516      if func_name.split(".")[-1] in ("setup_module", "teardown_module"):
517          return incorrect
518      # Dont check estimator_checks module
519      if func_name.split(".")[2] == "estimator_checks":
520          return incorrect
521      # Get the arguments from the function signature
522      param_signature = list(filter(lambda x: x not in ignore, _get_args(func)))
523      # drop self
524      if len(param_signature) > 0 and param_signature[0] == "self":
525          param_signature.remove("self")
526  
527      # Analyze function's docstring
528      if doc is None:
529          records = []
530          with warnings.catch_warnings(record=True):
531              warnings.simplefilter("error", UserWarning)
532              try:
533                  doc = docscrape.FunctionDoc(func)
534              except UserWarning as exp:
535                  if "potentially wrong underline length" in str(exp):
536                      # Catch warning raised as of numpydoc 1.2 when
537                      # the underline length for a section of a docstring
538                      # is not consistent.
539                      message = str(exp).split("\n")[:3]
540                      incorrect += [f"In function: {func_name}"] + message
541                      return incorrect
542                  records.append(str(exp))
543              except Exception as exp:
544                  incorrect += [func_name + " parsing error: " + str(exp)]
545                  return incorrect
546          if len(records):
547              raise RuntimeError("Error for %s:\n%s" % (func_name, records[0]))
548  
549      param_docs = []
550      for name, type_definition, param_doc in doc["Parameters"]:
551          # Type hints are empty only if parameter name ended with :
552          if not type_definition.strip():
553              if ":" in name and name[: name.index(":")][-1:].strip():
554                  incorrect += [
555                      func_name
556                      + " There was no space between the param name and colon (%r)" % name
557                  ]
558              elif name.rstrip().endswith(":"):
559                  incorrect += [
560                      func_name
561                      + " Parameter %r has an empty type spec. Remove the colon"
562                      % (name.lstrip())
563                  ]
564  
565          # Create a list of parameters to compare with the parameters gotten
566          # from the func signature
567          if "*" not in name:
568              param_docs.append(name.split(":")[0].strip("` "))
569  
570      # If one of the docstring's parameters had an error then return that
571      # incorrect message
572      if len(incorrect) > 0:
573          return incorrect
574  
575      # Remove the parameters that should be ignored from list
576      param_docs = list(filter(lambda x: x not in ignore, param_docs))
577  
578      # The following is derived from pytest, Copyright (c) 2004-2017 Holger
579      # Krekel and others, Licensed under MIT License. See
580      # https://github.com/pytest-dev/pytest
581  
582      message = []
583      for i in range(min(len(param_docs), len(param_signature))):
584          if param_signature[i] != param_docs[i]:
585              message += [
586                  "There's a parameter name mismatch in function"
587                  " docstring w.r.t. function signature, at index %s"
588                  " diff: %r != %r" % (i, param_signature[i], param_docs[i])
589              ]
590              break
591      if len(param_signature) > len(param_docs):
592          message += [
593              "Parameters in function docstring have less items w.r.t."
594              " function signature, first missing item: %s"
595              % param_signature[len(param_docs)]
596          ]
597  
598      elif len(param_signature) < len(param_docs):
599          message += [
600              "Parameters in function docstring have more items w.r.t."
601              " function signature, first extra item: %s"
602              % param_docs[len(param_signature)]
603          ]
604  
605      # If there wasn't any difference in the parameters themselves between
606      # docstring and signature including having the same length then return
607      # empty list
608      if len(message) == 0:
609          return []
610  
611      import difflib
612      import pprint
613  
614      param_docs_formatted = pprint.pformat(param_docs).splitlines()
615      param_signature_formatted = pprint.pformat(param_signature).splitlines()
616  
617      message += ["Full diff:"]
618  
619      message.extend(
620          line.strip()
621          for line in difflib.ndiff(param_signature_formatted, param_docs_formatted)
622      )
623  
624      incorrect.extend(message)
625  
626      # Prepend function name
627      incorrect = ["In function: " + func_name] + incorrect
628  
629      return incorrect

641  def _diff_key(line):
642      """Key for grouping output from `context_diff`."""
643      if line.startswith("  "):
644          return "  "
645      elif line.startswith("- "):
646          return "- "
647      elif line.startswith("+ "):
648          return "+ "
649      elif line.startswith("! "):
650          return "! "
651      return None

654  def _get_diff_msg(docstrings_grouped):
655      """Get message showing the difference between type/desc docstrings of all objects. 
656   
657      `docstrings_grouped` keys should be the type/desc docstrings and values are a list 
658      of objects with that docstring. Objects with the same type/desc docstring are 
659      thus grouped together. 
660      """
661      msg_diff = ""
662      ref_str = ""
663      ref_group = []
664      for docstring, group in docstrings_grouped.items():
665          if not ref_str and not ref_group:
666              ref_str += docstring
667              ref_group.extend(group)
668          diff = list(
669              context_diff(
670                  ref_str.split(),
671                  docstring.split(),
672                  fromfile=str(ref_group),
673                  tofile=str(group),
674                  n=8,
675              )
676          )
677          # Add header
678          msg_diff += "".join((diff[:3]))
679          # Group consecutive 'diff' words to shorten error message
680          for start, group in groupby(diff[3:], key=_diff_key):
681              if start is None:
682                  msg_diff += "\n" + "\n".join(group)
683              else:
684                  msg_diff += "\n" + start + " ".join(word[2:] for word in group)
685          # Add new lines at end of diff, to separate comparisons
686          msg_diff += "\n\n"
687      return msg_diff

690  def _check_consistency_items(
691      items_docs, type_or_desc, section, n_objects, descr_regex_pattern=""
692  ):
693      """Helper to check docstring consistency of all `items_docs`. 
694   
695      If item is not present in all objects, checking is skipped and warning raised. 
696      If `regex` provided, match descriptions to all descriptions. 
697      """
698      skipped = []
699      for item_name, docstrings_grouped in items_docs.items():
700          # If item not found in all objects, skip
701          if sum([len(objs) for objs in docstrings_grouped.values()]) < n_objects:
702              skipped.append(item_name)
703          # If regex provided, match to all descriptions
704          elif type_or_desc == "description" and descr_regex_pattern:
705              not_matched = []
706              for docstring, group in docstrings_grouped.items():
707                  if not re.search(descr_regex_pattern, docstring):
708                      not_matched.extend(group)
709              if not_matched:
710                  msg = textwrap.fill(
711                      f"The description of {section[:-1]} '{item_name}' in {not_matched}"
712                      f" does not match 'descr_regex_pattern': {descr_regex_pattern} "
713                  )
714                  raise AssertionError(msg)
715          # Otherwise, if more than one key, docstrings not consistent between objects
716          elif len(docstrings_grouped.keys()) > 1:
717              msg_diff = _get_diff_msg(docstrings_grouped)
718              obj_groups = " and ".join(
719                  str(group) for group in docstrings_grouped.values()
720              )
721              msg = textwrap.fill(
722                  f"The {type_or_desc} of {section[:-1]} '{item_name}' is inconsistent "
723                  f"between {obj_groups}:"
724              )
725              msg += msg_diff
726              raise AssertionError(msg)
727      if skipped:
728          warnings.warn(
729              f"Checking was skipped for {section}: {skipped} as they were "
730              "not found in all objects."
731          )

734  def assert_docstring_consistency(
735      objects,
736      include_params=False,
737      exclude_params=None,
738      include_attrs=False,
739      exclude_attrs=None,
740      include_returns=False,
741      exclude_returns=None,
742      descr_regex_pattern=None,
743  ):
744      r"""Check consistency between docstring parameters/attributes/returns of objects. 
745   
746      Checks if parameters/attributes/returns have the same type specification and 
747      description (ignoring whitespace) across `objects`. Intended to be used for 
748      related classes/functions/data descriptors. 
749   
750      Entries that do not appear across all `objects` are ignored. 
751   
752      Parameters 
753      ---------- 
754      objects : list of {classes, functions, data descriptors} 
755          Objects to check. 
756          Objects may be classes, functions or data descriptors with docstrings that 
757          can be parsed by numpydoc. 
758   
759      include_params : list of str or bool, default=False 
760          List of parameters to be included. If True, all parameters are included, 
761          if False, checking is skipped for parameters. 
762          Can only be set if `exclude_params` is None. 
763   
764      exclude_params : list of str or None, default=None 
765          List of parameters to be excluded. If None, no parameters are excluded. 
766          Can only be set if `include_params` is True. 
767   
768      include_attrs : list of str or bool, default=False 
769          List of attributes to be included. If True, all attributes are included, 
770          if False, checking is skipped for attributes. 
771          Can only be set if `exclude_attrs` is None. 
772   
773      exclude_attrs : list of str or None, default=None 
774          List of attributes to be excluded. If None, no attributes are excluded. 
775          Can only be set if `include_attrs` is True. 
776   
777      include_returns : list of str or bool, default=False 
778          List of returns to be included. If True, all returns are included, 
779          if False, checking is skipped for returns. 
780          Can only be set if `exclude_returns` is None. 
781   
782      exclude_returns : list of str or None, default=None 
783          List of returns to be excluded. If None, no returns are excluded. 
784          Can only be set if `include_returns` is True. 
785   
786      descr_regex_pattern : str, default=None 
787          Regular expression to match to all descriptions of included 
788          parameters/attributes/returns. If None, will revert to default behavior 
789          of comparing descriptions between objects. 
790   
791      Examples 
792      -------- 
793      >>> from sklearn.metrics import (accuracy_score, classification_report, 
794      ... mean_absolute_error, mean_squared_error, median_absolute_error) 
795      >>> from sklearn.utils._testing import assert_docstring_consistency 
796      ... # doctest: +SKIP 
797      >>> assert_docstring_consistency([mean_absolute_error, mean_squared_error], 
798      ... include_params=['y_true', 'y_pred', 'sample_weight'])  # doctest: +SKIP 
799      >>> assert_docstring_consistency([median_absolute_error, mean_squared_error], 
800      ... include_params=True)  # doctest: +SKIP 
801      >>> assert_docstring_consistency([accuracy_score, classification_report], 
802      ... include_params=["y_true"], 
803      ... descr_regex_pattern=r"Ground truth \(correct\) (labels|target values)") 
804      ... # doctest: +SKIP 
805      """
806      from numpydoc.docscrape import NumpyDocString
807  
808      Args = namedtuple("args", ["include", "exclude", "arg_name"])
809  
810      def _create_args(include, exclude, arg_name, section_name):
811          if exclude and include is not True:
812              raise TypeError(
813                  f"The 'exclude_{arg_name}' argument can be set only when the "
814                  f"'include_{arg_name}' argument is True."
815              )
816          if include is False:
817              return {}
818          return {section_name: Args(include, exclude, arg_name)}
819  
820      section_args = {
821          **_create_args(include_params, exclude_params, "params", "Parameters"),
822          **_create_args(include_attrs, exclude_attrs, "attrs", "Attributes"),
823          **_create_args(include_returns, exclude_returns, "returns", "Returns"),
824      }
825  
826      objects_doc = dict()
827      for obj in objects:
828          if (
829              inspect.isdatadescriptor(obj)
830              or inspect.isfunction(obj)
831              or inspect.isclass(obj)
832          ):
833              objects_doc[obj.__name__] = NumpyDocString(inspect.getdoc(obj))
834          else:
835              raise TypeError(
836                  "All 'objects' must be one of: function, class or descriptor, "
837                  f"got a: {type(obj)}."
838              )
839  
840      n_objects = len(objects)
841      for section, args in section_args.items():
842          type_items = defaultdict(lambda: defaultdict(list))
843          desc_items = defaultdict(lambda: defaultdict(list))
844          for obj_name, obj_doc in objects_doc.items():
845              for item_name, type_def, desc in obj_doc[section]:
846                  if _check_item_included(item_name, args):
847                      # Normalize white space
848                      type_def = " ".join(type_def.strip().split())
849                      desc = " ".join(chain.from_iterable(line.split() for line in desc))
850                      # Use string type/desc as key, to group consistent objs together
851                      type_items[item_name][type_def].append(obj_name)
852                      desc_items[item_name][desc].append(obj_name)
853  
854          _check_consistency_items(type_items, "type specification", section, n_objects)
855          _check_consistency_items(
856              desc_items,
857              "description",
858              section,
859              n_objects,
860              descr_regex_pattern=descr_regex_pattern,
861          )

864  def assert_run_python_script_without_output(source_code, pattern=".+", timeout=60):
865      """Utility to check assertions in an independent Python subprocess. 
866   
867      The script provided in the source code should return 0 and the stdtout + 
868      stderr should not match the pattern `pattern`. 
869   
870      This is a port from cloudpickle https://github.com/cloudpipe/cloudpickle 
871   
872      Parameters 
873      ---------- 
874      source_code : str 
875          The Python source code to execute. 
876      pattern : str 
877          Pattern that the stdout + stderr should not match. By default, unless 
878          stdout + stderr are both empty, an error will be raised. 
879      timeout : int, default=60 
880          Time in seconds before timeout. 
881      """
882      fd, source_file = tempfile.mkstemp(suffix="_src_test_sklearn.py")
883      os.close(fd)
884      try:
885          with open(source_file, "wb") as f:
886              f.write(source_code.encode("utf-8"))
887          cmd = [sys.executable, source_file]
888          cwd = op.normpath(op.join(op.dirname(sklearn.__file__), ".."))
889          env = os.environ.copy()
890          try:
891              env["PYTHONPATH"] = os.pathsep.join([cwd, env["PYTHONPATH"]])
892          except KeyError:
893              env["PYTHONPATH"] = cwd
894          kwargs = {"cwd": cwd, "stderr": STDOUT, "env": env}
895          # If coverage is running, pass the config file to the subprocess
896          coverage_rc = os.environ.get("COVERAGE_PROCESS_START")
897          if coverage_rc:
898              kwargs["env"]["COVERAGE_PROCESS_START"] = coverage_rc
899  
900          kwargs["timeout"] = timeout
901          try:
902              try:
903                  out = check_output(cmd, **kwargs)
904              except CalledProcessError as e:
905                  raise RuntimeError(
906                      "script errored with output:\n%s" % e.output.decode("utf-8")
907                  )
908  
909              out = out.decode("utf-8")
910              if re.search(pattern, out):
911                  if pattern == ".+":
912                      expectation = "Expected no output"
913                  else:
914                      expectation = f"The output was not supposed to match {pattern!r}"
915  
916                  message = f"{expectation}, got the following output instead: {out!r}"
917                  raise AssertionError(message)
918          except TimeoutExpired as e:
919              raise RuntimeError(
920                  "script timeout, output so far:\n%s" % e.output.decode("utf-8")
921              )
922      finally:
923          os.unlink(source_file)

926  def _convert_container(
927      container,
928      constructor_name,
929      columns_name=None,
930      dtype=None,
931      minversion=None,
932      categorical_feature_names=None,
933  ):
934      """Convert a given container to a specific array-like with a dtype. 
935   
936      Parameters 
937      ---------- 
938      container : array-like 
939          The container to convert. 
940      constructor_name : {"list", "tuple", "array", "sparse", "dataframe", \ 
941              "series", "index", "slice", "sparse_csr", "sparse_csc", \ 
942              "sparse_csr_array", "sparse_csc_array", "pyarrow", "polars", \ 
943              "polars_series"} 
944          The type of the returned container. 
945      columns_name : index or array-like, default=None 
946          For pandas container supporting `columns_names`, it will affect 
947          specific names. 
948      dtype : dtype, default=None 
949          Force the dtype of the container. Does not apply to `"slice"` 
950          container. 
951      minversion : str, default=None 
952          Minimum version for package to install. 
953      categorical_feature_names : list of str, default=None 
954          List of column names to cast to categorical dtype. 
955   
956      Returns 
957      ------- 
958      converted_container 
959      """
960      if constructor_name == "list":
961          if dtype is None:
962              return list(container)
963          else:
964              return np.asarray(container, dtype=dtype).tolist()
965      elif constructor_name == "tuple":
966          if dtype is None:
967              return tuple(container)
968          else:
969              return tuple(np.asarray(container, dtype=dtype).tolist())
970      elif constructor_name == "array":
971          return np.asarray(container, dtype=dtype)
972      elif constructor_name in ("pandas", "dataframe"):
973          pd = pytest.importorskip("pandas", minversion=minversion)
974          result = pd.DataFrame(container, columns=columns_name, dtype=dtype, copy=False)
975          if categorical_feature_names is not None:
976              for col_name in categorical_feature_names:
977                  result[col_name] = result[col_name].astype("category")
978          return result
979      elif constructor_name == "pyarrow":
980          pa = pytest.importorskip("pyarrow", minversion=minversion)
981          array = np.asarray(container)
982          if columns_name is None:
983              columns_name = [f"col{i}" for i in range(array.shape[1])]
984          data = {name: array[:, i] for i, name in enumerate(columns_name)}
985          result = pa.Table.from_pydict(data)
986          if categorical_feature_names is not None:
987              for col_idx, col_name in enumerate(result.column_names):
988                  if col_name in categorical_feature_names:
989                      result = result.set_column(
990                          col_idx, col_name, result.column(col_name).dictionary_encode()
991                      )
992          return result
993      elif constructor_name == "polars":
994          pl = pytest.importorskip("polars", minversion=minversion)
995          result = pl.DataFrame(container, schema=columns_name, orient="row")
996          if categorical_feature_names is not None:
997              for col_name in categorical_feature_names:
998                  result = result.with_columns(pl.col(col_name).cast(pl.Categorical))
999          return result
1000     elif constructor_name == "series":
1001         pd = pytest.importorskip("pandas", minversion=minversion)
1002         return pd.Series(container, dtype=dtype)
1003     elif constructor_name == "polars_series":
1004         pl = pytest.importorskip("polars", minversion=minversion)
1005         return pl.Series(values=container)
1006     elif constructor_name == "index":
1007         pd = pytest.importorskip("pandas", minversion=minversion)
1008         return pd.Index(container, dtype=dtype)
1009     elif constructor_name == "slice":
1010         return slice(container[0], container[1])
1011     elif "sparse" in constructor_name:
1012         if not sp.sparse.issparse(container):
1013             # For scipy >= 1.13, sparse array constructed from 1d array may be
1014             # 1d or raise an exception. To avoid this, we make sure that the
1015             # input container is 2d. For more details, see
1016             # https://github.com/scipy/scipy/pull/18530#issuecomment-1878005149
1017             container = np.atleast_2d(container)
1018 
1019         if "array" in constructor_name and sp_version < parse_version("1.8"):
1020             raise ValueError(
1021                 f"{constructor_name} is only available with scipy>=1.8.0, got "
1022                 f"{sp_version}"
1023             )
1024         if constructor_name in ("sparse", "sparse_csr"):
1025             # sparse and sparse_csr are equivalent for legacy reasons
1026             return sp.sparse.csr_matrix(container, dtype=dtype)
1027         elif constructor_name == "sparse_csr_array":
1028             return sp.sparse.csr_array(container, dtype=dtype)
1029         elif constructor_name == "sparse_csc":
1030             return sp.sparse.csc_matrix(container, dtype=dtype)
1031         elif constructor_name == "sparse_csc_array":
1032             return sp.sparse.csc_array(container, dtype=dtype)

1035 def raises(expected_exc_type, match=None, may_pass=False, err_msg=None):
1036     """Context manager to ensure exceptions are raised within a code block. 
1037  
1038     This is similar to and inspired from pytest.raises, but supports a few 
1039     other cases. 
1040  
1041     This is only intended to be used in estimator_checks.py where we don't 
1042     want to use pytest. In the rest of the code base, just use pytest.raises 
1043     instead. 
1044  
1045     Parameters 
1046     ---------- 
1047     excepted_exc_type : Exception or list of Exception 
1048         The exception that should be raised by the block. If a list, the block 
1049         should raise one of the exceptions. 
1050     match : str or list of str, default=None 
1051         A regex that the exception message should match. If a list, one of 
1052         the entries must match. If None, match isn't enforced. 
1053     may_pass : bool, default=False 
1054         If True, the block is allowed to not raise an exception. Useful in 
1055         cases where some estimators may support a feature but others must 
1056         fail with an appropriate error message. By default, the context 
1057         manager will raise an exception if the block does not raise an 
1058         exception. 
1059     err_msg : str, default=None 
1060         If the context manager fails (e.g. the block fails to raise the 
1061         proper exception, or fails to match), then an AssertionError is 
1062         raised with this message. By default, an AssertionError is raised 
1063         with a default error message (depends on the kind of failure). Use 
1064         this to indicate how users should fix their estimators to pass the 
1065         checks. 
1066  
1067     Attributes 
1068     ---------- 
1069     raised_and_matched : bool 
1070         True if an exception was raised and a match was found, False otherwise. 
1071     """
1072     return _Raises(expected_exc_type, match, may_pass, err_msg)
1073 
1074 
1075 class _Raises(contextlib.AbstractContextManager):
1076     # see raises() for parameters
1077     def __init__(self, expected_exc_type, match, may_pass, err_msg):
1078         self.expected_exc_types = (
1079             expected_exc_type
1080             if isinstance(expected_exc_type, Iterable)
1081             else [expected_exc_type]
1082         )
1083         self.matches = [match] if isinstance(match, str) else match
1084         self.may_pass = may_pass
1085         self.err_msg = err_msg
1086         self.raised_and_matched = False
1087 
1088     def __exit__(self, exc_type, exc_value, _):
1089         # see
1090         # https://docs.python.org/2.5/whatsnew/pep-343.html#SECTION000910000000000000000
1091 
1092         if exc_type is None:  # No exception was raised in the block
1093             if self.may_pass:
1094                 return True  # CM is happy
1095             else:
1096                 err_msg = self.err_msg or f"Did not raise: {self.expected_exc_types}"
1097                 raise AssertionError(err_msg)
1098 
1099         if not any(
1100             issubclass(exc_type, expected_type)
1101             for expected_type in self.expected_exc_types
1102         ):
1103             if self.err_msg is not None:
1104                 raise AssertionError(self.err_msg) from exc_value
1105             else:
1106                 return False  # will re-raise the original exception
1107 
1108         if self.matches is not None:
1109             err_msg = self.err_msg or (
1110                 "The error message should contain one of the following "
1111                 "patterns:\n{}\nGot {}".format("\n".join(self.matches), str(exc_value))
1112             )
1113             if not any(re.search(match, str(exc_value)) for match in self.matches):
1114                 raise AssertionError(err_msg) from exc_value
1115             self.raised_and_matched = True
1116 
1117         return True
