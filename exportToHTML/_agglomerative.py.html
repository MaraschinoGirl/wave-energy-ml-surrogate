50   def _fix_connectivity(X, connectivity, affinity):
51       """ 
52       Fixes the connectivity matrix. 
53    
54       The different steps are: 
55    
56       - copies it 
57       - makes it symmetric 
58       - converts it to LIL if necessary 
59       - completes it if necessary. 
60    
61       Parameters 
62       ---------- 
63       X : array-like of shape (n_samples, n_features) 
64           Feature matrix representing `n_samples` samples to be clustered. 
65    
66       connectivity : sparse matrix, default=None 
67           Connectivity matrix. Defines for each sample the neighboring samples 
68           following a given structure of the data. The matrix is assumed to 
69           be symmetric and only the upper triangular half is used. 
70           Default is `None`, i.e, the Ward algorithm is unstructured. 
71    
72       affinity : {"euclidean", "precomputed"}, default="euclidean" 
73           Which affinity to use. At the moment `precomputed` and 
74           ``euclidean`` are supported. `euclidean` uses the 
75           negative squared Euclidean distance between points. 
76    
77       Returns 
78       ------- 
79       connectivity : sparse matrix 
80           The fixed connectivity matrix. 
81    
82       n_connected_components : int 
83           The number of connected components in the graph. 
84       """
85       n_samples = X.shape[0]
86       if connectivity.shape[0] != n_samples or connectivity.shape[1] != n_samples:
87           raise ValueError(
88               "Wrong shape for connectivity matrix: %s when X is %s"
89               % (connectivity.shape, X.shape)
90           )
91   
92       # Make the connectivity matrix symmetric:
93       connectivity = connectivity + connectivity.T
94   
95       # Convert connectivity matrix to LIL
96       if not sparse.issparse(connectivity):
97           connectivity = sparse.lil_matrix(connectivity)
98   
99       # `connectivity` is a sparse matrix at this point
100      if connectivity.format != "lil":
101          connectivity = connectivity.tolil()
102  
103      # Compute the number of nodes
104      n_connected_components, labels = connected_components(connectivity)
105  
106      if n_connected_components > 1:
107          warnings.warn(
108              "the number of connected components of the "
109              "connectivity matrix is %d > 1. Completing it to avoid "
110              "stopping the tree early." % n_connected_components,
111              stacklevel=2,
112          )
113          # XXX: Can we do without completing the matrix?
114          connectivity = _fix_connected_components(
115              X=X,
116              graph=connectivity,
117              n_connected_components=n_connected_components,
118              component_labels=labels,
119              metric=affinity,
120              mode="connectivity",
121          )
122  
123      return connectivity, n_connected_components

126  def _single_linkage_tree(
127      connectivity,
128      n_samples,
129      n_nodes,
130      n_clusters,
131      n_connected_components,
132      return_distance,
133  ):
134      """ 
135      Perform single linkage clustering on sparse data via the minimum 
136      spanning tree from scipy.sparse.csgraph, then using union-find to label. 
137      The parent array is then generated by walking through the tree. 
138      """
139      from scipy.sparse.csgraph import minimum_spanning_tree
140  
141      # explicitly cast connectivity to ensure safety
142      connectivity = connectivity.astype(np.float64, copy=False)
143  
144      # Ensure zero distances aren't ignored by setting them to "epsilon"
145      epsilon_value = np.finfo(dtype=connectivity.data.dtype).eps
146      connectivity.data[connectivity.data == 0] = epsilon_value
147  
148      # Use scipy.sparse.csgraph to generate a minimum spanning tree
149      mst = minimum_spanning_tree(connectivity.tocsr())
150  
151      # Convert the graph to scipy.cluster.hierarchy array format
152      mst = mst.tocoo()
153  
154      # Undo the epsilon values
155      mst.data[mst.data == epsilon_value] = 0
156  
157      mst_array = np.vstack([mst.row, mst.col, mst.data]).T
158  
159      # Sort edges of the min_spanning_tree by weight
160      mst_array = mst_array[np.argsort(mst_array.T[2], kind="mergesort"), :]
161  
162      # Convert edge list into standard hierarchical clustering format
163      single_linkage_tree = _hierarchical._single_linkage_label(mst_array)
164      children_ = single_linkage_tree[:, :2].astype(int)
165  
166      # Compute parents
167      parent = np.arange(n_nodes, dtype=np.intp)
168      for i, (left, right) in enumerate(children_, n_samples):
169          if n_clusters is not None and i >= n_nodes:
170              break
171          if left < n_nodes:
172              parent[left] = i
173          if right < n_nodes:
174              parent[right] = i
175  
176      if return_distance:
177          distances = single_linkage_tree[:, 2]
178          return children_, n_connected_components, n_samples, parent, distances
179      return children_, n_connected_components, n_samples, parent

195  def ward_tree(X, *, connectivity=None, n_clusters=None, return_distance=False):
196      """Ward clustering based on a Feature matrix. 
197   
198      Recursively merges the pair of clusters that minimally increases 
199      within-cluster variance. 
200   
201      The inertia matrix uses a Heapq-based representation. 
202   
203      This is the structured version, that takes into account some topological 
204      structure between samples. 
205   
206      Read more in the :ref:`User Guide <hierarchical_clustering>`. 
207   
208      Parameters 
209      ---------- 
210      X : array-like of shape (n_samples, n_features) 
211          Feature matrix representing `n_samples` samples to be clustered. 
212   
213      connectivity : {array-like, sparse matrix}, default=None 
214          Connectivity matrix. Defines for each sample the neighboring samples 
215          following a given structure of the data. The matrix is assumed to 
216          be symmetric and only the upper triangular half is used. 
217          Default is None, i.e, the Ward algorithm is unstructured. 
218   
219      n_clusters : int, default=None 
220          `n_clusters` should be less than `n_samples`.  Stop early the 
221          construction of the tree at `n_clusters.` This is useful to decrease 
222          computation time if the number of clusters is not small compared to the 
223          number of samples. In this case, the complete tree is not computed, thus 
224          the 'children' output is of limited use, and the 'parents' output should 
225          rather be used. This option is valid only when specifying a connectivity 
226          matrix. 
227   
228      return_distance : bool, default=False 
229          If `True`, return the distance between the clusters. 
230   
231      Returns 
232      ------- 
233      children : ndarray of shape (n_nodes-1, 2) 
234          The children of each non-leaf node. Values less than `n_samples` 
235          correspond to leaves of the tree which are the original samples. 
236          A node `i` greater than or equal to `n_samples` is a non-leaf 
237          node and has children `children_[i - n_samples]`. Alternatively 
238          at the i-th iteration, children[i][0] and children[i][1] 
239          are merged to form node `n_samples + i`. 
240   
241      n_connected_components : int 
242          The number of connected components in the graph. 
243   
244      n_leaves : int 
245          The number of leaves in the tree. 
246   
247      parents : ndarray of shape (n_nodes,) or None 
248          The parent of each node. Only returned when a connectivity matrix 
249          is specified, elsewhere 'None' is returned. 
250   
251      distances : ndarray of shape (n_nodes-1,) 
252          Only returned if `return_distance` is set to `True` (for compatibility). 
253          The distances between the centers of the nodes. `distances[i]` 
254          corresponds to a weighted Euclidean distance between 
255          the nodes `children[i, 1]` and `children[i, 2]`. If the nodes refer to 
256          leaves of the tree, then `distances[i]` is their unweighted Euclidean 
257          distance. Distances are updated in the following way 
258          (from scipy.hierarchy.linkage): 
259   
260          The new entry :math:`d(u,v)` is computed as follows, 
261   
262          .. math:: 
263   
264             d(u,v) = \\sqrt{\\frac{|v|+|s|} 
265                                 {T}d(v,s)^2 
266                          + \\frac{|v|+|t|} 
267                                 {T}d(v,t)^2 
268                          - \\frac{|v|} 
269                                 {T}d(s,t)^2} 
270   
271          where :math:`u` is the newly joined cluster consisting of 
272          clusters :math:`s` and :math:`t`, :math:`v` is an unused 
273          cluster in the forest, :math:`T=|v|+|s|+|t|`, and 
274          :math:`|*|` is the cardinality of its argument. This is also 
275          known as the incremental algorithm. 
276   
277      Examples 
278      -------- 
279      >>> import numpy as np 
280      >>> from sklearn.cluster import ward_tree 
281      >>> X = np.array([[1, 2], [1, 4], [1, 0], 
282      ...               [4, 2], [4, 4], [4, 0]]) 
283      >>> children, n_connected_components, n_leaves, parents = ward_tree(X) 
284      >>> children 
285      array([[0, 1], 
286             [3, 5], 
287             [2, 6], 
288             [4, 7], 
289             [8, 9]]) 
290      >>> n_connected_components 
291      1 
292      >>> n_leaves 
293      6 
294      """
295      X = np.asarray(X)
296      if X.ndim == 1:
297          X = np.reshape(X, (-1, 1))
298      n_samples, n_features = X.shape
299  
300      if connectivity is None:
301          from scipy.cluster import hierarchy  # imports PIL
302  
303          if n_clusters is not None:
304              warnings.warn(
305                  (
306                      "Partial build of the tree is implemented "
307                      "only for structured clustering (i.e. with "
308                      "explicit connectivity). The algorithm "
309                      "will build the full tree and only "
310                      "retain the lower branches required "
311                      "for the specified number of clusters"
312                  ),
313                  stacklevel=2,
314              )
315          X = np.require(X, requirements="W")
316          out = hierarchy.ward(X)
317          children_ = out[:, :2].astype(np.intp)
318  
319          if return_distance:
320              distances = out[:, 2]
321              return children_, 1, n_samples, None, distances
322          else:
323              return children_, 1, n_samples, None
324  
325      connectivity, n_connected_components = _fix_connectivity(
326          X, connectivity, affinity="euclidean"
327      )
328      if n_clusters is None:
329          n_nodes = 2 * n_samples - 1
330      else:
331          if n_clusters > n_samples:
332              raise ValueError(
333                  "Cannot provide more clusters than samples. "
334                  "%i n_clusters was asked, and there are %i "
335                  "samples." % (n_clusters, n_samples)
336              )
337          n_nodes = 2 * n_samples - n_clusters
338  
339      # create inertia matrix
340      coord_row = []
341      coord_col = []
342      A = []
343      for ind, row in enumerate(connectivity.rows):
344          A.append(row)
345          # We keep only the upper triangular for the moments
346          # Generator expressions are faster than arrays on the following
347          row = [i for i in row if i < ind]
348          coord_row.extend(
349              len(row)
350              * [
351                  ind,
352              ]
353          )
354          coord_col.extend(row)
355  
356      coord_row = np.array(coord_row, dtype=np.intp, order="C")
357      coord_col = np.array(coord_col, dtype=np.intp, order="C")
358  
359      # build moments as a list
360      moments_1 = np.zeros(n_nodes, order="C")
361      moments_1[:n_samples] = 1
362      moments_2 = np.zeros((n_nodes, n_features), order="C")
363      moments_2[:n_samples] = X
364      inertia = np.empty(len(coord_row), dtype=np.float64, order="C")
365      _hierarchical.compute_ward_dist(moments_1, moments_2, coord_row, coord_col, inertia)
366      inertia = list(zip(inertia, coord_row, coord_col))
367      heapify(inertia)
368  
369      # prepare the main fields
370      parent = np.arange(n_nodes, dtype=np.intp)
371      used_node = np.ones(n_nodes, dtype=bool)
372      children = []
373      if return_distance:
374          distances = np.empty(n_nodes - n_samples)
375  
376      not_visited = np.empty(n_nodes, dtype=bool, order="C")
377  
378      # recursive merge loop
379      for k in range(n_samples, n_nodes):
380          # identify the merge
381          while True:
382              inert, i, j = heappop(inertia)
383              if used_node[i] and used_node[j]:
384                  break
385          parent[i], parent[j] = k, k
386          children.append((i, j))
387          used_node[i] = used_node[j] = False
388          if return_distance:  # store inertia value
389              distances[k - n_samples] = inert
390  
391          # update the moments
392          moments_1[k] = moments_1[i] + moments_1[j]
393          moments_2[k] = moments_2[i] + moments_2[j]
394  
395          # update the structure matrix A and the inertia matrix
396          coord_col = []
397          not_visited.fill(1)
398          not_visited[k] = 0
399          _hierarchical._get_parents(A[i], coord_col, parent, not_visited)
400          _hierarchical._get_parents(A[j], coord_col, parent, not_visited)
401          # List comprehension is faster than a for loop
402          [A[col].append(k) for col in coord_col]
403          A.append(coord_col)
404          coord_col = np.array(coord_col, dtype=np.intp, order="C")
405          coord_row = np.empty(coord_col.shape, dtype=np.intp, order="C")
406          coord_row.fill(k)
407          n_additions = len(coord_row)
408          ini = np.empty(n_additions, dtype=np.float64, order="C")
409  
410          _hierarchical.compute_ward_dist(moments_1, moments_2, coord_row, coord_col, ini)
411  
412          # List comprehension is faster than a for loop
413          [heappush(inertia, (ini[idx], k, coord_col[idx])) for idx in range(n_additions)]
414  
415      # Separate leaves in children (empty lists up to now)
416      n_leaves = n_samples
417      # sort children to get consistent output with unstructured version
418      children = [c[::-1] for c in children]
419      children = np.array(children)  # return numpy array for efficient caching
420  
421      if return_distance:
422          # 2 is scaling factor to compare w/ unstructured version
423          distances = np.sqrt(2.0 * distances)
424          return children, n_connected_components, n_leaves, parent, distances
425      else:
426          return children, n_connected_components, n_leaves, parent

429  # single average and complete linkage
430  def linkage_tree(
431      X,
432      connectivity=None,
433      n_clusters=None,
434      linkage="complete",
435      affinity="euclidean",
436      return_distance=False,
437  ):
438      """Linkage agglomerative clustering based on a Feature matrix. 
439   
440      The inertia matrix uses a Heapq-based representation. 
441   
442      This is the structured version, that takes into account some topological 
443      structure between samples. 
444   
445      Read more in the :ref:`User Guide <hierarchical_clustering>`. 
446   
447      Parameters 
448      ---------- 
449      X : array-like of shape (n_samples, n_features) 
450          Feature matrix representing `n_samples` samples to be clustered. 
451   
452      connectivity : sparse matrix, default=None 
453          Connectivity matrix. Defines for each sample the neighboring samples 
454          following a given structure of the data. The matrix is assumed to 
455          be symmetric and only the upper triangular half is used. 
456          Default is `None`, i.e, the Ward algorithm is unstructured. 
457   
458      n_clusters : int, default=None 
459          Stop early the construction of the tree at `n_clusters`. This is 
460          useful to decrease computation time if the number of clusters is 
461          not small compared to the number of samples. In this case, the 
462          complete tree is not computed, thus the 'children' output is of 
463          limited use, and the 'parents' output should rather be used. 
464          This option is valid only when specifying a connectivity matrix. 
465   
466      linkage : {"average", "complete", "single"}, default="complete" 
467          Which linkage criteria to use. The linkage criterion determines which 
468          distance to use between sets of observation. 
469              - "average" uses the average of the distances of each observation of 
470                the two sets. 
471              - "complete" or maximum linkage uses the maximum distances between 
472                all observations of the two sets. 
473              - "single" uses the minimum of the distances between all 
474                observations of the two sets. 
475   
476      affinity : str or callable, default='euclidean' 
477          Which metric to use. Can be 'euclidean', 'manhattan', or any 
478          distance known to paired distance (see metric.pairwise). 
479   
480      return_distance : bool, default=False 
481          Whether or not to return the distances between the clusters. 
482   
483      Returns 
484      ------- 
485      children : ndarray of shape (n_nodes-1, 2) 
486          The children of each non-leaf node. Values less than `n_samples` 
487          correspond to leaves of the tree which are the original samples. 
488          A node `i` greater than or equal to `n_samples` is a non-leaf 
489          node and has children `children_[i - n_samples]`. Alternatively 
490          at the i-th iteration, children[i][0] and children[i][1] 
491          are merged to form node `n_samples + i`. 
492   
493      n_connected_components : int 
494          The number of connected components in the graph. 
495   
496      n_leaves : int 
497          The number of leaves in the tree. 
498   
499      parents : ndarray of shape (n_nodes, ) or None 
500          The parent of each node. Only returned when a connectivity matrix 
501          is specified, elsewhere 'None' is returned. 
502   
503      distances : ndarray of shape (n_nodes-1,) 
504          Returned when `return_distance` is set to `True`. 
505   
506          distances[i] refers to the distance between children[i][0] and 
507          children[i][1] when they are merged. 
508   
509      See Also 
510      -------- 
511      ward_tree : Hierarchical clustering with ward linkage. 
512      """
513      X = np.asarray(X)
514      if X.ndim == 1:
515          X = np.reshape(X, (-1, 1))
516      n_samples, n_features = X.shape
517  
518      linkage_choices = {
519          "complete": _hierarchical.max_merge,
520          "average": _hierarchical.average_merge,
521          "single": None,
522      }  # Single linkage is handled differently
523      try:
524          join_func = linkage_choices[linkage]
525      except KeyError as e:
526          raise ValueError(
527              "Unknown linkage option, linkage should be one of %s, but %s was given"
528              % (linkage_choices.keys(), linkage)
529          ) from e
530  
531      if affinity == "cosine" and np.any(~np.any(X, axis=1)):
532          raise ValueError("Cosine affinity cannot be used when X contains zero vectors")
533  
534      if connectivity is None:
535          from scipy.cluster import hierarchy  # imports PIL
536  
537          if n_clusters is not None:
538              warnings.warn(
539                  (
540                      "Partial build of the tree is implemented "
541                      "only for structured clustering (i.e. with "
542                      "explicit connectivity). The algorithm "
543                      "will build the full tree and only "
544                      "retain the lower branches required "
545                      "for the specified number of clusters"
546                  ),
547                  stacklevel=2,
548              )
549  
550          if affinity == "precomputed":
551              # for the linkage function of hierarchy to work on precomputed
552              # data, provide as first argument an ndarray of the shape returned
553              # by sklearn.metrics.pairwise_distances.
554              if X.shape[0] != X.shape[1]:
555                  raise ValueError(
556                      f"Distance matrix should be square, got matrix of shape {X.shape}"
557                  )
558              i, j = np.triu_indices(X.shape[0], k=1)
559              X = X[i, j]
560          elif affinity == "l2":
561              # Translate to something understood by scipy
562              affinity = "euclidean"
563          elif affinity in ("l1", "manhattan"):
564              affinity = "cityblock"
565          elif callable(affinity):
566              X = affinity(X)
567              i, j = np.triu_indices(X.shape[0], k=1)
568              X = X[i, j]
569          if (
570              linkage == "single"
571              and affinity != "precomputed"
572              and not callable(affinity)
573              and affinity in METRIC_MAPPING64
574          ):
575              # We need the fast cythonized metric from neighbors
576              dist_metric = DistanceMetric.get_metric(affinity)
577  
578              # The Cython routines used require contiguous arrays
579              X = np.ascontiguousarray(X, dtype=np.double)
580  
581              mst = _hierarchical.mst_linkage_core(X, dist_metric)
582              # Sort edges of the min_spanning_tree by weight
583              mst = mst[np.argsort(mst.T[2], kind="mergesort"), :]
584  
585              # Convert edge list into standard hierarchical clustering format
586              out = _hierarchical.single_linkage_label(mst)
587          else:
588              out = hierarchy.linkage(X, method=linkage, metric=affinity)
589          children_ = out[:, :2].astype(int, copy=False)
590  
591          if return_distance:
592              distances = out[:, 2]
593              return children_, 1, n_samples, None, distances
594          return children_, 1, n_samples, None
595  
596      connectivity, n_connected_components = _fix_connectivity(
597          X, connectivity, affinity=affinity
598      )
599      connectivity = connectivity.tocoo()
600      # Put the diagonal to zero
601      diag_mask = connectivity.row != connectivity.col
602      connectivity.row = connectivity.row[diag_mask]
603      connectivity.col = connectivity.col[diag_mask]
604      connectivity.data = connectivity.data[diag_mask]
605      del diag_mask
606  
607      if affinity == "precomputed":
608          distances = X[connectivity.row, connectivity.col].astype(np.float64, copy=False)
609      else:
610          # FIXME We compute all the distances, while we could have only computed
611          # the "interesting" distances
612          distances = paired_distances(
613              X[connectivity.row], X[connectivity.col], metric=affinity
614          )
615      connectivity.data = distances
616  
617      if n_clusters is None:
618          n_nodes = 2 * n_samples - 1
619      else:
620          assert n_clusters <= n_samples
621          n_nodes = 2 * n_samples - n_clusters
622  
623      if linkage == "single":
624          return _single_linkage_tree(
625              connectivity,
626              n_samples,
627              n_nodes,
628              n_clusters,
629              n_connected_components,
630              return_distance,
631          )
632  
633      if return_distance:
634          distances = np.empty(n_nodes - n_samples)
635      # create inertia heap and connection matrix
636      A = np.empty(n_nodes, dtype=object)
637      inertia = list()
638  
639      # LIL seems to the best format to access the rows quickly,
640      # without the numpy overhead of slicing CSR indices and data.
641      connectivity = connectivity.tolil()
642      # We are storing the graph in a list of IntFloatDict
643      for ind, (data, row) in enumerate(zip(connectivity.data, connectivity.rows)):
644          A[ind] = IntFloatDict(
645              np.asarray(row, dtype=np.intp), np.asarray(data, dtype=np.float64)
646          )
647          # We keep only the upper triangular for the heap
648          # Generator expressions are faster than arrays on the following
649          inertia.extend(
650              _hierarchical.WeightedEdge(d, ind, r) for r, d in zip(row, data) if r < ind
651          )
652      del connectivity
653  
654      heapify(inertia)
655  
656      # prepare the main fields
657      parent = np.arange(n_nodes, dtype=np.intp)
658      used_node = np.ones(n_nodes, dtype=np.intp)
659      children = []
660  
661      # recursive merge loop
662      for k in range(n_samples, n_nodes):
663          # identify the merge
664          while True:
665              edge = heappop(inertia)
666              if used_node[edge.a] and used_node[edge.b]:
667                  break
668          i = edge.a
669          j = edge.b
670  
671          if return_distance:
672              # store distances
673              distances[k - n_samples] = edge.weight
674  
675          parent[i] = parent[j] = k
676          children.append((i, j))
677          # Keep track of the number of elements per cluster
678          n_i = used_node[i]
679          n_j = used_node[j]
680          used_node[k] = n_i + n_j
681          used_node[i] = used_node[j] = False
682  
683          # update the structure matrix A and the inertia matrix
684          # a clever 'min', or 'max' operation between A[i] and A[j]
685          coord_col = join_func(A[i], A[j], used_node, n_i, n_j)
686          for col, d in coord_col:
687              A[col].append(k, d)
688              # Here we use the information from coord_col (containing the
689              # distances) to update the heap
690              heappush(inertia, _hierarchical.WeightedEdge(d, k, col))
691          A[k] = coord_col
692          # Clear A[i] and A[j] to save memory
693          A[i] = A[j] = 0
694  
695      # Separate leaves in children (empty lists up to now)
696      n_leaves = n_samples
697  
698      # # return numpy array for efficient caching
699      children = np.array(children)[:, ::-1]
700  
701      if return_distance:
702          return children, n_connected_components, n_leaves, parent, distances
703      return children, n_connected_components, n_leaves, parent
704  
705  
706  # Matching names to tree-building strategies
707  def _complete_linkage(*args, **kwargs):
708      kwargs["linkage"] = "complete"
709      return linkage_tree(*args, **kwargs)
710  
711  
712  def _average_linkage(*args, **kwargs):
713      kwargs["linkage"] = "average"
714      return linkage_tree(*args, **kwargs)
715  
716  
717  def _single_linkage(*args, **kwargs):
718      kwargs["linkage"] = "single"
719      return linkage_tree(*args, **kwargs)
720

723      ward=ward_tree,
724      complete=_complete_linkage,
725      average=_average_linkage,
726      single=_single_linkage,

733  def _hc_cut(n_clusters, children, n_leaves):
734      """Function cutting the ward tree for a given number of clusters. 
735   
736      Parameters 
737      ---------- 
738      n_clusters : int or ndarray 
739          The number of clusters to form. 
740   
741      children : ndarray of shape (n_nodes-1, 2) 
742          The children of each non-leaf node. Values less than `n_samples` 
743          correspond to leaves of the tree which are the original samples. 
744          A node `i` greater than or equal to `n_samples` is a non-leaf 
745          node and has children `children_[i - n_samples]`. Alternatively 
746          at the i-th iteration, children[i][0] and children[i][1] 
747          are merged to form node `n_samples + i`. 
748   
749      n_leaves : int 
750          Number of leaves of the tree. 
751   
752      Returns 
753      ------- 
754      labels : array [n_samples] 
755          Cluster labels for each point. 
756      """
757      if n_clusters > n_leaves:
758          raise ValueError(
759              "Cannot extract more clusters than samples: "
760              f"{n_clusters} clusters were given for a tree with {n_leaves} leaves."
761          )
762      # In this function, we store nodes as a heap to avoid recomputing
763      # the max of the nodes: the first element is always the smallest
764      # We use negated indices as heaps work on smallest elements, and we
765      # are interested in largest elements
766      # children[-1] is the root of the tree
767      nodes = [-(max(children[-1]) + 1)]
768      for _ in range(n_clusters - 1):
769          # As we have a heap, nodes[0] is the smallest element
770          these_children = children[-nodes[0] - n_leaves]
771          # Insert the 2 children and remove the largest node
772          heappush(nodes, -these_children[0])
773          heappushpop(nodes, -these_children[1])
774      label = np.zeros(n_leaves, dtype=np.intp)
775      for i, node in enumerate(nodes):
776          label[_hierarchical._hc_get_descendent(-node, children, n_leaves)] = i
777      return label

785      Agglomerative Clustering. 
786   
787      Recursively merges pair of clusters of sample data; uses linkage distance. 
788   
789      Read more in the :ref:`User Guide <hierarchical_clustering>`. 
790   
791      Parameters 
792      ---------- 
793      n_clusters : int or None, default=2 
794          The number of clusters to find. It must be ``None`` if 
795          ``distance_threshold`` is not ``None``.

798          Metric used to compute the linkage. Can be "euclidean", "l1", "l2", 
799          "manhattan", "cosine", or "precomputed". If linkage is "ward", only 
800          "euclidean" is accepted. If "precomputed", a distance matrix is needed 
801          as input for the fit method. If connectivity is None, linkage is

810          path to the caching directory.

839          the pairs of cluster that minimize this criterion.

842          - 'average' uses the average of the distances of each observation of 
843            the two sets.

845            all observations of the two sets. 
846          - 'single' uses the minimum of the distances between all observations

898          has feature names that are all strings. 
899   
900          .. versionadded:: 1.0 
901   
902      children_ : array-like of shape (n_samples-1, 2) 
903          The children of each non-leaf node. Values less than `n_samples` 
904          correspond to leaves of the tree which are the original samples. 
905          A node `i` greater than or equal to `n_samples` is a non-leaf 
906          node and has children `children_[i - n_samples]`. Alternatively 
907          at the i-th iteration, children[i][0] and children[i][1]

977              Training instances to cluster, or distances between instances if

991      def _fit(self, X):
992          """Fit without validation 
993   
994          Parameters 
995          ---------- 
996          X : ndarray of shape (n_samples, n_features) or (n_samples, n_samples) 
997              Training instances to cluster, or distances between instances if 
998              ``metric='precomputed'``. 
999   
1000         Returns 
1001         ------- 
1002         self : object 
1003             Returns the fitted instance. 
1004         """
1005         memory = check_memory(self.memory)
1006 
1007         if not ((self.n_clusters is None) ^ (self.distance_threshold is None)):
1008             raise ValueError(
1009                 "Exactly one of n_clusters and "
1010                 "distance_threshold has to be set, and the other "
1011                 "needs to be None."
1012             )
1013 
1014         if self.distance_threshold is not None and not self.compute_full_tree:
1015             raise ValueError(
1016                 "compute_full_tree must be True if distance_threshold is set."
1017             )
1018 
1019         if self.linkage == "ward" and self.metric != "euclidean":
1020             raise ValueError(
1021                 f"{self.metric} was provided as metric. Ward can only "
1022                 "work with euclidean distances."
1023             )
1024 
1025         tree_builder = _TREE_BUILDERS[self.linkage]
1026 
1027         connectivity = self.connectivity
1028         if self.connectivity is not None:
1029             if callable(self.connectivity):
1030                 connectivity = self.connectivity(X)
1031             connectivity = check_array(
1032                 connectivity, accept_sparse=["csr", "coo", "lil"]
1033             )
1034 
1035         n_samples = len(X)
1036         compute_full_tree = self.compute_full_tree
1037         if self.connectivity is None:
1038             compute_full_tree = True
1039         if compute_full_tree == "auto":
1040             if self.distance_threshold is not None:
1041                 compute_full_tree = True
1042             else:
1043                 # Early stopping is likely to give a speed up only for
1044                 # a large number of clusters. The actual threshold
1045                 # implemented here is heuristic
1046                 compute_full_tree = self.n_clusters < max(100, 0.02 * n_samples)
1047         n_clusters = self.n_clusters
1048         if compute_full_tree:
1049             n_clusters = None
1050 
1051         # Construct the tree
1052         kwargs = {}
1053         if self.linkage != "ward":
1054             kwargs["linkage"] = self.linkage
1055             kwargs["affinity"] = self.metric
1056 
1057         distance_threshold = self.distance_threshold
1058 
1059         return_distance = (distance_threshold is not None) or self.compute_distances
1060 
1061         out = memory.cache(tree_builder)(
1062             X,
1063             connectivity=connectivity,
1064             n_clusters=n_clusters,
1065             return_distance=return_distance,
1066             **kwargs,
1067         )
1068         (self.children_, self.n_connected_components_, self.n_leaves_, parents) = out[
1069             :4
1070         ]
1071 
1072         if return_distance:
1073             self.distances_ = out[-1]
1074 
1075         if self.distance_threshold is not None:  # distance_threshold is used
1076             self.n_clusters_ = (
1077                 np.count_nonzero(self.distances_ >= distance_threshold) + 1
1078             )
1079         else:  # n_clusters is used
1080             self.n_clusters_ = self.n_clusters
1081 
1082         # Cut the tree
1083         if compute_full_tree:
1084             self.labels_ = _hc_cut(self.n_clusters_, self.children_, self.n_leaves_)
1085         else:
1086             labels = _hierarchical.hc_get_heads(parents, copy=False)
1087             # copy to avoid holding a reference on the original array
1088             labels = np.copy(labels[:n_samples])
1089             # Reassign cluster numbers
1090             self.labels_ = np.searchsorted(np.unique(labels), labels)
1091         return self
