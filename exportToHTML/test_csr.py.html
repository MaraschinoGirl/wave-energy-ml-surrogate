78   def test_csr_empty_slices(matrix_input, axis, expected_shape):
79       # see gh-11127 for related discussion
80       slice_1 = matrix_input.toarray().shape[0] - 1
81       slice_2 = slice_1
82       slice_3 = slice_2 - 1
83   
84       if axis == 0:
85           actual_shape_1 = matrix_input[slice_1:slice_2, :].toarray().shape
86           actual_shape_2 = matrix_input[slice_1:slice_3, :].toarray().shape
87       elif axis == 1:
88           actual_shape_1 = matrix_input[:, slice_1:slice_2].toarray().shape
89           actual_shape_2 = matrix_input[:, slice_1:slice_3].toarray().shape
90       elif axis == 'both':
91           actual_shape_1 = matrix_input[slice_1:slice_2, slice_1:slice_2].toarray().shape
92           actual_shape_2 = matrix_input[slice_1:slice_3, slice_1:slice_3].toarray().shape
93   
94       assert actual_shape_1 == expected_shape
95       assert actual_shape_1 == actual_shape_2

117  def test_csr_hstack_int64():
118      """ 
119      Tests if hstack properly promotes to indices and indptr arrays to np.int64 
120      when using np.int32 during concatenation would result in either array 
121      overflowing. 
122      """
123      max_int32 = np.iinfo(np.int32).max
124  
125      # First case: indices would overflow with int32
126      data = [1.0]
127      row = [0]
128  
129      max_indices_1 = max_int32 - 1
130      max_indices_2 = 3
131  
132      # Individual indices arrays are representable with int32
133      col_1 = [max_indices_1 - 1]
134      col_2 = [max_indices_2 - 1]
135  
136      X_1 = csr_matrix((data, (row, col_1)))
137      X_2 = csr_matrix((data, (row, col_2)))
138  
139      assert max(max_indices_1 - 1, max_indices_2 - 1) < max_int32
140      assert X_1.indices.dtype == X_1.indptr.dtype == np.int32
141      assert X_2.indices.dtype == X_2.indptr.dtype == np.int32
142  
143      # ... but when concatenating their CSR matrices, the resulting indices
144      # array can't be represented with int32 and must be promoted to int64.
145      X_hs = hstack([X_1, X_2], format="csr")
146  
147      assert X_hs.indices.max() == max_indices_1 + max_indices_2 - 1
148      assert max_indices_1 + max_indices_2 - 1 > max_int32
149      assert X_hs.indices.dtype == X_hs.indptr.dtype == np.int64
150  
151      # Even if the matrices are empty, we must account for their size
152      # contribution so that we may safely set the final elements.
153      X_1_empty = csr_matrix(X_1.shape)
154      X_2_empty = csr_matrix(X_2.shape)
155      X_hs_empty = hstack([X_1_empty, X_2_empty], format="csr")
156  
157      assert X_hs_empty.shape == X_hs.shape
158      assert X_hs_empty.indices.dtype == np.int64
159  
160      # Should be just small enough to stay in int32 after stack. Note that
161      # we theoretically could support indices.max() == max_int32, but due to an
162      # edge-case in the underlying sparsetools code
163      # (namely the `coo_tocsr` routine),
164      # we require that max(X_hs_32.shape) < max_int32 as well.
165      # Hence we can only support max_int32 - 1.
166      col_3 = [max_int32 - max_indices_1 - 1]
167      X_3 = csr_matrix((data, (row, col_3)))
168      X_hs_32 = hstack([X_1, X_3], format="csr")
169      assert X_hs_32.indices.dtype == np.int32
170      assert X_hs_32.indices.max() == max_int32 - 1

172  @pytest.mark.parametrize("cls", [csr_matrix, csr_array, csc_matrix, csc_array])
173  def test_mixed_index_dtype_int_indexing(cls):
174      # https://github.com/scipy/scipy/issues/20182
175      rng = np.random.default_rng(0)
176      base_mtx = cls(sparse.random(50, 50, random_state=rng, density=0.1))
177      indptr_64bit = base_mtx.copy()
178      indices_64bit = base_mtx.copy()
179      indptr_64bit.indptr = base_mtx.indptr.astype(np.int64)
180      indices_64bit.indices = base_mtx.indices.astype(np.int64)
181  
182      for mtx in [base_mtx, indptr_64bit, indices_64bit]:
183          np.testing.assert_array_equal(
184              mtx[[1,2], :].toarray(),
185              base_mtx[[1, 2], :].toarray()
186          )
187          np.testing.assert_array_equal(
188              mtx[:, [1, 2]].toarray(),
189              base_mtx[:, [1, 2]].toarray()
190          )

192  def test_broadcast_to():
193      a = np.array([1, 0, 2])
194      b = np.array([3])
195      e = np.zeros((0,))
196      res_a = csr_array(a)._broadcast_to((2,3))
197      res_b = csr_array(b)._broadcast_to((4,))
198      res_c = csr_array(b)._broadcast_to((2,4))
199      res_d = csr_array(b)._broadcast_to((1,))
200      res_e = csr_array(e)._broadcast_to((4,0))
201      assert_array_equal(res_a.toarray(), np.broadcast_to(a, (2,3)))
202      assert_array_equal(res_b.toarray(), np.broadcast_to(b, (4,)))
203      assert_array_equal(res_c.toarray(), np.broadcast_to(b, (2,4)))
204      assert_array_equal(res_d.toarray(), np.broadcast_to(b, (1,)))
205      assert_array_equal(res_e.toarray(), np.broadcast_to(e, (4,0)))
206  
207      with pytest.raises(ValueError, match="cannot be broadcast"):
208          csr_matrix([[1, 2, 0], [3, 0, 1]])._broadcast_to(shape=(2, 1))
209  
210      with pytest.raises(ValueError, match="cannot be broadcast"):
211          csr_matrix([[0, 1, 2]])._broadcast_to(shape=(3, 2))
212  
213      with pytest.raises(ValueError, match="cannot be broadcast"):
214          csr_array([0, 1, 2])._broadcast_to(shape=(3, 2))
