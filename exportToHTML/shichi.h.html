5    /*                                                     shichi.c 
6     * 
7     *     Hyperbolic sine and cosine integrals 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double x, Chi, Shi, shichi(); 
14    * 
15    * shichi( x, &Chi, &Shi ); 
16    * 
17    * 
18    * DESCRIPTION: 
19    * 
20    * Approximates the integrals 
21    * 
22    *                            x 
23    *                            - 
24    *                           | |   cosh t - 1 
25    *   Chi(x) = eul + ln x +   |    -----------  dt, 
26    *                         | |          t 
27    *                          - 
28    *                          0 
29    * 
30    *               x 
31    *               - 
32    *              | |  sinh t 
33    *   Shi(x) =   |    ------  dt 
34    *            | |       t 
35    *             - 
36    *             0 
37    * 
38    * where eul = 0.57721566490153286061 is Euler's constant. 
39    * The integrals are evaluated by power series for x < 8 
40    * and by Chebyshev expansions for x between 8 and 88. 
41    * For large x, both functions approach exp(x)/2x. 
42    * Arguments greater than 88 in magnitude return INFINITY. 
43    * 
44    * 
45    * ACCURACY: 
46    * 
47    * Test interval 0 to 88. 
48    *                      Relative error: 
49    * arithmetic   function  # trials      peak         rms 
50    *    IEEE         Shi      30000       6.9e-16     1.6e-16 
51    *        Absolute error, except relative when |Chi| > 1: 
52    *    IEEE         Chi      30000       8.4e-16     1.4e-16 
53    */

116          /* 
117           * Evaluate 3F0(a1, a2, a3; z) 
118           * 
119           * The series is only asymptotic, so this requires z large enough. 
120           */
121          XSF_HOST_DEVICE inline double hyp3f0(double a1, double a2, double a3, double z) {
122              int n, maxiter;
123              double err, sum, term, m;
124  
125              m = std::pow(z, -1.0 / 3);
126              if (m < 50) {
127                  maxiter = m;
128              } else {
129                  maxiter = 50;
130              }
131  
132              term = 1.0;
133              sum = term;
134              for (n = 0; n < maxiter; ++n) {
135                  term *= (a1 + n) * (a2 + n) * (a3 + n) * z / (n + 1);
136                  sum += term;
137                  if (std::abs(term) < 1e-13 * std::abs(sum) || term == 0) {
138                      break;
139                  }
140              }
141  
142              err = std::abs(term);
143  
144              if (err > 1e-13 * std::abs(sum)) {
145                  return std::numeric_limits<double>::quiet_NaN();
146              }
147  
148              return sum;
149          }

153      /* Sine and cosine integrals */
154      XSF_HOST_DEVICE inline int shichi(double x, double *si, double *ci) {
155          double k, z, c, s, a, b;
156          short sign;
157  
158          if (x < 0.0) {
159              sign = -1;
160              x = -x;
161          } else {
162              sign = 0;
163          }
164  
165          if (x == 0.0) {
166              *si = 0.0;
167              *ci = -std::numeric_limits<double>::infinity();
168              return (0);
169          }
170  
171          if (x >= 8.0) {
172              goto chb;
173          }
174  
175          if (x >= 88.0) {
176              goto asymp;
177          }
178  
179          z = x * x;
180  
181          /*     Direct power series expansion   */
182          a = 1.0;
183          s = 1.0;
184          c = 0.0;
185          k = 2.0;
186  
187          do {
188              a *= z / k;
189              c += a / k;
190              k += 1.0;
191              a /= k;
192              s += a / k;
193              k += 1.0;
194          } while (std::abs(a / s) > detail::MACHEP);
195  
196          s *= x;
197          goto done;
198  
199      chb:
200          /* Chebyshev series expansions */
201          if (x < 18.0) {
202              a = (576.0 / x - 52.0) / 10.0;
203              k = std::exp(x) / x;
204              s = k * chbevl(a, detail::shichi_S1, 22);
205              c = k * chbevl(a, detail::shichi_C1, 23);
206              goto done;
207          }
208  
209          if (x <= 88.0) {
210              a = (6336.0 / x - 212.0) / 70.0;
211              k = std::exp(x) / x;
212              s = k * chbevl(a, detail::shichi_S2, 23);
213              c = k * chbevl(a, detail::shichi_C2, 24);
214              goto done;
215          }
216  
217      asymp:
218          if (x > 1000) {
219              *si = std::numeric_limits<double>::infinity();
220              *ci = std::numeric_limits<double>::infinity();
221          } else {
222              /* Asymptotic expansions 
223               * http://functions.wolfram.com/GammaBetaErf/CoshIntegral/06/02/ 
224               * http://functions.wolfram.com/GammaBetaErf/SinhIntegral/06/02/0001/ 
225               */
226              a = detail::hyp3f0(0.5, 1, 1, 4.0 / (x * x));
227              b = detail::hyp3f0(1, 1, 1.5, 4.0 / (x * x));
228              *si = std::cosh(x) / x * a + std::sinh(x) / (x * x) * b;
229              *ci = std::sinh(x) / x * a + std::cosh(x) / (x * x) * b;
230          }
231          if (sign) {
232              *si = -*si;
233          }
234          return 0;
235  
236      done:
237          if (sign) {
238              s = -s;
239          }
240  
241          *si = s;
242  
243          *ci = detail::SCIPY_EULER + std::log(x) + c;
244          return (0);
245      }
