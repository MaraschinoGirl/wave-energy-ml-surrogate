21   def _is_jax_zero_gradient_array(x):
22       """Return True if `x` is a zero-gradient array. 
23    
24       These arrays are a design quirk of Jax that may one day be removed. 
25       See https://github.com/google/jax/issues/20620. 
26       """
27       if 'numpy' not in sys.modules or 'jax' not in sys.modules:
28           return False
29   
30       import numpy as np
31       import jax
32   
33       return isinstance(x, np.ndarray) and x.dtype == jax.float0

38    
39       This function does not import NumPy if it has not already been imported 
40       and is therefore cheap to use. 
41    
42       This also returns True for `ndarray` subclasses and NumPy scalar objects. 
43    
44       See Also 
45       -------- 
46    
47       array_namespace

417  def _check_api_version(api_version):
418      if api_version in ['2021.12', '2022.12']:
419          warnings.warn(f"The {api_version} version of the array API specification was requested but the returned namespace is actually version 2023.12")
420      elif api_version is not None and api_version not in ['2021.12', '2022.12',
421                                                           '2023.12']:
422          raise ValueError("Only the 2023.12 version of the array API specification is currently supported")

424  def array_namespace(*xs, api_version=None, use_compat=None):
425      """ 
426      Get the array API compatible namespace for the arrays `xs`. 
427   
428      Parameters 
429      ---------- 
430      xs: arrays 
431          one or more arrays. 
432   
433      api_version: str 
434          The newest version of the spec that you need support for (currently 
435          the compat library wrapped APIs support v2023.12). 
436   
437      use_compat: bool or None 
438          If None (the default), the native namespace will be returned if it is 
439          already array API compatible, otherwise a compat wrapper is used. If 
440          True, the compat library wrapped library will be returned. If False, 
441          the native library namespace is returned. 
442   
443      Returns 
444      ------- 
445   
446      out: namespace 
447          The array API compatible namespace corresponding to the arrays in `xs`. 
448   
449      Raises 
450      ------ 
451      TypeError 
452          If `xs` contains arrays from different array libraries or contains a 
453          non-array. 
454   
455   
456      Typical usage is to pass the arguments of a function to 
457      `array_namespace()` at the top of a function to get the corresponding 
458      array API namespace: 
459   
460      .. code:: python 
461   
462         def your_function(x, y): 
463             xp = array_api_compat.array_namespace(x, y) 
464             # Now use xp as the array library namespace 
465             return xp.mean(x, axis=0) + 2*xp.std(y, axis=0) 
466   
467   
468      Wrapped array namespaces can also be imported directly. For example, 
469      `array_namespace(np.array(...))` will return `array_api_compat.numpy`. 
470      This function will also work for any array library not wrapped by 
471      array-api-compat if it explicitly defines `__array_namespace__ 
472      <https://data-apis.org/array-api/latest/API_specification/generated/array_api.array.__array_namespace__.html>`__ 
473      (the wrapped namespace is always preferred if it exists). 
474   
475      See Also 
476      -------- 
477   
478      is_array_api_obj 
479      is_numpy_array 
480      is_cupy_array 
481      is_torch_array 
482      is_dask_array 
483      is_jax_array 
484      is_pydata_sparse_array 
485   
486      """
487      if use_compat not in [None, True, False]:
488          raise ValueError("use_compat must be None, True, or False")
489  
490      _use_compat = use_compat in [None, True]
491  
492      namespaces = set()
493      for x in xs:
494          if is_numpy_array(x):
495              from .. import numpy as numpy_namespace
496              import numpy as np
497              if use_compat is True:
498                  _check_api_version(api_version)
499                  namespaces.add(numpy_namespace)
500              elif use_compat is False:
501                  namespaces.add(np)
502              else:
503                  # numpy 2.0+ have __array_namespace__, however, they are not yet fully array API
504                  # compatible.
505                  namespaces.add(numpy_namespace)
506          elif is_cupy_array(x):
507              if _use_compat:
508                  _check_api_version(api_version)
509                  from .. import cupy as cupy_namespace
510                  namespaces.add(cupy_namespace)
511              else:
512                  import cupy as cp
513                  namespaces.add(cp)
514          elif is_torch_array(x):
515              if _use_compat:
516                  _check_api_version(api_version)
517                  from .. import torch as torch_namespace
518                  namespaces.add(torch_namespace)
519              else:
520                  import torch
521                  namespaces.add(torch)
522          elif is_dask_array(x):
523              if _use_compat:
524                  _check_api_version(api_version)
525                  from ..dask import array as dask_namespace
526                  namespaces.add(dask_namespace)
527              else:
528                  import dask.array as da
529                  namespaces.add(da)
530          elif is_jax_array(x):
531              if use_compat is True:
532                  _check_api_version(api_version)
533                  raise ValueError("JAX does not have an array-api-compat wrapper")
534              elif use_compat is False:
535                  import jax.numpy as jnp
536              else:
537                  # JAX v0.4.32 and newer implements the array API directly in jax.numpy.
538                  # For older JAX versions, it is available via jax.experimental.array_api.
539                  import jax.numpy
540                  if hasattr(jax.numpy, "__array_api_version__"):
541                      jnp = jax.numpy
542                  else:
543                      import jax.experimental.array_api as jnp
544              namespaces.add(jnp)
545          elif is_pydata_sparse_array(x):
546              if use_compat is True:
547                  _check_api_version(api_version)
548                  raise ValueError("`sparse` does not have an array-api-compat wrapper")
549              else:
550                  import sparse
551              # `sparse` is already an array namespace. We do not have a wrapper
552              # submodule for it.
553              namespaces.add(sparse)
554          elif hasattr(x, '__array_namespace__'):
555              if use_compat is True:
556                  raise ValueError("The given array does not have an array-api-compat wrapper")
557              namespaces.add(x.__array_namespace__(api_version=api_version))
558          else:
559              # TODO: Support Python scalars?
560              raise TypeError(f"{type(x).__name__} is not a supported array type")
561  
562      if not namespaces:
563          raise TypeError("Unrecognized array input")
564  
565      if len(namespaces) != 1:
566          raise TypeError(f"Multiple namespaces for array inputs: {namespaces}")
567  
568      xp, = namespaces
569  
570      return xp

589  # device() is not on numpy.ndarray or dask.array and to_device() is not on numpy.ndarray
590  # or cupy.ndarray. They are not included in array objects of this library
591  # because this library just reuses the respective ndarray classes without
592  # wrapping or subclassing them. These helper functions can be used instead of
593  # the wrapper functions for libraries that need to support both NumPy/CuPy and
594  # other libraries that use devices.
595  def device(x: Array, /) -> Device:
596      """ 
597      Hardware device the array data resides on. 
598   
599      This is equivalent to `x.device` according to the `standard 
600      <https://data-apis.org/array-api/latest/API_specification/generated/array_api.array.device.html>`__. 
601      This helper is included because some array libraries either do not have 
602      the `device` attribute or include it with an incompatible API. 
603   
604      Parameters 
605      ---------- 
606      x: array 
607          array instance from an array API compatible library. 
608   
609      Returns 
610      ------- 
611      out: device 
612          a ``device`` object (see the `Device Support <https://data-apis.org/array-api/latest/design_topics/device_support.html>`__ 
613          section of the array API specification). 
614   
615      Notes 
616      ----- 
617   
618      For NumPy the device is always `"cpu"`. For Dask, the device is always a 
619      special `DASK_DEVICE` object. 
620   
621      See Also 
622      -------- 
623   
624      to_device : Move array data to a different device. 
625   
626      """
627      if is_numpy_array(x):
628          return "cpu"
629      elif is_dask_array(x):
630          # Peek at the metadata of the jax array to determine type
631          try:
632              import numpy as np
633              if isinstance(x._meta, np.ndarray):
634                  # Must be on CPU since backed by numpy
635                  return "cpu"
636          except ImportError:
637              pass
638          return _DASK_DEVICE
639      elif is_jax_array(x):
640          # JAX has .device() as a method, but it is being deprecated so that it
641          # can become a property, in accordance with the standard. In order for
642          # this function to not break when JAX makes the flip, we check for
643          # both here.
644          if inspect.ismethod(x.device):
645              return x.device()
646          else:
647              return x.device
648      elif is_pydata_sparse_array(x):
649          # `sparse` will gain `.device`, so check for this first.
650          x_device = getattr(x, 'device', None)
651          if x_device is not None:
652              return x_device
653          # Everything but DOK has this attr.
654          try:
655              inner = x.data
656          except AttributeError:
657              return "cpu"
658          # Return the device of the constituent array
659          return device(inner)
660      return x.device

665  # Based on cupy.array_api.Array.to_device
666  def _cupy_to_device(x, device, /, stream=None):
667      import cupy as cp
668      from cupy.cuda import Device as _Device
669      from cupy.cuda import stream as stream_module
670      from cupy_backends.cuda.api import runtime
671  
672      if device == x.device:
673          return x
674      elif device == "cpu":
675          # allowing us to use `to_device(x, "cpu")`
676          # is useful for portable test swapping between
677          # host and device backends
678          return x.get()
679      elif not isinstance(device, _Device):
680          raise ValueError(f"Unsupported device {device!r}")
681      else:
682          # see cupy/cupy#5985 for the reason how we handle device/stream here
683          prev_device = runtime.getDevice()
684          prev_stream: stream_module.Stream = None
685          if stream is not None:
686              prev_stream = stream_module.get_current_stream()
687              # stream can be an int as specified in __dlpack__, or a CuPy stream
688              if isinstance(stream, int):
689                  stream = cp.cuda.ExternalStream(stream)
690              elif isinstance(stream, cp.cuda.Stream):
691                  pass
692              else:
693                  raise ValueError('the input stream is not recognized')
694              stream.use()
695          try:
696              runtime.setDevice(device.id)
697              arr = x.copy()
698          finally:
699              runtime.setDevice(prev_device)
700              if stream is not None:
701                  prev_stream.use()
702          return arr

709  def to_device(x: Array, device: Device, /, *, stream: Optional[Union[int, Any]] = None) -> Array:
710      """ 
711      Copy the array from the device on which it currently resides to the specified ``device``. 
712   
713      This is equivalent to `x.to_device(device, stream=stream)` according to 
714      the `standard 
715      <https://data-apis.org/array-api/latest/API_specification/generated/array_api.array.to_device.html>`__. 
716      This helper is included because some array libraries do not have the 
717      `to_device` method. 
718   
719      Parameters 
720      ---------- 
721   
722      x: array 
723          array instance from an array API compatible library. 
724   
725      device: device 
726          a ``device`` object (see the `Device Support <https://data-apis.org/array-api/latest/design_topics/device_support.html>`__ 
727          section of the array API specification). 
728   
729      stream: Optional[Union[int, Any]] 
730          stream object to use during copy. In addition to the types supported 
731          in ``array.__dlpack__``, implementations may choose to support any 
732          library-specific stream object with the caveat that any code using 
733          such an object would not be portable. 
734   
735      Returns 
736      ------- 
737   
738      out: array 
739          an array with the same data and data type as ``x`` and located on the 
740          specified ``device``. 
741   
742      Notes 
743      ----- 
744   
745      For NumPy, this function effectively does nothing since the only supported 
746      device is the CPU. For CuPy, this method supports CuPy CUDA 
747      :external+cupy:class:`Device <cupy.cuda.Device>` and 
748      :external+cupy:class:`Stream <cupy.cuda.Stream>` objects. For PyTorch, 
749      this is the same as :external+torch:meth:`x.to(device) <torch.Tensor.to>` 
750      (the ``stream`` argument is not supported in PyTorch). 
751   
752      See Also 
753      -------- 
754   
755      device : Hardware device the array data resides on. 
756   
757      """
758      if is_numpy_array(x):
759          if stream is not None:
760              raise ValueError("The stream argument to to_device() is not supported")
761          if device == 'cpu':
762              return x
763          raise ValueError(f"Unsupported device {device!r}")
764      elif is_cupy_array(x):
765          # cupy does not yet have to_device
766          return _cupy_to_device(x, device, stream=stream)
767      elif is_torch_array(x):
768          return _torch_to_device(x, device, stream=stream)
769      elif is_dask_array(x):
770          if stream is not None:
771              raise ValueError("The stream argument to to_device() is not supported")
772          # TODO: What if our array is on the GPU already?
773          if device == 'cpu':
774              return x
775          raise ValueError(f"Unsupported device {device!r}")
776      elif is_jax_array(x):
777          if not hasattr(x, "__array_namespace__"):
778              # In JAX v0.4.31 and older, this import adds to_device method to x.
779              import jax.experimental.array_api # noqa: F401
780          return x.to_device(device, stream=stream)
781      elif is_pydata_sparse_array(x) and device == _device(x):
782          # Perform trivial check to return the same array if
783          # device is same instead of err-ing.
784          return x
785      return x.to_device(device, stream=stream)
786  
787  def size(x):
788      """ 
789      Return the total number of elements of x. 
790   
791      This is equivalent to `x.size` according to the `standard 
792      <https://data-apis.org/array-api/latest/API_specification/generated/array_api.array.size.html>`__. 
793      This helper is included because PyTorch defines `size` in an 
794      :external+torch:meth:`incompatible way <torch.Tensor.size>`. 
795   
796      """
797      if None in x.shape:
798          return None
799      return math.prod(x.shape)
