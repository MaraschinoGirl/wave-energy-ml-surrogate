255  function hookVar(stream) {
256    // variables
257    // @@prefix.varName @varName
258    // varName can be quoted with ` or ' or "
259    // ref: http://dev.mysql.com/doc/refman/5.5/en/user-variables.html
260    if (stream.eat("@")) {
261      stream.match('session.');
262      stream.match('local.');
263      stream.match('global.');
264    }
265  
266    if (stream.eat("'")) {
267      stream.match(/^.*'/);
268      return "string.special";
269    } else if (stream.eat('"')) {
270      stream.match(/^.*"/);
271      return "string.special";
272    } else if (stream.eat("`")) {
273      stream.match(/^.*`/);
274      return "string.special";
275    } else if (stream.match(/^[0-9a-zA-Z$\.\_]+/)) {
276      return "string.special";
277    }
278    return null;
