1    # coding: utf-8
2    """Find the path to xgboost dynamic library files."""
3    
4    import os
5    import platform
6    import sys
7    from typing import List
8    
9    
10   class XGBoostLibraryNotFound(Exception):
11       """Error thrown by when xgboost is not found"""
12   
13   
14   def is_sphinx_build() -> bool:
15       """`XGBOOST_BUILD_DOC` is used by the sphinx conf.py to skip building the C++ code."""
16       return bool(os.environ.get("XGBOOST_BUILD_DOC", False))
17   
18   
19   def find_lib_path() -> List[str]:
20       """Find the path to xgboost dynamic library files. 
21    
22       Returns 
23       ------- 
24       lib_path 
25          List of all found library path to xgboost 
26       """
27       curr_path = os.path.dirname(os.path.abspath(os.path.expanduser(__file__)))
28       dll_path = [
29           # normal, after installation `lib` is copied into Python package tree.
30           os.path.join(curr_path, "lib"),
31           # editable installation, no copying is performed.
32           os.path.join(curr_path, os.path.pardir, os.path.pardir, "lib"),
33           # use libxgboost from a system prefix, if available.  This should be the last
34           # option.
35           os.path.join(sys.base_prefix, "lib"),
36       ]
37   
38       if sys.platform == "win32":
39           # On Windows, Conda may install libs in different paths
40           dll_path.extend(
41               [
42                   os.path.join(sys.base_prefix, "bin"),
43                   os.path.join(sys.base_prefix, "Library"),
44                   os.path.join(sys.base_prefix, "Library", "bin"),
45                   os.path.join(sys.base_prefix, "Library", "lib"),
46                   os.path.join(sys.base_prefix, "Library", "mingw-w64"),
47                   os.path.join(sys.base_prefix, "Library", "mingw-w64", "bin"),
48                   os.path.join(sys.base_prefix, "Library", "mingw-w64", "lib"),
49               ]
50           )
51           dll_path = [os.path.join(p, "xgboost.dll") for p in dll_path]
52       elif sys.platform.startswith(("linux", "freebsd", "emscripten")):
53           dll_path = [os.path.join(p, "libxgboost.so") for p in dll_path]
54       elif sys.platform == "darwin":
55           dll_path = [os.path.join(p, "libxgboost.dylib") for p in dll_path]
56       elif sys.platform == "cygwin":
57           dll_path = [os.path.join(p, "cygxgboost.dll") for p in dll_path]
58       if platform.system() == "OS400":
59           dll_path = [os.path.join(p, "libxgboost.so") for p in dll_path]
60   
61       lib_path = [p for p in dll_path if os.path.exists(p) and os.path.isfile(p)]
62   
63       if not lib_path and not is_sphinx_build():
64           link = "https://xgboost.readthedocs.io/en/stable/install.html"
65           msg = (
66               "Cannot find XGBoost Library in the candidate path.  "
67               + "List of candidates:\n- "
68               + ("\n- ".join(dll_path))
69               + "\nXGBoost Python package path: "
70               + curr_path
71               + "\nsys.base_prefix: "
72               + sys.base_prefix
73               + "\nSee: "
74               + link
75               + " for installing XGBoost."
76           )
77           raise XGBoostLibraryNotFound(msg)
78       return lib_path
