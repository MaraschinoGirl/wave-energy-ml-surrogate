11   struct svm_node

46   struct svm_parameter
47   {
48   	int svm_type;
49   	int kernel_type;
50   	int degree;	/* for poly */
51   	double gamma;	/* for poly/rbf/sigmoid */
52   	double coef0;	/* for poly/sigmoid */
53   
54   	/* these are for training only */
55   	double cache_size; /* in MB */
56   	double eps;	/* stopping criteria */
57   	double C;	/* for C_SVC, EPSILON_SVR and NU_SVR */
58   	int nr_weight;		/* for C_SVC */
59   	int *weight_label;	/* for C_SVC */
60   	double* weight;		/* for C_SVC */
61   	double nu;	/* for NU_SVC, ONE_CLASS, and NU_SVR */
62   	double p;	/* for EPSILON_SVR */
63   	int shrinking;	/* use the shrinking heuristics */
64   	int probability; /* do probability estimates */
65   	int max_iter; /* ceiling on Solver runtime */
66       int random_seed; /* seed for random number generator */
67   };

72   struct svm_model
73   {
74   	struct svm_parameter param;	/* parameter */
75   	int nr_class;		/* number of classes, = 2 in regression/one class svm */
76   	int l;			/* total #SV */
77   	struct svm_node *SV;		/* SVs (SV[l]) */
78   	double **sv_coef;	/* coefficients for SVs in decision functions (sv_coef[k-1][l]) */
79   	int *n_iter;		/* number of iterations run by the optimization routine to fit the model */
80   
81   	int *sv_ind;            /* index of support vectors */
82   
83   	double *rho;		/* constants in decision functions (rho[k*(k-1)/2]) */
84   	double *probA;		/* pairwise probability information */
85   	double *probB;
86   
87   	/* for classification only */
88   
89   	int *label;		/* label of each class (label[k]) */
90   	int *nSV;		/* number of SVs for each class (nSV[k]) */
91   				/* nSV[0] + nSV[1] + ... + nSV[k-1] = l */
92   	/* XXX */
93   	int free_sv;		/* 1 if svm_model is created by svm_load_model*/
94   				/* 0 if svm_model is created by svm_train */
95   };
