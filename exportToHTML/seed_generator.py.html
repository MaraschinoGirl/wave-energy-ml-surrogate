12   @keras_export("keras.random.SeedGenerator")
13   class SeedGenerator:
14       """Generates variable seeds upon each call to a function generating 
15       random numbers. 
16    
17       In Keras, all random number generators (such as 
18       `keras.random.normal()`) are stateless, meaning that if you pass an 
19       integer seed to them (such as `seed=42`), they will return the same 
20       values for repeated calls. To get different values for each 
21       call, a `SeedGenerator` providing the state of the random generator 
22       has to be used. 
23    
24       Note that all the random number generators have a default seed of None, 
25       which implies that an internal global SeedGenerator is used. 
26       If you need to decouple the RNG from the global state you can provide 
27       a local `StateGenerator` with either a deterministic or random initial 
28       state. 
29    
30       Remark concerning the JAX backen: Note that the use of a local 
31       `StateGenerator` as seed argument is required for JIT compilation of 
32       RNG with the JAX backend, because the use of global state is not 
33       supported. 
34    
35       Example: 
36    
37       ```python 
38       seed_gen = keras.random.SeedGenerator(seed=42) 
39       values = keras.random.normal(shape=(2, 3), seed=seed_gen) 
40       new_values = keras.random.normal(shape=(2, 3), seed=seed_gen) 
41       ``` 
42    
43       Usage in a layer: 
44    
45       ```python 
46       class Dropout(keras.Layer): 
47           def __init__(self, **kwargs): 
48               super().__init__(**kwargs) 
49               self.seed_generator = keras.random.SeedGenerator(1337) 
50    
51           def call(self, x, training=False): 
52               if training: 
53                   return keras.random.dropout( 
54                       x, rate=0.5, seed=self.seed_generator 
55                   ) 
56               return x 
57       ``` 
58       """
59   
60       def __init__(self, seed=None, name=None, **kwargs):
61           if name is None:
62               name = auto_name(self.__class__.__name__)
63           self.name = name
64   
65           custom_backend = kwargs.pop("backend", None)
66           if kwargs:
67               raise ValueError(f"Unrecognized keyword arguments: {kwargs}")
68           if custom_backend is not None:
69               self.backend = custom_backend
70           else:
71               self.backend = backend
72   
73           self._initial_seed = seed
74           if seed is None:
75               seed = make_default_seed()
76   
77           if not isinstance(seed, int):
78               raise ValueError(
79                   f"Argument `seed` must be an integer. Received: seed={seed}"
80               )
81   
82           def seed_initializer(*args, **kwargs):
83               dtype = kwargs.get("dtype", None)
84               return self.backend.convert_to_tensor([seed, 0], dtype=dtype)
85   
86           with self.backend.name_scope(self.name, caller=self):
87               self.state = self.backend.Variable(
88                   seed_initializer,
89                   shape=(2,),
90                   dtype=self.backend.random_seed_dtype(),
91                   trainable=False,
92                   aggregation="none",
93                   name="seed_generator_state",
94               )
95   
96       def next(self, ordered=True):
97           seed_state = self.state
98           # Use * 1 to create a copy
99           new_seed_value = seed_state.value * 1
100          if ordered:
101              increment = self.backend.convert_to_tensor(
102                  np.array([0, 1]), dtype=seed_state.dtype
103              )
104              self.state.assign(self.backend.numpy.add(seed_state, increment))
105          else:
106              # This produces a sequence of near-unique numbers
107              # between 0 and 1M
108              self.state.assign((seed_state + 1) * 5387 % 933199)
109          return new_seed_value
110  
111      def get_config(self):
112          return {"seed": self._initial_seed}
113  
114      @classmethod
115      def from_config(cls, config):
116          return cls(**config)

119  def global_seed_generator():
120      if jax_utils.is_in_jax_tracing_scope():
121          raise ValueError(
122              "[JAX RNG] When tracing a JAX function, "
123              "you should only use seeded random ops, e.g. "
124              "you should create a `SeedGenerator` instance, attach it "
125              "to your layer/model, and pass the instance as the `seed` "
126              "argument when calling random ops. Unseeded random ops "
127              "would get incorrectly traced by JAX and would become constant "
128              "after tracing. Example:\n\n"
129              "```\n"
130              "# Make sure to set the seed generator as a layer attribute\n"
131              "self.seed_generator = keras.random.SeedGenerator(seed=1337)\n"
132              "...\n"
133              "out = keras.random.normal(shape=(1,), seed=self.seed_generator)\n"
134              "```"
135          )
136      gen = global_state.get_global_attribute("global_seed_generator")
137      if gen is None:
138          gen = SeedGenerator()
139          global_state.set_global_attribute("global_seed_generator", gen)
140      return gen

143  def make_default_seed():
144      return python_random.randint(1, int(1e9))

147  def draw_seed(seed):
148      from keras.src.backend import convert_to_tensor
149      from keras.src.backend import random_seed_dtype
150  
151      if isinstance(seed, SeedGenerator):
152          return seed.next()
153      elif isinstance(seed, int):
154          return convert_to_tensor([seed, 0], dtype=random_seed_dtype())
155      elif seed is None:
156          return global_seed_generator().next(ordered=False)
157      raise ValueError(
158          "Argument `seed` must be either an integer "
159          "or an instance of `SeedGenerator`. "
160          f"Received: seed={seed} (of type {type(seed)})"
161      )
