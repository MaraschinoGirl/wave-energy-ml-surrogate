1    class export:
2        """Decorator to export a public API in a given package. 
3     
4        Example usage: 
5     
6        ```python 
7        @export(package="keras_tuner", path="keras_tuner.applications.HyperResNet") 
8        class HyperResNet: 
9            ... 
10       ``` 
11    
12       You can also pass a list of paths as `path`, to make 
13       the same symbol visible under various aliases: 
14    
15       ```python 
16       @export( 
17           package="keras_tuner", 
18           path=[ 
19               "keras_tuner.applications.HyperResNet", 
20               "keras_tuner.applications.resnet.HyperResNet", 
21           ]) 
22       class HyperResNet: 
23           ... 
24       ``` 
25    
26       **Note:** All export packages must start with the package name. 
27       Yes, that is redundant, but that is a helpful sanity check. 
28       The expectation is that each package will customize 
29       `export_api` to provide a default value for `package`, 
30       which will serve to validate all `path` values 
31       and avoid users inadvertendly ending up with non-exported 
32       symbols due to a bad path (e.g. `path="applications.HyperResNet"` 
33       instead of `path="keras_tuner.applications.HyperResNet"`). 
34       """
35   
36       def __init__(self, package, path):
37           if isinstance(path, str):
38               export_paths = [path]
39           elif isinstance(path, list):
40               export_paths = path
41           else:
42               raise ValueError(
43                   f"Invalid type for `path` argument: "
44                   f"Received '{path}' "
45                   f"of type {type(path)}"
46               )
47   
48           for p in export_paths:
49               if not p.startswith(package + "."):
50                   raise ValueError(
51                       f"All `export_path` values should start with '{package}.'. "
52                       f"Received: path={path}"
53                   )
54   
55           self.package = package
56           self.path = path
57   
58       def __call__(self, symbol):
59           if hasattr(symbol, "_api_export_path") and symbol._api_export_symbol_id == id(
60               symbol
61           ):
62               raise ValueError(
63                   f"Symbol {symbol} is already exported as '{symbol._api_export_path}'. "
64                   f"Cannot also export it to '{self.path}'."
65               )
66           symbol._api_export_path = self.path
67           symbol._api_export_symbol_id = id(symbol)
68           return symbol
