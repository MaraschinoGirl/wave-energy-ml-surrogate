1166     if (last >= 0 && ins <= 0 && ins == sections[last + 1])
1167         sections[last] += len;
1168     else if (last >= 0 && len == 0 && sections[last] == 0)
1169         sections[last + 1] += ins;
1170     else if (forceJoin) {
1171         sections[last] += len;
1172         sections[last + 1] += ins;
1173     }
1174     else
1175         sections.push(len, ins);

1190 function iterChanges(desc, f, individual) {
1191     let inserted = desc.inserted;
1192     for (let posA = 0, posB = 0, i = 0; i < desc.sections.length;) {

1216 function mapSet(setA, setB, before, mkSet = false) {
1217     // Produce a copy of setA that applies to the document after setB
1218     // has been applied (assuming both start at the same document).
1219     let sections = [], insert = mkSet ? [] : null;
1220     let a = new SectionIter(setA), b = new SectionIter(setB);
1221     // Iterate over both sets in parallel. inserted tracks, for changes
1222     // in A that have to be processed piece-by-piece, whether their
1223     // content has been inserted already, and refers to the section
1224     // index.
1225     for (let inserted = -1;;) {
1226         if (a.done && b.len || b.done && a.len) {
1227             throw new Error("Mismatched change set lengths");
1228         }
1229         else if (a.ins == -1 && b.ins == -1) {
1230             // Move across ranges skipped by both sets.
1231             let len = Math.min(a.len, b.len);
1232             addSection(sections, len, -1);
1233             a.forward(len);
1234             b.forward(len);
1235         }
1236         else if (b.ins >= 0 && (a.ins < 0 || inserted == a.i || a.off == 0 && (b.len < a.len || b.len == a.len && !before))) {
1237             // If there's a change in B that comes before the next change in
1238             // A (ordered by start pos, then len, then before flag), skip
1239             // that (and process any changes in A it covers).
1240             let len = b.len;
1241             addSection(sections, b.ins, -1);
1242             while (len) {
1243                 let piece = Math.min(a.len, len);
1244                 if (a.ins >= 0 && inserted < a.i && a.len <= piece) {
1245                     addSection(sections, 0, a.ins);
1246                     if (insert)
1247                         addInsert(insert, sections, a.text);
1248                     inserted = a.i;
1249                 }
1250                 a.forward(piece);
1251                 len -= piece;
1252             }
1253             b.next();
1254         }
1255         else if (a.ins >= 0) {
1256             // Process the part of a change in A up to the start of the next
1257             // non-deletion change in B (if overlapping).
1258             let len = 0, left = a.len;
1259             while (left) {
1260                 if (b.ins == -1) {
1261                     let piece = Math.min(left, b.len);
1262                     len += piece;
1263                     left -= piece;
1264                     b.forward(piece);
1265                 }
1266                 else if (b.ins == 0 && b.len < left) {
1267                     left -= b.len;
1268                     b.next();
1269                 }
1270                 else {
1271                     break;
1272                 }
1273             }
1274             addSection(sections, len, inserted < a.i ? a.ins : 0);
1275             if (insert && inserted < a.i)
1276                 addInsert(insert, sections, a.text);
1277             inserted = a.i;
1278             a.forward(a.len - left);
1279         }
1280         else if (a.done && b.done) {
1281             return insert ? ChangeSet.createSet(sections, insert) : ChangeDesc.create(sections);
1282         }
1283         else {
1284             throw new Error("Mismatched change set lengths");
1285         }
1286     }

1288 function composeSets(setA, setB, mkSet = false) {
1289     let sections = [];
1290     let insert = mkSet ? [] : null;
1291     let a = new SectionIter(setA), b = new SectionIter(setB);
1292     for (let open = false;;) {
1293         if (a.done && b.done) {
1294             return insert ? ChangeSet.createSet(sections, insert) : ChangeDesc.create(sections);
1295         }
1296         else if (a.ins == 0) { // Deletion in A
1297             addSection(sections, a.len, 0, open);
1298             a.next();
1299         }
1300         else if (b.len == 0 && !b.done) { // Insertion in B
1301             addSection(sections, 0, b.ins, open);
1302             if (insert)
1303                 addInsert(insert, sections, b.text);
1304             b.next();
1305         }
1306         else if (a.done || b.done) {
1307             throw new Error("Mismatched change set lengths");
1308         }
1309         else {
1310             let len = Math.min(a.len2, b.len), sectionLen = sections.length;
1311             if (a.ins == -1) {
1312                 let insB = b.ins == -1 ? -1 : b.off ? 0 : b.ins;
1313                 addSection(sections, len, insB, open);
1314                 if (insert && insB)
1315                     addInsert(insert, sections, b.text);
1316             }
1317             else if (b.ins == -1) {
1318                 addSection(sections, a.off ? 0 : a.len, len, open);
1319                 if (insert)
1320                     addInsert(insert, sections, a.textBit(len));
1321             }
1322             else {
1323                 addSection(sections, a.off ? 0 : a.len, b.off ? 0 : b.ins, open);
1324                 if (insert && !b.off)
1325                     addInsert(insert, sections, b.text);
1326             }
1327             open = (a.ins > len || b.ins >= 0 && b.len > len) && (open || sections.length > sectionLen);
1328             a.forward2(len);
1329             b.forward(len);
1330         }
1331     }

1333 class SectionIter {
1334     constructor(set) {
1335         this.set = set;
1336         this.i = 0;
1337         this.next();
1338     }
1339     next() {
1340         let { sections } = this.set;
1341         if (this.i < sections.length) {
1342             this.len = sections[this.i++];
1343             this.ins = sections[this.i++];
1344         }
1345         else {
1346             this.len = 0;
1347             this.ins = -2;
1348         }
1349         this.off = 0;
1350     }
1351     get done() { return this.ins == -2; }
1352     get len2() { return this.ins < 0 ? this.len : this.ins; }
1353     get text() {
1354         let { inserted } = this.set, index = (this.i - 2) >> 1;
1355         return index >= inserted.length ? Text.empty : inserted[index];
1356     }
1357     textBit(len) {
1358         let { inserted } = this.set, index = (this.i - 2) >> 1;
1359         return index >= inserted.length && !len ? Text.empty
1360             : inserted[index].slice(this.off, len == null ? undefined : this.off + len);
1361     }
1362     forward(len) {
1363         if (len == this.len)
1364             this.next();
1365         else {
1366             this.len -= len;
1367             this.off += len;
1368         }
1369     }
1370     forward2(len) {
1371         if (this.ins == -1)
1372             this.forward(len);
1373         else if (len == this.ins)
1374             this.next();
1375         else {
1376             this.ins -= len;
1377             this.off += len;
1378         }
1379     }

1425     any.

1435     lines of different length.
1436     */

1463         return EditorSelection.range(this.anchor, head);
1464     }

1467     */

1469         return this.anchor == other.anchor && this.head == other.head &&

1471     }
1472     /**
1473     Return a JSON-serializable object representing the range.
1474     */

1479     */
1480     static fromJSON(json) {
1481         if (!json || typeof json.anchor != "number" || typeof json.head != "number")

1604             (bidiLevel == null ? 7 : Math.min(6, bidiLevel)) |
