27   def _array_indexing(array, key, key_dtype, axis):
28       """Index an array or scipy.sparse consistently across NumPy version."""
29       xp, is_array_api = get_namespace(array)
30       if is_array_api:
31           return xp.take(array, key, axis=axis)
32       if issparse(array) and key_dtype == "bool":
33           key = np.asarray(key)
34       if isinstance(key, tuple):
35           key = list(key)
36       return array[key, ...] if axis == 0 else array[:, key]

39   def _pandas_indexing(X, key, key_dtype, axis):
40       """Index a pandas dataframe or a series."""
41       if _is_arraylike_not_scalar(key):
42           key = np.asarray(key)
43   
44       if key_dtype == "int" and not (isinstance(key, slice) or np.isscalar(key)):
45           # using take() instead of iloc[] ensures the return value is a "proper"
46           # copy that will not raise SettingWithCopyWarning
47           return X.take(key, axis=axis)
48       else:
49           # check whether we should index with loc or iloc
50           indexer = X.iloc if key_dtype == "int" else X.loc
51           return indexer[:, key] if axis else indexer[key]

54   def _list_indexing(X, key, key_dtype):
55       """Index a Python list."""
56       if np.isscalar(key) or isinstance(key, slice):
57           # key is a slice or a scalar
58           return X[key]
59       if key_dtype == "bool":
60           # key is a boolean array-like
61           return list(compress(X, key))
62       # key is a integer array-like of key
63       return [X[idx] for idx in key]

66   def _polars_indexing(X, key, key_dtype, axis):
67       """Indexing X with polars interchange protocol."""
68       # Polars behavior is more consistent with lists
69       if isinstance(key, np.ndarray):
70           # Convert each element of the array to a Python scalar
71           key = key.tolist()
72       elif not (np.isscalar(key) or isinstance(key, slice)):
73           key = list(key)
74   
75       if axis == 1:
76           # Here we are certain to have a polars DataFrame; which can be indexed with
77           # integer and string scalar, and list of integer, string and boolean
78           return X[:, key]
79   
80       if key_dtype == "bool":
81           # Boolean mask can be indexed in the same way for Series and DataFrame (axis=0)
82           return X.filter(key)
83   
84       # Integer scalar and list of integer can be indexed in the same way for Series and
85       # DataFrame (axis=0)
86       X_indexed = X[key]
87       if np.isscalar(key) and len(X.shape) == 2:
88           # `X_indexed` is a DataFrame with a single row; we return a Series to be
89           # consistent with pandas
90           pl = sys.modules["polars"]
91           return pl.Series(X_indexed.row(0))
92       return X_indexed

95   def _determine_key_type(key, accept_slice=True):
96       """Determine the data type of key. 
97    
98       Parameters 
99       ---------- 
100      key : scalar, slice or array-like 
101          The key from which we want to infer the data type. 
102   
103      accept_slice : bool, default=True 
104          Whether or not to raise an error if the key is a slice. 
105   
106      Returns 
107      ------- 
108      dtype : {'int', 'str', 'bool', None} 
109          Returns the data type of key. 
110      """
111      err_msg = (
112          "No valid specification of the columns. Only a scalar, list or "
113          "slice of all integers or all strings, or boolean mask is "
114          "allowed"
115      )
116  
117      dtype_to_str = {int: "int", str: "str", bool: "bool", np.bool_: "bool"}
118      array_dtype_to_str = {
119          "i": "int",
120          "u": "int",
121          "b": "bool",
122          "O": "str",
123          "U": "str",
124          "S": "str",
125      }
126  
127      if key is None:
128          return None
129      if isinstance(key, tuple(dtype_to_str.keys())):
130          try:
131              return dtype_to_str[type(key)]
132          except KeyError:
133              raise ValueError(err_msg)
134      if isinstance(key, slice):
135          if not accept_slice:
136              raise TypeError(
137                  "Only array-like or scalar are supported. A Python slice was given."
138              )
139          if key.start is None and key.stop is None:
140              return None
141          key_start_type = _determine_key_type(key.start)
142          key_stop_type = _determine_key_type(key.stop)
143          if key_start_type is not None and key_stop_type is not None:
144              if key_start_type != key_stop_type:
145                  raise ValueError(err_msg)
146          if key_start_type is not None:
147              return key_start_type
148          return key_stop_type
149      # TODO(1.9) remove UserList when the force_int_remainder_cols param
150      # of ColumnTransformer is removed
151      if isinstance(key, (list, tuple, UserList)):
152          unique_key = set(key)
153          key_type = {_determine_key_type(elt) for elt in unique_key}
154          if not key_type:
155              return None
156          if len(key_type) != 1:
157              raise ValueError(err_msg)
158          return key_type.pop()
159      if hasattr(key, "dtype"):
160          xp, is_array_api = get_namespace(key)
161          # NumPy arrays are special-cased in their own branch because the Array API
162          # cannot handle object/string-based dtypes that are often used to index
163          # columns of dataframes by names.
164          if is_array_api and not _is_numpy_namespace(xp):
165              if xp.isdtype(key.dtype, "bool"):
166                  return "bool"
167              elif xp.isdtype(key.dtype, "integral"):
168                  return "int"
169              else:
170                  raise ValueError(err_msg)
171          else:
172              try:
173                  return array_dtype_to_str[key.dtype.kind]
174              except KeyError:
175                  raise ValueError(err_msg)
176      raise ValueError(err_msg)

179  def _safe_indexing(X, indices, *, axis=0):
180      """Return rows, items or columns of X using indices. 
181   
182      .. warning:: 
183   
184          This utility is documented, but **private**. This means that 
185          backward compatibility might be broken without any deprecation 
186          cycle. 
187   
188      Parameters 
189      ---------- 
190      X : array-like, sparse-matrix, list, pandas.DataFrame, pandas.Series 
191          Data from which to sample rows, items or columns. `list` are only 
192          supported when `axis=0`. 
193      indices : bool, int, str, slice, array-like 
194          - If `axis=0`, boolean and integer array-like, integer slice, 
195            and scalar integer are supported. 
196          - If `axis=1`: 
197              - to select a single column, `indices` can be of `int` type for 
198                all `X` types and `str` only for dataframe. The selected subset 
199                will be 1D, unless `X` is a sparse matrix in which case it will 
200                be 2D. 
201              - to select multiples columns, `indices` can be one of the 
202                following: `list`, `array`, `slice`. The type used in 
203                these containers can be one of the following: `int`, 'bool' and 
204                `str`. However, `str` is only supported when `X` is a dataframe. 
205                The selected subset will be 2D. 
206      axis : int, default=0 
207          The axis along which `X` will be subsampled. `axis=0` will select 
208          rows while `axis=1` will select columns. 
209   
210      Returns 
211      ------- 
212      subset 
213          Subset of X on axis 0 or 1. 
214   
215      Notes 
216      ----- 
217      CSR, CSC, and LIL sparse matrices are supported. COO sparse matrices are 
218      not supported. 
219   
220      Examples 
221      -------- 
222      >>> import numpy as np 
223      >>> from sklearn.utils import _safe_indexing 
224      >>> data = np.array([[1, 2], [3, 4], [5, 6]]) 
225      >>> _safe_indexing(data, 0, axis=0)  # select the first row 
226      array([1, 2]) 
227      >>> _safe_indexing(data, 0, axis=1)  # select the first column 
228      array([1, 3, 5]) 
229      """
230      if indices is None:
231          return X
232  
233      if axis not in (0, 1):
234          raise ValueError(
235              "'axis' should be either 0 (to index rows) or 1 (to index "
236              " column). Got {} instead.".format(axis)
237          )
238  
239      indices_dtype = _determine_key_type(indices)
240  
241      if axis == 0 and indices_dtype == "str":
242          raise ValueError("String indexing is not supported with 'axis=0'")
243  
244      if axis == 1 and isinstance(X, list):
245          raise ValueError("axis=1 is not supported for lists")
246  
247      if axis == 1 and hasattr(X, "shape") and len(X.shape) != 2:
248          raise ValueError(
249              "'X' should be a 2D NumPy array, 2D sparse matrix or "
250              "dataframe when indexing the columns (i.e. 'axis=1'). "
251              "Got {} instead with {} dimension(s).".format(type(X), len(X.shape))
252          )
253  
254      if (
255          axis == 1
256          and indices_dtype == "str"
257          and not (_is_pandas_df(X) or _use_interchange_protocol(X))
258      ):
259          raise ValueError(
260              "Specifying the columns using strings is only supported for dataframes."
261          )
262  
263      if hasattr(X, "iloc"):
264          # TODO: we should probably use _is_pandas_df_or_series(X) instead but this
265          # would require updating some tests such as test_train_test_split_mock_pandas.
266          return _pandas_indexing(X, indices, indices_dtype, axis=axis)
267      elif _is_polars_df_or_series(X):
268          return _polars_indexing(X, indices, indices_dtype, axis=axis)
269      elif hasattr(X, "shape"):
270          return _array_indexing(X, indices, indices_dtype, axis=axis)
271      else:
272          return _list_indexing(X, indices, indices_dtype)

275  def _safe_assign(X, values, *, row_indexer=None, column_indexer=None):
276      """Safe assignment to a numpy array, sparse matrix, or pandas dataframe. 
277   
278      Parameters 
279      ---------- 
280      X : {ndarray, sparse-matrix, dataframe} 
281          Array to be modified. It is expected to be 2-dimensional. 
282   
283      values : ndarray 
284          The values to be assigned to `X`. 
285   
286      row_indexer : array-like, dtype={int, bool}, default=None 
287          A 1-dimensional array to select the rows of interest. If `None`, all 
288          rows are selected. 
289   
290      column_indexer : array-like, dtype={int, bool}, default=None 
291          A 1-dimensional array to select the columns of interest. If `None`, all 
292          columns are selected. 
293      """
294      row_indexer = slice(None, None, None) if row_indexer is None else row_indexer
295      column_indexer = (
296          slice(None, None, None) if column_indexer is None else column_indexer
297      )
298  
299      if hasattr(X, "iloc"):  # pandas dataframe
300          with warnings.catch_warnings():
301              # pandas >= 1.5 raises a warning when using iloc to set values in a column
302              # that does not have the same type as the column being set. It happens
303              # for instance when setting a categorical column with a string.
304              # In the future the behavior won't change and the warning should disappear.
305              # TODO(1.3): check if the warning is still raised or remove the filter.
306              warnings.simplefilter("ignore", FutureWarning)
307              X.iloc[row_indexer, column_indexer] = values
308      else:  # numpy array or sparse matrix
309          X[row_indexer, column_indexer] = values

312  def _get_column_indices_for_bool_or_int(key, n_columns):
313      # Convert key into list of positive integer indexes
314      try:
315          idx = _safe_indexing(np.arange(n_columns), key)
316      except IndexError as e:
317          raise ValueError(
318              f"all features must be in [0, {n_columns - 1}] or [-{n_columns}, 0]"
319          ) from e
320      return np.atleast_1d(idx).tolist()

323  def _get_column_indices(X, key):
324      """Get feature column indices for input data X and key. 
325   
326      For accepted values of `key`, see the docstring of 
327      :func:`_safe_indexing`. 
328      """
329      key_dtype = _determine_key_type(key)
330      if _use_interchange_protocol(X):
331          return _get_column_indices_interchange(X.__dataframe__(), key, key_dtype)
332  
333      n_columns = X.shape[1]
334      if isinstance(key, (list, tuple)) and not key:
335          # we get an empty list
336          return []
337      elif key_dtype in ("bool", "int"):
338          return _get_column_indices_for_bool_or_int(key, n_columns)
339      else:
340          try:
341              all_columns = X.columns
342          except AttributeError:
343              raise ValueError(
344                  "Specifying the columns using strings is only supported for dataframes."
345              )
346          if isinstance(key, str):
347              columns = [key]
348          elif isinstance(key, slice):
349              start, stop = key.start, key.stop
350              if start is not None:
351                  start = all_columns.get_loc(start)
352              if stop is not None:
353                  # pandas indexing with strings is endpoint included
354                  stop = all_columns.get_loc(stop) + 1
355              else:
356                  stop = n_columns + 1
357              return list(islice(range(n_columns), start, stop))
358          else:
359              columns = list(key)
360  
361          try:
362              column_indices = []
363              for col in columns:
364                  col_idx = all_columns.get_loc(col)
365                  if not isinstance(col_idx, numbers.Integral):
366                      raise ValueError(
367                          f"Selected columns, {columns}, are not unique in dataframe"
368                      )
369                  column_indices.append(col_idx)
370  
371          except KeyError as e:
372              raise ValueError("A given column is not a column of the dataframe") from e
373  
374          return column_indices

377  def _get_column_indices_interchange(X_interchange, key, key_dtype):
378      """Same as _get_column_indices but for X with __dataframe__ protocol."""
379  
380      n_columns = X_interchange.num_columns()
381  
382      if isinstance(key, (list, tuple)) and not key:
383          # we get an empty list
384          return []
385      elif key_dtype in ("bool", "int"):
386          return _get_column_indices_for_bool_or_int(key, n_columns)
387      else:
388          column_names = list(X_interchange.column_names())
389  
390          if isinstance(key, slice):
391              if key.step not in [1, None]:
392                  raise NotImplementedError("key.step must be 1 or None")
393              start, stop = key.start, key.stop
394              if start is not None:
395                  start = column_names.index(start)
396  
397              if stop is not None:
398                  stop = column_names.index(stop) + 1
399              else:
400                  stop = n_columns + 1
401              return list(islice(range(n_columns), start, stop))
402  
403          selected_columns = [key] if np.isscalar(key) else key
404  
405          try:
406              return [column_names.index(col) for col in selected_columns]
407          except ValueError as e:
408              raise ValueError("A given column is not a column of the dataframe") from e

411  @validate_params(
412      {
413          "replace": ["boolean"],
414          "n_samples": [Interval(numbers.Integral, 1, None, closed="left"), None],
415          "random_state": ["random_state"],
416          "stratify": ["array-like", "sparse matrix", None],
417      },
418      prefer_skip_nested_validation=True,
419  )
420  def resample(*arrays, replace=True, n_samples=None, random_state=None, stratify=None):
421      """Resample arrays or sparse matrices in a consistent way. 
422   
423      The default strategy implements one step of the bootstrapping 
424      procedure. 
425   
426      Parameters 
427      ---------- 
428      *arrays : sequence of array-like of shape (n_samples,) or \ 
429              (n_samples, n_outputs) 
430          Indexable data-structures can be arrays, lists, dataframes or scipy 
431          sparse matrices with consistent first dimension. 
432   
433      replace : bool, default=True 
434          Implements resampling with replacement. If False, this will implement 
435          (sliced) random permutations. 
436   
437      n_samples : int, default=None 
438          Number of samples to generate. If left to None this is 
439          automatically set to the first dimension of the arrays. 
440          If replace is False it should not be larger than the length of 
441          arrays. 
442   
443      random_state : int, RandomState instance or None, default=None 
444          Determines random number generation for shuffling 
445          the data. 
446          Pass an int for reproducible results across multiple function calls. 
447          See :term:`Glossary <random_state>`. 
448   
449      stratify : {array-like, sparse matrix} of shape (n_samples,) or \ 
450              (n_samples, n_outputs), default=None 
451          If not None, data is split in a stratified fashion, using this as 
452          the class labels. 
453   
454      Returns 
455      ------- 
456      resampled_arrays : sequence of array-like of shape (n_samples,) or \ 
457              (n_samples, n_outputs) 
458          Sequence of resampled copies of the collections. The original arrays 
459          are not impacted. 
460   
461      See Also 
462      -------- 
463      shuffle : Shuffle arrays or sparse matrices in a consistent way. 
464   
465      Examples 
466      -------- 
467      It is possible to mix sparse and dense arrays in the same run:: 
468   
469        >>> import numpy as np 
470        >>> X = np.array([[1., 0.], [2., 1.], [0., 0.]]) 
471        >>> y = np.array([0, 1, 2]) 
472   
473        >>> from scipy.sparse import coo_matrix 
474        >>> X_sparse = coo_matrix(X) 
475   
476        >>> from sklearn.utils import resample 
477        >>> X, X_sparse, y = resample(X, X_sparse, y, random_state=0) 
478        >>> X 
479        array([[1., 0.], 
480               [2., 1.], 
481               [1., 0.]]) 
482   
483        >>> X_sparse 
484        <Compressed Sparse Row sparse matrix of dtype 'float64' 
485            with 4 stored elements and shape (3, 2)> 
486   
487        >>> X_sparse.toarray() 
488        array([[1., 0.], 
489               [2., 1.], 
490               [1., 0.]]) 
491   
492        >>> y 
493        array([0, 1, 0]) 
494   
495        >>> resample(y, n_samples=2, random_state=0) 
496        array([0, 1]) 
497   
498      Example using stratification:: 
499   
500        >>> y = [0, 0, 1, 1, 1, 1, 1, 1, 1] 
501        >>> resample(y, n_samples=5, replace=False, stratify=y, 
502        ...          random_state=0) 
503        [1, 1, 1, 0, 1] 
504      """
505      max_n_samples = n_samples
506      random_state = check_random_state(random_state)
507  
508      if len(arrays) == 0:
509          return None
510  
511      first = arrays[0]
512      n_samples = first.shape[0] if hasattr(first, "shape") else len(first)
513  
514      if max_n_samples is None:
515          max_n_samples = n_samples
516      elif (max_n_samples > n_samples) and (not replace):
517          raise ValueError(
518              "Cannot sample %d out of arrays with dim %d when replace is False"
519              % (max_n_samples, n_samples)
520          )
521  
522      check_consistent_length(*arrays)
523  
524      if stratify is None:
525          if replace:
526              indices = random_state.randint(0, n_samples, size=(max_n_samples,))
527          else:
528              indices = np.arange(n_samples)
529              random_state.shuffle(indices)
530              indices = indices[:max_n_samples]
531      else:
532          # Code adapted from StratifiedShuffleSplit()
533          y = check_array(stratify, ensure_2d=False, dtype=None)
534          if y.ndim == 2:
535              # for multi-label y, map each distinct row to a string repr
536              # using join because str(row) uses an ellipsis if len(row) > 1000
537              y = np.array([" ".join(row.astype("str")) for row in y])
538  
539          classes, y_indices = np.unique(y, return_inverse=True)
540          n_classes = classes.shape[0]
541  
542          class_counts = np.bincount(y_indices)
543  
544          # Find the sorted list of instances for each class:
545          # (np.unique above performs a sort, so code is O(n logn) already)
546          class_indices = np.split(
547              np.argsort(y_indices, kind="mergesort"), np.cumsum(class_counts)[:-1]
548          )
549  
550          n_i = _approximate_mode(class_counts, max_n_samples, random_state)
551  
552          indices = []
553  
554          for i in range(n_classes):
555              indices_i = random_state.choice(class_indices[i], n_i[i], replace=replace)
556              indices.extend(indices_i)
557  
558          indices = random_state.permutation(indices)
559  
560      # convert sparse matrices to CSR for row-based indexing
561      arrays = [a.tocsr() if issparse(a) else a for a in arrays]
562      resampled_arrays = [_safe_indexing(a, indices) for a in arrays]
563      if len(resampled_arrays) == 1:
564          # syntactic sugar for the unit argument case
565          return resampled_arrays[0]
566      else:
567          return resampled_arrays

570  def shuffle(*arrays, random_state=None, n_samples=None):
571      """Shuffle arrays or sparse matrices in a consistent way. 
572   
573      This is a convenience alias to ``resample(*arrays, replace=False)`` to do 
574      random permutations of the collections. 
575   
576      Parameters 
577      ---------- 
578      *arrays : sequence of indexable data-structures 
579          Indexable data-structures can be arrays, lists, dataframes or scipy 
580          sparse matrices with consistent first dimension. 
581   
582      random_state : int, RandomState instance or None, default=None 
583          Determines random number generation for shuffling 
584          the data. 
585          Pass an int for reproducible results across multiple function calls. 
586          See :term:`Glossary <random_state>`. 
587   
588      n_samples : int, default=None 
589          Number of samples to generate. If left to None this is 
590          automatically set to the first dimension of the arrays.  It should 
591          not be larger than the length of arrays. 
592   
593      Returns 
594      ------- 
595      shuffled_arrays : sequence of indexable data-structures 
596          Sequence of shuffled copies of the collections. The original arrays 
597          are not impacted. 
598   
599      See Also 
600      -------- 
601      resample : Resample arrays or sparse matrices in a consistent way. 
602   
603      Examples 
604      -------- 
605      It is possible to mix sparse and dense arrays in the same run:: 
606   
607        >>> import numpy as np 
608        >>> X = np.array([[1., 0.], [2., 1.], [0., 0.]]) 
609        >>> y = np.array([0, 1, 2]) 
610   
611        >>> from scipy.sparse import coo_matrix 
612        >>> X_sparse = coo_matrix(X) 
613   
614        >>> from sklearn.utils import shuffle 
615        >>> X, X_sparse, y = shuffle(X, X_sparse, y, random_state=0) 
616        >>> X 
617        array([[0., 0.], 
618               [2., 1.], 
619               [1., 0.]]) 
620   
621        >>> X_sparse 
622        <Compressed Sparse Row sparse matrix of dtype 'float64' 
623            with 3 stored elements and shape (3, 2)> 
624   
625        >>> X_sparse.toarray() 
626        array([[0., 0.], 
627               [2., 1.], 
628               [1., 0.]]) 
629   
630        >>> y 
631        array([2, 1, 0]) 
632   
633        >>> shuffle(y, n_samples=2, random_state=0) 
634        array([0, 1]) 
635      """
636      return resample(
637          *arrays, replace=False, n_samples=n_samples, random_state=random_state
638      )
639
