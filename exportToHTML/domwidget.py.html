4    """Contains the DOMWidget class"""
5    
6    from traitlets import Bool, Unicode
7    from .widget import Widget, widget_serialization
8    from .trait_types import InstanceDict, TypedTuple
9    from .widget_layout import Layout
10   from .widget_style import Style
11   
12   
13   class DOMWidget(Widget):
14       """Widget that can be inserted into the DOM 
15    
16       Parameters 
17       ---------- 
18       tooltip: str 
19          tooltip caption 
20       layout: InstanceDict(Layout) 
21          widget layout 
22       """
23   
24       _model_name = Unicode('DOMWidgetModel').tag(sync=True)
25       _dom_classes = TypedTuple(trait=Unicode(), help="CSS classes applied to widget DOM element").tag(sync=True)
26       tabbable = Bool(help="Is widget tabbable?", allow_none=True, default_value=None).tag(sync=True)
27       tooltip = Unicode(None, allow_none=True, help="A tooltip caption.").tag(sync=True)
28       layout = InstanceDict(Layout).tag(sync=True, **widget_serialization)
29   
30       def add_class(self, className):
31           """ 
32           Adds a class to the top level element of the widget. 
33    
34           Doesn't add the class if it already exists. 
35           """
36           if className not in self._dom_classes:
37               self._dom_classes = list(self._dom_classes) + [className]
38           return self
39   
40       def remove_class(self, className):
41           """ 
42           Removes a class from the top level element of the widget. 
43    
44           Doesn't remove the class if it doesn't exist. 
45           """
46           if className in self._dom_classes:
47               self._dom_classes = [c for c in self._dom_classes if c != className]
48           return self
49   
50       def focus(self):
51           """ 
52           Focus on the widget. 
53           """
54           self.send({'do':'focus'})
55   
56       def blur(self):
57           """ 
58           Blur the widget. 
59           """
60           self.send({'do':'blur'})
61   
62       def _repr_keys(self):
63           for key in super()._repr_keys():
64               # Exclude layout if it had the default value
65               if key == 'layout':
66                   value = getattr(self, key)
67                   if repr(value) == '%s()' % value.__class__.__name__:
68                       continue
69               yield key
70           # We also need to include _dom_classes in repr for reproducibility
71           if self._dom_classes:
72               yield '_dom_classes'
