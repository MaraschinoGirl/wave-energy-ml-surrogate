95       def test_continuous_squareness_and_shape(self):
96           nsq = np.ones((3, 2))
97           sq = np.eye(3)
98           assert_raises(ValueError, solve_continuous_lyapunov, nsq, sq)
99           assert_raises(ValueError, solve_continuous_lyapunov, sq, nsq)
100          assert_raises(ValueError, solve_continuous_lyapunov, sq, np.eye(2))

102      def check_continuous_case(self, a, q):
103          x = solve_continuous_lyapunov(a, q)
104          assert_array_almost_equal(
105                            np.dot(a, x) + np.dot(x, a.conj().transpose()), q)
106  
107      def check_discrete_case(self, a, q, method=None):
108          x = solve_discrete_lyapunov(a, q, method=method)
109          assert_array_almost_equal(
110                        np.dot(np.dot(a, x), a.conj().transpose()) - x, -1.0*q)

116              self.check_discrete_case(case[0], case[1], method='direct')
117              self.check_discrete_case(case[0], case[1], method='bilinear')

317      @pytest.mark.parametrize("j, case", enumerate(cases))
318      def test_solve_continuous_are(self, j, case):
319          """Checks if 0 = XA + A'X - XB(R)^{-1} B'X + Q is true"""
320          a, b, q, r, knownfailure = case
321          if knownfailure:
322              pytest.xfail(reason=knownfailure)
323  
324          dec = self.min_decimal[j]
325          x = solve_continuous_are(a, b, q, r)
326          res = x @ a + a.conj().T @ x + q
327          out_fact = x @ b
328          res -= out_fact @ solve(np.atleast_2d(r), out_fact.conj().T)
329          assert_array_almost_equal(res, np.zeros_like(res), decimal=dec)

548          """Checks if X = A'XA-(A'XB)(R+B'XB)^-1(B'XA)+Q) is true"""
549          a, b, q, r, knownfailure = case
550          if knownfailure:
551              pytest.xfail(reason=knownfailure)
552  
553          atol = self.max_tol[j]
554  
555          x = solve_discrete_are(a, b, q, r)
556          bH = b.conj().T
557          xa, xb = x @ a, x @ b
558  
559          res = a.conj().T @ xa - x + q

565          # preferred
566          assert_allclose(res, np.zeros_like(res), atol=atol)
