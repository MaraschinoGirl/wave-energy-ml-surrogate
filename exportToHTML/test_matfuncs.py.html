27   def _burkardt_13_power(n, p):
28       """ 
29       A helper function for testing matrix functions. 
30    
31       Parameters 
32       ---------- 
33       n : integer greater than 1 
34           Order of the square matrix to be returned. 
35       p : non-negative integer 
36           Power of the matrix. 
37    
38       Returns 
39       ------- 
40       out : ndarray representing a square matrix 
41           A Forsythe matrix of order n, raised to the power p. 
42    
43       """
44       # Input validation.
45       if n != int(n) or n < 2:
46           raise ValueError('n must be an integer greater than 1')
47       n = int(n)
48       if p != int(p) or p < 0:
49           raise ValueError('p must be a non-negative integer')
50       p = int(p)
51   
52       # Construct the matrix explicitly.
53       a, b = divmod(p, n)
54       large = np.power(10.0, -n*a)
55       small = large * np.power(10.0, -n)
56       return np.diag([large]*(n-b), b) + np.diag([small]*b, b-n)

229      def test_burkardt_1(self):
230          # This matrix is diagonal.
231          # The calculation of the matrix exponential is simple.
232          #
233          # This is the first of a series of matrix exponential tests
234          # collected by John Burkardt from the following sources.
235          #
236          # Alan Laub,
237          # Review of "Linear System Theory" by Joao Hespanha,
238          # SIAM Review,
239          # Volume 52, Number 4, December 2010, pages 779--781.
240          #
241          # Cleve Moler and Charles Van Loan,
242          # Nineteen Dubious Ways to Compute the Exponential of a Matrix,
243          # Twenty-Five Years Later,
244          # SIAM Review,
245          # Volume 45, Number 1, March 2003, pages 3--49.
246          #
247          # Cleve Moler,
248          # Cleve's Corner: A Balancing Act for the Matrix Exponential,
249          # 23 July 2012.
250          #
251          # Robert Ward,
252          # Numerical computation of the matrix exponential
253          # with accuracy estimate,
254          # SIAM Journal on Numerical Analysis,
255          # Volume 14, Number 4, September 1977, pages 600--610.
256          exp1 = np.exp(1)
257          exp2 = np.exp(2)
258          A = np.array([
259              [1, 0],
260              [0, 2],
261              ], dtype=float)
262          desired = np.array([
263              [exp1, 0],
264              [0, exp2],
265              ], dtype=float)
266          actual = expm(A)
267          assert_allclose(actual, desired)

469      def test_burkardt_13(self):
470          # This is Ward's example #4.
471          # This is a version of the Forsythe matrix.
472          # The eigenvector problem is badly conditioned.
473          # Ward's algorithm has difficulty estimating the accuracy
474          # of its results for this problem.
475          #
476          # Check the construction of one instance of this family of matrices.
477          A4_actual = _burkardt_13_power(4, 1)
478          A4_desired = [[0, 1, 0, 0],
479                        [0, 0, 1, 0],
480                        [0, 0, 0, 1],
481                        [1e-4, 0, 0, 0]]
482          assert_allclose(A4_actual, A4_desired)
483          # Check the expm for a few instances.
484          for n in (2, 3, 4, 10):
485              # Approximate expm using Taylor series.
486              # This works well for this matrix family
487              # because each matrix in the summation,
488              # even before dividing by the factorial,
489              # is entrywise positive with max entry 10**(-floor(p/n)*n).
490              k = max(1, int(np.ceil(16/n)))
491              desired = np.zeros((n, n), dtype=float)
492              for p in range(n*k):
493                  Ap = _burkardt_13_power(n, p)
494                  assert_equal(np.min(Ap), 0)
495                  assert_allclose(np.max(Ap), np.power(10, -np.floor(p/n)*n))
496                  desired += Ap / factorial(p)
497              actual = expm(_burkardt_13_power(n, 1))
498              assert_allclose(actual, desired)

517          # Test pascal triangle.
518          # Nilpotent exponential, used to trigger a failure (gh-8029)
519  
520          for scale in [1.0, 1e-3, 1e-6]:
521              for n in range(0, 80, 3):

523                  if np.any(sc < 1e-300):
524                      break

526                  A = np.diag(np.arange(1, n + 1), -1) * scale
527                  B = expm(A)
528  
529                  got = B
530                  expected = binom(np.arange(n + 1)[:,None],
531                                   np.arange(n + 1)[None,:]) * sc[None,:] / sc[:,None]
