37   class SKLBase(BaseEstimator):
38       """Base class for scikit-learn wrappers. 
39    
40       Note that there are sources of randomness in model initialization and 
41       training. Refer to [Reproducibility in Keras Models]( 
42       https://keras.io/examples/keras_recipes/reproducibility_recipes/) on how to 
43       control randomness. 
44    
45       Args: 
46           model: `Model`. 
47               An instance of `Model`, or a callable returning such an object. 
48               Note that if input is a `Model`, it will be cloned using 
49               `keras.models.clone_model` before being fitted, unless 
50               `warm_start=True`. 
51               The `Model` instance needs to be passed as already compiled. 
52               If callable, it must accept at least `X` and `y` as keyword 
53               arguments. Other arguments must be accepted if passed as 
54               `model_kwargs` by the user. 
55           warm_start: bool, defaults to `False`. 
56               Whether to reuse the model weights from the previous fit. If `True`, 
57               the given model won't be cloned and the weights from the previous 
58               fit will be reused. 
59           model_kwargs: dict, defaults to `None`. 
60               Keyword arguments passed to `model`, if `model` is callable. 
61           fit_kwargs: dict, defaults to `None`. 
62               Keyword arguments passed to `model.fit`. These can also be passed 
63               directly to the `fit` method of the scikit-learn wrapper. The 
64               values passed directly to the `fit` method take precedence over 
65               these. 
66    
67       Attributes: 
68           model_ : `Model` 
69               The fitted model. 
70           history_ : dict 
71               The history of the fit, returned by `model.fit`. 
72       """
73   
74       def __init__(
75           self,
76           model,
77           warm_start=False,
78           model_kwargs=None,
79           fit_kwargs=None,
80       ):
81           assert_sklearn_installed(self.__class__.__name__)
82           self.model = model
83           self.warm_start = warm_start
84           self.model_kwargs = model_kwargs
85           self.fit_kwargs = fit_kwargs
86   
87       def _more_tags(self):
88           return {"non_deterministic": True}
89   
90       def __sklearn_tags__(self):
91           tags = super().__sklearn_tags__()
92           tags.non_deterministic = True
93           return tags
94   
95       def __sklearn_clone__(self):
96           """Return a deep copy of the model. 
97    
98           This is used by the `sklearn.base.clone` function. 
99           """
100          model = (
101              self.model if callable(self.model) else copy.deepcopy(self.model)
102          )
103          return type(self)(
104              model=model,
105              warm_start=self.warm_start,
106              model_kwargs=self.model_kwargs,
107          )
108  
109      @property
110      def epoch_(self):
111          """The current training epoch."""
112          return getattr(self, "history_", {}).get("epoch", 0)
113  
114      def set_fit_request(self, **kwargs):
115          """Set requested parameters by the fit method. 
116   
117          Please see [scikit-learn's metadata routing]( 
118          https://scikit-learn.org/stable/metadata_routing.html) for more 
119          details. 
120   
121   
122          Arguments: 
123              kwargs : dict 
124                  Arguments should be of the form `param_name=alias`, and `alias` 
125                  can be one of `{True, False, None, str}`. 
126   
127          Returns: 
128              self 
129          """
130          if not _routing_enabled():
131              raise RuntimeError(
132                  "This method is only available when metadata routing is "
133                  "enabled. You can enable it using "
134                  "sklearn.set_config(enable_metadata_routing=True)."
135              )
136  
137          self._metadata_request = sklearn.utils.metadata_routing.MetadataRequest(
138              owner=self.__class__.__name__
139          )
140          for param, alias in kwargs.items():
141              self._metadata_request.score.add_request(param=param, alias=alias)
142          return self
143  
144      def _get_model(self, X, y):
145          if isinstance(self.model, Model):
146              return clone_model(self.model)
147          else:
148              args = self.model_kwargs or {}
149              return self.model(X=X, y=y, **args)
150  
151      def fit(self, X, y, **kwargs):
152          """Fit the model. 
153   
154          Args: 
155              X: array-like, shape=(n_samples, n_features) 
156                  The input samples. 
157              y: array-like, shape=(n_samples,) or (n_samples, n_outputs) 
158                  The targets. 
159              **kwargs: keyword arguments passed to `model.fit` 
160          """
161          X, y = _validate_data(self, X, y)
162          y = self._process_target(y, reset=True)
163          model = self._get_model(X, y)
164          _check_model(model)
165  
166          fit_kwargs = self.fit_kwargs or {}
167          fit_kwargs.update(kwargs)
168          self.history_ = model.fit(X, y, **fit_kwargs)
169  
170          self.model_ = model
171          return self
172  
173      def predict(self, X):
174          """Predict using the model."""
175          sklearn.base.check_is_fitted(self)
176          X = _validate_data(self, X, reset=False)
177          raw_output = self.model_.predict(X)
178          return self._reverse_process_target(raw_output)
179  
180      def _process_target(self, y, reset=False):
181          """Regressors are NOOP here, classifiers do OHE."""
182          # This is here to raise the right error in case of invalid target
183          type_of_target(y, raise_unknown=True)
184          if reset:
185              self._target_encoder = TargetReshaper().fit(y)
186          return self._target_encoder.transform(y)
187  
188      def _reverse_process_target(self, y):
189          """Regressors are NOOP here, classifiers reverse OHE."""
190          return self._target_encoder.inverse_transform(y)

193  @keras_export("keras.wrappers.SKLearnClassifier")
194  class SKLearnClassifier(ClassifierMixin, SKLBase):
195      """scikit-learn compatible classifier wrapper for Keras models. 
196   
197      Note that there are sources of randomness in model initialization and 
198      training. Refer to [Reproducibility in Keras Models]( 
199      https://keras.io/examples/keras_recipes/reproducibility_recipes/) on how to 
200      control randomness. 
201   
202      Args: 
203          model: `Model`. 
204              An instance of `Model`, or a callable returning such an object. 
205              Note that if input is a `Model`, it will be cloned using 
206              `keras.models.clone_model` before being fitted, unless 
207              `warm_start=True`. 
208              The `Model` instance needs to be passed as already compiled. 
209              If callable, it must accept at least `X` and `y` as keyword 
210              arguments. Other arguments must be accepted if passed as 
211              `model_kwargs` by the user. 
212          warm_start: bool, defaults to `False`. 
213              Whether to reuse the model weights from the previous fit. If `True`, 
214              the given model won't be cloned and the weights from the previous 
215              fit will be reused. 
216          model_kwargs: dict, defaults to `None`. 
217              Keyword arguments passed to `model`, if `model` is callable. 
218          fit_kwargs: dict, defaults to `None`. 
219              Keyword arguments passed to `model.fit`. These can also be passed 
220              directly to the `fit` method of the scikit-learn wrapper. The 
221              values passed directly to the `fit` method take precedence over 
222              these. 
223   
224      Attributes: 
225          model_ : `Model` 
226              The fitted model. 
227          history_ : dict 
228              The history of the fit, returned by `model.fit`. 
229          classes_ : array-like, shape=(n_classes,) 
230              The classes labels. 
231   
232      Example: 
233      Here we use a function which creates a basic MLP model dynamically 
234      choosing the input and output shapes. We will use this to create our 
235      scikit-learn model. 
236   
237      ``` python 
238      from keras.src.layers import Dense, Input, Model 
239   
240      def dynamic_model(X, y, loss, layers=[10]): 
241          # Creates a basic MLP model dynamically choosing the input and 
242          # output shapes. 
243          n_features_in = X.shape[1] 
244          inp = Input(shape=(n_features_in,)) 
245   
246          hidden = inp 
247          for layer_size in layers: 
248              hidden = Dense(layer_size, activation="relu")(hidden) 
249   
250          n_outputs = y.shape[1] if len(y.shape) > 1 else 1 
251          out = [Dense(n_outputs, activation="softmax")(hidden)] 
252          model = Model(inp, out) 
253          model.compile(loss=loss, optimizer="rmsprop") 
254   
255          return model 
256      ``` 
257   
258      You can then use this function to create a scikit-learn compatible model 
259      and fit it on some data. 
260   
261      ``` python 
262      from sklearn.datasets import make_classification 
263      from keras.wrappers import SKLearnClassifier 
264   
265      X, y = make_classification(n_samples=1000, n_features=10, n_classes=3) 
266      est = SKLearnClassifier( 
267          model=dynamic_model, 
268          model_kwargs={ 
269              "loss": "categorical_crossentropy", 
270              "layers": [20, 20, 20], 
271          }, 
272      ) 
273   
274      est.fit(X, y, epochs=5) 
275      ``` 
276      """
277  
278      def _process_target(self, y, reset=False):
279          """Classifiers do OHE."""
280          target_type = type_of_target(y, raise_unknown=True)
281          if target_type not in ["binary", "multiclass"]:
282              raise ValueError(
283                  "Only binary and multiclass target types are supported."
284                  f" Target type: {target_type}"
285              )
286          if reset:
287              self._target_encoder = sklearn.pipeline.make_pipeline(
288                  TargetReshaper(),
289                  sklearn.preprocessing.OneHotEncoder(sparse_output=False),
290              ).fit(y)
291              self.classes_ = np.unique(y)
292              if len(self.classes_) == 1:
293                  raise ValueError(
294                      "Classifier can't train when only one class is present."
295                  )
296          return self._target_encoder.transform(y)
297  
298      def _more_tags(self):
299          # required to be compatible with scikit-learn<1.6
300          return {"poor_score": True}
301  
302      def __sklearn_tags__(self):
303          tags = super().__sklearn_tags__()
304          tags.classifier_tags.poor_score = True
305          return tags

399  @keras_export("keras.wrappers.SKLearnTransformer")
400  class SKLearnTransformer(TransformerMixin, SKLBase):
401      """scikit-learn compatible transformer wrapper for Keras models. 
402   
403      Note that this is a scikit-learn compatible transformer, and not a 
404      transformer in the deep learning sense. 
405   
406      Also note that there are sources of randomness in model initialization and 
407      training. Refer to [Reproducibility in Keras Models]( 
408      https://keras.io/examples/keras_recipes/reproducibility_recipes/) on how to 
409      control randomness. 
410   
411      Args: 
412          model: `Model`. 
413              An instance of `Model`, or a callable returning such an object. 
414              Note that if input is a `Model`, it will be cloned using 
415              `keras.models.clone_model` before being fitted, unless 
416              `warm_start=True`. 
417              The `Model` instance needs to be passed as already compiled. 
418              If callable, it must accept at least `X` and `y` as keyword 
419              arguments. Other arguments must be accepted if passed as 
420              `model_kwargs` by the user. 
421          warm_start: bool, defaults to `False`. 
422              Whether to reuse the model weights from the previous fit. If `True`, 
423              the given model won't be cloned and the weights from the previous 
424              fit will be reused. 
425          model_kwargs: dict, defaults to `None`. 
426              Keyword arguments passed to `model`, if `model` is callable. 
427          fit_kwargs: dict, defaults to `None`. 
428              Keyword arguments passed to `model.fit`. These can also be passed 
429              directly to the `fit` method of the scikit-learn wrapper. The 
430              values passed directly to the `fit` method take precedence over 
431              these. 
432   
433      Attributes: 
434          model_ : `Model` 
435              The fitted model. 
436          history_ : dict 
437              The history of the fit, returned by `model.fit`. 
438   
439      Example: 
440      A common use case for a scikit-learn transformer, is to have a step 
441      which gives you the embedding of your data. Here we assume 
442      `my_package.my_model` is a Keras model which takes the input and gives 
443      embeddings of the data, and `my_package.my_data` is your dataset loader. 
444   
445      ``` python 
446      from my_package import my_model, my_data 
447      from keras.wrappers import SKLearnTransformer 
448      from sklearn.frozen import FrozenEstimator # requires scikit-learn>=1.6 
449      from sklearn.pipeline import make_pipeline 
450      from sklearn.ensemble import HistGradientBoostingClassifier 
451   
452      X, y = my_data() 
453   
454      trs = FrozenEstimator(SKLearnTransformer(model=my_model)) 
455      pipe = make_pipeline(trs, HistGradientBoostingClassifier()) 
456      pipe.fit(X, y) 
457      ``` 
458   
459      Note that in the above example, `FrozenEstimator` prevents any further 
460      training of the transformer step in the pipeline, which can be the case 
461      if you don't want to change the embedding model at hand. 
462      """
463  
464      def transform(self, X):
465          """Transform the data. 
466   
467          Args: 
468              X: array-like, shape=(n_samples, n_features) 
469                  The input samples. 
470   
471          Returns: 
472              X_transformed: array-like, shape=(n_samples, n_features) 
473                  The transformed data. 
474          """
475          sklearn.base.check_is_fitted(self)
476          X = _validate_data(self, X, reset=False)
477          return self.model_.predict(X)
478  
479      def _more_tags(self):
480          # required to be compatible with scikit-learn<1.6
481          return {
482              "preserves_dtype": [],
483          }
484  
485      def __sklearn_tags__(self):
486          tags = super().__sklearn_tags__()
487          tags.transformer_tags.preserves_dtype = []
488          return tags
