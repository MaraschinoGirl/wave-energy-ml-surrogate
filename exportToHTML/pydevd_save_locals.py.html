17       return save_locals_impl is not None

20   def save_locals(frame):
21       """ 
22       Copy values from locals_dict into the fast stack slots in the given frame. 
23    
24       Note: the 'save_locals' branch had a different approach wrapping the frame (much more code, but it gives ideas 
25       on how to save things partially, not the 'whole' locals). 
26       """
27       if not isinstance(frame, frame_type):
28           # Fix exception when changing Django variable (receiving DjangoTemplateFrame)
29           return
30   
31       if save_locals_impl is not None:
32           try:
33               save_locals_impl(frame)
34           except:
35               pass

38   def make_save_locals_impl():
39       """ 
40       Factory for the 'save_locals_impl' method. This may seem like a complicated pattern but it is essential that the method is created at 
41       module load time. Inner imports after module load time would cause an occasional debugger deadlock due to the importer lock and debugger 
42       lock being taken in different order in  different threads. 
43       """
44       try:
45           if "__pypy__" in sys.builtin_module_names:
46               import __pypy__  # @UnresolvedImport
47   
48               save_locals = __pypy__.locals_to_fast
49       except:
50           pass
51       else:
52           if "__pypy__" in sys.builtin_module_names:
53   
54               def save_locals_pypy_impl(frame):
55                   save_locals(frame)
56   
57               return save_locals_pypy_impl
58   
59       if IS_PY313_OR_GREATER:
60           # No longer needed in Python 3.13 (deprecated)
61           # See PEP 667
62           return None
63   
64       try:
65           import ctypes
66   
67           locals_to_fast = ctypes.pythonapi.PyFrame_LocalsToFast
68       except:
69           pass
70       else:
71   
72           def save_locals_ctypes_impl(frame):
73               locals_to_fast(ctypes.py_object(frame), ctypes.c_int(0))
74   
75           return save_locals_ctypes_impl
76   
77       return None
78   
79   
80   save_locals_impl = make_save_locals_impl()

82   _SENTINEL = []  # Any mutable will do.
83   
84   
85   def update_globals_and_locals(updated_globals, initial_globals, frame):
86       # We don't have the locals and passed all in globals, so, we have to
87       # manually choose how to update the variables.
88       #
89       # Note that the current implementation is a bit tricky: it does work in general
90       # but if we do something as 'some_var = 10' and 'some_var' is already defined to have
91       # the value '10' in the globals, we won't actually put that value in the locals
92       # (which means that the frame locals won't be updated).
93       # Still, the approach to have a single namespace was chosen because it was the only
94       # one that enabled creating and using variables during the same evaluation.
95       assert updated_globals is not None
96       f_locals = None
97   
98       removed = set(initial_globals).difference(updated_globals)
99   
100      for key, val in updated_globals.items():
101          if val is not initial_globals.get(key, _SENTINEL):
102              if f_locals is None:
103                  # Note: we call f_locals only once because each time
104                  # we call it the values may be reset.
105                  f_locals = frame.f_locals
106  
107              f_locals[key] = val
108  
109      if removed:
110          if f_locals is None:
111              # Note: we call f_locals only once because each time
112              # we call it the values may be reset.
113              f_locals = frame.f_locals
114  
115          for key in removed:
116              try:
117                  del f_locals[key]
118              except Exception:
119                  # Python 3.13.0 has issues here:
120                  # https://github.com/python/cpython/pull/125616
121                  # This should be backported from the pull request
122                  # but we still need to handle it in this version
123                  try:
124                      if key in f_locals:
125                          f_locals[key] = None
126                  except Exception as e:
127                      pydev_log.info("Unable to remove key: %s from locals. Exception: %s", key, e)
128  
129      if f_locals is not None:
130          save_locals(frame)
