31   def OnPaint_1(hwnd, msg, wp, lp):
32       dc, ps = win32gui.BeginPaint(hwnd)
33       win32gui.SetGraphicsMode(dc, win32con.GM_ADVANCED)
34       br = win32gui.CreateSolidBrush(win32api.RGB(255, 0, 0))
35       win32gui.SelectObject(dc, br)
36       angle = win32gui.GetWindowLong(hwnd, win32con.GWL_USERDATA)
37       win32gui.SetWindowLong(hwnd, win32con.GWL_USERDATA, angle + 2)
38       r_angle = angle * (math.pi / 180)
39       win32gui.SetWorldTransform(
40           dc,
41           {
42               "M11": math.cos(r_angle),
43               "M12": math.sin(r_angle),
44               "M21": math.sin(r_angle) * -1,
45               "M22": math.cos(r_angle),
46               "Dx": 250,
47               "Dy": 250,
48           },
49       )
50       win32gui.MoveToEx(dc, 250, 250)
51       win32gui.BeginPath(dc)
52       win32gui.Pie(dc, 10, 70, 200, 200, 350, 350, 75, 10)
53       win32gui.Chord(dc, 200, 200, 850, 0, 350, 350, 75, 10)
54       win32gui.LineTo(dc, 300, 300)
55       win32gui.LineTo(dc, 100, 20)
56       win32gui.LineTo(dc, 20, 100)
57       win32gui.LineTo(dc, 400, 0)
58       win32gui.LineTo(dc, 0, 400)
59       win32gui.EndPath(dc)
60       win32gui.StrokeAndFillPath(dc)
61       win32gui.EndPaint(hwnd, ps)
62       return 0

109  def TestSetWorldTransform():
110      wc = win32gui.WNDCLASS()
111      wc.lpszClassName = "test_win32gui_1"
112      wc.style = win32con.CS_GLOBALCLASS | win32con.CS_VREDRAW | win32con.CS_HREDRAW
113      wc.hbrBackground = win32con.COLOR_WINDOW + 1
114      wc.lpfnWndProc = wndproc_1
115      class_atom = win32gui.RegisterClass(wc)
116      hwnd = win32gui.CreateWindow(
117          wc.lpszClassName,
118          "Spin the Lobster!",
119          win32con.WS_CAPTION | win32con.WS_VISIBLE,
120          100,
121          100,
122          900,
123          900,
124          0,
125          0,
126          0,
127          None,
128      )
129      for x in range(500):
130          win32gui.InvalidateRect(hwnd, None, True)
131          win32gui.PumpWaitingMessages()
132          time.sleep(0.01)
133      win32gui.DestroyWindow(hwnd)
134      win32gui.UnregisterClass(wc.lpszClassName, None)

142      wc.lpfnWndProc = wndproc_2
143      class_atom = win32gui.RegisterClass(wc)
144      hwnd = win32gui.CreateWindowEx(
145          0,
146          class_atom,
147          "Kaleidoscope",
148          win32con.WS_CAPTION
149          | win32con.WS_VISIBLE
150          | win32con.WS_THICKFRAME
151          | win32con.WS_SYSMENU,
152          100,
153          100,
154          900,
155          900,
156          0,
157          0,
158          0,
159          None,
160      )
161      s = win32gui.GetWindowLong(hwnd, win32con.GWL_EXSTYLE)
162      win32gui.SetWindowLong(hwnd, win32con.GWL_EXSTYLE, s | win32con.WS_EX_LAYERED)
163      win32gui.SetLayeredWindowAttributes(hwnd, 0, 175, win32con.LWA_ALPHA)
164      for x in range(30):
165          win32gui.InvalidateRect(hwnd, None, True)
166          win32gui.PumpWaitingMessages()
167          time.sleep(0.3)
168      win32gui.DestroyWindow(hwnd)
169      win32gui.UnregisterClass(class_atom, None)
