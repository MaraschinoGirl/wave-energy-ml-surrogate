8    class ListDialog(dialog.Dialog):
9        def __init__(self, title, list):
10           dialog.Dialog.__init__(self, self._maketemplate(title))
11           self.HookMessage(self.on_size, win32con.WM_SIZE)
12           self.HookNotify(self.OnListItemChange, commctrl.LVN_ITEMCHANGED)
13           self.HookCommand(self.OnListClick, win32ui.IDC_LIST1)
14           self.items = list
15   
16       def _maketemplate(self, title):
17           style = win32con.WS_DLGFRAME | win32con.WS_SYSMENU | win32con.WS_VISIBLE
18           ls = (
19               win32con.WS_CHILD
20               | win32con.WS_VISIBLE
21               | commctrl.LVS_ALIGNLEFT
22               | commctrl.LVS_REPORT
23           )
24           bs = win32con.WS_CHILD | win32con.WS_VISIBLE
25           return [
26               [title, (0, 0, 200, 200), style, None, (8, "MS Sans Serif")],
27               ["SysListView32", None, win32ui.IDC_LIST1, (0, 0, 200, 200), ls],
28               [128, "OK", win32con.IDOK, (10, 0, 50, 14), bs | win32con.BS_DEFPUSHBUTTON],
29               [128, "Cancel", win32con.IDCANCEL, (0, 0, 50, 14), bs],
30           ]
31   
32       def FillList(self):
33           size = self.GetWindowRect()
34           width = size[2] - size[0] - (10)
35           itemDetails = (commctrl.LVCFMT_LEFT, width, "Item", 0)
36           self.itemsControl.InsertColumn(0, itemDetails)
37           index = 0
38           for item in self.items:
39               index = self.itemsControl.InsertItem(index + 1, str(item), 0)
40   
41       def OnListClick(self, id, code):
42           if code == commctrl.NM_DBLCLK:
43               self.EndDialog(win32con.IDOK)
44           return 1
45   
46       def OnListItemChange(self, std, extra):
47           (
48               (hwndFrom, idFrom, code),
49               (
50                   itemNotify,
51                   sub,
52                   newState,
53                   oldState,
54                   change,
55                   point,
56                   lparam,
57               ),
58           ) = (std, extra)
59           oldSel = (oldState & commctrl.LVIS_SELECTED) != 0
60           newSel = (newState & commctrl.LVIS_SELECTED) != 0
61           if oldSel != newSel:
62               try:
63                   self.selecteditem = itemNotify
64                   self.butOK.EnableWindow(1)
65               except win32ui.error:
66                   self.selecteditem = None
67   
68       def OnInitDialog(self):
69           rc = dialog.Dialog.OnInitDialog(self)
70           self.itemsControl = self.GetDlgItem(win32ui.IDC_LIST1)
71           self.butOK = self.GetDlgItem(win32con.IDOK)
72           self.butCancel = self.GetDlgItem(win32con.IDCANCEL)
73   
74           self.FillList()
75   
76           size = self.GetWindowRect()
77           self.LayoutControls(size[2] - size[0], size[3] - size[1])
78           self.butOK.EnableWindow(0)  # wait for first selection
79           return rc
80   
81       def LayoutControls(self, w, h):
82           self.itemsControl.MoveWindow((0, 0, w, h - 30))
83           self.butCancel.MoveWindow((10, h - 24, 60, h - 4))
84           self.butOK.MoveWindow((w - 60, h - 24, w - 10, h - 4))
85   
86       def on_size(self, params):
87           lparam = params[3]
88           w = win32api.LOWORD(lparam)
89           h = win32api.HIWORD(lparam)
90           self.LayoutControls(w, h)

93   class ListsDialog(ListDialog):
94       def __init__(self, title, list, colHeadings=["Item"]):
95           ListDialog.__init__(self, title, list)
96           self.colHeadings = colHeadings
97   
98       def FillList(self):
99           index = 0
100          size = self.GetWindowRect()
101          width = (
102              size[2] - size[0] - (10) - win32api.GetSystemMetrics(win32con.SM_CXVSCROLL)
103          )
104          numCols = len(self.colHeadings)
105  
106          for col in self.colHeadings:
107              itemDetails = (commctrl.LVCFMT_LEFT, int(width / numCols), col, 0)
108              self.itemsControl.InsertColumn(index, itemDetails)
109              index += 1
110          index = 0
111          for items in self.items:
112              index = self.itemsControl.InsertItem(index + 1, str(items[0]), 0)
113              for itemno in range(1, numCols):
114                  item = items[itemno]
115                  self.itemsControl.SetItemText(index, itemno, str(item))

118  def SelectFromList(title, lst):
119      dlg = ListDialog(title, lst)
120      if dlg.DoModal() == win32con.IDOK:
121          return dlg.selecteditem
122      else:
123          return None
