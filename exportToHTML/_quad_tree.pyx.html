36   cdef class _QuadTree:
37       """Array-based representation of a QuadTree. 
38    
39       This class is currently working for indexing 2D data (regular QuadTree) and 
40       for indexing 3D data (OcTree). It is planned to split the 2 implementations 
41       using `Cython.Tempita` to save some memory for QuadTree. 
42    
43       Note that this code is currently internally used only by the Barnes-Hut 
44       method in `sklearn.manifold.TSNE`. It is planned to be refactored and 
45       generalized in the future to be compatible with nearest neighbors API of 
46       `sklearn.neighbors` with 2D and 3D data.

77       def build_tree(self, X):
78           """Build a tree from an array of points X."""
79           cdef:
80               int i
81               float32_t[3] pt
82               float32_t[3] min_bounds, max_bounds
83   
84           # validate X and prepare for query
85           # X = check_array(X, dtype=float32_t, order='C')
86           n_samples = X.shape[0]
87   
88           capacity = 100
89           self._resize(capacity)
90           m = np.min(X, axis=0)
91           M = np.max(X, axis=0)
92           # Scale the maximum to get all points strictly in the tree bounding box
93           # The 3 bounds are for positive, negative and small values
94           M = np.maximum(M * (1. + 1e-3 * np.sign(M)), M + 1e-3)
95           for i in range(self.n_dimensions):
96               min_bounds[i] = m[i]
97               max_bounds[i] = M[i]
98   
99               if self.verbose > 10:
100                  printf("[QuadTree] bounding box axis %i : [%f, %f]\n",
101                         i, min_bounds[i], max_bounds[i])
102  
103          # Create the initial node with boundaries from the dataset
104          self._init_root(min_bounds, max_bounds)
105  
106          for i in range(n_samples):
107              for j in range(self.n_dimensions):
108                  pt[j] = X[i, j]
109              self.insert_point(pt, i)
110  
111          # Shrink the cells array to reduce memory usage
112          self._resize(capacity=self.cell_count)

114      cdef int insert_point(self, float32_t[3] point, intp_t point_index,
115                            intp_t cell_id=0) except -1 nogil:
116          """Insert a point in the QuadTree."""
117          cdef int ax
118          cdef intp_t selected_child
119          cdef Cell* cell = &self.cells[cell_id]
120          cdef intp_t n_point = cell.cumulative_size
121  
122          if self.verbose > 10:
123              printf("[QuadTree] Inserting depth %li\n", cell.depth)
124  
125          # Assert that the point is in the right range
126          if DEBUGFLAG:
127              self._check_point_in_cell(point, cell)
128  
129          # If the cell is an empty leaf, insert the point in it
130          if cell.cumulative_size == 0:
131              cell.cumulative_size = 1
132              self.n_points += 1
133              for i in range(self.n_dimensions):
134                  cell.barycenter[i] = point[i]
135              cell.point_index = point_index
136              if self.verbose > 10:
137                  printf("[QuadTree] inserted point %li in cell %li\n",
138                         point_index, cell_id)
139              return cell_id
140  
141          # If the cell is not a leaf, update cell internals and
142          # recurse in selected child
143          if not cell.is_leaf:
144              for ax in range(self.n_dimensions):
145                  # barycenter update using a weighted mean
146                  cell.barycenter[ax] = (
147                      n_point * cell.barycenter[ax] + point[ax]) / (n_point + 1)
148  
149              # Increase the size of the subtree starting from this cell
150              cell.cumulative_size += 1
151  
152              # Insert child in the correct subtree
153              selected_child = self._select_child(point, cell)
154              if self.verbose > 49:
155                  printf("[QuadTree] selected child %li\n", selected_child)
156              if selected_child == -1:
157                  self.n_points += 1
158                  return self._insert_point_in_new_child(point, cell, point_index)
159              return self.insert_point(point, point_index, selected_child)
160  
161          # Finally, if the cell is a leaf with a point already inserted,
162          # split the cell in n_cells_per_cell if the point is not a duplicate.
163          # If it is a duplicate, increase the size of the leaf and return.
164          if self._is_duplicate(point, cell.barycenter):
165              if self.verbose > 10:
166                  printf("[QuadTree] found a duplicate!\n")
167              cell.cumulative_size += 1
168              self.n_points += 1
169              return cell_id
170  
171          # In a leaf, the barycenter correspond to the only point included
172          # in it.
173          self._insert_point_in_new_child(cell.barycenter, cell, cell.point_index,
174                                          cell.cumulative_size)
175          return self.insert_point(point, point_index, cell_id)

177      # XXX: This operation is not Thread safe
178      cdef intp_t _insert_point_in_new_child(
179          self, float32_t[3] point, Cell* cell, intp_t point_index, intp_t size=1
180      ) noexcept nogil:
181          """Create a child of cell which will contain point."""
182  
183          # Local variable definition
184          cdef:
185              intp_t cell_id, cell_child_id, parent_id
186              float32_t[3] save_point
187              float32_t width
188              Cell* child
189              int i
190  
191          # If the maximal capacity of the Tree have been reached, double the capacity
192          # We need to save the current cell id and the current point to retrieve them
193          # in case the reallocation
194          if self.cell_count + 1 > self.capacity:
195              parent_id = cell.cell_id
196              for i in range(self.n_dimensions):
197                  save_point[i] = point[i]
198              self._resize(SIZE_MAX)
199              cell = &self.cells[parent_id]
200              point = save_point
201  
202          # Get an empty cell and initialize it
203          cell_id = self.cell_count
204          self.cell_count += 1
205          child = &self.cells[cell_id]
206  
207          self._init_cell(child, cell.cell_id, cell.depth + 1)
208          child.cell_id = cell_id
209  
210          # Set the cell as an inner cell of the Tree
211          cell.is_leaf = False
212          cell.point_index = -1
213  
214          # Set the correct boundary for the cell, store the point in the cell
215          # and compute its index in the children array.
216          cell_child_id = 0
217          for i in range(self.n_dimensions):
218              cell_child_id *= 2
219              if point[i] >= cell.center[i]:
220                  cell_child_id += 1
221                  child.min_bounds[i] = cell.center[i]
222                  child.max_bounds[i] = cell.max_bounds[i]
223              else:
224                  child.min_bounds[i] = cell.min_bounds[i]
225                  child.max_bounds[i] = cell.center[i]
226              child.center[i] = (child.min_bounds[i] + child.max_bounds[i]) / 2.
227              width = child.max_bounds[i] - child.min_bounds[i]
228  
229              child.barycenter[i] = point[i]
230              child.squared_max_width = max(child.squared_max_width, width*width)
231  
232          # Store the point info and the size to account for duplicated points
233          child.point_index = point_index
234          child.cumulative_size = size
235  
236          # Store the child cell in the correct place in children
237          cell.children[cell_child_id] = child.cell_id
238  
239          if DEBUGFLAG:
240              # Assert that the point is in the right range
241              self._check_point_in_cell(point, child)
242          if self.verbose > 10:
243              printf("[QuadTree] inserted point %li in new child %li\n",
244                     point_index, cell_id)
245  
246          return cell_id

248      cdef bint _is_duplicate(self, float32_t[3] point1, float32_t[3] point2) noexcept nogil:
249          """Check if the two given points are equals."""
250          cdef int i
251          cdef bint res = True
252          for i in range(self.n_dimensions):
253              # Use EPSILON to avoid numerical error that would overgrow the tree
254              res &= fabsf(point1[i] - point2[i]) <= EPSILON
255          return res

257      cdef intp_t _select_child(self, float32_t[3] point, Cell* cell) noexcept nogil:
258          """Select the child of cell which contains the given query point."""
259          cdef:
260              int i
261              intp_t selected_child = 0
262  
263          for i in range(self.n_dimensions):
264              # Select the correct child cell to insert the point by comparing
265              # it to the borders of the cells using precomputed center.
266              selected_child *= 2
267              if point[i] >= cell.center[i]:
268                  selected_child += 1
269          return cell.children[selected_child]

300      cdef int _check_point_in_cell(self, float32_t[3] point, Cell* cell
301                                    ) except -1 nogil:
302          """Check that the given point is in the cell boundaries."""
303  
304          if self.verbose >= 50:
305              if self.n_dimensions == 3:
306                  printf("[QuadTree] Checking point (%f, %f, %f) in cell %li "
307                         "([%f/%f, %f/%f, %f/%f], size %li)\n",
308                         point[0], point[1], point[2], cell.cell_id,
309                         cell.min_bounds[0], cell.max_bounds[0], cell.min_bounds[1],
310                         cell.max_bounds[1], cell.min_bounds[2], cell.max_bounds[2],
311                         cell.cumulative_size)
312              else:
313                  printf("[QuadTree] Checking point (%f, %f) in cell %li "
314                         "([%f/%f, %f/%f], size %li)\n",
315                         point[0], point[1], cell.cell_id, cell.min_bounds[0],
316                         cell.max_bounds[0], cell.min_bounds[1],
317                         cell.max_bounds[1], cell.cumulative_size)
318  
319          for i in range(self.n_dimensions):
320              if (cell.min_bounds[i] > point[i] or
321                      cell.max_bounds[i] <= point[i]):
322                  with gil:
323                      msg = "[QuadTree] InsertionError: point out of cell "
324                      msg += "boundary.\nAxis %li: cell [%f, %f]; point %f\n"
325  
326                      msg %= i, cell.min_bounds[i],  cell.max_bounds[i], point[i]
327                      raise ValueError(msg)

329      def _check_coherence(self):
330          """Check the coherence of the cells of the tree. 
331   
332          Check that the info stored in each cell is compatible with the info 
333          stored in descendent and sibling cells. Raise a ValueError if this 
334          fails. 
335          """
336          for cell in self.cells[:self.cell_count]:
337              # Check that the barycenter of inserted point is within the cell
338              # boundaries
339              self._check_point_in_cell(cell.barycenter, &cell)
340  
341              if not cell.is_leaf:
342                  # Compute the number of point in children and compare with
343                  # its cummulative_size.
344                  n_points = 0
345                  for idx in range(self.n_cells_per_cell):
346                      child_id = cell.children[idx]
347                      if child_id != -1:
348                          child = self.cells[child_id]
349                          n_points += child.cumulative_size
350                          assert child.cell_id == child_id, (
351                              "Cell id not correctly initialized.")
352                  if n_points != cell.cumulative_size:
353                      raise ValueError(
354                          "Cell {} is incoherent. Size={} but found {} points "
355                          "in children. ({})"
356                          .format(cell.cell_id, cell.cumulative_size,
357                                  n_points, cell.children))
358  
359          # Make sure that the number of point in the tree correspond to the
360          # cumulative size in root cell.
361          if self.n_points != self.cells[0].cumulative_size:
362              raise ValueError(
363                  "QuadTree is incoherent. Size={} but found {} points "
364                  "in children."
365                  .format(self.n_points, self.cells[0].cumulative_size))

367      cdef long summarize(self, float32_t[3] point, float32_t* results,
368                          float squared_theta=.5, intp_t cell_id=0, long idx=0
369                          ) noexcept nogil:
370          """Summarize the tree compared to a query point. 
371   
372          Input arguments 
373          --------------- 
374          point : array (n_dimensions) 
375               query point to construct the summary. 
376          cell_id : integer, optional (default: 0) 
377              current cell of the tree summarized. This should be set to 0 for 
378              external calls. 
379          idx : integer, optional (default: 0) 
380              current index in the result array. This should be set to 0 for 
381              external calls 
382          squared_theta: float, optional (default: .5) 
383              threshold to decide whether the node is sufficiently far 
384              from the query point to be a good summary. The formula is such that 
385              the node is a summary if 
386                  node_width^2 / dist_node_point^2 < squared_theta. 
387              Note that the argument should be passed as theta^2 to avoid 
388              computing square roots of the distances. 
389   
390          Output arguments 
391          ---------------- 
392          results : array (n_samples * (n_dimensions+2)) 
393              result will contain a summary of the tree information compared to 
394              the query point: 
395              - results[idx:idx+n_dimensions] contains the coordinate-wise 
396                  difference between the query point and the summary cell idx. 
397                  This is useful in t-SNE to compute the negative forces. 
398              - result[idx+n_dimensions+1] contains the squared euclidean 
399                  distance to the summary cell idx. 
400              - result[idx+n_dimensions+2] contains the number of point of the 
401                  tree contained in the summary cell idx. 
402   
403          Return 
404          ------ 
405          idx : integer 
406              number of elements in the results array. 
407          """
408          cdef:
409              int i, idx_d = idx + self.n_dimensions
410              bint duplicate = True
411              Cell* cell = &self.cells[cell_id]
412  
413          results[idx_d] = 0.
414          for i in range(self.n_dimensions):
415              results[idx + i] = point[i] - cell.barycenter[i]
416              results[idx_d] += results[idx + i] * results[idx + i]
417              duplicate &= fabsf(results[idx + i]) <= EPSILON
418  
419          # Do not compute self interactions
420          if duplicate and cell.is_leaf:
421              return idx
422  
423          # Check whether we can use this node as a summary
424          # It's a summary node if the angular size as measured from the point
425          # is relatively small (w.r.t. theta) or if it is a leaf node.
426          # If it can be summarized, we use the cell center of mass
427          # Otherwise, we go a higher level of resolution and into the leaves.
428          if cell.is_leaf or (
429                  (cell.squared_max_width / results[idx_d]) < squared_theta):
430              results[idx_d + 1] = <float32_t> cell.cumulative_size
431              return idx + self.n_dimensions + 2
432  
433          else:
434              # Recursively compute the summary in nodes
435              for c in range(self.n_cells_per_cell):
436                  child_id = cell.children[c]
437                  if child_id != -1:
438                      idx = self.summarize(point, results, squared_theta,
439                                           child_id, idx)
440  
441          return idx

459      cdef int _get_cell(self, float32_t[3] point, intp_t cell_id=0
460                         ) except -1 nogil:
461          """guts of get_cell. 
462   
463          Return the id of the cell containing the query point or raise ValueError 
464          if the point is not in the tree"""
465          cdef:
466              intp_t selected_child
467              Cell* cell = &self.cells[cell_id]
468  
469          if cell.is_leaf:
470              if self._is_duplicate(cell.barycenter, point):
471                  if self.verbose > 99:
472                      printf("[QuadTree] Found point in cell: %li\n",
473                             cell.cell_id)
474                  return cell_id
475              with gil:
476                  raise ValueError("Query point not in the Tree.")
477  
478          selected_child = self._select_child(point, cell)
479          if selected_child > 0:
480              if self.verbose > 99:
481                  printf("[QuadTree] Selected_child: %li\n", selected_child)
482              return self._get_cell(point, selected_child)
483          with gil:
484              raise ValueError("Query point not in the Tree.")

563      cdef int _resize(self, intp_t capacity) except -1 nogil:
564          """Resize all inner arrays to `capacity`, if `capacity` == -1, then 
565             double the size of the inner arrays. 
566   
567          Returns -1 in case of failure to allocate memory (and raise MemoryError) 
568          or 0 otherwise. 
569          """
570          if self._resize_c(capacity) != 0:
571              # Acquire gil only if we need to raise
572              with gil:
573                  raise MemoryError()

575      cdef int _resize_c(self, intp_t capacity=SIZE_MAX) except -1 nogil:
576          """Guts of _resize 
577   
578          Returns -1 in case of failure to allocate memory (and raise MemoryError) 
579          or 0 otherwise. 
580          """
581          if capacity == self.capacity and self.cells != NULL:
582              return 0
583  
584          if <size_t> capacity == SIZE_MAX:
585              if self.capacity == 0:
586                  capacity = 9  # default initial value to min
587              else:
588                  capacity = 2 * self.capacity
589  
590          safe_realloc(&self.cells, capacity)
591  
592          # if capacity smaller than cell_count, adjust the counter
593          if capacity < self.cell_count:
594              self.cell_count = capacity
595  
596          self.capacity = capacity
597          return 0
