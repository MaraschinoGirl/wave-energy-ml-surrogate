18   # Notes:
19   # - IN views are read-only, OUT views are write-only
20   # - In a lot of functions here, we pass feature_idx and the whole 2d
21   #   histograms arrays instead of just histograms[feature_idx]. This is because
22   #   Cython generated C code will have strange Python interactions (likely
23   #   related to the GIL release and the custom histogram dtype) when using 1d
24   #   histogram arrays that come from 2d arrays.
25   # - The for loops are un-wrapped, for example:
26   #
27   #   for i in range(n):
28   #       array[i] = i
29   #
30   #   will become
31   #
32   #   for i in range(n // 4):
33   #       array[i] = i
34   #       array[i + 1] = i + 1
35   #       array[i + 2] = i + 2
36   #       array[i + 3] = i + 3
37   #
38   #   This is to hint gcc that it can auto-vectorize these 4 operations and
39   #   perform them all at once.

42   @cython.final
43   cdef class HistogramBuilder:
44       """A Histogram builder... used to build histograms. 
45    
46       A histogram is an array with n_bins entries of type HISTOGRAM_DTYPE. Each 
47       feature has its own histogram. A histogram contains the sum of gradients 
48       and hessians of all the samples belonging to each bin. 
49    
50       There are different ways to build a histogram: 
51       - by subtraction: hist(child) = hist(parent) - hist(sibling) 
52       - from scratch. In this case we have routines that update the hessians 
53         or not (not useful when hessians are constant for some losses e.g. 
54         least squares). Also, there's a special case for the root which 
55         contains all the samples, leading to some possible optimizations. 
56         Overall all the implementations look the same, and are optimized for 
57         cache hit. 
58    
59       Parameters 
60       ---------- 
61       X_binned : ndarray of int, shape (n_samples, n_features) 
62           The binned input samples. Must be Fortran-aligned. 
63       n_bins : int 
64           The total number of bins, including the bin for missing values. Used 
65           to define the shape of the histograms. 
66       gradients : ndarray, shape (n_samples,) 
67           The gradients of each training sample. Those are the gradients of the 
68           loss w.r.t the predictions, evaluated at iteration i - 1. 
69       hessians : ndarray, shape (n_samples,) 
70           The hessians of each training sample. Those are the hessians of the 
71           loss w.r.t the predictions, evaluated at iteration i - 1. 
72       hessians_are_constant : bool 
73           Whether hessians are constant. 
74       """
75       cdef public:
76           const X_BINNED_DTYPE_C [::1, :] X_binned
77           unsigned int n_features
78           unsigned int n_bins
79           G_H_DTYPE_C [::1] gradients
80           G_H_DTYPE_C [::1] hessians
81           G_H_DTYPE_C [::1] ordered_gradients
82           G_H_DTYPE_C [::1] ordered_hessians
83           uint8_t hessians_are_constant
84           int n_threads
85   
86       def __init__(self, const X_BINNED_DTYPE_C [::1, :] X_binned,
87                    unsigned int n_bins, G_H_DTYPE_C [::1] gradients,
88                    G_H_DTYPE_C [::1] hessians,
89                    uint8_t hessians_are_constant,
90                    int n_threads):
91   
92           self.X_binned = X_binned
93           self.n_features = X_binned.shape[1]
94           # Note: all histograms will have <n_bins> bins, but some of the
95           # bins may be unused if a feature has a small number of unique values.
96           self.n_bins = n_bins
97           self.gradients = gradients
98           self.hessians = hessians
99           # for root node, gradients and hessians are already ordered
100          self.ordered_gradients = gradients.copy()
101          self.ordered_hessians = hessians.copy()
102          self.hessians_are_constant = hessians_are_constant
103          self.n_threads = n_threads
104  
105      def compute_histograms_brute(
106          HistogramBuilder self,
107          const unsigned int [::1] sample_indices,       # IN
108          const unsigned int [:] allowed_features=None,  # IN
109      ):
110          """Compute the histograms of the node by scanning through all the data. 
111   
112          For a given feature, the complexity is O(n_samples) 
113   
114          Parameters 
115          ---------- 
116          sample_indices : array of int, shape (n_samples_at_node,) 
117              The indices of the samples at the node to split. 
118   
119          allowed_features : None or ndarray, dtype=np.uint32 
120              Indices of the features that are allowed by interaction constraints to be 
121              split. 
122   
123          Returns 
124          ------- 
125          histograms : ndarray of HISTOGRAM_DTYPE, shape (n_features, n_bins) 
126              The computed histograms of the current node. 
127          """
128          cdef:
129              int n_samples
130              int feature_idx
131              int f_idx
132              int i
133              # need local views to avoid python interactions
134              uint8_t hessians_are_constant = self.hessians_are_constant
135              int n_allowed_features = self.n_features
136              G_H_DTYPE_C [::1] ordered_gradients = self.ordered_gradients
137              G_H_DTYPE_C [::1] gradients = self.gradients
138              G_H_DTYPE_C [::1] ordered_hessians = self.ordered_hessians
139              G_H_DTYPE_C [::1] hessians = self.hessians
140              # Histograms will be initialized to zero later within a prange
141              hist_struct [:, ::1] histograms = np.empty(
142                  shape=(self.n_features, self.n_bins),
143                  dtype=HISTOGRAM_DTYPE
144              )
145              bint has_interaction_cst = allowed_features is not None
146              int n_threads = self.n_threads
147  
148          if has_interaction_cst:
149              n_allowed_features = allowed_features.shape[0]
150  
151          with nogil:
152              n_samples = sample_indices.shape[0]
153  
154              # Populate ordered_gradients and ordered_hessians. (Already done
155              # for root) Ordering the gradients and hessians helps to improve
156              # cache hit.
157              if sample_indices.shape[0] != gradients.shape[0]:
158                  if hessians_are_constant:
159                      for i in prange(n_samples, schedule='static',
160                                      num_threads=n_threads):
161                          ordered_gradients[i] = gradients[sample_indices[i]]
162                  else:
163                      for i in prange(n_samples, schedule='static',
164                                      num_threads=n_threads):
165                          ordered_gradients[i] = gradients[sample_indices[i]]
166                          ordered_hessians[i] = hessians[sample_indices[i]]
167  
168              # Compute histogram of each feature
169              for f_idx in prange(
170                  n_allowed_features, schedule='static', num_threads=n_threads
171              ):
172                  if has_interaction_cst:
173                      feature_idx = allowed_features[f_idx]
174                  else:
175                      feature_idx = f_idx
176  
177                  self._compute_histogram_brute_single_feature(
178                      feature_idx, sample_indices, histograms
179                  )
180  
181          return histograms
182  
183      cdef void _compute_histogram_brute_single_feature(
184              HistogramBuilder self,
185              const int feature_idx,
186              const unsigned int [::1] sample_indices,  # IN
187              hist_struct [:, ::1] histograms) noexcept nogil:  # OUT
188          """Compute the histogram for a given feature."""
189  
190          cdef:
191              unsigned int n_samples = sample_indices.shape[0]
192              const X_BINNED_DTYPE_C [::1] X_binned = \
193                  self.X_binned[:, feature_idx]
194              unsigned int root_node = X_binned.shape[0] == n_samples
195              G_H_DTYPE_C [::1] ordered_gradients = \
196                  self.ordered_gradients[:n_samples]
197              G_H_DTYPE_C [::1] ordered_hessians = \
198                  self.ordered_hessians[:n_samples]
199              uint8_t hessians_are_constant = \
200                  self.hessians_are_constant
201  
202          # Set histograms to zero.
203          memset(&histograms[feature_idx, 0], 0, self.n_bins * sizeof(hist_struct))
204  
205          if root_node:
206              if hessians_are_constant:
207                  _build_histogram_root_no_hessian(feature_idx, X_binned,
208                                                   ordered_gradients,
209                                                   histograms)
210              else:
211                  _build_histogram_root(feature_idx, X_binned,
212                                        ordered_gradients, ordered_hessians,
213                                        histograms)
214          else:
215              if hessians_are_constant:
216                  _build_histogram_no_hessian(feature_idx,
217                                              sample_indices, X_binned,
218                                              ordered_gradients, histograms)
219              else:
220                  _build_histogram(feature_idx, sample_indices,
221                                   X_binned, ordered_gradients,
222                                   ordered_hessians, histograms)

224      def compute_histograms_subtraction(
225          HistogramBuilder self,
226          hist_struct [:, ::1] parent_histograms,        # IN and OUT
227          hist_struct [:, ::1] sibling_histograms,       # IN
228          const unsigned int [:] allowed_features=None,  # IN
229      ):
230          """Compute the histograms of the node using the subtraction trick. 
231   
232          hist(parent) = hist(left_child) + hist(right_child) 
233   
234          For a given feature, the complexity is O(n_bins). This is much more 
235          efficient than compute_histograms_brute, but it's only possible for one 
236          of the siblings. 
237   
238          Parameters 
239          ---------- 
240          parent_histograms : ndarray of HISTOGRAM_DTYPE, \ 
241                  shape (n_features, n_bins) 
242              The histograms of the parent. 
243          sibling_histograms : ndarray of HISTOGRAM_DTYPE, \ 
244                  shape (n_features, n_bins) 
245              The histograms of the sibling. 
246          allowed_features : None or ndarray, dtype=np.uint32 
247              Indices of the features that are allowed by interaction constraints to be 
248              split. 
249   
250          Returns 
251          ------- 
252          histograms : ndarray of HISTOGRAM_DTYPE, shape(n_features, n_bins) 
253              The computed histograms of the current node. 
254              We repurpose parent_histograms for this and don't need to allocate new 
255              memory. 
256          """
257  
258          cdef:
259              int feature_idx
260              int f_idx
261              int n_allowed_features = self.n_features
262              bint has_interaction_cst = allowed_features is not None
263              int n_threads = self.n_threads
264  
265          if has_interaction_cst:
266              n_allowed_features = allowed_features.shape[0]
267  
268          # Compute histogram of each feature
269          for f_idx in prange(n_allowed_features, schedule='static', nogil=True,
270                              num_threads=n_threads):
271              if has_interaction_cst:
272                  feature_idx = allowed_features[f_idx]
273              else:
274                  feature_idx = f_idx
275  
276              _subtract_histograms(
277                  feature_idx,
278                  self.n_bins,
279                  parent_histograms,
280                  sibling_histograms,
281              )
282          return parent_histograms

285  cpdef void _build_histogram_naive(
286          const int feature_idx,
287          unsigned int [:] sample_indices,  # IN
288          X_BINNED_DTYPE_C [:] binned_feature,  # IN
289          G_H_DTYPE_C [:] ordered_gradients,  # IN
290          G_H_DTYPE_C [:] ordered_hessians,  # IN
291          hist_struct [:, :] out) noexcept nogil:  # OUT
292      """Build histogram in a naive way, without optimizing for cache hit. 
293   
294      Used in tests to compare with the optimized version."""
295      cdef:
296          unsigned int i
297          unsigned int n_samples = sample_indices.shape[0]
298          unsigned int sample_idx
299          unsigned int bin_idx
300  
301      for i in range(n_samples):
302          sample_idx = sample_indices[i]
303          bin_idx = binned_feature[sample_idx]
304          out[feature_idx, bin_idx].sum_gradients += ordered_gradients[i]
305          out[feature_idx, bin_idx].sum_hessians += ordered_hessians[i]
306          out[feature_idx, bin_idx].count += 1

309  cpdef void _subtract_histograms(
310          const int feature_idx,
311          unsigned int n_bins,
312          hist_struct [:, ::1] hist_a,  # IN and OUT
313          hist_struct [:, ::1] hist_b,  # IN
314  ) noexcept nogil:  # OUT
315      """compute hist_a = hist_a - hist_b"""
316      # Note that subtraction of large sums of floating point numbers, as we have here,
317      # can exhibit catastrophic cancallation. This is in particular true for gradients
318      # as they can be positive and negative, while hessians are non-negative.
319      # Remember that gradients and hessians are originally computed in
320      # G_H_DTYPE_C = float32 precision. Therefore, if sum_gradients and sum_hessians are
321      # float64, we don't loose precision. But if we also used float32 for summation, we
322      # would need to take care of floating point errors.
323      #
324      # Note that we could protect for negative hessians by setting:
325      #     sum_hessians = max(0, sum_hessians)
326      # But as we use float64 for summing float32, that's veeeery unlikely.
327      cdef:
328          unsigned int i = 0
329      for i in range(n_bins):
330          hist_a[feature_idx, i].sum_gradients -= hist_b[feature_idx, i].sum_gradients
331          hist_a[feature_idx, i].sum_hessians -= hist_b[feature_idx, i].sum_hessians
332          hist_a[feature_idx, i].count -= hist_b[feature_idx, i].count

335  cpdef void _build_histogram(
336          const int feature_idx,
337          const unsigned int [::1] sample_indices,  # IN
338          const X_BINNED_DTYPE_C [::1] binned_feature,  # IN
339          const G_H_DTYPE_C [::1] ordered_gradients,  # IN
340          const G_H_DTYPE_C [::1] ordered_hessians,  # IN
341          hist_struct [:, ::1] out) noexcept nogil:  # OUT
342      """Return histogram for a given feature."""
343      cdef:
344          unsigned int i = 0
345          unsigned int n_node_samples = sample_indices.shape[0]
346          unsigned int unrolled_upper = (n_node_samples // 4) * 4
347  
348          unsigned int bin_0
349          unsigned int bin_1
350          unsigned int bin_2
351          unsigned int bin_3
352          unsigned int bin_idx
353  
354      for i in range(0, unrolled_upper, 4):
355          bin_0 = binned_feature[sample_indices[i]]
356          bin_1 = binned_feature[sample_indices[i + 1]]
357          bin_2 = binned_feature[sample_indices[i + 2]]
358          bin_3 = binned_feature[sample_indices[i + 3]]
359  
360          out[feature_idx, bin_0].sum_gradients += ordered_gradients[i]
361          out[feature_idx, bin_1].sum_gradients += ordered_gradients[i + 1]
362          out[feature_idx, bin_2].sum_gradients += ordered_gradients[i + 2]
363          out[feature_idx, bin_3].sum_gradients += ordered_gradients[i + 3]
364  
365          out[feature_idx, bin_0].sum_hessians += ordered_hessians[i]
366          out[feature_idx, bin_1].sum_hessians += ordered_hessians[i + 1]
367          out[feature_idx, bin_2].sum_hessians += ordered_hessians[i + 2]
368          out[feature_idx, bin_3].sum_hessians += ordered_hessians[i + 3]
369  
370          out[feature_idx, bin_0].count += 1
371          out[feature_idx, bin_1].count += 1
372          out[feature_idx, bin_2].count += 1
373          out[feature_idx, bin_3].count += 1
374  
375      for i in range(unrolled_upper, n_node_samples):
376          bin_idx = binned_feature[sample_indices[i]]
377          out[feature_idx, bin_idx].sum_gradients += ordered_gradients[i]
378          out[feature_idx, bin_idx].sum_hessians += ordered_hessians[i]
379          out[feature_idx, bin_idx].count += 1
