18   m1.seek(100)
19   m1.write_byte(b"?")
20   m1.seek(-1, 1)
21   assert m1.read_byte() == b"?"
22   
23   ## A reopened named mapping should have exact same size as original mapping
24   m2 = mmapfile.mmapfile(Name=mapping_name, File=None, MaximumSize=fsize * 2)
25   assert m2.size() == m1.size()
26   m1.seek(0, 0)
27   m1.write(fsize * b"s")
28   assert m2.read(fsize) == fsize * b"s"

52   ## Test a file with size larger than 32 bits
53   ## need 10 GB free on drive where your temp folder lives
54   fname_large = tempfile.mktemp()
55   mapping_name = "Pywin32_large_mmap"
56   offsetdata = b"This is start of offset"
57   
58   ## Deliberately use odd numbers to test rounding logic
59   fsize = (1024 * 1024 * 1024 * 10) + 333
60   offset = (1024 * 1024 * 32) + 42
61   view_size = (1024 * 1024 * 16) + 111
62   
63   ## round mapping size and view size up to multiple of system page size
64   if fsize % page_size:
65       fsize += page_size - (fsize % page_size)
66   if view_size % page_size:
67       view_size += page_size - (view_size % page_size)
68   ## round offset down to multiple of allocation granularity
69   offset -= offset % alloc_size
70   
71   m1 = None
72   m2 = None
73   try:
74       try:
75           m1 = mmapfile.mmapfile(fname_large, mapping_name, fsize, 0, offset * 2)
76       except mmapfile.error as exc:
77           # if we don't have enough disk-space, that's OK.
78           if exc.winerror != winerror.ERROR_DISK_FULL:
79               raise
80           print("skipping large file test - need", fsize, "available bytes.")
81       else:
82           m1.seek(offset)
83           m1.write(offsetdata)
84   
85           ## When reopening an existing mapping without passing a file handle, you have
86           ##  to specify a positive size even though it's ignored
87           m2 = mmapfile.mmapfile(
88               File=None,
89               Name=mapping_name,
90               MaximumSize=1,
91               FileOffset=offset,
92               NumberOfBytesToMap=view_size,
93           )
94           assert m2.read(len(offsetdata)) == offsetdata
95   finally:
96       if m1 is not None:
97           m1.close()
98       if m2 is not None:
99           m2.close()
100      if os.path.exists(fname_large):
101          os.remove(fname_large)
