2        pygments.lexers.sql 
3        ~~~~~~~~~~~~~~~~~~~ 
4     
5        Lexers for various SQL dialects and related interactive sessions. 
6     
7        Postgres specific lexers: 
8     
9        `PostgresLexer` 
10           A SQL lexer for the PostgreSQL dialect. Differences w.r.t. the SQL 
11           lexer are: 
12    
13           - keywords and data types list parsed from the PG docs (run the 
14             `_postgres_builtins` module to update them); 
15           - Content of $-strings parsed using a specific lexer, e.g. the content 
16             of a PL/Python function is parsed using the Python lexer; 
17           - parse PG specific constructs: E-strings, $-strings, U&-strings, 
18             different operators and punctuation. 
19    
20       `PlPgsqlLexer` 
21           A lexer for the PL/pgSQL language. Adds a few specific construct on 
22           top of the PG SQL lexer (such as <<label>>). 
23    
24       `PostgresConsoleLexer` 
25           A lexer to highlight an interactive psql session: 
26    
27           - identifies the prompt and does its best to detect the end of command 
28             in multiline statement where not all the lines are prefixed by a 
29             prompt, telling them apart from the output; 
30           - highlights errors in the output and notification levels; 
31           - handles psql backslash commands. 
32    
33       `PostgresExplainLexer` 
34           A lexer to highlight Postgres execution plan. 
35    
36       The ``tests/examplefiles`` contains a few test files with data to be 
37       parsed by these lexers. 
38

40       :license: BSD, see LICENSE for details. 
41   """

89   def language_callback(lexer, match):
90       """Parse the content of a $-string using a lexer 
91    
92       The lexer is chosen looking for a nearby LANGUAGE or assumed as 
93       plpgsql if inside a DO statement and no LANGUAGE has been found. 
94       """
95       lx = None
96       m = language_re.match(lexer.text[match.end():match.end()+100])
97       if m is not None:
98           lx = lexer._get_lexer(m.group(1))
99       else:
100          m = list(language_re.finditer(
101              lexer.text[max(0, match.start()-100):match.start()]))
102          if m:
103              lx = lexer._get_lexer(m[-1].group(1))
104          else:
105              m = list(do_re.finditer(
106                  lexer.text[max(0, match.start()-25):match.start()]))
107              if m:
108                  lx = lexer._get_lexer('plpgsql')
109  
110      # 1 = $, 2 = delimiter, 3 = $
111      yield (match.start(1), String, match.group(1))
112      yield (match.start(2), String.Delimiter, match.group(2))
113      yield (match.start(3), String, match.group(3))
114      # 4 = string contents
115      if lx:
116          yield from lx.get_tokens_unprocessed(match.group(4))
117      else:
118          yield (match.start(4), String, match.group(4))
119      # 5 = $, 6 = delimiter, 7 = $
120      yield (match.start(5), String, match.group(5))
121      yield (match.start(6), String.Delimiter, match.group(6))
122      yield (match.start(7), String, match.group(7))

125  class PostgresBase:
126      """Base class for Postgres-related lexers. 
127   
128      This is implemented as a mixin to avoid the Lexer metaclass kicking in. 
129      this way the different lexer don't have a common Lexer ancestor. If they 
130      had, _tokens could be created on this ancestor and not updated for the 
131      other classes, resulting e.g. in PL/pgSQL parsed as SQL. This shortcoming 
132      seem to suggest that regexp lexers are not really subclassable. 
133      """
134      def get_tokens_unprocessed(self, text, *args):
135          # Have a copy of the entire text to be used by `language_callback`.
136          self.text = text
137          yield from super().get_tokens_unprocessed(text, *args)
138  
139      def _get_lexer(self, lang):
140          if lang.lower() == 'sql':
141              return get_lexer_by_name('postgresql', **self.options)
142  
143          tries = [lang]
144          if lang.startswith('pl'):
145              tries.append(lang[2:])
146          if lang.endswith('u'):
147              tries.append(lang[:-1])
148          if lang.startswith('pl') and lang.endswith('u'):
149              tries.append(lang[2:-1])
150  
151          for lx in tries:
152              try:
153                  return get_lexer_by_name(lx, **self.options)
154              except ClassNotFound:
155                  pass
156          else:
157              # TODO: better logging
158              # print >>sys.stderr, "language not found:", lang
159              return None

218  class PlPgsqlLexer(PostgresBase, RegexLexer):
219      """ 
220      Handle the extra syntax in Pl/pgSQL language. 
221      """
222      name = 'PL/pgSQL'
223      aliases = ['plpgsql']
224      mimetypes = ['text/x-plpgsql']
225      url = 'https://www.postgresql.org/docs/current/plpgsql.html'
226      version_added = '1.5'
227  
228      flags = re.IGNORECASE
229      # FIXME: use inheritance
230      tokens = {name: state[:] for (name, state) in PostgresLexer.tokens.items()}
231  
232      # extend the keywords list
233      for i, pattern in enumerate(tokens['root']):
234          if pattern[1] == Keyword:
235              tokens['root'][i] = (
236                  words(KEYWORDS + PLPGSQL_KEYWORDS, suffix=r'\b'),
237                  Keyword)
238              del i
239              break
240      else:
241          assert 0, "SQL keywords not found"
242  
243      # Add specific PL/pgSQL rules (before the SQL ones)
244      tokens['root'][:0] = [
245          (r'\%[a-z]\w*\b', Name.Builtin),     # actually, a datatype
246          (r':=', Operator),
247          (r'\<\<[a-z]\w*\>\>', Name.Label),
248          (r'\#[a-z]\w*\b', Keyword.Pseudo),   # #variable_conflict
249      ]

289  class lookahead:
290      """Wrap an iterator and allow pushing back an item."""
291      def __init__(self, x):
292          self.iter = iter(x)
293          self._nextitem = None
294  
295      def __iter__(self):
296          return self
297  
298      def send(self, i):
299          self._nextitem = i
300          return i
301  
302      def __next__(self):
303          if self._nextitem is not None:
304              ni = self._nextitem
305              self._nextitem = None
306              return ni
307          return next(self.iter)
308      next = __next__

311  class PostgresConsoleLexer(Lexer):
312      """ 
313      Lexer for psql sessions. 
314      """
315  
316      name = 'PostgreSQL console (psql)'
317      aliases = ['psql', 'postgresql-console', 'postgres-console']
318      mimetypes = ['text/x-postgresql-psql']
319      url = 'https://www.postgresql.org'
320      version_added = '1.5'
321      _example = "psql/psql_session.txt"
322  
323      def get_tokens_unprocessed(self, data):
324          sql = PsqlRegexLexer(**self.options)
325  
326          lines = lookahead(line_re.findall(data))
327  
328          # prompt-output cycle
329          while 1:
330  
331              # consume the lines of the command: start with an optional prompt
332              # and continue until the end of command is detected
333              curcode = ''
334              insertions = []
335              for line in lines:
336                  # Identify a shell prompt in case of psql commandline example
337                  if line.startswith('$') and not curcode:
338                      lexer = get_lexer_by_name('console', **self.options)
339                      yield from lexer.get_tokens_unprocessed(line)
340                      break
341  
342                  # Identify a psql prompt
343                  mprompt = re_prompt.match(line)
344                  if mprompt is not None:
345                      insertions.append((len(curcode),
346                                         [(0, Generic.Prompt, mprompt.group())]))
347                      curcode += line[len(mprompt.group()):]
348                  else:
349                      curcode += line
350  
351                  # Check if this is the end of the command
352                  # TODO: better handle multiline comments at the end with
353                  # a lexer with an external state?
354                  if re_psql_command.match(curcode) \
355                     or re_end_command.search(curcode):
356                      break
357  
358              # Emit the combined stream of command and prompt(s)
359              yield from do_insertions(insertions,
360                                       sql.get_tokens_unprocessed(curcode))
361  
362              # Emit the output lines
363              out_token = Generic.Output
364              for line in lines:
365                  mprompt = re_prompt.match(line)
366                  if mprompt is not None:
367                      # push the line back to have it processed by the prompt
368                      lines.send(line)
369                      break
370  
371                  mmsg = re_message.match(line)
372                  if mmsg is not None:
373                      if mmsg.group(1).startswith("ERROR") \
374                         or mmsg.group(1).startswith("FATAL"):
375                          out_token = Generic.Error
376                      yield (mmsg.start(1), Generic.Strong, mmsg.group(1))
377                      yield (mmsg.start(2), out_token, mmsg.group(2))
378                  else:
379                      yield (0, out_token, line)
380              else:
381                  return

573      not recognize any special syntax except ANSI SQL.

584      tokens = {
585          'root': [
586              (r'\s+', Whitespace),
587              (r'--.*\n?', Comment.Single),
588              (r'/\*', Comment.Multiline, 'multiline-comments'),
589              (words((

704  class TransactSqlLexer(RegexLexer):
705      """ 
706      Transact-SQL (T-SQL) is Microsoft's and Sybase's proprietary extension to 
707      SQL. 
708   
709      The list of keywords includes ODBC and keywords reserved for future use.. 
710      """
711  
712      name = 'Transact-SQL'
713      aliases = ['tsql', 't-sql']
714      filenames = ['*.sql']
715      mimetypes = ['text/x-tsql']
716      url = 'https://www.tsql.info'
717      version_added = ''
718  
719      flags = re.IGNORECASE
720  
721      tokens = {
722          'root': [
723              (r'\s+', Whitespace),
724              (r'--.*[$|\n]?', Comment.Single),
725              (r'/\*', Comment.Multiline, 'multiline-comments'),
726              (words(_tsql_builtins.OPERATORS), Operator),
727              (words(_tsql_builtins.OPERATOR_WORDS, suffix=r'\b'), Operator.Word),
728              (words(_tsql_builtins.TYPES, suffix=r'\b'), Name.Class),
729              (words(_tsql_builtins.FUNCTIONS, suffix=r'\b'), Name.Function),
730              (r'(goto)(\s+)(\w+\b)', bygroups(Keyword, Whitespace, Name.Label)),
731              (words(_tsql_builtins.KEYWORDS, suffix=r'\b'), Keyword),
732              (r'(\[)([^]]+)(\])', bygroups(Operator, Name, Operator)),
733              (r'0x[0-9a-f]+', Number.Hex),
734              # Float variant 1, for example: 1., 1.e2, 1.2e3
735              (r'[0-9]+\.[0-9]*(e[+-]?[0-9]+)?', Number.Float),
736              # Float variant 2, for example: .1, .1e2
737              (r'\.[0-9]+(e[+-]?[0-9]+)?', Number.Float),
738              # Float variant 3, for example: 123e45
739              (r'[0-9]+e[+-]?[0-9]+', Number.Float),
740              (r'[0-9]+', Number.Integer),
741              (r"'(''|[^'])*'", String.Single),
742              (r'"(""|[^"])*"', String.Symbol),
743              (r'[;(),.]', Punctuation),
744              # Below we use \w even for the first "real" character because
745              # tokens starting with a digit have already been recognized
746              # as Number above.
747              (r'@@\w+', Name.Builtin),
748              (r'@\w+', Name.Variable),
749              (r'(\w+)(:)', bygroups(Name.Label, Punctuation)),
750              (r'#?#?\w+', Name),  # names for temp tables and anything else
751              (r'\?', Name.Variable.Magic),  # parameter for prepared statements
752          ],
753          'multiline-comments': [
754              (r'/\*', Comment.Multiline, 'multiline-comments'),
755              (r'\*/', Comment.Multiline, '#pop'),
756              (r'[^/*]+', Comment.Multiline),
757              (r'[/*]', Comment.Multiline)
758          ]
759      }
760  
761      def analyse_text(text):
762          rating = 0
763          if tsql_declare_re.search(text):
764              # Found T-SQL variable declaration.
765              rating = 1.0
766          else:
767              name_between_backtick_count = len(
768                  name_between_backtick_re.findall(text))
769              name_between_bracket_count = len(
770                  name_between_bracket_re.findall(text))
771              # We need to check if there are any names using
772              # backticks or brackets, as otherwise both are 0
773              # and 0 >= 2 * 0, so we would always assume it's true
774              dialect_name_count = name_between_backtick_count + name_between_bracket_count
775              if dialect_name_count >= 1 and \
776                 name_between_bracket_count >= 2 * name_between_backtick_count:
777                  # Found at least twice as many [name] as `name`.
778                  rating += 0.5
779              elif name_between_bracket_count > name_between_backtick_count:
780                  rating += 0.2
781              elif name_between_bracket_count > 0:
782                  rating += 0.1
783              if tsql_variable_re.search(text) is not None:
784                  rating += 0.1
785              if tsql_go_re.search(text) is not None:
786                  rating += 0.1
787          return rating

1129     def analyse_text(text):

1168                 if line.startswith('SQL error: '):

1174                                      sql.get_tokens_unprocessed(curcode))
