373          """Add the cwd to the sys.path ($PYTHONPATH)"""

376      def init_notebooks(self):
377          """Construct the list of notebooks. 
378   
379          If notebooks are passed on the command-line, 
380          they override (rather than add) notebooks specified in config files. 
381          Glob each notebook to replace notebook patterns with filenames. 
382          """
383  
384          # Specifying notebooks on the command-line overrides (rather than
385          # adds) the notebook list
386          patterns = self.extra_args if self.extra_args else self.notebooks
387  
388          # Use glob to replace all the notebook patterns with filenames.
389          filenames = []
390          for pattern in patterns:
391              # Use glob to find matching filenames.  Allow the user to convert
392              # notebooks without having to type the extension.
393              globbed_files = glob.glob(pattern, recursive=self.recursive_glob)
394              globbed_files.extend(glob.glob(pattern + ".ipynb", recursive=self.recursive_glob))
395              if not globbed_files:
396                  self.log.warning("pattern %r matched no files", pattern)
397  
398              for filename in globbed_files:
399                  if filename not in filenames:
400                      filenames.append(filename)
401          self.notebooks = filenames

403      def init_writer(self):
404          """Initialize the writer (which is stateless)"""
405          self._writer_class_changed({"new": self.writer_class})
406          if self.writer_factory:
407              self.writer = self.writer_factory(parent=self)
408              if hasattr(self.writer, "build_directory") and self.writer.build_directory != "":
409                  self.use_output_suffix = False
410  
411      def init_postprocessor(self):
412          """Initialize the postprocessor (which is stateless)"""
413          self._postprocessor_class_changed({"new": self.postprocessor_class})
414          if self.postprocessor_factory:
415              self.postprocessor = self.postprocessor_factory(parent=self)

417      def start(self):
418          """Run start after initialization process has completed"""
419          super().start()
420          self.convert_notebooks()

422      def _notebook_filename_to_name(self, notebook_filename):
423          """ 
424          Returns the notebook name from the notebook filename by 
425          applying `output_base` pattern and stripping extension 
426          """
427          basename = os.path.basename(notebook_filename)
428          notebook_name = basename[: basename.rfind(".")]
429          notebook_name = self.output_base.format(notebook_name=notebook_name)
430  
431          return notebook_name  # noqa: RET504
432  
433      def init_single_notebook_resources(self, notebook_filename):
434          """Step 1: Initialize resources 
435   
436          This initializes the resources dictionary for a single notebook. 
437   
438          Returns 
439          ------- 
440          dict 
441              resources dictionary for a single notebook that MUST include the following keys: 
442                  - config_dir: the location of the Jupyter config directory 
443                  - unique_key: the notebook name 
444                  - output_files_dir: a directory where output files (not 
445                    including the notebook itself) should be saved 
446          """
447          notebook_name = self._notebook_filename_to_name(notebook_filename)
448          self.log.debug("Notebook name is '%s'", notebook_name)
449  
450          # first initialize the resources we want to use
451          resources = {}
452          resources["config_dir"] = self.config_dir
453          resources["unique_key"] = notebook_name
454  
455          output_files_dir = self.output_files_dir.format(notebook_name=notebook_name)
456  
457          resources["output_files_dir"] = output_files_dir
458  
459          return resources
460  
461      def export_single_notebook(self, notebook_filename, resources, input_buffer=None):
462          """Step 2: Export the notebook 
463   
464          Exports the notebook to a particular format according to the specified 
465          exporter. This function returns the output and (possibly modified) 
466          resources from the exporter. 
467   
468          Parameters 
469          ---------- 
470          notebook_filename : str 
471              name of notebook file. 
472          resources : dict 
473          input_buffer : 
474              readable file-like object returning unicode. 
475              if not None, notebook_filename is ignored 
476   
477          Returns 
478          ------- 
479          output 
480          dict 
481              resources (possibly modified) 
482          """
483          try:
484              if input_buffer is not None:
485                  output, resources = self.exporter.from_file(input_buffer, resources=resources)
486              else:
487                  output, resources = self.exporter.from_filename(
488                      notebook_filename, resources=resources
489                  )
490          except ConversionException:
491              self.log.error("Error while converting '%s'", notebook_filename, exc_info=True)  # noqa: G201
492              self.exit(1)
493  
494          return output, resources
495  
496      def write_single_notebook(self, output, resources):
497          """Step 3: Write the notebook to file 
498   
499          This writes output from the exporter to file using the specified writer. 
500          It returns the results from the writer. 
501   
502          Parameters 
503          ---------- 
504          output : 
505          resources : dict 
506              resources for a single notebook including name, config directory 
507              and directory to save output 
508   
509          Returns 
510          ------- 
511          file 
512              results from the specified writer output of exporter 
513          """
514  
515          if "unique_key" not in resources:
516              msg = "unique_key MUST be specified in the resources, but it is not"
517              raise KeyError(msg)
518  
519          notebook_name = resources["unique_key"]
520          if self.use_output_suffix and self.output_base == "{notebook_name}":
521              notebook_name += resources.get("output_suffix", "")
522  
523          if not self.writer:
524              msg = "No writer object defined!"
525              raise ValueError(msg)
526          return self.writer.write(output, resources, notebook_name=notebook_name)
527  
528      def postprocess_single_notebook(self, write_results):
529          """Step 4: Post-process the written file 
530   
531          Only used if a postprocessor has been specified. After the 
532          converted notebook is written to a file in Step 3, this post-processes 
533          the notebook. 
534          """
535          # Post-process if post processor has been defined.
536          if hasattr(self, "postprocessor") and self.postprocessor:
537              self.postprocessor(write_results)

539      def convert_single_notebook(self, notebook_filename, input_buffer=None):
540          """Convert a single notebook. 
541   
542          Performs the following steps: 
543   
544              1. Initialize notebook resources 
545              2. Export the notebook to a particular format 
546              3. Write the exported notebook to file 
547              4. (Maybe) postprocess the written file 
548   
549          Parameters 
550          ---------- 
551          notebook_filename : str 
552          input_buffer : 
553              If input_buffer is not None, conversion is done and the buffer is 
554              used as source into a file basenamed by the notebook_filename 
555              argument. 
556          """
557          if input_buffer is None:
558              self.log.info("Converting notebook %s to %s", notebook_filename, self.export_format)
559          else:
560              self.log.info("Converting notebook into %s", self.export_format)
561  
562          resources = self.init_single_notebook_resources(notebook_filename)
563          output, resources = self.export_single_notebook(
564              notebook_filename, resources, input_buffer=input_buffer
565          )
566          write_results = self.write_single_notebook(output, resources)
567          self.postprocess_single_notebook(write_results)
568  
569      def convert_notebooks(self):
570          """Convert the notebooks in the self.notebooks traitlet"""
571  
572          # no notebooks to convert!
573          if len(self.notebooks) == 0 and not self.from_stdin:
574              self.print_help()
575              sys.exit(-1)
576  
577          if not self.export_format:
578              msg = (
579                  "Please specify an output format with '--to <format>'."
580                  f"\nThe following formats are available: {get_export_names()}"
581              )
582              raise ValueError(msg)
583  
584          # initialize the exporter
585          cls = get_exporter(self.export_format)
586          self.exporter = cls(config=self.config)
587  
588          # strip duplicate extension from output_base, to avoid Basename.ext.ext
589          if getattr(self.exporter, "file_extension", False):
590              base, ext = os.path.splitext(self.output_base)
591              if ext == self.exporter.file_extension:
592                  self.output_base = base
593  
594          # convert each notebook
595          if not self.from_stdin:
596              for notebook_filename in self.notebooks:
597                  self.convert_single_notebook(notebook_filename)
598          else:
599              input_buffer = unicode_stdin_stream()
600              # default name when conversion from stdin
601              self.convert_single_notebook("notebook.ipynb", input_buffer=input_buffer)
602              input_buffer.close()

604      def document_flag_help(self):
605          """ 
606          Return a string containing descriptions of all the flags. 
607          """
608          flags = "The following flags are defined:\n\n"
609          for flag, (cfg, fhelp) in self.flags.items():
610              flags += f"{flag}\n"
611              flags += indent(fill(fhelp, 80)) + "\n\n"
612              flags += indent(fill("Long Form: " + str(cfg), 80)) + "\n\n"
613          return flags

619          for alias, longname in self.aliases.items():
620              aliases += f"\t**{alias}** ({longname})\n\n"
621          return aliases

623      def document_config_options(self):
624          """ 
625          Provides a much improves version of the configuration documentation by 
626          breaking the configuration options into app, exporter, writer, 
627          preprocessor, postprocessor, and other sections. 
628          """
629          categories = {
630              category: [c for c in self._classes_inc_parents() if category in c.__name__.lower()]
631              for category in ["app", "exporter", "writer", "preprocessor", "postprocessor"]
632          }
633          accounted_for = {c for category in categories.values() for c in category}
634          categories["other"] = [c for c in self._classes_inc_parents() if c not in accounted_for]
635  
636          header = dedent(
637              """ 
638                          {section} Options 
639                          ----------------------- 
640   
641                          """
642          )
643          sections = ""
644          for category in categories:
645              sections += header.format(section=category.title())
646              if category in ["exporter", "preprocessor", "writer"]:
647                  sections += f".. image:: _static/{category}_inheritance.png\n\n"
648              sections += "\n".join(c.class_config_rst_doc() for c in categories[category])
649  
650          return sections.replace(" : ", r" \: ")
