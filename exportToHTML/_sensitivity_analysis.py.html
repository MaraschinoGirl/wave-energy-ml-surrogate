25   def f_ishigami(x: "npt.ArrayLike") -> "npt.NDArray[np.inexact[Any]]":
26       r"""Ishigami function. 
27    
28       .. math:: 
29    
30           Y(\mathbf{x}) = \sin x_1 + 7 \sin^2 x_2 + 0.1 x_3^4 \sin x_1 
31    
32       with :math:`\mathbf{x} \in [-\pi, \pi]^3`. 
33    
34       Parameters 
35       ---------- 
36       x : array_like ([x1, x2, x3], n) 
37    
38       Returns 
39       ------- 
40       f : array_like (n,) 
41           Function evaluation. 
42    
43       References 
44       ---------- 
45       .. [1] Ishigami, T. and T. Homma. "An importance quantification technique 
46          in uncertainty analysis for computer models." IEEE, 
47          :doi:`10.1109/ISUMA.1990.151285`, 1990. 
48       """
49       x = np.atleast_2d(x)
50       f_eval = (
51           np.sin(x[0])
52           + 7 * np.sin(x[1])**2
53           + 0.1 * (x[2]**4) * np.sin(x[0])
54       )
55       return f_eval

58   def sample_A_B(
59       n,
60       dists,
61       rng=None
62   ):
63       """Sample two matrices A and B. 
64    
65       Uses a Sobol' sequence with 2`d` columns to have 2 uncorrelated matrices. 
66       This is more efficient than using 2 random draw of Sobol'. 
67       See sec. 5 from [1]_. 
68    
69       Output shape is (d, n). 
70    
71       References 
72       ---------- 
73       .. [1] Saltelli, A., P. Annoni, I. Azzini, F. Campolongo, M. Ratto, and 
74          S. Tarantola. "Variance based sensitivity analysis of model 
75          output. Design and estimator for the total sensitivity index." 
76          Computer Physics Communications, 181(2):259-270, 
77          :doi:`10.1016/j.cpc.2009.09.018`, 2010. 
78       """
79       d = len(dists)
80       A_B = qmc.Sobol(d=2*d, seed=rng, bits=64).random(n).T
81       A_B = A_B.reshape(2, d, -1)
82       try:
83           for d_, dist in enumerate(dists):
84               A_B[:, d_] = dist.ppf(A_B[:, d_])
85       except AttributeError as exc:
86           message = "Each distribution in `dists` must have method `ppf`."
87           raise ValueError(message) from exc
88       return A_B

91   def sample_AB(A: np.ndarray, B: np.ndarray) -> np.ndarray:
92       """AB matrix. 
93    
94       AB: rows of B into A. Shape (d, d, n). 
95       - Copy A into d "pages" 
96       - In the first page, replace 1st rows of A with 1st row of B. 
97       ... 
98       - In the dth page, replace dth row of A with dth row of B. 
99       - return the stack of pages 
100      """
101      d, n = A.shape
102      AB = np.tile(A, (d, 1, 1))
103      i = np.arange(d)
104      AB[i, i] = B[i]
105      return AB

108  def saltelli_2010(
109      f_A: np.ndarray, f_B: np.ndarray, f_AB: np.ndarray
110  ) -> tuple[np.ndarray, np.ndarray]:
111      r"""Saltelli2010 formulation. 
112   
113      .. math:: 
114   
115          S_i = \frac{1}{N} \sum_{j=1}^N 
116          f(\mathbf{B})_j (f(\mathbf{AB}^{(i)})_j - f(\mathbf{A})_j) 
117   
118      .. math:: 
119   
120          S_{T_i} = \frac{1}{N} \sum_{j=1}^N 
121          (f(\mathbf{A})_j - f(\mathbf{AB}^{(i)})_j)^2 
122   
123      Parameters 
124      ---------- 
125      f_A, f_B : array_like (s, n) 
126          Function values at A and B, respectively 
127      f_AB : array_like (d, s, n) 
128          Function values at each of the AB pages 
129   
130      Returns 
131      ------- 
132      s, st : array_like (s, d) 
133          First order and total order Sobol' indices. 
134   
135      References 
136      ---------- 
137      .. [1] Saltelli, A., P. Annoni, I. Azzini, F. Campolongo, M. Ratto, and 
138         S. Tarantola. "Variance based sensitivity analysis of model 
139         output. Design and estimator for the total sensitivity index." 
140         Computer Physics Communications, 181(2):259-270, 
141         :doi:`10.1016/j.cpc.2009.09.018`, 2010. 
142      """
143      # Empirical variance calculated using output from A and B which are
144      # independent. Output of AB is not independent and cannot be used
145      var = np.var([f_A, f_B], axis=(0, -1))
146  
147      # We divide by the variance to have a ratio of variance
148      # this leads to eq. 2
149      s = np.mean(f_B * (f_AB - f_A), axis=-1) / var  # Table 2 (b)
150      st = 0.5 * np.mean((f_A - f_AB) ** 2, axis=-1) / var  # Table 2 (f)
151  
152      return s.T, st.T

161  @dataclass
162  class SobolResult:
163      first_order: np.ndarray
164      total_order: np.ndarray
165      _indices_method: Callable
166      _f_A: np.ndarray
167      _f_B: np.ndarray
168      _f_AB: np.ndarray
169      _A: np.ndarray | None = None
170      _B: np.ndarray | None = None
171      _AB: np.ndarray | None = None
172      _bootstrap_result: BootstrapResult | None = None
173  
174      def bootstrap(
175          self,
176          confidence_level: "DecimalNumber" = 0.95,
177          n_resamples: "IntNumber" = 999
178      ) -> BootstrapSobolResult:
179          """Bootstrap Sobol' indices to provide confidence intervals. 
180   
181          Parameters 
182          ---------- 
183          confidence_level : float, default: ``0.95`` 
184              The confidence level of the confidence intervals. 
185          n_resamples : int, default: ``999`` 
186              The number of resamples performed to form the bootstrap 
187              distribution of the indices. 
188   
189          Returns 
190          ------- 
191          res : BootstrapSobolResult 
192              Bootstrap result containing the confidence intervals and the 
193              bootstrap distribution of the indices. 
194   
195              An object with attributes: 
196   
197              first_order : BootstrapResult 
198                  Bootstrap result of the first order indices. 
199              total_order : BootstrapResult 
200                  Bootstrap result of the total order indices. 
201              See `BootstrapResult` for more details. 
202   
203          """
204          def statistic(idx):
205              f_A_ = self._f_A[:, idx]
206              f_B_ = self._f_B[:, idx]
207              f_AB_ = self._f_AB[..., idx]
208              return self._indices_method(f_A_, f_B_, f_AB_)
209  
210          n = self._f_A.shape[1]
211  
212          res = bootstrap(
213              [np.arange(n)], statistic=statistic, method="BCa",
214              n_resamples=n_resamples,
215              confidence_level=confidence_level,
216              bootstrap_result=self._bootstrap_result
217          )
218          self._bootstrap_result = res
219  
220          first_order = BootstrapResult(
221              confidence_interval=ConfidenceInterval(
222                  res.confidence_interval.low[0], res.confidence_interval.high[0]
223              ),
224              bootstrap_distribution=res.bootstrap_distribution[0],
225              standard_error=res.standard_error[0],
226          )
227          total_order = BootstrapResult(
228              confidence_interval=ConfidenceInterval(
229                  res.confidence_interval.low[1], res.confidence_interval.high[1]
230              ),
231              bootstrap_distribution=res.bootstrap_distribution[1],
232              standard_error=res.standard_error[1],
233          )
234  
235          return BootstrapSobolResult(
236              first_order=first_order, total_order=total_order
237          )

239  @_transition_to_rng('random_state', replace_doc=False)
240  def sobol_indices(
241      *,
242      func,
243      n,
244      dists=None,
245      method='saltelli_2010',
246      rng=None
247  ):
248      r"""Global sensitivity indices of Sobol'. 
249   
250      Parameters 
251      ---------- 
252      func : callable or dict(str, array_like) 
253          If `func` is a callable, function to compute the Sobol' indices from. 
254          Its signature must be:: 
255   
256              func(x: ArrayLike) -> ArrayLike 
257   
258          with ``x`` of shape ``(d, n)`` and output of shape ``(s, n)`` where: 
259   
260          - ``d`` is the input dimensionality of `func` 
261            (number of input variables), 
262          - ``s`` is the output dimensionality of `func` 
263            (number of output variables), and 
264          - ``n`` is the number of samples (see `n` below). 
265   
266          Function evaluation values must be finite. 
267   
268          If `func` is a dictionary, contains the function evaluations from three 
269          different arrays. Keys must be: ``f_A``, ``f_B`` and ``f_AB``. 
270          ``f_A`` and ``f_B`` should have a shape ``(s, n)`` and ``f_AB`` 
271          should have a shape ``(d, s, n)``. 
272          This is an advanced feature and misuse can lead to wrong analysis. 
273      n : int 
274          Number of samples used to generate the matrices ``A`` and ``B``. 
275          Must be a power of 2. The total number of points at which `func` is 
276          evaluated will be ``n*(d+2)``. 
277      dists : list(distributions), optional 
278          List of each parameter's distribution. The distribution of parameters 
279          depends on the application and should be carefully chosen. 
280          Parameters are assumed to be independently distributed, meaning there 
281          is no constraint nor relationship between their values. 
282   
283          Distributions must be an instance of a class with a ``ppf`` 
284          method. 
285   
286          Must be specified if `func` is a callable, and ignored otherwise. 
287      method : Callable or str, default: 'saltelli_2010' 
288          Method used to compute the first and total Sobol' indices. 
289   
290          If a callable, its signature must be:: 
291   
292              func(f_A: np.ndarray, f_B: np.ndarray, f_AB: np.ndarray) 
293              -> Tuple[np.ndarray, np.ndarray] 
294   
295          with ``f_A, f_B`` of shape ``(s, n)`` and ``f_AB`` of shape 
296          ``(d, s, n)``. 
297          These arrays contain the function evaluations from three different sets 
298          of samples. 
299          The output is a tuple of the first and total indices with 
300          shape ``(s, d)``. 
301          This is an advanced feature and misuse can lead to wrong analysis. 
302      rng : `numpy.random.Generator`, optional 
303          Pseudorandom number generator state. When `rng` is None, a new 
304          `numpy.random.Generator` is created using entropy from the 
305          operating system. Types other than `numpy.random.Generator` are 
306          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
307   
308          .. versionchanged:: 1.15.0 
309   
310              As part of the `SPEC-007 <https://scientific-python.org/specs/spec-0007/>`_ 
311              transition from use of `numpy.random.RandomState` to 
312              `numpy.random.Generator`, this keyword was changed from `random_state` to 
313              `rng`. For an interim period, both keywords will continue to work, although 
314              only one may be specified at a time. After the interim period, function 
315              calls using the `random_state` keyword will emit warnings. Following a 
316              deprecation period, the `random_state` keyword will be removed. 
317   
318      Returns 
319      ------- 
320      res : SobolResult 
321          An object with attributes: 
322   
323          first_order : ndarray of shape (s, d) 
324              First order Sobol' indices. 
325          total_order : ndarray of shape (s, d) 
326              Total order Sobol' indices. 
327   
328          And method: 
329   
330          bootstrap(confidence_level: float, n_resamples: int) 
331          -> BootstrapSobolResult 
332   
333              A method providing confidence intervals on the indices. 
334              See `scipy.stats.bootstrap` for more details. 
335   
336              The bootstrapping is done on both first and total order indices, 
337              and they are available in `BootstrapSobolResult` as attributes 
338              ``first_order`` and ``total_order``. 
339   
340      Notes 
341      ----- 
342      The Sobol' method [1]_, [2]_ is a variance-based Sensitivity Analysis which 
343      obtains the contribution of each parameter to the variance of the 
344      quantities of interest (QoIs; i.e., the outputs of `func`). 
345      Respective contributions can be used to rank the parameters and 
346      also gauge the complexity of the model by computing the 
347      model's effective (or mean) dimension. 
348   
349      .. note:: 
350   
351          Parameters are assumed to be independently distributed. Each 
352          parameter can still follow any distribution. In fact, the distribution 
353          is very important and should match the real distribution of the 
354          parameters. 
355   
356      It uses a functional decomposition of the variance of the function to 
357      explore 
358   
359      .. math:: 
360   
361          \mathbb{V}(Y) = \sum_{i}^{d} \mathbb{V}_i (Y) + \sum_{i<j}^{d} 
362          \mathbb{V}_{ij}(Y) + ... + \mathbb{V}_{1,2,...,d}(Y), 
363   
364      introducing conditional variances: 
365   
366      .. math:: 
367   
368          \mathbb{V}_i(Y) = \mathbb{\mathbb{V}}[\mathbb{E}(Y|x_i)] 
369          \qquad 
370          \mathbb{V}_{ij}(Y) = \mathbb{\mathbb{V}}[\mathbb{E}(Y|x_i x_j)] 
371          - \mathbb{V}_i(Y) - \mathbb{V}_j(Y), 
372   
373      Sobol' indices are expressed as 
374   
375      .. math:: 
376   
377          S_i = \frac{\mathbb{V}_i(Y)}{\mathbb{V}[Y]} 
378          \qquad 
379          S_{ij} =\frac{\mathbb{V}_{ij}(Y)}{\mathbb{V}[Y]}. 
380   
381      :math:`S_{i}` corresponds to the first-order term which apprises the 
382      contribution of the i-th parameter, while :math:`S_{ij}` corresponds to the 
383      second-order term which informs about the contribution of interactions 
384      between the i-th and the j-th parameters. These equations can be 
385      generalized to compute higher order terms; however, they are expensive to 
386      compute and their interpretation is complex. 
387      This is why only first order indices are provided. 
388   
389      Total order indices represent the global contribution of the parameters 
390      to the variance of the QoI and are defined as: 
391   
392      .. math:: 
393   
394          S_{T_i} = S_i + \sum_j S_{ij} + \sum_{j,k} S_{ijk} + ... 
395          = 1 - \frac{\mathbb{V}[\mathbb{E}(Y|x_{\sim i})]}{\mathbb{V}[Y]}. 
396   
397      First order indices sum to at most 1, while total order indices sum to at 
398      least 1. If there are no interactions, then first and total order indices 
399      are equal, and both first and total order indices sum to 1. 
400   
401      .. warning:: 
402   
403          Negative Sobol' values are due to numerical errors. Increasing the 
404          number of points `n` should help. 
405   
406          The number of sample required to have a good analysis increases with 
407          the dimensionality of the problem. e.g. for a 3 dimension problem, 
408          consider at minima ``n >= 2**12``. The more complex the model is, 
409          the more samples will be needed. 
410   
411          Even for a purely additive model, the indices may not sum to 1 due 
412          to numerical noise. 
413   
414      References 
415      ---------- 
416      .. [1] Sobol, I. M.. "Sensitivity analysis for nonlinear mathematical

438      Examples 
439      -------- 
440      The following is an example with the Ishigami function [6]_ 
441   
442      .. math:: 
443   
444          Y(\mathbf{x}) = \sin x_1 + 7 \sin^2 x_2 + 0.1 x_3^4 \sin x_1, 
445   
446      with :math:`\mathbf{x} \in [-\pi, \pi]^3`. This function exhibits strong 
447      non-linearity and non-monotonicity. 
448   
449      Remember, Sobol' indices assumes that samples are independently 
450      distributed. In this case we use a uniform distribution on each marginals. 
451   
452      >>> import numpy as np 
453      >>> from scipy.stats import sobol_indices, uniform 
454      >>> rng = np.random.default_rng() 
455      >>> def f_ishigami(x): 
456      ...     f_eval = ( 
457      ...         np.sin(x[0]) 
458      ...         + 7 * np.sin(x[1])**2 
459      ...         + 0.1 * (x[2]**4) * np.sin(x[0]) 
460      ...     ) 
461      ...     return f_eval 
462      >>> indices = sobol_indices( 
463      ...     func=f_ishigami, n=1024, 
464      ...     dists=[ 
465      ...         uniform(loc=-np.pi, scale=2*np.pi), 
466      ...         uniform(loc=-np.pi, scale=2*np.pi), 
467      ...         uniform(loc=-np.pi, scale=2*np.pi) 
468      ...     ], 
469      ...     rng=rng 
470      ... ) 
471      >>> indices.first_order 
472      array([0.31637954, 0.43781162, 0.00318825]) 
473      >>> indices.total_order 
474      array([0.56122127, 0.44287857, 0.24229595]) 
475   
476      Confidence interval can be obtained using bootstrapping. 
477   
478      >>> boot = indices.bootstrap() 
479   
480      Then, this information can be easily visualized. 
481   
482      >>> import matplotlib.pyplot as plt 
483      >>> fig, axs = plt.subplots(1, 2, figsize=(9, 4)) 
484      >>> _ = axs[0].errorbar( 
485      ...     [1, 2, 3], indices.first_order, fmt='o', 
486      ...     yerr=[ 
487      ...         indices.first_order - boot.first_order.confidence_interval.low, 
488      ...         boot.first_order.confidence_interval.high - indices.first_order 
489      ...     ], 
490      ... ) 
491      >>> axs[0].set_ylabel("First order Sobol' indices") 
492      >>> axs[0].set_xlabel('Input parameters') 
493      >>> axs[0].set_xticks([1, 2, 3]) 
494      >>> _ = axs[1].errorbar( 
495      ...     [1, 2, 3], indices.total_order, fmt='o', 
496      ...     yerr=[ 
497      ...         indices.total_order - boot.total_order.confidence_interval.low, 
498      ...         boot.total_order.confidence_interval.high - indices.total_order 
499      ...     ], 
500      ... ) 
501      >>> axs[1].set_ylabel("Total order Sobol' indices") 
502      >>> axs[1].set_xlabel('Input parameters') 
503      >>> axs[1].set_xticks([1, 2, 3]) 
504      >>> plt.tight_layout() 
505      >>> plt.show() 
506   
507      .. note:: 
508   
509          By default, `scipy.stats.uniform` has support ``[0, 1]``. 
510          Using the parameters ``loc`` and ``scale``, one obtains the uniform 
511          distribution on ``[loc, loc + scale]``. 
512   
513      This result is particularly interesting because the first order index 
514      :math:`S_{x_3} = 0` whereas its total order is :math:`S_{T_{x_3}} = 0.244`. 
515      This means that higher order interactions with :math:`x_3` are responsible 
516      for the difference. Almost 25% of the observed variance 
517      on the QoI is due to the correlations between :math:`x_3` and :math:`x_1`, 
518      although :math:`x_3` by itself has no impact on the QoI. 
519   
520      The following gives a visual explanation of Sobol' indices on this 
521      function. Let's generate 1024 samples in :math:`[-\pi, \pi]^3` and 
522      calculate the value of the output. 
523   
524      >>> from scipy.stats import qmc 
525      >>> n_dim = 3 
526      >>> p_labels = ['$x_1$', '$x_2$', '$x_3$'] 
527      >>> sample = qmc.Sobol(d=n_dim, seed=rng).random(1024) 
528      >>> sample = qmc.scale( 
529      ...     sample=sample, 
530      ...     l_bounds=[-np.pi, -np.pi, -np.pi], 
531      ...     u_bounds=[np.pi, np.pi, np.pi] 
532      ... ) 
533      >>> output = f_ishigami(sample.T) 
534   
535      Now we can do scatter plots of the output with respect to each parameter. 
536      This gives a visual way to understand how each parameter impacts the 
537      output of the function. 
538   
539      >>> fig, ax = plt.subplots(1, n_dim, figsize=(12, 4)) 
540      >>> for i in range(n_dim): 
541      ...     xi = sample[:, i] 
542      ...     ax[i].scatter(xi, output, marker='+') 
543      ...     ax[i].set_xlabel(p_labels[i]) 
544      >>> ax[0].set_ylabel('Y') 
545      >>> plt.tight_layout() 
546      >>> plt.show() 
547   
548      Now Sobol' goes a step further: 
549      by conditioning the output value by given values of the parameter 
550      (black lines), the conditional output mean is computed. It corresponds to 
551      the term :math:`\mathbb{E}(Y|x_i)`. Taking the variance of this term gives 
552      the numerator of the Sobol' indices. 
553   
554      >>> mini = np.min(output) 
555      >>> maxi = np.max(output) 
556      >>> n_bins = 10 
557      >>> bins = np.linspace(-np.pi, np.pi, num=n_bins, endpoint=False) 
558      >>> dx = bins[1] - bins[0] 
559      >>> fig, ax = plt.subplots(1, n_dim, figsize=(12, 4)) 
560      >>> for i in range(n_dim): 
561      ...     xi = sample[:, i] 
562      ...     ax[i].scatter(xi, output, marker='+') 
563      ...     ax[i].set_xlabel(p_labels[i]) 
564      ...     for bin_ in bins: 
565      ...         idx = np.where((bin_ <= xi) & (xi <= bin_ + dx)) 
566      ...         xi_ = xi[idx] 
567      ...         y_ = output[idx] 
568      ...         ave_y_ = np.mean(y_) 
569      ...         ax[i].plot([bin_ + dx/2] * 2, [mini, maxi], c='k') 
570      ...         ax[i].scatter(bin_ + dx/2, ave_y_, c='r') 
571      >>> ax[0].set_ylabel('Y') 
572      >>> plt.tight_layout() 
573      >>> plt.show() 
574   
575      Looking at :math:`x_3`, the variance 
576      of the mean is zero leading to :math:`S_{x_3} = 0`. But we can further 
577      observe that the variance of the output is not constant along the parameter 
578      values of :math:`x_3`. This heteroscedasticity is explained by higher order 
579      interactions. Moreover, an heteroscedasticity is also noticeable on 
580      :math:`x_1` leading to an interaction between :math:`x_3` and :math:`x_1`. 
581      On :math:`x_2`, the variance seems to be constant and thus null interaction 
582      with this parameter can be supposed. 
583   
584      This case is fairly simple to analyse visually---although it is only a 
585      qualitative analysis. Nevertheless, when the number of input parameters 
586      increases such analysis becomes unrealistic as it would be difficult to 
587      conclude on high-order terms. Hence the benefit of using Sobol' indices. 
588   
589      """
590      rng = check_random_state(rng)
591  
592      n_ = int(n)
593      if not (n_ & (n_ - 1) == 0) or n != n_:
594          raise ValueError(
595              "The balance properties of Sobol' points require 'n' "
596              "to be a power of 2."
597          )
598      n = n_
599  
600      if not callable(method):
601          indices_methods = {
602              "saltelli_2010": saltelli_2010,
603          }
604          try:
605              method = method.lower()  # type: ignore[assignment]
606              indices_method_ = indices_methods[method]
607          except KeyError as exc:
608              message = (
609                  f"{method!r} is not a valid 'method'. It must be one of"
610                  f" {set(indices_methods)!r} or a callable."
611              )
612              raise ValueError(message) from exc
613      else:
614          indices_method_ = method
615          sig = inspect.signature(indices_method_)
616  
617          if set(sig.parameters) != {'f_A', 'f_B', 'f_AB'}:
618              message = (
619                  "If 'method' is a callable, it must have the following"
620                  f" signature: {inspect.signature(saltelli_2010)}"
621              )
622              raise ValueError(message)
623  
624      def indices_method(f_A, f_B, f_AB):
625          """Wrap indices method to ensure proper output dimension. 
626   
627          1D when single output, 2D otherwise. 
628          """
629          return np.squeeze(indices_method_(f_A=f_A, f_B=f_B, f_AB=f_AB))
630  
631      if callable(func):
632          if dists is None:
633              raise ValueError(
634                  "'dists' must be defined when 'func' is a callable."
635              )
636  
637          def wrapped_func(x):
638              return np.atleast_2d(func(x))
639  
640          A, B = sample_A_B(n=n, dists=dists, rng=rng)
641          AB = sample_AB(A=A, B=B)
642  
643          f_A = wrapped_func(A)
644  
645          if f_A.shape[1] != n:
646              raise ValueError(
647                  "'func' output should have a shape ``(s, -1)`` with ``s`` "
648                  "the number of output."
649              )
650  
651          def funcAB(AB):
652              d, d, n = AB.shape
653              AB = np.moveaxis(AB, 0, -1).reshape(d, n*d)
654              f_AB = wrapped_func(AB)
655              return np.moveaxis(f_AB.reshape((-1, n, d)), -1, 0)
656  
657          f_B = wrapped_func(B)
658          f_AB = funcAB(AB)
659      else:
660          message = (
661              "When 'func' is a dictionary, it must contain the following "
662              "keys: 'f_A', 'f_B' and 'f_AB'."
663              "'f_A' and 'f_B' should have a shape ``(s, n)`` and 'f_AB' "
664              "should have a shape ``(d, s, n)``."
665          )
666          try:
667              f_A, f_B, f_AB = map(lambda arr: arr.copy(), np.atleast_2d(
668                  func['f_A'], func['f_B'], func['f_AB']
669              ))
670          except KeyError as exc:
671              raise ValueError(message) from exc
672  
673          if f_A.shape[1] != n or f_A.shape != f_B.shape or \
674                  f_AB.shape == f_A.shape or f_AB.shape[-1] % n != 0:
675              raise ValueError(message)
676  
677      # Normalization by mean
678      # Sobol', I. and Levitan, Y. L. (1999). On the use of variance reducing
679      # multipliers in monte carlo computations of a global sensitivity index.
680      # Computer Physics Communications, 117(1) :52-61.
681      mean = np.mean([f_A, f_B], axis=(0, -1)).reshape(-1, 1)
682      f_A -= mean
683      f_B -= mean
684      f_AB -= mean
685  
686      # Compute indices
687      # Filter warnings for constant output as var = 0
688      with np.errstate(divide='ignore', invalid='ignore'):
689          first_order, total_order = indices_method(f_A=f_A, f_B=f_B, f_AB=f_AB)
690  
691      # null variance means null indices
692      first_order[~np.isfinite(first_order)] = 0
693      total_order[~np.isfinite(total_order)] = 0
694  
695      res = dict(
696          first_order=first_order,
697          total_order=total_order,
698          _indices_method=indices_method,
699          _f_A=f_A,
700          _f_B=f_B,
701          _f_AB=f_AB
702      )
703  
704      if callable(func):
705          res.update(
706              dict(
707                  _A=A,
708                  _B=B,
709                  _AB=AB,
710              )
711          )
712  
713      return SobolResult(**res)
