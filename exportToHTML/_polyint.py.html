19   class _Interpolator1D:
20       """ 
21       Common features in univariate interpolation 
22    
23       Deal with input data type and interpolation axis rolling. The 
24       actual interpolator can assume the y-data is of shape (n, r) where 
25       `n` is the number of x-points, and `r` the number of variables, 
26       and use self.dtype as the y-data type. 
27    
28       Attributes 
29       ---------- 
30       _y_axis 
31           Axis along which the interpolation goes in the original array 
32       _y_extra_shape 
33           Additional trailing shape of the input arrays, excluding 
34           the interpolation axis. 
35       dtype 
36           Dtype of the y-data arrays. Can be set via _set_dtype, which 
37           forces it to be float or complex. 
38    
39       Methods 
40       ------- 
41       __call__ 
42       _prepare_x 
43       _finish_y 
44       _reshape_yi 
45       _set_yi 
46       _set_dtype 
47       _evaluate 
48    
49       """
50   
51       __slots__ = ('_y_axis', '_y_extra_shape', 'dtype')
52   
53       def __init__(self, xi=None, yi=None, axis=None):
54           self._y_axis = axis
55           self._y_extra_shape = None
56           self.dtype = None
57           if yi is not None:
58               self._set_yi(yi, xi=xi, axis=axis)
59   
60       def __call__(self, x):
61           """ 
62           Evaluate the interpolant 
63    
64           Parameters 
65           ---------- 
66           x : array_like 
67               Point or points at which to evaluate the interpolant. 
68    
69           Returns 
70           ------- 
71           y : array_like 
72               Interpolated values. Shape is determined by replacing 
73               the interpolation axis in the original array with the shape of `x`. 
74    
75           Notes 
76           ----- 
77           Input values `x` must be convertible to `float` values like `int` 
78           or `float`. 
79    
80           """
81           x, x_shape = self._prepare_x(x)
82           y = self._evaluate(x)
83           return self._finish_y(y, x_shape)
84   
85       def _evaluate(self, x):
86           """ 
87           Actually evaluate the value of the interpolator. 
88           """
89           raise NotImplementedError()
90   
91       def _prepare_x(self, x):
92           """Reshape input x array to 1-D"""
93           x = _asarray_validated(x, check_finite=False, as_inexact=True)
94           x_shape = x.shape
95           return x.ravel(), x_shape
96   
97       def _finish_y(self, y, x_shape):
98           """Reshape interpolated y back to an N-D array similar to initial y"""
99           y = y.reshape(x_shape + self._y_extra_shape)
100          if self._y_axis != 0 and x_shape != ():
101              nx = len(x_shape)
102              ny = len(self._y_extra_shape)
103              s = (list(range(nx, nx + self._y_axis))
104                   + list(range(nx)) + list(range(nx+self._y_axis, nx+ny)))
105              y = y.transpose(s)
106          return y
107  
108      def _reshape_yi(self, yi, check=False):
109          yi = np.moveaxis(np.asarray(yi), self._y_axis, 0)
110          if check and yi.shape[1:] != self._y_extra_shape:
111              ok_shape = (f"{self._y_extra_shape[-self._y_axis:]!r} + (N,) + "
112                          f"{self._y_extra_shape[:-self._y_axis]!r}")
113              raise ValueError(f"Data must be of shape {ok_shape}")
114          return yi.reshape((yi.shape[0], -1))
115  
116      def _set_yi(self, yi, xi=None, axis=None):
117          if axis is None:
118              axis = self._y_axis
119          if axis is None:
120              raise ValueError("no interpolation axis specified")
121  
122          yi = np.asarray(yi)
123  
124          shape = yi.shape
125          if shape == ():
126              shape = (1,)
127          if xi is not None and shape[axis] != len(xi):
128              raise ValueError("x and y arrays must be equal in length along "
129                               "interpolation axis.")
130  
131          self._y_axis = (axis % yi.ndim)
132          self._y_extra_shape = yi.shape[:self._y_axis] + yi.shape[self._y_axis+1:]
133          self.dtype = None
134          self._set_dtype(yi.dtype)
135  
136      def _set_dtype(self, dtype, union=False):
137          if np.issubdtype(dtype, np.complexfloating) \
138                 or np.issubdtype(self.dtype, np.complexfloating):
139              self.dtype = np.complex128
140          else:
141              if not union or self.dtype != np.complex128:
142                  self.dtype = np.float64

145  class _Interpolator1DWithDerivatives(_Interpolator1D):
146      def derivatives(self, x, der=None):
147          """ 
148          Evaluate several derivatives of the polynomial at the point `x` 
149   
150          Produce an array of derivatives evaluated at the point `x`. 
151   
152          Parameters 
153          ---------- 
154          x : array_like 
155              Point or points at which to evaluate the derivatives 
156          der : int or list or None, optional 
157              How many derivatives to evaluate, or None for all potentially 
158              nonzero derivatives (that is, a number equal to the number 
159              of points), or a list of derivatives to evaluate. This number 
160              includes the function value as the '0th' derivative. 
161   
162          Returns 
163          ------- 
164          d : ndarray 
165              Array with derivatives; ``d[j]`` contains the jth derivative. 
166              Shape of ``d[j]`` is determined by replacing the interpolation 
167              axis in the original array with the shape of `x`. 
168   
169          Examples 
170          -------- 
171          >>> from scipy.interpolate import KroghInterpolator 
172          >>> KroghInterpolator([0,0,0],[1,2,3]).derivatives(0) 
173          array([1.0,2.0,3.0]) 
174          >>> KroghInterpolator([0,0,0],[1,2,3]).derivatives([0,0]) 
175          array([[1.0,1.0], 
176                 [2.0,2.0], 
177                 [3.0,3.0]]) 
178   
179          """
180          x, x_shape = self._prepare_x(x)
181          y = self._evaluate_derivatives(x, der)
182  
183          y = y.reshape((y.shape[0],) + x_shape + self._y_extra_shape)
184          if self._y_axis != 0 and x_shape != ():
185              nx = len(x_shape)
186              ny = len(self._y_extra_shape)
187              s = ([0] + list(range(nx+1, nx + self._y_axis+1))
188                   + list(range(1, nx+1)) +
189                   list(range(nx+1+self._y_axis, nx+ny+1)))
190              y = y.transpose(s)
191          return y
192  
193      def derivative(self, x, der=1):
194          """ 
195          Evaluate a single derivative of the polynomial at the point `x`. 
196   
197          Parameters 
198          ---------- 
199          x : array_like 
200              Point or points at which to evaluate the derivatives 
201   
202          der : integer, optional 
203              Which derivative to evaluate (default: first derivative). 
204              This number includes the function value as 0th derivative. 
205   
206          Returns 
207          ------- 
208          d : ndarray 
209              Derivative interpolated at the x-points. Shape of `d` is 
210              determined by replacing the interpolation axis in the 
211              original array with the shape of `x`. 
212   
213          Notes 
214          ----- 
215          This may be computed by evaluating all derivatives up to the desired 
216          one (using self.derivatives()) and then discarding the rest. 
217   
218          """
219          x, x_shape = self._prepare_x(x)
220          y = self._evaluate_derivatives(x, der+1)
221          return self._finish_y(y[der], x_shape)
222  
223      def _evaluate_derivatives(self, x, der=None):
224          """ 
225          Actually evaluate the derivatives. 
226   
227          Parameters 
228          ---------- 
229          x : array_like 
230              1D array of points at which to evaluate the derivatives 
231          der : integer, optional 
232              The number of derivatives to evaluate, from 'order 0' (der=1) 
233              to order der-1.  If omitted, return all possibly-non-zero 
234              derivatives, ie 0 to order n-1. 
235   
236          Returns 
237          ------- 
238          d : ndarray 
239              Array of shape ``(der, x.size, self.yi.shape[1])`` containing 
240              the derivatives from 0 to der-1 
241          """
242          raise NotImplementedError()

245  class KroghInterpolator(_Interpolator1DWithDerivatives):
246      """ 
247      Interpolating polynomial for a set of points. 
248   
249      The polynomial passes through all the pairs ``(xi, yi)``. One may 
250      additionally specify a number of derivatives at each point `xi`; 
251      this is done by repeating the value `xi` and specifying the 
252      derivatives as successive `yi` values. 
253   
254      Allows evaluation of the polynomial and all its derivatives. 
255      For reasons of numerical stability, this function does not compute 
256      the coefficients of the polynomial, although they can be obtained 
257      by evaluating all the derivatives. 
258   
259      Parameters 
260      ---------- 
261      xi : array_like, shape (npoints, ) 
262          Known x-coordinates. Must be sorted in increasing order. 
263      yi : array_like, shape (..., npoints, ...) 
264          Known y-coordinates. When an xi occurs two or more times in 
265          a row, the corresponding yi's represent derivative values. The length of `yi` 
266          along the interpolation axis must be equal to the length of `xi`. Use the 
267          `axis` parameter to select the correct axis. 
268      axis : int, optional 
269          Axis in the `yi` array corresponding to the x-coordinate values. Defaults to 
270          ``axis=0``. 
271   
272      Notes 
273      ----- 
274      Be aware that the algorithms implemented here are not necessarily 
275      the most numerically stable known. Moreover, even in a world of 
276      exact computation, unless the x coordinates are chosen very 
277      carefully - Chebyshev zeros (e.g., cos(i*pi/n)) are a good choice - 
278      polynomial interpolation itself is a very ill-conditioned process 
279      due to the Runge phenomenon. In general, even with well-chosen 
280      x values, degrees higher than about thirty cause problems with 
281      numerical instability in this code. 
282   
283      Based on [1]_. 
284   
285      References 
286      ---------- 
287      .. [1] Krogh, "Efficient Algorithms for Polynomial Interpolation 
288          and Numerical Differentiation", 1970. 
289   
290      Examples 
291      -------- 
292      To produce a polynomial that is zero at 0 and 1 and has 
293      derivative 2 at 0, call 
294   
295      >>> from scipy.interpolate import KroghInterpolator 
296      >>> KroghInterpolator([0,0,1],[0,2,0]) 
297   
298      This constructs the quadratic :math:`2x^2-2x`. The derivative condition 
299      is indicated by the repeated zero in the `xi` array; the corresponding 
300      yi values are 0, the function value, and 2, the derivative value. 
301   
302      For another example, given `xi`, `yi`, and a derivative `ypi` for each 
303      point, appropriate arrays can be constructed as: 
304   
305      >>> import numpy as np 
306      >>> rng = np.random.default_rng() 
307      >>> xi = np.linspace(0, 1, 5) 
308      >>> yi, ypi = rng.random((2, 5)) 
309      >>> xi_k, yi_k = np.repeat(xi, 2), np.ravel(np.dstack((yi,ypi))) 
310      >>> KroghInterpolator(xi_k, yi_k) 
311   
312      To produce a vector-valued polynomial, supply a higher-dimensional 
313      array for `yi`: 
314   
315      >>> KroghInterpolator([0,1],[[2,3],[4,5]]) 
316   
317      This constructs a linear polynomial giving (2,3) at 0 and (4,5) at 1. 
318   
319      """
320  
321      def __init__(self, xi, yi, axis=0):
322          super().__init__(xi, yi, axis)
323  
324          self.xi = np.asarray(xi)
325          self.yi = self._reshape_yi(yi)
326          self.n, self.r = self.yi.shape
327  
328          if (deg := self.xi.size) > 30:
329              warnings.warn(f"{deg} degrees provided, degrees higher than about"
330                            " thirty cause problems with numerical instability "
331                            "with 'KroghInterpolator'", stacklevel=2)
332  
333          c = np.zeros((self.n+1, self.r), dtype=self.dtype)
334          c[0] = self.yi[0]
335          Vk = np.zeros((self.n, self.r), dtype=self.dtype)
336          for k in range(1, self.n):
337              s = 0
338              while s <= k and xi[k-s] == xi[k]:
339                  s += 1
340              s -= 1
341              Vk[0] = self.yi[k]/float_factorial(s)
342              for i in range(k-s):
343                  if xi[i] == xi[k]:
344                      raise ValueError("Elements of `xi` can't be equal.")
345                  if s == 0:
346                      Vk[i+1] = (c[i]-Vk[i])/(xi[i]-xi[k])
347                  else:
348                      Vk[i+1] = (Vk[i+1]-Vk[i])/(xi[i]-xi[k])
349              c[k] = Vk[k-s]
350          self.c = c
351  
352      def _evaluate(self, x):
353          pi = 1
354          p = np.zeros((len(x), self.r), dtype=self.dtype)
355          p += self.c[0,np.newaxis,:]
356          for k in range(1, self.n):
357              w = x - self.xi[k-1]
358              pi = w*pi
359              p += pi[:,np.newaxis] * self.c[k]
360          return p
361  
362      def _evaluate_derivatives(self, x, der=None):
363          n = self.n
364          r = self.r
365  
366          if der is None:
367              der = self.n
368  
369          pi = np.zeros((n, len(x)))
370          w = np.zeros((n, len(x)))
371          pi[0] = 1
372          p = np.zeros((len(x), self.r), dtype=self.dtype)
373          p += self.c[0, np.newaxis, :]
374  
375          for k in range(1, n):
376              w[k-1] = x - self.xi[k-1]
377              pi[k] = w[k-1] * pi[k-1]
378              p += pi[k, :, np.newaxis] * self.c[k]
379  
380          cn = np.zeros((max(der, n+1), len(x), r), dtype=self.dtype)
381          cn[:n+1, :, :] += self.c[:n+1, np.newaxis, :]
382          cn[0] = p
383          for k in range(1, n):
384              for i in range(1, n-k+1):
385                  pi[i] = w[k+i-1]*pi[i-1] + pi[i]
386                  cn[k] = cn[k] + pi[i, :, np.newaxis]*cn[k+i]
387              cn[k] *= float_factorial(k)
388  
389          cn[n, :, :] = 0
390          return cn[:der]
391

393  def krogh_interpolate(xi, yi, x, der=0, axis=0):
394      """ 
395      Convenience function for polynomial interpolation. 
396   
397      See `KroghInterpolator` for more details. 
398   
399      Parameters 
400      ---------- 
401      xi : array_like 
402          Interpolation points (known x-coordinates). 
403      yi : array_like 
404          Known y-coordinates, of shape ``(xi.size, R)``. Interpreted as 
405          vectors of length R, or scalars if R=1. 
406      x : array_like 
407          Point or points at which to evaluate the derivatives. 
408      der : int or list or None, optional 
409          How many derivatives to evaluate, or None for all potentially 
410          nonzero derivatives (that is, a number equal to the number 
411          of points), or a list of derivatives to evaluate. This number 
412          includes the function value as the '0th' derivative. 
413      axis : int, optional 
414          Axis in the `yi` array corresponding to the x-coordinate values. 
415   
416      Returns 
417      ------- 
418      d : ndarray 
419          If the interpolator's values are R-D then the 
420          returned array will be the number of derivatives by N by R. 
421          If `x` is a scalar, the middle dimension will be dropped; if 
422          the `yi` are scalars then the last dimension will be dropped. 
423   
424      See Also 
425      -------- 
426      KroghInterpolator : Krogh interpolator 
427   
428      Notes 
429      ----- 
430      Construction of the interpolating polynomial is a relatively expensive 
431      process. If you want to evaluate it repeatedly consider using the class 
432      KroghInterpolator (which is what this function uses). 
433   
434      Examples 
435      -------- 
436      We can interpolate 2D observed data using Krogh interpolation: 
437   
438      >>> import numpy as np 
439      >>> import matplotlib.pyplot as plt 
440      >>> from scipy.interpolate import krogh_interpolate 
441      >>> x_observed = np.linspace(0.0, 10.0, 11) 
442      >>> y_observed = np.sin(x_observed) 
443      >>> x = np.linspace(min(x_observed), max(x_observed), num=100) 
444      >>> y = krogh_interpolate(x_observed, y_observed, x) 
445      >>> plt.plot(x_observed, y_observed, "o", label="observation") 
446      >>> plt.plot(x, y, label="krogh interpolation") 
447      >>> plt.legend() 
448      >>> plt.show() 
449      """
450  
451      P = KroghInterpolator(xi, yi, axis=axis)
452      if der == 0:
453          return P(x)
454      elif _isscalar(der):
455          return P.derivative(x, der=der)
456      else:
457          return P.derivatives(x, der=np.amax(der)+1)[der]

460  def approximate_taylor_polynomial(f,x,degree,scale,order=None):
461      """ 
462      Estimate the Taylor polynomial of f at x by polynomial fitting. 
463   
464      Parameters 
465      ---------- 
466      f : callable 
467          The function whose Taylor polynomial is sought. Should accept 
468          a vector of `x` values. 
469      x : scalar 
470          The point at which the polynomial is to be evaluated. 
471      degree : int 
472          The degree of the Taylor polynomial 
473      scale : scalar 
474          The width of the interval to use to evaluate the Taylor polynomial. 
475          Function values spread over a range this wide are used to fit the 
476          polynomial. Must be chosen carefully. 
477      order : int or None, optional 
478          The order of the polynomial to be used in the fitting; `f` will be 
479          evaluated ``order+1`` times. If None, use `degree`. 
480   
481      Returns 
482      ------- 
483      p : poly1d instance 
484          The Taylor polynomial (translated to the origin, so that 
485          for example p(0)=f(x)). 
486   
487      Notes 
488      ----- 
489      The appropriate choice of "scale" is a trade-off; too large and the 
490      function differs from its Taylor polynomial too much to get a good 
491      answer, too small and round-off errors overwhelm the higher-order terms. 
492      The algorithm used becomes numerically unstable around order 30 even 
493      under ideal circumstances. 
494   
495      Choosing order somewhat larger than degree may improve the higher-order 
496      terms. 
497   
498      Examples 
499      -------- 
500      We can calculate Taylor approximation polynomials of sin function with 
501      various degrees: 
502   
503      >>> import numpy as np 
504      >>> import matplotlib.pyplot as plt 
505      >>> from scipy.interpolate import approximate_taylor_polynomial 
506      >>> x = np.linspace(-10.0, 10.0, num=100) 
507      >>> plt.plot(x, np.sin(x), label="sin curve") 
508      >>> for degree in np.arange(1, 15, step=2): 
509      ...     sin_taylor = approximate_taylor_polynomial(np.sin, 0, degree, 1, 
510      ...                                                order=degree + 2) 
511      ...     plt.plot(x, sin_taylor(x), label=f"degree={degree}") 
512      >>> plt.legend(bbox_to_anchor=(1.05, 1), loc='upper left', 
513      ...            borderaxespad=0.0, shadow=True) 
514      >>> plt.tight_layout() 
515      >>> plt.axis([-10, 10, -10, 10]) 
516      >>> plt.show() 
517   
518      """
519      if order is None:
520          order = degree
521  
522      n = order+1
523      # Choose n points that cluster near the endpoints of the interval in
524      # a way that avoids the Runge phenomenon. Ensure, by including the
525      # endpoint or not as appropriate, that one point always falls at x
526      # exactly.
527      xs = scale*np.cos(np.linspace(0,np.pi,n,endpoint=n % 1)) + x
528  
529      P = KroghInterpolator(xs, f(xs))
530      d = P.derivatives(x,der=degree+1)
531  
532      return np.poly1d((d/factorial(np.arange(degree+1)))[::-1])

535  class BarycentricInterpolator(_Interpolator1DWithDerivatives):
536      r"""Interpolating polynomial for a set of points. 
537   
538      Constructs a polynomial that passes through a given set of points. 
539      Allows evaluation of the polynomial and all its derivatives, 
540      efficient changing of the y-values to be interpolated, 
541      and updating by adding more x- and y-values. 
542   
543      For reasons of numerical stability, this function does not compute 
544      the coefficients of the polynomial. 
545   
546      The values `yi` need to be provided before the function is 
547      evaluated, but none of the preprocessing depends on them, so rapid 
548      updates are possible. 
549   
550      Parameters 
551      ---------- 
552      xi : array_like, shape (npoints, ) 
553          1-D array of x coordinates of the points the polynomial 
554          should pass through 
555      yi : array_like, shape (..., npoints, ...), optional 
556          N-D array of y coordinates of the points the polynomial should pass through. 
557          If None, the y values will be supplied later via the `set_y` method. 
558          The length of `yi` along the interpolation axis must be equal to the length 
559          of `xi`. Use the ``axis`` parameter to select correct axis. 
560      axis : int, optional 
561          Axis in the yi array corresponding to the x-coordinate values. Defaults 
562          to ``axis=0``. 
563      wi : array_like, optional 
564          The barycentric weights for the chosen interpolation points `xi`. 
565          If absent or None, the weights will be computed from `xi` (default). 
566          This allows for the reuse of the weights `wi` if several interpolants 
567          are being calculated using the same nodes `xi`, without re-computation. 
568      rng : {None, int, `numpy.random.Generator`}, optional 
569          If `rng` is passed by keyword, types other than `numpy.random.Generator` are 
570          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
571          If `rng` is already a ``Generator`` instance, then the provided instance is 
572          used. Specify `rng` for repeatable interpolation. 
573   
574          If this argument `random_state` is passed by keyword, 
575          legacy behavior for the argument `random_state` applies: 
576   
577          - If `random_state` is None (or `numpy.random`), the `numpy.random.RandomState` 
578            singleton is used. 
579          - If `random_state` is an int, a new ``RandomState`` instance is used, 
580            seeded with `random_state`. 
581          - If `random_state` is already a ``Generator`` or ``RandomState`` instance then 
582            that instance is used. 
583   
584          .. versionchanged:: 1.15.0 
585              As part of the `SPEC-007 <https://scientific-python.org/specs/spec-0007/>`_ 
586              transition from use of `numpy.random.RandomState` to 
587              `numpy.random.Generator` this keyword was changed from `random_state` to `rng`. 
588              For an interim period, both keywords will continue to work (only specify 
589              one of them). After the interim period using the `random_state` keyword will emit 
590              warnings. The behavior of the `random_state` and `rng` keywords is outlined above. 
591   
592      Notes 
593      ----- 
594      This class uses a "barycentric interpolation" method that treats 
595      the problem as a special case of rational function interpolation. 
596      This algorithm is quite stable, numerically, but even in a world of 
597      exact computation, unless the x coordinates are chosen very 
598      carefully - Chebyshev zeros (e.g., cos(i*pi/n)) are a good choice - 
599      polynomial interpolation itself is a very ill-conditioned process 
600      due to the Runge phenomenon. 
601   
602      Based on Berrut and Trefethen 2004, "Barycentric Lagrange Interpolation". 
603   
604      Examples 
605      -------- 
606      To produce a quintic barycentric interpolant approximating the function 
607      :math:`\sin x`, and its first four derivatives, using six randomly-spaced 
608      nodes in :math:`(0, \frac{\pi}{2})`: 
609   
610      >>> import numpy as np 
611      >>> import matplotlib.pyplot as plt 
612      >>> from scipy.interpolate import BarycentricInterpolator 
613      >>> rng = np.random.default_rng() 
614      >>> xi = rng.random(6) * np.pi/2 
615      >>> f, f_d1, f_d2, f_d3, f_d4 = np.sin, np.cos, lambda x: -np.sin(x), lambda x: -np.cos(x), np.sin 
616      >>> P = BarycentricInterpolator(xi, f(xi), random_state=rng) 
617      >>> fig, axs = plt.subplots(5, 1, sharex=True, layout='constrained', figsize=(7,10)) 
618      >>> x = np.linspace(0, np.pi, 100) 
619      >>> axs[0].plot(x, P(x), 'r:', x, f(x), 'k--', xi, f(xi), 'xk')

639      @_transition_to_rng("random_state", replace_doc=False)
640      def __init__(self, xi, yi=None, axis=0, *, wi=None, rng=None):
641          super().__init__(xi, yi, axis)
642  
643          rng = check_random_state(rng)
644  
645          self.xi = np.asarray(xi, dtype=np.float64)
646          self.set_yi(yi)
647          self.n = len(self.xi)
648  
649          # cache derivative object to avoid re-computing the weights with every call.
650          self._diff_cij = None
651  
652          if wi is not None:
653              self.wi = wi
654          else:
655              # See page 510 of Berrut and Trefethen 2004 for an explanation of the
656              # capacity scaling and the suggestion of using a random permutation of
657              # the input factors.
658              # At the moment, the permutation is not performed for xi that are
659              # appended later through the add_xi interface. It's not clear to me how
660              # to implement that and it seems that most situations that require
661              # these numerical stability improvements will be able to provide all
662              # the points to the constructor.
663              self._inv_capacity = 4.0 / (np.max(self.xi) - np.min(self.xi))
664              permute = rng.permutation(self.n, )
665              inv_permute = np.zeros(self.n, dtype=np.int32)
666              inv_permute[permute] = np.arange(self.n)
667              self.wi = np.zeros(self.n)
668  
669              for i in range(self.n):
670                  dist = self._inv_capacity * (self.xi[i] - self.xi[permute])
671                  dist[inv_permute[i]] = 1.0
672                  prod = np.prod(dist)
673                  if prod == 0.0:
674                      raise ValueError("Interpolation points xi must be"
675                                       " distinct.")
676                  self.wi[i] = 1.0 / prod
677  
678      def set_yi(self, yi, axis=None):
679          """ 
680          Update the y values to be interpolated 
681   
682          The barycentric interpolation algorithm requires the calculation 
683          of weights, but these depend only on the `xi`. The `yi` can be changed 
684          at any time. 
685   
686          Parameters 
687          ---------- 
688          yi : array_like 
689              The y-coordinates of the points the polynomial will pass through. 
690              If None, the y values must be supplied later. 
691          axis : int, optional 
692              Axis in the `yi` array corresponding to the x-coordinate values. 
693   
694          """
695          if yi is None:
696              self.yi = None
697              return
698          self._set_yi(yi, xi=self.xi, axis=axis)
699          self.yi = self._reshape_yi(yi)
700          self.n, self.r = self.yi.shape
701          self._diff_baryint = None
702  
703      def add_xi(self, xi, yi=None):
704          """ 
705          Add more x values to the set to be interpolated 
706   
707          The barycentric interpolation algorithm allows easy updating by 
708          adding more points for the polynomial to pass through. 
709   
710          Parameters 
711          ---------- 
712          xi : array_like 
713              The x coordinates of the points that the polynomial should pass 
714              through. 
715          yi : array_like, optional 
716              The y coordinates of the points the polynomial should pass through. 
717              Should have shape ``(xi.size, R)``; if R > 1 then the polynomial is 
718              vector-valued. 
719              If `yi` is not given, the y values will be supplied later. `yi` 
720              should be given if and only if the interpolator has y values 
721              specified. 
722   
723          Notes 
724          ----- 
725          The new points added by `add_xi` are not randomly permuted 
726          so there is potential for numerical instability, 
727          especially for a large number of points. If this 
728          happens, please reconstruct interpolation from scratch instead. 
729          """
730          if yi is not None:
731              if self.yi is None:
732                  raise ValueError("No previous yi value to update!")
733              yi = self._reshape_yi(yi, check=True)
734              self.yi = np.vstack((self.yi,yi))
735          else:
736              if self.yi is not None:
737                  raise ValueError("No update to yi provided!")
738          old_n = self.n
739          self.xi = np.concatenate((self.xi,xi))
740          self.n = len(self.xi)
741          self.wi **= -1
742          old_wi = self.wi
743          self.wi = np.zeros(self.n)
744          self.wi[:old_n] = old_wi
745          for j in range(old_n, self.n):
746              self.wi[:j] *= self._inv_capacity * (self.xi[j]-self.xi[:j])
747              self.wi[j] = np.multiply.reduce(
748                  self._inv_capacity * (self.xi[:j]-self.xi[j])
749              )
750          self.wi **= -1
751          self._diff_cij = None
752          self._diff_baryint = None

754      def __call__(self, x):
755          """Evaluate the interpolating polynomial at the points x 
756   
757          Parameters 
758          ---------- 
759          x : array_like 
760              Point or points at which to evaluate the interpolant. 
761   
762          Returns 
763          ------- 
764          y : array_like 
765              Interpolated values. Shape is determined by replacing 
766              the interpolation axis in the original array with the shape of `x`. 
767   
768          Notes 
769          ----- 
770          Currently the code computes an outer product between `x` and the 
771          weights, that is, it constructs an intermediate array of size 
772          ``(N, len(x))``, where N is the degree of the polynomial. 
773          """
774          return _Interpolator1D.__call__(self, x)

776      def _evaluate(self, x):
777          if x.size == 0:
778              p = np.zeros((0, self.r), dtype=self.dtype)
779          else:
780              c = x[..., np.newaxis] - self.xi
781              z = c == 0
782              c[z] = 1
783              c = self.wi / c
784              with np.errstate(divide='ignore'):
785                  p = np.dot(c, self.yi) / np.sum(c, axis=-1)[..., np.newaxis]
786              # Now fix where x==some xi
787              r = np.nonzero(z)
788              if len(r) == 1:  # evaluation at a scalar
789                  if len(r[0]) > 0:  # equals one of the points
790                      p = self.yi[r[0][0]]
791              else:
792                  p[r[:-1]] = self.yi[r[-1]]
793          return p

795      def derivative(self, x, der=1):
796          """ 
797          Evaluate a single derivative of the polynomial at the point x. 
798   
799          Parameters 
800          ---------- 
801          x : array_like 
802              Point or points at which to evaluate the derivatives 
803          der : integer, optional 
804              Which derivative to evaluate (default: first derivative). 
805              This number includes the function value as 0th derivative. 
806   
807          Returns 
808          ------- 
809          d : ndarray 
810              Derivative interpolated at the x-points. Shape of `d` is 
811              determined by replacing the interpolation axis in the 
812              original array with the shape of `x`. 
813          """
814          x, x_shape = self._prepare_x(x)
815          y = self._evaluate_derivatives(x, der+1, all_lower=False)
816          return self._finish_y(y, x_shape)
817  
818      def _evaluate_derivatives(self, x, der=None, all_lower=True):
819          # NB: der here is not the order of the highest derivative;
820          # instead, it is the size of the derivatives matrix that
821          # would be returned with all_lower=True, including the
822          # '0th' derivative (the undifferentiated function).
823          # E.g. to evaluate the 5th derivative alone, call
824          # _evaluate_derivatives(x, der=6, all_lower=False).
825  
826          if (not all_lower) and (x.size == 0 or self.r == 0):
827              return np.zeros((0, self.r), dtype=self.dtype)
828  
829          if (not all_lower) and der == 1:
830              return self._evaluate(x)
831  
832          if (not all_lower) and (der > self.n):
833              return np.zeros((len(x), self.r), dtype=self.dtype)
834  
835          if der is None:
836              der = self.n
837  
838          if all_lower and (x.size == 0 or self.r == 0):
839              return np.zeros((der, len(x), self.r), dtype=self.dtype)
840  
841          if self._diff_cij is None:
842              # c[i,j] = xi[i] - xi[j]
843              c = self.xi[:, np.newaxis] - self.xi
844  
845              # avoid division by 0 (diagonal entries are so far zero by construction)
846              np.fill_diagonal(c, 1)
847  
848              # c[i,j] = (w[j] / w[i]) / (xi[i] - xi[j]) (equation 9.4)
849              c = self.wi/ (c * self.wi[..., np.newaxis])
850  
851              # fill in correct diagonal entries: each column sums to 0
852              np.fill_diagonal(c, 0)
853  
854              # calculate diagonal
855              # c[j,j] = -sum_{i != j} c[i,j] (equation 9.5)
856              d = -c.sum(axis=1)
857              # c[i,j] = l_j(x_i)
858              np.fill_diagonal(c, d)
859  
860              self._diff_cij = c
861  
862          if self._diff_baryint is None:
863              # initialise and cache derivative interpolator and cijs;
864              # reuse weights wi (which depend only on interpolation points xi),
865              # to avoid unnecessary re-computation
866              self._diff_baryint = BarycentricInterpolator(xi=self.xi,
867                                                           yi=self._diff_cij @ self.yi,
868                                                           wi=self.wi)
869              self._diff_baryint._diff_cij = self._diff_cij
870  
871          if all_lower:
872              # assemble matrix of derivatives from order 0 to order der-1,
873              # in the format required by _Interpolator1DWithDerivatives.
874              cn = np.zeros((der, len(x), self.r), dtype=self.dtype)
875              for d in range(der):
876                  cn[d, :, :] = self._evaluate_derivatives(x, d+1, all_lower=False)
877              return cn
878  
879          # recursively evaluate only the derivative requested
880          return self._diff_baryint._evaluate_derivatives(x, der-1, all_lower=False)
881

883  def barycentric_interpolate(xi, yi, x, axis=0, *, der=0, rng=None):
884      """ 
885      Convenience function for polynomial interpolation. 
886   
887      Constructs a polynomial that passes through a given set of points, 
888      then evaluates the polynomial. For reasons of numerical stability, 
889      this function does not compute the coefficients of the polynomial. 
890   
891      This function uses a "barycentric interpolation" method that treats 
892      the problem as a special case of rational function interpolation. 
893      This algorithm is quite stable, numerically, but even in a world of 
894      exact computation, unless the `x` coordinates are chosen very 
895      carefully - Chebyshev zeros (e.g., cos(i*pi/n)) are a good choice - 
896      polynomial interpolation itself is a very ill-conditioned process 
897      due to the Runge phenomenon. 
898   
899      Parameters 
900      ---------- 
901      xi : array_like 
902          1-D array of x coordinates of the points the polynomial should 
903          pass through 
904      yi : array_like 
905          The y coordinates of the points the polynomial should pass through. 
906      x : scalar or array_like 
907          Point or points at which to evaluate the interpolant. 
908      axis : int, optional 
909          Axis in the `yi` array corresponding to the x-coordinate values. 
910      der : int or list or None, optional 
911          How many derivatives to evaluate, or None for all potentially 
912          nonzero derivatives (that is, a number equal to the number 
913          of points), or a list of derivatives to evaluate. This number 
914          includes the function value as the '0th' derivative. 
915      rng : `numpy.random.Generator`, optional 
916          Pseudorandom number generator state. When `rng` is None, a new 
917          `numpy.random.Generator` is created using entropy from the 
918          operating system. Types other than `numpy.random.Generator` are 
919          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
920   
921      Returns 
922      ------- 
923      y : scalar or array_like 
924          Interpolated values. Shape is determined by replacing 
925          the interpolation axis in the original array with the shape of `x`. 
926   
927      See Also 
928      -------- 
929      BarycentricInterpolator : Barycentric interpolator 
930   
931      Notes 
932      ----- 
933      Construction of the interpolation weights is a relatively slow process. 
934      If you want to call this many times with the same xi (but possibly 
935      varying yi or x) you should use the class `BarycentricInterpolator`. 
936      This is what this function uses internally. 
937   
938      Examples 
939      -------- 
940      We can interpolate 2D observed data using barycentric interpolation: 
941   
942      >>> import numpy as np 
943      >>> import matplotlib.pyplot as plt 
944      >>> from scipy.interpolate import barycentric_interpolate 
945      >>> x_observed = np.linspace(0.0, 10.0, 11) 
946      >>> y_observed = np.sin(x_observed) 
947      >>> x = np.linspace(min(x_observed), max(x_observed), num=100) 
948      >>> y = barycentric_interpolate(x_observed, y_observed, x) 
949      >>> plt.plot(x_observed, y_observed, "o", label="observation") 
950      >>> plt.plot(x, y, label="barycentric interpolation") 
951      >>> plt.legend() 
952      >>> plt.show() 
953   
954      """
955      P = BarycentricInterpolator(xi, yi, axis=axis, rng=rng)
956      if der == 0:
957          return P(x)
958      elif _isscalar(der):
959          return P.derivative(x, der=der)
960      else:
961          return P.derivatives(x, der=np.amax(der)+1)[der]
