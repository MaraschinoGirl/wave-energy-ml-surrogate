22   class AgglomerationTransform(TransformerMixin):
23       """ 
24       A class for feature agglomeration via the transform interface. 
25       """
26   
27       # This prevents ``set_split_inverse_transform`` to be generated for the
28       # non-standard ``Xt`` arg on ``inverse_transform``.
29       # TODO(1.7): remove when Xt is removed for inverse_transform.
30       __metadata_request__inverse_transform = {"Xt": metadata_routing.UNUSED}
31   
32       def transform(self, X):
33           """ 
34           Transform a new matrix using the built clustering. 
35    
36           Parameters 
37           ---------- 
38           X : array-like of shape (n_samples, n_features) or \ 
39                   (n_samples, n_samples) 
40               A M by N array of M observations in N dimensions or a length 
41               M array of M one-dimensional observations. 
42    
43           Returns 
44           ------- 
45           Y : ndarray of shape (n_samples, n_clusters) or (n_clusters,) 
46               The pooled values for each feature cluster. 
47           """
48           check_is_fitted(self)
49   
50           X = validate_data(self, X, reset=False)
51           if self.pooling_func == np.mean and not issparse(X):
52               size = np.bincount(self.labels_)
53               n_samples = X.shape[0]
54               # a fast way to compute the mean of grouped features
55               nX = np.array(
56                   [np.bincount(self.labels_, X[i, :]) / size for i in range(n_samples)]
57               )
58           else:
59               nX = [
60                   self.pooling_func(X[:, self.labels_ == l], axis=1)
61                   for l in np.unique(self.labels_)
62               ]
63               nX = np.array(nX).T
64           return nX
65   
66       def inverse_transform(self, X=None, *, Xt=None):
67           """ 
68           Inverse the transformation and return a vector of size `n_features`. 
69    
70           Parameters 
71           ---------- 
72           X : array-like of shape (n_samples, n_clusters) or (n_clusters,) 
73               The values to be assigned to each cluster of samples. 
74    
75           Xt : array-like of shape (n_samples, n_clusters) or (n_clusters,) 
76               The values to be assigned to each cluster of samples. 
77    
78               .. deprecated:: 1.5 
79                   `Xt` was deprecated in 1.5 and will be removed in 1.7. Use `X` instead. 
80    
81           Returns 
82           ------- 
83           X : ndarray of shape (n_samples, n_features) or (n_features,) 
84               A vector of size `n_samples` with the values of `Xred` assigned to 
85               each of the cluster of samples. 
86           """
87           X = _deprecate_Xt_in_inverse_transform(X, Xt)
88   
89           check_is_fitted(self)
90   
91           unil, inverse = np.unique(self.labels_, return_inverse=True)
92           return X[..., inverse]
