14   def usage_and_die(rc):
15       print()
16       print("This script is designed to copy and register the Python debug")
17       print("binaries.  It looks for pythonXX_d.dll, pythoncomXX_d.dll etc,")
18       print("and installs them to work correctly with Python debug builds.")
19       print()
20       print("You will generally find this script in the. zip file that")
21       print("included these _d files.  Please run this script from")
22       print("that directory")
23       sys.exit(rc)

48   def _docopy(src, dest):
49       orig_src = src
50       if not os.path.isfile(src):
51           src = os.path.join(os.path.split(sys.argv[0])[0], src)
52           print(
53               "Can not find {} or {} to copy".format(
54                   os.path.abspath(orig_src), os.path.abspath(src)
55               )
56           )
57           return 0
58       try:
59           shutil.copy(src, dest)
60           print(f"Copied {src} -> {dest}")
61           return 1
62       except:
63           print(f"Error copying '{src}' -> '{dest}'")
64           print(sys.exc_info()[1])
65           usage_and_die(3)

68   def _doregister(mod_name, dll_name):
69       assert os.path.isfile(dll_name), "Shouldn't get here if the file doesn't exist!"
70       try:
71           key = winreg.OpenKey(
72               winreg.HKEY_LOCAL_MACHINE,
73               f"Software\\Python\\PythonCore\\{sys.winver}\\Modules\\{mod_name}",
74           )
75       except winreg.error:
76           try:
77               key = winreg.OpenKey(
78                   winreg.HKEY_LOCAL_MACHINE,
79                   f"Software\\Python\\PythonCore\\{sys.winver}\\Modules\\{mod_name}",
80               )
81           except winreg.error:
82               print(
83                   "Could not find the existing '{}' module registered in the registry".format(
84                       mod_name
85                   )
86               )
87               usage_and_die(4)
88       # Create the debug key.
89       sub_key = winreg.CreateKey(key, "Debug")
90       winreg.SetValue(sub_key, None, winreg.REG_SZ, dll_name)
91       print(f"Registered '{dll_name}' in the registry")

94   def _domodule(mod_name, release_mod_filename):
95       path, fname = os.path.split(release_mod_filename)
96       base, ext = os.path.splitext(fname)
97       new_fname = base + "_d" + ext
98       if _docopy(new_fname, path):
99           _doregister(mod_name, os.path.abspath(os.path.join(path, new_fname)))

102  # First the main Python DLL.
103  path, fname = path, fname = os.path.split(win32api.GetModuleFileName(sys.dllhandle))
104  base, ext = os.path.splitext(fname)
105  _docopy(base + "_d" + ext, path)
106  
107  # Then pythoncom and pywintypes.
108  _domodule("pythoncom", pythoncom.__file__)
109  _domodule("pywintypes", pywintypes.__file__)
110  
111  print("System _d files were setup.")
