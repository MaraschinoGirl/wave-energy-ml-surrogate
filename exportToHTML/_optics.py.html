39   class OPTICS(ClusterMixin, BaseEstimator):
40       """Estimate clustering structure from vector array. 
41    
42       OPTICS (Ordering Points To Identify the Clustering Structure), closely 
43       related to DBSCAN, finds core sample of high density and expands clusters 
44       from them [1]_. Unlike DBSCAN, keeps cluster hierarchy for a variable 
45       neighborhood radius. Better suited for usage on large datasets than the 
46       current sklearn implementation of DBSCAN. 
47    
48       Clusters are then extracted using a DBSCAN-like method 
49       (cluster_method = 'dbscan') or an automatic 
50       technique proposed in [1]_ (cluster_method = 'xi'). 
51    
52       This implementation deviates from the original OPTICS by first performing 
53       k-nearest-neighborhood searches on all points to identify core sizes, then 
54       computing only the distances to unprocessed points when constructing the 
55       cluster order. Note that we do not employ a heap to manage the expansion 
56       candidates, so the time complexity will be O(n^2). 
57    
58       Read more in the :ref:`User Guide <optics>`. 
59    
60       Parameters 
61       ---------- 
62       min_samples : int > 1 or float between 0 and 1, default=5 
63           The number of samples in a neighborhood for a point to be considered as 
64           a core point. Also, up and down steep regions can't have more than 
65           ``min_samples`` consecutive non-steep points. Expressed as an absolute 
66           number or a fraction of the number of samples (rounded to be at least 
67           2). 
68    
69       max_eps : float, default=np.inf 
70           The maximum distance between two samples for one to be considered as 
71           in the neighborhood of the other. Default value of ``np.inf`` will 
72           identify clusters across all scales; reducing ``max_eps`` will result 
73           in shorter run times. 
74    
75       metric : str or callable, default='minkowski' 
76           Metric to use for distance computation. Any metric from scikit-learn 
77           or scipy.spatial.distance can be used. 
78    
79           If metric is a callable function, it is called on each 
80           pair of instances (rows) and the resulting value recorded. The callable 
81           should take two arrays as input and return one value indicating the 
82           distance between them. This works for Scipy's metrics, but is less 
83           efficient than passing the metric name as a string. If metric is 
84           "precomputed", `X` is assumed to be a distance matrix and must be 
85           square. 
86    
87           Valid values for metric are: 
88    
89           - from scikit-learn: ['cityblock', 'cosine', 'euclidean', 'l1', 'l2', 
90             'manhattan'] 
91    
92           - from scipy.spatial.distance: ['braycurtis', 'canberra', 'chebyshev', 
93             'correlation', 'dice', 'hamming', 'jaccard', 'kulsinski', 
94             'mahalanobis', 'minkowski', 'rogerstanimoto', 'russellrao', 
95             'seuclidean', 'sokalmichener', 'sokalsneath', 'sqeuclidean', 
96             'yule'] 
97    
98           Sparse matrices are only supported by scikit-learn metrics. 
99           See the documentation for scipy.spatial.distance for details on these 
100          metrics. 
101   
102          .. note:: 
103             `'kulsinski'` is deprecated from SciPy 1.9 and will be removed in SciPy 1.11. 
104   
105      p : float, default=2 
106          Parameter for the Minkowski metric from 
107          :class:`~sklearn.metrics.pairwise_distances`. When p = 1, this is 
108          equivalent to using manhattan_distance (l1), and euclidean_distance 
109          (l2) for p = 2. For arbitrary p, minkowski_distance (l_p) is used. 
110   
111      metric_params : dict, default=None 
112          Additional keyword arguments for the metric function. 
113   
114      cluster_method : str, default='xi' 
115          The extraction method used to extract clusters using the calculated 
116          reachability and ordering. Possible values are "xi" and "dbscan". 
117   
118      eps : float, default=None 
119          The maximum distance between two samples for one to be considered as 
120          in the neighborhood of the other. By default it assumes the same value 
121          as ``max_eps``. 
122          Used only when ``cluster_method='dbscan'``. 
123   
124      xi : float between 0 and 1, default=0.05 
125          Determines the minimum steepness on the reachability plot that 
126          constitutes a cluster boundary. For example, an upwards point in the 
127          reachability plot is defined by the ratio from one point to its 
128          successor being at most 1-xi. 
129          Used only when ``cluster_method='xi'``. 
130   
131      predecessor_correction : bool, default=True 
132          Correct clusters according to the predecessors calculated by OPTICS 
133          [2]_. This parameter has minimal effect on most datasets. 
134          Used only when ``cluster_method='xi'``. 
135   
136      min_cluster_size : int > 1 or float between 0 and 1, default=None 
137          Minimum number of samples in an OPTICS cluster, expressed as an 
138          absolute number or a fraction of the number of samples (rounded to be 
139          at least 2). If ``None``, the value of ``min_samples`` is used instead. 
140          Used only when ``cluster_method='xi'``. 
141   
142      algorithm : {'auto', 'ball_tree', 'kd_tree', 'brute'}, default='auto' 
143          Algorithm used to compute the nearest neighbors: 
144   
145          - 'ball_tree' will use :class:`~sklearn.neighbors.BallTree`. 
146          - 'kd_tree' will use :class:`~sklearn.neighbors.KDTree`. 
147          - 'brute' will use a brute-force search. 
148          - 'auto' (default) will attempt to decide the most appropriate 
149            algorithm based on the values passed to :meth:`fit` method. 
150   
151          Note: fitting on sparse input will override the setting of 
152          this parameter, using brute force. 
153   
154      leaf_size : int, default=30 
155          Leaf size passed to :class:`~sklearn.neighbors.BallTree` or 
156          :class:`~sklearn.neighbors.KDTree`. This can affect the speed of the 
157          construction and query, as well as the memory required to store the 
158          tree. The optimal value depends on the nature of the problem. 
159   
160      memory : str or object with the joblib.Memory interface, default=None 
161          Used to cache the output of the computation of the tree. 
162          By default, no caching is done. If a string is given, it is the 
163          path to the caching directory. 
164   
165      n_jobs : int, default=None 
166          The number of parallel jobs to run for neighbors search. 
167          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
168          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
169          for more details. 
170   
171      Attributes 
172      ---------- 
173      labels_ : ndarray of shape (n_samples,) 
174          Cluster labels for each point in the dataset given to fit(). 
175          Noisy samples and points which are not included in a leaf cluster 
176          of ``cluster_hierarchy_`` are labeled as -1. 
177   
178      reachability_ : ndarray of shape (n_samples,) 
179          Reachability distances per sample, indexed by object order. Use 
180          ``clust.reachability_[clust.ordering_]`` to access in cluster order. 
181   
182      ordering_ : ndarray of shape (n_samples,) 
183          The cluster ordered list of sample indices. 
184   
185      core_distances_ : ndarray of shape (n_samples,) 
186          Distance at which each sample becomes a core point, indexed by object 
187          order. Points which will never be core have a distance of inf. Use 
188          ``clust.core_distances_[clust.ordering_]`` to access in cluster order. 
189   
190      predecessor_ : ndarray of shape (n_samples,) 
191          Point that a sample was reached from, indexed by object order. 
192          Seed points have a predecessor of -1. 
193   
194      cluster_hierarchy_ : ndarray of shape (n_clusters, 2) 
195          The list of clusters in the form of ``[start, end]`` in each row, with 
196          all indices inclusive. The clusters are ordered according to 
197          ``(end, -start)`` (ascending) so that larger clusters encompassing 
198          smaller clusters come after those smaller ones. Since ``labels_`` does 
199          not reflect the hierarchy, usually 
200          ``len(cluster_hierarchy_) > np.unique(optics.labels_)``. Please also 
201          note that these indices are of the ``ordering_``, i.e. 
202          ``X[ordering_][start:end + 1]`` form a cluster. 
203          Only available when ``cluster_method='xi'``. 
204   
205      n_features_in_ : int 
206          Number of features seen during :term:`fit`. 
207   
208          .. versionadded:: 0.24 
209   
210      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
211          Names of features seen during :term:`fit`. Defined only when `X` 
212          has feature names that are all strings. 
213   
214          .. versionadded:: 1.0 
215   
216      See Also 
217      -------- 
218      DBSCAN : A similar clustering for a specified neighborhood radius (eps). 
219          Our implementation is optimized for runtime. 
220   
221      References 
222      ---------- 
223      .. [1] Ankerst, Mihael, Markus M. Breunig, Hans-Peter Kriegel, 
224         and JÃ¶rg Sander. "OPTICS: ordering points to identify the clustering 
225         structure." ACM SIGMOD Record 28, no. 2 (1999): 49-60. 
226   
227      .. [2] Schubert, Erich, Michael Gertz. 
228         "Improving the Cluster Structure Extracted from OPTICS Plots." Proc. of 
229         the Conference "Lernen, Wissen, Daten, Analysen" (LWDA) (2018): 318-329. 
230   
231      Examples 
232      -------- 
233      >>> from sklearn.cluster import OPTICS 
234      >>> import numpy as np 
235      >>> X = np.array([[1, 2], [2, 5], [3, 6], 
236      ...               [8, 7], [8, 8], [7, 3]]) 
237      >>> clustering = OPTICS(min_samples=2).fit(X) 
238      >>> clustering.labels_ 
239      array([0, 0, 0, 1, 1, 1]) 
240   
241      For a more detailed example see 
242      :ref:`sphx_glr_auto_examples_cluster_plot_optics.py`. 
243      """

302      @_fit_context(
303          # Optics.metric is not validated yet
304          prefer_skip_nested_validation=False
305      )
306      def fit(self, X, y=None):
307          """Perform OPTICS clustering. 
308   
309          Extracts an ordered list of points and reachability distances, and 
310          performs initial clustering using ``max_eps`` distance specified at 
311          OPTICS object instantiation. 
312   
313          Parameters 
314          ---------- 
315          X : {ndarray, sparse matrix} of shape (n_samples, n_features), or \ 
316                  (n_samples, n_samples) if metric='precomputed' 
317              A feature array, or array of distances between samples if 
318              metric='precomputed'. If a sparse matrix is provided, it will be 
319              converted into CSR format. 
320   
321          y : Ignored 
322              Not used, present for API consistency by convention. 
323   
324          Returns 
325          ------- 
326          self : object 
327              Returns a fitted instance of self. 
328          """
329          dtype = bool if self.metric in PAIRWISE_BOOLEAN_FUNCTIONS else float
330          if dtype is bool and X.dtype != bool:
331              msg = (
332                  "Data will be converted to boolean for"
333                  f" metric {self.metric}, to avoid this warning,"
334                  " you may convert the data prior to calling fit."
335              )
336              warnings.warn(msg, DataConversionWarning)
337  
338          X = validate_data(self, X, dtype=dtype, accept_sparse="csr")
339          if self.metric == "precomputed" and issparse(X):
340              X = X.copy()  # copy to avoid in-place modification
341              with warnings.catch_warnings():
342                  warnings.simplefilter("ignore", SparseEfficiencyWarning)
343                  # Set each diagonal to an explicit value so each point is its
344                  # own neighbor
345                  X.setdiag(X.diagonal())
346          memory = check_memory(self.memory)
347  
348          (
349              self.ordering_,
350              self.core_distances_,
351              self.reachability_,
352              self.predecessor_,
353          ) = memory.cache(compute_optics_graph)(
354              X=X,
355              min_samples=self.min_samples,
356              algorithm=self.algorithm,
357              leaf_size=self.leaf_size,
358              metric=self.metric,
359              metric_params=self.metric_params,
360              p=self.p,
361              n_jobs=self.n_jobs,
362              max_eps=self.max_eps,
363          )
364  
365          # Extract clusters from the calculated orders and reachability
366          if self.cluster_method == "xi":
367              labels_, clusters_ = cluster_optics_xi(
368                  reachability=self.reachability_,
369                  predecessor=self.predecessor_,
370                  ordering=self.ordering_,
371                  min_samples=self.min_samples,
372                  min_cluster_size=self.min_cluster_size,
373                  xi=self.xi,
374                  predecessor_correction=self.predecessor_correction,
375              )
376              self.cluster_hierarchy_ = clusters_
377          elif self.cluster_method == "dbscan":
378              if self.eps is None:
379                  eps = self.max_eps
380              else:
381                  eps = self.eps
382  
383              if eps > self.max_eps:
384                  raise ValueError(
385                      "Specify an epsilon smaller than %s. Got %s." % (self.max_eps, eps)
386                  )
387  
388              labels_ = cluster_optics_dbscan(
389                  reachability=self.reachability_,
390                  core_distances=self.core_distances_,
391                  ordering=self.ordering_,
392                  eps=eps,
393              )
394  
395          self.labels_ = labels_
396          return self

407  # OPTICS helper functions
408  def _compute_core_distances_(X, neighbors, min_samples, working_memory):
409      """Compute the k-th nearest neighbor of each sample. 
410   
411      Equivalent to neighbors.kneighbors(X, self.min_samples)[0][:, -1] 
412      but with more memory efficiency. 
413   
414      Parameters 
415      ---------- 
416      X : array-like of shape (n_samples, n_features) 
417          The data. 
418      neighbors : NearestNeighbors instance 
419          The fitted nearest neighbors estimator. 
420      working_memory : int, default=None 
421          The sought maximum memory for temporary distance matrix chunks. 
422          When None (default), the value of 
423          ``sklearn.get_config()['working_memory']`` is used. 
424   
425      Returns 
426      ------- 
427      core_distances : ndarray of shape (n_samples,) 
428          Distance at which each sample becomes a core point. 
429          Points which will never be core have a distance of inf. 
430      """
431      n_samples = X.shape[0]
432      core_distances = np.empty(n_samples)
433      core_distances.fill(np.nan)
434  
435      chunk_n_rows = get_chunk_n_rows(
436          row_bytes=16 * min_samples, max_n_rows=n_samples, working_memory=working_memory
437      )
438      slices = gen_batches(n_samples, chunk_n_rows)
439      for sl in slices:
440          core_distances[sl] = neighbors.kneighbors(X[sl], min_samples)[0][:, -1]
441      return core_distances

461  def compute_optics_graph(
462      X, *, min_samples, max_eps, metric, p, metric_params, algorithm, leaf_size, n_jobs
463  ):
464      """Compute the OPTICS reachability graph. 
465   
466      Read more in the :ref:`User Guide <optics>`. 
467   
468      Parameters 
469      ---------- 
470      X : {ndarray, sparse matrix} of shape (n_samples, n_features), or \ 
471              (n_samples, n_samples) if metric='precomputed' 
472          A feature array, or array of distances between samples if 
473          metric='precomputed'. 
474   
475      min_samples : int > 1 or float between 0 and 1 
476          The number of samples in a neighborhood for a point to be considered 
477          as a core point. Expressed as an absolute number or a fraction of the 
478          number of samples (rounded to be at least 2). 
479   
480      max_eps : float, default=np.inf 
481          The maximum distance between two samples for one to be considered as 
482          in the neighborhood of the other. Default value of ``np.inf`` will 
483          identify clusters across all scales; reducing ``max_eps`` will result 
484          in shorter run times. 
485   
486      metric : str or callable, default='minkowski' 
487          Metric to use for distance computation. Any metric from scikit-learn 
488          or scipy.spatial.distance can be used. 
489   
490          If metric is a callable function, it is called on each 
491          pair of instances (rows) and the resulting value recorded. The callable 
492          should take two arrays as input and return one value indicating the 
493          distance between them. This works for Scipy's metrics, but is less 
494          efficient than passing the metric name as a string. If metric is 
495          "precomputed", X is assumed to be a distance matrix and must be square. 
496   
497          Valid values for metric are: 
498   
499          - from scikit-learn: ['cityblock', 'cosine', 'euclidean', 'l1', 'l2', 
500            'manhattan'] 
501   
502          - from scipy.spatial.distance: ['braycurtis', 'canberra', 'chebyshev', 
503            'correlation', 'dice', 'hamming', 'jaccard', 'kulsinski', 
504            'mahalanobis', 'minkowski', 'rogerstanimoto', 'russellrao', 
505            'seuclidean', 'sokalmichener', 'sokalsneath', 'sqeuclidean', 
506            'yule'] 
507   
508          See the documentation for scipy.spatial.distance for details on these 
509          metrics. 
510   
511          .. note:: 
512             `'kulsinski'` is deprecated from SciPy 1.9 and will be removed in SciPy 1.11. 
513   
514      p : float, default=2 
515          Parameter for the Minkowski metric from 
516          :class:`~sklearn.metrics.pairwise_distances`. When p = 1, this is 
517          equivalent to using manhattan_distance (l1), and euclidean_distance 
518          (l2) for p = 2. For arbitrary p, minkowski_distance (l_p) is used. 
519   
520      metric_params : dict, default=None 
521          Additional keyword arguments for the metric function. 
522   
523      algorithm : {'auto', 'ball_tree', 'kd_tree', 'brute'}, default='auto' 
524          Algorithm used to compute the nearest neighbors: 
525   
526          - 'ball_tree' will use :class:`~sklearn.neighbors.BallTree`. 
527          - 'kd_tree' will use :class:`~sklearn.neighbors.KDTree`. 
528          - 'brute' will use a brute-force search. 
529          - 'auto' will attempt to decide the most appropriate algorithm 
530            based on the values passed to `fit` method. (default) 
531   
532          Note: fitting on sparse input will override the setting of 
533          this parameter, using brute force. 
534   
535      leaf_size : int, default=30 
536          Leaf size passed to :class:`~sklearn.neighbors.BallTree` or 
537          :class:`~sklearn.neighbors.KDTree`. This can affect the speed of the 
538          construction and query, as well as the memory required to store the 
539          tree. The optimal value depends on the nature of the problem. 
540   
541      n_jobs : int, default=None 
542          The number of parallel jobs to run for neighbors search. 
543          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
544          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
545          for more details. 
546   
547      Returns 
548      ------- 
549      ordering_ : array of shape (n_samples,) 
550          The cluster ordered list of sample indices. 
551   
552      core_distances_ : array of shape (n_samples,) 
553          Distance at which each sample becomes a core point, indexed by object 
554          order. Points which will never be core have a distance of inf. Use 
555          ``clust.core_distances_[clust.ordering_]`` to access in cluster order. 
556   
557      reachability_ : array of shape (n_samples,) 
558          Reachability distances per sample, indexed by object order. Use 
559          ``clust.reachability_[clust.ordering_]`` to access in cluster order. 
560   
561      predecessor_ : array of shape (n_samples,) 
562          Point that a sample was reached from, indexed by object order. 
563          Seed points have a predecessor of -1. 
564   
565      References 
566      ---------- 
567      .. [1] Ankerst, Mihael, Markus M. Breunig, Hans-Peter Kriegel, 
568         and JÃ¶rg Sander. "OPTICS: ordering points to identify the clustering 
569         structure." ACM SIGMOD Record 28, no. 2 (1999): 49-60. 
570   
571      Examples 
572      -------- 
573      >>> import numpy as np 
574      >>> from sklearn.cluster import compute_optics_graph 
575      >>> X = np.array([[1, 2], [2, 5], [3, 6], 
576      ...               [8, 7], [8, 8], [7, 3]]) 
577      >>> ordering, core_distances, reachability, predecessor = compute_optics_graph( 
578      ...     X, 
579      ...     min_samples=2, 
580      ...     max_eps=np.inf, 
581      ...     metric="minkowski", 
582      ...     p=2, 
583      ...     metric_params=None, 
584      ...     algorithm="auto", 
585      ...     leaf_size=30, 
586      ...     n_jobs=None, 
587      ... ) 
588      >>> ordering 
589      array([0, 1, 2, 5, 3, 4]) 
590      >>> core_distances 
591      array([3.16..., 1.41..., 1.41..., 1.        , 1.        , 
592             4.12...]) 
593      >>> reachability 
594      array([       inf, 3.16..., 1.41..., 4.12..., 1.        , 
595             5.        ]) 
596      >>> predecessor 
597      array([-1,  0,  1,  5,  3,  2]) 
598      """
599      n_samples = X.shape[0]
600      _validate_size(min_samples, n_samples, "min_samples")
601      if min_samples <= 1:
602          min_samples = max(2, int(min_samples * n_samples))
603  
604      # Start all points as 'unprocessed' ##
605      reachability_ = np.empty(n_samples)
606      reachability_.fill(np.inf)
607      predecessor_ = np.empty(n_samples, dtype=int)
608      predecessor_.fill(-1)
609  
610      nbrs = NearestNeighbors(
611          n_neighbors=min_samples,
612          algorithm=algorithm,
613          leaf_size=leaf_size,
614          metric=metric,
615          metric_params=metric_params,
616          p=p,
617          n_jobs=n_jobs,
618      )
619  
620      nbrs.fit(X)
621      # Here we first do a kNN query for each point, this differs from
622      # the original OPTICS that only used epsilon range queries.
623      # TODO: handle working_memory somehow?
624      core_distances_ = _compute_core_distances_(
625          X=X, neighbors=nbrs, min_samples=min_samples, working_memory=None
626      )
627      # OPTICS puts an upper limit on these, use inf for undefined.
628      core_distances_[core_distances_ > max_eps] = np.inf
629      np.around(
630          core_distances_,
631          decimals=np.finfo(core_distances_.dtype).precision,
632          out=core_distances_,
633      )
634  
635      # Main OPTICS loop. Not parallelizable. The order that entries are
636      # written to the 'ordering_' list is important!
637      # Note that this implementation is O(n^2) theoretically, but
638      # supposedly with very low constant factors.
639      processed = np.zeros(X.shape[0], dtype=bool)
640      ordering = np.zeros(X.shape[0], dtype=int)
641      for ordering_idx in range(X.shape[0]):
642          # Choose next based on smallest reachability distance
643          # (And prefer smaller ids on ties, possibly np.inf!)
644          index = np.where(processed == 0)[0]
645          point = index[np.argmin(reachability_[index])]
646  
647          processed[point] = True
648          ordering[ordering_idx] = point
649          if core_distances_[point] != np.inf:
650              _set_reach_dist(
651                  core_distances_=core_distances_,
652                  reachability_=reachability_,
653                  predecessor_=predecessor_,
654                  point_index=point,
655                  processed=processed,
656                  X=X,
657                  nbrs=nbrs,
658                  metric=metric,
659                  metric_params=metric_params,
660                  p=p,
661                  max_eps=max_eps,
662              )
663      if np.all(np.isinf(reachability_)):
664          warnings.warn(
665              (
666                  "All reachability values are inf. Set a larger"
667                  " max_eps or all data will be considered outliers."
668              ),
669              UserWarning,
670          )
671      return ordering, core_distances_, reachability_, predecessor_
672

674  def _set_reach_dist(
675      core_distances_,
676      reachability_,
677      predecessor_,
678      point_index,
679      processed,
680      X,
681      nbrs,
682      metric,
683      metric_params,
684      p,
685      max_eps,
686  ):
687      P = X[point_index : point_index + 1]
688      # Assume that radius_neighbors is faster without distances
689      # and we don't need all distances, nevertheless, this means
690      # we may be doing some work twice.
691      indices = nbrs.radius_neighbors(P, radius=max_eps, return_distance=False)[0]
692  
693      # Getting indices of neighbors that have not been processed
694      unproc = np.compress(~np.take(processed, indices), indices)
695      # Neighbors of current point are already processed.
696      if not unproc.size:
697          return
698  
699      # Only compute distances to unprocessed neighbors:
700      if metric == "precomputed":
701          dists = X[[point_index], unproc]
702          if isinstance(dists, np.matrix):
703              dists = np.asarray(dists)
704          dists = dists.ravel()
705      else:
706          _params = dict() if metric_params is None else metric_params.copy()
707          if metric == "minkowski" and "p" not in _params:
708              # the same logic as neighbors, p is ignored if explicitly set
709              # in the dict params
710              _params["p"] = p
711          dists = pairwise_distances(P, X[unproc], metric, n_jobs=None, **_params).ravel()
712  
713      rdists = np.maximum(dists, core_distances_[point_index])
714      np.around(rdists, decimals=np.finfo(rdists.dtype).precision, out=rdists)
715      improved = np.where(rdists < np.take(reachability_, unproc))
716      reachability_[unproc[improved]] = rdists[improved]
717      predecessor_[unproc[improved]] = point_index
718

729  def cluster_optics_dbscan(*, reachability, core_distances, ordering, eps):
730      """Perform DBSCAN extraction for an arbitrary epsilon. 
731   
732      Extracting the clusters runs in linear time. Note that this results in 
733      ``labels_`` which are close to a :class:`~sklearn.cluster.DBSCAN` with 
734      similar settings and ``eps``, only if ``eps`` is close to ``max_eps``. 
735   
736      Parameters 
737      ---------- 
738      reachability : ndarray of shape (n_samples,) 
739          Reachability distances calculated by OPTICS (``reachability_``). 
740   
741      core_distances : ndarray of shape (n_samples,) 
742          Distances at which points become core (``core_distances_``). 
743   
744      ordering : ndarray of shape (n_samples,) 
745          OPTICS ordered point indices (``ordering_``). 
746   
747      eps : float 
748          DBSCAN ``eps`` parameter. Must be set to < ``max_eps``. Results 
749          will be close to DBSCAN algorithm if ``eps`` and ``max_eps`` are close 
750          to one another. 
751   
752      Returns 
753      ------- 
754      labels_ : array of shape (n_samples,) 
755          The estimated labels. 
756   
757      Examples 
758      -------- 
759      >>> import numpy as np 
760      >>> from sklearn.cluster import cluster_optics_dbscan, compute_optics_graph 
761      >>> X = np.array([[1, 2], [2, 5], [3, 6], 
762      ...               [8, 7], [8, 8], [7, 3]]) 
763      >>> ordering, core_distances, reachability, predecessor = compute_optics_graph( 
764      ...     X, 
765      ...     min_samples=2, 
766      ...     max_eps=np.inf, 
767      ...     metric="minkowski", 
768      ...     p=2, 
769      ...     metric_params=None, 
770      ...     algorithm="auto", 
771      ...     leaf_size=30, 
772      ...     n_jobs=None, 
773      ... ) 
774      >>> eps = 4.5 
775      >>> labels = cluster_optics_dbscan( 
776      ...     reachability=reachability, 
777      ...     core_distances=core_distances, 
778      ...     ordering=ordering, 
779      ...     eps=eps, 
780      ... ) 
781      >>> labels 
782      array([0, 0, 0, 1, 1, 1]) 
783      """
784      n_samples = len(core_distances)
785      labels = np.zeros(n_samples, dtype=int)
786  
787      far_reach = reachability > eps
788      near_core = core_distances <= eps
789      labels[ordering] = np.cumsum(far_reach[ordering] & near_core[ordering]) - 1
790      labels[far_reach & ~near_core] = -1
791      return labels

813  def cluster_optics_xi(
814      *,
815      reachability,
816      predecessor,
817      ordering,
818      min_samples,
819      min_cluster_size=None,
820      xi=0.05,
821      predecessor_correction=True,
822  ):
823      """Automatically extract clusters according to the Xi-steep method. 
824   
825      Parameters 
826      ---------- 
827      reachability : ndarray of shape (n_samples,) 
828          Reachability distances calculated by OPTICS (`reachability_`). 
829   
830      predecessor : ndarray of shape (n_samples,) 
831          Predecessors calculated by OPTICS. 
832   
833      ordering : ndarray of shape (n_samples,) 
834          OPTICS ordered point indices (`ordering_`). 
835   
836      min_samples : int > 1 or float between 0 and 1 
837          The same as the min_samples given to OPTICS. Up and down steep regions 
838          can't have more then ``min_samples`` consecutive non-steep points. 
839          Expressed as an absolute number or a fraction of the number of samples 
840          (rounded to be at least 2). 
841   
842      min_cluster_size : int > 1 or float between 0 and 1, default=None 
843          Minimum number of samples in an OPTICS cluster, expressed as an 
844          absolute number or a fraction of the number of samples (rounded to be 
845          at least 2). If ``None``, the value of ``min_samples`` is used instead. 
846   
847      xi : float between 0 and 1, default=0.05 
848          Determines the minimum steepness on the reachability plot that 
849          constitutes a cluster boundary. For example, an upwards point in the 
850          reachability plot is defined by the ratio from one point to its 
851          successor being at most 1-xi. 
852   
853      predecessor_correction : bool, default=True 
854          Correct clusters based on the calculated predecessors. 
855   
856      Returns 
857      ------- 
858      labels : ndarray of shape (n_samples,) 
859          The labels assigned to samples. Points which are not included 
860          in any cluster are labeled as -1. 
861   
862      clusters : ndarray of shape (n_clusters, 2) 
863          The list of clusters in the form of ``[start, end]`` in each row, with 
864          all indices inclusive. The clusters are ordered according to ``(end, 
865          -start)`` (ascending) so that larger clusters encompassing smaller 
866          clusters come after such nested smaller clusters. Since ``labels`` does 
867          not reflect the hierarchy, usually ``len(clusters) > 
868          np.unique(labels)``. 
869   
870      Examples 
871      -------- 
872      >>> import numpy as np 
873      >>> from sklearn.cluster import cluster_optics_xi, compute_optics_graph 
874      >>> X = np.array([[1, 2], [2, 5], [3, 6], 
875      ...               [8, 7], [8, 8], [7, 3]]) 
876      >>> ordering, core_distances, reachability, predecessor = compute_optics_graph( 
877      ...     X, 
878      ...     min_samples=2, 
879      ...     max_eps=np.inf, 
880      ...     metric="minkowski", 
881      ...     p=2, 
882      ...     metric_params=None, 
883      ...     algorithm="auto", 
884      ...     leaf_size=30, 
885      ...     n_jobs=None 
886      ... ) 
887      >>> min_samples = 2 
888      >>> labels, clusters = cluster_optics_xi( 
889      ...     reachability=reachability, 
890      ...     predecessor=predecessor, 
891      ...     ordering=ordering, 
892      ...     min_samples=min_samples, 
893      ... ) 
894      >>> labels 
895      array([0, 0, 0, 1, 1, 1]) 
896      >>> clusters 
897      array([[0, 2], 
898             [3, 5], 
899             [0, 5]]) 
900      """
901      n_samples = len(reachability)
902      _validate_size(min_samples, n_samples, "min_samples")
903      if min_samples <= 1:
904          min_samples = max(2, int(min_samples * n_samples))
905      if min_cluster_size is None:
906          min_cluster_size = min_samples
907      _validate_size(min_cluster_size, n_samples, "min_cluster_size")
908      if min_cluster_size <= 1:
909          min_cluster_size = max(2, int(min_cluster_size * n_samples))
910  
911      clusters = _xi_cluster(
912          reachability[ordering],
913          predecessor[ordering],
914          ordering,
915          xi,
916          min_samples,
917          min_cluster_size,
918          predecessor_correction,
919      )
920      labels = _extract_xi_labels(ordering, clusters)
921      return labels, clusters

924  def _extend_region(steep_point, xward_point, start, min_samples):
925      """Extend the area until it's maximal. 
926   
927      It's the same function for both upward and downward reagions, depending on 
928      the given input parameters. Assuming: 
929   
930          - steep_{upward/downward}: bool array indicating whether a point is a 
931            steep {upward/downward}; 
932          - upward/downward: bool array indicating whether a point is 
933            upward/downward; 
934   
935      To extend an upward reagion, ``steep_point=steep_upward`` and 
936      ``xward_point=downward`` are expected, and to extend a downward region, 
937      ``steep_point=steep_downward`` and ``xward_point=upward``. 
938   
939      Parameters 
940      ---------- 
941      steep_point : ndarray of shape (n_samples,), dtype=bool 
942          True if the point is steep downward (upward). 
943   
944      xward_point : ndarray of shape (n_samples,), dtype=bool 
945          True if the point is an upward (respectively downward) point. 
946   
947      start : int 
948          The start of the xward region. 
949   
950      min_samples : int 
951         The same as the min_samples given to OPTICS. Up and down steep 
952         regions can't have more then ``min_samples`` consecutive non-steep 
953         points. 
954   
955      Returns 
956      ------- 
957      index : int 
958          The current index iterating over all the samples, i.e. where we are up 
959          to in our search. 
960   
961      end : int 
962          The end of the region, which can be behind the index. The region 
963          includes the ``end`` index. 
964      """
965      n_samples = len(steep_point)
966      non_xward_points = 0
967      index = start
968      end = start
969      # find a maximal area
970      while index < n_samples:
971          if steep_point[index]:
972              non_xward_points = 0
973              end = index
974          elif not xward_point[index]:
975              # it's not a steep point, but still goes up.
976              non_xward_points += 1
977              # region should include no more than min_samples consecutive
978              # non steep xward points.
979              if non_xward_points > min_samples:
980                  break
981          else:
982              return end
983          index += 1
984      return end
985

987  def _update_filter_sdas(sdas, mib, xi_complement, reachability_plot):
988      """Update steep down areas (SDAs) using the new maximum in between (mib) 
989      value, and the given complement of xi, i.e. ``1 - xi``. 
990      """
991      if np.isinf(mib):
992          return []
993      res = [
994          sda for sda in sdas if mib <= reachability_plot[sda["start"]] * xi_complement
995      ]
996      for sda in res:
997          sda["mib"] = max(sda["mib"], mib)
998      return res

1001 def _correct_predecessor(reachability_plot, predecessor_plot, ordering, s, e):
1002     """Correct for predecessors. 
1003  
1004     Applies Algorithm 2 of [1]_. 
1005  
1006     Input parameters are ordered by the computer OPTICS ordering. 
1007  
1008     .. [1] Schubert, Erich, Michael Gertz. 
1009        "Improving the Cluster Structure Extracted from OPTICS Plots." Proc. of 
1010        the Conference "Lernen, Wissen, Daten, Analysen" (LWDA) (2018): 318-329. 
1011     """
1012     while s < e:
1013         if reachability_plot[s] > reachability_plot[e]:
1014             return s, e
1015         p_e = predecessor_plot[e]
1016         for i in range(s, e):
1017             if p_e == ordering[i]:
1018                 return s, e
1019         e -= 1
1020     return None, None

1023 def _xi_cluster(
1024     reachability_plot,
1025     predecessor_plot,
1026     ordering,
1027     xi,
1028     min_samples,
1029     min_cluster_size,
1030     predecessor_correction,
1031 ):
1032     """Automatically extract clusters according to the Xi-steep method. 
1033  
1034     This is rouphly an implementation of Figure 19 of the OPTICS paper. 
1035  
1036     Parameters 
1037     ---------- 
1038     reachability_plot : array-like of shape (n_samples,) 
1039         The reachability plot, i.e. reachability ordered according to 
1040         the calculated ordering, all computed by OPTICS. 
1041  
1042     predecessor_plot : array-like of shape (n_samples,) 
1043         Predecessors ordered according to the calculated ordering. 
1044  
1045     xi : float, between 0 and 1 
1046         Determines the minimum steepness on the reachability plot that 
1047         constitutes a cluster boundary. For example, an upwards point in the 
1048         reachability plot is defined by the ratio from one point to its 
1049         successor being at most 1-xi. 
1050  
1051     min_samples : int > 1 
1052         The same as the min_samples given to OPTICS. Up and down steep regions 
1053         can't have more then ``min_samples`` consecutive non-steep points. 
1054  
1055     min_cluster_size : int > 1 
1056         Minimum number of samples in an OPTICS cluster. 
1057  
1058     predecessor_correction : bool 
1059         Correct clusters based on the calculated predecessors. 
1060  
1061     Returns 
1062     ------- 
1063     clusters : ndarray of shape (n_clusters, 2) 
1064         The list of clusters in the form of [start, end] in each row, with all 
1065         indices inclusive. The clusters are ordered in a way that larger 
1066         clusters encompassing smaller clusters come after those smaller 
1067         clusters. 
1068     """
1069 
1070     # Our implementation adds an inf to the end of reachability plot
1071     # this helps to find potential clusters at the end of the
1072     # reachability plot even if there's no upward region at the end of it.
1073     reachability_plot = np.hstack((reachability_plot, np.inf))
1074 
1075     xi_complement = 1 - xi
1076     sdas = []  # steep down areas, introduced in section 4.3.2 of the paper
1077     clusters = []
1078     index = 0
1079     mib = 0.0  # maximum in between, section 4.3.2
1080 
1081     # Our implementation corrects a mistake in the original
1082     # paper, i.e., in Definition 9 steep downward point,
1083     # r(p) * (1 - x1) <= r(p + 1) should be
1084     # r(p) * (1 - x1) >= r(p + 1)
1085     with np.errstate(invalid="ignore"):
1086         ratio = reachability_plot[:-1] / reachability_plot[1:]
1087         steep_upward = ratio <= xi_complement
1088         steep_downward = ratio >= 1 / xi_complement
1089         downward = ratio > 1
1090         upward = ratio < 1
1091 
1092     # the following loop is almost exactly as Figure 19 of the paper.
1093     # it jumps over the areas which are not either steep down or up areas
1094     for steep_index in iter(np.flatnonzero(steep_upward | steep_downward)):
1095         # just continue if steep_index has been a part of a discovered xward
1096         # area.
1097         if steep_index < index:
1098             continue
1099 
1100         mib = max(mib, np.max(reachability_plot[index : steep_index + 1]))
1101 
1102         # steep downward areas
1103         if steep_downward[steep_index]:
1104             sdas = _update_filter_sdas(sdas, mib, xi_complement, reachability_plot)
1105             D_start = steep_index
1106             D_end = _extend_region(steep_downward, upward, D_start, min_samples)
1107             D = {"start": D_start, "end": D_end, "mib": 0.0}
1108             sdas.append(D)
1109             index = D_end + 1
1110             mib = reachability_plot[index]
1111 
1112         # steep upward areas
1113         else:
1114             sdas = _update_filter_sdas(sdas, mib, xi_complement, reachability_plot)
1115             U_start = steep_index
1116             U_end = _extend_region(steep_upward, downward, U_start, min_samples)
1117             index = U_end + 1
1118             mib = reachability_plot[index]
1119 
1120             U_clusters = []
1121             for D in sdas:
1122                 c_start = D["start"]
1123                 c_end = U_end
1124 
1125                 # line (**), sc2*
1126                 if reachability_plot[c_end + 1] * xi_complement < D["mib"]:
1127                     continue
1128 
1129                 # Definition 11: criterion 4
1130                 D_max = reachability_plot[D["start"]]
1131                 if D_max * xi_complement >= reachability_plot[c_end + 1]:
1132                     # Find the first index from the left side which is almost
1133                     # at the same level as the end of the detected cluster.
1134                     while (
1135                         reachability_plot[c_start + 1] > reachability_plot[c_end + 1]
1136                         and c_start < D["end"]
1137                     ):
1138                         c_start += 1
1139                 elif reachability_plot[c_end + 1] * xi_complement >= D_max:
1140                     # Find the first index from the right side which is almost
1141                     # at the same level as the beginning of the detected
1142                     # cluster.
1143                     # Our implementation corrects a mistake in the original
1144                     # paper, i.e., in Definition 11 4c, r(x) < r(sD) should be
1145                     # r(x) > r(sD).
1146                     while reachability_plot[c_end - 1] > D_max and c_end > U_start:
1147                         c_end -= 1
1148 
1149                 # predecessor correction
1150                 if predecessor_correction:
1151                     c_start, c_end = _correct_predecessor(
1152                         reachability_plot, predecessor_plot, ordering, c_start, c_end
1153                     )
1154                 if c_start is None:
1155                     continue
1156 
1157                 # Definition 11: criterion 3.a
1158                 if c_end - c_start + 1 < min_cluster_size:
1159                     continue
1160 
1161                 # Definition 11: criterion 1
1162                 if c_start > D["end"]:
1163                     continue
1164 
1165                 # Definition 11: criterion 2
1166                 if c_end < U_start:
1167                     continue
1168 
1169                 U_clusters.append((c_start, c_end))
1170 
1171             # add smaller clusters first.
1172             U_clusters.reverse()
1173             clusters.extend(U_clusters)
1174 
1175     return np.array(clusters)
1176

1178 def _extract_xi_labels(ordering, clusters):
1179     """Extracts the labels from the clusters returned by `_xi_cluster`. 
1180     We rely on the fact that clusters are stored 
1181     with the smaller clusters coming before the larger ones. 
1182  
1183     Parameters 
1184     ---------- 
1185     ordering : array-like of shape (n_samples,) 
1186         The ordering of points calculated by OPTICS 
1187  
1188     clusters : array-like of shape (n_clusters, 2) 
1189         List of clusters i.e. (start, end) tuples, 
1190         as returned by `_xi_cluster`. 
1191  
1192     Returns 
1193     ------- 
1194     labels : ndarray of shape (n_samples,) 
1195     """
1196 
1197     labels = np.full(len(ordering), -1, dtype=int)
1198     label = 0
1199     for c in clusters:
1200         if not np.any(labels[c[0] : (c[1] + 1)] != -1):
1201             labels[c[0] : (c[1] + 1)] = label
1202             label += 1
1203     labels[ordering] = labels.copy()
1204     return labels
1205
