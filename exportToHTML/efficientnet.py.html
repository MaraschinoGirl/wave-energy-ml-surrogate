140  BASE_DOCSTRING = """Instantiates the {name} architecture. 
141   
142  Reference: 
143  - [EfficientNet: Rethinking Model Scaling for Convolutional Neural Networks]( 
144      https://arxiv.org/abs/1905.11946) (ICML 2019) 
145   
146  This function returns a Keras image classification model, 
147  optionally loaded with weights pre-trained on ImageNet. 
148   
149  For image classification use cases, see 
150  [this page for detailed examples]( 
151  https://keras.io/api/applications/#usage-examples-for-image-classification-models). 
152   
153  For transfer learning use cases, make sure to read the 
154  [guide to transfer learning & fine-tuning]( 
155  https://keras.io/guides/transfer_learning/). 
156   
157  Note: each Keras Application expects a specific kind of input preprocessing. 
158  For EfficientNet, input preprocessing is included as part of the model 
159  (as a `Rescaling` layer), and thus 
160  `keras.applications.efficientnet.preprocess_input` is actually a 
161  pass-through function. EfficientNet models expect their inputs to be float 
162  tensors of pixels with values in the `[0-255]` range. 
163   
164  Args: 
165      include_top: Whether to include the fully-connected 
166          layer at the top of the network. Defaults to `True`. 
167      weights: One of `None` (random initialization), 
168          `"imagenet"` (pre-training on ImageNet), 
169          or the path to the weights file to be loaded. 
170          Defaults to `"imagenet"`. 
171      input_tensor: Optional Keras tensor 
172          (i.e. output of `layers.Input()`) 
173          to use as image input for the model. 
174      input_shape: Optional shape tuple, only to be specified 
175          if `include_top` is False. 
176          It should have exactly 3 inputs channels. 
177      pooling: Optional pooling mode for feature extraction 
178          when `include_top` is `False`. Defaults to `None`. 
179          - `None` means that the output of the model will be 
180              the 4D tensor output of the 
181              last convolutional layer. 
182          - `avg` means that global average pooling 
183              will be applied to the output of the 
184              last convolutional layer, and thus 
185              the output of the model will be a 2D tensor. 
186          - `max` means that global max pooling will 
187              be applied. 
188      classes: Optional number of classes to classify images 
189          into, only to be specified if `include_top` is True, and 
190          if no `weights` argument is specified. 1000 is how many 
191          ImageNet classes there are. Defaults to `1000`. 
192      classifier_activation: A `str` or callable. The activation function to use 
193          on the "top" layer. Ignored unless `include_top=True`. Set 
194          `classifier_activation=None` to return the logits of the "top" layer. 
195          Defaults to `'softmax'`. 
196          When loading pretrained weights, `classifier_activation` can only 
197          be `None` or `"softmax"`. 
198      name: The name of the model (string). 
199   
200  Returns: 
201      A model instance. 
202  """

208  def EfficientNet(
209      width_coefficient,
210      depth_coefficient,
211      default_size,
212      dropout_rate=0.2,
213      drop_connect_rate=0.2,
214      depth_divisor=8,
215      activation="swish",
216      blocks_args="default",
217      name="efficientnet",
218      include_top=True,
219      weights="imagenet",
220      input_tensor=None,
221      input_shape=None,
222      pooling=None,
223      classes=1000,
224      classifier_activation="softmax",
225      weights_name=None,
226  ):
227      """Instantiates the EfficientNet architecture. 
228   
229      Args: 
230        width_coefficient: float, scaling coefficient for network width. 
231        depth_coefficient: float, scaling coefficient for network depth. 
232        default_size: integer, default input image size. 
233        dropout_rate: float, dropout rate before final classifier layer. 
234        drop_connect_rate: float, dropout rate at skip connections. 
235        depth_divisor: integer, a unit of network width. 
236        activation: activation function. 
237        blocks_args: list of dicts, parameters to construct block modules. 
238        name: string, model name. 
239        include_top: whether to include the fully-connected 
240            layer at the top of the network. 
241        weights: one of `None` (random initialization), 
242              'imagenet' (pre-training on ImageNet), 
243              or the path to the weights file to be loaded. 
244        input_tensor: optional Keras tensor 
245            (i.e. output of `layers.Input()`) 
246            to use as image input for the model. 
247        input_shape: optional shape tuple, only to be specified 
248            if `include_top` is False. 
249            It should have exactly 3 inputs channels. 
250        pooling: optional pooling mode for feature extraction 
251            when `include_top` is `False`. 
252            - `None` means that the output of the model will be 
253                the 4D tensor output of the 
254                last convolutional layer. 
255            - `avg` means that global average pooling 
256                will be applied to the output of the 
257                last convolutional layer, and thus 
258                the output of the model will be a 2D tensor. 
259            - `max` means that global max pooling will 
260                be applied. 
261        classes: optional number of classes to classify images 
262            into, only to be specified if `include_top` is True, and 
263            if no `weights` argument is specified. 
264        classifier_activation: A `str` or callable. The activation function to use 
265            on the "top" layer. Ignored unless `include_top=True`. Set 
266            `classifier_activation=None` to return the logits of the "top" layer. 
267   
268      Returns: 
269          A model instance. 
270      """
271      if blocks_args == "default":
272          blocks_args = DEFAULT_BLOCKS_ARGS
273  
274      if not (weights in {"imagenet", None} or file_utils.exists(weights)):
275          raise ValueError(
276              "The `weights` argument should be either "
277              "`None` (random initialization), `imagenet` "
278              "(pre-training on ImageNet), "
279              "or the path to the weights file to be loaded."
280          )
281  
282      if weights == "imagenet" and include_top and classes != 1000:
283          raise ValueError(
284              'If using `weights="imagenet"` with `include_top`'
285              " as true, `classes` should be 1000"
286          )
287  
288      # Determine proper input shape
289      input_shape = imagenet_utils.obtain_input_shape(
290          input_shape,
291          default_size=default_size,
292          min_size=32,
293          data_format=backend.image_data_format(),
294          require_flatten=include_top,
295          weights=weights,
296      )
297  
298      if input_tensor is None:
299          img_input = layers.Input(shape=input_shape)
300      else:
301          if not backend.is_keras_tensor(input_tensor):
302              img_input = layers.Input(tensor=input_tensor, shape=input_shape)
303          else:
304              img_input = input_tensor
305  
306      bn_axis = 3 if backend.image_data_format() == "channels_last" else 1
307  
308      def round_filters(filters, divisor=depth_divisor):
309          """Round number of filters based on depth multiplier."""
310          filters *= width_coefficient
311          new_filters = max(
312              divisor, int(filters + divisor / 2) // divisor * divisor
313          )
314          # Make sure that round down does not go down by more than 10%.
315          if new_filters < 0.9 * filters:
316              new_filters += divisor
317          return int(new_filters)
318  
319      def round_repeats(repeats):
320          """Round number of repeats based on depth multiplier."""
321          return int(math.ceil(depth_coefficient * repeats))
322  
323      # Build stem
324      x = img_input
325      x = layers.Rescaling(1.0 / 255.0)(x)
326      x = layers.Normalization(axis=bn_axis)(x)
327  
328      if weights == "imagenet":
329          # Note that the normalization layer uses square value of STDDEV as the
330          # variance for the layer: result = (input - mean) / sqrt(var)
331          # However, the original implementation uses (input - mean) / var to
332          # normalize the input, we need to divide another sqrt(var) to match the
333          # original implementation.
334          # See https://github.com/tensorflow/tensorflow/issues/49930 for more
335          # details
336          x = layers.Rescaling(
337              [1.0 / math.sqrt(stddev) for stddev in IMAGENET_STDDEV_RGB]
338          )(x)
339  
340      x = layers.ZeroPadding2D(
341          padding=imagenet_utils.correct_pad(x, 3), name="stem_conv_pad"
342      )(x)
343      x = layers.Conv2D(
344          round_filters(32),
345          3,
346          strides=2,
347          padding="valid",
348          use_bias=False,
349          kernel_initializer=CONV_KERNEL_INITIALIZER,
350          name="stem_conv",
351      )(x)
352      x = layers.BatchNormalization(axis=bn_axis, name="stem_bn")(x)
353      x = layers.Activation(activation, name="stem_activation")(x)
354  
355      # Build blocks
356      blocks_args = copy.deepcopy(blocks_args)
357  
358      b = 0
359      blocks = float(sum(round_repeats(args["repeats"]) for args in blocks_args))
360      for i, args in enumerate(blocks_args):
361          assert args["repeats"] > 0
362          # Update block input and output filters based on depth multiplier.
363          args["filters_in"] = round_filters(args["filters_in"])
364          args["filters_out"] = round_filters(args["filters_out"])
365  
366          for j in range(round_repeats(args.pop("repeats"))):
367              # The first block needs to take care of stride and filter size
368              # increase.
369              if j > 0:
370                  args["strides"] = 1
371                  args["filters_in"] = args["filters_out"]
372              x = block(
373                  x,
374                  activation,
375                  drop_connect_rate * b / blocks,
376                  name=f"block{i + 1}{chr(j + 97)}_",
377                  **args,
378              )
379              b += 1
380  
381      # Build top
382      x = layers.Conv2D(
383          round_filters(1280),
384          1,
385          padding="same",
386          use_bias=False,
387          kernel_initializer=CONV_KERNEL_INITIALIZER,
388          name="top_conv",
389      )(x)
390      x = layers.BatchNormalization(axis=bn_axis, name="top_bn")(x)
391      x = layers.Activation(activation, name="top_activation")(x)
392      if include_top:
393          x = layers.GlobalAveragePooling2D(name="avg_pool")(x)
394          if dropout_rate > 0:
395              x = layers.Dropout(dropout_rate, name="top_dropout")(x)
396          imagenet_utils.validate_activation(classifier_activation, weights)
397          x = layers.Dense(
398              classes,
399              activation=classifier_activation,
400              kernel_initializer=DENSE_KERNEL_INITIALIZER,
401              name="predictions",
402          )(x)
403      else:
404          if pooling == "avg":
405              x = layers.GlobalAveragePooling2D(name="avg_pool")(x)
406          elif pooling == "max":
407              x = layers.GlobalMaxPooling2D(name="max_pool")(x)
408  
409      # Ensure that the model takes into account
410      # any potential predecessors of `input_tensor`.
411      if input_tensor is not None:
412          inputs = operation_utils.get_source_inputs(input_tensor)
413      else:
414          inputs = img_input
415  
416      # Create model.
417      model = Functional(inputs, x, name=name)
418  
419      # Load weights.
420      if weights == "imagenet":
421          if include_top:
422              file_suffix = ".h5"
423              file_hash = WEIGHTS_HASHES[weights_name][0]
424          else:
425              file_suffix = "_notop.h5"
426              file_hash = WEIGHTS_HASHES[weights_name][1]
427          file_name = name + file_suffix
428          weights_path = file_utils.get_file(
429              file_name,
430              BASE_WEIGHTS_PATH + file_name,
431              cache_subdir="models",
432              file_hash=file_hash,
433          )
434          model.load_weights(weights_path)
435      elif weights is not None:
436          model.load_weights(weights)
437      return model
438

440  def block(
441      inputs,
442      activation="swish",
443      drop_rate=0.0,
444      name="",
445      filters_in=32,
446      filters_out=16,
447      kernel_size=3,
448      strides=1,
449      expand_ratio=1,
450      se_ratio=0.0,
451      id_skip=True,
452  ):
453      """An inverted residual block. 
454   
455      Args: 
456          inputs: input tensor. 
457          activation: activation function. 
458          drop_rate: float between 0 and 1, fraction of the input units to drop. 
459          name: string, block label. 
460          filters_in: integer, the number of input filters. 
461          filters_out: integer, the number of output filters. 
462          kernel_size: integer, the dimension of the convolution window. 
463          strides: integer, the stride of the convolution. 
464          expand_ratio: integer, scaling coefficient for the input filters. 
465          se_ratio: float between 0 and 1, fraction to squeeze the input filters. 
466          id_skip: boolean. 
467   
468      Returns: 
469          output tensor for the block. 
470      """
471      bn_axis = 3 if backend.image_data_format() == "channels_last" else 1
472  
473      # Expansion phase
474      filters = filters_in * expand_ratio
475      if expand_ratio != 1:
476          x = layers.Conv2D(
477              filters,
478              1,
479              padding="same",
480              use_bias=False,
481              kernel_initializer=CONV_KERNEL_INITIALIZER,
482              name=name + "expand_conv",
483          )(inputs)
484          x = layers.BatchNormalization(axis=bn_axis, name=name + "expand_bn")(x)
485          x = layers.Activation(activation, name=name + "expand_activation")(x)
486      else:
487          x = inputs
488  
489      # Depthwise Convolution
490      if strides == 2:
491          x = layers.ZeroPadding2D(
492              padding=imagenet_utils.correct_pad(x, kernel_size),
493              name=name + "dwconv_pad",
494          )(x)
495          conv_pad = "valid"
496      else:
497          conv_pad = "same"
498      x = layers.DepthwiseConv2D(
499          kernel_size,
500          strides=strides,
501          padding=conv_pad,
502          use_bias=False,
503          depthwise_initializer=CONV_KERNEL_INITIALIZER,
504          name=name + "dwconv",
505      )(x)
506      x = layers.BatchNormalization(axis=bn_axis, name=name + "bn")(x)
507      x = layers.Activation(activation, name=name + "activation")(x)
508  
509      # Squeeze and Excitation phase
510      if 0 < se_ratio <= 1:
511          filters_se = max(1, int(filters_in * se_ratio))
512          se = layers.GlobalAveragePooling2D(name=name + "se_squeeze")(x)
513          if bn_axis == 1:
514              se_shape = (filters, 1, 1)
515          else:
516              se_shape = (1, 1, filters)
517          se = layers.Reshape(se_shape, name=name + "se_reshape")(se)
518          se = layers.Conv2D(
519              filters_se,
520              1,
521              padding="same",
522              activation=activation,
523              kernel_initializer=CONV_KERNEL_INITIALIZER,
524              name=name + "se_reduce",
525          )(se)
526          se = layers.Conv2D(
527              filters,
528              1,
529              padding="same",
530              activation="sigmoid",
531              kernel_initializer=CONV_KERNEL_INITIALIZER,
532              name=name + "se_expand",
533          )(se)
534          x = layers.multiply([x, se], name=name + "se_excite")
535  
536      # Output phase
537      x = layers.Conv2D(
538          filters_out,
539          1,
540          padding="same",
541          use_bias=False,
542          kernel_initializer=CONV_KERNEL_INITIALIZER,
543          name=name + "project_conv",
544      )(x)
545      x = layers.BatchNormalization(axis=bn_axis, name=name + "project_bn")(x)
546      if id_skip and strides == 1 and filters_in == filters_out:
547          if drop_rate > 0:
548              x = layers.Dropout(
549                  drop_rate, noise_shape=(None, 1, 1, 1), name=name + "drop"
550              )(x)
551          x = layers.add([x, inputs], name=name + "add")
552      return x
