25   class CookieJarTest(unittest.TestCase):
26       def test_add(self):
27           cookie_jar = SimpleCookieJar()
28           cookie_jar.add("")
29           self.assertFalse(
30               cookie_jar.jar, "Cookie with no domain should not be added to the jar"
31           )
32   
33           cookie_jar = SimpleCookieJar()
34           cookie_jar.add("a=b")
35           self.assertFalse(
36               cookie_jar.jar, "Cookie with no domain should not be added to the jar"
37           )
38   
39           cookie_jar = SimpleCookieJar()
40           cookie_jar.add("a=b; domain=.abc")
41           self.assertTrue(".abc" in cookie_jar.jar)
42   
43           cookie_jar = SimpleCookieJar()
44           cookie_jar.add("a=b; domain=abc")
45           self.assertTrue(".abc" in cookie_jar.jar)
46           self.assertTrue("abc" not in cookie_jar.jar)
47   
48           cookie_jar = SimpleCookieJar()
49           cookie_jar.add("a=b; c=d; domain=abc")
50           self.assertEqual(cookie_jar.get("abc"), "a=b; c=d")
51           self.assertEqual(cookie_jar.get(None), "")
52   
53           cookie_jar = SimpleCookieJar()
54           cookie_jar.add("a=b; c=d; domain=abc")
55           cookie_jar.add("e=f; domain=abc")
56           self.assertEqual(cookie_jar.get("abc"), "a=b; c=d; e=f")
57   
58           cookie_jar = SimpleCookieJar()
59           cookie_jar.add("a=b; c=d; domain=abc")
60           cookie_jar.add("e=f; domain=.abc")
61           self.assertEqual(cookie_jar.get("abc"), "a=b; c=d; e=f")
62   
63           cookie_jar = SimpleCookieJar()
64           cookie_jar.add("a=b; c=d; domain=abc")
65           cookie_jar.add("e=f; domain=xyz")
66           self.assertEqual(cookie_jar.get("abc"), "a=b; c=d")
67           self.assertEqual(cookie_jar.get("xyz"), "e=f")
68           self.assertEqual(cookie_jar.get("something"), "")
69

107      def test_get(self):
108          cookie_jar = SimpleCookieJar()
109          cookie_jar.set("a=b; c=d; domain=abc.com")
110          self.assertEqual(cookie_jar.get("abc.com"), "a=b; c=d")
111          self.assertEqual(cookie_jar.get("x.abc.com"), "a=b; c=d")
112          self.assertEqual(cookie_jar.get("abc.com.es"), "")
113          self.assertEqual(cookie_jar.get("xabc.com"), "")
114  
115          cookie_jar.set("a=b; c=d; domain=.abc.com")
116          self.assertEqual(cookie_jar.get("abc.com"), "a=b; c=d")
117          self.assertEqual(cookie_jar.get("x.abc.com"), "a=b; c=d")
118          self.assertEqual(cookie_jar.get("abc.com.es"), "")
