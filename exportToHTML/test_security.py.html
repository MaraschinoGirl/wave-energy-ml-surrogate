13   class SecurityTests(unittest.TestCase):
14       def setUp(self):
15           self.pwr_sid = win32security.LookupAccountName("", "Power Users")[0]
16           try:
17               self.admin_sid = win32security.LookupAccountName("", "Administrator")[0]
18           except pywintypes.error as exc:
19               # in automation we see:
20               # pywintypes.error: (1332, 'LookupAccountName', 'No mapping between account names and security IDs was done.')
21               if exc.winerror != winerror.ERROR_NONE_MAPPED:
22                   raise
23               self.admin_sid = None
24   
25       def tearDown(self):
26           pass
27   
28       def testEqual(self):
29           if self.admin_sid is None:
30               raise TestSkipped("No 'Administrator' account is available")
31           self.assertEqual(
32               win32security.LookupAccountName("", "Administrator")[0],
33               win32security.LookupAccountName("", "Administrator")[0],
34           )
35   
36       def testNESID(self):
37           self.assertTrue(self.pwr_sid == self.pwr_sid)
38           if self.admin_sid:
39               self.assertTrue(self.pwr_sid != self.admin_sid)
40   
41       def testNEOther(self):
42           self.assertTrue(self.pwr_sid is not None)
43           self.assertTrue(None != self.pwr_sid)
44           self.assertFalse(self.pwr_sid is None)
45           self.assertFalse(None == self.pwr_sid)
46           self.assertNotEqual(None, self.pwr_sid)
47   
48       def testSIDInDict(self):
49           d = {"foo": self.pwr_sid}
50           self.assertEqual(d["foo"], self.pwr_sid)
51   
52       def testBuffer(self):
53           if self.admin_sid is None:
54               raise TestSkipped("No 'Administrator' account is available")
55           self.assertEqual(
56               memoryview(win32security.LookupAccountName("", "Administrator")[0]),
57               memoryview(win32security.LookupAccountName("", "Administrator")[0]),
58           )
59   
60       def testMemory(self):
61           pwr_sid = self.pwr_sid
62           admin_sid = self.admin_sid
63           sd1 = win32security.SECURITY_DESCRIPTOR()
64           sd2 = win32security.SECURITY_DESCRIPTOR()
65           sd3 = win32security.SECURITY_DESCRIPTOR()
66           dacl = win32security.ACL()
67           dacl.AddAccessAllowedAce(
68               win32security.ACL_REVISION, win32con.GENERIC_READ, pwr_sid
69           )
70           if admin_sid is not None:
71               dacl.AddAccessAllowedAce(
72                   win32security.ACL_REVISION, win32con.GENERIC_ALL, admin_sid
73               )
74           sd4 = win32security.SECURITY_DESCRIPTOR()
75           sacl = win32security.ACL()
76           if admin_sid is not None:
77               sacl.AddAuditAccessAce(
78                   win32security.ACL_REVISION, win32con.DELETE, admin_sid, 1, 1
79               )
80           sacl.AddAuditAccessAce(
81               win32security.ACL_REVISION, win32con.GENERIC_ALL, pwr_sid, 1, 1
82           )
83           for x in range(0, 200000):
84               if admin_sid is not None:
85                   sd1.SetSecurityDescriptorOwner(admin_sid, 0)
86               sd2.SetSecurityDescriptorGroup(pwr_sid, 0)
87               sd3.SetSecurityDescriptorDacl(1, dacl, 0)
88               sd4.SetSecurityDescriptorSacl(1, sacl, 0)

91   class DomainTests(unittest.TestCase):
92       def setUp(self):
93           self.ds_handle = None
94           try:
95               # saving the handle means the other test itself should bind faster.
96               self.ds_handle = win32security.DsBind()
97           except win32security.error as exc:
98               if exc.winerror != winerror.ERROR_NO_SUCH_DOMAIN:
99                   raise
100              raise TestSkipped(exc)
101  
102      def tearDown(self):
103          if self.ds_handle is not None:
104              self.ds_handle.close()

122      def testDsCrackNames(self):
123          h = win32security.DsBind()
124          fmt_offered = ntsecuritycon.DS_FQDN_1779_NAME
125          name = win32api.GetUserNameEx(fmt_offered)
126          result = win32security.DsCrackNames(h, 0, fmt_offered, fmt_offered, (name,))
127          self.assertEqual(name, result[0][2])

129      def testDsCrackNamesSyntax(self):
130          # Do a syntax check only - that allows us to avoid binding.
131          # But must use DS_CANONICAL_NAME (or _EX)
132          expected = win32api.GetUserNameEx(win32api.NameCanonical)
133          fmt_offered = ntsecuritycon.DS_FQDN_1779_NAME
134          name = win32api.GetUserNameEx(fmt_offered)
135          result = win32security.DsCrackNames(
136              None,
137              ntsecuritycon.DS_NAME_FLAG_SYNTACTICAL_ONLY,
138              fmt_offered,
139              ntsecuritycon.DS_CANONICAL_NAME,
140              (name,),
141          )
142          self.assertEqual(expected, result[0][2])
