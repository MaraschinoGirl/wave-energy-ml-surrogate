23   class ExplorerEvents:
24       def OnVisible(self, visible):
25           global bVisibleEventFired
26           bVisibleEventFired = 1

29   def TestExplorerEvents():
30       global bVisibleEventFired
31       try:
32           iexplore = win32com.client.DispatchWithEvents(
33               "InternetExplorer.Application", ExplorerEvents
34           )
35       except pythoncom.com_error as exc:
36           # In automation we see this error trying to connect to events
37           # It's a little surprising that the non-event tests seem to work, but
38           # whatever...
39           if exc.hresult not in HRESULTS_IN_AUTOMATION:
40               raise
41           print("IE events appear to not be available, so skipping this test")
42           return
43   
44       iexplore.Visible = 1
45       assert bVisibleEventFired, "The IE event did not appear to fire!"
46       iexplore.Quit()
47       iexplore = None
48   
49       bVisibleEventFired = 0
50       ie = win32com.client.Dispatch("InternetExplorer.Application")
51       ie_events = win32com.client.DispatchWithEvents(ie, ExplorerEvents)
52       ie.Visible = 1
53       assert bVisibleEventFired, "The IE event did not appear to fire!"
54       ie.Quit()
55       ie = None
56       print("IE Event tests worked.")

59   def TestObjectFromWindow():
60       # Check we can use ObjectFromLresult to get the COM object from the
61       hwnd = win32gui.FindWindow("IEFrame", None)
62       # Thanks https://stackoverflow.com/a/10154498/18450412 for the child stack on IE8+
63       for child_class in (
64           "Frame Tab",
65           "TabWindowClass",
66           "Shell DocObject View",
67           "Internet Explorer_Server",
68       ):
69           hwnd = win32gui.FindWindowEx(hwnd, 0, child_class, None)
70       # Once you have an 'Internet Explorer_Server',
71       # you can send a message and use ObjectFromLresult to get it back.
72       msg = win32gui.RegisterWindowMessage("WM_HTML_GETOBJECT")
73       rc, result = win32gui.SendMessageTimeout(
74           hwnd, msg, 0, 0, win32con.SMTO_ABORTIFHUNG, 1000
75       )
76       ob = pythoncom.ObjectFromLresult(result, pythoncom.IID_IDispatch, 0)
77       doc = Dispatch(ob)
78       # just to prove it works, set the background color of the document.
79       for color in "red green blue orange white".split():
80           doc.bgColor = color
81           time.sleep(0.2)

84   def TestExplorer(iexplore):
85       if not iexplore.Visible:
86           iexplore.Visible = -1
87       filename = os.path.join(os.path.dirname(__file__), "..\\readme.html")
88       iexplore.Navigate(win32api.GetFullPathName(filename))
89       win32api.Sleep(1000)
90       TestObjectFromWindow()
91       win32api.Sleep(3000)
92       try:
93           iexplore.Quit()
94       except (AttributeError, pythoncom.com_error):
95           # User got sick of waiting :)
96           pass
