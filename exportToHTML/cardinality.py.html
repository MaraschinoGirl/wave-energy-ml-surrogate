14   """Defines an enum for classifying RPC methods by streaming semantics."""
15   
16   import enum
17   
18   
19   @enum.unique
20   class Cardinality(enum.Enum):
21       """Describes the streaming semantics of an RPC method."""
22   
23       UNARY_UNARY = "request-unary/response-unary"
24       UNARY_STREAM = "request-unary/response-streaming"
25       STREAM_UNARY = "request-streaming/response-unary"
26       STREAM_STREAM = "request-streaming/response-streaming"
