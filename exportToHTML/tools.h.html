55       // Series evaluators.
56       template <typename Generator, typename T = generator_result_t<Generator>>
57       XSF_HOST_DEVICE T
58       series_eval(Generator &g, T init_val, real_type_t<T> tol, std::uint64_t max_terms, const char *func_name) {
59           /* Sum an infinite series to a given precision. 
60            * 
61            * g : a generator of terms for the series. 
62            * 
63            * init_val : A starting value that terms are added to. This argument determines the 
64            *     type of the result. 
65            * 
66            * tol : relative tolerance for stopping criterion. 
67            * 
68            * max_terms : The maximum number of terms to add before giving up and declaring 
69            *     non-convergence. 
70            * 
71            * func_name : The name of the function within SciPy where this call to series_eval 
72            *     will ultimately be used. This is needed to pass to set_error in case 
73            *     of non-convergence. 
74            */
75           T result = init_val;
76           T term;
77           for (std::uint64_t i = 0; i < max_terms; ++i) {
78               term = g();
79               result += term;
80               if (std::abs(term) < std::abs(result) * tol) {
81                   return result;
82               }
83           }
84           // Exceeded max terms without converging. Return NaN.
85           set_error(func_name, SF_ERROR_NO_RESULT, NULL);
86           return maybe_complex_NaN<T>();
87       }

89       template <typename Generator, typename T = generator_result_t<Generator>>
90       XSF_HOST_DEVICE T series_eval_fixed_length(Generator &g, T init_val, std::uint64_t num_terms) {
91           /* Sum a fixed number of terms from a series. 
92            * 
93            * g : a generator of terms for the series. 
94            * 
95            * init_val : A starting value that terms are added to. This argument determines the 
96            *     type of the result. 
97            * 
98            * max_terms : The number of terms from the series to sum. 
99            * 
100           */
101          T result = init_val;
102          for (std::uint64_t i = 0; i < num_terms; ++i) {
103              result += g();
104          }
105          return result;
106      }

108      /* Performs one step of Kahan summation. */
109      template <typename T>
110      XSF_HOST_DEVICE void kahan_step(T &sum, T &comp, T x) {
111          T y = x - comp;
112          T t = sum + y;
113          comp = (t - sum) - y;
114          sum = t;
115      }

117      /* Evaluates an infinite series using Kahan summation. 
118       * 
119       * Denote the series by 
120       * 
121       *   S = a[0] + a[1] + a[2] + ... 
122       * 
123       * And for n = 0, 1, 2, ..., denote its n-th partial sum by 
124       * 
125       *   S[n] = a[0] + a[1] + ... + a[n] 
126       * 
127       * This function computes S[0], S[1], ... until a[n] is sufficiently 
128       * small or if the maximum number of terms have been evaluated. 
129       * 
130       * Parameters 
131       * ---------- 
132       *   g 
133       *       Reference to generator that yields the sequence of values a[1], 
134       *       a[2], a[3], ... 
135       * 
136       *   tol 
137       *       Relative tolerance for convergence.  Specifically, stop iteration 
138       *       as soon as `abs(a[n]) <= tol * abs(S[n])` for some n >= 1. 
139       * 
140       *   max_terms 
141       *       Maximum number of terms after a[0] to evaluate.  It should be set 
142       *       large enough such that the convergence criterion is guaranteed 
143       *       to have been satisfied within that many terms if there is no 
144       *       rounding error. 
145       * 
146       *   init_val 
147       *       a[0].  Default is zero.  The type of this parameter (T) is used 
148       *       for intermediary computations as well as the result. 
149       * 
150       * Return Value 
151       * ------------ 
152       * If the convergence criterion is satisfied by some `n <= max_terms`, 
153       * returns `(S[n], n)`.  Otherwise, returns `(S[max_terms], 0)`. 
154       */
155      template <typename Generator, typename T = generator_result_t<Generator>>
156      XSF_HOST_DEVICE std::pair<T, std::uint64_t>
157      series_eval_kahan(Generator &&g, real_type_t<T> tol, std::uint64_t max_terms, T init_val = T(0)) {
158  
159          using std::abs;
160          T sum = init_val;
161          T comp = T(0);
162          for (std::uint64_t i = 0; i < max_terms; ++i) {
163              T term = g();
164              kahan_step(sum, comp, term);
165              if (abs(term) <= tol * abs(sum)) {
166                  return {sum, i + 1};
167              }
168          }
169          return {sum, 0};
170      }

172      /* Generator that yields the difference of successive convergents of a 
173       * continued fraction. 
174       * 
175       * Let f[n] denote the n-th convergent of a continued fraction: 
176       * 
177       *                 a[1]   a[2]       a[n] 
178       *   f[n] = b[0] + ------ ------ ... ---- 
179       *                 b[1] + b[2] +     b[n] 
180       * 
181       * with f[0] = b[0].  This generator yields the sequence of values 
182       * f[1]-f[0], f[2]-f[1], f[3]-f[2], ... 
183       * 
184       * Constructor Arguments 
185       * --------------------- 
186       *   cf 
187       *       Reference to generator that yields the terms of the continued 
188       *       fraction as (numerator, denominator) pairs, starting from 
189       *       (a[1], b[1]). 
190       * 
191       *       `cf` must outlive the ContinuedFractionSeriesGenerator object. 
192       * 
193       *       The constructed object always eagerly retrieves the next term 
194       *       of the continued fraction.  Specifically, (a[1], b[1]) is 
195       *       retrieved upon construction, and (a[n], b[n]) is retrieved after 
196       *       (n-1) calls of `()`. 
197       * 
198       * Type Arguments 
199       * -------------- 
200       *   T 
201       *       Type in which computations are performed and results are turned. 
202       * 
203       * Remarks 
204       * ------- 
205       * The series is computed using the recurrence relation described in [1]. 
206       * Let v[n], n >= 1 denote the terms of the series.  Then 
207       * 
208       *   v[1] = a[1] / b[1] 
209       *   v[n] = v[n-1] * r[n-1], n >= 2 
210       * 
211       * where 
212       * 
213       *                              -(a[n] + a[n] * r[n-1]) 
214       *   r[1] = 0,  r[n] = ------------------------------------------, n >= 2 
215       *                      (a[n] + a[n] * r[n-1]) + (b[n] * b[n-1]) 
216       * 
217       * No error checking is performed.  The caller must ensure that all terms 
218       * are finite and that intermediary computations do not trigger floating 
219       * point exceptions such as overflow. 
220       * 
221       * The numerical stability of this method depends on the characteristics 
222       * of the continued fraction being evaluated. 
223       * 
224       * Reference 
225       * --------- 
226       * [1] Gautschi, W. (1967). “Computational Aspects of Three-Term 
227       *     Recurrence Relations.” SIAM Review, 9(1):24-82. 
228       */
229      template <typename Generator, typename T = pair_value_t<generator_result_t<Generator>>>
230      class ContinuedFractionSeriesGenerator {
231  
232        public:
233          XSF_HOST_DEVICE explicit ContinuedFractionSeriesGenerator(Generator &cf) : cf_(cf) { init(); }
234  
235          XSF_HOST_DEVICE T operator()() {
236              T v = v_;
237              advance();
238              return v;
239          }
240  
241        private:
242          XSF_HOST_DEVICE void init() {
243              auto [num, denom] = cf_();
244              T a = num;
245              T b = denom;
246              r_ = T(0);
247              v_ = a / b;
248              b_ = b;
249          }
250  
251          XSF_HOST_DEVICE void advance() {
252              auto [num, denom] = cf_();
253              T a = num;
254              T b = denom;
255              T p = a + a * r_;
256              T q = p + b * b_;
257              r_ = -p / q;
258              v_ = v_ * r_;
259              b_ = b;
260          }
261  
262          Generator &cf_; // reference to continued fraction generator
263          T v_;           // v[n] == f[n] - f[n-1], n >= 1
264          T r_;           // r[1] = 0, r[n] = v[n]/v[n-1], n >= 2
265          T b_;           // last denominator, i.e. b[n-1]
266      };

274      XSF_HOST_DEVICE ContinuedFractionSeriesGenerator<Generator, T> continued_fraction_series(Generator &cf) {

278      /* Find initial bracket for a bracketing scalar root finder. A valid bracket is a pair of points a < b for 
279       * which the signs of f(a) and f(b) differ. If f(x0) = 0, where x0 is the initial guess, this bracket finder 
280       * will return the bracket (x0, x0). It is expected that the rootfinder will check if the bracket 
281       * endpoints are roots. 
282       * 
283       * This is a private function intended specifically for the situation where 
284       * the goal is to invert a CDF function F for a parametrized family of distributions with respect to one 
285       * parameter, when the other parameters are known, and where F is monotonic with respect to the unknown parameter. 
286       */
287      template <typename Function>
288      XSF_HOST_DEVICE inline std::tuple<double, double, double, double, int> bracket_root_for_cdf_inversion(
289          Function func, double x0, double xmin, double xmax, double step0_left,
290          double step0_right, double factor_left, double factor_right, bool increasing, std::uint64_t maxiter
291      ) {
292          double y0 = func(x0);
293  
294          if (y0 == 0) {
295              // Initial guess is correct.
296              return {x0, x0, y0, y0, 0};
297          }
298  
299          double y0_sgn = std::signbit(y0);
300  
301          bool search_left;
302          /* The frontier is the new leading endpoint of the expanding bracket. The 
303           * interior endpoint trails behind the frontier. In each step, the old frontier 
304           * endpoint becomes the new interior endpoint. */
305          double interior, frontier, y_interior, y_frontier, y_interior_sgn, y_frontier_sgn, boundary, factor;
306          if ((increasing && y0 < 0) || (!increasing && y0 > 0)) {
307              /* If func is increasing  and func(x_right) < 0 or if func is decreasing and 
308               *  f(y_right) > 0, we should expand the bracket to the right. */
309              interior = x0, y_interior = y0;
310              frontier = x0 + step0_right;
311              y_interior_sgn = y0_sgn;
312              search_left = false;
313              boundary = xmax;
314              factor = factor_right;
315          } else {
316              /* Otherwise we move and expand the bracket to the left. */
317              interior = x0, y_interior = y0;
318              frontier = x0 + step0_left;
319              y_interior_sgn = y0_sgn;
320              search_left = true;
321              boundary = xmin;
322              factor = factor_left;
323          }
324  
325          bool reached_boundary = false;
326          for (std::uint64_t i = 0; i < maxiter; i++) {
327              y_frontier = func(frontier);
328              y_frontier_sgn = std::signbit(y_frontier);
329              if (y_frontier_sgn != y_interior_sgn || (y_frontier == 0.0)) {
330                  /* Stopping condition, func evaluated at endpoints of bracket has opposing signs, 
331                   * meeting requirement for bracketing root finder. (Or endpoint has reached a 
332                   * zero.) */
333                  if (search_left) {
334                      /* Ensure we return an interval (a, b) with a < b. */
335                      std::swap(interior, frontier);
336                      std::swap(y_interior, y_frontier);
337                  }
338  		return {interior, frontier, y_interior, y_frontier, 0};
339              }
340              if (reached_boundary) {
341                  /* We've reached a boundary point without finding a root . */
342                  return {
343                      std::numeric_limits<double>::quiet_NaN(), std::numeric_limits<double>::quiet_NaN(),
344                      std::numeric_limits<double>::quiet_NaN(), std::numeric_limits<double>::quiet_NaN(),
345                      search_left ? 1 : 2
346                  };
347              }
348              double step = (frontier - interior) * factor;
349              interior = frontier;
350              y_interior = y_frontier;
351              y_interior_sgn = y_frontier_sgn;
352              frontier += step;
353              if ((search_left && frontier <= boundary) || (!search_left && frontier >= boundary)) {
354                  /* If the frontier has reached the boundary, set a flag so the algorithm will know 
355                   * not to search beyond this point. */
356                  frontier = boundary;
357                  reached_boundary = true;
358              }
359          }
360          /* Failed to converge within maxiter iterations. If maxiter is sufficiently high and 
361           * factor_left and factor_right are set appropriately, this should only happen due to 
362           * a bug in this function. Limiting the number of iterations is a defensive programming measure. */
363  	return {
364              std::numeric_limits<double>::quiet_NaN(), std::numeric_limits<double>::quiet_NaN(),
365              std::numeric_limits<double>::quiet_NaN(), std::numeric_limits<double>::quiet_NaN(), 3
366          };
367      }

369      /* Find root of a scalar function using Chandrupatla's algorithm */
370      template <typename Function>
371      XSF_HOST_DEVICE inline std::pair<double, int> find_root_chandrupatla(
372          Function func, double x1, double x2, double f1, double f2, double rtol,
373          double atol, std::uint64_t maxiter
374      ) {
375          if (f1 == 0) {
376              return {x1, 0};
377          }
378          if (f2 == 0) {
379              return {x2, 0};
380          }
381          double t = 0.5, x3, f3;
382          for (uint64_t i = 0; i < maxiter; i++) {
383              double x = x1 + t * (x2 - x1);
384              double f = func(x);
385              if (std::signbit(f) == std::signbit(f1)) {
386                  x3 = x1;
387                  x1 = x;
388                  f3 = f1;
389                  f1 = f;
390              } else {
391                  x3 = x2;
392                  x2 = x1;
393                  x1 = x;
394                  f3 = f2;
395                  f2 = f1;
396                  f1 = f;
397              }
398              double xm, fm;
399              if (std::abs(f2) < std::abs(f1)) {
400                  xm = x2;
401                  fm = f2;
402              } else {
403                  xm = x1;
404                  fm = f1;
405              }
406              double tol = 2.0 * rtol * std::abs(xm) + 0.5 * atol;
407              double tl = tol / std::abs(x2 - x1);
408              if (tl > 0.5 || fm == 0) {
409                  return {xm, 0};
410              }
411              double xi = (x1 - x2) / (x3 - x2);
412              double phi = (f1 - f2) / (f3 - f2);
413              double fl = 1.0 - std::sqrt(1.0 - xi);
414              double fh = std::sqrt(xi);
415  
416              if ((fl < phi) && (phi < fh)) {
417                  t = (f1 / (f2 - f1)) * (f3 / (f2 - f3)) + (f1 / (f3 - f1)) * (f2 / (f3 - f2)) * ((x3 - x1) / (x2 - x1));
418              } else {
419                  t = 0.5;
420              }
421              t = std::fmin(std::fmax(t, tl), 1.0 - tl);
422          }
423          return {std::numeric_limits<double>::quiet_NaN(), 1};
424      }
