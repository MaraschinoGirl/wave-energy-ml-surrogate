16   def check_library_installed(library):
17       """Check library is installed."""
18       try:
19           return importlib.import_module(library)
20       except ImportError as exc:
21           raise ImportError(
22               f"Setting output container to '{library}' requires {library} to be"
23               " installed"
24           ) from exc

27   def get_columns(columns):
28       if callable(columns):
29           try:
30               return columns()
31           except Exception:
32               return None
33       return columns

37   class ContainerAdapterProtocol(Protocol):
38       container_lib: str

40       def create_container(self, X_output, X_original, columns, inplace=False):
41           """Create container from `X_output` with additional metadata. 
42    
43           Parameters 
44           ---------- 
45           X_output : {ndarray, dataframe} 
46               Data to wrap. 
47    
48           X_original : {ndarray, dataframe} 
49               Original input dataframe. This is used to extract the metadata that should 
50               be passed to `X_output`, e.g. pandas row index. 
51    
52           columns : callable, ndarray, or None 
53               The column names or a callable that returns the column names. The 
54               callable is useful if the column names require some computation. If `None`, 
55               then no columns are passed to the container's constructor. 
56    
57           inplace : bool, default=False 
58               Whether or not we intend to modify `X_output` in-place. However, it does 
59               not guarantee that we return the same object if the in-place operation 
60               is not possible. 
61    
62           Returns 
63           ------- 
64           wrapped_output : container_type 
65               `X_output` wrapped into the container type. 
66           """

72           ---------- 
73           Xs: container 
74               Containers to be checked.

76           Returns 
77           ------- 
78           is_supported_container : bool 
79               True if X is a supported container. 
80           """

114  class PandasAdapter:
115      container_lib = "pandas"
116  
117      def create_container(self, X_output, X_original, columns, inplace=True):
118          pd = check_library_installed("pandas")
119          columns = get_columns(columns)
120  
121          if not inplace or not isinstance(X_output, pd.DataFrame):
122              # In all these cases, we need to create a new DataFrame
123  
124              # Unfortunately, we cannot use `getattr(container, "index")`
125              # because `list` exposes an `index` attribute.
126              if isinstance(X_output, pd.DataFrame):
127                  index = X_output.index
128              elif isinstance(X_original, pd.DataFrame):
129                  index = X_original.index
130              else:
131                  index = None
132  
133              # We don't pass columns here because it would intend columns selection
134              # instead of renaming.
135              X_output = _create_pandas_dataframe_from_non_pandas_container(
136                  X=X_output, index=index, copy=not inplace
137              )
138  
139          if columns is not None:
140              return self.rename_columns(X_output, columns)
141          return X_output
142  
143      def is_supported_container(self, X):
144          pd = check_library_installed("pandas")
145          return isinstance(X, pd.DataFrame)
146  
147      def rename_columns(self, X, columns):
148          # we cannot use `rename` since it takes a dictionary and at this stage we have
149          # potentially duplicate column names in `X`
150          X.columns = columns
151          return X
152  
153      def hstack(self, Xs):
154          pd = check_library_installed("pandas")
155          return pd.concat(Xs, axis=1)

158  class PolarsAdapter:
159      container_lib = "polars"
160  
161      def create_container(self, X_output, X_original, columns, inplace=True):
162          pl = check_library_installed("polars")
163          columns = get_columns(columns)
164          columns = columns.tolist() if isinstance(columns, np.ndarray) else columns
165  
166          if not inplace or not isinstance(X_output, pl.DataFrame):
167              # In all these cases, we need to create a new DataFrame
168              return pl.DataFrame(X_output, schema=columns, orient="row")
169  
170          if columns is not None:
171              return self.rename_columns(X_output, columns)
172          return X_output
173  
174      def is_supported_container(self, X):
175          pl = check_library_installed("polars")
176          return isinstance(X, pl.DataFrame)
177  
178      def rename_columns(self, X, columns):
179          # we cannot use `rename` since it takes a dictionary and at this stage we have
180          # potentially duplicate column names in `X`
181          X.columns = columns
182          return X
183  
184      def hstack(self, Xs):
185          pl = check_library_installed("polars")
186          return pl.concat(Xs, how="horizontal")

189  class ContainerAdaptersManager:
190      def __init__(self):
191          self.adapters = {}
192  
193      @property
194      def supported_outputs(self):
195          return {"default"} | set(self.adapters)
196  
197      def register(self, adapter):
198          self.adapters[adapter.container_lib] = adapter
199  
200  
201  ADAPTERS_MANAGER = ContainerAdaptersManager()
202  ADAPTERS_MANAGER.register(PandasAdapter())
203  ADAPTERS_MANAGER.register(PolarsAdapter())

206  def _get_adapter_from_container(container):
207      """Get the adapter that knows how to handle such container. 
208   
209      See :class:`sklearn.utils._set_output.ContainerAdapterProtocol` for more 
210      details. 
211      """
212      module_name = container.__class__.__module__.split(".")[0]
213      try:
214          return ADAPTERS_MANAGER.adapters[module_name]
215      except KeyError as exc:
216          available_adapters = list(ADAPTERS_MANAGER.adapters.keys())
217          raise ValueError(
218              "The container does not have a registered adapter in scikit-learn. "
219              f"Available adapters are: {available_adapters} while the container "
220              f"provided is: {container!r}."
221          ) from exc
222

224  def _get_container_adapter(method, estimator=None):
225      """Get container adapter."""
226      dense_config = _get_output_config(method, estimator)["dense"]
227      try:
228          return ADAPTERS_MANAGER.adapters[dense_config]
229      except KeyError:
230          return None

233  def _get_output_config(method, estimator=None):
234      """Get output config based on estimator and global configuration. 
235   
236      Parameters 
237      ---------- 
238      method : {"transform"} 
239          Estimator's method for which the output container is looked up. 
240   
241      estimator : estimator instance or None 
242          Estimator to get the output configuration from. If `None`, check global 
243          configuration is used. 
244   
245      Returns 
246      ------- 
247      config : dict 
248          Dictionary with keys: 
249   
250          - "dense": specifies the dense container for `method`. This can be 
251            `"default"` or `"pandas"`. 
252      """
253      est_sklearn_output_config = getattr(estimator, "_sklearn_output_config", {})
254      if method in est_sklearn_output_config:
255          dense_config = est_sklearn_output_config[method]
256      else:
257          dense_config = get_config()[f"{method}_output"]
258  
259      supported_outputs = ADAPTERS_MANAGER.supported_outputs
260      if dense_config not in supported_outputs:
261          raise ValueError(
262              f"output config must be in {sorted(supported_outputs)}, got {dense_config}"
263          )
264  
265      return {"dense": dense_config}

268  def _wrap_data_with_container(method, data_to_wrap, original_input, estimator):
269      """Wrap output with container based on an estimator's or global config. 
270   
271      Parameters 
272      ---------- 
273      method : {"transform"} 
274          Estimator's method to get container output for. 
275   
276      data_to_wrap : {ndarray, dataframe} 
277          Data to wrap with container. 
278   
279      original_input : {ndarray, dataframe} 
280          Original input of function. 
281   
282      estimator : estimator instance 
283          Estimator with to get the output configuration from. 
284   
285      Returns 
286      ------- 
287      output : {ndarray, dataframe} 
288          If the output config is "default" or the estimator is not configured 
289          for wrapping return `data_to_wrap` unchanged. 
290          If the output config is "pandas", return `data_to_wrap` as a pandas 
291          DataFrame. 
292      """
293      output_config = _get_output_config(method, estimator)
294  
295      if output_config["dense"] == "default" or not _auto_wrap_is_configured(estimator):
296          return data_to_wrap
297  
298      dense_config = output_config["dense"]
299      if issparse(data_to_wrap):
300          raise ValueError(
301              "The transformer outputs a scipy sparse matrix. "
302              "Try to set the transformer output to a dense array or disable "
303              f"{dense_config.capitalize()} output with set_output(transform='default')."
304          )
305  
306      adapter = ADAPTERS_MANAGER.adapters[dense_config]
307      return adapter.create_container(
308          data_to_wrap,
309          original_input,
310          columns=estimator.get_feature_names_out,
311      )

314  def _wrap_method_output(f, method):
315      """Wrapper used by `_SetOutputMixin` to automatically wrap methods."""
316  
317      @wraps(f)
318      def wrapped(self, X, *args, **kwargs):
319          data_to_wrap = f(self, X, *args, **kwargs)
320          if isinstance(data_to_wrap, tuple):
321              # only wrap the first output for cross decomposition
322              return_tuple = (
323                  _wrap_data_with_container(method, data_to_wrap[0], X, self),
324                  *data_to_wrap[1:],
325              )
326              # Support for namedtuples `_make` is a documented API for namedtuples:
327              # https://docs.python.org/3/library/collections.html#collections.somenamedtuple._make
328              if hasattr(type(data_to_wrap), "_make"):
329                  return type(data_to_wrap)._make(return_tuple)
330              return return_tuple
331  
332          return _wrap_data_with_container(method, data_to_wrap, X, self)
333  
334      return wrapped

337  def _auto_wrap_is_configured(estimator):
338      """Return True if estimator is configured for auto-wrapping the transform method. 
339   
340      `_SetOutputMixin` sets `_sklearn_auto_wrap_output_keys` to `set()` if auto wrapping 
341      is manually disabled. 
342      """
343      auto_wrap_output_keys = getattr(estimator, "_sklearn_auto_wrap_output_keys", set())
344      return (
345          hasattr(estimator, "get_feature_names_out")
346          and "transform" in auto_wrap_output_keys
347      )

350  class _SetOutputMixin:
351      """Mixin that dynamically wraps methods to return container based on config. 
352   
353      Currently `_SetOutputMixin` wraps `transform` and `fit_transform` and configures 
354      it based on `set_output` of the global configuration. 
355   
356      `set_output` is only defined if `get_feature_names_out` is defined and 
357      `auto_wrap_output_keys` is the default value. 
358      """
359  
360      def __init_subclass__(cls, auto_wrap_output_keys=("transform",), **kwargs):
361          super().__init_subclass__(**kwargs)
362  
363          # Dynamically wraps `transform` and `fit_transform` and configure it's
364          # output based on `set_output`.
365          if not (
366              isinstance(auto_wrap_output_keys, tuple) or auto_wrap_output_keys is None
367          ):
368              raise ValueError("auto_wrap_output_keys must be None or a tuple of keys.")
369  
370          if auto_wrap_output_keys is None:
371              cls._sklearn_auto_wrap_output_keys = set()
372              return
373  
374          # Mapping from method to key in configurations
375          method_to_key = {
376              "transform": "transform",
377              "fit_transform": "transform",
378          }
379          cls._sklearn_auto_wrap_output_keys = set()
380  
381          for method, key in method_to_key.items():
382              if not hasattr(cls, method) or key not in auto_wrap_output_keys:
383                  continue
384              cls._sklearn_auto_wrap_output_keys.add(key)
385  
386              # Only wrap methods defined by cls itself
387              if method not in cls.__dict__:
388                  continue
389              wrapped_method = _wrap_method_output(getattr(cls, method), key)
390              setattr(cls, method, wrapped_method)
391  
392      @available_if(_auto_wrap_is_configured)
393      def set_output(self, *, transform=None):
394          """Set output container. 
395   
396          See :ref:`sphx_glr_auto_examples_miscellaneous_plot_set_output.py` 
397          for an example on how to use the API. 
398   
399          Parameters 
400          ---------- 
401          transform : {"default", "pandas", "polars"}, default=None 
402              Configure output of `transform` and `fit_transform`. 
403   
404              - `"default"`: Default output format of a transformer 
405              - `"pandas"`: DataFrame output 
406              - `"polars"`: Polars output 
407              - `None`: Transform configuration is unchanged 
408   
409              .. versionadded:: 1.4 
410                  `"polars"` option was added. 
411   
412          Returns 
413          ------- 
414          self : estimator instance 
415              Estimator instance. 
416          """
417          if transform is None:
418              return self
419  
420          if not hasattr(self, "_sklearn_output_config"):
421              self._sklearn_output_config = {}
422  
423          self._sklearn_output_config["transform"] = transform
424          return self

427  def _safe_set_output(estimator, *, transform=None):
428      """Safely call estimator.set_output and error if it not available. 
429   
430      This is used by meta-estimators to set the output for child estimators. 
431   
432      Parameters 
433      ---------- 
434      estimator : estimator instance 
435          Estimator instance. 
436   
437      transform : {"default", "pandas", "polars"}, default=None 
438          Configure output of the following estimator's methods: 
439   
440          - `"transform"` 
441          - `"fit_transform"` 
442   
443          If `None`, this operation is a no-op. 
444   
445      Returns 
446      ------- 
447      estimator : estimator instance 
448          Estimator instance. 
449      """
450      set_output_for_transform = (
451          hasattr(estimator, "transform")
452          or hasattr(estimator, "fit_transform")
453          and transform is not None
454      )
455      if not set_output_for_transform:
456          # If estimator can not transform, then `set_output` does not need to be
457          # called.
458          return
459  
460      if not hasattr(estimator, "set_output"):
461          raise ValueError(
462              f"Unable to configure output for {estimator} because `set_output` "
463              "is not available."
464          )
465      return estimator.set_output(transform=transform)
