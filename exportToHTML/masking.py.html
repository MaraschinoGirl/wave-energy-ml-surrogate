5    def set_keras_mask(x, mask):
6        """Sets the Keras mask attribute for the given tensor in-place. 
7     
8        Args: 
9            x: Input tensor. 
10           mask: The mask tensor to be set. If `None`, the `_keras_mask` attribute 
11               will be cleared. 
12       """
13       set_tensor_attr(x, "_keras_mask", mask)

16   def get_keras_mask(x):

21
