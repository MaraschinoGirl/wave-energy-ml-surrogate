30   class ZMQDataPublisher(Configurable):
31       """A zmq data publisher."""
32   
33       topic = topic = CBytes(b"datapub")
34       session = Instance(Session, allow_none=True)
35       pub_socket = Any(allow_none=True)
36       parent_header = Dict({})
37   
38       def set_parent(self, parent):
39           """Set the parent for outbound messages."""
40           self.parent_header = extract_header(parent)
41   
42       def publish_data(self, data):
43           """publish a data_message on the IOPub channel 
44    
45           Parameters 
46           ---------- 
47           data : dict 
48               The data to be published. Think of it as a namespace. 
49           """
50           session = self.session
51           assert session is not None
52           buffers = serialize_object(
53               data,
54               buffer_threshold=session.buffer_threshold,
55               item_threshold=session.item_threshold,
56           )
57           content = json_clean(dict(keys=list(data.keys())))
58           session.send(
59               self.pub_socket,
60               "data_message",
61               content=content,
62               parent=self.parent_header,
63               buffers=buffers,
64               ident=self.topic,
65           )
