132      def analyse_text(text):
133          """Verilog code will use one of reg/wire/assign for sure, and that 
134          is not common elsewhere."""
135          result = 0
136          if 'reg' in text:
137              result += 0.1
138          if 'wire' in text:
139              result += 0.1
140          if 'assign' in text:
141              result += 0.1
142  
143          return result
