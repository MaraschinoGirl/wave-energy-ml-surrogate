10   class VeryPermissive:
11       def _dynamic_(self, name, lcid, wFlags, args):
12           if wFlags & pythoncom.DISPATCH_METHOD:
13               return getattr(self, name)(*args)
14   
15           if wFlags & pythoncom.DISPATCH_PROPERTYGET:
16               try:
17                   # to avoid problems with byref param handling, tuple results are converted to lists.
18                   ret = self.__dict__[name]
19                   if isinstance(ret, tuple):
20                       ret = list(ret)
21                   return ret
22               except KeyError:  # Probably a method request.
23                   raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)
24   
25           if wFlags & (
26               pythoncom.DISPATCH_PROPERTYPUT | pythoncom.DISPATCH_PROPERTYPUTREF
27           ):
28               setattr(self, name, args[0])
29               return
30   
31           raise COMException(scode=winerror.E_INVALIDARG, desc="invalid wFlags")
32   
33       def write(self, *args):
34           if len(args) == 0:
35               raise COMException(
36                   scode=winerror.DISP_E_BADPARAMCOUNT
37               )  # Probably call as PROPGET.
38   
39           for arg in args[:-1]:
40               print(arg, end=" ")
41           print(args[-1])
