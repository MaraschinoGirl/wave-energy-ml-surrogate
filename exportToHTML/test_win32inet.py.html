26   class CookieTests(unittest.TestCase):
27       def testCookies(self):
28           data = "TestData=Test"
29           InternetSetCookie("https://www.python.org", None, data)
30           got = InternetGetCookie("https://www.python.org", None)
31           # handle that there might already be cookies for the domain.
32           bits = (x.strip() for x in got.split(";"))
33           self.assertTrue(data in bits)
34   
35       def testCookiesEmpty(self):

37               InternetGetCookie("https://site-with-no-cookie.python.org", None)
38               self.fail("expected win32 exception")
39           except error as exc:
40               self.assertEqual(exc.winerror, winerror.ERROR_NO_MORE_ITEMS)

48       def testLongCanonicalize(self):
49           # a 4k URL causes the underlying API to request a bigger buffer"
50           big = "x" * 2048
51           ret = InternetCanonicalizeUrl(big + " " + big)
52           self.assertEqual(ret, big + "%20" + big)

55   class TestNetwork(unittest.TestCase):
56       def setUp(self):
57           self.hi = InternetOpen("test", INTERNET_OPEN_TYPE_DIRECT, None, None, 0)
58   
59       def tearDown(self):
60           self.hi.Close()
61   
62       def testPythonDotOrg(self):
63           hdl = InternetOpenUrl(
64               self.hi, "https://www.python.org", None, INTERNET_FLAG_EXISTING_CONNECT
65           )
66           chunks = []
67           while 1:
68               chunk = InternetReadFile(hdl, 1024)
69               if not chunk:
70                   break
71               chunks.append(chunk)
72           data = b"".join(chunks)
73           # This must appear somewhere on the main page!
74           self.assertGreater(data.find(b"Python"), 0, repr(data))

76       def testFtpCommand(self):
77           # ftp.python.org doesn't exist.  ftp.gnu.org is what Python's urllib
78           # test code uses.
79           # (As of 2020 it doesn't! Unsurprisingly, it's difficult to find a good
80           # test server. This test sometimes works, but often doesn't - so handle
81           # failure here as a "skip")
82           try:
83               hcon = InternetConnect(
84                   self.hi,
85                   "ftp.gnu.org",
86                   INTERNET_INVALID_PORT_NUMBER,

93               try:
94                   hftp = FtpCommand(hcon, True, FTP_TRANSFER_TYPE_ASCII, "NLST", 0)
95                   try:
96                       print("Connected - response info is", InternetGetLastResponseInfo())
97                       got = InternetReadFile(hftp, 2048)
98                       print("Read", len(got), "bytes")

100                      hftp.Close()
