14   @register
15   class _Media(DOMWidget, ValueWidget, CoreWidget):
16       """Base class for Image, Audio and Video widgets. 
17    
18       The `value` of this widget accepts a byte string.  The byte string is the 
19       raw data that you want the browser to display. 
20    
21       If you pass `"url"` to the `"format"` trait, `value` will be interpreted 
22       as a URL as bytes encoded in UTF-8. 
23       """
24   
25       # Define the custom state properties to sync with the front-end
26       value = CByteMemoryView(help="The media data as a memory view of bytes.").tag(sync=True)
27   
28       @classmethod
29       def _from_file(cls, tag, filename, **kwargs):
30           """ 
31           Create an :class:`Media` from a local file. 
32    
33           Parameters 
34           ---------- 
35           filename: str 
36               The location of a file to read into the value from disk. 
37    
38           **kwargs: 
39               The keyword arguments for `Media` 
40    
41           Returns an `Media` with the value set from the filename. 
42           """
43           value = cls._load_file_value(filename)
44   
45           if 'format' not in kwargs:
46               format = cls._guess_format(tag, filename)
47               if format is not None:
48                   kwargs['format'] = format
49   
50           return cls(value=value, **kwargs)
51   
52       @classmethod
53       def from_url(cls, url, **kwargs):
54           """ 
55           Create an :class:`Media` from a URL. 
56    
57           :code:`Media.from_url(url)` is equivalent to: 
58    
59           .. code-block: python 
60    
61               med = Media(value=url, format='url') 
62    
63           But both unicode and bytes arguments are allowed for ``url``. 
64    
65           Parameters 
66           ---------- 
67           url: [str, bytes] 
68               The location of a URL to load. 
69           """
70           if isinstance(url, str):
71               # If str, it needs to be encoded to bytes
72               url = url.encode('utf-8')
73   
74           return cls(value=url, format='url', **kwargs)
75   
76       def set_value_from_file(self, filename):
77           """ 
78           Convenience method for reading a file into `value`. 
79    
80           Parameters 
81           ---------- 
82           filename: str 
83               The location of a file to read into value from disk. 
84           """
85           value = self._load_file_value(filename)
86   
87           self.value = value
88   
89       @classmethod
90       def _load_file_value(cls, filename):
91           if getattr(filename, 'read', None) is not None:
92               return filename.read()
93           else:
94               with open(filename, 'rb') as f:
95                   return f.read()
96   
97       @classmethod
98       def _guess_format(cls, tag, filename):
99           # file objects may have a .name parameter
100          name = getattr(filename, 'name', None)
101          name = name or filename
102  
103          try:
104              mtype, _ = mimetypes.guess_type(name)
105              if not mtype.startswith('{}/'.format(tag)):
106                  return None
107  
108              return mtype[len('{}/'.format(tag)):]
109          except Exception:
110              return None
111  
112      def _get_repr(self, cls):
113          # Truncate the value in the repr, since it will
114          # typically be very, very large.
115          class_name = self.__class__.__name__
116  
117          # Return value first like a ValueWidget
118          signature = []
119  
120          sig_value = 'value={!r}'.format(self.value[:40].tobytes())
121          if self.value.nbytes > 40:
122              sig_value = sig_value[:-1]+"..."+sig_value[-1]
123          signature.append(sig_value)
124  
125          for key in super(cls, self)._repr_keys():
126              if key == 'value':
127                  continue
128              value = str(getattr(self, key))
129              signature.append('{}={!r}'.format(key, value))
130          signature = ', '.join(signature)
131          return '{}({})'.format(class_name, signature)

134  @register
135  class Image(_Media):
136      """Displays an image as a widget. 
137   
138      The `value` of this widget accepts a byte string.  The byte string is the 
139      raw image data that you want the browser to display.  You can explicitly 
140      define the format of the byte string using the `format` trait (which 
141      defaults to "png"). 
142   
143      If you pass `"url"` to the `"format"` trait, `value` will be interpreted 
144      as a URL as bytes encoded in UTF-8. 
145      """
146      _view_name = Unicode('ImageView').tag(sync=True)
147      _model_name = Unicode('ImageModel').tag(sync=True)
148  
149      # Define the custom state properties to sync with the front-end
150      format = Unicode('png', help="The format of the image.").tag(sync=True)
151      width = CUnicode(help="Width of the image in pixels. Use layout.width "
152                            "for styling the widget.").tag(sync=True)
153      height = CUnicode(help="Height of the image in pixels. Use layout.height "
154                             "for styling the widget.").tag(sync=True)
155  
156      def __init__(self, *args, **kwargs):
157          super().__init__(*args, **kwargs)
158  
159      @classmethod
160      def from_file(cls, filename, **kwargs):
161          return cls._from_file('image', filename, **kwargs)
162  
163      def __repr__(self):
164          return self._get_repr(Image)
