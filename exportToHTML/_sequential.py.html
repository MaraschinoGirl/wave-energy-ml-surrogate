28   class SequentialFeatureSelector(SelectorMixin, MetaEstimatorMixin, BaseEstimator):
29       """Transformer that performs Sequential Feature Selection. 
30    
31       This Sequential Feature Selector adds (forward selection) or 
32       removes (backward selection) features to form a feature subset in a 
33       greedy fashion. At each stage, this estimator chooses the best feature to 
34       add or remove based on the cross-validation score of an estimator. In 
35       the case of unsupervised learning, this Sequential Feature Selector 
36       looks only at the features (X), not the desired outputs (y). 
37    
38       Read more in the :ref:`User Guide <sequential_feature_selection>`. 
39    
40       .. versionadded:: 0.24 
41    
42       Parameters 
43       ---------- 
44       estimator : estimator instance 
45           An unfitted estimator. 
46    
47       n_features_to_select : "auto", int or float, default="auto" 
48           If `"auto"`, the behaviour depends on the `tol` parameter: 
49    
50           - if `tol` is not `None`, then features are selected while the score 
51             change does not exceed `tol`. 
52           - otherwise, half of the features are selected. 
53    
54           If integer, the parameter is the absolute number of features to select. 
55           If float between 0 and 1, it is the fraction of features to select. 
56    
57           .. versionadded:: 1.1 
58              The option `"auto"` was added in version 1.1. 
59    
60           .. versionchanged:: 1.3 
61              The default changed from `"warn"` to `"auto"` in 1.3. 
62    
63       tol : float, default=None 
64           If the score is not incremented by at least `tol` between two 
65           consecutive feature additions or removals, stop adding or removing. 
66    
67           `tol` can be negative when removing features using `direction="backward"`. 
68           `tol` is required to be strictly positive when doing forward selection. 
69           It can be useful to reduce the number of features at the cost of a small 
70           decrease in the score. 
71    
72           `tol` is enabled only when `n_features_to_select` is `"auto"`. 
73    
74           .. versionadded:: 1.1 
75    
76       direction : {'forward', 'backward'}, default='forward' 
77           Whether to perform forward selection or backward selection. 
78    
79       scoring : str or callable, default=None 
80           A single str (see :ref:`scoring_parameter`) or a callable 
81           (see :ref:`scoring_callable`) to evaluate the predictions on the test set. 
82    
83           NOTE that when using a custom scorer, it should return a single 
84           value. 
85    
86           If None, the estimator's score method is used. 
87    
88       cv : int, cross-validation generator or an iterable, default=None 
89           Determines the cross-validation splitting strategy. 
90           Possible inputs for cv are: 
91    
92           - None, to use the default 5-fold cross validation, 
93           - integer, to specify the number of folds in a `(Stratified)KFold`, 
94           - :term:`CV splitter`, 
95           - An iterable yielding (train, test) splits as arrays of indices. 
96    
97           For integer/None inputs, if the estimator is a classifier and ``y`` is 
98           either binary or multiclass, 
99           :class:`~sklearn.model_selection.StratifiedKFold` is used. In all other 
100          cases, :class:`~sklearn.model_selection.KFold` is used. These splitters 
101          are instantiated with `shuffle=False` so the splits will be the same 
102          across calls. 
103   
104          Refer :ref:`User Guide <cross_validation>` for the various 
105          cross-validation strategies that can be used here. 
106   
107      n_jobs : int, default=None 
108          Number of jobs to run in parallel. When evaluating a new feature to 
109          add or remove, the cross-validation procedure is parallel over the 
110          folds. 
111          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
112          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
113          for more details. 
114   
115      Attributes 
116      ---------- 
117      n_features_in_ : int 
118          Number of features seen during :term:`fit`. Only defined if the 
119          underlying estimator exposes such an attribute when fit. 
120   
121          .. versionadded:: 0.24 
122   
123      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
124          Names of features seen during :term:`fit`. Defined only when `X` 
125          has feature names that are all strings. 
126   
127          .. versionadded:: 1.0 
128   
129      n_features_to_select_ : int 
130          The number of features that were selected. 
131   
132      support_ : ndarray of shape (n_features,), dtype=bool 
133          The mask of selected features. 
134   
135      See Also 
136      -------- 
137      GenericUnivariateSelect : Univariate feature selector with configurable 
138          strategy. 
139      RFE : Recursive feature elimination based on importance weights. 
140      RFECV : Recursive feature elimination based on importance weights, with 
141          automatic selection of the number of features. 
142      SelectFromModel : Feature selection based on thresholds of importance 
143          weights. 
144   
145      Examples 
146      -------- 
147      >>> from sklearn.feature_selection import SequentialFeatureSelector 
148      >>> from sklearn.neighbors import KNeighborsClassifier 
149      >>> from sklearn.datasets import load_iris 
150      >>> X, y = load_iris(return_X_y=True) 
151      >>> knn = KNeighborsClassifier(n_neighbors=3) 
152      >>> sfs = SequentialFeatureSelector(knn, n_features_to_select=3) 
153      >>> sfs.fit(X, y) 
154      SequentialFeatureSelector(estimator=KNeighborsClassifier(n_neighbors=3), 
155                                n_features_to_select=3) 
156      >>> sfs.get_support() 
157      array([ True, False,  True,  True]) 
158      >>> sfs.transform(X).shape 
159      (150, 3) 
160      """
161

164          "n_features_to_select": [

176      def __init__(
177          self,
178          estimator,
179          *,
180          n_features_to_select="auto",
181          tol=None,
182          direction="forward",
183          scoring=None,
184          cv=5,
185          n_jobs=None,
186      ):
187          self.estimator = estimator
188          self.n_features_to_select = n_features_to_select
189          self.tol = tol
190          self.direction = direction
191          self.scoring = scoring
192          self.cv = cv
193          self.n_jobs = n_jobs
194  
195      @_fit_context(
196          # SequentialFeatureSelector.estimator is not validated yet
197          prefer_skip_nested_validation=False
198      )
199      def fit(self, X, y=None, **params):
200          """Learn the features to select from X. 
201   
202          Parameters 
203          ---------- 
204          X : array-like of shape (n_samples, n_features) 
205              Training vectors, where `n_samples` is the number of samples and 
206              `n_features` is the number of predictors. 
207   
208          y : array-like of shape (n_samples,), default=None 
209              Target values. This parameter may be ignored for 
210              unsupervised learning. 
211   
212          **params : dict, default=None 
213              Parameters to be passed to the underlying `estimator`, `cv` 
214              and `scorer` objects. 
215   
216              .. versionadded:: 1.6 
217   
218                  Only available if `enable_metadata_routing=True`, 
219                  which can be set by using 
220                  ``sklearn.set_config(enable_metadata_routing=True)``. 
221                  See :ref:`Metadata Routing User Guide <metadata_routing>` for 
222                  more details. 
223   
224          Returns 
225          ------- 
226          self : object 
227              Returns the instance itself. 
228          """
229          _raise_for_params(params, self, "fit")
230          tags = self.__sklearn_tags__()
231          X = validate_data(
232              self,
233              X,
234              accept_sparse="csc",
235              ensure_min_features=2,
236              ensure_all_finite=not tags.input_tags.allow_nan,
237          )
238          n_features = X.shape[1]
239  
240          if self.n_features_to_select == "auto":
241              if self.tol is not None:
242                  # With auto feature selection, `n_features_to_select_` will be updated
243                  # to `support_.sum()` after features are selected.
244                  self.n_features_to_select_ = n_features - 1
245              else:
246                  self.n_features_to_select_ = n_features // 2
247          elif isinstance(self.n_features_to_select, Integral):
248              if self.n_features_to_select >= n_features:
249                  raise ValueError("n_features_to_select must be < n_features.")
250              self.n_features_to_select_ = self.n_features_to_select
251          elif isinstance(self.n_features_to_select, Real):
252              self.n_features_to_select_ = int(n_features * self.n_features_to_select)
253  
254          if self.tol is not None and self.tol < 0 and self.direction == "forward":
255              raise ValueError(
256                  "tol must be strictly positive when doing forward selection"
257              )
258  
259          cv = check_cv(self.cv, y, classifier=is_classifier(self.estimator))
260  
261          cloned_estimator = clone(self.estimator)
262  
263          # the current mask corresponds to the set of features:
264          # - that we have already *selected* if we do forward selection
265          # - that we have already *excluded* if we do backward selection
266          current_mask = np.zeros(shape=n_features, dtype=bool)
267          n_iterations = (
268              self.n_features_to_select_
269              if self.n_features_to_select == "auto" or self.direction == "forward"
270              else n_features - self.n_features_to_select_
271          )
272  
273          old_score = -np.inf
274          is_auto_select = self.tol is not None and self.n_features_to_select == "auto"
275  
276          # We only need to verify the routing here and not use the routed params
277          # because internally the actual routing will also take place inside the
278          # `cross_val_score` function.
279          if _routing_enabled():
280              process_routing(self, "fit", **params)
281          for _ in range(n_iterations):
282              new_feature_idx, new_score = self._get_best_new_feature_score(
283                  cloned_estimator, X, y, cv, current_mask, **params
284              )
285              if is_auto_select and ((new_score - old_score) < self.tol):
286                  break
287  
288              old_score = new_score
289              current_mask[new_feature_idx] = True
290  
291          if self.direction == "backward":
292              current_mask = ~current_mask
293  
294          self.support_ = current_mask
295          self.n_features_to_select_ = self.support_.sum()
296  
297          return self
298  
299      def _get_best_new_feature_score(self, estimator, X, y, cv, current_mask, **params):
300          # Return the best new feature and its score to add to the current_mask,
301          # i.e. return the best new feature and its score to add (resp. remove)
302          # when doing forward selection (resp. backward selection).
303          # Feature will be added if the current score and past score are greater
304          # than tol when n_feature is auto,
305          candidate_feature_indices = np.flatnonzero(~current_mask)
306          scores = {}
307          for feature_idx in candidate_feature_indices:
308              candidate_mask = current_mask.copy()
309              candidate_mask[feature_idx] = True
310              if self.direction == "backward":
311                  candidate_mask = ~candidate_mask
312              X_new = X[:, candidate_mask]
313              scores[feature_idx] = cross_val_score(
314                  estimator,
315                  X_new,
316                  y,
317                  cv=cv,
318                  scoring=self.scoring,
319                  n_jobs=self.n_jobs,
320                  params=params,
321              ).mean()
322          new_feature_idx = max(scores, key=lambda feature_idx: scores[feature_idx])
323          return new_feature_idx, scores[new_feature_idx]
324  
325      def _get_support_mask(self):
326          check_is_fitted(self)
327          return self.support_

329      def __sklearn_tags__(self):
330          tags = super().__sklearn_tags__()
331          tags.input_tags.allow_nan = get_tags(self.estimator).input_tags.allow_nan
332          tags.input_tags.sparse = get_tags(self.estimator).input_tags.sparse
333          return tags

335      def get_metadata_routing(self):
336          """Get metadata routing of this object. 
337   
338          Please check :ref:`User Guide <metadata_routing>` on how the routing 
339          mechanism works. 
340   
341          .. versionadded:: 1.6 
342   
343          Returns 
344          ------- 
345          routing : MetadataRouter 
346              A :class:`~sklearn.utils.metadata_routing.MetadataRouter` encapsulating 
347              routing information. 
348          """
349          router = MetadataRouter(owner=self.__class__.__name__)
350          router.add(
351              estimator=self.estimator,
352              method_mapping=MethodMapping().add(caller="fit", callee="fit"),
353          )
354          router.add(
355              splitter=check_cv(self.cv, classifier=is_classifier(self.estimator)),
356              method_mapping=MethodMapping().add(caller="fit", callee="split"),
357          )
358          router.add(
359              scorer=check_scoring(self.estimator, scoring=self.scoring),
360              method_mapping=MethodMapping().add(caller="fit", callee="score"),
361          )
362          return router
