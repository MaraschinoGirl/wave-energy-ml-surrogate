60   class ParameterGrid:
61       """Grid of parameters with a discrete number of values for each. 
62    
63       Can be used to iterate over parameter value combinations with the 
64       Python built-in function iter. 
65       The order of the generated parameter combinations is deterministic. 
66    
67       Read more in the :ref:`User Guide <grid_search>`. 
68    
69       Parameters 
70       ---------- 
71       param_grid : dict of str to sequence, or sequence of such 
72           The parameter grid to explore, as a dictionary mapping estimator 
73           parameters to sequences of allowed values. 
74    
75           An empty dict signifies default parameters. 
76    
77           A sequence of dicts signifies a sequence of grids to search, and is 
78           useful to avoid exploring parameter combinations that make no sense 
79           or have no effect. See the examples below. 
80    
81       Examples 
82       -------- 
83       >>> from sklearn.model_selection import ParameterGrid 
84       >>> param_grid = {'a': [1, 2], 'b': [True, False]} 
85       >>> list(ParameterGrid(param_grid)) == ( 
86       ...    [{'a': 1, 'b': True}, {'a': 1, 'b': False}, 
87       ...     {'a': 2, 'b': True}, {'a': 2, 'b': False}]) 
88       True 
89    
90       >>> grid = [{'kernel': ['linear']}, {'kernel': ['rbf'], 'gamma': [1, 10]}] 
91       >>> list(ParameterGrid(grid)) == [{'kernel': 'linear'}, 
92       ...                               {'kernel': 'rbf', 'gamma': 1}, 
93       ...                               {'kernel': 'rbf', 'gamma': 10}] 
94       True 
95       >>> ParameterGrid(grid)[1] == {'kernel': 'rbf', 'gamma': 1} 
96       True 
97    
98       See Also 
99       -------- 
100      GridSearchCV : Uses :class:`ParameterGrid` to perform a full parallelized 
101          parameter search. 
102      """
103  
104      def __init__(self, param_grid):
105          if not isinstance(param_grid, (Mapping, Iterable)):
106              raise TypeError(
107                  f"Parameter grid should be a dict or a list, got: {param_grid!r} of"
108                  f" type {type(param_grid).__name__}"
109              )
110  
111          if isinstance(param_grid, Mapping):
112              # wrap dictionary in a singleton list to support either dict
113              # or list of dicts
114              param_grid = [param_grid]
115  
116          # check if all entries are dictionaries of lists
117          for grid in param_grid:
118              if not isinstance(grid, dict):
119                  raise TypeError(f"Parameter grid is not a dict ({grid!r})")
120              for key, value in grid.items():
121                  if isinstance(value, np.ndarray) and value.ndim > 1:
122                      raise ValueError(
123                          f"Parameter array for {key!r} should be one-dimensional, got:"
124                          f" {value!r} with shape {value.shape}"
125                      )
126                  if isinstance(value, str) or not isinstance(
127                      value, (np.ndarray, Sequence)
128                  ):
129                      raise TypeError(
130                          f"Parameter grid for parameter {key!r} needs to be a list or a"
131                          f" numpy array, but got {value!r} (of type "
132                          f"{type(value).__name__}) instead. Single values "
133                          "need to be wrapped in a list with one element."
134                      )
135                  if len(value) == 0:
136                      raise ValueError(
137                          f"Parameter grid for parameter {key!r} need "
138                          f"to be a non-empty sequence, got: {value!r}"
139                      )
140  
141          self.param_grid = param_grid
142  
143      def __iter__(self):
144          """Iterate over the points in the grid. 
145   
146          Returns 
147          ------- 
148          params : iterator over dict of str to any 
149              Yields dictionaries mapping each estimator parameter to one of its 
150              allowed values. 
151          """
152          for p in self.param_grid:
153              # Always sort the keys of a dictionary, for reproducibility
154              items = sorted(p.items())
155              if not items:
156                  yield {}
157              else:
158                  keys, values = zip(*items)
159                  for v in product(*values):
160                      params = dict(zip(keys, v))
161                      yield params
162  
163      def __len__(self):
164          """Number of points on the grid."""
165          # Product function that can handle iterables (np.prod can't).
166          product = partial(reduce, operator.mul)
167          return sum(
168              product(len(v) for v in p.values()) if p else 1 for p in self.param_grid
169          )
170  
171      def __getitem__(self, ind):
172          """Get the parameters that would be ``ind``th in iteration 
173   
174          Parameters 
175          ---------- 
176          ind : int 
177              The iteration index 
178   
179          Returns 
180          ------- 
181          params : dict of str to any 
182              Equal to list(self)[ind] 
183          """
184          # This is used to make discrete sampling without replacement memory
185          # efficient.
186          for sub_grid in self.param_grid:
187              # XXX: could memoize information used here
188              if not sub_grid:
189                  if ind == 0:
190                      return {}
191                  else:
192                      ind -= 1
193                      continue
194  
195              # Reverse so most frequent cycling parameter comes first
196              keys, values_lists = zip(*sorted(sub_grid.items())[::-1])
197              sizes = [len(v_list) for v_list in values_lists]
198              total = np.prod(sizes)
199  
200              if ind >= total:
201                  # Try the next grid
202                  ind -= total
203              else:
204                  out = {}
205                  for key, v_list, n in zip(keys, values_lists, sizes):
206                      ind, offset = divmod(ind, n)
207                      out[key] = v_list[offset]
208                  return out
209  
210          raise IndexError("ParameterGrid index out of range")

213  class ParameterSampler:
214      """Generator on parameters sampled from given distributions. 
215   
216      Non-deterministic iterable over random candidate combinations for hyper- 
217      parameter search. If all parameters are presented as a list, 
218      sampling without replacement is performed. If at least one parameter 
219      is given as a distribution, sampling with replacement is used. 
220      It is highly recommended to use continuous distributions for continuous 
221      parameters. 
222   
223      Read more in the :ref:`User Guide <grid_search>`. 
224   
225      Parameters 
226      ---------- 
227      param_distributions : dict 
228          Dictionary with parameters names (`str`) as keys and distributions 
229          or lists of parameters to try. Distributions must provide a ``rvs`` 
230          method for sampling (such as those from scipy.stats.distributions). 
231          If a list is given, it is sampled uniformly. 
232          If a list of dicts is given, first a dict is sampled uniformly, and 
233          then a parameter is sampled using that dict as above. 
234   
235      n_iter : int 
236          Number of parameter settings that are produced. 
237   
238      random_state : int, RandomState instance or None, default=None 
239          Pseudo random number generator state used for random uniform sampling 
240          from lists of possible values instead of scipy.stats distributions. 
241          Pass an int for reproducible output across multiple 
242          function calls. 
243          See :term:`Glossary <random_state>`. 
244   
245      Returns 
246      ------- 
247      params : dict of str to any 
248          **Yields** dictionaries mapping each estimator parameter to 
249          as sampled value. 
250   
251      Examples 
252      -------- 
253      >>> from sklearn.model_selection import ParameterSampler 
254      >>> from scipy.stats.distributions import expon 
255      >>> import numpy as np 
256      >>> rng = np.random.RandomState(0) 
257      >>> param_grid = {'a':[1, 2], 'b': expon()} 
258      >>> param_list = list(ParameterSampler(param_grid, n_iter=4, 
259      ...                                    random_state=rng)) 
260      >>> rounded_list = [dict((k, round(v, 6)) for (k, v) in d.items()) 
261      ...                 for d in param_list] 
262      >>> rounded_list == [{'b': 0.89856, 'a': 1}, 
263      ...                  {'b': 0.923223, 'a': 1}, 
264      ...                  {'b': 1.878964, 'a': 2}, 
265      ...                  {'b': 1.038159, 'a': 2}] 
266      True 
267      """
268  
269      def __init__(self, param_distributions, n_iter, *, random_state=None):
270          if not isinstance(param_distributions, (Mapping, Iterable)):
271              raise TypeError(
272                  "Parameter distribution is not a dict or a list,"
273                  f" got: {param_distributions!r} of type "
274                  f"{type(param_distributions).__name__}"
275              )
276  
277          if isinstance(param_distributions, Mapping):
278              # wrap dictionary in a singleton list to support either dict
279              # or list of dicts
280              param_distributions = [param_distributions]
281  
282          for dist in param_distributions:
283              if not isinstance(dist, dict):
284                  raise TypeError(
285                      "Parameter distribution is not a dict ({!r})".format(dist)
286                  )
287              for key in dist:
288                  if not isinstance(dist[key], Iterable) and not hasattr(
289                      dist[key], "rvs"
290                  ):
291                      raise TypeError(
292                          f"Parameter grid for parameter {key!r} is not iterable "
293                          f"or a distribution (value={dist[key]})"
294                      )
295          self.n_iter = n_iter
296          self.random_state = random_state
297          self.param_distributions = param_distributions
298  
299      def _is_all_lists(self):
300          return all(
301              all(not hasattr(v, "rvs") for v in dist.values())
302              for dist in self.param_distributions
303          )
304  
305      def __iter__(self):
306          rng = check_random_state(self.random_state)
307  
308          # if all distributions are given as lists, we want to sample without
309          # replacement
310          if self._is_all_lists():
311              # look up sampled parameter settings in parameter grid
312              param_grid = ParameterGrid(self.param_distributions)
313              grid_size = len(param_grid)
314              n_iter = self.n_iter
315  
316              if grid_size < n_iter:
317                  warnings.warn(
318                      "The total space of parameters %d is smaller "
319                      "than n_iter=%d. Running %d iterations. For exhaustive "
320                      "searches, use GridSearchCV." % (grid_size, self.n_iter, grid_size),
321                      UserWarning,
322                  )
323                  n_iter = grid_size
324              for i in sample_without_replacement(grid_size, n_iter, random_state=rng):
325                  yield param_grid[i]
326  
327          else:
328              for _ in range(self.n_iter):
329                  dist = rng.choice(self.param_distributions)
330                  # Always sort the keys of a dictionary, for reproducibility
331                  items = sorted(dist.items())
332                  params = dict()
333                  for k, v in items:
334                      if hasattr(v, "rvs"):
335                          params[k] = v.rvs(random_state=rng)
336                      else:
337                          params[k] = v[rng.randint(len(v))]
338                  yield params
339  
340      def __len__(self):
341          """Number of points that will be sampled."""
342          if self._is_all_lists():
343              grid_size = len(ParameterGrid(self.param_distributions))
344              return min(self.n_iter, grid_size)
345          else:
346              return self.n_iter

350      if not search_cv.refit:
351          raise AttributeError(
352              f"This {type(search_cv).__name__} instance was initialized with "
353              f"`refit=False`. {attr} is available only after refitting on the best "
354              "parameters. You can refit an estimator manually using the "
355              "`best_params_` attribute"

359  def _search_estimator_has(attr):
360      """Check if we can delegate a method to the underlying estimator. 
361   
362      Calling a prediction method will only be available if `refit=True`. In 
363      such case, we check first the fitted best estimator. If it is not 
364      fitted, we check the unfitted estimator. 
365   
366      Checking the unfitted estimator allows to use `hasattr` on the `SearchCV` 
367      instance even before calling `fit`. 
368      """
369  
370      def check(self):
371          _check_refit(self, attr)
372          if hasattr(self, "best_estimator_"):
373              # raise an AttributeError if `attr` does not exist
374              getattr(self.best_estimator_, attr)
375              return True
376          # raise an AttributeError if `attr` does not exist
377          getattr(self.estimator, attr)
378          return True
379  
380      return check

383  def _yield_masked_array_for_each_param(candidate_params):
384      """ 
385      Yield a masked array for each candidate param. 
386   
387      `candidate_params` is a sequence of params which were used in 
388      a `GridSearchCV`. We use masked arrays for the results, as not 
389      all params are necessarily present in each element of 
390      `candidate_params`. For example, if using `GridSearchCV` with 
391      a `SVC` model, then one might search over params like: 
392   
393          - kernel=["rbf"], gamma=[0.1, 1] 
394          - kernel=["poly"], degree=[1, 2] 
395   
396      and then param `'gamma'` would not be present in entries of 
397      `candidate_params` corresponding to `kernel='poly'`. 
398      """
399      n_candidates = len(candidate_params)
400      param_results = defaultdict(dict)
401  
402      for cand_idx, params in enumerate(candidate_params):
403          for name, value in params.items():
404              param_results["param_%s" % name][cand_idx] = value
405  
406      for key, param_result in param_results.items():
407          param_list = list(param_result.values())
408          try:
409              arr = np.array(param_list)
410          except ValueError:
411              # This can happen when param_list contains lists of different
412              # lengths, for example:
413              # param_list=[[1], [2, 3]]
414              arr_dtype = np.dtype(object)
415          else:
416              # There are two cases when we don't use the automatically inferred
417              # dtype when creating the array and we use object instead:
418              # - string dtype
419              # - when array.ndim > 1, that means that param_list was something
420              #   like a list of same-size sequences, which gets turned into a
421              #   multi-dimensional array but we want a 1d array
422              arr_dtype = arr.dtype if arr.dtype.kind != "U" and arr.ndim == 1 else object
423  
424          # Use one MaskedArray and mask all the places where the param is not
425          # applicable for that candidate (which may not contain all the params).
426          ma = MaskedArray(np.empty(n_candidates, dtype=arr_dtype), mask=True)
427          for index, value in param_result.items():
428              # Setting the value at an index unmasks that index
429              ma[index] = value
430          yield (key, ma)

433  class BaseSearchCV(MetaEstimatorMixin, BaseEstimator, metaclass=ABCMeta):
434      """Abstract base class for hyper parameter search with cross-validation."""
435  
436      _parameter_constraints: dict = {
437          "estimator": [HasMethods(["fit"])],
438          "scoring": [
439              StrOptions(set(get_scorer_names())),
440              callable,
441              list,
442              tuple,
443              dict,
444              None,
445          ],
446          "n_jobs": [numbers.Integral, None],
447          "refit": ["boolean", str, callable],
448          "cv": ["cv_object"],
449          "verbose": ["verbose"],
450          "pre_dispatch": [numbers.Integral, str],
451          "error_score": [StrOptions({"raise"}), numbers.Real],
452          "return_train_score": ["boolean"],
453      }

455      @abstractmethod
456      def __init__(
457          self,
458          estimator,
459          *,
460          scoring=None,
461          n_jobs=None,
462          refit=True,
463          cv=None,
464          verbose=0,
465          pre_dispatch="2*n_jobs",
466          error_score=np.nan,
467          return_train_score=True,
468      ):
469          self.scoring = scoring
470          self.estimator = estimator
471          self.n_jobs = n_jobs
472          self.refit = refit
473          self.cv = cv
474          self.verbose = verbose
475          self.pre_dispatch = pre_dispatch
476          self.error_score = error_score
477          self.return_train_score = return_train_score

496      def score(self, X, y=None, **params):
497          """Return the score on the given data, if the estimator has been refit. 
498   
499          This uses the score defined by ``scoring`` where provided, and the 
500          ``best_estimator_.score`` method otherwise. 
501   
502          Parameters 
503          ---------- 
504          X : array-like of shape (n_samples, n_features) 
505              Input data, where `n_samples` is the number of samples and 
506              `n_features` is the number of features. 
507   
508          y : array-like of shape (n_samples, n_output) \ 
509              or (n_samples,), default=None 
510              Target relative to X for classification or regression; 
511              None for unsupervised learning. 
512   
513          **params : dict 
514              Parameters to be passed to the underlying scorer(s). 
515   
516              .. versionadded:: 1.4 
517                  Only available if `enable_metadata_routing=True`. See 
518                  :ref:`Metadata Routing User Guide <metadata_routing>` for more 
519                  details. 
520   
521          Returns 
522          ------- 
523          score : float 
524              The score defined by ``scoring`` if provided, and the 
525              ``best_estimator_.score`` method otherwise. 
526          """
527          _check_refit(self, "score")
528          check_is_fitted(self)
529  
530          _raise_for_params(params, self, "score")
531  
532          if _routing_enabled():
533              score_params = process_routing(self, "score", **params).scorer["score"]
534          else:
535              score_params = dict()
536  
537          if self.scorer_ is None:
538              raise ValueError(
539                  "No score function explicitly defined, "
540                  "and the estimator doesn't provide one %s" % self.best_estimator_
541              )
542          if isinstance(self.scorer_, dict):
543              if self.multimetric_:
544                  scorer = self.scorer_[self.refit]
545              else:
546                  scorer = self.scorer_
547              return scorer(self.best_estimator_, X, y, **score_params)
548  
549          # callable
550          score = self.scorer_(self.best_estimator_, X, y, **score_params)
551          if self.multimetric_:
552              score = score[self.refit]
553          return score

749      def _run_search(self, evaluate_candidates):
750          """Repeatedly calls `evaluate_candidates` to conduct a search. 
751   
752          This method, implemented in sub-classes, makes it possible to 
753          customize the scheduling of evaluations: GridSearchCV and 
754          RandomizedSearchCV schedule evaluations for their whole parameter 
755          search space at once but other more sequential approaches are also 
756          possible: for instance is possible to iteratively schedule evaluations 
757          for new regions of the parameter search space based on previously 
758          collected evaluation results. This makes it possible to implement 
759          Bayesian optimization or more generally sequential model-based 
760          optimization by deriving from the BaseSearchCV abstract base class. 
761          For example, Successive Halving is implemented by calling 
762          `evaluate_candidates` multiples times (once per iteration of the SH 
763          process), each time passing a different set of candidates with `X` 
764          and `y` of increasing sizes. 
765   
766          Parameters 
767          ---------- 
768          evaluate_candidates : callable 
769              This callback accepts: 
770                  - a list of candidates, where each candidate is a dict of 
771                    parameter settings. 
772                  - an optional `cv` parameter which can be used to e.g. 
773                    evaluate candidates on different dataset splits, or 
774                    evaluate candidates on subsampled data (as done in the 
775                    SucessiveHaling estimators). By default, the original `cv` 
776                    parameter is used, and it is available as a private 
777                    `_checked_cv_orig` attribute. 
778                  - an optional `more_results` dict. Each key will be added to 
779                    the `cv_results_` attribute. Values should be lists of 
780                    length `n_candidates` 
781   
782              It returns a dict of all results so far, formatted like 
783              ``cv_results_``. 
784   
785              Important note (relevant whether the default cv is used or not): 
786              in randomized splitters, and unless the random_state parameter of 
787              cv was set to an int, calling cv.split() multiple times will 
788              yield different splits. Since cv.split() is called in 
789              evaluate_candidates, this means that candidates will be evaluated 
790              on different splits each time evaluate_candidates is called. This 
791              might be a methodological issue depending on the search strategy 
792              that you're implementing. To prevent randomized splitters from 
793              being used, you may use _split._yields_constant_splits() 
794   
795          Examples 
796          -------- 
797   
798          :: 
799   
800              def _run_search(self, evaluate_candidates): 
801                  'Try C=0.1 only if C=1 is better than C=10' 
802                  all_results = evaluate_candidates([{'C': 1}, {'C': 10}]) 
803                  score = all_results['mean_test_score'] 
804                  if score[0] < score[1]: 
805                      evaluate_candidates([{'C': 0.1}]) 
806          """
807          raise NotImplementedError("_run_search not implemented.")

887      @_fit_context(
888          # *SearchCV.estimator is not validated yet
889          prefer_skip_nested_validation=False
890      )
891      def fit(self, X, y=None, **params):
892          """Run fit with all sets of parameters. 
893   
894          Parameters 
895          ---------- 
896   
897          X : array-like of shape (n_samples, n_features) or (n_samples, n_samples) 
898              Training vectors, where `n_samples` is the number of samples and 
899              `n_features` is the number of features. For precomputed kernel or 
900              distance matrix, the expected shape of X is (n_samples, n_samples). 
901   
902          y : array-like of shape (n_samples, n_output) \ 
903              or (n_samples,), default=None 
904              Target relative to X for classification or regression; 
905              None for unsupervised learning. 
906   
907          **params : dict of str -> object 
908              Parameters passed to the ``fit`` method of the estimator, the scorer, 
909              and the CV splitter. 
910   
911              If a fit parameter is an array-like whose length is equal to 
912              `num_samples` then it will be split by cross-validation along with 
913              `X` and `y`. For example, the :term:`sample_weight` parameter is 
914              split because `len(sample_weights) = len(X)`. However, this behavior 
915              does not apply to `groups` which is passed to the splitter configured 
916              via the `cv` parameter of the constructor. Thus, `groups` is used 
917              *to perform the split* and determines which samples are 
918              assigned to the each side of the a split. 
919   
920          Returns 
921          ------- 
922          self : object 
923              Instance of fitted estimator. 
924          """
925          estimator = self.estimator
926          scorers, refit_metric = self._get_scorers()
927  
928          X, y = indexable(X, y)
929          params = _check_method_params(X, params=params)
930  
931          routed_params = self._get_routed_params_for_fit(params)
932  
933          cv_orig = check_cv(self.cv, y, classifier=is_classifier(estimator))
934          n_splits = cv_orig.get_n_splits(X, y, **routed_params.splitter.split)
935  
936          base_estimator = clone(self.estimator)
937  
938          parallel = Parallel(n_jobs=self.n_jobs, pre_dispatch=self.pre_dispatch)
939  
940          fit_and_score_kwargs = dict(
941              scorer=scorers,
942              fit_params=routed_params.estimator.fit,
943              score_params=routed_params.scorer.score,
944              return_train_score=self.return_train_score,
945              return_n_test_samples=True,
946              return_times=True,
947              return_parameters=False,
948              error_score=self.error_score,
949              verbose=self.verbose,
950          )
951          results = {}
952          with parallel:
953              all_candidate_params = []
954              all_out = []
955              all_more_results = defaultdict(list)
956  
957              def evaluate_candidates(candidate_params, cv=None, more_results=None):
958                  cv = cv or cv_orig
959                  candidate_params = list(candidate_params)
960                  n_candidates = len(candidate_params)
961  
962                  if self.verbose > 0:
963                      print(
964                          "Fitting {0} folds for each of {1} candidates,"
965                          " totalling {2} fits".format(
966                              n_splits, n_candidates, n_candidates * n_splits
967                          )
968                      )
969  
970                  out = parallel(
971                      delayed(_fit_and_score)(
972                          clone(base_estimator),
973                          X,
974                          y,
975                          train=train,
976                          test=test,
977                          parameters=parameters,
978                          split_progress=(split_idx, n_splits),
979                          candidate_progress=(cand_idx, n_candidates),
980                          **fit_and_score_kwargs,
981                      )
982                      for (cand_idx, parameters), (split_idx, (train, test)) in product(
983                          enumerate(candidate_params),
984                          enumerate(cv.split(X, y, **routed_params.splitter.split)),
985                      )
986                  )
987  
988                  if len(out) < 1:
989                      raise ValueError(
990                          "No fits were performed. "
991                          "Was the CV iterator empty? "
992                          "Were there no candidates?"
993                      )
994                  elif len(out) != n_candidates * n_splits:
995                      raise ValueError(
996                          "cv.split and cv.get_n_splits returned "
997                          "inconsistent results. Expected {} "
998                          "splits, got {}".format(n_splits, len(out) // n_candidates)
999                      )
1000 
1001                 _warn_or_raise_about_fit_failures(out, self.error_score)
1002 
1003                 # For callable self.scoring, the return type is only know after
1004                 # calling. If the return type is a dictionary, the error scores
1005                 # can now be inserted with the correct key. The type checking
1006                 # of out will be done in `_insert_error_scores`.
1007                 if callable(self.scoring):
1008                     _insert_error_scores(out, self.error_score)
1009 
1010                 all_candidate_params.extend(candidate_params)
1011                 all_out.extend(out)
1012 
1013                 if more_results is not None:
1014                     for key, value in more_results.items():
1015                         all_more_results[key].extend(value)
1016 
1017                 nonlocal results
1018                 results = self._format_results(
1019                     all_candidate_params, n_splits, all_out, all_more_results
1020                 )
1021 
1022                 return results
1023 
1024             self._run_search(evaluate_candidates)
1025 
1026             # multimetric is determined here because in the case of a callable
1027             # self.scoring the return type is only known after calling
1028             first_test_score = all_out[0]["test_scores"]
1029             self.multimetric_ = isinstance(first_test_score, dict)
1030 
1031             # check refit_metric now for a callable scorer that is multimetric
1032             if callable(self.scoring) and self.multimetric_:
1033                 self._check_refit_for_multimetric(first_test_score)
1034                 refit_metric = self.refit
1035 
1036         # For multi-metric evaluation, store the best_index_, best_params_ and
1037         # best_score_ iff refit is one of the scorer names
1038         # In single metric evaluation, refit_metric is "score"
1039         if self.refit or not self.multimetric_:
1040             self.best_index_ = self._select_best_index(
1041                 self.refit, refit_metric, results
1042             )
1043             if not callable(self.refit):
1044                 # With a non-custom callable, we can select the best score
1045                 # based on the best index
1046                 self.best_score_ = results[f"mean_test_{refit_metric}"][
1047                     self.best_index_
1048                 ]
1049             self.best_params_ = results["params"][self.best_index_]
1050 
1051         if self.refit:
1052             # here we clone the estimator as well as the parameters, since
1053             # sometimes the parameters themselves might be estimators, e.g.
1054             # when we search over different estimators in a pipeline.
1055             # ref: https://github.com/scikit-learn/scikit-learn/pull/26786
1056             self.best_estimator_ = clone(base_estimator).set_params(
1057                 **clone(self.best_params_, safe=False)
1058             )
1059 
1060             refit_start_time = time.time()
1061             if y is not None:
1062                 self.best_estimator_.fit(X, y, **routed_params.estimator.fit)
1063             else:
1064                 self.best_estimator_.fit(X, **routed_params.estimator.fit)
1065             refit_end_time = time.time()
1066             self.refit_time_ = refit_end_time - refit_start_time
1067 
1068             if hasattr(self.best_estimator_, "feature_names_in_"):
1069                 self.feature_names_in_ = self.best_estimator_.feature_names_in_
1070 
1071         # Store the only scorer not as a dict for single metric evaluation
1072         if isinstance(scorers, _MultimetricScorer):
1073             self.scorer_ = scorers._scorers
1074         else:
1075             self.scorer_ = scorers
1076 
1077         self.cv_results_ = results
1078         self.n_splits_ = n_splits
1079 
1080         return self

1082     def _format_results(self, candidate_params, n_splits, out, more_results=None):
1083         n_candidates = len(candidate_params)
1084         out = _aggregate_score_dicts(out)
1085 
1086         results = dict(more_results or {})
1087         for key, val in results.items():
1088             # each value is a list (as per evaluate_candidate's convention)
1089             # we convert it to an array for consistency with the other keys
1090             results[key] = np.asarray(val)
1091 
1092         def _store(key_name, array, weights=None, splits=False, rank=False):
1093             """A small helper to store the scores/times to the cv_results_"""
1094             # When iterated first by splits, then by parameters
1095             # We want `array` to have `n_candidates` rows and `n_splits` cols.
1096             array = np.array(array, dtype=np.float64).reshape(n_candidates, n_splits)
1097             if splits:
1098                 for split_idx in range(n_splits):
1099                     # Uses closure to alter the results
1100                     results["split%d_%s" % (split_idx, key_name)] = array[:, split_idx]
1101 
1102             array_means = np.average(array, axis=1, weights=weights)
1103             results["mean_%s" % key_name] = array_means
1104 
1105             if key_name.startswith(("train_", "test_")) and np.any(
1106                 ~np.isfinite(array_means)
1107             ):
1108                 warnings.warn(
1109                     (
1110                         f"One or more of the {key_name.split('_')[0]} scores "
1111                         f"are non-finite: {array_means}"
1112                     ),
1113                     category=UserWarning,
1114                 )
1115 
1116             # Weighted std is not directly available in numpy
1117             array_stds = np.sqrt(
1118                 np.average(
1119                     (array - array_means[:, np.newaxis]) ** 2, axis=1, weights=weights
1120                 )
1121             )
1122             results["std_%s" % key_name] = array_stds
1123 
1124             if rank:
1125                 # When the fit/scoring fails `array_means` contains NaNs, we
1126                 # will exclude them from the ranking process and consider them
1127                 # as tied with the worst performers.
1128                 if np.isnan(array_means).all():
1129                     # All fit/scoring routines failed.
1130                     rank_result = np.ones_like(array_means, dtype=np.int32)
1131                 else:
1132                     min_array_means = np.nanmin(array_means) - 1
1133                     array_means = np.nan_to_num(array_means, nan=min_array_means)
1134                     rank_result = rankdata(-array_means, method="min").astype(
1135                         np.int32, copy=False
1136                     )
1137                 results["rank_%s" % key_name] = rank_result
1138 
1139         _store("fit_time", out["fit_time"])
1140         _store("score_time", out["score_time"])
1141         # Store a list of param dicts at the key 'params'
1142         for param, ma in _yield_masked_array_for_each_param(candidate_params):
1143             results[param] = ma
1144         results["params"] = candidate_params
1145 
1146         test_scores_dict = _normalize_score_results(out["test_scores"])
1147         if self.return_train_score:
1148             train_scores_dict = _normalize_score_results(out["train_scores"])
1149 
1150         for scorer_name in test_scores_dict:
1151             # Computed the (weighted) mean and std for test scores alone
1152             _store(
1153                 "test_%s" % scorer_name,
1154                 test_scores_dict[scorer_name],
1155                 splits=True,
1156                 rank=True,
1157                 weights=None,
1158             )
1159             if self.return_train_score:
1160                 _store(
1161                     "train_%s" % scorer_name,
1162                     train_scores_dict[scorer_name],
1163                     splits=True,
1164                 )
1165 
1166         return results

1215 class GridSearchCV(BaseSearchCV):
1216     """Exhaustive search over specified parameter values for an estimator. 
1217  
1218     Important members are fit, predict. 
1219  
1220     GridSearchCV implements a "fit" and a "score" method. 
1221     It also implements "score_samples", "predict", "predict_proba", 
1222     "decision_function", "transform" and "inverse_transform" if they are 
1223     implemented in the estimator used. 
1224  
1225     The parameters of the estimator used to apply these methods are optimized 
1226     by cross-validated grid-search over a parameter grid. 
1227  
1228     Read more in the :ref:`User Guide <grid_search>`. 
1229  
1230     Parameters 
1231     ---------- 
1232     estimator : estimator object 
1233         This is assumed to implement the scikit-learn estimator interface. 
1234         Either estimator needs to provide a ``score`` function, 
1235         or ``scoring`` must be passed. 
1236  
1237     param_grid : dict or list of dictionaries 
1238         Dictionary with parameters names (`str`) as keys and lists of 
1239         parameter settings to try as values, or a list of such 
1240         dictionaries, in which case the grids spanned by each dictionary 
1241         in the list are explored. This enables searching over any sequence 
1242         of parameter settings. 
1243  
1244     scoring : str, callable, list, tuple or dict, default=None 
1245         Strategy to evaluate the performance of the cross-validated model on 
1246         the test set. 
1247  
1248         If `scoring` represents a single score, one can use: 
1249  
1250         - a single string (see :ref:`scoring_parameter`); 
1251         - a callable (see :ref:`scoring_callable`) that returns a single value. 
1252  
1253         If `scoring` represents multiple scores, one can use: 
1254  
1255         - a list or tuple of unique strings; 
1256         - a callable returning a dictionary where the keys are the metric 
1257           names and the values are the metric scores; 
1258         - a dictionary with metric names as keys and callables as values. 
1259  
1260         See :ref:`multimetric_grid_search` for an example. 
1261  
1262     n_jobs : int, default=None 
1263         Number of jobs to run in parallel. 
1264         ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
1265         ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
1266         for more details. 
1267  
1268         .. versionchanged:: v0.20 
1269            `n_jobs` default changed from 1 to None 
1270  
1271     refit : bool, str, or callable, default=True 
1272         Refit an estimator using the best found parameters on the whole 
1273         dataset. 
1274  
1275         For multiple metric evaluation, this needs to be a `str` denoting the 
1276         scorer that would be used to find the best parameters for refitting 
1277         the estimator at the end. 
1278  
1279         Where there are considerations other than maximum score in 
1280         choosing a best estimator, ``refit`` can be set to a function which 
1281         returns the selected ``best_index_`` given ``cv_results_``. In that 
1282         case, the ``best_estimator_`` and ``best_params_`` will be set 
1283         according to the returned ``best_index_`` while the ``best_score_`` 
1284         attribute will not be available. 
1285  
1286         The refitted estimator is made available at the ``best_estimator_`` 
1287         attribute and permits using ``predict`` directly on this 
1288         ``GridSearchCV`` instance. 
1289  
1290         Also for multiple metric evaluation, the attributes ``best_index_``, 
1291         ``best_score_`` and ``best_params_`` will only be available if 
1292         ``refit`` is set and all of them will be determined w.r.t this specific 
1293         scorer. 
1294  
1295         See ``scoring`` parameter to know more about multiple metric 
1296         evaluation.

1575     """Randomized search on hyper parameters. 
1576  
1577     RandomizedSearchCV implements a "fit" and a "score" method. 
1578     It also implements "score_samples", "predict", "predict_proba", 
1579     "decision_function", "transform" and "inverse_transform" if they are 
1580     implemented in the estimator used. 
1581  
1582     The parameters of the estimator used to apply these methods are optimized 
1583     by cross-validated search over parameter settings. 
1584  
1585     In contrast to GridSearchCV, not all parameter values are tried out, but 
1586     rather a fixed number of parameter settings is sampled from the specified 
1587     distributions. The number of parameter settings that are tried is 
1588     given by n_iter. 
1589  
1590     If all parameters are presented as a list, 
1591     sampling without replacement is performed. If at least one parameter 
1592     is given as a distribution, sampling with replacement is used. 
1593     It is highly recommended to use continuous distributions for continuous 
1594     parameters.

1881     Notes 
1882     ----- 
1883     The parameters selected are those that maximize the score of the held-out 
1884     data, according to the scoring parameter. 
1885  
1886     If `n_jobs` was set to a value higher than one, the data is copied for each 
1887     parameter setting(and not `n_jobs` times). This is done for efficiency 
1888     reasons if individual jobs take very little time, but may raise errors if 
1889     the dataset is large and not enough memory is available.  A workaround in 
1890     this case is to set `pre_dispatch`. Then, the memory is copied only 
1891     `pre_dispatch` many times. A reasonable value for `pre_dispatch` is `2 * 
1892     n_jobs`.
