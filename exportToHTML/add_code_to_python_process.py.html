90   @contextmanager
91   def _create_win_event(name):
92       from winappdbg.win32.kernel32 import CreateEventA, WaitForSingleObject, CloseHandle
93   
94       manual_reset = False  # i.e.: after someone waits it, automatically set to False.
95       initial_state = False
96       if not isinstance(name, bytes):
97           name = name.encode("utf-8")
98       event = CreateEventA(None, manual_reset, initial_state, name)
99       if not event:
100          raise ctypes.WinError()
101  
102      class _WinEvent(object):
103          def wait_for_event_set(self, timeout=None):
104              """ 
105              :param timeout: in seconds 
106              """
107              if timeout is None:
108                  timeout = 0xFFFFFFFF
109              else:
110                  timeout = int(timeout * 1000)
111              ret = WaitForSingleObject(event, timeout)
112              if ret in (0, 0x80):
113                  return True
114              elif ret == 0x102:
115                  # Timed out
116                  return False
117              else:
118                  raise ctypes.WinError()
119  
120      try:
121          yield _WinEvent()
122      finally:
123          CloseHandle(event)

135  def get_target_filename(is_target_process_64=None, prefix=None, extension=None):
136      # Note: we have an independent (and similar -- but not equal) version of this method in
137      # `pydevd_tracing.py` which should be kept synchronized with this one (we do a copy
138      # because the `pydevd_attach_to_process` is mostly independent and shouldn't be imported in the
139      # debugger -- the only situation where it's imported is if the user actually does an attach to
140      # process, through `attach_pydevd.py`, but this should usually be called from the IDE directly
141      # and not from the debugger).
142      libdir = os.path.dirname(os.path.abspath(__file__))
143  
144      if is_target_process_64 is None:
145          if IS_WINDOWS:
146              # i.e.: On windows the target process could have a different bitness (32bit is emulated on 64bit).
147              raise AssertionError("On windows it's expected that the target bitness is specified.")
148  
149          # For other platforms, just use the the same bitness of the process we're running in.
150          is_target_process_64 = is_python_64bit()
151  
152      arch = ""
153      if IS_WINDOWS:
154          # prefer not using platform.machine() when possible (it's a bit heavyweight as it may
155          # spawn a subprocess).
156          arch = os.environ.get("PROCESSOR_ARCHITEW6432", os.environ.get("PROCESSOR_ARCHITECTURE", ""))
157  
158      if not arch:
159          arch = platform.machine()
160          if not arch:
161              print("platform.machine() did not return valid value.")  # This shouldn't happen...
162              return None
163  
164      if IS_WINDOWS:
165          if not extension:
166              extension = ".dll"
167          suffix_64 = "amd64"
168          suffix_32 = "x86"
169  
170      elif IS_LINUX:
171          if not extension:
172              extension = ".so"
173          suffix_64 = "amd64"
174          suffix_32 = "x86"
175  
176      elif IS_MAC:
177          if not extension:
178              extension = ".dylib"
179          suffix_64 = "x86_64"
180          suffix_32 = "x86"
181  
182      else:
183          print("Unable to attach to process in platform: %s", sys.platform)
184          return None
185  
186      if arch.lower() not in ("amd64", "x86", "x86_64", "i386", "x86"):
187          # We don't support this processor by default. Still, let's support the case where the
188          # user manually compiled it himself with some heuristics.
189          #
190          # Ideally the user would provide a library in the format: "attach_<arch>.<extension>"
191          # based on the way it's currently compiled -- see:
192          # - windows/compile_windows.bat
193          # - linux_and_mac/compile_linux.sh
194          # - linux_and_mac/compile_mac.sh
195  
196          try:
197              found = [name for name in os.listdir(libdir) if name.startswith("attach_") and name.endswith(extension)]
198          except:
199              print("Error listing dir: %s" % (libdir,))
200              traceback.print_exc()
201              return None
202  
203          if prefix:
204              expected_name = prefix + arch + extension
205              expected_name_linux = prefix + "linux_" + arch + extension
206          else:
207              # Default is looking for the attach_ / attach_linux
208              expected_name = "attach_" + arch + extension
209              expected_name_linux = "attach_linux_" + arch + extension
210  
211          filename = None
212          if expected_name in found:  # Heuristic: user compiled with "attach_<arch>.<extension>"
213              filename = os.path.join(libdir, expected_name)
214  
215          elif IS_LINUX and expected_name_linux in found:  # Heuristic: user compiled with "attach_linux_<arch>.<extension>"
216              filename = os.path.join(libdir, expected_name_linux)
217  
218          elif len(found) == 1:  # Heuristic: user removed all libraries and just left his own lib.
219              filename = os.path.join(libdir, found[0])
220  
221          else:  # Heuristic: there's one additional library which doesn't seem to be our own. Find the odd one.
222              filtered = [name for name in found if not name.endswith((suffix_64 + extension, suffix_32 + extension))]
223              if len(filtered) == 1:  # If more than one is available we can't be sure...
224                  filename = os.path.join(libdir, found[0])
225  
226          if filename is None:
227              print("Unable to attach to process in arch: %s (did not find %s in %s)." % (arch, expected_name, libdir))
228              return None
229  
230          print("Using %s in arch: %s." % (filename, arch))
231  
232      else:
233          if is_target_process_64:
234              suffix = suffix_64
235          else:
236              suffix = suffix_32
237  
238          if not prefix:
239              # Default is looking for the attach_ / attach_linux
240              if IS_WINDOWS or IS_MAC:  # just the extension changes
241                  prefix = "attach_"
242              elif IS_LINUX:
243                  prefix = "attach_linux_"  # historically it has a different name
244              else:
245                  print("Unable to attach to process in platform: %s" % (sys.platform,))
246                  return None
247  
248          filename = os.path.join(libdir, "%s%s%s" % (prefix, suffix, extension))
249  
250      if not os.path.exists(filename):
251          print("Expected: %s to exist." % (filename,))
252          return None
253  
254      return filename

257  def run_python_code_windows(pid, python_code, connect_debugger_tracing=False, show_debug_info=0):
258      assert "'" not in python_code, "Having a single quote messes with our command."
259  
260      # Suppress winappdbg warning about sql package missing.
261      import warnings
262  
263      with warnings.catch_warnings():
264          warnings.simplefilter("ignore", category=ImportWarning)
265          from winappdbg.process import Process
266  
267      if not isinstance(python_code, bytes):
268          python_code = python_code.encode("utf-8")
269  
270      process = Process(pid)
271      bits = process.get_bits()
272      is_target_process_64 = bits == 64
273  
274      # Note: this restriction no longer applies (we create a process with the proper bitness from
275      # this process so that the attach works).
276      # if is_target_process_64 != is_python_64bit():
277      #     raise RuntimeError("The architecture of the Python used to connect doesn't match the architecture of the target.\n"
278      #     "Target 64 bits: %s\n"
279      #     "Current Python 64 bits: %s" % (is_target_process_64, is_python_64bit()))
280  
281      with _acquire_mutex("_pydevd_pid_attach_mutex_%s" % (pid,), 10):
282          print("--- Connecting to %s bits target (current process is: %s) ---" % (bits, 64 if is_python_64bit() else 32))
283          sys.stdout.flush()
284  
285          with _win_write_to_shared_named_memory(python_code, pid):
286              target_executable = get_target_filename(is_target_process_64, "inject_dll_", ".exe")
287              if not target_executable:
288                  raise RuntimeError("Could not find expected .exe file to inject dll in attach to process.")
289  
290              target_dll = get_target_filename(is_target_process_64)
291              if not target_dll:
292                  raise RuntimeError("Could not find expected .dll file in attach to process.")
293  
294              print("\n--- Injecting attach dll: %s into pid: %s ---" % (os.path.basename(target_dll), pid))
295              sys.stdout.flush()
296              args = [target_executable, str(pid), target_dll]
297              subprocess.check_call(args)
298  
299              # Now, if the first injection worked, go on to the second which will actually
300              # run the code.
301              target_dll_run_on_dllmain = get_target_filename(is_target_process_64, "run_code_on_dllmain_", ".dll")
302              if not target_dll_run_on_dllmain:
303                  raise RuntimeError("Could not find expected .dll in attach to process.")
304  
305              with _create_win_event("_pydevd_pid_event_%s" % (pid,)) as event:
306                  print("\n--- Injecting run code dll: %s into pid: %s ---" % (os.path.basename(target_dll_run_on_dllmain), pid))
307                  sys.stdout.flush()
308                  args = [target_executable, str(pid), target_dll_run_on_dllmain]
309                  subprocess.check_call(args)
310  
311                  if not event.wait_for_event_set(15):
312                      print("Timeout error: the attach may not have completed.")
313                      sys.stdout.flush()
314              print("--- Finished dll injection ---\n")
315              sys.stdout.flush()
316  
317      return 0

320  @contextmanager
321  def _acquire_mutex(mutex_name, timeout):
322      """ 
323      Only one process may be attaching to a pid, so, create a system mutex 
324      to make sure this holds in practice. 
325      """
326      from winappdbg.win32.kernel32 import CreateMutex, GetLastError, CloseHandle
327      from winappdbg.win32.defines import ERROR_ALREADY_EXISTS
328  
329      initial_time = time.time()
330      while True:
331          mutex = CreateMutex(None, True, mutex_name)
332          acquired = GetLastError() != ERROR_ALREADY_EXISTS
333          if acquired:
334              break
335          if time.time() - initial_time > timeout:
336              raise TimeoutError("Unable to acquire mutex to make attach before timeout.")
337          time.sleep(0.2)
338  
339      try:
340          yield
341      finally:
342          CloseHandle(mutex)

345  @contextmanager
346  def _win_write_to_shared_named_memory(python_code, pid):
347      # Use the definitions from winappdbg when possible.
348      from winappdbg.win32 import defines
349      from winappdbg.win32.kernel32 import (
350          CreateFileMapping,
351          MapViewOfFile,
352          CloseHandle,
353          UnmapViewOfFile,
354      )
355  
356      memmove = ctypes.cdll.msvcrt.memmove
357      memmove.argtypes = [
358          ctypes.c_void_p,
359          ctypes.c_void_p,
360          defines.SIZE_T,
361      ]
362      memmove.restype = ctypes.c_void_p
363  
364      # Note: BUFSIZE must be the same from run_code_in_memory.hpp
365      BUFSIZE = 2048
366      assert isinstance(python_code, bytes)
367      assert len(python_code) > 0, "Python code must not be empty."
368      # Note: -1 so that we're sure we'll add a \0 to the end.
369      assert len(python_code) < BUFSIZE - 1, "Python code must have at most %s bytes (found: %s)" % (BUFSIZE - 1, len(python_code))
370  
371      python_code += b"\0" * (BUFSIZE - len(python_code))
372      assert python_code.endswith(b"\0")
373  
374      INVALID_HANDLE_VALUE = -1
375      PAGE_READWRITE = 0x4
376      FILE_MAP_WRITE = 0x2
377      filemap = CreateFileMapping(INVALID_HANDLE_VALUE, 0, PAGE_READWRITE, 0, BUFSIZE, "__pydevd_pid_code_to_run__%s" % (pid,))
378  
379      if filemap == INVALID_HANDLE_VALUE or filemap is None:
380          raise Exception("Failed to create named file mapping (ctypes: CreateFileMapping): %s" % (filemap,))
381      try:
382          view = MapViewOfFile(filemap, FILE_MAP_WRITE, 0, 0, 0)
383          if not view:
384              raise Exception("Failed to create view of named file mapping (ctypes: MapViewOfFile).")
385  
386          try:
387              memmove(view, python_code, BUFSIZE)
388              yield
389          finally:
390              UnmapViewOfFile(view)
391      finally:
392          CloseHandle(filemap)

395  def run_python_code_linux(pid, python_code, connect_debugger_tracing=False, show_debug_info=0):
396      assert "'" not in python_code, "Having a single quote messes with our command."
397  
398      target_dll = get_target_filename()
399      if not target_dll:
400          libdir = os.path.dirname(os.path.abspath(__file__))
401          found = [name for name in os.listdir(libdir)]
402          raise RuntimeError(
403              "Could not find .so for attach to process.\nlibdir: %s\nAvailable in dir: %s"
404              % (
405                  libdir,
406                  found,
407              )
408          )
409      target_dll_name = os.path.splitext(os.path.basename(target_dll))[0]
410  
411      # Note: we currently don't support debug builds
412      is_debug = 0
413      # Note that the space in the beginning of each line in the multi-line is important!
414      cmd = [
415          "gdb",
416          "--nw",  # no gui interface
417          "--nh",  # no ~/.gdbinit
418          "--nx",  # no .gdbinit
419          #         '--quiet',  # no version number on startup
420          "--pid",
421          str(pid),
422          "--batch",
423          #         '--batch-silent',
424      ]
425  
426      # PYDEVD_GDB_SCAN_SHARED_LIBRARIES can be a list of strings with the shared libraries
427      # which should be scanned by default to make the attach to process (i.e.: libdl, libltdl, libc, libfreebl3).
428      #
429      # The default is scanning all shared libraries, but on some cases this can be in the 20-30
430      # seconds range for some corner cases.
431      # See: https://github.com/JetBrains/intellij-community/pull/1608
432      #
433      # By setting PYDEVD_GDB_SCAN_SHARED_LIBRARIES (to a comma-separated string), it's possible to
434      # specify just a few libraries to be loaded (not many are needed for the attach,
435      # but it can be tricky to pre-specify for all Linux versions as this may change
436      # across different versions).
437      #
438      # See: https://github.com/microsoft/debugpy/issues/762#issuecomment-947103844
439      # for a comment that explains the basic steps on how to discover what should be available
440      # in each case (mostly trying different versions based on the output of gdb).
441      #
442      # The upside is that for cases when too many libraries are loaded the attach could be slower
443      # and just specifying the one that is actually needed for the attach can make it much faster.
444      #
445      # The downside is that it may be dependent on the Linux version being attached to (which is the
446      # reason why this is no longer done by default -- see: https://github.com/microsoft/debugpy/issues/882).
447      gdb_load_shared_libraries = os.environ.get("PYDEVD_GDB_SCAN_SHARED_LIBRARIES", "").strip()
448      if gdb_load_shared_libraries:
449          print("PYDEVD_GDB_SCAN_SHARED_LIBRARIES set: %s." % (gdb_load_shared_libraries,))
450          cmd.extend(["--init-eval-command='set auto-solib-add off'"])  # Don't scan all libraries.
451  
452          for lib in gdb_load_shared_libraries.split(","):
453              lib = lib.strip()
454              cmd.extend(["--eval-command='sharedlibrary %s'" % (lib,)])  # Scan the specified library
455      else:
456          print("PYDEVD_GDB_SCAN_SHARED_LIBRARIES not set (scanning all libraries for needed symbols).")
457  
458      cmd.extend(["--eval-command='set scheduler-locking off'"])  # If on we'll deadlock.
459  
460      # Leave auto by default (it should do the right thing as we're attaching to a process in the
461      # current host).
462      cmd.extend(["--eval-command='set architecture auto'"])
463  
464      cmd.extend(
465          [
466              "--eval-command='call (void*)dlopen(\"%s\", 2)'" % target_dll,
467              "--eval-command='sharedlibrary %s'" % target_dll_name,
468              "--eval-command='call (int)DoAttach(%s, \"%s\", %s)'" % (is_debug, python_code, show_debug_info),
469          ]
470      )
471  
472      # print ' '.join(cmd)
473  
474      env = os.environ.copy()
475      # Remove the PYTHONPATH (if gdb has a builtin Python it could fail if we
476      # have the PYTHONPATH for a different python version or some forced encoding).
477      env.pop("PYTHONIOENCODING", None)
478      env.pop("PYTHONPATH", None)
479      print("Running: %s" % (" ".join(cmd)))
480      subprocess.check_call(" ".join(cmd), shell=True, env=env)
