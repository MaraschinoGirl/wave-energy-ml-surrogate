28   def barycenter_weights(X, Y, indices, reg=1e-3):
29       """Compute barycenter weights of X from Y along the first axis 
30    
31       We estimate the weights to assign to each point in Y[indices] to recover 
32       the point X[i]. The barycenter weights sum to 1. 
33    
34       Parameters 
35       ---------- 
36       X : array-like, shape (n_samples, n_dim) 
37    
38       Y : array-like, shape (n_samples, n_dim) 
39    
40       indices : array-like, shape (n_samples, n_dim) 
41               Indices of the points in Y used to compute the barycenter 
42    
43       reg : float, default=1e-3 
44           Amount of regularization to add for the problem to be 
45           well-posed in the case of n_neighbors > n_dim 
46    
47       Returns 
48       ------- 
49       B : array-like, shape (n_samples, n_neighbors) 
50    
51       Notes 
52       ----- 
53       See developers note for more information. 
54       """
55       X = check_array(X, dtype=FLOAT_DTYPES)
56       Y = check_array(Y, dtype=FLOAT_DTYPES)
57       indices = check_array(indices, dtype=int)
58   
59       n_samples, n_neighbors = indices.shape
60       assert X.shape[0] == n_samples
61   
62       B = np.empty((n_samples, n_neighbors), dtype=X.dtype)
63       v = np.ones(n_neighbors, dtype=X.dtype)
64   
65       # this might raise a LinalgError if G is singular and has trace
66       # zero
67       for i, ind in enumerate(indices):
68           A = Y[ind]
69           C = A - X[i]  # broadcasting
70           G = np.dot(C, C.T)
71           trace = np.trace(G)
72           if trace > 0:
73               R = reg * trace
74           else:
75               R = reg
76           G.flat[:: n_neighbors + 1] += R
77           w = solve(G, v, assume_a="pos")
78           B[i, :] = w / np.sum(w)
79       return B

82   def barycenter_kneighbors_graph(X, n_neighbors, reg=1e-3, n_jobs=None):
83       """Computes the barycenter weighted graph of k-Neighbors for points in X 
84    
85       Parameters 
86       ---------- 
87       X : {array-like, NearestNeighbors} 
88           Sample data, shape = (n_samples, n_features), in the form of a 
89           numpy array or a NearestNeighbors object. 
90    
91       n_neighbors : int 
92           Number of neighbors for each sample. 
93    
94       reg : float, default=1e-3 
95           Amount of regularization when solving the least-squares 
96           problem. Only relevant if mode='barycenter'. If None, use the 
97           default. 
98    
99       n_jobs : int or None, default=None 
100          The number of parallel jobs to run for neighbors search. 
101          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
102          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
103          for more details. 
104   
105      Returns 
106      ------- 
107      A : sparse matrix in CSR format, shape = [n_samples, n_samples] 
108          A[i, j] is assigned the weight of edge that connects i to j. 
109   
110      See Also 
111      -------- 
112      sklearn.neighbors.kneighbors_graph 
113      sklearn.neighbors.radius_neighbors_graph 
114      """
115      knn = NearestNeighbors(n_neighbors=n_neighbors + 1, n_jobs=n_jobs).fit(X)
116      X = knn._fit_X
117      n_samples = knn.n_samples_fit_
118      ind = knn.kneighbors(X, return_distance=False)[:, 1:]
119      data = barycenter_weights(X, X, ind, reg=reg)
120      indptr = np.arange(0, n_samples * n_neighbors + 1, n_neighbors)
121      return csr_matrix((data.ravel(), ind.ravel(), indptr), shape=(n_samples, n_samples))

124  def null_space(
125      M, k, k_skip=1, eigen_solver="arpack", tol=1e-6, max_iter=100, random_state=None
126  ):
127      """ 
128      Find the null space of a matrix M. 
129   
130      Parameters 
131      ---------- 
132      M : {array, matrix, sparse matrix, LinearOperator} 
133          Input covariance matrix: should be symmetric positive semi-definite 
134   
135      k : int 
136          Number of eigenvalues/vectors to return 
137   
138      k_skip : int, default=1 
139          Number of low eigenvalues to skip. 
140   
141      eigen_solver : {'auto', 'arpack', 'dense'}, default='arpack' 
142          auto : algorithm will attempt to choose the best method for input data 
143          arpack : use arnoldi iteration in shift-invert mode. 
144                      For this method, M may be a dense matrix, sparse matrix, 
145                      or general linear operator. 
146                      Warning: ARPACK can be unstable for some problems.  It is 
147                      best to try several random seeds in order to check results. 
148          dense  : use standard dense matrix operations for the eigenvalue 
149                      decomposition.  For this method, M must be an array 
150                      or matrix type.  This method should be avoided for 
151                      large problems. 
152   
153      tol : float, default=1e-6 
154          Tolerance for 'arpack' method. 
155          Not used if eigen_solver=='dense'. 
156   
157      max_iter : int, default=100 
158          Maximum number of iterations for 'arpack' method. 
159          Not used if eigen_solver=='dense' 
160   
161      random_state : int, RandomState instance, default=None 
162          Determines the random number generator when ``solver`` == 'arpack'. 
163          Pass an int for reproducible results across multiple function calls. 
164          See :term:`Glossary <random_state>`. 
165      """
166      if eigen_solver == "auto":
167          if M.shape[0] > 200 and k + k_skip < 10:
168              eigen_solver = "arpack"
169          else:
170              eigen_solver = "dense"
171  
172      if eigen_solver == "arpack":
173          v0 = _init_arpack_v0(M.shape[0], random_state)
174          try:
175              eigen_values, eigen_vectors = eigsh(
176                  M, k + k_skip, sigma=0.0, tol=tol, maxiter=max_iter, v0=v0
177              )
178          except RuntimeError as e:
179              raise ValueError(
180                  "Error in determining null-space with ARPACK. Error message: "
181                  "'%s'. Note that eigen_solver='arpack' can fail when the "
182                  "weight matrix is singular or otherwise ill-behaved. In that "
183                  "case, eigen_solver='dense' is recommended. See online "
184                  "documentation for more information." % e
185              ) from e
186  
187          return eigen_vectors[:, k_skip:], np.sum(eigen_values[k_skip:])
188      elif eigen_solver == "dense":
189          if hasattr(M, "toarray"):
190              M = M.toarray()
191          eigen_values, eigen_vectors = eigh(
192              M, subset_by_index=(k_skip, k + k_skip - 1), overwrite_a=True
193          )
194          index = np.argsort(np.abs(eigen_values))
195          return eigen_vectors[:, index], np.sum(eigen_values)
196      else:
197          raise ValueError("Unrecognized eigen_solver '%s'" % eigen_solver)

200  def _locally_linear_embedding(
201      X,
202      *,
203      n_neighbors,
204      n_components,
205      reg=1e-3,
206      eigen_solver="auto",
207      tol=1e-6,
208      max_iter=100,
209      method="standard",
210      hessian_tol=1e-4,
211      modified_tol=1e-12,
212      random_state=None,
213      n_jobs=None,
214  ):
215      nbrs = NearestNeighbors(n_neighbors=n_neighbors + 1, n_jobs=n_jobs)
216      nbrs.fit(X)
217      X = nbrs._fit_X
218  
219      N, d_in = X.shape
220  
221      if n_components > d_in:
222          raise ValueError(
223              "output dimension must be less than or equal to input dimension"
224          )
225      if n_neighbors >= N:
226          raise ValueError(
227              "Expected n_neighbors <= n_samples,  but n_samples = %d, n_neighbors = %d"
228              % (N, n_neighbors)
229          )
230  
231      M_sparse = eigen_solver != "dense"
232      M_container_constructor = lil_matrix if M_sparse else np.zeros
233  
234      if method == "standard":
235          W = barycenter_kneighbors_graph(
236              nbrs, n_neighbors=n_neighbors, reg=reg, n_jobs=n_jobs
237          )
238  
239          # we'll compute M = (I-W)'(I-W)
240          # depending on the solver, we'll do this differently
241          if M_sparse:
242              M = eye(*W.shape, format=W.format) - W
243              M = M.T * M
244          else:
245              M = (W.T * W - W.T - W).toarray()
246              M.flat[:: M.shape[0] + 1] += 1  # M = W' W - W' - W + I
247  
248      elif method == "hessian":
249          dp = n_components * (n_components + 1) // 2
250  
251          if n_neighbors <= n_components + dp:
252              raise ValueError(
253                  "for method='hessian', n_neighbors must be "
254                  "greater than "
255                  "[n_components * (n_components + 3) / 2]"
256              )
257  
258          neighbors = nbrs.kneighbors(
259              X, n_neighbors=n_neighbors + 1, return_distance=False
260          )
261          neighbors = neighbors[:, 1:]
262  
263          Yi = np.empty((n_neighbors, 1 + n_components + dp), dtype=np.float64)
264          Yi[:, 0] = 1
265  
266          M = M_container_constructor((N, N), dtype=np.float64)
267  
268          use_svd = n_neighbors > d_in
269  
270          for i in range(N):
271              Gi = X[neighbors[i]]
272              Gi -= Gi.mean(0)
273  
274              # build Hessian estimator
275              if use_svd:
276                  U = svd(Gi, full_matrices=0)[0]
277              else:
278                  Ci = np.dot(Gi, Gi.T)
279                  U = eigh(Ci)[1][:, ::-1]
280  
281              Yi[:, 1 : 1 + n_components] = U[:, :n_components]
282  
283              j = 1 + n_components
284              for k in range(n_components):
285                  Yi[:, j : j + n_components - k] = U[:, k : k + 1] * U[:, k:n_components]
286                  j += n_components - k
287  
288              Q, R = qr(Yi)
289  
290              w = Q[:, n_components + 1 :]
291              S = w.sum(0)
292  
293              S[np.where(abs(S) < hessian_tol)] = 1
294              w /= S
295  
296              nbrs_x, nbrs_y = np.meshgrid(neighbors[i], neighbors[i])
297              M[nbrs_x, nbrs_y] += np.dot(w, w.T)
298  
299      elif method == "modified":
300          if n_neighbors < n_components:
301              raise ValueError("modified LLE requires n_neighbors >= n_components")
302  
303          neighbors = nbrs.kneighbors(
304              X, n_neighbors=n_neighbors + 1, return_distance=False
305          )
306          neighbors = neighbors[:, 1:]
307  
308          # find the eigenvectors and eigenvalues of each local covariance
309          # matrix. We want V[i] to be a [n_neighbors x n_neighbors] matrix,
310          # where the columns are eigenvectors
311          V = np.zeros((N, n_neighbors, n_neighbors))
312          nev = min(d_in, n_neighbors)
313          evals = np.zeros([N, nev])
314  
315          # choose the most efficient way to find the eigenvectors
316          use_svd = n_neighbors > d_in
317  
318          if use_svd:
319              for i in range(N):
320                  X_nbrs = X[neighbors[i]] - X[i]
321                  V[i], evals[i], _ = svd(X_nbrs, full_matrices=True)
322              evals **= 2
323          else:
324              for i in range(N):
325                  X_nbrs = X[neighbors[i]] - X[i]
326                  C_nbrs = np.dot(X_nbrs, X_nbrs.T)
327                  evi, vi = eigh(C_nbrs)
328                  evals[i] = evi[::-1]
329                  V[i] = vi[:, ::-1]
330  
331          # find regularized weights: this is like normal LLE.
332          # because we've already computed the SVD of each covariance matrix,
333          # it's faster to use this rather than np.linalg.solve
334          reg = 1e-3 * evals.sum(1)
335  
336          tmp = np.dot(V.transpose(0, 2, 1), np.ones(n_neighbors))
337          tmp[:, :nev] /= evals + reg[:, None]
338          tmp[:, nev:] /= reg[:, None]
339  
340          w_reg = np.zeros((N, n_neighbors))
341          for i in range(N):
342              w_reg[i] = np.dot(V[i], tmp[i])
343          w_reg /= w_reg.sum(1)[:, None]
344  
345          # calculate eta: the median of the ratio of small to large eigenvalues
346          # across the points.  This is used to determine s_i, below
347          rho = evals[:, n_components:].sum(1) / evals[:, :n_components].sum(1)
348          eta = np.median(rho)
349  
350          # find s_i, the size of the "almost null space" for each point:
351          # this is the size of the largest set of eigenvalues
352          # such that Sum[v; v in set]/Sum[v; v not in set] < eta
353          s_range = np.zeros(N, dtype=int)
354          evals_cumsum = stable_cumsum(evals, 1)
355          eta_range = evals_cumsum[:, -1:] / evals_cumsum[:, :-1] - 1
356          for i in range(N):
357              s_range[i] = np.searchsorted(eta_range[i, ::-1], eta)
358          s_range += n_neighbors - nev  # number of zero eigenvalues
359  
360          # Now calculate M.
361          # This is the [N x N] matrix whose null space is the desired embedding
362          M = M_container_constructor((N, N), dtype=np.float64)
363  
364          for i in range(N):
365              s_i = s_range[i]
366  
367              # select bottom s_i eigenvectors and calculate alpha
368              Vi = V[i, :, n_neighbors - s_i :]
369              alpha_i = np.linalg.norm(Vi.sum(0)) / np.sqrt(s_i)
370  
371              # compute Householder matrix which satisfies
372              #  Hi*Vi.T*ones(n_neighbors) = alpha_i*ones(s)
373              # using prescription from paper
374              h = np.full(s_i, alpha_i) - np.dot(Vi.T, np.ones(n_neighbors))
375  
376              norm_h = np.linalg.norm(h)
377              if norm_h < modified_tol:
378                  h *= 0
379              else:
380                  h /= norm_h
381  
382              # Householder matrix is
383              #  >> Hi = np.identity(s_i) - 2*np.outer(h,h)
384              # Then the weight matrix is
385              #  >> Wi = np.dot(Vi,Hi) + (1-alpha_i) * w_reg[i,:,None]
386              # We do this much more efficiently:
387              Wi = Vi - 2 * np.outer(np.dot(Vi, h), h) + (1 - alpha_i) * w_reg[i, :, None]
388  
389              # Update M as follows:
390              # >> W_hat = np.zeros( (N,s_i) )
391              # >> W_hat[neighbors[i],:] = Wi
392              # >> W_hat[i] -= 1
393              # >> M += np.dot(W_hat,W_hat.T)
394              # We can do this much more efficiently:
395              nbrs_x, nbrs_y = np.meshgrid(neighbors[i], neighbors[i])
396              M[nbrs_x, nbrs_y] += np.dot(Wi, Wi.T)
397              Wi_sum1 = Wi.sum(1)
398              M[i, neighbors[i]] -= Wi_sum1
399              M[neighbors[i], [i]] -= Wi_sum1
400              M[i, i] += s_i
401  
402      elif method == "ltsa":
403          neighbors = nbrs.kneighbors(
404              X, n_neighbors=n_neighbors + 1, return_distance=False
405          )
406          neighbors = neighbors[:, 1:]
407  
408          M = M_container_constructor((N, N), dtype=np.float64)
409  
410          use_svd = n_neighbors > d_in
411  
412          for i in range(N):
413              Xi = X[neighbors[i]]
414              Xi -= Xi.mean(0)
415  
416              # compute n_components largest eigenvalues of Xi * Xi^T
417              if use_svd:
418                  v = svd(Xi, full_matrices=True)[0]
419              else:
420                  Ci = np.dot(Xi, Xi.T)
421                  v = eigh(Ci)[1][:, ::-1]
422  
423              Gi = np.zeros((n_neighbors, n_components + 1))
424              Gi[:, 1:] = v[:, :n_components]
425              Gi[:, 0] = 1.0 / np.sqrt(n_neighbors)
426  
427              GiGiT = np.dot(Gi, Gi.T)
428  
429              nbrs_x, nbrs_y = np.meshgrid(neighbors[i], neighbors[i])
430              M[nbrs_x, nbrs_y] -= GiGiT
431  
432              M[neighbors[i], neighbors[i]] += np.ones(shape=n_neighbors)
433  
434      if M_sparse:
435          M = M.tocsr()
436  
437      return null_space(
438          M,
439          n_components,
440          k_skip=1,
441          eigen_solver=eigen_solver,
442          tol=tol,
443          max_iter=max_iter,
444          random_state=random_state,
445      )

449      {
450          "X": ["array-like", NearestNeighbors],
451          "n_neighbors": [Interval(Integral, 1, None, closed="left")],

453          "reg": [Interval(Real, 0, None, closed="left")],
454          "eigen_solver": [StrOptions({"auto", "arpack", "dense"})],
455          "tol": [Interval(Real, 0, None, closed="left")],
456          "max_iter": [Interval(Integral, 1, None, closed="left")],
457          "method": [StrOptions({"standard", "hessian", "modified", "ltsa"})],
458          "hessian_tol": [Interval(Real, 0, None, closed="left")],
459          "modified_tol": [Interval(Real, 0, None, closed="left")],
460          "random_state": ["random_state"],
461          "n_jobs": [None, Integral],
462      },

464  )
465  def locally_linear_embedding(
466      X,
467      *,
468      n_neighbors,
469      n_components,
470      reg=1e-3,
471      eigen_solver="auto",
472      tol=1e-6,
473      max_iter=100,
474      method="standard",
475      hessian_tol=1e-4,
476      modified_tol=1e-12,
477      random_state=None,
478      n_jobs=None,
479  ):
480      """Perform a Locally Linear Embedding analysis on the data. 
481   
482      Read more in the :ref:`User Guide <locally_linear_embedding>`. 
483   
484      Parameters 
485      ---------- 
486      X : {array-like, NearestNeighbors} 
487          Sample data, shape = (n_samples, n_features), in the form of a 
488          numpy array or a NearestNeighbors object. 
489   
490      n_neighbors : int 
491          Number of neighbors to consider for each point. 
492   
493      n_components : int 
494          Number of coordinates for the manifold. 
495   
496      reg : float, default=1e-3 
497          Regularization constant, multiplies the trace of the local covariance 
498          matrix of the distances. 
499   
500      eigen_solver : {'auto', 'arpack', 'dense'}, default='auto' 
501          auto : algorithm will attempt to choose the best method for input data 
502   
503          arpack : use arnoldi iteration in shift-invert mode. 
504                      For this method, M may be a dense matrix, sparse matrix, 
505                      or general linear operator. 
506                      Warning: ARPACK can be unstable for some problems.  It is 
507                      best to try several random seeds in order to check results. 
508   
509          dense  : use standard dense matrix operations for the eigenvalue 
510                      decomposition.  For this method, M must be an array 
511                      or matrix type.  This method should be avoided for 
512                      large problems. 
513   
514      tol : float, default=1e-6 
515          Tolerance for 'arpack' method 
516          Not used if eigen_solver=='dense'. 
517   
518      max_iter : int, default=100 
519          Maximum number of iterations for the arpack solver. 
520   
521      method : {'standard', 'hessian', 'modified', 'ltsa'}, default='standard' 
522          standard : use the standard locally linear embedding algorithm. 
523                     see reference [1]_ 
524          hessian  : use the Hessian eigenmap method.  This method requires 
525                     n_neighbors > n_components * (1 + (n_components + 1) / 2. 
526                     see reference [2]_ 
527          modified : use the modified locally linear embedding algorithm. 
528                     see reference [3]_ 
529          ltsa     : use local tangent space alignment algorithm 
530                     see reference [4]_ 
531   
532      hessian_tol : float, default=1e-4 
533          Tolerance for Hessian eigenmapping method. 
534          Only used if method == 'hessian'. 
535   
536      modified_tol : float, default=1e-12 
537          Tolerance for modified LLE method. 
538          Only used if method == 'modified'. 
539   
540      random_state : int, RandomState instance, default=None 
541          Determines the random number generator when ``solver`` == 'arpack'. 
542          Pass an int for reproducible results across multiple function calls. 
543          See :term:`Glossary <random_state>`. 
544   
545      n_jobs : int or None, default=None 
546          The number of parallel jobs to run for neighbors search. 
547          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
548          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
549          for more details. 
550   
551      Returns 
552      ------- 
553      Y : ndarray of shape (n_samples, n_components) 
554          Embedding vectors. 
555   
556      squared_error : float 
557          Reconstruction error for the embedding vectors. Equivalent to 
558          ``norm(Y - W Y, 'fro')**2``, where W are the reconstruction weights. 
559   
560      References 
561      ---------- 
562   
563      .. [1] Roweis, S. & Saul, L. Nonlinear dimensionality reduction 
564          by locally linear embedding.  Science 290:2323 (2000). 
565      .. [2] Donoho, D. & Grimes, C. Hessian eigenmaps: Locally 
566          linear embedding techniques for high-dimensional data. 
567          Proc Natl Acad Sci U S A.  100:5591 (2003). 
568      .. [3] `Zhang, Z. & Wang, J. MLLE: Modified Locally Linear 
569          Embedding Using Multiple Weights. 
570          <https://citeseerx.ist.psu.edu/doc_view/pid/0b060fdbd92cbcc66b383bcaa9ba5e5e624d7ee3>`_ 
571      .. [4] Zhang, Z. & Zha, H. Principal manifolds and nonlinear 
572          dimensionality reduction via tangent space alignment. 
573          Journal of Shanghai Univ.  8:406 (2004) 
574   
575      Examples 
576      -------- 
577      >>> from sklearn.datasets import load_digits 
578      >>> from sklearn.manifold import locally_linear_embedding 
579      >>> X, _ = load_digits(return_X_y=True) 
580      >>> X.shape 
581      (1797, 64) 
582      >>> embedding, _ = locally_linear_embedding(X[:100],n_neighbors=5, n_components=2) 
583      >>> embedding.shape 
584      (100, 2) 
585      """
586      return _locally_linear_embedding(
587          X=X,
588          n_neighbors=n_neighbors,
589          n_components=n_components,
590          reg=reg,
591          eigen_solver=eigen_solver,
592          tol=tol,
593          max_iter=max_iter,
594          method=method,
595          hessian_tol=hessian_tol,
596          modified_tol=modified_tol,
597          random_state=random_state,
598          n_jobs=n_jobs,
599      )

602  class LocallyLinearEmbedding(
603      ClassNamePrefixFeaturesOutMixin,
604      TransformerMixin,
605      _UnstableArchMixin,
606      BaseEstimator,
607  ):
608      """Locally Linear Embedding. 
609   
610      Read more in the :ref:`User Guide <locally_linear_embedding>`.

612      Parameters 
613      ---------- 
614      n_neighbors : int, default=5 
615          Number of neighbors to consider for each point. 
616   
617      n_components : int, default=2 
618          Number of coordinates for the manifold. 
619   
620      reg : float, default=1e-3 
621          Regularization constant, multiplies the trace of the local covariance 
622          matrix of the distances. 
623   
624      eigen_solver : {'auto', 'arpack', 'dense'}, default='auto' 
625          The solver used to compute the eigenvectors. The available options are: 
626   
627          - `'auto'` : algorithm will attempt to choose the best method for input 
628            data. 
629          - `'arpack'` : use arnoldi iteration in shift-invert mode. For this 
630            method, M may be a dense matrix, sparse matrix, or general linear 
631            operator. 
632          - `'dense'`  : use standard dense matrix operations for the eigenvalue 
633            decomposition. For this method, M must be an array or matrix type. 
634            This method should be avoided for large problems. 
635   
636          .. warning:: 
637             ARPACK can be unstable for some problems.  It is best to try several 
638             random seeds in order to check results. 
639   
640      tol : float, default=1e-6 
641          Tolerance for 'arpack' method 
642          Not used if eigen_solver=='dense'. 
643   
644      max_iter : int, default=100 
645          Maximum number of iterations for the arpack solver. 
646          Not used if eigen_solver=='dense'. 
647   
648      method : {'standard', 'hessian', 'modified', 'ltsa'}, default='standard' 
649          - `standard`: use the standard locally linear embedding algorithm. see 
650            reference [1]_ 
651          - `hessian`: use the Hessian eigenmap method. This method requires 
652            ``n_neighbors > n_components * (1 + (n_components + 1) / 2``. see 
653            reference [2]_ 
654          - `modified`: use the modified locally linear embedding algorithm. 
655            see reference [3]_ 
656          - `ltsa`: use local tangent space alignment algorithm. see 
657            reference [4]_ 
658   
659      hessian_tol : float, default=1e-4 
660          Tolerance for Hessian eigenmapping method. 
661          Only used if ``method == 'hessian'``. 
662   
663      modified_tol : float, default=1e-12 
664          Tolerance for modified LLE method. 
665          Only used if ``method == 'modified'``. 
666   
667      neighbors_algorithm : {'auto', 'brute', 'kd_tree', 'ball_tree'}, \ 
668                            default='auto' 
669          Algorithm to use for nearest neighbors search, passed to 
670          :class:`~sklearn.neighbors.NearestNeighbors` instance. 
671   
672      random_state : int, RandomState instance, default=None 
673          Determines the random number generator when 
674          ``eigen_solver`` == 'arpack'. Pass an int for reproducible results 
675          across multiple function calls. See :term:`Glossary <random_state>`. 
676   
677      n_jobs : int or None, default=None 
678          The number of parallel jobs to run. 
679          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
680          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
681          for more details.

850      def transform(self, X):
851          """ 
852          Transform new points into embedding space. 
853   
854          Parameters 
855          ---------- 
856          X : array-like of shape (n_samples, n_features) 
857              Training set. 
858   
859          Returns 
860          ------- 
861          X_new : ndarray of shape (n_samples, n_components) 
862              Returns the instance itself. 
863   
864          Notes 
865          ----- 
866          Because of scaling performed by this method, it is discouraged to use 
867          it together with methods that are not scale-invariant (like SVMs). 
868          """
869          check_is_fitted(self)
870  
871          X = validate_data(self, X, reset=False)
872          ind = self.nbrs_.kneighbors(
873              X, n_neighbors=self.n_neighbors, return_distance=False
874          )
875          weights = barycenter_weights(X, self.nbrs_._fit_X, ind, reg=self.reg)
876          X_new = np.empty((X.shape[0], self.n_components))
877          for i in range(X.shape[0]):
878              X_new[i] = np.dot(self.embedding_[ind[i]].T, weights[i])
879          return X_new
