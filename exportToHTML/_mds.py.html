23   def _smacof_single(
24       dissimilarities,
25       metric=True,
26       n_components=2,
27       init=None,
28       max_iter=300,
29       verbose=0,
30       eps=1e-3,
31       random_state=None,
32       normalized_stress=False,
33   ):
34       """Computes multidimensional scaling using SMACOF algorithm. 
35    
36       Parameters 
37       ---------- 
38       dissimilarities : ndarray of shape (n_samples, n_samples) 
39           Pairwise dissimilarities between the points. Must be symmetric. 
40    
41       metric : bool, default=True 
42           Compute metric or nonmetric SMACOF algorithm. 
43           When ``False`` (i.e. non-metric MDS), dissimilarities with 0 are considered as 
44           missing values. 
45    
46       n_components : int, default=2 
47           Number of dimensions in which to immerse the dissimilarities. If an 
48           ``init`` array is provided, this option is overridden and the shape of 
49           ``init`` is used to determine the dimensionality of the embedding 
50           space. 
51    
52       init : ndarray of shape (n_samples, n_components), default=None 
53           Starting configuration of the embedding to initialize the algorithm. By 
54           default, the algorithm is initialized with a randomly chosen array. 
55    
56       max_iter : int, default=300 
57           Maximum number of iterations of the SMACOF algorithm for a single run. 
58    
59       verbose : int, default=0 
60           Level of verbosity. 
61    
62       eps : float, default=1e-3 
63           Relative tolerance with respect to stress at which to declare 
64           convergence. The value of `eps` should be tuned separately depending 
65           on whether or not `normalized_stress` is being used. 
66    
67       random_state : int, RandomState instance or None, default=None 
68           Determines the random number generator used to initialize the centers. 
69           Pass an int for reproducible results across multiple function calls. 
70           See :term:`Glossary <random_state>`. 
71    
72       normalized_stress : bool, default=False 
73           Whether use and return normed stress value (Stress-1) instead of raw 
74           stress calculated by default. Only supported in non-metric MDS. The 
75           caller must ensure that if `normalized_stress=True` then `metric=False` 
76    
77           .. versionadded:: 1.2 
78    
79       Returns 
80       ------- 
81       X : ndarray of shape (n_samples, n_components) 
82           Coordinates of the points in a ``n_components``-space. 
83    
84       stress : float 
85           The final value of the stress (sum of squared distance of the 
86           disparities and the distances for all constrained points). 
87           If `normalized_stress=True`, and `metric=False` returns Stress-1. 
88           A value of 0 indicates "perfect" fit, 0.025 excellent, 0.05 good, 
89           0.1 fair, and 0.2 poor [1]_. 
90    
91       n_iter : int 
92           The number of iterations corresponding to the best stress. 
93    
94       References 
95       ---------- 
96       .. [1] "Nonmetric multidimensional scaling: a numerical method" Kruskal, J. 
97              Psychometrika, 29 (1964) 
98    
99       .. [2] "Multidimensional scaling by optimizing goodness of fit to a nonmetric 
100             hypothesis" Kruskal, J. Psychometrika, 29, (1964) 
101   
102      .. [3] "Modern Multidimensional Scaling - Theory and Applications" Borg, I.; 
103             Groenen P. Springer Series in Statistics (1997) 
104      """
105      dissimilarities = check_symmetric(dissimilarities, raise_exception=True)
106  
107      n_samples = dissimilarities.shape[0]
108      random_state = check_random_state(random_state)
109  
110      sim_flat = ((1 - np.tri(n_samples)) * dissimilarities).ravel()
111      sim_flat_w = sim_flat[sim_flat != 0]
112      if init is None:
113          # Randomly choose initial configuration
114          X = random_state.uniform(size=n_samples * n_components)
115          X = X.reshape((n_samples, n_components))
116      else:
117          # overrides the parameter p
118          n_components = init.shape[1]
119          if n_samples != init.shape[0]:
120              raise ValueError(
121                  "init matrix should be of shape (%d, %d)" % (n_samples, n_components)
122              )
123          X = init
124  
125      old_stress = None
126      ir = IsotonicRegression()
127      for it in range(max_iter):
128          # Compute distance and monotonic regression
129          dis = euclidean_distances(X)
130  
131          if metric:
132              disparities = dissimilarities
133          else:
134              dis_flat = dis.ravel()
135              # dissimilarities with 0 are considered as missing values
136              dis_flat_w = dis_flat[sim_flat != 0]
137  
138              # Compute the disparities using a monotonic regression
139              disparities_flat = ir.fit_transform(sim_flat_w, dis_flat_w)
140              disparities = dis_flat.copy()
141              disparities[sim_flat != 0] = disparities_flat
142              disparities = disparities.reshape((n_samples, n_samples))
143              disparities *= np.sqrt(
144                  (n_samples * (n_samples - 1) / 2) / (disparities**2).sum()
145              )
146  
147          # Compute stress
148          stress = ((dis.ravel() - disparities.ravel()) ** 2).sum() / 2
149          if normalized_stress:
150              stress = np.sqrt(stress / ((disparities.ravel() ** 2).sum() / 2))
151          # Update X using the Guttman transform
152          dis[dis == 0] = 1e-5
153          ratio = disparities / dis
154          B = -ratio
155          B[np.arange(len(B)), np.arange(len(B))] += ratio.sum(axis=1)
156          X = 1.0 / n_samples * np.dot(B, X)
157  
158          dis = np.sqrt((X**2).sum(axis=1)).sum()
159          if verbose >= 2:
160              print("it: %d, stress %s" % (it, stress))
161          if old_stress is not None:
162              if (old_stress - stress / dis) < eps:
163                  if verbose:
164                      print("breaking at iteration %d with stress %s" % (it, stress))
165                  break
166          old_stress = stress / dis
167  
168      return X, stress, it + 1

177          "n_init": [Interval(Integral, 1, None, closed="left")],
178          "n_jobs": [Integral, None],
179          "max_iter": [Interval(Integral, 1, None, closed="left")],
180          "verbose": ["verbose"],
181          "eps": [Interval(Real, 0, None, closed="left")],
182          "random_state": ["random_state"],

184          "normalized_stress": ["boolean", StrOptions({"auto"})],

188  def smacof(
189      dissimilarities,
190      *,
191      metric=True,
192      n_components=2,
193      init=None,
194      n_init=8,
195      n_jobs=None,
196      max_iter=300,
197      verbose=0,
198      eps=1e-3,
199      random_state=None,
200      return_n_iter=False,
201      normalized_stress="auto",
202  ):
203      """Compute multidimensional scaling using the SMACOF algorithm. 
204   
205      The SMACOF (Scaling by MAjorizing a COmplicated Function) algorithm is a 
206      multidimensional scaling algorithm which minimizes an objective function 
207      (the *stress*) using a majorization technique. Stress majorization, also 
208      known as the Guttman Transform, guarantees a monotone convergence of 
209      stress, and is more powerful than traditional techniques such as gradient 
210      descent. 
211   
212      The SMACOF algorithm for metric MDS can be summarized by the following 
213      steps: 
214   
215      1. Set an initial start configuration, randomly or not. 
216      2. Compute the stress 
217      3. Compute the Guttman Transform 
218      4. Iterate 2 and 3 until convergence. 
219   
220      The nonmetric algorithm adds a monotonic regression step before computing 
221      the stress. 
222   
223      Parameters 
224      ---------- 
225      dissimilarities : array-like of shape (n_samples, n_samples) 
226          Pairwise dissimilarities between the points. Must be symmetric. 
227   
228      metric : bool, default=True 
229          Compute metric or nonmetric SMACOF algorithm. 
230          When ``False`` (i.e. non-metric MDS), dissimilarities with 0 are considered as 
231          missing values. 
232   
233      n_components : int, default=2 
234          Number of dimensions in which to immerse the dissimilarities. If an 
235          ``init`` array is provided, this option is overridden and the shape of 
236          ``init`` is used to determine the dimensionality of the embedding 
237          space. 
238   
239      init : array-like of shape (n_samples, n_components), default=None 
240          Starting configuration of the embedding to initialize the algorithm. By 
241          default, the algorithm is initialized with a randomly chosen array. 
242   
243      n_init : int, default=8 
244          Number of times the SMACOF algorithm will be run with different 
245          initializations. The final results will be the best output of the runs, 
246          determined by the run with the smallest final stress. If ``init`` is 
247          provided, this option is overridden and a single run is performed. 
248   
249      n_jobs : int, default=None 
250          The number of jobs to use for the computation. If multiple 
251          initializations are used (``n_init``), each run of the algorithm is 
252          computed in parallel. 
253   
254          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
255          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
256          for more details. 
257   
258      max_iter : int, default=300 
259          Maximum number of iterations of the SMACOF algorithm for a single run. 
260   
261      verbose : int, default=0 
262          Level of verbosity. 
263   
264      eps : float, default=1e-3 
265          Relative tolerance with respect to stress at which to declare 
266          convergence. The value of `eps` should be tuned separately depending 
267          on whether or not `normalized_stress` is being used. 
268   
269      random_state : int, RandomState instance or None, default=None 
270          Determines the random number generator used to initialize the centers. 
271          Pass an int for reproducible results across multiple function calls. 
272          See :term:`Glossary <random_state>`. 
273   
274      return_n_iter : bool, default=False 
275          Whether or not to return the number of iterations. 
276   
277      normalized_stress : bool or "auto" default="auto" 
278          Whether use and return normed stress value (Stress-1) instead of raw 
279          stress calculated by default. Only supported in non-metric MDS. 
280   
281          .. versionadded:: 1.2 
282   
283          .. versionchanged:: 1.4 
284             The default value changed from `False` to `"auto"` in version 1.4. 
285   
286      Returns 
287      ------- 
288      X : ndarray of shape (n_samples, n_components) 
289          Coordinates of the points in a ``n_components``-space. 
290   
291      stress : float 
292          The final value of the stress (sum of squared distance of the 
293          disparities and the distances for all constrained points). 
294          If `normalized_stress=True`, and `metric=False` returns Stress-1. 
295          A value of 0 indicates "perfect" fit, 0.025 excellent, 0.05 good, 
296          0.1 fair, and 0.2 poor [1]_. 
297   
298      n_iter : int 
299          The number of iterations corresponding to the best stress. Returned 
300          only if ``return_n_iter`` is set to ``True``. 
301   
302      References 
303      ---------- 
304      .. [1] "Nonmetric multidimensional scaling: a numerical method" Kruskal, J. 
305             Psychometrika, 29 (1964) 
306   
307      .. [2] "Multidimensional scaling by optimizing goodness of fit to a nonmetric 
308             hypothesis" Kruskal, J. Psychometrika, 29, (1964) 
309   
310      .. [3] "Modern Multidimensional Scaling - Theory and Applications" Borg, I.; 
311             Groenen P. Springer Series in Statistics (1997) 
312   
313      Examples 
314      -------- 
315      >>> import numpy as np 
316      >>> from sklearn.manifold import smacof 
317      >>> from sklearn.metrics import euclidean_distances 
318      >>> X = np.array([[0, 1, 2], [1, 0, 3],[2, 3, 0]]) 
319      >>> dissimilarities = euclidean_distances(X) 
320      >>> mds_result, stress = smacof(dissimilarities, n_components=2, random_state=42) 
321      >>> mds_result 
322      array([[ 0.05... -1.07... ], 
323             [ 1.74..., -0.75...], 
324             [-1.79...,  1.83...]]) 
325      >>> stress 
326      np.float64(0.0012...) 
327      """
328  
329      dissimilarities = check_array(dissimilarities)
330      random_state = check_random_state(random_state)
331  
332      if normalized_stress == "auto":
333          normalized_stress = not metric
334  
335      if normalized_stress and metric:
336          raise ValueError(
337              "Normalized stress is not supported for metric MDS. Either set"
338              " `normalized_stress=False` or use `metric=False`."
339          )
340      if hasattr(init, "__array__"):
341          init = np.asarray(init).copy()
342          if not n_init == 1:
343              warnings.warn(
344                  "Explicit initial positions passed: "
345                  "performing only one init of the MDS instead of %d" % n_init
346              )
347              n_init = 1
348  
349      best_pos, best_stress = None, None
350  
351      if effective_n_jobs(n_jobs) == 1:
352          for it in range(n_init):
353              pos, stress, n_iter_ = _smacof_single(
354                  dissimilarities,
355                  metric=metric,
356                  n_components=n_components,
357                  init=init,
358                  max_iter=max_iter,
359                  verbose=verbose,
360                  eps=eps,
361                  random_state=random_state,
362                  normalized_stress=normalized_stress,
363              )
364              if best_stress is None or stress < best_stress:
365                  best_stress = stress
366                  best_pos = pos.copy()
367                  best_iter = n_iter_
368      else:
369          seeds = random_state.randint(np.iinfo(np.int32).max, size=n_init)
370          results = Parallel(n_jobs=n_jobs, verbose=max(verbose - 1, 0))(
371              delayed(_smacof_single)(
372                  dissimilarities,
373                  metric=metric,
374                  n_components=n_components,
375                  init=init,
376                  max_iter=max_iter,
377                  verbose=verbose,
378                  eps=eps,
379                  random_state=seed,
380                  normalized_stress=normalized_stress,
381              )
382              for seed in seeds
383          )
384          positions, stress, n_iters = zip(*results)
385          best = np.argmin(stress)
386          best_stress = stress[best]
387          best_pos = positions[best]
388          best_iter = n_iters[best]
389  
390      if return_n_iter:
391          return best_pos, best_stress, best_iter
392      else:
393          return best_pos, best_stress

396  class MDS(BaseEstimator):
397      """Multidimensional scaling. 
398   
399      Read more in the :ref:`User Guide <multidimensional_scaling>`. 
400   
401      Parameters 
402      ---------- 
403      n_components : int, default=2 
404          Number of dimensions in which to immerse the dissimilarities. 
405   
406      metric : bool, default=True 
407          If ``True``, perform metric MDS; otherwise, perform nonmetric MDS. 
408          When ``False`` (i.e. non-metric MDS), dissimilarities with 0 are considered as 
409          missing values. 
410   
411      n_init : int, default=4 
412          Number of times the SMACOF algorithm will be run with different 
413          initializations. The final results will be the best output of the runs, 
414          determined by the run with the smallest final stress. 
415   
416      max_iter : int, default=300 
417          Maximum number of iterations of the SMACOF algorithm for a single run. 
418   
419      verbose : int, default=0 
420          Level of verbosity. 
421   
422      eps : float, default=1e-3 
423          Relative tolerance with respect to stress at which to declare 
424          convergence. The value of `eps` should be tuned separately depending 
425          on whether or not `normalized_stress` is being used. 
426   
427      n_jobs : int, default=None 
428          The number of jobs to use for the computation. If multiple 
429          initializations are used (``n_init``), each run of the algorithm is 
430          computed in parallel. 
431   
432          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
433          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
434          for more details. 
435   
436      random_state : int, RandomState instance or None, default=None 
437          Determines the random number generator used to initialize the centers. 
438          Pass an int for reproducible results across multiple function calls. 
439          See :term:`Glossary <random_state>`. 
440   
441      dissimilarity : {'euclidean', 'precomputed'}, default='euclidean' 
442          Dissimilarity measure to use: 
443   
444          - 'euclidean': 
445              Pairwise Euclidean distances between points in the dataset. 
446   
447          - 'precomputed': 
448              Pre-computed dissimilarities are passed directly to ``fit`` and 
449              ``fit_transform``. 
450   
451      normalized_stress : bool or "auto" default="auto" 
452          Whether use and return normed stress value (Stress-1) instead of raw 
453          stress calculated by default. Only supported in non-metric MDS. 
454   
455          .. versionadded:: 1.2 
456   
457          .. versionchanged:: 1.4 
458             The default value changed from `False` to `"auto"` in version 1.4. 
459   
460      Attributes 
461      ---------- 
462      embedding_ : ndarray of shape (n_samples, n_components) 
463          Stores the position of the dataset in the embedding space. 
464   
465      stress_ : float 
466          The final value of the stress (sum of squared distance of the 
467          disparities and the distances for all constrained points). 
468          If `normalized_stress=True`, and `metric=False` returns Stress-1. 
469          A value of 0 indicates "perfect" fit, 0.025 excellent, 0.05 good, 
470          0.1 fair, and 0.2 poor [1]_. 
471   
472      dissimilarity_matrix_ : ndarray of shape (n_samples, n_samples) 
473          Pairwise dissimilarities between the points. Symmetric matrix that: 
474   
475          - either uses a custom dissimilarity matrix by setting `dissimilarity` 
476            to 'precomputed'; 
477          - or constructs a dissimilarity matrix from data using 
478            Euclidean distances. 
479   
480      n_features_in_ : int 
481          Number of features seen during :term:`fit`. 
482   
483          .. versionadded:: 0.24 
484   
485      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
486          Names of features seen during :term:`fit`. Defined only when `X` 
487          has feature names that are all strings. 
488   
489          .. versionadded:: 1.0 
490   
491      n_iter_ : int 
492          The number of iterations corresponding to the best stress. 
493   
494      See Also 
495      -------- 
496      sklearn.decomposition.PCA : Principal component analysis that is a linear 
497          dimensionality reduction method. 
498      sklearn.decomposition.KernelPCA : Non-linear dimensionality reduction using 
499          kernels and PCA. 
500      TSNE : T-distributed Stochastic Neighbor Embedding. 
501      Isomap : Manifold learning based on Isometric Mapping. 
502      LocallyLinearEmbedding : Manifold learning using Locally Linear Embedding. 
503      SpectralEmbedding : Spectral embedding for non-linear dimensionality. 
504   
505      References 
506      ---------- 
507      .. [1] "Nonmetric multidimensional scaling: a numerical method" Kruskal, J. 
508         Psychometrika, 29 (1964) 
509   
510      .. [2] "Multidimensional scaling by optimizing goodness of fit to a nonmetric 
511         hypothesis" Kruskal, J. Psychometrika, 29, (1964) 
512   
513      .. [3] "Modern Multidimensional Scaling - Theory and Applications" Borg, I.; 
514         Groenen P. Springer Series in Statistics (1997) 
515   
516      Examples 
517      -------- 
518      >>> from sklearn.datasets import load_digits 
519      >>> from sklearn.manifold import MDS 
520      >>> X, _ = load_digits(return_X_y=True) 
521      >>> X.shape 
522      (1797, 64) 
523      >>> embedding = MDS(n_components=2, normalized_stress='auto') 
524      >>> X_transformed = embedding.fit_transform(X[:100]) 
525      >>> X_transformed.shape 
526      (100, 2) 
527   
528      For a more detailed example of usage, see 
529      :ref:`sphx_glr_auto_examples_manifold_plot_mds.py`. 
530   
531      For a comparison of manifold learning techniques, see 
532      :ref:`sphx_glr_auto_examples_manifold_plot_compare_methods.py`. 
533      """

578      def fit(self, X, y=None, init=None):
579          """ 
580          Compute the position of the points in the embedding space. 
581   
582          Parameters 
583          ---------- 
584          X : array-like of shape (n_samples, n_features) or \ 
585                  (n_samples, n_samples) 
586              Input data. If ``dissimilarity=='precomputed'``, the input should 
587              be the dissimilarity matrix. 
588   
589          y : Ignored 
590              Not used, present for API consistency by convention. 
591   
592          init : ndarray of shape (n_samples, n_components), default=None 
593              Starting configuration of the embedding to initialize the SMACOF 
594              algorithm. By default, the algorithm is initialized with a randomly 
595              chosen array. 
596   
597          Returns 
598          ------- 
599          self : object 
600              Fitted estimator. 
601          """
602          self.fit_transform(X, init=init)
603          return self
604  
605      @_fit_context(prefer_skip_nested_validation=True)
606      def fit_transform(self, X, y=None, init=None):
607          """ 
608          Fit the data from `X`, and returns the embedded coordinates. 
609   
610          Parameters 
611          ---------- 
612          X : array-like of shape (n_samples, n_features) or \ 
613                  (n_samples, n_samples) 
614              Input data. If ``dissimilarity=='precomputed'``, the input should 
615              be the dissimilarity matrix. 
616   
617          y : Ignored 
618              Not used, present for API consistency by convention. 
619   
620          init : ndarray of shape (n_samples, n_components), default=None 
621              Starting configuration of the embedding to initialize the SMACOF 
622              algorithm. By default, the algorithm is initialized with a randomly 
623              chosen array. 
624   
625          Returns 
626          ------- 
627          X_new : ndarray of shape (n_samples, n_components) 
628              X transformed in the new space. 
629          """
630          X = validate_data(self, X)
631          if X.shape[0] == X.shape[1] and self.dissimilarity != "precomputed":
632              warnings.warn(
633                  "The MDS API has changed. ``fit`` now constructs an"
634                  " dissimilarity matrix from data. To use a custom "
635                  "dissimilarity matrix, set "
636                  "``dissimilarity='precomputed'``."
637              )
638  
639          if self.dissimilarity == "precomputed":
640              self.dissimilarity_matrix_ = X
641          elif self.dissimilarity == "euclidean":
642              self.dissimilarity_matrix_ = euclidean_distances(X)
643  
644          self.embedding_, self.stress_, self.n_iter_ = smacof(
645              self.dissimilarity_matrix_,
646              metric=self.metric,
647              n_components=self.n_components,
648              init=init,
649              n_init=self.n_init,
650              n_jobs=self.n_jobs,
651              max_iter=self.max_iter,
652              verbose=self.verbose,
653              eps=self.eps,
654              random_state=self.random_state,
655              return_n_iter=True,
656              normalized_stress=self.normalized_stress,
657          )
658  
659          return self.embedding_
660
