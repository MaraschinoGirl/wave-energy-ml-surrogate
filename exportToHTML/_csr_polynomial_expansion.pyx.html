8    # We use the following verbatim block to determine whether the current
9    # platform's compiler supports 128-bit integer values intrinsically.
10   # This should work for GCC and CLANG on 64-bit architectures, but doesn't for
11   # MSVC on any architecture. We prefer to use 128-bit integers when possible
12   # because the intermediate calculations have a non-trivial risk of overflow. It
13   # is, however, very unlikely to come up on an average use case, hence 64-bit
14   # integers (i.e. `long long`) are "good enough" for most common cases. There is
15   # not much we can do to efficiently mitigate  the overflow risk on the Windows
16   # platform at this time. Consider this a "best effort" design decision that
17   # could be revisited later in case someone comes up with a safer option that
18   # does not hurt the performance of the common cases.
19   # See `test_sizeof_LARGEST_INT_t()`for more information on exact type expectations.
20   cdef extern from *:
21       """ 
22       #ifdef __SIZEOF_INT128__ 
23           typedef __int128 LARGEST_INT_t; 
24       #elif (__clang__ || __EMSCRIPTEN__) && !__i386__ 
25           typedef _BitInt(128) LARGEST_INT_t; 
26       #else 
27           typedef long long LARGEST_INT_t; 
28       #endif 
29       """
30       ctypedef long long LARGEST_INT_t

55   cdef inline int64_t _deg2_column(
56       LARGEST_INT_t n_features,
57       LARGEST_INT_t i,
58       LARGEST_INT_t j,
59       FLAG_t interaction_only
60   ) nogil:
61       """Compute the index of the column for a degree 2 expansion 
62    
63       n_features is the dimensionality of the input data, i and j are the indices 
64       for the columns involved in the expansion. 
65       """
66       if interaction_only:
67           return n_features * i - i * (i + 3) / 2 - 1 + j
68       else:
69           return n_features * i - i* (i + 1) / 2 + j
70   
71   
72   cdef inline int64_t _deg3_column(
73       LARGEST_INT_t n_features,
74       LARGEST_INT_t i,
75       LARGEST_INT_t j,
76       LARGEST_INT_t k,
77       FLAG_t interaction_only
78   ) nogil:
79       """Compute the index of the column for a degree 3 expansion 
80    
81       n_features is the dimensionality of the input data, i, j and k are the indices 
82       for the columns involved in the expansion. 
83       """
84       if interaction_only:
85           return (
86               (
87                   (3 * n_features) * (n_features * i - i**2)
88                   + i * (i**2 + 11) - (3 * j) * (j + 3)
89               ) / 6 + i**2 + n_features * (j - 1 - 2 * i) + k
90           )
91       else:
92           return (
93               (
94                   (3 * n_features) * (n_features * i - i**2)
95                   + i ** 3 - i - (3 * j) * (j + 1)
96               ) / 6 + n_features * j + k
97           )

100  def py_calc_expanded_nnz_deg2(n, interaction_only):
101      return n * (n + 1) // 2 - interaction_only * n
102  
103  
104  def py_calc_expanded_nnz_deg3(n, interaction_only):
105      return n * (n**2 + 3 * n + 2) // 6 - interaction_only * n**2

108  cpdef int64_t _calc_expanded_nnz(
109      LARGEST_INT_t n,
110      FLAG_t interaction_only,
111      LARGEST_INT_t degree
112  ):
113      """ 
114      Calculates the number of non-zero interaction terms generated by the 
115      non-zero elements of a single row. 
116      """
117      # This is the maximum value before the intermediate computation
118      # d**2 + d overflows
119      # Solution to d**2 + d = maxint64
120      # SymPy: solve(x**2 + x - int64_max, x)
121      cdef int64_t MAX_SAFE_INDEX_CALC_DEG2 = 3037000499
122  
123      # This is the maximum value before the intermediate computation
124      # d**3 + 3 * d**2 + 2*d overflows
125      # Solution to d**3 + 3 * d**2 + 2*d = maxint64
126      # SymPy: solve(x * (x**2 + 3 * x + 2) - int64_max, x)
127      cdef int64_t MAX_SAFE_INDEX_CALC_DEG3 = 2097151
128  
129      if degree == 2:
130          # Only need to check when not using 128-bit integers
131          if sizeof(LARGEST_INT_t) < 16 and n <= MAX_SAFE_INDEX_CALC_DEG2:
132              return n * (n + 1) / 2 - interaction_only * n
133          return <int64_t> py_calc_expanded_nnz_deg2(n, interaction_only)
134      else:
135          # Only need to check when not using 128-bit integers
136          if sizeof(LARGEST_INT_t) < 16 and n <= MAX_SAFE_INDEX_CALC_DEG3:
137              return n * (n**2 + 3 * n + 2) / 6 - interaction_only * n**2
138          return <int64_t> py_calc_expanded_nnz_deg3(n, interaction_only)

140  cpdef int64_t _calc_total_nnz(
141      INDEX_A_t[:] indptr,
142      FLAG_t interaction_only,
143      int64_t degree,
144  ):
145      """ 
146      Calculates the number of non-zero interaction terms generated by the 
147      non-zero elements across all rows for a single degree. 
148      """
149      cdef int64_t total_nnz=0
150      cdef intp_t row_idx
151      for row_idx in range(len(indptr) - 1):
152          total_nnz += _calc_expanded_nnz(
153              indptr[row_idx + 1] - indptr[row_idx],
154              interaction_only,
155              degree
156          )
157      return total_nnz

160  cpdef void _csr_polynomial_expansion(
161      const DATA_t[:] data,           # IN READ-ONLY
162      const INDEX_A_t[:] indices,     # IN READ-ONLY
163      const INDEX_A_t[:] indptr,      # IN READ-ONLY
164      INDEX_A_t n_features,
165      DATA_t[:] result_data,          # OUT
166      INDEX_B_t[:] result_indices,    # OUT
167      INDEX_B_t[:] result_indptr,     # OUT
168      FLAG_t interaction_only,
169      FLAG_t degree
170  ):
171      """ 
172      Perform a second or third degree polynomial or interaction expansion on a 
173      compressed sparse row (CSR) matrix. The method used only takes products of 
174      non-zero features. For a matrix with density :math:`d`, this results in a 
175      speedup on the order of :math:`(1/d)^k` where :math:`k` is the degree of 
176      the expansion, assuming all rows are of similar density. 
177   
178      Parameters 
179      ---------- 
180      data : memory view on nd-array 
181          The "data" attribute of the input CSR matrix. 
182   
183      indices : memory view on nd-array 
184          The "indices" attribute of the input CSR matrix. 
185   
186      indptr : memory view on nd-array 
187          The "indptr" attribute of the input CSR matrix. 
188   
189      n_features : int 
190          The dimensionality of the input CSR matrix. 
191   
192      result_data : nd-array 
193          The output CSR matrix's "data" attribute. 
194          It is modified by this routine. 
195   
196      result_indices : nd-array 
197          The output CSR matrix's "indices" attribute. 
198          It is modified by this routine. 
199   
200      result_indptr : nd-array 
201          The output CSR matrix's "indptr" attribute. 
202          It is modified by this routine. 
203   
204      interaction_only : int 
205          0 for a polynomial expansion, 1 for an interaction expansion. 
206   
207      degree : int 
208          The degree of the expansion. This must be either 2 or 3. 
209   
210      References 
211      ---------- 
212      "Leveraging Sparsity to Speed Up Polynomial Feature Expansions of CSR 
213      Matrices Using K-Simplex Numbers" by Andrew Nystrom and John Hughes. 
214      """
215  
216      # Make the arrays that will form the CSR matrix of the expansion.
217      cdef INDEX_A_t row_i, row_starts, row_ends, i, j, k, i_ptr, j_ptr, k_ptr
218      cdef INDEX_B_t expanded_index=0, num_cols_in_row, col
219      with nogil:
220          result_indptr[0] = indptr[0]
221          for row_i in range(indptr.shape[0]-1):
222              row_starts = indptr[row_i]
223              row_ends = indptr[row_i + 1]
224              num_cols_in_row = 0
225              for i_ptr in range(row_starts, row_ends):
226                  i = indices[i_ptr]
227                  for j_ptr in range(i_ptr + interaction_only, row_ends):
228                      j = indices[j_ptr]
229                      if degree == 2:
230                          col = <INDEX_B_t> _deg2_column(
231                              n_features,
232                              i, j,
233                              interaction_only
234                          )
235                          result_indices[expanded_index] = col
236                          result_data[expanded_index] = (
237                              data[i_ptr] * data[j_ptr]
238                          )
239                          expanded_index += 1
240                          num_cols_in_row += 1
241                      else:
242                          # degree == 3
243                          for k_ptr in range(j_ptr + interaction_only, row_ends):
244                              k = indices[k_ptr]
245                              col = <INDEX_B_t> _deg3_column(
246                                  n_features,
247                                  i, j, k,
248                                  interaction_only
249                              )
250                              result_indices[expanded_index] = col
251                              result_data[expanded_index] = (
252                                  data[i_ptr] * data[j_ptr] * data[k_ptr]
253                              )
254                              expanded_index += 1
255                              num_cols_in_row += 1
256  
257              result_indptr[row_i+1] = result_indptr[row_i] + num_cols_in_row
258      return
