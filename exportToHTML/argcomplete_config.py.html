27   def get_argcomplete_cwords() -> t.Optional[t.List[str]]:
28       """Get current words prior to completion point 
29    
30       This is normally done in the `argcomplete.CompletionFinder` constructor, 
31       but is exposed here to allow `traitlets` to follow dynamic code-paths such 
32       as determining whether to evaluate a subcommand. 
33       """
34       if "_ARGCOMPLETE" not in os.environ:
35           return None
36   
37       comp_line = os.environ["COMP_LINE"]
38       comp_point = int(os.environ["COMP_POINT"])
39       # argcomplete.debug("splitting COMP_LINE for:", comp_line, comp_point)
40       comp_words: t.List[str]
41       try:
42           (
43               cword_prequote,
44               cword_prefix,
45               cword_suffix,
46               comp_words,
47               last_wordbreak_pos,
48           ) = argcomplete.split_line(comp_line, comp_point)  # type:ignore[attr-defined,no-untyped-call]
49       except ModuleNotFoundError:
50           return None
51   
52       # _ARGCOMPLETE is set by the shell script to tell us where comp_words
53       # should start, based on what we're completing.
54       # 1: <script> [args]
55       # 2: python <script> [args]
56       # 3: python -m <module> [args]
57       start = int(os.environ["_ARGCOMPLETE"]) - 1
58       comp_words = comp_words[start:]
59   
60       # argcomplete.debug("prequote=", cword_prequote, "prefix=", cword_prefix, "suffix=", cword_suffix, "words=", comp_words, "last=", last_wordbreak_pos)
61       return comp_words  # noqa: RET504

64   def increment_argcomplete_index() -> None:
65       """Assumes ``$_ARGCOMPLETE`` is set and `argcomplete` is importable 
66    
67       Increment the index pointed to by ``$_ARGCOMPLETE``, which is used to 
68       determine which word `argcomplete` should start evaluating the command-line. 
69       This may be useful to "inform" `argcomplete` that we have already evaluated 
70       the first word as a subcommand. 
71       """
72       try:
73           os.environ["_ARGCOMPLETE"] = str(int(os.environ["_ARGCOMPLETE"]) + 1)
74       except Exception:
75           try:
76               argcomplete.debug("Unable to increment $_ARGCOMPLETE", os.environ["_ARGCOMPLETE"])  # type:ignore[attr-defined,no-untyped-call]
77           except (KeyError, ModuleNotFoundError):
78               pass

81   class ExtendedCompletionFinder(CompletionFinder):
82       """An extension of CompletionFinder which dynamically completes class-trait based options 
83    
84       This finder adds a few functionalities: 
85    
86       1. When completing options, it will add ``--Class.`` to the list of completions, for each 
87       class in `Application.classes` that could complete the current option. 
88       2. If it detects that we are currently trying to complete an option related to ``--Class.``, 
89       it will add the corresponding config traits of Class to the `ArgumentParser` instance, 
90       so that the traits' completers can be used. 
91       3. If there are any subcommands, they are added as completions for the first word 
92    
93       Note that we are avoiding adding all config traits of all classes to the `ArgumentParser`, 
94       which would be easier but would add more runtime overhead and would also make completions 
95       appear more spammy. 
96    
97       These changes do require using the internals of `argcomplete.CompletionFinder`. 
98       """
99   
100      _parser: argparse.ArgumentParser
101      config_classes: t.List[t.Any] = []  # Configurables
102      subcommands: t.List[str] = []
103  
104      def match_class_completions(self, cword_prefix: str) -> t.List[t.Tuple[t.Any, str]]:
105          """Match the word to be completed against our Configurable classes 
106   
107          Check if cword_prefix could potentially match against --{class}. for any class 
108          in Application.classes. 
109          """
110          class_completions = [(cls, f"--{cls.__name__}.") for cls in self.config_classes]
111          matched_completions = class_completions
112          if "." in cword_prefix:
113              cword_prefix = cword_prefix[: cword_prefix.index(".") + 1]
114              matched_completions = [(cls, c) for (cls, c) in class_completions if c == cword_prefix]
115          elif len(cword_prefix) > 0:
116              matched_completions = [
117                  (cls, c) for (cls, c) in class_completions if c.startswith(cword_prefix)
118              ]
119          return matched_completions
120  
121      def inject_class_to_parser(self, cls: t.Any) -> None:
122          """Add dummy arguments to our ArgumentParser for the traits of this class 
123   
124          The argparse-based loader currently does not actually add any class traits to 
125          the constructed ArgumentParser, only the flags & aliaes. In order to work nicely 
126          with argcomplete's completers functionality, this method adds dummy arguments 
127          of the form --Class.trait to the ArgumentParser instance. 
128   
129          This method should be called selectively to reduce runtime overhead and to avoid 
130          spamming options across all of Application.classes. 
131          """
132          try:
133              for traitname, trait in cls.class_traits(config=True).items():
134                  completer = trait.metadata.get("argcompleter") or getattr(
135                      trait, "argcompleter", None
136                  )
137                  multiplicity = trait.metadata.get("multiplicity")
138                  self._parser.add_argument(  # type: ignore[attr-defined]
139                      f"--{cls.__name__}.{traitname}",
140                      type=str,
141                      help=trait.help,
142                      nargs=multiplicity,
143                      # metavar=traitname,
144                  ).completer = completer
145                  # argcomplete.debug(f"added --{cls.__name__}.{traitname}")
146          except AttributeError:
147              pass

149      def _get_completions(
150          self, comp_words: t.List[str], cword_prefix: str, *args: t.Any
151      ) -> t.List[str]:
152          """Overridden to dynamically append --Class.trait arguments if appropriate 
153   
154          Warning: 
155              This does not (currently) support completions of the form 
156              --Class1.Class2.<...>.trait, although this is valid for traitlets. 
157              Part of the reason is that we don't currently have a way to identify 
158              which classes may be used with Class1 as a parent. 
159   
160          Warning: 
161              This is an internal method in CompletionFinder and so the API might 
162              be subject to drift. 
163          """
164          # Try to identify if we are completing something related to --Class. for
165          # a known Class, if we are then add the Class config traits to our ArgumentParser.
166          prefix_chars = self._parser.prefix_chars
167          is_option = len(cword_prefix) > 0 and cword_prefix[0] in prefix_chars
168          if is_option:
169              # If we are currently completing an option, check if it could
170              # match with any of the --Class. completions. If there's exactly
171              # one matched class, then expand out the --Class.trait options.
172              matched_completions = self.match_class_completions(cword_prefix)
173              if len(matched_completions) == 1:
174                  matched_cls = matched_completions[0][0]
175                  self.inject_class_to_parser(matched_cls)
176          elif len(comp_words) > 0 and "." in comp_words[-1] and not is_option:
177              # If not an option, perform a hacky check to see if we are completing
178              # an argument for an already present --Class.trait option. Search backwards
179              # for last option (based on last word starting with prefix_chars), and see
180              # if it is of the form --Class.trait. Note that if multiplicity="+", these
181              # arguments might conflict with positional arguments.
182              for prev_word in comp_words[::-1]:
183                  if len(prev_word) > 0 and prev_word[0] in prefix_chars:
184                      matched_completions = self.match_class_completions(prev_word)
185                      if matched_completions:
186                          matched_cls = matched_completions[0][0]
187                          self.inject_class_to_parser(matched_cls)
188                      break
189  
190          completions: t.List[str]
191          completions = super()._get_completions(comp_words, cword_prefix, *args)  # type:ignore[no-untyped-call]
192  
193          # For subcommand-handling: it is difficult to get this to work
194          # using argparse subparsers, because the ArgumentParser accepts
195          # arbitrary extra_args, which ends up masking subparsers.
196          # Instead, check if comp_words only consists of the script,
197          # if so check if any subcommands start with cword_prefix.
198          if self.subcommands and len(comp_words) == 1:
199              argcomplete.debug("Adding subcommands for", cword_prefix)  # type:ignore[attr-defined,no-untyped-call]
200              completions.extend(subc for subc in self.subcommands if subc.startswith(cword_prefix))
201  
202          return completions
203  
204      def _get_option_completions(
205          self, parser: argparse.ArgumentParser, cword_prefix: str
206      ) -> t.List[str]:
207          """Overridden to add --Class. completions when appropriate"""
208          completions: t.List[str]
209          completions = super()._get_option_completions(parser, cword_prefix)  # type:ignore[no-untyped-call]
210          if cword_prefix.endswith("."):
211              return completions
212  
213          matched_completions = self.match_class_completions(cword_prefix)
214          if len(matched_completions) > 1:
215              completions.extend(opt for cls, opt in matched_completions)
216          # If there is exactly one match, we would expect it to have already
217          # been handled by the options dynamically added in _get_completions().
218          # However, maybe there's an edge cases missed here, for example if the
219          # matched class has no configurable traits.
220          return completions
