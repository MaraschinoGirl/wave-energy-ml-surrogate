9    class TestChatterjeeXi:
10       @pytest.mark.parametrize('case', [
11           dict(y_cont=True, statistic=-0.303030303030303, pvalue=0.9351329808526656),
12           dict(y_cont=False, statistic=0.07407407407407396, pvalue=0.3709859367123997)])
13       def test_against_R_XICOR(self, case):
14           # Test against R package XICOR, e.g.
15           # library(XICOR)
16           # options(digits=16)
17           # x = c(0.11027287231363914, 0.8154770102474279, 0.7073943466920335,
18           #       0.6651317324378386, 0.6905752850115503, 0.06115250587536558,
19           #       0.5209906494474178, 0.3155763519785274, 0.18405731803625924,
20           #       0.8613557911541495)
21           # y = c(0.8402081904493103, 0.5946972833914318, 0.23481606164114155,
22           #       0.49754786197715384, 0.9146460831206026, 0.5848057749217579,
23           #       0.7620801065573549, 0.31410063302647495, 0.7935620302236199,
24           #       0.5423085761365468)
25           # xicor(x, y, ties=FALSE, pvalue=TRUE)
26   
27           rng = np.random.default_rng(25982435982346983)
28           x = rng.random(size=10)
29   
30           y = (rng.random(size=10) if case['y_cont']
31                else rng.integers(0, 5, size=10))
32           res = stats.chatterjeexi(x, y, y_continuous=case['y_cont'])
33   
34           assert_allclose(res.statistic, case['statistic'])
35           assert_allclose(res.pvalue, case['pvalue'])
36   
37       @pytest.mark.parametrize('y_continuous', (False, True))
38       def test_permutation_asymptotic(self, y_continuous):
39           # XICOR doesn't seem to perform the permutation test as advertised, so
40           # compare the result of a permutation test against an asymptotic test.
41           rng = np.random.default_rng(2524579827426)
42           n = np.floor(rng.uniform(100, 150)).astype(int)
43           shape = (2, n)
44           x = rng.random(size=shape)
45           y = (rng.random(size=shape) if y_continuous
46                else rng.integers(0, 10, size=shape))
47           method = stats.PermutationMethod(rng=rng)
48           res = stats.chatterjeexi(x, y, method=method,
49                                    y_continuous=y_continuous, axis=-1)
50           ref = stats.chatterjeexi(x, y, y_continuous=y_continuous, axis=-1)
51           np.testing.assert_allclose(res.statistic, ref.statistic, rtol=1e-15)
52           np.testing.assert_allclose(res.pvalue, ref.pvalue, rtol=2e-2)
53   
54       def test_input_validation(self):
55           rng = np.random.default_rng(25932435798274926)
56           x, y = rng.random(size=(2, 10))
57   
58           message = 'Array shapes are incompatible for broadcasting.'
59           with pytest.raises(ValueError, match=message):
60               stats.chatterjeexi(x, y[:-1])
61   
62           message = '...axis 10 is out of bounds for array...'
63           with pytest.raises(ValueError, match=message):
64               stats.chatterjeexi(x, y, axis=10)
65   
66           message = '`y_continuous` must be boolean.'
67           with pytest.raises(ValueError, match=message):
68               stats.chatterjeexi(x, y, y_continuous='a herring')
69   
70           message = "`method` must be 'asymptotic' or"
71           with pytest.raises(ValueError, match=message):
72               stats.chatterjeexi(x, y, method='ekki ekii')
73   
74       def test_special_cases(self):
75           message = 'One or more sample arguments is too small...'
76           with pytest.warns(SmallSampleWarning, match=message):
77               res = stats.chatterjeexi([1], [2])
78   
79           assert np.isnan(res.statistic)
80           assert np.isnan(res.pvalue)
