32   def _find_suite():
33       root = os.environ.get("JSON_SCHEMA_TEST_SUITE")
34       if root is not None:
35           return Path(root)
36   
37       root = Path(jsonschema.__file__).parent.parent / "json"
38       if not root.is_dir():  # pragma: no cover
39           raise ValueError(
40               (
41                   "Can't find the JSON-Schema-Test-Suite directory. "
42                   "Set the 'JSON_SCHEMA_TEST_SUITE' environment "
43                   "variable or run the tests from alongside a checkout "
44                   "of the suite."
45               ),
46           )
47       return root

50   @frozen

56       def __attrs_post_init__(self):
57           jsonschema_suite = self._root.joinpath("bin", "jsonschema_suite")
58           argv = [sys.executable, str(jsonschema_suite), "remotes"]
59           remotes = subprocess.check_output(argv).decode("utf-8")
60   
61           resources = json.loads(remotes)
62   
63           li = "http://localhost:1234/locationIndependentIdentifierPre2019.json"
64           li4 = "http://localhost:1234/locationIndependentIdentifierDraft4.json"
65   
66           registry = Registry().with_resources(
67               [
68                   (
69                       li,
70                       referencing.jsonschema.DRAFT7.create_resource(
71                           contents=resources.pop(li),
72                       ),
73                   ),
74                   (
75                       li4,
76                       referencing.jsonschema.DRAFT4.create_resource(
77                           contents=resources.pop(li4),
78                       ),
79                   ),
80               ],
81           ).with_contents(
82               resources.items(),
83               default_specification=referencing.jsonschema.DRAFT202012,
84           )
85           object.__setattr__(self, "_remotes", registry)

102  @frozen
103  class Version:
104  
105      _path: Path
106      _remotes: referencing.jsonschema.SchemaRegistry
107  
108      name: str
109  
110      def benchmark(self, **kwargs):  # pragma: no cover
111          for case in self.cases():
112              case.benchmark(**kwargs)

114      def cases(self) -> Iterable[_Case]:
115          return self._cases_in(paths=self._path.glob("*.json"))
116  
117      def format_cases(self) -> Iterable[_Case]:
118          return self._cases_in(paths=self._path.glob("optional/format/*.json"))
119  
120      def optional_cases_of(self, name: str) -> Iterable[_Case]:
121          return self._cases_in(paths=[self._path / "optional" / f"{name}.json"])
122  
123      def to_unittest_testcase(self, *groups, **kwargs):
124          name = kwargs.pop("name", "Test" + self.name.title().replace("-", ""))
125          methods = {
126              method.__name__: method
127              for method in (
128                  test.to_unittest_method(**kwargs)
129                  for group in groups
130                  for case in group
131                  for test in case.tests
132              )
133          }
134          cls = type(name, (unittest.TestCase,), methods)
135  
136          # We're doing crazy things, so if they go wrong, like a function
137          # behaving differently on some other interpreter, just make them
138          # not happen.
139          with suppress(Exception):
140              cls.__module__ = _someone_save_us_the_module_of_the_caller()
141  
142          return cls
143  
144      def _cases_in(self, paths: Iterable[Path]) -> Iterable[_Case]:
145          for path in paths:
146              for case in json.loads(path.read_text(encoding="utf-8")):
147                  yield _Case.from_dict(
148                      case,
149                      version=self,
150                      subject=path.stem,
151                      remotes=self._remotes,
152                  )

155  @frozen
156  class _Case:
157  
158      version: Version
159  
160      subject: str
161      description: str
162      schema: Mapping[str, Any] | bool
163      tests: list[_Test]
164      comment: str | None = None
165      specification: Sequence[dict[str, str]] = ()
166  
167      @classmethod
168      def from_dict(cls, data, remotes, **kwargs):
169          data.update(kwargs)
170          tests = [
171              _Test(
172                  version=data["version"],
173                  subject=data["subject"],
174                  case_description=data["description"],
175                  schema=data["schema"],
176                  remotes=remotes,
177                  **test,
178              ) for test in data.pop("tests")
179          ]
180          return cls(tests=tests, **data)
181  
182      def benchmark(self, runner: pyperf.Runner, **kwargs):  # pragma: no cover
183          for test in self.tests:
184              runner.bench_func(
185                  test.fully_qualified_name,
186                  partial(test.validate_ignoring_errors, **kwargs),
187              )

190  @frozen(repr=False)
191  class _Test:
192  
193      version: Version
194  
195      subject: str
196      case_description: str
197      description: str
198  
199      data: Any
200      schema: Mapping[str, Any] | bool
201  
202      valid: bool
203  
204      _remotes: referencing.jsonschema.SchemaRegistry
205  
206      comment: str | None = None
207  
208      def __repr__(self):  # pragma: no cover
209          return f"<Test {self.fully_qualified_name}>"
210  
211      @property
212      def fully_qualified_name(self):  # pragma: no cover
213          return " > ".join(  # noqa: FLY002
214              [
215                  self.version.name,
216                  self.subject,
217                  self.case_description,
218                  self.description,
219              ],
220          )
221  
222      def to_unittest_method(self, skip=lambda test: None, **kwargs):
223          if self.valid:
224              def fn(this):
225                  self.validate(**kwargs)
226          else:
227              def fn(this):
228                  with this.assertRaises(jsonschema.ValidationError):
229                      self.validate(**kwargs)
230  
231          fn.__name__ = "_".join(
232              [
233                  "test",
234                  _DELIMITERS.sub("_", self.subject),
235                  _DELIMITERS.sub("_", self.case_description),
236                  _DELIMITERS.sub("_", self.description),
237              ],
238          )
239          reason = skip(self)
240          if reason is None or os.environ.get("JSON_SCHEMA_DEBUG", "0") != "0":
241              return fn
242          elif os.environ.get("JSON_SCHEMA_EXPECTED_FAILURES", "0") != "0":  # pragma: no cover  # noqa: E501
243              return unittest.expectedFailure(fn)
244          else:
245              return unittest.skip(reason)(fn)
246  
247      def validate(self, Validator, **kwargs):
248          Validator.check_schema(self.schema)
249          validator = Validator(
250              schema=self.schema,
251              registry=self._remotes,
252              **kwargs,
253          )
254          if os.environ.get("JSON_SCHEMA_DEBUG", "0") != "0":  # pragma: no cover
255              breakpoint()  # noqa: T100
256          validator.validate(instance=self.data)
257  
258      def validate_ignoring_errors(self, Validator):  # pragma: no cover
259          with suppress(jsonschema.ValidationError):
260              self.validate(Validator=Validator)

263  def _someone_save_us_the_module_of_the_caller():
264      """ 
265      The FQON of the module 2nd stack frames up from here. 
266   
267      This is intended to allow us to dynamically return test case classes that 
268      are indistinguishable from being defined in the module that wants them. 
269   
270      Otherwise, trial will mis-print the FQON, and copy pasting it won't re-run 
271      the class that really is running. 
272   
273      Save us all, this is all so so so so so terrible. 
274      """
275  
276      return sys._getframe(2).f_globals["__name__"]
