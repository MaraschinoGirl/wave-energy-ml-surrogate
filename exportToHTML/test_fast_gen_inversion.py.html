32   def test_random_state():
33       # fixed seed
34       gen = FastGeneratorInversion(stats.norm(), random_state=68734509)
35       x1 = gen.rvs(size=10)
36       gen.random_state = 68734509
37       x2 = gen.rvs(size=10)
38       assert_array_equal(x1, x2)
39   
40       # Generator
41       urng = np.random.default_rng(20375857)
42       gen = FastGeneratorInversion(stats.norm(), random_state=urng)
43       x1 = gen.rvs(size=10)
44       gen.random_state = np.random.default_rng(20375857)
45       x2 = gen.rvs(size=10)
46       assert_array_equal(x1, x2)
47   
48       # RandomState
49       urng = np.random.RandomState(2364)
50       gen = FastGeneratorInversion(stats.norm(), random_state=urng)
51       x1 = gen.rvs(size=10)
52       gen.random_state = np.random.RandomState(2364)
53       x2 = gen.rvs(size=10)
54       assert_array_equal(x1, x2)
55   
56       # if evaluate_error is called, it must not interfere with the random_state
57       # used by rvs
58       gen = FastGeneratorInversion(stats.norm(), random_state=68734509)
59       x1 = gen.rvs(size=10)
60       _ = gen.evaluate_error(size=5)  # this will generate 5 uniform rvs
61       x2 = gen.rvs(size=10)
62       gen.random_state = 68734509
63       x3 = gen.rvs(size=20)
64       assert_array_equal(x2, x3[10:])

106  @pytest.mark.parametrize(("distname, args"), dists_with_params)
107  def test_rvs_and_ppf(distname, args):
108      # check sample against rvs generated by rv_continuous
109      urng = np.random.default_rng(9807324628097097)
110      rng1 = getattr(stats, distname)(*args)
111      rvs1 = rng1.rvs(size=500, random_state=urng)
112      rng2 = FastGeneratorInversion(rng1, random_state=urng)
113      rvs2 = rng2.rvs(size=500)
114      assert stats.cramervonmises_2samp(rvs1, rvs2).pvalue > 0.01
115  
116      # check ppf
117      q = [0.001, 0.1, 0.5, 0.9, 0.999]
118      assert_allclose(rng1.ppf(q), rng2.ppf(q), atol=1e-10)

181  def test_domain():
182      # only a basic check that the domain argument is passed to the
183      # UNU.RAN generators
184      rng = FastGeneratorInversion(stats.norm(), domain=(-1, 1))
185      r = rng.rvs(size=100)
186      assert -1 <= r.min() < r.max() <= 1
187  
188      # if loc and scale are used, new domain is loc + scale*domain
189      loc, scale = 3.5, 1.3
190      dist = stats.norm(loc=loc, scale=scale)
191      rng = FastGeneratorInversion(dist, domain=(-1.5, 2))
192      r = rng.rvs(size=100)
193      lb, ub = loc - scale * 1.5, loc + scale * 2
194      assert lb <= r.min() < r.max() <= ub

197  @pytest.mark.parametrize(("distname, args, expected"),
198                           [("beta", (3.5, 2.5), (0, 1)),
199                            ("norm", (), (-np.inf, np.inf))])
200  def test_support(distname, args, expected):
201      # test that the support is updated if truncation and loc/scale are applied
202      # use beta distribution since it is a transformed betaprime distribution,
203      # so it is important that the correct support is considered
204      # (i.e., the support of beta is (0,1), while betaprime is (0, inf))
205      dist = getattr(stats, distname)(*args)
206      rng = FastGeneratorInversion(dist)
207      assert_array_equal(rng.support(), expected)
208      rng.loc = 1
209      rng.scale = 2
210      assert_array_equal(rng.support(), 1 + 2*np.array(expected))

236  def test_non_rvs_methods_with_domain():
237      # as a first step, compare truncated normal against stats.truncnorm
238      rng = FastGeneratorInversion(stats.norm(), domain=(2.3, 3.2))
239      trunc_norm = stats.truncnorm(2.3, 3.2)
240      # take values that are inside and outside the domain
241      x = (2.0, 2.4, 3.0, 3.4)
242      p = (0.01, 0.5, 0.99)
243      assert_allclose(rng._cdf(x), trunc_norm.cdf(x))
244      assert_allclose(rng._ppf(p), trunc_norm.ppf(p))
245      loc, scale = 2, 3
246      rng.loc = 2
247      rng.scale = 3
248      trunc_norm = stats.truncnorm(2.3, 3.2, loc=loc, scale=scale)
249      x = np.array(x) * scale + loc
250      assert_allclose(rng._cdf(x), trunc_norm.cdf(x))
251      assert_allclose(rng._ppf(p), trunc_norm.ppf(p))
252  
253      # do another sanity check with beta distribution
254      # in that case, it is important to use the correct domain since beta
255      # is a transformation of betaprime which has a different support
256      rng = FastGeneratorInversion(stats.beta(2.5, 3.5), domain=(0.3, 0.7))
257      rng.loc = 2
258      rng.scale = 2.5
259      # the support is 2.75, , 3.75 (2 + 2.5 * 0.3, 2 + 2.5 * 0.7)
260      assert_array_equal(rng.support(), (2.75, 3.75))
261      x = np.array([2.74, 2.76, 3.74, 3.76])
262      # the cdf needs to be zero outside of the domain
263      y_cdf = rng._cdf(x)
264      assert_array_equal((y_cdf[0], y_cdf[3]), (0, 1))
265      assert np.min(y_cdf[1:3]) > 0
266      # ppf needs to map 0 and 1 to the boundaries
267      assert_allclose(rng._ppf(y_cdf), (2.75, 2.76, 3.74, 3.75))

297  def test_domain_argus_large_chi():
298      # for large chi, the Gamma distribution is used and the domain has to be
299      # transformed. this is a test to ensure that the transformation works
300      chi, lb, ub = 5.5, 0.25, 0.75
301      rng = FastGeneratorInversion(stats.argus(chi), domain=(lb, ub))
302      rng.random_state = 4574
303      r = rng.rvs(size=500)
304      assert lb <= r.min() < r.max() <= ub
305      # perform goodness of fit test with conditional cdf
306      cdf = stats.argus(chi).cdf
307      prob = cdf(ub) - cdf(lb)
308      assert stats.cramervonmises(r, lambda x: cdf(x) / prob).pvalue > 0.05

336  @pytest.mark.xfail_on_32bit(
337      "NumericalInversePolynomial.qrvs fails for Win 32-bit"
338  )
339  class TestQRVS:
340      def test_input_validation(self):
341          gen = FastGeneratorInversion(stats.norm())
342  
343          match = "`qmc_engine` must be an instance of..."
344          with pytest.raises(ValueError, match=match):
345              gen.qrvs(qmc_engine=0)
346  
347          match = "`d` must be consistent with dimension of `qmc_engine`."
348          with pytest.raises(ValueError, match=match):
349              gen.qrvs(d=3, qmc_engine=stats.qmc.Halton(2))
350  
351      qrngs = [None, stats.qmc.Sobol(1, seed=0), stats.qmc.Halton(3, seed=0)]
352      # `size=None` should not add anything to the shape, `size=1` should
353      sizes = [
354          (None, tuple()),
355          (1, (1,)),
356          (4, (4,)),
357          ((4,), (4,)),
358          ((2, 4), (2, 4)),
359      ]
360      # Neither `d=None` nor `d=1` should add anything to the shape
361      ds = [(None, tuple()), (1, tuple()), (3, (3,))]
362  
363      @pytest.mark.parametrize("qrng", qrngs)
364      @pytest.mark.parametrize("size_in, size_out", sizes)
365      @pytest.mark.parametrize("d_in, d_out", ds)
366      def test_QRVS_shape_consistency(self, qrng, size_in, size_out,
367                                      d_in, d_out):
368          gen = FastGeneratorInversion(stats.norm())
369  
370          # If d and qrng.d are inconsistent, an error is raised
371          if d_in is not None and qrng is not None and qrng.d != d_in:
372              match = "`d` must be consistent with dimension of `qmc_engine`."
373              with pytest.raises(ValueError, match=match):
374                  gen.qrvs(size_in, d=d_in, qmc_engine=qrng)
375              return
376  
377          # Sometimes d is really determined by qrng
378          if d_in is None and qrng is not None and qrng.d != 1:
379              d_out = (qrng.d,)
380  
381          shape_expected = size_out + d_out
382  
383          qrng2 = deepcopy(qrng)
384          qrvs = gen.qrvs(size=size_in, d=d_in, qmc_engine=qrng)
385          if size_in is not None:
386              assert qrvs.shape == shape_expected
387  
388          if qrng2 is not None:
389              uniform = qrng2.random(np.prod(size_in) or 1)
390              qrvs2 = stats.norm.ppf(uniform).reshape(shape_expected)
391              assert_allclose(qrvs, qrvs2, atol=1e-12)

393      def test_QRVS_size_tuple(self):
394          # QMCEngine samples are always of shape (n, d). When `size` is a tuple,
395          # we set `n = prod(size)` in the call to qmc_engine.random, transform
396          # the sample, and reshape it to the final dimensions. When we reshape,
397          # we need to be careful, because the _columns_ of the sample returned
398          # by a QMCEngine are "independent"-ish, but the elements within the
399          # columns are not. We need to make sure that this doesn't get mixed up
400          # by reshaping: qrvs[..., i] should remain "independent"-ish of
401          # qrvs[..., i+1], but the elements within qrvs[..., i] should be
402          # transformed from the same low-discrepancy sequence.
403  
404          gen = FastGeneratorInversion(stats.norm())
405  
406          size = (3, 4)
407          d = 5
408          qrng = stats.qmc.Halton(d, seed=0)
409          qrng2 = stats.qmc.Halton(d, seed=0)
410  
411          uniform = qrng2.random(np.prod(size))
412  
413          qrvs = gen.qrvs(size=size, d=d, qmc_engine=qrng)
414          qrvs2 = stats.norm.ppf(uniform)
415  
416          for i in range(d):
417              sample = qrvs[..., i]
418              sample2 = qrvs2[:, i].reshape(size)
419              assert_allclose(sample, sample2, atol=1e-12)

422  def test_burr_overflow():
423      # this case leads to an overflow error if math.exp is used
424      # in the definition of the burr pdf instead of np.exp
425      # a direct implementation of the PDF as x**(-c-1) / (1+x**(-c))**(d+1)
426      # also leads to an overflow error in the setup
427      args = (1.89128135, 0.30195177)
428      with suppress_warnings() as sup:
429          # filter potential overflow warning
430          sup.filter(RuntimeWarning)
431          gen = FastGeneratorInversion(stats.burr(*args))
432      u_error, _ = gen.evaluate_error(random_state=4326)
433      assert u_error <= 1e-10
