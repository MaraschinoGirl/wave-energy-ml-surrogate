61   class _ColorPalette(list):
62       """Set the color palette in a with statement, otherwise be a list."""
63       def __enter__(self):
64           """Open the context."""
65           from .rcmod import set_palette
66           self._orig_palette = color_palette()
67           set_palette(self)
68           return self
69   
70       def __exit__(self, *args):
71           """Close the context."""
72           from .rcmod import set_palette
73           set_palette(self._orig_palette)

75       def as_hex(self):
76           """Return a color palette with hex codes instead of RGB values."""
77           hex = [mpl.colors.rgb2hex(rgb) for rgb in self]
78           return _ColorPalette(hex)

80       def _repr_html_(self):
81           """Rich display of the color palette in an HTML frontend."""
82           s = 55
83           n = len(self)
84           html = f'<svg  width="{n * s}" height="{s}">'
85           for i, c in enumerate(self.as_hex()):
86               html += (
87                   f'<rect x="{i * s}" y="0" width="{s}" height="{s}" style="fill:{c};'
88                   'stroke-width:2;stroke:rgb(255,255,255)"/>'
89               )
90           html += '</svg>'
91           return html

94   def _patch_colormap_display():
95       """Simplify the rich display of matplotlib color maps in a notebook."""
96       def _repr_png_(self):
97           """Generate a PNG representation of the Colormap."""
98           import io
99           from PIL import Image
100          import numpy as np
101          IMAGE_SIZE = (400, 50)
102          X = np.tile(np.linspace(0, 1, IMAGE_SIZE[0]), (IMAGE_SIZE[1], 1))
103          pixels = self(X, bytes=True)
104          png_bytes = io.BytesIO()
105          Image.fromarray(pixels).save(png_bytes, format='png')
106          return png_bytes.getvalue()
107  
108      def _repr_html_(self):
109          """Generate an HTML representation of the Colormap."""
110          import base64
111          png_bytes = self._repr_png_()
112          png_base64 = base64.b64encode(png_bytes).decode('ascii')
113          return ('<img '
114                  + 'alt="' + self.name + ' color map" '
115                  + 'title="' + self.name + '"'
116                  + 'src="data:image/png;base64,' + png_base64 + '">')
117  
118      mpl.colors.Colormap._repr_png_ = _repr_png_
119      mpl.colors.Colormap._repr_html_ = _repr_html_

122  def color_palette(palette=None, n_colors=None, desat=None, as_cmap=False):
123      """Return a list of colors or continuous colormap defining a palette. 
124   
125      Possible ``palette`` values include: 
126          - Name of a seaborn palette (deep, muted, bright, pastel, dark, colorblind) 
127          - Name of matplotlib colormap 
128          - 'husl' or 'hls' 
129          - 'ch:<cubehelix arguments>' 
130          - 'light:<color>', 'dark:<color>', 'blend:<color>,<color>', 
131          - A sequence of colors in any format matplotlib accepts 
132   
133      Calling this function with ``palette=None`` will return the current 
134      matplotlib color cycle. 
135   
136      This function can also be used in a ``with`` statement to temporarily 
137      set the color cycle for a plot or set of plots. 
138   
139      See the :ref:`tutorial <palette_tutorial>` for more information. 
140   
141      Parameters 
142      ---------- 
143      palette : None, string, or sequence, optional 
144          Name of palette or None to return current palette. If a sequence, input 
145          colors are used but possibly cycled and desaturated. 
146      n_colors : int, optional 
147          Number of colors in the palette. If ``None``, the default will depend 
148          on how ``palette`` is specified. Named palettes default to 6 colors, 
149          but grabbing the current palette or passing in a list of colors will 
150          not change the number of colors unless this is specified. Asking for 
151          more colors than exist in the palette will cause it to cycle. Ignored 
152          when ``as_cmap`` is True. 
153      desat : float, optional 
154          Proportion to desaturate each color by. 
155      as_cmap : bool 
156          If True, return a :class:`matplotlib.colors.ListedColormap`. 
157   
158      Returns 
159      ------- 
160      list of RGB tuples or :class:`matplotlib.colors.ListedColormap` 
161   
162      See Also 
163      -------- 
164      set_palette : Set the default color cycle for all plots. 
165      set_color_codes : Reassign color codes like ``"b"``, ``"g"``, etc. to 
166                        colors from one of the seaborn palettes. 
167   
168      Examples 
169      -------- 
170   
171      .. include:: ../docstrings/color_palette.rst 
172   
173      """
174      if palette is None:
175          palette = get_color_cycle()
176          if n_colors is None:
177              n_colors = len(palette)
178  
179      elif not isinstance(palette, str):
180          palette = palette
181          if n_colors is None:
182              n_colors = len(palette)
183      else:
184  
185          if n_colors is None:
186              # Use all colors in a qualitative palette or 6 of another kind
187              n_colors = QUAL_PALETTE_SIZES.get(palette, 6)
188  
189          if palette in SEABORN_PALETTES:
190              # Named "seaborn variant" of matplotlib default color cycle
191              palette = SEABORN_PALETTES[palette]
192  
193          elif palette == "hls":
194              # Evenly spaced colors in cylindrical RGB space
195              palette = hls_palette(n_colors, as_cmap=as_cmap)
196  
197          elif palette == "husl":
198              # Evenly spaced colors in cylindrical Lab space
199              palette = husl_palette(n_colors, as_cmap=as_cmap)
200  
201          elif palette.lower() == "jet":
202              # Paternalism
203              raise ValueError("No.")
204  
205          elif palette.startswith("ch:"):
206              # Cubehelix palette with params specified in string
207              args, kwargs = _parse_cubehelix_args(palette)
208              palette = cubehelix_palette(n_colors, *args, **kwargs, as_cmap=as_cmap)
209  
210          elif palette.startswith("light:"):
211              # light palette to color specified in string
212              _, color = palette.split(":")
213              reverse = color.endswith("_r")
214              if reverse:
215                  color = color[:-2]
216              palette = light_palette(color, n_colors, reverse=reverse, as_cmap=as_cmap)
217  
218          elif palette.startswith("dark:"):
219              # light palette to color specified in string
220              _, color = palette.split(":")
221              reverse = color.endswith("_r")
222              if reverse:
223                  color = color[:-2]
224              palette = dark_palette(color, n_colors, reverse=reverse, as_cmap=as_cmap)
225  
226          elif palette.startswith("blend:"):
227              # blend palette between colors specified in string
228              _, colors = palette.split(":")
229              colors = colors.split(",")
230              palette = blend_palette(colors, n_colors, as_cmap=as_cmap)
231  
232          else:
233              try:
234                  # Perhaps a named matplotlib colormap?
235                  palette = mpl_palette(palette, n_colors, as_cmap=as_cmap)
236              except (ValueError, KeyError):  # Error class changed in mpl36
237                  raise ValueError(f"{palette!r} is not a valid palette name")
238  
239      if desat is not None:
240          palette = [desaturate(c, desat) for c in palette]
241  
242      if not as_cmap:
243  
244          # Always return as many colors as we asked for
245          pal_cycle = cycle(palette)
246          palette = [next(pal_cycle) for _ in range(n_colors)]
247  
248          # Always return in r, g, b tuple format
249          try:
250              palette = map(mpl.colors.colorConverter.to_rgb, palette)
251              palette = _ColorPalette(palette)
252          except ValueError:
253              raise ValueError(f"Could not generate a palette for {palette}")
254  
255      return palette

258  def hls_palette(n_colors=6, h=.01, l=.6, s=.65, as_cmap=False):  # noqa
259      """ 
260      Return hues with constant lightness and saturation in the HLS system. 
261   
262      The hues are evenly sampled along a circular path. The resulting palette will be 
263      appropriate for categorical or cyclical data. 
264   
265      The `h`, `l`, and `s` values should be between 0 and 1. 
266   
267      .. note:: 
268          While the separation of the resulting colors will be mathematically 
269          constant, the HLS system does not construct a perceptually-uniform space, 
270          so their apparent intensity will vary. 
271   
272      Parameters 
273      ---------- 
274      n_colors : int 
275          Number of colors in the palette. 
276      h : float 
277          The value of the first hue. 
278      l : float 
279          The lightness value. 
280      s : float 
281          The saturation intensity. 
282      as_cmap : bool 
283          If True, return a matplotlib colormap object. 
284   
285      Returns 
286      ------- 
287      palette 
288          list of RGB tuples or :class:`matplotlib.colors.ListedColormap` 
289   
290      See Also 
291      -------- 
292      husl_palette : Make a palette using evenly spaced hues in the HUSL system. 
293   
294      Examples 
295      -------- 
296      .. include:: ../docstrings/hls_palette.rst 
297   
298      """
299      if as_cmap:
300          n_colors = 256
301      hues = np.linspace(0, 1, int(n_colors) + 1)[:-1]
302      hues += h
303      hues %= 1
304      hues -= hues.astype(int)
305      palette = [colorsys.hls_to_rgb(h_i, l, s) for h_i in hues]
306      if as_cmap:
307          return mpl.colors.ListedColormap(palette, "hls")
308      else:
309          return _ColorPalette(palette)

312  def husl_palette(n_colors=6, h=.01, s=.9, l=.65, as_cmap=False):  # noqa
313      """ 
314      Return hues with constant lightness and saturation in the HUSL system. 
315   
316      The hues are evenly sampled along a circular path. The resulting palette will be 
317      appropriate for categorical or cyclical data. 
318   
319      The `h`, `l`, and `s` values should be between 0 and 1. 
320   
321      This function is similar to :func:`hls_palette`, but it uses a nonlinear color 
322      space that is more perceptually uniform. 
323   
324      Parameters 
325      ---------- 
326      n_colors : int 
327          Number of colors in the palette. 
328      h : float 
329          The value of the first hue. 
330      l : float 
331          The lightness value. 
332      s : float 
333          The saturation intensity. 
334      as_cmap : bool 
335          If True, return a matplotlib colormap object. 
336   
337      Returns 
338      ------- 
339      palette 
340          list of RGB tuples or :class:`matplotlib.colors.ListedColormap` 
341   
342      See Also 
343      -------- 
344      hls_palette : Make a palette using evenly spaced hues in the HSL system. 
345   
346      Examples 
347      -------- 
348      .. include:: ../docstrings/husl_palette.rst 
349   
350      """
351      if as_cmap:
352          n_colors = 256
353      hues = np.linspace(0, 1, int(n_colors) + 1)[:-1]
354      hues += h
355      hues %= 1
356      hues *= 359
357      s *= 99
358      l *= 99  # noqa
359      palette = [_color_to_rgb((h_i, s, l), input="husl") for h_i in hues]
360      if as_cmap:
361          return mpl.colors.ListedColormap(palette, "hsl")
362      else:
363          return _ColorPalette(palette)

366  def mpl_palette(name, n_colors=6, as_cmap=False):
367      """ 
368      Return a palette or colormap from the matplotlib registry. 
369   
370      For continuous palettes, evenly-spaced discrete samples are chosen while 
371      excluding the minimum and maximum value in the colormap to provide better 
372      contrast at the extremes. 
373   
374      For qualitative palettes (e.g. those from colorbrewer), exact values are 
375      indexed (rather than interpolated), but fewer than `n_colors` can be returned 
376      if the palette does not define that many. 
377   
378      Parameters 
379      ---------- 
380      name : string 
381          Name of the palette. This should be a named matplotlib colormap. 
382      n_colors : int 
383          Number of discrete colors in the palette. 
384   
385      Returns 
386      ------- 
387      list of RGB tuples or :class:`matplotlib.colors.ListedColormap` 
388   
389      Examples 
390      -------- 
391      .. include:: ../docstrings/mpl_palette.rst 
392   
393      """
394      if name.endswith("_d"):
395          sub_name = name[:-2]
396          if sub_name.endswith("_r"):
397              reverse = True
398              sub_name = sub_name[:-2]
399          else:
400              reverse = False
401          pal = color_palette(sub_name, 2) + ["#333333"]
402          if reverse:
403              pal = pal[::-1]
404          cmap = blend_palette(pal, n_colors, as_cmap=True)
405      else:
406          cmap = get_colormap(name)
407  
408      if name in MPL_QUAL_PALS:
409          bins = np.linspace(0, 1, MPL_QUAL_PALS[name])[:n_colors]
410      else:
411          bins = np.linspace(0, 1, int(n_colors) + 2)[1:-1]
412      palette = list(map(tuple, cmap(bins)[:, :3]))
413  
414      if as_cmap:
415          return cmap
416      else:
417          return _ColorPalette(palette)

420  def _color_to_rgb(color, input):
421      """Add some more flexibility to color choices."""
422      if input == "hls":
423          color = colorsys.hls_to_rgb(*color)
424      elif input == "husl":
425          color = husl.husl_to_rgb(*color)
426          color = tuple(np.clip(color, 0, 1))
427      elif input == "xkcd":
428          color = xkcd_rgb[color]
429  
430      return mpl.colors.to_rgb(color)

433  def dark_palette(color, n_colors=6, reverse=False, as_cmap=False, input="rgb"):
434      """Make a sequential palette that blends from dark to ``color``. 
435   
436      This kind of palette is good for data that range between relatively 
437      uninteresting low values and interesting high values. 
438   
439      The ``color`` parameter can be specified in a number of ways, including 
440      all options for defining a color in matplotlib and several additional 
441      color spaces that are handled by seaborn. You can also use the database 
442      of named colors from the XKCD color survey. 
443   
444      If you are using the IPython notebook, you can also choose this palette 
445      interactively with the :func:`choose_dark_palette` function. 
446   
447      Parameters 
448      ---------- 
449      color : base color for high values 
450          hex, rgb-tuple, or html color name 
451      n_colors : int, optional 
452          number of colors in the palette 
453      reverse : bool, optional 
454          if True, reverse the direction of the blend 
455      as_cmap : bool, optional 
456          If True, return a :class:`matplotlib.colors.ListedColormap`. 
457      input : {'rgb', 'hls', 'husl', xkcd'} 
458          Color space to interpret the input color. The first three options 
459          apply to tuple inputs and the latter applies to string inputs. 
460   
461      Returns 
462      ------- 
463      palette 
464          list of RGB tuples or :class:`matplotlib.colors.ListedColormap` 
465   
466      See Also 
467      -------- 
468      light_palette : Create a sequential palette with bright low values. 
469      diverging_palette : Create a diverging palette with two colors. 
470   
471      Examples 
472      -------- 
473      .. include:: ../docstrings/dark_palette.rst 
474   
475      """
476      rgb = _color_to_rgb(color, input)
477      hue, sat, _ = husl.rgb_to_husl(*rgb)
478      gray_s, gray_l = .15 * sat, 15
479      gray = _color_to_rgb((hue, gray_s, gray_l), input="husl")
480      colors = [rgb, gray] if reverse else [gray, rgb]
481      return blend_palette(colors, n_colors, as_cmap)

532  def diverging_palette(h_neg, h_pos, s=75, l=50, sep=1, n=6,  # noqa
533                        center="light", as_cmap=False):
534      """Make a diverging palette between two HUSL colors. 
535   
536      If you are using the IPython notebook, you can also choose this palette 
537      interactively with the :func:`choose_diverging_palette` function. 
538   
539      Parameters 
540      ---------- 
541      h_neg, h_pos : float in [0, 359] 
542          Anchor hues for negative and positive extents of the map. 
543      s : float in [0, 100], optional 
544          Anchor saturation for both extents of the map. 
545      l : float in [0, 100], optional 
546          Anchor lightness for both extents of the map. 
547      sep : int, optional 
548          Size of the intermediate region. 
549      n : int, optional 
550          Number of colors in the palette (if not returning a cmap) 
551      center : {"light", "dark"}, optional 
552          Whether the center of the palette is light or dark 
553      as_cmap : bool, optional 
554          If True, return a :class:`matplotlib.colors.ListedColormap`. 
555   
556      Returns 
557      ------- 
558      palette 
559          list of RGB tuples or :class:`matplotlib.colors.ListedColormap` 
560   
561      See Also 
562      -------- 
563      dark_palette : Create a sequential palette with dark values. 
564      light_palette : Create a sequential palette with light values. 
565   
566      Examples 
567      -------- 
568      .. include: ../docstrings/diverging_palette.rst 
569   
570      """
571      palfunc = dict(dark=dark_palette, light=light_palette)[center]
572      n_half = int(128 - (sep // 2))
573      neg = palfunc((h_neg, s, l), n_half, reverse=True, input="husl")
574      pos = palfunc((h_pos, s, l), n_half, input="husl")
575      midpoint = dict(light=[(.95, .95, .95)], dark=[(.133, .133, .133)])[center]
576      mid = midpoint * sep
577      pal = blend_palette(np.concatenate([neg, mid, pos]), n, as_cmap=as_cmap)
578      return pal

581  def blend_palette(colors, n_colors=6, as_cmap=False, input="rgb"):
582      """Make a palette that blends between a list of colors. 
583   
584      Parameters 
585      ---------- 
586      colors : sequence of colors in various formats interpreted by `input` 
587          hex code, html color name, or tuple in `input` space. 
588      n_colors : int, optional 
589          Number of colors in the palette. 
590      as_cmap : bool, optional 
591          If True, return a :class:`matplotlib.colors.ListedColormap`. 
592   
593      Returns 
594      ------- 
595      palette 
596          list of RGB tuples or :class:`matplotlib.colors.ListedColormap` 
597   
598      Examples 
599      -------- 
600      .. include: ../docstrings/blend_palette.rst 
601   
602      """
603      colors = [_color_to_rgb(color, input) for color in colors]
604      name = "blend"
605      pal = mpl.colors.LinearSegmentedColormap.from_list(name, colors)
606      if not as_cmap:
607          rgb_array = pal(np.linspace(0, 1, int(n_colors)))[:, :3]  # no alpha
608          pal = _ColorPalette(map(tuple, rgb_array))
609      return pal

612  def xkcd_palette(colors):
613      """Make a palette with color names from the xkcd color survey. 
614   
615      See xkcd for the full list of colors: https://xkcd.com/color/rgb/ 
616   
617      This is just a simple wrapper around the `seaborn.xkcd_rgb` dictionary. 
618   
619      Parameters

665  def cubehelix_palette(n_colors=6, start=0, rot=.4, gamma=1.0, hue=0.8,
666                        light=.85, dark=.15, reverse=False, as_cmap=False):
667      """Make a sequential palette from the cubehelix system. 
668   
669      This produces a colormap with linearly-decreasing (or increasing) 
670      brightness. That means that information will be preserved if printed to 
671      black and white or viewed by someone who is colorblind.  "cubehelix" is 
672      also available as a matplotlib-based palette, but this function gives the 
673      user more control over the look of the palette and has a different set of 
674      defaults. 
675   
676      In addition to using this function, it is also possible to generate a 
677      cubehelix palette generally in seaborn using a string starting with 
678      `ch:` and containing other parameters (e.g. `"ch:s=.25,r=-.5"`). 
679   
680      Parameters 
681      ---------- 
682      n_colors : int 
683          Number of colors in the palette. 
684      start : float, 0 <= start <= 3 
685          The hue value at the start of the helix. 
686      rot : float 
687          Rotations around the hue wheel over the range of the palette. 
688      gamma : float 0 <= gamma 
689          Nonlinearity to emphasize dark (gamma < 1) or light (gamma > 1) colors. 
690      hue : float, 0 <= hue <= 1 
691          Saturation of the colors. 
692      dark : float 0 <= dark <= 1 
693          Intensity of the darkest color in the palette. 
694      light : float 0 <= light <= 1 
695          Intensity of the lightest color in the palette. 
696      reverse : bool 
697          If True, the palette will go from dark to light. 
698      as_cmap : bool 
699          If True, return a :class:`matplotlib.colors.ListedColormap`. 
700   
701      Returns 
702      ------- 
703      palette 
704          list of RGB tuples or :class:`matplotlib.colors.ListedColormap` 
705   
706      See Also 
707      -------- 
708      choose_cubehelix_palette : Launch an interactive widget to select cubehelix 
709                                 palette parameters. 
710      dark_palette : Create a sequential palette with dark low values. 
711      light_palette : Create a sequential palette with bright low values. 
712   
713      References 
714      ---------- 
715      Green, D. A. (2011). "A colour scheme for the display of astronomical 
716      intensity images". Bulletin of the Astromical Society of India, Vol. 39, 
717      p. 289-295. 
718   
719      Examples 
720      -------- 
721      .. include:: ../docstrings/cubehelix_palette.rst 
722   
723      """
724      def get_color_function(p0, p1):
725          # Copied from matplotlib because it lives in private module
726          def color(x):
727              # Apply gamma factor to emphasise low or high intensity values
728              xg = x ** gamma
729  
730              # Calculate amplitude and angle of deviation from the black
731              # to white diagonal in the plane of constant
732              # perceived intensity.
733              a = hue * xg * (1 - xg) / 2
734  
735              phi = 2 * np.pi * (start / 3 + rot * x)
736  
737              return xg + a * (p0 * np.cos(phi) + p1 * np.sin(phi))
738          return color
739  
740      cdict = {
741          "red": get_color_function(-0.14861, 1.78277),
742          "green": get_color_function(-0.29227, -0.90649),
743          "blue": get_color_function(1.97294, 0.0),
744      }
745  
746      cmap = mpl.colors.LinearSegmentedColormap("cubehelix", cdict)
747  
748      x = np.linspace(light, dark, int(n_colors))
749      pal = cmap(x)[:, :3].tolist()
750      if reverse:
751          pal = pal[::-1]
752  
753      if as_cmap:
754          x_256 = np.linspace(light, dark, 256)
755          if reverse:
756              x_256 = x_256[::-1]
757          pal_256 = cmap(x_256)
758          cmap = mpl.colors.ListedColormap(pal_256, "seaborn_cubehelix")
759          return cmap
760      else:
761          return _ColorPalette(pal)
762

764  def _parse_cubehelix_args(argstr):
765      """Turn stringified cubehelix params into args/kwargs."""
766  
767      if argstr.startswith("ch:"):
768          argstr = argstr[3:]
769  
770      if argstr.endswith("_r"):
771          reverse = True
772          argstr = argstr[:-2]
773      else:
774          reverse = False
775  
776      if not argstr:
777          return [], {"reverse": reverse}
778  
779      all_args = argstr.split(",")
780  
781      args = [float(a.strip(" ")) for a in all_args if "=" not in a]
782  
783      kwargs = [a.split("=") for a in all_args if "=" in a]
784      kwargs = {k.strip(" "): float(v.strip(" ")) for k, v in kwargs}
785  
786      kwarg_map = dict(
787          s="start", r="rot", g="gamma",
788          h="hue", l="light", d="dark",  # noqa: E741
789      )
790  
791      kwargs = {kwarg_map.get(k, k): v for k, v in kwargs.items()}
792  
793      if reverse:
794          kwargs["reverse"] = True
795  
796      return args, kwargs

799  def set_color_codes(palette="deep"):
800      """Change how matplotlib color shorthands are interpreted. 
801   
802      Calling this will change how shorthand codes like "b" or "g" 
803      are interpreted by matplotlib in subsequent plots. 
804   
805      Parameters 
806      ---------- 
807      palette : {deep, muted, pastel, dark, bright, colorblind} 
808          Named seaborn palette to use as the source of colors. 
809   
810      See Also 
811      -------- 
812      set : Color codes can be set through the high-level seaborn style 
813            manager. 
814      set_palette : Color codes can also be set through the function that 
815                    sets the matplotlib color cycle. 
816   
817      """
818      if palette == "reset":
819          colors = [
820              (0., 0., 1.),
821              (0., .5, 0.),
822              (1., 0., 0.),
823              (.75, 0., .75),
824              (.75, .75, 0.),
825              (0., .75, .75),
826              (0., 0., 0.)
827          ]
828      elif not isinstance(palette, str):
829          err = "set_color_codes requires a named seaborn palette"
830          raise TypeError(err)
831      elif palette in SEABORN_PALETTES:
832          if not palette.endswith("6"):
833              palette = palette + "6"
834          colors = SEABORN_PALETTES[palette] + [(.1, .1, .1)]
835      else:
836          err = f"Cannot set colors with palette '{palette}'"
837          raise ValueError(err)
838  
839      for code, color in zip("bgrmyck", colors):
840          rgb = mpl.colors.colorConverter.to_rgb(color)
841          mpl.colors.colorConverter.colors[code] = rgb
