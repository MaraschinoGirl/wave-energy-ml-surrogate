59       def test_custom_builder_class(self):
60           # Verify that you can pass in a custom Builder class and
61           # it'll be instantiated with the appropriate keyword arguments.
62           class Mock(object):
63               def __init__(self, **kwargs):
64                   self.called_with = kwargs
65                   self.is_xml = True
66                   self.store_line_numbers = False
67                   self.cdata_list_attributes = []
68                   self.preserve_whitespace_tags = []
69                   self.string_containers = {}
70                   self.attribute_dict_class = XMLAttributeDict
71                   self.attribute_value_list_class = AttributeValueList
72   
73               def initialize_soup(self, soup):
74                   pass
75   
76               def feed(self, markup):
77                   self.fed = markup
78   
79               def reset(self):
80                   pass
81   
82               def ignore(self, ignore):
83                   pass
84   
85               set_up_substitutions = can_be_empty_element = ignore
86   
87               def prepare_markup(self, *args, **kwargs):
88                   yield (
89                       "prepared markup",
90                       "original encoding",
91                       "declared encoding",
92                       "contains replacement characters",
93                   )
94   
95           kwargs = dict(
96               var="value",
97               # This is a deprecated BS3-era keyword argument, which
98               # will be stripped out.
99               convertEntities=True,
100          )
101          with warnings.catch_warnings(record=True):
102              soup = BeautifulSoup("", builder=Mock, **kwargs)
103          assert isinstance(soup.builder, Mock)
104          assert dict(var="value") == soup.builder.called_with
105          assert "prepared markup" == soup.builder.fed
106  
107          # You can also instantiate the TreeBuilder yourself. In this
108          # case, that specific object is used and any keyword arguments
109          # to the BeautifulSoup constructor are ignored.
110          builder = Mock(**kwargs)
111          with warnings.catch_warnings(record=True) as w:
112              soup = BeautifulSoup(
113                  "",
114                  builder=builder,
115                  ignored_value=True,
116              )
117          msg = str(w[0].message)
118          assert msg.startswith(
119              "Keyword arguments to the BeautifulSoup constructor will be ignored."
120          )
121          assert builder == soup.builder
122          assert kwargs == builder.called_with

145      def test_cdata_list_attributes(self):
146          # Most attribute values are represented as scalars, but the
147          # HTML standard says that some attributes, like 'class' have
148          # space-separated lists as values.
149          markup = '<a id=" an id " class=" a class "></a>'
150          soup = self.soup(markup)
151  
152          # Note that the spaces are stripped for 'class' but not for 'id'.
153          a = soup.a
154          assert " an id " == a["id"]
155          assert ["a", "class"] == a["class"]
156  
157          # TreeBuilder takes an argument called 'multi_valued_attributes'  which lets
158          # you customize or disable this. As always, you can customize the TreeBuilder
159          # by passing in a keyword argument to the BeautifulSoup constructor.
160          soup = self.soup(markup, builder=default_builder, multi_valued_attributes=None)
161          assert " a class " == soup.a["class"]
162  
163          # Here are two ways of saying that `id` is a multi-valued
164          # attribute in this context, but 'class' is not.
165          for switcheroo in ({"*": "id"}, {"a": "id"}):
166              with warnings.catch_warnings(record=True):
167                  # This will create a warning about not explicitly
168                  # specifying a parser, but we'll ignore it.
169                  soup = self.soup(
170                      markup, builder=None, multi_valued_attributes=switcheroo
171                  )
172              a = soup.a
173              assert ["an", "id"] == a["id"]
174              assert " a class " == a["class"]

292  class TestWarnings(SoupTest):
293      # Note that some of the tests in this class create BeautifulSoup
294      # objects directly rather than using self.soup(). That's
295      # because SoupTest.soup is defined in a different file,
296      # which will throw off the assertion in _assert_warning
297      # that the code that triggered the warning is in the same
298      # file as the test.
299  
300      def _assert_warning(
301          self, warnings: Iterable[warnings.WarningMessage], cls: Type[Warning]
302      ) -> warnings.WarningMessage:
303          for w in warnings:
304              if isinstance(w.message, cls):
305                  assert w.filename == __file__
306                  return w
307          raise Exception("%s warning not found in %r" % (cls, warnings))
308  
309      def _assert_no_parser_specified(self, w: Iterable[warnings.WarningMessage]) -> None:
310          warning = self._assert_warning(w, GuessedAtParserWarning)
311          message = str(warning.message)
312          assert message.startswith(GuessedAtParserWarning.MESSAGE[:60])
313  
314      def test_warning_if_no_parser_specified(self):
315          with warnings.catch_warnings(record=True) as w:
316              BeautifulSoup("<a><b></b></a>")
317          self._assert_no_parser_specified(w)

549  class TestEncodingConversion(SoupTest):
550      # Test Beautiful Soup's ability to decode and encode from various
551      # encodings.
552  
553      def setup_method(self):
554          self.unicode_data = '<html><head><meta charset="utf-8"/></head><body><foo>Sacr\N{LATIN SMALL LETTER E WITH ACUTE} bleu!</foo></body></html>'
555          self.utf8_data = self.unicode_data.encode("utf-8")
556          # Just so you know what it looks like.
557          assert (
558              self.utf8_data
559              == b'<html><head><meta charset="utf-8"/></head><body><foo>Sacr\xc3\xa9 bleu!</foo></body></html>'
560          )
561  
562      def test_ascii_in_unicode_out(self):
563          # ASCII input is converted to Unicode. The original_encoding
564          # attribute is set to 'utf-8', a superset of ASCII.
565          chardet = dammit._chardet_dammit
566          logging.disable(logging.WARNING)
567          try:
568  
569              def noop(str):
570                  return None
571  
572              # Disable chardet, which will realize that the ASCII is ASCII.
573              dammit._chardet_dammit = noop
574              ascii = b"<foo>a</foo>"
575              soup_from_ascii = self.soup(ascii)
576              unicode_output = soup_from_ascii.decode()
577              assert isinstance(unicode_output, str)
578              assert unicode_output == self.document_for(ascii.decode())
579              assert soup_from_ascii.original_encoding.lower() == "utf-8"
580          finally:
581              logging.disable(logging.NOTSET)
582              dammit._chardet_dammit = chardet
583  
584      def test_unicode_in_unicode_out(self):
585          # Unicode input is left alone. The original_encoding attribute
586          # is not set.
587          soup_from_unicode = self.soup(self.unicode_data)
588          assert soup_from_unicode.decode() == self.unicode_data
589          assert soup_from_unicode.foo.string == "Sacr\xe9 bleu!"
590          assert soup_from_unicode.original_encoding is None
591  
592      def test_utf8_in_unicode_out(self):
593          # UTF-8 input is converted to Unicode. The original_encoding
594          # attribute is set.
595          soup_from_utf8 = self.soup(self.utf8_data)
596          assert soup_from_utf8.decode() == self.unicode_data
597          assert soup_from_utf8.foo.string == "Sacr\xe9 bleu!"
598  
599      def test_utf8_out(self):
600          # The internal data structures can be encoded as UTF-8.
601          soup_from_unicode = self.soup(self.unicode_data)
602          assert soup_from_unicode.encode("utf-8") == self.utf8_data
603
