8    class Adam(torch_parallel_optimizer.TorchParallelOptimizer, optimizers.Adam):
9        def _parallel_update_step(
10           self,
11           grads,
12           variables,
13           learning_rate,
14       ):
15           keras_variables = variables
16           variables = [v.value for v in variables]
17   
18           dtype = variables[0].dtype
19           lr = ops.cast(learning_rate, dtype)
20           local_step = ops.cast(self.iterations + 1, dtype)
21   
22           beta_1_power = ops.power(ops.cast(self.beta_1, dtype), local_step)
23           beta_2_power = ops.power(ops.cast(self.beta_2, dtype), local_step)
24           alpha = lr * ops.sqrt(1 - beta_2_power) / (1 - beta_1_power)
25   
26           m_list = [
27               self._momentums[self._get_variable_index(variable)].value
28               for variable in keras_variables
29           ]
30           v_list = [
31               self._velocities[self._get_variable_index(variable)].value
32               for variable in keras_variables
33           ]
34   
35           torch._foreach_mul_(m_list, self.beta_1)
36           torch._foreach_add_(m_list, grads, alpha=1 - self.beta_1)
37   
38           torch._foreach_mul_(v_list, self.beta_2)
39           torch._foreach_add_(
40               v_list, torch._foreach_mul(grads, grads), alpha=1 - self.beta_2
41           )
42   
43           if self.amsgrad:
44               v_hat_list = [
45                   self._velocity_hats[self._get_variable_index(variable)].value
46                   for variable in keras_variables
47               ]
48               torch._foreach_maximum_(v_hat_list, v_list)
49               v_list = v_hat_list
50   
51           torch._foreach_add_(
52               variables,
53               torch._foreach_div(
54                   torch._foreach_mul(m_list, alpha),
55                   torch._foreach_add(torch._foreach_sqrt(v_list), self.epsilon),
56               ),
57               alpha=-1,
58           )
