7    /* 
8     * Convert matrix to sparse representation suitable for liblinear. x is 
9     * expected to be an array of length n_samples*n_features. 
10    * 
11    * Whether the matrix is densely or sparsely populated, the fastest way to 
12    * convert it to liblinear's sparse format is to calculate the amount of memory 
13    * needed and allocate a single big block. 
14    * 
15    * Special care must be taken with indices, since liblinear indices start at 1 
16    * and not at 0. 
17    * 
18    * If bias is > 0, we append an item at the end. 
19    */
20   static struct feature_node **dense_to_sparse(char *x, int double_precision,
21           int n_samples, int n_features, int n_nonzero, double bias)
22   {
23       float *x32 = (float *)x;
24       double *x64 = (double *)x;
25       struct feature_node **sparse;
26       int i, j;                           /* number of nonzero elements in row i */
27       struct feature_node *T;             /* pointer to the top of the stack */
28       int have_bias = (bias > 0);
29   
30       sparse = malloc (n_samples * sizeof(struct feature_node *));
31       if (sparse == NULL)
32           return NULL;
33   
34       n_nonzero += (have_bias+1) * n_samples;
35       T = malloc (n_nonzero * sizeof(struct feature_node));
36       if (T == NULL) {
37           free(sparse);
38           return NULL;
39       }
40   
41       for (i=0; i<n_samples; ++i) {
42           sparse[i] = T;
43   
44           for (j=1; j<=n_features; ++j) {
45               if (double_precision) {
46                   if (*x64 != 0) {
47                       T->value = *x64;
48                       T->index = j;
49                       ++ T;
50                   }
51                   ++ x64; /* go to next element */
52               } else {
53                   if (*x32 != 0) {
54                       T->value = *x32;
55                       T->index = j;
56                       ++ T;
57                   }
58                   ++ x32; /* go to next element */
59               }
60           }
61   
62           /* set bias element */
63           if (have_bias) {
64                   T->value = bias;
65                   T->index = j;
66                   ++ T;
67               }
68   
69           /* set sentinel */
70           T->index = -1;
71           ++ T;
72       }
73   
74       return sparse;
75   }

78   /* 
79    * Convert scipy.sparse.csr to liblinear's sparse data structure 
80    */
81   static struct feature_node **csr_to_sparse(char *x, int double_precision,
82           int *indices, int *indptr, int n_samples, int n_features, int n_nonzero,
83           double bias)
84   {
85       float *x32 = (float *)x;
86       double *x64 = (double *)x;
87       struct feature_node **sparse;
88       int i, j=0, k=0, n;
89       struct feature_node *T;
90       int have_bias = (bias > 0);
91   
92       sparse = malloc (n_samples * sizeof(struct feature_node *));
93       if (sparse == NULL)
94           return NULL;
95   
96       n_nonzero += (have_bias+1) * n_samples;
97       T = malloc (n_nonzero * sizeof(struct feature_node));
98       if (T == NULL) {
99           free(sparse);
100          return NULL;
101      }
102  
103      for (i=0; i<n_samples; ++i) {
104          sparse[i] = T;
105          n = indptr[i+1] - indptr[i]; /* count elements in row i */
106  
107          for (j=0; j<n; ++j) {
108              T->value = double_precision ? x64[k] : x32[k];
109              T->index = indices[k] + 1; /* liblinear uses 1-based indexing */
110              ++T;
111              ++k;
112          }
113  
114          if (have_bias) {
115              T->value = bias;
116              T->index = n_features + 1;
117              ++T;
118              ++j;
119          }
120  
121          /* set sentinel */
122          T->index = -1;
123          ++T;
124      }
125  
126      return sparse;
127  }

129  struct problem * set_problem(char *X, int double_precision_X, int n_samples,
130          int n_features, int n_nonzero, double bias, char* sample_weight,
131          char *Y)
132  {
133      struct problem *problem;
134      /* not performant but simple */
135      problem = malloc(sizeof(struct problem));
136      if (problem == NULL) return NULL;
137      problem->l = n_samples;
138      problem->n = n_features + (bias > 0);
139      problem->y = (double *) Y;
140      problem->W = (double *) sample_weight;
141      problem->x = dense_to_sparse(X, double_precision_X, n_samples, n_features,
142                          n_nonzero, bias);
143      problem->bias = bias;
144  
145      if (problem->x == NULL) {
146          free(problem);
147          return NULL;
148      }
149  
150      return problem;
151  }

153  struct problem * csr_set_problem (char *X, int double_precision_X,
154          char *indices, char *indptr, int n_samples, int n_features,
155          int n_nonzero, double bias, char *sample_weight, char *Y)
156  {
157      struct problem *problem;
158      problem = malloc (sizeof (struct problem));
159      if (problem == NULL) return NULL;
160      problem->l = n_samples;
161      problem->n = n_features + (bias > 0);
162      problem->y = (double *) Y;
163      problem->W = (double *) sample_weight;
164      problem->x = csr_to_sparse(X, double_precision_X, (int *) indices,
165                          (int *) indptr, n_samples, n_features, n_nonzero, bias);
166      problem->bias = bias;
167  
168      if (problem->x == NULL) {
169          free(problem);
170          return NULL;
171      }
172  
173      return problem;
174  }

177  /* Create a parameter struct with and return it */
178  struct parameter *set_parameter(int solver_type, double eps, double C,
179                                  Py_ssize_t nr_weight, char *weight_label,
180                                  char *weight, int max_iter, unsigned seed,
181                                  double epsilon)
182  {
183      struct parameter *param = malloc(sizeof(struct parameter));
184      if (param == NULL)
185          return NULL;
186  
187      set_seed(seed);
188      param->solver_type = solver_type;
189      param->eps = eps;
190      param->C = C;
191      param->p = epsilon;  // epsilon for epsilon-SVR
192      param->nr_weight = (int) nr_weight;
193      param->weight_label = (int *) weight_label;
194      param->weight = (double *) weight;
195      param->max_iter = max_iter;
196      return param;
197  }
