9    from __future__ import annotations
10   
11   from pathlib import Path
12   
13   import tornado.web
14   
15   
16   class Terminal(tornado.web.UIModule):
17       """A terminal UI module."""
18   
19       def render(self, ws_url: str, cols: int = 80, rows: int = 25) -> str:
20           """Render the module."""
21           return (
22               '<div class="terminado-container" '
23               f'data-ws-url="{ws_url}" '
24               f'data-rows="{rows}" data-cols="{cols}"/>'
25           )
26   
27       def javascript_files(self) -> list[str]:
28           """Get the list of JS files to include."""
29           # TODO: Can we calculate these dynamically?
30           return ["/xstatic/termjs/term.js", "/static/terminado.js"]
31   
32       def embedded_javascript(self) -> str:
33           """Get the embedded JS content as a string."""
34           file = Path(__file__).parent / "uimod_embed.js"
35           with file.open() as f:
36               return f.read()
