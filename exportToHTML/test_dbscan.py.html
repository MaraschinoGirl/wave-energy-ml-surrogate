23   def test_dbscan_similarity():
24       # Tests the DBSCAN algorithm with a similarity array.
25       # Parameters chosen specifically for this task.
26       eps = 0.15
27       min_samples = 10
28       # Compute similarities
29       D = distance.squareform(distance.pdist(X))
30       D /= np.max(D)
31       # Compute DBSCAN
32       core_samples, labels = dbscan(
33           D, metric="precomputed", eps=eps, min_samples=min_samples
34       )
35       # number of clusters, ignoring noise if present
36       n_clusters_1 = len(set(labels)) - (1 if -1 in labels else 0)
37   
38       assert n_clusters_1 == n_clusters
39   
40       db = DBSCAN(metric="precomputed", eps=eps, min_samples=min_samples)
41       labels = db.fit(D).labels_
42   
43       n_clusters_2 = len(set(labels)) - int(-1 in labels)
44       assert n_clusters_2 == n_clusters

79       D = pairwise_distances(X)
80       nn = NearestNeighbors(radius=0.9).fit(X)

85       core_sparse, labels_sparse = dbscan(
86           D_sparse, eps=0.8, min_samples=10, metric="precomputed"

89       assert_array_equal(core_dense, core_sparse)
90       assert_array_equal(labels_dense, labels_sparse)

93   def test_dbscan_sparse_precomputed_different_eps():
94       # test that precomputed neighbors graph is filtered if computed with
95       # a radius larger than DBSCAN's eps.
96       lower_eps = 0.2
97       nn = NearestNeighbors(radius=lower_eps).fit(X)
98       D_sparse = nn.radius_neighbors_graph(X, mode="distance")
99       dbscan_lower = dbscan(D_sparse, eps=lower_eps, metric="precomputed")
100  
101      higher_eps = lower_eps + 0.7
102      nn = NearestNeighbors(radius=higher_eps).fit(X)
103      D_sparse = nn.radius_neighbors_graph(X, mode="distance")
104      dbscan_higher = dbscan(D_sparse, eps=lower_eps, metric="precomputed")
105  
106      assert_array_equal(dbscan_lower[0], dbscan_higher[0])
107      assert_array_equal(dbscan_lower[1], dbscan_higher[1])

125  @pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
126  def test_dbscan_input_not_modified_precomputed_sparse_nodiag(csr_container):
127      """Check that we don't modify in-place the pre-computed sparse matrix. 
128   
129      Non-regression test for: 
130      https://github.com/scikit-learn/scikit-learn/issues/27508 
131      """
132      X = np.random.RandomState(0).rand(10, 10)
133      # Add zeros on the diagonal that will be implicit when creating
134      # the sparse matrix. If `X` is modified in-place, the zeros from
135      # the diagonal will be made explicit.
136      np.fill_diagonal(X, 0)
137      X = csr_container(X)
138      assert all(row != col for row, col in zip(*X.nonzero()))
139      X_copy = X.copy()
140      dbscan(X, metric="precomputed")
141      # Make sure that we did not modify `X` in-place even by creating
142      # explicit 0s values.
143      assert X.nnz == X_copy.nnz
144      assert_array_equal(X.toarray(), X_copy.toarray())

185  def test_dbscan_metric_params():
186      # Tests that DBSCAN works with the metrics_params argument.
187      eps = 0.8
188      min_samples = 10
189      p = 1
190  
191      # Compute DBSCAN with metric_params arg
192  
193      with warnings.catch_warnings(record=True) as warns:
194          db = DBSCAN(
195              metric="minkowski",
196              metric_params={"p": p},
197              eps=eps,
198              p=None,
199              min_samples=min_samples,
200              algorithm="ball_tree",
201          ).fit(X)
202      assert not warns, warns[0].message
203      core_sample_1, labels_1 = db.core_sample_indices_, db.labels_
204  
205      # Test that sample labels are the same as passing Minkowski 'p' directly
206      db = DBSCAN(
207          metric="minkowski", eps=eps, min_samples=min_samples, algorithm="ball_tree", p=p
208      ).fit(X)
209      core_sample_2, labels_2 = db.core_sample_indices_, db.labels_
210  
211      assert_array_equal(core_sample_1, core_sample_2)
212      assert_array_equal(labels_1, labels_2)
213  
214      # Minkowski with p=1 should be equivalent to Manhattan distance
215      db = DBSCAN(
216          metric="manhattan", eps=eps, min_samples=min_samples, algorithm="ball_tree"
217      ).fit(X)
218      core_sample_3, labels_3 = db.core_sample_indices_, db.labels_
219  
220      assert_array_equal(core_sample_1, core_sample_3)
221      assert_array_equal(labels_1, labels_3)
222  
223      with pytest.warns(
224          SyntaxWarning,
225          match=(
226              "Parameter p is found in metric_params. "
227              "The corresponding parameter from __init__ "
228              "is ignored."
229          ),
230      ):
231          # Test that checks p is ignored in favor of metric_params={'p': <val>}
232          db = DBSCAN(
233              metric="minkowski",
234              metric_params={"p": p},
235              eps=eps,
236              p=p + 1,
237              min_samples=min_samples,
238              algorithm="ball_tree",
239          ).fit(X)
240          core_sample_4, labels_4 = db.core_sample_indices_, db.labels_
241  
242      assert_array_equal(core_sample_1, core_sample_4)
243      assert_array_equal(labels_1, labels_4)

308  def test_weighted_dbscan(global_random_seed):
309      # ensure sample_weight is validated
310      with pytest.raises(ValueError):
311          dbscan([[0], [1]], sample_weight=[2])
312      with pytest.raises(ValueError):
313          dbscan([[0], [1]], sample_weight=[2, 3, 4])
314  
315      # ensure sample_weight has an effect
316      assert_array_equal([], dbscan([[0], [1]], sample_weight=None, min_samples=6)[0])
317      assert_array_equal([], dbscan([[0], [1]], sample_weight=[5, 5], min_samples=6)[0])
318      assert_array_equal([0], dbscan([[0], [1]], sample_weight=[6, 5], min_samples=6)[0])
319      assert_array_equal(
320          [0, 1], dbscan([[0], [1]], sample_weight=[6, 6], min_samples=6)[0]
321      )
322  
323      # points within eps of each other:
324      assert_array_equal(
325          [0, 1], dbscan([[0], [1]], eps=1.5, sample_weight=[5, 1], min_samples=6)[0]
326      )
327      # and effect of non-positive and non-integer sample_weight:
328      assert_array_equal(
329          [], dbscan([[0], [1]], sample_weight=[5, 0], eps=1.5, min_samples=6)[0]
330      )
331      assert_array_equal(
332          [0, 1], dbscan([[0], [1]], sample_weight=[5.9, 0.1], eps=1.5, min_samples=6)[0]
333      )
334      assert_array_equal(
335          [0, 1], dbscan([[0], [1]], sample_weight=[6, 0], eps=1.5, min_samples=6)[0]
336      )
337      assert_array_equal(
338          [], dbscan([[0], [1]], sample_weight=[6, -1], eps=1.5, min_samples=6)[0]
339      )
340  
341      # for non-negative sample_weight, cores should be identical to repetition
342      rng = np.random.RandomState(global_random_seed)
343      sample_weight = rng.randint(0, 5, X.shape[0])
344      core1, label1 = dbscan(X, sample_weight=sample_weight)
345      assert len(label1) == len(X)
346  
347      X_repeated = np.repeat(X, sample_weight, axis=0)
348      core_repeated, label_repeated = dbscan(X_repeated)
349      core_repeated_mask = np.zeros(X_repeated.shape[0], dtype=bool)
350      core_repeated_mask[core_repeated] = True
351      core_mask = np.zeros(X.shape[0], dtype=bool)
352      core_mask[core1] = True
353      assert_array_equal(np.repeat(core_mask, sample_weight), core_repeated_mask)
354  
355      # sample_weight should work with precomputed distance matrix
356      D = pairwise_distances(X)
357      core3, label3 = dbscan(D, sample_weight=sample_weight, metric="precomputed")
358      assert_array_equal(core1, core3)
359      assert_array_equal(label1, label3)
360  
361      # sample_weight should work with estimator
362      est = DBSCAN().fit(X, sample_weight=sample_weight)
363      core4 = est.core_sample_indices_
364      label4 = est.labels_
365      assert_array_equal(core1, core4)
366      assert_array_equal(label1, label4)
367  
368      est = DBSCAN()
369      label5 = est.fit_predict(X, sample_weight=sample_weight)
370      core5 = est.core_sample_indices_
371      assert_array_equal(core1, core5)
372      assert_array_equal(label1, label5)
373      assert_array_equal(label1, est.labels_)

376  @pytest.mark.parametrize("algorithm", ["brute", "kd_tree", "ball_tree"])
377  def test_dbscan_core_samples_toy(algorithm):
378      X = [[0], [2], [3], [4], [6], [8], [10]]
379      n_samples = len(X)
380  
381      # Degenerate case: every sample is a core sample, either with its own
382      # cluster or including other close core samples.
383      core_samples, labels = dbscan(X, algorithm=algorithm, eps=1, min_samples=1)
384      assert_array_equal(core_samples, np.arange(n_samples))
385      assert_array_equal(labels, [0, 1, 1, 1, 2, 3, 4])
386  
387      # With eps=1 and min_samples=2 only the 3 samples from the denser area
388      # are core samples. All other points are isolated and considered noise.
389      core_samples, labels = dbscan(X, algorithm=algorithm, eps=1, min_samples=2)
390      assert_array_equal(core_samples, [1, 2, 3])
391      assert_array_equal(labels, [-1, 0, 0, 0, -1, -1, -1])
392  
393      # Only the sample in the middle of the dense area is core. Its two
394      # neighbors are edge samples. Remaining samples are noise.
395      core_samples, labels = dbscan(X, algorithm=algorithm, eps=1, min_samples=3)
396      assert_array_equal(core_samples, [2])
397      assert_array_equal(labels, [-1, 0, 0, 0, -1, -1, -1])
398  
399      # It's no longer possible to extract core samples with eps=1:
400      # everything is noise.
401      core_samples, labels = dbscan(X, algorithm=algorithm, eps=1, min_samples=4)
402      assert_array_equal(core_samples, [])
403      assert_array_equal(labels, np.full(n_samples, -1.0))
