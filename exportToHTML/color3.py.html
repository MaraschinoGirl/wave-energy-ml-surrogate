8    class RGBA(collections.namedtuple('RGBA', ['red', 'green', 'blue', 'alpha'])):
9        """An RGBA color. 
10    
11       A tuple of four floats in the 0..1 range: ``(red, green, blue, alpha)``. 
12    
13       .. attribute:: red 
14    
15           Convenience access to the red channel. Same as ``rgba[0]``. 
16    
17       .. attribute:: green 
18    
19           Convenience access to the green channel. Same as ``rgba[1]``. 
20    
21       .. attribute:: blue 
22    
23           Convenience access to the blue channel. Same as ``rgba[2]``. 
24    
25       .. attribute:: alpha 
26    
27           Convenience access to the alpha channel. Same as ``rgba[3]``. 
28    
29       """

32   def parse_color(input):
33       """Parse a color value as defined in CSS Color Level 3. 
34    
35       https://www.w3.org/TR/css-color-3/ 
36    
37       :type input: :obj:`str` or :term:`iterable` 
38       :param input: A string or an iterable of :term:`component values`. 
39       :returns: 
40           * :obj:`None` if the input is not a valid color value. 
41             (No exception is raised.) 
42           * The string ``'currentColor'`` for the ``currentColor`` keyword 
43           * Or a :class:`RGBA` object for every other values 
44             (including keywords, HSL and HSLA.) 
45             The alpha channel is clipped to [0, 1] 
46             but red, green, or blue can be out of range 
47             (eg. ``rgb(-10%, 120%, 0%)`` is represented as 
48             ``(-0.1, 1.2, 0, 1)``.) 
49    
50       """
51       if isinstance(input, str):
52           token = parse_one_component_value(input, skip_comments=True)
53       else:
54           token = input
55       if token.type == 'ident':
56           return _COLOR_KEYWORDS.get(token.lower_value)
57       elif token.type == 'hash':
58           for multiplier, regexp in _HASH_REGEXPS:
59               match = regexp(token.value)
60               if match:
61                   channels = [
62                       int(group * multiplier, 16) / 255
63                       for group in match.groups()]
64                   if len(channels) == 3:
65                       channels.append(1.)
66                   return RGBA(*channels)
67       elif token.type == 'function':
68           args = _parse_comma_separated(token.arguments)
69           if args:
70               name = token.lower_name
71               if name == 'rgb':
72                   return _parse_rgb(args, alpha=1.)
73               elif name == 'rgba':
74                   alpha = _parse_alpha(args[3:])
75                   if alpha is not None:
76                       return _parse_rgb(args[:3], alpha)
77               elif name == 'hsl':
78                   return _parse_hsl(args, alpha=1.)
79               elif name == 'hsla':
80                   alpha = _parse_alpha(args[3:])
81                   if alpha is not None:
82                       return _parse_hsl(args[:3], alpha)

85   def _parse_alpha(args):
86       """Parse a list of one alpha value. 
87    
88       If args is a list of a single INTEGER or NUMBER token, 
89       return its value clipped to the 0..1 range. Otherwise, return None. 
90    
91       """
92       if len(args) == 1 and args[0].type == 'number':
93           return min(1, max(0, args[0].value))

96   def _parse_rgb(args, alpha):
97       """Parse a list of RGB channels. 
98    
99       If args is a list of 3 INTEGER tokens or 3 PERCENTAGE tokens, return RGB 
100      values as a tuple of 3 floats in 0..1. Otherwise, return None. 
101   
102      """
103      types = [arg.type for arg in args]
104      if (types == ['number', 'number', 'number'] and
105              all(a.is_integer for a in args)):
106          r, g, b = [arg.int_value / 255 for arg in args[:3]]
107          return RGBA(r, g, b, alpha)
108      elif types == ['percentage', 'percentage', 'percentage']:
109          r, g, b = [arg.value / 100 for arg in args[:3]]
110          return RGBA(r, g, b, alpha)
111

113  def _parse_hsl(args, alpha):
114      """Parse a list of HSL channels. 
115   
116      If args is a list of 1 NUMBER token and 2 PERCENTAGE tokens, return RGB 
117      values as a tuple of 3 floats in 0..1. Otherwise, return None. 
118   
119      """
120      types = [arg.type for arg in args]
121      if types == ['number', 'percentage', 'percentage']:
122          r, g, b = hls_to_rgb(
123              args[0].value / 360, args[2].value / 100, args[1].value / 100)
124          return RGBA(r, g, b, alpha)

127  def _parse_comma_separated(tokens):
128      """Parse a list of tokens (typically the content of a function token) 
129      as arguments made of a single token each, separated by mandatory commas, 
130      with optional white space around each argument. 
131   
132      return the argument list without commas or white space; 
133      or None if the function token content do not match the description above. 
134   
135      """
136      tokens = [token for token in tokens
137                if token.type not in ('whitespace', 'comment')]
138      if not tokens:
139          return []
140      if len(tokens) % 2 == 1 and all(token == ',' for token in tokens[1::2]):
141          return tokens[::2]

144  _HASH_REGEXPS = (
145      (2, re.compile('^{}$'.format(4 * '([\\da-f])'), re.I).match),
146      (1, re.compile('^{}$'.format(4 * '([\\da-f]{2})'), re.I).match),
147      (2, re.compile('^{}$'.format(3 * '([\\da-f])'), re.I).match),
148      (1, re.compile('^{}$'.format(3 * '([\\da-f]{2})'), re.I).match),
