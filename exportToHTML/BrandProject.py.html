1    # BrandProject.py
2    #
3    # Brand a VSS project with a "build number", then optionally
4    # stamp DLL/EXE files with version information.
5    
6    import getopt
7    import os
8    import sys
9    
10   import bulkstamp
11   import vssutil
12   import win32api
13   
14   
15   def BrandProject(
16       vssProjectName,
17       descFile,
18       stampPath,
19       filesToSubstitute,
20       buildDesc=None,
21       auto=0,
22       bRebrand=0,
23   ):
24       # vssProjectName -- The name of the VSS project to brand.
25       # descFile -- A test file containing descriptions of the files in the release.
26       # stampPath -- The full path to where the files referenced in descFile can be found.
27       path = win32api.GetFullPathName(stampPath)
28   
29       build = vssutil.MakeNewBuildNo(vssProjectName, buildDesc, auto, bRebrand)
30       if build is None:
31           print("Cancelled")
32           return
33   
34       bulkstamp.scan(build, stampPath, descFile)
35       for infile, outfile in filesToSubstitute:
36           vssutil.SubstituteVSSInFile(vssProjectName, infile, outfile)
37       return 1
38   
39   
40   def usage(msg):
41       print(msg)
42       print(
43           f"""\ 
44   {os.path.basename(sys.argv[0])} Usage:
45   {os.path.basename(sys.argv[0])} [options] vssProject descFile stampPath
46   
47   Automatically brand a VSS project with an automatically incremented
48   build number, and stamp DLL/EXE files with the build number.
49   
50   Checks that no files are checked out in the project, and finds the last
51   build number, and suggests the next number.
52   
53   Options:
54   -a     - Auto increment the build number, and brand (otherwise prompt
55            for the build number after looking for the previous)
56   -r     - Restamp the files with the existing build number.
57   -d     - A description for the VSS Label.
58   -f infile=outfile - Substitute special VSS labels in the specified text
59                       file with the text extracted from VSS.
60   """
61       )
62       sys.exit(1)
63   
64   
65   if __name__ == "__main__":
66       import getopt
67   
68       try:
69           opts, args = getopt.getopt(sys.argv[1:], "af:d:r")
70       except getopt.GetoptError as msg:
71           usage(msg)
72       bAuto = bRebrand = 0
73       stampFiles = []
74       desc = None
75       for opt, val in opts:
76           if opt == "-a":
77               bAuto = 1
78           if opt == "-f":
79               infile, outfile = val.split("=", 2)
80               stampFiles.append((infile, outfile))
81           if opt == "-d":
82               desc = val
83           if opt == "-r":
84               bRebrand = 1
85       if len(args) < 3:
86           usage("You must specify the required arguments")
87       vssProjectName = "$\\" + args[0]
88       descFile = args[1]
89       path = args[2]
90       try:
91           os.stat(descFile)
92       except OSError:
93           usage("The description file '%s' can not be found" % (descFile))
94       if not os.path.isdir(path):
95           usage("The path to the files to stamp '%s' does not exist" % (path))
96   
97       BrandProject(vssProjectName, descFile, path, stampFiles, desc, bAuto, bRebrand)
