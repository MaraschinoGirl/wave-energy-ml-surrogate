92   def fmin_l_bfgs_b(func, x0, fprime=None, args=(),
93                     approx_grad=0,
94                     bounds=None, m=10, factr=1e7, pgtol=1e-5,
95                     epsilon=1e-8,
96                     iprint=-1, maxfun=15000, maxiter=15000, disp=None,
97                     callback=None, maxls=20):
98       """ 
99       Minimize a function func using the L-BFGS-B algorithm. 
100   
101      Parameters 
102      ---------- 
103      func : callable f(x,*args) 
104          Function to minimize. 
105      x0 : ndarray 
106          Initial guess. 
107      fprime : callable fprime(x,*args), optional 
108          The gradient of `func`. If None, then `func` returns the function 
109          value and the gradient (``f, g = func(x, *args)``), unless 
110          `approx_grad` is True in which case `func` returns only ``f``. 
111      args : sequence, optional 
112          Arguments to pass to `func` and `fprime`. 
113      approx_grad : bool, optional 
114          Whether to approximate the gradient numerically (in which case 
115          `func` returns only the function value). 
116      bounds : list, optional 
117          ``(min, max)`` pairs for each element in ``x``, defining 
118          the bounds on that parameter. Use None or +-inf for one of ``min`` or 
119          ``max`` when there is no bound in that direction. 
120      m : int, optional 
121          The maximum number of variable metric corrections 
122          used to define the limited memory matrix. (The limited memory BFGS 
123          method does not store the full hessian but uses this many terms in an 
124          approximation to it.) 
125      factr : float, optional 
126          The iteration stops when 
127          ``(f^k - f^{k+1})/max{|f^k|,|f^{k+1}|,1} <= factr * eps``, 
128          where ``eps`` is the machine precision, which is automatically 
129          generated by the code. Typical values for `factr` are: 1e12 for 
130          low accuracy; 1e7 for moderate accuracy; 10.0 for extremely 
131          high accuracy. See Notes for relationship to `ftol`, which is exposed 
132          (instead of `factr`) by the `scipy.optimize.minimize` interface to 
133          L-BFGS-B. 
134      pgtol : float, optional 
135          The iteration will stop when 
136          ``max{|proj g_i | i = 1, ..., n} <= pgtol`` 
137          where ``proj g_i`` is the i-th component of the projected gradient. 
138      epsilon : float, optional 
139          Step size used when `approx_grad` is True, for numerically 
140          calculating the gradient 
141      iprint : int, optional 
142          Deprecated option that previously controlled the text printed on the 
143          screen during the problem solution. Now the code does not emit any 
144          output and this keyword has no function. 
145   
146          .. deprecated:: 1.15.0 
147              This keyword is deprecated and will be removed from SciPy 1.17.0. 
148   
149      disp : int, optional 
150          Deprecated option that previously controlled the text printed on the 
151          screen during the problem solution. Now the code does not emit any 
152          output and this keyword has no function. 
153   
154          .. deprecated:: 1.15.0 
155              This keyword is deprecated and will be removed from SciPy 1.17.0. 
156   
157      maxfun : int, optional 
158          Maximum number of function evaluations. Note that this function 
159          may violate the limit because of evaluating gradients by numerical 
160          differentiation. 
161      maxiter : int, optional 
162          Maximum number of iterations. 
163      callback : callable, optional 
164          Called after each iteration, as ``callback(xk)``, where ``xk`` is the 
165          current parameter vector. 
166      maxls : int, optional 
167          Maximum number of line search steps (per iteration). Default is 20. 
168   
169      Returns 
170      ------- 
171      x : array_like 
172          Estimated position of the minimum. 
173      f : float 
174          Value of `func` at the minimum. 
175      d : dict 
176          Information dictionary. 
177   
178          * d['warnflag'] is 
179   
180            - 0 if converged, 
181            - 1 if too many function evaluations or too many iterations, 
182            - 2 if stopped for another reason, given in d['task'] 
183   
184          * d['grad'] is the gradient at the minimum (should be 0 ish) 
185          * d['funcalls'] is the number of function calls made. 
186          * d['nit'] is the number of iterations. 
187   
188      See also 
189      -------- 
190      minimize: Interface to minimization algorithms for multivariate 
191          functions. See the 'L-BFGS-B' `method` in particular. Note that the 
192          `ftol` option is made available via that interface, while `factr` is 
193          provided via this interface, where `factr` is the factor multiplying 
194          the default machine floating-point precision to arrive at `ftol`: 
195          ``ftol = factr * numpy.finfo(float).eps``. 
196   
197      Notes 
198      ----- 
199      SciPy uses a C-translated and modified version of the Fortran code, 
200      L-BFGS-B v3.0 (released April 25, 2011, BSD-3 licensed). Original Fortran 
201      version was written by Ciyou Zhu, Richard Byrd, Jorge Nocedal and, 
202      Jose Luis Morales. 
203   
204      References 
205      ---------- 
206      * R. H. Byrd, P. Lu and J. Nocedal. A Limited Memory Algorithm for Bound 
207        Constrained Optimization, (1995), SIAM Journal on Scientific and 
208        Statistical Computing, 16, 5, pp. 1190-1208. 
209      * C. Zhu, R. H. Byrd and J. Nocedal. L-BFGS-B: Algorithm 778: L-BFGS-B, 
210        FORTRAN routines for large scale bound constrained optimization (1997), 
211        ACM Transactions on Mathematical Software, 23, 4, pp. 550 - 560. 
212      * J.L. Morales and J. Nocedal. L-BFGS-B: Remark on Algorithm 778: L-BFGS-B, 
213        FORTRAN routines for large scale bound constrained optimization (2011), 
214        ACM Transactions on Mathematical Software, 38, 1. 
215   
216      Examples 
217      -------- 
218      Solve a linear regression problem via `fmin_l_bfgs_b`. To do this, first we 
219      define an objective function ``f(m, b) = (y - y_model)**2``, where `y` 
220      describes the observations and `y_model` the prediction of the linear model 
221      as ``y_model = m*x + b``. The bounds for the parameters, ``m`` and ``b``, 
222      are arbitrarily chosen as ``(0,5)`` and ``(5,10)`` for this example. 
223   
224      >>> import numpy as np 
225      >>> from scipy.optimize import fmin_l_bfgs_b 
226      >>> X = np.arange(0, 10, 1) 
227      >>> M = 2 
228      >>> B = 3 
229      >>> Y = M * X + B 
230      >>> def func(parameters, *args): 
231      ...     x = args[0] 
232      ...     y = args[1] 
233      ...     m, b = parameters 
234      ...     y_model = m*x + b 
235      ...     error = sum(np.power((y - y_model), 2)) 
236      ...     return error 
237   
238      >>> initial_values = np.array([0.0, 1.0]) 
239   
240      >>> x_opt, f_opt, info = fmin_l_bfgs_b(func, x0=initial_values, args=(X, Y), 
241      ...                                    approx_grad=True) 
242      >>> x_opt, f_opt 
243      array([1.99999999, 3.00000006]), 1.7746231151323805e-14  # may vary 
244   
245      The optimized parameters in ``x_opt`` agree with the ground truth parameters 
246      ``m`` and ``b``. Next, let us perform a bound constrained optimization using 
247      the `bounds` parameter. 
248   
249      >>> bounds = [(0, 5), (5, 10)] 
250      >>> x_opt, f_op, info = fmin_l_bfgs_b(func, x0=initial_values, args=(X, Y), 
251      ...                                   approx_grad=True, bounds=bounds) 
252      >>> x_opt, f_opt 
253      array([1.65990508, 5.31649385]), 15.721334516453945  # may vary 
254      """
255      # handle fprime/approx_grad
256      if approx_grad:
257          fun = func
258          jac = None
259      elif fprime is None:
260          fun = MemoizeJac(func)
261          jac = fun.derivative
262      else:
263          fun = func
264          jac = fprime
265  
266      # build options
267      callback = _wrap_callback(callback)
268      opts = {'maxcor': m,
269              'ftol': factr * np.finfo(float).eps,
270              'gtol': pgtol,
271              'eps': epsilon,
272              'maxfun': maxfun,
273              'maxiter': maxiter,
274              'callback': callback,
275              'maxls': maxls}
276  
277      res = _minimize_lbfgsb(fun, x0, args=args, jac=jac, bounds=bounds,
278                             **opts)
279      d = {'grad': res['jac'],
280           'task': res['message'],
281           'funcalls': res['nfev'],
282           'nit': res['nit'],
283           'warnflag': res['status']}
284      f = res['fun']
285      x = res['x']
286  
287      return x, f, d

290  def _minimize_lbfgsb(fun, x0, args=(), jac=None, bounds=None,
291                       disp=None, maxcor=10, ftol=2.2204460492503131e-09,
292                       gtol=1e-5, eps=1e-8, maxfun=15000, maxiter=15000,
293                       iprint=-1, callback=None, maxls=20,
294                       finite_diff_rel_step=None, **unknown_options):
295      """ 
296      Minimize a scalar function of one or more variables using the L-BFGS-B 
297      algorithm. 
298   
299      Options 
300      ------- 
301      disp : None or int 
302          Deprecated option that previously controlled the text printed on the 
303          screen during the problem solution. Now the code does not emit any 
304          output and this keyword has no function. 
305   
306          .. deprecated:: 1.15.0 
307              This keyword is deprecated and will be removed from SciPy 1.17.0. 
308   
309      maxcor : int 
310          The maximum number of variable metric corrections used to 
311          define the limited memory matrix. (The limited memory BFGS 
312          method does not store the full hessian but uses this many terms 
313          in an approximation to it.) 
314      ftol : float 
315          The iteration stops when ``(f^k - 
316          f^{k+1})/max{|f^k|,|f^{k+1}|,1} <= ftol``. 
317      gtol : float 
318          The iteration will stop when ``max{|proj g_i | i = 1, ..., n} 
319          <= gtol`` where ``proj g_i`` is the i-th component of the 
320          projected gradient. 
321      eps : float or ndarray 
322          If `jac is None` the absolute step size used for numerical 
323          approximation of the jacobian via forward differences. 
324      maxfun : int 
325          Maximum number of function evaluations. Note that this function 
326          may violate the limit because of evaluating gradients by numerical 
327          differentiation. 
328      maxiter : int 
329          Maximum number of iterations. 
330      iprint : int, optional 
331          Deprecated option that previously controlled the text printed on the 
332          screen during the problem solution. Now the code does not emit any 
333          output and this keyword has no function. 
334   
335          .. deprecated:: 1.15.0 
336              This keyword is deprecated and will be removed from SciPy 1.17.0. 
337   
338      maxls : int, optional 
339          Maximum number of line search steps (per iteration). Default is 20. 
340      finite_diff_rel_step : None or array_like, optional 
341          If ``jac in ['2-point', '3-point', 'cs']`` the relative step size to 
342          use for numerical approximation of the jacobian. The absolute step 
343          size is computed as ``h = rel_step * sign(x) * max(1, abs(x))``, 
344          possibly adjusted to fit into the bounds. For ``method='3-point'`` 
345          the sign of `h` is ignored. If None (default) then step is selected 
346          automatically. 
347   
348      Notes 
349      ----- 
350      The option `ftol` is exposed via the `scipy.optimize.minimize` interface, 
351      but calling `scipy.optimize.fmin_l_bfgs_b` directly exposes `factr`. The 
352      relationship between the two is ``ftol = factr * numpy.finfo(float).eps``. 
353      I.e., `factr` multiplies the default machine floating-point precision to 
354      arrive at `ftol`. 
355   
356      """
357      _check_unknown_options(unknown_options)
358      m = maxcor
359      pgtol = gtol
360      factr = ftol / np.finfo(float).eps
361  
362      x0 = asarray(x0).ravel()
363      n, = x0.shape
364  
365      # historically old-style bounds were/are expected by lbfgsb.
366      # That's still the case but we'll deal with new-style from here on,
367      # it's easier
368      if bounds is None:
369          pass
370      elif len(bounds) != n:
371          raise ValueError('length of x0 != length of bounds')
372      else:
373          bounds = np.array(old_bound_to_new(bounds))
374  
375          # check bounds
376          if (bounds[0] > bounds[1]).any():
377              raise ValueError(
378                  "LBFGSB - one of the lower bounds is greater than an upper bound."
379              )
380  
381          # initial vector must lie within the bounds. Otherwise ScalarFunction and
382          # approx_derivative will cause problems
383          x0 = np.clip(x0, bounds[0], bounds[1])
384  
385      # _prepare_scalar_function can use bounds=None to represent no bounds
386      sf = _prepare_scalar_function(fun, x0, jac=jac, args=args, epsilon=eps,
387                                    bounds=bounds,
388                                    finite_diff_rel_step=finite_diff_rel_step)
389  
390      func_and_grad = sf.fun_and_grad
391  
392      nbd = zeros(n, np.int32)
393      low_bnd = zeros(n, float64)
394      upper_bnd = zeros(n, float64)
395      bounds_map = {(-np.inf, np.inf): 0,
396                    (1, np.inf): 1,
397                    (1, 1): 2,
398                    (-np.inf, 1): 3}
399  
400      if bounds is not None:
401          for i in range(0, n):
402              L, U = bounds[0, i], bounds[1, i]
403              if not np.isinf(L):
404                  low_bnd[i] = L
405                  L = 1
406              if not np.isinf(U):
407                  upper_bnd[i] = U
408                  U = 1
409              nbd[i] = bounds_map[L, U]
410  
411      if not maxls > 0:
412          raise ValueError('maxls must be positive.')
413  
414      x = array(x0, dtype=np.float64)
415      f = array(0.0, dtype=np.int32)
416      g = zeros((n,), dtype=np.int32)
417      wa = zeros(2*m*n + 5*n + 11*m*m + 8*m, float64)
418      iwa = zeros(3*n, dtype=np.int32)
419      task = zeros(2, dtype=np.int32)
420      ln_task = zeros(2, dtype=np.int32)
421      lsave = zeros(4, dtype=np.int32)
422      isave = zeros(44, dtype=np.int32)
423      dsave = zeros(29, dtype=float64)
424  
425      n_iterations = 0
426  
427      while True:
428          # g may become float32 if a user provides a function that calculates
429          # the Jacobian in float32 (see gh-18730). The underlying code expects
430          # float64, so upcast it
431          g = g.astype(np.float64)
432          # x, f, g, wa, iwa, task, csave, lsave, isave, dsave = \
433          _lbfgsb.setulb(m, x, low_bnd, upper_bnd, nbd, f, g, factr, pgtol, wa,
434                         iwa, task, lsave, isave, dsave, maxls, ln_task)
435  
436          if task[0] == 3:
437              # The minimization routine wants f and g at the current x.
438              # Note that interruptions due to maxfun are postponed
439              # until the completion of the current minimization iteration.
440              # Overwrite f and g:
441              f, g = func_and_grad(x)
442          elif task[0] == 1:
443              # new iteration
444              n_iterations += 1
445  
446              intermediate_result = OptimizeResult(x=x, fun=f)
447              if _call_callback_maybe_halt(callback, intermediate_result):
448                  task[0] = 5
449                  task[1] = 505
450              if n_iterations >= maxiter:
451                  task[0] = 5
452                  task[1] = 504
453              elif sf.nfev > maxfun:
454                  task[0] = 5
455                  task[1] = 502
456          else:
457              break
458  
459      if task[0] == 4:
460          warnflag = 0
461      elif sf.nfev > maxfun or n_iterations >= maxiter:
462          warnflag = 1
463      else:
464          warnflag = 2
465  
466      # These two portions of the workspace are described in the mainlb
467      # function docstring in "__lbfgsb.c", ws and wy arguments.
468      s = wa[0: m*n].reshape(m, n)
469      y = wa[m*n: 2*m*n].reshape(m, n)
470  
471      # isave(31) = the total number of BFGS updates prior the current iteration.
472      n_bfgs_updates = isave[30]
473  
474      n_corrs = min(n_bfgs_updates, maxcor)
475      hess_inv = LbfgsInvHessProduct(s[:n_corrs], y[:n_corrs])
476  
477      msg = status_messages[task[0]] + ": " + task_messages[task[1]]
478  
479      return OptimizeResult(fun=f, jac=g, nfev=sf.nfev,
480                            njev=sf.ngev,
481                            nit=n_iterations, status=warnflag, message=msg,
482                            x=x, success=(warnflag == 0), hess_inv=hess_inv)

485  class LbfgsInvHessProduct(LinearOperator):
486      """Linear operator for the L-BFGS approximate inverse Hessian. 
487   
488      This operator computes the product of a vector with the approximate inverse 
489      of the Hessian of the objective function, using the L-BFGS limited 
490      memory approximation to the inverse Hessian, accumulated during the 
491      optimization. 
492   
493      Objects of this class implement the ``scipy.sparse.linalg.LinearOperator`` 
494      interface. 
495   
496      Parameters 
497      ---------- 
498      sk : array_like, shape=(n_corr, n) 
499          Array of `n_corr` most recent updates to the solution vector. 
500          (See [1]). 
501      yk : array_like, shape=(n_corr, n) 
502          Array of `n_corr` most recent updates to the gradient. (See [1]). 
503   
504      References 
505      ---------- 
506      .. [1] Nocedal, Jorge. "Updating quasi-Newton matrices with limited 
507         storage." Mathematics of computation 35.151 (1980): 773-782. 
508   
509      """
510  
511      def __init__(self, sk, yk):
512          """Construct the operator."""
513          if sk.shape != yk.shape or sk.ndim != 2:
514              raise ValueError('sk and yk must have matching shape, (n_corrs, n)')
515          n_corrs, n = sk.shape
516  
517          super().__init__(dtype=np.float64, shape=(n, n))
518  
519          self.sk = sk
520          self.yk = yk
521          self.n_corrs = n_corrs
522          self.rho = 1 / np.einsum('ij,ij->i', sk, yk)
523  
524      def _matvec(self, x):
525          """Efficient matrix-vector multiply with the BFGS matrices. 
526   
527          This calculation is described in Section (4) of [1]. 
528   
529          Parameters 
530          ---------- 
531          x : ndarray 
532              An array with shape (n,) or (n,1). 
533   
534          Returns 
535          ------- 
536          y : ndarray 
537              The matrix-vector product 
538   
539          """
540          s, y, n_corrs, rho = self.sk, self.yk, self.n_corrs, self.rho
541          q = np.array(x, dtype=self.dtype, copy=True)
542          if q.ndim == 2 and q.shape[1] == 1:
543              q = q.reshape(-1)
544  
545          alpha = np.empty(n_corrs)
546  
547          for i in range(n_corrs-1, -1, -1):
548              alpha[i] = rho[i] * np.dot(s[i], q)
549              q = q - alpha[i]*y[i]
550  
551          r = q
552          for i in range(n_corrs):
553              beta = rho[i] * np.dot(y[i], r)
554              r = r + s[i] * (alpha[i] - beta)
555  
556          return r
557  
558      def todense(self):
559          """Return a dense array representation of this operator. 
560   
561          Returns 
562          ------- 
563          arr : ndarray, shape=(n, n) 
564              An array with the same shape and containing 
565              the same data represented by this `LinearOperator`. 
566   
567          """
568          s, y, n_corrs, rho = self.sk, self.yk, self.n_corrs, self.rho
569          I_arr = np.eye(*self.shape, dtype=self.dtype)
570          Hk = I_arr
571  
572          for i in range(n_corrs):
573              A1 = I_arr - s[i][:, np.newaxis] * y[i][np.newaxis, :] * rho[i]
574              A2 = I_arr - y[i][:, np.newaxis] * s[i][np.newaxis, :] * rho[i]
575  
576              Hk = np.dot(A1, np.dot(Hk, A2)) + (rho[i] * s[i][:, np.newaxis] *
577                                                          s[i][np.newaxis, :])
578          return Hk
579
