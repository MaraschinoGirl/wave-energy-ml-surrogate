3    import win32ui  # isort: skip # Must be imported before dde !
4    import dde
5    from pywin.mfc import object
6    
7    
8    class MySystemTopic(object.Object):
9        def __init__(self):
10           object.Object.__init__(self, dde.CreateServerSystemTopic())
11   
12       def Exec(self, cmd):
13           print("System Topic asked to exec", cmd)
14   
15   
16   class MyOtherTopic(object.Object):
17       def __init__(self, topicName):
18           object.Object.__init__(self, dde.CreateTopic(topicName))
19   
20       def Exec(self, cmd):
21           print("Other Topic asked to exec", cmd)
22   
23   
24   class MyRequestTopic(object.Object):
25       def __init__(self, topicName):
26           topic = dde.CreateTopic(topicName)
27           topic.AddItem(dde.CreateStringItem(""))
28           object.Object.__init__(self, topic)
29   
30       def Request(self, aString):
31           print("Request Topic asked to compute length of:", aString)
32           return str(len(aString))
33   
34   
35   server = dde.CreateServer()
36   server.AddTopic(MySystemTopic())
37   server.AddTopic(MyOtherTopic("RunAnyCommand"))
38   server.AddTopic(MyRequestTopic("ComputeStringLength"))
39   server.Create("RunAny")
40   
41   while 1:
42       win32ui.PumpWaitingMessages(0, -1)
