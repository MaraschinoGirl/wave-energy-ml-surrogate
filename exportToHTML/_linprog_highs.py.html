34   def _highs_to_scipy_status_message(highs_status, highs_message):
35       """Converts HiGHS status number/message to SciPy status number/message"""
36   
37       scipy_statuses_messages = {
38           None: (4, "HiGHS did not provide a status code. "),
39           HighsModelStatus.kNotset: (4, ""),
40           HighsModelStatus.kLoadError: (4, ""),
41           HighsModelStatus.kModelError: (2, ""),
42           HighsModelStatus.kPresolveError: (4, ""),
43           HighsModelStatus.kSolveError: (4, ""),
44           HighsModelStatus.kPostsolveError: (4, ""),
45           HighsModelStatus.kModelEmpty: (4, ""),
46           HighsModelStatus.kObjectiveBound: (4, ""),
47           HighsModelStatus.kObjectiveTarget: (4, ""),
48           HighsModelStatus.kOptimal: (0, "Optimization terminated successfully. "),
49           HighsModelStatus.kTimeLimit: (1, "Time limit reached. "),
50           HighsModelStatus.kIterationLimit: (1, "Iteration limit reached. "),
51           HighsModelStatus.kInfeasible: (2, "The problem is infeasible. "),
52           HighsModelStatus.kUnbounded: (3, "The problem is unbounded. "),
53           HighsModelStatus.kUnboundedOrInfeasible: (4, "The problem is unbounded "
54                                                     "or infeasible. ")}
55       unrecognized = (4, "The HiGHS status code was not recognized. ")
56       scipy_status, scipy_message = (
57           scipy_statuses_messages.get(highs_status, unrecognized))
58       hstat = int(highs_status) if highs_status is not None else None
59       scipy_message = (f"{scipy_message}"
60                        f"(HiGHS Status {hstat}: {highs_message})")
61       return scipy_status, scipy_message

64   def _replace_inf(x):
65       # Replace `np.inf` with kHighsInf
66       infs = np.isinf(x)
67       with np.errstate(invalid="ignore"):
68           x[infs] = np.sign(x[infs])*kHighsInf
69       return x

72   def _convert_to_highs_enum(option, option_str, choices):
73       # If option is in the choices we can look it up, if not use
74       # the default value taken from function signature and warn:
75       try:
76           return choices[option.lower()]
77       except AttributeError:
78           return choices[option]
79       except KeyError:
80           sig = inspect.signature(_linprog_highs)
81           default_str = sig.parameters[option_str].default
82           warn(f"Option {option_str} is {option}, but only values in "
83                f"{set(choices.keys())} are allowed. Using default: "
84                f"{default_str}.",
85                OptimizeWarning, stacklevel=3)
86           return choices[default_str]

89   def _linprog_highs(lp, solver, time_limit=None, presolve=True,
90                      disp=False, maxiter=None,
91                      dual_feasibility_tolerance=None,
92                      primal_feasibility_tolerance=None,
93                      ipm_optimality_tolerance=None,
94                      simplex_dual_edge_weight_strategy=None,
95                      mip_rel_gap=None,
96                      mip_max_nodes=None,
97                      **unknown_options):
98       r""" 
99       Solve the following linear programming problem using one of the HiGHS 
100      solvers: 
101   
102      User-facing documentation is in _linprog_doc.py. 
103   
104      Parameters 
105      ---------- 
106      lp :  _LPProblem 
107          A ``scipy.optimize._linprog_util._LPProblem`` ``namedtuple``. 
108      solver : "ipm" or "simplex" or None 
109          Which HiGHS solver to use.  If ``None``, "simplex" will be used. 
110   
111      Options 
112      ------- 
113      maxiter : int 
114          The maximum number of iterations to perform in either phase. For 
115          ``solver='ipm'``, this does not include the number of crossover 
116          iterations.  Default is the largest possible value for an ``int`` 
117          on the platform. 
118      disp : bool 
119          Set to ``True`` if indicators of optimization status are to be printed 
120          to the console each iteration; default ``False``. 
121      time_limit : float 
122          The maximum time in seconds allotted to solve the problem; default is 
123          the largest possible value for a ``double`` on the platform. 
124      presolve : bool 
125          Presolve attempts to identify trivial infeasibilities, 
126          identify trivial unboundedness, and simplify the problem before 
127          sending it to the main solver. It is generally recommended 
128          to keep the default setting ``True``; set to ``False`` if presolve is 
129          to be disabled. 
130      dual_feasibility_tolerance : double 
131          Dual feasibility tolerance.  Default is 1e-07. 
132          The minimum of this and ``primal_feasibility_tolerance`` 
133          is used for the feasibility tolerance when ``solver='ipm'``. 
134      primal_feasibility_tolerance : double 
135          Primal feasibility tolerance.  Default is 1e-07. 
136          The minimum of this and ``dual_feasibility_tolerance`` 
137          is used for the feasibility tolerance when ``solver='ipm'``. 
138      ipm_optimality_tolerance : double 
139          Optimality tolerance for ``solver='ipm'``.  Default is 1e-08. 
140          Minimum possible value is 1e-12 and must be smaller than the largest 
141          possible value for a ``double`` on the platform. 
142      simplex_dual_edge_weight_strategy : str (default: None) 
143          Strategy for simplex dual edge weights. The default, ``None``, 
144          automatically selects one of the following. 
145   
146          ``'dantzig'`` uses Dantzig's original strategy of choosing the most 
147          negative reduced cost. 
148   
149          ``'devex'`` uses the strategy described in [15]_. 
150   
151          ``steepest`` uses the exact steepest edge strategy as described in 
152          [16]_. 
153   
154          ``'steepest-devex'`` begins with the exact steepest edge strategy 
155          until the computation is too costly or inexact and then switches to 
156          the devex method. 
157   
158          Currently, using ``None`` always selects ``'steepest-devex'``, but this 
159          may change as new options become available. 
160   
161      mip_max_nodes : int 
162          The maximum number of nodes allotted to solve the problem; default is 
163          the largest possible value for a ``HighsInt`` on the platform. 
164          Ignored if not using the MIP solver. 
165      unknown_options : dict 
166          Optional arguments not used by this particular solver. If 
167          ``unknown_options`` is non-empty, a warning is issued listing all 
168          unused options. 
169   
170      Returns 
171      ------- 
172      sol : dict 
173          A dictionary consisting of the fields: 
174   
175              x : 1D array 
176                  The values of the decision variables that minimizes the 
177                  objective function while satisfying the constraints. 
178              fun : float 
179                  The optimal value of the objective function ``c @ x``. 
180              slack : 1D array 
181                  The (nominally positive) values of the slack, 
182                  ``b_ub - A_ub @ x``. 
183              con : 1D array 
184                  The (nominally zero) residuals of the equality constraints, 
185                  ``b_eq - A_eq @ x``. 
186              success : bool 
187                  ``True`` when the algorithm succeeds in finding an optimal 
188                  solution. 
189              status : int 
190                  An integer representing the exit status of the algorithm. 
191   
192                  ``0`` : Optimization terminated successfully. 
193   
194                  ``1`` : Iteration or time limit reached. 
195   
196                  ``2`` : Problem appears to be infeasible. 
197   
198                  ``3`` : Problem appears to be unbounded. 
199   
200                  ``4`` : The HiGHS solver ran into a problem. 
201   
202              message : str 
203                  A string descriptor of the exit status of the algorithm. 
204              nit : int 
205                  The total number of iterations performed. 
206                  For ``solver='simplex'``, this includes iterations in all 
207                  phases. For ``solver='ipm'``, this does not include 
208                  crossover iterations. 
209              crossover_nit : int 
210                  The number of primal/dual pushes performed during the 
211                  crossover routine for ``solver='ipm'``.  This is ``0`` 
212                  for ``solver='simplex'``. 
213              ineqlin : OptimizeResult 
214                  Solution and sensitivity information corresponding to the 
215                  inequality constraints, `b_ub`. A dictionary consisting of the 
216                  fields: 
217   
218                  residual : np.ndnarray 
219                      The (nominally positive) values of the slack variables, 
220                      ``b_ub - A_ub @ x``.  This quantity is also commonly 
221                      referred to as "slack". 
222   
223                  marginals : np.ndarray 
224                      The sensitivity (partial derivative) of the objective 
225                      function with respect to the right-hand side of the 
226                      inequality constraints, `b_ub`. 
227   
228              eqlin : OptimizeResult 
229                  Solution and sensitivity information corresponding to the 
230                  equality constraints, `b_eq`.  A dictionary consisting of the 
231                  fields: 
232   
233                  residual : np.ndarray 
234                      The (nominally zero) residuals of the equality constraints, 
235                      ``b_eq - A_eq @ x``. 
236   
237                  marginals : np.ndarray 
238                      The sensitivity (partial derivative) of the objective 
239                      function with respect to the right-hand side of the 
240                      equality constraints, `b_eq`. 
241   
242              lower, upper : OptimizeResult 
243                  Solution and sensitivity information corresponding to the 
244                  lower and upper bounds on decision variables, `bounds`. 
245   
246                  residual : np.ndarray 
247                      The (nominally positive) values of the quantity 
248                      ``x - lb`` (lower) or ``ub - x`` (upper). 
249   
250                  marginals : np.ndarray 
251                      The sensitivity (partial derivative) of the objective 
252                      function with respect to the lower and upper 
253                      `bounds`. 
254   
255              mip_node_count : int 
256                  The number of subproblems or "nodes" solved by the MILP 
257                  solver. Only present when `integrality` is not `None`. 
258   
259              mip_dual_bound : float 
260                  The MILP solver's final estimate of the lower bound on the 
261                  optimal solution. Only present when `integrality` is not 
262                  `None`. 
263   
264              mip_gap : float 
265                  The difference between the final objective function value 
266                  and the final dual bound, scaled by the final objective 
267                  function value. Only present when `integrality` is not 
268                  `None`. 
269   
270      Notes 
271      ----- 
272      The result fields `ineqlin`, `eqlin`, `lower`, and `upper` all contain 
273      `marginals`, or partial derivatives of the objective function with respect 
274      to the right-hand side of each constraint. These partial derivatives are 
275      also referred to as "Lagrange multipliers", "dual values", and 
276      "shadow prices". The sign convention of `marginals` is opposite that 
277      of Lagrange multipliers produced by many nonlinear solvers. 
278   
279      References 
280      ---------- 
281      .. [15] Harris, Paula MJ. "Pivot selection methods of the Devex LP code." 
282              Mathematical programming 5.1 (1973): 1-28. 
283      .. [16] Goldfarb, Donald, and John Ker Reid. "A practicable steepest-edge 
284              simplex algorithm." Mathematical Programming 12.1 (1977): 361-371. 
285      """
286      if unknown_options:
287          message = (f"Unrecognized options detected: {unknown_options}. "
288                     "These will be passed to HiGHS verbatim.")
289          warn(message, OptimizeWarning, stacklevel=3)
290  
291      # Map options to HiGHS enum values
292      simplex_dual_edge_weight_strategy_enum = _convert_to_highs_enum(
293          simplex_dual_edge_weight_strategy,
294          'simplex_dual_edge_weight_strategy',
295          choices={'dantzig': \
296                   SimplexEdgeWeightStrategy.kSimplexEdgeWeightStrategyDantzig,
297                   'devex': \
298                   SimplexEdgeWeightStrategy.kSimplexEdgeWeightStrategyDevex,
299                   'steepest-devex': \
300                   SimplexEdgeWeightStrategy.kSimplexEdgeWeightStrategyChoose,
301                   'steepest': \
302                   SimplexEdgeWeightStrategy.kSimplexEdgeWeightStrategySteepestEdge,
303                   None: None})
304  
305      c, A_ub, b_ub, A_eq, b_eq, bounds, x0, integrality = lp
306  
307      lb, ub = bounds.T.copy()  # separate bounds, copy->C-cntgs
308      # highs_wrapper solves LHS <= A*x <= RHS, not equality constraints
309      with np.errstate(invalid="ignore"):
310          lhs_ub = -np.ones_like(b_ub)*np.inf  # LHS of UB constraints is -inf
311      rhs_ub = b_ub  # RHS of UB constraints is b_ub
312      lhs_eq = b_eq  # Equality constraint is inequality
313      rhs_eq = b_eq  # constraint with LHS=RHS
314      lhs = np.concatenate((lhs_ub, lhs_eq))
315      rhs = np.concatenate((rhs_ub, rhs_eq))
316  
317      if issparse(A_ub) or issparse(A_eq):
318          A = vstack((A_ub, A_eq))
319      else:
320          A = np.vstack((A_ub, A_eq))
321      A = csc_matrix(A)
322  
323      options = {
324          'presolve': presolve,
325          'sense': ObjSense.kMinimize,
326          'solver': solver,
327          'time_limit': time_limit,
328          'highs_debug_level': HighsDebugLevel.kHighsDebugLevelNone,
329          'dual_feasibility_tolerance': dual_feasibility_tolerance,
330          'ipm_optimality_tolerance': ipm_optimality_tolerance,
331          'log_to_console': disp,
332          'mip_max_nodes': mip_max_nodes,
333          'output_flag': disp,
334          'primal_feasibility_tolerance': primal_feasibility_tolerance,
335          'simplex_dual_edge_weight_strategy':
336              simplex_dual_edge_weight_strategy_enum,
337          'simplex_strategy': SimplexStrategy.kSimplexStrategyDual,
338          'ipm_iteration_limit': maxiter,
339          'simplex_iteration_limit': maxiter,
340          'mip_rel_gap': mip_rel_gap,
341      }
342      options.update(unknown_options)
343  
344      # np.inf doesn't work; use very large constant
345      rhs = _replace_inf(rhs)
346      lhs = _replace_inf(lhs)
347      lb = _replace_inf(lb)
348      ub = _replace_inf(ub)
349  
350      if integrality is None or np.sum(integrality) == 0:
351          integrality = np.empty(0)
352      else:
353          integrality = np.array(integrality)
354  
355      res = _highs_wrapper(c, A.indptr, A.indices, A.data, lhs, rhs,
356                           lb, ub, integrality.astype(np.uint8), options)
357  
358      # HiGHS represents constraints as lhs/rhs, so
359      # Ax + s = b => Ax = b - s
360      # and we need to split up s by A_ub and A_eq
361      if 'slack' in res:
362          slack = res['slack']
363          con = np.array(slack[len(b_ub):])
364          slack = np.array(slack[:len(b_ub)])
365      else:
366          slack, con = None, None
367  
368      # lagrange multipliers for equalities/inequalities and upper/lower bounds
369      if 'lambda' in res:
370          lamda = res['lambda']
371          marg_ineqlin = np.array(lamda[:len(b_ub)])
372          marg_eqlin = np.array(lamda[len(b_ub):])
373          marg_upper = np.array(res['marg_bnds'][1, :])
374          marg_lower = np.array(res['marg_bnds'][0, :])
375      else:
376          marg_ineqlin, marg_eqlin = None, None
377          marg_upper, marg_lower = None, None
378  
379      # this needs to be updated if we start choosing the solver intelligently
380  
381      # Convert to scipy-style status and message
382      highs_status = res.get('status', None)
383      highs_message = res.get('message', None)
384      status, message = _highs_to_scipy_status_message(highs_status,
385                                                       highs_message)
386  
387      x = res['x'] # is None if not set
388      sol = {'x': x,
389             'slack': slack,
390             'con': con,
391             'ineqlin': OptimizeResult({
392                 'residual': slack,
393                 'marginals': marg_ineqlin,
394             }),
395             'eqlin': OptimizeResult({
396                 'residual': con,
397                 'marginals': marg_eqlin,
398             }),
399             'lower': OptimizeResult({
400                 'residual': None if x is None else x - lb,
401                 'marginals': marg_lower,
402             }),
403             'upper': OptimizeResult({
404                 'residual': None if x is None else ub - x,
405                 'marginals': marg_upper
406              }),
407             'fun': res.get('fun'),
408             'status': status,
409             'success': res['status'] == HighsModelStatus.kOptimal,
410             'message': message,
411             'nit': res.get('simplex_nit', 0) or res.get('ipm_nit', 0),
412             'crossover_nit': res.get('crossover_nit'),
413             }
414  
415      if np.any(x) and integrality is not None:
416          sol.update({
417              'mip_node_count': res.get('mip_node_count', 0),
418              'mip_dual_bound': res.get('mip_dual_bound', 0.0),
419              'mip_gap': res.get('mip_gap', 0.0),
420          })
421  
422      return sol
