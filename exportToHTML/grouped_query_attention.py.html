15   @keras_export("keras.layers.GroupQueryAttention")
16   class GroupedQueryAttention(Layer):
17       """Grouped Query Attention layer. 
18    
19       This is an implementation of grouped-query attention introduced by 
20       [Ainslie et al., 2023](https://arxiv.org/abs/2305.13245). Here 
21       `num_key_value_heads` denotes number of groups, setting 
22       `num_key_value_heads` to 1 is equivalent to multi-query attention, and 
23       when `num_key_value_heads` is equal to `num_query_heads` it is equivalent 
24       to multi-head attention. 
25    
26       This layer first projects `query`, `key`, and `value` tensors. Then, `key` 
27       and `value` are repeated to match the number of heads of `query`. 
28    
29       Then, the `query` is scaled and dot-producted with `key` tensors. These are 
30       softmaxed to obtain attention probabilities. The value tensors are then 
31       interpolated by these probabilities and concatenated back to a single 
32       tensor. 
33    
34       Args: 
35           head_dim: Size of each attention head. 
36           num_query_heads: Number of query attention heads. 
37           num_key_value_heads: Number of key and value attention heads. 
38           dropout: Dropout probability. 
39           use_bias: Boolean, whether the dense layers use bias vectors/matrices. 
40           flash_attention: If `None`, the layer attempts to use flash 
41               attention for faster and more memory-efficient attention 
42               computations when possible. This behavior can be configured using 
43               `keras.config.enable_flash_attention()` or 
44               `keras.config.disable_flash_attention()`. 
45           kernel_initializer: Initializer for dense layer kernels. 
46           bias_initializer: Initializer for dense layer biases. 
47           kernel_regularizer: Regularizer for dense layer kernels. 
48           bias_regularizer: Regularizer for dense layer biases. 
49           activity_regularizer: Regularizer for dense layer activity. 
50           kernel_constraint: Constraint for dense layer kernels. 
51           bias_constraint: Constraint for dense layer kernels. 
52           seed: Optional integer to seed the dropout layer. 
53    
54       Call arguments: 
55           query: Query tensor of shape `(batch_dim, target_seq_len, feature_dim)`, 
56               where `batch_dim` is batch size, `target_seq_len` is the length of 
57               target sequence, and `feature_dim` is dimension of feature. 
58           value: Value tensor of shape `(batch_dim, source_seq_len, feature_dim)`, 
59               where `batch_dim` is batch size, `source_seq_len` is the length of 
60               source sequence, and `feature_dim` is dimension of feature. 
61           key: Optional key tensor of shape 
62               `(batch_dim, source_seq_len, feature_dim)`. If not given, will use 
63               `value` for both `key` and `value`, which is most common case. 
64           attention_mask: A boolean mask of shape 
65               `(batch_dim, target_seq_len, source_seq_len)`, that prevents 
66               attention to certain positions. The boolean mask specifies which 
67               query elements can attend to which key elements, where 1 indicates 
68               attention and 0 indicates no attention. Broadcasting can happen for 
69               the missing batch dimensions and the head dimension. 
70           return_attention_scores: A boolean to indicate whether the output 
71               should be `(attention_output, attention_scores)` if `True`, or 
72               `attention_output` if `False`. Defaults to `False`. 
73           training: Python boolean indicating whether the layer should behave in 
74               training mode (adding dropout) or in inference mode (no dropout). 
75               Will go with either using the training mode of the parent 
76               layer/model or `False` (inference) if there is no parent layer. 
77           use_causal_mask: A boolean to indicate whether to apply a causal mask to 
78               prevent tokens from attending to future tokens (e.g., used in a 
79               decoder Transformer). 
80    
81       Returns: 
82           attention_output: Result of the computation, of shape 
83               `(batch_dim, target_seq_len, feature_dim)`, where `target_seq_len` 
84               is for target sequence length and `feature_dim` is the query input 
85               last dim. 
86           attention_scores: (Optional) attention coefficients of shape 
87               `(batch_dim, num_query_heads, target_seq_len, source_seq_len)`. 
88       """
89   
90       def __init__(
91           self,
92           head_dim,
93           num_query_heads,
94           num_key_value_heads,
95           dropout=0.0,
96           use_bias=True,
97           flash_attention=None,
98           kernel_initializer="glorot_uniform",
99           bias_initializer="zeros",
100          kernel_regularizer=None,
101          bias_regularizer=None,
102          activity_regularizer=None,
103          kernel_constraint=None,
104          bias_constraint=None,
105          seed=None,
106          **kwargs,
107      ):
108          super().__init__(**kwargs)
109          self.supports_masking = True
110          self.head_dim = head_dim
111          self.num_query_heads = num_query_heads
112          self.num_key_value_heads = num_key_value_heads
113          if num_query_heads % num_key_value_heads != 0:
114              raise ValueError(
115                  "`num_query_heads` must be divisible by `num_key_value_heads`."
116              )
117          self.num_repeats = num_query_heads // num_key_value_heads
118          self.dropout = dropout
119          self.use_bias = use_bias
120          self._flash_attention = flash_attention or is_flash_attention_enabled()
121          self.kernel_initializer = initializers.get(kernel_initializer)
122          self.bias_initializer = initializers.get(bias_initializer)
123          self.kernel_regularizer = regularizers.get(kernel_regularizer)
124          self.bias_regularizer = regularizers.get(bias_regularizer)
125          self.activity_regularizer = regularizers.get(activity_regularizer)
126          self.kernel_constraint = constraints.get(kernel_constraint)
127          self.bias_constraint = constraints.get(bias_constraint)
128          self.seed = seed
129  
130          self._inverse_sqrt_head_dim = 1.0 / math.sqrt(float(self.head_dim))
131          self._return_attention_scores = False
132  
133          # Check for flash attention constraints
134          if self._flash_attention and self.dropout > 0.0:
135              raise ValueError(
136                  "Dropout is not supported when flash attention is enabled. "
137                  "Please set dropout to 0.0 to use flash attention."
138              )

140      def build(
141          self,
142          query_shape,
143          value_shape,
144          key_shape=None,
145      ):
146          # Einsum variables:
147          # b = batch size
148          # q = query length
149          # k = key/value length
150          # m = model dim
151          # u = num query heads
152          # v = num key/value heads
153          # h = head dim
154          key_shape = value_shape if key_shape is None else key_shape
155          self.feature_dim = query_shape[-1]
156          self._query_dense = EinsumDense(
157              "bqm,muh->bquh",
158              output_shape=(None, self.num_query_heads, self.head_dim),
159              bias_axes="uh" if self.use_bias else None,
160              name="query",
161              **self._get_common_kwargs_for_sublayer(),
162          )
163          self._query_dense.build(query_shape)
164  
165          self._key_dense = EinsumDense(
166              "bkm,mvh->bkvh",
167              output_shape=(None, self.num_key_value_heads, self.head_dim),
168              bias_axes="vh" if self.use_bias else None,
169              name="key",
170              **self._get_common_kwargs_for_sublayer(),
171          )
172          self._key_dense.build(key_shape)
173  
174          self._value_dense = EinsumDense(
175              "bkm,mvh->bkvh",
176              output_shape=(None, self.num_key_value_heads, self.head_dim),
177              bias_axes="vh" if self.use_bias else None,
178              name="value",
179              **self._get_common_kwargs_for_sublayer(),
180          )
181          self._value_dense.build(value_shape)
182  
183          self._softmax = Softmax(axis=-1, dtype=self.dtype_policy)
184          self._dropout_layer = Dropout(
185              rate=self.dropout, dtype=self.dtype_policy, seed=self.seed
186          )
187  
188          self._dot_product_equation = "bquh,bkuh->buqk"
189          self._combine_equation = "buqk,bkuh->bquh"
190  
191          self._output_dense = EinsumDense(
192              "bquh,uhm->bqm",
193              output_shape=(None, self.feature_dim),
194              bias_axes="m" if self.use_bias else None,
195              name="attention_output",
196              **self._get_common_kwargs_for_sublayer(),
197          )
198          self._output_dense.build(
199              (None, None, self.num_query_heads, self.head_dim)
200          )
201          self.built = True
202  
203      def _get_common_kwargs_for_sublayer(self):
204          common_kwargs = dict(
205              kernel_regularizer=self.kernel_regularizer,
206              bias_regularizer=self.bias_regularizer,
207              activity_regularizer=self.activity_regularizer,
208              kernel_constraint=self.kernel_constraint,
209              bias_constraint=self.bias_constraint,
210              dtype=self.dtype_policy,
211          )
212          # Create new clone of kernel/bias initializer, so that we don't reuse
213          # the initializer instance, which could lead to same init value since
214          # initializer is stateless.
215          kernel_initializer = self.kernel_initializer.__class__.from_config(
216              self.kernel_initializer.get_config()
217          )
218          bias_initializer = self.bias_initializer.__class__.from_config(
219              self.bias_initializer.get_config()
220          )
221          common_kwargs["kernel_initializer"] = kernel_initializer
222          common_kwargs["bias_initializer"] = bias_initializer
223          return common_kwargs
224  
225      def call(
226          self,
227          query,
228          value,
229          key=None,
230          query_mask=None,
231          value_mask=None,
232          key_mask=None,
233          attention_mask=None,
234          return_attention_scores=False,
235          training=None,
236          use_causal_mask=False,
237      ):
238          self._return_attention_scores = return_attention_scores
239          if key is None:
240              key = value
241  
242          attention_mask = self._compute_attention_mask(
243              query,
244              value,
245              query_mask=query_mask,
246              value_mask=value_mask,
247              key_mask=key_mask,
248              attention_mask=attention_mask,
249              use_causal_mask=use_causal_mask,
250          )
251  
252          query = self._query_dense(query)
253          key = self._key_dense(key)
254          value = self._value_dense(value)
255  
256          key = ops.repeat(
257              key, self.num_repeats, axis=2
258          )  # (batch_dim, source_seq_len, query_heads, head_dim)
259          value = ops.repeat(
260              value, self.num_repeats, axis=2
261          )  # (batch_dim, source_seq_len, query_heads, head_dim)
262  
263          output, scores = self._compute_attention(
264              query,
265              key,
266              value,
267              attention_mask=attention_mask,
268              training=training,
269          )
270  
271          output = self._output_dense(
272              output
273          )  # (batch_dim, target_seq_len, feature_dim)
274  
275          if return_attention_scores:
276              return output, scores
277          return output

279      def _compute_attention_mask(
280          self,
281          query,
282          value,
283          query_mask=None,
284          value_mask=None,
285          key_mask=None,
286          attention_mask=None,
287          use_causal_mask=False,
288      ):
289          """Computes the attention mask, using the Keras masks of the inputs. 
290   
291          * The `query`'s mask is reshaped from [B, T] to [B, T, 1]. 
292          * The `value`'s mask is reshaped from [B, S] to [B, 1, S]. 
293          * The `key`'s mask is reshaped from [B, S] to [B, 1, S]. The `key`'s 
294            mask is ignored if `key` is `None` or if `key is value`. 
295          * If `use_causal_mask=True`, then the causal mask is computed. Its shape 
296            is [1, T, S]. 
297   
298          All defined masks are merged using a logical AND operation (`&`). 
299   
300          In general, if the `query` and `value` are masked, then there is no need 
301          to define the `attention_mask`. 
302   
303          Args: 
304              query: Projected query tensor of shape `(B, T, N, key_dim)`. 
305              key: Projected key tensor of shape `(B, T, N, key_dim)`. 
306              value: Projected value tensor of shape `(B, T, N, value_dim)`. 
307              attention_mask: a boolean mask of shape `(B, T, S)`, that prevents 
308                  attention to certain positions. 
309              use_causal_mask: A boolean to indicate whether to apply a causal 
310                  mask to prevent tokens from attending to future tokens (e.g., 
311                  used in a decoder Transformer). 
312   
313          Returns: 
314              attention_mask: a boolean mask of shape `(B, T, S)`, that prevents 
315                  attention to certain positions, based on the Keras masks of the 
316                  `query`, `key`, `value`, and `attention_mask` tensors, and the 
317                  causal mask if `use_causal_mask=True`. 
318          """
319          auto_mask = None
320          if query_mask is not None:
321              query_mask = ops.cast(query_mask, "bool")  # defensive casting
322              # B = batch size, T = max query length
323              auto_mask = ops.expand_dims(query_mask, -1)  # shape is [B, T, 1]
324          if value_mask is not None:
325              value_mask = ops.cast(value_mask, "bool")  # defensive casting
326              # B = batch size, S == max value length
327              mask = ops.expand_dims(value_mask, -2)  # shape is [B, 1, S]
328              auto_mask = mask if auto_mask is None else auto_mask & mask
329          if key_mask is not None:
330              key_mask = ops.cast(key_mask, "bool")  # defensive casting
331              # B == batch size, S == max key length == max value length
332              mask = ops.expand_dims(key_mask, -2)  # shape is [B, 1, S]
333              auto_mask = mask if auto_mask is None else auto_mask & mask
334          if use_causal_mask:
335              # the shape of the causal mask is [1, T, S]
336              mask = self._compute_causal_mask(query, value)
337              auto_mask = mask if auto_mask is None else auto_mask & mask
338          if auto_mask is not None:
339              # merge attention_mask & automatic mask, to shape [B, T, S]
340              attention_mask = (
341                  auto_mask
342                  if attention_mask is None
343                  else ops.cast(attention_mask, bool) & auto_mask
344              )
345          return attention_mask
346  
347      def _compute_causal_mask(self, query, value=None):
348          """Computes a causal mask (e.g., for masked self-attention layers). 
349   
350          For example, if query and value both contain sequences of length 4, 
351          this function returns a boolean tensor equal to: 
352   
353          ``` 
354          [[[True,  False, False, False], 
355            [True,  True,  False, False], 
356            [True,  True,  True,  False], 
357            [True,  True,  True,  True]]] 
358          ``` 
359   
360          Args: 
361              query: query tensor of shape `(B, T, ...)`. 
362              value: value tensor of shape `(B, S, ...)` (optional, defaults to 
363                  query). 
364   
365          Returns: 
366              mask: a boolean tensor of shape `(1, T, S)` containing a lower 
367                  triangular matrix of shape `(T, S)`. 
368          """
369          q_seq_length = ops.shape(query)[1]
370          v_seq_length = q_seq_length if value is None else ops.shape(value)[1]
371          ones_mask = ops.ones((1, q_seq_length, v_seq_length), dtype="int32")
372          row_index = ops.cumsum(ones_mask, axis=-2)
373          col_index = ops.cumsum(ones_mask, axis=-1)
374          return ops.greater_equal(row_index, col_index)

376      def _compute_attention(
377          self, query, key, value, attention_mask=None, training=None
378      ):
379          # Check for flash attention constraints
380          if self._flash_attention and self._return_attention_scores:
381              raise ValueError(
382                  "Returning attention scores is not supported when flash "
383                  "attention is enabled. Please disable flash attention to access"
384                  " attention scores."
385              )
386  
387          # Determine whether to use dot-product attention
388          use_dot_product_attention = not (
389              self.dropout > 0.0
390              or self._return_attention_scores
391              or (len(query.shape) != 4)
392          )
393  
394          if use_dot_product_attention:
395              if attention_mask is not None:
396                  # Ensure attention_mask has the correct shape for broadcasting
397                  # Expected shape: [batch_size, num_heads, query_seq_len,
398                  # key_seq_len].
399                  mask_expansion_axis = -1 * 2 - 1
400                  len_attention_scores_shape = 4  # Only accepts 4D inputs
401                  for _ in range(
402                      len_attention_scores_shape - len(attention_mask.shape)
403                  ):
404                      attention_mask = ops.expand_dims(
405                          attention_mask, axis=mask_expansion_axis
406                      )
407                  attention_mask = ops.cast(attention_mask, dtype="bool")
408              # Directly compute the attention output using dot-product attention
409              attention_output = ops.dot_product_attention(
410                  query=query,
411                  key=key,
412                  value=value,
413                  bias=None,
414                  mask=attention_mask,
415                  scale=self._inverse_sqrt_head_dim,
416                  is_causal=False,
417                  flash_attention=self._flash_attention,
418              )
419              return attention_output, None
420  
421          # Default behavior without flash attention, with explicit attention
422          # scores
423          query = ops.multiply(
424              query, ops.cast(self._inverse_sqrt_head_dim, query.dtype)
425          )
426          # Take the dot product between "query" and "key" to get the raw
427          # attention scores.
428          scores = ops.einsum(
429              self._dot_product_equation, query, key
430          )  # (batch_dim, query_heads, target_seq_len, source_seq_len)
431          scores = self._masked_softmax(scores, attention_mask=attention_mask)
432          # This is actually dropping out entire tokens to attend to, which might
433          # seem a bit unusual, but is taken from the original Transformer paper.
434          if self.dropout > 0.0:
435              scores_dropout = self._dropout_layer(scores, training=training)
436          else:
437              scores_dropout = scores
438          output = ops.einsum(self._combine_equation, scores_dropout, value)
439          return output, scores
440  
441      def _masked_softmax(self, scores, attention_mask=None):
442          # Normalize the attention scores to probabilities.
443          # scores = [B, N, T, S]
444          if attention_mask is not None:
445              # The expand dim happens starting from the `num_heads` dimension,
446              # (<batch_dims>, num_heads, <query_attention_dims,
447              # key_attention_dims>)
448              mask_expansion_axis = -1 * 2 - 1
449              for _ in range(len(scores.shape) - len(attention_mask.shape)):
450                  attention_mask = ops.expand_dims(
451                      attention_mask, axis=mask_expansion_axis
452                  )
453          return self._softmax(scores, mask=attention_mask)
454  
455      def compute_output_shape(
456          self,
457          query_shape,
458          value_shape,
459          key_shape=None,
460      ):
461          if key_shape is None:
462              key_shape = value_shape
463  
464          if query_shape[-1] != value_shape[-1]:
465              raise ValueError(
466                  "The last dimension of `query_shape` and `value_shape` "
467                  f"must be equal, but are {query_shape[-1]}, {value_shape[-1]}. "
468                  "Received: query_shape={query_shape}, value_shape={value_shape}"
469              )
470  
471          if value_shape[1:-1] != key_shape[1:-1]:
472              raise ValueError(
473                  "All dimensions of `value` and `key`, except the last one, "
474                  f"must be equal. Received: value_shape={value_shape} and "
475                  f"key_shape={key_shape}"
476              )
477  
478          return query_shape
