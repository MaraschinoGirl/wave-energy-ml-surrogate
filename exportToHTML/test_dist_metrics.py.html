64   @pytest.mark.parametrize(
65       "metric_param_grid", METRICS_DEFAULT_PARAMS, ids=lambda params: params[0]
66   )
67   @pytest.mark.parametrize("X, Y", [(X64, Y64), (X32, Y32), (X_mmap, Y_mmap)])
68   @pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
69   def test_cdist(metric_param_grid, X, Y, csr_container):
70       metric, param_grid = metric_param_grid
71       keys = param_grid.keys()
72       X_csr, Y_csr = csr_container(X), csr_container(Y)
73       for vals in itertools.product(*param_grid.values()):
74           kwargs = dict(zip(keys, vals))
75           rtol_dict = {}
76           if metric == "mahalanobis" and X.dtype == np.float32:
77               # Computation of mahalanobis differs between
78               # the scipy and scikit-learn implementation.
79               # Hence, we increase the relative tolerance.
80               # TODO: Inspect slight numerical discrepancy
81               # with scipy
82               rtol_dict = {"rtol": 1e-6}
83   
84           # TODO: Remove when scipy minimum version >= 1.7.0
85           # scipy supports 0<p<1 for minkowski metric >= 1.7.0
86           if metric == "minkowski":
87               p = kwargs["p"]
88               if sp_version < parse_version("1.7.0") and p < 1:
89                   pytest.skip("scipy does not support 0<p<1 for minkowski metric < 1.7.0")
90   
91           D_scipy_cdist = cdist(X, Y, metric, **kwargs)
92   
93           dm = DistanceMetric.get_metric(metric, X.dtype, **kwargs)
94   
95           # DistanceMetric.pairwise must be consistent for all
96           # combinations of formats in {sparse, dense}.
97           D_sklearn = dm.pairwise(X, Y)
98           assert D_sklearn.flags.c_contiguous
99           assert_allclose(D_sklearn, D_scipy_cdist, **rtol_dict)
100  
101          D_sklearn = dm.pairwise(X_csr, Y_csr)
102          assert D_sklearn.flags.c_contiguous
103          assert_allclose(D_sklearn, D_scipy_cdist, **rtol_dict)
104  
105          D_sklearn = dm.pairwise(X_csr, Y)
106          assert D_sklearn.flags.c_contiguous
107          assert_allclose(D_sklearn, D_scipy_cdist, **rtol_dict)
108  
109          D_sklearn = dm.pairwise(X, Y_csr)
110          assert D_sklearn.flags.c_contiguous
111          assert_allclose(D_sklearn, D_scipy_cdist, **rtol_dict)

197  @pytest.mark.parametrize(
198      "metric_param_grid", METRICS_DEFAULT_PARAMS, ids=lambda params: params[0]
199  )
200  def test_distance_metrics_dtype_consistency(metric_param_grid):
201      # DistanceMetric must return similar distances for both float32 and float64
202      # input data.
203      metric, param_grid = metric_param_grid
204      keys = param_grid.keys()
205  
206      # Choose rtol to make sure that this test is robust to changes in the random
207      # seed in the module-level test data generation code.
208      rtol = 1e-5
209  
210      for vals in itertools.product(*param_grid.values()):
211          kwargs = dict(zip(keys, vals))
212          dm64 = DistanceMetric.get_metric(metric, np.float64, **kwargs)
213          dm32 = DistanceMetric.get_metric(metric, np.float32, **kwargs)
214  
215          D64 = dm64.pairwise(X64)
216          D32 = dm32.pairwise(X32)
217  
218          assert D64.dtype == np.float64
219          assert D32.dtype == np.float32
220  
221          # assert_allclose introspects the dtype of the input arrays to decide
222          # which rtol value to use by default but in this case we know that D32
223          # is not computed with the same precision so we set rtol manually.
224          assert_allclose(D64, D32, rtol=rtol)
225  
226          D64 = dm64.pairwise(X64, Y64)
227          D32 = dm32.pairwise(X32, Y32)
228          assert_allclose(D64, D32, rtol=rtol)

254  @pytest.mark.parametrize("writable_kwargs", [True, False])
255  @pytest.mark.parametrize(
256      "metric_param_grid", METRICS_DEFAULT_PARAMS, ids=lambda params: params[0]
257  )
258  @pytest.mark.parametrize("X", [X64, X32])
259  def test_pickle(writable_kwargs, metric_param_grid, X):
260      metric, param_grid = metric_param_grid
261      keys = param_grid.keys()
262      for vals in itertools.product(*param_grid.values()):
263          if any(isinstance(val, np.ndarray) for val in vals):
264              vals = copy.deepcopy(vals)
265              for val in vals:
266                  if isinstance(val, np.ndarray):
267                      val.setflags(write=writable_kwargs)
268          kwargs = dict(zip(keys, vals))
269          dm = DistanceMetric.get_metric(metric, X.dtype, **kwargs)
270          D1 = dm.pairwise(X)
271          dm2 = pickle.loads(pickle.dumps(dm))
272          D2 = dm2.pairwise(X)
273          assert_allclose(D1, D2)

286  @pytest.mark.parametrize("X, Y", [(X64, Y64), (X32, Y32), (X_mmap, Y_mmap)])
287  @pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
288  def test_haversine_metric(X, Y, csr_container):
289      # The Haversine DistanceMetric only works on 2 features.
290      X = np.asarray(X[:, :2])
291      Y = np.asarray(Y[:, :2])
292  
293      X_csr, Y_csr = csr_container(X), csr_container(Y)
294  
295      # Haversine is not supported by scipy.special.distance.{cdist,pdist}
296      # So we reimplement it to have a reference.
297      def haversine_slow(x1, x2):
298          return 2 * np.arcsin(
299              np.sqrt(
300                  np.sin(0.5 * (x1[0] - x2[0])) ** 2
301                  + np.cos(x1[0]) * np.cos(x2[0]) * np.sin(0.5 * (x1[1] - x2[1])) ** 2
302              )
303          )
304  
305      D_reference = np.zeros((X_csr.shape[0], Y_csr.shape[0]))
306      for i, xi in enumerate(X):
307          for j, yj in enumerate(Y):
308              D_reference[i, j] = haversine_slow(xi, yj)
309  
310      haversine = DistanceMetric.get_metric("haversine", X.dtype)
311  
312      D_sklearn = haversine.pairwise(X, Y)
313      assert_allclose(
314          haversine.dist_to_rdist(D_sklearn), np.sin(0.5 * D_reference) ** 2, rtol=1e-6
315      )
316  
317      assert_allclose(D_sklearn, D_reference)
318  
319      D_sklearn = haversine.pairwise(X_csr, Y_csr)
320      assert D_sklearn.flags.c_contiguous
321      assert_allclose(D_sklearn, D_reference)
322  
323      D_sklearn = haversine.pairwise(X_csr, Y)
324      assert D_sklearn.flags.c_contiguous
325      assert_allclose(D_sklearn, D_reference)
326  
327      D_sklearn = haversine.pairwise(X, Y_csr)
328      assert D_sklearn.flags.c_contiguous
329      assert_allclose(D_sklearn, D_reference)
