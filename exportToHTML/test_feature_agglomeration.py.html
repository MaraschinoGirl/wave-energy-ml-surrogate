16   def test_feature_agglomeration():
17       n_clusters = 1
18       X = np.array([0, 0, 1]).reshape(1, 3)  # (n_samples, n_features)
19   
20       agglo_mean = FeatureAgglomeration(n_clusters=n_clusters, pooling_func=np.mean)
21       agglo_median = FeatureAgglomeration(n_clusters=n_clusters, pooling_func=np.median)
22       agglo_mean.fit(X)
23       agglo_median.fit(X)
24   
25       assert np.size(np.unique(agglo_mean.labels_)) == n_clusters
26       assert np.size(np.unique(agglo_median.labels_)) == n_clusters
27       assert np.size(agglo_mean.labels_) == X.shape[1]
28       assert np.size(agglo_median.labels_) == X.shape[1]
29   
30       # Test transform
31       Xt_mean = agglo_mean.transform(X)
32       Xt_median = agglo_median.transform(X)
33       assert Xt_mean.shape[1] == n_clusters
34       assert Xt_median.shape[1] == n_clusters
35       assert Xt_mean == np.array([1 / 3.0])
36       assert Xt_median == np.array([0.0])
37   
38       # Test inverse transform
39       X_full_mean = agglo_mean.inverse_transform(Xt_mean)
40       X_full_median = agglo_median.inverse_transform(Xt_median)
41       assert np.unique(X_full_mean[0]).size == n_clusters
42       assert np.unique(X_full_median[0]).size == n_clusters
43   
44       assert_array_almost_equal(agglo_mean.transform(X_full_mean), Xt_mean)
45       assert_array_almost_equal(agglo_median.transform(X_full_median), Xt_median)

48   def test_feature_agglomeration_feature_names_out():
49       """Check `get_feature_names_out` for `FeatureAgglomeration`."""
50       X, _ = make_blobs(n_features=6, random_state=0)
51       agglo = FeatureAgglomeration(n_clusters=3)
52       agglo.fit(X)
53       n_clusters = agglo.n_clusters_
54   
55       names_out = agglo.get_feature_names_out()
56       assert_array_equal(
57           [f"featureagglomeration{i}" for i in range(n_clusters)], names_out
58       )
