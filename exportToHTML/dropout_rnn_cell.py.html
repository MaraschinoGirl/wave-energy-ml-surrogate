1    from keras.src import backend
2    from keras.src import ops

4    
5    class DropoutRNNCell:
6        """Object that holds dropout-related functionality for RNN cells. 
7     
8        This class is not a standalone RNN cell. It suppose to be used with a RNN 
9        cell by multiple inheritance. Any cell that mix with class should have 
10       following fields: 
11    
12       - `dropout`: a float number in the range `[0, 1]`. 
13           Dropout rate for the input tensor. 
14       - `recurrent_dropout`: a float number in the range `[0, 1]`. 
15           Dropout rate for the recurrent connections. 
16       - `seed_generator`, an instance of `backend.random.SeedGenerator`. 
17    
18       This object will create and cache dropout masks, and reuse them for 
19       all incoming steps, so that the same mask is used for every step. 
20       """
21   
22       def _create_dropout_mask(self, step_input, dropout_rate):
23           count = getattr(self, "dropout_mask_count", None)
24           ones = ops.ones_like(step_input)
25           if count is None:
26               return backend.random.dropout(
27                   ones, rate=dropout_rate, seed=self.seed_generator
28               )
29           else:
30               return [
31                   backend.random.dropout(
32                       ones, rate=dropout_rate, seed=self.seed_generator
33                   )
34                   for _ in range(count)
35               ]
36   
37       def get_dropout_mask(self, step_input):
38           if not hasattr(self, "_dropout_mask"):
39               self._dropout_mask = None
40           if self._dropout_mask is None and self.dropout > 0:
41               self._dropout_mask = self._create_dropout_mask(
42                   step_input, self.dropout
43               )
44           return self._dropout_mask
45   
46       def get_recurrent_dropout_mask(self, step_input):
47           if not hasattr(self, "_recurrent_dropout_mask"):
48               self._recurrent_dropout_mask = None
49           if self._recurrent_dropout_mask is None and self.recurrent_dropout > 0:
50               self._recurrent_dropout_mask = self._create_dropout_mask(
51                   step_input, self.recurrent_dropout
52               )
53           return self._recurrent_dropout_mask
54   
55       def reset_dropout_mask(self):
56           """Reset the cached dropout mask if any. 
57    
58           The RNN layer invokes this in the `call()` method 
59           so that the cached mask is cleared after calling `cell.call()`. The 
60           mask should be cached across all timestep within the same batch, but 
61           shouldn't be cached between batches. 
62           """
63           self._dropout_mask = None
64   
65       def reset_recurrent_dropout_mask(self):
66           self._recurrent_dropout_mask = None
