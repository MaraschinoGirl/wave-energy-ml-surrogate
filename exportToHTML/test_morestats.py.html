341      def test_weibull_min_case_A(self):
342          # data and reference values from `anderson` reference [7]
343          x = np.array([225, 171, 198, 189, 189, 135, 162, 135, 117, 162])
344          res = stats.anderson(x, 'weibull_min')
345          m, loc, scale = res.fit_result.params
346          assert_allclose((m, loc, scale), (2.38, 99.02, 78.23), rtol=2e-3)
347          assert_allclose(res.statistic, 0.260, rtol=1e-3)
348          assert res.statistic < res.critical_values[0]
349  
350          c = 1 / m  # ~0.42
351          assert_allclose(c, 1/2.38, rtol=2e-3)
352          # interpolate between rows for c=0.4 and c=0.45, indices -3 and -2
353          As40 = _Avals_weibull[-3]
354          As45 = _Avals_weibull[-2]
355          As_ref = As40 + (c - 0.4)/(0.45 - 0.4) * (As45 - As40)
356          # atol=1e-3 because results are rounded up to the next third decimal
357          assert np.all(res.critical_values > As_ref)
358          assert_allclose(res.critical_values, As_ref, atol=1e-3)

662          x1 = [-5, 1, 5, 10, 15, 20, 25]  # high scale, loc=10
663          x2 = [7.5, 8.5, 9.5, 10.5, 11.5, 12.5]  # low scale, loc=10
664          # ratio of scales is greater than 1. So, the
665          # p-value must be high when `alternative='less'`

667          statistic, pval = stats.ansari(x1, x2)
668          pval_l = stats.ansari(x1, x2, alternative='less').pvalue
669          pval_g = stats.ansari(x1, x2, alternative='greater').pvalue

673          # mass under the calculated statistic.

675          assert_allclose(pval_g + pval_l, 1 + prob, atol=1e-12)
676          # also check if one of the one-sided p-value equals half the
677          # two-sided p-value and the other one-sided p-value is its
678          # compliment.

680          assert_allclose(pval_l, 1+prob-pval/2, atol=1e-12)
681          # sanity check. The result should flip if
682          # we exchange x and y.
683          pval_l_reverse = stats.ansari(x2, x1, alternative='less').pvalue
