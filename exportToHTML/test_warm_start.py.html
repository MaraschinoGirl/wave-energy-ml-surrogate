17   def _assert_predictor_equal(gb_1, gb_2, X):
18       """Assert that two HistGBM instances are identical."""
19       # Check identical nodes for each tree
20       for pred_ith_1, pred_ith_2 in zip(gb_1._predictors, gb_2._predictors):
21           for predictor_1, predictor_2 in zip(pred_ith_1, pred_ith_2):
22               assert_array_equal(predictor_1.nodes, predictor_2.nodes)
23   
24       # Check identical predictions
25       assert_allclose(gb_1.predict(X), gb_2.predict(X))

112  def test_warm_start_early_stopping(GradientBoosting, X, y, scoring):
113      # Make sure that early stopping occurs after a small number of iterations
114      # when fitting a second time with warm starting.
115  
116      n_iter_no_change = 5
117      gb = GradientBoosting(
118          n_iter_no_change=n_iter_no_change,
119          max_iter=10000,
120          early_stopping=True,
121          random_state=42,
122          warm_start=True,
123          tol=1e-3,
124          scoring=scoring,
125      )
126      gb.fit(X, y)
127      n_iter_first_fit = gb.n_iter_
128      gb.fit(X, y)
129      n_iter_second_fit = gb.n_iter_
130      assert 0 < n_iter_second_fit - n_iter_first_fit < n_iter_no_change

169  
170      # Check that both predictors have the same train_score_ and
171      # validation_score_ attributes
172      assert_allclose(gb_1.train_score_, gb_2.train_score_)
173      assert_allclose(gb_1.validation_score_, gb_2.validation_score_)

175      # Check that both predictors are equal
176      _assert_predictor_equal(gb_1, gb_2, X)

186  @pytest.mark.parametrize("rng_type", ("none", "int", "instance"))
187  def test_random_seeds_warm_start(GradientBoosting, X, y, rng_type):
188      # Make sure the seeds for train/val split and small trainset subsampling
189      # are correctly set in a warm start context.
190      def _get_rng(rng_type):
191          # Helper to avoid consuming rngs
192          if rng_type == "none":
193              return None
194          elif rng_type == "int":
195              return 42
196          else:
197              return np.random.RandomState(0)
198  
199      random_state = _get_rng(rng_type)
200      gb_1 = GradientBoosting(early_stopping=True, max_iter=2, random_state=random_state)
201      gb_1.set_params(scoring=check_scoring(gb_1))
202      gb_1.fit(X, y)
203      random_seed_1_1 = gb_1._random_seed
204  
205      gb_1.fit(X, y)
206      random_seed_1_2 = gb_1._random_seed  # clear the old state, different seed
207  
208      random_state = _get_rng(rng_type)
209      gb_2 = GradientBoosting(
210          early_stopping=True, max_iter=2, random_state=random_state, warm_start=True
211      )
212      gb_2.set_params(scoring=check_scoring(gb_2))
213      gb_2.fit(X, y)  # inits state
214      random_seed_2_1 = gb_2._random_seed
215      gb_2.fit(X, y)  # clears old state and equals est
216      random_seed_2_2 = gb_2._random_seed
217  
218      # Without warm starting, the seeds should be
219      # * all different if random state is None
220      # * all equal if random state is an integer
221      # * different when refitting and equal with a new estimator (because
222      #   the random state is mutated)
223      if rng_type == "none":
224          assert random_seed_1_1 != random_seed_1_2 != random_seed_2_1
225      elif rng_type == "int":
226          assert random_seed_1_1 == random_seed_1_2 == random_seed_2_1
227      else:
228          assert random_seed_1_1 == random_seed_2_1 != random_seed_1_2
229  
230      # With warm starting, the seeds must be equal
231      assert random_seed_2_1 == random_seed_2_2
