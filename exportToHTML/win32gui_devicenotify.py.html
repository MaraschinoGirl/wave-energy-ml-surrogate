13   # These device GUIDs are from Ioevent.h in the Windows SDK.  Ideally they

15   GUID_DEVINTERFACE_USB_DEVICE = "{A5DCBF10-6530-11D2-901F-00C04FB951ED}"

18   # WM_DEVICECHANGE message handler.
19   def OnDeviceChange(hwnd, msg, wp, lp):
20       # Unpack the 'lp' into the appropriate DEV_BROADCAST_* structure,
21       # using the self-identifying data inside the DEV_BROADCAST_HDR.
22       info = win32gui_struct.UnpackDEV_BROADCAST(lp)
23       print("Device change notification:", wp, str(info))
24       if (
25           wp == win32con.DBT_DEVICEQUERYREMOVE
26           and info.devicetype == win32con.DBT_DEVTYP_HANDLE
27       ):
28           # Our handle is stored away in the structure - just close it
29           print("Device being removed - closing handle")
30           win32file.CloseHandle(info.handle)
31           # and cancel our notifications - if it gets plugged back in we get
32           # the same notification and try and close the same handle...
33           win32gui.UnregisterDeviceNotification(info.hdevnotify)
34       return True

37   def TestDeviceNotifications(dir_names):
38       wc = win32gui.WNDCLASS()
39       wc.lpszClassName = "test_devicenotify"
40       wc.style = win32con.CS_GLOBALCLASS | win32con.CS_VREDRAW | win32con.CS_HREDRAW
41       wc.hbrBackground = win32con.COLOR_WINDOW + 1
42       wc.lpfnWndProc = {win32con.WM_DEVICECHANGE: OnDeviceChange}
43       class_atom = win32gui.RegisterClass(wc)
44       hwnd = win32gui.CreateWindow(
45           wc.lpszClassName,
46           "Testing some devices",
47           # no need for it to be visible.
48           win32con.WS_CAPTION,
49           100,
50           100,
51           900,
52           900,
53           0,
54           0,
55           0,
56           None,
57       )
58   
59       hdevs = []
60       # Watch for all USB device notifications
61       filter = win32gui_struct.PackDEV_BROADCAST_DEVICEINTERFACE(
62           GUID_DEVINTERFACE_USB_DEVICE
63       )
64       hdev = win32gui.RegisterDeviceNotification(
65           hwnd, filter, win32con.DEVICE_NOTIFY_WINDOW_HANDLE
66       )
67       hdevs.append(hdev)
68       # and create handles for all specified directories
69       for d in dir_names:
70           hdir = win32file.CreateFile(
71               d,
72               winnt.FILE_LIST_DIRECTORY,
73               winnt.FILE_SHARE_READ | winnt.FILE_SHARE_WRITE | winnt.FILE_SHARE_DELETE,
74               None,  # security attributes
75               win32con.OPEN_EXISTING,
76               win32con.FILE_FLAG_BACKUP_SEMANTICS
77               | win32con.FILE_FLAG_OVERLAPPED,  # required privileges: SE_BACKUP_NAME and SE_RESTORE_NAME.
78               None,
79           )
80   
81           filter = win32gui_struct.PackDEV_BROADCAST_HANDLE(hdir)
82           hdev = win32gui.RegisterDeviceNotification(
83               hwnd, filter, win32con.DEVICE_NOTIFY_WINDOW_HANDLE
84           )
85           hdevs.append(hdev)
86   
87       # now start a message pump and wait for messages to be delivered.
88       print("Watching", len(hdevs), "handles - press Ctrl+C to terminate, or")
89       print("add and remove some USB devices...")
90       if not dir_names:
91           print("(Note you can also pass paths to watch on the command-line - eg,")
92           print("pass the root of an inserted USB stick to see events specific to")
93           print("that volume)")
94       while 1:
95           win32gui.PumpWaitingMessages()
96           time.sleep(0.01)
97       win32gui.DestroyWindow(hwnd)
98       win32gui.UnregisterClass(wc.lpszClassName, None)

105      # Then remove and insert the device.
