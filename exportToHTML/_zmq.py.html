160  @cfunc
161  @inline
162  @C.exceptval(-1)
163  def _check_rc(rc: C.int, error_without_errno: bint = False) -> C.int:
164      """internal utility for checking zmq return condition 
165   
166      and raising the appropriate Exception class 
167      """
168      errno: C.int = _zmq_errno()
169      PyErr_CheckSignals()
170      if errno == 0 and not error_without_errno:
171          return 0
172      if rc == -1:  # if rc < -1, it's a bug in libzmq. Should we warn?
173          if errno == EINTR:
174              raise InterruptedSystemCall(errno)
175          elif errno == EAGAIN:
176              raise Again(errno)
177          elif errno == ZMQ_ETERM:
178              raise ContextTerminated(errno)
179          else:
180              raise ZMQError(errno)
181      return 0

193  @cfunc
194  @nogil
195  def free_python_msg(data: p_void, vhint: p_void) -> C.int:
196      """A pure-C function for DECREF'ing Python-owned message data. 
197   
198      Sends a message on a PUSH socket 
199   
200      The hint is a `zhint` struct with two values: 
201   
202      sock (void *): pointer to the Garbage Collector's PUSH socket 
203      id (size_t): the id to be used to construct a zmq_msg_t that should be sent on a PUSH socket, 
204         signaling the Garbage Collector to remove its reference to the object. 
205   
206      When the Garbage Collector's PULL socket receives the message, 
207      it deletes its reference to the object, 
208      allowing Python to free the memory. 
209      """
210      msg = declare(zmq_msg_t)
211      msg_ptr: pointer(zmq_msg_t) = address(msg)
212      hint: pointer(_zhint) = cast(pointer(_zhint), vhint)
213      rc: C.int
214  
215      if hint != NULL:
216          zmq_msg_init_size(msg_ptr, sizeof(size_t))
217          memcpy(zmq_msg_data(msg_ptr), address(hint.id), sizeof(size_t))
218          rc = mutex_lock(hint.mutex)
219          if rc != 0:
220              fprintf(cstderr, "pyzmq-gc mutex lock failed rc=%d\n", rc)
221          rc = zmq_msg_send(msg_ptr, hint.sock, 0)
222          if rc < 0:
223              # gc socket could have been closed, e.g. during process teardown.
224              # If so, ignore the failure because there's nothing to do.
225              if _zmq_errno() != ZMQ_ENOTSOCK:
226                  fprintf(
227                      cstderr, "pyzmq-gc send failed: %s\n", zmq_strerror(_zmq_errno())
228                  )
229          rc = mutex_unlock(hint.mutex)
230          if rc != 0:
231              fprintf(cstderr, "pyzmq-gc mutex unlock failed rc=%d\n", rc)
232  
233          zmq_msg_close(msg_ptr)
234          free(hint)
235          return 0

250  @inline
251  def _asbuffer(obj, data_c: pointer(p_void), writable: bint = False) -> size_t:
252      """Get a C buffer from a memoryview"""
253      pybuf = declare(Py_buffer)
254      flags: C.int = PyBUF_ANY_CONTIGUOUS
255      if writable:
256          flags |= PyBUF_WRITABLE
257      rc: C.int = PyObject_GetBuffer(obj, address(pybuf), flags)
258      if rc < 0:
259          raise ValueError("Couldn't create buffer")
260      data_c[0] = pybuf.buf
261      data_size: size_t = pybuf.len
262      PyBuffer_Release(address(pybuf))
263      return data_size

269  @cclass
270  class Frame:
271      def __init__(
272          self, data=None, track=False, copy=None, copy_threshold=None, **kwargs
273      ):
274          rc: C.int
275          data_c: p_char = NULL
276          data_len_c: Py_ssize_t = 0
277          hint: pointer(_zhint)
278          if copy_threshold is None:
279              copy_threshold = zmq.COPY_THRESHOLD
280  
281          c_copy_threshold: C.size_t = 0
282          if copy_threshold is not None:
283              c_copy_threshold = copy_threshold
284  
285          zmq_msg_ptr: pointer(zmq_msg_t) = address(self.zmq_msg)
286          # init more as False
287          self.more = False
288  
289          # Save the data object in case the user wants the the data as a str.
290          self._data = data
291          self._failed_init = True  # bool switch for dealloc
292          self._buffer = None  # buffer view of data
293          self._bytes = None  # bytes copy of data
294  
295          self.tracker_event = None
296          self.tracker = None
297          # self.tracker should start finished
298          # except in the case where we are sharing memory with libzmq
299          if track:
300              self.tracker = zmq._FINISHED_TRACKER
301  
302          if isinstance(data, str):
303              raise TypeError("Str objects not allowed. Only: bytes, buffer interfaces.")
304  
305          if data is None:
306              rc = zmq_msg_init(zmq_msg_ptr)
307              _check_rc(rc)
308              self._failed_init = False
309              return
310  
311          data_len_c = _asbuffer(data, cast(pointer(p_void), address(data_c)))
312  
313          # copy unspecified, apply copy_threshold
314          c_copy: bint = True
315          if copy is None:
316              if c_copy_threshold and data_len_c < c_copy_threshold:
317                  c_copy = True
318              else:
319                  c_copy = False
320          else:
321              c_copy = copy
322  
323          if c_copy:
324              # copy message data instead of sharing memory
325              rc = zmq_msg_init_size(zmq_msg_ptr, data_len_c)
326              _check_rc(rc)
327              memcpy(zmq_msg_data(zmq_msg_ptr), data_c, data_len_c)
328              self._failed_init = False
329              return
330  
331          # Getting here means that we are doing a true zero-copy Frame,
332          # where libzmq and Python are sharing memory.
333          # Hook up garbage collection with MessageTracker and zmq_free_fn
334  
335          # Event and MessageTracker for monitoring when zmq is done with data:
336          if track:
337              evt = Event()
338              self.tracker_event = evt
339              self.tracker = zmq.MessageTracker(evt)
340          # create the hint for zmq_free_fn
341          # two pointers: the gc context and a message to be sent to the gc PULL socket
342          # allows libzmq to signal to Python when it is done with Python-owned memory.
343          global _gc
344          if _gc is None:
345              from zmq.utils.garbage import gc as _gc
346  
347          hint: pointer(_zhint) = cast(pointer(_zhint), malloc(sizeof(_zhint)))
348          hint.id = _gc.store(data, self.tracker_event)
349          if not _gc._push_mutex:
350              hint.mutex = mutex_allocate()
351              _gc._push_mutex = cast(size_t, hint.mutex)
352          else:
353              hint.mutex = cast(pointer(mutex_t), cast(size_t, _gc._push_mutex))
354          hint.sock = cast(p_void, cast(size_t, _gc._push_socket.underlying))
355  
356          rc = zmq_msg_init_data(
357              zmq_msg_ptr,
358              cast(p_void, data_c),
359              data_len_c,
360              cast(pointer(zmq_free_fn), free_python_msg),
361              cast(p_void, hint),
362          )
363          if rc != 0:
364              free(hint)
365              _check_rc(rc)
366          self._failed_init = False
367  
368      def __dealloc__(self):
369          if self._failed_init:
370              return
371          # decrease the 0MQ ref-count of zmq_msg
372          with nogil:
373              rc: C.int = zmq_msg_close(address(self.zmq_msg))
374          _check_rc(rc)
375  
376      def __copy__(self):
377          return self.fast_copy()
378  
379      def fast_copy(self) -> Frame:
380          new_msg: Frame = Frame()
381          # This does not copy the contents, but just increases the ref-count
382          # of the zmq_msg by one.
383          zmq_msg_copy(address(new_msg.zmq_msg), address(self.zmq_msg))
384          # Copy the ref to data so the copy won't create a copy when str is
385          # called.
386          if self._data is not None:
387              new_msg._data = self._data
388          if self._buffer is not None:
389              new_msg._buffer = self._buffer
390          if self._bytes is not None:
391              new_msg._bytes = self._bytes
392  
393          # Frame copies share the tracker and tracker_event
394          new_msg.tracker_event = self.tracker_event
395          new_msg.tracker = self.tracker
396  
397          return new_msg

403          buffer.buf = zmq_msg_data(address(self.zmq_msg))

408          buffer.format = "B"
409          buffer.ndim = 1
410          buffer.shape = address(buffer.len)
411          buffer.strides = NULL
412          buffer.suboffsets = NULL
413          buffer.itemsize = 1
414          buffer.internal = NULL

417          """Return the length of the message in bytes."""

419          return sz

423          """A memoryview of the message contents."""
424          _buffer = self._buffer and self._buffer()

426              return _buffer

433          """The message content as a Python bytes object. 
434   
435          The first time this property is accessed, a copy of the message 
436          contents is made. From then on that same copy of the message is

439          if self._bytes is None:

443      def get(self, option):
444          """ 
445          Get a Frame option or property. 
446   
447          See the 0MQ API documentation for zmq_msg_get and zmq_msg_gets 
448          for details on specific options. 
449   
450          .. versionadded:: libzmq-3.2 
451          .. versionadded:: 13.0 
452   
453          .. versionchanged:: 14.3 
454              add support for zmq_msg_gets (requires libzmq-4.1) 
455              All message properties are strings. 
456   
457          .. versionchanged:: 17.0 
458              Added support for `routing_id` and `group`. 
459              Only available if draft API is enabled 
460              with libzmq >= 4.2. 
461          """
462          rc: C.int = 0
463          property_c: p_char = NULL
464  
465          # zmq_msg_get
466          if isinstance(option, int):
467              rc = zmq_msg_get(address(self.zmq_msg), option)
468              _check_rc(rc)
469              return rc
470  
471          if option == 'routing_id':
472              routing_id: uint32_t = zmq_msg_routing_id(address(self.zmq_msg))
473              if routing_id == 0:
474                  _check_rc(-1)
475              return routing_id
476          elif option == 'group':
477              buf = zmq_msg_group(address(self.zmq_msg))
478              if buf == NULL:
479                  _check_rc(-1)
480              return buf.decode('utf8')
481  
482          # zmq_msg_gets
483          _check_version((4, 1), "get string properties")
484          if isinstance(option, str):
485              option = option.encode('utf8')
486  
487          if not isinstance(option, bytes):
488              raise TypeError(f"expected str, got: {option!r}")
489  
490          property_c = option
491  
492          result: p_char = cast(p_char, zmq_msg_gets(address(self.zmq_msg), property_c))
493          if result == NULL:
494              _check_rc(-1)
495          return result.decode('utf8')

497      def set(self, option, value):
498          """Set a Frame option. 
499   
500          See the 0MQ API documentation for zmq_msg_set 
501          for details on specific options. 
502   
503          .. versionadded:: libzmq-3.2 
504          .. versionadded:: 13.0 
505          .. versionchanged:: 17.0 
506              Added support for `routing_id` and `group`. 
507              Only available if draft API is enabled 
508              with libzmq >= 4.2. 
509          """
510          rc: C.int
511  
512          if option == 'routing_id':

522              return
523  
524          rc = zmq_msg_set(address(self.zmq_msg), option, value)

528  @cclass
529  class Context:
530      """ 
531      Manage the lifecycle of a 0MQ context. 
532   
533      Parameters 
534      ---------- 
535      io_threads : int 
536          The number of IO threads. 
537      """
538  
539      def __init__(self, io_threads: C.int = 1, shadow: size_t = 0):
540          self.handle = NULL
541          self._pid = 0
542          self._shadow = False
543  
544          if shadow:
545              self.handle = cast(p_void, shadow)
546              self._shadow = True
547          else:
548              self._shadow = False
549              self.handle = zmq_ctx_new()
550  
551          if self.handle == NULL:
552              raise ZMQError()
553  
554          rc: C.int = 0
555          if not self._shadow:
556              rc = zmq_ctx_set(self.handle, ZMQ_IO_THREADS, io_threads)
557              _check_rc(rc)
558  
559          self.closed = False
560          self._pid = getpid()
561  
562      @property
563      def underlying(self):
564          """The address of the underlying libzmq context"""
565          return cast(size_t, self.handle)
566  
567      @cfunc
568      @inline
569      def _term(self) -> C.int:
570          rc: C.int = 0
571          if self.handle != NULL and not self.closed and getpid() == self._pid:
572              with nogil:
573                  rc = zmq_ctx_destroy(self.handle)
574          self.handle = NULL
575          return rc
576  
577      def term(self):
578          """ 
579          Close or terminate the context. 
580   
581          This can be called to close the context by hand. If this is not called, 
582          the context will automatically be closed when it is garbage collected. 
583          """
584          rc: C.int = self._term()
585          try:
586              _check_rc(rc)
587          except InterruptedSystemCall:
588              # ignore interrupted term
589              # see PEP 475 notes about close & EINTR for why
590              pass
591  
592          self.closed = True
593  
594      def set(self, option: C.int, optval):
595          """ 
596          Set a context option. 
597   
598          See the 0MQ API documentation for zmq_ctx_set 
599          for details on specific options. 
600   
601          .. versionadded:: libzmq-3.2 
602          .. versionadded:: 13.0 
603   
604          Parameters 
605          ---------- 
606          option : int 
607              The option to set.  Available values will depend on your 
608              version of libzmq.  Examples include:: 
609   
610                  zmq.IO_THREADS, zmq.MAX_SOCKETS 
611   
612          optval : int 
613              The value of the option to set. 
614          """
615          optval_int_c: C.int
616          rc: C.int
617  
618          if self.closed:
619              raise RuntimeError("Context has been destroyed")
620  
621          if not isinstance(optval, int):
622              raise TypeError(f'expected int, got: {optval!r}')
623          optval_int_c = optval
624          rc = zmq_ctx_set(self.handle, option, optval_int_c)
625          _check_rc(rc)

628          """ 
629          Get the value of a context option. 
630   
631          See the 0MQ API documentation for zmq_ctx_get 
632          for details on specific options. 
633   
634          .. versionadded:: libzmq-3.2 
635          .. versionadded:: 13.0

643                  zmq.IO_THREADS, zmq.MAX_SOCKETS

672  @cclass
673  class Socket:
674      """ 
675      A 0MQ socket. 
676   
677      These objects will generally be constructed via the socket() method of a Context object. 
678   
679      Note: 0MQ Sockets are *not* threadsafe. **DO NOT** share them across threads. 
680   
681      Parameters 
682      ---------- 
683      context : Context 
684          The 0MQ Context this Socket belongs to. 
685      socket_type : int 
686          The socket type, which can be any of the 0MQ socket types: 
687          REQ, REP, PUB, SUB, PAIR, DEALER, ROUTER, PULL, PUSH, XPUB, XSUB. 
688   
689      See Also 
690      -------- 
691      .Context.socket : method for creating a socket bound to a Context. 
692      """
693  
694      def __init__(
695          self,
696          context=None,
697          socket_type: C.int = -1,
698          shadow: size_t = 0,
699          copy_threshold=None,
700      ):
701          # pre-init
702          self.handle = NULL
703          self._pid = 0
704          self._shadow = False
705          self.context = None
706  
707          if copy_threshold is None:
708              copy_threshold = zmq.COPY_THRESHOLD
709          self.copy_threshold = copy_threshold
710  
711          self.handle = NULL
712          self.context = context
713          if shadow:
714              self._shadow = True
715              self.handle = cast(p_void, shadow)
716          else:
717              if context is None:
718                  raise TypeError("context must be specified")
719              if socket_type < 0:
720                  raise TypeError("socket_type must be specified")
721              self._shadow = False

905      def bind(self, addr: str | bytes):
906          """ 
907          Bind the socket to an address. 
908   
909          This causes the socket to listen on a network port. Sockets on the 
910          other side of this connection will use ``Socket.connect(addr)`` to 
911          connect to this socket. 
912   
913          Parameters 
914          ---------- 
915          addr : str 
916              The address string. This has the form 'protocol://interface:port', 
917              for example 'tcp://127.0.0.1:5555'. Protocols supported include 
918              tcp, udp, pgm, epgm, inproc and ipc. If the address is unicode, it is 
919              encoded to utf-8 first. 
920          """
921          c_addr: p_char = _c_addr(addr)
922          _check_closed(self)
923          rc: C.int = zmq_bind(self.handle, c_addr)
924          if rc != 0:
925              _errno: C.int = _zmq_errno()
926              _ipc_max: C.int = get_ipc_path_max_len()

928                  path = addr.split('://', 1)[-1]

930                      f'ipc path "{path}" is longer than {_ipc_max} '
931                      'characters (sizeof(sockaddr_un.sun_path)). '
932                      'zmq.IPC_PATH_MAX_LEN constant can be used '
933                      'to check addr length (if it is defined).'
934                  )
935                  raise ZMQError(msg=msg)
936              elif _errno == ENOENT:
937                  path = addr.split('://', 1)[-1]
938                  msg = f'No such file or directory for ipc path "{path}".'
939                  raise ZMQError(msg=msg)
940          while True:
941              try:
942                  _check_rc(rc)
943              except InterruptedSystemCall:
944                  rc = zmq_bind(self.handle, c_addr)

946              else:
947                  break

1081     def send(self, data, flags=0, copy: bint = True, track: bint = False):
1082         """ 
1083         Send a single zmq message frame on this socket. 
1084  
1085         This queues the message to be sent by the IO thread at a later time. 
1086  
1087         With flags=NOBLOCK, this raises :class:`ZMQError` if the queue is full; 
1088         otherwise, this waits until space is available. 
1089         See :class:`Poller` for more general non-blocking I/O. 
1090  
1091         Parameters 
1092         ---------- 
1093         data : bytes, Frame, memoryview 
1094             The content of the message. This can be any object that provides 
1095             the Python buffer API (`memoryview(data)` can be called). 
1096         flags : int 
1097             0, NOBLOCK, SNDMORE, or NOBLOCK|SNDMORE. 
1098         copy : bool 
1099             Should the message be sent in a copying or non-copying manner. 
1100         track : bool 
1101             Should the message be tracked for notification that ZMQ has 
1102             finished with it? (ignored if copy=True) 
1103  
1104         Returns 
1105         ------- 
1106         None : if `copy` or not track 
1107             None if message was sent, raises an exception otherwise. 
1108         MessageTracker : if track and not copy 
1109             a MessageTracker object, whose `done` property will 
1110             be False until the send is completed. 
1111  
1112         Raises 
1113         ------ 
1114         TypeError 
1115             If a unicode object is passed 
1116         ValueError 
1117             If `track=True`, but an untracked Frame is passed. 
1118         ZMQError 
1119             for any of the reasons zmq_msg_send might fail (including 
1120             if NOBLOCK is set and the outgoing queue is full). 
1121  
1122         """
1123         _check_closed(self)
1124 
1125         if isinstance(data, str):
1126             raise TypeError("unicode not allowed, use send_string")
1127 
1128         if copy and not isinstance(data, Frame):
1129             return _send_copy(self.handle, data, flags)
1130         else:
1131             if isinstance(data, Frame):
1132                 if track and not data.tracker:
1133                     raise ValueError('Not a tracked message')
1134                 msg = data
1135             else:
1136                 if self.copy_threshold:
1137                     buf = memoryview(data)
1138                     nbytes: size_t = buf.nbytes
1139                     copy_threshold: size_t = self.copy_threshold
1140                     # always copy messages smaller than copy_threshold
1141                     if nbytes < copy_threshold:
1142                         _send_copy(self.handle, buf, flags)
1143                         return zmq._FINISHED_TRACKER
1144                 msg = Frame(data, track=track, copy_threshold=self.copy_threshold)
1145             return _send_frame(self.handle, msg, flags)

1193     def recv_into(self, buffer, /, *, nbytes=0, flags=0) -> C.int:
1194         """ 
1195         Receive up to nbytes bytes from the socket, 
1196         storing the data into a buffer rather than allocating a new Frame. 
1197  
1198         The next message frame can be discarded by receiving into an empty buffer:: 
1199  
1200             sock.recv_into(bytearray()) 
1201  
1202         .. versionadded:: 26.4 
1203  
1204         Parameters 
1205         ---------- 
1206         buffer : memoryview 
1207             Any object providing the buffer interface (i.e. `memoryview(buffer)` works), 
1208             where the memoryview is contiguous and writable. 
1209         nbytes: int, default=0 
1210             The maximum number of bytes to receive. 
1211             If nbytes is not specified (or 0), receive up to the size available in the given buffer. 
1212             If the next frame is larger than this, the frame will be truncated and message content discarded. 
1213         flags: int, default=0 
1214             See `socket.recv` 
1215  
1216         Returns 
1217         ------- 
1218         bytes_received: int 
1219             Returns the number of bytes received. 
1220             This is always the size of the received frame. 
1221             If the returned `bytes_received` is larger than `nbytes` (or size of `buffer` if `nbytes=0`), 
1222             the message has been truncated and the rest of the frame discarded. 
1223             Truncated data cannot be recovered. 
1224  
1225         Raises 
1226         ------ 
1227         ZMQError 
1228             for any of the reasons `zmq_recv` might fail. 
1229         BufferError 
1230             for invalid buffers, such as readonly or not contiguous. 
1231         """
1232         c_flags: C.int = flags
1233         _check_closed(self)
1234         c_nbytes: size_t = nbytes
1235         if c_nbytes < 0:
1236             raise ValueError(f"{nbytes=} must be non-negative")
1237         view = memoryview(buffer)
1238         c_data = declare(pointer(C.void))
1239         view_bytes: C.size_t = _asbuffer(view, address(c_data), True)
1240         if nbytes == 0:
1241             c_nbytes = view_bytes
1242         elif c_nbytes > view_bytes:
1243             raise ValueError(f"{nbytes=} too big for memoryview of {view_bytes}B")
1244 
1245         # call zmq_recv, with retries
1246         while True:
1247             with nogil:
1248                 rc: C.int = zmq_recv(self.handle, c_data, c_nbytes, c_flags)
1249             try:
1250                 _check_rc(rc)
1251             except InterruptedSystemCall:
1252                 continue
1253             else:
1254                 return rc

1540 def zmq_poll(sockets, timeout: C.int = -1):
1541     """zmq_poll(sockets, timeout=-1) 
1542  
1543     Poll a set of 0MQ sockets, native file descs. or sockets. 
1544  
1545     Parameters 
1546     ---------- 
1547     sockets : list of tuples of (socket, flags) 
1548         Each element of this list is a two-tuple containing a socket 
1549         and a flags. The socket may be a 0MQ socket or any object with 
1550         a ``fileno()`` method. The flags can be zmq.POLLIN (for detecting 
1551         for incoming messages), zmq.POLLOUT (for detecting that send is OK) 
1552         or zmq.POLLIN|zmq.POLLOUT for detecting both. 
1553     timeout : int 
1554         The number of milliseconds to poll for. Negative means no timeout. 
1555     """
1556     rc: C.int
1557     i: C.int
1558     fileno: fd_t
1559     events: C.int
1560     pollitems: pointer(zmq_pollitem_t) = NULL
1561     nsockets: C.int = len(sockets)
1562 
1563     if nsockets == 0:
1564         return []
1565 
1566     pollitems = cast(pointer(zmq_pollitem_t), malloc(nsockets * sizeof(zmq_pollitem_t)))
1567     if pollitems == NULL:
1568         raise MemoryError("Could not allocate poll items")
1569 
1570     for i in range(nsockets):
1571         s, events = sockets[i]
1572         if isinstance(s, Socket):
1573             pollitems[i].socket = cast(Socket, s).handle
1574             pollitems[i].fd = 0
1575             pollitems[i].events = events
1576             pollitems[i].revents = 0
1577         elif isinstance(s, int):
1578             fileno = s
1579             pollitems[i].socket = NULL
1580             pollitems[i].fd = fileno
1581             pollitems[i].events = events
1582             pollitems[i].revents = 0
1583         elif hasattr(s, 'fileno'):
1584             try:
1585                 fileno = int(s.fileno())
1586             except Exception:
1587                 free(pollitems)
1588                 raise ValueError('fileno() must return a valid integer fd')
1589             else:
1590                 pollitems[i].socket = NULL
1591                 pollitems[i].fd = fileno
1592                 pollitems[i].events = events
1593                 pollitems[i].revents = 0
1594         else:
1595             free(pollitems)
1596             raise TypeError(
1597                 "Socket must be a 0MQ socket, an integer fd or have "
1598                 f"a fileno() method: {s!r}"
1599             )
1600 
1601     ms_passed: C.int = 0
1602     tic: C.int
1603     try:
1604         while True:
1605             start: C.int = monotonic()
1606             with nogil:
1607                 rc = zmq_poll_c(pollitems, nsockets, timeout)
1608             try:
1609                 _check_rc(rc)
1610             except InterruptedSystemCall:
1611                 if timeout > 0:
1612                     tic = monotonic()
1613                     ms_passed = int(1000 * (tic - start))
1614                     if ms_passed < 0:
1615                         # don't allow negative ms_passed,
1616                         # which can happen on old Python versions without time.monotonic.
1617                         warnings.warn(
1618                             f"Negative elapsed time for interrupted poll: {ms_passed}."
1619                             "  Did the clock change?",
1620                             RuntimeWarning,
1621                         )
1622                         # treat this case the same as no time passing,
1623                         # since it should be rare and not happen twice in a row.
1624                         ms_passed = 0
1625                     timeout = max(0, timeout - ms_passed)
1626                 continue
1627             else:
1628                 break
1629     except Exception:
1630         free(pollitems)
1631         raise
1632 
1633     results = []
1634     for i in range(nsockets):
1635         revents = pollitems[i].revents
1636         # for compatibility with select.poll:
1637         # - only return sockets with non-zero status
1638         # - return the fd for plain sockets
1639         if revents > 0:
1640             if pollitems[i].socket != NULL:
1641                 s = sockets[i][0]
1642             else:
1643                 s = pollitems[i].fd
1644             results.append((s, revents))
1645 
1646     free(pollitems)
1647     return results

1732 # but supports ROUTER-ROUTER devices
1733 @cfunc
1734 @inline
1735 @nogil
1736 def _mq_relay(
1737     in_socket: p_void,
1738     out_socket: p_void,
1739     side_socket: p_void,
1740     msg: zmq_msg_t,
1741     side_msg: zmq_msg_t,
1742     id_msg: zmq_msg_t,
1743     swap_ids: bint,
1744 ) -> C.int:
1745     rc: C.int
1746     flags: C.int
1747     flagsz = declare(size_t)
1748     more = declare(int)
1749     flagsz = sizeof(int)
1750 
1751     if swap_ids:  # both router, must send second identity first
1752         # recv two ids into msg, id_msg
1753         rc = zmq_msg_recv(address(msg), in_socket, 0)
1754         if rc < 0:
1755             return rc
1756 
1757         rc = zmq_msg_recv(address(id_msg), in_socket, 0)
1758         if rc < 0:
1759             return rc
1760 
1761         # send second id (id_msg) first
1762         # !!!! always send a copy before the original !!!!
1763         rc = zmq_msg_copy(address(side_msg), address(id_msg))
1764         if rc < 0:
1765             return rc
1766         rc = zmq_msg_send(address(side_msg), out_socket, ZMQ_SNDMORE)
1767         if rc < 0:
1768             return rc
1769         rc = zmq_msg_send(address(id_msg), side_socket, ZMQ_SNDMORE)
1770         if rc < 0:
1771             return rc
1772         # send first id (msg) second
1773         rc = zmq_msg_copy(address(side_msg), address(msg))
1774         if rc < 0:
1775             return rc
1776         rc = zmq_msg_send(address(side_msg), out_socket, ZMQ_SNDMORE)
1777         if rc < 0:
1778             return rc
1779         rc = zmq_msg_send(address(msg), side_socket, ZMQ_SNDMORE)
1780         if rc < 0:
1781             return rc
1782     while True:
1783         rc = zmq_msg_recv(address(msg), in_socket, 0)
1784         if rc < 0:
1785             return rc
1786         # assert (rc == 0)
1787         rc = zmq_getsockopt(in_socket, ZMQ_RCVMORE, address(more), address(flagsz))
1788         if rc < 0:
1789             return rc
1790         flags = 0
1791         if more:
1792             flags |= ZMQ_SNDMORE
1793 
1794         rc = zmq_msg_copy(address(side_msg), address(msg))
1795         if rc < 0:
1796             return rc
1797         if flags:
1798             rc = zmq_msg_send(address(side_msg), out_socket, flags)
1799             if rc < 0:
1800                 return rc
1801             # only SNDMORE for side-socket
1802             rc = zmq_msg_send(address(msg), side_socket, ZMQ_SNDMORE)
1803             if rc < 0:
1804                 return rc
1805         else:
1806             rc = zmq_msg_send(address(side_msg), out_socket, 0)
1807             if rc < 0:
1808                 return rc
1809             rc = zmq_msg_send(address(msg), side_socket, 0)
1810             if rc < 0:
1811                 return rc
1812             break
1813     return rc

1816 @cfunc
1817 @inline
1818 @nogil
1819 def _mq_inline(
1820     in_socket: p_void,
1821     out_socket: p_void,
1822     side_socket: p_void,
1823     in_msg_ptr: pointer(zmq_msg_t),
1824     out_msg_ptr: pointer(zmq_msg_t),
1825     swap_ids: bint,
1826 ) -> C.int:
1827     """ 
1828     inner C function for monitored_queue 
1829     """
1830 
1831     msg: zmq_msg_t = declare(zmq_msg_t)
1832     rc: C.int = zmq_msg_init(address(msg))
1833     id_msg = declare(zmq_msg_t)
1834     rc = zmq_msg_init(address(id_msg))
1835     if rc < 0:
1836         return rc
1837     side_msg = declare(zmq_msg_t)
1838     rc = zmq_msg_init(address(side_msg))
1839     if rc < 0:
1840         return rc
1841 
1842     items = declare(zmq_pollitem_t[2])
1843     items[0].socket = in_socket
1844     items[0].events = ZMQ_POLLIN
1845     items[0].fd = items[0].revents = 0
1846     items[1].socket = out_socket
1847     items[1].events = ZMQ_POLLIN
1848     items[1].fd = items[1].revents = 0
1849 
1850     while True:
1851         # wait for the next message to process
1852         rc = zmq_poll_c(address(items[0]), 2, -1)
1853         if rc < 0:
1854             return rc
1855         if items[0].revents & ZMQ_POLLIN:
1856             # send in_prefix to side socket
1857             rc = zmq_msg_copy(address(side_msg), in_msg_ptr)
1858             if rc < 0:
1859                 return rc
1860             rc = zmq_msg_send(address(side_msg), side_socket, ZMQ_SNDMORE)
1861             if rc < 0:
1862                 return rc
1863             # relay the rest of the message
1864             rc = _mq_relay(
1865                 in_socket, out_socket, side_socket, msg, side_msg, id_msg, swap_ids
1866             )
1867             if rc < 0:
1868                 return rc
1869         if items[1].revents & ZMQ_POLLIN:
1870             # send out_prefix to side socket
1871             rc = zmq_msg_copy(address(side_msg), out_msg_ptr)
1872             if rc < 0:
1873                 return rc
1874             rc = zmq_msg_send(address(side_msg), side_socket, ZMQ_SNDMORE)
1875             if rc < 0:
1876                 return rc
1877             # relay the rest of the message
1878             rc = _mq_relay(
1879                 out_socket, in_socket, side_socket, msg, side_msg, id_msg, swap_ids
1880             )
1881             if rc < 0:
1882                 return rc
1883     return rc

1886 def monitored_queue(
1887     in_socket: Socket,
1888     out_socket: Socket,
1889     mon_socket: Socket,
1890     in_prefix: bytes = b'in',
1891     out_prefix: bytes = b'out',
1892 ):
1893     """ 
1894     Start a monitored queue device. 
1895  
1896     A monitored queue is very similar to the zmq.proxy device (monitored queue came first). 
1897  
1898     Differences from zmq.proxy: 
1899  
1900     - monitored_queue supports both in and out being ROUTER sockets 
1901       (via swapping IDENTITY prefixes). 
1902     - monitor messages are prefixed, making in and out messages distinguishable. 
1903  
1904     Parameters 
1905     ---------- 
1906     in_socket : zmq.Socket 
1907         One of the sockets to the Queue. Its messages will be prefixed with 
1908         'in'. 
1909     out_socket : zmq.Socket 
1910         One of the sockets to the Queue. Its messages will be prefixed with 
1911         'out'. The only difference between in/out socket is this prefix. 
1912     mon_socket : zmq.Socket 
1913         This socket sends out every message received by each of the others 
1914         with an in/out prefix specifying which one it was. 
1915     in_prefix : str 
1916         Prefix added to broadcast messages from in_socket. 
1917     out_prefix : str 
1918         Prefix added to broadcast messages from out_socket. 
1919     """
1920     ins: p_void = in_socket.handle
1921     outs: p_void = out_socket.handle
1922     mons: p_void = mon_socket.handle
1923     in_msg = declare(zmq_msg_t)
1924     out_msg = declare(zmq_msg_t)
1925     swap_ids: bint
1926     msg_c: p_void = NULL
1927     msg_c_len = declare(Py_ssize_t)
1928     rc: C.int
1929 
1930     # force swap_ids if both ROUTERs
1931     swap_ids = in_socket.type == ZMQ_ROUTER and out_socket.type == ZMQ_ROUTER
1932 
1933     # build zmq_msg objects from str prefixes
1934     msg_c_len = _asbuffer(in_prefix, address(msg_c))
1935     rc = zmq_msg_init_size(address(in_msg), msg_c_len)
1936     _check_rc(rc)
1937 
1938     memcpy(zmq_msg_data(address(in_msg)), msg_c, zmq_msg_size(address(in_msg)))
1939 
1940     msg_c_len = _asbuffer(out_prefix, address(msg_c))
1941 
1942     rc = zmq_msg_init_size(address(out_msg), msg_c_len)
1943     _check_rc(rc)
1944 
1945     while True:
1946         with nogil:
1947             memcpy(
1948                 zmq_msg_data(address(out_msg)), msg_c, zmq_msg_size(address(out_msg))
1949             )
1950             rc = _mq_inline(
1951                 ins, outs, mons, address(in_msg), address(out_msg), swap_ids
1952             )
1953         try:
1954             _check_rc(rc)
1955         except InterruptedSystemCall:
1956             continue
1957         else:
1958             break
1959     return rc
