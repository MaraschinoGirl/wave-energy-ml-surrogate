16   def _process_predict_proba(*, y_pred, target_type, classes, pos_label):
17       """Get the response values when the response method is `predict_proba`. 
18    
19       This function process the `y_pred` array in the binary and multi-label cases. 
20       In the binary case, it selects the column corresponding to the positive 
21       class. In the multi-label case, it stacks the predictions if they are not 
22       in the "compressed" format `(n_samples, n_outputs)`. 
23    
24       Parameters 
25       ---------- 
26       y_pred : ndarray 
27           Output of `estimator.predict_proba`. The shape depends on the target type: 
28    
29           - for binary classification, it is a 2d array of shape `(n_samples, 2)`; 
30           - for multiclass classification, it is a 2d array of shape 
31             `(n_samples, n_classes)`; 
32           - for multilabel classification, it is either a list of 2d arrays of shape 
33             `(n_samples, 2)` (e.g. `RandomForestClassifier` or `KNeighborsClassifier`) or 
34             an array of shape `(n_samples, n_outputs)` (e.g. `MLPClassifier` or 
35             `RidgeClassifier`). 
36    
37       target_type : {"binary", "multiclass", "multilabel-indicator"} 
38           Type of the target. 
39    
40       classes : ndarray of shape (n_classes,) or list of such arrays 
41           Class labels as reported by `estimator.classes_`. 
42    
43       pos_label : int, float, bool or str 
44           Only used with binary and multiclass targets. 
45    
46       Returns 
47       ------- 
48       y_pred : ndarray of shape (n_samples,), (n_samples, n_classes) or \ 
49               (n_samples, n_output) 
50           Compressed predictions format as requested by the metrics. 
51       """
52       if target_type == "binary" and y_pred.shape[1] < 2:
53           # We don't handle classifiers trained on a single class.
54           raise ValueError(
55               f"Got predict_proba of shape {y_pred.shape}, but need "
56               "classifier with two classes."
57           )
58   
59       if target_type == "binary":
60           col_idx = np.flatnonzero(classes == pos_label)[0]
61           return y_pred[:, col_idx]
62       elif target_type == "multilabel-indicator":
63           # Use a compress format of shape `(n_samples, n_output)`.
64           # Only `MLPClassifier` and `RidgeClassifier` return an array of shape
65           # `(n_samples, n_outputs)`.
66           if isinstance(y_pred, list):
67               # list of arrays of shape `(n_samples, 2)`
68               return np.vstack([p[:, -1] for p in y_pred]).T
69           else:
70               # array of shape `(n_samples, n_outputs)`
71               return y_pred
72   
73       return y_pred

76   def _process_decision_function(*, y_pred, target_type, classes, pos_label):
77       """Get the response values when the response method is `decision_function`. 
78    
79       This function process the `y_pred` array in the binary and multi-label cases. 
80       In the binary case, it inverts the sign of the score if the positive label 
81       is not `classes[1]`. In the multi-label case, it stacks the predictions if 
82       they are not in the "compressed" format `(n_samples, n_outputs)`. 
83    
84       Parameters 
85       ---------- 
86       y_pred : ndarray 
87           Output of `estimator.decision_function`. The shape depends on the target type: 
88    
89           - for binary classification, it is a 1d array of shape `(n_samples,)` where the 
90             sign is assuming that `classes[1]` is the positive class; 
91           - for multiclass classification, it is a 2d array of shape 
92             `(n_samples, n_classes)`; 
93           - for multilabel classification, it is a 2d array of shape `(n_samples, 
94             n_outputs)`. 
95    
96       target_type : {"binary", "multiclass", "multilabel-indicator"} 
97           Type of the target. 
98    
99       classes : ndarray of shape (n_classes,) or list of such arrays 
100          Class labels as reported by `estimator.classes_`. 
101   
102      pos_label : int, float, bool or str 
103          Only used with binary and multiclass targets. 
104   
105      Returns 
106      ------- 
107      y_pred : ndarray of shape (n_samples,), (n_samples, n_classes) or \ 
108              (n_samples, n_output) 
109          Compressed predictions format as requested by the metrics. 
110      """
111      if target_type == "binary" and pos_label == classes[0]:
112          return -1 * y_pred
113      return y_pred

116  def _get_response_values(
117      estimator,
118      X,
119      response_method,
120      pos_label=None,
121      return_response_method_used=False,
122  ):
123      """Compute the response values of a classifier, an outlier detector, or a regressor. 
124   
125      The response values are predictions such that it follows the following shape: 
126   
127      - for binary classification, it is a 1d array of shape `(n_samples,)`; 
128      - for multiclass classification, it is a 2d array of shape `(n_samples, n_classes)`; 
129      - for multilabel classification, it is a 2d array of shape `(n_samples, n_outputs)`; 
130      - for outlier detection, it is a 1d array of shape `(n_samples,)`; 
131      - for regression, it is a 1d array of shape `(n_samples,)`. 
132   
133      If `estimator` is a binary classifier, also return the label for the 
134      effective positive class. 
135   
136      This utility is used primarily in the displays and the scikit-learn scorers. 
137   
138      .. versionadded:: 1.3 
139   
140      Parameters 
141      ---------- 
142      estimator : estimator instance 
143          Fitted classifier, outlier detector, or regressor or a 
144          fitted :class:`~sklearn.pipeline.Pipeline` in which the last estimator is a 
145          classifier, an outlier detector, or a regressor. 
146   
147      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
148          Input values. 
149   
150      response_method : {"predict_proba", "predict_log_proba", "decision_function", \ 
151              "predict"} or list of such str 
152          Specifies the response method to use get prediction from an estimator 
153          (i.e. :term:`predict_proba`, :term:`predict_log_proba`, 
154          :term:`decision_function` or :term:`predict`). Possible choices are: 
155   
156          - if `str`, it corresponds to the name to the method to return; 
157          - if a list of `str`, it provides the method names in order of 
158            preference. The method returned corresponds to the first method in 
159            the list and which is implemented by `estimator`. 
160   
161      pos_label : int, float, bool or str, default=None 
162          The class considered as the positive class when computing 
163          the metrics. If `None` and target is 'binary', `estimators.classes_[1]` is 
164          considered as the positive class. 
165   
166      return_response_method_used : bool, default=False 
167          Whether to return the response method used to compute the response 
168          values. 
169   
170          .. versionadded:: 1.4 
171   
172      Returns 
173      ------- 
174      y_pred : ndarray of shape (n_samples,), (n_samples, n_classes) or \ 
175              (n_samples, n_outputs) 
176          Target scores calculated from the provided `response_method` 
177          and `pos_label`. 
178   
179      pos_label : int, float, bool, str or None 
180          The class considered as the positive class when computing 
181          the metrics. Returns `None` if `estimator` is a regressor or an outlier 
182          detector. 
183   
184      response_method_used : str 
185          The response method used to compute the response values. Only returned 
186          if `return_response_method_used` is `True`. 
187   
188          .. versionadded:: 1.4 
189   
190      Raises 
191      ------ 
192      ValueError 
193          If `pos_label` is not a valid label. 
194          If the shape of `y_pred` is not consistent for binary classifier. 
195          If the response method can be applied to a classifier only and 
196          `estimator` is a regressor. 
197      """
198      from sklearn.base import is_classifier, is_outlier_detector  # noqa
199  
200      if is_classifier(estimator):
201          prediction_method = _check_response_method(estimator, response_method)
202          classes = estimator.classes_
203          target_type = type_of_target(classes)
204  
205          if target_type in ("binary", "multiclass"):
206              if pos_label is not None and pos_label not in classes.tolist():
207                  raise ValueError(
208                      f"pos_label={pos_label} is not a valid label: It should be "
209                      f"one of {classes}"
210                  )
211              elif pos_label is None and target_type == "binary":
212                  pos_label = classes[-1]
213  
214          y_pred = prediction_method(X)
215  
216          if prediction_method.__name__ in ("predict_proba", "predict_log_proba"):
217              y_pred = _process_predict_proba(
218                  y_pred=y_pred,
219                  target_type=target_type,
220                  classes=classes,
221                  pos_label=pos_label,
222              )
223          elif prediction_method.__name__ == "decision_function":
224              y_pred = _process_decision_function(
225                  y_pred=y_pred,
226                  target_type=target_type,
227                  classes=classes,
228                  pos_label=pos_label,
229              )
230      elif is_outlier_detector(estimator):
231          prediction_method = _check_response_method(estimator, response_method)
232          y_pred, pos_label = prediction_method(X), None
233      else:  # estimator is a regressor
234          if response_method != "predict":
235              raise ValueError(
236                  f"{estimator.__class__.__name__} should either be a classifier to be "
237                  f"used with response_method={response_method} or the response_method "
238                  "should be 'predict'. Got a regressor with response_method="
239                  f"{response_method} instead."
240              )
241          prediction_method = estimator.predict
242          y_pred, pos_label = prediction_method(X), None
243  
244      if return_response_method_used:
245          return y_pred, pos_label, prediction_method.__name__
246      return y_pred, pos_label

249  def _get_response_values_binary(
250      estimator, X, response_method, pos_label=None, return_response_method_used=False
251  ):
252      """Compute the response values of a binary classifier. 
253   
254      Parameters 
255      ---------- 
256      estimator : estimator instance 
257          Fitted classifier or a fitted :class:`~sklearn.pipeline.Pipeline` 
258          in which the last estimator is a binary classifier. 
259   
260      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
261          Input values. 
262   
263      response_method : {'auto', 'predict_proba', 'decision_function'} 
264          Specifies whether to use :term:`predict_proba` or 
265          :term:`decision_function` as the target response. If set to 'auto', 
266          :term:`predict_proba` is tried first and if it does not exist 
267          :term:`decision_function` is tried next. 
268   
269      pos_label : int, float, bool or str, default=None 
270          The class considered as the positive class when computing 
271          the metrics. By default, `estimators.classes_[1]` is 
272          considered as the positive class. 
273   
274      return_response_method_used : bool, default=False 
275          Whether to return the response method used to compute the response 
276          values. 
277   
278          .. versionadded:: 1.5 
279   
280      Returns 
281      ------- 
282      y_pred : ndarray of shape (n_samples,) 
283          Target scores calculated from the provided response_method 
284          and pos_label. 
285   
286      pos_label : int, float, bool or str 
287          The class considered as the positive class when computing 
288          the metrics. 
289   
290      response_method_used : str 
291          The response method used to compute the response values. Only returned 
292          if `return_response_method_used` is `True`. 
293   
294          .. versionadded:: 1.5 
295      """
296      classification_error = "Expected 'estimator' to be a binary classifier."
297  
298      check_is_fitted(estimator)
299      if not is_classifier(estimator):
300          raise ValueError(
301              classification_error + f" Got {estimator.__class__.__name__} instead."
302          )
303      elif len(estimator.classes_) != 2:
304          raise ValueError(
305              classification_error + f" Got {len(estimator.classes_)} classes instead."
306          )
307  
308      if response_method == "auto":
309          response_method = ["predict_proba", "decision_function"]
310  
311      return _get_response_values(
312          estimator,
313          X,
314          response_method,
315          pos_label=pos_label,
316          return_response_method_used=return_response_method_used,
317      )
