1    """Statistical transformations for visualization. 
2     
3    This module is currently private, but is being written to eventually form part 
4    of the public API. 
5     
6    The classes should behave roughly in the style of scikit-learn. 
7     
8    - All data-independent parameters should be passed to the class constructor. 
9    - Each class should implement a default transformation that is exposed through 
10     __call__. These are currently written for vector arguments, but I think 
11     consuming a whole `plot_data` DataFrame and return it with transformed 
12     variables would make more sense. 
13   - Some class have data-dependent preprocessing that should be cached and used 
14     multiple times (think defining histogram bins off all data and then counting 
15     observations within each bin multiple times per data subsets). These currently 
16     have unique names, but it would be good to have a common name. Not quite 
17     `fit`, but something similar. 
18   - Alternatively, the transform interface could take some information about grouping 
19     variables and do a groupby internally. 
20   - Some classes should define alternate transforms that might make the most sense 
21     with a different function. For example, KDE usually evaluates the distribution 
22     on a regular grid, but it would be useful for it to transform at the actual 
23     datapoints. Then again, this could be controlled by a parameter at  the time of 
24     class instantiation. 
25    
26   """

33       _no_scipy = False

42   class KDE:
43       """Univariate and bivariate kernel density estimator."""
44       def __init__(
45           self, *,
46           bw_method=None,
47           bw_adjust=1,
48           gridsize=200,
49           cut=3,
50           clip=None,
51           cumulative=False,
52       ):
53           """Initialize the estimator with its parameters. 
54    
55           Parameters 
56           ---------- 
57           bw_method : string, scalar, or callable, optional 
58               Method for determining the smoothing bandwidth to use; passed to 
59               :class:`scipy.stats.gaussian_kde`. 
60           bw_adjust : number, optional 
61               Factor that multiplicatively scales the value chosen using 
62               ``bw_method``. Increasing will make the curve smoother. See Notes. 
63           gridsize : int, optional 
64               Number of points on each dimension of the evaluation grid. 
65           cut : number, optional 
66               Factor, multiplied by the smoothing bandwidth, that determines how 
67               far the evaluation grid extends past the extreme datapoints. When 
68               set to 0, truncate the curve at the data limits. 
69           clip : pair of numbers or None, or a pair of such pairs 
70               Do not evaluate the density outside of these limits. 
71           cumulative : bool, optional 
72               If True, estimate a cumulative distribution function. Requires scipy. 
73    
74           """
75           if clip is None:
76               clip = None, None
77   
78           self.bw_method = bw_method
79           self.bw_adjust = bw_adjust
80           self.gridsize = gridsize
81           self.cut = cut
82           self.clip = clip
83           self.cumulative = cumulative
84   
85           if cumulative and _no_scipy:
86               raise RuntimeError("Cumulative KDE evaluation requires scipy")
87   
88           self.support = None
89   
90       def _define_support_grid(self, x, bw, cut, clip, gridsize):
91           """Create the grid of evaluation points depending for vector x."""
92           clip_lo = -np.inf if clip[0] is None else clip[0]
93           clip_hi = +np.inf if clip[1] is None else clip[1]
94           gridmin = max(x.min() - bw * cut, clip_lo)
95           gridmax = min(x.max() + bw * cut, clip_hi)
96           return np.linspace(gridmin, gridmax, gridsize)
97   
98       def _define_support_univariate(self, x, weights):
99           """Create a 1D grid of evaluation points."""
100          kde = self._fit(x, weights)
101          bw = np.sqrt(kde.covariance.squeeze())
102          grid = self._define_support_grid(
103              x, bw, self.cut, self.clip, self.gridsize
104          )
105          return grid
106  
107      def _define_support_bivariate(self, x1, x2, weights):
108          """Create a 2D grid of evaluation points."""
109          clip = self.clip
110          if clip[0] is None or np.isscalar(clip[0]):
111              clip = (clip, clip)
112  
113          kde = self._fit([x1, x2], weights)
114          bw = np.sqrt(np.diag(kde.covariance).squeeze())
115  
116          grid1 = self._define_support_grid(
117              x1, bw[0], self.cut, clip[0], self.gridsize
118          )
119          grid2 = self._define_support_grid(
120              x2, bw[1], self.cut, clip[1], self.gridsize
121          )
122  
123          return grid1, grid2
124  
125      def define_support(self, x1, x2=None, weights=None, cache=True):
126          """Create the evaluation grid for a given data set."""
127          if x2 is None:
128              support = self._define_support_univariate(x1, weights)
129          else:
130              support = self._define_support_bivariate(x1, x2, weights)
131  
132          if cache:
133              self.support = support
134  
135          return support
136  
137      def _fit(self, fit_data, weights=None):
138          """Fit the scipy kde while adding bw_adjust logic and version check."""
139          fit_kws = {"bw_method": self.bw_method}
140          if weights is not None:
141              fit_kws["weights"] = weights
142  
143          kde = gaussian_kde(fit_data, **fit_kws)
144          kde.set_bandwidth(kde.factor * self.bw_adjust)
145  
146          return kde
147  
148      def _eval_univariate(self, x, weights=None):
149          """Fit and evaluate a univariate on univariate data."""
150          support = self.support
151          if support is None:
152              support = self.define_support(x, cache=False)
153  
154          kde = self._fit(x, weights)
155  
156          if self.cumulative:
157              s_0 = support[0]
158              density = np.array([
159                  kde.integrate_box_1d(s_0, s_i) for s_i in support
160              ])
161          else:
162              density = kde(support)
163  
164          return density, support
165  
166      def _eval_bivariate(self, x1, x2, weights=None):
167          """Fit and evaluate a univariate on bivariate data."""
168          support = self.support
169          if support is None:
170              support = self.define_support(x1, x2, cache=False)
171  
172          kde = self._fit([x1, x2], weights)
173  
174          if self.cumulative:
175  
176              grid1, grid2 = support
177              density = np.zeros((grid1.size, grid2.size))
178              p0 = grid1.min(), grid2.min()
179              for i, xi in enumerate(grid1):
180                  for j, xj in enumerate(grid2):
181                      density[i, j] = kde.integrate_box(p0, (xi, xj))
182  
183          else:
184  
185              xx1, xx2 = np.meshgrid(*support)
186              density = kde([xx1.ravel(), xx2.ravel()]).reshape(xx1.shape)
187  
188          return density, support
189  
190      def __call__(self, x1, x2=None, weights=None):
191          """Fit and evaluate on univariate or bivariate data."""
192          if x2 is None:
193              return self._eval_univariate(x1, weights)
194          else:
195              return self._eval_bivariate(x1, x2, weights)

198  # Note: we no longer use this for univariate histograms in histplot,
199  # preferring _stats.Hist. We'll deprecate this once we have a bivariate Stat class.
200  class Histogram:
201      """Univariate and bivariate histogram estimator."""
202      def __init__(
203          self,
204          stat="count",
205          bins="auto",
206          binwidth=None,
207          binrange=None,
208          discrete=False,
209          cumulative=False,
210      ):
211          """Initialize the estimator with its parameters. 
212   
213          Parameters 
214          ---------- 
215          stat : str 
216              Aggregate statistic to compute in each bin. 
217   
218              - `count`: show the number of observations in each bin 
219              - `frequency`: show the number of observations divided by the bin width 
220              - `probability` or `proportion`: normalize such that bar heights sum to 1 
221              - `percent`: normalize such that bar heights sum to 100 
222              - `density`: normalize such that the total area of the histogram equals 1 
223   
224          bins : str, number, vector, or a pair of such values 
225              Generic bin parameter that can be the name of a reference rule, 
226              the number of bins, or the breaks of the bins. 
227              Passed to :func:`numpy.histogram_bin_edges`. 
228          binwidth : number or pair of numbers 
229              Width of each bin, overrides ``bins`` but can be used with 
230              ``binrange``. 
231          binrange : pair of numbers or a pair of pairs 
232              Lowest and highest value for bin edges; can be used either 
233              with ``bins`` or ``binwidth``. Defaults to data extremes. 
234          discrete : bool or pair of bools 
235              If True, set ``binwidth`` and ``binrange`` such that bin 
236              edges cover integer values in the dataset. 
237          cumulative : bool 
238              If True, return the cumulative statistic. 
239   
240          """
241          stat_choices = [
242              "count", "frequency", "density", "probability", "proportion", "percent",
243          ]
244          _check_argument("stat", stat_choices, stat)
245  
246          self.stat = stat
247          self.bins = bins
248          self.binwidth = binwidth
249          self.binrange = binrange
250          self.discrete = discrete
251          self.cumulative = cumulative
252  
253          self.bin_kws = None
254  
255      def _define_bin_edges(self, x, weights, bins, binwidth, binrange, discrete):
256          """Inner function that takes bin parameters as arguments."""
257          if binrange is None:
258              start, stop = x.min(), x.max()
259          else:
260              start, stop = binrange
261  
262          if discrete:
263              bin_edges = np.arange(start - .5, stop + 1.5)
264          elif binwidth is not None:
265              step = binwidth
266              bin_edges = np.arange(start, stop + step, step)
267              # Handle roundoff error (maybe there is a less clumsy way?)
268              if bin_edges.max() < stop or len(bin_edges) < 2:
269                  bin_edges = np.append(bin_edges, bin_edges.max() + step)
270          else:
271              bin_edges = np.histogram_bin_edges(
272                  x, bins, binrange, weights,
273              )
274          return bin_edges
275  
276      def define_bin_params(self, x1, x2=None, weights=None, cache=True):
277          """Given data, return numpy.histogram parameters to define bins."""
278          if x2 is None:
279  
280              bin_edges = self._define_bin_edges(
281                  x1, weights, self.bins, self.binwidth, self.binrange, self.discrete,
282              )
283  
284              if isinstance(self.bins, (str, Number)):
285                  n_bins = len(bin_edges) - 1
286                  bin_range = bin_edges.min(), bin_edges.max()
287                  bin_kws = dict(bins=n_bins, range=bin_range)
288              else:
289                  bin_kws = dict(bins=bin_edges)
290  
291          else:
292  
293              bin_edges = []
294              for i, x in enumerate([x1, x2]):
295  
296                  # Resolve out whether bin parameters are shared
297                  # or specific to each variable
298  
299                  bins = self.bins
300                  if not bins or isinstance(bins, (str, Number)):
301                      pass
302                  elif isinstance(bins[i], str):
303                      bins = bins[i]
304                  elif len(bins) == 2:
305                      bins = bins[i]
306  
307                  binwidth = self.binwidth
308                  if binwidth is None:
309                      pass
310                  elif not isinstance(binwidth, Number):
311                      binwidth = binwidth[i]
312  
313                  binrange = self.binrange
314                  if binrange is None:
315                      pass
316                  elif not isinstance(binrange[0], Number):
317                      binrange = binrange[i]
318  
319                  discrete = self.discrete
320                  if not isinstance(discrete, bool):
321                      discrete = discrete[i]
322  
323                  # Define the bins for this variable
324  
325                  bin_edges.append(self._define_bin_edges(
326                      x, weights, bins, binwidth, binrange, discrete,
327                  ))
328  
329              bin_kws = dict(bins=tuple(bin_edges))
330  
331          if cache:
332              self.bin_kws = bin_kws
333  
334          return bin_kws
335  
336      def _eval_bivariate(self, x1, x2, weights):
337          """Inner function for histogram of two variables."""
338          bin_kws = self.bin_kws
339          if bin_kws is None:
340              bin_kws = self.define_bin_params(x1, x2, cache=False)
341  
342          density = self.stat == "density"
343  
344          hist, *bin_edges = np.histogram2d(
345              x1, x2, **bin_kws, weights=weights, density=density
346          )
347  
348          area = np.outer(
349              np.diff(bin_edges[0]),
350              np.diff(bin_edges[1]),
351          )
352  
353          if self.stat == "probability" or self.stat == "proportion":
354              hist = hist.astype(float) / hist.sum()
355          elif self.stat == "percent":
356              hist = hist.astype(float) / hist.sum() * 100
357          elif self.stat == "frequency":
358              hist = hist.astype(float) / area
359  
360          if self.cumulative:
361              if self.stat in ["density", "frequency"]:
362                  hist = (hist * area).cumsum(axis=0).cumsum(axis=1)
363              else:
364                  hist = hist.cumsum(axis=0).cumsum(axis=1)
365  
366          return hist, bin_edges
367  
368      def _eval_univariate(self, x, weights):
369          """Inner function for histogram of one variable."""
370          bin_kws = self.bin_kws
371          if bin_kws is None:
372              bin_kws = self.define_bin_params(x, weights=weights, cache=False)
373  
374          density = self.stat == "density"
375          hist, bin_edges = np.histogram(
376              x, **bin_kws, weights=weights, density=density,
377          )
378  
379          if self.stat == "probability" or self.stat == "proportion":
380              hist = hist.astype(float) / hist.sum()
381          elif self.stat == "percent":
382              hist = hist.astype(float) / hist.sum() * 100
383          elif self.stat == "frequency":
384              hist = hist.astype(float) / np.diff(bin_edges)
385  
386          if self.cumulative:
387              if self.stat in ["density", "frequency"]:
388                  hist = (hist * np.diff(bin_edges)).cumsum()
389              else:
390                  hist = hist.cumsum()
391  
392          return hist, bin_edges
393  
394      def __call__(self, x1, x2=None, weights=None):
395          """Count the occurrences in each bin, maybe normalize."""
396          if x2 is None:
397              return self._eval_univariate(x1, weights)
398          else:
399              return self._eval_bivariate(x1, x2, weights)

402  class ECDF:
403      """Univariate empirical cumulative distribution estimator."""
404      def __init__(self, stat="proportion", complementary=False):
405          """Initialize the class with its parameters 
406   
407          Parameters 
408          ---------- 
409          stat : {{"proportion", "percent", "count"}} 
410              Distribution statistic to compute. 
411          complementary : bool 
412              If True, use the complementary CDF (1 - CDF) 
413   
414          """
415          _check_argument("stat", ["count", "percent", "proportion"], stat)
416          self.stat = stat
417          self.complementary = complementary
418  
419      def _eval_bivariate(self, x1, x2, weights):
420          """Inner function for ECDF of two variables."""
421          raise NotImplementedError("Bivariate ECDF is not implemented")
422  
423      def _eval_univariate(self, x, weights):
424          """Inner function for ECDF of one variable."""
425          sorter = x.argsort()
426          x = x[sorter]
427          weights = weights[sorter]
428          y = weights.cumsum()
429  
430          if self.stat in ["percent", "proportion"]:
431              y = y / y.max()
432          if self.stat == "percent":
433              y = y * 100
434  
435          x = np.r_[-np.inf, x]
436          y = np.r_[0, y]
437  
438          if self.complementary:
439              y = y.max() - y
440  
441          return y, x
442  
443      def __call__(self, x1, x2=None, weights=None):
444          """Return proportion or count of observations below each sorted datapoint."""
445          x1 = np.asarray(x1)
446          if weights is None:
447              weights = np.ones_like(x1)
448          else:
449              weights = np.asarray(weights)
450  
451          if x2 is None:
452              return self._eval_univariate(x1, weights)
453          else:
454              return self._eval_bivariate(x1, x2, weights)

457  class EstimateAggregator:
458  
459      def __init__(self, estimator, errorbar=None, **boot_kws):
460          """ 
461          Data aggregator that produces an estimate and error bar interval. 
462   
463          Parameters 
464          ---------- 
465          estimator : callable or string 
466              Function (or method name) that maps a vector to a scalar. 
467          errorbar : string, (string, number) tuple, or callable 
468              Name of errorbar method (either "ci", "pi", "se", or "sd"), or a tuple 
469              with a method name and a level parameter, or a function that maps from a 
470              vector to a (min, max) interval, or None to hide errorbar. See the 
471              :doc:`errorbar tutorial </tutorial/error_bars>` for more information. 
472          boot_kws 
473              Additional keywords are passed to bootstrap when error_method is "ci". 
474   
475          """
476          self.estimator = estimator
477  
478          method, level = _validate_errorbar_arg(errorbar)
479          self.error_method = method
480          self.error_level = level
481  
482          self.boot_kws = boot_kws
483  
484      def __call__(self, data, var):
485          """Aggregate over `var` column of `data` with estimate and error interval."""
486          vals = data[var]
487          if callable(self.estimator):
488              # You would think we could pass to vals.agg, and yet:
489              # https://github.com/mwaskom/seaborn/issues/2943
490              estimate = self.estimator(vals)
491          else:
492              estimate = vals.agg(self.estimator)
493  
494          # Options that produce no error bars
495          if self.error_method is None:
496              err_min = err_max = np.nan
497          elif len(data) <= 1:
498              err_min = err_max = np.nan
499  
500          # Generic errorbars from user-supplied function
501          elif callable(self.error_method):
502              err_min, err_max = self.error_method(vals)
503  
504          # Parametric options
505          elif self.error_method == "sd":
506              half_interval = vals.std() * self.error_level
507              err_min, err_max = estimate - half_interval, estimate + half_interval
508          elif self.error_method == "se":
509              half_interval = vals.sem() * self.error_level
510              err_min, err_max = estimate - half_interval, estimate + half_interval
511  
512          # Nonparametric options
513          elif self.error_method == "pi":
514              err_min, err_max = _percentile_interval(vals, self.error_level)
515          elif self.error_method == "ci":
516              units = data.get("units", None)
517              boots = bootstrap(vals, units=units, func=self.estimator, **self.boot_kws)
518              err_min, err_max = _percentile_interval(boots, self.error_level)
519  
520          return pd.Series({var: estimate, f"{var}min": err_min, f"{var}max": err_max})

523  class WeightedAggregator:
524  
525      def __init__(self, estimator, errorbar=None, **boot_kws):
526          """ 
527          Data aggregator that produces a weighted estimate and error bar interval. 
528   
529          Parameters 
530          ---------- 
531          estimator : string 
532              Function (or method name) that maps a vector to a scalar. Currently 
533              supports only "mean". 
534          errorbar : string or (string, number) tuple 
535              Name of errorbar method or a tuple with a method name and a level parameter. 
536              Currently the only supported method is "ci". 
537          boot_kws 
538              Additional keywords are passed to bootstrap when error_method is "ci". 
539   
540          """
541          if estimator != "mean":
542              # Note that, while other weighted estimators may make sense (e.g. median),
543              # I'm not aware of an implementation in our dependencies. We can add one
544              # in seaborn later, if there is sufficient interest. For now, limit to mean.
545              raise ValueError(f"Weighted estimator must be 'mean', not {estimator!r}.")
546          self.estimator = estimator
547  
548          method, level = _validate_errorbar_arg(errorbar)
549          if method is not None and method != "ci":
550              # As with the estimator, weighted 'sd' or 'pi' error bars may make sense.
551              # But we'll keep things simple for now and limit to (bootstrap) CI.
552              raise ValueError(f"Error bar method must be 'ci', not {method!r}.")
553          self.error_method = method
554          self.error_level = level
555  
556          self.boot_kws = boot_kws
557  
558      def __call__(self, data, var):
559          """Aggregate over `var` column of `data` with estimate and error interval."""
560          vals = data[var]
561          weights = data["weight"]
562  
563          estimate = np.average(vals, weights=weights)
564  
565          if self.error_method == "ci" and len(data) > 1:
566  
567              def error_func(x, w):
568                  return np.average(x, weights=w)
569  
570              boots = bootstrap(vals, weights, func=error_func, **self.boot_kws)
571              err_min, err_max = _percentile_interval(boots, self.error_level)
572  
573          else:
574              err_min = err_max = np.nan
575  
576          return pd.Series({var: estimate, f"{var}min": err_min, f"{var}max": err_max})

579  class LetterValues:
580  
581      def __init__(self, k_depth, outlier_prop, trust_alpha):
582          """ 
583          Compute percentiles of a distribution using various tail stopping rules. 
584   
585          Parameters 
586          ---------- 
587          k_depth: "tukey", "proportion", "trustworthy", or "full" 
588              Stopping rule for choosing tail percentiled to show: 
589   
590              - tukey: Show a similar number of outliers as in a conventional boxplot. 
591              - proportion: Show approximately `outlier_prop` outliers. 
592              - trust_alpha: Use `trust_alpha` level for most extreme tail percentile. 
593   
594          outlier_prop: float 
595              Parameter for `k_depth="proportion"` setting the expected outlier rate. 
596          trust_alpha: float 
597              Parameter for `k_depth="trustworthy"` setting the confidence threshold. 
598   
599          Notes 
600          ----- 
601          Based on the proposal in this paper: 
602          https://vita.had.co.nz/papers/letter-value-plot.pdf 
603   
604          """
605          k_options = ["tukey", "proportion", "trustworthy", "full"]
606          if isinstance(k_depth, str):
607              _check_argument("k_depth", k_options, k_depth)
608          elif not isinstance(k_depth, int):
609              err = (
610                  "The `k_depth` parameter must be either an integer or string "
611                  f"(one of {k_options}), not {k_depth!r}."
612              )
613              raise TypeError(err)
614  
615          self.k_depth = k_depth
616          self.outlier_prop = outlier_prop
617          self.trust_alpha = trust_alpha
618  
619      def _compute_k(self, n):
620  
621          # Select the depth, i.e. number of boxes to draw, based on the method
622          if self.k_depth == "full":
623              # extend boxes to 100% of the data
624              k = int(np.log2(n)) + 1
625          elif self.k_depth == "tukey":
626              # This results with 5-8 points in each tail
627              k = int(np.log2(n)) - 3
628          elif self.k_depth == "proportion":
629              k = int(np.log2(n)) - int(np.log2(n * self.outlier_prop)) + 1
630          elif self.k_depth == "trustworthy":
631              normal_quantile_func = np.vectorize(NormalDist().inv_cdf)
632              point_conf = 2 * normal_quantile_func(1 - self.trust_alpha / 2) ** 2
633              k = int(np.log2(n / point_conf)) + 1
634          else:
635              # Allow having k directly specified as input
636              k = int(self.k_depth)
637  
638          return max(k, 1)
639  
640      def __call__(self, x):
641          """Evaluate the letter values."""
642          k = self._compute_k(len(x))
643          exp = np.arange(k + 1, 1, -1), np.arange(2, k + 2)
644          levels = k + 1 - np.concatenate([exp[0], exp[1][1:]])
645          percentiles = 100 * np.concatenate([0.5 ** exp[0], 1 - 0.5 ** exp[1]])
646          if self.k_depth == "full":
647              percentiles[0] = 0
648              percentiles[-1] = 100
649          values = np.percentile(x, percentiles)
650          fliers = np.asarray(x[(x < values.min()) | (x > values.max())])
651          median = np.percentile(x, 50)
652  
653          return {
654              "k": k,
655              "levels": levels,
656              "percs": percentiles,
657              "values": values,
658              "fliers": fliers,
659              "median": median,
660          }
661

663  def _percentile_interval(data, width):
664      """Return a percentile interval from data of a given width."""
665      edge = (100 - width) / 2
666      percentiles = edge, 100 - edge
667      return np.nanpercentile(data, percentiles)

670  def _validate_errorbar_arg(arg):
671      """Check type and value of errorbar argument and assign default level."""
672      DEFAULT_LEVELS = {
673          "ci": 95,
674          "pi": 95,
675          "se": 1,
676          "sd": 1,
677      }
678  
679      usage = "`errorbar` must be a callable, string, or (string, number) tuple"
680  
681      if arg is None:
682          return None, None
683      elif callable(arg):
684          return arg, None
685      elif isinstance(arg, str):
686          method = arg
687          level = DEFAULT_LEVELS.get(method, None)
688      else:
689          try:
690              method, level = arg
691          except (ValueError, TypeError) as err:
692              raise err.__class__(usage) from err
693  
694      _check_argument("errorbar", list(DEFAULT_LEVELS), method)
695      if level is not None and not isinstance(level, Number):
696          raise TypeError(usage)
697  
698      return method, level
