54   class BaseThresholdClassifier(ClassifierMixin, MetaEstimatorMixin, BaseEstimator):
55       """Base class for binary classifiers that set a non-default decision threshold. 
56    
57       In this base class, we define the following interface: 
58    
59       - the validation of common parameters in `fit`; 
60       - the different prediction methods that can be used with the classifier. 
61    
62       .. versionadded:: 1.5 
63    
64       Parameters 
65       ---------- 
66       estimator : estimator instance 
67           The binary classifier, fitted or not, for which we want to optimize 
68           the decision threshold used during `predict`. 
69    
70       response_method : {"auto", "decision_function", "predict_proba"}, default="auto" 
71           Methods by the classifier `estimator` corresponding to the 
72           decision function for which we want to find a threshold. It can be: 
73    
74           * if `"auto"`, it will try to invoke, for each classifier, 
75             `"predict_proba"` or `"decision_function"` in that order. 
76           * otherwise, one of `"predict_proba"` or `"decision_function"`. 
77             If the method is not implemented by the classifier, it will raise an 
78             error. 
79       """
80   
81       _parameter_constraints: dict = {
82           "estimator": [
83               HasMethods(["fit", "predict_proba"]),
84               HasMethods(["fit", "decision_function"]),
85           ],
86           "response_method": [StrOptions({"auto", "predict_proba", "decision_function"})],
87       }
88   
89       def __init__(self, estimator, *, response_method="auto"):
90           self.estimator = estimator
91           self.response_method = response_method
92   
93       def _get_response_method(self):
94           """Define the response method."""
95           if self.response_method == "auto":
96               response_method = ["predict_proba", "decision_function"]
97           else:
98               response_method = self.response_method
99           return response_method
100  
101      @_fit_context(
102          # *ThresholdClassifier*.estimator is not validated yet
103          prefer_skip_nested_validation=False
104      )
105      def fit(self, X, y, **params):
106          """Fit the classifier. 
107   
108          Parameters 
109          ---------- 
110          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
111              Training data. 
112   
113          y : array-like of shape (n_samples,) 
114              Target values. 
115   
116          **params : dict 
117              Parameters to pass to the `fit` method of the underlying 
118              classifier. 
119   
120          Returns 
121          ------- 
122          self : object 
123              Returns an instance of self. 
124          """
125          _raise_for_params(params, self, None)
126  
127          X, y = indexable(X, y)
128  
129          y_type = type_of_target(y, input_name="y")
130          if y_type != "binary":
131              raise ValueError(
132                  f"Only binary classification is supported. Unknown label type: {y_type}"
133              )
134  
135          self._fit(X, y, **params)
136  
137          if hasattr(self.estimator_, "n_features_in_"):
138              self.n_features_in_ = self.estimator_.n_features_in_
139          if hasattr(self.estimator_, "feature_names_in_"):
140              self.feature_names_in_ = self.estimator_.feature_names_in_
141  
142          return self
143  
144      @property
145      def classes_(self):
146          """Classes labels."""
147          return self.estimator_.classes_
148  
149      @available_if(_estimator_has("predict_proba"))
150      def predict_proba(self, X):
151          """Predict class probabilities for `X` using the fitted estimator. 
152   
153          Parameters 
154          ---------- 
155          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
156              Training vectors, where `n_samples` is the number of samples and 
157              `n_features` is the number of features. 
158   
159          Returns 
160          ------- 
161          probabilities : ndarray of shape (n_samples, n_classes) 
162              The class probabilities of the input samples. 
163          """
164          _check_is_fitted(self)
165          estimator = getattr(self, "estimator_", self.estimator)
166          return estimator.predict_proba(X)
167  
168      @available_if(_estimator_has("predict_log_proba"))
169      def predict_log_proba(self, X):
170          """Predict logarithm class probabilities for `X` using the fitted estimator. 
171   
172          Parameters 
173          ---------- 
174          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
175              Training vectors, where `n_samples` is the number of samples and 
176              `n_features` is the number of features. 
177   
178          Returns 
179          ------- 
180          log_probabilities : ndarray of shape (n_samples, n_classes) 
181              The logarithm class probabilities of the input samples. 
182          """
183          _check_is_fitted(self)
184          estimator = getattr(self, "estimator_", self.estimator)
185          return estimator.predict_log_proba(X)
186  
187      @available_if(_estimator_has("decision_function"))
188      def decision_function(self, X):
189          """Decision function for samples in `X` using the fitted estimator. 
190   
191          Parameters 
192          ---------- 
193          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
194              Training vectors, where `n_samples` is the number of samples and 
195              `n_features` is the number of features. 
196   
197          Returns 
198          ------- 
199          decisions : ndarray of shape (n_samples,) 
200              The decision function computed the fitted estimator. 
201          """
202          _check_is_fitted(self)
203          estimator = getattr(self, "estimator_", self.estimator)
204          return estimator.decision_function(X)
205  
206      def __sklearn_tags__(self):
207          tags = super().__sklearn_tags__()
208          tags.classifier_tags.multi_class = False
209          tags.input_tags.sparse = get_tags(self.estimator).input_tags.sparse
210          return tags

213  class FixedThresholdClassifier(BaseThresholdClassifier):
214      """Binary classifier that manually sets the decision threshold. 
215   
216      This classifier allows to change the default decision threshold used for 
217      converting posterior probability estimates (i.e. output of `predict_proba`) or 
218      decision scores (i.e. output of `decision_function`) into a class label. 
219   
220      Here, the threshold is not optimized and is set to a constant value. 
221   
222      Read more in the :ref:`User Guide <FixedThresholdClassifier>`. 
223   
224      .. versionadded:: 1.5 
225   
226      Parameters 
227      ---------- 
228      estimator : estimator instance 
229          The binary classifier, fitted or not, for which we want to optimize 
230          the decision threshold used during `predict`. 
231   
232      threshold : {"auto"} or float, default="auto" 
233          The decision threshold to use when converting posterior probability estimates 
234          (i.e. output of `predict_proba`) or decision scores (i.e. output of 
235          `decision_function`) into a class label. When `"auto"`, the threshold is set 
236          to 0.5 if `predict_proba` is used as `response_method`, otherwise it is set to 
237          0 (i.e. the default threshold for `decision_function`). 
238   
239      pos_label : int, float, bool or str, default=None 
240          The label of the positive class. Used to process the output of the 
241          `response_method` method. When `pos_label=None`, if `y_true` is in `{-1, 1}` or 
242          `{0, 1}`, `pos_label` is set to 1, otherwise an error will be raised. 
243   
244      response_method : {"auto", "decision_function", "predict_proba"}, default="auto" 
245          Methods by the classifier `estimator` corresponding to the 
246          decision function for which we want to find a threshold. It can be: 
247   
248          * if `"auto"`, it will try to invoke `"predict_proba"` or `"decision_function"` 
249            in that order. 
250          * otherwise, one of `"predict_proba"` or `"decision_function"`. 
251            If the method is not implemented by the classifier, it will raise an 
252            error. 
253   
254      Attributes 
255      ---------- 
256      estimator_ : estimator instance 
257          The fitted classifier used when predicting. 
258   
259      classes_ : ndarray of shape (n_classes,) 
260          The class labels. 
261   
262      n_features_in_ : int 
263          Number of features seen during :term:`fit`. Only defined if the 
264          underlying estimator exposes such an attribute when fit. 
265   
266      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
267          Names of features seen during :term:`fit`. Only defined if the 
268          underlying estimator exposes such an attribute when fit. 
269   
270      See Also 
271      -------- 
272      sklearn.model_selection.TunedThresholdClassifierCV : Classifier that post-tunes 
273          the decision threshold based on some metrics and using cross-validation. 
274      sklearn.calibration.CalibratedClassifierCV : Estimator that calibrates 
275          probabilities. 
276   
277      Examples 
278      -------- 
279      >>> from sklearn.datasets import make_classification 
280      >>> from sklearn.linear_model import LogisticRegression 
281      >>> from sklearn.metrics import confusion_matrix 
282      >>> from sklearn.model_selection import FixedThresholdClassifier, train_test_split 
283      >>> X, y = make_classification( 
284      ...     n_samples=1_000, weights=[0.9, 0.1], class_sep=0.8, random_state=42 
285      ... ) 
286      >>> X_train, X_test, y_train, y_test = train_test_split( 
287      ...     X, y, stratify=y, random_state=42 
288      ... ) 
289      >>> classifier = LogisticRegression(random_state=0).fit(X_train, y_train) 
290      >>> print(confusion_matrix(y_test, classifier.predict(X_test))) 
291      [[217   7] 
292       [ 19   7]] 
293      >>> classifier_other_threshold = FixedThresholdClassifier( 
294      ...     classifier, threshold=0.1, response_method="predict_proba" 
295      ... ).fit(X_train, y_train) 
296      >>> print(confusion_matrix(y_test, classifier_other_threshold.predict(X_test))) 
297      [[184  40] 
298       [  6  20]] 
299      """
300  
301      _parameter_constraints: dict = {
302          **BaseThresholdClassifier._parameter_constraints,
303          "threshold": [StrOptions({"auto"}), Real],
304          "pos_label": [Real, str, "boolean", None],
305      }
306  
307      def __init__(
308          self,
309          estimator,
310          *,
311          threshold="auto",
312          pos_label=None,
313          response_method="auto",
314      ):
315          super().__init__(estimator=estimator, response_method=response_method)
316          self.pos_label = pos_label
317          self.threshold = threshold
318  
319      @property
320      def classes_(self):
321          if estimator := getattr(self, "estimator_", None):
322              return estimator.classes_
323          try:
324              check_is_fitted(self.estimator)
325              return self.estimator.classes_
326          except NotFittedError:
327              raise AttributeError(
328                  "The underlying estimator is not fitted yet."
329              ) from NotFittedError
330  
331      def _fit(self, X, y, **params):
332          """Fit the classifier. 
333   
334          Parameters 
335          ---------- 
336          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
337              Training data. 
338   
339          y : array-like of shape (n_samples,) 
340              Target values. 
341   
342          **params : dict 
343              Parameters to pass to the `fit` method of the underlying 
344              classifier. 
345   
346          Returns 
347          ------- 
348          self : object 
349              Returns an instance of self. 
350          """
351          routed_params = process_routing(self, "fit", **params)
352          self.estimator_ = clone(self.estimator).fit(X, y, **routed_params.estimator.fit)
353          return self
354  
355      def predict(self, X):
356          """Predict the target of new samples. 
357   
358          Parameters 
359          ---------- 
360          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
361              The samples, as accepted by `estimator.predict`. 
362   
363          Returns 
364          ------- 
365          class_labels : ndarray of shape (n_samples,) 
366              The predicted class. 
367          """
368          _check_is_fitted(self)
369  
370          estimator = getattr(self, "estimator_", self.estimator)
371  
372          y_score, _, response_method_used = _get_response_values_binary(
373              estimator,
374              X,
375              self._get_response_method(),
376              pos_label=self.pos_label,
377              return_response_method_used=True,
378          )
379  
380          if self.threshold == "auto":
381              decision_threshold = 0.5 if response_method_used == "predict_proba" else 0.0
382          else:
383              decision_threshold = self.threshold
384  
385          return _threshold_scores_to_class_labels(
386              y_score, decision_threshold, self.classes_, self.pos_label
387          )

390          """Get metadata routing of this object. 
391

408  def _fit_and_score_over_thresholds(
409      classifier,
410      X,
411      y,
412      *,
413      fit_params,
414      train_idx,
415      val_idx,
416      curve_scorer,
417      score_params,
418  ):
419      """Fit a classifier and compute the scores for different decision thresholds. 
420   
421      Parameters 
422      ---------- 
423      classifier : estimator instance 
424          The classifier to fit and use for scoring. If `classifier` is already fitted, 
425          it will be used as is. 
426   
427      X : {array-like, sparse matrix} of shape (n_samples, n_features) 
428          The entire dataset. 
429   
430      y : array-like of shape (n_samples,) 
431          The entire target vector. 
432   
433      fit_params : dict 
434          Parameters to pass to the `fit` method of the underlying classifier. 
435   
436      train_idx : ndarray of shape (n_train_samples,) or None 
437          The indices of the training set. If `None`, `classifier` is expected to be 
438          already fitted. 
439   
440      val_idx : ndarray of shape (n_val_samples,) 
441          The indices of the validation set used to score `classifier`. If `train_idx`, 
442          the entire set will be used. 
443   
444      curve_scorer : scorer instance 
445          The scorer taking `classifier` and the validation set as input and outputting 
446          decision thresholds and scores as a curve. Note that this is different from 
447          the usual scorer that output a single score value: 
448   
449          * when `score_method` is one of the four constraint metrics, the curve scorer 
450            will output a curve of two scores parametrized by the decision threshold, e.g. 
451            TPR/TNR or precision/recall curves for each threshold; 
452          * otherwise, the curve scorer will output a single score value for each 
453            threshold. 
454   
455      score_params : dict 
456          Parameters to pass to the `score` method of the underlying scorer. 
457   
458      Returns 
459      ------- 
460      scores : ndarray of shape (thresholds,) or tuple of such arrays 
461          The scores computed for each decision threshold. When TPR/TNR or precision/ 
462          recall are computed, `scores` is a tuple of two arrays. 
463   
464      potential_thresholds : ndarray of shape (thresholds,) 
465          The decision thresholds used to compute the scores. They are returned in 
466          ascending order. 
467      """
468  
469      if train_idx is not None:
470          X_train, X_val = _safe_indexing(X, train_idx), _safe_indexing(X, val_idx)
471          y_train, y_val = _safe_indexing(y, train_idx), _safe_indexing(y, val_idx)
472          fit_params_train = _check_method_params(X, fit_params, indices=train_idx)
473          score_params_val = _check_method_params(X, score_params, indices=val_idx)
474          classifier.fit(X_train, y_train, **fit_params_train)
475      else:  # prefit estimator, only a validation set is provided
476          X_val, y_val, score_params_val = X, y, score_params
477  
478      return curve_scorer(classifier, X_val, y_val, **score_params_val)

481  def _mean_interpolated_score(target_thresholds, cv_thresholds, cv_scores):
482      """Compute the mean interpolated score across folds by defining common thresholds. 
483   
484      Parameters 
485      ---------- 
486      target_thresholds : ndarray of shape (thresholds,) 
487          The thresholds to use to compute the mean score. 
488   
489      cv_thresholds : ndarray of shape (n_folds, thresholds_fold) 
490          The thresholds used to compute the scores for each fold. 
491   
492      cv_scores : ndarray of shape (n_folds, thresholds_fold) 
493          The scores computed for each threshold for each fold. 
494   
495      Returns 
496      ------- 
497      mean_score : ndarray of shape (thresholds,) 
498          The mean score across all folds for each target threshold. 
499      """
500      return np.mean(
501          [
502              np.interp(target_thresholds, split_thresholds, split_score)
503              for split_thresholds, split_score in zip(cv_thresholds, cv_scores)
504          ],
505          axis=0,
506      )

509  class TunedThresholdClassifierCV(BaseThresholdClassifier):
510      """Classifier that post-tunes the decision threshold using cross-validation. 
511   
512      This estimator post-tunes the decision threshold (cut-off point) that is 
513      used for converting posterior probability estimates (i.e. output of 
514      `predict_proba`) or decision scores (i.e. output of `decision_function`) 
515      into a class label. The tuning is done by optimizing a binary metric, 
516      potentially constrained by a another metric. 
517   
518      Read more in the :ref:`User Guide <TunedThresholdClassifierCV>`. 
519   
520      .. versionadded:: 1.5 
521   
522      Parameters 
523      ---------- 
524      estimator : estimator instance 
525          The classifier, fitted or not, for which we want to optimize 
526          the decision threshold used during `predict`. 
527   
528      scoring : str or callable, default="balanced_accuracy" 
529          The objective metric to be optimized. Can be one of: 
530   
531          * a string associated to a scoring function for binary classification 
532            (see :ref:`scoring_parameter`); 
533          * a scorer callable object created with :func:`~sklearn.metrics.make_scorer`; 
534   
535      response_method : {"auto", "decision_function", "predict_proba"}, default="auto" 
536          Methods by the classifier `estimator` corresponding to the 
537          decision function for which we want to find a threshold. It can be: 
538   
539          * if `"auto"`, it will try to invoke, for each classifier, 
540            `"predict_proba"` or `"decision_function"` in that order. 
541          * otherwise, one of `"predict_proba"` or `"decision_function"`. 
542            If the method is not implemented by the classifier, it will raise an 
543            error. 
544   
545      thresholds : int or array-like, default=100 
546          The number of decision threshold to use when discretizing the output of the 
547          classifier `method`. Pass an array-like to manually specify the thresholds 
548          to use. 
549   
550      cv : int, float, cross-validation generator, iterable or "prefit", default=None 
551          Determines the cross-validation splitting strategy to train classifier. 
552          Possible inputs for cv are: 
553   
554          * `None`, to use the default 5-fold stratified K-fold cross validation; 
555          * An integer number, to specify the number of folds in a stratified k-fold; 
556          * A float number, to specify a single shuffle split. The floating number should 
557            be in (0, 1) and represent the size of the validation set; 
558          * An object to be used as a cross-validation generator; 
559          * An iterable yielding train, test splits; 
560          * `"prefit"`, to bypass the cross-validation. 
561   
562          Refer :ref:`User Guide <cross_validation>` for the various 
563          cross-validation strategies that can be used here. 
564   
565          .. warning:: 
566              Using `cv="prefit"` and passing the same dataset for fitting `estimator` 
567              and tuning the cut-off point is subject to undesired overfitting. You can 
568              refer to :ref:`TunedThresholdClassifierCV_no_cv` for an example. 
569   
570              This option should only be used when the set used to fit `estimator` is 
571              different from the one used to tune the cut-off point (by calling 
572              :meth:`TunedThresholdClassifierCV.fit`). 
573   
574      refit : bool, default=True 
575          Whether or not to refit the classifier on the entire training set once 
576          the decision threshold has been found. 
577          Note that forcing `refit=False` on cross-validation having more 
578          than a single split will raise an error. Similarly, `refit=True` in 
579          conjunction with `cv="prefit"` will raise an error. 
580   
581      n_jobs : int, default=None 
582          The number of jobs to run in parallel. When `cv` represents a 
583          cross-validation strategy, the fitting and scoring on each data split 
584          is done in parallel. ``None`` means 1 unless in a 
585          :obj:`joblib.parallel_backend` context. ``-1`` means using all 
586          processors. See :term:`Glossary <n_jobs>` for more details. 
587   
588      random_state : int, RandomState instance or None, default=None 
589          Controls the randomness of cross-validation when `cv` is a float. 
590          See :term:`Glossary <random_state>`. 
591   
592      store_cv_results : bool, default=False 
593          Whether to store all scores and thresholds computed during the cross-validation 
594          process. 
595   
596      Attributes 
597      ---------- 
598      estimator_ : estimator instance 
599          The fitted classifier used when predicting. 
600   
601      best_threshold_ : float 
602          The new decision threshold. 
603   
604      best_score_ : float or None 
605          The optimal score of the objective metric, evaluated at `best_threshold_`. 
606   
607      cv_results_ : dict or None 
608          A dictionary containing the scores and thresholds computed during the 
609          cross-validation process. Only exist if `store_cv_results=True`. The 
610          keys are `"thresholds"` and `"scores"`. 
611   
612      classes_ : ndarray of shape (n_classes,) 
613          The class labels. 
614   
615      n_features_in_ : int 
616          Number of features seen during :term:`fit`. Only defined if the 
617          underlying estimator exposes such an attribute when fit. 
618   
619      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
620          Names of features seen during :term:`fit`. Only defined if the 
621          underlying estimator exposes such an attribute when fit. 
622   
623      See Also 
624      -------- 
625      sklearn.model_selection.FixedThresholdClassifier : Classifier that uses a 
626          constant threshold. 
627      sklearn.calibration.CalibratedClassifierCV : Estimator that calibrates 
628          probabilities. 
629   
630      Examples 
631      -------- 
632      >>> from sklearn.datasets import make_classification 
633      >>> from sklearn.ensemble import RandomForestClassifier 
634      >>> from sklearn.metrics import classification_report 
635      >>> from sklearn.model_selection import TunedThresholdClassifierCV, train_test_split 
636      >>> X, y = make_classification( 
637      ...     n_samples=1_000, weights=[0.9, 0.1], class_sep=0.8, random_state=42 
638      ... ) 
639      >>> X_train, X_test, y_train, y_test = train_test_split( 
640      ...     X, y, stratify=y, random_state=42 
641      ... ) 
642      >>> classifier = RandomForestClassifier(random_state=0).fit(X_train, y_train) 
643      >>> print(classification_report(y_test, classifier.predict(X_test))) 
644                    precision    recall  f1-score   support 
645      <BLANKLINE> 
646                 0       0.94      0.99      0.96       224 
647                 1       0.80      0.46      0.59        26 
648      <BLANKLINE> 
649          accuracy                           0.93       250 
650         macro avg       0.87      0.72      0.77       250 
651      weighted avg       0.93      0.93      0.92       250 
652      <BLANKLINE> 
653      >>> classifier_tuned = TunedThresholdClassifierCV( 
654      ...     classifier, scoring="balanced_accuracy" 
655      ... ).fit(X_train, y_train) 
656      >>> print( 
657      ...     f"Cut-off point found at {classifier_tuned.best_threshold_:.3f}" 
658      ... ) 
659      Cut-off point found at 0.342 
660      >>> print(classification_report(y_test, classifier_tuned.predict(X_test))) 
661                    precision    recall  f1-score   support

713      def _fit(self, X, y, **params):
714          """Fit the classifier and post-tune the decision threshold. 
715   
716          Parameters 
717          ---------- 
718          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
719              Training data. 
720   
721          y : array-like of shape (n_samples,) 
722              Target values. 
723   
724          **params : dict 
725              Parameters to pass to the `fit` method of the underlying 
726              classifier and to the `scoring` scorer. 
727   
728          Returns 
729          ------- 
730          self : object 
731              Returns an instance of self. 
732          """
733          if isinstance(self.cv, Real) and 0 < self.cv < 1:
734              cv = StratifiedShuffleSplit(
735                  n_splits=1, test_size=self.cv, random_state=self.random_state
736              )
737          elif self.cv == "prefit":
738              if self.refit is True:
739                  raise ValueError("When cv='prefit', refit cannot be True.")
740              try:
741                  check_is_fitted(self.estimator, "classes_")
742              except NotFittedError as exc:
743                  raise NotFittedError(
744                      """When cv='prefit', `estimator` must be fitted."""
745                  ) from exc
746              cv = self.cv
747          else:
748              cv = check_cv(self.cv, y=y, classifier=True)
749              if self.refit is False and cv.get_n_splits() > 1:
750                  raise ValueError("When cv has several folds, refit cannot be False.")
751  
752          routed_params = process_routing(self, "fit", **params)
753          self._curve_scorer = self._get_curve_scorer()
754  
755          # in the following block, we:
756          # - define the final classifier `self.estimator_` and train it if necessary
757          # - define `classifier` to be used to post-tune the decision threshold
758          # - define `split` to be used to fit/score `classifier`
759          if cv == "prefit":
760              self.estimator_ = self.estimator
761              classifier = self.estimator_
762              splits = [(None, range(_num_samples(X)))]
763          else:
764              self.estimator_ = clone(self.estimator)
765              classifier = clone(self.estimator)
766              splits = cv.split(X, y, **routed_params.splitter.split)
767  
768              if self.refit:
769                  # train on the whole dataset
770                  X_train, y_train, fit_params_train = X, y, routed_params.estimator.fit
771              else:
772                  # single split cross-validation
773                  train_idx, _ = next(cv.split(X, y, **routed_params.splitter.split))
774                  X_train = _safe_indexing(X, train_idx)
775                  y_train = _safe_indexing(y, train_idx)
776                  fit_params_train = _check_method_params(
777                      X, routed_params.estimator.fit, indices=train_idx
778                  )
779  
780              self.estimator_.fit(X_train, y_train, **fit_params_train)
781  
782          cv_scores, cv_thresholds = zip(
783              *Parallel(n_jobs=self.n_jobs)(
784                  delayed(_fit_and_score_over_thresholds)(
785                      clone(classifier) if cv != "prefit" else classifier,
786                      X,
787                      y,
788                      fit_params=routed_params.estimator.fit,
789                      train_idx=train_idx,
790                      val_idx=val_idx,
791                      curve_scorer=self._curve_scorer,
792                      score_params=routed_params.scorer.score,
793                  )
794                  for train_idx, val_idx in splits
795              )
796          )
797  
798          if any(np.isclose(th[0], th[-1]) for th in cv_thresholds):
799              raise ValueError(
800                  "The provided estimator makes constant predictions. Therefore, it is "
801                  "impossible to optimize the decision threshold."
802              )
803  
804          # find the global min and max thresholds across all folds
805          min_threshold = min(
806              split_thresholds.min() for split_thresholds in cv_thresholds
807          )
808          max_threshold = max(
809              split_thresholds.max() for split_thresholds in cv_thresholds
810          )
811          if isinstance(self.thresholds, Integral):
812              decision_thresholds = np.linspace(
813                  min_threshold, max_threshold, num=self.thresholds
814              )
815          else:
816              decision_thresholds = np.asarray(self.thresholds)
817  
818          objective_scores = _mean_interpolated_score(
819              decision_thresholds, cv_thresholds, cv_scores
820          )
821          best_idx = objective_scores.argmax()
822          self.best_score_ = objective_scores[best_idx]
823          self.best_threshold_ = decision_thresholds[best_idx]
824          if self.store_cv_results:
825              self.cv_results_ = {
826                  "thresholds": decision_thresholds,
827                  "scores": objective_scores,
828              }
829  
830          return self
831  
832      def predict(self, X):
833          """Predict the target of new samples. 
834   
835          Parameters 
836          ---------- 
837          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
838              The samples, as accepted by `estimator.predict`. 
839   
840          Returns 
841          ------- 
842          class_labels : ndarray of shape (n_samples,) 
843              The predicted class. 
844          """
845          check_is_fitted(self, "estimator_")
846          pos_label = self._curve_scorer._get_pos_label()
847          y_score, _ = _get_response_values_binary(
848              self.estimator_,
849              X,
850              self._get_response_method(),
851              pos_label=pos_label,
852          )
853  
854          return _threshold_scores_to_class_labels(
855              y_score, self.best_threshold_, self.classes_, pos_label
856          )

858      def get_metadata_routing(self):
859          """Get metadata routing of this object. 
860   
861          Please check :ref:`User Guide <metadata_routing>` on how the routing 
862          mechanism works. 
863   
864          Returns 
865          ------- 
866          routing : MetadataRouter 
867              A :class:`~sklearn.utils.metadata_routing.MetadataRouter` encapsulating 
868              routing information. 
869          """
870          router = (
871              MetadataRouter(owner=self.__class__.__name__)
872              .add(
873                  estimator=self.estimator,
874                  method_mapping=MethodMapping().add(callee="fit", caller="fit"),
875              )
876              .add(
877                  splitter=self.cv,
878                  method_mapping=MethodMapping().add(callee="split", caller="fit"),
879              )
880              .add(
881                  scorer=self._get_curve_scorer(),
882                  method_mapping=MethodMapping().add(callee="score", caller="fit"),
883              )
884          )
885          return router

887      def _get_curve_scorer(self):
888          """Get the curve scorer based on the objective metric used."""
889          scoring = check_scoring(self.estimator, scoring=self.scoring)
890          curve_scorer = _CurveScorer.from_scorer(
891              scoring, self._get_response_method(), self.thresholds
892          )
