102  def int_parser(s, default_value=0):

104          return int(s)
105      except Exception:

150  def _build_debug_options(flags):
151      """Build string representation of debug options from the launch config."""
152      return ";".join(DEBUG_OPTIONS_BY_FLAG[flag] for flag in flags or [] if flag in DEBUG_OPTIONS_BY_FLAG)

155  def _parse_debug_options(opts):
156      """Debug options are semicolon separated key=value pairs"""
157      options = {}
158      if not opts:
159          return options
160  
161      for opt in opts.split(";"):
162          try:
163              key, value = opt.split("=")
164          except ValueError:
165              continue
166          try:
167              options[key] = DEBUG_OPTIONS_PARSER[key](value)
168          except KeyError:
169              continue
170  
171      return options
172  
173  
174  def _extract_debug_options(opts, flags=None):
175      """Return the debug options encoded in the given value. 
176   
177      "opts" is a semicolon-separated string of "key=value" pairs. 
178      "flags" is a list of strings. 
179   
180      If flags is provided then it is used as a fallback. 
181   
182      The values come from the launch config: 
183   
184       { 
185           type:'python', 
186           request:'launch'|'attach', 
187           name:'friendly name for debug config', 
188           debugOptions:[ 
189               'RedirectOutput', 'Django' 
190           ], 
191           options:'REDIRECT_OUTPUT=True;DJANGO_DEBUG=True' 
192       } 
193   
194      Further information can be found here: 
195   
196      https://code.visualstudio.com/docs/editor/debugging#_launchjson-attributes 
197      """
198      if not opts:
199          opts = _build_debug_options(flags)
200      return _parse_debug_options(opts)
