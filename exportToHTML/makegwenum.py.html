17   def is_interface_enum(enumtype):
18       return not (enumtype[0].isupper() and enumtype[2].isupper())

21   def _write_enumifc_cpp(f, interface):
22       enumtype = interface.name[5:]
23       if is_interface_enum(enumtype):
24           # Assume an interface.
25           enum_interface = "I" + enumtype[:-1]
26           converter = "PyObject *ob = PyCom_PyObjectFromIUnknown(rgVar[i], IID_{enum_interface}, FALSE);".format(
27               **locals()
28           )
29           arraydeclare = (
30               "{enum_interface} **rgVar = new {enum_interface} *[celt];".format(
31                   **locals()
32               )
33           )
34       else:
35           # Enum of a simple structure
36           converter = "PyObject *ob = PyCom_PyObjectFrom{enumtype}(&rgVar[i]);".format(
37               **locals()
38           )
39           arraydeclare = "{enumtype} *rgVar = new {enumtype}[celt];".format(**locals())
40   
41       f.write(
42           """

45   // Interface Implementation

48       PyIUnknown(pdisp)

51   }} 
52

54   {{ 
55   }}

89       if (  HRESULT_CODE(hr) != ERROR_NO_MORE_ITEMS && FAILED(hr) )

92           return PyCom_BuildPyException(hr,pIE{enumtype}, IID_IE{enumtype});

119  PyObject *PyIEnum{enumtype}::Skip(PyObject *self, PyObject *args) 
120  {{ 
121      long celt; 
122      if ( !PyArg_ParseTuple(args, "l:Skip", &celt) ) 
123          return NULL; 
124   
125      IEnum{enumtype} *pIE{enumtype} = GetI(self); 
126      if ( pIE{enumtype} == NULL ) 
127          return NULL; 
128   
129      PY_INTERFACE_PRECALL; 
130      HRESULT hr = pIE{enumtype}->Skip(celt); 
131      PY_INTERFACE_POSTCALL; 
132      if ( FAILED(hr) ) 
133          return PyCom_BuildPyException(hr, pIE{enumtype}, IID_IE{enumtype}); 
134   
135      Py_INCREF(Py_None); 
136      return Py_None;

198  def _write_enumgw_cpp(f, interface):
199      enumtype = interface.name[5:]
200      if is_interface_enum(enumtype):
201          # Assume an interface.
202          enum_interface = "I" + enumtype[:-1]
203          converter = "if ( !PyCom_InterfaceFromPyObject(ob, IID_{enum_interface}, (void **)&rgVar[i], FALSE) )".format(
204              **locals()
205          )
206          argdeclare = "{enum_interface} __RPC_FAR * __RPC_FAR *rgVar".format(**locals())
207      else:
208          argdeclare = "{enumtype} __RPC_FAR *rgVar".format(**locals())
209          converter = "if ( !PyCom_PyObjectAs{enumtype}(ob, &rgVar[i]) )".format(
210              **locals()
211          )
212      f.write(
213          """ 
214  // --------------------------------------------------- 
215  // 
216  // Gateway Implementation 
217   
218  // Std delegation 
219  STDMETHODIMP_(ULONG) PyGEnum{enumtype}::AddRef(void) {{return PyGatewayBase::AddRef();}} 
220  STDMETHODIMP_(ULONG) PyGEnum{enumtype}::Release(void) {{return PyGatewayBase::Release();}} 
221  STDMETHODIMP PyGEnum{enumtype}::QueryInterface(REFIID iid, void ** obj) {{return PyGatewayBase::QueryInterface(iid, obj);}} 
222  STDMETHODIMP PyGEnum{enumtype}::GetTypeInfoCount(UINT FAR* pctInfo) {{return PyGatewayBase::GetTypeInfoCount(pctInfo);}} 
223  STDMETHODIMP PyGEnum{enumtype}::GetTypeInfo(UINT itinfo, LCID lcid, ITypeInfo FAR* FAR* pptInfo) {{return PyGatewayBase::GetTypeInfo(itinfo, lcid, pptInfo);}} 
224  STDMETHODIMP PyGEnum{enumtype}::GetIDsOfNames(REFIID refiid, OLECHAR FAR* FAR* rgszNames, UINT cNames, LCID lcid, DISPID FAR* rgdispid) {{return PyGatewayBase::GetIDsOfNames( refiid, rgszNames, cNames, lcid, rgdispid);}} 
225  STDMETHODIMP PyGEnum{enumtype}::Invoke(DISPID dispid, REFIID riid, LCID lcid, WORD wFlags, DISPPARAMS FAR* params, VARIANT FAR* pVarResult, EXCEPINFO FAR* pexcepinfo, UINT FAR* puArgErr) {{return PyGatewayBase::Invoke( dispid, riid, lcid, wFlags, params, pVarResult, pexcepinfo, puArgErr);}}

227  STDMETHODIMP PyGEnum{enumtype}::Next( 
228              /* [in] */ ULONG celt, 
229              /* [length_is][size_is][out] */ {argdeclare}, 
230              /* [out] */ ULONG __RPC_FAR *pCeltFetched) 
231  {{ 
232      PY_GATEWAY_METHOD; 
233      PyObject *result; 
234      HRESULT hr = InvokeViaPolicy("Next", &result, "i", celt); 
235      if ( FAILED(hr) ) 
236          return hr; 
237   
238      if ( !PySequence_Check(result) ) 
239          goto error; 
240      int len; 
241      len = PyObject_Length(result); 
242      if ( len == -1 ) 
243          goto error; 
244      if ( len > (int)celt) 
245          len = celt; 
246   
247      if ( pCeltFetched ) 
248          *pCeltFetched = len; 
249   
250      int i; 
251      for ( i = 0; i < len; ++i ) 
252      {{ 
253          PyObject *ob = PySequence_GetItem(result, i); 
254          if ( ob == NULL ) 
255              goto error; 
256   
257          {converter} 
258          {{ 
259              Py_DECREF(result); 
260              return PyCom_SetCOMErrorFromPyException(IID_IEnum{enumtype}); 
261          }} 
262      }} 
263   
264      Py_DECREF(result); 
265   
266      return len < (int)celt ? S_FALSE : S_OK; 
267   
268    error: 
269      PyErr_Clear(); // just in case 
270      Py_DECREF(result); 
271      return PyCom_HandleIEnumNoSequence(IID_IEnum{enumtype}); 
272  }}

287  STDMETHODIMP PyGEnum{enumtype}::Clone( 
288              /* [out] */ IEnum{enumtype} __RPC_FAR *__RPC_FAR *ppEnum) 
289  {{ 
290      PY_GATEWAY_METHOD; 
291      PyObject * result; 
292      HRESULT hr = InvokeViaPolicy("Clone", &result); 
293      if ( FAILED(hr) ) 
294          return hr; 
295   
296      /* 
297      ** Make sure we have the right kind of object: we should have some kind 
298      ** of IUnknown subclass wrapped into a PyIUnknown instance. 
299      */ 
300      if ( !PyIBase::is_object(result, &PyIUnknown::type) ) 
301      {{ 
302          /* the wrong kind of object was returned to us */ 
303          Py_DECREF(result); 
304          return PyCom_SetCOMErrorFromSimple(E_FAIL, IID_IEnum{enumtype}); 
305      }} 
306   
307      /* 
308      ** Get the IUnknown out of the thing. note that the Python ob maintains 
309      ** a reference, so we don't have to explicitly AddRef() here. 
310      */ 
311      IUnknown *punk = ((PyIUnknown *)result)->m_obj; 
312      if ( !punk ) 
313      {{ 
314          /* damn. the object was released. */ 
315          Py_DECREF(result); 
316          return PyCom_SetCOMErrorFromSimple(E_FAIL, IID_IEnum{enumtype}); 
317      }} 
318   
319      /* 
320      ** Get the interface we want. note it is returned with a refcount. 
321      ** This QI is actually going to instantiate a PyGEnum{enumtype}. 
322      */ 
323      hr = punk->QueryInterface(IID_IEnum{enumtype}, (LPVOID *)ppEnum); 
324   
325      /* done with the result; this DECREF is also for <punk> */ 
326      Py_DECREF(result); 
327   
328      return PyCom_CheckIEnumNextResult(hr, IID_IEnum{enumtype}); 
329  }} 
330  """.format(**locals())
331      )
