37   def test_simple():
38       # Principle of Lars is to keep covariances tied and decreasing
39   
40       # also test verbose output
41       import sys
42       from io import StringIO
43   
44       old_stdout = sys.stdout
45       try:
46           sys.stdout = StringIO()
47   
48           _, _, coef_path_ = linear_model.lars_path(X, y, method="lar", verbose=10)
49   
50           sys.stdout = old_stdout
51   
52           for i, coef_ in enumerate(coef_path_.T):
53               res = y - np.dot(X, coef_)
54               cov = np.dot(X.T, res)
55               C = np.max(abs(cov))
56               eps = 1e-3
57               ocur = len(cov[C - eps < abs(cov)])
58               if i < X.shape[1]:
59                   assert ocur == i + 1
60               else:
61                   # no more than max_pred variables can go into the active set
62                   assert ocur == X.shape[1]
63       finally:
64           sys.stdout = old_stdout

231  def test_rank_deficient_design():
232      # consistency test that checks that LARS Lasso is handling rank
233      # deficient input data (with n_features < rank) in the same way
234      # as coordinate descent Lasso
235      y = [5, 0, 5]
236      for X in ([[5, 0], [0, 5], [10, 10]], [[10, 10, 0], [1e-32, 0, 0], [0, 0, 1]]):
237          # To be able to use the coefs to compute the objective function,
238          # we need to turn off normalization
239          lars = linear_model.LassoLars(0.1)
240          coef_lars_ = lars.fit(X, y).coef_
241          obj_lars = 1.0 / (2.0 * 3.0) * linalg.norm(
242              y - np.dot(X, coef_lars_)
243          ) ** 2 + 0.1 * linalg.norm(coef_lars_, 1)
244          coord_descent = linear_model.Lasso(0.1, tol=1e-6)
245          coef_cd_ = coord_descent.fit(X, y).coef_
246          obj_cd = (1.0 / (2.0 * 3.0)) * linalg.norm(
247              y - np.dot(X, coef_cd_)
248          ) ** 2 + 0.1 * linalg.norm(coef_cd_, 1)
249          assert obj_lars < obj_cd * (1.0 + 1e-8)

252  def test_lasso_lars_vs_lasso_cd():
253      # Test that LassoLars and Lasso using coordinate descent give the
254      # same results.
255      X = 3 * diabetes.data
256  
257      alphas, _, lasso_path = linear_model.lars_path(X, y, method="lasso")
258      lasso_cd = linear_model.Lasso(fit_intercept=False, tol=1e-8)
259      for c, a in zip(lasso_path.T, alphas):
260          if a == 0:
261              continue
262          lasso_cd.alpha = a
263          lasso_cd.fit(X, y)
264          error = linalg.norm(c - lasso_cd.coef_)
265          assert error < 0.01
266  
267      # similar test, with the classifiers
268      for alpha in np.linspace(1e-2, 1 - 1e-2, 20):
269          clf1 = linear_model.LassoLars(alpha=alpha).fit(X, y)
270          clf2 = linear_model.Lasso(alpha=alpha, tol=1e-8).fit(X, y)
271          err = linalg.norm(clf1.coef_ - clf2.coef_)
272          assert err < 1e-3
273  
274      # same test, with normalized data
275      X = diabetes.data
276      X = X - X.sum(axis=0)
277      X /= np.linalg.norm(X, axis=0)
278      alphas, _, lasso_path = linear_model.lars_path(X, y, method="lasso")
279      lasso_cd = linear_model.Lasso(fit_intercept=False, tol=1e-8)
280      for c, a in zip(lasso_path.T, alphas):
281          if a == 0:
282              continue
283          lasso_cd.alpha = a
284          lasso_cd.fit(X, y)
285          error = linalg.norm(c - lasso_cd.coef_)
286          assert error < 0.01

334      # Test lasso lars on a very ill-conditioned design, and check that
335      # it does not blow up, and stays somewhat close to a solution given
336      # by the coordinate descent solver

362  def test_lasso_lars_vs_lasso_cd_ill_conditioned2():
363      # Create an ill-conditioned situation in which the LARS has to go
364      # far in the path to converge, and check that LARS and coordinate
365      # descent give the same answers
366      # Note it used to be the case that Lars had to use the drop for good
367      # strategy for this but this is no longer the case with the
368      # equality_tolerance checks
369      X = [[1e20, 1e20, 0], [-1e-32, 0, 0], [1, 1, 1]]
370      y = [10, 10, 1]
371      alpha = 0.0001
372  
373      def objective_function(coef):
374          return 1.0 / (2.0 * len(X)) * linalg.norm(
375              y - np.dot(X, coef)
376          ) ** 2 + alpha * linalg.norm(coef, 1)
377  
378      lars = linear_model.LassoLars(alpha=alpha)
379      warning_message = "Regressors in active set degenerate."
380      with pytest.warns(ConvergenceWarning, match=warning_message):
381          lars.fit(X, y)
382      lars_coef_ = lars.coef_
383      lars_obj = objective_function(lars_coef_)
384  
385      coord_descent = linear_model.Lasso(alpha=alpha, tol=1e-4)
386      cd_coef_ = coord_descent.fit(X, y).coef_
387      cd_obj = objective_function(cd_coef_)
388  
389      assert lars_obj < cd_obj * (1.0 + 1e-8)

415      estimators = [

570  def test_lasso_lars_vs_lasso_cd_positive():
571      # Test that LassoLars and Lasso using coordinate descent give the
572      # same results when using the positive option
573  
574      # This test is basically a copy of the above with additional positive
575      # option. However for the middle part, the comparison of coefficient values
576      # for a range of alphas, we had to make an adaptations. See below.
577  
578      # not normalized data
579      X = 3 * diabetes.data
580  
581      alphas, _, lasso_path = linear_model.lars_path(X, y, method="lasso", positive=True)
582      lasso_cd = linear_model.Lasso(fit_intercept=False, tol=1e-8, positive=True)
583      for c, a in zip(lasso_path.T, alphas):
584          if a == 0:
585              continue
586          lasso_cd.alpha = a
587          lasso_cd.fit(X, y)
588          error = linalg.norm(c - lasso_cd.coef_)
589          assert error < 0.01
590  
591      # The range of alphas chosen for coefficient comparison here is restricted
592      # as compared with the above test without the positive option. This is due
593      # to the circumstance that the Lars-Lasso algorithm does not converge to
594      # the least-squares-solution for small alphas, see 'Least Angle Regression'
595      # by Efron et al 2004. The coefficients are typically in congruence up to
596      # the smallest alpha reached by the Lars-Lasso algorithm and start to
597      # diverge thereafter.  See
598      # https://gist.github.com/michigraber/7e7d7c75eca694c7a6ff
599  
600      for alpha in np.linspace(6e-1, 1 - 1e-2, 20):
601          clf1 = linear_model.LassoLars(
602              fit_intercept=False, alpha=alpha, positive=True
603          ).fit(X, y)
604          clf2 = linear_model.Lasso(
605              fit_intercept=False, alpha=alpha, tol=1e-8, positive=True
606          ).fit(X, y)
607          err = linalg.norm(clf1.coef_ - clf2.coef_)
608          assert err < 1e-3
609  
610      # normalized data
611      X = diabetes.data - diabetes.data.sum(axis=0)
612      X /= np.linalg.norm(X, axis=0)
613      alphas, _, lasso_path = linear_model.lars_path(X, y, method="lasso", positive=True)
614      lasso_cd = linear_model.Lasso(fit_intercept=False, tol=1e-8, positive=True)
615      for c, a in zip(lasso_path.T[:-1], alphas[:-1]):  # don't include alpha=0
616          lasso_cd.alpha = a
617          lasso_cd.fit(X, y)
618          error = linalg.norm(c - lasso_cd.coef_)
619          assert error < 0.01
