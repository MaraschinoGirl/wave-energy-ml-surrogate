12   class AutoreloadTest(unittest.TestCase):
13       def setUp(self):
14           # When these tests fail the output sometimes exceeds the default maxDiff.
15           self.maxDiff = 1024
16   
17           self.path = mkdtemp()
18   
19           # Most test apps run themselves twice via autoreload. The first time it manually triggers
20           # a reload (could also do this by touching a file but this is faster since filesystem
21           # timestamps are not necessarily high resolution). The second time it exits directly
22           # so that the autoreload wrapper (if it is used) doesn't catch it.
23           #
24           # The last line of each such test's "main" program should be
25           #     exec(open("run_twice_magic.py").read())
26           self.write_files(
27               {
28                   "run_twice_magic.py": """ 
29                       import os 
30                       import sys 
31    
32                       import tornado.autoreload 
33    
34                       sys.stdout.flush() 
35    
36                       if "TESTAPP_STARTED" not in os.environ: 
37                           os.environ["TESTAPP_STARTED"] = "1" 
38                           tornado.autoreload._reload() 
39                       else: 
40                           os._exit(0) 
41                   """
42               }
43           )

45       def tearDown(self):
46           try:
47               shutil.rmtree(self.path)
48           except OSError:
49               # Windows disallows deleting files that are in use by
50               # another process, and even though we've waited for our
51               # child process below, it appears that its lock on these
52               # files is not guaranteed to be released by this point.
53               # Sleep and try again (once).
54               time.sleep(1)
55               shutil.rmtree(self.path)
56   
57       def write_files(self, tree, base_path=None):
58           """Write a directory tree to self.path. 
59    
60           tree is a dictionary mapping file names to contents, or 
61           sub-dictionaries representing subdirectories. 
62           """
63           if base_path is None:
64               base_path = self.path
65           for name, contents in tree.items():
66               if isinstance(contents, dict):
67                   os.mkdir(os.path.join(base_path, name))
68                   self.write_files(contents, os.path.join(base_path, name))
69               else:
70                   with open(os.path.join(base_path, name), "w", encoding="utf-8") as f:
71                       f.write(textwrap.dedent(contents))
72   
73       def run_subprocess(self, args):
74           # Make sure the tornado module under test is available to the test
75           # application
76           pythonpath = os.getcwd()
77           if "PYTHONPATH" in os.environ:
78               pythonpath += os.pathsep + os.environ["PYTHONPATH"]
79   
80           p = Popen(
81               args,
82               stdout=subprocess.PIPE,
83               env=dict(os.environ, PYTHONPATH=pythonpath),
84               cwd=self.path,
85               universal_newlines=True,
86               encoding="utf-8",
87           )
88   
89           # This timeout needs to be fairly generous for pypy due to jit
90           # warmup costs.
91           for i in range(40):
92               if p.poll() is not None:
93                   break
94               time.sleep(0.1)
95           else:
96               p.kill()
97               raise Exception("subprocess failed to terminate")
98   
99           out = p.communicate()[0]
100          self.assertEqual(p.returncode, 0)
101          return out

103      def test_reload(self):
104          main = """\ 
105  import sys 
106   
107  # In module mode, the path is set to the parent directory and we can import testapp. 
108  try: 
109      import testapp 
110  except ImportError: 
111      print("import testapp failed") 
112  else: 
113      print("import testapp succeeded") 
114   
115  spec = getattr(sys.modules[__name__], '__spec__', None) 
116  print(f"Starting {__name__=}, __spec__.name={getattr(spec, 'name', None)}") 
117  exec(open("run_twice_magic.py").read()) 
118  """
119  
120          # Create temporary test application
121          self.write_files(
122              {
123                  "testapp": {
124                      "__init__.py": "",
125                      "__main__.py": main,
126                  },
127              }
128          )
129  
130          # The autoreload wrapper should support all the same modes as the python interpreter.
131          # The wrapper itself should have no effect on this test so we try all modes with and
132          # without it.
133          for wrapper in [False, True]:
134              with self.subTest(wrapper=wrapper):
135                  with self.subTest(mode="module"):
136                      if wrapper:
137                          base_args = [sys.executable, "-m", "tornado.autoreload"]
138                      else:
139                          base_args = [sys.executable]
140                      # In module mode, the path is set to the parent directory and we can import
141                      # testapp. Also, the __spec__.name is set to the fully qualified module name.
142                      out = self.run_subprocess(base_args + ["-m", "testapp"])
143                      self.assertEqual(
144                          out,
145                          (
146                              "import testapp succeeded\n"
147                              + "Starting __name__='__main__', __spec__.name=testapp.__main__\n"
148                          )
149                          * 2,
150                      )
151  
152                  with self.subTest(mode="file"):
153                      out = self.run_subprocess(base_args + ["testapp/__main__.py"])
154                      # In file mode, we do not expect the path to be set so we can import testapp,
155                      # but when the wrapper is used the -m argument to the python interpreter
156                      # does this for us.
157                      expect_import = (
158                          "import testapp succeeded"
159                          if wrapper
160                          else "import testapp failed"
161                      )
162                      # In file mode there is no qualified module spec.
163                      self.assertEqual(
164                          out,
165                          f"{expect_import}\nStarting __name__='__main__', __spec__.name=None\n"
166                          * 2,
167                      )
168  
169                  with self.subTest(mode="directory"):
170                      # Running as a directory finds __main__.py like a module. It does not manipulate
171                      # sys.path but it does set a spec with a name of exactly __main__.
172                      out = self.run_subprocess(base_args + ["testapp"])
173                      expect_import = (
174                          "import testapp succeeded"
175                          if wrapper
176                          else "import testapp failed"
177                      )
178                      self.assertEqual(
179                          out,
180                          f"{expect_import}\nStarting __name__='__main__', __spec__.name=__main__\n"
181                          * 2,
182                      )

184      def test_reload_wrapper_preservation(self):
185          # This test verifies that when `python -m tornado.autoreload`
186          # is used on an application that also has an internal
187          # autoreload, the reload wrapper is preserved on restart.
188          main = """\ 
189  import sys 
190   
191  # This import will fail if path is not set up correctly 
192  import testapp 
193   
194  if 'tornado.autoreload' not in sys.modules: 
195      raise Exception('started without autoreload wrapper') 
196   
197  print('Starting') 
198  exec(open("run_twice_magic.py").read()) 
199  """
200  
201          self.write_files(
202              {
203                  "testapp": {
204                      "__init__.py": "",
205                      "__main__.py": main,
206                  },
207              }
208          )
209  
210          out = self.run_subprocess(
211              [sys.executable, "-m", "tornado.autoreload", "-m", "testapp"]
212          )
213          self.assertEqual(out, "Starting\n" * 2)

245          main = """\ 
246  import os 
247  import sys 
248   
249  if "TESTAPP_STARTED" in os.environ: 
250      print("exiting cleanly")

253      print("reloading")

255  """
256

258          self.write_files({"main.py": main})
