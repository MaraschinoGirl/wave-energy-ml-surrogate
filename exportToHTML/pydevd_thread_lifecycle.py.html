11   def pydevd_find_thread_by_id(thread_id):
12       try:
13           threads = threading.enumerate()
14           for i in threads:
15               tid = get_thread_id(i)
16               if thread_id == tid or thread_id.endswith("|" + tid):
17                   return i
18   
19           # This can happen when a request comes for a thread which was previously removed.
20           pydev_log.info("Could not find thread %s.", thread_id)
21           pydev_log.info("Available: %s.", ([get_thread_id(t) for t in threads],))
22       except:
23           pydev_log.exception()
24   
25       return None

28   def mark_thread_suspended(thread, stop_reason: int, original_step_cmd: int = -1):
29       info = set_additional_thread_info(thread)
30       info.suspend_type = PYTHON_SUSPEND
31       if original_step_cmd != -1:
32           stop_reason = original_step_cmd
33       thread.stop_reason = stop_reason
34   
35       # Note: don't set the 'pydev_original_step_cmd' here if unset.
36   
37       if info.pydev_step_cmd == -1:
38           # If the step command is not specified, set it to step into
39           # to make sure it'll break as soon as possible.
40           info.pydev_step_cmd = CMD_STEP_INTO
41           info.pydev_step_stop = None
42   
43       # Mark as suspended as the last thing.
44       info.pydev_state = STATE_SUSPEND
45       info.update_stepping_info()
46       return info

49   def internal_run_thread(thread, set_additional_thread_info):
50       info = set_additional_thread_info(thread)
51       info.pydev_original_step_cmd = -1
52       info.pydev_step_cmd = -1
53       info.pydev_step_stop = None
54       info.pydev_state = STATE_RUN
55       info.update_stepping_info()

58   def resume_threads(thread_id, except_thread=None):
59       pydev_log.info("Resuming threads: %s (except thread: %s)", thread_id, except_thread)
60       threads = []
61       if thread_id == "*":
62           threads = pydevd_utils.get_non_pydevd_threads()
63   
64       elif thread_id.startswith("__frame__:"):
65           pydev_log.critical("Can't make tasklet run: %s", thread_id)
66   
67       else:
68           threads = [pydevd_find_thread_by_id(thread_id)]
69   
70       for t in threads:
71           if t is None or t is except_thread:
72               pydev_log.info("Skipped resuming thread: %s", t)
73               continue
74   
75           internal_run_thread(t, set_additional_thread_info=set_additional_thread_info)

78   def suspend_all_threads(py_db, except_thread):
79       """ 
80       Suspend all except the one passed as a parameter. 
81       :param except_thread: 
82       """
83       if PYDEVD_USE_SYS_MONITORING:
84           pydevd_sys_monitoring.update_monitor_events(suspend_requested=True)
85   
86       pydev_log.info("Suspending all threads except: %s", except_thread)
87       all_threads = pydevd_utils.get_non_pydevd_threads()
88       for t in all_threads:
89           if getattr(t, "pydev_do_not_trace", None):
90               pass  # skip some other threads, i.e. ipython history saving thread from debug console
91           else:
92               if t is except_thread:
93                   continue
94               info = mark_thread_suspended(t, CMD_THREAD_SUSPEND)
95               frame = info.get_topmost_frame(t)
96   
97               # Reset the tracing as in this case as it could've set scopes to be untraced.
98               if frame is not None:
99                   try:
100                      py_db.set_trace_for_frame_and_parents(t.ident, frame)
101                  finally:
102                      frame = None
103  
104      if PYDEVD_USE_SYS_MONITORING:
105          # After suspending the frames we need the monitoring to be reset.
106          pydevd_sys_monitoring.restart_events()
