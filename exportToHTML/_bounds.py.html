6    from numbers import Real
7    
8    import numpy as np
9    
10   from ..preprocessing import LabelBinarizer
11   from ..utils._param_validation import Interval, StrOptions, validate_params
12   from ..utils.extmath import safe_sparse_dot
13   from ..utils.validation import check_array, check_consistent_length
14   
15   
16   @validate_params(
17       {
18           "X": ["array-like", "sparse matrix"],
19           "y": ["array-like"],
20           "loss": [StrOptions({"squared_hinge", "log"})],
21           "fit_intercept": ["boolean"],
22           "intercept_scaling": [Interval(Real, 0, None, closed="neither")],
23       },
24       prefer_skip_nested_validation=True,
25   )
26   def l1_min_c(X, y, *, loss="squared_hinge", fit_intercept=True, intercept_scaling=1.0):
27       """Return the lowest bound for C. 
28    
29       The lower bound for C is computed such that for C in (l1_min_C, infinity) 
30       the model is guaranteed not to be empty. This applies to l1 penalized 
31       classifiers, such as LinearSVC with penalty='l1' and 
32       linear_model.LogisticRegression with penalty='l1'. 
33    
34       This value is valid if class_weight parameter in fit() is not set. 
35    
36       Parameters 
37       ---------- 
38       X : {array-like, sparse matrix} of shape (n_samples, n_features) 
39           Training vector, where `n_samples` is the number of samples and 
40           `n_features` is the number of features. 
41    
42       y : array-like of shape (n_samples,) 
43           Target vector relative to X. 
44    
45       loss : {'squared_hinge', 'log'}, default='squared_hinge' 
46           Specifies the loss function. 
47           With 'squared_hinge' it is the squared hinge loss (a.k.a. L2 loss). 
48           With 'log' it is the loss of logistic regression models. 
49    
50       fit_intercept : bool, default=True 
51           Specifies if the intercept should be fitted by the model. 
52           It must match the fit() method parameter. 
53    
54       intercept_scaling : float, default=1.0 
55           When fit_intercept is True, instance vector x becomes 
56           [x, intercept_scaling], 
57           i.e. a "synthetic" feature with constant value equals to 
58           intercept_scaling is appended to the instance vector. 
59           It must match the fit() method parameter. 
60    
61       Returns 
62       ------- 
63       l1_min_c : float 
64           Minimum value for C. 
65    
66       Examples 
67       -------- 
68       >>> from sklearn.svm import l1_min_c 
69       >>> from sklearn.datasets import make_classification 
70       >>> X, y = make_classification(n_samples=100, n_features=20, random_state=42) 
71       >>> print(f"{l1_min_c(X, y, loss='squared_hinge', fit_intercept=True):.4f}") 
72       0.0044 
73       """
74   
75       X = check_array(X, accept_sparse="csc")
76       check_consistent_length(X, y)
77   
78       Y = LabelBinarizer(neg_label=-1).fit_transform(y).T
79       # maximum absolute value over classes and features
80       den = np.max(np.abs(safe_sparse_dot(Y, X)))
81       if fit_intercept:
82           bias = np.full(
83               (np.size(y), 1), intercept_scaling, dtype=np.array(intercept_scaling).dtype
84           )
85           den = max(den, abs(np.dot(Y, bias)).max())
86   
87       if den == 0.0:
88           raise ValueError(
89               "Ill-posed l1_min_c calculation: l1 will always "
90               "select zero coefficients for this data"
91           )
92       if loss == "squared_hinge":
93           return 0.5 / den
94       else:  # loss == 'log':
95           return 2.0 / den
