219  class TestReadDirectly:
220  
221      """ 
222          Feature: Read data directly from Dataset into a Numpy array 
223      """
224  
225      @pytest.mark.parametrize(
226          'source_shape,dest_shape,source_sel,dest_sel',
227          [
228              ((100,), (100,), np.s_[0:10], np.s_[50:60]),
229              ((70,), (100,), np.s_[50:60], np.s_[90:]),
230              ((30, 10), (20, 20), np.s_[:20, :], np.s_[:, :10]),
231              ((5, 7, 9), (6,), np.s_[2, :6, 3], np.s_[:]),
232          ])
233      def test_read_direct(self, writable_file, source_shape, dest_shape, source_sel, dest_sel):
234          source_values = np.arange(product(source_shape), dtype="int64").reshape(source_shape)
235          dset = writable_file.create_dataset("dset", source_shape, data=source_values)
236          arr = np.full(dest_shape, -1, dtype="int64")
237          expected = arr.copy()
238          expected[dest_sel] = source_values[source_sel]
239  
240          dset.read_direct(arr, source_sel, dest_sel)
241          np.testing.assert_array_equal(arr, expected)
242  
243      def test_no_sel(self, writable_file):
244          dset = writable_file.create_dataset("dset", (10,), data=np.arange(10, dtype="int64"))

246          dset.read_direct(arr)
247          np.testing.assert_array_equal(arr, np.arange(10, dtype="int64"))
248
