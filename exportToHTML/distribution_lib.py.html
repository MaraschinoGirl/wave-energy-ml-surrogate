42   @keras_export("keras.distribution.initialize")
43   def initialize(job_addresses=None, num_processes=None, process_id=None):
44       """Initialize the distribution system for multi-host/process setting. 
45    
46       Calling `initialize` will prepare the backend for execution on multi-host 
47       GPU or TPUs. It should be called before any computations. 
48    
49       Note that the parameters can also be injected via environment variables, 
50       which can be better controlled by the launch script at startup time. 
51       For certain backend that also rely on the environment variables to 
52       configure, Keras will properly forward them. 
53    
54       Args: 
55           job_addresses: string. Comma separated IP addresses for all the jobs 
56               that will form the whole computation cluster. Note that for JAX 
57               backend, only the address for job 0 (coodinator) is needed. For 
58               certain runtime like cloud TPU, this value can be `None`, and the 
59               backend will figure it out with the TPU environment variables. You 
60               can also config this value via environment variable 
61               `KERAS_DISTRIBUTION_JOB_ADDRESSES`. 
62           num_processes: int. The number of worker/processes that will form the 
63               whole computation cluster. For certain runtime like cloud TPU, this 
64               value can be `None`, and the backend will figure it out with the TPU 
65               environment variables. You can also configure this value via 
66               environment variable `KERAS_DISTRIBUTION_NUM_PROCESSES`. 
67           process_id: int. The ID number of the current worker/process. The value 
68               should be ranged from `0` to `num_processes - 1`. `0` will indicate 
69               the current worker/process is the master/coordinate job. You can 
70               also configure this value via environment variable 
71               `KERAS_DISTRIBUTION_PROCESS_ID`. 
72    
73           Example: 
74               Suppose there are two GPU processes, and process 0 is running at 
75               address `10.0.0.1:1234`, and process 1 is running at address 
76               `10.0.0.2:2345`. To configure such cluster, you can run 
77    
78           On process 0: 
79           ```python 
80           keras.distribute.initialize( 
81               job_addresses="10.0.0.1:1234,10.0.0.2:2345", 
82               num_processes=2, 
83               process_id=0) 
84           ``` 
85    
86           On process 1: 
87           ```python 
88           keras.distribute.initialize( 
89               job_addresses="10.0.0.1:1234,10.0.0.2:2345", 
90               num_processes=2, 
91               process_id=1) 
92           ``` 
93    
94           or via the environment variables: 
95           On process 0: 
96           ```python 
97           os.environ[ 
98               "KERAS_DISTRIBUTION_JOB_ADDRESSES"] = "10.0.0.1:1234,10.0.0.2:2345" 
99           os.environ["KERAS_DISTRIBUTION_NUM_PROCESSES"] = "2" 
100          os.environ["KERAS_DISTRIBUTION_PROCESS_ID"] = "0" 
101          keras.distribute.initialize() 
102          ``` 
103   
104          On process 1: 
105          ```python 
106          os.environ[ 
107              "KERAS_DISTRIBUTION_JOB_ADDRESSES"] = "10.0.0.1:1234,10.0.0.2:2345" 
108          os.environ["KERAS_DISTRIBUTION_NUM_PROCESSES"] = "2" 
109          os.environ["KERAS_DISTRIBUTION_PROCESS_ID"] = "1" 
110          keras.distribute.initialize() 
111          ``` 
112   
113          Also note that for JAX backend, the `job_addresses` can be further 
114          reduced to just the master/coordinator address, which is 
115          `10.0.0.1:1234`. 
116      """
117      if (
118          job_addresses is None
119          and "KERAS_DISTRIBUTION_JOB_ADDRESSES" in os.environ
120      ):
121          job_addresses = os.environ["KERAS_DISTRIBUTION_JOB_ADDRESSES"]
122      if (
123          num_processes is None
124          and "KERAS_DISTRIBUTION_NUM_PROCESSES" in os.environ
125      ):
126          num_processes = int(os.environ["KERAS_DISTRIBUTION_NUM_PROCESSES"])
127      if process_id is None and "KERAS_DISTRIBUTION_PROCESS_ID" in os.environ:
128          process_id = int(os.environ["KERAS_DISTRIBUTION_PROCESS_ID"])
129      distribution_lib.initialize(job_addresses, num_processes, process_id)

132  @keras_export("keras.distribution.DeviceMesh")
133  class DeviceMesh:
134      """A cluster of computation devices for distributed computation. 
135   
136      This API is aligned with `jax.sharding.Mesh` and `tf.dtensor.Mesh`, which 
137      represents the computation devices in the global context. 
138   
139      See more details in [jax.sharding.Mesh]( 
140          https://jax.readthedocs.io/en/latest/jax.sharding.html#jax.sharding.Mesh) 
141      and [tf.dtensor.Mesh]( 
142          https://www.tensorflow.org/api_docs/python/tf/experimental/dtensor/Mesh). 
143   
144      Args: 
145          shape: tuple of list of integers. The shape of the overall 
146              `DeviceMesh`, e.g. `(8,)` for a data parallel only distribution, 
147              or `(4, 2)` for a model+data parallel distribution. 
148          axis_names: List of string. The logical name of the each axis for 
149              the `DeviceMesh`. The length of the `axis_names` should match to 
150              the rank of the `shape`. The `axis_names` will be used to 
151              match/create the `TensorLayout` when distribute the data and 
152              variables. 
153          devices: Optional list of devices. Defaults to all the available 
154              devices locally from `keras.distribution.list_devices()`. 
155      """
156  
157      def __init__(
158          self,
159          shape,
160          axis_names,
161          devices=None,
162      ):
163          if not shape or not axis_names:
164              raise ValueError(
165                  "Shape and axis_names cannot be empty. Received: "
166                  f"shape={shape}, axis_names={axis_names}"
167              )
168  
169          if len(shape) != len(axis_names):
170              raise ValueError(
171                  "Shape and axis_names should have same size. "
172                  f"Received: shape={shape}, axis_names={axis_names}"
173              )
174          if devices is None:
175              devices = list_devices()
176          devices = np.array(devices)
177          if np.prod(shape) != np.prod(devices.shape):
178              raise ValueError(
179                  "Shape does not match the number of devices. "
180                  f"Received: shape={shape}; devices.shape="
181                  f"{devices.shape}"
182              )
183  
184          self._shape = shape
185          self._axis_names = axis_names
186          self._devices = np.reshape(devices, shape)
187  
188      @property
189      def shape(self):
190          return self._shape
191  
192      @property
193      def axis_names(self):
194          return self._axis_names
195  
196      @property
197      def devices(self):
198          return self._devices
199  
200      def __repr__(self):
201          return (
202              f"<{self.__class__.__name__} "
203              f"shape={self.shape}, axis_names={self.axis_names}>"
204          )
205  
206      def __str__(self):
207          return self.__repr__()

210  @keras_export("keras.distribution.TensorLayout")
211  class TensorLayout:
212      """A layout to apply to a tensor. 
213   
214      This API is aligned with `jax.sharding.NamedSharding` 
215      and `tf.dtensor.Layout`. 
216   
217      See more details in [jax.sharding.NamedSharding]( 
218          https://jax.readthedocs.io/en/latest/jax.sharding.html#jax.sharding.NamedSharding) 
219      and [tf.dtensor.Layout]( 
220          https://www.tensorflow.org/api_docs/python/tf/experimental/dtensor/Layout). 
221   
222      Args: 
223          axes: tuple of strings that should map to the `axis_names` in 
224              a `DeviceMesh`. For any dimensions that doesn't need any sharding, 
225              A `None` can be used a placeholder. 
226          device_mesh: Optional `DeviceMesh` that will be used to create 
227              the layout. The actual mapping of tensor to physical device 
228              is not known until the mesh is specified. 
229      """
230  
231      def __init__(self, axes, device_mesh=None):
232          self._axes = tuple(axes)
233          self._device_mesh = device_mesh
234          self._validate_axes()
235  
236      @property
237      def axes(self):
238          return self._axes
239  
240      @property
241      def device_mesh(self):
242          return self._device_mesh
243  
244      @device_mesh.setter
245      def device_mesh(self, device_mesh):
246          if self._device_mesh is not None:
247              raise ValueError(
248                  "Cannot override device mesh value. Existing "
249                  f"value is {self._device_mesh}"
250              )
251          self._device_mesh = device_mesh
252          self._validate_axes()
253  
254      def _validate_axes(self):
255          if self._device_mesh:
256              valid_axis_names = set(self._device_mesh.axis_names)
257              axis_names = set(self._axes) - set([None])
258              if axis_names - valid_axis_names:
259                  raise ValueError(
260                      "Invalid axis names for Layout. Valid axis "
261                      f"names: {valid_axis_names}, Got {axis_names}"
262                  )
263  
264      def __repr__(self):
265          return (
266              f"<{self.__class__.__name__} "
267              f"axes={self.axes}, device_mesh={self.device_mesh}>"
268          )
269  
270      def __str__(self):
271          return self.__repr__()

274  class Distribution:
275      """Base class for variable distribution strategies. 
276   
277      A `Distribution` has following key functionalities: 
278   
279      1. Distribute the model variables to a `DeviceMesh`. 
280      2. Distribute the input data to a `DeviceMesh`. 
281      3. Distribute an intermediate state tensor in the model. 
282   
283      It can create a context scope so that the framework to properly detect the 
284      `Distribution` and distribute the variable/data accordingly. 
285   
286      Args: 
287          device_mesh: A `DeviceMesh` instance. 
288      """
289  
290      def __init__(self, device_mesh, batch_dim_name=None):
291          self._device_mesh = device_mesh
292          self._batch_dim_name = batch_dim_name
293  
294      def get_data_layout(self, data_shape):
295          """Retrieve the `TensorLayout` for the input data. 
296   
297          Args: 
298              data_shape: shape for the input data in list or tuple format. 
299   
300          Returns: 
301              The `TensorLayout` for the data, which can be used by 
302              `backend.distribute_value()` to redistribute a input data. 
303          """
304          raise NotImplementedError()
305  
306      def get_variable_layout(self, variable):
307          """Retrieve the `TensorLayout` for the variable. 
308   
309          Args: 
310              variable: A `Variable` instance. 
311   
312          return: 
313              The `TensorLayout` for the variable, which can be used by 
314              `backend.distribute_value()` to redistribute a variable. 
315          """
316          raise NotImplementedError()
317  
318      def get_tensor_layout(self, path):
319          """Retrieve the `TensorLayout` for the intermediate tensor. 
320   
321          Args: 
322              path: a string path for the corresponding tensor. 
323   
324          return: 
325              The `TensorLayout` for the intermediate tensor, which can be used 
326              by `backend.relayout()` to reshard the tensor. Could also return 
327              None. 
328          """
329          raise NotImplementedError()
330  
331      @contextlib.contextmanager
332      def scope(self):
333          """Context manager to make the `Distribution` current."""
334          original_scope = distribution()
335          set_distribution(self)
336          try:
337              yield
338          finally:
339              set_distribution(original_scope)
340  
341      @property
342      def device_mesh(self):
343          return self._device_mesh
344  
345      @property
346      def batch_dim_name(self):
347          return self._batch_dim_name
348  
349      def distribute_dataset(self, dataset):
350          """Create a distributed dataset instance from the original user dataset. 
351   
352          Args: 
353              dataset: the original global dataset instance. Only 
354              `tf.data.Dataset` is supported at the moment. 
355   
356          Returns: 
357              a sharded `tf.data.Dataset` instance, which will produce data for 
358              the current local worker/process. 
359          """
360          raise NotImplementedError()
361  
362      def __repr__(self):
363          return f"<{self.__class__.__name__} device_mesh={self.device_mesh}>"
364  
365      def __str__(self):
366          return self.__repr__()

369  @keras_export("keras.distribution.DataParallel")
370  class DataParallel(Distribution):
371      """Distribution for data parallelism. 
372   
373      You can choose to create this instance by either specifying 
374      the `device_mesh` or `devices` arguments (but not both). 
375   
376      The `device_mesh` argument is expected to be a `DeviceMesh` instance, 
377      and is expected to be 1D only. In case that the mesh has multiple axes, 
378      then the first axis will be treated as the data parallel dimension 
379      (and a warning will be raised). 
380   
381      When a list of `devices` are provided, they will be used to construct a 
382      1D mesh. 
383   
384      When both `mesh` and `devices` are absent, then `list_devices()` 
385      will be used to detect any available devices and create a 1D mesh from 
386      them. 
387   
388      Args: 
389          device_mesh: Optional `DeviceMesh` instance. 
390          devices: Optional list of devices. 
391          auto_shard_dataset: Automatically shard the dataset amongst processes. 
392              Defaults to true. 
393      """
394  
395      def __init__(self, device_mesh=None, devices=None, auto_shard_dataset=True):
396          if device_mesh:
397              self._initialize_with_device_mesh(device_mesh)
398          elif devices:
399              self._initialize_mesh_from_devices(devices)
400          else:
401              self._initialize_mesh_from_list_devices()
402  
403          # Those following attributes might get convert to public methods.
404          self._num_process = distribution_lib.num_processes()
405          self._process_id = distribution_lib.process_id()
406          self._is_multi_process = self._num_process > 1
407          self._auto_shard_dataset = auto_shard_dataset
408  
409      def _initialize_with_device_mesh(self, device_mesh):
410          if not isinstance(device_mesh, DeviceMesh):
411              raise ValueError(
412                  "Expect `mesh` to be an instance of `DeviceMesh`. "
413                  f"Received: mesh={device_mesh} (of type {type(device_mesh)})"
414              )
415          super().__init__(device_mesh, device_mesh.axis_names[0])
416          if self.device_mesh.devices.ndim != 1:
417              warnings.warn(
418                  "Expect the input mesh to be 1D, but received "
419                  "mesh.devices.ndim=%d. "
420                  "The first axis will be used for data-parallel sharding.",
421                  device_mesh.devices.ndim,
422              )
423  
424      def _initialize_mesh_from_devices(self, devices):
425          devices = np.array(devices)
426          device_mesh = DeviceMesh(
427              shape=devices.shape,
428              axis_names=[DEFAULT_BATCH_DIM_NAME],
429              devices=devices,
430          )
431          super().__init__(device_mesh, DEFAULT_BATCH_DIM_NAME)
432  
433      def _initialize_mesh_from_list_devices(self):
434          devices = np.array(list_devices())
435          device_mesh = DeviceMesh(
436              shape=devices.shape,
437              axis_names=[DEFAULT_BATCH_DIM_NAME],
438              devices=devices,
439          )
440          super().__init__(device_mesh, DEFAULT_BATCH_DIM_NAME)
441  
442      def get_data_layout(self, data_shape):
443          data_shard_spec = [None] * len(data_shape)
444          data_shard_spec[0] = self.batch_dim_name  # Shard on the first dim
445          return TensorLayout(data_shard_spec, self.device_mesh)
446  
447      def get_variable_layout(self, variable):
448          variable_shard_spec = [None] * len(variable.shape)
449          return TensorLayout(variable_shard_spec, self.device_mesh)
450  
451      def get_tensor_layout(self, path):
452          # For data parallel training, the intermediate state is not changed.
453          return None
454  
455      def distribute_dataset(self, dataset):
456          from tensorflow.python.data.experimental.ops import (
457              distribute as tf_data_distribute,
458          )
459  
460          from keras.src.utils.module_utils import tensorflow as tf
461  
462          if not isinstance(dataset, tf.data.Dataset):
463              raise ValueError(
464                  "Only `tf.data.Dataset` is supported for "
465                  f"sharding, got {type(dataset)}"
466              )
467          if not self._is_multi_process or not self._auto_shard_dataset:
468              return dataset
469  
470          batch_size = tf_data_distribute.compute_batch_size(dataset)
471          if batch_size.numpy() < 0:
472              raise ValueError(
473                  "The batch size of the input dataset is "
474                  "unknown. Please config the batch size for "
475                  "the input dataset, e.g via `dataset.batch(batch_size)`"
476              )
477          per_worker_batch_size = tf_data_distribute.batch_sizes_for_worker(
478              global_batch_size=batch_size,
479              num_workers=self._num_process,
480              num_replicas_per_worker=1,  # We hard code this for now.
481              worker_index=self._process_id,
482          )
483          distributed_dataset = dataset.rebatch(per_worker_batch_size)
484          distributed_dataset = tf_data_distribute._AutoShardDataset(
485              distributed_dataset,
486              num_workers=self._num_process,
487              index=self._process_id,
488              num_replicas=self._num_process,
489          )
490          return distributed_dataset.prefetch(tf.data.AUTOTUNE)

493  @keras_export("keras.distribution.ModelParallel")
494  class ModelParallel(Distribution):
495      """Distribution that shards model variables. 
496   
497      Compare to `DataParallel` which replicates the variables across all devices, 
498      `ModelParallel` allows you to shard variables in addition to the input data. 
499   
500      To construct a `ModelParallel` distribution, you need to provide a 
501      `DeviceMesh` and a `LayoutMap`. 
502   
503      1. `DeviceMesh` contains physical device information. The axis names in 
504          the mesh will be used to map the variable and data layout. 
505      2. `LayoutMap` contains the mapping between variable paths to their 
506          corresponding `TensorLayout`. 
507   
508      Example: 
509   
510      ```python 
511      devices = list_devices()    # Assume there are 8 devices. 
512   
513      # Create a mesh with 2 devices for data parallelism and 4 devices for 
514      # model parallelism. 
515      device_mesh = DeviceMesh(shape=(2, 4), axis_names=('batch', 'model'), 
516                               devices=devices) 
517      # Create a layout map that shard the `Dense` layer and `Conv2D` 
518      # layer variables on the last dimension. 
519      # Based on the `device_mesh`, this means the variables 
520      # will be split across 4 devices. Any other variable that doesn't 
521      # match any key in the layout map will be fully replicated. 
522      layout_map = LayoutMap(device_mesh) 
523      layout_map['dense.*kernel'] = (None, 'model') 
524      layout_map['dense.*bias'] = ('model',) 
525      layout_map['conv2d.*kernel'] = (None, None, None, 'model') 
526      layout_map['conv2d.*bias'] = ('model',) 
527   
528      distribution = ModelParallel( 
529          layout_map=layout_map, 
530          batch_dim_name='batch', 
531      ) 
532   
533      # Set the global distribution, or via `with distribution.scope():` 
534      set_distribution(distribution) 
535   
536      model = model_creation() 
537      model.compile() 
538      model.fit(data) 
539      ``` 
540   
541      You can quickly update the device mesh shape to change the sharding factor 
542      of the variables. E.g. 
543   
544      ```python 
545      # With only the shape change for the device mesh, the variables will be 
546      # sharded across 8 devices instead of 4, which further reduces the memory 
547      # footprint of variables on each of the device. 
548      device_mesh = DeviceMesh( 
549          shape=(1, 8), 
550          axis_names=('batch', 'model'), 
551          devices=devices, 
552      ) 
553      ``` 
554   
555      To figure out a proper layout mapping rule for all the model variables, you 
556      can first list out all the model variable paths, which will be used as the 
557      key to map the variables to `TensorLayout`. 
558   
559      e.g. 
560   
561      ```python 
562      model = create_model() 
563      for v in model.variables: 
564          print(v.path) 
565      ``` 
566   
567      Args: 
568          layout_map: `LayoutMap` instance which map the variable path to the 
569              corresponding tensor layout. 
570          batch_dim_name: Optional string, the axis name in the device mesh 
571              (of the `layout_map` object) 
572              that will be used to distribute data. If unspecified, the 
573              first axis from the device mesh will be used. 
574      """
575  
576      def __init__(self, *, layout_map=None, batch_dim_name=None, **kwargs):
577          kwargs.pop("device_mesh", None)
578          if layout_map is None:
579              raise ValueError("You must specify a layout_map argument.")
580          if not isinstance(layout_map, LayoutMap):
581              raise ValueError(
582                  "Argument `layout_map` must be a `LayoutMap` instance. "
583                  f"Received: layout_map={layout_map}"
584              )
585          device_mesh = layout_map.device_mesh
586          super().__init__(device_mesh)
587          self._layout_map = layout_map
588          self._batch_dim_name = batch_dim_name or self.device_mesh.axis_names[0]
589  
590          # Those following attributes might get convert to public methods.
591          self._num_process = distribution_lib.num_processes()
592          self._process_id = distribution_lib.process_id()
593          self._is_multi_process = self._num_process > 1
594  
595      def get_data_layout(self, data_shape):
596          data_shard_spec = [None] * len(data_shape)
597          data_shard_spec[0] = self.batch_dim_name  # Shard on the first dim
598          return TensorLayout(data_shard_spec, self.device_mesh)
599  
600      def get_variable_layout(self, variable):
601          variable_layout = self._layout_map[variable.path]
602          if variable_layout is not None:
603              return variable_layout
604          variable_shard_spec = [None] * len(variable.shape)
605          return TensorLayout(variable_shard_spec, self.device_mesh)
606  
607      def get_tensor_layout(self, path):
608          return self._layout_map[path]
609  
610      def distribute_dataset(self, dataset):
611          from tensorflow.python.data.experimental.ops import (
612              distribute as tf_data_distribute,
613          )
614  
615          from keras.src.utils.module_utils import tensorflow as tf
616  
617          if not isinstance(dataset, tf.data.Dataset):
618              raise ValueError(
619                  "Only `tf.data.Dataset` is supported for "
620                  f"sharding, got {type(dataset)}"
621              )
622          if not self._is_multi_process:
623              return dataset
624  
625          global_batch_size = tf_data_distribute.compute_batch_size(dataset)
626          if global_batch_size.numpy() < 0:
627              raise ValueError(
628                  "The batch size of the input dataset is "
629                  "unknown. Please config the batch size for "
630                  "the input dataset, e.g via `dataset.batch(batch_size)`"
631              )
632  
633          # We need to compute the per-process/worker/host batch size.
634          # This will depend on how many model replicas we have on each process.
635          # Note that this might be smaller than one if model replicas are sharded
636          # across multiple processes.
637          mesh_batch_dim_index = self.device_mesh.axis_names.index(
638              self.batch_dim_name
639          )
640          num_model_replicas = self.device_mesh.shape[mesh_batch_dim_index]
641          if num_model_replicas == 1:
642              # No sharding is needed in this case. Each process will have the
643              # global batch size, and data from the iterator will need to be
644              # replicated across all processes.
645              return dataset.prefetch(tf.data.AUTOTUNE)
646          num_model_replicas_per_process = num_model_replicas / self._num_process
647          if num_model_replicas_per_process >= 1:
648              # Each process will have one or more full model replicas. Data will
649              # be sharded across all processes without replication.
650              if global_batch_size % self._num_process != 0:
651                  raise ValueError(
652                      "Global batch size must be divisible by the number of "
653                      f"processes. `global_batch_size`={global_batch_size} and "
654                      f"`num_process`={self._num_process}"
655                  )
656              per_process_batch_size = global_batch_size // self._num_process
657              distributed_dataset = dataset.rebatch(per_process_batch_size)
658              distributed_dataset = distributed_dataset.shard(
659                  num_shards=self._num_process,
660                  index=self._process_id,
661              )
662              return distributed_dataset.prefetch(tf.data.AUTOTUNE)
663          else:
664              # Model replicas are sharded across multiple processes. Data will be
665              # sharded across model replicas, and replicated across processes
666              # within the same model replica.
667              if global_batch_size % num_model_replicas != 0:
668                  raise ValueError(
669                      "Global batch size must be divisible by the number of "
670                      f"replicas. `global_batch_size`={global_batch_size} and "
671                      f"`num_model_replicas`={num_model_replicas}"
672                  )
673              per_process_batch_size = global_batch_size // num_model_replicas
674              distributed_dataset = dataset.rebatch(per_process_batch_size)
675              processes_per_replica = self._num_process // num_model_replicas
676              # TODO: Figure out what the convention is for data sharding id.
677              data_shard_id = self._process_id % processes_per_replica
678              distributed_dataset = distributed_dataset.shard(
679                  num_shards=num_model_replicas,
680                  index=data_shard_id,
681              )
682              return distributed_dataset.prefetch(tf.data.AUTOTUNE)

685  @keras_export("keras.distribution.LayoutMap")
686  class LayoutMap(collections.abc.MutableMapping):
687      """A dict-like object that maps string to `TensorLayout` instances. 
688   
689      `LayoutMap` uses a string as key and a `TensorLayout` as value. There is a 
690      behavior difference between a normal Python dict and this class. The string 
691      key will be treated as a regex when retrieving the value. See the docstring 
692      of `get` for more details. 
693   
694      See below for a usage example. You can define the naming schema 
695      of the `TensorLayout`, and then retrieve the corresponding 
696      `TensorLayout` instance. 
697   
698      In the normal case, the key to query is usually the `variable.path`, which 
699      is the identifier of the variable. 
700   
701      As shortcut, tuple or list of axis names are also allowed when inserting 
702      as value, and will be converted to `TensorLayout`. 
703   
704      ```python 
705      layout_map = LayoutMap(device_mesh) 
706      layout_map['dense.*kernel'] = (None, 'model') 
707      layout_map['dense.*bias'] = ('model',) 
708      layout_map['conv2d.*kernel'] = (None, None, None, 'model') 
709      layout_map['conv2d.*bias'] = ('model',) 
710   
711      layout_1 = layout_map['dense_1.kernel']             # layout_1 == layout_2d 
712      layout_2 = layout_map['dense_1.bias']               # layout_2 == layout_1d 
713      layout_3 = layout_map['dense_2.kernel']             # layout_3 == layout_2d 
714      layout_4 = layout_map['dense_2.bias']               # layout_4 == layout_1d 
715      layout_5 = layout_map['my_model/conv2d_123/kernel'] # layout_5 == layout_4d 
716      layout_6 = layout_map['my_model/conv2d_123/bias']   # layout_6 == layout_1d 
717      layout_7 = layout_map['my_model/conv3d_1/kernel']   # layout_7 == None 
718      layout_8 = layout_map['my_model/conv3d_1/bias']     # layout_8 == None 
719      ``` 
720   
721      Args: 
722          device_mesh: `keras.distribution.DeviceMesh` instance. 
723      """
724  
725      def __init__(self, device_mesh):
726          self._layout_map = collections.OrderedDict()
727          self._device_mesh = device_mesh
728  
729      def __getitem__(self, key):
730          """Retrieves the corresponding layout by the string key. 
731   
732          When there isn't an exact match, all the existing keys in the layout map 
733          will be treated as a regex and map against the input key again. When 
734          there are multiple matches for the regex, an `ValueError` will be 
735          raised. Returns `None` if there isn't any match found. 
736   
737          Args: 
738              key: String key to query a layout. 
739   
740          Returns: 
741              Corresponding layout based on the query. 
742          """
743          if key in self._layout_map:
744              return self._layout_map[key]
745  
746          matching_keys = []
747          for k in self._layout_map:
748              if re.search(k, key):
749                  matching_keys.append(k)
750          if len(matching_keys) > 1:
751              raise ValueError(
752                  f"Path '{key}' matches multiple layout "
753                  f"specification keys: {matching_keys}. Please make "
754                  "sure each tensor/variable path only matches at most "
755                  "one layout specification key in the LayoutMap."
756              )
757          elif len(matching_keys) == 1:
758              return self._layout_map[matching_keys[0]]
759          return None
760  
761      def __setitem__(self, key, layout):
762          """Insert TensorLayout to the LayoutMap. 
763   
764          Args: 
765              key: String key for the `TensorLayout`. 
766              layout: The `TensorLayout`. As a shortcut, tuple of string and None 
767                  are also acceptable, and will be converted to `TensorLayout`. 
768          """
769          if key in self._layout_map:
770              raise ValueError(
771                  f"{key} already exist in the LayoutMap with "
772                  f"value {self._layout_map[key]}. Please make sure to "
773                  "not use duplicated keys."
774              )
775          if isinstance(layout, tuple):
776              layout = TensorLayout(axes=layout, device_mesh=None)
777  
778          if not isinstance(layout, TensorLayout):
779              raise ValueError(
780                  f"{layout} should be a TensorLayout type, got {type(layout)}"
781              )
782          self._maybe_populate_device_mesh(layout)
783          self._layout_map[key] = layout
784  
785      def __delitem__(self, key):
786          # let the dict to handle the key missing error
787          return self._layout_map.pop(key)
788  
789      def __len__(self):
790          return len(self._layout_map)
791  
792      def __iter__(self):
793          return iter(self._layout_map)
794  
795      @property
796      def device_mesh(self):
797          return self._device_mesh
798  
799      def _maybe_populate_device_mesh(self, layout):
800          if layout.device_mesh is None and self.device_mesh is not None:
801              layout.device_mesh = self.device_mesh
802  
803  
804  LayoutMap.get.__doc__ = LayoutMap.__getitem__.__doc__
