51   def MAKE_WPARAM(wParam):
52       """ 
53       Convert arguments to the WPARAM type. 
54       Used automatically by SendMessage, PostMessage, etc. 
55       You shouldn't need to call this function. 
56       """
57       wParam = ctypes.cast(wParam, LPVOID).value
58       if wParam is None:
59           wParam = 0
60       return wParam
61   
62   
63   def MAKE_LPARAM(lParam):
64       """ 
65       Convert arguments to the LPARAM type. 
66       Used automatically by SendMessage, PostMessage, etc. 
67       You shouldn't need to call this function. 
68       """
69       return ctypes.cast(lParam, LPARAM)

412  class Point(object):
413      """ 
414      Python wrapper over the L{POINT} class. 
415   
416      @type x: int 
417      @ivar x: Horizontal coordinate 
418      @type y: int 
419      @ivar y: Vertical coordinate 
420      """
421  
422      def __init__(self, x=0, y=0):
423          """ 
424          @see: L{POINT} 
425          @type  x: int 
426          @param x: Horizontal coordinate 
427          @type  y: int 
428          @param y: Vertical coordinate 
429          """
430          self.x = x
431          self.y = y
432  
433      def __iter__(self):
434          return (self.x, self.y).__iter__()
435  
436      def __len__(self):
437          return 2
438  
439      def __getitem__(self, index):
440          return (self.x, self.y)[index]
441  
442      def __setitem__(self, index, value):
443          if index == 0:
444              self.x = value
445          elif index == 1:
446              self.y = value
447          else:
448              raise IndexError("index out of range")
449  
450      @property
451      def _as_parameter_(self):
452          """ 
453          Compatibility with ctypes. 
454          Allows passing transparently a Point object to an API call. 
455          """
456          return POINT(self.x, self.y)
457  
458      def screen_to_client(self, hWnd):
459          """ 
460          Translates window screen coordinates to client coordinates. 
461   
462          @see: L{client_to_screen}, L{translate} 
463   
464          @type  hWnd: int or L{HWND} or L{system.Window} 
465          @param hWnd: Window handle. 
466   
467          @rtype:  L{Point} 
468          @return: New object containing the translated coordinates. 
469          """
470          return ScreenToClient(hWnd, self)
471  
472      def client_to_screen(self, hWnd):
473          """ 
474          Translates window client coordinates to screen coordinates. 
475   
476          @see: L{screen_to_client}, L{translate} 
477   
478          @type  hWnd: int or L{HWND} or L{system.Window} 
479          @param hWnd: Window handle. 
480   
481          @rtype:  L{Point} 
482          @return: New object containing the translated coordinates. 
483          """
484          return ClientToScreen(hWnd, self)
485  
486      def translate(self, hWndFrom=HWND_DESKTOP, hWndTo=HWND_DESKTOP):
487          """ 
488          Translate coordinates from one window to another. 
489   
490          @note: To translate multiple points it's more efficient to use the 
491              L{MapWindowPoints} function instead. 
492   
493          @see: L{client_to_screen}, L{screen_to_client} 
494   
495          @type  hWndFrom: int or L{HWND} or L{system.Window} 
496          @param hWndFrom: Window handle to translate from. 
497              Use C{HWND_DESKTOP} for screen coordinates. 
498   
499          @type  hWndTo: int or L{HWND} or L{system.Window} 
500          @param hWndTo: Window handle to translate to. 
501              Use C{HWND_DESKTOP} for screen coordinates. 
502   
503          @rtype:  L{Point} 
504          @return: New object containing the translated coordinates. 
505          """
506          return MapWindowPoints(hWndFrom, hWndTo, [self])

509  class Rect(object):
510      """ 
511      Python wrapper over the L{RECT} class. 
512   
513      @type   left: int 
514      @ivar   left: Horizontal coordinate for the top left corner. 
515      @type    top: int 
516      @ivar    top: Vertical coordinate for the top left corner. 
517      @type  right: int 
518      @ivar  right: Horizontal coordinate for the bottom right corner. 
519      @type bottom: int 
520      @ivar bottom: Vertical coordinate for the bottom right corner. 
521   
522      @type  width: int 
523      @ivar  width: Width in pixels. Same as C{right - left}. 
524      @type height: int 
525      @ivar height: Height in pixels. Same as C{bottom - top}. 
526      """
527  
528      def __init__(self, left=0, top=0, right=0, bottom=0):
529          """ 
530          @see: L{RECT} 
531          @type    left: int 
532          @param   left: Horizontal coordinate for the top left corner. 
533          @type     top: int 
534          @param    top: Vertical coordinate for the top left corner. 
535          @type   right: int 
536          @param  right: Horizontal coordinate for the bottom right corner. 
537          @type  bottom: int 
538          @param bottom: Vertical coordinate for the bottom right corner. 
539          """
540          self.left = left
541          self.top = top
542          self.right = right
543          self.bottom = bottom
544  
545      def __iter__(self):
546          return (self.left, self.top, self.right, self.bottom).__iter__()
547  
548      def __len__(self):
549          return 2
550  
551      def __getitem__(self, index):
552          return (self.left, self.top, self.right, self.bottom)[index]
553  
554      def __setitem__(self, index, value):
555          if index == 0:
556              self.left = value
557          elif index == 1:
558              self.top = value
559          elif index == 2:
560              self.right = value
561          elif index == 3:
562              self.bottom = value
563          else:
564              raise IndexError("index out of range")
565  
566      @property
567      def _as_parameter_(self):
568          """ 
569          Compatibility with ctypes. 
570          Allows passing transparently a Point object to an API call. 
571          """
572          return RECT(self.left, self.top, self.right, self.bottom)
573  
574      def __get_width(self):
575          return self.right - self.left
576  
577      def __get_height(self):
578          return self.bottom - self.top
579  
580      def __set_width(self, value):
581          self.right = value - self.left
582  
583      def __set_height(self, value):
584          self.bottom = value - self.top
585  
586      width = property(__get_width, __set_width)
587      height = property(__get_height, __set_height)
588  
589      def screen_to_client(self, hWnd):
590          """ 
591          Translates window screen coordinates to client coordinates. 
592   
593          @see: L{client_to_screen}, L{translate} 
594   
595          @type  hWnd: int or L{HWND} or L{system.Window} 
596          @param hWnd: Window handle. 
597   
598          @rtype:  L{Rect} 
599          @return: New object containing the translated coordinates. 
600          """
601          topleft = ScreenToClient(hWnd, (self.left, self.top))
602          bottomright = ScreenToClient(hWnd, (self.bottom, self.right))
603          return Rect(topleft.x, topleft.y, bottomright.x, bottomright.y)
604  
605      def client_to_screen(self, hWnd):
606          """ 
607          Translates window client coordinates to screen coordinates. 
608   
609          @see: L{screen_to_client}, L{translate} 
610   
611          @type  hWnd: int or L{HWND} or L{system.Window} 
612          @param hWnd: Window handle. 
613   
614          @rtype:  L{Rect} 
615          @return: New object containing the translated coordinates. 
616          """
617          topleft = ClientToScreen(hWnd, (self.left, self.top))
618          bottomright = ClientToScreen(hWnd, (self.bottom, self.right))
619          return Rect(topleft.x, topleft.y, bottomright.x, bottomright.y)
620  
621      def translate(self, hWndFrom=HWND_DESKTOP, hWndTo=HWND_DESKTOP):
622          """ 
623          Translate coordinates from one window to another. 
624   
625          @see: L{client_to_screen}, L{screen_to_client} 
626   
627          @type  hWndFrom: int or L{HWND} or L{system.Window} 
628          @param hWndFrom: Window handle to translate from. 
629              Use C{HWND_DESKTOP} for screen coordinates. 
630   
631          @type  hWndTo: int or L{HWND} or L{system.Window} 
632          @param hWndTo: Window handle to translate to. 
633              Use C{HWND_DESKTOP} for screen coordinates. 
634   
635          @rtype:  L{Rect} 
636          @return: New object containing the translated coordinates. 
637          """
638          points = [(self.left, self.top), (self.right, self.bottom)]
639          return MapWindowPoints(hWndFrom, hWndTo, points)

642  class WindowPlacement(object):
643      """ 
644      Python wrapper over the L{WINDOWPLACEMENT} class. 
645      """
646  
647      def __init__(self, wp=None):
648          """ 
649          @type  wp: L{WindowPlacement} or L{WINDOWPLACEMENT} 
650          @param wp: Another window placement object. 
651          """
652  
653          # Initialize all properties with empty values.
654          self.flags = 0
655          self.showCmd = 0
656          self.ptMinPosition = Point()
657          self.ptMaxPosition = Point()
658          self.rcNormalPosition = Rect()
659  
660          # If a window placement was given copy it's properties.
661          if wp:
662              self.flags = wp.flags
663              self.showCmd = wp.showCmd
664              self.ptMinPosition = Point(wp.ptMinPosition.x, wp.ptMinPosition.y)
665              self.ptMaxPosition = Point(wp.ptMaxPosition.x, wp.ptMaxPosition.y)
666              self.rcNormalPosition = Rect(
667                  wp.rcNormalPosition.left,
668                  wp.rcNormalPosition.top,
669                  wp.rcNormalPosition.right,
670                  wp.rcNormalPosition.bottom,
671              )
672  
673      @property
674      def _as_parameter_(self):
675          """ 
676          Compatibility with ctypes. 
677          Allows passing transparently a Point object to an API call. 
678          """
