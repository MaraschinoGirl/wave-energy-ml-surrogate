9    class Control(window.Wnd):
10       """An ActiveX control base class.  A new class must be derived from both 
11       this class and the Events class.  See the demos for more details. 
12       """
13   
14       def __init__(self):
15           self.__dict__["_dispobj_"] = None
16           window.Wnd.__init__(self)
17   
18       def _GetControlCLSID(self):
19           return self.CLSID
20   
21       def _GetDispatchClass(self):
22           return self.default_interface
23   
24       def _GetEventMap(self):
25           return self.default_source._dispid_to_func_
26   
27       def CreateControl(self, windowTitle, style, rect, parent, id, lic_string=None):
28           clsid = str(self._GetControlCLSID())
29           self.__dict__["_obj_"] = win32ui.CreateControl(
30               clsid, windowTitle, style, rect, parent, id, None, False, lic_string
31           )
32           klass = self._GetDispatchClass()
33           dispobj = klass(win32uiole.GetIDispatchForWindow(self._obj_))
34           self.HookOleEvents()
35           self.__dict__["_dispobj_"] = dispobj
36   
37       def HookOleEvents(self):
38           dict = self._GetEventMap()
39           for dispid, methodName in dict.items():
40               if hasattr(self, methodName):
41                   self._obj_.HookOleEvent(getattr(self, methodName), dispid)
42   
43       def __getattr__(self, attr):
44           # Delegate attributes to the windows and the Dispatch object for this class
45           try:
46               return window.Wnd.__getattr__(self, attr)
47           except AttributeError:
48               pass
49           return getattr(self._dispobj_, attr)
50   
51       def __setattr__(self, attr, value):
52           if hasattr(self.__dict__, attr):
53               self.__dict__[attr] = value
54               return
55           try:
56               if self._dispobj_:
57                   self._dispobj_.__setattr__(attr, value)
58                   return
59           except AttributeError:
60               pass
61           self.__dict__[attr] = value
62   
63   
64   def MakeControlClass(controlClass, name=None):
65       """Given a CoClass in a generated .py file, this function will return a Class 
66       object which can be used as an OCX control. 
67    
68       This function is used when you do not want to handle any events from the OCX 
69       control.  If you need events, then you should derive a class from both the 
70       activex.Control class and the CoClass 
71       """
72       if name is None:
73           name = controlClass.__name__
74       return type("OCX" + name, (Control, controlClass), {})
75   
76   
77   def MakeControlInstance(controlClass, name=None):
78       """As for MakeControlClass(), but returns an instance of the class."""
79       return MakeControlClass(controlClass, name)()
