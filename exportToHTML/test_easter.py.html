5    from __future__ import annotations

7    from datetime import datetime

9    import pytest

11   from pandas.tests.tseries.offsets.common import assert_offset_equal

13   from pandas.tseries.offsets import Easter
14   
15   
16   class TestEaster:
17       @pytest.mark.parametrize(
18           "offset,date,expected",
19           [
20               (Easter(), datetime(2010, 1, 1), datetime(2010, 4, 4)),
21               (Easter(), datetime(2010, 4, 5), datetime(2011, 4, 24)),
22               (Easter(2), datetime(2010, 1, 1), datetime(2011, 4, 24)),
23               (Easter(), datetime(2010, 4, 4), datetime(2011, 4, 24)),
24               (Easter(2), datetime(2010, 4, 4), datetime(2012, 4, 8)),
25               (-Easter(), datetime(2011, 1, 1), datetime(2010, 4, 4)),
26               (-Easter(), datetime(2010, 4, 5), datetime(2010, 4, 4)),
27               (-Easter(2), datetime(2011, 1, 1), datetime(2009, 4, 12)),
28               (-Easter(), datetime(2010, 4, 4), datetime(2009, 4, 12)),
29               (-Easter(2), datetime(2010, 4, 4), datetime(2008, 3, 23)),
30           ],
31       )
32       def test_offset(self, offset, date, expected):
33           assert_offset_equal(offset, date, expected)
