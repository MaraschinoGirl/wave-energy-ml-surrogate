5    /*                                                     polevl.c 
6     *                                                     p1evl.c 
7     * 
8     *     Evaluate polynomial 
9     * 
10    * 
11    * 
12    * SYNOPSIS: 
13    * 
14    * int N; 
15    * double x, y, coef[N+1], polevl[]; 
16    * 
17    * y = polevl( x, coef, N ); 
18    * 
19    * 
20    * 
21    * DESCRIPTION: 
22    * 
23    * Evaluates polynomial of degree N: 
24    * 
25    *                     2          N 
26    * y  =  C  + C x + C x  +...+ C x 
27    *        0    1     2          N 
28    * 
29    * Coefficients are stored in reverse order: 
30    * 
31    * coef[0] = C  , ..., coef[N] = C  . 
32    *            N                   0 
33    * 
34    * The function p1evl() assumes that c_N = 1.0 so that coefficent 
35    * is omitted from the array.  Its calling arguments are 
36    * otherwise the same as polevl(). 
37    * 
38    * 
39    * SPEED: 
40    * 
41    * In the interest of speed, there are no checks for out 
42    * of bounds arithmetic.  This routine is used by most of 
43    * the functions in the library.  Depending on available 
44    * equipment features, the user may wish to rewrite the 
45    * program in microcode or assembly language. 
46    * 
47    */

70       XSF_HOST_DEVICE inline double polevl(double x, const double coef[], int N) {
71           double ans;
72           int i;
73           const double *p;
74   
75           p = coef;
76           ans = *p++;
77           i = N;
78   
79           do {
80               ans = ans * x + *p++;
81           } while (--i);
82   
83           return (ans);
84       }

86       /*                                                     p1evl() */
87       /*                                          N 
88        * Evaluate polynomial when coefficient of x  is 1.0. 
89        * That is, C_{N} is assumed to be 1, and that coefficient 
90        * is not included in the input array coef. 
91        * coef must have length N and contain the polynomial coefficients 
92        * stored as 
93        *     coef[0] = C_{N-1} 
94        *     coef[1] = C_{N-2} 
95        *          ... 
96        *     coef[N-2] = C_1 
97        *     coef[N-1] = C_0 
98        * Otherwise same as polevl. 
99        */
100  
101      XSF_HOST_DEVICE inline double p1evl(double x, const double coef[], int N) {
102          double ans;
103          const double *p;
104          int i;
105  
106          p = coef;
107          ans = x + *p++;
108          i = N - 1;
109  
110          do
111              ans = ans * x + *p++;
112          while (--i);
113  
114          return (ans);
115      }

119      /* The function ratevl is only used once in cephes/lanczos.h. */
120      XSF_HOST_DEVICE inline double ratevl(double x, const double num[], int M, const double denom[], int N) {
121          int i, dir;
122          double y, num_ans, denom_ans;
123          double absx = std::abs(x);
124          const double *p;
125  
126          if (absx > 1) {
127              /* Evaluate as a polynomial in 1/x. */
128              dir = -1;
129              p = num + M;
130              y = 1 / x;
131          } else {
132              dir = 1;
133              p = num;
134              y = x;
135          }
136  
137          /* Evaluate the numerator */
138          num_ans = *p;
139          p += dir;
140          for (i = 1; i <= M; i++) {
141              num_ans = num_ans * y + *p;
142              p += dir;
143          }
144  
145          /* Evaluate the denominator */
146          if (absx > 1) {
147              p = denom + N;
148          } else {
149              p = denom;
150          }
151  
152          denom_ans = *p;
153          p += dir;
154          for (i = 1; i <= N; i++) {
155              denom_ans = denom_ans * y + *p;
156              p += dir;
157          }
158  
159          if (absx > 1) {
160              i = M - N;
161              return std::pow(x, i) * num_ans / denom_ans;
162          } else {
163              return num_ans / denom_ans;
164          }
165      }
