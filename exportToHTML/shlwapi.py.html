86   # BOOL IsOS(
87   #     DWORD dwOS
88   # );
89   def IsOS(dwOS):
90       try:
91           _IsOS = windll.shlwapi.IsOS
92           _IsOS.argtypes = [DWORD]
93           _IsOS.restype = bool
94       except AttributeError:
95           # According to MSDN, on Windows versions prior to Vista
96           # this function is exported only by ordinal number 437.
97           # http://msdn.microsoft.com/en-us/library/bb773795%28VS.85%29.aspx
98           _GetProcAddress = windll.kernel32.GetProcAddress
99           _GetProcAddress.argtypes = [HINSTANCE, DWORD]
100          _GetProcAddress.restype = LPVOID
101          _IsOS = windll.kernel32.GetProcAddress(windll.shlwapi._handle, 437)
102          _IsOS = WINFUNCTYPE(bool, DWORD)(_IsOS)
103      return _IsOS(dwOS)

263  # BOOL PathRelativePathTo(
264  #   _Out_  LPTSTR pszPath,
265  #   _In_   LPCTSTR pszFrom,
266  #   _In_   DWORD dwAttrFrom,
267  #   _In_   LPCTSTR pszTo,
268  #   _In_   DWORD dwAttrTo
269  # );
270  def PathRelativePathToA(pszFrom=None, dwAttrFrom=FILE_ATTRIBUTE_DIRECTORY, pszTo=None, dwAttrTo=FILE_ATTRIBUTE_DIRECTORY):
271      _PathRelativePathToA = windll.shlwapi.PathRelativePathToA
272      _PathRelativePathToA.argtypes = [LPSTR, LPSTR, DWORD, LPSTR, DWORD]
273      _PathRelativePathToA.restype = bool
274      _PathRelativePathToA.errcheck = RaiseIfZero
275  
276      # Make the paths absolute or the function fails.
277      if pszFrom:
278          pszFrom = GetFullPathNameA(pszFrom)[0]
279      else:
280          pszFrom = GetCurrentDirectoryA()
281      if pszTo:
282          pszTo = GetFullPathNameA(pszTo)[0]
283      else:
284          pszTo = GetCurrentDirectoryA()
285  
286      # Argh, this function doesn't receive an output buffer size!
287      # We'll try to guess the maximum possible buffer size.
288      dwPath = max((len(pszFrom) + len(pszTo)) * 2 + 1, MAX_PATH + 1)
289      pszPath = ctypes.create_string_buffer("", dwPath)
290  
291      # Also, it doesn't set the last error value.
292      # Whoever coded it must have been drunk or tripping on acid. Or both.
293      # The only failure conditions I've seen were invalid paths, paths not
294      # on the same drive, or the path is not absolute.
295      SetLastError(ERROR_INVALID_PARAMETER)
296  
297      _PathRelativePathToA(pszPath, pszFrom, dwAttrFrom, pszTo, dwAttrTo)
298      return pszPath.value
299  
300

412  # BOOL PathFindOnPath(
413  #     LPTSTR pszFile,
414  #     LPCTSTR *ppszOtherDirs
415  # );
416  def PathFindOnPathA(pszFile, ppszOtherDirs=None):
417      _PathFindOnPathA = windll.shlwapi.PathFindOnPathA
418      _PathFindOnPathA.argtypes = [LPSTR, LPSTR]
419      _PathFindOnPathA.restype = bool
420  
421      pszFile = ctypes.create_string_buffer(pszFile, MAX_PATH)
422      if not ppszOtherDirs:
423          ppszOtherDirs = None
424      else:
425          szArray = ""
426          for pszOtherDirs in ppszOtherDirs:
427              if pszOtherDirs:
428                  szArray = "%s%s\0" % (szArray, pszOtherDirs)
429          szArray = szArray + "\0"
430          pszOtherDirs = ctypes.create_string_buffer(szArray)
431          ppszOtherDirs = ctypes.pointer(pszOtherDirs)
432      if _PathFindOnPathA(pszFile, ppszOtherDirs):
433          return pszFile.value
434      return None
435
