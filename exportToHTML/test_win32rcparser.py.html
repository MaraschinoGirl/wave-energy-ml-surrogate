9    class TestParser(unittest.TestCase):
10       def setUp(self):
11           rc_file = os.path.join(os.path.dirname(__file__), "win32rcparser", "test.rc")
12           self.resources = win32rcparser.Parse(rc_file)
13   
14       def testStrings(self):
15           for sid, expected in (
16               ("IDS_TEST_STRING4", "Test 'single quoted' string"),
17               ("IDS_TEST_STRING1", 'Test "quoted" string'),
18               ("IDS_TEST_STRING3", 'String with single " quote'),
19               ("IDS_TEST_STRING2", "Test string"),
20           ):
21               got = self.resources.stringTable[sid].value
22               self.assertEqual(got, expected)
23   
24       def testStandardIds(self):
25           for idc in "IDOK IDCANCEL".split():
26               correct = getattr(win32con, idc)
27               self.assertEqual(self.resources.names[correct], idc)
28               self.assertEqual(self.resources.ids[idc], correct)
29   
30       def testTabStop(self):
31           d = self.resources.dialogs["IDD_TEST_DIALOG2"]
32           tabstop_names = ["IDC_EDIT1", "IDOK"]  # should have WS_TABSTOP
33           tabstop_ids = [self.resources.ids[name] for name in tabstop_names]
34           notabstop_names = ["IDC_EDIT2"]  # should have WS_TABSTOP
35           notabstop_ids = [self.resources.ids[name] for name in notabstop_names]
36           num_ok = 0
37           for cdef in d[1:]:  # skip dlgdef
38               # print(cdef)
39               cid = cdef[2]
40               style = cdef[-2]
41               styleex = cdef[-1]
42               if cid in tabstop_ids:
43                   self.assertEqual(style & win32con.WS_TABSTOP, win32con.WS_TABSTOP)
44                   num_ok += 1
45               elif cid in notabstop_ids:
46                   self.assertEqual(style & win32con.WS_TABSTOP, 0)
47                   num_ok += 1
48           self.assertEqual(num_ok, len(tabstop_ids) + len(notabstop_ids))
