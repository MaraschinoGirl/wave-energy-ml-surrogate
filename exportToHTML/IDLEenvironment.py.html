24   def GetIDLEModule(module):
25       try:
26           # First get it from Pythonwin it is exists.
27           modname = "pywin.idle." + module
28           __import__(modname)
29       except ImportError as details:
30           msg = (
31               f"The IDLE extension '{module}' can not be located.\r\n\r\n"
32               "Please correct the installation and restart the"
33               f" application.\r\n\r\n{details}"
34           )
35           win32ui.MessageBox(msg)
36           return None
37       mod = sys.modules[modname]
38       mod.TclError = TextError  # A hack that can go soon!
39       return mod

42   # A class that is injected into the IDLE auto-indent extension.
43   # It allows for decent performance when opening a new file,
44   # as auto-indent uses the tokenizer module to determine indents.
45   # The default AutoIndent readline method works OK, but it goes through
46   # this layer of Tk index indirection for every single line.  For large files
47   # without indents (and even small files with indents :-) it was pretty slow!
48   def fast_readline(self):
49       if self.finished:
50           val = ""
51       else:
52           if "_scint_lines" not in self.__dict__:
53               # XXX - note - assumes this is only called once the file is loaded!
54               self._scint_lines = self.text.edit.GetTextRange().split("\n")
55           sl = self._scint_lines
56           i = self.i = self.i + 1
57           if i >= len(sl):
58               val = ""
59           else:
60               val = sl[i] + "\n"
61       return val.encode(default_scintilla_encoding)

72   class IDLEEditorWindow:
73       def __init__(self, edit):
74           self.edit = edit
75           self.text = TkText(edit)
76           self.extensions = {}
77           self.extension_menus = {}
78   
79       def close(self):
80           self.edit = self.text = None
81           self.extension_menus = None
82           try:
83               for ext in self.extensions.values():
84                   closer = getattr(ext, "close", None)
85                   if closer is not None:
86                       closer()
87           finally:
88               self.extensions = {}
89   
90       def IDLEExtension(self, extension):
91           ext = self.extensions.get(extension)
92           if ext is not None:
93               return ext
94           mod = GetIDLEModule(extension)
95           if mod is None:
96               return None
97           klass = getattr(mod, extension)
98           ext = self.extensions[extension] = klass(self)
99           # Find and bind all the events defined in the extension.
100          events = [item for item in dir(klass) if item[-6:] == "_event"]
101          for event in events:
102              name = "<<{}>>".format(event[:-6].replace("_", "-"))
103              self.edit.bindings.bind(name, getattr(ext, event))
104          return ext

106      def GetMenuItems(self, menu_name):
107          # Get all menu items for the menu name (eg, "edit")
108          bindings = self.edit.bindings
109          ret = []
110          for ext in self.extensions.values():
111              menudefs = getattr(ext, "menudefs", [])
112              for name, items in menudefs:
113                  if name == menu_name:
114                      for text, event in [item for item in items if item is not None]:
115                          text = text.replace("&", "&&")
116                          text = text.replace("_", "&")
117                          ret.append((text, event))
118          return ret

123      def askinteger(
124          self, caption, prompt, parent=None, initialvalue=0, minvalue=None, maxvalue=None
125      ):
126          while 1:
127              rc = GetSimpleInput(prompt, str(initialvalue), caption)
128              if rc is None:
129                  return 0  # Correct "cancel" semantics?
130              err = None
131              try:
132                  rc = int(rc)
133              except ValueError:
134                  err = "Please enter an integer"
135              if not err and minvalue is not None and rc < minvalue:
136                  err = f"Please enter an integer greater then or equal to {minvalue}"
137              if not err and maxvalue is not None and rc > maxvalue:
138                  err = f"Please enter an integer less then or equal to {maxvalue}"
139              if err:
140                  win32ui.MessageBox(err, caption, win32con.MB_OK)
141                  continue
142              return rc

196  def TkOffsetToIndex(offset, edit):
197      lineoff = 0
198      # May be 1 > actual end if we pretended there was a trailing '\n'
199      offset = min(offset, edit.GetTextLength())
200      line = edit.LineFromChar(offset)
201      lineIndex = edit.LineIndex(line)
202      return "%d.%d" % (line + 1, offset - lineIndex)

205  def _NextTok(str, pos):
206      # Returns (token, endPos)
207      end = len(str)
208      if pos >= end:
209          return None, 0
210      while pos < end and str[pos] in string.whitespace:
211          pos += 1
212      # Special case for +-
213      if str[pos] in "+-":
214          return str[pos], pos + 1
215      # Digits also a special case.
216      endPos = pos
217      while endPos < end and str[endPos] in string.digits + ".":
218          endPos += 1
219      if pos != endPos:
220          return str[pos:endPos], endPos
221      endPos = pos
222      while endPos < end and str[endPos] not in string.whitespace + string.digits + "+-":
223          endPos += 1
224      if pos != endPos:
225          return str[pos:endPos], endPos
226      return None, 0

229  def TkIndexToOffset(bm, edit, marks):
230      base, nextTokPos = _NextTok(bm, 0)
231      if base is None:
232          raise ValueError("Empty bookmark ID!")
233      if base.find(".") > 0:
234          try:
235              line, col = base.split(".", 2)
236              if col == "first" or col == "last":
237                  # Tag name
238                  if line != "sel":
239                      raise ValueError("Tags aren't here!")
240                  sel = edit.GetSel()
241                  if sel[0] == sel[1]:
242                      raise EmptyRange
243                  if col == "first":
244                      pos = sel[0]
245                  else:
246                      pos = sel[1]
247              else:
248                  # Lines are 1 based for tkinter
249                  line = int(line) - 1
250                  if line > edit.GetLineCount():
251                      pos = edit.GetTextLength() + 1
252                  else:
253                      pos = edit.LineIndex(line)
254                      if pos == -1:
255                          pos = edit.GetTextLength()
256                      pos += int(col)
257          except (ValueError, IndexError):
258              raise ValueError("Unexpected literal in '%s'" % base)
259      elif base == "insert":
260          pos = edit.GetSel()[0]
261      elif base == "end":
262          pos = edit.GetTextLength()
263          # Pretend there is a trailing '\n' if necessary
264          if pos and edit.SCIGetCharAt(pos - 1) != "\n":
265              pos += 1
266      else:
267          try:
268              pos = marks[base]
269          except KeyError:
270              raise ValueError("Unsupported base offset or undefined mark '%s'" % base)
271  
272      while 1:
273          word, nextTokPos = _NextTok(bm, nextTokPos)
274          if word is None:
275              break
276          if word in ("+", "-"):
277              num, nextTokPos = _NextTok(bm, nextTokPos)
278              if num is None:
279                  raise ValueError("+/- operator needs 2 args")
280              what, nextTokPos = _NextTok(bm, nextTokPos)
281              if what is None:
282                  raise ValueError("+/- operator needs 2 args")
283              if what[0] != "c":
284                  raise ValueError("+/- only supports chars")
285              if word == "+":
286                  pos += int(num)
287              else:
288                  pos -= int(num)
289          elif word == "wordstart":
290              while pos > 0 and edit.SCIGetCharAt(pos - 1) in wordchars:
291                  pos -= 1
292          elif word == "wordend":
293              end = edit.GetTextLength()
294              while pos < end and edit.SCIGetCharAt(pos) in wordchars:
295                  pos += 1
296          elif word == "linestart":
297              while pos > 0 and edit.SCIGetCharAt(pos - 1) not in "\n\r":
298                  pos -= 1
299          elif word == "lineend":
300              end = edit.GetTextLength()
301              while pos < end and edit.SCIGetCharAt(pos) not in "\n\r":
302                  pos += 1
303          else:
304              raise ValueError("Unsupported relative offset '%s'" % word)
305      return max(pos, 0)  # Tkinter is tollerant of -ve indexes - we aren't

339      def _fix_indexes(self, start, end):
340          # first some magic to handle skipping over utf8 extended chars.
341          while start > 0 and ord(self.edit.SCIGetCharAt(start)) & 0xC0 == 0x80:
342              start -= 1
343          while (
344              end < self.edit.GetTextLength()
345              and ord(self.edit.SCIGetCharAt(end)) & 0xC0 == 0x80
346          ):
347              end += 1
348          # now handling fixing \r\n->\n disparities...
349          if (
350              start > 0
351              and self.edit.SCIGetCharAt(start) == "\n"
352              and self.edit.SCIGetCharAt(start - 1) == "\r"
353          ):
354              start -= 1
355          if (
356              end < self.edit.GetTextLength()
357              and self.edit.SCIGetCharAt(end - 1) == "\r"
358              and self.edit.SCIGetCharAt(end) == "\n"
359          ):
360              end += 1
361          return start, end

379      def get(self, start, end=None):
380          try:
381              start = self._getoffset(start)
382              if end is None:
383                  end = start + 1
384              else:
385                  end = self._getoffset(end)
386          except EmptyRange:
387              return ""
388          # Simple semantic checks to conform to the Tk text interface
389          if end <= start:
390              return ""
391          max = self.edit.GetTextLength()
392          checkEnd = 0
393          if end > max:
394              end = max
395              checkEnd = 1
396          start, end = self._fix_indexes(start, end)
397          ret = self.edit.GetTextRange(start, end)
398          # pretend a trailing '\n' exists if necessary.
399          if checkEnd and (not ret or ret[-1] != "\n"):
400              ret += "\n"
401          return ret.replace("\r", "")

403      def index(self, spec):
404          try:
405              return self._getindex(self._getoffset(spec))
406          except EmptyRange:
407              return ""

409      def insert(self, pos, text):
410          try:
411              pos = self._getoffset(pos)
412          except EmptyRange:
413              raise TextError("Empty range")
414          self.edit.SetSel((pos, pos))
415          # IDLE only deals with "\n" - we will be nicer
416  
417          bits = text.split("\n")
418          self.edit.SCIAddText(bits[0])
419          for bit in bits[1:]:
420              self.edit.SCINewline()
421              self.edit.SCIAddText(bit)
422  
423      def delete(self, start, end=None):
424          try:
425              start = self._getoffset(start)
426              if end is not None:
427                  end = self._getoffset(end)
428          except EmptyRange:
429              raise TextError("Empty range")
430          # If end is specified and == start, then we must delete nothing.
431          if start == end:
432              return
433          # If end is not specified, delete one char
434          if end is None:
435              end = start + 1
436          else:
437              # Tk says not to delete in this case, but our control would.
438              if end < start:
439                  return
440          if start == self.edit.GetTextLength():
441              return  # Nothing to delete.
442          old = self.edit.GetSel()[0]  # Lose a selection
443          # Hack for partial '\r\n' and UTF-8 char removal
444          start, end = self._fix_indexes(start, end)
445          self.edit.SetSel((start, end))
446          self.edit.Clear()
447          if old >= start and old < end:
448              old = start
449          elif old >= end:
450              old -= end - start
451          self.edit.SetSel(old)
