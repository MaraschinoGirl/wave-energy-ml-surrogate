8    class TypelibSpec:
9        def __init__(self, clsid, lcid, major, minor, flags=0):
10           self.clsid = str(clsid)
11           self.lcid = int(lcid)
12           # We avoid assuming 'major' or 'minor' are integers - when
13           # read from the registry there is some confusion about if
14           # they are base 10 or base 16 (they *should* be base 16, but
15           # how they are written is beyond our control.)
16           self.major = major
17           self.minor = minor
18           self.dll = None
19           self.desc = None
20           self.ver_desc = None
21           self.flags = flags
22   
23       # For the SelectList
24       def __getitem__(self, item):
25           if item == 0:
26               return self.ver_desc
27           raise IndexError("Can't index me!")
28   
29       def __lt__(self, other):
30           me = (
31               (self.ver_desc or "").lower(),
32               (self.desc or "").lower(),
33               self.major,
34               self.minor,
35           )
36           them = (
37               (other.ver_desc or "").lower(),
38               (other.desc or "").lower(),
39               other.major,
40               other.minor,
41           )
42           return me < them
43   
44       def __eq__(self, other):
45           return (
46               (self.ver_desc or "").lower() == (other.ver_desc or "").lower()
47               and (self.desc or "").lower() == (other.desc or "").lower()
48               and self.major == other.major
49               and self.minor == other.minor
50           )
51   
52       def Resolve(self):
53           if self.dll is None:
54               return 0
55           tlb = pythoncom.LoadTypeLib(self.dll)
56           self.FromTypelib(tlb, None)
57           return 1
58   
59       def FromTypelib(self, typelib, dllName=None):
60           la = typelib.GetLibAttr()
61           self.clsid = str(la[0])
62           self.lcid = la[1]
63           self.major = la[3]
64           self.minor = la[4]
65           if dllName:
66               self.dll = dllName

69   def EnumKeys(root):
70       index = 0
71       ret = []
72       while 1:
73           try:
74               item = win32api.RegEnumKey(root, index)
75           except win32api.error:
76               break
77           try:
78               # Note this doesn't handle REG_EXPAND_SZ, but the implementation
79               # here doesn't need to - that is handled as the data is read.
80               val = win32api.RegQueryValue(root, item)
81           except win32api.error:
82               val = ""  # code using this assumes a string.
83   
84           ret.append((item, val))
85           index += 1
86       return ret

94   def EnumTlbs(excludeFlags=0):
95       """Return a list of TypelibSpec objects, one for each registered library."""
96       key = win32api.RegOpenKey(win32con.HKEY_CLASSES_ROOT, "Typelib")
97       iids = EnumKeys(key)
98       results = []
99       for iid, crap in iids:
100          try:
101              key2 = win32api.RegOpenKey(key, str(iid))
102          except win32api.error:
103              # A few good reasons for this, including "access denied".
104              continue
105          for version, tlbdesc in EnumKeys(key2):
106              major_minor = version.split(".", 1)
107              if len(major_minor) < 2:
108                  major_minor.append("0")
109              # For some reason, this code used to assume the values were hex.
110              # This seems to not be true - particularly for CDO 1.21
111              # *sigh* - it appears there are no rules here at all, so when we need
112              # to know the info, we must load the tlb by filename and request it.
113              # The Resolve() method on the TypelibSpec does this.
114              # For this reason, keep the version numbers as strings - that
115              # way we can't be wrong!  Let code that really needs an int to work
116              # out what to do.  FWIW, https://www.betaarchive.com/wiki/index.php?title=Microsoft_KB_Archive/816970
117              # is pretty clear that they *should* be hex.
118              major = major_minor[0]
119              minor = major_minor[1]
120              key3 = win32api.RegOpenKey(key2, str(version))
121              try:
122                  # The "FLAGS" are at this point
123                  flags = int(win32api.RegQueryValue(key3, "FLAGS"))
124              except (win32api.error, ValueError):
125                  flags = 0
126              if flags & excludeFlags == 0:
127                  for lcid, crap in EnumKeys(key3):
128                      try:
129                          lcid = int(lcid)
130                      except ValueError:  # not an LCID entry
131                          continue
132                      # Check for both "{lcid}\win32" and "{lcid}\win64" keys.
133                      try:
134                          key4 = win32api.RegOpenKey(key3, f"{lcid}\\win32")
135                      except win32api.error:
136                          try:
137                              key4 = win32api.RegOpenKey(key3, f"{lcid}\\win64")
138                          except win32api.error:
139                              continue
140                      try:
141                          dll, typ = win32api.RegQueryValueEx(key4, None)
142                          if typ == win32con.REG_EXPAND_SZ:
143                              dll = win32api.ExpandEnvironmentStrings(dll)
144                      except win32api.error:
145                          dll = None
146                      spec = TypelibSpec(iid, lcid, major, minor, flags)
147                      spec.dll = dll
148                      spec.desc = tlbdesc
149                      spec.ver_desc = tlbdesc + " (" + version + ")"
150                      results.append(spec)
151      return results

154  def FindTlbsWithDescription(desc):
155      """Find all installed type libraries with the specified description"""
156      ret = []
157      items = EnumTlbs()
158      for item in items:
159          if item.desc == desc:
160              ret.append(item)
161      return ret

164  def SelectTlb(title="Select Library", excludeFlags=0):
165      """Display a list of all the type libraries, and select one.   Returns None if cancelled"""
166      import pywin.dialogs.list
167  
168      items = EnumTlbs(excludeFlags)
169      # fixup versions - we assume hex (see __init__ above)
170      for i in items:
171          i.major = int(i.major, 16)
172          i.minor = int(i.minor, 16)
173      items.sort()
174      rc = pywin.dialogs.list.SelectFromLists(title, items, ["Type Library"])
175      if rc is None:
176          return None
177      return items[rc]
