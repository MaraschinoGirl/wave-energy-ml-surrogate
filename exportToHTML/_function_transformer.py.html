30   def _identity(X):
31       """The identity function."""
32       return X

35   class FunctionTransformer(TransformerMixin, BaseEstimator):
36       """Constructs a transformer from an arbitrary callable. 
37    
38       A FunctionTransformer forwards its X (and optionally y) arguments to a 
39       user-defined function or function object and returns the result of this 
40       function. This is useful for stateless transformations such as taking the 
41       log of frequencies, doing custom scaling, etc. 
42    
43       Note: If a lambda is used as the function, then the resulting 
44       transformer will not be pickleable. 
45    
46       .. versionadded:: 0.17 
47    
48       Read more in the :ref:`User Guide <function_transformer>`. 
49    
50       Parameters 
51       ---------- 
52       func : callable, default=None 
53           The callable to use for the transformation. This will be passed 
54           the same arguments as transform, with args and kwargs forwarded. 
55           If func is None, then func will be the identity function. 
56    
57       inverse_func : callable, default=None 
58           The callable to use for the inverse transformation. This will be 
59           passed the same arguments as inverse transform, with args and 
60           kwargs forwarded. If inverse_func is None, then inverse_func 
61           will be the identity function. 
62    
63       validate : bool, default=False 
64           Indicate that the input X array should be checked before calling 
65           ``func``. The possibilities are: 
66    
67           - If False, there is no input validation. 
68           - If True, then X will be converted to a 2-dimensional NumPy array or 
69             sparse matrix. If the conversion is not possible an exception is 
70             raised. 
71    
72           .. versionchanged:: 0.22 
73              The default of ``validate`` changed from True to False. 
74    
75       accept_sparse : bool, default=False 
76           Indicate that func accepts a sparse matrix as input. If validate is 
77           False, this has no effect. Otherwise, if accept_sparse is false, 
78           sparse matrix inputs will cause an exception to be raised. 
79    
80       check_inverse : bool, default=True 
81          Whether to check that or ``func`` followed by ``inverse_func`` leads to 
82          the original inputs. It can be used for a sanity check, raising a 
83          warning when the condition is not fulfilled. 
84    
85          .. versionadded:: 0.20 
86    
87       feature_names_out : callable, 'one-to-one' or None, default=None 
88           Determines the list of feature names that will be returned by the 
89           `get_feature_names_out` method. If it is 'one-to-one', then the output 
90           feature names will be equal to the input feature names. If it is a 
91           callable, then it must take two positional arguments: this 
92           `FunctionTransformer` (`self`) and an array-like of input feature names 
93           (`input_features`). It must return an array-like of output feature 
94           names. The `get_feature_names_out` method is only defined if 
95           `feature_names_out` is not None. 
96    
97           See ``get_feature_names_out`` for more details. 
98    
99           .. versionadded:: 1.1 
100   
101      kw_args : dict, default=None 
102          Dictionary of additional keyword arguments to pass to func. 
103   
104          .. versionadded:: 0.18 
105   
106      inv_kw_args : dict, default=None 
107          Dictionary of additional keyword arguments to pass to inverse_func. 
108   
109          .. versionadded:: 0.18 
110   
111      Attributes 
112      ---------- 
113      n_features_in_ : int 
114          Number of features seen during :term:`fit`. 
115   
116          .. versionadded:: 0.24 
117   
118      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
119          Names of features seen during :term:`fit`. Defined only when `X` has feature 
120          names that are all strings. 
121   
122          .. versionadded:: 1.0 
123   
124      See Also 
125      -------- 
126      MaxAbsScaler : Scale each feature by its maximum absolute value. 
127      StandardScaler : Standardize features by removing the mean and 
128          scaling to unit variance. 
129      LabelBinarizer : Binarize labels in a one-vs-all fashion. 
130      MultiLabelBinarizer : Transform between iterable of iterables 
131          and a multilabel format. 
132   
133      Notes 
134      ----- 
135      If `func` returns an output with a `columns` attribute, then the columns is enforced 
136      to be consistent with the output of `get_feature_names_out`. 
137   
138      Examples 
139      -------- 
140      >>> import numpy as np 
141      >>> from sklearn.preprocessing import FunctionTransformer 
142      >>> transformer = FunctionTransformer(np.log1p) 
143      >>> X = np.array([[0, 1], [2, 3]]) 
144      >>> transformer.transform(X) 
145      array([[0.       , 0.6931...], 
146             [1.0986..., 1.3862...]]) 
147      """
148  
149      _parameter_constraints: dict = {
150          "func": [callable, None],
151          "inverse_func": [callable, None],
152          "validate": ["boolean"],
153          "accept_sparse": ["boolean"],
154          "check_inverse": ["boolean"],
155          "feature_names_out": [callable, StrOptions({"one-to-one"}), None],
156          "kw_args": [dict, None],
157          "inv_kw_args": [dict, None],
158      }

160      def __init__(
161          self,
162          func=None,
163          inverse_func=None,
164          *,
165          validate=False,
166          accept_sparse=False,
167          check_inverse=True,
168          feature_names_out=None,
169          kw_args=None,
170          inv_kw_args=None,
171      ):
172          self.func = func
173          self.inverse_func = inverse_func
174          self.validate = validate
175          self.accept_sparse = accept_sparse
176          self.check_inverse = check_inverse
177          self.feature_names_out = feature_names_out
178          self.kw_args = kw_args
179          self.inv_kw_args = inv_kw_args
180  
181      def _check_input(self, X, *, reset):
182          if self.validate:
183              return validate_data(self, X, accept_sparse=self.accept_sparse, reset=reset)
184          elif reset:
185              # Set feature_names_in_ and n_features_in_ even if validate=False
186              # We run this only when reset==True to store the attributes but not
187              # validate them, because validate=False
188              _check_n_features(self, X, reset=reset)
189              _check_feature_names(self, X, reset=reset)
190          return X
191  
192      def _check_inverse_transform(self, X):
193          """Check that func and inverse_func are the inverse."""
194          idx_selected = slice(None, None, max(1, X.shape[0] // 100))
195          X_round_trip = self.inverse_transform(self.transform(X[idx_selected]))
196  
197          if hasattr(X, "dtype"):
198              dtypes = [X.dtype]
199          elif hasattr(X, "dtypes"):
200              # Dataframes can have multiple dtypes
201              dtypes = X.dtypes
202  
203          if not all(np.issubdtype(d, np.number) for d in dtypes):
204              raise ValueError(
205                  "'check_inverse' is only supported when all the elements in `X` is"
206                  " numerical."
207              )
208  
209          if not _allclose_dense_sparse(X[idx_selected], X_round_trip):
210              warnings.warn(
211                  (
212                      "The provided functions are not strictly"
213                      " inverse of each other. If you are sure you"
214                      " want to proceed regardless, set"
215                      " 'check_inverse=False'."
216                  ),
217                  UserWarning,
218              )

220      @_fit_context(prefer_skip_nested_validation=True)
221      def fit(self, X, y=None):
222          """Fit transformer by checking X. 
223   
224          If ``validate`` is ``True``, ``X`` will be checked. 
225   
226          Parameters 
227          ---------- 
228          X : {array-like, sparse-matrix} of shape (n_samples, n_features) \ 
229                  if `validate=True` else any object that `func` can handle 
230              Input array. 
231   
232          y : Ignored 
233              Not used, present here for API consistency by convention. 
234   
235          Returns 
236          ------- 
237          self : object 
238              FunctionTransformer class instance. 
239          """
240          X = self._check_input(X, reset=True)
241          if self.check_inverse and not (self.func is None or self.inverse_func is None):
242              self._check_inverse_transform(X)
243          return self
244  
245      def transform(self, X):
246          """Transform X using the forward function. 
247   
248          Parameters 
249          ---------- 
250          X : {array-like, sparse-matrix} of shape (n_samples, n_features) \ 
251                  if `validate=True` else any object that `func` can handle 
252              Input array. 
253   
254          Returns 
255          ------- 
256          X_out : array-like, shape (n_samples, n_features) 
257              Transformed input. 
258          """
259          X = self._check_input(X, reset=False)
260          out = self._transform(X, func=self.func, kw_args=self.kw_args)
261          output_config = _get_output_config("transform", self)["dense"]
262  
263          if hasattr(out, "columns") and self.feature_names_out is not None:
264              # check the consistency between the column provided by `transform` and
265              # the the column names provided by `get_feature_names_out`.
266              feature_names_out = self.get_feature_names_out()
267              if list(out.columns) != list(feature_names_out):
268                  # we can override the column names of the output if it is inconsistent
269                  # with the column names provided by `get_feature_names_out` in the
270                  # following cases:
271                  # * `func` preserved the column names between the input and the output
272                  # * the input column names are all numbers
273                  # * the output is requested to be a DataFrame (pandas or polars)
274                  feature_names_in = getattr(
275                      X, "feature_names_in_", _get_feature_names(X)
276                  )
277                  same_feature_names_in_out = feature_names_in is not None and list(
278                      feature_names_in
279                  ) == list(out.columns)
280                  not_all_str_columns = not all(
281                      isinstance(col, str) for col in out.columns
282                  )
283                  if same_feature_names_in_out or not_all_str_columns:
284                      adapter = _get_adapter_from_container(out)
285                      out = adapter.create_container(
286                          X_output=out,
287                          X_original=out,
288                          columns=feature_names_out,
289                          inplace=False,
290                      )
291                  else:
292                      raise ValueError(
293                          "The output generated by `func` have different column names "
294                          "than the ones provided by `get_feature_names_out`. "
295                          f"Got output with columns names: {list(out.columns)} and "
296                          "`get_feature_names_out` returned: "
297                          f"{list(self.get_feature_names_out())}. "
298                          "The column names can be overridden by setting "
299                          "`set_output(transform='pandas')` or "
300                          "`set_output(transform='polars')` such that the column names "
301                          "are set to the names provided by `get_feature_names_out`."
302                      )
303  
304          if self.feature_names_out is None:
305              warn_msg = (
306                  "When `set_output` is configured to be '{0}', `func` should return "
307                  "a {0} DataFrame to follow the `set_output` API  or `feature_names_out`"
308                  " should be defined."
309              )
310              if output_config == "pandas" and not _is_pandas_df(out):
311                  warnings.warn(warn_msg.format("pandas"))
312              elif output_config == "polars" and not _is_polars_df(out):
313                  warnings.warn(warn_msg.format("polars"))
314  
315          return out
316  
317      def inverse_transform(self, X):
318          """Transform X using the inverse function. 
319   
320          Parameters 
321          ---------- 
322          X : {array-like, sparse-matrix} of shape (n_samples, n_features) \ 
323                  if `validate=True` else any object that `inverse_func` can handle 
324              Input array. 
325   
326          Returns 
327          ------- 
328          X_out : array-like, shape (n_samples, n_features) 
329              Transformed input. 
330          """
331          if self.validate:
332              X = check_array(X, accept_sparse=self.accept_sparse)
333          return self._transform(X, func=self.inverse_func, kw_args=self.inv_kw_args)
334  
335      @available_if(lambda self: self.feature_names_out is not None)
336      def get_feature_names_out(self, input_features=None):
337          """Get output feature names for transformation. 
338   
339          This method is only defined if `feature_names_out` is not None. 
340   
341          Parameters 
342          ---------- 
343          input_features : array-like of str or None, default=None 
344              Input feature names. 
345   
346              - If `input_features` is None, then `feature_names_in_` is 
347                used as the input feature names. If `feature_names_in_` is not 
348                defined, then names are generated: 
349                `[x0, x1, ..., x(n_features_in_ - 1)]`. 
350              - If `input_features` is array-like, then `input_features` must 
351                match `feature_names_in_` if `feature_names_in_` is defined. 
352   
353          Returns 
354          ------- 
355          feature_names_out : ndarray of str objects 
356              Transformed feature names. 
357   
358              - If `feature_names_out` is 'one-to-one', the input feature names 
359                are returned (see `input_features` above). This requires 
360                `feature_names_in_` and/or `n_features_in_` to be defined, which 
361                is done automatically if `validate=True`. Alternatively, you can 
362                set them in `func`. 
363              - If `feature_names_out` is a callable, then it is called with two 
364                arguments, `self` and `input_features`, and its return value is 
365                returned by this method. 
366          """
367          if hasattr(self, "n_features_in_") or input_features is not None:
368              input_features = _check_feature_names_in(self, input_features)
369          if self.feature_names_out == "one-to-one":
370              names_out = input_features
371          elif callable(self.feature_names_out):
372              names_out = self.feature_names_out(self, input_features)
373          else:
374              raise ValueError(
375                  f"feature_names_out={self.feature_names_out!r} is invalid. "
376                  'It must either be "one-to-one" or a callable with two '
377                  "arguments: the function transformer and an array-like of "
378                  "input feature names. The callable must return an array-like "
379                  "of output feature names."
380              )
381          return np.asarray(names_out, dtype=object)
382  
383      def _transform(self, X, func=None, kw_args=None):
384          if func is None:
385              func = _identity
386  
387          return func(X, **(kw_args if kw_args else {}))

389      def __sklearn_is_fitted__(self):
390          """Return True since FunctionTransfomer is stateless."""
391          return True
392  
393      def __sklearn_tags__(self):
394          tags = super().__sklearn_tags__()
395          tags.no_validation = not self.validate
396          tags.requires_fit = False
397          tags.input_tags.sparse = not self.validate or self.accept_sparse
398          return tags

400      def set_output(self, *, transform=None):
401          """Set output container. 
402   
403          See :ref:`sphx_glr_auto_examples_miscellaneous_plot_set_output.py` 
404          for an example on how to use the API. 
405   
406          Parameters 
407          ---------- 
408          transform : {"default", "pandas", "polars"}, default=None 
409              Configure output of `transform` and `fit_transform`. 
410   
411              - `"default"`: Default output format of a transformer 
412              - `"pandas"`: DataFrame output 
413              - `"polars"`: Polars output 
414              - `None`: Transform configuration is unchanged 
415   
416              .. versionadded:: 1.4 
417                  `"polars"` option was added. 
418   
419          Returns 
420          ------- 
421          self : estimator instance 
422              Estimator instance. 
423          """
424          if not hasattr(self, "_sklearn_output_config"):
425              self._sklearn_output_config = {}
426  
427          self._sklearn_output_config["transform"] = transform
428          return self

430      def _get_function_name(self):
431          """Get the name display of the `func` used in HTML representation."""
432          if hasattr(self.func, "__name__"):
433              return self.func.__name__
434          if isinstance(self.func, partial):
435              return self.func.func.__name__
436          return f"{self.func.__class__.__name__}(...)"
