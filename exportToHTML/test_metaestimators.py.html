8    class AvailableParameterEstimator:
9        """This estimator's `available` parameter toggles the presence of a method"""
10   
11       def __init__(self, available=True, return_value=1):
12           self.available = available
13           self.return_value = return_value
14   
15       @available_if(lambda est: est.available)
16       def available_func(self):
17           """This is a mock available_if function"""
18           return self.return_value

22       assert "This is a mock available_if function" in str(
23           AvailableParameterEstimator.__dict__["available_func"].__doc__
24       )
25       assert "This is a mock available_if function" in str(
26           AvailableParameterEstimator.available_func.__doc__

28       assert "This is a mock available_if function" in str(

33   def test_available_if():

35       assert not hasattr(AvailableParameterEstimator(available=False), "available_func")

38   def test_available_if_unbound_method():
39       # This is a non regression test for:
40       # https://github.com/scikit-learn/scikit-learn/issues/20614
41       # to make sure that decorated functions can be used as an unbound method,
42       # for instance when monkeypatching.
43       est = AvailableParameterEstimator()
44       AvailableParameterEstimator.available_func(est)
45   
46       est = AvailableParameterEstimator(available=False)
47       with pytest.raises(
48           AttributeError,
49           match="This 'AvailableParameterEstimator' has no attribute 'available_func'",
50       ):
51           AvailableParameterEstimator.available_func(est)

54   def test_available_if_methods_can_be_pickled():
55       """Check that available_if methods can be pickled. 
56    
57       Non-regression test for #21344. 
58       """
59       return_value = 10
60       est = AvailableParameterEstimator(available=True, return_value=return_value)
61       pickled_bytes = pickle.dumps(est.available_func)
62       unpickled_func = pickle.loads(pickled_bytes)
63       assert unpickled_func() == return_value
