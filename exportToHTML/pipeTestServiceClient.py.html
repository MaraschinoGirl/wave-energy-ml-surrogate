40   def CallPipe(fn, args):
41       ret = None
42       retryCount = 0
43       while retryCount < 8:  # Keep looping until user cancels.
44           retryCount += 1
45           try:
46               return fn(*args)
47           except win32api.error as exc:
48               if exc.winerror == winerror.ERROR_PIPE_BUSY:
49                   win32api.Sleep(5000)
50                   continue
51               else:
52                   raise
53   
54       raise RuntimeError("Could not make a connection to the server")

101  def stressTestClient(server, numThreads, numMessages):
102      import _thread
103  
104      thread_waits = []
105      for t_num in range(numThreads):
106          # Note I could just wait on thread handles (after calling DuplicateHandle)
107          # See the service itself for an example of waiting for the clients...
108          wait = CreateEvent(None, 0, 0, None)
109          thread_waits.append(wait)
110          _thread.start_new_thread(stressThread, (server, numMessages, wait))
111      # Wait for all threads to finish.
112      WaitForMultipleObjects(thread_waits, 1, INFINITE)
