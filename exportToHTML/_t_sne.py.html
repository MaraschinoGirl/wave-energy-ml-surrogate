39   def _joint_probabilities(distances, desired_perplexity, verbose):
40       """Compute joint probabilities p_ij from distances. 
41    
42       Parameters 
43       ---------- 
44       distances : ndarray of shape (n_samples * (n_samples-1) / 2,) 
45           Distances of samples are stored as condensed matrices, i.e. 
46           we omit the diagonal and duplicate entries and store everything 
47           in a one-dimensional array. 
48    
49       desired_perplexity : float 
50           Desired perplexity of the joint probability distributions. 
51    
52       verbose : int 
53           Verbosity level. 
54    
55       Returns 
56       ------- 
57       P : ndarray of shape (n_samples * (n_samples-1) / 2,) 
58           Condensed joint probability matrix. 
59       """
60       # Compute conditional probabilities such that they approximately match
61       # the desired perplexity
62       distances = distances.astype(np.float32, copy=False)
63       conditional_P = _utils._binary_search_perplexity(
64           distances, desired_perplexity, verbose
65       )
66       P = conditional_P + conditional_P.T
67       sum_P = np.maximum(np.sum(P), MACHINE_EPSILON)
68       P = np.maximum(squareform(P) / sum_P, MACHINE_EPSILON)
69       return P

72   def _joint_probabilities_nn(distances, desired_perplexity, verbose):
73       """Compute joint probabilities p_ij from distances using just nearest 
74       neighbors. 
75    
76       This method is approximately equal to _joint_probabilities. The latter 
77       is O(N), but limiting the joint probability to nearest neighbors improves 
78       this substantially to O(uN). 
79    
80       Parameters 
81       ---------- 
82       distances : sparse matrix of shape (n_samples, n_samples) 
83           Distances of samples to its n_neighbors nearest neighbors. All other 
84           distances are left to zero (and are not materialized in memory). 
85           Matrix should be of CSR format. 
86    
87       desired_perplexity : float 
88           Desired perplexity of the joint probability distributions. 
89    
90       verbose : int 
91           Verbosity level. 
92    
93       Returns 
94       ------- 
95       P : sparse matrix of shape (n_samples, n_samples) 
96           Condensed joint probability matrix with only nearest neighbors. Matrix 
97           will be of CSR format. 
98       """
99       t0 = time()
100      # Compute conditional probabilities such that they approximately match
101      # the desired perplexity
102      distances.sort_indices()
103      n_samples = distances.shape[0]
104      distances_data = distances.data.reshape(n_samples, -1)
105      distances_data = distances_data.astype(np.float32, copy=False)
106      conditional_P = _utils._binary_search_perplexity(
107          distances_data, desired_perplexity, verbose
108      )
109      assert np.all(np.isfinite(conditional_P)), "All probabilities should be finite"
110  
111      # Symmetrize the joint probability distribution using sparse operations
112      P = csr_matrix(
113          (conditional_P.ravel(), distances.indices, distances.indptr),
114          shape=(n_samples, n_samples),
115      )
116      P = P + P.T
117  
118      # Normalize the joint probability distribution
119      sum_P = np.maximum(P.sum(), MACHINE_EPSILON)
120      P /= sum_P
121  
122      assert np.all(np.abs(P.data) <= 1.0)
123      if verbose >= 2:
124          duration = time() - t0
125          print("[t-SNE] Computed conditional probabilities in {:.3f}s".format(duration))
126      return P

129  def _kl_divergence(
130      params,
131      P,
132      degrees_of_freedom,
133      n_samples,
134      n_components,
135      skip_num_points=0,
136      compute_error=True,
137  ):
138      """t-SNE objective function: gradient of the KL divergence 
139      of p_ijs and q_ijs and the absolute error. 
140   
141      Parameters 
142      ---------- 
143      params : ndarray of shape (n_params,) 
144          Unraveled embedding. 
145   
146      P : ndarray of shape (n_samples * (n_samples-1) / 2,) 
147          Condensed joint probability matrix. 
148   
149      degrees_of_freedom : int 
150          Degrees of freedom of the Student's-t distribution. 
151   
152      n_samples : int 
153          Number of samples. 
154   
155      n_components : int 
156          Dimension of the embedded space. 
157   
158      skip_num_points : int, default=0 
159          This does not compute the gradient for points with indices below 
160          `skip_num_points`. This is useful when computing transforms of new 
161          data where you'd like to keep the old data fixed. 
162   
163      compute_error: bool, default=True 
164          If False, the kl_divergence is not computed and returns NaN. 
165   
166      Returns 
167      ------- 
168      kl_divergence : float 
169          Kullback-Leibler divergence of p_ij and q_ij. 
170   
171      grad : ndarray of shape (n_params,) 
172          Unraveled gradient of the Kullback-Leibler divergence with respect to 
173          the embedding. 
174      """
175      X_embedded = params.reshape(n_samples, n_components)
176  
177      # Q is a heavy-tailed distribution: Student's t-distribution
178      dist = pdist(X_embedded, "sqeuclidean")
179      dist /= degrees_of_freedom
180      dist += 1.0
181      dist **= (degrees_of_freedom + 1.0) / -2.0
182      Q = np.maximum(dist / (2.0 * np.sum(dist)), MACHINE_EPSILON)
183  
184      # Optimization trick below: np.dot(x, y) is faster than
185      # np.sum(x * y) because it calls BLAS
186  
187      # Objective: C (Kullback-Leibler divergence of P and Q)
188      if compute_error:
189          kl_divergence = 2.0 * np.dot(P, np.log(np.maximum(P, MACHINE_EPSILON) / Q))
190      else:
191          kl_divergence = np.nan
192  
193      # Gradient: dC/dY
194      # pdist always returns double precision distances. Thus we need to take
195      grad = np.ndarray((n_samples, n_components), dtype=params.dtype)
196      PQd = squareform((P - Q) * dist)
197      for i in range(skip_num_points, n_samples):
198          grad[i] = np.dot(np.ravel(PQd[i], order="K"), X_embedded[i] - X_embedded)
199      grad = grad.ravel()
200      c = 2.0 * (degrees_of_freedom + 1.0) / degrees_of_freedom
201      grad *= c
202  
203      return kl_divergence, grad

206  def _kl_divergence_bh(
207      params,
208      P,
209      degrees_of_freedom,
210      n_samples,
211      n_components,
212      angle=0.5,
213      skip_num_points=0,
214      verbose=False,
215      compute_error=True,
216      num_threads=1,
217  ):
218      """t-SNE objective function: KL divergence of p_ijs and q_ijs. 
219   
220      Uses Barnes-Hut tree methods to calculate the gradient that 
221      runs in O(NlogN) instead of O(N^2). 
222   
223      Parameters 
224      ---------- 
225      params : ndarray of shape (n_params,) 
226          Unraveled embedding. 
227   
228      P : sparse matrix of shape (n_samples, n_sample) 
229          Sparse approximate joint probability matrix, computed only for the 
230          k nearest-neighbors and symmetrized. Matrix should be of CSR format. 
231   
232      degrees_of_freedom : int 
233          Degrees of freedom of the Student's-t distribution. 
234   
235      n_samples : int 
236          Number of samples. 
237   
238      n_components : int 
239          Dimension of the embedded space. 
240   
241      angle : float, default=0.5 
242          This is the trade-off between speed and accuracy for Barnes-Hut T-SNE. 
243          'angle' is the angular size (referred to as theta in [3]) of a distant 
244          node as measured from a point. If this size is below 'angle' then it is 
245          used as a summary node of all points contained within it. 
246          This method is not very sensitive to changes in this parameter 
247          in the range of 0.2 - 0.8. Angle less than 0.2 has quickly increasing 
248          computation time and angle greater 0.8 has quickly increasing error. 
249   
250      skip_num_points : int, default=0 
251          This does not compute the gradient for points with indices below 
252          `skip_num_points`. This is useful when computing transforms of new 
253          data where you'd like to keep the old data fixed. 
254   
255      verbose : int, default=False 
256          Verbosity level. 
257   
258      compute_error: bool, default=True 
259          If False, the kl_divergence is not computed and returns NaN. 
260   
261      num_threads : int, default=1 
262          Number of threads used to compute the gradient. This is set here to 
263          avoid calling _openmp_effective_n_threads for each gradient step. 
264   
265      Returns 
266      ------- 
267      kl_divergence : float 
268          Kullback-Leibler divergence of p_ij and q_ij. 
269   
270      grad : ndarray of shape (n_params,) 
271          Unraveled gradient of the Kullback-Leibler divergence with respect to 
272          the embedding. 
273      """
274      params = params.astype(np.float32, copy=False)
275      X_embedded = params.reshape(n_samples, n_components)
276  
277      val_P = P.data.astype(np.float32, copy=False)
278      neighbors = P.indices.astype(np.int64, copy=False)
279      indptr = P.indptr.astype(np.int64, copy=False)
280  
281      grad = np.zeros(X_embedded.shape, dtype=np.float32)
282      error = _barnes_hut_tsne.gradient(
283          val_P,
284          X_embedded,
285          neighbors,
286          indptr,
287          grad,
288          angle,
289          n_components,
290          verbose,
291          dof=degrees_of_freedom,
292          compute_error=compute_error,
293          num_threads=num_threads,
294      )
295      c = 2.0 * (degrees_of_freedom + 1.0) / degrees_of_freedom
296      grad = grad.ravel()
297      grad *= c
298  
299      return error, grad

302  def _gradient_descent(
303      objective,
304      p0,
305      it,
306      max_iter,
307      n_iter_check=1,
308      n_iter_without_progress=300,
309      momentum=0.8,
310      learning_rate=200.0,
311      min_gain=0.01,
312      min_grad_norm=1e-7,
313      verbose=0,
314      args=None,
315      kwargs=None,
316  ):
317      """Batch gradient descent with momentum and individual gains. 
318   
319      Parameters 
320      ---------- 
321      objective : callable 
322          Should return a tuple of cost and gradient for a given parameter 
323          vector. When expensive to compute, the cost can optionally 
324          be None and can be computed every n_iter_check steps using 
325          the objective_error function. 
326   
327      p0 : array-like of shape (n_params,) 
328          Initial parameter vector. 
329   
330      it : int 
331          Current number of iterations (this function will be called more than 
332          once during the optimization). 
333   
334      max_iter : int 
335          Maximum number of gradient descent iterations. 
336   
337      n_iter_check : int, default=1 
338          Number of iterations before evaluating the global error. If the error 
339          is sufficiently low, we abort the optimization. 
340   
341      n_iter_without_progress : int, default=300 
342          Maximum number of iterations without progress before we abort the 
343          optimization. 
344   
345      momentum : float within (0.0, 1.0), default=0.8 
346          The momentum generates a weight for previous gradients that decays 
347          exponentially. 
348   
349      learning_rate : float, default=200.0 
350          The learning rate for t-SNE is usually in the range [10.0, 1000.0]. If 
351          the learning rate is too high, the data may look like a 'ball' with any 
352          point approximately equidistant from its nearest neighbours. If the 
353          learning rate is too low, most points may look compressed in a dense 
354          cloud with few outliers. 
355   
356      min_gain : float, default=0.01 
357          Minimum individual gain for each parameter. 
358   
359      min_grad_norm : float, default=1e-7 
360          If the gradient norm is below this threshold, the optimization will 
361          be aborted. 
362   
363      verbose : int, default=0 
364          Verbosity level. 
365   
366      args : sequence, default=None 
367          Arguments to pass to objective function. 
368   
369      kwargs : dict, default=None 
370          Keyword arguments to pass to objective function. 
371   
372      Returns 
373      ------- 
374      p : ndarray of shape (n_params,) 
375          Optimum parameters. 
376   
377      error : float 
378          Optimum. 
379   
380      i : int 
381          Last iteration. 
382      """
383      if args is None:
384          args = []
385      if kwargs is None:
386          kwargs = {}
387  
388      p = p0.copy().ravel()
389      update = np.zeros_like(p)
390      gains = np.ones_like(p)
391      error = np.finfo(float).max
392      best_error = np.finfo(float).max
393      best_iter = i = it
394  
395      tic = time()
396      for i in range(it, max_iter):
397          check_convergence = (i + 1) % n_iter_check == 0
398          # only compute the error when needed
399          kwargs["compute_error"] = check_convergence or i == max_iter - 1
400  
401          error, grad = objective(p, *args, **kwargs)
402  
403          inc = update * grad < 0.0
404          dec = np.invert(inc)
405          gains[inc] += 0.2
406          gains[dec] *= 0.8
407          np.clip(gains, min_gain, np.inf, out=gains)
408          grad *= gains
409          update = momentum * update - learning_rate * grad
410          p += update
411  
412          if check_convergence:
413              toc = time()
414              duration = toc - tic
415              tic = toc
416              grad_norm = linalg.norm(grad)
417  
418              if verbose >= 2:
419                  print(
420                      "[t-SNE] Iteration %d: error = %.7f,"
421                      " gradient norm = %.7f"
422                      " (%s iterations in %0.3fs)"
423                      % (i + 1, error, grad_norm, n_iter_check, duration)
424                  )
425  
426              if error < best_error:
427                  best_error = error
428                  best_iter = i
429              elif i - best_iter > n_iter_without_progress:
430                  if verbose >= 2:
431                      print(
432                          "[t-SNE] Iteration %d: did not make any progress "
433                          "during the last %d episodes. Finished."
434                          % (i + 1, n_iter_without_progress)
435                      )
436                  break
437              if grad_norm <= min_grad_norm:
438                  if verbose >= 2:
439                      print(
440                          "[t-SNE] Iteration %d: gradient norm %f. Finished."
441                          % (i + 1, grad_norm)
442                      )
443                  break
444  
445      return p, error, i

448  @validate_params(
449      {
450          "X": ["array-like", "sparse matrix"],
451          "X_embedded": ["array-like", "sparse matrix"],
452          "n_neighbors": [Interval(Integral, 1, None, closed="left")],
453          "metric": [StrOptions(set(_VALID_METRICS) | {"precomputed"}), callable],
454      },
455      prefer_skip_nested_validation=True,
456  )
457  def trustworthiness(X, X_embedded, *, n_neighbors=5, metric="euclidean"):
458      r"""Indicate to what extent the local structure is retained. 
459   
460      The trustworthiness is within [0, 1]. It is defined as 
461   
462      .. math:: 
463   
464          T(k) = 1 - \frac{2}{nk (2n - 3k - 1)} \sum^n_{i=1} 
465              \sum_{j \in \mathcal{N}_{i}^{k}} \max(0, (r(i, j) - k)) 
466   
467      where for each sample i, :math:`\mathcal{N}_{i}^{k}` are its k nearest 
468      neighbors in the output space, and every sample j is its :math:`r(i, j)`-th 
469      nearest neighbor in the input space. In other words, any unexpected nearest 
470      neighbors in the output space are penalised in proportion to their rank in 
471      the input space. 
472   
473      Parameters 
474      ---------- 
475      X : {array-like, sparse matrix} of shape (n_samples, n_features) or \ 
476          (n_samples, n_samples) 
477          If the metric is 'precomputed' X must be a square distance 
478          matrix. Otherwise it contains a sample per row. 
479   
480      X_embedded : {array-like, sparse matrix} of shape (n_samples, n_components) 
481          Embedding of the training data in low-dimensional space. 
482   
483      n_neighbors : int, default=5 
484          The number of neighbors that will be considered. Should be fewer than 
485          `n_samples / 2` to ensure the trustworthiness to lies within [0, 1], as 
486          mentioned in [1]_. An error will be raised otherwise. 
487   
488      metric : str or callable, default='euclidean' 
489          Which metric to use for computing pairwise distances between samples 
490          from the original input space. If metric is 'precomputed', X must be a 
491          matrix of pairwise distances or squared distances. Otherwise, for a list 
492          of available metrics, see the documentation of argument metric in 
493          `sklearn.pairwise.pairwise_distances` and metrics listed in 
494          `sklearn.metrics.pairwise.PAIRWISE_DISTANCE_FUNCTIONS`. Note that the 
495          "cosine" metric uses :func:`~sklearn.metrics.pairwise.cosine_distances`. 
496   
497          .. versionadded:: 0.20 
498   
499      Returns 
500      ------- 
501      trustworthiness : float 
502          Trustworthiness of the low-dimensional embedding. 
503   
504      References 
505      ---------- 
506      .. [1] Jarkko Venna and Samuel Kaski. 2001. Neighborhood 
507             Preservation in Nonlinear Projection Methods: An Experimental Study. 
508             In Proceedings of the International Conference on Artificial Neural Networks 
509             (ICANN '01). Springer-Verlag, Berlin, Heidelberg, 485-491. 
510   
511      .. [2] Laurens van der Maaten. Learning a Parametric Embedding by Preserving 
512             Local Structure. Proceedings of the Twelfth International Conference on 
513             Artificial Intelligence and Statistics, PMLR 5:384-391, 2009. 
514   
515      Examples 
516      -------- 
517      >>> from sklearn.datasets import make_blobs 
518      >>> from sklearn.decomposition import PCA 
519      >>> from sklearn.manifold import trustworthiness 
520      >>> X, _ = make_blobs(n_samples=100, n_features=10, centers=3, random_state=42) 
521      >>> X_embedded = PCA(n_components=2).fit_transform(X) 
522      >>> print(f"{trustworthiness(X, X_embedded, n_neighbors=5):.2f}") 
523      0.92 
524      """
525      n_samples = _num_samples(X)
526      if n_neighbors >= n_samples / 2:
527          raise ValueError(
528              f"n_neighbors ({n_neighbors}) should be less than n_samples / 2"
529              f" ({n_samples / 2})"
530          )
531      dist_X = pairwise_distances(X, metric=metric)
532      if metric == "precomputed":
533          dist_X = dist_X.copy()
534      # we set the diagonal to np.inf to exclude the points themselves from
535      # their own neighborhood
536      np.fill_diagonal(dist_X, np.inf)
537      ind_X = np.argsort(dist_X, axis=1)
538      # `ind_X[i]` is the index of sorted distances between i and other samples
539      ind_X_embedded = (
540          NearestNeighbors(n_neighbors=n_neighbors)
541          .fit(X_embedded)
542          .kneighbors(return_distance=False)
543      )
544  
545      # We build an inverted index of neighbors in the input space: For sample i,
546      # we define `inverted_index[i]` as the inverted index of sorted distances:
547      # inverted_index[i][ind_X[i]] = np.arange(1, n_sample + 1)
548      inverted_index = np.zeros((n_samples, n_samples), dtype=int)
549      ordered_indices = np.arange(n_samples + 1)
550      inverted_index[ordered_indices[:-1, np.newaxis], ind_X] = ordered_indices[1:]
551      ranks = (
552          inverted_index[ordered_indices[:-1, np.newaxis], ind_X_embedded] - n_neighbors
553      )
554      t = np.sum(ranks[ranks > 0])
555      t = 1.0 - t * (
556          2.0 / (n_samples * n_neighbors * (2.0 * n_samples - 3.0 * n_neighbors - 1.0))
557      )
558      return t

561  class TSNE(ClassNamePrefixFeaturesOutMixin, TransformerMixin, BaseEstimator):
562      """T-distributed Stochastic Neighbor Embedding. 
563   
564      t-SNE [1] is a tool to visualize high-dimensional data. It converts 
565      similarities between data points to joint probabilities and tries 
566      to minimize the Kullback-Leibler divergence between the joint 
567      probabilities of the low-dimensional embedding and the 
568      high-dimensional data. t-SNE has a cost function that is not convex, 
569      i.e. with different initializations we can get different results. 
570   
571      It is highly recommended to use another dimensionality reduction 
572      method (e.g. PCA for dense data or TruncatedSVD for sparse data) 
573      to reduce the number of dimensions to a reasonable amount (e.g. 50) 
574      if the number of features is very high. This will suppress some 
575      noise and speed up the computation of pairwise distances between 
576      samples. For more tips see Laurens van der Maaten's FAQ [2]. 
577   
578      Read more in the :ref:`User Guide <t_sne>`. 
579   
580      Parameters 
581      ---------- 
582      n_components : int, default=2 
583          Dimension of the embedded space. 
584   
585      perplexity : float, default=30.0 
586          The perplexity is related to the number of nearest neighbors that 
587          is used in other manifold learning algorithms. Larger datasets 
588          usually require a larger perplexity. Consider selecting a value 
589          between 5 and 50. Different values can result in significantly 
590          different results. The perplexity must be less than the number 
591          of samples. 
592   
593      early_exaggeration : float, default=12.0 
594          Controls how tight natural clusters in the original space are in 
595          the embedded space and how much space will be between them. For 
596          larger values, the space between natural clusters will be larger 
597          in the embedded space. Again, the choice of this parameter is not 
598          very critical. If the cost function increases during initial 
599          optimization, the early exaggeration factor or the learning rate 
600          might be too high. 
601   
602      learning_rate : float or "auto", default="auto" 
603          The learning rate for t-SNE is usually in the range [10.0, 1000.0]. If 
604          the learning rate is too high, the data may look like a 'ball' with any 
605          point approximately equidistant from its nearest neighbours. If the 
606          learning rate is too low, most points may look compressed in a dense 
607          cloud with few outliers. If the cost function gets stuck in a bad local 
608          minimum increasing the learning rate may help. 
609          Note that many other t-SNE implementations (bhtsne, FIt-SNE, openTSNE, 
610          etc.) use a definition of learning_rate that is 4 times smaller than 
611          ours. So our learning_rate=200 corresponds to learning_rate=800 in 
612          those other implementations. The 'auto' option sets the learning_rate 
613          to `max(N / early_exaggeration / 4, 50)` where N is the sample size, 
614          following [4] and [5]. 
615   
616          .. versionchanged:: 1.2 
617             The default value changed to `"auto"`. 
618   
619      max_iter : int, default=1000 
620          Maximum number of iterations for the optimization. Should be at 
621          least 250. 
622   
623          .. versionchanged:: 1.5 
624              Parameter name changed from `n_iter` to `max_iter`. 
625   
626      n_iter_without_progress : int, default=300 
627          Maximum number of iterations without progress before we abort the 
628          optimization, used after 250 initial iterations with early 
629          exaggeration. Note that progress is only checked every 50 iterations so 
630          this value is rounded to the next multiple of 50. 
631   
632          .. versionadded:: 0.17 
633             parameter *n_iter_without_progress* to control stopping criteria. 
634   
635      min_grad_norm : float, default=1e-7 
636          If the gradient norm is below this threshold, the optimization will 
637          be stopped. 
638   
639      metric : str or callable, default='euclidean' 
640          The metric to use when calculating distance between instances in a 
641          feature array. If metric is a string, it must be one of the options 
642          allowed by scipy.spatial.distance.pdist for its metric parameter, or 
643          a metric listed in pairwise.PAIRWISE_DISTANCE_FUNCTIONS. 
644          If metric is "precomputed", X is assumed to be a distance matrix. 
645          Alternatively, if metric is a callable function, it is called on each 
646          pair of instances (rows) and the resulting value recorded. The callable 
647          should take two arrays from X as input and return a value indicating 
648          the distance between them. The default is "euclidean" which is 
649          interpreted as squared euclidean distance. 
650   
651      metric_params : dict, default=None 
652          Additional keyword arguments for the metric function. 
653   
654          .. versionadded:: 1.1 
655   
656      init : {"random", "pca"} or ndarray of shape (n_samples, n_components), \ 
657              default="pca" 
658          Initialization of embedding. 
659          PCA initialization cannot be used with precomputed distances and is 
660          usually more globally stable than random initialization. 
661   
662          .. versionchanged:: 1.2 
663             The default value changed to `"pca"`. 
664   
665      verbose : int, default=0 
666          Verbosity level. 
667   
668      random_state : int, RandomState instance or None, default=None 
669          Determines the random number generator. Pass an int for reproducible 
670          results across multiple function calls. Note that different 
671          initializations might result in different local minima of the cost 
672          function. See :term:`Glossary <random_state>`. 
673   
674      method : {'barnes_hut', 'exact'}, default='barnes_hut' 
675          By default the gradient calculation algorithm uses Barnes-Hut 
676          approximation running in O(NlogN) time. method='exact' 
677          will run on the slower, but exact, algorithm in O(N^2) time. The 
678          exact algorithm should be used when nearest-neighbor errors need 
679          to be better than 3%. However, the exact method cannot scale to 
680          millions of examples. 
681   
682          .. versionadded:: 0.17 
683             Approximate optimization *method* via the Barnes-Hut. 
684   
685      angle : float, default=0.5 
686          Only used if method='barnes_hut' 
687          This is the trade-off between speed and accuracy for Barnes-Hut T-SNE. 
688          'angle' is the angular size (referred to as theta in [3]) of a distant 
689          node as measured from a point. If this size is below 'angle' then it is 
690          used as a summary node of all points contained within it. 
691          This method is not very sensitive to changes in this parameter 
692          in the range of 0.2 - 0.8. Angle less than 0.2 has quickly increasing 
693          computation time and angle greater 0.8 has quickly increasing error. 
694   
695      n_jobs : int, default=None 
696          The number of parallel jobs to run for neighbors search. This parameter 
697          has no impact when ``metric="precomputed"`` or 
698          (``metric="euclidean"`` and ``method="exact"``). 
699          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
700          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
701          for more details. 
702   
703          .. versionadded:: 0.22 
704   
705      n_iter : int 
706          Maximum number of iterations for the optimization. Should be at 
707          least 250. 
708   
709          .. deprecated:: 1.5 
710              `n_iter` was deprecated in version 1.5 and will be removed in 1.7. 
711              Please use `max_iter` instead. 
712

860      def _check_params_vs_input(self, X):
861          if self.perplexity >= X.shape[0]:
862              raise ValueError("perplexity must be less than n_samples")
863  
864      def _fit(self, X, skip_num_points=0):
865          """Private function to fit the model using X as training data."""
866  
867          if isinstance(self.init, str) and self.init == "pca" and issparse(X):
868              raise TypeError(
869                  "PCA initialization is currently not supported "
870                  "with the sparse input matrix. Use "
871                  'init="random" instead.'
872              )
873  
874          if self.learning_rate == "auto":
875              # See issue #18018
876              self.learning_rate_ = X.shape[0] / self.early_exaggeration / 4
877              self.learning_rate_ = np.maximum(self.learning_rate_, 50)
878          else:
879              self.learning_rate_ = self.learning_rate
880  
881          if self.method == "barnes_hut":
882              X = validate_data(
883                  self,
884                  X,
885                  accept_sparse=["csr"],
886                  ensure_min_samples=2,
887                  dtype=[np.float32, np.float64],
888              )
889          else:
890              X = validate_data(
891                  self,
892                  X,
893                  accept_sparse=["csr", "csc", "coo"],
894                  dtype=[np.float32, np.float64],
895              )
896          if self.metric == "precomputed":
897              if isinstance(self.init, str) and self.init == "pca":
898                  raise ValueError(
899                      'The parameter init="pca" cannot be used with metric="precomputed".'
900                  )
901              if X.shape[0] != X.shape[1]:
902                  raise ValueError("X should be a square distance matrix")
903  
904              check_non_negative(
905                  X,
906                  (
907                      "TSNE.fit(). With metric='precomputed', X "
908                      "should contain positive distances."
909                  ),
910              )
911  
912              if self.method == "exact" and issparse(X):
913                  raise TypeError(
914                      'TSNE with method="exact" does not accept sparse '
915                      'precomputed distance matrix. Use method="barnes_hut" '
916                      "or provide the dense distance matrix."
917                  )
918  
919          if self.method == "barnes_hut" and self.n_components > 3:
920              raise ValueError(
921                  "'n_components' should be inferior to 4 for the "
922                  "barnes_hut algorithm as it relies on "
923                  "quad-tree or oct-tree."
924              )
925          random_state = check_random_state(self.random_state)
926  
927          n_samples = X.shape[0]
928  
929          neighbors_nn = None
930          if self.method == "exact":
931              # Retrieve the distance matrix, either using the precomputed one or
932              # computing it.
933              if self.metric == "precomputed":
934                  distances = X
935              else:
936                  if self.verbose:
937                      print("[t-SNE] Computing pairwise distances...")
938  
939                  if self.metric == "euclidean":
940                      # Euclidean is squared here, rather than using **= 2,
941                      # because euclidean_distances already calculates
942                      # squared distances, and returns np.sqrt(dist) for
943                      # squared=False.
944                      # Also, Euclidean is slower for n_jobs>1, so don't set here
945                      distances = pairwise_distances(X, metric=self.metric, squared=True)
946                  else:
947                      metric_params_ = self.metric_params or {}
948                      distances = pairwise_distances(
949                          X, metric=self.metric, n_jobs=self.n_jobs, **metric_params_
950                      )
951  
952              if np.any(distances < 0):
953                  raise ValueError(
954                      "All distances should be positive, the metric given is not correct"
955                  )
956  
957              if self.metric != "euclidean":
958                  distances **= 2
959  
960              # compute the joint probability distribution for the input space
961              P = _joint_probabilities(distances, self.perplexity, self.verbose)
962              assert np.all(np.isfinite(P)), "All probabilities should be finite"
963              assert np.all(P >= 0), "All probabilities should be non-negative"
964              assert np.all(
965                  P <= 1
966              ), "All probabilities should be less or then equal to one"
967  
968          else:
969              # Compute the number of nearest neighbors to find.
970              # LvdM uses 3 * perplexity as the number of neighbors.
971              # In the event that we have very small # of points
972              # set the neighbors to n - 1.
973              n_neighbors = min(n_samples - 1, int(3.0 * self.perplexity + 1))
974  
975              if self.verbose:
976                  print("[t-SNE] Computing {} nearest neighbors...".format(n_neighbors))
977  
978              # Find the nearest neighbors for every point
979              knn = NearestNeighbors(
980                  algorithm="auto",
981                  n_jobs=self.n_jobs,
982                  n_neighbors=n_neighbors,
983                  metric=self.metric,
984                  metric_params=self.metric_params,
985              )
986              t0 = time()
987              knn.fit(X)
988              duration = time() - t0
989              if self.verbose:
990                  print(
991                      "[t-SNE] Indexed {} samples in {:.3f}s...".format(
992                          n_samples, duration
993                      )
994                  )
995  
996              t0 = time()
997              distances_nn = knn.kneighbors_graph(mode="distance")
998              duration = time() - t0
999              if self.verbose:
1000                 print(
1001                     "[t-SNE] Computed neighbors for {} samples in {:.3f}s...".format(
1002                         n_samples, duration
1003                     )
1004                 )
1005 
1006             # Free the memory used by the ball_tree
1007             del knn
1008 
1009             # knn return the euclidean distance but we need it squared
1010             # to be consistent with the 'exact' method. Note that the
1011             # the method was derived using the euclidean method as in the
1012             # input space. Not sure of the implication of using a different
1013             # metric.
1014             distances_nn.data **= 2
1015 
1016             # compute the joint probability distribution for the input space
1017             P = _joint_probabilities_nn(distances_nn, self.perplexity, self.verbose)
1018 
1019         if isinstance(self.init, np.ndarray):
1020             X_embedded = self.init
1021         elif self.init == "pca":
1022             pca = PCA(
1023                 n_components=self.n_components,
1024                 svd_solver="randomized",
1025                 random_state=random_state,
1026             )
1027             # Always output a numpy array, no matter what is configured globally
1028             pca.set_output(transform="default")
1029             X_embedded = pca.fit_transform(X).astype(np.float32, copy=False)
1030             # PCA is rescaled so that PC1 has standard deviation 1e-4 which is
1031             # the default value for random initialization. See issue #18018.
1032             X_embedded = X_embedded / np.std(X_embedded[:, 0]) * 1e-4
1033         elif self.init == "random":
1034             # The embedding is initialized with iid samples from Gaussians with
1035             # standard deviation 1e-4.
1036             X_embedded = 1e-4 * random_state.standard_normal(
1037                 size=(n_samples, self.n_components)
1038             ).astype(np.float32)
1039 
1040         # Degrees of freedom of the Student's t-distribution. The suggestion
1041         # degrees_of_freedom = n_components - 1 comes from
1042         # "Learning a Parametric Embedding by Preserving Local Structure"
1043         # Laurens van der Maaten, 2009.
1044         degrees_of_freedom = max(self.n_components - 1, 1)
1045 
1046         return self._tsne(
1047             P,
1048             degrees_of_freedom,
1049             n_samples,
1050             X_embedded=X_embedded,
1051             neighbors=neighbors_nn,
1052             skip_num_points=skip_num_points,
1053         )
1054 
1055     def _tsne(
1056         self,
1057         P,
1058         degrees_of_freedom,
1059         n_samples,
1060         X_embedded,
1061         neighbors=None,
1062         skip_num_points=0,
1063     ):
1064         """Runs t-SNE."""
1065         # t-SNE minimizes the Kullback-Leiber divergence of the Gaussians P
1066         # and the Student's t-distributions Q. The optimization algorithm that
1067         # we use is batch gradient descent with two stages:
1068         # * initial optimization with early exaggeration and momentum at 0.5
1069         # * final optimization with momentum at 0.8
1070         params = X_embedded.ravel()
1071 
1072         opt_args = {
1073             "it": 0,
1074             "n_iter_check": self._N_ITER_CHECK,
1075             "min_grad_norm": self.min_grad_norm,
1076             "learning_rate": self.learning_rate_,
1077             "verbose": self.verbose,
1078             "kwargs": dict(skip_num_points=skip_num_points),
1079             "args": [P, degrees_of_freedom, n_samples, self.n_components],
1080             "n_iter_without_progress": self._EXPLORATION_MAX_ITER,
1081             "max_iter": self._EXPLORATION_MAX_ITER,
1082             "momentum": 0.5,
1083         }
1084         if self.method == "barnes_hut":
1085             obj_func = _kl_divergence_bh
1086             opt_args["kwargs"]["angle"] = self.angle
1087             # Repeat verbose argument for _kl_divergence_bh
1088             opt_args["kwargs"]["verbose"] = self.verbose
1089             # Get the number of threads for gradient computation here to
1090             # avoid recomputing it at each iteration.
1091             opt_args["kwargs"]["num_threads"] = _openmp_effective_n_threads()
1092         else:
1093             obj_func = _kl_divergence
1094 
1095         # Learning schedule (part 1): do 250 iteration with lower momentum but
1096         # higher learning rate controlled via the early exaggeration parameter
1097         P *= self.early_exaggeration
1098         params, kl_divergence, it = _gradient_descent(obj_func, params, **opt_args)
1099         if self.verbose:
1100             print(
1101                 "[t-SNE] KL divergence after %d iterations with early exaggeration: %f"
1102                 % (it + 1, kl_divergence)
1103             )
1104 
1105         # Learning schedule (part 2): disable early exaggeration and finish
1106         # optimization with a higher momentum at 0.8
1107         P /= self.early_exaggeration
1108         remaining = self._max_iter - self._EXPLORATION_MAX_ITER
1109         if it < self._EXPLORATION_MAX_ITER or remaining > 0:
1110             opt_args["max_iter"] = self._max_iter
1111             opt_args["it"] = it + 1
1112             opt_args["momentum"] = 0.8
1113             opt_args["n_iter_without_progress"] = self.n_iter_without_progress
1114             params, kl_divergence, it = _gradient_descent(obj_func, params, **opt_args)
1115 
1116         # Save the final number of iterations
1117         self.n_iter_ = it
1118 
1119         if self.verbose:
1120             print(
1121                 "[t-SNE] KL divergence after %d iterations: %f"
1122                 % (it + 1, kl_divergence)
1123             )
1124 
1125         X_embedded = params.reshape(n_samples, self.n_components)
1126         self.kl_divergence_ = kl_divergence
1127 
1128         return X_embedded
