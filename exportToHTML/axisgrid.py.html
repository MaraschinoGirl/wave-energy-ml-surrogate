126      def add_legend(self, legend_data=None, title=None, label_order=None,
127                     adjust_subtitles=False, **kwargs):
128          """Draw a legend, maybe placing it outside axes and resizing the figure. 
129   
130          Parameters 
131          ---------- 
132          legend_data : dict 
133              Dictionary mapping label names (or two-element tuples where the 
134              second element is a label name) to matplotlib artist handles. The 
135              default reads from ``self._legend_data``. 
136          title : string 
137              Title for the legend. The default reads from ``self._hue_var``. 
138          label_order : list of labels 
139              The order that the legend entries should appear in. The default 
140              reads from ``self.hue_names``. 
141          adjust_subtitles : bool 
142              If True, modify entries with invisible artists to left-align 
143              the labels and set the font size to that of a title. 
144          kwargs : key, value pairings 
145              Other keyword arguments are passed to the underlying legend methods 
146              on the Figure or Axes object. 
147   
148          Returns 
149          ------- 
150          self : Grid instance 
151              Returns self for easy chaining. 
152   
153          """
154          # Find the data for the legend
155          if legend_data is None:
156              legend_data = self._legend_data
157          if label_order is None:
158              if self.hue_names is None:
159                  label_order = list(legend_data.keys())
160              else:
161                  label_order = list(map(utils.to_utf8, self.hue_names))
162  
163          blank_handle = mpl.patches.Patch(alpha=0, linewidth=0)
164          handles = [legend_data.get(lab, blank_handle) for lab in label_order]
165          title = self._hue_var if title is None else title
166          title_size = mpl.rcParams["legend.title_fontsize"]
167  
168          # Unpack nested labels from a hierarchical legend
169          labels = []
170          for entry in label_order:
171              if isinstance(entry, tuple):
172                  _, label = entry
173              else:
174                  label = entry
175              labels.append(label)
176  
177          # Set default legend kwargs
178          kwargs.setdefault("scatterpoints", 1)
179  
180          if self._legend_out:
181  
182              kwargs.setdefault("frameon", False)
183              kwargs.setdefault("loc", "center right")
184  
185              # Draw a full-figure legend outside the grid
186              figlegend = self._figure.legend(handles, labels, **kwargs)
187  
188              self._legend = figlegend
189              figlegend.set_title(title, prop={"size": title_size})
190  
191              if adjust_subtitles:
192                  adjust_legend_subtitles(figlegend)
193  
194              # Draw the plot to set the bounding boxes correctly
195              _draw_figure(self._figure)
196  
197              # Calculate and set the new width of the figure so the legend fits
198              legend_width = figlegend.get_window_extent().width / self._figure.dpi
199              fig_width, fig_height = self._figure.get_size_inches()
200              self._figure.set_size_inches(fig_width + legend_width, fig_height)
201  
202              # Draw the plot again to get the new transformations
203              _draw_figure(self._figure)
204  
205              # Now calculate how much space we need on the right side
206              legend_width = figlegend.get_window_extent().width / self._figure.dpi
207              space_needed = legend_width / (fig_width + legend_width)
208              margin = .04 if self._margin_titles else .01
209              self._space_needed = margin + space_needed
210              right = 1 - self._space_needed
211  
212              # Place the subplot axes to give space for the legend
213              self._figure.subplots_adjust(right=right)
214              self._tight_layout_rect[2] = right
215  
216          else:
217              # Draw a legend in the first axis
218              ax = self.axes.flat[0]
219              kwargs.setdefault("loc", "best")
220  
221              leg = ax.legend(handles, labels, **kwargs)
222              leg.set_title(title, prop={"size": title_size})
223              self._legend = leg
224  
225              if adjust_subtitles:
226                  adjust_legend_subtitles(leg)
227  
228          return self

230      def _update_legend_data(self, ax):
231          """Extract the legend data from an axes object and save it."""
232          data = {}
233  
234          # Get data directly from the legend, which is necessary
235          # for newer functions that don't add labeled proxy artists
236          if ax.legend_ is not None and self._extract_legend_handles:
237              handles = get_legend_handles(ax.legend_)
238              labels = [t.get_text() for t in ax.legend_.texts]
239              data.update({label: handle for handle, label in zip(handles, labels)})
240  
241          handles, labels = ax.get_legend_handles_labels()
242          data.update({label: handle for handle, label in zip(handles, labels)})
243  
244          self._legend_data.update(data)
245  
246          # Now clear the legend
247          ax.legend_ = None

249      def _get_palette(self, data, hue, hue_order, palette):
250          """Get a list of colors for the hue variable."""
251          if hue is None:
252              palette = color_palette(n_colors=1)
253  
254          else:
255              hue_names = categorical_order(data[hue], hue_order)
256              n_colors = len(hue_names)
257  
258              # By default use either the current color palette or HUSL
259              if palette is None:
260                  current_palette = utils.get_color_cycle()
261                  if n_colors > len(current_palette):
262                      colors = color_palette("husl", n_colors)
263                  else:
264                      colors = color_palette(n_colors=n_colors)
265  
266              # Allow for palette to map from hue variable names
267              elif isinstance(palette, dict):
268                  color_names = [palette[h] for h in hue_names]
269                  colors = color_palette(color_names, n_colors)
270  
271              # Otherwise act as if we just got a list of colors
272              else:
273                  colors = color_palette(palette, n_colors)
274  
275              palette = color_palette(colors, n_colors)
276  
277          return palette

371      def __init__(
372          self, data, *,
373          row=None, col=None, hue=None, col_wrap=None,
374          sharex=True, sharey=True, height=3, aspect=1, palette=None,
375          row_order=None, col_order=None, hue_order=None, hue_kws=None,
376          dropna=False, legend_out=True, despine=True,
377          margin_titles=False, xlim=None, ylim=None, subplot_kws=None,
378          gridspec_kws=None,
379      ):
380  
381          super().__init__()
382          data = handle_data_source(data)
383  
384          # Determine the hue facet layer information
385          hue_var = hue
386          if hue is None:
387              hue_names = None
388          else:
389              hue_names = categorical_order(data[hue], hue_order)
390  
391          colors = self._get_palette(data, hue, hue_order, palette)
392  
393          # Set up the lists of names for the row and column facet variables
394          if row is None:
395              row_names = []
396          else:
397              row_names = categorical_order(data[row], row_order)
398  
399          if col is None:
400              col_names = []
401          else:
402              col_names = categorical_order(data[col], col_order)
403  
404          # Additional dict of kwarg -> list of values for mapping the hue var
405          hue_kws = hue_kws if hue_kws is not None else {}
406  
407          # Make a boolean mask that is True anywhere there is an NA
408          # value in one of the faceting variables, but only if dropna is True
409          none_na = np.zeros(len(data), bool)
410          if dropna:
411              row_na = none_na if row is None else data[row].isnull()
412              col_na = none_na if col is None else data[col].isnull()
413              hue_na = none_na if hue is None else data[hue].isnull()
414              not_na = ~(row_na | col_na | hue_na)
415          else:
416              not_na = ~none_na
417  
418          # Compute the grid shape
419          ncol = 1 if col is None else len(col_names)
420          nrow = 1 if row is None else len(row_names)
421          self._n_facets = ncol * nrow
422  
423          self._col_wrap = col_wrap
424          if col_wrap is not None:
425              if row is not None:
426                  err = "Cannot use `row` and `col_wrap` together."
427                  raise ValueError(err)
428              ncol = col_wrap
429              nrow = int(np.ceil(len(col_names) / col_wrap))
430          self._ncol = ncol
431          self._nrow = nrow
432  
433          # Calculate the base figure size
434          # This can get stretched later by a legend
435          # TODO this doesn't account for axis labels
436          figsize = (ncol * height * aspect, nrow * height)
437  
438          # Validate some inputs
439          if col_wrap is not None:
440              margin_titles = False
441  
442          # Build the subplot keyword dictionary
443          subplot_kws = {} if subplot_kws is None else subplot_kws.copy()
444          gridspec_kws = {} if gridspec_kws is None else gridspec_kws.copy()
445          if xlim is not None:
446              subplot_kws["xlim"] = xlim
447          if ylim is not None:
448              subplot_kws["ylim"] = ylim
449  
450          # --- Initialize the subplot grid
451  
452          with _disable_autolayout():
453              fig = plt.figure(figsize=figsize)
454  
455          if col_wrap is None:
456  
457              kwargs = dict(squeeze=False,
458                            sharex=sharex, sharey=sharey,
459                            subplot_kw=subplot_kws,
460                            gridspec_kw=gridspec_kws)
461  
462              axes = fig.subplots(nrow, ncol, **kwargs)
463  
464              if col is None and row is None:
465                  axes_dict = {}
466              elif col is None:
467                  axes_dict = dict(zip(row_names, axes.flat))
468              elif row is None:
469                  axes_dict = dict(zip(col_names, axes.flat))
470              else:
471                  facet_product = product(row_names, col_names)
472                  axes_dict = dict(zip(facet_product, axes.flat))
473  
474          else:
475  
476              # If wrapping the col variable we need to make the grid ourselves
477              if gridspec_kws:
478                  warnings.warn("`gridspec_kws` ignored when using `col_wrap`")
479  
480              n_axes = len(col_names)
481              axes = np.empty(n_axes, object)
482              axes[0] = fig.add_subplot(nrow, ncol, 1, **subplot_kws)
483              if sharex:
484                  subplot_kws["sharex"] = axes[0]
485              if sharey:
486                  subplot_kws["sharey"] = axes[0]
487              for i in range(1, n_axes):
488                  axes[i] = fig.add_subplot(nrow, ncol, i + 1, **subplot_kws)
489  
490              axes_dict = dict(zip(col_names, axes))
491  
492          # --- Set up the class attributes
493  
494          # Attributes that are part of the public API but accessed through
495          # a  property so that Sphinx adds them to the auto class doc
496          self._figure = fig
497          self._axes = axes
498          self._axes_dict = axes_dict
499          self._legend = None
500  
501          # Public attributes that aren't explicitly documented
502          # (It's not obvious that having them be public was a good idea)
503          self.data = data
504          self.row_names = row_names
505          self.col_names = col_names
506          self.hue_names = hue_names
507          self.hue_kws = hue_kws
508  
509          # Next the private variables
510          self._nrow = nrow
511          self._row_var = row
512          self._ncol = ncol
513          self._col_var = col
514  
515          self._margin_titles = margin_titles
516          self._margin_titles_texts = []
517          self._col_wrap = col_wrap
518          self._hue_var = hue_var
519          self._colors = colors
520          self._legend_out = legend_out
521          self._legend_data = {}
522          self._x_var = None
523          self._y_var = None
524          self._sharex = sharex
525          self._sharey = sharey
526          self._dropna = dropna
527          self._not_na = not_na
528  
529          # --- Make the axes look good
530  
531          self.set_titles()
532          self.tight_layout()
533  
534          if despine:
535              self.despine()
536  
537          if sharex in [True, 'col']:
538              for ax in self._not_bottom_axes:
539                  for label in ax.get_xticklabels():
540                      label.set_visible(False)
541                  ax.xaxis.offsetText.set_visible(False)
542                  ax.xaxis.label.set_visible(False)
543  
544          if sharey in [True, 'row']:
545              for ax in self._not_left_axes:
546                  for label in ax.get_yticklabels():
547                      label.set_visible(False)
548                  ax.yaxis.offsetText.set_visible(False)
549                  ax.yaxis.label.set_visible(False)

643      def facet_data(self):
644          """Generator for name indices and data subsets for each facet. 
645   
646          Yields 
647          ------ 
648          (i, j, k), data_ijk : tuple of ints, DataFrame 
649              The ints provide an index into the {row, col, hue}_names attribute, 
650              and the dataframe contains a subset of the full data corresponding 
651              to each facet. The generator yields subsets that correspond with 
652              the self.axes.flat iterator, or self.axes[i, j] when `col_wrap` 
653              is None. 
654   
655          """
656          data = self.data
657  
658          # Construct masks for the row variable
659          if self.row_names:
660              row_masks = [data[self._row_var] == n for n in self.row_names]
661          else:
662              row_masks = [np.repeat(True, len(self.data))]
663  
664          # Construct masks for the column variable
665          if self.col_names:
666              col_masks = [data[self._col_var] == n for n in self.col_names]
667          else:
668              col_masks = [np.repeat(True, len(self.data))]
669  
670          # Construct masks for the hue variable
671          if self.hue_names:
672              hue_masks = [data[self._hue_var] == n for n in self.hue_names]
673          else:

678                                                      enumerate(col_masks),
679                                                      enumerate(hue_masks)):

683      def map(self, func, *args, **kwargs):
684          """Apply a plotting function to each facet's subset of the data. 
685   
686          Parameters 
687          ---------- 
688          func : callable 
689              A plotting function that takes data and keyword arguments. It 
690              must plot to the currently active matplotlib Axes and take a 
691              `color` keyword argument. If faceting on the `hue` dimension, 
692              it must also take a `label` keyword argument. 
693          args : strings 
694              Column names in self.data that identify variables with data to 
695              plot. The data for each variable is passed to `func` in the 
696              order the variables are specified in the call. 
697          kwargs : keyword arguments 
698              All keyword arguments are passed to the plotting function. 
699   
700          Returns 
701          ------- 
702          self : object 
703              Returns self. 
704   
705          """
706          # If color was a keyword argument, grab it here
707          kw_color = kwargs.pop("color", None)
708  
709          # How we use the function depends on where it comes from
710          func_module = str(getattr(func, "__module__", ""))
711  
712          # Check for categorical plots without order information
713          if func_module == "seaborn.categorical":
714              if "order" not in kwargs:
715                  warning = ("Using the {} function without specifying "
716                             "`order` is likely to produce an incorrect "
717                             "plot.".format(func.__name__))
718                  warnings.warn(warning)
719              if len(args) == 3 and "hue_order" not in kwargs:
720                  warning = ("Using the {} function without specifying "
721                             "`hue_order` is likely to produce an incorrect "
722                             "plot.".format(func.__name__))
723                  warnings.warn(warning)
724  
725          # Iterate over the data subsets
726          for (row_i, col_j, hue_k), data_ijk in self.facet_data():
727  
728              # If this subset is null, move on
729              if not data_ijk.values.size:
730                  continue
731  
732              # Get the current axis
733              modify_state = not func_module.startswith("seaborn")
734              ax = self.facet_axis(row_i, col_j, modify_state)
735  
736              # Decide what color to plot with
737              kwargs["color"] = self._facet_color(hue_k, kw_color)
738  
739              # Insert the other hue aesthetics if appropriate
740              for kw, val_list in self.hue_kws.items():
741                  kwargs[kw] = val_list[hue_k]
742  
743              # Insert a label in the keyword arguments for the legend
744              if self._hue_var is not None:
745                  kwargs["label"] = utils.to_utf8(self.hue_names[hue_k])
746  
747              # Get the actual data we are going to plot with
748              plot_data = data_ijk[list(args)]
749              if self._dropna:
750                  plot_data = plot_data.dropna()
751              plot_args = [v for k, v in plot_data.items()]
752  
753              # Some matplotlib functions don't handle pandas objects correctly
754              if func_module.startswith("matplotlib"):
755                  plot_args = [v.values for v in plot_args]
756  
757              # Draw the plot
758              self._facet_plot(func, ax, plot_args, kwargs)
759  
760          # Finalize the annotations and layout
761          self._finalize_grid(args[:2])
762  
763          return self

948          ----------

976                  template = col_template

1177 class PairGrid(Grid):
1178     """Subplot grid for plotting pairwise relationships in a dataset. 
1179  
1180     This object maps each variable in a dataset onto a column and row in a 
1181     grid of multiple axes. Different axes-level plotting functions can be 
1182     used to draw bivariate plots in the upper and lower triangles, and the 
1183     marginal distribution of each variable can be shown on the diagonal. 
1184  
1185     Several different common plots can be generated in a single line using 
1186     :func:`pairplot`. Use :class:`PairGrid` when you need more flexibility. 
1187  
1188     See the :ref:`tutorial <grid_tutorial>` for more information. 
1189  
1190     """
1191     def __init__(
1192         self, data, *, hue=None, vars=None, x_vars=None, y_vars=None,
1193         hue_order=None, palette=None, hue_kws=None, corner=False, diag_sharey=True,
1194         height=2.5, aspect=1, layout_pad=.5, despine=True, dropna=False,
1195     ):
1196         """Initialize the plot figure and PairGrid object. 
1197  
1198         Parameters 
1199         ---------- 
1200         data : DataFrame 
1201             Tidy (long-form) dataframe where each column is a variable and 
1202             each row is an observation. 
1203         hue : string (variable name) 
1204             Variable in ``data`` to map plot aspects to different colors. This 
1205             variable will be excluded from the default x and y variables. 
1206         vars : list of variable names 
1207             Variables within ``data`` to use, otherwise use every column with 
1208             a numeric datatype. 
1209         {x, y}_vars : lists of variable names 
1210             Variables within ``data`` to use separately for the rows and 
1211             columns of the figure; i.e. to make a non-square plot. 
1212         hue_order : list of strings 
1213             Order for the levels of the hue variable in the palette 
1214         palette : dict or seaborn color palette 
1215             Set of colors for mapping the ``hue`` variable. If a dict, keys 
1216             should be values  in the ``hue`` variable. 
1217         hue_kws : dictionary of param -> list of values mapping 
1218             Other keyword arguments to insert into the plotting call to let 
1219             other plot attributes vary across levels of the hue variable (e.g. 
1220             the markers in a scatterplot). 
1221         corner : bool 
1222             If True, don't add axes to the upper (off-diagonal) triangle of the 
1223             grid, making this a "corner" plot. 
1224         height : scalar 
1225             Height (in inches) of each facet. 
1226         aspect : scalar 
1227             Aspect * height gives the width (in inches) of each facet. 
1228         layout_pad : scalar 
1229             Padding between axes; passed to ``fig.tight_layout``. 
1230         despine : boolean 
1231             Remove the top and right spines from the plots. 
1232         dropna : boolean 
1233             Drop missing values from the data before plotting. 
1234  
1235         See Also 
1236         -------- 
1237         pairplot : Easily drawing common uses of :class:`PairGrid`. 
1238         FacetGrid : Subplot grid for plotting conditional relationships. 
1239  
1240         Examples 
1241         -------- 
1242  
1243         .. include:: ../docstrings/PairGrid.rst 
1244  
1245         """
1246 
1247         super().__init__()
1248         data = handle_data_source(data)
1249 
1250         # Sort out the variables that define the grid
1251         numeric_cols = self._find_numeric_cols(data)
1252         if hue in numeric_cols:
1253             numeric_cols.remove(hue)
1254         if vars is not None:
1255             x_vars = list(vars)
1256             y_vars = list(vars)
1257         if x_vars is None:
1258             x_vars = numeric_cols
1259         if y_vars is None:
1260             y_vars = numeric_cols
1261 
1262         if np.isscalar(x_vars):
1263             x_vars = [x_vars]
1264         if np.isscalar(y_vars):
1265             y_vars = [y_vars]
1266 
1267         self.x_vars = x_vars = list(x_vars)
1268         self.y_vars = y_vars = list(y_vars)
1269         self.square_grid = self.x_vars == self.y_vars
1270 
1271         if not x_vars:
1272             raise ValueError("No variables found for grid columns.")
1273         if not y_vars:
1274             raise ValueError("No variables found for grid rows.")
1275 
1276         # Create the figure and the array of subplots
1277         figsize = len(x_vars) * height * aspect, len(y_vars) * height
1278 
1279         with _disable_autolayout():
1280             fig = plt.figure(figsize=figsize)
1281 
1282         axes = fig.subplots(len(y_vars), len(x_vars),
1283                             sharex="col", sharey="row",
1284                             squeeze=False)
1285 
1286         # Possibly remove upper axes to make a corner grid
1287         # Note: setting up the axes is usually the most time-intensive part
1288         # of using the PairGrid. We are foregoing the speed improvement that
1289         # we would get by just not setting up the hidden axes so that we can
1290         # avoid implementing fig.subplots ourselves. But worth thinking about.
