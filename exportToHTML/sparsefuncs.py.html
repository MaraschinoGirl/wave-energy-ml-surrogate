23   def _raise_typeerror(X):
24       """Raises a TypeError if X is not a CSR or CSC matrix"""
25       input_type = X.format if sp.issparse(X) else type(X)
26       err = "Expected a CSR or CSC sparse matrix, got %s." % input_type
27       raise TypeError(err)

30   def _raise_error_wrong_axis(axis):
31       if axis not in (0, 1):
32           raise ValueError(
33               "Unknown axis value: %d. Use 0 for rows, or 1 for columns" % axis
34           )

37   def inplace_csr_column_scale(X, scale):
38       """Inplace column scaling of a CSR matrix. 
39    
40       Scale each feature of the data matrix by multiplying with specific scale 
41       provided by the caller assuming a (n_samples, n_features) shape. 
42    
43       Parameters 
44       ---------- 
45       X : sparse matrix of shape (n_samples, n_features) 
46           Matrix to normalize using the variance of the features. 
47           It should be of CSR format. 
48    
49       scale : ndarray of shape (n_features,), dtype={np.float32, np.float64} 
50           Array of precomputed feature-wise values to use for scaling. 
51    
52       Examples 
53       -------- 
54       >>> from sklearn.utils import sparsefuncs 
55       >>> from scipy import sparse 
56       >>> import numpy as np 
57       >>> indptr = np.array([0, 3, 4, 4, 4]) 
58       >>> indices = np.array([0, 1, 2, 2]) 
59       >>> data = np.array([8, 1, 2, 5]) 
60       >>> scale = np.array([2, 3, 2]) 
61       >>> csr = sparse.csr_matrix((data, indices, indptr)) 
62       >>> csr.todense() 
63       matrix([[8, 1, 2], 
64               [0, 0, 5], 
65               [0, 0, 0], 
66               [0, 0, 0]]) 
67       >>> sparsefuncs.inplace_csr_column_scale(csr, scale) 
68       >>> csr.todense() 
69       matrix([[16,  3,  4], 
70               [ 0,  0, 10], 
71               [ 0,  0,  0], 
72               [ 0,  0,  0]]) 
73       """
74       assert scale.shape[0] == X.shape[1]
75       X.data *= scale.take(X.indices, mode="clip")

78   def inplace_csr_row_scale(X, scale):
79       """Inplace row scaling of a CSR matrix. 
80    
81       Scale each sample of the data matrix by multiplying with specific scale 
82       provided by the caller assuming a (n_samples, n_features) shape. 
83    
84       Parameters 
85       ---------- 
86       X : sparse matrix of shape (n_samples, n_features) 
87           Matrix to be scaled. It should be of CSR format. 
88    
89       scale : ndarray of float of shape (n_samples,) 
90           Array of precomputed sample-wise values to use for scaling. 
91       """
92       assert scale.shape[0] == X.shape[0]
93       X.data *= np.repeat(scale, np.diff(X.indptr))

96   def mean_variance_axis(X, axis, weights=None, return_sum_weights=False):
97       """Compute mean and variance along an axis on a CSR or CSC matrix. 
98    
99       Parameters 
100      ---------- 
101      X : sparse matrix of shape (n_samples, n_features) 
102          Input data. It can be of CSR or CSC format. 
103   
104      axis : {0, 1} 
105          Axis along which the axis should be computed. 
106   
107      weights : ndarray of shape (n_samples,) or (n_features,), default=None 
108          If axis is set to 0 shape is (n_samples,) or 
109          if axis is set to 1 shape is (n_features,). 
110          If it is set to None, then samples are equally weighted. 
111   
112          .. versionadded:: 0.24 
113   
114      return_sum_weights : bool, default=False 
115          If True, returns the sum of weights seen for each feature 
116          if `axis=0` or each sample if `axis=1`. 
117   
118          .. versionadded:: 0.24 
119   
120      Returns 
121      ------- 
122   
123      means : ndarray of shape (n_features,), dtype=floating 
124          Feature-wise means. 
125   
126      variances : ndarray of shape (n_features,), dtype=floating 
127          Feature-wise variances. 
128   
129      sum_weights : ndarray of shape (n_features,), dtype=floating 
130          Returned if `return_sum_weights` is `True`. 
131   
132      Examples 
133      -------- 
134      >>> from sklearn.utils import sparsefuncs 
135      >>> from scipy import sparse 
136      >>> import numpy as np 
137      >>> indptr = np.array([0, 3, 4, 4, 4]) 
138      >>> indices = np.array([0, 1, 2, 2]) 
139      >>> data = np.array([8, 1, 2, 5]) 
140      >>> scale = np.array([2, 3, 2]) 
141      >>> csr = sparse.csr_matrix((data, indices, indptr)) 
142      >>> csr.todense() 
143      matrix([[8, 1, 2], 
144              [0, 0, 5], 
145              [0, 0, 0], 
146              [0, 0, 0]]) 
147      >>> sparsefuncs.mean_variance_axis(csr, axis=0) 
148      (array([2.  , 0.25, 1.75]), array([12.    ,  0.1875,  4.1875])) 
149      """
150      _raise_error_wrong_axis(axis)
151  
152      if sp.issparse(X) and X.format == "csr":
153          if axis == 0:
154              return _csr_mean_var_axis0(
155                  X, weights=weights, return_sum_weights=return_sum_weights
156              )
157          else:
158              return _csc_mean_var_axis0(
159                  X.T, weights=weights, return_sum_weights=return_sum_weights
160              )
161      elif sp.issparse(X) and X.format == "csc":
162          if axis == 0:
163              return _csc_mean_var_axis0(
164                  X, weights=weights, return_sum_weights=return_sum_weights
165              )
166          else:
167              return _csr_mean_var_axis0(
168                  X.T, weights=weights, return_sum_weights=return_sum_weights
169              )
170      else:
171          _raise_typeerror(X)

174  def incr_mean_variance_axis(X, *, axis, last_mean, last_var, last_n, weights=None):
175      """Compute incremental mean and variance along an axis on a CSR or CSC matrix. 
176   
177      last_mean, last_var are the statistics computed at the last step by this 
178      function. Both must be initialized to 0-arrays of the proper size, i.e. 
179      the number of features in X. last_n is the number of samples encountered 
180      until now. 
181   
182      Parameters 
183      ---------- 
184      X : CSR or CSC sparse matrix of shape (n_samples, n_features) 
185          Input data. 
186   
187      axis : {0, 1} 
188          Axis along which the axis should be computed. 
189   
190      last_mean : ndarray of shape (n_features,) or (n_samples,), dtype=floating 
191          Array of means to update with the new data X. 
192          Should be of shape (n_features,) if axis=0 or (n_samples,) if axis=1. 
193   
194      last_var : ndarray of shape (n_features,) or (n_samples,), dtype=floating 
195          Array of variances to update with the new data X. 
196          Should be of shape (n_features,) if axis=0 or (n_samples,) if axis=1. 
197   
198      last_n : float or ndarray of shape (n_features,) or (n_samples,), \ 
199              dtype=floating 
200          Sum of the weights seen so far, excluding the current weights 
201          If not float, it should be of shape (n_features,) if 
202          axis=0 or (n_samples,) if axis=1. If float it corresponds to 
203          having same weights for all samples (or features). 
204   
205      weights : ndarray of shape (n_samples,) or (n_features,), default=None 
206          If axis is set to 0 shape is (n_samples,) or 
207          if axis is set to 1 shape is (n_features,). 
208          If it is set to None, then samples are equally weighted. 
209   
210          .. versionadded:: 0.24 
211   
212      Returns 
213      ------- 
214      means : ndarray of shape (n_features,) or (n_samples,), dtype=floating 
215          Updated feature-wise means if axis = 0 or 
216          sample-wise means if axis = 1. 
217   
218      variances : ndarray of shape (n_features,) or (n_samples,), dtype=floating 
219          Updated feature-wise variances if axis = 0 or 
220          sample-wise variances if axis = 1. 
221   
222      n : ndarray of shape (n_features,) or (n_samples,), dtype=integral 
223          Updated number of seen samples per feature if axis=0 
224          or number of seen features per sample if axis=1. 
225   
226          If weights is not None, n is a sum of the weights of the seen 
227          samples or features instead of the actual number of seen 
228          samples or features. 
229   
230      Notes 
231      ----- 
232      NaNs are ignored in the algorithm. 
233   
234      Examples 
235      -------- 
236      >>> from sklearn.utils import sparsefuncs 
237      >>> from scipy import sparse 
238      >>> import numpy as np 
239      >>> indptr = np.array([0, 3, 4, 4, 4]) 
240      >>> indices = np.array([0, 1, 2, 2]) 
241      >>> data = np.array([8, 1, 2, 5]) 
242      >>> scale = np.array([2, 3, 2]) 
243      >>> csr = sparse.csr_matrix((data, indices, indptr)) 
244      >>> csr.todense() 
245      matrix([[8, 1, 2], 
246              [0, 0, 5], 
247              [0, 0, 0], 
248              [0, 0, 0]]) 
249      >>> sparsefuncs.incr_mean_variance_axis( 
250      ...     csr, axis=0, last_mean=np.zeros(3), last_var=np.zeros(3), last_n=2 
251      ... ) 
252      (array([1.3..., 0.1..., 1.1...]), array([8.8..., 0.1..., 3.4...]), 
253      array([6., 6., 6.])) 
254      """
255      _raise_error_wrong_axis(axis)
256  
257      if not (sp.issparse(X) and X.format in ("csc", "csr")):
258          _raise_typeerror(X)
259  
260      if np.size(last_n) == 1:
261          last_n = np.full(last_mean.shape, last_n, dtype=last_mean.dtype)
262  
263      if not (np.size(last_mean) == np.size(last_var) == np.size(last_n)):
264          raise ValueError("last_mean, last_var, last_n do not have the same shapes.")
265  
266      if axis == 1:
267          if np.size(last_mean) != X.shape[0]:
268              raise ValueError(
269                  "If axis=1, then last_mean, last_n, last_var should be of "
270                  f"size n_samples {X.shape[0]} (Got {np.size(last_mean)})."
271              )
272      else:  # axis == 0
273          if np.size(last_mean) != X.shape[1]:
274              raise ValueError(
275                  "If axis=0, then last_mean, last_n, last_var should be of "
276                  f"size n_features {X.shape[1]} (Got {np.size(last_mean)})."
277              )
278  
279      X = X.T if axis == 1 else X
280  
281      if weights is not None:
282          weights = _check_sample_weight(weights, X, dtype=X.dtype)
283  
284      return _incr_mean_var_axis0(
285          X, last_mean=last_mean, last_var=last_var, last_n=last_n, weights=weights
286      )

289  def inplace_column_scale(X, scale):
290      """Inplace column scaling of a CSC/CSR matrix. 
291   
292      Scale each feature of the data matrix by multiplying with specific scale 
293      provided by the caller assuming a (n_samples, n_features) shape. 
294   
295      Parameters 
296      ---------- 
297      X : sparse matrix of shape (n_samples, n_features) 
298          Matrix to normalize using the variance of the features. It should be 
299          of CSC or CSR format. 
300   
301      scale : ndarray of shape (n_features,), dtype={np.float32, np.float64} 
302          Array of precomputed feature-wise values to use for scaling. 
303   
304      Examples 
305      -------- 
306      >>> from sklearn.utils import sparsefuncs 
307      >>> from scipy import sparse 
308      >>> import numpy as np 
309      >>> indptr = np.array([0, 3, 4, 4, 4]) 
310      >>> indices = np.array([0, 1, 2, 2]) 
311      >>> data = np.array([8, 1, 2, 5]) 
312      >>> scale = np.array([2, 3, 2]) 
313      >>> csr = sparse.csr_matrix((data, indices, indptr)) 
314      >>> csr.todense() 
315      matrix([[8, 1, 2], 
316              [0, 0, 5], 
317              [0, 0, 0], 
318              [0, 0, 0]]) 
319      >>> sparsefuncs.inplace_column_scale(csr, scale) 
320      >>> csr.todense() 
321      matrix([[16,  3,  4], 
322              [ 0,  0, 10], 
323              [ 0,  0,  0], 
324              [ 0,  0,  0]]) 
325      """
326      if sp.issparse(X) and X.format == "csc":
327          inplace_csr_row_scale(X.T, scale)
328      elif sp.issparse(X) and X.format == "csr":
329          inplace_csr_column_scale(X, scale)
330      else:
331          _raise_typeerror(X)

378  def inplace_swap_row_csc(X, m, n):
379      """Swap two rows of a CSC matrix in-place. 
380   
381      Parameters 
382      ---------- 
383      X : sparse matrix of shape (n_samples, n_features) 
384          Matrix whose two rows are to be swapped. It should be of 
385          CSC format. 
386   
387      m : int 
388          Index of the row of X to be swapped. 
389   
390      n : int 
391          Index of the row of X to be swapped. 
392      """
393      for t in [m, n]:
394          if isinstance(t, np.ndarray):
395              raise TypeError("m and n should be valid integers")
396  
397      if m < 0:
398          m += X.shape[0]
399      if n < 0:
400          n += X.shape[0]
401  
402      m_mask = X.indices == m
403      X.indices[X.indices == n] = m
404      X.indices[m_mask] = n

407  def inplace_swap_row_csr(X, m, n):
408      """Swap two rows of a CSR matrix in-place. 
409   
410      Parameters 
411      ---------- 
412      X : sparse matrix of shape (n_samples, n_features) 
413          Matrix whose two rows are to be swapped. It should be of 
414          CSR format. 
415   
416      m : int 
417          Index of the row of X to be swapped. 
418   
419      n : int 
420          Index of the row of X to be swapped. 
421      """
422      for t in [m, n]:
423          if isinstance(t, np.ndarray):
424              raise TypeError("m and n should be valid integers")
425  
426      if m < 0:
427          m += X.shape[0]
428      if n < 0:
429          n += X.shape[0]
430  
431      # The following swapping makes life easier since m is assumed to be the
432      # smaller integer below.
433      if m > n:
434          m, n = n, m
435  
436      indptr = X.indptr
437      m_start = indptr[m]
438      m_stop = indptr[m + 1]
439      n_start = indptr[n]
440      n_stop = indptr[n + 1]
441      nz_m = m_stop - m_start
442      nz_n = n_stop - n_start
443  
444      if nz_m != nz_n:
445          # Modify indptr first
446          X.indptr[m + 2 : n] += nz_n - nz_m
447          X.indptr[m + 1] = m_start + nz_n
448          X.indptr[n] = n_stop - nz_m
449  
450      X.indices = np.concatenate(
451          [
452              X.indices[:m_start],
453              X.indices[n_start:n_stop],
454              X.indices[m_stop:n_start],
455              X.indices[m_start:m_stop],
456              X.indices[n_stop:],
457          ]
458      )
459      X.data = np.concatenate(
460          [
461              X.data[:m_start],
462              X.data[n_start:n_stop],
463              X.data[m_stop:n_start],
464              X.data[m_start:m_stop],
465              X.data[n_stop:],
466          ]
467      )

600  def count_nonzero(X, axis=None, sample_weight=None):
601      """A variant of X.getnnz() with extension to weighting on axis 0. 
602   
603      Useful in efficiently calculating multilabel metrics. 
604   
605      Parameters 
606      ---------- 
607      X : sparse matrix of shape (n_samples, n_labels) 
608          Input data. It should be of CSR format. 
609   
610      axis : {0, 1}, default=None 
611          The axis on which the data is aggregated. 
612   
613      sample_weight : array-like of shape (n_samples,), default=None 
614          Weight for each row of X. 
615   
616      Returns 
617      ------- 
618      nnz : int, float, ndarray of shape (n_samples,) or ndarray of shape (n_features,) 
619          Number of non-zero values in the array along a given axis. Otherwise, 
620          the total number of non-zero values in the array is returned. 
621      """
622      if axis == -1:
623          axis = 1
624      elif axis == -2:
625          axis = 0
626      elif X.format != "csr":
627          raise TypeError("Expected CSR sparse format, got {0}".format(X.format))
628  
629      # We rely here on the fact that np.diff(Y.indptr) for a CSR
630      # will return the number of nonzero entries in each row.
631      # A bincount over Y.indices will return the number of nonzeros
632      # in each column. See ``csr_matrix.getnnz`` in scipy >= 0.14.
633      if axis is None:
634          if sample_weight is None:
635              return X.nnz
636          else:
637              return np.dot(np.diff(X.indptr), sample_weight)
638      elif axis == 1:
639          out = np.diff(X.indptr)
640          if sample_weight is None:
641              # astype here is for consistency with axis=0 dtype
642              return out.astype("intp")
643          return out * sample_weight
644      elif axis == 0:
645          if sample_weight is None:
646              return np.bincount(X.indices, minlength=X.shape[1])
647          else:
648              weights = np.repeat(sample_weight, np.diff(X.indptr))
649              return np.bincount(X.indices, minlength=X.shape[1], weights=weights)
650      else:
651          raise ValueError("Unsupported axis: {0}".format(axis))

654  def _get_median(data, n_zeros):
655      """Compute the median of data with n_zeros additional zeros. 
656   
657      This function is used to support sparse matrices; it modifies data 
658      in-place. 
659      """
660      n_elems = len(data) + n_zeros
661      if not n_elems:
662          return np.nan
663      n_negative = np.count_nonzero(data < 0)
664      middle, is_odd = divmod(n_elems, 2)
665      data.sort()
666  
667      if is_odd:
668          return _get_elem_at_rank(middle, data, n_negative, n_zeros)
669  
670      return (
671          _get_elem_at_rank(middle - 1, data, n_negative, n_zeros)
672          + _get_elem_at_rank(middle, data, n_negative, n_zeros)
673      ) / 2.0
674  
675  
676  def _get_elem_at_rank(rank, data, n_negative, n_zeros):
677      """Find the value in data augmented with n_zeros for the given rank"""
678      if rank < n_negative:
679          return data[rank]
680      if rank - n_negative < n_zeros:
681          return 0
682      return data[rank - n_zeros]

685  def csc_median_axis_0(X):
686      """Find the median across axis 0 of a CSC matrix. 
687   
688      It is equivalent to doing np.median(X, axis=0). 
689   
690      Parameters 
691      ---------- 
692      X : sparse matrix of shape (n_samples, n_features) 
693          Input data. It should be of CSC format. 
694   
695      Returns 
696      ------- 
697      median : ndarray of shape (n_features,) 
698          Median. 
699      """
700      if not (sp.issparse(X) and X.format == "csc"):
701          raise TypeError("Expected matrix of CSC format, got %s" % X.format)
702  
703      indptr = X.indptr
704      n_samples, n_features = X.shape
705      median = np.zeros(n_features)
706  
707      for f_ind, (start, end) in enumerate(zip(indptr[:-1], indptr[1:])):
708          # Prevent modifying X in place
709          data = np.copy(X.data[start:end])
710          nz = n_samples - data.size
711          median[f_ind] = _get_median(data, nz)
712  
713      return median

716  def _implicit_column_offset(X, offset):
717      """Create an implicitly offset linear operator. 
718   
719      This is used by PCA on sparse data to avoid densifying the whole data 
720      matrix. 
721   
722      Params 
723      ------ 
724          X : sparse matrix of shape (n_samples, n_features) 
725          offset : ndarray of shape (n_features,) 
726   
727      Returns 
728      ------- 
729      centered : LinearOperator 
730      """
731      offset = offset[None, :]
732      XT = X.T
733      return LinearOperator(
734          matvec=lambda x: X @ x - offset @ x,
735          matmat=lambda x: X @ x - offset @ x,
736          rmatvec=lambda x: XT @ x - (offset * x.sum()),
737          rmatmat=lambda x: XT @ x - offset.T @ x.sum(axis=0)[None, :],
738          dtype=X.dtype,
739          shape=X.shape,
740      )
