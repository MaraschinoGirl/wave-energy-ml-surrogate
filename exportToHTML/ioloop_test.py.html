260          """When a file object is used instead of a numeric file descriptor, 
261          the object should be closed (by IOLoop.close(all_fds=True), 
262          not just the fd. 
263          """
264  
265          # Use a socket since they are supported by IOLoop on all platforms.

282          io_loop = IOLoop()
283          io_loop.add_handler(socket_wrapper, lambda fd, events: None, IOLoop.READ)
284          io_loop.close(all_fds=True)
285          self.assertTrue(socket_wrapper.closed)

457          self.io_loop = IOLoop(make_current=False)
458          # The new IOLoop is not initially made current.
459          self.assertIsNone(IOLoop.current(instance=False))

471              self.assertIs(self.current_io_loop, self.io_loop)

473              self.assertIsNone(IOLoop.current(instance=False))

631      def simulate_calls(self, pc, durations):
632          """Simulate a series of calls to the PeriodicCallback. 
633   
634          Pass a list of call durations in seconds (negative values 
635          work to simulate clock adjustments during the call, or more or 
636          less equivalently, between calls). This method returns the 
637          times at which each call would be made. 
638          """
639          calls = []
640          now = 1000
641          pc._next_timeout = now
642          for d in durations:
643              pc._update_next(now)
644              calls.append(pc._next_timeout)
645              now = pc._next_timeout + d
646          return calls

661          expected = [
662              1010,
663              1020,
664              1030,  # first 3 calls on schedule
665              1050,
666              1070,  # next 2 delayed one cycle
667              1100,
668              1130,  # next 2 delayed 2 cycles
669              1170,
670              1210,  # next 2 delayed 3 cycles
671              1220,
672              1230,  # then back on schedule.
673          ]
674  
675          pc = PeriodicCallback(self.dummy, 10000)
676          self.assertEqual(self.simulate_calls(pc, call_durations), expected)
677  
678      def test_clock_backwards(self):
679          pc = PeriodicCallback(self.dummy, 10000)
680          # Backwards jumps are ignored, potentially resulting in a
681          # slightly slow schedule (although we assume that when
682          # time.time() and time.monotonic() are different, time.time()
683          # is getting adjusted by NTP and is therefore more accurate)
684          self.assertEqual(
685              self.simulate_calls(pc, [-2, -1, -3, -2, 0]), [1010, 1020, 1030, 1040, 1050]
686          )
687  
688          # For big jumps, we should perhaps alter the schedule, but we
689          # don't currently. This trace shows that we run callbacks
690          # every 10s of time.time(), but the first and second calls are
691          # 110s of real time apart because the backwards jump is
692          # ignored.
693          self.assertEqual(self.simulate_calls(pc, [-100, 0, 0]), [1010, 1020, 1030])

695      def test_jitter(self):

700  
701          def mock_random():
702              return random_times.pop(0)
703  
704          with mock.patch("random.random", mock_random):
705              self.assertEqual(self.simulate_calls(pc, call_durations), expected)
