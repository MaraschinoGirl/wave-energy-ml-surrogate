14       def test__should_pretty_print(self):
15           # Test the rules about when a tag should be pretty-printed.
16           tag = self.soup("").new_tag("a_tag")
17   
18           # No list of whitespace-preserving tags -> pretty-print
19           tag._preserve_whitespace_tags = None
20           assert True is tag._should_pretty_print(0)
21   
22           # List exists but tag is not on the list -> pretty-print
23           tag.preserve_whitespace_tags = ["some_other_tag"]
24           assert True is tag._should_pretty_print(1)
25   
26           # Indent level is None -> don't pretty-print
27           assert False is tag._should_pretty_print(None)
28   
29           # Tag is on the whitespace-preserving list -> don't pretty-print
30           tag.preserve_whitespace_tags = ["some_other_tag", "a_tag"]
31           assert False is tag._should_pretty_print(1)

33       def test_len(self):
34           """The length of a Tag is its number of children."""
35           soup = self.soup("<top>1<b>2</b>3</top>")
36   
37           # The BeautifulSoup object itself contains one element: the
38           # <top> tag.
39           assert len(soup.contents) == 1
40           assert len(soup) == 1
41   
42           # The <top> tag contains three elements: the text node "1", the
43           # <b> tag, and the text node "3".
44           assert len(soup.top) == 3
45           assert len(soup.top.contents) == 3

144      def test_string_methods_inside_special_string_container_tags(self):
145          # Strings inside tags like <script> are generally ignored by
146          # methods like get_text, because they're not what humans
147          # consider 'text'. But if you call get_text on the <script>
148          # tag itself, those strings _are_ considered to be 'text',
149          # because there's nothing else you might be looking for.
150  
151          style = self.soup("<div>a<style>Some CSS</style></div>")
152          template = self.soup(
153              "<div>a<template><p>Templated <b>text</b>.</p><!--With a comment.--></template></div>"
154          )
155          script = self.soup("<div>a<script><!--a comment-->Some text</script></div>")
156  
157          assert style.div.get_text() == "a"
158          assert list(style.div.strings) == ["a"]
159          assert style.div.style.get_text() == "Some CSS"
160          assert list(style.div.style.strings) == ["Some CSS"]
161  
162          # The comment is not picked up here. That's because it was
163          # parsed into a Comment object, which is not considered
164          # interesting by template.strings.
165          assert template.div.get_text() == "a"
166          assert list(template.div.strings) == ["a"]
167          assert template.div.template.get_text() == "Templated text."
168          assert list(template.div.template.strings) == ["Templated ", "text", "."]
169  
170          # The comment is included here, because it didn't get parsed
171          # into a Comment object--it's part of the Script string.
172          assert script.div.get_text() == "a"
173          assert list(script.div.strings) == ["a"]
174          assert script.div.script.get_text() == "<!--a comment-->Some text"
175          assert list(script.div.script.strings) == ["<!--a comment-->Some text"]

178  class TestMultiValuedAttributes(SoupTest):
179      """Test the behavior of multi-valued attributes like 'class'. 
180   
181      The values of such attributes are always presented as lists. 
182      """

184      def test_single_value_becomes_list(self):
185          soup = self.soup("<a class='foo'>")
186          assert ["foo"] == soup.a["class"]
187  
188      def test_multiple_values_becomes_list(self):
189          soup = self.soup("<a class='foo bar'>")
190          assert ["foo", "bar"] == soup.a["class"]

192      def test_multiple_values_separated_by_weird_whitespace(self):
193          soup = self.soup("<a class='foo\tbar\nbaz'>")
194          assert ["foo", "bar", "baz"] == soup.a["class"]

218          # It's possible to change which attributes of which tags

230      def test_hidden_tag_is_invisible(self):
231          # Setting .hidden on a tag makes it invisible in output, but
232          # leaves its contents visible.
233          #
234          # This is not a documented or supported feature of Beautiful
235          # Soup (e.g. NavigableString doesn't support .hidden even
236          # though it could), but some people use it and it's not
237          # hurting anything to verify that it keeps working.
238          #
239          soup = self.soup('<div id="1"><span id="2">a string</span></div>')
240          soup.span.hidden = True
