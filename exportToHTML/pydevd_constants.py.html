391  def protect_libraries_from_patching():
392      """ 
393      In this function we delete some modules from `sys.modules` dictionary and import them again inside 
394        `_pydev_saved_modules` in order to save their original copies there. After that we can use these 
395        saved modules within the debugger to protect them from patching by external libraries (e.g. gevent). 
396      """
397      patched = [
398          "threading",
399          "thread",
400          "_thread",
401          "time",
402          "socket",
403          "queue",
404          "select",
405          "xmlrpclib",
406          "SimpleXMLRPCServer",
407          "BaseHTTPServer",
408          "SocketServer",
409          "xmlrpc.client",
410          "xmlrpc.server",
411          "http.server",
412          "socketserver",
413      ]
414  
415      for name in patched:
416          try:
417              __import__(name)
418          except:
419              pass
420  
421      patched_modules = dict([(k, v) for k, v in sys.modules.items() if k in patched])
422  
423      for name in patched_modules:
424          del sys.modules[name]
425  
426      # import for side effects
427      import _pydev_bundle._pydev_saved_modules
428  
429      for name in patched_modules:
430          sys.modules[name] = patched_modules[name]

450      class ForkSafeLock(object):
451          """ 
452          A lock which is fork-safe (when a fork is done, `pydevd_constants.after_fork()` 
453          should be called to reset the locks in the new process to avoid deadlocks 
454          from a lock which was locked during the fork). 
455   
456          Note: 
457              Unlike `threading.Lock` this class is not completely atomic, so, doing: 
458   
459              lock = ForkSafeLock() 
460              with lock: 
461                  ... 
462   
463              is different than using `threading.Lock` directly because the tracing may 
464              find an additional function call on `__enter__` and on `__exit__`, so, it's 
465              not recommended to use this in all places, only where the forking may be important 
466              (so, for instance, the locks on PyDB should not be changed to this lock because 
467              of that -- and those should all be collected in the new process because PyDB itself 
468              should be completely cleared anyways). 
469   
470              It's possible to overcome this limitation by using `ForkSafeLock.acquire` and 
471              `ForkSafeLock.release` instead of the context manager (as acquire/release are 
472              bound to the original implementation, whereas __enter__/__exit__ is not due to Python 
473              limitations). 
474          """
475  
476          def __init__(self, rlock=False):
477              self._rlock = rlock
478              self._init()
479              _fork_safe_locks.append(weakref.ref(self))
480  
481          def __enter__(self):
482              return self._lock.__enter__()
483  
484          def __exit__(self, exc_type, exc_val, exc_tb):
485              return self._lock.__exit__(exc_type, exc_val, exc_tb)
486  
487          def _init(self):
488              if self._rlock:
489                  self._lock = threading.RLock()
490              else:
491                  self._lock = thread.allocate_lock()
492  
493              self.acquire = self._lock.acquire
494              self.release = self._lock.release
495              _fork_safe_locks.append(weakref.ref(self))
496

498  def after_fork():
499      """ 
500      Must be called after a fork operation (will reset the ForkSafeLock). 
501      """
502      global _fork_safe_locks
503      locks = _fork_safe_locks[:]
504      _fork_safe_locks = []
505      for lock in locks:
506          lock = lock()
507          if lock is not None:
508              lock._init()

541  def iter_chars(b):
542      # In Python 2, we can iterate bytes or str with individual characters, but Python 3 onwards
543      # changed that behavior so that when iterating bytes we actually get ints!
544      if isinstance(b, bytes):
545          # i.e.: do something as struct.unpack('3c', b)
546          return iter(struct.unpack(str(len(b)) + "c", b))
547      return iter(b)

614  def clear_cached_thread_id(thread):
615      with _thread_id_lock:
616          try:
617              if thread.__pydevd_id__ != "console_main":
618                  # The console_main is a special thread id used in the console and its id should never be reset
619                  # (otherwise we may no longer be able to get its variables -- see: https://www.brainwy.com/tracker/PyDev/776).
620                  del thread.__pydevd_id__
621          except AttributeError:
622              pass
623

625  # Don't let threads be collected (so that id(thread) is guaranteed to be unique).
626  _thread_id_to_thread_found = {}
627  
628  
629  def _get_or_compute_thread_id_with_lock(thread, is_current_thread):
630      with _thread_id_lock:
631          # We do a new check with the lock in place just to be sure that nothing changed
632          tid = getattr(thread, "__pydevd_id__", None)
633          if tid is not None:
634              return tid
635  
636          _thread_id_to_thread_found[id(thread)] = thread
637  
638          # Note: don't use thread.ident because a new thread may have the
639          # same id from an old thread.
640          pid = get_pid()
641          tid = "pid_%s_id_%s" % (pid, id(thread))
642  
643          thread.__pydevd_id__ = tid
644  
645      return tid

648  def get_current_thread_id(thread):
649      """ 
650      Note: the difference from get_current_thread_id to get_thread_id is that 
651      for the current thread we can get the thread id while the thread.ident 
652      is still not set in the Thread instance. 
653      """
654      try:
655          # Fast path without getting lock.
656          tid = thread.__pydevd_id__
657          if tid is None:
658              # Fix for https://www.brainwy.com/tracker/PyDev/645
659              # if __pydevd_id__ is None, recalculate it... also, use an heuristic
660              # that gives us always the same id for the thread (using thread.ident or id(thread)).
661              raise AttributeError()
662      except AttributeError:
663          tid = _get_or_compute_thread_id_with_lock(thread, is_current_thread=True)
664  
665      return tid

761  def call_only_once(func):
762      """ 
763      To be used as a decorator 
764   
765      @call_only_once 
766      def func(): 
767          print 'Calling func only this time' 
768   
769      Actually, in PyDev it must be called as: 
770   
771      func = call_only_once(func) to support older versions of Python. 
772      """
773  
774      def new_func(*args, **kwargs):
775          if not new_func._called:
776              new_func._called = True
777              return func(*args, **kwargs)

779      new_func._called = False
780      return new_func
