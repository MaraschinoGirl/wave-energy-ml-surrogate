10   from h5py import h5
11   
12   from .common import TestCase
13   
14   def fixnames():
15       cfg = h5.get_config()
16       cfg.complex_names = ('r','i')
17   
18   class TestH5(TestCase):
19   
20       def test_config(self):
21           cfg = h5.get_config()
22           self.assertIsInstance(cfg, h5.H5PYConfig)
23           cfg2 = h5.get_config()
24           self.assertIs(cfg, cfg2)
25   
26       def test_cnames_get(self):
27           cfg = h5.get_config()
28           self.assertEqual(cfg.complex_names, ('r','i'))
29   
30       def test_cnames_set(self):
31           self.addCleanup(fixnames)
32           cfg = h5.get_config()
33           cfg.complex_names = ('q','x')
34           self.assertEqual(cfg.complex_names, ('q','x'))
35   
36       def test_cnames_set_exc(self):
37           self.addCleanup(fixnames)
38           cfg = h5.get_config()
39           with self.assertRaises(TypeError):
40               cfg.complex_names = ('q','i','v')
41           self.assertEqual(cfg.complex_names, ('r','i'))
42   
43       def test_repr(self):
44           cfg = h5.get_config()
45           repr(cfg)
