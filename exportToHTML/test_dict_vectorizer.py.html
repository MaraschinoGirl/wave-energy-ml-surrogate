16   @pytest.mark.parametrize("sparse", (True, False))
17   @pytest.mark.parametrize("dtype", (int, np.float32, np.int16))
18   @pytest.mark.parametrize("sort", (True, False))
19   @pytest.mark.parametrize("iterable", (True, False))
20   def test_dictvectorizer(sparse, dtype, sort, iterable):
21       D = [{"foo": 1, "bar": 3}, {"bar": 4, "baz": 2}, {"bar": 1, "quux": 1, "quuux": 2}]
22   
23       v = DictVectorizer(sparse=sparse, dtype=dtype, sort=sort)
24       X = v.fit_transform(iter(D) if iterable else D)
25   
26       assert sp.issparse(X) == sparse
27       assert X.shape == (3, 5)
28       assert X.sum() == 14
29       assert v.inverse_transform(X) == D
30   
31       if sparse:
32           # CSR matrices can't be compared for equality
33           assert_array_equal(
34               X.toarray(), v.transform(iter(D) if iterable else D).toarray()
35           )
36       else:
37           assert_array_equal(X, v.transform(iter(D) if iterable else D))
38   
39       if sort:
40           assert v.feature_names_ == sorted(v.feature_names_)

43   def test_feature_selection():
44       # make two feature dicts with two useful features and a bunch of useless
45       # ones, in terms of chi2
46       d1 = dict([("useless%d" % i, 10) for i in range(20)], useful1=1, useful2=20)
47       d2 = dict([("useless%d" % i, 10) for i in range(20)], useful1=20, useful2=1)
48   
49       for indices in (True, False):
50           v = DictVectorizer().fit([d1, d2])
51           X = v.transform([d1, d2])
52           sel = SelectKBest(chi2, k=2).fit(X, [0, 1])
53   
54           v.restrict(sel.get_support(indices=indices), indices=indices)
55           assert_array_equal(v.get_feature_names_out(), ["useful1", "useful2"])

58   def test_one_of_k():
59       D_in = [
60           {"version": "1", "ham": 2},
61           {"version": "2", "spam": 0.3},
62           {"version=3": True, "spam": -1},
63       ]
64       v = DictVectorizer()
65       X = v.fit_transform(D_in)
66       assert X.shape == (3, 5)
67   
68       D_out = v.inverse_transform(X)
69       assert D_out[0] == {"version=1": 1, "ham": 2}
70   
71       names = v.get_feature_names_out()
72       assert "version=2" in names
73       assert "version" not in names

76   def test_iterable_value():
77       D_names = ["ham", "spam", "version=1", "version=2", "version=3"]
78       X_expected = [
79           [2.0, 0.0, 2.0, 1.0, 0.0],
80           [0.0, 0.3, 0.0, 1.0, 0.0],
81           [0.0, -1.0, 0.0, 0.0, 1.0],
82       ]
83       D_in = [
84           {"version": ["1", "2", "1"], "ham": 2},
85           {"version": "2", "spam": 0.3},
86           {"version=3": True, "spam": -1},
87       ]
88       v = DictVectorizer()
89       X = v.fit_transform(D_in)
90       X = X.toarray()
91       assert_array_equal(X, X_expected)
92   
93       D_out = v.inverse_transform(X)
94       assert D_out[0] == {"version=1": 2, "version=2": 1, "ham": 2}
95   
96       names = v.get_feature_names_out()
97   
98       assert_array_equal(names, D_names)

104          "Only iterables of string are supported."

130  def test_unseen_or_no_features():
131      D = [{"camelot": 0, "spamalot": 1}]
132      for sparse in [True, False]:
133          v = DictVectorizer(sparse=sparse).fit(D)
134  
135          X = v.transform({"push the pram a lot": 2})
136          if sparse:
137              X = X.toarray()
138          assert_array_equal(X, np.zeros((1, 2)))
139  
140          X = v.transform({})
141          if sparse:
142              X = X.toarray()
143          assert_array_equal(X, np.zeros((1, 2)))
144  
145          with pytest.raises(ValueError, match="empty"):
146              v.transform([])

149  def test_deterministic_vocabulary(global_random_seed):
150      # Generate equal dictionaries with different memory layouts
151      items = [("%03d" % i, i) for i in range(1000)]
152      rng = Random(global_random_seed)
153      d_sorted = dict(items)
154      rng.shuffle(items)
155      d_shuffled = dict(items)
156  
157      # check that the memory layout does not impact the resulting vocabulary
158      v_1 = DictVectorizer().fit([d_sorted])
159      v_2 = DictVectorizer().fit([d_shuffled])
160  
161      assert v_1.vocabulary_ == v_2.vocabulary_

173  def test_dictvectorizer_dense_sparse_equivalence():
174      """Check the equivalence between between sparse and dense DictVectorizer. 
175      Non-regression test for: 
176      https://github.com/scikit-learn/scikit-learn/issues/19978 
177      """
178      movie_entry_fit = [
179          {"category": ["thriller", "drama"], "year": 2003},
180          {"category": ["animation", "family"], "year": 2011},
181          {"year": 1974},
182      ]
183      movie_entry_transform = [{"category": ["thriller"], "unseen_feature": "3"}]
184      dense_vectorizer = DictVectorizer(sparse=False)
185      sparse_vectorizer = DictVectorizer(sparse=True)
186  
187      dense_vector_fit = dense_vectorizer.fit_transform(movie_entry_fit)
188      sparse_vector_fit = sparse_vectorizer.fit_transform(movie_entry_fit)
189  
190      assert not sp.issparse(dense_vector_fit)
191      assert sp.issparse(sparse_vector_fit)
192  
193      assert_allclose(dense_vector_fit, sparse_vector_fit.toarray())
194  
195      dense_vector_transform = dense_vectorizer.transform(movie_entry_transform)
196      sparse_vector_transform = sparse_vectorizer.transform(movie_entry_transform)
197  
198      assert not sp.issparse(dense_vector_transform)
199      assert sp.issparse(sparse_vector_transform)
200  
201      assert_allclose(dense_vector_transform, sparse_vector_transform.toarray())
202  
203      dense_inverse_transform = dense_vectorizer.inverse_transform(dense_vector_transform)
204      sparse_inverse_transform = sparse_vectorizer.inverse_transform(
205          sparse_vector_transform
206      )
207  
208      expected_inverse = [{"category=thriller": 1.0}]
209      assert dense_inverse_transform == expected_inverse
210      assert sparse_inverse_transform == expected_inverse

214      """Check that we raise an error when the value associated to a feature 
215      is not supported. 
216   
217      Non-regression test for: 
218      https://github.com/scikit-learn/scikit-learn/issues/19489

227      with pytest.raises(TypeError, match=err_msg):
228          vectorizer.fit_transform(X)

231  def test_dict_vectorizer_get_feature_names_out():
232      """Check that integer feature names are converted to strings in 
233      feature_names_out."""
234  
235      X = [{1: 2, 3: 4}, {2: 4}]
236      dv = DictVectorizer(sparse=False).fit(X)
237  
238      feature_names = dv.get_feature_names_out()
239      assert isinstance(feature_names, np.ndarray)
240      assert feature_names.dtype == object
241      assert_array_equal(feature_names, ["1", "2", "3"])
