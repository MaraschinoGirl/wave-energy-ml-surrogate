53       def test_gh_15650_mp(self, a, b, x, expected):
54           # See https://github.com/scipy/scipy/issues/15650
55           # b == 1, |a| < 0.25, 0 < x < 1
56           #
57           # This purpose of this test is to check the accuracy of results
58           # in the region that was impacted by gh-15650.
59           #
60           # Reference values computed with mpmath using the script:
61           #
62           # import itertools as it
63           # import numpy as np
64           #
65           # from mpmath import mp
66           #
67           # rng = np.random.default_rng(1234)
68           #
69           # cases = []
70           # for a, x in it.product(
71           #         np.random.uniform(-0.25, 0.25, size=6),
72           #         np.logspace(-5, -1, 4),
73           # ):
74           #     with mp.workdps(100):
75           #         cases.append((float(a), 1.0, float(x), float(mp.hyperu(a, 1.0, x))))
76           assert_allclose(sc.hyperu(a, b, x), expected, rtol=1e-13)

78       def test_gh_15650_sanity(self):
79           # The purpose of this test is to sanity check hyperu in the region that
80           # was impacted by gh-15650 by making sure there are no excessively large
81           # results, as were reported there.
82           a = np.linspace(-0.5, 0.5, 500)
83           x = np.linspace(1e-6, 1e-1, 500)
84           a, x = np.meshgrid(a, x)
85           results = sc.hyperu(a, 1.0, x)
86           assert np.all(np.abs(results) < 1e3)

214      # The "legacy edge cases" mentioned in the comments in the following
215      # tests refers to the behavior of hyp1f1(a, b, x) when b is a nonpositive
216      # integer.  In some subcases, the behavior of SciPy does not match that
217      # of Boost (1.81+), mpmath and Mathematica (via Wolfram Alpha online).
218      # If the handling of these edges cases is changed to agree with those
219      # libraries, these test will have to be updated.
220  
221      @pytest.mark.parametrize('b', [0, -1, -5])
222      def test_legacy_case1(self, b):
223          # Test results of hyp1f1(0, n, x) for n <= 0.
224          # This is a legacy edge case.
225          # Boost (versions greater than 1.80), Mathematica (via Wolfram Alpha
226          # online) and mpmath all return 1 in this case, but SciPy's hyp1f1
227          # returns inf.
228          assert_equal(sc.hyp1f1(0, b, [-1.5, 0, 1.5]), [np.inf, np.inf, np.inf])
