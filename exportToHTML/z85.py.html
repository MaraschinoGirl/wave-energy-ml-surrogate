z85.py 
1    """Python implementation of Z85 85-bit encoding 
2     
3    Z85 encoding is a plaintext encoding for a bytestring interpreted as 32bit integers. 
4    Since the chunks are 32bit, a bytestring must be a multiple of 4 bytes. 
5    See ZMQ RFC 32 for details. 
6     
7     
8    """
9    
10   # Copyright (C) PyZMQ Developers
11   # Distributed under the terms of the Modified BSD License.
12   from __future__ import annotations
13   
14   import struct
15   
16   # Z85CHARS is the base 85 symbol table
17   Z85CHARS = b"0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ.-:+=^!/*?&<>()[]{}@%$#"
18   # Z85MAP maps integers in [0,84] to the appropriate character in Z85CHARS
19   Z85MAP = {c: idx for idx, c in enumerate(Z85CHARS)}
20   
21   _85s = [85**i for i in range(5)][::-1]
22   
23   
24   def encode(rawbytes):
25       """encode raw bytes into Z85"""
26       # Accepts only byte arrays bounded to 4 bytes
27       if len(rawbytes) % 4:
28           raise ValueError(f"length must be multiple of 4, not {len(rawbytes)}")
29   
30       nvalues = len(rawbytes) // 4
31       values = struct.unpack(f'>{nvalues:d}I', rawbytes)
32       encoded = []
33       for v in values:
34           for offset in _85s:
35               encoded.append(Z85CHARS[(v // offset) % 85])
36   
37       return bytes(encoded)
38   
39   
40   def decode(z85bytes):
41       """decode Z85 bytes to raw bytes, accepts ASCII string"""
42       if isinstance(z85bytes, str):
43           try:
44               z85bytes = z85bytes.encode('ascii')
45           except UnicodeEncodeError:
46               raise ValueError('string argument should contain only ASCII characters')
47   
48       if len(z85bytes) % 5:
49           raise ValueError(f"Z85 length must be multiple of 5, not {len(z85bytes)}")
50   
51       nvalues = len(z85bytes) // 5
52       values = []
53       for i in range(0, len(z85bytes), 5):
54           value = 0
55           for j, offset in enumerate(_85s):
56               value += Z85MAP[z85bytes[i + j]] * offset
57           values.append(value)
58       return struct.pack(f'>{nvalues:d}I', *values)
