23       def __init__(self, arg1, shape=None, dtype=None, copy=False, *, maxprint=None):
24           _spbase.__init__(self, arg1, maxprint=maxprint)
25           self.dtype = getdtype(dtype, arg1, default=float)
26   
27           # First get the shape
28           if issparse(arg1):
29               if arg1.format == "lil" and copy:
30                   A = arg1.copy()
31               else:
32                   A = arg1.tolil()
33   
34               if dtype is not None:
35                   newdtype = getdtype(dtype)
36                   A = A.astype(newdtype, copy=False)
37   
38               self._shape = check_shape(A.shape)
39               self.dtype = A.dtype
40               self.rows = A.rows
41               self.data = A.data
42           elif isinstance(arg1,tuple):
43               if isshape(arg1):
44                   if shape is not None:
45                       raise ValueError('invalid use of shape parameter')
46                   M, N = arg1
47                   self._shape = check_shape((M, N))
48                   self.rows = np.empty((M,), dtype=object)
49                   self.data = np.empty((M,), dtype=object)
50                   for i in range(M):
51                       self.rows[i] = []
52                       self.data[i] = []
53               else:
54                   raise TypeError('unrecognized lil_array constructor usage')
55           else:
56               # assume A is dense
57               try:
58                   A = self._ascontainer(arg1)
59               except TypeError as e:
60                   raise TypeError('unsupported matrix type') from e
61               if isinstance(self, sparray) and A.ndim != 2:
62                   raise ValueError(f"LIL arrays don't support {A.ndim}D input. Use 2D")
63               A = self._csr_container(A, dtype=dtype).tolil()
64   
65               self._shape = check_shape(A.shape)
66               self.dtype = getdtype(A.dtype)
67               self.rows = A.rows
68               self.data = A.data

95       def _getnnz(self, axis=None):
96           if axis is None:
97               return sum([len(rowvals) for rowvals in self.data])
98           if axis < 0:
99               axis += 2
100          if axis == 0:
101              out = np.zeros(self.shape[1], dtype=np.intp)
102              for row in self.rows:
103                  out[row] += 1
104              return out
105          elif axis == 1:
106              return np.array([len(rowvals) for rowvals in self.data], dtype=np.intp)
107          else:
108              raise ValueError('axis out of bounds')

112      def count_nonzero(self, axis=None):
113          if axis is None:
114              return sum(np.count_nonzero(rowvals) for rowvals in self.data)
115  
116          if axis < 0:
117              axis += 2
118          if axis == 0:
119              out = np.zeros(self.shape[1], dtype=np.intp)
120              for row, data in zip(self.rows, self.data):
121                  mask = [c for c, d in zip(row, data) if d != 0]
122                  out[mask] += 1
123              return out
124          elif axis == 1:
125              return np.array(
126                  [np.count_nonzero(rowvals) for rowvals in self.data], dtype=np.intp,
127              )
128          else:
129              raise ValueError('axis out of bounds')

141      def getrow(self, i):
142          """Returns a copy of the 'i'th row. 
143          """
144          M, N = self.shape
145          if i < 0:
146              i += M
147          if i < 0 or i >= M:
148              raise IndexError('row index out of bounds')
149          new = self._lil_container((1, N), dtype=self.dtype)
150          new.rows[0] = self.rows[i][:]
151          new.data[0] = self.data[i][:]
152          return new

212      def _get_arrayXarray(self, row, col):
213          # inner indexing
214          i, j = map(np.atleast_2d, _prepare_index_for_memoryview(row, col))
215          new = self._lil_container(i.shape, dtype=self.dtype)
216          _csparsetools.lil_fancy_get(self.shape[0], self.shape[1],
217                                      self.rows, self.data,
218                                      new.rows, new.data,
219                                      i, j)

222      def _get_row_ranges(self, rows, col_slice):
223          """ 
224          Fast path for indexing in the case where column index is slice. 
225   
226          This gains performance improvement over brute force by more 
227          efficient skipping of zeros, by accessing the elements 
228          column-wise in order. 
229   
230          Parameters 
231          ---------- 
232          rows : sequence or range 
233              Rows indexed. If range, must be within valid bounds. 
234          col_slice : slice 
235              Columns indexed 
236   
237          """
238          j_start, j_stop, j_stride = col_slice.indices(self.shape[1])
239          col_range = range(j_start, j_stop, j_stride)
240          nj = len(col_range)
241          new = self._lil_container((len(rows), nj), dtype=self.dtype)
242  
243          _csparsetools.lil_get_row_ranges(self.shape[0], self.shape[1],
244                                           self.rows, self.data,
245                                           new.rows, new.data,
246                                           rows,
247                                           j_start, j_stop, j_stride, nj)
248  
249          return new

267      def __setitem__(self, key, x):
268          if isinstance(key, tuple) and len(key) == 2:
269              row, col = key
270              # Fast path for simple (int, int) indexing.
271              if isinstance(row, INT_TYPES) and isinstance(col, INT_TYPES):
272                  x = self.dtype.type(x)
273                  if x.size > 1:
274                      raise ValueError("Trying to assign a sequence to an item")
275                  return self._set_intXint(row, col, x)
276              # Fast path for full-matrix sparse assignment.
277              if (isinstance(row, slice) and isinstance(col, slice) and
278                      row == slice(None) and col == slice(None) and
279                      issparse(x) and x.shape == self.shape):
280                  x = self._lil_container(x, dtype=self.dtype)
281                  self.rows = x.rows
282                  self.data = x.data
283                  return
284          # Everything else takes the normal path.
285          IndexMixin.__setitem__(self, key, x)

287      def _mul_scalar(self, other):
288          if other == 0:
289              # Multiply by zero: return the zero matrix
290              new = self._lil_container(self.shape, dtype=self.dtype)
291          else:
292              res_dtype = upcast_scalar(self.dtype, other)
293  
294              new = self.copy()
295              new = new.astype(res_dtype)
296              # Multiply this scalar by every element.
297              for j, rowvals in enumerate(new.data):
298                  new.data[j] = [val*other for val in rowvals]
299          return new
300  
301      def __truediv__(self, other):           # self / other
302          if isscalarlike(other):
303              new = self.copy()
304              new.dtype = np.result_type(self, other)
305              # Divide every element by this scalar
306              for j, rowvals in enumerate(new.data):
307                  new.data[j] = [val/other for val in rowvals]
308              return new
309          else:
310              return self.tocsr() / other

312      def copy(self):
313          M, N = self.shape
314          new = self._lil_container(self.shape, dtype=self.dtype)
315          # This is ~14x faster than calling deepcopy() on rows and data.
316          _csparsetools.lil_get_row_ranges(M, N, self.rows, self.data,
317                                           new.rows, new.data, range(M),
318                                           0, N, 1, N)
319          return new

323      def reshape(self, *args, **kwargs):
324          shape = check_shape(args, self.shape)
325          order, copy = check_reshape_kwargs(kwargs)
326  
327          # Return early if reshape is not required
328          if shape == self.shape:
329              if copy:
330                  return self.copy()
331              else:
332                  return self
333  
334          new = self._lil_container(shape, dtype=self.dtype)
335  
336          if order == 'C':
337              ncols = self.shape[1]
338              for i, row in enumerate(self.rows):
339                  for col, j in enumerate(row):
340                      new_r, new_c = np.unravel_index(i * ncols + j, shape)
341                      new[new_r, new_c] = self[i, j]
342          elif order == 'F':
343              nrows = self.shape[0]
344              for i, row in enumerate(self.rows):
345                  for col, j in enumerate(row):
346                      new_r, new_c = np.unravel_index(i + j * nrows, shape, order)
347                      new[new_r, new_c] = self[i, j]
348          else:
349              raise ValueError("'order' must be 'C' or 'F'")
350  
351          return new

355      def resize(self, *shape):
356          shape = check_shape(shape)
357          new_M, new_N = shape
358          M, N = self.shape
359  
360          if new_M < M:
361              self.rows = self.rows[:new_M]
362              self.data = self.data[:new_M]
363          elif new_M > M:
364              self.rows = np.resize(self.rows, new_M)
365              self.data = np.resize(self.data, new_M)
366              for i in range(M, new_M):
367                  self.rows[i] = []
368                  self.data[i] = []
369  
370          if new_N < N:
371              for row, data in zip(self.rows, self.data):
372                  trunc = bisect_left(row, new_N)
373                  del row[trunc:]
374                  del data[trunc:]
375  
376          self._shape = shape

402      def tocsr(self, copy=False):
403          M, N = self.shape
404          if M == 0 or N == 0:
405              return self._csr_container((M, N), dtype=self.dtype)
406  
407          # construct indptr array
408          if M*N <= np.iinfo(np.int32).max:
409              # fast path: it is known that 64-bit indexing will not be needed.
410              idx_dtype = np.int32
411              indptr = np.empty(M + 1, dtype=idx_dtype)
412              indptr[0] = 0
413              _csparsetools.lil_get_lengths(self.rows, indptr[1:])
414              np.cumsum(indptr, out=indptr)
415              nnz = indptr[-1]
416          else:
417              idx_dtype = self._get_index_dtype(maxval=N)
418              lengths = np.empty(M, dtype=idx_dtype)
419              _csparsetools.lil_get_lengths(self.rows, lengths)
420              nnz = lengths.sum(dtype=np.int64)
421              idx_dtype = self._get_index_dtype(maxval=max(N, nnz))
422              indptr = np.empty(M + 1, dtype=idx_dtype)
423              indptr[0] = 0
424              np.cumsum(lengths, dtype=idx_dtype, out=indptr[1:])
425  
426          indices = np.empty(nnz, dtype=idx_dtype)
427          data = np.empty(nnz, dtype=self.dtype)
428          _csparsetools.lil_flatten_to_array(self.rows, indices)
429          _csparsetools.lil_flatten_to_array(self.data, data)
430  
431          # init csr matrix
432          return self._csr_container((data, indices, indptr), shape=self.shape)

437  def _prepare_index_for_memoryview(i, j, x=None):
438      """ 
439      Convert index and data arrays to form suitable for passing to the 
440      Cython fancy getset routines. 
441   
442      The conversions are necessary since to (i) ensure the integer 
443      index arrays are in one of the accepted types, and (ii) to ensure 
444      the arrays are writable so that Cython memoryview support doesn't 
445      choke on them. 
446   
447      Parameters 
448      ---------- 
449      i, j 
450          Index arrays 
451      x : optional 
452          Data arrays 
453   
454      Returns 
455      ------- 
456      i, j, x 
457          Re-formatted arrays (x is omitted, if input was None) 
458   
459      """
460      if i.dtype > j.dtype:
461          j = j.astype(i.dtype)
462      elif i.dtype < j.dtype:
463          i = i.astype(j.dtype)
464  
465      if not i.flags.writeable or i.dtype not in (np.int32, np.int64):
466          i = i.astype(np.intp)
467      if not j.flags.writeable or j.dtype not in (np.int32, np.int64):
468          j = j.astype(np.intp)
469  
470      if x is not None:
471          if not x.flags.writeable:
472              x = x.copy()
473          return i, j, x
474      else:
475          return i, j

504  # This namespace class separates array from matrix with isinstance
505  class lil_array(_lil_base, sparray):
506      """ 
507      Row-based LIst of Lists sparse array. 
508   
509      This is a structure for constructing sparse arrays incrementally. 
510      Note that inserting a single item can take linear time in the worst case; 
511      to construct the array efficiently, make sure the items are pre-sorted by 
512      index, per row. 
513   
514      This can be instantiated in several ways: 
515          lil_array(D) 
516              where D is a 2-D ndarray 
517   
518          lil_array(S) 
519              with another sparse array or matrix S (equivalent to S.tolil()) 
520   
521          lil_array((M, N), [dtype]) 
522              to construct an empty array with shape (M, N) 
523              dtype is optional, defaulting to dtype='d'. 
524   
525      Attributes 
526      ---------- 
527      dtype : dtype 
528          Data type of the array 
529      shape : 2-tuple 
530          Shape of the array 
531      ndim : int 
532          Number of dimensions (this is always 2) 
533      nnz 
534      size 
535      data 
536          LIL format data array of the array 
537      rows 
538          LIL format row index array of the array 
539      T 
540   
541      Notes 
542      ----- 
543      Sparse arrays can be used in arithmetic operations: they support 
544      addition, subtraction, multiplication, division, and matrix power. 
545   
546      Advantages of the LIL format 
547          - supports flexible slicing 
548          - changes to the array sparsity structure are efficient 
549   
550      Disadvantages of the LIL format 
551          - arithmetic operations LIL + LIL are slow (consider CSR or CSC) 
552          - slow column slicing (consider CSC) 
553          - slow matrix vector products (consider CSR or CSC) 
554   
555      Intended Usage 
556          - LIL is a convenient format for constructing sparse arrays 
557          - once an array has been constructed, convert to CSR or 
558            CSC format for fast arithmetic and matrix vector operations 
559          - consider using the COO format when constructing large arrays 
560   
561      Data Structure 
562          - An array (``self.rows``) of rows, each of which is a sorted 
563            list of column indices of non-zero elements. 
564          - The corresponding nonzero values are stored in similar 
565            fashion in ``self.data``. 
566   
567      """
