11   typedef int (_RunCodeInMemoryInAttachedDll)();
12   
13   HINSTANCE globalDllInstance = NULL;
14   
15   class NotificationHelper {
16   public:
17   #pragma warning( push )
18   #pragma warning( disable : 4722 )
19   // disable c4722 here: Destructor never returns warning. Compiler sees ExitThread and assumes that
20   // there is a potential memory leak.
21       ~NotificationHelper(){
22           std::string eventName("_pydevd_pid_event_");
23           eventName += std::to_string(GetCurrentProcessId());
24   
25           // When we finish we need to set the event that the caller is waiting for and
26           // unload the dll (if we don't exit this dll we won't be able to reattach later on).
27           auto event = CreateEventA(nullptr, false, false, eventName.c_str());
28           if (event != nullptr) {
29               SetEvent(event);
30               CloseHandle(event);
31           }
32           FreeLibraryAndExitThread(globalDllInstance, 0);
33       }
34   #pragma warning( pop )
35   };
36   
37   DWORD WINAPI RunCodeInThread(LPVOID lpParam){
38       NotificationHelper notificationHelper; // When we exit the scope the destructor should take care of the cleanup.
39       
40   #ifdef BITS_32
41       HMODULE attachModule = GetModuleHandleA("attach_x86.dll");
42   #else
43       HMODULE attachModule = GetModuleHandleA("attach_amd64.dll");
44   #endif
45   
46       if (attachModule == nullptr) {
47           std::cout << "Error: unable to get attach_x86.dll or attach_amd64.dll module handle." << std::endl;
48           return 900;
49       }
50   
51       _RunCodeInMemoryInAttachedDll* runCode = reinterpret_cast < _RunCodeInMemoryInAttachedDll* > (GetProcAddress(attachModule, "RunCodeInMemoryInAttachedDll"));
52       if (runCode == nullptr) {
53           std::cout << "Error: unable to GetProcAddress(attachModule, RunCodeInMemoryInAttachedDll) from attach_x86.dll or attach_amd64.dll." << std::endl;
54           return 901;
55       }
56   
57       runCode();
58       return 0;
59   }

61   /** 
62    * When the dll is loaded we create a thread that will call 'RunCodeInMemoryInAttachedDll' 
63    * in the attach dll (when completed we unload this library for a reattach to work later on). 
64    */
65   BOOL WINAPI DllMain(
66     _In_ HINSTANCE hinstDLL,
67     _In_ DWORD     fdwReason,
68     _In_ LPVOID    lpvReserved
69   ){
70       if(fdwReason == DLL_PROCESS_ATTACH){
71           globalDllInstance = hinstDLL;
72           DWORD threadId;
73           CreateThread(nullptr, 0, &RunCodeInThread, nullptr, 0, &threadId);
74       }
75       else if(fdwReason == DLL_PROCESS_DETACH){
76       }
77       return true;
78   }
