48   def make_error(function: _NamedFuncPointer) -> OSError:
49       code = get_last_error()
50       exception = OSError()
51       exception.winerror = code
52       exception.function = function.__name__
53       exception.strerror = FormatError(code).strip()
54       return exception
55   
56   
57   def check_null(result: int | None, function: _NamedFuncPointer, *_) -> int:
58       if result is None:
59           raise make_error(function)
60       return result
61   
62   
63   def check_false(result: int | None, function: _NamedFuncPointer, *_) -> Literal[True]:
64       if not bool(result):
65           raise make_error(function)
66       else:
67           return True

98   def BeginUpdateResource(filename: str, delete: bool):
99       """Get a handle that can be used by the :func:`UpdateResource`. 
100   
101      Parameters 
102      ---------- 
103      fileName : str 
104          The filename of the module to load. 
105      delete : bool 
106          When true all existing resources are deleted 
107   
108      Returns 
109      ------- 
110      result : hModule 
111          Handle of the resource. 
112   
113      """
114      return _BeginUpdateResource(filename, delete)
115  
116  
117  def EndUpdateResource(handle: int, discard: bool) -> None:
118      """End the update resource of the handle. 
119   
120      Parameters 
121      ---------- 
122      handle : hModule 
123          The handle of the resource as it is returned 
124          by :func:`BeginUpdateResource` 
125   
126      discard : bool 
127          When True all writes are discarded. 
128   
129      """
130      _EndUpdateResource(handle, discard)

133  def UpdateResource(
134      handle: int,
135      type: str | int,
136      name: str | int,
137      data: Iterable[SupportsIndex] | SupportsIndex | SupportsBytes | ReadableBuffer,
138      language: int = LANG_NEUTRAL,
139  ) -> None:
140      """Update a resource. 
141   
142      Parameters 
143      ---------- 
144      handle : hModule 
145          The handle of the resource file as returned by 
146          :func:`BeginUpdateResource`. 
147   
148      type : str | int 
149          The type of resource to update. 
150   
151      name : str | int 
152          The name or Id of the resource to update. 
153   
154      data : bytes-like 
155          A bytes like object is expected. 
156   
157      language : int 
158          Language to use, default is LANG_NEUTRAL. 
159   
160      """
161      lp_data = bytes(data)
162      _UpdateResource(
163          handle, LPCWSTR(type), LPCWSTR(name), language, lp_data, len(lp_data)
164      )
