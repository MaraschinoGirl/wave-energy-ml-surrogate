3    from tensorflow.keras.models import Sequential
4    from tensorflow.keras.layers import LSTM, Dense, Input, Dropout, BatchNormalization
5    from tensorflow.keras.optimizers import Adam
6    from tensorflow.keras.callbacks import EarlyStopping
7    from tensorflow.keras.losses import Huber
8    
9    def train_lstm_model(X, y, epochs=100, batch_size=32):
10       if len(X.shape) != 3:
11           raise ValueError(f"LSTM input must be 3D. Got shape: {X.shape}")
12   
13       model = Sequential([
14           Input(shape=(X.shape[1], X.shape[2])),
15           LSTM(128, return_sequences=True),
16           Dropout(0.3),
17           BatchNormalization(),
18           LSTM(64, return_sequences=False),
19           Dropout(0.3),
20           Dense(64, activation='relu'),
21           Dense(1)
22       ])
23   
24       model.compile(
25           optimizer=Adam(0.001),
26           loss=Huber(delta=1.0),   # Can switch back to 'mse' if needed
27           metrics=['mae']
28       )
29   
30       early_stop = EarlyStopping(
31           monitor='val_loss',
32           patience=15,
33           restore_best_weights=True,
34           verbose=1
35       )
36   
37       history = model.fit(
38           X, y,
39           validation_split=0.2,
40           epochs=epochs,
41           batch_size=batch_size,
42           callbacks=[early_stop],
43           verbose=1
44       )
45   
46       return model, history
47
