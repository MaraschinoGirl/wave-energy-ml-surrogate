22       def transpose(self, axes=None, copy=False):
23           if axes is not None and axes != (1, 0):
24               raise ValueError("Sparse arrays/matrices do not support "
25                                 "an 'axes' parameter because swapping "
26                                 "dimensions is the only logical permutation.")
27   
28           if self.ndim == 1:
29               return self.copy() if copy else self
30           M, N = self.shape
31           return self._csc_container((self.data, self.indices,
32                                       self.indptr), shape=(N, M), copy=copy)

36       def tolil(self, copy=False):
37           if self.ndim != 2:
38               raise ValueError("Cannot convert a 1d sparse array to lil format")
39           lil = self._lil_container(self.shape, dtype=self.dtype)
40   
41           self.sum_duplicates()
42           ptr,ind,dat = self.indptr,self.indices,self.data
43           rows, data = lil.rows, lil.data
44   
45           for n in range(self.shape[0]):
46               start = ptr[n]
47               end = ptr[n+1]
48               rows[n] = ind[start:end].tolist()
49               data[n] = dat[start:end].tolist()
50   
51           return lil

63       def tocsc(self, copy=False):
64           if self.ndim != 2:
65               raise ValueError("Cannot convert a 1d sparse array to csc format")
66           M, N = self.shape
67           idx_dtype = self._get_index_dtype((self.indptr, self.indices),
68                                       maxval=max(self.nnz, M))
69           indptr = np.empty(N + 1, dtype=idx_dtype)
70           indices = np.empty(self.nnz, dtype=idx_dtype)
71           data = np.empty(self.nnz, dtype=upcast(self.dtype))
72   
73           csr_tocsc(M, N,
74                     self.indptr.astype(idx_dtype),
75                     self.indices.astype(idx_dtype),
76                     self.data,
77                     indptr,
78                     indices,
79                     data)
80   
81           A = self._csc_container((data, indices, indptr), shape=self.shape)
82           A.has_sorted_indices = True
83           return A

87       def tobsr(self, blocksize=None, copy=True):
88           if self.ndim != 2:
89               raise ValueError("Cannot convert a 1d sparse array to bsr format")
90           if blocksize is None:
91               from ._spfuncs import estimate_blocksize
92               return self.tobsr(blocksize=estimate_blocksize(self))
93   
94           elif blocksize == (1,1):
95               arg1 = (self.data.reshape(-1,1,1),self.indices,self.indptr)
96               return self._bsr_container(arg1, shape=self.shape, copy=copy)
97   
98           else:
99               R,C = blocksize
100              M,N = self.shape
101  
102              if R < 1 or C < 1 or M % R != 0 or N % C != 0:
103                  raise ValueError(f'invalid blocksize {blocksize}')
104  
105              blks = csr_count_blocks(M,N,R,C,self.indptr,self.indices)
106  
107              idx_dtype = self._get_index_dtype((self.indptr, self.indices),
108                                          maxval=max(N//C, blks))
109              indptr = np.empty(M//R+1, dtype=idx_dtype)
110              indices = np.empty(blks, dtype=idx_dtype)
111              data = np.zeros((blks,R,C), dtype=self.dtype)
112  
113              csr_tobsr(M, N, R, C,
114                        self.indptr.astype(idx_dtype),
115                        self.indices.astype(idx_dtype),
116                        self.data,
117                        indptr, indices, data.ravel())
118  
119              return self._bsr_container(
120                  (data, indices, indptr), shape=self.shape
121              )

133      def __iter__(self):
134          if self.ndim == 1:
135              zero = self.dtype.type(0)
136              u = 0
137              for v, d in zip(self.indices, self.data):
138                  for _ in range(v - u):
139                      yield zero
140                  yield d
141                  u = v + 1
142              for _ in range(self.shape[0] - u):
143                  yield zero
144              return
145  
146          indptr = np.zeros(2, dtype=self.indptr.dtype)
147          # return 1d (sparray) or 2drow (spmatrix)
148          shape = self.shape[1:] if isinstance(self, sparray) else (1, self.shape[1])
149          i0 = 0
150          for i1 in self.indptr[1:]:
151              indptr[1] = i1 - i0
152              indices = self.indices[i0:i1]
153              data = self.data[i0:i1]
154              yield self.__class__((data, indices, indptr), shape=shape, copy=True)
155              i0 = i1

157      def _getrow(self, i):
158          """Returns a copy of row i of the matrix, as a (1 x n) 
159          CSR matrix (row vector). 
160          """
161          if self.ndim == 1:
162              if i not in (0, -1):
163                  raise IndexError(f'index ({i}) out of range')
164              return self.reshape((1, self.shape[0]), copy=True)
165  
166          M, N = self.shape
167          i = int(i)
168          if i < 0:
169              i += M
170          if i < 0 or i >= M:
171              raise IndexError('index (%d) out of range' % i)
172          indptr, indices, data = get_csr_submatrix(
173              M, N, self.indptr, self.indices, self.data, i, i + 1, 0, N)
174          return self.__class__((data, indices, indptr), shape=(1, N),
175                                dtype=self.dtype, copy=False)
176  
177      def _getcol(self, i):
178          """Returns a copy of column i. A (m x 1) sparse array (column vector). 
179          """
180          if self.ndim == 1:
181              raise ValueError("getcol not provided for 1d arrays. Use indexing A[j]")
182          M, N = self.shape
183          i = int(i)
184          if i < 0:
185              i += N
186          if i < 0 or i >= N:
187              raise IndexError('index (%d) out of range' % i)
188          indptr, indices, data = get_csr_submatrix(
189              M, N, self.indptr, self.indices, self.data, 0, M, i, i + 1)
190          return self.__class__((data, indices, indptr), shape=(M, 1),
191                                dtype=self.dtype, copy=False)

193      def _get_int(self, idx):
194          spot = np.flatnonzero(self.indices == idx)
195          if spot.size:
196              return self.data[spot[0]]
197          return self.data.dtype.type(0)

199      def _get_slice(self, idx):
200          if idx == slice(None):
201              return self.copy()
202          if idx.step in (1, None):
203              ret = self._get_submatrix(0, idx, copy=True)
204              return ret.reshape(ret.shape[-1])
205          return self._minor_slice(idx)
206  
207      def _get_array(self, idx):
208          idx_dtype = self._get_index_dtype(self.indices)
209          idx = np.asarray(idx, dtype=idx_dtype)
210          if idx.size == 0:
211              return self.__class__([], dtype=self.dtype)
212  
213          M, N = 1, self.shape[0]
214          row = np.zeros_like(idx, dtype=idx_dtype)
215          col = np.asarray(idx, dtype=idx_dtype)
216          val = np.empty(row.size, dtype=self.dtype)
217          csr_sample_values(M, N, self.indptr, self.indices, self.data,
218                            row.size, row, col, val)
219  
220          new_shape = col.shape if col.shape[0] > 1 else (col.shape[0],)
221          return self.__class__(val.reshape(new_shape))
222  
223      def _get_intXarray(self, row, col):
224          return self._getrow(row)._minor_index_fancy(col)
225  
226      def _get_intXslice(self, row, col):
227          if col.step in (1, None):
228              return self._get_submatrix(row, col, copy=True)
229          # TODO: uncomment this once it's faster:
230          # return self._getrow(row)._minor_slice(col)
231  
232          M, N = self.shape
233          start, stop, stride = col.indices(N)
234  
235          ii, jj = self.indptr[row:row+2]
236          row_indices = self.indices[ii:jj]
237          row_data = self.data[ii:jj]
238  
239          if stride > 0:
240              ind = (row_indices >= start) & (row_indices < stop)
241          else:
242              ind = (row_indices <= start) & (row_indices > stop)
243  
244          if abs(stride) > 1:
245              ind &= (row_indices - start) % stride == 0
246  
247          row_indices = (row_indices[ind] - start) // stride
248          row_data = row_data[ind]
249          row_indptr = np.array([0, len(row_indices)])
250  
251          if stride < 0:
252              row_data = row_data[::-1]
253              row_indices = abs(row_indices[::-1])
254  
255          shape = (1, max(0, int(np.ceil(float(stop - start) / stride))))
256          return self.__class__((row_data, row_indices, row_indptr), shape=shape,
257                                dtype=self.dtype, copy=False)
258  
259      def _get_sliceXint(self, row, col):
260          if row.step in (1, None):
261              return self._get_submatrix(row, col, copy=True)
262          return self._major_slice(row)._get_submatrix(minor=col)
263  
264      def _get_sliceXarray(self, row, col):
265          return self._major_slice(row)._minor_index_fancy(col)
266  
267      def _get_arrayXint(self, row, col):
268          res = self._major_index_fancy(row)._get_submatrix(minor=col)
269          if row.ndim > 1:
270              return res.reshape(row.shape)
271          return res
272  
273      def _get_arrayXslice(self, row, col):
274          if col.step not in (1, None):
275              col = np.arange(*col.indices(self.shape[1]))
276              return self._get_arrayXarray(row, col)
277          return self._major_index_fancy(row)._get_submatrix(minor=col)

283          x = np.broadcast_to(x, idx.shape)

289   
290      Parameters

314  class csr_array(_csr_base, sparray):
315      """ 
316      Compressed Sparse Row array. 
317   
318      This can be instantiated in several ways: 
319          csr_array(D) 
320              where D is a 2-D ndarray 
321   
322          csr_array(S) 
323              with another sparse array or matrix S (equivalent to S.tocsr()) 
324   
325          csr_array((M, N), [dtype]) 
326              to construct an empty array with shape (M, N) 
327              dtype is optional, defaulting to dtype='d'. 
328   
329          csr_array((data, (row_ind, col_ind)), [shape=(M, N)]) 
330              where ``data``, ``row_ind`` and ``col_ind`` satisfy the 
331              relationship ``a[row_ind[k], col_ind[k]] = data[k]``. 
332   
333          csr_array((data, indices, indptr), [shape=(M, N)]) 
334              is the standard CSR representation where the column indices for 
335              row i are stored in ``indices[indptr[i]:indptr[i+1]]`` and their 
336              corresponding values are stored in ``data[indptr[i]:indptr[i+1]]``. 
337              If the shape parameter is not supplied, the array dimensions 
338              are inferred from the index arrays. 
339   
340      Attributes 
341      ---------- 
342      dtype : dtype 
343          Data type of the array 
344      shape : 2-tuple 
345          Shape of the array 
346      ndim : int 
347          Number of dimensions (this is always 2) 
348      nnz 
349      size 
350      data 
351          CSR format data array of the array 
352      indices 
353          CSR format index array of the array 
354      indptr 
355          CSR format index pointer array of the array 
356      has_sorted_indices 
357      has_canonical_format 
358      T 
359   
360      Notes 
361      ----- 
362   
363      Sparse arrays can be used in arithmetic operations: they support 
364      addition, subtraction, multiplication, division, and matrix power. 
365   
366      Advantages of the CSR format 
367        - efficient arithmetic operations CSR + CSR, CSR * CSR, etc. 
368        - efficient row slicing 
369        - fast matrix vector products 
370   
371      Disadvantages of the CSR format 
372        - slow column slicing operations (consider CSC) 
373        - changes to the sparsity structure are expensive (consider LIL or DOK) 
374   
375      Canonical Format 
376          - Within each row, indices are sorted by column. 
377          - There are no duplicate entries. 
378   
379      Examples 
380      -------- 
381   
382      >>> import numpy as np 
383      >>> from scipy.sparse import csr_array 
384      >>> csr_array((3, 4), dtype=np.int8).toarray() 
385      array([[0, 0, 0, 0], 
386             [0, 0, 0, 0], 
387             [0, 0, 0, 0]], dtype=int8) 
388   
389      >>> row = np.array([0, 0, 1, 2, 2, 2]) 
390      >>> col = np.array([0, 2, 2, 0, 1, 2]) 
391      >>> data = np.array([1, 2, 3, 4, 5, 6]) 
392      >>> csr_array((data, (row, col)), shape=(3, 3)).toarray() 
393      array([[1, 0, 2], 
394             [0, 0, 3], 
395             [4, 5, 6]]) 
396   
397      >>> indptr = np.array([0, 2, 3, 6]) 
398      >>> indices = np.array([0, 2, 2, 0, 1, 2]) 
399      >>> data = np.array([1, 2, 3, 4, 5, 6]) 
400      >>> csr_array((data, indices, indptr), shape=(3, 3)).toarray() 
401      array([[1, 0, 2], 
402             [0, 0, 3], 
403             [4, 5, 6]]) 
404   
405      Duplicate entries are summed together: 
406   
407      >>> row = np.array([0, 1, 2, 0]) 
408      >>> col = np.array([0, 1, 1, 0]) 
409      >>> data = np.array([1, 2, 4, 8]) 
410      >>> csr_array((data, (row, col)), shape=(3, 3)).toarray() 
411      array([[9, 0, 0], 
412             [0, 2, 0], 
413             [0, 4, 0]]) 
414   
415      As an example of how to construct a CSR array incrementally, 
416      the following snippet builds a term-document array from texts: 
417   
418      >>> docs = [["hello", "world", "hello"], ["goodbye", "cruel", "world"]] 
419      >>> indptr = [0] 
420      >>> indices = [] 
421      >>> data = [] 
422      >>> vocabulary = {} 
423      >>> for d in docs: 
424      ...     for term in d: 
425      ...         index = vocabulary.setdefault(term, len(vocabulary)) 
426      ...         indices.append(index) 
427      ...         data.append(1) 
428      ...     indptr.append(len(indices)) 
429      ... 
430      >>> csr_array((data, indices, indptr), dtype=int).toarray() 
431      array([[2, 1, 0, 0], 
432             [0, 1, 1, 1]]) 
433   
434      """
