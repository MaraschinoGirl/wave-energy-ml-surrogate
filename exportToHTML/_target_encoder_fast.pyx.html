20   def _fit_encoding_fast(
21       INT_DTYPE[:, ::1] X_int,
22       const Y_DTYPE[:] y,
23       int64_t[::1] n_categories,
24       double smooth,
25       double y_mean,
26   ):
27       """Fit a target encoding on X_int and y. 
28    
29       This implementation uses Eq 7 from [1] to compute the encoding. 
30       As stated in the paper, Eq 7 is the same as Eq 3. 
31    
32       [1]: Micci-Barreca, Daniele. "A preprocessing scheme for high-cardinality 
33            categorical attributes in classification and prediction problems" 
34       """
35       cdef:
36           int64_t sample_idx, feat_idx, cat_idx, n_cats
37           INT_DTYPE X_int_tmp
38           int n_samples = X_int.shape[0]
39           int n_features = X_int.shape[1]
40           double smooth_sum = smooth * y_mean
41           int64_t max_n_cats = np.max(n_categories)
42           double[::1] sums = np.empty(max_n_cats, dtype=np.float64)
43           double[::1] counts = np.empty(max_n_cats, dtype=np.float64)
44           list encodings = []
45           double[::1] current_encoding
46           # Gives access to encodings without gil
47           vector[double*] encoding_vec
48   
49       encoding_vec.resize(n_features)
50       for feat_idx in range(n_features):
51           current_encoding = np.empty(shape=n_categories[feat_idx], dtype=np.float64)
52           encoding_vec[feat_idx] = &current_encoding[0]
53           encodings.append(np.asarray(current_encoding))
54   
55       with nogil:
56           for feat_idx in range(n_features):
57               n_cats = n_categories[feat_idx]
58   
59               for cat_idx in range(n_cats):
60                   sums[cat_idx] = smooth_sum
61                   counts[cat_idx] = smooth
62   
63               for sample_idx in range(n_samples):
64                   X_int_tmp = X_int[sample_idx, feat_idx]
65                   # -1 are unknown categories, which are not counted
66                   if X_int_tmp == -1:
67                       continue
68                   sums[X_int_tmp] += y[sample_idx]
69                   counts[X_int_tmp] += 1.0
70   
71               for cat_idx in range(n_cats):
72                   if counts[cat_idx] == 0:
73                       encoding_vec[feat_idx][cat_idx] = y_mean
74                   else:
75                       encoding_vec[feat_idx][cat_idx] = sums[cat_idx] / counts[cat_idx]
76   
77       return encodings

80   def _fit_encoding_fast_auto_smooth(
81       INT_DTYPE[:, ::1] X_int,
82       const Y_DTYPE[:] y,
83       int64_t[::1] n_categories,
84       double y_mean,
85       double y_variance,
86   ):
87       """Fit a target encoding on X_int and y with auto smoothing. 
88    
89       This implementation uses Eq 5 and 6 from [1]. 
90    
91       [1]: Micci-Barreca, Daniele. "A preprocessing scheme for high-cardinality 
92            categorical attributes in classification and prediction problems" 
93       """
94       cdef:
95           int64_t sample_idx, feat_idx, cat_idx, n_cats
96           INT_DTYPE X_int_tmp
97           double diff
98           int n_samples = X_int.shape[0]
99           int n_features = X_int.shape[1]
100          int64_t max_n_cats = np.max(n_categories)
101          double[::1] means = np.empty(max_n_cats, dtype=np.float64)
102          int64_t[::1] counts = np.empty(max_n_cats, dtype=np.int64)
103          double[::1] sum_of_squared_diffs = np.empty(max_n_cats, dtype=np.float64)
104          double lambda_
105          list encodings = []
106          double[::1] current_encoding
107          # Gives access to encodings without gil
108          vector[double*] encoding_vec
109  
110      encoding_vec.resize(n_features)
111      for feat_idx in range(n_features):
112          current_encoding = np.empty(shape=n_categories[feat_idx], dtype=np.float64)
113          encoding_vec[feat_idx] = &current_encoding[0]
114          encodings.append(np.asarray(current_encoding))
115  
116      # TODO: parallelize this with OpenMP prange. When n_features >= n_threads, it's
117      # probably good to parallelize the outer loop. When n_features is too small,
118      # then it would probably better to parallelize the nested loops on n_samples and
119      # n_cats, but the code to handle thread-local temporary variables might be
120      # significantly more complex.
121      with nogil:
122          for feat_idx in range(n_features):
123              n_cats = n_categories[feat_idx]
124  
125              for cat_idx in range(n_cats):
126                  means[cat_idx] = 0.0
127                  counts[cat_idx] = 0
128                  sum_of_squared_diffs[cat_idx] = 0.0
129  
130              # first pass to compute the mean
131              for sample_idx in range(n_samples):
132                  X_int_tmp = X_int[sample_idx, feat_idx]
133  
134                  # -1 are unknown categories, which are not counted
135                  if X_int_tmp == -1:
136                      continue
137                  counts[X_int_tmp] += 1
138                  means[X_int_tmp] += y[sample_idx]
139  
140              for cat_idx in range(n_cats):
141                  means[cat_idx] /= counts[cat_idx]
142  
143              # second pass to compute the sum of squared differences
144              for sample_idx in range(n_samples):
145                  X_int_tmp = X_int[sample_idx, feat_idx]
146                  if X_int_tmp == -1:
147                      continue
148                  diff = y[sample_idx] - means[X_int_tmp]
149                  sum_of_squared_diffs[X_int_tmp] += diff * diff
150  
151              for cat_idx in range(n_cats):
152                  lambda_ = (
153                      y_variance * counts[cat_idx] /
154                      (y_variance * counts[cat_idx] + sum_of_squared_diffs[cat_idx] /
155                       counts[cat_idx])
156                  )
157                  if isnan(lambda_):
158                      # A nan can happen when:
159                      # 1. counts[cat_idx] == 0
160                      # 2. y_variance == 0 and sum_of_squared_diffs[cat_idx] == 0
161                      encoding_vec[feat_idx][cat_idx] = y_mean
162                  else:
163                      encoding_vec[feat_idx][cat_idx] = (
164                          lambda_ * means[cat_idx] + (1 - lambda_) * y_mean
165                      )
166  
167      return encodings
