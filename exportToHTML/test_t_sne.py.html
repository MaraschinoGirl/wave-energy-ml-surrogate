53   def test_gradient_descent_stops():
54       # Test stopping conditions of gradient descent.
55       class ObjectiveSmallGradient:
56           def __init__(self):
57               self.it = -1
58   
59           def __call__(self, _, compute_error=True):
60               self.it += 1
61               return (10 - self.it) / 10.0, np.array([1e-5])
62   
63       def flat_function(_, compute_error=True):
64           return 0.0, np.ones(1)
65   
66       # Gradient norm
67       old_stdout = sys.stdout
68       sys.stdout = StringIO()
69       try:
70           _, error, it = _gradient_descent(
71               ObjectiveSmallGradient(),
72               np.zeros(1),
73               0,
74               max_iter=100,
75               n_iter_without_progress=100,
76               momentum=0.0,
77               learning_rate=0.0,
78               min_gain=0.0,
79               min_grad_norm=1e-5,
80               verbose=2,
81           )
82       finally:
83           out = sys.stdout.getvalue()
84           sys.stdout.close()
85           sys.stdout = old_stdout
86       assert error == 1.0
87       assert it == 0
88       assert "gradient norm" in out
89   
90       # Maximum number of iterations without improvement
91       old_stdout = sys.stdout
92       sys.stdout = StringIO()
93       try:
94           _, error, it = _gradient_descent(
95               flat_function,
96               np.zeros(1),
97               0,
98               max_iter=100,
99               n_iter_without_progress=10,
100              momentum=0.0,
101              learning_rate=0.0,
102              min_gain=0.0,
103              min_grad_norm=0.0,
104              verbose=2,
105          )
106      finally:
107          out = sys.stdout.getvalue()
108          sys.stdout.close()
109          sys.stdout = old_stdout
110      assert error == 0.0
111      assert it == 11
112      assert "did not make any progress" in out
113  
114      # Maximum number of iterations
115      old_stdout = sys.stdout
116      sys.stdout = StringIO()
117      try:
118          _, error, it = _gradient_descent(
119              ObjectiveSmallGradient(),
120              np.zeros(1),
121              0,
122              max_iter=11,
123              n_iter_without_progress=100,
124              momentum=0.0,
125              learning_rate=0.0,

139  def test_binary_search():
140      # Test if the binary search finds Gaussians with desired perplexity.
141      random_state = check_random_state(0)
142      data = random_state.randn(50, 5)
143      distances = pairwise_distances(data).astype(np.float32)
144      desired_perplexity = 25.0
145      P = _binary_search_perplexity(distances, desired_perplexity, verbose=0)
146      P = np.maximum(P, np.finfo(np.double).eps)
147      mean_perplexity = np.mean(
148          [np.exp(-np.sum(P[i] * np.log(P[i]))) for i in range(P.shape[0])]
149      )
150      assert_almost_equal(mean_perplexity, desired_perplexity, decimal=3)

165  def test_binary_search_neighbors():
166      # Binary perplexity search approximation.
167      # Should be approximately equal to the slow method when we use
168      # all points as neighbors.
169      n_samples = 200
170      desired_perplexity = 25.0
171      random_state = check_random_state(0)
172      data = random_state.randn(n_samples, 2).astype(np.float32, copy=False)
173      distances = pairwise_distances(data)
174      P1 = _binary_search_perplexity(distances, desired_perplexity, verbose=0)
175  
176      # Test that when we use all the neighbors the results are identical
177      n_neighbors = n_samples - 1
178      nn = NearestNeighbors().fit(data)
179      distance_graph = nn.kneighbors_graph(n_neighbors=n_neighbors, mode="distance")
180      distances_nn = distance_graph.data.astype(np.float32, copy=False)
181      distances_nn = distances_nn.reshape(n_samples, n_neighbors)
182      P2 = _binary_search_perplexity(distances_nn, desired_perplexity, verbose=0)
183  
184      indptr = distance_graph.indptr
185      P1_nn = np.array(
186          [
187              P1[k, distance_graph.indices[indptr[k] : indptr[k + 1]]]
188              for k in range(n_samples)
189          ]
190      )
191      assert_array_almost_equal(P1_nn, P2, decimal=4)
192  
193      # Test that the highest P_ij are the same when fewer neighbors are used
194      for k in np.linspace(150, n_samples - 1, 5):
195          k = int(k)
196          topn = k * 10  # check the top 10 * k entries out of k * k entries
197          distance_graph = nn.kneighbors_graph(n_neighbors=k, mode="distance")
198          distances_nn = distance_graph.data.astype(np.float32, copy=False)
199          distances_nn = distances_nn.reshape(n_samples, k)
200          P2k = _binary_search_perplexity(distances_nn, desired_perplexity, verbose=0)
201          assert_array_almost_equal(P1_nn, P2, decimal=2)
202          idx = np.argsort(P1.ravel())[::-1]
203          P1top = P1.ravel()[idx][:topn]
204          idx = np.argsort(P2k.ravel())[::-1]
205          P2top = P2k.ravel()[idx][:topn]
206          assert_array_almost_equal(P1top, P2top, decimal=2)

209  def test_binary_perplexity_stability():
210      # Binary perplexity search should be stable.
211      # The binary_search_perplexity had a bug wherein the P array
212      # was uninitialized, leading to sporadically failing tests.
213      n_neighbors = 10
214      n_samples = 100
215      random_state = check_random_state(0)
216      data = random_state.randn(n_samples, 5)
217      nn = NearestNeighbors().fit(data)
218      distance_graph = nn.kneighbors_graph(n_neighbors=n_neighbors, mode="distance")
219      distances = distance_graph.data.astype(np.float32, copy=False)
220      distances = distances.reshape(n_samples, n_neighbors)
221      last_P = None
222      desired_perplexity = 3
223      for _ in range(100):
224          P = _binary_search_perplexity(distances.copy(), desired_perplexity, verbose=0)
225          P1 = _joint_probabilities_nn(distance_graph, desired_perplexity, verbose=0)
226          # Convert the sparse matrix to a dense one for testing
227          P1 = P1.toarray()
228          if last_P is None:
229              last_P = P
230              last_P1 = P1
231          else:
232              assert_array_almost_equal(P, last_P, decimal=4)
233              assert_array_almost_equal(P1, last_P1, decimal=4)

236  def test_gradient():
237      # Test gradient of Kullback-Leibler divergence.
238      random_state = check_random_state(0)
239  
240      n_samples = 50
241      n_features = 2
242      n_components = 2
243      alpha = 1.0
244  
245      distances = random_state.randn(n_samples, n_features).astype(np.float32)
246      distances = np.abs(distances.dot(distances.T))
247      np.fill_diagonal(distances, 0.0)
248      X_embedded = random_state.randn(n_samples, n_components).astype(np.float32)
249  
250      P = _joint_probabilities(distances, desired_perplexity=25.0, verbose=0)
251  
252      def fun(params):
253          return _kl_divergence(params, P, alpha, n_samples, n_components)[0]
254  
255      def grad(params):
256          return _kl_divergence(params, P, alpha, n_samples, n_components)[1]
257  
258      assert_almost_equal(check_grad(fun, grad, X_embedded.ravel()), 0.0, decimal=5)

842  def test_min_grad_norm():
843      # Make sure that the parameter min_grad_norm is used correctly
844      random_state = check_random_state(0)
845      X = random_state.randn(100, 2)
846      min_grad_norm = 0.002
847      tsne = TSNE(min_grad_norm=min_grad_norm, verbose=2, random_state=0, method="exact")
848  
849      old_stdout = sys.stdout
850      sys.stdout = StringIO()
851      try:
852          tsne.fit_transform(X)
853      finally:
854          out = sys.stdout.getvalue()
855          sys.stdout.close()
856          sys.stdout = old_stdout
857  
858      lines_out = out.split("\n")
859  
860      # extract the gradient norm from the verbose output
861      gradient_norm_values = []
862      for line in lines_out:
863          # When the computation is Finished just an old gradient norm value
864          # is repeated that we do not need to store
865          if "Finished" in line:
866              break
867  
868          start_grad_norm = line.find("gradient norm")
869          if start_grad_norm >= 0:
870              line = line[start_grad_norm:]
871              line = line.replace("gradient norm = ", "").split(" ")[0]
872              gradient_norm_values.append(float(line))
873  
874      # Compute how often the gradient norm is smaller than min_grad_norm
875      gradient_norm_values = np.array(gradient_norm_values)
876      n_smaller_gradient_norms = len(
877          gradient_norm_values[gradient_norm_values <= min_grad_norm]
878      )
879  
880      # The gradient norm can be smaller than min_grad_norm at most once,
881      # because in the moment it becomes smaller the optimization stops
882      assert n_smaller_gradient_norms <= 1

917  @pytest.mark.parametrize("method", ["barnes_hut", "exact"])
918  def test_uniform_grid(method):
919      """Make sure that TSNE can approximately recover a uniform 2D grid 
920   
921      Due to ties in distances between point in X_2d_grid, this test is platform 
922      dependent for ``method='barnes_hut'`` due to numerical imprecision. 
923   
924      Also, t-SNE is not assured to converge to the right solution because bad 
925      initialization can lead to convergence to bad local minimum (the 
926      optimization problem is non-convex). To avoid breaking the test too often, 
927      we re-run t-SNE from the final point when the convergence is not good 
928      enough. 
929      """
930      seeds = range(3)
931      max_iter = 500
932      for seed in seeds:
933          tsne = TSNE(
934              n_components=2,
935              init="random",
936              random_state=seed,
937              perplexity=50,
938              max_iter=max_iter,
939              method=method,
940              learning_rate="auto",
941          )
942          Y = tsne.fit_transform(X_2d_grid)
943  
944          try_name = "{}_{}".format(method, seed)
945          try:
946              assert_uniform_grid(Y, try_name)
947          except AssertionError:
948              # If the test fails a first time, re-run with init=Y to see if
949              # this was caused by a bad initialization. Note that this will
950              # also run an early_exaggeration step.
951              try_name += ":rerun"
952              tsne.init = Y
953              Y = tsne.fit_transform(X_2d_grid)
954              assert_uniform_grid(Y, try_name)

957  def assert_uniform_grid(Y, try_name=None):
958      # Ensure that the resulting embedding leads to approximately
959      # uniformly spaced points: the distance to the closest neighbors
960      # should be non-zero and approximately constant.
961      nn = NearestNeighbors(n_neighbors=1).fit(Y)
962      dist_to_nn = nn.kneighbors(return_distance=True)[0].ravel()
963      assert dist_to_nn.min() > 0.1
964  
965      smallest_to_mean = dist_to_nn.min() / np.mean(dist_to_nn)
966      largest_to_mean = dist_to_nn.max() / np.mean(dist_to_nn)
967  
968      assert smallest_to_mean > 0.5, try_name
969      assert largest_to_mean < 2, try_name
