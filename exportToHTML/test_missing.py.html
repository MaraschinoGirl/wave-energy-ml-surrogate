15           (0, False),
16           (0.0, False),
17           (None, False),
18           ("", False),
19           ("nan", False),
20           ([np.nan], False),
21           (9867966753463435747313673, False),  # Python int that overflows with C type
22       ],
23   )
24   def test_is_scalar_nan(value, result):
25       assert is_scalar_nan(value) is result
26       # make sure that we are returning a Python bool
27       assert isinstance(is_scalar_nan(value), bool)
