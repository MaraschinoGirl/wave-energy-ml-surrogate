22   class ContrastMatrix:
23       """A simple container for a matrix used for coding categorical factors. 
24    
25       Attributes: 
26    
27       .. attribute:: matrix 
28    
29          A 2d ndarray, where each column corresponds to one column of the 
30          resulting design matrix, and each row contains the entries for a single 
31          categorical variable level. Usually n-by-n for a full rank coding or 
32          n-by-(n-1) for a reduced rank coding, though other options are 
33          possible. 
34    
35       .. attribute:: column_suffixes 
36    
37          A list of strings to be appended to the factor name, to produce the 
38          final column names. E.g. for treatment coding the entries will look 
39          like ``"[T.level1]"``. 
40       """
41   
42       def __init__(self, matrix, column_suffixes):
43           self.matrix = np.asarray(matrix)
44           self.column_suffixes = column_suffixes
45           if self.matrix.shape[1] != len(column_suffixes):
46               raise PatsyError("matrix and column_suffixes don't conform")
47   
48       __repr__ = repr_pretty_delegate
49   
50       def _repr_pretty_(self, p, cycle):
51           repr_pretty_impl(p, self, [self.matrix, self.column_suffixes])
52   
53       __getstate__ = no_pickling

70   # This always produces an object of the type that Python calls 'str' (whether
71   # that be a Python 2 string-of-bytes or a Python 3 string-of-unicode). It does
72   # *not* make any particular guarantees about being reversible or having other
73   # such useful programmatic properties -- it just produces something that will
74   # be nice for users to look at.
75   def _obj_to_readable_str(obj):
76       if isinstance(obj, str):
77           return obj
78       elif isinstance(obj, bytes):
79           try:
80               return obj.decode("utf-8")
81           except UnicodeDecodeError:
82               return repr(obj)
83       else:
84           return repr(obj)

118  def _get_level(levels, level_ref):
119      if level_ref in levels:
120          return levels.index(level_ref)
121      if isinstance(level_ref, int):
122          if level_ref < 0:
123              level_ref += len(levels)
124          if not (0 <= level_ref < len(levels)):
125              raise PatsyError("specified level %r is out of range" % (level_ref,))
126          return level_ref
127      raise PatsyError("specified level %r not found" % (level_ref,))

143  class Treatment:
144      """Treatment coding (also known as dummy coding). 
145   
146      This is the default coding. 
147   
148      For reduced-rank coding, one level is chosen as the "reference", and its 
149      mean behaviour is represented by the intercept. Each column of the 
150      resulting matrix represents the difference between the mean of one level 
151      and this reference level. 
152   
153      For full-rank coding, classic "dummy" coding is used, and each column of 
154      the resulting matrix represents the mean of the corresponding level. 
155   
156      The reference level defaults to the first level, or can be specified 
157      explicitly. 
158   
159      .. ipython:: python 
160   
161         # reduced rank 
162         dmatrix("C(a, Treatment)", balanced(a=3)) 
163         # full rank 
164         dmatrix("0 + C(a, Treatment)", balanced(a=3)) 
165         # Setting a reference level 
166         dmatrix("C(a, Treatment(1))", balanced(a=3)) 
167         dmatrix("C(a, Treatment('a2'))", balanced(a=3)) 
168   
169      Equivalent to R ``contr.treatment``. The R documentation suggests that 
170      using ``Treatment(reference=-1)`` will produce contrasts that are 
171      "equivalent to those produced by many (but not all) SAS procedures". 
172      """
173  
174      def __init__(self, reference=None):
175          self.reference = reference
176  
177      def code_with_intercept(self, levels):
178          return _dummy_code(levels)
179  
180      def code_without_intercept(self, levels):
181          if self.reference is None:
182              reference = 0
183          else:
184              reference = _get_level(levels, self.reference)
185          eye = np.eye(len(levels) - 1)
186          contrasts = np.vstack(
187              (eye[:reference, :], np.zeros((1, len(levels) - 1)), eye[reference:, :])
188          )
189          names = _name_levels("T.", levels[:reference] + levels[reference + 1 :])
190          return ContrastMatrix(contrasts, names)
191  
192      __getstate__ = no_pickling
193  
194  
195  def test_Treatment():
196      t1 = Treatment()
197      matrix = t1.code_with_intercept(["a", "b", "c"])
198      assert matrix.column_suffixes == ["[a]", "[b]", "[c]"]
199      assert np.allclose(matrix.matrix, [[1, 0, 0], [0, 1, 0], [0, 0, 1]])
200      matrix = t1.code_without_intercept(["a", "b", "c"])
201      assert matrix.column_suffixes == ["[T.b]", "[T.c]"]
202      assert np.allclose(matrix.matrix, [[0, 0], [1, 0], [0, 1]])
203      matrix = Treatment(reference=1).code_without_intercept(["a", "b", "c"])
204      assert matrix.column_suffixes == ["[T.a]", "[T.c]"]
205      assert np.allclose(matrix.matrix, [[1, 0], [0, 0], [0, 1]])
206      matrix = Treatment(reference=-2).code_without_intercept(["a", "b", "c"])
207      assert matrix.column_suffixes == ["[T.a]", "[T.c]"]
208      assert np.allclose(matrix.matrix, [[1, 0], [0, 0], [0, 1]])
209      matrix = Treatment(reference="b").code_without_intercept(["a", "b", "c"])
210      assert matrix.column_suffixes == ["[T.a]", "[T.c]"]
211      assert np.allclose(matrix.matrix, [[1, 0], [0, 0], [0, 1]])
212      # Make sure the default is always the first level, even if there is a
213      # different level called 0.

215      assert matrix.column_suffixes == ["[T.1]", "[T.0]"]
216      assert np.allclose(matrix.matrix, [[0, 0], [1, 0], [0, 1]])

219  class Poly(object):
220      """Orthogonal polynomial contrast coding. 
221   
222      This coding scheme treats the levels as ordered samples from an underlying 
223      continuous scale, whose effect takes an unknown functional form which is 
224      `Taylor-decomposed`__ into the sum of a linear, quadratic, etc. components. 
225   
226      .. __: https://en.wikipedia.org/wiki/Taylor_series 
227   
228      For reduced-rank coding, you get a linear column, a quadratic column, 
229      etc., up to the number of levels provided. 
230   
231      For full-rank coding, the same scheme is used, except that the zero-order 
232      constant polynomial is also included. I.e., you get an intercept column 
233      included as part of your categorical term. 
234   
235      By default the levels are treated as equally spaced, but you can override 
236      this by providing a value for the `scores` argument. 
237   
238      Examples: 
239   
240      .. ipython:: python 
241   
242         # Reduced rank 
243         dmatrix("C(a, Poly)", balanced(a=4)) 
244         # Full rank 
245         dmatrix("0 + C(a, Poly)", balanced(a=3)) 
246         # Explicit scores 
247         dmatrix("C(a, Poly([1, 2, 10]))", balanced(a=3)) 
248   
249      This is equivalent to R's ``contr.poly``. (But note that in R, reduced 
250      rank encodings are always dummy-coded, regardless of what contrast you 
251      have set.) 
252      """
253  
254      def __init__(self, scores=None):
255          self.scores = scores
256  
257      def _code_either(self, intercept, levels):
258          n = len(levels)
259          scores = self.scores
260          if scores is None:
261              scores = np.arange(n)
262          scores = np.asarray(scores, dtype=float)
263          if len(scores) != n:
264              raise PatsyError(
265                  "number of levels (%s) does not match"
266                  " number of scores (%s)" % (n, len(scores))
267              )
268          # Strategy: just make a matrix whose columns are naive linear,
269          # quadratic, etc., functions of the raw scores, and then use 'qr' to
270          # orthogonalize each column against those to its left.
271          scores -= scores.mean()
272          raw_poly = scores.reshape((-1, 1)) ** np.arange(n).reshape((1, -1))
273          q, r = np.linalg.qr(raw_poly)
274          q *= np.sign(np.diag(r))
275          q /= np.sqrt(np.sum(q**2, axis=1))
276          # The constant term is always all 1's -- we don't normalize it.
277          q[:, 0] = 1
278          names = [".Constant", ".Linear", ".Quadratic", ".Cubic"]
279          names += ["^%s" % (i,) for i in range(4, n)]
280          names = names[:n]
281          if intercept:
282              return ContrastMatrix(q, names)
283          else:
284              # We always include the constant/intercept column as something to
285              # orthogonalize against, but we don't always return it:
286              return ContrastMatrix(q[:, 1:], names[1:])
287  
288      def code_with_intercept(self, levels):
289          return self._code_either(True, levels)
290  
291      def code_without_intercept(self, levels):
292          return self._code_either(False, levels)
293  
294      __getstate__ = no_pickling
295

365  class Sum(object):
366      """Deviation coding (also known as sum-to-zero coding). 
367   
368      Compares the mean of each level to the mean-of-means. (In a balanced 
369      design, compares the mean of each level to the overall mean.) 
370   
371      For full-rank coding, a standard intercept term is added. 
372   
373      One level must be omitted to avoid redundancy; by default this is the last 
374      level, but this can be adjusted via the `omit` argument. 
375   
376      .. warning:: There are multiple definitions of 'deviation coding' in 
377         use. Make sure this is the one you expect before trying to interpret 
378         your results! 
379   
380      Examples: 
381   
382      .. ipython:: python 
383   
384         # Reduced rank 
385         dmatrix("C(a, Sum)", balanced(a=4)) 
386         # Full rank 
387         dmatrix("0 + C(a, Sum)", balanced(a=4)) 
388         # Omit a different level 
389         dmatrix("C(a, Sum(1))", balanced(a=3)) 
390         dmatrix("C(a, Sum('a1'))", balanced(a=3)) 
391   
392      This is equivalent to R's `contr.sum`. 
393      """
394  
395      def __init__(self, omit=None):
396          self.omit = omit
397  
398      def _omit_i(self, levels):
399          if self.omit is None:
400              # We assume below that this is positive
401              return len(levels) - 1
402          else:
403              return _get_level(levels, self.omit)
404  
405      def _sum_contrast(self, levels):
406          n = len(levels)
407          omit_i = self._omit_i(levels)
408          eye = np.eye(n - 1)
409          out = np.empty((n, n - 1))
410          out[:omit_i, :] = eye[:omit_i, :]
411          out[omit_i, :] = -1
412          out[omit_i + 1 :, :] = eye[omit_i:, :]
413          return out
414  
415      def code_with_intercept(self, levels):
416          contrast = self.code_without_intercept(levels)
417          matrix = np.column_stack((np.ones(len(levels)), contrast.matrix))
418          column_suffixes = ["[mean]"] + contrast.column_suffixes
419          return ContrastMatrix(matrix, column_suffixes)
420  
421      def code_without_intercept(self, levels):
422          matrix = self._sum_contrast(levels)
423          omit_i = self._omit_i(levels)
424          included_levels = levels[:omit_i] + levels[omit_i + 1 :]
425          return ContrastMatrix(matrix, _name_levels("S.", included_levels))
426  
427      __getstate__ = no_pickling
428

468  class Helmert(object):
469      """Helmert contrasts. 
470   
471      Compares the second level with the first, the third with the average of 
472      the first two, and so on. 
473   
474      For full-rank coding, a standard intercept term is added. 
475   
476      .. warning:: There are multiple definitions of 'Helmert coding' in 
477         use. Make sure this is the one you expect before trying to interpret 
478         your results! 
479   
480      Examples: 
481   
482      .. ipython:: python 
483   
484         # Reduced rank 
485         dmatrix("C(a, Helmert)", balanced(a=4)) 
486         # Full rank 
487         dmatrix("0 + C(a, Helmert)", balanced(a=4)) 
488   
489      This is equivalent to R's `contr.helmert`. 
490      """
491  
492      def _helmert_contrast(self, levels):
493          n = len(levels)
494          # http://www.ats.ucla.edu/stat/sas/webbooks/reg/chapter5/sasreg5.htm#HELMERT
495          # contr = np.eye(n - 1)
496          # int_range = np.arange(n - 1., 1, -1)
497          # denom = np.repeat(int_range, np.arange(n - 2, 0, -1))
498          # contr[np.tril_indices(n - 1, -1)] = -1. / denom
499  
500          # http://www.ats.ucla.edu/stat/r/library/contrast_coding.htm#HELMERT
501          # contr = np.zeros((n - 1., n - 1))
502          # int_range = np.arange(n, 1, -1)
503          # denom = np.repeat(int_range[:-1], np.arange(n - 2, 0, -1))
504          # contr[np.diag_indices(n - 1)] = (int_range - 1.) / int_range
505          # contr[np.tril_indices(n - 1, -1)] = -1. / denom
506          # contr = np.vstack((contr, -1./int_range))
507  
508          # r-like
509          contr = np.zeros((n, n - 1))
510          contr[1:][np.diag_indices(n - 1)] = np.arange(1, n)
511          contr[np.triu_indices(n - 1)] = -1
512          return contr
513  
514      def code_with_intercept(self, levels):
515          contrast = np.column_stack(
516              (np.ones(len(levels)), self._helmert_contrast(levels))
517          )
518          column_suffixes = _name_levels("H.", ["intercept"] + list(levels[1:]))
519          return ContrastMatrix(contrast, column_suffixes)
520  
521      def code_without_intercept(self, levels):
522          contrast = self._helmert_contrast(levels)
523          return ContrastMatrix(contrast, _name_levels("H.", levels[1:]))
524  
525      __getstate__ = no_pickling
526  
527  
528  def test_Helmert():
529      t1 = Helmert()
530      for levels in (["a", "b", "c", "d"], ("a", "b", "c", "d")):
531          matrix = t1.code_with_intercept(levels)
532          assert matrix.column_suffixes == ["[H.intercept]", "[H.b]", "[H.c]", "[H.d]"]

537          matrix = t1.code_without_intercept(levels)
538          assert matrix.column_suffixes == ["[H.b]", "[H.c]", "[H.d]"]

544  class Diff(object):
545      """Backward difference coding. 
546   
547      This coding scheme is useful for ordered factors, and compares the mean of 
548      each level with the preceding level. So you get the second level minus the 
549      first, the third level minus the second, etc. 
550   
551      For full-rank coding, a standard intercept term is added (which gives the 
552      mean value for the first level). 
553   
554      Examples: 
555   
556      .. ipython:: python 
557   
558         # Reduced rank 
559         dmatrix("C(a, Diff)", balanced(a=3)) 
560         # Full rank 
561         dmatrix("0 + C(a, Diff)", balanced(a=3)) 
562      """
563  
564      def _diff_contrast(self, levels):
565          nlevels = len(levels)
566          contr = np.zeros((nlevels, nlevels - 1))
567          int_range = np.arange(1, nlevels)
568          upper_int = np.repeat(int_range, int_range)
569          row_i, col_i = np.triu_indices(nlevels - 1)
570          # we want to iterate down the columns not across the rows
571          # it would be nice if the index functions had a row/col order arg
572          col_order = np.argsort(col_i)
573          contr[row_i[col_order], col_i[col_order]] = (upper_int - nlevels) / float(
574              nlevels
575          )
576          lower_int = np.repeat(int_range, int_range[::-1])
577          row_i, col_i = np.tril_indices(nlevels - 1)
578          # we want to iterate down the columns not across the rows
579          col_order = np.argsort(col_i)
580          contr[row_i[col_order] + 1, col_i[col_order]] = lower_int / float(nlevels)
581          return contr
582  
583      def code_with_intercept(self, levels):
584          contrast = np.column_stack((np.ones(len(levels)), self._diff_contrast(levels)))
585          return ContrastMatrix(contrast, _name_levels("D.", levels))
586  
587      def code_without_intercept(self, levels):
588          contrast = self._diff_contrast(levels)
589          return ContrastMatrix(contrast, _name_levels("D.", levels[:-1]))

620  # contrast can be:
621  #   -- a ContrastMatrix
622  #   -- a simple np.ndarray
623  #   -- an object with code_with_intercept and code_without_intercept methods
624  #   -- a function returning one of the above
625  #   -- None, in which case the above rules are applied to 'default'
626  # This function always returns a ContrastMatrix.
627  def code_contrast_matrix(intercept, levels, contrast, default=None):
628      if contrast is None:
629          contrast = default
630      if callable(contrast):
631          contrast = contrast()
632      if isinstance(contrast, ContrastMatrix):
633          return contrast
634      as_array = np.asarray(contrast)
635      if safe_issubdtype(as_array.dtype, np.number):
636          return ContrastMatrix(
637              as_array, _name_levels("custom", range(as_array.shape[1]))
638          )
639      if intercept:
640          return contrast.code_with_intercept(levels)
641      else:
642          return contrast.code_without_intercept(levels)
