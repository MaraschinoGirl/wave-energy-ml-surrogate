132      def analyse_text(text):
133          """This is very difficult to guess relative to other business languages. 
134          -> in conjunction with BEGIN/END seems relatively rare though."""
135          result = 0
136  
137          if '->' in text:
138              result += 0.01
139          if 'BEGIN' in text:
140              result += 0.01
141          if 'END' in text:
142              result += 0.01
143  
144          return result
