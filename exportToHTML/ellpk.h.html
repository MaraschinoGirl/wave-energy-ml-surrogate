5    /*                                                     ellpk.c 
6     * 
7     *     Complete elliptic integral of the first kind 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double m1, y, ellpk(); 
14    * 
15    * y = ellpk( m1 ); 
16    * 
17    * 
18    * 
19    * DESCRIPTION: 
20    * 
21    * Approximates the integral 
22    * 
23    * 
24    * 
25    *            pi/2 
26    *             - 
27    *            | | 
28    *            |           dt 
29    * K(m)  =    |    ------------------ 
30    *            |                   2 
31    *          | |    sqrt( 1 - m sin t ) 
32    *           - 
33    *            0 
34    * 
35    * where m = 1 - m1, using the approximation 
36    * 
37    *     P(x)  -  log x Q(x). 
38    * 
39    * The argument m1 is used internally rather than m so that the logarithmic 
40    * singularity at m = 1 will be shifted to the origin; this 
41    * preserves maximum accuracy. 
42    * 
43    * K(0) = pi/2. 
44    * 
45    * ACCURACY: 
46    * 
47    *                      Relative error: 
48    * arithmetic   domain     # trials      peak         rms 
49    *    IEEE       0,1        30000       2.5e-16     6.8e-17 
50    * 
51    * ERROR MESSAGES: 
52    * 
53    *   message         condition      value returned 
54    * ellpk domain       x<0, x>1           0.0 
55    * 
56    */
57

91       XSF_HOST_DEVICE inline double ellpk(double x) {
92   
93           if (x < 0.0) {
94               set_error("ellpk", SF_ERROR_DOMAIN, NULL);
95               return (std::numeric_limits<double>::quiet_NaN());
96           }
97   
98           if (x > 1.0) {
99               if (std::isinf(x)) {
100                  return 0.0;
101              }
102              return ellpk(1 / x) / std::sqrt(x);
103          }
104  
105          if (x > detail::MACHEP) {
106              return (polevl(x, detail::ellpk_P, 10) - std::log(x) * polevl(x, detail::ellpk_Q, 10));
107          } else {
108              if (x == 0.0) {
109                  set_error("ellpk", SF_ERROR_SINGULAR, NULL);
110                  return (std::numeric_limits<double>::infinity());
111              } else {
112                  return (detail::ellpk_C1 - 0.5 * std::log(x));
113              }
114          }
115      }
