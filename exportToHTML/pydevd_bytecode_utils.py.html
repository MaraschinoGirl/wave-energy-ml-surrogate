59   class Target(object):
60       __slots__ = ["arg", "lineno", "endlineno", "startcol", "endcol", "offset", "children_targets"]
61   
62       def __init__(
63           self,
64           arg,
65           lineno,
66           offset,
67           children_targets=(),
68           # These are optional (only Python 3.11 onwards).
69           endlineno=-1,
70           startcol=-1,
71           endcol=-1,
72       ):
73           self.arg = arg
74           self.lineno = lineno
75           self.endlineno = endlineno
76           self.startcol = startcol
77           self.endcol = endcol
78   
79           self.offset = offset
80           self.children_targets = children_targets

82       def __repr__(self):
83           ret = []
84           for s in self.__slots__:
85               ret.append("%s: %s" % (s, getattr(self, s)))
86           return "Target(%s)" % ", ".join(ret)

91   class _TargetIdHashable(object):
92       def __init__(self, target):
93           self.target = target
94   
95       def __eq__(self, other):
96           if not hasattr(other, "target"):
97               return
98           return other.target is self.target
99   
100      def __ne__(self, other):
101          return not self == other
102  
103      def __hash__(self):
104          return id(self.target)

123      def _getname(self, instr):
124          if instr.opcode in _opcode.hascompare:
125              cmp_op = dis.cmp_op[instr.arg]
126              if cmp_op not in ("exception match", "BAD"):
127                  return _COMP_OP_MAP.get(cmp_op, cmp_op)
128          return instr.arg

130      def _getcallname(self, instr):
131          if instr.name == "BINARY_SUBSCR":
132              return "__getitem__().__call__"
133          if instr.name == "CALL_FUNCTION":
134              # Note: previously a '__call__().__call__' was returned, but this was a bit weird
135              # and on Python 3.9 this construct could appear for some internal things where
136              # it wouldn't be expected.
137              # Note: it'd be what we had in func()().
138              return None
139          if instr.name == "MAKE_FUNCTION":
140              return "__func__().__call__"
141          if instr.name == "LOAD_ASSERTION_ERROR":
142              return "AssertionError"
143          name = self._getname(instr)
144          if isinstance(name, CodeType):
145              name = name.co_qualname  # Note: only available for Python 3.11
146          if isinstance(name, _Variable):
147              name = name.name
148          if isinstance(name, tuple):
149              # Load attr in Python 3.12 comes with (bool, name)
150              if len(name) == 2 and isinstance(name[0], bool) and isinstance(name[1], str):
151                  name = name[1]
152  
153          if not isinstance(name, str):
154              return None
155          if name.endswith(">"):  # xxx.<listcomp>, xxx.<lambda>, ...
156              return name.split(".")[-1]
157          return name

220      def _handle_call_from_instr(self, func_name_instr, func_call_instr):
221          self.load_attrs.pop(_TargetIdHashable(func_name_instr), None)
222          call_name = self._getcallname(func_name_instr)
223          target = None
224          if not call_name:
225              pass  # Ignore if we can't identify a name
226          elif call_name in ("<listcomp>", "<genexpr>", "<setcomp>", "<dictcomp>"):
227              code_obj = self.func_name_id_to_code_object[_TargetIdHashable(func_name_instr)]
228              if code_obj is not None and GO_INTO_INNER_CODES:
229                  children_targets = _get_smart_step_into_targets(code_obj)
230                  if children_targets:
231                      # i.e.: we have targets inside of a <listcomp> or <genexpr>.
232                      # Note that to actually match this in the debugger we need to do matches on 2 frames,
233                      # the one with the <listcomp> and then the actual target inside the <listcomp>.
234                      target = Target(call_name, func_name_instr.lineno, func_call_instr.offset, children_targets)
235                      self.function_calls.append(target)
236  
237          else:
238              # Ok, regular call
239              target = Target(call_name, func_name_instr.lineno, func_call_instr.offset)
240              self.function_calls.append(target)
241  
242          if DEBUG and target is not None:
243              print("Created target", target)
244          self._stack.append(func_call_instr)  # Keep the func call as the result

246      def on_COMPARE_OP(self, instr):
247          try:
248              _right = self._stack.pop()
249          except IndexError:
250              return
251          try:
252              _left = self._stack.pop()
253          except IndexError:
254              return
255  
256          cmp_op = dis.cmp_op[instr.arg]
257          if cmp_op not in ("exception match", "BAD"):
258              self.function_calls.append(Target(self._getname(instr), instr.lineno, instr.offset))
259  
260          self._stack.append(instr)

302      def on_MAKE_FUNCTION(self, instr):
303          if not IS_PY311_OR_GREATER:
304              # The qualifier name is no longer put in the stack.
305              qualname = self._stack.pop()
306              code_obj_instr = self._stack.pop()
307          else:
308              # In 3.11 the code object has a co_qualname which we can use.
309              qualname = code_obj_instr = self._stack.pop()
310  
311          arg = instr.arg
312          if arg & 0x08:
313              _func_closure = self._stack.pop()
314          if arg & 0x04:
315              _func_annotations = self._stack.pop()
316          if arg & 0x02:
317              _func_kwdefaults = self._stack.pop()
318          if arg & 0x01:
319              _func_defaults = self._stack.pop()
320  
321          call_name = self._getcallname(qualname)
322          if call_name in ("<genexpr>", "<listcomp>", "<setcomp>", "<dictcomp>"):
323              if isinstance(code_obj_instr.arg, CodeType):
324                  self.func_name_id_to_code_object[_TargetIdHashable(qualname)] = code_obj_instr.arg
325          self._stack.append(qualname)

658          # This would actually pop 2 and leave the value in the stack.
659          # In a += 1 it pop `a` and `1` and leave the resulting value
660          # for a load. In our case, let's just pop the `1` and leave the `a`
661          # instead of leaving the INPLACE_ADD bytecode.
662          try:
663              self._stack.pop()
664          except IndexError:
665              pass
666  
667      on_INPLACE_POWER = on_INPLACE_ADD
668      on_INPLACE_MULTIPLY = on_INPLACE_ADD
669      on_INPLACE_MATRIX_MULTIPLY = on_INPLACE_ADD
670      on_INPLACE_TRUE_DIVIDE = on_INPLACE_ADD
671      on_INPLACE_FLOOR_DIVIDE = on_INPLACE_ADD
672      on_INPLACE_MODULO = on_INPLACE_ADD
673      on_INPLACE_SUBTRACT = on_INPLACE_ADD
674      on_INPLACE_RSHIFT = on_INPLACE_ADD
675      on_INPLACE_LSHIFT = on_INPLACE_ADD
676      on_INPLACE_AND = on_INPLACE_ADD
677      on_INPLACE_OR = on_INPLACE_ADD
678      on_INPLACE_XOR = on_INPLACE_ADD
679  
680      def on_DUP_TOP(self, instr):
681          try:
682              i = self._stack[-1]
683          except IndexError:
684              # ok (in the start of block)
685              self._stack.append(instr)
686          else:
687              self._stack.append(i)
688  
689      def on_DUP_TOP_TWO(self, instr):
690          if len(self._stack) == 0:
691              self._stack.append(instr)
692              return
693  
694          if len(self._stack) == 1:
695              i = self._stack[-1]
696              self._stack.append(i)
697              self._stack.append(instr)
698              return
699  
700          i = self._stack[-1]
701          j = self._stack[-2]
702          self._stack.append(j)
703          self._stack.append(i)
704  
705      def on_BUILD_SLICE(self, instr):
706          for _ in range(instr.arg):
707              try:
708                  self._stack.pop()
709              except IndexError:
710                  pass
711          self._stack.append(instr)
712  
713      def on_STORE_SUBSCR(self, instr):
714          try:
715              self._stack.pop()
716              self._stack.pop()
717              self._stack.pop()
718          except IndexError:
719              pass
720  
721      def on_DELETE_SUBSCR(self, instr):
722          try:
723              self._stack.pop()
724              self._stack.pop()
725          except IndexError:
726              pass

729      # some evaluation.
730      on_PRINT_EXPR = on_POP_TOP

732      on_LABEL = _no_stack_change
733      on_UNARY_POSITIVE = _no_stack_change

735      on_UNARY_NOT = _no_stack_change

742  def _get_smart_step_into_targets(code):
743      """ 
744      :return list(Target) 
745      """
746      b = bytecode.Bytecode.from_code(code)
747      cfg = bytecode_cfg.ControlFlowGraph.from_bytecode(b)
748  
749      ret = []
750  
751      for block in cfg:
752          if DEBUG:
753              print("\nStart block----")
754          stack = _StackInterpreter(block)
755          for instr in block:
756              if isinstance(instr, (Label,)):
757                  # No name for these
758                  continue
759              try:
760                  func_name = "on_%s" % (instr.name,)
761                  func = getattr(stack, func_name, None)
762  
763                  if func is None:
764                      if STRICT_MODE:
765                          raise AssertionError("%s not found." % (func_name,))
766                      else:
767                          if DEBUG:
768                              print("Skipping: %s." % (func_name,))
769  
770                          continue
771                  func(instr)
772  
773                  if DEBUG:
774                      if instr.name != "CACHE":  # Filter the ones we don't want to see.
775                          print("\nHandled: ", instr, ">>", stack._getname(instr), "<<")
776                          print("New stack:")
777                          for entry in stack._stack:
778                              print("    arg:", stack._getname(entry), "(", entry, ")")
779              except:
780                  if STRICT_MODE:
781                      raise  # Error in strict mode.
782                  else:
783                      # In non-strict mode, log it (if in verbose mode) and keep on going.
784                      if DebugInfoHolder.DEBUG_TRACE_LEVEL >= 2:
785                          pydev_log.exception("Exception computing step into targets (handled).")
786  
787          ret.extend(stack.function_calls)
788          # No longer considering attr loads as calls (while in theory sometimes it's possible
789          # that something as `some.attr` can turn out to be a property which could be stepped
790          # in, it's not that common in practice and can be surprising for users, so, disabling
791          # step into from stepping into properties).
792          # ret.extend(stack.load_attrs.values())
793  
794          if DEBUG:
795              print("\nEnd block----")
796      return ret

799  # Note that the offset is unique within the frame (so, we can use it as the target id).
800  # Also, as the offset is the instruction offset within the frame, it's possible to
801  # to inspect the parent frame for frame.f_lasti to know where we actually are (as the
802  # caller name may not always match the new frame name).
803  class Variant(object):
804      __slots__ = ["name", "is_visited", "line", "offset", "call_order", "children_variants", "parent", "endlineno", "startcol", "endcol"]
805  
806      def __init__(self, name, is_visited, line, offset, call_order, children_variants=None, endlineno=-1, startcol=-1, endcol=-1):
807          self.name = name
808          self.is_visited = is_visited
809          self.line = line
810          self.endlineno = endlineno
811          self.startcol = startcol
812          self.endcol = endcol
813          self.offset = offset
814          self.call_order = call_order
815          self.children_variants = children_variants
816          self.parent = None
817          if children_variants:
818              for variant in children_variants:
819                  variant.parent = self
820  
821      def __repr__(self):
822          ret = []
823          for s in self.__slots__:
824              if s == "parent":
825                  try:
826                      parent = self.parent
827                  except AttributeError:
828                      ret.append("%s: <not set>" % (s,))
829                  else:
830                      if parent is None:
831                          ret.append("parent: None")
832                      else:
833                          ret.append("parent: %s (%s)" % (parent.name, parent.offset))
834                  continue
835  
836              if s == "children_variants":
837                  ret.append("children_variants: %s" % (len(self.children_variants) if self.children_variants else 0))
838                  continue
839  
840              try:
841                  ret.append("%s= %s" % (s, getattr(self, s)))
842              except AttributeError:
843                  ret.append("%s: <not set>" % (s,))
844          return "Variant(%s)" % ", ".join(ret)
845  
846      __str__ = __repr__

849  def _convert_target_to_variant(target, start_line, end_line, call_order_cache: dict, lasti: int, base: int):
850      name = target.arg
851      if not isinstance(name, str):
852          return
853      if target.lineno > end_line:
854          return
855      if target.lineno < start_line:
856          return
857  
858      call_order = call_order_cache.get(name, 0) + 1
859      call_order_cache[name] = call_order
860      is_visited = target.offset <= lasti
861  
862      children_targets = target.children_targets
863      children_variants = None
864      if children_targets:
865          children_variants = [
866              _convert_target_to_variant(child, start_line, end_line, call_order_cache, lasti, base) for child in target.children_targets
867          ]
868  
869      return Variant(
870          name,
871          is_visited,
872          target.lineno - base,
873          target.offset,
874          call_order,
875          children_variants,
876          # Only really matter in Python 3.11
877          target.endlineno - base if target.endlineno >= 0 else -1,
878          target.startcol,
879          target.endcol,
880      )

883  def calculate_smart_step_into_variants(frame, start_line, end_line, base=0):
884      """ 
885      Calculate smart step into variants for the given line range. 
886      :param frame: 
887      :type frame: :py:class:`types.FrameType` 
888      :param start_line: 
889      :param end_line: 
890      :return: A list of call names from the first to the last. 
891      :note: it's guaranteed that the offsets appear in order. 
892      :raise: :py:class:`RuntimeError` if failed to parse the bytecode or if dis cannot be used. 
893      """
894      if IS_PY311_OR_GREATER:
895          from . import pydevd_bytecode_utils_py311
896  
897          return pydevd_bytecode_utils_py311.calculate_smart_step_into_variants(frame, start_line, end_line, base)
898  
899      variants = []
900      code = frame.f_code
901      lasti = frame.f_lasti
902  
903      call_order_cache = {}
904      if DEBUG:
905          print("dis.dis:")
906          if IS_PY311_OR_GREATER:
907              dis.dis(code, show_caches=False)
908          else:
909              dis.dis(code)
910  
911      for target in _get_smart_step_into_targets(code):
912          variant = _convert_target_to_variant(target, start_line, end_line, call_order_cache, lasti, base)
913          if variant is None:
914              continue
915          variants.append(variant)
916  
917      return variants

920  def get_smart_step_into_variant_from_frame_offset(frame_f_lasti, variants):
921      """ 
922      Given the frame.f_lasti, return the related `Variant`. 
923   
924      :note: if the offset is found before any variant available or no variants are 
925             available, None is returned. 
926   
927      :rtype: Variant|NoneType 
928      """
929      if not variants:
930          return None
931  
932      i = bisect(KeyifyList(variants, lambda entry: entry.offset), frame_f_lasti)
933  
934      if i == 0:
935          return None
936  
937      else:
938          return variants[i - 1]
