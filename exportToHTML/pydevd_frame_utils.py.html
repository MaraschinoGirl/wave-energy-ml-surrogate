83       while trace is not None:
84           filename = trace.tb_frame.f_code.co_filename
85           if filename in ("<frozen importlib._bootstrap>", "<frozen importlib._bootstrap_external>"):
86               # Do not stop on inner exceptions in py3 while importing
87               return True

89           # ImportError should appear in a user's code, not inside debugger
90           for file in FILES_WITH_IMPORT_HOOKS:
91               if filename.endswith(file):
92                   return True
93   
94           trace = trace.tb_next

107  class _LineColInfo:
108      def __init__(self, lineno, end_lineno, colno, end_colno):
109          self.lineno = lineno
110          self.end_lineno = end_lineno
111          self.colno = colno
112          self.end_colno = end_colno
113  
114      def map_columns_to_line(self, original_line: str):
115          """ 
116          The columns internally are actually based on bytes. 
117   
118          Also, the position isn't always the ideal one as the start may not be 
119          what we want (if the user has many subscripts in the line the start 
120          will always be the same and only the end would change). 
121          For more details see: 
122          https://github.com/microsoft/debugpy/issues/1099#issuecomment-1303403995 
123   
124          So, this function maps the start/end columns to the position to be shown in the editor. 
125          """
126          colno = _utf8_byte_offset_to_character_offset(original_line, self.colno)
127          end_colno = _utf8_byte_offset_to_character_offset(original_line, self.end_colno)
128  
129          if self.lineno == self.end_lineno:
130              try:
131                  ret = _extract_caret_anchors_in_bytes_from_line_segment(original_line[colno:end_colno])
132                  if ret is not None:
133                      return (
134                          _utf8_byte_offset_to_character_offset(original_line, ret[0] + self.colno),
135                          _utf8_byte_offset_to_character_offset(original_line, ret[1] + self.colno),
136                      )
137              except Exception:
138                  pass  # Suppress exception
139  
140          return colno, end_colno

148  def _utf8_byte_offset_to_character_offset(s: str, offset: int):
149      byte_offset = 0
150      char_offset = 0
151      offset = offset or 0
152  
153      for char_offset, character in enumerate(s):
154          byte_offset += 1
155  
156          codepoint = ord(character)
157  
158          if codepoint >= _utf8_with_4_bytes:
159              byte_offset += 3
160  
161          elif codepoint >= _utf8_with_3_bytes:
162              byte_offset += 2
163  
164          elif codepoint >= _utf8_with_2_bytes:
165              byte_offset += 1
166  
167          if byte_offset > offset:
168              break
169      else:
170          char_offset += 1
171  
172      return char_offset

175  # Based on traceback._extract_caret_anchors_in_bytes_from_line_segment (Python 3.11.0)
176  def _extract_caret_anchors_in_bytes_from_line_segment(segment: str):
177      import ast
178  
179      try:
180          segment = segment.encode("utf-8")
181      except UnicodeEncodeError:
182          return None
183      try:
184          tree = ast.parse(segment)
185      except SyntaxError:
186          return None
187  
188      if len(tree.body) != 1:
189          return None
190  
191      statement = tree.body[0]
192      if isinstance(statement, ast.Expr):
193          expr = statement.value
194          if isinstance(expr, ast.BinOp):
195              operator_str = segment[expr.left.end_col_offset : expr.right.col_offset]
196              operator_offset = len(operator_str) - len(operator_str.lstrip())
197  
198              left_anchor = expr.left.end_col_offset + operator_offset
199              right_anchor = left_anchor + 1
200              if operator_offset + 1 < len(operator_str) and not operator_str[operator_offset + 1] == ord(b" "):
201                  right_anchor += 1
202              return left_anchor, right_anchor
203          if isinstance(expr, ast.Subscript):
204              return expr.value.end_col_offset, expr.slice.end_col_offset + 1
205  
206      return None

209  class FramesList(object):
210      def __init__(self):
211          self._frames = []
212  
213          # If available, the line number for the frame will be gotten from this dict,
214          # otherwise frame.f_lineno will be used (needed for unhandled exceptions as
215          # the place where we report may be different from the place where it's raised).
216          self.frame_id_to_lineno = {}
217          self.frame_id_to_line_col_info: Dict[Any, _LineColInfo] = {}
218  
219          self.exc_type = None
220          self.exc_desc = None
221          self.trace_obj = None
222  
223          # This may be set to set the current frame (for the case where we have
224          # an unhandled exception where we want to show the root bu we have a different
225          # executing frame).
226          self.current_frame = None
227  
228          # This is to know whether an exception was extracted from a __cause__ or __context__.
229          self.exc_context_msg = ""
230  
231          self.chained_frames_list = None
232  
233      def append(self, frame):
234          self._frames.append(frame)
235  
236      def last_frame(self):
237          return self._frames[-1]
238  
239      def __len__(self):
240          return len(self._frames)
241  
242      def __iter__(self):
243          return iter(self._frames)
244  
245      def __repr__(self):
246          lst = ["FramesList("]
247  
248          lst.append("\n    exc_type: ")
249          lst.append(str(self.exc_type))
250  
251          lst.append("\n    exc_desc: ")
252          lst.append(str(self.exc_desc))
253  
254          lst.append("\n    trace_obj: ")
255          lst.append(str(self.trace_obj))
256  
257          lst.append("\n    current_frame: ")
258          lst.append(str(self.current_frame))
259  
260          for frame in self._frames:
261              lst.append("\n    ")
262              lst.append(repr(frame))
263              lst.append(",")
264  
265          if self.chained_frames_list is not None:
266              lst.append("\n--- Chained ---\n")
267              lst.append(str(self.chained_frames_list))
268  
269          lst.append("\n)")
270  
271          return "".join(lst)

305  def create_frames_list_from_exception_cause(trace_obj, frame, exc_type, exc_desc, memo):
306      lst = []
307      msg = "<Unknown context>"
308      try:
309          exc_cause = getattr(exc_desc, "__cause__", None)
310          msg = _cause_message
311      except Exception:
312          exc_cause = None
313  
314      if exc_cause is None:
315          try:
316              exc_cause = getattr(exc_desc, "__context__", None)
317              msg = _context_message
318          except Exception:
319              exc_cause = None
320  
321      if exc_cause is None or id(exc_cause) in memo:
322          return None
323  
324      # The traceback module does this, so, let's play safe here too...
325      memo.add(id(exc_cause))
326  
327      tb = exc_cause.__traceback__
328      frames_list = FramesList()
329      frames_list.exc_type = type(exc_cause)
330      frames_list.exc_desc = exc_cause
331      frames_list.trace_obj = tb
332      frames_list.exc_context_msg = msg
333  
334      while tb is not None:
335          # Note: we don't use the actual tb.tb_frame because if the cause of the exception
336          # uses the same frame object, the id(frame) would be the same and the frame_id_to_lineno
337          # would be wrong as the same frame needs to appear with 2 different lines.
338          lst.append((_DummyFrameWrapper(tb.tb_frame, tb.tb_lineno, None), tb.tb_lineno, _get_line_col_info_from_tb(tb)))
339          tb = tb.tb_next
340  
341      for tb_frame, tb_lineno, line_col_info in lst:
342          frames_list.append(tb_frame)
343          frames_list.frame_id_to_lineno[id(tb_frame)] = tb_lineno
344          frames_list.frame_id_to_line_col_info[id(tb_frame)] = line_col_info
345  
346      return frames_list

372  def create_frames_list_from_traceback(trace_obj, frame, exc_type, exc_desc, exception_type=None):
373      """ 
374      :param trace_obj: 
375          This is the traceback from which the list should be created. 
376   
377      :param frame: 
378          This is the first frame to be considered (i.e.: topmost frame). If None is passed, all 
379          the frames from the traceback are shown (so, None should be passed for unhandled exceptions). 
380   
381      :param exception_type: 
382          If this is an unhandled exception or user unhandled exception, we'll not trim the stack to create from the passed 
383          frame, rather, we'll just mark the frame in the frames list. 
384      """
385      lst = []
386  
387      tb = trace_obj
388      if tb is not None and tb.tb_frame is not None:
389          f = tb.tb_frame.f_back
390          while f is not None:
391              lst.insert(0, (f, f.f_lineno, None))
392              f = f.f_back
393  
394      while tb is not None:
395          lst.append((tb.tb_frame, tb.tb_lineno, _get_line_col_info_from_tb(tb)))
396          tb = tb.tb_next
397  
398      frames_list = None
399  
400      for tb_frame, tb_lineno, line_col_info in reversed(lst):
401          if frames_list is None and ((frame is tb_frame) or (frame is None) or (exception_type == EXCEPTION_TYPE_USER_UNHANDLED)):
402              frames_list = FramesList()
403  
404          if frames_list is not None:
405              frames_list.append(tb_frame)
406              frames_list.frame_id_to_lineno[id(tb_frame)] = tb_lineno
407              frames_list.frame_id_to_line_col_info[id(tb_frame)] = line_col_info
408  
409      if frames_list is None and frame is not None:
410          # Fallback (shouldn't happen in practice).
411          pydev_log.info("create_frames_list_from_traceback did not find topmost frame in list.")
412          frames_list = create_frames_list_from_frame(frame)
413  
414      frames_list.exc_type = exc_type
415      frames_list.exc_desc = exc_desc
416      frames_list.trace_obj = trace_obj
417  
418      if exception_type == EXCEPTION_TYPE_USER_UNHANDLED:
419          frames_list.current_frame = frame
420      elif exception_type == EXCEPTION_TYPE_UNHANDLED:
421          if len(frames_list) > 0:
422              frames_list.current_frame = frames_list.last_frame()
423  
424      curr = frames_list
425      memo = set()
426      memo.add(id(exc_desc))
427  
428      while True:
429          chained = create_frames_list_from_exception_cause(None, None, None, curr.exc_desc, memo)
430          if chained is None:
431              break
432          else:
433              curr.chained_frames_list = chained
434              curr = chained
435  
436      return frames_list
