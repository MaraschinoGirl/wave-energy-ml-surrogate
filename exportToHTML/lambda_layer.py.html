12   @keras_export("keras.layers.Lambda")
13   class Lambda(Layer):
14       """Wraps arbitrary expressions as a `Layer` object. 
15    
16       The `Lambda` layer exists so that arbitrary expressions can be used 
17       as a `Layer` when constructing Sequential 
18       and Functional API models. `Lambda` layers are best suited for simple 
19       operations or quick experimentation. For more advanced use cases, 
20       prefer writing new subclasses of `Layer`. 
21    
22       WARNING: `Lambda` layers have (de)serialization limitations! 
23    
24       The main reason to subclass `Layer` instead of using a 
25       `Lambda` layer is saving and inspecting a model. `Lambda` layers 
26       are saved by serializing the Python bytecode, which is fundamentally 
27       non-portable and potentially unsafe. 
28       They should only be loaded in the same environment where 
29       they were saved. Subclassed layers can be saved in a more portable way 
30       by overriding their `get_config()` method. Models that rely on 
31       subclassed Layers are also often easier to visualize and reason about. 
32    
33       Example: 
34    
35       ```python 
36       # add a x -> x^2 layer 
37       model.add(Lambda(lambda x: x ** 2)) 
38       ``` 
39    
40       Args: 
41           function: The function to be evaluated. Takes input tensor as first 
42               argument. 
43           output_shape: Expected output shape from function. This argument 
44               can usually be inferred if not explicitly provided. 
45               Can be a tuple or function. If a tuple, it only specifies 
46               the first dimension onward; sample dimension is assumed 
47               either the same as the input: 
48               `output_shape = (input_shape[0], ) + output_shape` or, 
49               the input is `None` and the sample dimension is also `None`: 
50               `output_shape = (None, ) + output_shape`. 
51               If a function, it specifies the 
52               entire shape as a function of the input shape: 
53               `output_shape = f(input_shape)`. 
54           mask: Either None (indicating no masking) or a callable with the same 
55               signature as the `compute_mask` layer method, or a tensor 
56               that will be returned as output mask regardless 
57               of what the input is. 
58           arguments: Optional dictionary of keyword arguments to be passed to the 
59               function. 
60       """
61   
62       def __init__(
63           self, function, output_shape=None, mask=None, arguments=None, **kwargs
64       ):
65           super().__init__(**kwargs)
66   
67           self.arguments = arguments or {}
68           self.function = function
69   
70           if mask is not None:
71               self.supports_masking = True
72           else:
73               self.supports_masking = False
74           self.mask = mask
75           self._output_shape = output_shape
76   
77           # Warning on every invocation will be quite irksome in Eager mode.
78           self._already_warned = False
79   
80           function_args = inspect.getfullargspec(function).args
81           self._fn_expects_training_arg = "training" in function_args
82           self._fn_expects_mask_arg = "mask" in function_args
83   
84       def compute_output_shape(self, input_shape):
85           if self._output_shape is None:
86               # Leverage backend shape inference
87               try:
88                   inputs = tree.map_shape_structure(
89                       lambda x: backend.KerasTensor(x, dtype=self.compute_dtype),
90                       input_shape,
91                   )
92                   output_spec = backend.compute_output_spec(self.call, inputs)
93                   return tree.map_structure(lambda x: x.shape, output_spec)
94               except:
95                   raise NotImplementedError(
96                       "We could not automatically infer the shape of "
97                       "the Lambda's output. Please specify the `output_shape` "
98                       "argument for this Lambda layer."
99                   )
100  
101          if callable(self._output_shape):
102              return self._output_shape(input_shape)
103  
104          # Output shapes are passed directly and don't include batch dimension.
105          batch_size = tree.flatten(input_shape)[0]
106  
107          def _add_batch(shape):
108              return (batch_size,) + shape
109  
110          return tree.map_shape_structure(_add_batch, self._output_shape)
111  
112      def call(self, inputs, mask=None, training=None):
113          # We must copy for thread safety,
114          # but it only needs to be a shallow copy.
115          kwargs = {k: v for k, v in self.arguments.items()}
116          if self._fn_expects_mask_arg:
117              kwargs["mask"] = mask
118          if self._fn_expects_training_arg:
119              kwargs["training"] = training
120          return self.function(inputs, **kwargs)
121  
122      def compute_mask(self, inputs, mask=None):
123          if callable(self.mask):
124              return self.mask(inputs, mask)
125          return self.mask
126  
127      def get_config(self):
128          config = {
129              "function": self._serialize_function_to_config(self.function),
130          }
131          if self._output_shape is not None:
132              if callable(self._output_shape):
133                  output_shape = self._serialize_function_to_config(
134                      self._output_shape
135                  )
136              else:
137                  output_shape = self._output_shape
138              config["output_shape"] = output_shape
139          if self.mask is not None:
140              if callable(self.mask):
141                  mask = self._serialize_function_to_config(self.mask)
142              else:
143                  mask = serialization_lib.serialize_keras_object(self.mask)
144              config["mask"] = mask
145          config["arguments"] = serialization_lib.serialize_keras_object(
146              self.arguments
147          )
148          base_config = super().get_config()
149          return {**base_config, **config}
150  
151      def _serialize_function_to_config(self, fn):
152          if isinstance(fn, types.LambdaType) and fn.__name__ == "<lambda>":
153              code, defaults, closure = python_utils.func_dump(fn)
154              return {
155                  "class_name": "__lambda__",
156                  "config": {
157                      "code": code,
158                      "defaults": defaults,
159                      "closure": closure,
160                  },
161              }
162          elif callable(fn):
163              return serialization_lib.serialize_keras_object(fn)
164          raise ValueError(
165              "Invalid input type for serialization. "
166              f"Received: {fn} of type {type(fn)}."
167          )
168  
169      @staticmethod
170      def _raise_for_lambda_deserialization(arg_name, safe_mode):
171          if safe_mode:
172              raise ValueError(
173                  "The `{arg_name}` of this `Lambda` layer is a Python lambda. "
174                  "Deserializing it is unsafe. If you trust the source of the "
175                  "config artifact, you can override this error "
176                  "by passing `safe_mode=False` "
177                  "to `from_config()`, or calling "
178                  "`keras.config.enable_unsafe_deserialization()."
179              )
180  
181      @classmethod
182      def from_config(cls, config, custom_objects=None, safe_mode=None):
183          safe_mode = safe_mode or serialization_lib.in_safe_mode()
184          fn_config = config["function"]
185          if (
186              isinstance(fn_config, dict)
187              and "class_name" in fn_config
188              and fn_config["class_name"] == "__lambda__"
189          ):
190              cls._raise_for_lambda_deserialization("function", safe_mode)
191              inner_config = fn_config["config"]
192              fn = python_utils.func_load(
193                  inner_config["code"],
194                  defaults=inner_config["defaults"],
195                  closure=inner_config["closure"],
196              )
197              config["function"] = fn
198          else:
199              config["function"] = serialization_lib.deserialize_keras_object(
200                  fn_config, custom_objects=custom_objects
201              )
202          if "output_shape" in config:
203              fn_config = config["output_shape"]
204              if (
205                  isinstance(fn_config, dict)
206                  and "class_name" in fn_config
207                  and fn_config["class_name"] == "__lambda__"
208              ):
209                  cls._raise_for_lambda_deserialization("function", safe_mode)
210                  inner_config = fn_config["config"]
211                  fn = python_utils.func_load(
212                      inner_config["code"],
213                      defaults=inner_config["defaults"],
214                      closure=inner_config["closure"],
215                  )
216                  config["output_shape"] = fn
217              else:
218                  output_shape = serialization_lib.deserialize_keras_object(
219                      fn_config, custom_objects=custom_objects
220                  )
221                  if isinstance(output_shape, list) and all(
222                      isinstance(e, (int, type(None))) for e in output_shape
223                  ):
224                      output_shape = tuple(output_shape)
225                  config["output_shape"] = output_shape
226  
227          if "arguments" in config:
228              config["arguments"] = serialization_lib.deserialize_keras_object(
229                  config["arguments"], custom_objects=custom_objects
230              )
231          return cls(**config)
