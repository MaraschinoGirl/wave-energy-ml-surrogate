1    """Algorithms to support fitting routines in seaborn plotting functions."""
2    import numpy as np
3    import warnings
4    
5    
6    def bootstrap(*args, **kwargs):
7        """Resample one or more arrays with replacement and store aggregate values. 
8     
9        Positional arguments are a sequence of arrays to bootstrap along the first 
10       axis and pass to a summary function. 
11    
12       Keyword arguments: 
13           n_boot : int, default=10000 
14               Number of iterations 
15           axis : int, default=None 
16               Will pass axis to ``func`` as a keyword argument. 
17           units : array, default=None 
18               Array of sampling unit IDs. When used the bootstrap resamples units 
19               and then observations within units instead of individual 
20               datapoints. 
21           func : string or callable, default="mean" 
22               Function to call on the args that are passed in. If string, uses as 
23               name of function in the numpy namespace. If nans are present in the 
24               data, will try to use nan-aware version of named function. 
25           seed : Generator | SeedSequence | RandomState | int | None 
26               Seed for the random number generator; useful if you want 
27               reproducible resamples. 
28    
29       Returns 
30       ------- 
31       boot_dist: array 
32           array of bootstrapped statistic values 
33    
34       """
35       # Ensure list of arrays are same length
36       if len(np.unique(list(map(len, args)))) > 1:
37           raise ValueError("All input arrays must have the same length")
38       n = len(args[0])
39   
40       # Default keyword arguments
41       n_boot = kwargs.get("n_boot", 10000)
42       func = kwargs.get("func", "mean")
43       axis = kwargs.get("axis", None)
44       units = kwargs.get("units", None)
45       random_seed = kwargs.get("random_seed", None)
46       if random_seed is not None:
47           msg = "`random_seed` has been renamed to `seed` and will be removed"
48           warnings.warn(msg)
49       seed = kwargs.get("seed", random_seed)
50       if axis is None:
51           func_kwargs = dict()
52       else:
53           func_kwargs = dict(axis=axis)
54   
55       # Initialize the resampler
56       if isinstance(seed, np.random.RandomState):
57           rng = seed
58       else:
59           rng = np.random.default_rng(seed)
60   
61       # Coerce to arrays
62       args = list(map(np.asarray, args))
63       if units is not None:
64           units = np.asarray(units)
65   
66       if isinstance(func, str):
67   
68           # Allow named numpy functions
69           f = getattr(np, func)
70   
71           # Try to use nan-aware version of function if necessary
72           missing_data = np.isnan(np.sum(np.column_stack(args)))
73   
74           if missing_data and not func.startswith("nan"):
75               nanf = getattr(np, f"nan{func}", None)
76               if nanf is None:
77                   msg = f"Data contain nans but no nan-aware version of `{func}` found"
78                   warnings.warn(msg, UserWarning)
79               else:
80                   f = nanf
81   
82       else:
83           f = func
84   
85       # Handle numpy changes
86       try:
87           integers = rng.integers
88       except AttributeError:
89           integers = rng.randint
90   
91       # Do the bootstrap
92       if units is not None:
93           return _structured_bootstrap(args, n_boot, units, f,
94                                        func_kwargs, integers)
95   
96       boot_dist = []
97       for i in range(int(n_boot)):
98           resampler = integers(0, n, n, dtype=np.intp)  # intp is indexing dtype
99           sample = [a.take(resampler, axis=0) for a in args]
100          boot_dist.append(f(*sample, **func_kwargs))
101      return np.array(boot_dist)

104  def _structured_bootstrap(args, n_boot, units, func, func_kwargs, integers):
105      """Resample units instead of datapoints."""
106      unique_units = np.unique(units)
107      n_units = len(unique_units)
108  
109      args = [[a[units == unit] for unit in unique_units] for a in args]
110  
111      boot_dist = []
112      for i in range(int(n_boot)):
113          resampler = integers(0, n_units, n_units, dtype=np.intp)
114          sample = [[a[i] for i in resampler] for a in args]
115          lengths = map(len, sample[0])
116          resampler = [integers(0, n, n, dtype=np.intp) for n in lengths]
117          sample = [[c.take(r, axis=0) for c, r in zip(a, resampler)] for a in sample]
118          sample = list(map(np.concatenate, sample))
119          boot_dist.append(func(*sample, **func_kwargs))
120      return np.array(boot_dist)
