1    import pytest

7    
8    @pytest.mark.parametrize("csr_container", CSR_CONTAINERS)

10       random_state = check_random_state(0)
11       X = random_state.rand(5, 4)
12       X_csr = csr_container(X)
13       mask = [False, False, True, True, True]
14   
15       mask = safe_mask(X, mask)
16       assert X[mask].shape[0] == 3
17   
18       mask = safe_mask(X_csr, mask)
19       assert X_csr[mask].shape[0] == 3
