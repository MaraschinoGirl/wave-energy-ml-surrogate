70   class Pattern(object):
71       """ 
72       Base class for search patterns. 
73    
74       The following L{Pattern} subclasses are provided by WinAppDbg: 
75        - L{BytePattern} 
76        - L{TextPattern} 
77        - L{RegExpPattern} 
78        - L{HexPattern} 
79    
80       @see: L{Search.search_process} 
81       """
82   
83       def __init__(self, pattern):
84           """ 
85           Class constructor. 
86    
87           The only mandatory argument should be the pattern string. 
88    
89           This method B{MUST} be reimplemented by subclasses of L{Pattern}. 
90           """
91           raise NotImplementedError()
92   
93       def __len__(self):
94           """ 
95           Returns the maximum expected length of the strings matched by this 
96           pattern. Exact behavior is implementation dependent. 
97    
98           Ideally it should be an exact value, but in some cases it's not 
99           possible to calculate so an upper limit should be returned instead. 
100   
101          If that's not possible either an exception must be raised. 
102   
103          This value will be used to calculate the required buffer size when 
104          doing buffered searches. 
105   
106          This method B{MUST} be reimplemented by subclasses of L{Pattern}. 
107          """
108          raise NotImplementedError()
109  
110      def read(self, process, address, size):
111          """ 
112          Reads the requested number of bytes from the process memory at the 
113          given address. 
114   
115          Subclasses of L{Pattern} tipically don't need to reimplement this 
116          method. 
117          """
118          return process.read(address, size)
119  
120      def find(self, buffer, pos=None):
121          """ 
122          Searches for the pattern in the given buffer, optionally starting at 
123          the given position within the buffer. 
124   
125          This method B{MUST} be reimplemented by subclasses of L{Pattern}. 
126   
127          @type  buffer: str 
128          @param buffer: Buffer to search on. 
129   
130          @type  pos: int 
131          @param pos: 
132              (Optional) Position within the buffer to start searching from. 
133   
134          @rtype:  tuple( int, int ) 
135          @return: Tuple containing the following: 
136               - Position within the buffer where a match is found, or C{-1} if 
137                 no match was found. 
138               - Length of the matched data if a match is found, or undefined if 
139                 no match was found. 
140          """
141          raise NotImplementedError()
142  
143      def found(self, address, size, data):
144          """ 
145          This method gets called when a match is found. 
146   
147          This allows subclasses of L{Pattern} to filter out unwanted results, 
148          or modify the results before giving them to the caller of 
149          L{Search.search_process}. 
150   
151          If the return value is C{None} the result is skipped. 
152   
153          Subclasses of L{Pattern} don't need to reimplement this method unless 
154          filtering is needed. 
155   
156          @type  address: int 
157          @param address: The memory address where the pattern was found. 
158   
159          @type  size: int 
160          @param size: The size of the data that matches the pattern. 
161   
162          @type  data: str 
163          @param data: The data that matches the pattern. 
164   
165          @rtype:  tuple( int, int, str ) 
166          @return: Tuple containing the following: 
167               * The memory address where the pattern was found. 
168               * The size of the data that matches the pattern. 
169               * The data that matches the pattern. 
170          """
171          return (address, size, data)

177  class BytePattern(Pattern):
178      """ 
179      Fixed byte pattern. 
180   
181      @type pattern: str 
182      @ivar pattern: Byte string to search for. 
183   
184      @type length: int 
185      @ivar length: Length of the byte pattern. 
186      """
187  
188      def __init__(self, pattern):
189          """ 
190          @type  pattern: str 
191          @param pattern: Byte string to search for. 
192          """
193          self.pattern = str(pattern)
194          self.length = len(pattern)
195  
196      def __len__(self):
197          """ 
198          Returns the exact length of the pattern. 
199   
200          @see: L{Pattern.__len__} 
201          """
202          return self.length
203  
204      def find(self, buffer, pos=None):
205          return buffer.find(self.pattern, pos), self.length

213  class TextPattern(BytePattern):
214      """ 
215      Text pattern. 
216   
217      @type isUnicode: bool 
218      @ivar isUnicode: C{True} if the text to search for is a compat.unicode string, 
219          C{False} otherwise. 
220   
221      @type encoding: str 
222      @ivar encoding: Encoding for the text parameter. 
223          Only used when the text to search for is a Unicode string. 
224          Don't change unless you know what you're doing! 
225   
226      @type caseSensitive: bool 
227      @ivar caseSensitive: C{True} of the search is case sensitive, 
228          C{False} otherwise. 
229      """
230  
231      def __init__(self, text, encoding="utf-16le", caseSensitive=False):
232          """ 
233          @type  text: str or compat.unicode 
234          @param text: Text to search for. 
235   
236          @type  encoding: str 
237          @param encoding: (Optional) Encoding for the text parameter. 
238              Only used when the text to search for is a Unicode string. 
239              Don't change unless you know what you're doing! 
240   
241          @type  caseSensitive: bool 
242          @param caseSensitive: C{True} of the search is case sensitive, 
243              C{False} otherwise. 
244          """
245          self.isUnicode = isinstance(text, compat.unicode)
246          self.encoding = encoding
247          self.caseSensitive = caseSensitive
248          if not self.caseSensitive:
249              pattern = text.lower()
250          if self.isUnicode:
251              pattern = text.encode(encoding)
252          super(TextPattern, self).__init__(pattern)
253  
254      def read(self, process, address, size):
255          data = super(TextPattern, self).read(address, size)
256          if not self.caseSensitive:
257              if self.isUnicode:
258                  try:
259                      encoding = self.encoding
260                      text = data.decode(encoding, "replace")
261                      text = text.lower()
262                      new_data = text.encode(encoding, "replace")
263                      if len(data) == len(new_data):
264                          data = new_data
265                      else:
266                          data = data.lower()
267                  except Exception:
268                      data = data.lower()
269              else:
270                  data = data.lower()
271          return data
272  
273      def found(self, address, size, data):
274          if self.isUnicode:
275              try:
276                  data = compat.unicode(data, self.encoding)
277              except Exception:
278                  ##                traceback.print_exc()    # XXX DEBUG
279                  return None
280          return (address, size, data)

286  class RegExpPattern(Pattern):
287      """ 
288      Regular expression pattern. 
289   
290      @type pattern: str 
291      @ivar pattern: Regular expression in text form. 
292   
293      @type flags: int 
294      @ivar flags: Regular expression flags. 
295   
296      @type regexp: re.compile 
297      @ivar regexp: Regular expression in compiled form. 
298   
299      @type maxLength: int 
300      @ivar maxLength: 
301          Maximum expected length of the strings matched by this regular 
302          expression. 
303   
304          This value will be used to calculate the required buffer size when 
305          doing buffered searches. 
306   
307          Ideally it should be an exact value, but in some cases it's not 
308          possible to calculate so an upper limit should be given instead. 
309   
310          If that's not possible either, C{None} should be used. That will 
311          cause an exception to be raised if this pattern is used in a 
312          buffered search. 
313      """
314  
315      def __init__(self, regexp, flags=0, maxLength=None):
316          """ 
317          @type  regexp: str 
318          @param regexp: Regular expression string. 
319   
320          @type  flags: int 
321          @param flags: Regular expression flags. 
322   
323          @type  maxLength: int 
324          @param maxLength: Maximum expected length of the strings matched by 
325              this regular expression. 
326   
327              This value will be used to calculate the required buffer size when 
328              doing buffered searches. 
329   
330              Ideally it should be an exact value, but in some cases it's not 
331              possible to calculate so an upper limit should be given instead. 
332   
333              If that's not possible either, C{None} should be used. That will 
334              cause an exception to be raised if this pattern is used in a 
335              buffered search. 
336          """
337          self.pattern = regexp
338          self.flags = flags
339          self.regexp = re.compile(regexp, flags)
340          self.maxLength = maxLength
341  
342      def __len__(self):
343          """ 
344          Returns the maximum expected length of the strings matched by this 
345          pattern. This value is taken from the C{maxLength} argument of the 
346          constructor if this class. 
347   
348          Ideally it should be an exact value, but in some cases it's not 
349          possible to calculate so an upper limit should be returned instead. 
350   
351          If that's not possible either an exception must be raised. 
352   
353          This value will be used to calculate the required buffer size when 
354          doing buffered searches. 
355          """
356          if self.maxLength is None:
357              raise NotImplementedError()
358          return self.maxLength
359  
360      def find(self, buffer, pos=None):
361          if not pos:  # make sure pos is an int
362              pos = 0
363          match = self.regexp.search(buffer, pos)
364          if match:
365              start, end = match.span()
366              return start, end - start
367          return -1, 0

373  class HexPattern(RegExpPattern):
374      """ 
375      Hexadecimal pattern. 
376   
377      Hex patterns must be in this form:: 
378          "68 65 6c 6c 6f 20 77 6f 72 6c 64"  # "hello world" 
379   
380      Spaces are optional. Capitalization of hex digits doesn't matter. 
381      This is exactly equivalent to the previous example:: 
382          "68656C6C6F20776F726C64"            # "hello world" 
383   
384      Wildcards are allowed, in the form of a C{?} sign in any hex digit:: 
385          "5? 5? c3"          # pop register / pop register / ret 
386          "b8 ?? ?? ?? ??"    # mov eax, immediate value 
387   
388      @type pattern: str 
389      @ivar pattern: Hexadecimal pattern. 
390      """
391  
392      def __new__(cls, pattern):
393          """ 
394          If the pattern is completely static (no wildcards are present) a 
395          L{BytePattern} is created instead. That's because searching for a 
396          fixed byte pattern is faster than searching for a regular expression. 
397          """
398          if "?" not in pattern:
399              return BytePattern(HexInput.hexadecimal(pattern))
400          return object.__new__(cls, pattern)
401  
402      def __init__(self, hexa):
403          """ 
404          Hex patterns must be in this form:: 
405              "68 65 6c 6c 6f 20 77 6f 72 6c 64"  # "hello world" 
406   
407          Spaces are optional. Capitalization of hex digits doesn't matter. 
408          This is exactly equivalent to the previous example:: 
409              "68656C6C6F20776F726C64"            # "hello world" 
410   
411          Wildcards are allowed, in the form of a C{?} sign in any hex digit:: 
412              "5? 5? c3"          # pop register / pop register / ret 
413              "b8 ?? ?? ?? ??"    # mov eax, immediate value 
414   
415          @type  hexa: str 
416          @param hexa: Pattern to search for. 
417          """
418          maxLength = len([x for x in hexa if x in "?0123456789ABCDEFabcdef"]) / 2
419          super(HexPattern, self).__init__(HexInput.pattern(hexa), maxLength=maxLength)

437      @staticmethod
438      def search_process(process, pattern, minAddr=None, maxAddr=None, bufferPages=None, overlapping=False):
439          """ 
440          Search for the given pattern within the process memory. 
441   
442          @type  process: L{Process} 
443          @param process: Process to search. 
444   
445          @type  pattern: L{Pattern} 
446          @param pattern: Pattern to search for. 
447              It must be an instance of a subclass of L{Pattern}. 
448   
449              The following L{Pattern} subclasses are provided by WinAppDbg: 
450               - L{BytePattern} 
451               - L{TextPattern} 
452               - L{RegExpPattern} 
453               - L{HexPattern} 
454   
455              You can also write your own subclass of L{Pattern} for customized 
456              searches. 
457   
458          @type  minAddr: int 
459          @param minAddr: (Optional) Start the search at this memory address. 
460   
461          @type  maxAddr: int 
462          @param maxAddr: (Optional) Stop the search at this memory address. 
463   
464          @type  bufferPages: int 
465          @param bufferPages: (Optional) Number of memory pages to buffer when 
466              performing the search. Valid values are: 
467               - C{0} or C{None}: 
468                 Automatically determine the required buffer size. May not give 
469                 complete results for regular expressions that match variable 
470                 sized strings. 
471               - C{> 0}: Set the buffer size, in memory pages. 
472               - C{< 0}: Disable buffering entirely. This may give you a little 
473                 speed gain at the cost of an increased memory usage. If the 
474                 target process has very large contiguous memory regions it may 
475                 actually be slower or even fail. It's also the only way to 
476                 guarantee complete results for regular expressions that match 
477                 variable sized strings. 
478   
479          @type  overlapping: bool 
480          @param overlapping: C{True} to allow overlapping results, C{False} 
481              otherwise. 
482   
483              Overlapping results yield the maximum possible number of results. 
484   
485              For example, if searching for "AAAA" within "AAAAAAAA" at address 
486              C{0x10000}, when overlapping is turned off the following matches 
487              are yielded:: 
488                  (0x10000, 4, "AAAA") 
489                  (0x10004, 4, "AAAA") 
490   
491              If overlapping is turned on, the following matches are yielded:: 
492                  (0x10000, 4, "AAAA") 
493                  (0x10001, 4, "AAAA") 
494                  (0x10002, 4, "AAAA") 
495                  (0x10003, 4, "AAAA") 
496                  (0x10004, 4, "AAAA") 
497   
498              As you can see, the middle results are overlapping the last two. 
499   
500          @rtype:  iterator of tuple( int, int, str ) 
501          @return: An iterator of tuples. Each tuple contains the following: 
502               - The memory address where the pattern was found. 
503               - The size of the data that matches the pattern. 
504               - The data that matches the pattern. 
505   
506          @raise WindowsError: An error occurred when querying or reading the 
507              process memory. 
508          """
509  
510          # Do some namespace lookups of symbols we'll be using frequently.
511          MEM_COMMIT = win32.MEM_COMMIT
512          PAGE_GUARD = win32.PAGE_GUARD
513          page = MemoryAddresses.pageSize
514          read = pattern.read
515          find = pattern.find
516  
517          # Calculate the address range.
518          if minAddr is None:
519              minAddr = 0
520          if maxAddr is None:
521              maxAddr = win32.LPVOID(-1).value  # XXX HACK
522  
523          # Calculate the buffer size from the number of pages.
524          if bufferPages is None:
525              try:
526                  size = MemoryAddresses.align_address_to_page_end(len(pattern)) + page
527              except NotImplementedError:
528                  size = None
529          elif bufferPages > 0:
530              size = page * (bufferPages + 1)
531          else:
532              size = None
533  
534          # Get the memory map of the process.
535          memory_map = process.iter_memory_map(minAddr, maxAddr)
536  
537          # Perform search with buffering enabled.
538          if size:
539              # Loop through all memory blocks containing data.
540              buffer = ""  # buffer to hold the memory data
541              prev_addr = 0  # previous memory block address
542              last = 0  # position of the last match
543              delta = 0  # delta of last read address and start of buffer
544              for mbi in memory_map:
545                  # Skip blocks with no data to search on.
546                  if not mbi.has_content():
547                      continue
548  
549                  # Get the address and size of this block.
550                  address = mbi.BaseAddress  # current address to search on
551                  block_size = mbi.RegionSize  # total size of the block
552                  if address >= maxAddr:
553                      break
554                  end = address + block_size  # end address of the block
555  
556                  # If the block is contiguous to the previous block,
557                  # coalesce the new data in the buffer.
558                  if delta and address == prev_addr:
559                      buffer += read(process, address, page)
560  
561                  # If not, clear the buffer and read new data.
562                  else:
563                      buffer = read(process, address, min(size, block_size))
564                      last = 0
565                      delta = 0
566  
567                  # Search for the pattern in this block.
568                  while 1:
569                      # Yield each match of the pattern in the buffer.
570                      pos, length = find(buffer, last)
571                      while pos >= last:
572                          match_addr = address + pos - delta
573                          if minAddr <= match_addr < maxAddr:
574                              result = pattern.found(match_addr, length, buffer[pos : pos + length])
575                              if result is not None:
576                                  yield result
577                          if overlapping:
578                              last = pos + 1
579                          else:
580                              last = pos + length
581                          pos, length = find(buffer, last)
582  
583                      # Advance to the next page.
584                      address = address + page
585                      block_size = block_size - page
586                      prev_addr = address
587  
588                      # Fix the position of the last match.
589                      last = last - page
590                      if last < 0:
591                          last = 0
592  
593                      # Remove the first page in the buffer.
594                      buffer = buffer[page:]
595                      delta = page
596  
597                      # If we haven't reached the end of the block yet,
598                      # read the next page in the block and keep seaching.
599                      if address < end:
600                          buffer = buffer + read(process, address, page)
601  
602                      # Otherwise, we're done searching this block.
603                      else:
604                          break
605  
606          # Perform search with buffering disabled.
607          else:
608              # Loop through all memory blocks containing data.
609              for mbi in memory_map:
610                  # Skip blocks with no data to search on.
611                  if not mbi.has_content():
612                      continue
613  
614                  # Get the address and size of this block.
615                  address = mbi.BaseAddress
616                  block_size = mbi.RegionSize
617                  if address >= maxAddr:
618                      break
619  
620                  # Read the whole memory region.
621                  buffer = process.read(address, block_size)
622  
623                  # Search for the pattern in this region.
624                  pos, length = find(buffer)
625                  last = 0
626                  while pos >= last:
627                      match_addr = address + pos
628                      if minAddr <= match_addr < maxAddr:
629                          result = pattern.found(match_addr, length, buffer[pos : pos + length])
630                          if result is not None:
631                              yield result
632                      if overlapping:
633                          last = pos + 1
634                      else:
635                          last = pos + length
636                      pos, length = find(buffer, last)

638      @classmethod
639      def extract_ascii_strings(cls, process, minSize=4, maxSize=1024):
640          """ 
641          Extract ASCII strings from the process memory. 
642   
643          @type  process: L{Process} 
644          @param process: Process to search. 
645   
646          @type  minSize: int 
647          @param minSize: (Optional) Minimum size of the strings to search for. 
648   
649          @type  maxSize: int 
650          @param maxSize: (Optional) Maximum size of the strings to search for. 
651   
652          @rtype:  iterator of tuple(int, int, str) 
653          @return: Iterator of strings extracted from the process memory. 
654              Each tuple contains the following: 
655               - The memory address where the string was found. 
656               - The size of the string. 
657               - The string. 
658          """
659          regexp = r"[\s\w\!\@\#\$\%%\^\&\*\(\)\{\}\[\]\~\`\'\"\:\;\.\,\\\/\-\+\=\_\<\>]{%d,%d}\0" % (minSize, maxSize)
660          pattern = RegExpPattern(regexp, 0, maxSize)
661          return cls.search_process(process, pattern, overlapping=False)
