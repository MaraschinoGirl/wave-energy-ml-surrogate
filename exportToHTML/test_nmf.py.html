81   def test_initialize_close():
82       # Test NNDSVD error
83       # Test that _initialize_nmf error is less than the standard deviation of
84       # the entries in the matrix.
85       rng = np.random.mtrand.RandomState(42)
86       A = np.abs(rng.randn(10, 10))
87       W, H = nmf._initialize_nmf(A, 10, init="nndsvd")

152  def test_nmf_true_reconstruction():
153      # Test that the fit is not too far away from an exact solution
154      # (by construction)
155      n_samples = 15
156      n_features = 10
157      n_components = 5
158      beta_loss = 1
159      batch_size = 3
160      max_iter = 1000
161  
162      rng = np.random.mtrand.RandomState(42)
163      W_true = np.zeros([n_samples, n_components])
164      W_array = np.abs(rng.randn(n_samples))
165      for j in range(n_components):
166          W_true[j % n_samples, j] = W_array[j % n_samples]
167      H_true = np.zeros([n_components, n_features])
168      H_array = np.abs(rng.randn(n_components))
169      for j in range(n_features):
170          H_true[j % n_components, j] = H_array[j % n_components]
171      X = np.dot(W_true, H_true)
172  
173      model = NMF(
174          n_components=n_components,
175          solver="mu",
176          beta_loss=beta_loss,
177          max_iter=max_iter,
178          random_state=0,
179      )
180      transf = model.fit_transform(X)
181      X_calc = np.dot(transf, model.components_)
182  
183      assert model.reconstruction_err_ < 0.1
184      assert_allclose(X, X_calc)
185  
186      mbmodel = MiniBatchNMF(
187          n_components=n_components,
188          beta_loss=beta_loss,
189          batch_size=batch_size,
190          random_state=0,
191          max_iter=max_iter,
192      )
193      transf = mbmodel.fit_transform(X)
194      X_calc = np.dot(transf, mbmodel.components_)
195  
196      assert mbmodel.reconstruction_err_ < 0.1
197      assert_allclose(X, X_calc, atol=1)

419  def _beta_divergence_dense(X, W, H, beta):
420      """Compute the beta-divergence of X and W.H for dense array only. 
421   
422      Used as a reference for testing nmf._beta_divergence. 
423      """
424      WH = np.dot(W, H)
425  
426      if beta == 2:
427          return squared_norm(X - WH) / 2
428  
429      WH_Xnonzero = WH[X != 0]
430      X_nonzero = X[X != 0]
431      np.maximum(WH_Xnonzero, 1e-9, out=WH_Xnonzero)
432  
433      if beta == 1:
434          res = np.sum(X_nonzero * np.log(X_nonzero / WH_Xnonzero))
435          res += WH.sum() - X.sum()
436  
437      elif beta == 0:
438          div = X_nonzero / WH_Xnonzero
439          res = np.sum(div) - X.size - np.sum(np.log(div))
440      else:
441          res = (X_nonzero**beta).sum()
442          res += (beta - 1) * (WH**beta).sum()
443          res -= beta * (X_nonzero * (WH_Xnonzero ** (beta - 1))).sum()
444          res /= beta * (beta - 1)
445  
446      return res

451      # Compare _beta_divergence with the reference _beta_divergence_dense
452      n_samples = 20
453      n_features = 10
454      n_components = 5
455      beta_losses = [0.0, 0.5, 1.0, 1.5, 2.0, 3.0]

463  
464      for beta in beta_losses:

502  @pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
503  def test_nmf_multiplicative_update_sparse(csr_container):
504      # Compare sparse and dense input in multiplicative update NMF
505      # Also test continuity of the results with respect to beta_loss parameter
506      n_samples = 20
507      n_features = 10
508      n_components = 5
509      alpha = 0.1
510      l1_ratio = 0.5
511      n_iter = 20
512  
513      # initialization
514      rng = np.random.mtrand.RandomState(1337)
515      X = rng.randn(n_samples, n_features)
516      X = np.abs(X)
517      X_csr = csr_container(X)
518      W0, H0 = nmf._initialize_nmf(X, n_components, init="random", random_state=42)
519  
520      for beta_loss in (-1.2, 0, 0.2, 1.0, 2.0, 2.5):
521          # Reference with dense array X
522          W, H = W0.copy(), H0.copy()
523          W1, H1, _ = non_negative_factorization(
524              X,
525              W,
526              H,
527              n_components,
528              init="custom",
529              update_H=True,
530              solver="mu",
531              beta_loss=beta_loss,
532              max_iter=n_iter,
533              alpha_W=alpha,
534              l1_ratio=l1_ratio,
535              random_state=42,
536          )
537  
538          # Compare with sparse X
539          W, H = W0.copy(), H0.copy()
540          W2, H2, _ = non_negative_factorization(
541              X_csr,
542              W,
543              H,
544              n_components,
545              init="custom",
546              update_H=True,
547              solver="mu",
548              beta_loss=beta_loss,
549              max_iter=n_iter,
550              alpha_W=alpha,
551              l1_ratio=l1_ratio,
552              random_state=42,
553          )
554  
555          assert_allclose(W1, W2, atol=1e-7)
556          assert_allclose(H1, H2, atol=1e-7)
557  
558          # Compare with almost same beta_loss, since some values have a specific
559          # behavior, but the results should be continuous w.r.t beta_loss
560          beta_loss -= 1.0e-5
561          W, H = W0.copy(), H0.copy()
562          W3, H3, _ = non_negative_factorization(
563              X_csr,
564              W,
565              H,
566              n_components,
567              init="custom",
568              update_H=True,
569              solver="mu",
570              beta_loss=beta_loss,
571              max_iter=n_iter,
572              alpha_W=alpha,
573              l1_ratio=l1_ratio,
574              random_state=42,
575          )

636  def test_nmf_regularization(Estimator, solver):
637      # Test the effect of L1 and L2 regularizations
638      n_samples = 6
639      n_features = 5
640      n_components = 3
641      rng = np.random.mtrand.RandomState(42)
642      X = np.abs(rng.randn(n_samples, n_features))
643  
644      # L1 regularization should increase the number of zeros
645      l1_ratio = 1.0
646      regul = Estimator(
647          n_components=n_components,
648          alpha_W=0.5,
649          l1_ratio=l1_ratio,
650          random_state=42,
651          **solver,
652      )
653      model = Estimator(
654          n_components=n_components,
655          alpha_W=0.0,
656          l1_ratio=l1_ratio,
657          random_state=42,
658          **solver,
659      )
660  
661      W_regul = regul.fit_transform(X)
662      W_model = model.fit_transform(X)
663  
664      H_regul = regul.components_
665      H_model = model.components_
666  
667      eps = np.finfo(np.float64).eps
668      W_regul_n_zeros = W_regul[W_regul <= eps].size
669      W_model_n_zeros = W_model[W_model <= eps].size
670      H_regul_n_zeros = H_regul[H_regul <= eps].size
671      H_model_n_zeros = H_model[H_model <= eps].size
672  
673      assert W_regul_n_zeros > W_model_n_zeros
674      assert H_regul_n_zeros > H_model_n_zeros
675  
676      # L2 regularization should decrease the sum of the squared norm
677      # of the matrices W and H
678      l1_ratio = 0.0
679      regul = Estimator(
680          n_components=n_components,
681          alpha_W=0.5,
682          l1_ratio=l1_ratio,
683          random_state=42,
684          **solver,
685      )
686      model = Estimator(
687          n_components=n_components,
688          alpha_W=0.0,
689          l1_ratio=l1_ratio,
690          random_state=42,
691          **solver,
692      )
693  
694      W_regul = regul.fit_transform(X)
695      W_model = model.fit_transform(X)
696  
697      H_regul = regul.components_
698      H_model = model.components_
699  
700      assert (linalg.norm(W_model)) ** 2.0 + (linalg.norm(H_model)) ** 2.0 > (
701          linalg.norm(W_regul)
702      ) ** 2.0 + (linalg.norm(H_regul)) ** 2.0

705  @pytest.mark.filterwarnings("ignore::sklearn.exceptions.ConvergenceWarning")
706  @pytest.mark.parametrize("solver", ("cd", "mu"))
707  def test_nmf_decreasing(solver):
708      # test that the objective function is decreasing at each iteration
709      n_samples = 20
710      n_features = 15
711      n_components = 10
712      alpha = 0.1
713      l1_ratio = 0.5
714      tol = 0.0
715  
716      # initialization
717      rng = np.random.mtrand.RandomState(42)
718      X = rng.randn(n_samples, n_features)
719      np.abs(X, X)
720      W0, H0 = nmf._initialize_nmf(X, n_components, init="random", random_state=42)
721  
722      for beta_loss in (-1.2, 0, 0.2, 1.0, 2.0, 2.5):
723          if solver != "mu" and beta_loss != 2:
724              # not implemented
725              continue
726          W, H = W0.copy(), H0.copy()
727          previous_loss = None
728          for _ in range(30):
729              # one more iteration starting from the previous results
730              W, H, _ = non_negative_factorization(
731                  X,
732                  W,
733                  H,
734                  beta_loss=beta_loss,
735                  init="custom",
736                  n_components=n_components,
737                  max_iter=1,
738                  alpha_W=alpha,
739                  solver=solver,
740                  tol=tol,
741                  l1_ratio=l1_ratio,
742                  verbose=0,
743                  random_state=0,
744                  update_H=True,
745              )
746  
747              loss = (
748                  nmf._beta_divergence(X, W, H, beta_loss)
749                  + alpha * l1_ratio * n_features * W.sum()
750                  + alpha * l1_ratio * n_samples * H.sum()
751                  + alpha * (1 - l1_ratio) * n_features * (W**2).sum()
752                  + alpha * (1 - l1_ratio) * n_samples * (H**2).sum()
753              )
754              if previous_loss is not None:
755                  assert previous_loss > loss
756              previous_loss = loss
