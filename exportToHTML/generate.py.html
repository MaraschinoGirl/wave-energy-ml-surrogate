15   def generate_api_files(package, code_directory="src", verbose=False, target_directory=None):
16       """Writes out API export `__init__.py` files. 
17    
18       Given a codebase structured as such: 
19    
20       ``` 
21       package/ 
22       ...src/ 
23       ......__init__.py 
24       ......(Python files that use e.g. `@export_api(package="package", export_path="package.x.y.Z")`) 
25       ``` 
26    
27       this script generates `__init__.py` files within `package/` 
28       to export the public API described by the `@api_export` calls. 
29    
30       Important notes: 
31    
32       * Any existing `__init__.py` files in `package/` but outside of 
33           `package/code_directory/` may be overwritten. 
34       * This script must be run in an environment that includes 
35           all dependencies used by `package`. Make sure to install 
36           them before running the script. 
37       """
38       if verbose:
39           print(
40               f"Generating files for package '{package}' "
41               f"from sources found in '{package}/{code_directory}'."
42           )
43   
44       if not os.path.exists(package):
45           raise ValueError(f"No directory named '{package}'.")
46       if not os.path.exists(os.path.join(package, code_directory)):
47           raise ValueError(f"No directory named '{package}/{code_directory}'.")
48   
49       # Make list of all Python files (modules) to visit.
50       codebase_walk_entry_points = []
51       for root, _, files in os.walk(os.path.join(package, code_directory)):
52           for fname in files:
53               if fname == "__init__.py":
54                   codebase_walk_entry_points.append(".".join(root.split("/")))
55               elif fname.endswith(".py") and not fname.endswith("_test.py"):
56                   module_name = fname[:-3]
57                   codebase_walk_entry_points.append(
58                       ".".join(root.split("/")) + "." + module_name
59                   )
60   
61       # Import all Python modules found in the code directory.
62       sys.path.insert(0, os.getcwd())
63       modules = []
64       for entry_point in codebase_walk_entry_points:
65           mod = importlib.import_module(entry_point, package=".")
66           modules.append(mod)
67   
68       if verbose:
69           print("Compiling list of symbols to export.")
70   
71       # Populate list of all symbols to register.
72       all_symbols = set()
73       for module in modules:
74           for name in dir(module):
75               symbol = getattr(module, name)
76               if not hasattr(symbol, "_api_export_path"):
77                   continue
78               if symbol._api_export_symbol_id != id(symbol):
79                   # This symbol is a non-exported subclass
80                   # of an exported symbol.
81                   continue
82               if not all(
83                   [
84                       path.startswith(package + ".")
85                       for path in to_list(symbol._api_export_path)
86                   ]
87               ):
88                   continue
89               all_symbols.add(symbol)
90   
91       # Generate __init__ files content.
92       init_files_content = {}
93       for symbol in all_symbols:
94           if verbose:
95               print(f"...processing symbol '{symbol.__name__}'")
96           for export_path in to_list(symbol._api_export_path):
97               export_modules = export_path.split(".")
98               if export_modules[0] == package and target_directory is not None:
99                   export_modules = [export_modules[0], target_directory] + export_modules[1:]
100              export_name = export_modules[-1]
101              parent_path = os.path.join(*export_modules[:-1])
102              if parent_path not in init_files_content:
103                  init_files_content[parent_path] = []
104              init_files_content[parent_path].append(
105                  {"symbol": symbol, "export_name": export_name}
106              )
107              for i in range(1, len(export_modules[:-1])):
108                  intermediate_path = os.path.join(*export_modules[:i])
109                  if intermediate_path not in init_files_content:
110                      init_files_content[intermediate_path] = []
111                  init_files_content[intermediate_path].append(
112                      {
113                          "module": export_modules[i],
114                          "location": ".".join(export_modules[:i]),
115                      }
116                  )
117  
118      if verbose:
119          print("Writing out API files.")
120  
121      # Go over init_files_content, make dirs,
122      # create __init__.py file, populate file with public symbol imports.
123      for path, contents in init_files_content.items():
124          os.makedirs(path, exist_ok=True)
125          init_file_lines = []
126          modules_included = set()
127          for symbol_metadata in contents:
128              if "symbol" in symbol_metadata:
129                  symbol = symbol_metadata["symbol"]
130                  name = symbol_metadata["export_name"]
131                  if name == symbol.__name__:
132                      init_file_lines.append(
133                          f"from {symbol.__module__} import {name}"
134                      )
135                  else:
136                      init_file_lines.append(
137                          f"from {symbol.__module__} import {symbol.__name__} as {name}"
138                      )
139              elif "module" in symbol_metadata:
140                  if symbol_metadata["module"] not in modules_included:
141                      init_file_lines.append(
142                          f"from {'.'.join(path.split('/'))} import {symbol_metadata['module']}"
143                      )
144                      modules_included.add(symbol_metadata["module"])
145  
146          init_path = os.path.join(path, "__init__.py")
147          if verbose:
148              print(f"...writing {init_path}")
149          init_file_lines = sorted(init_file_lines)
150          with open(init_path, "w") as f:
151              contents = INIT_FILE_HEADER + "\n".join(init_file_lines) + "\n"
152              f.write(contents)

155  def to_list(x):
156      if isinstance(x, (list, tuple)):
157          return list(x)
158      elif isinstance(x, str):
159          return [x]
160      return []
