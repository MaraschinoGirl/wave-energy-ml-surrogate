6    const playwright = require('playwright');
7    const path = require('path');
8    const fs = require('fs');
9    
10   // eslint-disable-next-line no-redeclare
11   const URL = process.argv[2];
12   const BROWSER_VAR = 'JLAB_BROWSER_TYPE';
13   const BROWSER = process.env[BROWSER_VAR] || 'chromium';
14   const HEADLESS_VAR = 'JLAB_BROWSER_HEADLESS';
15   const HEADLESS = process.env[HEADLESS_VAR] === 'false' ? false : true;
16   const OUTPUT_VAR = 'JLAB_BROWSER_CHECK_OUTPUT';
17   const OUTPUT = process.env[OUTPUT_VAR];
18   
19   let nextScreenshot = 0;
20   const screenshotStem = `screenshot-${+new Date()}`;
21   
22   if (OUTPUT) {
23     console.log(`Screenshots will be saved in ${OUTPUT}...`);
24     if (!fs.existsSync(OUTPUT)) {
25       console.log(`Creating ${OUTPUT}...`);
26       fs.mkdirSync(OUTPUT, { recursive: true });
27     }
28   }
29   
30   async function main() {
31     /* eslint-disable no-console */
32     console.info(`Starting headless ${BROWSER}...`);
33     let testError = null;
34   
35     const pwBrowser = playwright[BROWSER];
36     const browser = await pwBrowser.launch({
37       headless: HEADLESS,
38       logger: {
39         isEnabled: () => !!OUTPUT,
40         log: (name, severity, message, args) => console.log(name, message)
41       }
42     });
43   
44     const context = await browser.newContext();
45     const page = await context.newPage();
46   
47     async function screenshot() {
48       if (!OUTPUT) {
49         return;
50       }
51       const screenshotPath = path.join(
52         OUTPUT,
53         `${screenshotStem}-${++nextScreenshot}.png`
54       );
55       console.log(`Capturing screenshot ${screenshotPath}...`);
56       await page.screenshot({
57         type: 'png',
58         path: screenshotPath
59       });
60     }
61   
62     console.info('Navigating to page:', URL);
63     await page.goto(URL);
64     console.info('Waiting for page to load...');
65   
66     // Wait for the local file to redirect on notebook >= 6.0
67     await page.waitForNavigation();
68   
69     console.log('Waiting for page content..');
70   
71     try {
72       await page.locator('#jupyter-config-data').waitFor({ state: 'attached' });
73     } catch (reason) {
74       console.error('Error loading JupyterLab page:', reason);
75       // Limit to 1000 characters
76       console.error((await page.content()).substring(0, 1000));
77       testError = reason;
78     }
79   
80     console.log('Waiting for #main selector...');
81     await page.waitForSelector('#main', { timeout: 100000 });
82   
83     console.log('Waiting for #browserTest selector...');
84     const el = await page.waitForSelector('#browserTest', {
85       timeout: 100000,
86       state: 'attached'
87     });
88     console.log('Waiting for application to start...');
89   
90     try {
91       await page.waitForSelector('.completed', { state: 'attached' });
92     } catch (e) {
93       testError = e;
94     }
95   
96     await screenshot();
97   
98     const textContent = await el.getProperty('textContent');
99     const errors = JSON.parse(await textContent.jsonValue());
100  
101    for (let error of errors) {
102      console.error(`Parsed an error from text content: ${error.message}`, error);
103      testError = true;
104    }
105  
106    await browser.close();
107  
108    if (testError) {
109      throw testError;
110    }
111    console.info('Browser test complete');
112  }
113  
114  // Stop the process if an error is raised in the async function.
115  process.on('unhandledRejection', up => {
116    throw up;
117  });
118  
119  void main();
