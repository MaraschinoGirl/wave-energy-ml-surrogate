21   def _check_func(checker, argname, thefunc, x0, args, numinputs,
22                   output_shape=None):
23       res = atleast_1d(thefunc(*((x0[:numinputs],) + args)))
24       if (output_shape is not None) and (shape(res) != output_shape):
25           if (output_shape[0] != 1):
26               if len(output_shape) > 1:
27                   if output_shape[1] == 1:
28                       return shape(res)
29               msg = f"{checker}: there is a mismatch between the input and output " \
30                     f"shape of the '{argname}' argument"
31               func_name = getattr(thefunc, '__name__', None)
32               if func_name:
33                   msg += f" '{func_name}'."
34               else:
35                   msg += "."
36               msg += f'Shape should be {output_shape} but it is {shape(res)}.'
37               raise TypeError(msg)
38       if issubdtype(res.dtype, inexact):
39           dt = res.dtype
40       else:
41           dt = dtype(float)
42       return shape(res), dt

45   def fsolve(func, x0, args=(), fprime=None, full_output=0,
46              col_deriv=0, xtol=1.49012e-8, maxfev=0, band=None,
47              epsfcn=None, factor=100, diag=None):
48       """ 
49       Find the roots of a function. 
50    
51       Return the roots of the (non-linear) equations defined by 
52       ``func(x) = 0`` given a starting estimate. 
53    
54       Parameters 
55       ---------- 
56       func : callable ``f(x, *args)`` 
57           A function that takes at least one (possibly vector) argument, 
58           and returns a value of the same length. 
59       x0 : ndarray 
60           The starting estimate for the roots of ``func(x) = 0``. 
61       args : tuple, optional 
62           Any extra arguments to `func`. 
63       fprime : callable ``f(x, *args)``, optional 
64           A function to compute the Jacobian of `func` with derivatives 
65           across the rows. By default, the Jacobian will be estimated. 
66       full_output : bool, optional 
67           If True, return optional outputs. 
68       col_deriv : bool, optional 
69           Specify whether the Jacobian function computes derivatives down 
70           the columns (faster, because there is no transpose operation). 
71       xtol : float, optional 
72           The calculation will terminate if the relative error between two 
73           consecutive iterates is at most `xtol`. 
74       maxfev : int, optional 
75           The maximum number of calls to the function. If zero, then 
76           ``100*(N+1)`` is the maximum where N is the number of elements 
77           in `x0`. 
78       band : tuple, optional 
79           If set to a two-sequence containing the number of sub- and 
80           super-diagonals within the band of the Jacobi matrix, the 
81           Jacobi matrix is considered banded (only for ``fprime=None``). 
82       epsfcn : float, optional 
83           A suitable step length for the forward-difference 
84           approximation of the Jacobian (for ``fprime=None``). If 
85           `epsfcn` is less than the machine precision, it is assumed 
86           that the relative errors in the functions are of the order of 
87           the machine precision. 
88       factor : float, optional 
89           A parameter determining the initial step bound 
90           (``factor * || diag * x||``). Should be in the interval 
91           ``(0.1, 100)``. 
92       diag : sequence, optional 
93           N positive entries that serve as a scale factors for the 
94           variables. 
95    
96       Returns 
97       ------- 
98       x : ndarray 
99           The solution (or the result of the last iteration for 
100          an unsuccessful call). 
101      infodict : dict 
102          A dictionary of optional outputs with the keys: 
103   
104          ``nfev`` 
105              number of function calls 
106          ``njev`` 
107              number of Jacobian calls 
108          ``fvec`` 
109              function evaluated at the output 
110          ``fjac`` 
111              the orthogonal matrix, q, produced by the QR 
112              factorization of the final approximate Jacobian 
113              matrix, stored column wise 
114          ``r`` 
115              upper triangular matrix produced by QR factorization 
116              of the same matrix 
117          ``qtf`` 
118              the vector ``(transpose(q) * fvec)`` 
119   
120      ier : int 
121          An integer flag.  Set to 1 if a solution was found, otherwise refer 
122          to `mesg` for more information. 
123      mesg : str 
124          If no solution is found, `mesg` details the cause of failure. 
125   
126      See Also 
127      -------- 
128      root : Interface to root finding algorithms for multivariate 
129             functions. See the ``method='hybr'`` in particular. 
130   
131      Notes 
132      ----- 
133      ``fsolve`` is a wrapper around MINPACK's hybrd and hybrj algorithms. 
134   
135      Examples 
136      -------- 
137      Find a solution to the system of equations: 
138      ``x0*cos(x1) = 4,  x1*x0 - x1 = 5``. 
139   
140      >>> import numpy as np 
141      >>> from scipy.optimize import fsolve 
142      >>> def func(x): 
143      ...     return [x[0] * np.cos(x[1]) - 4, 
144      ...             x[1] * x[0] - x[1] - 5] 
145      >>> root = fsolve(func, [1, 1]) 
146      >>> root 
147      array([6.50409711, 0.90841421]) 
148      >>> np.isclose(func(root), [0.0, 0.0])  # func(root) should be almost 0.0. 
149      array([ True,  True]) 
150   
151      """
152      def _wrapped_func(*fargs):
153          """ 
154          Wrapped `func` to track the number of times 
155          the function has been called. 
156          """
157          _wrapped_func.nfev += 1
158          return func(*fargs)
159  
160      _wrapped_func.nfev = 0
161  
162      options = {'col_deriv': col_deriv,
163                 'xtol': xtol,
164                 'maxfev': maxfev,
165                 'band': band,
166                 'eps': epsfcn,
167                 'factor': factor,
168                 'diag': diag}
169  
170      res = _root_hybr(_wrapped_func, x0, args, jac=fprime, **options)
171      res.nfev = _wrapped_func.nfev
172  
173      if full_output:
174          x = res['x']
175          info = {k: res.get(k)
176                      for k in ('nfev', 'njev', 'fjac', 'r', 'qtf') if k in res}
177          info['fvec'] = res['fun']
178          return x, info, res['status'], res['message']
179      else:
180          status = res['status']
181          msg = res['message']
182          if status == 0:
183              raise TypeError(msg)
184          elif status == 1:
185              pass
186          elif status in [2, 3, 4, 5]:
187              warnings.warn(msg, RuntimeWarning, stacklevel=2)
188          else:
189              raise TypeError(msg)
190          return res['x']

193  def _root_hybr(func, x0, args=(), jac=None,
194                 col_deriv=0, xtol=1.49012e-08, maxfev=0, band=None, eps=None,
195                 factor=100, diag=None, **unknown_options):
196      """ 
197      Find the roots of a multivariate function using MINPACK's hybrd and 
198      hybrj routines (modified Powell method). 
199   
200      Options 
201      ------- 
202      col_deriv : bool 
203          Specify whether the Jacobian function computes derivatives down 
204          the columns (faster, because there is no transpose operation). 
205      xtol : float 
206          The calculation will terminate if the relative error between two 
207          consecutive iterates is at most `xtol`. 
208      maxfev : int 
209          The maximum number of calls to the function. If zero, then 
210          ``100*(N+1)`` is the maximum where N is the number of elements 
211          in `x0`. 
212      band : tuple 
213          If set to a two-sequence containing the number of sub- and 
214          super-diagonals within the band of the Jacobi matrix, the 
215          Jacobi matrix is considered banded (only for ``jac=None``). 
216      eps : float 
217          A suitable step length for the forward-difference 
218          approximation of the Jacobian (for ``jac=None``). If 
219          `eps` is less than the machine precision, it is assumed 
220          that the relative errors in the functions are of the order of 
221          the machine precision. 
222      factor : float 
223          A parameter determining the initial step bound 
224          (``factor * || diag * x||``). Should be in the interval 
225          ``(0.1, 100)``. 
226      diag : sequence 
227          N positive entries that serve as a scale factors for the 
228          variables. 
229   
230      """
231      _check_unknown_options(unknown_options)
232      epsfcn = eps
233  
234      x0 = asarray(x0).flatten()
235      n = len(x0)
236      if not isinstance(args, tuple):
237          args = (args,)
238      shape, dtype = _check_func('fsolve', 'func', func, x0, args, n, (n,))
239      if epsfcn is None:
240          epsfcn = finfo(dtype).eps
241      Dfun = jac
242      if Dfun is None:
243          if band is None:
244              ml, mu = -10, -10
245          else:
246              ml, mu = band[:2]
247          if maxfev == 0:
248              maxfev = 200 * (n + 1)
249          retval = _minpack._hybrd(func, x0, args, 1, xtol, maxfev,
250                                   ml, mu, epsfcn, factor, diag)
251      else:
252          _check_func('fsolve', 'fprime', Dfun, x0, args, n, (n, n))
253          if (maxfev == 0):
254              maxfev = 100 * (n + 1)
255          retval = _minpack._hybrj(func, Dfun, x0, args, 1,
256                                   col_deriv, xtol, maxfev, factor, diag)
257  
258      x, status = retval[0], retval[-1]
259  
260      errors = {0: "Improper input parameters were entered.",
261                1: "The solution converged.",
262                2: "The number of calls to function has "
263                    "reached maxfev = %d." % maxfev,
264                3: f"xtol={xtol:f} is too small, no further improvement "
265                    "in the approximate\n solution is possible.",
266                4: "The iteration is not making good progress, as measured "
267                    "by the \n improvement from the last five "
268                    "Jacobian evaluations.",
269                5: "The iteration is not making good progress, "
270                    "as measured by the \n improvement from the last "
271                    "ten iterations.",
272                'unknown': "An error occurred."}
273  
274      info = retval[1]
275      info['fun'] = info.pop('fvec')
276      sol = OptimizeResult(x=x, success=(status == 1), status=status,
277                           method="hybr")
278      sol.update(info)
279      try:
280          sol['message'] = errors[status]
281      except KeyError:
282          sol['message'] = errors['unknown']
283  
284      return sol

291  def leastsq(func, x0, args=(), Dfun=None, full_output=False,
292              col_deriv=False, ftol=1.49012e-8, xtol=1.49012e-8,
293              gtol=0.0, maxfev=0, epsfcn=None, factor=100, diag=None):
294      """ 
295      Minimize the sum of squares of a set of equations. 
296   
297      :: 
298   
299          x = arg min(sum(func(y)**2,axis=0)) 
300                   y 
301   
302      Parameters 
303      ---------- 
304      func : callable 
305          Should take at least one (possibly length ``N`` vector) argument and 
306          returns ``M`` floating point numbers. It must not return NaNs or 
307          fitting might fail. ``M`` must be greater than or equal to ``N``. 
308      x0 : ndarray 
309          The starting estimate for the minimization. 
310      args : tuple, optional 
311          Any extra arguments to func are placed in this tuple. 
312      Dfun : callable, optional 
313          A function or method to compute the Jacobian of func with derivatives 
314          across the rows. If this is None, the Jacobian will be estimated. 
315      full_output : bool, optional 
316          If ``True``, return all optional outputs (not just `x` and `ier`). 
317      col_deriv : bool, optional 
318          If ``True``, specify that the Jacobian function computes derivatives 
319          down the columns (faster, because there is no transpose operation). 
320      ftol : float, optional 
321          Relative error desired in the sum of squares. 
322      xtol : float, optional 
323          Relative error desired in the approximate solution. 
324      gtol : float, optional 
325          Orthogonality desired between the function vector and the columns of 
326          the Jacobian. 
327      maxfev : int, optional 
328          The maximum number of calls to the function. If `Dfun` is provided, 
329          then the default `maxfev` is 100*(N+1) where N is the number of elements 
330          in x0, otherwise the default `maxfev` is 200*(N+1). 
331      epsfcn : float, optional 
332          A variable used in determining a suitable step length for the forward- 
333          difference approximation of the Jacobian (for Dfun=None). 
334          Normally the actual step length will be sqrt(epsfcn)*x 
335          If epsfcn is less than the machine precision, it is assumed that the 
336          relative errors are of the order of the machine precision. 
337      factor : float, optional 
338          A parameter determining the initial step bound 
339          (``factor * || diag * x||``). Should be in interval ``(0.1, 100)``. 
340      diag : sequence, optional 
341          N positive entries that serve as a scale factors for the variables. 
342   
343      Returns 
344      ------- 
345      x : ndarray 
346          The solution (or the result of the last iteration for an unsuccessful 
347          call). 
348      cov_x : ndarray 
349          The inverse of the Hessian. `fjac` and `ipvt` are used to construct an 
350          estimate of the Hessian. A value of None indicates a singular matrix, 
351          which means the curvature in parameters `x` is numerically flat. To 
352          obtain the covariance matrix of the parameters `x`, `cov_x` must be 
353          multiplied by the variance of the residuals -- see curve_fit. Only 
354          returned if `full_output` is ``True``. 
355      infodict : dict 
356          a dictionary of optional outputs with the keys: 
357   
358          ``nfev`` 
359              The number of function calls 
360          ``fvec`` 
361              The function evaluated at the output 
362          ``fjac`` 
363              A permutation of the R matrix of a QR 
364              factorization of the final approximate 
365              Jacobian matrix, stored column wise. 
366              Together with ipvt, the covariance of the 
367              estimate can be approximated. 
368          ``ipvt`` 
369              An integer array of length N which defines 
370              a permutation matrix, p, such that 
371              fjac*p = q*r, where r is upper triangular 
372              with diagonal elements of nonincreasing 
373              magnitude. Column j of p is column ipvt(j) 
374              of the identity matrix. 
375          ``qtf`` 
376              The vector (transpose(q) * fvec). 
377   
378          Only returned if `full_output` is ``True``. 
379      mesg : str 
380          A string message giving information about the cause of failure. 
381          Only returned if `full_output` is ``True``. 
382      ier : int 
383          An integer flag. If it is equal to 1, 2, 3 or 4, the solution was 
384          found. Otherwise, the solution was not found. In either case, the 
385          optional output variable 'mesg' gives more information. 
386   
387      See Also 
388      -------- 
389      least_squares : Newer interface to solve nonlinear least-squares problems 
390          with bounds on the variables. See ``method='lm'`` in particular. 
391   
392      Notes 
393      ----- 
394      "leastsq" is a wrapper around MINPACK's lmdif and lmder algorithms. 
395   
396      cov_x is a Jacobian approximation to the Hessian of the least squares 
397      objective function. 
398      This approximation assumes that the objective function is based on the 
399      difference between some observed target data (ydata) and a (non-linear) 
400      function of the parameters `f(xdata, params)` :: 
401   
402             func(params) = ydata - f(xdata, params) 
403   
404      so that the objective function is :: 
405   
406             min   sum((ydata - f(xdata, params))**2, axis=0) 
407           params 
408   
409      The solution, `x`, is always a 1-D array, regardless of the shape of `x0`, 
410      or whether `x0` is a scalar. 
411   
412      Examples 
413      -------- 
414      >>> from scipy.optimize import leastsq 
415      >>> def func(x): 
416      ...     return 2*(x-3)**2+1 
417      >>> leastsq(func, 0) 
418      (array([2.99999999]), 1) 
419   
420      """
421      x0 = asarray(x0).flatten()
422      n = len(x0)
423      if not isinstance(args, tuple):
424          args = (args,)
425      shape, dtype = _check_func('leastsq', 'func', func, x0, args, n)
426      m = shape[0]
427  
428      if n > m:
429          raise TypeError(f"Improper input: func input vector length N={n} must"
430                          f" not exceed func output vector length M={m}")
431  
432      if epsfcn is None:
433          epsfcn = finfo(dtype).eps
434  
435      if Dfun is None:
436          if maxfev == 0:
437              maxfev = 200*(n + 1)
438          retval = _minpack._lmdif(func, x0, args, full_output, ftol, xtol,
439                                   gtol, maxfev, epsfcn, factor, diag)
440      else:
441          if col_deriv:
442              _check_func('leastsq', 'Dfun', Dfun, x0, args, n, (n, m))
443          else:
444              _check_func('leastsq', 'Dfun', Dfun, x0, args, n, (m, n))
445          if maxfev == 0:
446              maxfev = 100 * (n + 1)
447          retval = _minpack._lmder(func, Dfun, x0, args, full_output,
448                                   col_deriv, ftol, xtol, gtol, maxfev,
449                                   factor, diag)
450  
451      errors = {0: ["Improper input parameters.", TypeError],
452                1: ["Both actual and predicted relative reductions "
453                    f"in the sum of squares\n  are at most {ftol:f}", None],
454                2: ["The relative error between two consecutive "
455                    f"iterates is at most {xtol:f}", None],
456                3: ["Both actual and predicted relative reductions in "
457                    f"the sum of squares\n  are at most {ftol:f} and the "
458                    "relative error between two consecutive "
459                    f"iterates is at \n  most {xtol:f}", None],
460                4: ["The cosine of the angle between func(x) and any "
461                    f"column of the\n  Jacobian is at most {gtol:f} in "
462                    "absolute value", None],
463                5: ["Number of calls to function has reached "
464                    "maxfev = %d." % maxfev, ValueError],
465                6: [f"ftol={ftol:f} is too small, no further reduction "
466                    "in the sum of squares\n  is possible.",
467                    ValueError],
468                7: [f"xtol={xtol:f} is too small, no further improvement in "
469                    "the approximate\n  solution is possible.",
470                    ValueError],
471                8: [f"gtol={gtol:f} is too small, func(x) is orthogonal to the "
472                    "columns of\n  the Jacobian to machine precision.", ValueError]}
473  
474      # The FORTRAN return value (possible return values are >= 0 and <= 8)
475      info = retval[-1]
476  
477      if full_output:
478          cov_x = None
479          if info in LEASTSQ_SUCCESS:
480              # This was
481              # perm = take(eye(n), retval[1]['ipvt'] - 1, 0)
482              # r = triu(transpose(retval[1]['fjac'])[:n, :])
483              # R = dot(r, perm)
484              # cov_x = inv(dot(transpose(R), R))
485              # but the explicit dot product was not necessary and sometimes
486              # the result was not symmetric positive definite. See gh-4555.
487              perm = retval[1]['ipvt']
488              n = len(perm)
489              r = triu(transpose(retval[1]['fjac'])[:n, :])
490              inv_triu = linalg.get_lapack_funcs('trtri', (r,))
491              try:
492                  # inverse of permuted matrix is a permutation of matrix inverse
493                  invR, trtri_info = inv_triu(r)  # default: upper, non-unit diag
494                  if trtri_info != 0:  # explicit comparison for readability
495                      raise LinAlgError(f'trtri returned info {trtri_info}')
496                  invR[perm] = invR.copy()
497                  cov_x = invR @ invR.T
498              except (LinAlgError, ValueError):
499                  pass
500          return (retval[0], cov_x) + retval[1:-1] + (errors[info][0], info)
501      else:
502          if info in LEASTSQ_FAILURE:
503              warnings.warn(errors[info][0], RuntimeWarning, stacklevel=2)
504          elif info == 0:
505              raise errors[info][1](errors[info][0])
506          return retval[0], info

509  def _lightweight_memoizer(f):
510      # very shallow memoization to address gh-13670: only remember the first set
511      # of parameters and corresponding function value, and only attempt to use
512      # them twice (the number of times the function is evaluated at x0).
513      def _memoized_func(params):
514          if _memoized_func.skip_lookup:
515              return f(params)
516  
517          if np.all(_memoized_func.last_params == params):
518              return _memoized_func.last_val
519          elif _memoized_func.last_params is not None:
520              _memoized_func.skip_lookup = True
521  
522          val = f(params)
523  
524          if _memoized_func.last_params is None:
525              _memoized_func.last_params = np.copy(params)
526              _memoized_func.last_val = val
527  
528          return val
529  
530      _memoized_func.last_params = None
531      _memoized_func.last_val = None
532      _memoized_func.skip_lookup = False
533      return _memoized_func

536  def _wrap_func(func, xdata, ydata, transform):
537      if transform is None:
538          def func_wrapped(params):
539              return func(xdata, *params) - ydata
540      elif transform.size == 1 or transform.ndim == 1:
541          def func_wrapped(params):
542              return transform * (func(xdata, *params) - ydata)
543      else:
544          # Chisq = (y - yd)^T C^{-1} (y-yd)
545          # transform = L such that C = L L^T
546          # C^{-1} = L^{-T} L^{-1}
547          # Chisq = (y - yd)^T L^{-T} L^{-1} (y-yd)
548          # Define (y-yd)' = L^{-1} (y-yd)
549          # by solving
550          # L (y-yd)' = (y-yd)
551          # and minimize (y-yd)'^T (y-yd)'
552          def func_wrapped(params):
553              return solve_triangular(transform, func(xdata, *params) - ydata, lower=True)
554      return func_wrapped

557  def _wrap_jac(jac, xdata, transform):
558      if transform is None:
559          def jac_wrapped(params):
560              return jac(xdata, *params)
561      elif transform.ndim == 1:
562          def jac_wrapped(params):
563              return transform[:, np.newaxis] * np.asarray(jac(xdata, *params))
564      else:
565          def jac_wrapped(params):
566              return solve_triangular(transform,
567                                      np.asarray(jac(xdata, *params)),
568                                      lower=True)
569      return jac_wrapped

572  def _initialize_feasible(lb, ub):
573      p0 = np.ones_like(lb)
574      lb_finite = np.isfinite(lb)
575      ub_finite = np.isfinite(ub)
576  
577      mask = lb_finite & ub_finite
578      p0[mask] = 0.5 * (lb[mask] + ub[mask])
579  
580      mask = lb_finite & ~ub_finite
581      p0[mask] = lb[mask] + 1
582  
583      mask = ~lb_finite & ub_finite
584      p0[mask] = ub[mask] - 1
585  
586      return p0

589  def curve_fit(f, xdata, ydata, p0=None, sigma=None, absolute_sigma=False,
590                check_finite=None, bounds=(-np.inf, np.inf), method=None,
591                jac=None, *, full_output=False, nan_policy=None,
592                **kwargs):
593      """ 
594      Use non-linear least squares to fit a function, f, to data. 
595   
596      Assumes ``ydata = f(xdata, *params) + eps``. 
597   
598      Parameters 
599      ---------- 
600      f : callable 
601          The model function, f(x, ...). It must take the independent 
602          variable as the first argument and the parameters to fit as 
603          separate remaining arguments. 
604      xdata : array_like 
605          The independent variable where the data is measured. 
606          Should usually be an M-length sequence or an (k,M)-shaped array for 
607          functions with k predictors, and each element should be float 
608          convertible if it is an array like object. 
609      ydata : array_like 
610          The dependent data, a length M array - nominally ``f(xdata, ...)``. 
611      p0 : array_like, optional 
612          Initial guess for the parameters (length N). If None, then the 
613          initial values will all be 1 (if the number of parameters for the 
614          function can be determined using introspection, otherwise a 
615          ValueError is raised). 
616      sigma : None or scalar or M-length sequence or MxM array, optional 
617          Determines the uncertainty in `ydata`. If we define residuals as 
618          ``r = ydata - f(xdata, *popt)``, then the interpretation of `sigma` 
619          depends on its number of dimensions: 
620   
621          - A scalar or 1-D `sigma` should contain values of standard deviations of 
622            errors in `ydata`. In this case, the optimized function is 
623            ``chisq = sum((r / sigma) ** 2)``. 
624   
625          - A 2-D `sigma` should contain the covariance matrix of 
626            errors in `ydata`. In this case, the optimized function is 
627            ``chisq = r.T @ inv(sigma) @ r``. 
628   
629            .. versionadded:: 0.19 
630   
631          None (default) is equivalent of 1-D `sigma` filled with ones. 
632      absolute_sigma : bool, optional 
633          If True, `sigma` is used in an absolute sense and the estimated parameter 
634          covariance `pcov` reflects these absolute values. 
635   
636          If False (default), only the relative magnitudes of the `sigma` values matter. 
637          The returned parameter covariance matrix `pcov` is based on scaling 
638          `sigma` by a constant factor. This constant is set by demanding that the 
639          reduced `chisq` for the optimal parameters `popt` when using the 
640          *scaled* `sigma` equals unity. In other words, `sigma` is scaled to 
641          match the sample variance of the residuals after the fit. Default is False. 
642          Mathematically, 
643          ``pcov(absolute_sigma=False) = pcov(absolute_sigma=True) * chisq(popt)/(M-N)`` 
644      check_finite : bool, optional 
645          If True, check that the input arrays do not contain nans of infs, 
646          and raise a ValueError if they do. Setting this parameter to 
647          False may silently produce nonsensical results if the input arrays 
648          do contain nans. Default is True if `nan_policy` is not specified 
649          explicitly and False otherwise. 
650      bounds : 2-tuple of array_like or `Bounds`, optional 
651          Lower and upper bounds on parameters. Defaults to no bounds. 
652          There are two ways to specify the bounds: 
653   
654          - Instance of `Bounds` class. 
655   
656          - 2-tuple of array_like: Each element of the tuple must be either 
657            an array with the length equal to the number of parameters, or a 
658            scalar (in which case the bound is taken to be the same for all 
659            parameters). Use ``np.inf`` with an appropriate sign to disable 
660            bounds on all or some parameters. 
661   
662      method : {'lm', 'trf', 'dogbox'}, optional 
663          Method to use for optimization. See `least_squares` for more details. 
664          Default is 'lm' for unconstrained problems and 'trf' if `bounds` are 
665          provided. The method 'lm' won't work when the number of observations 
666          is less than the number of variables, use 'trf' or 'dogbox' in this 
667          case. 
668   
669          .. versionadded:: 0.17 
670      jac : callable, string or None, optional 
671          Function with signature ``jac(x, ...)`` which computes the Jacobian 
672          matrix of the model function with respect to parameters as a dense 
673          array_like structure. It will be scaled according to provided `sigma`. 
674          If None (default), the Jacobian will be estimated numerically. 
675          String keywords for 'trf' and 'dogbox' methods can be used to select 
676          a finite difference scheme, see `least_squares`. 
677   
678          .. versionadded:: 0.18 
679      full_output : boolean, optional 
680          If True, this function returns additional information: `infodict`, 
681          `mesg`, and `ier`. 
682   
683          .. versionadded:: 1.9 
684      nan_policy : {'raise', 'omit', None}, optional 
685          Defines how to handle when input contains nan. 
686          The following options are available (default is None): 
687   
688          * 'raise': throws an error 
689          * 'omit': performs the calculations ignoring nan values 
690          * None: no special handling of NaNs is performed 
691            (except what is done by check_finite); the behavior when NaNs 
692            are present is implementation-dependent and may change. 
693   
694          Note that if this value is specified explicitly (not None), 
695          `check_finite` will be set as False. 
696   
697          .. versionadded:: 1.11 
698      **kwargs 
699          Keyword arguments passed to `leastsq` for ``method='lm'`` or 
700          `least_squares` otherwise. 
701   
702      Returns 
703      ------- 
704      popt : array 
705          Optimal values for the parameters so that the sum of the squared 
706          residuals of ``f(xdata, *popt) - ydata`` is minimized. 
707      pcov : 2-D array 
708          The estimated approximate covariance of popt. The diagonals provide 
709          the variance of the parameter estimate. To compute one standard 
710          deviation errors on the parameters, use 
711          ``perr = np.sqrt(np.diag(pcov))``. Note that the relationship between 
712          `cov` and parameter error estimates is derived based on a linear 
713          approximation to the model function around the optimum [1]_. 
714          When this approximation becomes inaccurate, `cov` may not provide an 
715          accurate measure of uncertainty. 
716   
717          How the `sigma` parameter affects the estimated covariance 
718          depends on `absolute_sigma` argument, as described above. 
719   
720          If the Jacobian matrix at the solution doesn't have a full rank, then 
721          'lm' method returns a matrix filled with ``np.inf``, on the other hand 
722          'trf'  and 'dogbox' methods use Moore-Penrose pseudoinverse to compute 
723          the covariance matrix. Covariance matrices with large condition numbers 
724          (e.g. computed with `numpy.linalg.cond`) may indicate that results are 
725          unreliable. 
726      infodict : dict (returned only if `full_output` is True) 
727          a dictionary of optional outputs with the keys: 
728   
729          ``nfev`` 
730              The number of function calls. Methods 'trf' and 'dogbox' do not 
731              count function calls for numerical Jacobian approximation, 
732              as opposed to 'lm' method. 
733          ``fvec`` 
734              The residual values evaluated at the solution, for a 1-D `sigma` 
735              this is ``(f(x, *popt) - ydata)/sigma``. 
736          ``fjac`` 
737              A permutation of the R matrix of a QR 
738              factorization of the final approximate 
739              Jacobian matrix, stored column wise. 
740              Together with ipvt, the covariance of the 
741              estimate can be approximated. 
742              Method 'lm' only provides this information. 
743          ``ipvt`` 
744              An integer array of length N which defines 
745              a permutation matrix, p, such that 
746              fjac*p = q*r, where r is upper triangular 
747              with diagonal elements of nonincreasing 
748              magnitude. Column j of p is column ipvt(j) 
749              of the identity matrix. 
750              Method 'lm' only provides this information. 
751          ``qtf`` 
752              The vector (transpose(q) * fvec). 
753              Method 'lm' only provides this information. 
754   
755          .. versionadded:: 1.9 
756      mesg : str (returned only if `full_output` is True) 
757          A string message giving information about the solution. 
758   
759          .. versionadded:: 1.9 
760      ier : int (returned only if `full_output` is True) 
761          An integer flag. If it is equal to 1, 2, 3 or 4, the solution was 
762          found. Otherwise, the solution was not found. In either case, the 
763          optional output variable `mesg` gives more information. 
764   
765          .. versionadded:: 1.9 
766   
767      Raises 
768      ------ 
769      ValueError 
770          if either `ydata` or `xdata` contain NaNs, or if incompatible options 
771          are used. 
772   
773      RuntimeError 
774          if the least-squares minimization fails. 
775   
776      OptimizeWarning 
777          if covariance of the parameters can not be estimated.

785      Notes 
786      ----- 
787      Users should ensure that inputs `xdata`, `ydata`, and the output of `f` 
788      are ``float64``, or else the optimization may return incorrect results. 
789   
790      With ``method='lm'``, the algorithm uses the Levenberg-Marquardt algorithm 
791      through `leastsq`. Note that this algorithm can only deal with 
792      unconstrained problems. 
793   
794      Box constraints can be handled by methods 'trf' and 'dogbox'. Refer to 
795      the docstring of `least_squares` for more information. 
796   
797      Parameters to be fitted must have similar scale. Differences of multiple 
798      orders of magnitude can lead to incorrect results. For the 'trf' and 
799      'dogbox' methods, the `x_scale` keyword argument can be used to scale 
800      the parameters.

802      References 
803      ---------- 
804      .. [1] K. Vugrin et al. Confidence region estimation techniques for nonlinear 
805             regression in groundwater flow: Three case studies. Water Resources 
806             Research, Vol. 43, W03423, :doi:`10.1029/2005WR004804`

808      Examples 
809      -------- 
810      >>> import numpy as np 
811      >>> import matplotlib.pyplot as plt 
812      >>> from scipy.optimize import curve_fit 
813   
814      >>> def func(x, a, b, c): 
815      ...     return a * np.exp(-b * x) + c 
816   
817      Define the data to be fit with some noise: 
818   
819      >>> xdata = np.linspace(0, 4, 50) 
820      >>> y = func(xdata, 2.5, 1.3, 0.5) 
821      >>> rng = np.random.default_rng() 
822      >>> y_noise = 0.2 * rng.normal(size=xdata.size) 
823      >>> ydata = y + y_noise 
824      >>> plt.plot(xdata, ydata, 'b-', label='data') 
825   
826      Fit for the parameters a, b, c of the function `func`: 
827   
828      >>> popt, pcov = curve_fit(func, xdata, ydata) 
829      >>> popt 
830      array([2.56274217, 1.37268521, 0.47427475]) 
831      >>> plt.plot(xdata, func(xdata, *popt), 'r-', 
832      ...          label='fit: a=%5.3f, b=%5.3f, c=%5.3f' % tuple(popt)) 
833   
834      Constrain the optimization to the region of ``0 <= a <= 3``, 
835      ``0 <= b <= 1`` and ``0 <= c <= 0.5``: 
836   
837      >>> popt, pcov = curve_fit(func, xdata, ydata, bounds=(0, [3., 1., 0.5])) 
838      >>> popt 
839      array([2.43736712, 1.        , 0.34463856]) 
840      >>> plt.plot(xdata, func(xdata, *popt), 'g--', 
841      ...          label='fit: a=%5.3f, b=%5.3f, c=%5.3f' % tuple(popt)) 
842   
843      >>> plt.xlabel('x') 
844      >>> plt.ylabel('y') 
845      >>> plt.legend() 
846      >>> plt.show() 
847   
848      For reliable results, the model `func` should not be overparametrized; 
849      redundant parameters can cause unreliable covariance matrices and, in some 
850      cases, poorer quality fits. As a quick check of whether the model may be 
851      overparameterized, calculate the condition number of the covariance matrix: 
852   
853      >>> np.linalg.cond(pcov) 
854      34.571092161547405  # may vary 
855   
856      The value is small, so it does not raise much concern. If, however, we were 
857      to add a fourth parameter ``d`` to `func` with the same effect as ``a``: 
858   
859      >>> def func2(x, a, b, c, d): 
860      ...     return a * d * np.exp(-b * x) + c  # a and d are redundant 
861      >>> popt, pcov = curve_fit(func2, xdata, ydata) 
862      >>> np.linalg.cond(pcov) 
863      1.13250718925596e+32  # may vary 
864   
865      Such a large value is cause for concern. The diagonal elements of the 
866      covariance matrix, which is related to uncertainty of the fit, gives more 
867      information: 
868   
869      >>> np.diag(pcov) 
870      array([1.48814742e+29, 3.78596560e-02, 5.39253738e-03, 2.76417220e+28])  # may vary 
871   
872      Note that the first and last terms are much larger than the other elements, 
873      suggesting that the optimal values of these parameters are ambiguous and 
874      that only one of these parameters is needed in the model. 
875   
876      If the optimal parameters of `f` differ by multiple orders of magnitude, the 
877      resulting fit can be inaccurate. Sometimes, `curve_fit` can fail to find any 
878      results: 
879   
880      >>> ydata = func(xdata, 500000, 0.01, 15) 
881      >>> try: 
882      ...     popt, pcov = curve_fit(func, xdata, ydata, method = 'trf') 
883      ... except RuntimeError as e: 
884      ...     print(e) 
885      Optimal parameters not found: The maximum number of function evaluations is 
886      exceeded. 
887   
888      If parameter scale is roughly known beforehand, it can be defined in 
889      `x_scale` argument: 
890   
891      >>> popt, pcov = curve_fit(func, xdata, ydata, method = 'trf', 
892      ...                        x_scale = [1000, 1, 1]) 
893      >>> popt 
894      array([5.00000000e+05, 1.00000000e-02, 1.49999999e+01]) 
895      """
896      if p0 is None:
897          # determine number of parameters by inspecting the function
898          sig = _getfullargspec(f)
899          args = sig.args
900          if len(args) < 2:
901              raise ValueError("Unable to determine number of fit parameters.")
902          n = len(args) - 1
903      else:
904          p0 = np.atleast_1d(p0)
905          n = p0.size
906  
907      if isinstance(bounds, Bounds):
908          lb, ub = bounds.lb, bounds.ub
909      else:
910          lb, ub = prepare_bounds(bounds, n)
911      if p0 is None:
912          p0 = _initialize_feasible(lb, ub)
913  
914      bounded_problem = np.any((lb > -np.inf) | (ub < np.inf))
915      if method is None:
916          if bounded_problem:
917              method = 'trf'
918          else:
919              method = 'lm'
920  
921      if method == 'lm' and bounded_problem:
922          raise ValueError("Method 'lm' only works for unconstrained problems. "
923                           "Use 'trf' or 'dogbox' instead.")
924  
925      if check_finite is None:
926          check_finite = True if nan_policy is None else False
927  
928      # optimization may produce garbage for float32 inputs, cast them to float64
929      if check_finite:
930          ydata = np.asarray_chkfinite(ydata, float)
931      else:
932          ydata = np.asarray(ydata, float)
933  
934      if isinstance(xdata, (list, tuple, np.ndarray)):
935          # `xdata` is passed straight to the user-defined `f`, so allow
936          # non-array_like `xdata`.
937          if check_finite:
938              xdata = np.asarray_chkfinite(xdata, float)
939          else:
940              xdata = np.asarray(xdata, float)
941  
942      if ydata.size == 0:
943          raise ValueError("`ydata` must not be empty!")
944  
945      # nan handling is needed only if check_finite is False because if True,
946      # the x-y data are already checked, and they don't contain nans.
947      if not check_finite and nan_policy is not None:
948          if nan_policy == "propagate":
949              raise ValueError("`nan_policy='propagate'` is not supported "
950                               "by this function.")
951  
952          policies = [None, 'raise', 'omit']
953          x_contains_nan, nan_policy = _contains_nan(xdata, nan_policy,
954                                                     policies=policies)
955          y_contains_nan, nan_policy = _contains_nan(ydata, nan_policy,
956                                                     policies=policies)
957  
958          if (x_contains_nan or y_contains_nan) and nan_policy == 'omit':
959              # ignore NaNs for N dimensional arrays
960              has_nan = np.isnan(xdata)
961              has_nan = has_nan.any(axis=tuple(range(has_nan.ndim-1)))
962              has_nan |= np.isnan(ydata)
963  
964              xdata = xdata[..., ~has_nan]
965              ydata = ydata[~has_nan]
966  
967              # Also omit the corresponding entries from sigma
968              if sigma is not None:
969                  sigma = np.asarray(sigma)
970                  if sigma.ndim == 1:
971                      sigma = sigma[~has_nan]
972                  elif sigma.ndim == 2:
973                      sigma = sigma[~has_nan, :]
974                      sigma = sigma[:, ~has_nan]
975  
976      # Determine type of sigma
977      if sigma is not None:
978          sigma = np.asarray(sigma)
979  
980          # if 1-D or a scalar, sigma are errors, define transform = 1/sigma
981          if sigma.size == 1 or sigma.shape == (ydata.size,):
982              transform = 1.0 / sigma
983          # if 2-D, sigma is the covariance matrix,
984          # define transform = L such that L L^T = C
985          elif sigma.shape == (ydata.size, ydata.size):
986              try:
987                  # scipy.linalg.cholesky requires lower=True to return L L^T = A
988                  transform = cholesky(sigma, lower=True)
989              except LinAlgError as e:
990                  raise ValueError("`sigma` must be positive definite.") from e
991          else:
992              raise ValueError("`sigma` has incorrect shape.")
993      else:
994          transform = None
995  
996      func = _lightweight_memoizer(_wrap_func(f, xdata, ydata, transform))
997  
998      if callable(jac):
999          jac = _lightweight_memoizer(_wrap_jac(jac, xdata, transform))
1000     elif jac is None and method != 'lm':
1001         jac = '2-point'
1002 
1003     if 'args' in kwargs:
1004         # The specification for the model function `f` does not support
1005         # additional arguments. Refer to the `curve_fit` docstring for
1006         # acceptable call signatures of `f`.
1007         raise ValueError("'args' is not a supported keyword argument.")
1008 
1009     if method == 'lm':
1010         # if ydata.size == 1, this might be used for broadcast.
1011         if ydata.size != 1 and n > ydata.size:
1012             raise TypeError(f"The number of func parameters={n} must not"
1013                             f" exceed the number of data points={ydata.size}")
1014         res = leastsq(func, p0, Dfun=jac, full_output=1, **kwargs)
1015         popt, pcov, infodict, errmsg, ier = res
1016         ysize = len(infodict['fvec'])
1017         cost = np.sum(infodict['fvec'] ** 2)
1018         if ier not in [1, 2, 3, 4]:
1019             raise RuntimeError("Optimal parameters not found: " + errmsg)
1020     else:
1021         # Rename maxfev (leastsq) to max_nfev (least_squares), if specified.
1022         if 'max_nfev' not in kwargs:
1023             kwargs['max_nfev'] = kwargs.pop('maxfev', None)
1024 
1025         res = least_squares(func, p0, jac=jac, bounds=bounds, method=method,
1026                             **kwargs)
1027 
1028         if not res.success:
1029             raise RuntimeError("Optimal parameters not found: " + res.message)
1030 
1031         infodict = dict(nfev=res.nfev, fvec=res.fun)
1032         ier = res.status
1033         errmsg = res.message
1034 
1035         ysize = len(res.fun)
1036         cost = 2 * res.cost  # res.cost is half sum of squares!
1037         popt = res.x
1038 
1039         # Do Moore-Penrose inverse discarding zero singular values.
1040         _, s, VT = svd(res.jac, full_matrices=False)
1041         threshold = np.finfo(float).eps * max(res.jac.shape) * s[0]
1042         s = s[s > threshold]
1043         VT = VT[:s.size]
1044         pcov = np.dot(VT.T / s**2, VT)
1045 
1046     warn_cov = False
1047     if pcov is None or np.isnan(pcov).any():
1048         # indeterminate covariance
1049         pcov = zeros((len(popt), len(popt)), dtype=float)
1050         pcov.fill(inf)
1051         warn_cov = True
1052     elif not absolute_sigma:
1053         if ysize > p0.size:
1054             s_sq = cost / (ysize - p0.size)
1055             pcov = pcov * s_sq
1056         else:
1057             pcov.fill(inf)
1058             warn_cov = True
1059 
1060     if warn_cov:
1061         warnings.warn('Covariance of the parameters could not be estimated',
1062                       category=OptimizeWarning, stacklevel=2)
1063 
1064     if full_output:
1065         return popt, pcov, infodict, errmsg, ier
1066     else:
1067         return popt, pcov

1070 def check_gradient(fcn, Dfcn, x0, args=(), col_deriv=0):
1071     """Perform a simple check on the gradient for correctness. 
1072  
1073     """
1074 
1075     x = atleast_1d(x0)
1076     n = len(x)
1077     x = x.reshape((n,))
1078     fvec = atleast_1d(fcn(x, *args))
1079     m = len(fvec)
1080     fvec = fvec.reshape((m,))
1081     ldfjac = m
1082     fjac = atleast_1d(Dfcn(x, *args))
1083     fjac = fjac.reshape((m, n))
1084     if col_deriv == 0:
1085         fjac = transpose(fjac)
1086 
1087     xp = zeros((n,), float)
1088     err = zeros((m,), float)
1089     fvecp = None
1090     _minpack._chkder(m, n, x, fvec, fjac, ldfjac, xp, fvecp, 1, err)
1091 
1092     fvecp = atleast_1d(fcn(xp, *args))
1093     fvecp = fvecp.reshape((m,))
1094     _minpack._chkder(m, n, x, fvec, fjac, ldfjac, xp, fvecp, 2, err)
1095 
1096     good = (prod(greater(err, 0.5), axis=0))
1097 
1098     return (good, err)

1101 def _del2(p0, p1, d):
1102     return p0 - np.square(p1 - p0) / d

1105 def _relerr(actual, desired):
1106     return (actual - desired) / desired

1109 def _fixed_point_helper(func, x0, args, xtol, maxiter, use_accel):
1110     p0 = x0
1111     for i in range(maxiter):
1112         p1 = func(p0, *args)
1113         if use_accel:
1114             p2 = func(p1, *args)
1115             d = p2 - 2.0 * p1 + p0
1116             p = _lazywhere(d != 0, (p0, p1, d), f=_del2, fillvalue=p2)
1117         else:
1118             p = p1
1119         relerr = _lazywhere(p0 != 0, (p, p0), f=_relerr, fillvalue=p)
1120         if np.all(np.abs(relerr) < xtol):
1121             return p
1122         p0 = p
1123     msg = "Failed to converge after %d iterations, value is %s" % (maxiter, p)
1124     raise RuntimeError(msg)

1127 def fixed_point(func, x0, args=(), xtol=1e-8, maxiter=500, method='del2'):
1128     """ 
1129     Find a fixed point of the function. 
1130  
1131     Given a function of one or more variables and a starting point, find a 
1132     fixed point of the function: i.e., where ``func(x0) == x0``. 
1133  
1134     Parameters 
1135     ---------- 
1136     func : function 
1137         Function to evaluate. 
1138     x0 : array_like 
1139         Fixed point of function. 
1140     args : tuple, optional 
1141         Extra arguments to `func`. 
1142     xtol : float, optional 
1143         Convergence tolerance, defaults to 1e-08. 
1144     maxiter : int, optional 
1145         Maximum number of iterations, defaults to 500. 
1146     method : {"del2", "iteration"}, optional 
1147         Method of finding the fixed-point, defaults to "del2", 
1148         which uses Steffensen's Method with Aitken's ``Del^2`` 
1149         convergence acceleration [1]_. The "iteration" method simply iterates 
1150         the function until convergence is detected, without attempting to 
1151         accelerate the convergence. 
1152  
1153     References 
1154     ---------- 
1155     .. [1] Burden, Faires, "Numerical Analysis", 5th edition, pg. 80 
1156  
1157     Examples 
1158     -------- 
1159     >>> import numpy as np 
1160     >>> from scipy import optimize 
1161     >>> def func(x, c1, c2): 
1162     ...    return np.sqrt(c1/(x+c2)) 
1163     >>> c1 = np.array([10,12.]) 
1164     >>> c2 = np.array([3, 5.]) 
1165     >>> optimize.fixed_point(func, [1.2, 1.3], args=(c1,c2)) 
1166     array([ 1.4920333 ,  1.37228132]) 
1167  
1168     """
1169     use_accel = {'del2': True, 'iteration': False}[method]
1170     x0 = _asarray_validated(x0, as_inexact=True)
1171     return _fixed_point_helper(func, x0, args, xtol, maxiter, use_accel)
1172
