19   def make_regression_with_outliers(n_samples=50, n_features=20):
20       rng = np.random.RandomState(0)
21       # Generate data with outliers by replacing 10% of the samples with noise.
22       X, y = make_regression(
23           n_samples=n_samples, n_features=n_features, random_state=0, noise=0.05
24       )
25   
26       # Replace 10% of the sample with noise.
27       num_noise = int(0.1 * n_samples)
28       random_samples = rng.randint(0, n_samples, num_noise)
29       X[random_samples, :] = 2.0 * rng.normal(0, 1, (num_noise, X.shape[1]))
30       return X, y

33   def test_huber_equals_lr_for_high_epsilon():
34       # Test that Ridge matches LinearRegression for large epsilon
35       X, y = make_regression_with_outliers()
36       lr = LinearRegression()
37       lr.fit(X, y)
38       huber = HuberRegressor(epsilon=1e3, alpha=0.0)
39       huber.fit(X, y)
40       assert_almost_equal(huber.coef_, lr.coef_, 3)
41       assert_almost_equal(huber.intercept_, lr.intercept_, 2)

51   def test_huber_gradient():
52       # Test that the gradient calculated by _huber_loss_and_gradient is correct
53       rng = np.random.RandomState(1)
54       X, y = make_regression_with_outliers()
55       sample_weight = rng.randint(1, 3, (y.shape[0]))
56   
57       def loss_func(x, *args):
58           return _huber_loss_and_gradient(x, *args)[0]
59   
60       def grad_func(x, *args):
61           return _huber_loss_and_gradient(x, *args)[1]
62   
63       # Check using optimize.check_grad that the gradients are equal.
64       for _ in range(5):
65           # Check for both fit_intercept and otherwise.
66           for n_features in [X.shape[1] + 1, X.shape[1] + 2]:
67               w = rng.randn(n_features)
68               w[-1] = np.abs(w[-1])
69               grad_same = optimize.check_grad(
70                   loss_func, grad_func, w, X, y, 0.01, 0.1, sample_weight
71               )
72               assert_almost_equal(grad_same, 1e-6, 4)

75   @pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
76   def test_huber_sample_weights(csr_container):
77       # Test sample_weights implementation in HuberRegressor"""
78   
79       X, y = make_regression_with_outliers()
80       huber = HuberRegressor()
81       huber.fit(X, y)
82       huber_coef = huber.coef_
83       huber_intercept = huber.intercept_
84   
85       # Rescale coefs before comparing with assert_array_almost_equal to make
86       # sure that the number of decimal places used is somewhat insensitive to
87       # the amplitude of the coefficients and therefore to the scale of the
88       # data and the regularization parameter
89       scale = max(np.mean(np.abs(huber.coef_)), np.mean(np.abs(huber.intercept_)))
90   
91       huber.fit(X, y, sample_weight=np.ones(y.shape[0]))
92       assert_array_almost_equal(huber.coef_ / scale, huber_coef / scale)
93       assert_array_almost_equal(huber.intercept_ / scale, huber_intercept / scale)
94   
95       X, y = make_regression_with_outliers(n_samples=5, n_features=20)
96       X_new = np.vstack((X, np.vstack((X[1], X[1], X[3]))))
97       y_new = np.concatenate((y, [y[1]], [y[1]], [y[3]]))
98       huber.fit(X_new, y_new)
99       huber_coef = huber.coef_
100      huber_intercept = huber.intercept_
101      sample_weight = np.ones(X.shape[0])
102      sample_weight[1] = 3
103      sample_weight[3] = 2
104      huber.fit(X, y, sample_weight=sample_weight)
105  
106      assert_array_almost_equal(huber.coef_ / scale, huber_coef / scale)
107      assert_array_almost_equal(huber.intercept_ / scale, huber_intercept / scale)
108  
109      # Test sparse implementation with sample weights.
110      X_csr = csr_container(X)
111      huber_sparse = HuberRegressor()
112      huber_sparse.fit(X_csr, y, sample_weight=sample_weight)
113      assert_array_almost_equal(huber_sparse.coef_ / scale, huber_coef / scale)

129  def test_huber_scaling_invariant():
130      # Test that outliers filtering is scaling independent.
131      X, y = make_regression_with_outliers()
132      huber = HuberRegressor(fit_intercept=False, alpha=0.0)
133      huber.fit(X, y)
134      n_outliers_mask_1 = huber.outliers_
135      assert not np.all(n_outliers_mask_1)
136  
137      huber.fit(X, 2.0 * y)
138      n_outliers_mask_2 = huber.outliers_
139      assert_array_equal(n_outliers_mask_2, n_outliers_mask_1)
140  
141      huber.fit(2.0 * X, 2.0 * y)
142      n_outliers_mask_3 = huber.outliers_
143      assert_array_equal(n_outliers_mask_3, n_outliers_mask_1)

146  def test_huber_and_sgd_same_results():
147      # Test they should converge to same coefficients for same parameters
148  
149      X, y = make_regression_with_outliers(n_samples=10, n_features=2)
150  
151      # Fit once to find out the scale parameter. Scale down X and y by scale
152      # so that the scale parameter is optimized to 1.0
153      huber = HuberRegressor(fit_intercept=False, alpha=0.0, epsilon=1.35)
154      huber.fit(X, y)
155      X_scale = X / huber.scale_
156      y_scale = y / huber.scale_
157      huber.fit(X_scale, y_scale)
158      assert_almost_equal(huber.scale_, 1.0, 3)
159  
160      sgdreg = SGDRegressor(
161          alpha=0.0,
162          loss="huber",
163          shuffle=True,
164          random_state=0,
165          max_iter=10000,
166          fit_intercept=False,
167          epsilon=1.35,
168          tol=None,
169      )
170      sgdreg.fit(X_scale, y_scale)
171      assert_array_almost_equal(huber.coef_, sgdreg.coef_, 1)

174  def test_huber_warm_start():
175      X, y = make_regression_with_outliers()
176      huber_warm = HuberRegressor(alpha=1.0, max_iter=10000, warm_start=True, tol=1e-1)
177  
178      huber_warm.fit(X, y)
179      huber_warm_coef = huber_warm.coef_.copy()
180      huber_warm.fit(X, y)
181  
182      # SciPy performs the tol check after doing the coef updates, so
183      # these would be almost same but not equal.
184      assert_array_almost_equal(huber_warm.coef_, huber_warm_coef, 1)
185  
186      assert huber_warm.n_iter_ == 0

189  def test_huber_better_r2_score():
190      # Test that huber returns a better r2 score than non-outliers"""
191      X, y = make_regression_with_outliers()
192      huber = HuberRegressor(alpha=0.01)
193      huber.fit(X, y)
194      linear_loss = np.dot(X, huber.coef_) + huber.intercept_ - y
195      mask = np.abs(linear_loss) < huber.epsilon * huber.scale_
196      huber_score = huber.score(X[mask], y[mask])
197      huber_outlier_score = huber.score(X[~mask], y[~mask])
198  
199      # The Ridge regressor should be influenced by the outliers and hence
200      # give a worse score on the non-outliers as compared to the huber
201      # regressor.
202      ridge = Ridge(alpha=0.01)
203      ridge.fit(X, y)
204      ridge_score = ridge.score(X[mask], y[mask])
205      ridge_outlier_score = ridge.score(X[~mask], y[~mask])
206      assert huber_score > ridge_score
207  
208      # The huber model should also fit poorly on the outliers.
209      assert ridge_outlier_score > huber_outlier_score
