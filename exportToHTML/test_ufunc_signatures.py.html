test_ufunc_signatures.py 
1    """Test that all ufuncs have float32-preserving signatures. 
2     
3    This was once guaranteed through the code generation script for 
4    generating ufuncs, `scipy/special/_generate_pyx.py`. Starting with 
5    gh-20260, SciPy developers have begun moving to generate ufuncs 
6    through direct use of the NumPy C API (through C++). Existence of 
7    float32 preserving signatures must now be tested since it is no 
8    longer guaranteed. 
9    """
10   
11   import numpy as np
12   import pytest
13   import scipy.special._ufuncs
14   import scipy.special._gufuncs
15   
16   _ufuncs = []
17   for funcname in dir(scipy.special._ufuncs):
18       _ufuncs.append(getattr(scipy.special._ufuncs, funcname))
19   for funcname in dir(scipy.special._gufuncs):
20       _ufuncs.append(getattr(scipy.special._gufuncs, funcname))
21   
22   # Not all module members are actually ufuncs
23   _ufuncs = [func for func in _ufuncs if isinstance(func, np.ufunc)]
24   
25   @pytest.mark.parametrize("ufunc", _ufuncs)
26   def test_ufunc_signatures(ufunc):
27   
28       # From _generate_pyx.py
29       # "Don't add float32 versions of ufuncs with integer arguments, as this
30       # can lead to incorrect dtype selection if the integer arguments are
31       # arrays, but float arguments are scalars.
32       # For instance sph_harm(0,[0],0,0).dtype == complex64
33       # This may be a NumPy bug, but we need to work around it.
34       # cf. gh-4895, https://github.com/numpy/numpy/issues/5895"
35       types = set(sig for sig in ufunc.types
36                   if not ("l" in sig or "i" in sig or "q" in sig or "p" in sig))
37   
38       # Generate the full expanded set of signatures which should exist. There
39       # should be matching float and double versions of any existing signature.
40       expanded_types = set()
41       for sig in types:
42           expanded_types.update(
43               [sig.replace("d", "f").replace("D", "F"),
44                sig.replace("f", "d").replace("F", "D")]
45           )
46       assert types == expanded_types
