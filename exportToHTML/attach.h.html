15   #ifndef _ATTACH_DLL_H_
16   #define _ATTACH_DLL_H_
17   
18   #if defined DLL_EXPORT
19   #define DECLDIR __declspec(dllexport)
20   #else
21   #define DECLDIR __declspec(dllimport)
22   #endif
23   
24   
25   extern "C"
26   {
27       DECLDIR int AttachAndRunPythonCode(const char *command, int *result );
28       
29       /* 
30        * Helper to print debug information from the current process 
31        */
32       DECLDIR int PrintDebugInfo();
33       
34       /* 
35       Could be used with ctypes (note that the threading should be initialized, so,  
36       doing it in a thread as below is recommended): 
37        
38       def check(): 
39            
40           import ctypes 
41           lib = ctypes.cdll.LoadLibrary(r'C:\...\attach_x86.dll') 
42           print 'result', lib.AttachDebuggerTracing(0) 
43            
44       t = threading.Thread(target=check) 
45       t.start() 
46       t.join() 
47       */
48       DECLDIR int AttachDebuggerTracing(
49           bool showDebugInfo, 
50           void* pSetTraceFunc, // Actually PyObject*, but we don't want to include it here.
51           void* pTraceFunc,  // Actually PyObject*, but we don't want to include it here.
52           unsigned int threadId,
53           void* pPyNone  // Actually PyObject*, but we don't want to include it here.
54       );
55   }
56   
57   #endif
