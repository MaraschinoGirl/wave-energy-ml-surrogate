12   class RocCurveDisplay(_BinaryClassifierCurveDisplayMixin):
13       """ROC Curve visualization. 
14    
15       It is recommend to use 
16       :func:`~sklearn.metrics.RocCurveDisplay.from_estimator` or 
17       :func:`~sklearn.metrics.RocCurveDisplay.from_predictions` to create 
18       a :class:`~sklearn.metrics.RocCurveDisplay`. All parameters are 
19       stored as attributes. 
20    
21       Read more in the :ref:`User Guide <visualizations>`. 
22    
23       Parameters 
24       ---------- 
25       fpr : ndarray 
26           False positive rate. 
27    
28       tpr : ndarray 
29           True positive rate. 
30    
31       roc_auc : float, default=None 
32           Area under ROC curve. If None, the roc_auc score is not shown. 
33    
34       estimator_name : str, default=None 
35           Name of estimator. If None, the estimator name is not shown. 
36    
37       pos_label : int, float, bool or str, default=None 
38           The class considered as the positive class when computing the roc auc 
39           metrics. By default, `estimators.classes_[1]` is considered 
40           as the positive class. 
41    
42           .. versionadded:: 0.24 
43    
44       Attributes 
45       ---------- 
46       line_ : matplotlib Artist 
47           ROC Curve. 
48    
49       chance_level_ : matplotlib Artist or None 
50           The chance level line. It is `None` if the chance level is not plotted. 
51    
52           .. versionadded:: 1.3 
53    
54       ax_ : matplotlib Axes 
55           Axes with ROC Curve. 
56    
57       figure_ : matplotlib Figure 
58           Figure containing the curve. 
59    
60       See Also 
61       -------- 
62       roc_curve : Compute Receiver operating characteristic (ROC) curve. 
63       RocCurveDisplay.from_estimator : Plot Receiver Operating Characteristic 
64           (ROC) curve given an estimator and some data. 
65       RocCurveDisplay.from_predictions : Plot Receiver Operating Characteristic 
66           (ROC) curve given the true and predicted values. 
67       roc_auc_score : Compute the area under the ROC curve. 
68    
69       Examples 
70       -------- 
71       >>> import matplotlib.pyplot as plt 
72       >>> import numpy as np 
73       >>> from sklearn import metrics 
74       >>> y = np.array([0, 0, 1, 1]) 
75       >>> pred = np.array([0.1, 0.4, 0.35, 0.8]) 
76       >>> fpr, tpr, thresholds = metrics.roc_curve(y, pred) 
77       >>> roc_auc = metrics.auc(fpr, tpr) 
78       >>> display = metrics.RocCurveDisplay(fpr=fpr, tpr=tpr, roc_auc=roc_auc, 
79       ...                                   estimator_name='example estimator') 
80       >>> display.plot() 
81       <...> 
82       >>> plt.show() 
83       """
84   
85       def __init__(self, *, fpr, tpr, roc_auc=None, estimator_name=None, pos_label=None):
86           self.estimator_name = estimator_name
87           self.fpr = fpr
88           self.tpr = tpr
89           self.roc_auc = roc_auc
90           self.pos_label = pos_label
91   
92       def plot(
93           self,
94           ax=None,
95           *,
96           name=None,
97           plot_chance_level=False,
98           chance_level_kw=None,
99           despine=False,
100          **kwargs,
101      ):
102          """Plot visualization. 
103   
104          Extra keyword arguments will be passed to matplotlib's ``plot``. 
105   
106          Parameters 
107          ---------- 
108          ax : matplotlib axes, default=None 
109              Axes object to plot on. If `None`, a new figure and axes is 
110              created. 
111   
112          name : str, default=None 
113              Name of ROC Curve for labeling. If `None`, use `estimator_name` if 
114              not `None`, otherwise no labeling is shown. 
115   
116          plot_chance_level : bool, default=False 
117              Whether to plot the chance level. 
118   
119              .. versionadded:: 1.3 
120   
121          chance_level_kw : dict, default=None 
122              Keyword arguments to be passed to matplotlib's `plot` for rendering 
123              the chance level line. 
124   
125              .. versionadded:: 1.3 
126   
127          despine : bool, default=False 
128              Whether to remove the top and right spines from the plot. 
129   
130              .. versionadded:: 1.6 
131   
132          **kwargs : dict 
133              Keyword arguments to be passed to matplotlib's `plot`. 
134   
135          Returns 
136          ------- 
137          display : :class:`~sklearn.metrics.RocCurveDisplay` 
138              Object that stores computed values. 
139          """
140          self.ax_, self.figure_, name = self._validate_plot_params(ax=ax, name=name)
141  
142          default_line_kwargs = {}
143          if self.roc_auc is not None and name is not None:
144              default_line_kwargs["label"] = f"{name} (AUC = {self.roc_auc:0.2f})"
145          elif self.roc_auc is not None:
146              default_line_kwargs["label"] = f"AUC = {self.roc_auc:0.2f}"
147          elif name is not None:
148              default_line_kwargs["label"] = name
149  
150          line_kwargs = _validate_style_kwargs(default_line_kwargs, kwargs)
151  
152          default_chance_level_line_kw = {
153              "label": "Chance level (AUC = 0.5)",
154              "color": "k",
155              "linestyle": "--",
156          }
157  
158          if chance_level_kw is None:
159              chance_level_kw = {}
160  
161          chance_level_kw = _validate_style_kwargs(
162              default_chance_level_line_kw, chance_level_kw
163          )
164  
165          (self.line_,) = self.ax_.plot(self.fpr, self.tpr, **line_kwargs)
166          info_pos_label = (
167              f" (Positive label: {self.pos_label})" if self.pos_label is not None else ""
168          )
169  
170          xlabel = "False Positive Rate" + info_pos_label
171          ylabel = "True Positive Rate" + info_pos_label
172          self.ax_.set(
173              xlabel=xlabel,
174              xlim=(-0.01, 1.01),
175              ylabel=ylabel,
176              ylim=(-0.01, 1.01),
177              aspect="equal",
178          )
179  
180          if plot_chance_level:
181              (self.chance_level_,) = self.ax_.plot((0, 1), (0, 1), **chance_level_kw)
182          else:
183              self.chance_level_ = None
184  
185          if despine:
186              _despine(self.ax_)
187  
188          if "label" in line_kwargs or "label" in chance_level_kw:
189              self.ax_.legend(loc="lower right")
190  
191          return self

193      @classmethod
194      def from_estimator(
195          cls,
196          estimator,
197          X,
198          y,
199          *,
200          sample_weight=None,
201          drop_intermediate=True,
202          response_method="auto",
203          pos_label=None,
204          name=None,
205          ax=None,
206          plot_chance_level=False,
207          chance_level_kw=None,
208          despine=False,
209          **kwargs,
210      ):
211          """Create a ROC Curve display from an estimator. 
212   
213          Parameters 
214          ---------- 
215          estimator : estimator instance 
216              Fitted classifier or a fitted :class:`~sklearn.pipeline.Pipeline` 
217              in which the last estimator is a classifier. 
218   
219          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
220              Input values. 
221   
222          y : array-like of shape (n_samples,) 
223              Target values. 
224   
225          sample_weight : array-like of shape (n_samples,), default=None 
226              Sample weights. 
227   
228          drop_intermediate : bool, default=True 
229              Whether to drop some suboptimal thresholds which would not appear 
230              on a plotted ROC curve. This is useful in order to create lighter 
231              ROC curves. 
232   
233          response_method : {'predict_proba', 'decision_function', 'auto'} \ 
234                  default='auto' 
235              Specifies whether to use :term:`predict_proba` or 
236              :term:`decision_function` as the target response. If set to 'auto', 
237              :term:`predict_proba` is tried first and if it does not exist 
238              :term:`decision_function` is tried next. 
239   
240          pos_label : int, float, bool or str, default=None 
241              The class considered as the positive class when computing the roc auc 
242              metrics. By default, `estimators.classes_[1]` is considered 
243              as the positive class. 
244   
245          name : str, default=None 
246              Name of ROC Curve for labeling. If `None`, use the name of the 
247              estimator. 
248   
249          ax : matplotlib axes, default=None 
250              Axes object to plot on. If `None`, a new figure and axes is created. 
251   
252          plot_chance_level : bool, default=False 
253              Whether to plot the chance level. 
254   
255              .. versionadded:: 1.3 
256   
257          chance_level_kw : dict, default=None 
258              Keyword arguments to be passed to matplotlib's `plot` for rendering 
259              the chance level line. 
260   
261              .. versionadded:: 1.3 
262   
263          despine : bool, default=False 
264              Whether to remove the top and right spines from the plot. 
265   
266              .. versionadded:: 1.6 
267   
268          **kwargs : dict 
269              Keyword arguments to be passed to matplotlib's `plot`. 
270   
271          Returns 
272          ------- 
273          display : :class:`~sklearn.metrics.RocCurveDisplay` 
274              The ROC Curve display. 
275   
276          See Also 
277          -------- 
278          roc_curve : Compute Receiver operating characteristic (ROC) curve. 
279          RocCurveDisplay.from_predictions : ROC Curve visualization given the 
280              probabilities of scores of a classifier. 
281          roc_auc_score : Compute the area under the ROC curve. 
282   
283          Examples 
284          -------- 
285          >>> import matplotlib.pyplot as plt 
286          >>> from sklearn.datasets import make_classification 
287          >>> from sklearn.metrics import RocCurveDisplay 
288          >>> from sklearn.model_selection import train_test_split 
289          >>> from sklearn.svm import SVC 
290          >>> X, y = make_classification(random_state=0) 
291          >>> X_train, X_test, y_train, y_test = train_test_split( 
292          ...     X, y, random_state=0) 
293          >>> clf = SVC(random_state=0).fit(X_train, y_train) 
294          >>> RocCurveDisplay.from_estimator( 
295          ...    clf, X_test, y_test) 
296          <...> 
297          >>> plt.show() 
298          """
299          y_pred, pos_label, name = cls._validate_and_get_response_values(
300              estimator,
301              X,
302              y,
303              response_method=response_method,
304              pos_label=pos_label,
305              name=name,
306          )
307  
308          return cls.from_predictions(
309              y_true=y,
310              y_pred=y_pred,
311              sample_weight=sample_weight,
312              drop_intermediate=drop_intermediate,
313              name=name,
314              ax=ax,
315              pos_label=pos_label,
316              plot_chance_level=plot_chance_level,
317              chance_level_kw=chance_level_kw,
318              despine=despine,
319              **kwargs,
320          )
321  
322      @classmethod
323      def from_predictions(
324          cls,
325          y_true,
326          y_pred,
327          *,
328          sample_weight=None,
329          drop_intermediate=True,
330          pos_label=None,
331          name=None,
332          ax=None,
333          plot_chance_level=False,
334          chance_level_kw=None,
335          despine=False,
336          **kwargs,
337      ):
338          """Plot ROC curve given the true and predicted values. 
339   
340          Read more in the :ref:`User Guide <visualizations>`. 
341   
342          .. versionadded:: 1.0 
343   
344          Parameters 
345          ---------- 
346          y_true : array-like of shape (n_samples,) 
347              True labels. 
348   
349          y_pred : array-like of shape (n_samples,) 
350              Target scores, can either be probability estimates of the positive 
351              class, confidence values, or non-thresholded measure of decisions 
352              (as returned by “decision_function” on some classifiers). 
353   
354          sample_weight : array-like of shape (n_samples,), default=None 
355              Sample weights. 
356   
357          drop_intermediate : bool, default=True 
358              Whether to drop some suboptimal thresholds which would not appear 
359              on a plotted ROC curve. This is useful in order to create lighter 
360              ROC curves. 
361   
362          pos_label : int, float, bool or str, default=None 
363              The label of the positive class. When `pos_label=None`, if `y_true` 
364              is in {-1, 1} or {0, 1}, `pos_label` is set to 1, otherwise an 
365              error will be raised. 
366   
367          name : str, default=None 
368              Name of ROC curve for labeling. If `None`, name will be set to 
369              `"Classifier"`. 
370   
371          ax : matplotlib axes, default=None 
372              Axes object to plot on. If `None`, a new figure and axes is 
373              created. 
374   
375          plot_chance_level : bool, default=False 
376              Whether to plot the chance level. 
377   
378              .. versionadded:: 1.3 
379   
380          chance_level_kw : dict, default=None 
381              Keyword arguments to be passed to matplotlib's `plot` for rendering 
382              the chance level line. 
383   
384              .. versionadded:: 1.3 
385   
386          despine : bool, default=False 
387              Whether to remove the top and right spines from the plot. 
388   
389              .. versionadded:: 1.6 
390   
391          **kwargs : dict 
392              Additional keywords arguments passed to matplotlib `plot` function. 
393   
394          Returns 
395          ------- 
396          display : :class:`~sklearn.metrics.RocCurveDisplay` 
397              Object that stores computed values. 
398   
399          See Also 
400          -------- 
401          roc_curve : Compute Receiver operating characteristic (ROC) curve. 
402          RocCurveDisplay.from_estimator : ROC Curve visualization given an 
403              estimator and some data. 
404          roc_auc_score : Compute the area under the ROC curve. 
405   
406          Examples 
407          -------- 
408          >>> import matplotlib.pyplot as plt 
409          >>> from sklearn.datasets import make_classification 
410          >>> from sklearn.metrics import RocCurveDisplay 
411          >>> from sklearn.model_selection import train_test_split 
412          >>> from sklearn.svm import SVC 
413          >>> X, y = make_classification(random_state=0) 
414          >>> X_train, X_test, y_train, y_test = train_test_split( 
415          ...     X, y, random_state=0) 
416          >>> clf = SVC(random_state=0).fit(X_train, y_train) 
417          >>> y_pred = clf.decision_function(X_test) 
418          >>> RocCurveDisplay.from_predictions( 
419          ...    y_test, y_pred) 
420          <...> 
421          >>> plt.show() 
422          """
423          pos_label_validated, name = cls._validate_from_predictions_params(
424              y_true, y_pred, sample_weight=sample_weight, pos_label=pos_label, name=name
425          )
426  
427          fpr, tpr, _ = roc_curve(
428              y_true,
429              y_pred,
430              pos_label=pos_label,
431              sample_weight=sample_weight,
432              drop_intermediate=drop_intermediate,
433          )
434          roc_auc = auc(fpr, tpr)
435  
436          viz = cls(
437              fpr=fpr,
438              tpr=tpr,
439              roc_auc=roc_auc,
440              estimator_name=name,
441              pos_label=pos_label_validated,
442          )
443  
444          return viz.plot(
445              ax=ax,
446              name=name,
447              plot_chance_level=plot_chance_level,
448              chance_level_kw=chance_level_kw,
449              despine=despine,
450              **kwargs,
451          )
