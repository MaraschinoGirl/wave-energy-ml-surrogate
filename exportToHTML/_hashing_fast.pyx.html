16   def transform(raw_X, Py_ssize_t n_features, dtype,
17                 bint alternate_sign=1, unsigned int seed=0):
18       """Guts of FeatureHasher.transform. 
19    
20       Returns 
21       ------- 
22       n_samples : integer 
23       indices, indptr, values : lists 
24           For constructing a scipy.sparse.csr_matrix. 
25    
26       """
27       cdef int32_t h
28       cdef double value
29   
30       cdef vector[int32_t] indices
31       cdef vector[int64_t] indptr
32       indptr.push_back(0)
33   
34       # Since Python array does not understand Numpy dtypes, we grow the indices
35       # and values arrays ourselves. Use a Py_ssize_t capacity for safety.
36       cdef Py_ssize_t capacity = 8192     # arbitrary
37       cdef int64_t size = 0
38       cdef cnp.ndarray values = np.empty(capacity, dtype=dtype)
39   
40       for x in raw_X:
41           for f, v in x:
42               if isinstance(v, (str, unicode)):
43                   f = "%s%s%s" % (f, '=', v)
44                   value = 1
45               else:
46                   value = v
47   
48               if value == 0:
49                   continue
50   
51               if isinstance(f, unicode):
52                   f = (<unicode>f).encode("utf-8")
53               # Need explicit type check because Murmurhash does not propagate
54               # all exceptions. Add "except *" there?
55               elif not isinstance(f, bytes):
56                   raise TypeError("feature names must be strings")
57   
58               h = murmurhash3_bytes_s32(<bytes>f, seed)
59   
60               if h == - 2147483648:
61                   # abs(-2**31) is undefined behavior because h is a `np.int32`
62                   # The following is defined such that it is equal to: abs(-2**31) % n_features
63                   indices.push_back((2147483647 - (n_features - 1)) % n_features)
64               else:
65                   indices.push_back(abs(h) % n_features)
66               # improve inner product preservation in the hashed space
67               if alternate_sign:
68                   value *= (h >= 0) * 2 - 1
69               values[size] = value
70               size += 1
71   
72               if size == capacity:
73                   capacity *= 2
74                   # can't use resize member because there might be multiple
75                   # references to the arrays due to Cython's error checking
76                   values = np.resize(values, capacity)
77   
78           indptr.push_back(size)
79   
80       indices_array = vector_to_nd_array(&indices)
81       indptr_array = vector_to_nd_array(&indptr)
82   
83       if indptr_array[indptr_array.shape[0]-1] > np.iinfo(np.int32).max:  # = 2**31 - 1
84           # both indices and indptr have the same dtype in CSR arrays
85           indices_array = indices_array.astype(np.int64, copy=False)
86       else:
87           indptr_array = indptr_array.astype(np.int32, copy=False)
88   
89       return (indices_array, indptr_array, values[:size])
