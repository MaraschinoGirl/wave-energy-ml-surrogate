104  def parse_cmdline(argv=None):
105      """ 
106      Parses command line and returns test directories, verbosity, test filter and test suites 
107   
108      usage: 
109          runfiles.py  -v|--verbosity <level>  -t|--tests <Test.test1,Test2>  dirs|files 
110   
111      Multiprocessing options: 
112      jobs=number (with the number of jobs to be used to run the tests) 
113      split_jobs='module'|'tests' 
114          if == module, a given job will always receive all the tests from a module 
115          if == tests, the tests will be split independently of their originating module (default) 
116   
117      --exclude_files  = comma-separated list of patterns with files to exclude (fnmatch style) 
118      --include_files = comma-separated list of patterns with files to include (fnmatch style) 
119      --exclude_tests = comma-separated list of patterns with test names to exclude (fnmatch style) 
120   
121      Note: if --tests is given, --exclude_files, --include_files and --exclude_tests are ignored! 
122      """
123      if argv is None:
124          argv = sys.argv
125  
126      verbosity = 2
127      include_tests = None
128      tests = None
129      port = None
130      jobs = 1
131      split_jobs = "tests"
132      files_to_tests = {}
133      coverage_output_dir = None
134      coverage_include = None
135      exclude_files = None
136      exclude_tests = None
137      include_files = None
138      django = False
139  
140      from _pydev_bundle._pydev_getopt import gnu_getopt
141  
142      optlist, dirs = gnu_getopt(
143          argv[1:],
144          "",
145          [
146              "verbosity=",
147              "tests=",
148              "port=",
149              "config_file=",
150              "jobs=",
151              "split_jobs=",
152              "include_tests=",
153              "include_files=",
154              "exclude_files=",
155              "exclude_tests=",
156              "coverage_output_dir=",
157              "coverage_include=",
158              "django=",
159          ],
160      )
161  
162      for opt, value in optlist:
163          if opt in ("-v", "--verbosity"):
164              verbosity = value
165  
166          elif opt in ("-p", "--port"):
167              port = int(value)
168  
169          elif opt in ("-j", "--jobs"):
170              jobs = int(value)
171  
172          elif opt in ("-s", "--split_jobs"):
173              split_jobs = value
174              if split_jobs not in ("module", "tests"):
175                  raise AssertionError('Expected split to be either "module" or "tests". Was :%s' % (split_jobs,))
176  
177          elif opt in (
178              "-d",
179              "--coverage_output_dir",
180          ):
181              coverage_output_dir = value.strip()
182  
183          elif opt in (
184              "-i",
185              "--coverage_include",
186          ):
187              coverage_include = value.strip()
188  
189          elif opt in ("-I", "--include_tests"):
190              include_tests = value.split(",")
191  
192          elif opt in ("-E", "--exclude_files"):
193              exclude_files = value.split(",")
194  
195          elif opt in ("-F", "--include_files"):
196              include_files = value.split(",")
197  
198          elif opt in ("-e", "--exclude_tests"):
199              exclude_tests = value.split(",")
200  
201          elif opt in ("-t", "--tests"):
202              tests = value.split(",")
203  
204          elif opt in ("--django",):
205              django = value.strip() in ["true", "True", "1"]
206  
207          elif opt in ("-c", "--config_file"):
208              config_file = value.strip()
209              if os.path.exists(config_file):
210                  f = open(config_file, "r")
211                  try:
212                      config_file_contents = f.read()
213                  finally:
214                      f.close()
215  
216                  if config_file_contents:
217                      config_file_contents = config_file_contents.strip()
218  
219                  if config_file_contents:
220                      for line in config_file_contents.splitlines():
221                          file_and_test = line.split("|")
222                          if len(file_and_test) == 2:
223                              file, test = file_and_test
224                              if file in files_to_tests:
225                                  files_to_tests[file].append(test)
226                              else:
227                                  files_to_tests[file] = [test]
228  
229              else:
230                  sys.stderr.write("Could not find config file: %s\n" % (config_file,))
231  
232      filter_tests_env_var = os.environ.get("PYDEV_RUNFILES_FILTER_TESTS", None)
233      if filter_tests_env_var:
234          loaded = json.loads(filter_tests_env_var)
235          include = loaded["include"]
236          for path, name in include:
237              existing = files_to_tests.get(path)
238              if not existing:
239                  existing = files_to_tests[path] = []
240              existing.append(name)
241          # Note: at this point exclude or `*` is not handled.
242          # Clients need to do all the filtering on their side (could
243          # change to have `exclude` and support `*` entries).
244  
245      if type([]) != type(dirs):
246          dirs = [dirs]
247  
248      ret_dirs = []
249      for d in dirs:
250          if "|" in d:
251              # paths may come from the ide separated by |
252              ret_dirs.extend(d.split("|"))
253          else:
254              ret_dirs.append(d)
255  
256      verbosity = int(verbosity)
257  
258      if tests:
259          if verbosity > 4:
260              sys.stdout.write("--tests provided. Ignoring --exclude_files, --exclude_tests and --include_files\n")
261          exclude_files = exclude_tests = include_files = None
262  
263      config = Configuration(
264          ret_dirs,
265          verbosity,
266          include_tests,
267          tests,
268          port,
269          files_to_tests,
270          jobs,
271          split_jobs,
272          coverage_output_dir,
273          coverage_include,
274          exclude_files=exclude_files,
275          exclude_tests=exclude_tests,
276          include_files=include_files,
277          django=django,
278      )
279  
280      if verbosity > 5:
281          sys.stdout.write(str(config) + "\n")
282      return config

288  class PydevTestRunner(object):
289      """finds and runs a file or directory of files as a unit test"""
290  
291      __py_extensions = ["*.py", "*.pyw"]
292      __exclude_files = ["__init__.*"]
293  
294      # Just to check that only this attributes will be written to this file
295      __slots__ = [
296          "verbosity",  # Always used
297          "files_to_tests",  # If this one is given, the ones below are not used
298          "files_or_dirs",  # Files or directories received in the command line
299          "include_tests",  # The filter used to collect the tests
300          "tests",  # Strings with the tests to be run
301          "jobs",  # Integer with the number of jobs that should be used to run the test cases
302          "split_jobs",  # String with 'tests' or 'module' (how should the jobs be split)
303          "configuration",
304          "coverage",
305      ]
306  
307      def __init__(self, configuration):
308          self.verbosity = configuration.verbosity
309  
310          self.jobs = configuration.jobs
311          self.split_jobs = configuration.split_jobs
312  
313          files_to_tests = configuration.files_to_tests
314          if files_to_tests:
315              self.files_to_tests = files_to_tests
316              self.files_or_dirs = list(files_to_tests.keys())
317              self.tests = None
318          else:
319              self.files_to_tests = {}
320              self.files_or_dirs = configuration.files_or_dirs
321              self.tests = configuration.tests
322  
323          self.configuration = configuration
324          self.__adjust_path()
325  
326      def __adjust_path(self):
327          """add the current file or directory to the python path"""
328          path_to_append = None
329          for n in range(len(self.files_or_dirs)):
330              dir_name = self.__unixify(self.files_or_dirs[n])
331              if os.path.isdir(dir_name):
332                  if not dir_name.endswith("/"):
333                      self.files_or_dirs[n] = dir_name + "/"
334                  path_to_append = os.path.normpath(dir_name)
335              elif os.path.isfile(dir_name):
336                  path_to_append = os.path.dirname(dir_name)
337              else:
338                  if not os.path.exists(dir_name):
339                      block_line = "*" * 120
340                      sys.stderr.write("\n%s\n* PyDev test runner error: %s does not exist.\n%s\n" % (block_line, dir_name, block_line))
341                      return
342                  msg = "unknown type. \n%s\nshould be file or a directory.\n" % (dir_name)
343                  raise RuntimeError(msg)
344          if path_to_append is not None:
345              # Add it as the last one (so, first things are resolved against the default dirs and
346              # if none resolves, then we try a relative import).
347              sys.path.append(path_to_append)

349      def __is_valid_py_file(self, fname):
350          """tests that a particular file contains the proper file extension 
351          and is not in the list of files to exclude"""
352          is_valid_fname = 0
353          for invalid_fname in self.__class__.__exclude_files:
354              is_valid_fname += int(not fnmatch.fnmatch(fname, invalid_fname))
355          if_valid_ext = 0
356          for ext in self.__class__.__py_extensions:
357              if_valid_ext += int(fnmatch.fnmatch(fname, ext))
358          return is_valid_fname > 0 and if_valid_ext > 0
359  
360      def __unixify(self, s):
361          """stupid windows. converts the backslash to forwardslash for consistency"""
362          return os.path.normpath(s).replace(os.sep, "/")
363  
364      def __importify(self, s, dir=False):
365          """turns directory separators into dots and removes the ".py*" extension 
366          so the string can be used as import statement"""
367          if not dir:
368              dirname, fname = os.path.split(s)
369  
370              if fname.count(".") > 1:
371                  # if there's a file named xxx.xx.py, it is not a valid module, so, let's not load it...
372                  return
373  
374              imp_stmt_pieces = [dirname.replace("\\", "/").replace("/", "."), os.path.splitext(fname)[0]]
375  
376              if len(imp_stmt_pieces[0]) == 0:
377                  imp_stmt_pieces = imp_stmt_pieces[1:]
378  
379              return ".".join(imp_stmt_pieces)
380  
381          else:  # handle dir
382              return s.replace("\\", "/").replace("/", ".")
383  
384      def __add_files(self, pyfiles, root, files):
385          """if files match, appends them to pyfiles. used by os.path.walk fcn"""
386          for fname in files:
387              if self.__is_valid_py_file(fname):
388                  name_without_base_dir = self.__unixify(os.path.join(root, fname))
389                  pyfiles.append(name_without_base_dir)
390  
391      def find_import_files(self):
392          """return a list of files to import"""
393          if self.files_to_tests:
394              pyfiles = self.files_to_tests.keys()
395          else:
396              pyfiles = []
397  
398              for base_dir in self.files_or_dirs:
399                  if os.path.isdir(base_dir):
400                      for root, dirs, files in os.walk(base_dir):
401                          # Note: handling directories that should be excluded from the search because
402                          # they don't have __init__.py
403                          exclude = {}
404                          for d in dirs:
405                              for init in ["__init__.py", "__init__.pyo", "__init__.pyc", "__init__.pyw", "__init__$py.class"]:
406                                  if os.path.exists(os.path.join(root, d, init).replace("\\", "/")):
407                                      break
408                              else:
409                                  exclude[d] = 1
410  
411                          if exclude:
412                              new = []
413                              for d in dirs:
414                                  if d not in exclude:
415                                      new.append(d)
416  
417                              dirs[:] = new
418  
419                          self.__add_files(pyfiles, root, files)
420  
421                  elif os.path.isfile(base_dir):
422                      pyfiles.append(base_dir)
423  
424          if self.configuration.exclude_files or self.configuration.include_files:
425              ret = []
426              for f in pyfiles:
427                  add = True
428                  basename = os.path.basename(f)
429                  if self.configuration.include_files:
430                      add = False
431  
432                      for pat in self.configuration.include_files:
433                          if fnmatch.fnmatchcase(basename, pat):
434                              add = True
435                              break
436  
437                  if not add:
438                      if self.verbosity > 3:
439                          sys.stdout.write(
440                              "Skipped file: %s (did not match any include_files pattern: %s)\n" % (f, self.configuration.include_files)
441                          )
442  
443                  elif self.configuration.exclude_files:
444                      for pat in self.configuration.exclude_files:
445                          if fnmatch.fnmatchcase(basename, pat):
446                              if self.verbosity > 3:
447                                  sys.stdout.write("Skipped file: %s (matched exclude_files pattern: %s)\n" % (f, pat))
448  
449                              elif self.verbosity > 2:
450                                  sys.stdout.write("Skipped file: %s\n" % (f,))
451  
452                              add = False
453                              break
454  
455                  if add:
456                      if self.verbosity > 3:
457                          sys.stdout.write("Adding file: %s for test discovery.\n" % (f,))
458                      ret.append(f)
459  
460              pyfiles = ret
461  
462          return pyfiles

464      def __get_module_from_str(self, modname, print_exception, pyfile):
465          """Import the module in the given import path. 
466          * Returns the "final" module, so importing "coilib40.subject.visu" 
467          returns the "visu" module, not the "coilib40" as returned by __import__"""
468          try:
469              mod = __import__(modname)
470              for part in modname.split(".")[1:]:
471                  mod = getattr(mod, part)
472              return mod
473          except:
474              if print_exception:
475                  from _pydev_runfiles import pydev_runfiles_xml_rpc
476                  from _pydevd_bundle import pydevd_io
477  
478                  buf_err = pydevd_io.start_redirect(keep_original_redirection=True, std="stderr")
479                  buf_out = pydevd_io.start_redirect(keep_original_redirection=True, std="stdout")
480                  try:
481                      import traceback
482  
483                      traceback.print_exc()
484                      sys.stderr.write("ERROR: Module: %s could not be imported (file: %s).\n" % (modname, pyfile))
485                  finally:
486                      pydevd_io.end_redirect("stderr")
487                      pydevd_io.end_redirect("stdout")
488  
489                  pydev_runfiles_xml_rpc.notifyTest("error", buf_out.getvalue(), buf_err.getvalue(), pyfile, modname, 0)
490  
491              return None

493      def remove_duplicates_keeping_order(self, seq):
494          seen = set()
495          seen_add = seen.add
496          return [x for x in seq if not (x in seen or seen_add(x))]
497  
498      def find_modules_from_files(self, pyfiles):
499          """returns a list of modules given a list of files"""
500          # let's make sure that the paths we want are in the pythonpath...
501          imports = [(s, self.__importify(s)) for s in pyfiles]
502  
503          sys_path = [os.path.normpath(path) for path in sys.path]
504          sys_path = self.remove_duplicates_keeping_order(sys_path)
505  
506          system_paths = []
507          for s in sys_path:
508              system_paths.append(self.__importify(s, True))
509  
510          ret = []
511          for pyfile, imp in imports:
512              if imp is None:
513                  continue  # can happen if a file is not a valid module
514              choices = []
515              for s in system_paths:
516                  if imp.startswith(s):
517                      add = imp[len(s) + 1 :]
518                      if add:
519                          choices.append(add)
520                      # sys.stdout.write(' ' + add + ' ')
521  
522              if not choices:
523                  sys.stdout.write("PYTHONPATH not found for file: %s\n" % imp)
524              else:
525                  for i, import_str in enumerate(choices):
526                      print_exception = i == len(choices) - 1
527                      mod = self.__get_module_from_str(import_str, print_exception, pyfile)
528                      if mod is not None:
529                          ret.append((pyfile, mod, import_str))
530                          break
531  
532          return ret

537      class GetTestCaseNames:
538          """Yes, we need a class for that (cannot use outer context on jython 2.1)"""
539  
540          def __init__(self, accepted_classes, accepted_methods):
541              self.accepted_classes = accepted_classes
542              self.accepted_methods = accepted_methods
543  
544          def __call__(self, testCaseClass):
545              """Return a sorted sequence of method names found within testCaseClass"""
546              testFnNames = []
547              className = testCaseClass.__name__
548  
549              if className in self.accepted_classes:
550                  for attrname in dir(testCaseClass):
551                      # If a class is chosen, we select all the 'test' methods'
552                      if attrname.startswith("test") and hasattr(getattr(testCaseClass, attrname), "__call__"):
553                          testFnNames.append(attrname)
554  
555              else:
556                  for attrname in dir(testCaseClass):
557                      # If we have the class+method name, we must do a full check and have an exact match.
558                      if className + "." + attrname in self.accepted_methods:
559                          if hasattr(getattr(testCaseClass, attrname), "__call__"):
560                              testFnNames.append(attrname)
561  
562              # sorted() is not available in jython 2.1
563              testFnNames.sort()
564              return testFnNames

566      def _decorate_test_suite(self, suite, pyfile, module_name):
567          import unittest
568  
569          if isinstance(suite, unittest.TestSuite):
570              add = False
571              suite.__pydev_pyfile__ = pyfile
572              suite.__pydev_module_name__ = module_name
573  
574              for t in suite._tests:
575                  t.__pydev_pyfile__ = pyfile
576                  t.__pydev_module_name__ = module_name
577                  if self._decorate_test_suite(t, pyfile, module_name):
578                      add = True
579  
580              return add
581  
582          elif isinstance(suite, unittest.TestCase):
583              return True
584  
585          else:
586              return False

588      def find_tests_from_modules(self, file_and_modules_and_module_name):
589          """returns the unittests given a list of modules"""
590          # Use our own suite!
591          from _pydev_runfiles import pydev_runfiles_unittest
592          import unittest
593  
594          unittest.TestLoader.suiteClass = pydev_runfiles_unittest.PydevTestSuite
595          loader = unittest.TestLoader()
596  
597          ret = []
598          if self.files_to_tests:
599              for pyfile, m, module_name in file_and_modules_and_module_name:
600                  accepted_classes = {}
601                  accepted_methods = {}
602                  tests = self.files_to_tests[pyfile]
603                  for t in tests:
604                      accepted_methods[t] = t
605  
606                  loader.getTestCaseNames = self.GetTestCaseNames(accepted_classes, accepted_methods)
607  
608                  suite = loader.loadTestsFromModule(m)
609                  if self._decorate_test_suite(suite, pyfile, module_name):
610                      ret.append(suite)
611              return ret
612  
613          if self.tests:
614              accepted_classes = {}
615              accepted_methods = {}
616  
617              for t in self.tests:
618                  splitted = t.split(".")
619                  if len(splitted) == 1:
620                      accepted_classes[t] = t
621  
622                  elif len(splitted) == 2:
623                      accepted_methods[t] = t
624  
625              loader.getTestCaseNames = self.GetTestCaseNames(accepted_classes, accepted_methods)
626  
627          for pyfile, m, module_name in file_and_modules_and_module_name:
628              suite = loader.loadTestsFromModule(m)
629              if self._decorate_test_suite(suite, pyfile, module_name):
630                  ret.append(suite)
631  
632          return ret

634      def filter_tests(self, test_objs, internal_call=False):
635          """based on a filter name, only return those tests that have 
636          the test case names that match"""
637          import unittest
638  
639          if not internal_call:
640              if not self.configuration.include_tests and not self.tests and not self.configuration.exclude_tests:
641                  # No need to filter if we have nothing to filter!
642                  return test_objs
643  
644              if self.verbosity > 1:
645                  if self.configuration.include_tests:
646                      sys.stdout.write("Tests to include: %s\n" % (self.configuration.include_tests,))
647  
648                  if self.tests:
649                      sys.stdout.write("Tests to run: %s\n" % (self.tests,))
650  
651                  if self.configuration.exclude_tests:
652                      sys.stdout.write("Tests to exclude: %s\n" % (self.configuration.exclude_tests,))
653  
654          test_suite = []
655          for test_obj in test_objs:
656              if isinstance(test_obj, unittest.TestSuite):
657                  # Note: keep the suites as they are and just 'fix' the tests (so, don't use the iter_tests).
658                  if test_obj._tests:
659                      test_obj._tests = self.filter_tests(test_obj._tests, True)
660                      if test_obj._tests:  # Only add the suite if we still have tests there.
661                          test_suite.append(test_obj)
662  
663              elif isinstance(test_obj, unittest.TestCase):
664                  try:
665                      testMethodName = test_obj._TestCase__testMethodName
666                  except AttributeError:
667                      # changed in python 2.5
668                      testMethodName = test_obj._testMethodName
669  
670                  add = True
671                  if self.configuration.exclude_tests:
672                      for pat in self.configuration.exclude_tests:
673                          if fnmatch.fnmatchcase(testMethodName, pat):
674                              if self.verbosity > 3:
675                                  sys.stdout.write("Skipped test: %s (matched exclude_tests pattern: %s)\n" % (testMethodName, pat))
676  
677                              elif self.verbosity > 2:
678                                  sys.stdout.write("Skipped test: %s\n" % (testMethodName,))
679  
680                              add = False
681                              break
682  
683                  if add:
684                      if self.__match_tests(self.tests, test_obj, testMethodName):
685                          include = True
686                          if self.configuration.include_tests:
687                              include = False
688                              for pat in self.configuration.include_tests:
689                                  if fnmatch.fnmatchcase(testMethodName, pat):
690                                      include = True
691                                      break
692                          if include:
693                              test_suite.append(test_obj)
694                          else:
695                              if self.verbosity > 3:
696                                  sys.stdout.write(
697                                      "Skipped test: %s (did not match any include_tests pattern %s)\n"
698                                      % (
699                                          testMethodName,
700                                          self.configuration.include_tests,
701                                      )
702                                  )
703          return test_suite
704  
705      def iter_tests(self, test_objs):
706          # Note: not using yield because of Jython 2.1.
707          import unittest
708  
709          tests = []
710          for test_obj in test_objs:
711              if isinstance(test_obj, unittest.TestSuite):
712                  tests.extend(self.iter_tests(test_obj._tests))
713  
714              elif isinstance(test_obj, unittest.TestCase):
715                  tests.append(test_obj)
716          return tests
717  
718      def list_test_names(self, test_objs):
719          names = []
720          for tc in self.iter_tests(test_objs):
721              try:
722                  testMethodName = tc._TestCase__testMethodName
723              except AttributeError:
724                  # changed in python 2.5
725                  testMethodName = tc._testMethodName
726              names.append(testMethodName)
727          return names
728  
729      def __match_tests(self, tests, test_case, test_method_name):
730          if not tests:
731              return 1
732  
733          for t in tests:
734              class_and_method = t.split(".")
735              if len(class_and_method) == 1:
736                  # only class name
737                  if class_and_method[0] == test_case.__class__.__name__:
738                      return 1
739  
740              elif len(class_and_method) == 2:
741                  if class_and_method[0] == test_case.__class__.__name__ and class_and_method[1] == test_method_name:
742                      return 1
743  
744          return 0
745  
746      def __match(self, filter_list, name):
747          """returns whether a test name matches the test filter"""
748          if filter_list is None:
749              return 1
750          for f in filter_list:
751              if re.match(f, name):
752                  return 1
753          return 0
754  
755      def run_tests(self, handle_coverage=True):
756          """runs all tests"""
757          sys.stdout.write("Finding files... ")
758          files = self.find_import_files()
759          if self.verbosity > 3:
760              sys.stdout.write("%s ... done.\n" % (self.files_or_dirs))
761          else:
762              sys.stdout.write("done.\n")
763          sys.stdout.write("Importing test modules ... ")
764  
765          if self.configuration.django:
766              import django
767  
768              if hasattr(django, "setup"):
769                  django.setup()
770  
771          if handle_coverage:
772              coverage_files, coverage = start_coverage_support(self.configuration)
773  
774          file_and_modules_and_module_name = self.find_modules_from_files(files)
775          sys.stdout.write("done.\n")
776  
777          all_tests = self.find_tests_from_modules(file_and_modules_and_module_name)
778          all_tests = self.filter_tests(all_tests)
779  
780          from _pydev_runfiles import pydev_runfiles_unittest
781  
782          test_suite = pydev_runfiles_unittest.PydevTestSuite(all_tests)
783          from _pydev_runfiles import pydev_runfiles_xml_rpc
784  
785          pydev_runfiles_xml_rpc.notifyTestsCollected(test_suite.countTestCases())
786  
787          start_time = time.time()
788  
789          def run_tests():
790              executed_in_parallel = False
791              if self.jobs > 1:
792                  from _pydev_runfiles import pydev_runfiles_parallel
793  
794                  # What may happen is that the number of jobs needed is lower than the number of jobs requested
795                  # (e.g.: 2 jobs were requested for running 1 test) -- in which case execute_tests_in_parallel will
796                  # return False and won't run any tests.
797                  executed_in_parallel = pydev_runfiles_parallel.execute_tests_in_parallel(
798                      all_tests, self.jobs, self.split_jobs, self.verbosity, coverage_files, self.configuration.coverage_include
799                  )
800  
801              if not executed_in_parallel:
802                  # If in coverage, we don't need to pass anything here (coverage is already enabled for this execution).
803                  runner = pydev_runfiles_unittest.PydevTextTestRunner(stream=sys.stdout, descriptions=1, verbosity=self.verbosity)
804                  sys.stdout.write("\n")
805                  runner.run(test_suite)
806  
807          if self.configuration.django:
808              get_django_test_suite_runner()(run_tests).run_tests([])
809          else:
810              run_tests()
811  
812          if handle_coverage:
813              coverage.stop()
814              coverage.save()
815  
816          total_time = "Finished in: %.2f secs." % (time.time() - start_time,)
817          pydev_runfiles_xml_rpc.notifyTestRunFinished(total_time)

824      global DJANGO_TEST_SUITE_RUNNER

826          return DJANGO_TEST_SUITE_RUNNER
827      try:

834                  django.setup()
835                  DiscoverRunner.__init__(self)

838              def build_suite(self, *args, **kwargs):
839                  pass
840  
841              def suite_result(self, *args, **kwargs):
842                  pass
843  
844              def run_suite(self, *args, **kwargs):
845                  self.on_run_suite()
846  
847      except:
848          # django < 1.8
849          try:
850              from django.test.simple import DjangoTestSuiteRunner
851          except:
852

854                  def __init__(self):
855                      pass
856

858                      raise AssertionError(

868                  pass
869

871                  pass
872
