34   def _is_syntax_valid(filepath):
35     with open(filepath, 'r') as f:
36       try:
37         ast.parse(f.read())
38       except SyntaxError:
39         return False
40     return True

43   class AutoFormatTestMeta(type):
44   
45     def __new__(mcs, name, bases, inst_dict):
46       # Helper function to generate a test method
47       def auto_format_test_generator(input_file):
48         def test(self):
49           with open(input_file, 'r') as handle:
50             src = handle.read()
51           t = ast.parse(src)
52           auto_formatted = codegen.to_str(t)
53           self.assertMultiLineEqual(src, auto_formatted)
54         return test
55   
56       # Add a test method for each input file
57       test_method_prefix = 'test_auto_format_'
58       data_dir = os.path.join(TESTDATA_DIR, 'codegen')
59       for dirpath, _, files in os.walk(data_dir):
60         for filename in files:
61           if filename.endswith('.in'):
62             full_path = os.path.join(dirpath, filename)
63             inst_dict[test_method_prefix + filename[:-3]] = unittest.skipIf(
64                 not _is_syntax_valid(full_path),
65                 'Test contains syntax not supported by this version.',
66                 )(auto_format_test_generator(full_path))
67       return type.__new__(mcs, name, bases, inst_dict)
68

70   class AutoFormatTest(with_metaclass(AutoFormatTestMeta, test_utils.TestCase)):
71     """Tests that code without formatting info is printed neatly."""
72   
73     def test_imports(self):
74       src = 'from a import b\nimport c, d\nfrom ..e import f, g\n'
75       t = ast.parse(src)
76       self.assertEqual(src, pasta.dump(t))
77   
78     @test_utils.requires_features('exec_node')
79     def test_exec_node_default(self):
80       src = 'exec foo in bar'
81       t = ast.parse(src)
82       self.assertEqual('exec(foo, bar)\n', pasta.dump(t))
83   
84     @test_utils.requires_features('bytes_node')
85     def test_bytes(self):
86       src = "b'foo'"
87       t = ast.parse(src)
88       self.assertEqual("b'foo'\n", pasta.dump(t))
89   
90     def test_default_indentation(self):
91       for indent in ('  ', '    ', '\t'):
92         src ='def a():\n' + indent + 'b\n'
93         t = pasta.parse(src)
94         t.body.extend(ast.parse('def c(): d').body)
95         self.assertEqual(codegen.to_str(t),
96                          src + 'def c():\n' + indent + 'd\n')

99   def suite():

101    result.addTests(unittest.makeSuite(AutoFormatTest))
