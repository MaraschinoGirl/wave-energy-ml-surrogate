4    class DegenerateDataWarning(RuntimeWarning):
5        """Warns when data is degenerate and results may not be reliable."""
6        def __init__(self, msg=None):
7            if msg is None:
8                msg = ("Degenerate data encountered; results may not be reliable.")
9            self.args = (msg,)
