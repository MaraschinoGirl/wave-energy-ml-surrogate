14   def toarray(a):
15       if isinstance(a, np.ndarray) or isscalarlike(a):
16           return a
17       return a.toarray()

20   formats_for_minmax = [bsr_array, coo_array, csc_array, csr_array]
21   formats_for_minmax_supporting_1d = [coo_array, csr_array]

24   @pytest.mark.parametrize("spcreator", formats_for_minmax_supporting_1d)
25   class Test_MinMaxMixin1D:
26       def test_minmax(self, spcreator):
27           D = np.arange(5)
28           X = spcreator(D)
29   
30           assert_equal(X.min(), 0)
31           assert_equal(X.max(), 4)
32           assert_equal((-X).min(), -4)
33           assert_equal((-X).max(), 0)
34   
35       def test_minmax_axis(self, spcreator):
36           D = np.arange(50)
37           X = spcreator(D)
38   
39           for axis in [0, -1]:
40               assert_array_equal(
41                   toarray(X.max(axis=axis)), D.max(axis=axis, keepdims=True)
42               )
43               assert_array_equal(
44                   toarray(X.min(axis=axis)), D.min(axis=axis, keepdims=True)
45               )
46           for axis in [-2, 1]:
47               with pytest.raises(ValueError, match="axis out of range"):
48                   X.min(axis=axis)
49               with pytest.raises(ValueError, match="axis out of range"):
50                   X.max(axis=axis)
51   
52       def test_numpy_minmax(self, spcreator):
53           dat = np.array([0, 1, 2])
54           datsp = spcreator(dat)
55           assert_array_equal(np.min(datsp), np.min(dat))
56           assert_array_equal(np.max(datsp), np.max(dat))

59       def test_argmax(self, spcreator):
60           D1 = np.array([-1, 5, 2, 3])
61           D2 = np.array([0, 0, -1, -2])
62           D3 = np.array([-1, -2, -3, -4])
63           D4 = np.array([1, 2, 3, 4])
64           D5 = np.array([1, 2, 0, 0])
65   
66           for D in [D1, D2, D3, D4, D5]:
67               mat = spcreator(D)
68   
69               assert_equal(mat.argmax(), np.argmax(D))
70               assert_equal(mat.argmin(), np.argmin(D))
71   
72               assert_equal(mat.argmax(axis=0), np.argmax(D, axis=0))
73               assert_equal(mat.argmin(axis=0), np.argmin(D, axis=0))
74   
75           D6 = np.empty((0,))
76   
77           for axis in [None, 0]:
78               mat = spcreator(D6)
79               with pytest.raises(ValueError, match="to an empty matrix"):
80                   mat.argmin(axis=axis)
81               with pytest.raises(ValueError, match="to an empty matrix"):
82                   mat.argmax(axis=axis)

85   @pytest.mark.parametrize("spcreator", formats_for_minmax)
86   class Test_ShapeMinMax2DWithAxis:
87       def test_minmax(self, spcreator):
88           dat = np.array([[-1, 5, 0, 3], [0, 0, -1, -2], [0, 0, 1, 2]])
89           datsp = spcreator(dat)
90   
91           for (spminmax, npminmax) in [
92               (datsp.min, np.min),
93               (datsp.max, np.max),
94               (datsp.nanmin, np.nanmin),
95               (datsp.nanmax, np.nanmax),
96           ]:
97               for ax, result_shape in [(0, (4,)), (1, (3,))]:
98                   assert_equal(toarray(spminmax(axis=ax)), npminmax(dat, axis=ax))
99                   assert_equal(spminmax(axis=ax).shape, result_shape)
100                  assert spminmax(axis=ax).format == "coo"
101  
102          for spminmax in [datsp.argmin, datsp.argmax]:
103              for ax in [0, 1]:
104                  assert isinstance(spminmax(axis=ax), np.ndarray)
105  
106          # verify spmatrix behavior
107          spmat_form = {
108              'coo': coo_matrix,
109              'csr': csr_matrix,
110              'csc': csc_matrix,
111              'bsr': bsr_matrix,
112          }
113          datspm = spmat_form[datsp.format](dat)
114  
115          for spm, npm in [
116              (datspm.min, np.min),
117              (datspm.max, np.max),
118              (datspm.nanmin, np.nanmin),
119              (datspm.nanmax, np.nanmax),
120          ]:
121              for ax, result_shape in [(0, (1, 4)), (1, (3, 1))]:
122                  assert_equal(toarray(spm(axis=ax)), npm(dat, axis=ax, keepdims=True))
123                  assert_equal(spm(axis=ax).shape, result_shape)
124                  assert spm(axis=ax).format == "coo"
125  
126          for spminmax in [datspm.argmin, datspm.argmax]:
127              for ax in [0, 1]:
128                  assert isinstance(spminmax(axis=ax), np.ndarray)
