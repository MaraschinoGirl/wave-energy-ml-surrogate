10   def setup(app):
11       """Setup as a sphinx extension."""

13       # This is only a lexer, so adding it below to pygments appears sufficient.
14       # But if somebody knows what the right API usage should be to do that via

17       metadata = {"parallel_read_safe": True, "parallel_write_safe": True}
18       return metadata
19   
20   
21   # Register the extension as a valid pygments lexer.

23   # require using setuptools entrypoints: http://pygments.org/docs/plugins
24   
25   ipy3 = IPyLexer()
26   
27   highlighting.lexers["ipython"] = ipy3
28   highlighting.lexers["ipython3"] = ipy3
