14   def export_onnx(model, filepath, verbose=None, input_signature=None, **kwargs):
15       """Export the model as a ONNX artifact for inference. 
16    
17       This method lets you export a model to a lightweight ONNX artifact 
18       that contains the model's forward pass only (its `call()` method) 
19       and can be served via e.g. ONNX Runtime. 
20    
21       The original code of the model (including any custom layers you may 
22       have used) is *no longer* necessary to reload the artifact -- it is 
23       entirely standalone. 
24    
25       Args: 
26           filepath: `str` or `pathlib.Path` object. The path to save the artifact. 
27           verbose: `bool`. Whether to print a message during export. Defaults to 
28               `None`, which uses the default value set by different backends and 
29               formats. 
30           input_signature: Optional. Specifies the shape and dtype of the model 
31               inputs. Can be a structure of `keras.InputSpec`, `tf.TensorSpec`, 
32               `backend.KerasTensor`, or backend tensor. If not provided, it will 
33               be automatically computed. Defaults to `None`. 
34           **kwargs: Additional keyword arguments. 
35    
36       **Note:** This feature is currently supported only with TensorFlow, JAX and 
37       Torch backends. 
38    
39       **Note:** The dtype policy must be "float32" for the model. You can further 
40       optimize the ONNX artifact using the ONNX toolkit. Learn more here: 
41       [https://onnxruntime.ai/docs/performance/](https://onnxruntime.ai/docs/performance/). 
42    
43       **Note:** The dynamic shape feature is not yet supported with Torch 
44       backend. As a result, you must fully define the shapes of the inputs using 
45       `input_signature`. If `input_signature` is not provided, all instances of 
46       `None` (such as the batch size) will be replaced with `1`. 
47    
48       Example: 
49    
50       ```python 
51       # Export the model as a ONNX artifact 
52       model.export("path/to/location", format="onnx") 
53    
54       # Load the artifact in a different process/environment 
55       ort_session = onnxruntime.InferenceSession("path/to/location") 
56       ort_inputs = { 
57           k.name: v for k, v in zip(ort_session.get_inputs(), input_data) 
58       } 
59       predictions = ort_session.run(None, ort_inputs) 
60       ``` 
61       """
62       actual_verbose = verbose
63       if actual_verbose is None:
64           actual_verbose = True  # Defaults to `True` for all backends.
65   
66       if input_signature is None:
67           input_signature = get_input_signature(model)
68           if not input_signature or not model._called:
69               raise ValueError(
70                   "The model provided has never called. "
71                   "It must be called at least once before export."
72               )
73   
74       if backend.backend() in ("tensorflow", "jax"):
75           from keras.src.utils.module_utils import tf2onnx
76   
77           input_signature = tree.map_structure(
78               make_tf_tensor_spec, input_signature
79           )
80           decorated_fn = get_concrete_fn(model, input_signature, **kwargs)
81   
82           # Use `tf2onnx` to convert the `decorated_fn` to the ONNX format.
83           patch_tf2onnx()  # TODO: Remove this once `tf2onnx` supports numpy 2.
84           tf2onnx.convert.from_function(
85               decorated_fn, input_signature, output_path=filepath
86           )
87   
88       elif backend.backend() == "torch":
89           import torch
90   
91           sample_inputs = tree.map_structure(
92               lambda x: convert_spec_to_tensor(x, replace_none_number=1),
93               input_signature,
94           )
95           sample_inputs = tuple(sample_inputs)
96           # TODO: Make dict model exportable.
97           if any(isinstance(x, dict) for x in sample_inputs):
98               raise ValueError(
99                   "Currently, `export_onnx` in the torch backend doesn't support "
100                  "dictionaries as inputs."
101              )
102  
103          if hasattr(model, "eval"):
104              model.eval()
105          with warnings.catch_warnings():
106              # Suppress some unuseful warnings.
107              warnings.filterwarnings(
108                  "ignore",
109                  message=r".*\n.*\n*.*\n*.*export will treat it as a constant.*",
110              )
111              warnings.filterwarnings(
112                  "ignore",
113                  message=r".*not properly registered as a submodule,.*",
114              )
115              warnings.filterwarnings(
116                  "ignore",
117                  message=r".*which is what 'get_attr' Nodes typically target.*",
118              )
119              warnings.filterwarnings(
120                  "ignore",
121                  message=r".*underlying reference in the owning GraphModule.*",
122              )
123              warnings.filterwarnings(
124                  "ignore", message=r".*suppressed about get_attr references.*"
125              )
126              try:
127                  # Try the TorchDynamo-based ONNX exporter first.
128                  onnx_program = torch.onnx.export(
129                      model, sample_inputs, verbose=actual_verbose, dynamo=True
130                  )
131                  if hasattr(onnx_program, "optimize"):
132                      onnx_program.optimize()  # Only supported by torch>=2.6.0.
133                  onnx_program.save(filepath)
134              except:
135                  if verbose is None:
136                      # Set to `False` due to file system leakage issue:
137                      # https://github.com/keras-team/keras/issues/20826
138                      actual_verbose = False
139  
140                  # Fall back to the TorchScript-based ONNX exporter.
141                  torch.onnx.export(
142                      model, sample_inputs, filepath, verbose=actual_verbose
143                  )
144      else:
145          raise NotImplementedError(
146              "`export_onnx` is only compatible with TensorFlow, JAX and "
147              "Torch backends."
148          )
149  
150      if actual_verbose:
151          io_utils.print_msg(f"Saved artifact at '{filepath}'.")
152

154  def _check_jax_kwargs(kwargs):
155      kwargs = kwargs.copy()
156      if "is_static" not in kwargs:
157          kwargs["is_static"] = True
158      if "jax2tf_kwargs" not in kwargs:
159          # TODO: These options will be deprecated in JAX. We need to
160          # find another way to export ONNX.
161          kwargs["jax2tf_kwargs"] = {
162              "enable_xla": False,
163              "native_serialization": False,
164          }
165      if kwargs["is_static"] is not True:
166          raise ValueError(
167              "`is_static` must be `True` in `kwargs` when using the jax backend."
168          )
169      if kwargs["jax2tf_kwargs"]["enable_xla"] is not False:
170          raise ValueError(
171              "`enable_xla` must be `False` in `kwargs['jax2tf_kwargs']` "
172              "when using the jax backend."
173          )
174      if kwargs["jax2tf_kwargs"]["native_serialization"] is not False:
175          raise ValueError(
176              "`native_serialization` must be `False` in "
177              "`kwargs['jax2tf_kwargs']` when using the jax backend."
178          )
179      return kwargs

182  def get_concrete_fn(model, input_signature, **kwargs):
183      """Get the `tf.function` associated with the model."""
184      if backend.backend() == "jax":
185          kwargs = _check_jax_kwargs(kwargs)
186      export_archive = ExportArchive()
187      export_archive.track_and_add_endpoint(
188          DEFAULT_ENDPOINT_NAME, model, input_signature, **kwargs
189      )
190      if backend.backend() == "tensorflow":
191          export_archive._filter_and_track_resources()
192      return export_archive._get_concrete_fn(DEFAULT_ENDPOINT_NAME)
