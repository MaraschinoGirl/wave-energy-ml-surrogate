8    import numpy as np
9    from scipy.special import gammaln
10   
11   
12   def expected_mutual_information(contingency, int64_t n_samples):
13       """Calculate the expected mutual information for two labelings."""
14       cdef:
15           float64_t emi = 0
16           int64_t n_rows, n_cols
17           float64_t term2, term3, gln
18           int64_t[::1] a_view, b_view
19           float64_t[::1] term1
20           float64_t[::1] gln_a, gln_b, gln_Na, gln_Nb, gln_Nnij, log_Nnij
21           float64_t[::1] log_a, log_b
22           Py_ssize_t i, j, nij
23           int64_t start, end
24   
25       n_rows, n_cols = contingency.shape
26       a = np.ravel(contingency.sum(axis=1).astype(np.int64, copy=False))
27       b = np.ravel(contingency.sum(axis=0).astype(np.int64, copy=False))
28       a_view = a
29       b_view = b
30   
31       # any labelling with zero entropy implies EMI = 0
32       if a.size == 1 or b.size == 1:
33           return 0.0
34   
35       # There are three major terms to the EMI equation, which are multiplied to
36       # and then summed over varying nij values.
37       # While nijs[0] will never be used, having it simplifies the indexing.
38       nijs = np.arange(0, max(np.max(a), np.max(b)) + 1, dtype='float')
39       nijs[0] = 1  # Stops divide by zero warnings. As its not used, no issue.
40       # term1 is nij / N
41       term1 = nijs / n_samples
42       # term2 is log((N*nij) / (a * b)) == log(N * nij) - log(a * b)
43       log_a = np.log(a)
44       log_b = np.log(b)
45       # term2 uses log(N * nij) = log(N) + log(nij)
46       log_Nnij = np.log(n_samples) + np.log(nijs)
47       # term3 is large, and involved many factorials. Calculate these in log
48       # space to stop overflows.
49       gln_a = gammaln(a + 1)
50       gln_b = gammaln(b + 1)
51       gln_Na = gammaln(n_samples - a + 1)
52       gln_Nb = gammaln(n_samples - b + 1)
53       gln_Nnij = gammaln(nijs + 1) + gammaln(n_samples + 1)
54   
55       # emi itself is a summation over the various values.
56       for i in range(n_rows):
57           for j in range(n_cols):
58               start = max(1, a_view[i] - n_samples + b_view[j])
59               end = min(a_view[i], b_view[j]) + 1
60               for nij in range(start, end):
61                   term2 = log_Nnij[nij] - log_a[i] - log_b[j]
62                   # Numerators are positive, denominators are negative.
63                   gln = (gln_a[i] + gln_b[j] + gln_Na[i] + gln_Nb[j]
64                          - gln_Nnij[nij] - lgamma(a_view[i] - nij + 1)
65                          - lgamma(b_view[j] - nij + 1)
66                          - lgamma(n_samples - a_view[i] - b_view[j] + nij + 1))
67                   term3 = exp(gln)
68                   emi += (term1[nij] * term2 * term3)
69       return emi
