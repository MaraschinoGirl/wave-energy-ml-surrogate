4    
5    from keras.src import backend
6    from keras.src import ops
7    from keras.src.api_export import keras_export
8    from keras.src.visualization.draw_segmentation_masks import (
9        draw_segmentation_masks,
10   )
11   from keras.src.visualization.plot_image_gallery import plot_image_gallery

14   @keras_export("keras.visualization.plot_segmentation_mask_gallery")
15   def plot_segmentation_mask_gallery(
16       images,
17       num_classes,
18       value_range=(0, 255),
19       y_true=None,
20       y_pred=None,
21       color_mapping=None,
22       blend=True,
23       alpha=0.8,
24       ignore_index=-1,
25       data_format=None,
26       **kwargs,
27   ):
28       """Plots a gallery of images with corresponding segmentation masks. 
29    
30       Args: 
31           images: A 4D tensor or NumPy array of images. Shape should be 
32               `(batch_size, height, width, channels)`. 
33           num_classes: The number of segmentation classes.  Class indices should 
34               start from `1`.  Class `0` will be treated as background and 
35               ignored if `ignore_index` is not 0. 
36           value_range: A tuple specifying the value range of the images 
37               (e.g., `(0, 255)` or `(0, 1)`). Defaults to `(0, 255)`. 
38           y_true: A 3D/4D tensor or NumPy array representing the ground truth 
39               segmentation masks. Shape should be `(batch_size, height, width)` or 
40               `(batch_size, height, width, 1)`. Defaults to `None`. 
41           y_pred: A 3D/4D tensor or NumPy array representing the predicted 
42               segmentation masks.  Shape should be the same as `y_true`. 
43               Defaults to `None`. 
44           color_mapping: A dictionary mapping class indices to RGB colors. 
45               If `None`, a default color palette is used. Class indices start 
46               from `1`. Defaults to `None`. 
47           blend: Whether to blend the masks with the input image using the 
48               `alpha` value. If `False`, the masks are drawn directly on the 
49               images without blending. Defaults to `True`. 
50           alpha: The opacity of the segmentation masks (a float between 0 and 1). 
51               Defaults to `0.8`. 
52           ignore_index: The class index to ignore when drawing masks. 
53               Defaults to `-1`. 
54           data_format: The image data format `"channels_last"` or 
55               `"channels_first"`. Defaults to the Keras backend data format. 
56           kwargs: Additional keyword arguments to be passed to 
57               `keras.visualization.plot_image_gallery`. 
58    
59       Returns: 
60           The output of `keras.visualization.plot_image_gallery`. 
61    
62       Raises: 
63           ValueError: If `images` is not a 4D tensor/array. 
64       """
65       data_format = data_format or backend.image_data_format()
66       image_shape = ops.shape(images)
67       if len(image_shape) != 4:
68           raise ValueError(
69               "`images` must be batched 4D tensor. "
70               f"Received: images.shape={image_shape}"
71           )
72       if data_format == "channels_first":
73           images = ops.transpose(images, (0, 2, 3, 1))
74   
75       batch_size = image_shape[0] if len(image_shape) == 4 else 1
76   
77       rows = batch_size
78       cols = 1
79   
80       if y_true is not None:
81           cols += 1
82   
83       if y_pred is not None:
84           cols += 1
85   
86       images_np = ops.convert_to_numpy(images)
87   
88       draw_masks_fn = functools.partial(
89           draw_segmentation_masks,
90           num_classes=num_classes,
91           color_mapping=color_mapping,
92           alpha=alpha,
93           ignore_index=ignore_index,
94           blend=blend,
95       )
96   
97       if y_true is not None:
98           if data_format == "channels_first":
99               y_true = ops.transpose(y_true, (0, 2, 3, 1))
100          y_true = ops.cast(y_true, "int32")
101          true_masks_drawn = draw_masks_fn(images_np, y_true)
102  
103      if y_pred is not None:
104          if data_format == "channels_first":
105              y_pred = ops.transpose(y_pred, (0, 2, 3, 1))
106          y_pred = ops.cast(y_pred, "int32")
107          predicted_masks_drawn = draw_masks_fn(images_np, y_pred)
108  
109      images_with_masks = []
110      for i in range(batch_size):
111          images_with_masks.append(images_np[i])
112          if y_true is not None:
113              images_with_masks.append(true_masks_drawn[i])
114          if y_pred is not None:
115              images_with_masks.append(predicted_masks_drawn[i])
116  
117      gallery_images = np.stack(images_with_masks, axis=0)
118  
119      return plot_image_gallery(
120          gallery_images, value_range=value_range, rows=rows, cols=cols, **kwargs
121      )
