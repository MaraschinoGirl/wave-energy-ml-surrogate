19   def _get_sys_info():
20       """System information 
21    
22       Returns 
23       ------- 
24       sys_info : dict 
25           system and Python version information 
26    
27       """
28       python = sys.version.replace("\n", " ")
29   
30       blob = [
31           ("python", python),
32           ("executable", sys.executable),
33           ("machine", platform.platform()),
34       ]
35   
36       return dict(blob)
37   
38   
39   def _get_deps_info():
40       """Overview of the installed version of main dependencies 
41    
42       This function does not import the modules to collect the version numbers 
43       but instead relies on standard Python package metadata. 
44    
45       Returns 
46       ------- 
47       deps_info: dict 
48           version information on relevant Python libraries 
49    
50       """
51       deps = [
52           "pip",
53           "setuptools",
54           "numpy",
55           "scipy",
56           "Cython",
57           "pandas",
58           "matplotlib",
59           "joblib",
60           "threadpoolctl",
61       ]
62   
63       deps_info = {
64           "sklearn": __version__,
65       }
66   
67       from importlib.metadata import PackageNotFoundError, version
68   
69       for modname in deps:
70           try:
71               deps_info[modname] = version(modname)
72           except PackageNotFoundError:
73               deps_info[modname] = None
74       return deps_info

77   def show_versions():
78       """Print useful debugging information" 
79    
80       .. versionadded:: 0.20 
81    
82       Examples 
83       -------- 
84       >>> from sklearn import show_versions 
85       >>> show_versions()  # doctest: +SKIP 
86       """
87   
88       sys_info = _get_sys_info()
89       deps_info = _get_deps_info()
90   
91       print("\nSystem:")
92       for k, stat in sys_info.items():
93           print("{k:>10}: {stat}".format(k=k, stat=stat))
94   
95       print("\nPython dependencies:")
96       for k, stat in deps_info.items():
97           print("{k:>13}: {stat}".format(k=k, stat=stat))
98   
99       print(
100          "\n{k}: {stat}".format(
101              k="Built with OpenMP", stat=_openmp_parallelism_enabled()
102          )
103      )
104  
105      # show threadpoolctl results
106      threadpool_results = threadpool_info()
107      if threadpool_results:
108          print()
109          print("threadpoolctl info:")
110  
111          for i, result in enumerate(threadpool_results):
112              for key, val in result.items():
113                  print(f"{key:>15}: {val}")
114              if i != len(threadpool_results) - 1:
115                  print()
