10   class NearestNeighbors(KNeighborsMixin, RadiusNeighborsMixin, NeighborsBase):
11       """Unsupervised learner for implementing neighbor searches. 
12    
13       Read more in the :ref:`User Guide <unsupervised_neighbors>`. 
14    
15       .. versionadded:: 0.9 
16    
17       Parameters 
18       ---------- 
19       n_neighbors : int, default=5 
20           Number of neighbors to use by default for :meth:`kneighbors` queries. 
21    
22       radius : float, default=1.0 
23           Range of parameter space to use by default for :meth:`radius_neighbors` 
24           queries. 
25    
26       algorithm : {'auto', 'ball_tree', 'kd_tree', 'brute'}, default='auto' 
27           Algorithm used to compute the nearest neighbors: 
28    
29           - 'ball_tree' will use :class:`BallTree` 
30           - 'kd_tree' will use :class:`KDTree` 
31           - 'brute' will use a brute-force search. 
32           - 'auto' will attempt to decide the most appropriate algorithm 
33             based on the values passed to :meth:`fit` method. 
34    
35           Note: fitting on sparse input will override the setting of 
36           this parameter, using brute force. 
37    
38       leaf_size : int, default=30 
39           Leaf size passed to BallTree or KDTree.  This can affect the 
40           speed of the construction and query, as well as the memory 
41           required to store the tree.  The optimal value depends on the 
42           nature of the problem. 
43    
44       metric : str or callable, default='minkowski' 
45           Metric to use for distance computation. Default is "minkowski", which 
46           results in the standard Euclidean distance when p = 2. See the 
47           documentation of `scipy.spatial.distance 
48           <https://docs.scipy.org/doc/scipy/reference/spatial.distance.html>`_ and 
49           the metrics listed in 
50           :class:`~sklearn.metrics.pairwise.distance_metrics` for valid metric 
51           values. 
52    
53           If metric is "precomputed", X is assumed to be a distance matrix and 
54           must be square during fit. X may be a :term:`sparse graph`, in which 
55           case only "nonzero" elements may be considered neighbors. 
56    
57           If metric is a callable function, it takes two arrays representing 1D 
58           vectors as inputs and must return one value indicating the distance 
59           between those vectors. This works for Scipy's metrics, but is less 
60           efficient than passing the metric name as a string. 
61    
62       p : float (positive), default=2 
63           Parameter for the Minkowski metric from 
64           sklearn.metrics.pairwise.pairwise_distances. When p = 1, this is 
65           equivalent to using manhattan_distance (l1), and euclidean_distance 
66           (l2) for p = 2. For arbitrary p, minkowski_distance (l_p) is used. 
67    
68       metric_params : dict, default=None 
69           Additional keyword arguments for the metric function. 
70    
71       n_jobs : int, default=None 
72           The number of parallel jobs to run for neighbors search. 
73           ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
74           ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
75           for more details. 
76    
77       Attributes 
78       ---------- 
79       effective_metric_ : str 
80           Metric used to compute distances to neighbors. 
81    
82       effective_metric_params_ : dict 
83           Parameters for the metric used to compute distances to neighbors. 
84    
85       n_features_in_ : int 
86           Number of features seen during :term:`fit`. 
87    
88           .. versionadded:: 0.24 
89    
90       feature_names_in_ : ndarray of shape (`n_features_in_`,) 
91           Names of features seen during :term:`fit`. Defined only when `X` 
92           has feature names that are all strings. 
93    
94           .. versionadded:: 1.0 
95    
96       n_samples_fit_ : int 
97           Number of samples in the fitted data. 
98    
99       See Also 
100      -------- 
101      KNeighborsClassifier : Classifier implementing the k-nearest neighbors 
102          vote. 
103      RadiusNeighborsClassifier : Classifier implementing a vote among neighbors 
104          within a given radius. 
105      KNeighborsRegressor : Regression based on k-nearest neighbors. 
106      RadiusNeighborsRegressor : Regression based on neighbors within a fixed 
107          radius. 
108      BallTree : Space partitioning data structure for organizing points in a 
109          multi-dimensional space, used for nearest neighbor search. 
110   
111      Notes 
112      ----- 
113      See :ref:`Nearest Neighbors <neighbors>` in the online documentation 
114      for a discussion of the choice of ``algorithm`` and ``leaf_size``. 
115   
116      https://en.wikipedia.org/wiki/K-nearest_neighbors_algorithm 
117   
118      Examples 
119      -------- 
120      >>> import numpy as np 
121      >>> from sklearn.neighbors import NearestNeighbors 
122      >>> samples = [[0, 0, 2], [1, 0, 0], [0, 0, 1]] 
123      >>> neigh = NearestNeighbors(n_neighbors=2, radius=0.4) 
124      >>> neigh.fit(samples) 
125      NearestNeighbors(...) 
126      >>> neigh.kneighbors([[0, 0, 1.3]], 2, return_distance=False) 
127      array([[2, 0]]...) 
128      >>> nbrs = neigh.radius_neighbors( 
129      ...    [[0, 0, 1.3]], 0.4, return_distance=False 
130      ... ) 
131      >>> np.asarray(nbrs[0][0]) 
132      array(2) 
133      """
134  
135      def __init__(
136          self,
137          *,
138          n_neighbors=5,
139          radius=1.0,
140          algorithm="auto",
141          leaf_size=30,
142          metric="minkowski",
143          p=2,
144          metric_params=None,
145          n_jobs=None,
146      ):
147          super().__init__(
148              n_neighbors=n_neighbors,
149              radius=radius,
150              algorithm=algorithm,
151              leaf_size=leaf_size,
152              metric=metric,
153              p=p,
154              metric_params=metric_params,
155              n_jobs=n_jobs,
156          )

158      @_fit_context(
159          # NearestNeighbors.metric is not validated yet
160          prefer_skip_nested_validation=False
161      )
162      def fit(self, X, y=None):
163          """Fit the nearest neighbors estimator from the training dataset. 
164   
165          Parameters 
166          ---------- 
167          X : {array-like, sparse matrix} of shape (n_samples, n_features) or \ 
168                  (n_samples, n_samples) if metric='precomputed' 
169              Training data. 
170   
171          y : Ignored 
172              Not used, present for API consistency by convention. 
173   
174          Returns 
175          ------- 
176          self : NearestNeighbors 
177              The fitted nearest neighbors estimator. 
178          """
179          return self._fit(X)
