30   class TestTypeGeneration(BaseSelection):
31   
32       """ 
33           Internal feature: Determine output types from dataset dtype and fields. 
34       """
35   
36       def test_simple(self):
37           """ Non-compound types are handled appropriately """
38           dt = np.dtype('i')
39           out, format = sel2.read_dtypes(dt, ())
40           self.assertEqual(out, format)
41           self.assertEqual(out, np.dtype('i'))
42   
43       def test_simple_fieldexc(self):
44           """ Field names for non-field types raises ValueError """
45           dt = np.dtype('i')
46           with self.assertRaises(ValueError):
47               out, format = sel2.read_dtypes(dt, ('a',))
48   
49       def test_compound_simple(self):
50           """ Compound types with elemental subtypes """
51           dt = np.dtype( [('a','i'), ('b','f'), ('c','|S10')] )
52   
53           # Implicit selection of all fields -> all fields
54           out, format = sel2.read_dtypes(dt, ())
55           self.assertEqual(out, format)
56           self.assertEqual(out, dt)
57   
58           # Explicit selection of fields -> requested fields
59           out, format = sel2.read_dtypes(dt, ('a','b'))
60           self.assertEqual(out, format)
61           self.assertEqual(out, np.dtype( [('a','i'), ('b','f')] ))
62   
63           # Explicit selection of exactly one field -> no fields
64           out, format = sel2.read_dtypes(dt, ('a',))
65           self.assertEqual(out, np.dtype('i'))
66           self.assertEqual(format, np.dtype( [('a','i')] ))
67   
68           # Field does not appear in named typed
69           with self.assertRaises(ValueError):
70               out, format = sel2.read_dtypes(dt, ('j', 'k'))

95   class TestSelection(BaseSelection):
96   
97       """ High-level routes to generate a selection 
98       """
99   
100      def test_selection(self):
101          dset = self.f.create_dataset('dset', (100,100))
102          regref = dset.regionref[0:100, 0:100]
103  
104          # args is list, return a FancySelection
105          st = sel.select((10,), list([1,2,3]), dset)
106          self.assertIsInstance(st, sel.FancySelection)
107  
108          # args[0] is tuple, return a FancySelection
109          st = sel.select((10,), ((1, 2, 3),), dset)
110          self.assertIsInstance(st, sel.FancySelection)
111  
112          # args is a Boolean mask, return a PointSelection
113          st1 = sel.select((5,), np.array([True,False,False,False,True]), dset)
114          self.assertIsInstance(st1, sel.PointSelection)
115  
116          # args is int, return a SimpleSelection
117          st2 = sel.select((10,), 1, dset)
118          self.assertIsInstance(st2, sel.SimpleSelection)
119  
120          # args is str, should be rejected
121          with self.assertRaises(TypeError):
122              sel.select((100,), "foo", dset)
123  
124          # args is RegionReference, return a Selection instance
125          st3 = sel.select((100,100), regref, dset)
126          self.assertIsInstance(st3, sel.Selection)
127  
128          # args is RegionReference, but dataset is None
129          with self.assertRaises(TypeError):
130              sel.select((100,), regref, None)
131  
132          # args is RegionReference, but its shape doesn't match dataset shape
133          with self.assertRaises(TypeError):
134              sel.select((100,), regref, dset)
135  
136          # args is a single Selection instance, return the arg
137          st4 = sel.select((100,100), st3, dset)
138          self.assertEqual(st4,st3)
139  
140          # args is a single Selection instance, but args shape doesn't match Shape
141          with self.assertRaises(TypeError):
142              sel.select((100,), st3, dset)
