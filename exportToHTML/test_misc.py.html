22       def check_dump_bytecode(self, code, expected, lineno=None):
23           with contextlib.redirect_stdout(io.StringIO()) as stderr:
24               if lineno is not None:
25                   bytecode.dump_bytecode(code, lineno=True)
26               else:
27                   bytecode.dump_bytecode(code)
28               output = stderr.getvalue()
29   
30           self.assertEqual(output, expected)
