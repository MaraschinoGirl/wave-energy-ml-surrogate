28   def scatterd(x,
29                y,
30                z=None,
31                s=150,
32                c=[0, 0.1, 0.4],
33                labels=None,
34                marker='o',
35                alpha=0.8,
36                edgecolor='#000000',
37                gradient='opaque',
38                opaque_type='per_class',
39                density=True,
40                density_on_top=False,
41                norm=False,
42                cmap='tab20c',
43                figsize=(25, 15),
44                dpi=100,
45                legend=None,
46                jitter=None,
47                xlabel='x-axis', ylabel='y-axis', title='', fontsize=24, fontcolor=None, grid=True, fontweight='normal',
48                args_density = {'cmap': 'Reds', 'fill': True, 'thresh': 0.05, 'bw_adjust': .6, 'alpha': 0.66, 'legend': False, 'cbar': False},
49                visible=True,
50                fig=None,
51                ax=None,
52                verbose='info'):
53       """Make scaterplot. 
54    
55       Parameters 
56       ---------- 
57       x : numpy array 
58           1D Coordinates x-axis. 
59       y : numpy array 
60           1D Coordinates y-axis. 
61       z : numpy array 
62           1D Coordinates z-axis. 
63       s: Int or list/array of sizes with same size as X 
64           Size of the samples. 
65       c: list/array of RGB colors with same size as X 
66           Color of samples in RGB colors. 
67       labels: list of labels with same size as X 
68           labels of the samples. 
69       marker: list/array of strings (default: 'o'). 
70           Marker for the samples. 
71               * 'x' : All data points get this marker 
72               * ('.', 'o', 'v', '^', '<', '>', '8', 's', 'p', '*', 'h', 'H', 'D', 'd', 'P', 'X') : Specify per sample the marker type. 
73               * [0,3,0,1,2,1,..] : It is also possible to provide a list of labels. The markers will be assigned accordingly. 
74       alpha : float, default: 0.8 
75           The alpha blending value, between 0 (transparent) and 1 (opaque). 
76       edgecolors : (default: 'face') 
77           The edge color of the marker. Possible values: 
78               * 'face': The edge color will always be the same as the face color. 
79               * 'none': No patch boundary will be drawn. 
80               * '#FFFFFF' : A color or sequence of colors. 
81       gradient : String, (default: 'opaque') 
82           Hex color to make a lineair gradient using the density. 
83               * None: Do not use gradient. 
84               * opaque: Towards the edges the points become more opaque and thus not visible. 
85               * '#FFFFFF': Towards the edges it smooths into this color 
86       opaque_type : String, optional 
87               * 'per_class': Transprancy is determined on the density within the class label (y) 
88               * 'all': Transprancy is determined on all available data points 
89               * 'lineair': Transprancy is lineair set within the class label (y) 
90       density : Bool (default: False) 
91           Include the kernel density in the scatter plot. 
92       density_on_top : bool, (default: False) 
93           True : The density is the highest layer. 
94           False : The density is the lowest layer. 
95       xlabel : String, optional 
96           Xlabel. The default is None. 
97       ylabel : String, optional 
98           Ylabel. The default is None. 
99       title : String, optional 
100          Title of the figure. The default is None. 
101      fontsize : String, optional 
102          The fontsize of the y labels that are plotted in the graph. The default is 16. 
103      fontcolor: list/array of RGB colors with same size as X (default : None) 
104          None : Use same colorscheme as for c 
105          '#000000' : If the input is a single color, all fonts will get this color. 
106      grid: str or bool (default: False) 
107          False or None : Do not plot grid 
108          True : Set axis color to: '#dddddd' 
109          '#dddddd' : Specify color with hex 
110      norm : Bool, optional 
111          Normalize datapoints. The default is False. 
112      legend : int, default: 0 
113          None: Set automatically based number of labels. 
114          False : Disable. 
115          True : Best position. 
116          1 : 'upper right' 
117          2 : 'upper left' 
118          3 : 'lower left' 
119          4 : 'lower right' 
120      jitter : float, default: None 
121          Add jitter to data points as random normal data. Values of 0.01 is usually good for one-hot data seperation. 
122              * None or False: Do not add jitter 
123              * True : adds 0.01 
124              * 0.05 : Specify the amount jitter to add. 
125      cmap : String, optional 
126          'Set1'       (default) 
127          'Set2' 
128          'rainbow' 
129          'bwr'        Blue-white-red 
130          'binary' or 'binary_r' 
131          'seismic'    Blue-white-red 
132          'Blues'      white-to-blue 
133          'Reds'       white-to-red 
134          'Pastel1'    Discrete colors 
135          'Paired'     Discrete colors 
136          'Set1'       Discrete colors 
137      figsize : tuple, optional 
138          Figure size. The default is (15,10). 
139      visible : Bool, default: True 
140          Visible status of the Figure. When False, figure is created on the background. 
141      args_density : dict() 
142          Dictionary containing arguments for kernel density plotting. 
143      verbose : int, (default: 'info') 
144          Print progress to screen. The default is 'info' 
145          60: None, 40: error, 30: warning, 20: info, 10: debug 
146   
147      Returns 
148      ------- 
149      Fig, ax 
150   
151      Examples 
152      -------- 
153      >>> # Import library 
154      >>> from scatterd import scatterd, import_example 
155      >>> 
156      >>> # Import example 
157      >>> df = import_example() 
158      >>> 
159      >>> # Simple scatter 
160      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], edgecolor='#FFFFFF') 
161      >>> 
162      >>> # Scatter with labels 
163      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], labels=df['labx']) 
164      >>> 
165      >>> # Scatter with labels 
166      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], labels=df['labx']) 
167      >>> 
168      >>> # Scatter with gradient 
169      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], z=df['tsneY'], labels=df['labx'], gradient='#FFFFFF') 
170      >>> 
171      >>> # Change cmap 
172      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], labels=df['labx'], gradient='#FFFFFF', cmap='Set2') 
173      >>> 
174      >>> # Scatter with density 
175      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], labels=df['labx'], density=True) 
176      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], labels=df['labx'], density=False, gradient='#FFFFFF', edgecolor='#FFFFFF') 
177      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], labels=df['labx'], density=True, gradient='#FFFFFF', edgecolor='#FFFFFF') 
178      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], labels=df['labx'], density=True, gradient='#FFFFFF', c=None) 
179      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], labels=df['labx'], density=True, density_on_top=True, args_density={'alpha': 0.3}, gradient='#FFFFFF', edgecolor='#FFFFFF', grid=True, fontweight='normal', fontsize=26, legend=2) 
180      >>> 
181      >>> # Scatter with density and gradient 
182      >>> fig, ax = scatterd(df['tsneX'], df['tsneY'], labels=df['labx'], density=True, gradient='#FFFFFF') 
183      >>> 
184   
185      References 
186      ------- 
187      * github: https://github.com/erdogant/scatterd 
188      * Documentation: https://erdogant.github.io/scatterd/ 
189      * Colormap: https://matplotlib.org/examples/color/colormaps_reference.html 
190   
191      """
192      # Set the logger
193      set_logger(verbose=verbose)
194  
195      if len(x)!=len(y): raise Exception('[scatterd] >Error: input parameter x should be the same size of y.')
196      if isinstance(c, str): raise Exception('[scatterd] >Error: input parameter c(olors) should be RGB of type tuple [0,0,0] .')
197      if not isinstance(s, int) and len(s)!=len(x): raise Exception('[scatterd] >Error: input parameter s(ize) should be of same size of X.')
198      if (z is not None) and len(x)!=len(z): raise Exception('[scatterd] >Error: input parameter z should be the same size of x and y.')
199      if s is None: s=0
200      if c is None: s, c = 0, [0, 0, 0]
201      if isinstance(s, (int, float)) and s==0: fontsize=0
202      zorder = None if density_on_top else 10
203  
204      # Defaults
205      defaults_density = {'cmap': 'Reds', 'thresh': 0.05, 'bw_adjust': .6, 'alpha': 0.66, 'legend': False, 'cbar': False, 'fill': True}
206      args_density = {**defaults_density, **args_density}
207  
208      # Preprocessing
209      X, labels = _preprocessing(x, y, z, labels, jitter, norm)
210      # Set color
211      c_rgb, opaque = set_colors(X, labels, c, cmap, gradient=gradient, opaque_type=opaque_type)
212      # Set fontcolor
213      fontcolor = set_fontcolor(fontcolor, labels, X, cmap)
214      # Set size
215      s = set_size(X, s)
216      # Set size
217      alpha = set_alpha(X, alpha, gradient, opaque)
218      # Set marker
219      marker = set_marker(X, marker)
220      # Bootup figure
221      fig, ax = init_figure(ax, z, dpi, figsize, visible, fig, fontsize)
222      # Set figure properties
223      ax = _set_figure_properties(X, labels, fontcolor, fontsize, xlabel, ylabel, title, grid, fontweight, zorder, ax)
224  
225      # Add density as bottom layer to the scatterplot
226      if density:
227          logger.info('Add density layer')
228          ax = sns.kdeplot(x=X[:, 0], y=X[:, 1], ax=ax, **args_density)
229  
230      # Scatter all at once
231      if (labels is None) and isinstance(marker, str):
232          logger.info('Create scatterplot (fast)')
233          # Do not try to plot legend.
234          legend = -1
235          if z is None:
236              ax.scatter(X[:, 0], X[:, 1], c=c_rgb, s=s, edgecolor=edgecolor, marker=marker, alpha=alpha, zorder=zorder)
237          else:
238              ax.scatter(X[:, 0], X[:, 1], X[:, 2], s=s, c=c_rgb, edgecolor=edgecolor, marker=marker, alpha=alpha, zorder=zorder)
239      else:
240          logger.info('Create scatterplot')
241          uilabels = np.unique(labels)
242          for label in uilabels:
243              Iloc1 = label==labels
244              # Extract the unique markers
245              for m in np.unique(marker[Iloc1]):
246                  Iloc2 = np.logical_and(Iloc1, m==marker)
247                  if z is None:
248                      ax.scatter(X[Iloc2, 0], X[Iloc2, 1], c=c_rgb[Iloc2], s=s[Iloc2], edgecolor=edgecolor, marker=m, label=label, alpha=alpha[Iloc2], zorder=zorder)
249                  else:
250                      # Note: alpha[Iloc2] throws an error
251                      ax.scatter(X[Iloc2, 0], X[Iloc2, 1], X[Iloc2, 2], s=s[Iloc2], c=c_rgb[Iloc2], edgecolor=edgecolor, marker=m, label=label, alpha=0.8, zorder=zorder)
252  
253      # Show legend (only if labels are present)
254      if isinstance(legend, bool): legend = 0 if legend else -1
255      if legend is None: legend = -1 if len(np.unique(labels))>20 else 0
256      if (legend>=0) and (labels is not None): ax.legend(loc=legend, fontsize=14)
257  
258      # Return
259      return fig, ax

263  def _set_figure_properties(X, labels, fontcolor, fontsize, xlabel, ylabel, title, grid, fontweight, zorder, ax):
264      # Set axis fontsizes
265      if grid is None: grid=False
266      if grid is True: grid='#dddddd'
267      None if zorder is None else zorder + 1
268      font = {'family': 'DejaVu Sans', 'weight': fontweight, 'size': np.maximum(fontsize, 1)}
269      matplotlib.rc('font', **font)
270      for item in ([ax.title, ax.xaxis.label, ax.yaxis.label] + ax.get_xticklabels() + ax.get_yticklabels()):
271          item.set_fontsize(20)
272  
273      # Plot labels
274      if (labels is not None) and (fontcolor is not None):
275          for uilabel in fontcolor.keys():
276              # Compute median for better center compared to mean
277              XYmean = np.mean(X[labels==uilabel, :], axis=0)
278              if X.shape[1]==2:
279                  ax.text(XYmean[0], XYmean[1], str(uilabel), color=fontcolor.get(uilabel), fontdict={'weight': fontweight, 'size': fontsize}, zorder=zorder)
280              else:
281                  ax.text(XYmean[0], XYmean[1], XYmean[2], str(uilabel), color=fontcolor.get(uilabel), fontdict={'weight': fontweight, 'size': fontsize})
282  
283      # Labels on axis
284      if (xlabel is not None) and (xlabel!=''): ax.set_xlabel(xlabel)
285      if (ylabel is not None) and (ylabel!=''): ax.set_ylabel(ylabel)
286      if (title is not None) and (title!=''): ax.set_title(title)
287  
288      # set background to none
289      ax.set_facecolor('none')
290  
291      # Set grid and axis to grey
292      if isinstance(grid, str):
293          ax.grid(True, color=grid)
294          # Color all lines in the plot the same grid color
295          # for child in ax.get_children():
296              # if isinstance(child, matplotlib.spines.Spine):
297                  # child.set_color(grid)
298  
299      return ax

303  def set_colors(X, labels, c, cmap='tab20c', gradient=None, opaque_type='per_class', verbose=None):
304      """Set colors."""
305      # Set the logger
306      if verbose is not None: set_logger(verbose=verbose)
307      if c is None: return None
308  
309      # The default is all dots to black
310      c_rgb = np.repeat([0, 0, 0], X.shape[0], axis=0).reshape(-1, 3)
311      # Create opaque levels
312      opaque = np.array([1.0] * X.shape[0])
313  
314      # Change on input c
315      if len(c)==1 and isinstance(c, list): c = c[0]
316      if len(c)==3 and isinstance(c[0], (int, float)):
317          logger.debug('Colors are all set to %s.' %(c))
318          c_rgb = np.repeat([c], X.shape[0], axis=0).reshape(-1, 3)
319  
320      if X.shape[0]==len(c):
321          logger.debug('Colors are set to input colors defined in [c].')
322          c_rgb=c
323  
324      if labels is not None:
325          # Create unqiue colors for labels if there are multiple classes or in case cmap and gradient is used.
326          logger.debug('Colors are based on the input [labels] and on [cmap].')
327          if labels is None: labels = np.repeat(0, X.shape[0])
328          c_rgb = colourmap.fromlist(labels, X=X, cmap=cmap, scheme='rgb', method='matplotlib', gradient=gradient, opaque_type=opaque_type, verbose=verbose)[0]
329          # Add extra column with transparancy
330          if gradient=='opaque' and c_rgb.shape[1]==4:
331              # Set the minimum transparancy level at 0.1
332              opaque = np.maximum(c_rgb[:, -1], 0.1)
333              c_rgb = c_rgb[:, :3]
334  
335      # Return
336      return c_rgb, opaque

339  # %% Jitter
340  def jitter_func(x, jitter=0.01):
341      """Add jitter to data. 
342   
343      Noise is generated from random normal distribution and added to the data. 
344   
345      Parameters 
346      ---------- 
347      x : numpy array 
348          input data. 
349      jitter : float, optional 
350          Strength of generated noise. The default is 0.01. 
351   
352      Returns 
353      ------- 
354      x : array-like 
355          Data including noise. 
356   
357      """
358      if jitter is None or jitter is False: jitter=0
359      if jitter is True: jitter=0.01
360      if jitter>0 and x is not None:
361          x = x + np.random.normal(0, jitter, size=len(x))
362      return x

366  def _preprocessing(x, y, z, labels, jitter, norm=False):
367      # Add jitter
368      x = jitter_func(x, jitter=jitter)
369      y = jitter_func(y, jitter=jitter)
370      z = jitter_func(z, jitter=jitter)
371  
372      # Combine into array
373      if z is not None:
374          X = np.c_[x, y, z]
375      else:
376          X = np.c_[x, y]
377      # Normalize data
378      if norm:
379          X = normalize_between_0_and_1(X)
380      # Labels
381      # if labels is None:
382      #     labels=np.zeros_like(y).astype(int)
383      # Return
384      return X, labels

387  # %% Normalize between [0-1]
388  def normalize_between_0_and_1(X):
389      x_min, x_max = np.min(X, 0), np.max(X, 0)
390      out = (X - x_min) / (x_max - x_min)
391      out[np.isnan(out)]=1
392      return out

396  def set_fontcolor(fontcolor, label, X, cmap, verbose=None):
397      # Set the logger
398      if verbose is not None: set_logger(verbose=verbose)
399      if (fontcolor is not None) and colourmap.is_hex_color(fontcolor, verbose=0):
400          fontcolor = colourmap.hex2rgb(fontcolor)
401  
402      # Set font colors
403      if (fontcolor is None) and (label is None):
404          # fontcolor = {np.unique(label)[0]: [0, 0, 0]}
405          pass
406      elif (fontcolor is None) and (len(np.unique(label))==1):
407          pass
408      elif (fontcolor is not None) and (len(fontcolor)==3 or len(fontcolor)==1):
409          fontcolor_dict = {}
410          for lab in np.unique(label):
411              fontcolor_dict[lab]=fontcolor
412          fontcolor=fontcolor_dict
413      elif (fontcolor is not None) and (label is None):
414          logger.warning('Without label, there is no font(color) to print.')
415      elif (fontcolor is None) and (label is not None):
416          _, fontcolor = colourmap.fromlist(label, cmap=cmap, method='matplotlib', verbose=get_logger())
417      elif (fontcolor is not None) and (label is not None) and (len(fontcolor)==X.shape[0]):
418          _, fontcolor = colourmap.fromlist(fontcolor, cmap=cmap, method='matplotlib', verbose=get_logger())
419      elif (fontcolor is not None) and (label is not None) and ((isinstance(fontcolor[0], int)) or (isinstance(fontcolor[0], float))):
420          _, tmpcolors = colourmap.fromlist(label, cmap=cmap, method='matplotlib', verbose=get_logger())
421          # list(map(lambda x: tmpcolors.update({x: fontcolor}), [*tmpcolors.keys()]))
422          fontcolor = tmpcolors
423      else:
424          raise Exception('[scatterd] >ERROR : fontcolor input is not correct.')
425  
426      return fontcolor

430      """Set size."""

433      s = np.maximum(s, 0)

437  def set_alpha(X, alpha, gradient, opaque, verbose=None):
438      """Set alpha."""
439      # Set the logger
440      if verbose is not None: set_logger(verbose=verbose)
441      if alpha is None: alpha=0.8
442      if isinstance(alpha, (int, float)):
443          alpha = np.repeat(alpha, X.shape[0])
444      if gradient=='opaque':
445          logger.info('Set alpha based on density because of the parameter: [%s]' %(gradient))
446          alpha = opaque
447      # Minimum size should be 0 (dots will not be showed)
448      alpha = np.maximum(alpha, 0)
449      return alpha

452  def set_marker(X, marker):
453      """Set markers."""
454      markers = np.array(['o', 'v', 's', 'p', '*', 'h', 'H', 'D', 'd', 'P', 'X', '.', '^', '>', '<', '8'])
455      # markers_list = dict(zip(markers, np.arange(0,len(markers))))
456      # In case single str
457      if isinstance(marker, str) and np.isin(marker, markers):
458          return np.repeat(marker, X.shape[0])
459      # In case array of markers.
460      if np.all(np.isin(marker, markers)) and len(marker)==X.shape[0]:
461          return marker
462      # In case array of labels.
463      if len(marker)==X.shape[0]:
464          d = dict(zip(set(marker), range(len(set(marker)))))
465          marker_num = [d[i] for i in marker]
466          return markers[np.mod(marker_num, len(markers))]
467      # Return
468      return np.repeat('o', X.shape[0])

471  def init_figure(ax, z, dpi, figsize, visible, fig, fontsize):
472      """Initialize figure."""
473      if ax is None:
474          fig = plt.figure(figsize=figsize, dpi=dpi)
475          # Adjust global font settings
476          plt.rcParams['font.size'] = np.maximum(fontsize, 1)  # Set an appropriate font size
477          if z is None:
478              ax = fig.add_subplot()
479          else:
480              ax = fig.add_subplot(projection='3d')
481              # Rotate
482              ax.view_init(-140, 60)
483  
484      if fig is not None:
485          fig.set_visible(visible)
486  
487      return fig, ax

491  def convert_verbose_to_new(verbose):
492      """Convert old verbosity to the new."""
493      # In case the new verbosity is used, convert to the old one.
494      if verbose is None: verbose=0
495      if not isinstance(verbose, str) and verbose<10:
496          status_map = {
497              'None': 'silent',
498              0: 'silent',
499              6: 'silent',
500              1: 'critical',
501              2: 'warning',
502              3: 'info',
503              4: 'debug',
504              5: 'debug'}
505          if verbose>=2: print('[scatterd]> WARNING use the standardized verbose status. The status [1-6] will be deprecated in future versions.')
506          return status_map.get(verbose, 0)
507      else:
508          return verbose

516  # %%
517  def set_logger(verbose: [str, int] = 'info'):
518      """Set the logger for verbosity messages. 
519   
520      Parameters 
521      ---------- 
522      verbose : [str, int], default is 'info' or 20 
523          Set the verbose messages using string or integer values. 
524          * [0, 60, None, 'silent', 'off', 'no']: No message. 
525          * [10, 'debug']: Messages from debug level and higher. 
526          * [20, 'info']: Messages from info level and higher. 
527          * [30, 'warning']: Messages from warning level and higher. 
528          * [50, 'critical']: Messages from critical level and higher. 
529   
530      Returns 
531      ------- 
532      None. 
533   
534      > # Set the logger to warning 
535      > set_logger(verbose='warning') 
536      > # Test with different messages 
537      > logger.debug("Hello debug") 
538      > logger.info("Hello info") 
539      > logger.warning("Hello warning") 
540      > logger.critical("Hello critical") 
541   
542      """
543      # Set 0 and None as no messages.
544      if (verbose==0) or (verbose is None):
545          verbose=60
546      # Convert to verbose
547      verbose = convert_verbose_to_new(verbose)
548      # Convert str to levels
549      if isinstance(verbose, str):
550          levels = {'silent': 60,
551                    'off': 60,
552                    'no': 60,
553                    'debug': 10,
554                    'info': 20,
555                    'warning': 30,
556                    'error': 50,
557                    'critical': 50}
558          verbose = levels[verbose]
559  
560      # Show examples
561      logger.setLevel(verbose)
562      logger.debug('Set verbose to %s' %(verbose))
