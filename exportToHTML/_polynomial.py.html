42   def _create_expansion(X, interaction_only, deg, n_features, cumulative_size=0):
43       """Helper function for creating and appending sparse expansion matrices"""
44   
45       total_nnz = _calc_total_nnz(X.indptr, interaction_only, deg)
46       expanded_col = _calc_expanded_nnz(n_features, interaction_only, deg)
47   
48       if expanded_col == 0:
49           return None
50       # This only checks whether each block needs 64bit integers upon
51       # expansion. We prefer to keep int32 indexing where we can,
52       # since currently SciPy's CSR construction downcasts when possible,
53       # so we prefer to avoid an unnecessary cast. The dtype may still
54       # change in the concatenation process if needed.
55       # See: https://github.com/scipy/scipy/issues/16569
56       max_indices = expanded_col - 1
57       max_indptr = total_nnz
58       max_int32 = np.iinfo(np.int32).max
59       needs_int64 = max(max_indices, max_indptr) > max_int32
60       index_dtype = np.int64 if needs_int64 else np.int32
61   
62       # This is a pretty specific bug that is hard to work around by a user,
63       # hence we do not detail the entire bug and all possible avoidance
64       # mechnasisms. Instead we recommend upgrading scipy or shrinking their data.
65       cumulative_size += expanded_col
66       if (
67           sp_version < parse_version("1.8.0")
68           and cumulative_size - 1 > max_int32
69           and not needs_int64
70       ):
71           raise ValueError(
72               "In scipy versions `<1.8.0`, the function `scipy.sparse.hstack`"
73               " sometimes produces negative columns when the output shape contains"
74               " `n_cols` too large to be represented by a 32bit signed"
75               " integer. To avoid this error, either use a version"
76               " of scipy `>=1.8.0` or alter the `PolynomialFeatures`"
77               " transformer to produce fewer than 2^31 output features."
78           )
79   
80       # Result of the expansion, modified in place by the
81       # `_csr_polynomial_expansion` routine.
82       expanded_data = np.empty(shape=total_nnz, dtype=X.data.dtype)
83       expanded_indices = np.empty(shape=total_nnz, dtype=index_dtype)
84       expanded_indptr = np.empty(shape=X.indptr.shape[0], dtype=index_dtype)
85       _csr_polynomial_expansion(
86           X.data,
87           X.indices,
88           X.indptr,
89           X.shape[1],
90           expanded_data,
91           expanded_indices,
92           expanded_indptr,
93           interaction_only,
94           deg,
95       )
96       return sparse.csr_matrix(
97           (expanded_data, expanded_indices, expanded_indptr),
98           shape=(X.indptr.shape[0] - 1, expanded_col),
99           dtype=X.dtype,
100      )

103  class PolynomialFeatures(TransformerMixin, BaseEstimator):
104      """Generate polynomial and interaction features. 
105   
106      Generate a new feature matrix consisting of all polynomial combinations 
107      of the features with degree less than or equal to the specified degree. 
108      For example, if an input sample is two dimensional and of the form 
109      [a, b], the degree-2 polynomial features are [1, a, b, a^2, ab, b^2]. 
110   
111      Read more in the :ref:`User Guide <polynomial_features>`. 
112   
113      Parameters 
114      ---------- 
115      degree : int or tuple (min_degree, max_degree), default=2 
116          If a single int is given, it specifies the maximal degree of the 
117          polynomial features. If a tuple `(min_degree, max_degree)` is passed, 
118          then `min_degree` is the minimum and `max_degree` is the maximum 
119          polynomial degree of the generated features. Note that `min_degree=0` 
120          and `min_degree=1` are equivalent as outputting the degree zero term is 
121          determined by `include_bias`. 
122   
123      interaction_only : bool, default=False 
124          If `True`, only interaction features are produced: features that are 
125          products of at most `degree` *distinct* input features, i.e. terms with 
126          power of 2 or higher of the same input feature are excluded: 
127   
128          - included: `x[0]`, `x[1]`, `x[0] * x[1]`, etc. 
129          - excluded: `x[0] ** 2`, `x[0] ** 2 * x[1]`, etc. 
130   
131      include_bias : bool, default=True 
132          If `True` (default), then include a bias column, the feature in which 
133          all polynomial powers are zero (i.e. a column of ones - acts as an 
134          intercept term in a linear model). 
135   
136      order : {'C', 'F'}, default='C' 
137          Order of output array in the dense case. `'F'` order is faster to 
138          compute, but may slow down subsequent estimators. 
139   
140          .. versionadded:: 0.21 
141   
142      Attributes 
143      ---------- 
144      powers_ : ndarray of shape (`n_output_features_`, `n_features_in_`) 
145          `powers_[i, j]` is the exponent of the jth input in the ith output. 
146   
147      n_features_in_ : int 
148          Number of features seen during :term:`fit`. 
149   
150          .. versionadded:: 0.24 
151   
152      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
153          Names of features seen during :term:`fit`. Defined only when `X` 
154          has feature names that are all strings. 
155   
156          .. versionadded:: 1.0 
157   
158      n_output_features_ : int 
159          The total number of polynomial output features. The number of output 
160          features is computed by iterating over all suitably sized combinations 
161          of input features. 
162   
163      See Also 
164      -------- 
165      SplineTransformer : Transformer that generates univariate B-spline bases 
166          for features. 
167   
168      Notes 
169      ----- 
170      Be aware that the number of features in the output array scales 
171      polynomially in the number of features of the input array, and 
172      exponentially in the degree. High degrees can cause overfitting. 
173   
174      See :ref:`examples/linear_model/plot_polynomial_interpolation.py 
175      <sphx_glr_auto_examples_linear_model_plot_polynomial_interpolation.py>` 
176   
177      Examples 
178      -------- 
179      >>> import numpy as np 
180      >>> from sklearn.preprocessing import PolynomialFeatures 
181      >>> X = np.arange(6).reshape(3, 2) 
182      >>> X 
183      array([[0, 1], 
184             [2, 3], 
185             [4, 5]]) 
186      >>> poly = PolynomialFeatures(2) 
187      >>> poly.fit_transform(X) 
188      array([[ 1.,  0.,  1.,  0.,  0.,  1.], 
189             [ 1.,  2.,  3.,  4.,  6.,  9.], 
190             [ 1.,  4.,  5., 16., 20., 25.]]) 
191      >>> poly = PolynomialFeatures(interaction_only=True) 
192      >>> poly.fit_transform(X) 
193      array([[ 1.,  0.,  1.,  0.], 
194             [ 1.,  2.,  3.,  6.], 
195             [ 1.,  4.,  5., 20.]]) 
196      """
197  
198      _parameter_constraints: dict = {
199          "degree": [Interval(Integral, 0, None, closed="left"), "array-like"],
200          "interaction_only": ["boolean"],
201          "include_bias": ["boolean"],
202          "order": [StrOptions({"C", "F"})],

205      def __init__(
206          self, degree=2, *, interaction_only=False, include_bias=True, order="C"
207      ):
208          self.degree = degree
209          self.interaction_only = interaction_only
210          self.include_bias = include_bias
211          self.order = order

213      @staticmethod
214      def _combinations(
215          n_features, min_degree, max_degree, interaction_only, include_bias
216      ):
217          comb = combinations if interaction_only else combinations_w_r
218          start = max(1, min_degree)
219          iter = chain.from_iterable(
220              comb(range(n_features), i) for i in range(start, max_degree + 1)
221          )
222          if include_bias:
223              iter = chain(comb(range(n_features), 0), iter)
224          return iter
225  
226      @staticmethod
227      def _num_combinations(
228          n_features, min_degree, max_degree, interaction_only, include_bias
229      ):
230          """Calculate number of terms in polynomial expansion 
231   
232          This should be equivalent to counting the number of terms returned by 
233          _combinations(...) but much faster. 
234          """
235  
236          if interaction_only:
237              combinations = sum(
238                  [
239                      comb(n_features, i, exact=True)
240                      for i in range(max(1, min_degree), min(max_degree, n_features) + 1)
241                  ]
242              )
243          else:
244              combinations = comb(n_features + max_degree, max_degree, exact=True) - 1
245              if min_degree > 0:
246                  d = min_degree - 1
247                  combinations -= comb(n_features + d, d, exact=True) - 1
248  
249          if include_bias:
250              combinations += 1
251  
252          return combinations

254      @property
255      def powers_(self):
256          """Exponent for each of the inputs in the output."""
257          check_is_fitted(self)
258  
259          combinations = self._combinations(
260              n_features=self.n_features_in_,
261              min_degree=self._min_degree,
262              max_degree=self._max_degree,
263              interaction_only=self.interaction_only,
264              include_bias=self.include_bias,
265          )
266          return np.vstack(
267              [np.bincount(c, minlength=self.n_features_in_) for c in combinations]
268          )

270      def get_feature_names_out(self, input_features=None):
271          """Get output feature names for transformation. 
272   
273          Parameters 
274          ---------- 
275          input_features : array-like of str or None, default=None 
276              Input features. 
277   
278              - If `input_features is None`, then `feature_names_in_` is 
279                used as feature names in. If `feature_names_in_` is not defined, 
280                then the following input feature names are generated: 
281                `["x0", "x1", ..., "x(n_features_in_ - 1)"]`. 
282              - If `input_features` is an array-like, then `input_features` must 
283                match `feature_names_in_` if `feature_names_in_` is defined. 
284   
285          Returns 
286          ------- 
287          feature_names_out : ndarray of str objects 
288              Transformed feature names. 
289          """
290          powers = self.powers_
291          input_features = _check_feature_names_in(self, input_features)
292          feature_names = []
293          for row in powers:
294              inds = np.where(row)[0]
295              if len(inds):
296                  name = " ".join(
297                      (
298                          "%s^%d" % (input_features[ind], exp)
299                          if exp != 1
300                          else input_features[ind]
301                      )
302                      for ind, exp in zip(inds, row[inds])
303                  )
304              else:
305                  name = "1"
306              feature_names.append(name)
307          return np.asarray(feature_names, dtype=object)

309      @_fit_context(prefer_skip_nested_validation=True)
310      def fit(self, X, y=None):
311          """ 
312          Compute number of output features. 
313   
314          Parameters 
315          ---------- 
316          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
317              The data. 
318   
319          y : Ignored 
320              Not used, present here for API consistency by convention. 
321   
322          Returns 
323          ------- 
324          self : object 
325              Fitted transformer. 
326          """
327          _, n_features = validate_data(self, X, accept_sparse=True).shape
328  
329          if isinstance(self.degree, Integral):
330              if self.degree == 0 and not self.include_bias:
331                  raise ValueError(
332                      "Setting degree to zero and include_bias to False would result in"
333                      " an empty output array."
334                  )
335  
336              self._min_degree = 0
337              self._max_degree = self.degree
338          elif (
339              isinstance(self.degree, collections.abc.Iterable) and len(self.degree) == 2
340          ):
341              self._min_degree, self._max_degree = self.degree
342              if not (
343                  isinstance(self._min_degree, Integral)
344                  and isinstance(self._max_degree, Integral)
345                  and self._min_degree >= 0
346                  and self._min_degree <= self._max_degree
347              ):
348                  raise ValueError(
349                      "degree=(min_degree, max_degree) must "
350                      "be non-negative integers that fulfil "
351                      "min_degree <= max_degree, got "
352                      f"{self.degree}."
353                  )
354              elif self._max_degree == 0 and not self.include_bias:
355                  raise ValueError(
356                      "Setting both min_degree and max_degree to zero and include_bias to"
357                      " False would result in an empty output array."
358                  )
359          else:
360              raise ValueError(
361                  "degree must be a non-negative int or tuple "
362                  "(min_degree, max_degree), got "
363                  f"{self.degree}."
364              )
365  
366          self.n_output_features_ = self._num_combinations(
367              n_features=n_features,
368              min_degree=self._min_degree,
369              max_degree=self._max_degree,
370              interaction_only=self.interaction_only,
371              include_bias=self.include_bias,
372          )
373          if self.n_output_features_ > np.iinfo(np.intp).max:
374              msg = (
375                  "The output that would result from the current configuration would"
376                  f" have {self.n_output_features_} features which is too large to be"
377                  f" indexed by {np.intp().dtype.name}. Please change some or all of the"
378                  " following:\n- The number of features in the input, currently"
379                  f" {n_features=}\n- The range of degrees to calculate, currently"
380                  f" [{self._min_degree}, {self._max_degree}]\n- Whether to include only"
381                  f" interaction terms, currently {self.interaction_only}\n- Whether to"
382                  f" include a bias term, currently {self.include_bias}."
383              )
384              if (
385                  np.intp == np.int32
386                  and self.n_output_features_ <= np.iinfo(np.int64).max
387              ):  # pragma: nocover
388                  msg += (
389                      "\nNote that the current Python runtime has a limited 32 bit "
390                      "address space and that this configuration would have been "
391                      "admissible if run on a 64 bit Python runtime."
392                  )
393              raise ValueError(msg)
394          # We also record the number of output features for
395          # _min_degree = 0
396          self._n_out_full = self._num_combinations(
397              n_features=n_features,
398              min_degree=0,
399              max_degree=self._max_degree,
400              interaction_only=self.interaction_only,
401              include_bias=self.include_bias,
402          )
403  
404          return self

406      def transform(self, X):
407          """Transform data to polynomial features. 
408   
409          Parameters 
410          ---------- 
411          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
412              The data to transform, row by row. 
413   
414              Prefer CSR over CSC for sparse input (for speed), but CSC is 
415              required if the degree is 4 or higher. If the degree is less than 
416              4 and the input format is CSC, it will be converted to CSR, have 
417              its polynomial features generated, then converted back to CSC. 
418   
419              If the degree is 2 or 3, the method described in "Leveraging 
420              Sparsity to Speed Up Polynomial Feature Expansions of CSR Matrices 
421              Using K-Simplex Numbers" by Andrew Nystrom and John Hughes is 
422              used, which is much faster than the method used on CSC input. For 
423              this reason, a CSC input will be converted to CSR, and the output 
424              will be converted back to CSC prior to being returned, hence the 
425              preference of CSR. 
426   
427          Returns 
428          ------- 
429          XP : {ndarray, sparse matrix} of shape (n_samples, NP) 
430              The matrix of features, where `NP` is the number of polynomial 
431              features generated from the combination of inputs. If a sparse 
432              matrix is provided, it will be converted into a sparse 
433              `csr_matrix`. 
434          """
435          check_is_fitted(self)
436  
437          X = validate_data(
438              self,
439              X,
440              order="F",
441              dtype=FLOAT_DTYPES,
442              reset=False,
443              accept_sparse=("csr", "csc"),
444          )
445  
446          n_samples, n_features = X.shape
447          max_int32 = np.iinfo(np.int32).max
448          if sparse.issparse(X) and X.format == "csr":
449              if self._max_degree > 3:
450                  return self.transform(X.tocsc()).tocsr()
451              to_stack = []
452              if self.include_bias:
453                  to_stack.append(
454                      sparse.csr_matrix(np.ones(shape=(n_samples, 1), dtype=X.dtype))
455                  )
456              if self._min_degree <= 1 and self._max_degree > 0:
457                  to_stack.append(X)
458  
459              cumulative_size = sum(mat.shape[1] for mat in to_stack)
460              for deg in range(max(2, self._min_degree), self._max_degree + 1):
461                  expanded = _create_expansion(
462                      X=X,
463                      interaction_only=self.interaction_only,
464                      deg=deg,
465                      n_features=n_features,
466                      cumulative_size=cumulative_size,
467                  )
468                  if expanded is not None:
469                      to_stack.append(expanded)
470                      cumulative_size += expanded.shape[1]
471              if len(to_stack) == 0:
472                  # edge case: deal with empty matrix
473                  XP = sparse.csr_matrix((n_samples, 0), dtype=X.dtype)
474              else:
475                  # `scipy.sparse.hstack` breaks in scipy<1.9.2
476                  # when `n_output_features_ > max_int32`
477                  all_int32 = all(mat.indices.dtype == np.int32 for mat in to_stack)
478                  if (
479                      sp_version < parse_version("1.9.2")
480                      and self.n_output_features_ > max_int32
481                      and all_int32
482                  ):
483                      raise ValueError(  # pragma: no cover
484                          "In scipy versions `<1.9.2`, the function `scipy.sparse.hstack`"
485                          " produces negative columns when:\n1. The output shape contains"
486                          " `n_cols` too large to be represented by a 32bit signed"
487                          " integer.\n2. All sub-matrices to be stacked have indices of"
488                          " dtype `np.int32`.\nTo avoid this error, either use a version"
489                          " of scipy `>=1.9.2` or alter the `PolynomialFeatures`"
490                          " transformer to produce fewer than 2^31 output features"
491                      )
492                  XP = sparse.hstack(to_stack, dtype=X.dtype, format="csr")
493          elif sparse.issparse(X) and X.format == "csc" and self._max_degree < 4:
494              return self.transform(X.tocsr()).tocsc()
495          elif sparse.issparse(X):
496              combinations = self._combinations(
497                  n_features=n_features,
498                  min_degree=self._min_degree,
499                  max_degree=self._max_degree,
500                  interaction_only=self.interaction_only,
501                  include_bias=self.include_bias,
502              )
503              columns = []
504              for combi in combinations:
505                  if combi:
506                      out_col = 1
507                      for col_idx in combi:
508                          out_col = X[:, [col_idx]].multiply(out_col)
509                      columns.append(out_col)
510                  else:
511                      bias = sparse.csc_matrix(np.ones((X.shape[0], 1)))
512                      columns.append(bias)
513              XP = sparse.hstack(columns, dtype=X.dtype).tocsc()
514          else:
515              # Do as if _min_degree = 0 and cut down array after the
516              # computation, i.e. use _n_out_full instead of n_output_features_.
517              XP = np.empty(
518                  shape=(n_samples, self._n_out_full), dtype=X.dtype, order=self.order
519              )
520  
521              # What follows is a faster implementation of:
522              # for i, comb in enumerate(combinations):
523              #     XP[:, i] = X[:, comb].prod(1)
524              # This implementation uses two optimisations.
525              # First one is broadcasting,
526              # multiply ([X1, ..., Xn], X1) -> [X1 X1, ..., Xn X1]
527              # multiply ([X2, ..., Xn], X2) -> [X2 X2, ..., Xn X2]
528              # ...
529              # multiply ([X[:, start:end], X[:, start]) -> ...
530              # Second optimisation happens for degrees >= 3.
531              # Xi^3 is computed reusing previous computation:
532              # Xi^3 = Xi^2 * Xi.
533  
534              # degree 0 term
535              if self.include_bias:
536                  XP[:, 0] = 1
537                  current_col = 1
538              else:
539                  current_col = 0
540  
541              if self._max_degree == 0:
542                  return XP
543  
544              # degree 1 term
545              XP[:, current_col : current_col + n_features] = X
546              index = list(range(current_col, current_col + n_features))
547              current_col += n_features
548              index.append(current_col)
549  
550              # loop over degree >= 2 terms
551              for _ in range(2, self._max_degree + 1):
552                  new_index = []
553                  end = index[-1]
554                  for feature_idx in range(n_features):
555                      start = index[feature_idx]
556                      new_index.append(current_col)
557                      if self.interaction_only:
558                          start += index[feature_idx + 1] - index[feature_idx]
559                      next_col = current_col + end - start
560                      if next_col <= current_col:
561                          break
562                      # XP[:, start:end] are terms of degree d - 1
563                      # that exclude feature #feature_idx.
564                      np.multiply(
565                          XP[:, start:end],
566                          X[:, feature_idx : feature_idx + 1],
567                          out=XP[:, current_col:next_col],
568                          casting="no",
569                      )
570                      current_col = next_col
571  
572                  new_index.append(current_col)
573                  index = new_index
574  
575              if self._min_degree > 1:
576                  n_XP, n_Xout = self._n_out_full, self.n_output_features_
577                  if self.include_bias:
578                      Xout = np.empty(
579                          shape=(n_samples, n_Xout), dtype=XP.dtype, order=self.order
580                      )
581                      Xout[:, 0] = 1
582                      Xout[:, 1:] = XP[:, n_XP - n_Xout + 1 :]
583                  else:
584                      Xout = XP[:, n_XP - n_Xout :].copy()
585                  XP = Xout
586          return XP

594  class SplineTransformer(TransformerMixin, BaseEstimator):
595      """Generate univariate B-spline bases for features. 
596   
597      Generate a new feature matrix consisting of 
598      `n_splines=n_knots + degree - 1` (`n_knots - 1` for 
599      `extrapolation="periodic"`) spline basis functions 
600      (B-splines) of polynomial order=`degree` for each feature. 
601   
602      In order to learn more about the SplineTransformer class go to: 
603      :ref:`sphx_glr_auto_examples_applications_plot_cyclical_feature_engineering.py` 
604   
605      Read more in the :ref:`User Guide <spline_transformer>`. 
606   
607      .. versionadded:: 1.0 
608   
609      Parameters 
610      ---------- 
611      n_knots : int, default=5 
612          Number of knots of the splines if `knots` equals one of 
613          {'uniform', 'quantile'}. Must be larger or equal 2. Ignored if `knots` 
614          is array-like. 
615   
616      degree : int, default=3 
617          The polynomial degree of the spline basis. Must be a non-negative 
618          integer. 
619   
620      knots : {'uniform', 'quantile'} or array-like of shape \ 
621          (n_knots, n_features), default='uniform' 
622          Set knot positions such that first knot <= features <= last knot. 
623   
624          - If 'uniform', `n_knots` number of knots are distributed uniformly 
625            from min to max values of the features. 
626          - If 'quantile', they are distributed uniformly along the quantiles of 
627            the features. 
628          - If an array-like is given, it directly specifies the sorted knot 
629            positions including the boundary knots. Note that, internally, 
630            `degree` number of knots are added before the first knot, the same 
631            after the last knot. 
632   
633      extrapolation : {'error', 'constant', 'linear', 'continue', 'periodic'}, \ 
634          default='constant' 
635          If 'error', values outside the min and max values of the training 
636          features raises a `ValueError`. If 'constant', the value of the 
637          splines at minimum and maximum value of the features is used as 
638          constant extrapolation. If 'linear', a linear extrapolation is used. 
639          If 'continue', the splines are extrapolated as is, i.e. option 
640          `extrapolate=True` in :class:`scipy.interpolate.BSpline`. If 
641          'periodic', periodic splines with a periodicity equal to the distance 
642          between the first and last knot are used. Periodic splines enforce 
643          equal function values and derivatives at the first and last knot. 
644          For example, this makes it possible to avoid introducing an arbitrary 
645          jump between Dec 31st and Jan 1st in spline features derived from a 
646          naturally periodic "day-of-year" input feature. In this case it is 
647          recommended to manually set the knot values to control the period. 
648   
649      include_bias : bool, default=True 
650          If False, then the last spline element inside the data range 
651          of a feature is dropped. As B-splines sum to one over the spline basis 
652          functions for each data point, they implicitly include a bias term, 
653          i.e. a column of ones. It acts as an intercept term in a linear models. 
654   
655      order : {'C', 'F'}, default='C' 
656          Order of output array in the dense case. `'F'` order is faster to compute, but 
657          may slow down subsequent estimators. 
658   
659      sparse_output : bool, default=False 
660          Will return sparse CSR matrix if set True else will return an array. This 
661          option is only available with `scipy>=1.8`. 
662   
663          .. versionadded:: 1.2 
664   
665      Attributes 
666      ---------- 
667      bsplines_ : list of shape (n_features,) 
668          List of BSplines objects, one for each feature. 
669   
670      n_features_in_ : int 
671          The total number of input features. 
672   
673      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
674          Names of features seen during :term:`fit`. Defined only when `X` 
675          has feature names that are all strings. 
676   
677          .. versionadded:: 1.0 
678   
679      n_features_out_ : int 
680          The total number of output features, which is computed as 
681          `n_features * n_splines`, where `n_splines` is 
682          the number of bases elements of the B-splines, 
683          `n_knots + degree - 1` for non-periodic splines and 
684          `n_knots - 1` for periodic ones. 
685          If `include_bias=False`, then it is only 
686          `n_features * (n_splines - 1)`. 
687   
688      See Also 
689      -------- 
690      KBinsDiscretizer : Transformer that bins continuous data into intervals. 
691   
692      PolynomialFeatures : Transformer that generates polynomial and interaction 
693          features. 
694   
695      Notes 
696      ----- 
697      High degrees and a high number of knots can cause overfitting. 
698   
699      See :ref:`examples/linear_model/plot_polynomial_interpolation.py 
700      <sphx_glr_auto_examples_linear_model_plot_polynomial_interpolation.py>`. 
701   
702      Examples 
703      -------- 
704      >>> import numpy as np 
705      >>> from sklearn.preprocessing import SplineTransformer 
706      >>> X = np.arange(6).reshape(6, 1) 
707      >>> spline = SplineTransformer(degree=2, n_knots=3) 
708      >>> spline.fit_transform(X) 
709      array([[0.5 , 0.5 , 0.  , 0.  ], 
710             [0.18, 0.74, 0.08, 0.  ], 
711             [0.02, 0.66, 0.32, 0.  ], 
712             [0.  , 0.32, 0.66, 0.02], 
713             [0.  , 0.08, 0.74, 0.18], 
714             [0.  , 0.  , 0.5 , 0.5 ]]) 
715      """

748      @staticmethod
749      def _get_base_knot_positions(X, n_knots=10, knots="uniform", sample_weight=None):
750          """Calculate base knot positions. 
751   
752          Base knots such that first knot <= feature <= last knot. For the 
753          B-spline construction with scipy.interpolate.BSpline, 2*degree knots 
754          beyond the base interval are added. 
755   
756          Returns 
757          ------- 
758          knots : ndarray of shape (n_knots, n_features), dtype=np.float64 
759              Knot positions (points) of base interval. 
760          """
761          if knots == "quantile":
762              percentiles = 100 * np.linspace(
763                  start=0, stop=1, num=n_knots, dtype=np.float64
764              )
765  
766              if sample_weight is None:
767                  knots = np.percentile(X, percentiles, axis=0)
768              else:
769                  knots = np.array(
770                      [
771                          _weighted_percentile(X, sample_weight, percentile)
772                          for percentile in percentiles
773                      ]
774                  )
775  
776          else:
777              # knots == 'uniform':
778              # Note that the variable `knots` has already been validated and
779              # `else` is therefore safe.
780              # Disregard observations with zero weight.
781              mask = slice(None, None, 1) if sample_weight is None else sample_weight > 0
782              x_min = np.amin(X[mask], axis=0)
783              x_max = np.amax(X[mask], axis=0)
784  
785              knots = np.linspace(
786                  start=x_min,
787                  stop=x_max,
788                  num=n_knots,
789                  endpoint=True,
790                  dtype=np.float64,
791              )
792  
793          return knots

795      def get_feature_names_out(self, input_features=None):
796          """Get output feature names for transformation. 
797   
798          Parameters 
799          ---------- 
800          input_features : array-like of str or None, default=None 
801              Input features. 
802   
803              - If `input_features` is `None`, then `feature_names_in_` is 
804                used as feature names in. If `feature_names_in_` is not defined, 
805                then the following input feature names are generated: 
806                `["x0", "x1", ..., "x(n_features_in_ - 1)"]`. 
807              - If `input_features` is an array-like, then `input_features` must 
808                match `feature_names_in_` if `feature_names_in_` is defined. 
809   
810          Returns 
811          ------- 
812          feature_names_out : ndarray of str objects 
813              Transformed feature names. 
814          """
815          check_is_fitted(self, "n_features_in_")
816          n_splines = self.bsplines_[0].c.shape[1]
817  
818          input_features = _check_feature_names_in(self, input_features)
819          feature_names = []
820          for i in range(self.n_features_in_):
821              for j in range(n_splines - 1 + self.include_bias):
822                  feature_names.append(f"{input_features[i]}_sp_{j}")
823          return np.asarray(feature_names, dtype=object)

825      @_fit_context(prefer_skip_nested_validation=True)
826      def fit(self, X, y=None, sample_weight=None):
827          """Compute knot positions of splines. 
828   
829          Parameters 
830          ---------- 
831          X : array-like of shape (n_samples, n_features) 
832              The data. 
833   
834          y : None 
835              Ignored. 
836   
837          sample_weight : array-like of shape (n_samples,), default = None 
838              Individual weights for each sample. Used to calculate quantiles if 
839              `knots="quantile"`. For `knots="uniform"`, zero weighted 
840              observations are ignored for finding the min and max of `X`. 
841   
842          Returns 
843          ------- 
844          self : object 
845              Fitted transformer. 
846          """
847          X = validate_data(
848              self,
849              X,
850              reset=True,
851              accept_sparse=False,
852              ensure_min_samples=2,
853              ensure_2d=True,
854          )
855          if sample_weight is not None:
856              sample_weight = _check_sample_weight(sample_weight, X, dtype=X.dtype)
857  
858          _, n_features = X.shape
859  
860          if isinstance(self.knots, str):
861              base_knots = self._get_base_knot_positions(
862                  X, n_knots=self.n_knots, knots=self.knots, sample_weight=sample_weight
863              )
864          else:
865              base_knots = check_array(self.knots, dtype=np.float64)
866              if base_knots.shape[0] < 2:
867                  raise ValueError("Number of knots, knots.shape[0], must be >= 2.")
868              elif base_knots.shape[1] != n_features:
869                  raise ValueError("knots.shape[1] == n_features is violated.")
870              elif not np.all(np.diff(base_knots, axis=0) > 0):
871                  raise ValueError("knots must be sorted without duplicates.")
872  
873          if self.sparse_output and sp_version < parse_version("1.8.0"):
874              raise ValueError(
875                  "Option sparse_output=True is only available with scipy>=1.8.0, "
876                  f"but here scipy=={sp_version} is used."
877              )
878  
879          # number of knots for base interval
880          n_knots = base_knots.shape[0]
881  
882          if self.extrapolation == "periodic" and n_knots <= self.degree:
883              raise ValueError(
884                  "Periodic splines require degree < n_knots. Got n_knots="
885                  f"{n_knots} and degree={self.degree}."
886              )
887  
888          # number of splines basis functions
889          if self.extrapolation != "periodic":
890              n_splines = n_knots + self.degree - 1
891          else:
892              # periodic splines have self.degree less degrees of freedom
893              n_splines = n_knots - 1
894  
895          degree = self.degree
896          n_out = n_features * n_splines
897          # We have to add degree number of knots below, and degree number knots
898          # above the base knots in order to make the spline basis complete.
899          if self.extrapolation == "periodic":
900              # For periodic splines the spacing of the first / last degree knots
901              # needs to be a continuation of the spacing of the last / first
902              # base knots.
903              period = base_knots[-1] - base_knots[0]
904              knots = np.r_[
905                  base_knots[-(degree + 1) : -1] - period,
906                  base_knots,
907                  base_knots[1 : (degree + 1)] + period,
908              ]
909  
910          else:
911              # Eilers & Marx in "Flexible smoothing with B-splines and
912              # penalties" https://doi.org/10.1214/ss/1038425655 advice
913              # against repeating first and last knot several times, which
914              # would have inferior behaviour at boundaries if combined with
915              # a penalty (hence P-Spline). We follow this advice even if our
916              # splines are unpenalized. Meaning we do not:
917              # knots = np.r_[
918              #     np.tile(base_knots.min(axis=0), reps=[degree, 1]),
919              #     base_knots,
920              #     np.tile(base_knots.max(axis=0), reps=[degree, 1])
921              # ]
922              # Instead, we reuse the distance of the 2 fist/last knots.
923              dist_min = base_knots[1] - base_knots[0]
924              dist_max = base_knots[-1] - base_knots[-2]
925  
926              knots = np.r_[
927                  np.linspace(
928                      base_knots[0] - degree * dist_min,
929                      base_knots[0] - dist_min,
930                      num=degree,
931                  ),
932                  base_knots,
933                  np.linspace(
934                      base_knots[-1] + dist_max,
935                      base_knots[-1] + degree * dist_max,
936                      num=degree,
937                  ),
938              ]
939  
940          # With a diagonal coefficient matrix, we get back the spline basis
941          # elements, i.e. the design matrix of the spline.
942          # Note, BSpline appreciates C-contiguous float64 arrays as c=coef.
943          coef = np.eye(n_splines, dtype=np.float64)
944          if self.extrapolation == "periodic":
945              coef = np.concatenate((coef, coef[:degree, :]))
946  
947          extrapolate = self.extrapolation in ["periodic", "continue"]
948  
949          bsplines = [
950              BSpline.construct_fast(
951                  knots[:, i], coef, self.degree, extrapolate=extrapolate
952              )
953              for i in range(n_features)
954          ]
955          self.bsplines_ = bsplines
956  
957          self.n_features_out_ = n_out - n_features * (1 - self.include_bias)
958          return self

960      def transform(self, X):
961          """Transform each feature data to B-splines. 
962   
963          Parameters 
964          ---------- 
965          X : array-like of shape (n_samples, n_features) 
966              The data to transform. 
967   
968          Returns 
969          ------- 
970          XBS : {ndarray, sparse matrix} of shape (n_samples, n_features * n_splines) 
971              The matrix of features, where n_splines is the number of bases 
972              elements of the B-splines, n_knots + degree - 1. 
973          """
974          check_is_fitted(self)
975  
976          X = validate_data(self, X, reset=False, accept_sparse=False, ensure_2d=True)
977  
978          n_samples, n_features = X.shape
979          n_splines = self.bsplines_[0].c.shape[1]
980          degree = self.degree
981  
982          # TODO: Remove this condition, once scipy 1.10 is the minimum version.
983          #       Only scipy => 1.10 supports design_matrix(.., extrapolate=..).
984          #       The default (implicit in scipy < 1.10) is extrapolate=False.
985          scipy_1_10 = sp_version >= parse_version("1.10.0")
986          # Note: self.bsplines_[0].extrapolate is True for extrapolation in
987          # ["periodic", "continue"]
988          if scipy_1_10:
989              use_sparse = self.sparse_output
990              kwargs_extrapolate = {"extrapolate": self.bsplines_[0].extrapolate}
991          else:
992              use_sparse = self.sparse_output and not self.bsplines_[0].extrapolate
993              kwargs_extrapolate = dict()
994  
995          # Note that scipy BSpline returns float64 arrays and converts input
996          # x=X[:, i] to c-contiguous float64.
997          n_out = self.n_features_out_ + n_features * (1 - self.include_bias)
998          if X.dtype in FLOAT_DTYPES:
999              dtype = X.dtype
1000         else:
1001             dtype = np.float64
1002         if use_sparse:
1003             output_list = []
1004         else:
1005             XBS = np.zeros((n_samples, n_out), dtype=dtype, order=self.order)
1006 
1007         for i in range(n_features):
1008             spl = self.bsplines_[i]
1009 
1010             if self.extrapolation in ("continue", "error", "periodic"):
1011                 if self.extrapolation == "periodic":
1012                     # With periodic extrapolation we map x to the segment
1013                     # [spl.t[k], spl.t[n]].
1014                     # This is equivalent to BSpline(.., extrapolate="periodic")
1015                     # for scipy>=1.0.0.
1016                     n = spl.t.size - spl.k - 1
1017                     # Assign to new array to avoid inplace operation
1018                     x = spl.t[spl.k] + (X[:, i] - spl.t[spl.k]) % (
1019                         spl.t[n] - spl.t[spl.k]
1020                     )
1021                 else:
1022                     x = X[:, i]
1023 
1024                 if use_sparse:
1025                     XBS_sparse = BSpline.design_matrix(
1026                         x, spl.t, spl.k, **kwargs_extrapolate
1027                     )
1028                     if self.extrapolation == "periodic":
1029                         # See the construction of coef in fit. We need to add the last
1030                         # degree spline basis function to the first degree ones and
1031                         # then drop the last ones.
1032                         # Note: See comment about SparseEfficiencyWarning below.
1033                         XBS_sparse = XBS_sparse.tolil()
1034                         XBS_sparse[:, :degree] += XBS_sparse[:, -degree:]
1035                         XBS_sparse = XBS_sparse[:, :-degree]
1036                 else:
1037                     XBS[:, (i * n_splines) : ((i + 1) * n_splines)] = spl(x)
1038             else:  # extrapolation in ("constant", "linear")
1039                 xmin, xmax = spl.t[degree], spl.t[-degree - 1]
1040                 # spline values at boundaries
1041                 f_min, f_max = spl(xmin), spl(xmax)
1042                 mask = (xmin <= X[:, i]) & (X[:, i] <= xmax)
1043                 if use_sparse:
1044                     mask_inv = ~mask
1045                     x = X[:, i].copy()
1046                     # Set some arbitrary values outside boundary that will be reassigned
1047                     # later.
1048                     x[mask_inv] = spl.t[self.degree]
1049                     XBS_sparse = BSpline.design_matrix(x, spl.t, spl.k)
1050                     # Note: Without converting to lil_matrix we would get:
1051                     # scipy.sparse._base.SparseEfficiencyWarning: Changing the sparsity
1052                     # structure of a csr_matrix is expensive. lil_matrix is more
1053                     # efficient.
1054                     if np.any(mask_inv):
1055                         XBS_sparse = XBS_sparse.tolil()
1056                         XBS_sparse[mask_inv, :] = 0
1057                 else:
1058                     XBS[mask, (i * n_splines) : ((i + 1) * n_splines)] = spl(X[mask, i])
1059 
1060             # Note for extrapolation:
1061             # 'continue' is already returned as is by scipy BSplines
1062             if self.extrapolation == "error":
1063                 # BSpline with extrapolate=False does not raise an error, but
1064                 # outputs np.nan.
1065                 if (use_sparse and np.any(np.isnan(XBS_sparse.data))) or (
1066                     not use_sparse
1067                     and np.any(
1068                         np.isnan(XBS[:, (i * n_splines) : ((i + 1) * n_splines)])
1069                     )
1070                 ):
1071                     raise ValueError(
1072                         "X contains values beyond the limits of the knots."
1073                     )
1074             elif self.extrapolation == "constant":
1075                 # Set all values beyond xmin and xmax to the value of the
1076                 # spline basis functions at those two positions.
1077                 # Only the first degree and last degree number of splines
1078                 # have non-zero values at the boundaries.
1079 
1080                 mask = X[:, i] < xmin
1081                 if np.any(mask):
1082                     if use_sparse:
1083                         # Note: See comment about SparseEfficiencyWarning above.
1084                         XBS_sparse = XBS_sparse.tolil()
1085                         XBS_sparse[mask, :degree] = f_min[:degree]
1086 
1087                     else:
1088                         XBS[mask, (i * n_splines) : (i * n_splines + degree)] = f_min[
1089                             :degree
1090                         ]
1091 
1092                 mask = X[:, i] > xmax
1093                 if np.any(mask):
1094                     if use_sparse:
1095                         # Note: See comment about SparseEfficiencyWarning above.
1096                         XBS_sparse = XBS_sparse.tolil()
1097                         XBS_sparse[mask, -degree:] = f_max[-degree:]
1098                     else:
1099                         XBS[
1100                             mask,
1101                             ((i + 1) * n_splines - degree) : ((i + 1) * n_splines),
1102                         ] = f_max[-degree:]
1103 
1104             elif self.extrapolation == "linear":
1105                 # Continue the degree first and degree last spline bases
1106                 # linearly beyond the boundaries, with slope = derivative at
1107                 # the boundary.
1108                 # Note that all others have derivative = value = 0 at the
1109                 # boundaries.
1110 
1111                 # spline derivatives = slopes at boundaries
1112                 fp_min, fp_max = spl(xmin, nu=1), spl(xmax, nu=1)
1113                 # Compute the linear continuation.
1114                 if degree <= 1:
1115                     # For degree=1, the derivative of 2nd spline is not zero at
1116                     # boundary. For degree=0 it is the same as 'constant'.
1117                     degree += 1
1118                 for j in range(degree):
1119                     mask = X[:, i] < xmin
1120                     if np.any(mask):
1121                         linear_extr = f_min[j] + (X[mask, i] - xmin) * fp_min[j]
1122                         if use_sparse:
1123                             # Note: See comment about SparseEfficiencyWarning above.
1124                             XBS_sparse = XBS_sparse.tolil()
1125                             XBS_sparse[mask, j] = linear_extr
1126                         else:
1127                             XBS[mask, i * n_splines + j] = linear_extr
1128 
1129                     mask = X[:, i] > xmax
1130                     if np.any(mask):
1131                         k = n_splines - 1 - j
1132                         linear_extr = f_max[k] + (X[mask, i] - xmax) * fp_max[k]
1133                         if use_sparse:
1134                             # Note: See comment about SparseEfficiencyWarning above.
1135                             XBS_sparse = XBS_sparse.tolil()
1136                             XBS_sparse[mask, k : k + 1] = linear_extr[:, None]
1137                         else:
1138                             XBS[mask, i * n_splines + k] = linear_extr
1139 
1140             if use_sparse:
1141                 XBS_sparse = XBS_sparse.tocsr()
1142                 output_list.append(XBS_sparse)
1143 
1144         if use_sparse:
1145             # TODO: Remove this conditional error when the minimum supported version of
1146             # SciPy is 1.9.2
1147             # `scipy.sparse.hstack` breaks in scipy<1.9.2
1148             # when `n_features_out_ > max_int32`
1149             max_int32 = np.iinfo(np.int32).max
1150             all_int32 = True
1151             for mat in output_list:
1152                 all_int32 &= mat.indices.dtype == np.int32
1153             if (
1154                 sp_version < parse_version("1.9.2")
1155                 and self.n_features_out_ > max_int32
1156                 and all_int32
1157             ):
1158                 raise ValueError(
1159                     "In scipy versions `<1.9.2`, the function `scipy.sparse.hstack`"
1160                     " produces negative columns when:\n1. The output shape contains"
1161                     " `n_cols` too large to be represented by a 32bit signed"
1162                     " integer.\n. All sub-matrices to be stacked have indices of"
1163                     " dtype `np.int32`.\nTo avoid this error, either use a version"
1164                     " of scipy `>=1.9.2` or alter the `SplineTransformer`"
1165                     " transformer to produce fewer than 2^31 output features"
1166                 )
1167             XBS = sparse.hstack(output_list, format="csr")
1168         elif self.sparse_output:
1169             # TODO: Remove ones scipy 1.10 is the minimum version. See comments above.
1170             XBS = sparse.csr_matrix(XBS)
1171 
1172         if self.include_bias:
1173             return XBS
1174         else:
1175             # We throw away one spline basis per feature.
1176             # We chose the last one.
1177             indices = [j for j in range(XBS.shape[1]) if (j + 1) % n_splines != 0]
1178             return XBS[:, indices]
