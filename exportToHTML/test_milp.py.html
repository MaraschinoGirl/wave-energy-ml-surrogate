100      res = milp(c=c, constraints=(A, b, b), bounds=(0, 1), integrality=1)
101      assert res.status == 0
102      assert res.success
103      msg = "Optimization terminated successfully. (HiGHS Status 7:"
104      assert res.message.startswith(msg)
105      assert isinstance(res.x, np.ndarray)
106      assert isinstance(res.fun, float)
107      assert isinstance(res.mip_node_count, int)
108      assert isinstance(res.mip_dual_bound, float)
109      assert isinstance(res.mip_gap, float)

113                 options={'time_limit': 0.05})
114      assert res.status == 1
115      assert not res.success
116      msg = "Time limit reached. (HiGHS Status 13:"
117      assert res.message.startswith(msg)
118      assert (res.fun is res.mip_dual_bound is res.mip_gap
119              is res.mip_node_count is res.x is None)

145  def test_milp_1():
146      # solve magic square problem
147      n = 3
148      A, b, c, numbers, M = magic_square(n)
149      A = sparse.csc_array(A)  # confirm that sparse arrays are accepted
150      res = milp(c=c*0, constraints=(A, b, b), bounds=(0, 1), integrality=1)
151  
152      # check that solution is a magic square
153      x = np.round(res.x)
154      s = (numbers.flatten() * x).reshape(n**2, n, n)
155      square = np.sum(s, axis=0)
156      np.testing.assert_allclose(square.sum(axis=0), M)
157      np.testing.assert_allclose(square.sum(axis=1), M)
158      np.testing.assert_allclose(np.diag(square).sum(), M)
159      np.testing.assert_allclose(np.diag(square[:, ::-1]).sum(), M)

162  def test_milp_2():
163      # solve MIP with inequality constraints and all integer constraints
164      # source: slide 5,
165      # https://www.cs.upc.edu/~erodri/webpage/cps/theory/lp/milp/slides.pdf
166      # also check that `milp` accepts all valid ways of specifying constraints
167      c = -np.ones(2)
168      A = [[-2, 2], [-8, 10]]
169      b_l = [1, -np.inf]
170      b_u = [np.inf, 13]
171      linear_constraint = LinearConstraint(A, b_l, b_u)
172  
173      # solve original problem
174      res1 = milp(c=c, constraints=(A, b_l, b_u), integrality=True)
175      res2 = milp(c=c, constraints=linear_constraint, integrality=True)
176      res3 = milp(c=c, constraints=[(A, b_l, b_u)], integrality=True)
177      res4 = milp(c=c, constraints=[linear_constraint], integrality=True)
178      res5 = milp(c=c, integrality=True,
179                  constraints=[(A[:1], b_l[:1], b_u[:1]),
180                               (A[1:], b_l[1:], b_u[1:])])
181      res6 = milp(c=c, integrality=True,
182                  constraints=[LinearConstraint(A[:1], b_l[:1], b_u[:1]),
183                               LinearConstraint(A[1:], b_l[1:], b_u[1:])])
184      res7 = milp(c=c, integrality=True,
185                  constraints=[(A[:1], b_l[:1], b_u[:1]),
186                               LinearConstraint(A[1:], b_l[1:], b_u[1:])])
187      xs = np.array([res1.x, res2.x, res3.x, res4.x, res5.x, res6.x, res7.x])
188      funs = np.array([res1.fun, res2.fun, res3.fun,
189                       res4.fun, res5.fun, res6.fun, res7.fun])
190      np.testing.assert_allclose(xs, np.broadcast_to([1, 2], xs.shape))
191      np.testing.assert_allclose(funs, -3)
192  
193      # solve relaxed problem
194      res = milp(c=c, constraints=(A, b_l, b_u))
195      np.testing.assert_allclose(res.x, [4, 4.5])
196      np.testing.assert_allclose(res.fun, -8.5)

199  def test_milp_3():
200      # solve MIP with inequality constraints and all integer constraints
201      # source: https://en.wikipedia.org/wiki/Integer_programming#Example
202      c = [0, -1]
203      A = [[-1, 1], [3, 2], [2, 3]]
204      b_u = [1, 12, 12]
205      b_l = np.full_like(b_u, -np.inf, dtype=np.float64)
206      constraints = LinearConstraint(A, b_l, b_u)
207  
208      integrality = np.ones_like(c)
209  
210      # solve original problem
211      res = milp(c=c, constraints=constraints, integrality=integrality)
212      assert_allclose(res.fun, -2)
213      # two optimal solutions possible, just need one of them
214      assert np.allclose(res.x, [1, 2]) or np.allclose(res.x, [2, 2])
215  
216      # solve relaxed problem
217      res = milp(c=c, constraints=constraints)
218      assert_allclose(res.fun, -2.8)
219      assert_allclose(res.x, [1.8, 2.8])

305  def test_milp_timeout_16545(options, msg):
306      # Ensure solution is not thrown away if MILP solver times out
307      # -- see gh-16545
308      rng = np.random.default_rng(5123833489170494244)
309      A = rng.integers(0, 5, size=(100, 100))
310      b_lb = np.full(100, fill_value=-np.inf)
311      b_ub = np.full(100, fill_value=25)
312      constraints = LinearConstraint(A, b_lb, b_ub)
313      variable_lb = np.zeros(100)
314      variable_ub = np.ones(100)
315      variable_bounds = Bounds(variable_lb, variable_ub)
316      integrality = np.ones(100)
317      c_vector = -np.ones(100)
318      res = milp(
319          c_vector,
320          integrality=integrality,
321          bounds=variable_bounds,
322          constraints=constraints,
323          options=options,
324      )
325  
326      assert res.message.startswith(msg)
327      assert res["x"] is not None
328  
329      # ensure solution is feasible
330      x = res["x"]
331      tol = 1e-8  # sometimes needed due to finite numerical precision
332      assert np.all(b_lb - tol <= A @ x) and np.all(A @ x <= b_ub + tol)
333      assert np.all(variable_lb - tol <= x) and np.all(x <= variable_ub + tol)
334      assert np.allclose(x, np.round(x))

338      # `milp` failed when exactly three constraints were passed
339      # Ensure that this is no longer the case.
340      rng = np.random.default_rng(5123833489170494244)
341      A = rng.integers(0, 5, size=(6, 6))
342      bl = np.full(6, fill_value=-np.inf)
343      bu = np.full(6, fill_value=10)
344      constraints = [LinearConstraint(A[:2], bl[:2], bu[:2]),
345                     LinearConstraint(A[2:4], bl[2:4], bu[2:4]),
346                     LinearConstraint(A[4:], bl[4:], bu[4:])]
347      constraints2 = [(A[:2], bl[:2], bu[:2]),
348                      (A[2:4], bl[2:4], bu[2:4]),
349                      (A[4:], bl[4:], bu[4:])]

351      ub = np.ones(6)
352      variable_bounds = Bounds(lb, ub)
353      c = -np.ones(6)

355      res2 = milp(c, bounds=variable_bounds, constraints=constraints2)

357      assert res1.success and res2.success
358      assert_allclose(res1.x, ref.x)
359      assert_allclose(res2.x, ref.x)

363  def test_mip_rel_gap_passdown():
364      # Solve problem with decreasing mip_gap to make sure mip_rel_gap decreases
365      # Adapted from test_linprog::TestLinprogHiGHSMIP::test_mip_rel_gap_passdown
366      # MIP taken from test_mip_6 above
367      A_eq = np.array([[22, 13, 26, 33, 21, 3, 14, 26],
368                       [39, 16, 22, 28, 26, 30, 23, 24],
369                       [18, 14, 29, 27, 30, 38, 26, 26],
370                       [41, 26, 28, 36, 18, 38, 16, 26]])
371      b_eq = np.array([7872, 10466, 11322, 12058])
372      c = np.array([2, 10, 13, 17, 7, 5, 7, 3])
373  
374      mip_rel_gaps = [0.25, 0.01, 0.001]
375      sol_mip_gaps = []
376      for mip_rel_gap in mip_rel_gaps:
377          res = milp(c=c, bounds=(0, np.inf), constraints=(A_eq, b_eq, b_eq),
378                     integrality=True, options={"mip_rel_gap": mip_rel_gap})
379          # assert that the solution actually has mip_gap lower than the
380          # required mip_rel_gap supplied
381          assert res.mip_gap <= mip_rel_gap
382          # check that `res.mip_gap` is as defined in the documentation
383          assert res.mip_gap == (res.fun - res.mip_dual_bound)/res.fun
384          sol_mip_gaps.append(res.mip_gap)
385  
386      # make sure that the mip_rel_gap parameter is actually doing something
387      # check that differences between solution gaps are declining
388      # monotonically with the mip_rel_gap parameter.
389      assert np.all(np.diff(sol_mip_gaps) < 0)
