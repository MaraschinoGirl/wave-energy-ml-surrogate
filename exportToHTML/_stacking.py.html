88       def _concatenate_predictions(self, X, predictions):
89           """Concatenate the predictions of each first layer learner and 
90           possibly the input dataset `X`. 
91    
92           If `X` is sparse and `self.passthrough` is False, the output of 
93           `transform` will be dense (the predictions). If `X` is sparse 
94           and `self.passthrough` is True, the output of `transform` will 
95           be sparse. 
96    
97           This helper is in charge of ensuring the predictions are 2D arrays and 
98           it will drop one of the probability column when using probabilities 
99           in the binary case. Indeed, the p(y|c=0) = 1 - p(y|c=1) 
100   
101          When `y` type is `"multilabel-indicator"`` and the method used is 
102          `predict_proba`, `preds` can be either a `ndarray` of shape 
103          `(n_samples, n_class)` or for some estimators a list of `ndarray`. 
104          This function will drop one of the probability column in this situation as well. 
105          """
106          X_meta = []
107          for est_idx, preds in enumerate(predictions):
108              if isinstance(preds, list):
109                  # `preds` is here a list of `n_targets` 2D ndarrays of
110                  # `n_classes` columns. The k-th column contains the
111                  # probabilities of the samples belonging the k-th class.
112                  #
113                  # Since those probabilities must sum to one for each sample,
114                  # we can work with probabilities of `n_classes - 1` classes.
115                  # Hence we drop the first column.
116                  for pred in preds:
117                      X_meta.append(pred[:, 1:])
118              elif preds.ndim == 1:
119                  # Some estimator return a 1D array for predictions
120                  # which must be 2-dimensional arrays.
121                  X_meta.append(preds.reshape(-1, 1))
122              elif (
123                  self.stack_method_[est_idx] == "predict_proba"
124                  and len(self.classes_) == 2
125              ):
126                  # Remove the first column when using probabilities in
127                  # binary classification because both features `preds` are perfectly
128                  # collinear.
129                  X_meta.append(preds[:, 1:])
130              else:
131                  X_meta.append(preds)
132  
133          self._n_feature_outs = [pred.shape[1] for pred in X_meta]
134          if self.passthrough:
135              X_meta.append(X)
136              if sparse.issparse(X):
137                  return sparse.hstack(X_meta, format=X.format)
138  
139          return np.hstack(X_meta)

141      @staticmethod
142      def _method_name(name, estimator, method):
143          if estimator == "drop":
144              return None
145          if method == "auto":
146              method = ["predict_proba", "decision_function", "predict"]
147          try:
148              method_name = _check_response_method(estimator, method).__name__
149          except AttributeError as e:
150              raise ValueError(
151                  f"Underlying estimator {name} does not implement the method {method}."
152              ) from e
153  
154          return method_name

156      @_fit_context(
157          # estimators in Stacking*.estimators are not validated yet
158          prefer_skip_nested_validation=False
159      )
160      def fit(self, X, y, **fit_params):
161          """Fit the estimators. 
162   
163          Parameters 
164          ---------- 
165          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
166              Training vectors, where `n_samples` is the number of samples and 
167              `n_features` is the number of features. 
168   
169          y : array-like of shape (n_samples,) 
170              Target values. 
171   
172          **fit_params : dict 
173              Dict of metadata, potentially containing sample_weight as a 
174              key-value pair. If sample_weight is not present, then samples are 
175              equally weighted. Note that sample_weight is supported only if all 
176              underlying estimators support sample weights. 
177   
178              .. versionadded:: 1.6 
179   
180          Returns 
181          ------- 
182          self : object 
183          """
184          # all_estimators contains all estimators, the one to be fitted and the
185          # 'drop' string.
186          names, all_estimators = self._validate_estimators()
187          self._validate_final_estimator()
188  
189          stack_method = [self.stack_method] * len(all_estimators)
190  
191          if _routing_enabled():
192              routed_params = process_routing(self, "fit", **fit_params)
193          else:
194              routed_params = Bunch()
195              for name in names:
196                  routed_params[name] = Bunch(fit={})
197                  if "sample_weight" in fit_params:
198                      routed_params[name].fit["sample_weight"] = fit_params[
199                          "sample_weight"
200                      ]
201  
202          if self.cv == "prefit":
203              self.estimators_ = []
204              for estimator in all_estimators:
205                  if estimator != "drop":
206                      check_is_fitted(estimator)
207                      self.estimators_.append(estimator)
208          else:
209              # Fit the base estimators on the whole training data. Those
210              # base estimators will be used in transform, predict, and
211              # predict_proba. They are exposed publicly.
212              self.estimators_ = Parallel(n_jobs=self.n_jobs)(
213                  delayed(_fit_single_estimator)(
214                      clone(est), X, y, routed_params[name]["fit"]
215                  )
216                  for name, est in zip(names, all_estimators)
217                  if est != "drop"
218              )
219  
220          self.named_estimators_ = Bunch()
221          est_fitted_idx = 0
222          for name_est, org_est in zip(names, all_estimators):
223              if org_est != "drop":
224                  current_estimator = self.estimators_[est_fitted_idx]
225                  self.named_estimators_[name_est] = current_estimator
226                  est_fitted_idx += 1
227                  if hasattr(current_estimator, "feature_names_in_"):
228                      self.feature_names_in_ = current_estimator.feature_names_in_
229              else:
230                  self.named_estimators_[name_est] = "drop"
231  
232          self.stack_method_ = [
233              self._method_name(name, est, meth)
234              for name, est, meth in zip(names, all_estimators, stack_method)
235          ]
236  
237          if self.cv == "prefit":
238              # Generate predictions from prefit models
239              predictions = [
240                  getattr(estimator, predict_method)(X)
241                  for estimator, predict_method in zip(all_estimators, self.stack_method_)
242                  if estimator != "drop"
243              ]
244          else:
245              # To train the meta-classifier using the most data as possible, we use
246              # a cross-validation to obtain the output of the stacked estimators.
247              # To ensure that the data provided to each estimator are the same,
248              # we need to set the random state of the cv if there is one and we
249              # need to take a copy.
250              cv = check_cv(self.cv, y=y, classifier=is_classifier(self))
251              if hasattr(cv, "random_state") and cv.random_state is None:
252                  cv.random_state = np.random.RandomState()
253  
254              predictions = Parallel(n_jobs=self.n_jobs)(
255                  delayed(cross_val_predict)(
256                      clone(est),
257                      X,
258                      y,
259                      cv=deepcopy(cv),
260                      method=meth,
261                      n_jobs=self.n_jobs,
262                      params=routed_params[name]["fit"],
263                      verbose=self.verbose,
264                  )
265                  for name, est, meth in zip(names, all_estimators, self.stack_method_)
266                  if est != "drop"
267              )
268  
269          # Only not None or not 'drop' estimators will be used in transform.
270          # Remove the None from the method as well.
271          self.stack_method_ = [
272              meth
273              for (meth, est) in zip(self.stack_method_, all_estimators)
274              if est != "drop"
275          ]
276  
277          X_meta = self._concatenate_predictions(X, predictions)
278          _fit_single_estimator(self.final_estimator_, X_meta, y, fit_params=fit_params)
279  
280          return self
281  
282      @property
283      def n_features_in_(self):
284          """Number of features seen during :term:`fit`."""
285          try:
286              check_is_fitted(self)
287          except NotFittedError as nfe:
288              raise AttributeError(
289                  f"{self.__class__.__name__} object has no attribute n_features_in_"
290              ) from nfe
291          return self.estimators_[0].n_features_in_
292  
293      def _transform(self, X):
294          """Concatenate and return the predictions of the estimators."""
295          check_is_fitted(self)
296          predictions = [
297              getattr(est, meth)(X)
298              for est, meth in zip(self.estimators_, self.stack_method_)
299              if est != "drop"
300          ]
301          return self._concatenate_predictions(X, predictions)

303      def get_feature_names_out(self, input_features=None):
304          """Get output feature names for transformation. 
305   
306          Parameters 
307          ---------- 
308          input_features : array-like of str or None, default=None 
309              Input features. The input feature names are only used when `passthrough` is 
310              `True`. 
311   
312              - If `input_features` is `None`, then `feature_names_in_` is 
313                used as feature names in. If `feature_names_in_` is not defined, 
314                then names are generated: `[x0, x1, ..., x(n_features_in_ - 1)]`. 
315              - If `input_features` is an array-like, then `input_features` must 
316                match `feature_names_in_` if `feature_names_in_` is defined. 
317   
318              If `passthrough` is `False`, then only the names of `estimators` are used 
319              to generate the output feature names. 
320   
321          Returns 
322          ------- 
323          feature_names_out : ndarray of str objects 
324              Transformed feature names. 
325          """
326          check_is_fitted(self, "n_features_in_")
327          input_features = _check_feature_names_in(
328              self, input_features, generate_names=self.passthrough
329          )
330  
331          class_name = self.__class__.__name__.lower()
332          non_dropped_estimators = (
333              name for name, est in self.estimators if est != "drop"
334          )
335          meta_names = []
336          for est, n_features_out in zip(non_dropped_estimators, self._n_feature_outs):
337              if n_features_out == 1:
338                  meta_names.append(f"{class_name}_{est}")
339              else:
340                  meta_names.extend(
341                      f"{class_name}_{est}{i}" for i in range(n_features_out)
342                  )
343  
344          if self.passthrough:
345              return np.concatenate((meta_names, input_features))
346  
347          return np.asarray(meta_names, dtype=object)

349      @available_if(
350          _estimator_has("predict", delegates=("final_estimator_", "final_estimator"))
351      )
352      def predict(self, X, **predict_params):
353          """Predict target for X. 
354   
355          Parameters 
356          ---------- 
357          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
358              Training vectors, where `n_samples` is the number of samples and 
359              `n_features` is the number of features. 
360   
361          **predict_params : dict of str -> obj 
362              Parameters to the `predict` called by the `final_estimator`. Note 
363              that this may be used to return uncertainties from some estimators 
364              with `return_std` or `return_cov`. Be aware that it will only 
365              account for uncertainty in the final estimator. 
366   
367          Returns 
368          ------- 
369          y_pred : ndarray of shape (n_samples,) or (n_samples, n_output) 
370              Predicted targets. 
371          """
372  
373          check_is_fitted(self)
374          return self.final_estimator_.predict(self.transform(X), **predict_params)

387      def get_metadata_routing(self):
388          """Get metadata routing of this object. 
389   
390          Please check :ref:`User Guide <metadata_routing>` on how the routing 
391          mechanism works. 
392   
393          .. versionadded:: 1.6 
394   
395          Returns 
396          ------- 
397          routing : MetadataRouter 
398              A :class:`~sklearn.utils.metadata_routing.MetadataRouter` encapsulating 
399              routing information. 
400          """
401          router = MetadataRouter(owner=self.__class__.__name__)
402  
403          # `self.estimators` is a list of (name, est) tuples
404          for name, estimator in self.estimators:
405              router.add(
406                  **{name: estimator},
407                  method_mapping=MethodMapping().add(callee="fit", caller="fit"),
408              )
409  
410          try:
411              final_estimator_ = self.final_estimator_
412          except AttributeError:
413              final_estimator_ = self.final_estimator
414  
415          router.add(
416              final_estimator_=final_estimator_,
417              method_mapping=MethodMapping().add(caller="predict", callee="predict"),
418          )
419  
420          return router

423  class StackingClassifier(ClassifierMixin, _BaseStacking):
424      """Stack of estimators with a final classifier. 
425   
426      Stacked generalization consists in stacking the output of individual 
427      estimator and use a classifier to compute the final prediction. Stacking 
428      allows to use the strength of each individual estimator by using their 
429      output as input of a final estimator. 
430   
431      Note that `estimators_` are fitted on the full `X` while `final_estimator_` 
432      is trained using cross-validated predictions of the base estimators using 
433      `cross_val_predict`. 
434   
435      Read more in the :ref:`User Guide <stacking>`. 
436   
437      .. versionadded:: 0.22 
438   
439      Parameters 
440      ---------- 
441      estimators : list of (str, estimator) 
442          Base estimators which will be stacked together. Each element of the 
443          list is defined as a tuple of string (i.e. name) and an estimator 
444          instance. An estimator can be set to 'drop' using `set_params`. 
445   
446          The type of estimator is generally expected to be a classifier. 
447          However, one can pass a regressor for some use case (e.g. ordinal 
448          regression). 
449   
450      final_estimator : estimator, default=None 
451          A classifier which will be used to combine the base estimators. 
452          The default classifier is a 
453          :class:`~sklearn.linear_model.LogisticRegression`. 
454   
455      cv : int, cross-validation generator, iterable, or "prefit", default=None 
456          Determines the cross-validation splitting strategy used in 
457          `cross_val_predict` to train `final_estimator`. Possible inputs for 
458          cv are: 
459   
460          * None, to use the default 5-fold cross validation, 
461          * integer, to specify the number of folds in a (Stratified) KFold, 
462          * An object to be used as a cross-validation generator, 
463          * An iterable yielding train, test splits, 
464          * `"prefit"`, to assume the `estimators` are prefit. In this case, the 
465            estimators will not be refitted. 
466   
467          For integer/None inputs, if the estimator is a classifier and y is 
468          either binary or multiclass, 
469          :class:`~sklearn.model_selection.StratifiedKFold` is used. 
470          In all other cases, :class:`~sklearn.model_selection.KFold` is used. 
471          These splitters are instantiated with `shuffle=False` so the splits 
472          will be the same across calls. 
473   
474          Refer :ref:`User Guide <cross_validation>` for the various 
475          cross-validation strategies that can be used here. 
476   
477          If "prefit" is passed, it is assumed that all `estimators` have 
478          been fitted already. The `final_estimator_` is trained on the `estimators` 
479          predictions on the full training set and are **not** cross validated 
480          predictions. Please note that if the models have been trained on the same 
481          data to train the stacking model, there is a very high risk of overfitting. 
482   
483          .. versionadded:: 1.1 
484              The 'prefit' option was added in 1.1 
485   
486          .. note:: 
487             A larger number of split will provide no benefits if the number 
488             of training samples is large enough. Indeed, the training time 
489             will increase. ``cv`` is not used for model evaluation but for 
490             prediction. 
491   
492      stack_method : {'auto', 'predict_proba', 'decision_function', 'predict'}, \ 
493              default='auto' 
494          Methods called for each base estimator. It can be: 
495   
496          * if 'auto', it will try to invoke, for each estimator, 
497            `'predict_proba'`, `'decision_function'` or `'predict'` in that 
498            order. 
499          * otherwise, one of `'predict_proba'`, `'decision_function'` or 
500            `'predict'`. If the method is not implemented by the estimator, it 
501            will raise an error. 
502   
503      n_jobs : int, default=None 
504          The number of jobs to run in parallel for `fit` of all `estimators`. 
505          `None` means 1 unless in a `joblib.parallel_backend` context. -1 means 
506          using all processors. See :term:`Glossary <n_jobs>` for more details. 
507   
508      passthrough : bool, default=False 
509          When False, only the predictions of estimators will be used as 
510          training data for `final_estimator`. When True, the 
511          `final_estimator` is trained on the predictions as well as the 
512          original training data. 
513   
514      verbose : int, default=0 
515          Verbosity level. 
516   
517      Attributes 
518      ---------- 
519      classes_ : ndarray of shape (n_classes,) or list of ndarray if `y` \ 
520          is of type `"multilabel-indicator"`. 
521          Class labels. 
522   
523      estimators_ : list of estimators 
524          The elements of the `estimators` parameter, having been fitted on the 
525          training data. If an estimator has been set to `'drop'`, it 
526          will not appear in `estimators_`. When `cv="prefit"`, `estimators_` 
527          is set to `estimators` and is not fitted again. 
528   
529      named_estimators_ : :class:`~sklearn.utils.Bunch` 
530          Attribute to access any fitted sub-estimators by name. 
531   
532      n_features_in_ : int 
533          Number of features seen during :term:`fit`. Only defined if the 
534          underlying estimator exposes such an attribute when fit. 
535   
536          .. versionadded:: 0.24 
537   
538      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
539          Names of features seen during :term:`fit`. Only defined if the 
540          underlying estimators expose such an attribute when fit. 
541   
542          .. versionadded:: 1.0 
543   
544      final_estimator_ : estimator 
545          The classifier fit on the output of `estimators_` and responsible for 
546          final predictions. 
547   
548      stack_method_ : list of str 
549          The method used by each base estimator. 
550   
551      See Also 
552      -------- 
553      StackingRegressor : Stack of estimators with a final regressor. 
554   
555      Notes 
556      ----- 
557      When `predict_proba` is used by each estimator (i.e. most of the time for 
558      `stack_method='auto'` or specifically for `stack_method='predict_proba'`), 
559      the first column predicted by each estimator will be dropped in the case 
560      of a binary classification problem. Indeed, both feature will be perfectly 
561      collinear. 
562   
563      In some cases (e.g. ordinal regression), one can pass regressors as the 
564      first layer of the :class:`StackingClassifier`. However, note that `y` will 
565      be internally encoded in a numerically increasing order or lexicographic 
566      order. If this ordering is not adequate, one should manually numerically 
567      encode the classes in the desired order. 
568   
569      References 
570      ---------- 
571      .. [1] Wolpert, David H. "Stacked generalization." Neural networks 5.2 
572         (1992): 241-259. 
573   
574      Examples 
575      -------- 
576      >>> from sklearn.datasets import load_iris 
577      >>> from sklearn.ensemble import RandomForestClassifier 
578      >>> from sklearn.svm import LinearSVC 
579      >>> from sklearn.linear_model import LogisticRegression 
580      >>> from sklearn.preprocessing import StandardScaler 
581      >>> from sklearn.pipeline import make_pipeline 
582      >>> from sklearn.ensemble import StackingClassifier 
583      >>> X, y = load_iris(return_X_y=True) 
584      >>> estimators = [ 
585      ...     ('rf', RandomForestClassifier(n_estimators=10, random_state=42)), 
586      ...     ('svr', make_pipeline(StandardScaler(), 
587      ...                           LinearSVC(random_state=42))) 
588      ... ] 
589      >>> clf = StackingClassifier( 
590      ...     estimators=estimators, final_estimator=LogisticRegression() 
591      ... ) 
592      >>> from sklearn.model_selection import train_test_split 
593      >>> X_train, X_test, y_train, y_test = train_test_split( 
594      ...     X, y, stratify=y, random_state=42 
595      ... ) 
596      >>> clf.fit(X_train, y_train).score(X_test, y_test) 
597      0.9... 
598      """
599

629          self._clone_final_estimator(default=LogisticRegression())
630          if not is_classifier(self.final_estimator_):

632                  "'final_estimator' parameter should be a classifier. Got {}".format(

637      def _validate_estimators(self):
638          """Overload the method of `_BaseHeterogeneousEnsemble` to be more 
639          lenient towards the type of `estimators`. 
640   
641          Regressors can be accepted for some cases such as ordinal regression. 
642          """
643          if len(self.estimators) == 0:
644              raise ValueError(
645                  "Invalid 'estimators' attribute, 'estimators' should be a "
646                  "non-empty list of (string, estimator) tuples."

667          Parameters 
668          ---------- 
669          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
670              Training vectors, where `n_samples` is the number of samples and 
671              `n_features` is the number of features.

673          y : array-like of shape (n_samples,) 
674              Target values. Note that `y` will be internally encoded in 
675              numerically increasing order or lexicographic order. If the order 
676              matter (e.g. for ordinal regression), one should numerically encode 
677              the target `y` before calling :term:`fit`. 
678   
679          sample_weight : array-like of shape (n_samples,), default=None 
680              Sample weights. If None, then samples are equally weighted. 
681              Note that this is supported only if all underlying estimators 
682              support sample weights.

692                  more details. 
693   
694          Returns 
695          ------- 
696          self : object 
697              Returns a fitted instance of estimator.

700          check_classification_targets(y)
701          if type_of_target(y) == "multilabel-indicator":
702              self._label_encoder = [LabelEncoder().fit(yk) for yk in y.T]
703              self.classes_ = [le.classes_ for le in self._label_encoder]
704              y_encoded = np.array(
705                  [
706                      self._label_encoder[target_idx].transform(target)
707                      for target_idx, target in enumerate(y.T)
708                  ]
709              ).T
710          else:
711              self._label_encoder = LabelEncoder().fit(y)
712              self.classes_ = self._label_encoder.classes_
713              y_encoded = self._label_encoder.transform(y)
714  
715          if sample_weight is not None:
716              fit_params["sample_weight"] = sample_weight
717          return super().fit(X, y_encoded, **fit_params)

755          else:

757              routed_params = Bunch()

770          else:

852  class StackingRegressor(RegressorMixin, _BaseStacking):
853      """Stack of estimators with a final regressor. 
854   
855      Stacked generalization consists in stacking the output of individual 
856      estimator and use a regressor to compute the final prediction. Stacking 
857      allows to use the strength of each individual estimator by using their 
858      output as input of a final estimator. 
859   
860      Note that `estimators_` are fitted on the full `X` while `final_estimator_` 
861      is trained using cross-validated predictions of the base estimators using 
862      `cross_val_predict`. 
863   
864      Read more in the :ref:`User Guide <stacking>`. 
865   
866      .. versionadded:: 0.22 
867   
868      Parameters 
869      ---------- 
870      estimators : list of (str, estimator) 
871          Base estimators which will be stacked together. Each element of the 
872          list is defined as a tuple of string (i.e. name) and an estimator 
873          instance. An estimator can be set to 'drop' using `set_params`. 
874   
875      final_estimator : estimator, default=None 
876          A regressor which will be used to combine the base estimators. 
877          The default regressor is a :class:`~sklearn.linear_model.RidgeCV`. 
878   
879      cv : int, cross-validation generator, iterable, or "prefit", default=None 
880          Determines the cross-validation splitting strategy used in 
881          `cross_val_predict` to train `final_estimator`. Possible inputs for 
882          cv are: 
883   
884          * None, to use the default 5-fold cross validation, 
885          * integer, to specify the number of folds in a (Stratified) KFold, 
886          * An object to be used as a cross-validation generator, 
887          * An iterable yielding train, test splits, 
888          * `"prefit"`, to assume the `estimators` are prefit. In this case, the 
889            estimators will not be refitted. 
890   
891          For integer/None inputs, if the estimator is a classifier and y is 
892          either binary or multiclass, 
893          :class:`~sklearn.model_selection.StratifiedKFold` is used. 
894          In all other cases, :class:`~sklearn.model_selection.KFold` is used. 
895          These splitters are instantiated with `shuffle=False` so the splits 
896          will be the same across calls. 
897   
898          Refer :ref:`User Guide <cross_validation>` for the various 
899          cross-validation strategies that can be used here. 
900   
901          If "prefit" is passed, it is assumed that all `estimators` have 
902          been fitted already. The `final_estimator_` is trained on the `estimators` 
903          predictions on the full training set and are **not** cross validated 
904          predictions. Please note that if the models have been trained on the same 
905          data to train the stacking model, there is a very high risk of overfitting. 
906   
907          .. versionadded:: 1.1 
908              The 'prefit' option was added in 1.1 
909   
910          .. note:: 
911             A larger number of split will provide no benefits if the number 
912             of training samples is large enough. Indeed, the training time 
913             will increase. ``cv`` is not used for model evaluation but for 
914             prediction. 
915   
916      n_jobs : int, default=None 
917          The number of jobs to run in parallel for `fit` of all `estimators`. 
918          `None` means 1 unless in a `joblib.parallel_backend` context. -1 means 
919          using all processors. See :term:`Glossary <n_jobs>` for more details. 
920   
921      passthrough : bool, default=False 
922          When False, only the predictions of estimators will be used as 
923          training data for `final_estimator`. When True, the 
924          `final_estimator` is trained on the predictions as well as the 
925          original training data. 
926   
927      verbose : int, default=0 
928          Verbosity level. 
929   
930      Attributes 
931      ---------- 
932      estimators_ : list of estimators 
933          The elements of the `estimators` parameter, having been fitted on the 
934          training data. If an estimator has been set to `'drop'`, it 
935          will not appear in `estimators_`. When `cv="prefit"`, `estimators_` 
936          is set to `estimators` and is not fitted again. 
937   
938      named_estimators_ : :class:`~sklearn.utils.Bunch` 
939          Attribute to access any fitted sub-estimators by name. 
940   
941      n_features_in_ : int 
942          Number of features seen during :term:`fit`. Only defined if the 
943          underlying estimator exposes such an attribute when fit. 
944   
945          .. versionadded:: 0.24 
946   
947      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
948          Names of features seen during :term:`fit`. Only defined if the 
949          underlying estimators expose such an attribute when fit. 
950   
951          .. versionadded:: 1.0 
952   
953      final_estimator_ : estimator 
954          The regressor fit on the output of `estimators_` and responsible for 
955          final predictions. 
956   
957      stack_method_ : list of str 
958          The method used by each base estimator. 
959   
960      See Also 
961      -------- 
962      StackingClassifier : Stack of estimators with a final classifier. 
963   
964      References 
965      ---------- 
966      .. [1] Wolpert, David H. "Stacked generalization." Neural networks 5.2 
967         (1992): 241-259. 
968   
969      Examples 
970      -------- 
971      >>> from sklearn.datasets import load_diabetes 
972      >>> from sklearn.linear_model import RidgeCV 
973      >>> from sklearn.svm import LinearSVR 
974      >>> from sklearn.ensemble import RandomForestRegressor 
975      >>> from sklearn.ensemble import StackingRegressor 
976      >>> X, y = load_diabetes(return_X_y=True) 
977      >>> estimators = [ 
978      ...     ('lr', RidgeCV()), 
979      ...     ('svr', LinearSVR(random_state=42)) 
980      ... ] 
981      >>> reg = StackingRegressor( 
982      ...     estimators=estimators, 
983      ...     final_estimator=RandomForestRegressor(n_estimators=10, 
984      ...                                           random_state=42) 
985      ... ) 
986      >>> from sklearn.model_selection import train_test_split 
987      >>> X_train, X_test, y_train, y_test = train_test_split( 
988      ...     X, y, random_state=42 
989      ... ) 
990      >>> reg.fit(X_train, y_train).score(X_test, y_test) 
991      0.3... 
992      """
