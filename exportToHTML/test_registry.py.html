6    import logging

16   class TestAppHandlerRegistry(AppHandlerTest):
17       def test_node_not_available(self):
18           # patch should be applied on `jupyterlab.commands` and not on `jupyterlab_server.process`
19           # See https://docs.python.org/3/library/unittest.mock.html#where-to-patch
20           with patch("jupyterlab.commands.which") as which:
21               which.side_effect = ValueError("Command not found")
22   
23               logger = logging.getLogger("jupyterlab")
24               config = commands._yarn_config(logger)
25   
26               which.assert_called_once_with("node")
27               self.assertDictEqual(config, {"yarn config": {}, "npm config": {}})
28   
29       def test_yarn_config(self):
30           with patch("subprocess.check_output") as check_output:
31               yarn_registry = "https://private.yarn/manager"
32               check_output.return_value = b"\n".join(
33                   [
34                       b'{"type":"info","data":"yarn config"}',
35                       b'{"type":"inspect","data":{"registry":"'
36                       + bytes(yarn_registry, "utf-8")
37                       + b'"}}',
38                       b'{"type":"info","data":"npm config"}',
39                       b'{"type":"inspect","data":{"registry":"'
40                       + bytes(yarn_registry, "utf-8")
41                       + b'"}}',
42                   ]
43               )
44               logger = logging.getLogger("jupyterlab")
45               config = commands._yarn_config(logger)
46   
47               self.assertDictEqual(
48                   config,
49                   {
50                       "yarn config": {"registry": yarn_registry},
51                       "npm config": {"registry": yarn_registry},
52                   },
53               )
54   
55       def test_yarn_config_failure(self):
56           with patch("subprocess.check_output") as check_output:
57               check_output.side_effect = subprocess.CalledProcessError(
58                   1, ["yarn", "config", "list"], b"", stderr=b"yarn config failed."
59               )
60   
61               logger = logging.getLogger("jupyterlab")
62               config = commands._yarn_config(logger)
63   
64               self.assertDictEqual(config, {"yarn config": {}, "npm config": {}})

86       def test_populate_staging(self):
87           with patch("subprocess.check_output") as check_output:
88               yarn_registry = "https://private.yarn/manager"
89               check_output.return_value = b"\n".join(
90                   [
91                       b'{"type":"info","data":"yarn config"}',
92                       b'{"type":"inspect","data":{"registry":"'
93                       + bytes(yarn_registry, "utf-8")
94                       + b'"}}',
95                       b'{"type":"info","data":"npm config"}',
96                       b'{"type":"inspect","data":{"registry":"'
97                       + bytes(yarn_registry, "utf-8")
98                       + b'"}}',
99                   ]
100              )
101  
102              staging = pjoin(self.app_dir, "staging")
103              handler = commands._AppHandler(commands.AppOptions())
104              handler._populate_staging()
105  
106              lock_path = pjoin(staging, "yarn.lock")
107              with open(lock_path) as f:
108                  lock = f.read()
109  
110              # yarn >=2.x does not record the registry in the lockfile
111              self.assertNotIn(commands.YARN_DEFAULT_REGISTRY, lock)
112              self.assertNotIn(yarn_registry, lock)
