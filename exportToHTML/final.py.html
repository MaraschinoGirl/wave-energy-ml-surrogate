16   from types import FunctionType
17   from typing import Callable, TypeVar, Union
18   
19   _WrappedMethod = TypeVar("_WrappedMethod", bound=Union[FunctionType, Callable])
20   
21   
22   def final(method: _WrappedMethod) -> _WrappedMethod:
23       """Decorator to indicate that the decorated method is finalized and cannot be overridden. 
24       The decorator code is executed while loading class. Using this method 
25       should have minimal runtime performance implications. 
26       Currently, only methods with @override are checked. 
27    
28       How to use: 
29       from overrides import final 
30    
31       class SuperClass(object): 
32           @final 
33           def method(self): 
34             return 2 
35    
36       class SubClass(SuperClass): 
37           @override 
38           def method(self): #causes an error 
39               return 1 
40    
41       :raises AssertionError: if there exists a match in sub classes for the method name 
42       :return: method 
43       """
44       setattr(method, "__final__", True)
45       return method
