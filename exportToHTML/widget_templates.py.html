34   @doc_subst(_doc_snippets)
35   class LayoutProperties(HasTraits):
36       """Mixin class for layout templates 
37    
38       This class handles mainly style attributes (height, grid_gap etc.) 
39    
40       Parameters 
41       ---------- 
42    
43       {style_params} 
44    
45    
46       Note 
47       ---- 
48    
49       This class is only meant to be used in inheritance as mixin with other 
50       classes. It will not work, unless `self.layout` attribute is defined. 
51    
52       """
53   
54       # style attributes (passed to Layout)
55       grid_gap = Unicode(
56           None,
57           allow_none=True,
58           help="The grid-gap CSS attribute.")
59       justify_content = CaselessStrEnum(
60           ['flex-start', 'flex-end', 'center',
61            'space-between', 'space-around'],
62           allow_none=True,
63           help="The justify-content CSS attribute.")
64       align_items = CaselessStrEnum(
65           ['top', 'bottom',
66            'flex-start', 'flex-end', 'center',
67            'baseline', 'stretch'],
68           allow_none=True, help="The align-items CSS attribute.")
69       width = Unicode(
70           None,
71           allow_none=True,
72           help="The width CSS attribute.")
73       height = Unicode(
74           None,
75           allow_none=True,
76           help="The width CSS attribute.")
77   
78   
79       def __init__(self, **kwargs):
80           super().__init__(**kwargs)
81           self._property_rewrite = defaultdict(dict)
82           self._property_rewrite['align_items'] = {'top': 'flex-start',
83                                                    'bottom': 'flex-end'}
84           self._copy_layout_props()
85           self._set_observers()
86   
87       def _delegate_to_layout(self, change):
88           "delegate the trait types to their counterparts in self.layout"
89           value, name = change['new'], change['name']
90           value = self._property_rewrite[name].get(value, value)
91           setattr(self.layout, name, value) # pylint: disable=no-member
92   
93       def _set_observers(self):
94           "set observers on all layout properties defined in this class"
95           _props = LayoutProperties.class_trait_names()
96           self.observe(self._delegate_to_layout, _props)
97   
98       def _copy_layout_props(self):
99   
100          _props = LayoutProperties.class_trait_names()
101  
102          for prop in _props:
103              value = getattr(self, prop)
104              if value:
105                  value = self._property_rewrite[prop].get(value, value)
106                  setattr(self.layout, prop, value) #pylint: disable=no-member

108  @doc_subst(_doc_snippets)
109  class AppLayout(GridBox, LayoutProperties):
110      """ Define an application like layout of widgets. 
111   
112      Parameters 
113      ---------- 
114   
115      header: instance of Widget 
116      left_sidebar: instance of Widget 
117      center: instance of Widget 
118      right_sidebar: instance of Widget 
119      footer: instance of Widget 
120          widgets to fill the positions in the layout 
121   
122      merge: bool 
123          flag to say whether the empty positions should be automatically merged 
124   
125      pane_widths: list of numbers/strings 
126          the fraction of the total layout width each of the central panes should occupy 
127          (left_sidebar, 
128          center, right_sidebar) 
129   
130      pane_heights: list of numbers/strings 
131          the fraction of the width the vertical space that the panes should occupy 
132           (left_sidebar, center, right_sidebar) 
133   
134      {style_params} 
135   
136      Examples 
137      -------- 
138   
139      """
140  
141      # widget positions
142      header = Instance(Widget, allow_none=True)
143      footer = Instance(Widget, allow_none=True)
144      left_sidebar = Instance(Widget, allow_none=True)
145      right_sidebar = Instance(Widget, allow_none=True)
146      center = Instance(Widget, allow_none=True)
147  
148      # extra args
149      pane_widths = Tuple(CUnicode(), CUnicode(), CUnicode(),
150                          default_value=['1fr', '2fr', '1fr'])
151      pane_heights = Tuple(CUnicode(), CUnicode(), CUnicode(),
152                           default_value=['1fr', '3fr', '1fr'])
153  
154      merge = Bool(default_value=True)
155  
156      def __init__(self, **kwargs):
157          super().__init__(**kwargs)
158          self._update_layout()
159  
160      @staticmethod
161      def _size_to_css(size):
162          if re.match(r'\d+\.?\d*(px|fr|%)$', size):
163              return size
164          if re.match(r'\d+\.?\d*$', size):
165              return size + 'fr'
166  
167          raise TypeError("the pane sizes must be in one of the following formats: "
168                          "'10px', '10fr', 10 (will be converted to '10fr')."
169                          "Got '{}'".format(size))
170  
171      def _convert_sizes(self, size_list):
172          return list(map(self._size_to_css, size_list))
173  
174      def _update_layout(self):
175  
176          grid_template_areas = [["header", "header", "header"],
177                                 ["left-sidebar", "center", "right-sidebar"],
178                                 ["footer", "footer", "footer"]]
179  
180          grid_template_columns = self._convert_sizes(self.pane_widths)
181          grid_template_rows = self._convert_sizes(self.pane_heights)
182  
183          all_children = {'header': self.header,
184                          'footer': self.footer,
185                          'left-sidebar': self.left_sidebar,
186                          'right-sidebar': self.right_sidebar,
187                          'center': self.center}
188  
189          children = {position : child
190                      for position, child in all_children.items()
191                      if child is not None}
192  
193          if not children:
194              return
195  
196          for position, child in children.items():
197              child.layout.grid_area = position
198  
199          if self.merge:
200  
201              if len(children) == 1:
202                  position = list(children.keys())[0]
203                  grid_template_areas = [[position, position, position],
204                                         [position, position, position],
205                                         [position, position, position]]
206  
207              else:
208                  if self.center is None:
209                      for row in grid_template_areas:
210                          del row[1]
211                      del grid_template_columns[1]
212  
213                  if self.left_sidebar is None:
214                      grid_template_areas[1][0] = grid_template_areas[1][1]
215  
216                  if self.right_sidebar is None:
217                      grid_template_areas[1][-1] = grid_template_areas[1][-2]
218  
219                  if (self.left_sidebar is None and
220                          self.right_sidebar is None and
221                          self.center is None):
222                      grid_template_areas = [['header'], ['footer']]
223                      grid_template_columns = ['1fr']
224                      grid_template_rows = ['1fr', '1fr']
225  
226                  if self.header is None:
227                      del grid_template_areas[0]
228                      del grid_template_rows[0]
229  
230                  if self.footer is None:
231                      del grid_template_areas[-1]
232                      del grid_template_rows[-1]
233  
234  
235          grid_template_areas_css = "\n".join('"{}"'.format(" ".join(line))
236                                              for line in grid_template_areas)
237  
238          self.layout.grid_template_columns = " ".join(grid_template_columns)
239          self.layout.grid_template_rows = " ".join(grid_template_rows)
240          self.layout.grid_template_areas = grid_template_areas_css
241  
242          self.children = tuple(children.values())
243  
244      @observe("footer", "header", "center", "left_sidebar", "right_sidebar", "merge",
245               "pane_widths", "pane_heights")
246      def _child_changed(self, change): #pylint: disable=unused-argument
247          self._update_layout()

251  class GridspecLayout(GridBox, LayoutProperties):
252      """ Define a N by M grid layout 
253   
254      Parameters 
255      ---------- 
256   
257      n_rows : int 
258          number of rows in the grid 
259   
260      n_columns : int 
261          number of columns in the grid 
262   
263      {style_params} 
264   
265      Examples 
266      -------- 
267   
268      >>> from ipywidgets import GridspecLayout, Button, Layout 
269      >>> layout = GridspecLayout(n_rows=4, n_columns=2, height='200px') 
270      >>> layout[:3, 0] = Button(layout=Layout(height='auto', width='auto')) 
271      >>> layout[1:, 1] = Button(layout=Layout(height='auto', width='auto')) 
272      >>> layout[-1, 0] = Button(layout=Layout(height='auto', width='auto')) 
273      >>> layout[0, 1] = Button(layout=Layout(height='auto', width='auto')) 
274      >>> layout 
275      """
276  
277      n_rows = Integer()
278      n_columns = Integer()
279  
280      def __init__(self, n_rows=None, n_columns=None, **kwargs):
281          super().__init__(**kwargs)
282          self.n_rows = n_rows
283          self.n_columns = n_columns
284          self._grid_template_areas = [['.'] * self.n_columns for i in range(self.n_rows)]
285  
286          self._grid_template_rows = 'repeat(%d, 1fr)' % (self.n_rows,)
287          self._grid_template_columns = 'repeat(%d, 1fr)' % (self.n_columns,)
288          self._children = {}
289          self._id_count = 0
290  
291      @validate('n_rows', 'n_columns')
292      def _validate_integer(self, proposal):
293          if proposal['value'] > 0:
294              return proposal['value']
295          raise TraitError('n_rows and n_columns must be positive integer')
296  
297      def _get_indices_from_slice(self, row, column):
298          "convert a two-dimensional slice to a list of rows and column indices"
299  
300          if isinstance(row, slice):
301              start, stop, stride = row.indices(self.n_rows)
302              rows = range(start, stop, stride)
303          else:
304              rows = [row]
305  
306          if isinstance(column, slice):
307              start, stop, stride = column.indices(self.n_columns)
308              columns = range(start, stop, stride)
309          else:
310              columns = [column]
311  
312          return rows, columns
313  
314      def __setitem__(self, key, value):
315          row, column = key
316          self._id_count += 1
317          obj_id = 'widget%03d' % self._id_count
318          value.layout.grid_area = obj_id
319  
320          rows, columns = self._get_indices_from_slice(row, column)
321  
322          for row in rows:
323              for column in columns:
324                  current_value = self._grid_template_areas[row][column]
325                  if current_value != '.' and current_value in self._children:
326                      del self._children[current_value]
327                  self._grid_template_areas[row][column] = obj_id
328  
329          self._children[obj_id] = value
330          self._update_layout()
331  
332      def __getitem__(self, key):
333          rows, columns = self._get_indices_from_slice(*key)
334  
335          obj_id = None
336          for row in rows:
337              for column in columns:
338                  new_obj_id = self._grid_template_areas[row][column]
339                  obj_id = obj_id or new_obj_id
340                  if obj_id != new_obj_id:
341                      raise TypeError('The slice spans several widgets, but '
342                                      'only a single widget can be retrieved '
343                                      'at a time')
344  
345          return self._children[obj_id]
346  
347      def _update_layout(self):
348  
349          grid_template_areas_css = "\n".join('"{}"'.format(" ".join(line))
350                                              for line in self._grid_template_areas)
351  
352          self.layout.grid_template_columns = self._grid_template_columns
353          self.layout.grid_template_rows = self._grid_template_rows
354          self.layout.grid_template_areas = grid_template_areas_css
355          self.children = tuple(self._children.values())

359  class TwoByTwoLayout(GridBox, LayoutProperties):
360      """ Define a layout with 2x2 regular grid. 
361   
362      Parameters 
363      ---------- 
364   
365      top_left: instance of Widget 
366      top_right: instance of Widget 
367      bottom_left: instance of Widget 
368      bottom_right: instance of Widget 
369          widgets to fill the positions in the layout 
370   
371      merge: bool 
372          flag to say whether the empty positions should be automatically merged 
373   
374      {style_params} 
375   
376      Examples 
377      -------- 
378   
379      >>> from ipywidgets import TwoByTwoLayout, Button 
380      >>> TwoByTwoLayout(top_left=Button(description="Top left"), 
381      ...                top_right=Button(description="Top right"), 
382      ...                bottom_left=Button(description="Bottom left"), 
383      ...                bottom_right=Button(description="Bottom right")) 
384   
385      """
386  
387      # widget positions
388      top_left = Instance(Widget, allow_none=True)
389      top_right = Instance(Widget, allow_none=True)
390      bottom_left = Instance(Widget, allow_none=True)
391      bottom_right = Instance(Widget, allow_none=True)
392  
393      # extra args
394      merge = Bool(default_value=True)
395  
396      def __init__(self, **kwargs):
397          super().__init__(**kwargs)
398          self._update_layout()
399  
400      def _update_layout(self):
401  
402  
403          grid_template_areas = [["top-left", "top-right"],
404                                 ["bottom-left", "bottom-right"]]
405  
406          all_children = {'top-left' : self.top_left,
407                          'top-right' : self.top_right,
408                          'bottom-left' : self.bottom_left,
409                          'bottom-right' : self.bottom_right}
410  
411          children = {position : child
412                      for position, child in all_children.items()
413                      if child is not None}
414  
415          if not children:
416              return
417  
418          for position, child in children.items():
419              child.layout.grid_area = position
420  
421          if self.merge:
422  
423              if len(children) == 1:
424                  position = list(children.keys())[0]
425                  grid_template_areas = [[position, position],
426                                         [position, position]]
427              else:
428                  columns = ['left', 'right']
429                  for i, column in enumerate(columns):
430                      top, bottom = children.get('top-' + column), children.get('bottom-' + column)
431                      i_neighbour = (i + 1) % 2
432                      if top is None and bottom is None:
433                          # merge each cell in this column with the neighbour on the same row
434                          grid_template_areas[0][i] = grid_template_areas[0][i_neighbour]
435                          grid_template_areas[1][i] = grid_template_areas[1][i_neighbour]
436                      elif top is None:
437                          # merge with the cell below
438                          grid_template_areas[0][i] = grid_template_areas[1][i]
439                      elif bottom is None:
440                          # merge with the cell above
441                          grid_template_areas[1][i] = grid_template_areas[0][i]
442  
443          grid_template_areas_css = "\n".join('"{}"'.format(" ".join(line))
444                                              for line in grid_template_areas)
445  
446          self.layout.grid_template_columns = '1fr 1fr'
447          self.layout.grid_template_rows = '1fr 1fr'
448          self.layout.grid_template_areas = grid_template_areas_css
449  
450          self.children = tuple(children.values())
451  
452      @observe("top_left", "bottom_left", "top_right", "bottom_right", "merge")
453      def _child_changed(self, change): #pylint: disable=unused-argument
454          self._update_layout()
