51   def get_params(signature):
52       in_, out = signature.split("->")
53       in_params = []
54       out_params = []
55       for i, typecode in enumerate(in_):
56           in_params.append(f"{dtype_map[typecode]} in{i}")
57       for i, typecode in enumerate(out):
58           out_params.append(f"{dtype_map[typecode]} out{i}")
59       in_params = ", ".join(in_params)
60       out_params = ", ".join(out_params)
61       return in_params, out_params

64   def get_sample_input(signature, xp):
65       dtype_map = {

67           "d": xp.float64,
68           "F": xp.complex64,
69           "D": xp.complex128,

71           "l": xp.int64,
72       }
73   
74       in_, _ = signature.split("->")
75       args = []
76       for typecode in in_:
77           args.append(xp.zeros(2, dtype=dtype_map[typecode]))
78       return args
