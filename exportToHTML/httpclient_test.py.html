142      def get(self) -> None:
143          # set Content-Encoding manually to avoid automatic gzip encoding
144          self.set_header("Content-Type", "text/plain")
145          self.set_header("Content-Encoding", "gzip")
146          # Triggering the potential bug seems to depend on input length.
147          # This length is taken from the bad-response example reported in
148          # https://github.com/tornadoweb/tornado/pull/2875 (uncompressed).
149          text = "".join("Hello World {}\n".format(i) for i in range(9000))[:149051]
150          body = gzip.compress(text.encode(), compresslevel=6) + b"\00"
151          self.write(body)

230          # over several ioloop iterations, but the connection is already closed.
231          sock, port = bind_unused_port()
232          with closing(sock):

236                  # fake an HTTP server using chunked encoding where the final chunks

247  1 
248  1

347      def test_method_after_redirect(self):
348          # Legacy redirect codes (301, 302) convert POST requests to GET.
349          for status in [301, 302, 303]:
350              url = "/redirect?url=/all_methods&status=%d" % status
351              resp = self.fetch(url, method="POST", body=b"")
352              self.assertEqual(b"GET", resp.body)
353  
354              # Other methods are left alone, except for 303 redirect, depending on client
355              for method in ["GET", "OPTIONS", "PUT", "DELETE"]:
356                  resp = self.fetch(url, method=method, allow_nonstandard_methods=True)
357                  if status in [301, 302]:
358                      self.assertEqual(utf8(method), resp.body)
359                  else:
360                      self.assertIn(resp.body, [utf8(method), b"GET"])
361  
362              # HEAD is different so check it separately.
363              resp = self.fetch(url, method="HEAD")
364              self.assertEqual(200, resp.code)
365              self.assertEqual(b"", resp.body)
366  
367          # Newer redirects always preserve the original method.
368          for status in [307, 308]:
369              url = "/redirect?url=/all_methods&status=307"
370              for method in ["GET", "OPTIONS", "POST", "PUT", "DELETE"]:
371                  resp = self.fetch(url, method=method, allow_nonstandard_methods=True)
372                  self.assertEqual(method, to_unicode(resp.body))
373              resp = self.fetch(url, method="HEAD")
374              self.assertEqual(200, resp.code)
375              self.assertEqual(b"", resp.body)

455      def test_header_callback(self):
456          first_line = []
457          headers = {}
458          chunks = []
459  
460          def header_callback(header_line):
461              if header_line.startswith("HTTP/1.1 101"):
462                  # Upgrading to HTTP/2
463                  pass
464              elif header_line.startswith("HTTP/"):
465                  first_line.append(header_line)
466              elif header_line != "\r\n":
467                  k, v = header_line.split(":", 1)
468                  headers[k.lower()] = v.strip()
469  
470          def streaming_callback(chunk):
471              # All header callbacks are run before any streaming callbacks,
472              # so the header data is available to process the data as it
473              # comes in.
474              self.assertEqual(headers["content-type"], "text/html; charset=UTF-8")
475              chunks.append(chunk)
476  
477          self.fetch(
478              "/chunk",
479              header_callback=header_callback,
480              streaming_callback=streaming_callback,
481          )
482          self.assertEqual(len(first_line), 1, first_line)
483          self.assertRegex(first_line[0], "HTTP/[0-9]\\.[0-9] 200.*\r\n")
484          self.assertEqual(chunks, [b"asdf", b"qwer"])

730          # prohibits a bare CR specifically and "a recipient MAY recognize a single LF as a line
731          # terminator" so we check each character separately as well as the (redundant) CRLF pair.

788  class SyncHTTPClientTest(unittest.TestCase):
789      def setUp(self):
790          self.server_ioloop = IOLoop(make_current=False)
791          event = threading.Event()
792  
793          @gen.coroutine
794          def init_server():
795              sock, self.port = bind_unused_port()
796              app = Application([("/", HelloWorldHandler)])
797              self.server = HTTPServer(app)
798              self.server.add_socket(sock)
799              event.set()
800  
801          def start():
802              self.server_ioloop.run_sync(init_server)
803              self.server_ioloop.start()
804  
805          self.server_thread = threading.Thread(target=start)
806          self.server_thread.start()
807          event.wait()
808  
809          self.http_client = HTTPClient()
810  
811      def tearDown(self):
812          def stop_server():
813              self.server.stop()
814              # Delay the shutdown of the IOLoop by several iterations because
815              # the server may still have some cleanup work left when
816              # the client finishes with the response (this is noticeable
817              # with http/2, which leaves a Future with an unexamined
818              # StreamClosedError on the loop).
819  
820              @gen.coroutine
821              def slow_stop():
822                  yield self.server.close_all_connections()
823                  # The number of iterations is difficult to predict. Typically,
824                  # one is sufficient, although sometimes it needs more.
825                  for i in range(5):
826                      yield
827                  self.server_ioloop.stop()
828  
829              self.server_ioloop.add_callback(slow_stop)
830  
831          self.server_ioloop.add_callback(stop_server)
832          self.server_thread.join()
833          self.http_client.close()
834          self.server_ioloop.close(all_fds=True)
835  
836      def get_url(self, path):
837          return "http://127.0.0.1:%d%s" % (self.port, path)
838  
839      def test_sync_client(self):
840          response = self.http_client.fetch(self.get_url("/"))
841          self.assertEqual(b"Hello world!", response.body)
842  
843      def test_sync_client_error(self):
844          # Synchronous HTTPClient raises errors directly; no need for
845          # response.rethrow()
846          with self.assertRaises(HTTPError) as assertion:
847              self.http_client.fetch(self.get_url("/notfound"))
848          self.assertEqual(assertion.exception.code, 404)
