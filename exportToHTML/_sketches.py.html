10   from scipy.sparse import csc_matrix

12   __all__ = ['clarkson_woodruff_transform']

15   def cwt_matrix(n_rows, n_columns, rng=None):
16       r""" 
17       Generate a matrix S which represents a Clarkson-Woodruff transform. 
18    
19       Given the desired size of matrix, the method returns a matrix S of size 
20       (n_rows, n_columns) where each column has all the entries set to 0 
21       except for one position which has been randomly set to +1 or -1 with 
22       equal probability. 
23    
24       Parameters 
25       ---------- 
26       n_rows : int 
27           Number of rows of S 
28       n_columns : int 
29           Number of columns of S 
30       rng : `numpy.random.Generator`, optional 
31           Pseudorandom number generator state. When `rng` is None, a new 
32           `numpy.random.Generator` is created using entropy from the 
33           operating system. Types other than `numpy.random.Generator` are 
34           passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
35    
36    
37       Returns 
38       ------- 
39       S : (n_rows, n_columns) csc_matrix 
40           The returned matrix has ``n_columns`` nonzero entries. 
41    
42       Notes 
43       ----- 
44       Given a matrix A, with probability at least 9/10, 
45       .. math:: \|SA\| = (1 \pm \epsilon)\|A\| 
46       Where the error epsilon is related to the size of S. 
47       """
48       rng = check_random_state(rng)
49       rows = rng_integers(rng, 0, n_rows, n_columns)
50       cols = np.arange(n_columns+1)
51       signs = rng.choice([1, -1], n_columns)
52       S = csc_matrix((signs, rows, cols), shape=(n_rows, n_columns))
53       return S

56   @_transition_to_rng("seed", position_num=2)
57   def clarkson_woodruff_transform(input_matrix, sketch_size, rng=None):
58       r""" 
59       Applies a Clarkson-Woodruff Transform/sketch to the input matrix. 
60    
61       Given an input_matrix ``A`` of size ``(n, d)``, compute a matrix ``A'`` of 
62       size (sketch_size, d) so that 
63    
64       .. math:: \|Ax\| \approx \|A'x\| 
65    
66       with high probability via the Clarkson-Woodruff Transform, otherwise 
67       known as the CountSketch matrix. 
68    
69       Parameters 
70       ---------- 
71       input_matrix : array_like 
72           Input matrix, of shape ``(n, d)``. 
73       sketch_size : int 
74           Number of rows for the sketch. 
75       rng : `numpy.random.Generator`, optional 
76           Pseudorandom number generator state. When `rng` is None, a new 
77           `numpy.random.Generator` is created using entropy from the 
78           operating system. Types other than `numpy.random.Generator` are 
79           passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
80    
81       Returns 
82       ------- 
83       A' : array_like 
84           Sketch of the input matrix ``A``, of size ``(sketch_size, d)``. 
85    
86       Notes 
87       ----- 
88       To make the statement 
89    
90       .. math:: \|Ax\| \approx \|A'x\| 
91    
92       precise, observe the following result which is adapted from the 
93       proof of Theorem 14 of [2]_ via Markov's Inequality. If we have 
94       a sketch size ``sketch_size=k`` which is at least 
95    
96       .. math:: k \geq \frac{2}{\epsilon^2\delta} 
97    
98       Then for any fixed vector ``x``, 
99    
100      .. math:: \|Ax\| = (1\pm\epsilon)\|A'x\| 
101   
102      with probability at least one minus delta. 
103   
104      This implementation takes advantage of sparsity: computing 
105      a sketch takes time proportional to ``A.nnz``. Data ``A`` which 
106      is in ``scipy.sparse.csc_matrix`` format gives the quickest 
107      computation time for sparse input. 
108   
109      >>> import numpy as np 
110      >>> from scipy import linalg 
111      >>> from scipy import sparse 
112      >>> rng = np.random.default_rng() 
113      >>> n_rows, n_columns, density, sketch_n_rows = 15000, 100, 0.01, 200 
114      >>> A = sparse.rand(n_rows, n_columns, density=density, format='csc') 
115      >>> B = sparse.rand(n_rows, n_columns, density=density, format='csr') 
116      >>> C = sparse.rand(n_rows, n_columns, density=density, format='coo') 
117      >>> D = rng.standard_normal((n_rows, n_columns)) 
118      >>> SA = linalg.clarkson_woodruff_transform(A, sketch_n_rows) # fastest 
119      >>> SB = linalg.clarkson_woodruff_transform(B, sketch_n_rows) # fast 
120      >>> SC = linalg.clarkson_woodruff_transform(C, sketch_n_rows) # slower 
121      >>> SD = linalg.clarkson_woodruff_transform(D, sketch_n_rows) # slowest 
122   
123      That said, this method does perform well on dense inputs, just slower 
124      on a relative scale. 
125   
126      References 
127      ---------- 
128      .. [1] Kenneth L. Clarkson and David P. Woodruff. Low rank approximation 
129             and regression in input sparsity time. In STOC, 2013. 
130      .. [2] David P. Woodruff. Sketching as a tool for numerical linear algebra. 
131             In Foundations and Trends in Theoretical Computer Science, 2014. 
132   
133      Examples 
134      -------- 
135      Create a big dense matrix ``A`` for the example: 
136   
137      >>> import numpy as np 
138      >>> from scipy import linalg 
139      >>> n_rows, n_columns  = 15000, 100 
140      >>> rng = np.random.default_rng() 
141      >>> A = rng.standard_normal((n_rows, n_columns)) 
142   
143      Apply the transform to create a new matrix with 200 rows: 
144   
145      >>> sketch_n_rows = 200 
146      >>> sketch = linalg.clarkson_woodruff_transform(A, sketch_n_rows, seed=rng) 
147      >>> sketch.shape 
148      (200, 100) 
149   
150      Now with high probability, the true norm is close to the sketched norm 
151      in absolute value. 
152   
153      >>> linalg.norm(A) 
154      1224.2812927123198 
155      >>> linalg.norm(sketch) 
156      1226.518328407333 
157   
158      Similarly, applying our sketch preserves the solution to a linear 
159      regression of :math:`\min \|Ax - b\|`. 
160   
161      >>> b = rng.standard_normal(n_rows) 
162      >>> x = linalg.lstsq(A, b)[0] 
163      >>> Ab = np.hstack((A, b.reshape(-1, 1))) 
164      >>> SAb = linalg.clarkson_woodruff_transform(Ab, sketch_n_rows, seed=rng) 
165      >>> SA, Sb = SAb[:, :-1], SAb[:, -1] 
166      >>> x_sketched = linalg.lstsq(SA, Sb)[0] 
167   
168      As with the matrix norm example, ``linalg.norm(A @ x - b)`` is close 
169      to ``linalg.norm(A @ x_sketched - b)`` with high probability. 
170   
171      >>> linalg.norm(A @ x - b) 
172      122.83242365433877 
173      >>> linalg.norm(A @ x_sketched - b) 
174      166.58473879945151 
175   
176      """
177      S = cwt_matrix(sketch_size, input_matrix.shape[0], rng=rng)
178      return S.dot(input_matrix)
