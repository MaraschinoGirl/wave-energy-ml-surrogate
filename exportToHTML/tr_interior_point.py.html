1    """Trust-region interior point method. 
2     
3    References 
4    ---------- 
5    .. [1] Byrd, Richard H., Mary E. Hribar, and Jorge Nocedal. 
6           "An interior point algorithm for large-scale nonlinear 
7           programming." SIAM Journal on Optimization 9.4 (1999): 877-900. 
8    .. [2] Byrd, Richard H., Guanghui Liu, and Jorge Nocedal. 
9           "On the local behavior of an interior point method for 
10          nonlinear programming." Numerical analysis 1997 (1997): 37-56. 
11   .. [3] Nocedal, Jorge, and Stephen J. Wright. "Numerical optimization" 
12          Second Edition (2006). 
13   """
14   
15   import scipy.sparse as sps
16   import numpy as np
17   from .equality_constrained_sqp import equality_constrained_sqp
18   from scipy.sparse.linalg import LinearOperator
19   
20   __all__ = ['tr_interior_point']
21   
22   
23   class BarrierSubproblem:
24       """ 
25       Barrier optimization problem: 
26           minimize fun(x) - barrier_parameter*sum(log(s)) 
27           subject to: constr_eq(x)     = 0 
28                     constr_ineq(x) + s = 0 
29       """
30   
31       def __init__(self, x0, s0, fun, grad, lagr_hess, n_vars, n_ineq, n_eq,
32                    constr, jac, barrier_parameter, tolerance,
33                    enforce_feasibility, global_stop_criteria,
34                    xtol, fun0, grad0, constr_ineq0, jac_ineq0, constr_eq0,
35                    jac_eq0, finite_diff_bounds):
36           # Store parameters
37           self.n_vars = n_vars
38           self.x0 = x0
39           self.s0 = s0
40           self.fun = fun
41           self.grad = grad
42           self.lagr_hess = lagr_hess
43           self.constr = constr
44           self.jac = jac
45           self.barrier_parameter = barrier_parameter
46           self.tolerance = tolerance
47           self.n_eq = n_eq
48           self.n_ineq = n_ineq
49           self.enforce_feasibility = enforce_feasibility
50           self.global_stop_criteria = global_stop_criteria
51           self.xtol = xtol
52           self.fun0 = self._compute_function(fun0, constr_ineq0, s0)
53           self.grad0 = self._compute_gradient(grad0)
54           self.constr0 = self._compute_constr(constr_ineq0, constr_eq0, s0)
55           self.jac0 = self._compute_jacobian(jac_eq0, jac_ineq0, s0)
56           self.terminate = False
57           self.lb = finite_diff_bounds[0]
58           self.ub = finite_diff_bounds[1]
59   
60       def update(self, barrier_parameter, tolerance):
61           self.barrier_parameter = barrier_parameter
62           self.tolerance = tolerance
63   
64       def get_slack(self, z):
65           return z[self.n_vars:self.n_vars+self.n_ineq]
66   
67       def get_variables(self, z):
68           return z[:self.n_vars]
69   
70       def function_and_constraints(self, z):
71           """Returns barrier function and constraints at given point. 
72    
73           For z = [x, s], returns barrier function: 
74               function(z) = fun(x) - barrier_parameter*sum(log(s)) 
75           and barrier constraints: 
76               constraints(z) = [   constr_eq(x)     ] 
77                                [ constr_ineq(x) + s ] 
78    
79           """
80           # Get variables and slack variables
81           x = self.get_variables(z)
82           s = self.get_slack(z)
83   
84           # Compute function and constraints,
85           # making sure x is within any strict bounds
86           if np.any((x < self.lb) | (x > self.ub)):
87               # If x is out of the strict bounds, set f = inf,
88               # and just set both equality and inequality
89               # constraints to 0 since we can't evaluate
90               # them separately.
91               f = np.inf
92               c_eq = np.full(self.n_eq, 0.)
93               c_ineq = np.full(self.n_ineq, 0.)
94           else:
95               f = self.fun(x)
96               c_eq, c_ineq = self.constr(x)
97   
98           # Return objective function and constraints
99           return (self._compute_function(f, c_ineq, s),
100                  self._compute_constr(c_ineq, c_eq, s))
101  
102      def _compute_function(self, f, c_ineq, s):
103          # Use technique from Nocedal and Wright book, ref [3]_, p.576,
104          # to guarantee constraints from `enforce_feasibility`
105          # stay feasible along iterations.
106          s[self.enforce_feasibility] = -c_ineq[self.enforce_feasibility]
107          log_s = [np.log(s_i) if s_i > 0 else -np.inf for s_i in s]
108          # Compute barrier objective function
109          return f - self.barrier_parameter*np.sum(log_s)
110  
111      def _compute_constr(self, c_ineq, c_eq, s):
112          # Compute barrier constraint
113          return np.hstack((c_eq,
114                            c_ineq + s))
115  
116      def scaling(self, z):
117          """Returns scaling vector. 
118          Given by: 
119              scaling = [ones(n_vars), s] 
120          """
121          s = self.get_slack(z)
122          diag_elements = np.hstack((np.ones(self.n_vars), s))
123  
124          # Diagonal matrix
125          def matvec(vec):
126              return diag_elements*vec
127          return LinearOperator((self.n_vars+self.n_ineq,
128                                 self.n_vars+self.n_ineq),
129                                matvec)
130  
131      def gradient_and_jacobian(self, z):
132          """Returns scaled gradient. 
133   
134          Return scaled gradient: 
135              gradient = [             grad(x)             ] 
136                         [ -barrier_parameter*ones(n_ineq) ] 
137          and scaled Jacobian matrix: 
138              jacobian = [  jac_eq(x)  0  ] 
139                         [ jac_ineq(x) S  ] 
140          Both of them scaled by the previously defined scaling factor. 
141          """
142          # Get variables and slack variables
143          x = self.get_variables(z)
144          s = self.get_slack(z)
145          # Compute first derivatives
146          g = self.grad(x)
147          J_eq, J_ineq = self.jac(x)
148          # Return gradient and Jacobian
149          return (self._compute_gradient(g),
150                  self._compute_jacobian(J_eq, J_ineq, s))
151  
152      def _compute_gradient(self, g):
153          return np.hstack((g, -self.barrier_parameter*np.ones(self.n_ineq)))
154  
155      def _compute_jacobian(self, J_eq, J_ineq, s):
156          if self.n_ineq == 0:
157              return J_eq
158          else:
159              if sps.issparse(J_eq) or sps.issparse(J_ineq):
160                  # It is expected that J_eq and J_ineq
161                  # are already `csr_matrix` because of
162                  # the way ``BoxConstraint``, ``NonlinearConstraint``
163                  # and ``LinearConstraint`` are defined.
164                  J_eq = sps.csr_matrix(J_eq)
165                  J_ineq = sps.csr_matrix(J_ineq)
166                  return self._assemble_sparse_jacobian(J_eq, J_ineq, s)
167              else:
168                  S = np.diag(s)
169                  zeros = np.zeros((self.n_eq, self.n_ineq))
170                  # Convert to matrix
171                  if sps.issparse(J_ineq):
172                      J_ineq = J_ineq.toarray()
173                  if sps.issparse(J_eq):
174                      J_eq = J_eq.toarray()
175                  # Concatenate matrices
176                  return np.block([[J_eq, zeros],
177                                   [J_ineq, S]])
178  
179      def _assemble_sparse_jacobian(self, J_eq, J_ineq, s):
180          """Assemble sparse Jacobian given its components. 
181   
182          Given ``J_eq``, ``J_ineq`` and ``s`` returns: 
183              jacobian = [ J_eq,     0     ] 
184                         [ J_ineq, diag(s) ] 
185   
186          It is equivalent to: 
187              sps.bmat([[ J_eq,   None    ], 
188                        [ J_ineq, diag(s) ]], "csr") 
189          but significantly more efficient for this 
190          given structure. 
191          """
192          n_vars, n_ineq, n_eq = self.n_vars, self.n_ineq, self.n_eq
193          J_aux = sps.vstack([J_eq, J_ineq], "csr")
194          indptr, indices, data = J_aux.indptr, J_aux.indices, J_aux.data
195          new_indptr = indptr + np.hstack((np.zeros(n_eq, dtype=int),
196                                           np.arange(n_ineq+1, dtype=int)))
197          size = indices.size+n_ineq
198          new_indices = np.empty(size)
199          new_data = np.empty(size)
200          mask = np.full(size, False, bool)
201          mask[new_indptr[-n_ineq:]-1] = True
202          new_indices[mask] = n_vars+np.arange(n_ineq)
203          new_indices[~mask] = indices
204          new_data[mask] = s
205          new_data[~mask] = data
206          J = sps.csr_matrix((new_data, new_indices, new_indptr),
207                             (n_eq + n_ineq, n_vars + n_ineq))
208          return J
209  
210      def lagrangian_hessian_x(self, z, v):
211          """Returns Lagrangian Hessian (in relation to `x`) -> Hx"""
212          x = self.get_variables(z)
213          # Get lagrange multipliers related to nonlinear equality constraints
214          v_eq = v[:self.n_eq]
215          # Get lagrange multipliers related to nonlinear ineq. constraints
216          v_ineq = v[self.n_eq:self.n_eq+self.n_ineq]
217          lagr_hess = self.lagr_hess
218          return lagr_hess(x, v_eq, v_ineq)
219  
220      def lagrangian_hessian_s(self, z, v):
221          """Returns scaled Lagrangian Hessian (in relation to`s`) -> S Hs S"""
222          s = self.get_slack(z)
223          # Using the primal formulation:
224          #     S Hs S = diag(s)*diag(barrier_parameter/s**2)*diag(s).
225          # Reference [1]_ p. 882, formula (3.1)
226          primal = self.barrier_parameter
227          # Using the primal-dual formulation
228          #     S Hs S = diag(s)*diag(v/s)*diag(s)
229          # Reference [1]_ p. 883, formula (3.11)
230          primal_dual = v[-self.n_ineq:]*s
231          # Uses the primal-dual formulation for
232          # positives values of v_ineq, and primal
233          # formulation for the remaining ones.
234          return np.where(v[-self.n_ineq:] > 0, primal_dual, primal)
235  
236      def lagrangian_hessian(self, z, v):
237          """Returns scaled Lagrangian Hessian"""
238          # Compute Hessian in relation to x and s
239          Hx = self.lagrangian_hessian_x(z, v)
240          if self.n_ineq > 0:
241              S_Hs_S = self.lagrangian_hessian_s(z, v)
242  
243          # The scaled Lagragian Hessian is:
244          #     [ Hx    0    ]
245          #     [ 0   S Hs S ]
246          def matvec(vec):
247              vec_x = self.get_variables(vec)
248              vec_s = self.get_slack(vec)
249              if self.n_ineq > 0:
250                  return np.hstack((Hx.dot(vec_x), S_Hs_S*vec_s))
251              else:
252                  return Hx.dot(vec_x)
253          return LinearOperator((self.n_vars+self.n_ineq,
254                                 self.n_vars+self.n_ineq),
255                                matvec)
256  
257      def stop_criteria(self, state, z, last_iteration_failed,
258                        optimality, constr_violation,
259                        trust_radius, penalty, cg_info):
260          """Stop criteria to the barrier problem. 
261          The criteria here proposed is similar to formula (2.3) 
262          from [1]_, p.879. 
263          """
264          x = self.get_variables(z)
265          if self.global_stop_criteria(state, x,
266                                       last_iteration_failed,
267                                       trust_radius, penalty,
268                                       cg_info,
269                                       self.barrier_parameter,
270                                       self.tolerance):
271              self.terminate = True
272              return True
273          else:
274              g_cond = (optimality < self.tolerance and
275                        constr_violation < self.tolerance)
276              x_cond = trust_radius < self.xtol
277              return g_cond or x_cond

280  def tr_interior_point(fun, grad, lagr_hess, n_vars, n_ineq, n_eq,
281                        constr, jac, x0, fun0, grad0,
282                        constr_ineq0, jac_ineq0, constr_eq0,
283                        jac_eq0, stop_criteria,
284                        enforce_feasibility, xtol, state,
285                        initial_barrier_parameter,
286                        initial_tolerance,
287                        initial_penalty,
288                        initial_trust_radius,
289                        factorization_method,
290                        finite_diff_bounds):
291      """Trust-region interior points method. 
292   
293      Solve problem: 
294          minimize fun(x) 
295          subject to: constr_ineq(x) <= 0 
296                      constr_eq(x) = 0 
297      using trust-region interior point method described in [1]_. 
298      """
299      # BOUNDARY_PARAMETER controls the decrease on the slack
300      # variables. Represents ``tau`` from [1]_ p.885, formula (3.18).
301      BOUNDARY_PARAMETER = 0.995
302      # BARRIER_DECAY_RATIO controls the decay of the barrier parameter
303      # and of the subproblem tolerance. Represents ``theta`` from [1]_ p.879.
304      BARRIER_DECAY_RATIO = 0.2
305      # TRUST_ENLARGEMENT controls the enlargement on trust radius
306      # after each iteration
307      TRUST_ENLARGEMENT = 5
308  
309      # Default enforce_feasibility
310      if enforce_feasibility is None:
311          enforce_feasibility = np.zeros(n_ineq, bool)
312      # Initial Values
313      barrier_parameter = initial_barrier_parameter
314      tolerance = initial_tolerance
315      trust_radius = initial_trust_radius
316      # Define initial value for the slack variables
317      s0 = np.maximum(-1.5*constr_ineq0, np.ones(n_ineq))
318      # Define barrier subproblem
319      subprob = BarrierSubproblem(
320          x0, s0, fun, grad, lagr_hess, n_vars, n_ineq, n_eq, constr, jac,
321          barrier_parameter, tolerance, enforce_feasibility,
322          stop_criteria, xtol, fun0, grad0, constr_ineq0, jac_ineq0,
323          constr_eq0, jac_eq0, finite_diff_bounds)
324      # Define initial parameter for the first iteration.
325      z = np.hstack((x0, s0))
326      fun0_subprob, constr0_subprob = subprob.fun0, subprob.constr0
327      grad0_subprob, jac0_subprob = subprob.grad0, subprob.jac0
328      # Define trust region bounds
329      trust_lb = np.hstack((np.full(subprob.n_vars, -np.inf),
330                            np.full(subprob.n_ineq, -BOUNDARY_PARAMETER)))
331      trust_ub = np.full(subprob.n_vars+subprob.n_ineq, np.inf)
332  
333      # Solves a sequence of barrier problems
334      while True:
335          # Solve SQP subproblem
336          z, state = equality_constrained_sqp(
337              subprob.function_and_constraints,
338              subprob.gradient_and_jacobian,
339              subprob.lagrangian_hessian,
340              z, fun0_subprob, grad0_subprob,
341              constr0_subprob, jac0_subprob, subprob.stop_criteria,
342              state, initial_penalty, trust_radius,
343              factorization_method, trust_lb, trust_ub, subprob.scaling)
344          if subprob.terminate:
345              break
346          # Update parameters
347          trust_radius = max(initial_trust_radius,
348                             TRUST_ENLARGEMENT*state.tr_radius)
349          # TODO: Use more advanced strategies from [2]_
350          # to update this parameters.
351          barrier_parameter *= BARRIER_DECAY_RATIO
352          tolerance *= BARRIER_DECAY_RATIO
353          # Update Barrier Problem
354          subprob.update(barrier_parameter, tolerance)
355          # Compute initial values for next iteration
356          fun0_subprob, constr0_subprob = subprob.function_and_constraints(z)
357          grad0_subprob, jac0_subprob = subprob.gradient_and_jacobian(z)
358  
359      # Get x and s
360      x = subprob.get_variables(z)
361      return x, state
