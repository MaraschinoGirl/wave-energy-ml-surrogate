14   def chunk_generator(gen, chunksize):
15       """Chunk generator, ``gen`` into lists of length ``chunksize``. The last 
16       chunk may have a length less than ``chunksize``."""
17       while True:
18           chunk = list(islice(gen, chunksize))
19           if chunk:
20               yield chunk
21           else:
22               return

25   @validate_params(
26       {
27           "n": [Interval(Integral, 1, None, closed="left")],
28           "batch_size": [Interval(Integral, 1, None, closed="left")],
29           "min_batch_size": [Interval(Integral, 0, None, closed="left")],
30       },
31       prefer_skip_nested_validation=True,
32   )
33   def gen_batches(n, batch_size, *, min_batch_size=0):
34       """Generator to create slices containing `batch_size` elements from 0 to `n`. 
35    
36       The last slice may contain less than `batch_size` elements, when 
37       `batch_size` does not divide `n`. 
38    
39       Parameters 
40       ---------- 
41       n : int 
42           Size of the sequence. 
43       batch_size : int 
44           Number of elements in each batch. 
45       min_batch_size : int, default=0 
46           Minimum number of elements in each batch. 
47    
48       Yields 
49       ------ 
50       slice of `batch_size` elements 
51    
52       See Also 
53       -------- 
54       gen_even_slices: Generator to create n_packs slices going up to n. 
55    
56       Examples 
57       -------- 
58       >>> from sklearn.utils import gen_batches 
59       >>> list(gen_batches(7, 3)) 
60       [slice(0, 3, None), slice(3, 6, None), slice(6, 7, None)] 
61       >>> list(gen_batches(6, 3)) 
62       [slice(0, 3, None), slice(3, 6, None)] 
63       >>> list(gen_batches(2, 3)) 
64       [slice(0, 2, None)] 
65       >>> list(gen_batches(7, 3, min_batch_size=0)) 
66       [slice(0, 3, None), slice(3, 6, None), slice(6, 7, None)] 
67       >>> list(gen_batches(7, 3, min_batch_size=2)) 
68       [slice(0, 3, None), slice(3, 7, None)] 
69       """
70       start = 0
71       for _ in range(int(n // batch_size)):
72           end = start + batch_size
73           if end + min_batch_size > n:
74               continue
75           yield slice(start, end)
76           start = end
77       if start < n:
78           yield slice(start, n)

81   @validate_params(
82       {
83           "n": [Interval(Integral, 1, None, closed="left")],
84           "n_packs": [Interval(Integral, 1, None, closed="left")],
85           "n_samples": [Interval(Integral, 1, None, closed="left"), None],
86       },
87       prefer_skip_nested_validation=True,
88   )
89   def gen_even_slices(n, n_packs, *, n_samples=None):
90       """Generator to create `n_packs` evenly spaced slices going up to `n`. 
91    
92       If `n_packs` does not divide `n`, except for the first `n % n_packs` 
93       slices, remaining slices may contain fewer elements. 
94    
95       Parameters 
96       ---------- 
97       n : int 
98           Size of the sequence. 
99       n_packs : int 
100          Number of slices to generate. 
101      n_samples : int, default=None 
102          Number of samples. Pass `n_samples` when the slices are to be used for 
103          sparse matrix indexing; slicing off-the-end raises an exception, while 
104          it works for NumPy arrays. 
105   
106      Yields 
107      ------ 
108      `slice` representing a set of indices from 0 to n. 
109   
110      See Also 
111      -------- 
112      gen_batches: Generator to create slices containing batch_size elements 
113          from 0 to n. 
114   
115      Examples 
116      -------- 
117      >>> from sklearn.utils import gen_even_slices 
118      >>> list(gen_even_slices(10, 1)) 
119      [slice(0, 10, None)] 
120      >>> list(gen_even_slices(10, 10)) 
121      [slice(0, 1, None), slice(1, 2, None), ..., slice(9, 10, None)] 
122      >>> list(gen_even_slices(10, 5)) 
123      [slice(0, 2, None), slice(2, 4, None), ..., slice(8, 10, None)] 
124      >>> list(gen_even_slices(10, 3)) 
125      [slice(0, 4, None), slice(4, 7, None), slice(7, 10, None)] 
126      """
127      start = 0
128      for pack_num in range(n_packs):
129          this_n = n // n_packs
130          if pack_num < n % n_packs:
131              this_n += 1
132          if this_n > 0:
133              end = start + this_n
134              if n_samples is not None:
135                  end = min(n_samples, end)
136              yield slice(start, end, None)
137              start = end

140  def get_chunk_n_rows(row_bytes, *, max_n_rows=None, working_memory=None):
141      """Calculate how many rows can be processed within `working_memory`. 
142   
143      Parameters 
144      ---------- 
145      row_bytes : int 
146          The expected number of bytes of memory that will be consumed 
147          during the processing of each row. 
148      max_n_rows : int, default=None 
149          The maximum return value. 
150      working_memory : int or float, default=None 
151          The number of rows to fit inside this number of MiB will be 
152          returned. When None (default), the value of 
153          ``sklearn.get_config()['working_memory']`` is used. 
154   
155      Returns 
156      ------- 
157      int 
158          The number of rows which can be processed within `working_memory`. 
159   
160      Warns 
161      ----- 
162      Issues a UserWarning if `row_bytes exceeds `working_memory` MiB. 
163      """
164  
165      if working_memory is None:
166          working_memory = get_config()["working_memory"]
167  
168      chunk_n_rows = int(working_memory * (2**20) // row_bytes)
169      if max_n_rows is not None:
170          chunk_n_rows = min(chunk_n_rows, max_n_rows)
171      if chunk_n_rows < 1:
172          warnings.warn(
173              "Could not adhere to working_memory config. "
174              "Currently %.0fMiB, %.0fMiB required."
175              % (working_memory, np.ceil(row_bytes * 2**-20))
176          )
177          chunk_n_rows = 1
178      return chunk_n_rows
