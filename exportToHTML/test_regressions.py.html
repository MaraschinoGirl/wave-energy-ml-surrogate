11   def test_issue_11():
12       # Give a sensible error message for level mismatches
13       # (At some points we've failed to put an origin= on these errors)
14       env = EvalEnvironment.capture()
15       data = {"X": [0, 1, 2, 3], "Y": [1, 2, 3, 4]}
16       formula = "C(X) + Y"
17       new_data = {"X": [0, 0, 1, 2, 3, 3, 4], "Y": [1, 2, 3, 4, 5, 6, 7]}
18       info = dmatrix(formula, data)
19       try:
20           build_design_matrices([info.design_info], new_data)
21       except PatsyError as e:
22           assert e.origin == Origin(formula, 0, 4)
23       else:
24           assert False
