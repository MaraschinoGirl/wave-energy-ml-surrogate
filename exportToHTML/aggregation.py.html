18   @dataclass
19   class Agg(Stat):
20       """ 
21       Aggregate data along the value axis using given method. 
22    
23       Parameters 
24       ---------- 
25       func : str or callable 
26           Name of a :class:`pandas.Series` method or a vector -> scalar function. 
27    
28       See Also 
29       -------- 
30       objects.Est : Aggregation with error bars. 
31    
32       Examples 
33       -------- 
34       .. include:: ../docstrings/objects.Agg.rst 
35    
36       """
37       func: str | Callable[[Vector], float] = "mean"
38   
39       group_by_orient: ClassVar[bool] = True
40   
41       def __call__(
42           self, data: DataFrame, groupby: GroupBy, orient: str, scales: dict[str, Scale],
43       ) -> DataFrame:
44   
45           var = {"x": "y", "y": "x"}.get(orient)
46           res = (
47               groupby
48               .agg(data, {var: self.func})
49               .dropna(subset=[var])
50               .reset_index(drop=True)
51           )
52           return res

55   @dataclass
56   class Est(Stat):
57       """ 
58       Calculate a point estimate and error bar interval. 
59    
60       For more information about the various `errorbar` choices, see the 
61       :doc:`errorbar tutorial </tutorial/error_bars>`. 
62    
63       Additional variables: 
64    
65       - **weight**: When passed to a layer that uses this stat, a weighted estimate 
66         will be computed. Note that use of weights currently limits the choice of 
67         function and error bar method  to `"mean"` and `"ci"`, respectively. 
68    
69       Parameters 
70       ---------- 
71       func : str or callable 
72           Name of a :class:`numpy.ndarray` method or a vector -> scalar function. 
73       errorbar : str, (str, float) tuple, or callable 
74           Name of errorbar method (one of "ci", "pi", "se" or "sd"), or a tuple 
75           with a method name ane a level parameter, or a function that maps from a 
76           vector to a (min, max) interval. 
77       n_boot : int 
78          Number of bootstrap samples to draw for "ci" errorbars. 
79       seed : int 
80           Seed for the PRNG used to draw bootstrap samples. 
81    
82       Examples 
83       -------- 
84       .. include:: ../docstrings/objects.Est.rst 
85    
86       """
87       func: str | Callable[[Vector], float] = "mean"
88       errorbar: str | tuple[str, float] = ("ci", 95)
89       n_boot: int = 1000
90       seed: int | None = None
91   
92       group_by_orient: ClassVar[bool] = True
93   
94       def _process(
95           self, data: DataFrame, var: str, estimator: EstimateAggregator
96       ) -> DataFrame:
97           # Needed because GroupBy.apply assumes func is DataFrame -> DataFrame
98           # which we could probably make more general to allow Series return
99           res = estimator(data, var)
100          return pd.DataFrame([res])
101  
102      def __call__(
103          self, data: DataFrame, groupby: GroupBy, orient: str, scales: dict[str, Scale],
104      ) -> DataFrame:
105  
106          boot_kws = {"n_boot": self.n_boot, "seed": self.seed}
107          if "weight" in data:
108              engine = WeightedAggregator(self.func, self.errorbar, **boot_kws)
109          else:
110              engine = EstimateAggregator(self.func, self.errorbar, **boot_kws)
111  
112          var = {"x": "y", "y": "x"}[orient]
113          res = (
114              groupby
115              .apply(data, self._process, var, engine)
116              .dropna(subset=[var])
117              .reset_index(drop=True)
118          )
119  
120          res = res.fillna({f"{var}min": res[var], f"{var}max": res[var]})
121  
122          return res
