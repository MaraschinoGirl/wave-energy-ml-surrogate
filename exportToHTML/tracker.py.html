12   def get_family(addr: str) -> int:
13       """Get network family from address."""
14       return socket.getaddrinfo(addr, None)[0][0]

17   class RabitTracker:
18       """Tracker for the collective used in XGBoost, acting as a coordinator between 
19       workers. 
20    
21       Parameters 
22       ---------- 
23    
24       n_workers: 
25    
26           The total number of workers in the communication group. 
27    
28       host_ip: 
29    
30           The IP address of the tracker node. XGBoost can try to guess one by probing with 
31           sockets. But it's best to explicitly pass an address. 
32    
33       port: 
34    
35           The port this tracker should listen to. XGBoost can query an available port from 
36           the OS, this configuration is useful for restricted network environments. 
37    
38       sortby: 
39    
40           How to sort the workers for rank assignment. The default is host, but users can 
41           set the `DMLC_TASK_ID` via arguments of :py:meth:`~xgboost.collective.init` and 
42           obtain deterministic rank assignment through sorting by task name. Available 
43           options are: 
44    
45             - host 
46             - task 
47    
48       timeout : 
49    
50           Timeout for constructing (bootstrap) and shutting down the communication group, 
51           doesn't apply to communication when the group is up and running. 
52    
53           The timeout value should take the time of data loading and pre-processing into 
54           account, due to potential lazy execution. By default the Tracker doesn't have 
55           any timeout to avoid pre-mature aborting. 
56    
57           The :py:meth:`.wait_for` method has a different timeout parameter that can stop 
58           the tracker even if the tracker is still being used. A value error is raised 
59           when timeout is reached. 
60    
61       Examples 
62       -------- 
63    
64       .. code-block:: python 
65    
66           from xgboost.tracker import RabitTracker 
67           from xgboost import collective as coll 
68    
69           tracker = RabitTracker(host_ip="127.0.0.1", n_workers=2) 
70           tracker.start() 
71    
72           with coll.CommunicatorContext(**tracker.worker_args()): 
73               ret = coll.broadcast("msg", 0) 
74               assert str(ret) == "msg" 
75    
76       """
77   
78       @unique
79       class _SortBy(IntEnum):
80           HOST = 0
81           TASK = 1
82   
83       @_deprecate_positional_args
84       def __init__(  # pylint: disable=too-many-arguments
85           self,
86           n_workers: int,
87           host_ip: Optional[str],
88           port: int = 0,
89           *,
90           sortby: str = "host",
91           timeout: int = 0,
92       ) -> None:
93   
94           handle = ctypes.c_void_p()
95           if sortby not in ("host", "task"):
96               raise ValueError("Expecting either 'host' or 'task' for sortby.")
97           if host_ip is not None:
98               get_family(host_ip)  # use python socket to stop early for invalid address
99           args = make_jcargs(
100              host=host_ip,
101              n_workers=n_workers,
102              port=port,
103              dmlc_communicator="rabit",
104              sortby=self._SortBy.HOST if sortby == "host" else self._SortBy.TASK,
105              timeout=int(timeout),
106          )
107          _check_call(_LIB.XGTrackerCreate(args, ctypes.byref(handle)))
108          self.handle = handle
109  
110      def free(self) -> None:
111          """Internal function for testing."""
112          if hasattr(self, "handle"):
113              handle = self.handle
114              del self.handle
115              _check_call(_LIB.XGTrackerFree(handle))
116  
117      def __del__(self) -> None:
118          self.free()
119  
120      def start(self) -> None:
121          """Start the tracker. Once started, the client still need to call the 
122          :py:meth:`wait_for` method in order to wait for it to finish (think of it as a 
123          thread). 
124   
125          """
126          _check_call(_LIB.XGTrackerRun(self.handle, make_jcargs()))
127  
128      def wait_for(self, timeout: Optional[int] = None) -> None:
129          """Wait for the tracker to finish all the work and shutdown. When timeout is 
130          reached, a value error is raised. By default we don't have timeout since we 
131          don't know how long it takes for the model to finish training. 
132   
133          """
134          _check_call(_LIB.XGTrackerWaitFor(self.handle, make_jcargs(timeout=timeout)))
135  
136      def worker_args(self) -> Dict[str, Union[str, int]]:
137          """Get arguments for workers."""
138          c_env = ctypes.c_char_p()
139          _check_call(_LIB.XGTrackerWorkerArgs(self.handle, ctypes.byref(c_env)))
140          assert c_env.value is not None
141          env = json.loads(c_env.value)
142          return env
