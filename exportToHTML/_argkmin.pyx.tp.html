34       @classmethod
35       def compute(
36           cls,
37           X,
38           Y,
39           intp_t k,
40           metric="euclidean",
41           chunk_size=None,
42           dict metric_kwargs=None,
43           str strategy=None,
44           bint return_distance=False,
45       ):
46           """Compute the argkmin reduction.
47   
48           This classmethod is responsible for introspecting the arguments
49           values to dispatch to the most appropriate implementation of
50           :class:`ArgKmin{{name_suffix}}`.
51   
52           This allows decoupling the API entirely from the implementation details
53           whilst maintaining RAII: all temporarily allocated datastructures necessary
54           for the concrete implementation are therefore freed when this classmethod
55           returns.
56   
57           No instance should directly be created outside of this class method.
58           """
59           # Limit the number of threads in second level of nested parallelism for BLAS
60           # to avoid threads over-subscription (in DOT or GEMM for instance).
61           with _get_threadpool_controller().limit(limits=1, user_api='blas'):
62             if metric in ("euclidean", "sqeuclidean"):
63                 # Specialized implementation of ArgKmin for the Euclidean distance
64                 # for the dense-dense and sparse-sparse cases.
65                 # This implementation computes the distances by chunk using
66                 # a decomposition of the Squared Euclidean distance.
67                 # This specialisation has an improved arithmetic intensity for both
68                 # the dense and sparse settings, allowing in most case speed-ups of
69                 # several orders of magnitude compared to the generic ArgKmin
70                 # implementation.
71                 # Note that squared norms of X and Y are precomputed in the
72                 # constructor of this class by issuing BLAS calls that may use
73                 # multithreading (depending on the BLAS implementation), hence calling
74                 # the constructor needs to be protected under the threadpool_limits
75                 # context, along with the main calls to _parallel_on_Y and
76                 # _parallel_on_X.
77                 # For more information see MiddleTermComputer.
78                 use_squared_distances = metric == "sqeuclidean"
79                 pda = EuclideanArgKmin{{name_suffix}}(
80                     X=X, Y=Y, k=k,
81                     use_squared_distances=use_squared_distances,
82                     chunk_size=chunk_size,
83                     strategy=strategy,
84                     metric_kwargs=metric_kwargs,
85                 )
86             else:
87                 # Fall back on a generic implementation that handles most scipy
88                 # metrics by computing the distances between 2 vectors at a time.
89                 pda = ArgKmin{{name_suffix}}(
90                     datasets_pair=DatasetsPair{{name_suffix}}.get_for(X, Y, metric, metric_kwargs),
91                     k=k,
92                     chunk_size=chunk_size,
93                     strategy=strategy,
94                 )
95   
96             if pda.execute_in_parallel_on_Y:
97                 pda._parallel_on_Y()
98             else:
99                 pda._parallel_on_X()
100  
101          return pda._finalize_results(return_distance)

103      def __init__(
104          self,
105          DatasetsPair{{name_suffix}} datasets_pair,
106          chunk_size=None,
107          strategy=None,
108          intp_t k=1,
109      ):
110          super().__init__(
111              datasets_pair=datasets_pair,
112              chunk_size=chunk_size,
113              strategy=strategy,
114          )
115          self.k = check_scalar(k, "k", Integral, min_val=1)
116  
117          # Allocating pointers to datastructures but not the datastructures themselves.
118          # There are as many pointers as effective threads.
119          #
120          # For the sake of explicitness:
121          #   - when parallelizing on X, the pointers of those heaps are referencing
122          #   (with proper offsets) addresses of the two main heaps (see below)
123          #   - when parallelizing on Y, the pointers of those heaps are referencing
124          #   small heaps which are thread-wise-allocated and whose content will be
125          #   merged with the main heaps'.
126          self.heaps_r_distances_chunks = <float64_t **> malloc(
127              sizeof(float64_t *) * self.chunks_n_threads
128          )
129          self.heaps_indices_chunks = <intp_t **> malloc(
130              sizeof(intp_t *) * self.chunks_n_threads
131          )
132  
133          # Main heaps which will be returned as results by `ArgKmin{{name_suffix}}.compute`.
134          self.argkmin_indices = np.full((self.n_samples_X, self.k), 0, dtype=np.intp)
135          self.argkmin_distances = np.full((self.n_samples_X, self.k), DBL_MAX, dtype=np.float64)
136  
137      def __dealloc__(self):
138          if self.heaps_indices_chunks is not NULL:
139              free(self.heaps_indices_chunks)
140  
141          if self.heaps_r_distances_chunks is not NULL:
142              free(self.heaps_r_distances_chunks)

144      cdef void _compute_and_reduce_distances_on_chunks(
145          self,
146          intp_t X_start,
147          intp_t X_end,
148          intp_t Y_start,
149          intp_t Y_end,
150          intp_t thread_num,
151      ) noexcept nogil:
152          cdef:
153              intp_t i, j
154              intp_t n_samples_X = X_end - X_start
155              intp_t n_samples_Y = Y_end - Y_start
156              float64_t *heaps_r_distances = self.heaps_r_distances_chunks[thread_num]
157              intp_t *heaps_indices = self.heaps_indices_chunks[thread_num]
158  
159          # Pushing the distances and their associated indices on a heap
160          # which by construction will keep track of the argkmin.
161          for i in range(n_samples_X):
162              for j in range(n_samples_Y):
163                  heap_push(
164                      values=heaps_r_distances + i * self.k,
165                      indices=heaps_indices + i * self.k,
166                      size=self.k,
167                      val=self.datasets_pair.surrogate_dist(X_start + i, Y_start + j),
168                      val_idx=Y_start + j,
169                  )

171      cdef void _parallel_on_X_init_chunk(
172          self,
173          intp_t thread_num,
174          intp_t X_start,
175          intp_t X_end,
176      ) noexcept nogil:
177          # As this strategy is embarrassingly parallel, we can set each
178          # thread's heaps pointer to the proper position on the main heaps.
179          self.heaps_r_distances_chunks[thread_num] = &self.argkmin_distances[X_start, 0]
180          self.heaps_indices_chunks[thread_num] = &self.argkmin_indices[X_start, 0]

182      cdef void _parallel_on_X_prange_iter_finalize(
183          self,
184          intp_t thread_num,
185          intp_t X_start,
186          intp_t X_end,
187      ) noexcept nogil:
188          cdef:
189              intp_t idx
190  
191          # Sorting the main heaps portion associated to `X[X_start:X_end]`
192          # in ascending order w.r.t the distances.
193          for idx in range(X_end - X_start):
194              simultaneous_sort(
195                  self.heaps_r_distances_chunks[thread_num] + idx * self.k,
196                  self.heaps_indices_chunks[thread_num] + idx * self.k,
197                  self.k
198              )

200      cdef void _parallel_on_Y_init(
201          self,
202      ) noexcept nogil:
203          cdef:
204              # Maximum number of scalar elements (the last chunks can be smaller)
205              intp_t heaps_size = self.X_n_samples_chunk * self.k
206              intp_t thread_num
207  
208          # The allocation is done in parallel for data locality purposes: this way
209          # the heaps used in each threads are allocated in pages which are closer
210          # to the CPU core used by the thread.
211          # See comments about First Touch Placement Policy:
212          # https://www.openmp.org/wp-content/uploads/openmp-webinar-vanderPas-20210318.pdf #noqa
213          for thread_num in prange(self.chunks_n_threads, schedule='static', nogil=True,
214                                   num_threads=self.chunks_n_threads):
215              # As chunks of X are shared across threads, so must their
216              # heaps. To solve this, each thread has its own heaps
217              # which are then synchronised back in the main ones.
218              self.heaps_r_distances_chunks[thread_num] = <float64_t *> malloc(
219                  heaps_size * sizeof(float64_t)
220              )
221              self.heaps_indices_chunks[thread_num] = <intp_t *> malloc(
222                  heaps_size * sizeof(intp_t)
223              )

225      cdef void _parallel_on_Y_parallel_init(
226          self,
227          intp_t thread_num,
228          intp_t X_start,
229          intp_t X_end,
230      ) noexcept nogil:
231          # Initialising heaps (memset can't be used here)
232          for idx in range(self.X_n_samples_chunk * self.k):
233              self.heaps_r_distances_chunks[thread_num][idx] = DBL_MAX
234              self.heaps_indices_chunks[thread_num][idx] = -1

236      @final
237      cdef void _parallel_on_Y_synchronize(
238          self,
239          intp_t X_start,
240          intp_t X_end,
241      ) noexcept nogil:
242          cdef:
243              intp_t idx, jdx, thread_num
244          with nogil, parallel(num_threads=self.effective_n_threads):
245              # Synchronising the thread heaps with the main heaps.
246              # This is done in parallel sample-wise (no need for locks).
247              #
248              # This might break each thread's data locality as each heap which
249              # was allocated in a thread is being now being used in several threads.
250              #
251              # Still, this parallel pattern has shown to be efficient in practice.
252              for idx in prange(X_end - X_start, schedule="static"):
253                  for thread_num in range(self.chunks_n_threads):
254                      for jdx in range(self.k):
255                          heap_push(
256                              values=&self.argkmin_distances[X_start + idx, 0],
257                              indices=&self.argkmin_indices[X_start + idx, 0],
258                              size=self.k,
259                              val=self.heaps_r_distances_chunks[thread_num][idx * self.k + jdx],
260                              val_idx=self.heaps_indices_chunks[thread_num][idx * self.k + jdx],
261                          )
262  
263      cdef void _parallel_on_Y_finalize(
264          self,
265      ) noexcept nogil:
266          cdef:
267              intp_t idx, thread_num
268  
269          with nogil, parallel(num_threads=self.chunks_n_threads):
270              # Deallocating temporary datastructures
271              for thread_num in prange(self.chunks_n_threads, schedule='static'):
272                  free(self.heaps_r_distances_chunks[thread_num])
273                  free(self.heaps_indices_chunks[thread_num])
274  
275              # Sorting the main in ascending order w.r.t the distances.
276              # This is done in parallel sample-wise (no need for locks).
277              for idx in prange(self.n_samples_X, schedule='static'):
278                  simultaneous_sort(
279                      &self.argkmin_distances[idx, 0],
280                      &self.argkmin_indices[idx, 0],
281                      self.k,
282                  )
283          return

285      cdef void compute_exact_distances(self) noexcept nogil:
286          cdef:
287              intp_t i, j
288              float64_t[:, ::1] distances = self.argkmin_distances
289          for i in prange(self.n_samples_X, schedule='static', nogil=True,
290                          num_threads=self.effective_n_threads):
291              for j in range(self.k):
292                  distances[i, j] = self.datasets_pair.distance_metric._rdist_to_dist(
293                      # Guard against potential -0., causing nan production.
294                      max(distances[i, j], 0.)
295                  )
296  
297      def _finalize_results(self, bint return_distance=False):
298          if return_distance:
299              # We need to recompute distances because we relied on
300              # surrogate distances for the reduction.
301              self.compute_exact_distances()
302  
303              # Values are returned identically to the way `KNeighborsMixin.kneighbors`
304              # returns values. This is counter-intuitive but this allows not using
305              # complex adaptations where `ArgKmin.compute` is called.
306              return np.asarray(self.argkmin_distances), np.asarray(self.argkmin_indices)
307  
308          return np.asarray(self.argkmin_indices)

472          self,
473          intp_t X_start,
474          intp_t X_end,
475          intp_t Y_start,
476          intp_t Y_end,
477          intp_t thread_num,
478      ) noexcept nogil:
479          cdef:
480              intp_t i, j
481              float64_t sqeuclidean_dist_i_j
482              intp_t n_X = X_end - X_start
483              intp_t n_Y = Y_end - Y_start
484              float64_t * dist_middle_terms = self.middle_term_computer._compute_dist_middle_terms(
485                  X_start, X_end, Y_start, Y_end, thread_num
486              )

488              intp_t * heaps_indices = self.heaps_indices_chunks[thread_num]

491          # which keep tracks of the argkmin.
492          for i in range(n_X):
493              for j in range(n_Y):
494                  sqeuclidean_dist_i_j = (
495                      self.X_norm_squared[i + X_start] +
496                      dist_middle_terms[i * n_Y + j] +
497                      self.Y_norm_squared[j + Y_start]

504                  heap_push(

506                      indices=heaps_indices + i * self.k,
507                      size=self.k,
508                      val=sqeuclidean_dist_i_j,
