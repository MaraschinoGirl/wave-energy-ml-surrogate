9    def get_coverage_files(coverage_output_dir, number_of_files):
10       base_dir = coverage_output_dir
11       ret = []
12       i = 0
13       while len(ret) < number_of_files:
14           while True:
15               f = os.path.join(base_dir, ".coverage.%s" % i)
16               i += 1
17               if not os.path.exists(f):
18                   ret.append(f)
19                   break  # Break only inner for.
20       return ret

38   def start_coverage_support_from_params(coverage_output_dir, coverage_output_file, jobs, coverage_include):
39       coverage_files = []
40       coverage_instance = Null()
41       if coverage_output_dir or coverage_output_file:
42           try:
43               import coverage  # @UnresolvedImport
44           except:
45               sys.stderr.write("Error: coverage module could not be imported\n")
46               sys.stderr.write("Please make sure that the coverage module (http://nedbatchelder.com/code/coverage/)\n")
47               sys.stderr.write("is properly installed in your interpreter: %s\n" % (sys.executable,))
48   
49               import traceback
50   
51               traceback.print_exc()
52           else:
53               if coverage_output_dir:
54                   if not os.path.exists(coverage_output_dir):
55                       sys.stderr.write("Error: directory for coverage output (%s) does not exist.\n" % (coverage_output_dir,))
56   
57                   elif not os.path.isdir(coverage_output_dir):
58                       sys.stderr.write("Error: expected (%s) to be a directory.\n" % (coverage_output_dir,))
59   
60                   else:
61                       n = jobs
62                       if n <= 0:
63                           n += 1
64                       n += 1  # Add 1 more for the current process (which will do the initial import).
65                       coverage_files = get_coverage_files(coverage_output_dir, n)
66                       os.environ["COVERAGE_FILE"] = coverage_files.pop(0)
67   
68                       coverage_instance = coverage.coverage(source=[coverage_include])
69                       coverage_instance.start()
70   
71               elif coverage_output_file:
72                   # Client of parallel run.
73                   os.environ["COVERAGE_FILE"] = coverage_output_file
74                   coverage_instance = coverage.coverage(source=[coverage_include])
75                   coverage_instance.start()
76   
77       return coverage_files, coverage_instance
