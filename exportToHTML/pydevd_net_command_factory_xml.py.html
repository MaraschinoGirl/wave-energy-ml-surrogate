127      def make_get_thread_stack_message(self, py_db, seq, thread_id, topmost_frame, fmt, must_be_suspended=False, start_frame=0, levels=0):
128          """ 
129          Returns thread stack as XML. 
130   
131          :param must_be_suspended: If True and the thread is not suspended, returns None. 
132          """
133          try:
134              # If frame is None, the return is an empty frame list.
135              cmd_text = ['<xml><thread id="%s">' % (thread_id,)]
136  
137              if topmost_frame is not None:
138                  try:
139                      # : :type suspended_frames_manager: SuspendedFramesManager
140                      suspended_frames_manager = py_db.suspended_frames_manager
141                      frames_list = suspended_frames_manager.get_frames_list(thread_id)
142                      if frames_list is None:
143                          # Could not find stack of suspended frame...
144                          if must_be_suspended:
145                              return None
146                          else:
147                              frames_list = pydevd_frame_utils.create_frames_list_from_frame(topmost_frame)
148  
149                      cmd_text.append(self.make_thread_stack_str(py_db, frames_list))
150                  finally:
151                      topmost_frame = None
152              cmd_text.append("</thread></xml>")
153              return NetCommand(CMD_GET_THREAD_STACK, seq, "".join(cmd_text))
154          except:
155              return self.make_error_message(seq, get_exception_traceback_str())

197      def _iter_visible_frames_info(self, py_db, frames_list, flatten_chained=False):
198          assert frames_list.__class__ == FramesList
199          is_chained = False
200          while True:
201              for frame in frames_list:
202                  show_as_current_frame = frame is frames_list.current_frame
203                  if frame.f_code is None:
204                      pydev_log.info("Frame without f_code: %s", frame)
205                      continue  # IronPython sometimes does not have it!
206  
207                  method_name = frame.f_code.co_name  # method name (if in method) or ? if global
208                  if method_name is None:
209                      pydev_log.info("Frame without co_name: %s", frame)
210                      continue  # IronPython sometimes does not have it!
211  
212                  if is_chained:
213                      method_name = "[Chained Exc: %s] %s" % (frames_list.exc_desc, method_name)
214  
215                  abs_path_real_path_and_base = get_abs_path_real_path_and_base_from_frame(frame)
216                  if py_db.get_file_type(frame, abs_path_real_path_and_base) == py_db.PYDEV_FILE:
217                      # Skip pydevd files.
218                      frame = frame.f_back
219                      continue
220  
221                  frame_id = id(frame)
222                  lineno = frames_list.frame_id_to_lineno.get(frame_id, frame.f_lineno)
223                  line_col_info = frames_list.frame_id_to_line_col_info.get(frame_id)
224  
225                  filename_in_utf8, lineno, changed = py_db.source_mapping.map_to_client(abs_path_real_path_and_base[0], lineno)
226                  new_filename_in_utf8, applied_mapping = pydevd_file_utils.map_file_to_client(filename_in_utf8)
227                  applied_mapping = applied_mapping or changed
228  
229                  yield (
230                      frame_id,
231                      frame,
232                      method_name,
233                      abs_path_real_path_and_base[0],
234                      new_filename_in_utf8,
235                      lineno,
236                      applied_mapping,
237                      show_as_current_frame,
238                      line_col_info,
239                  )
240  
241              if not flatten_chained:
242                  break
243  
244              frames_list = frames_list.chained_frames_list
245              if frames_list is None or len(frames_list) == 0:
246                  break
247              is_chained = True

249      def make_thread_stack_str(self, py_db, frames_list):
250          assert frames_list.__class__ == FramesList
251          make_valid_xml_value = pydevd_xml.make_valid_xml_value
252          cmd_text_list = []
253          append = cmd_text_list.append
254  
255          try:
256              for (
257                  frame_id,
258                  frame,
259                  method_name,
260                  _original_filename,
261                  filename_in_utf8,
262                  lineno,
263                  _applied_mapping,
264                  _show_as_current_frame,
265                  line_col_info,
266              ) in self._iter_visible_frames_info(py_db, frames_list, flatten_chained=True):
267                  # print("file is ", filename_in_utf8)
268                  # print("line is ", lineno)
269  
270                  # Note: variables are all gotten 'on-demand'.
271                  append('<frame id="%s" name="%s" ' % (frame_id, make_valid_xml_value(method_name)))
272                  append('file="%s" line="%s">' % (quote(make_valid_xml_value(filename_in_utf8), "/>_= \t"), lineno))
273                  append("</frame>")
274          except:
275              pydev_log.exception()
276  
277          return "".join(cmd_text_list)

279      def make_thread_suspend_str(
280          self,
281          py_db,
282          thread_id,
283          frames_list,
284          stop_reason=None,
285          message=None,
286          trace_suspend_type="trace",
287      ):
288          """ 
289          :return tuple(str,str): 
290              Returns tuple(thread_suspended_str, thread_stack_str). 
291   
292              i.e.: 
293              ( 
294                  ''' 
295                      <xml> 
296                          <thread id="id" stop_reason="reason"> 
297                              <frame id="id" name="functionName " file="file" line="line"> 
298                              </frame> 
299                          </thread> 
300                      </xml> 
301                  ''' 
302                  , 
303                  ''' 
304                  <frame id="id" name="functionName " file="file" line="line"> 
305                  </frame> 
306                  ''' 
307              ) 
308          """
309          assert frames_list.__class__ == FramesList
310          make_valid_xml_value = pydevd_xml.make_valid_xml_value
311          cmd_text_list = []
312          append = cmd_text_list.append
313  
314          cmd_text_list.append("<xml>")
315          if message:
316              message = make_valid_xml_value(message)
317  
318          append('<thread id="%s"' % (thread_id,))
319          if stop_reason is not None:
320              append(' stop_reason="%s"' % (stop_reason,))
321          if message is not None:
322              append(' message="%s"' % (message,))
323          if trace_suspend_type is not None:
324              append(' suspend_type="%s"' % (trace_suspend_type,))
325          append(">")
326          thread_stack_str = self.make_thread_stack_str(py_db, frames_list)
327          append(thread_stack_str)
328          append("</thread></xml>")
329  
330          return "".join(cmd_text_list), thread_stack_str

438      def make_get_exception_details_message(self, py_db, seq, thread_id, topmost_frame):
439          """Returns exception details as XML"""
440          try:
441              # If the debugger is not suspended, just return the thread and its id.
442              cmd_text = ['<xml><thread id="%s" ' % (thread_id,)]
443  
444              if topmost_frame is not None:
445                  try:
446                      frame = topmost_frame
447                      topmost_frame = None
448                      while frame is not None:
449                          if frame.f_code.co_name == "do_wait_suspend" and frame.f_code.co_filename.endswith("pydevd.py"):
450                              arg = frame.f_locals.get("arg", None)
451                              if arg is not None:
452                                  exc_type, exc_desc, _thread_suspend_str, thread_stack_str = self._make_send_curr_exception_trace_str(
453                                      py_db, thread_id, *arg
454                                  )
455                                  cmd_text.append('exc_type="%s" ' % (exc_type,))
456                                  cmd_text.append('exc_desc="%s" ' % (exc_desc,))
457                                  cmd_text.append(">")
458                                  cmd_text.append(thread_stack_str)
459                                  break
460                          frame = frame.f_back
461                      else:
462                          cmd_text.append(">")
463                  finally:
464                      frame = None
465              cmd_text.append("</thread></xml>")
466              return NetCommand(CMD_GET_EXCEPTION_DETAILS, seq, "".join(cmd_text))
467          except:
468              return self.make_error_message(seq, get_exception_traceback_str())

530      def make_evaluation_timeout_msg(self, py_db, expression, thread):
531          msg = """pydevd: Evaluating: %s did not finish after %.2f seconds. 
532  This may mean a number of things: 
533  - This evaluation is really slow and this is expected. 
534      In this case it's possible to silence this error by raising the timeout, setting the 
535      PYDEVD_WARN_EVALUATION_TIMEOUT environment variable to a bigger value. 
536   
537  - The evaluation may need other threads running while it's running: 
538      In this case, you may need to manually let other paused threads continue. 
539   
540      Alternatively, it's also possible to skip breaking on a particular thread by setting a 
541      `pydev_do_not_trace = True` attribute in the related threading.Thread instance 
542      (if some thread should always be running and no breakpoints are expected to be hit in it). 
543   
544  - The evaluation is deadlocked: 
545      In this case you may set the PYDEVD_THREAD_DUMP_ON_WARN_EVALUATION_TIMEOUT 
546      environment variable to true so that a thread dump is shown along with this message and 
547      optionally, set the PYDEVD_INTERRUPT_THREAD_TIMEOUT to some value so that the debugger
