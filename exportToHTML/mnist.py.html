1    """MNIST handwritten digits dataset."""
2    
3    import numpy as np
4    
5    from keras.src.api_export import keras_export
6    from keras.src.utils.file_utils import get_file
7    
8    
9    @keras_export("keras.datasets.mnist.load_data")
10   def load_data(path="mnist.npz"):
11       """Loads the MNIST dataset. 
12    
13       This is a dataset of 60,000 28x28 grayscale images of the 10 digits, 
14       along with a test set of 10,000 images. 
15       More info can be found at the 
16       [MNIST homepage](http://yann.lecun.com/exdb/mnist/). 
17    
18       Args: 
19           path: path where to cache the dataset locally 
20               (relative to `~/.keras/datasets`). 
21    
22       Returns: 
23           Tuple of NumPy arrays: `(x_train, y_train), (x_test, y_test)`. 
24    
25       **`x_train`**: `uint8` NumPy array of grayscale image data with shapes 
26         `(60000, 28, 28)`, containing the training data. Pixel values range 
27         from 0 to 255. 
28    
29       **`y_train`**: `uint8` NumPy array of digit labels (integers in range 0-9) 
30         with shape `(60000,)` for the training data. 
31    
32       **`x_test`**: `uint8` NumPy array of grayscale image data with shapes 
33         `(10000, 28, 28)`, containing the test data. Pixel values range 
34         from 0 to 255. 
35    
36       **`y_test`**: `uint8` NumPy array of digit labels (integers in range 0-9) 
37         with shape `(10000,)` for the test data. 
38    
39       Example: 
40    
41       ```python 
42       (x_train, y_train), (x_test, y_test) = keras.datasets.mnist.load_data() 
43       assert x_train.shape == (60000, 28, 28) 
44       assert x_test.shape == (10000, 28, 28) 
45       assert y_train.shape == (60000,) 
46       assert y_test.shape == (10000,) 
47       ``` 
48    
49       License: 
50    
51       Yann LeCun and Corinna Cortes hold the copyright of MNIST dataset, 
52       which is a derivative work from original NIST datasets. 
53       MNIST dataset is made available under the terms of the 
54       [Creative Commons Attribution-Share Alike 3.0 license.]( 
55           https://creativecommons.org/licenses/by-sa/3.0/) 
56       """
57       origin_folder = (
58           "https://storage.googleapis.com/tensorflow/tf-keras-datasets/"
59       )
60       path = get_file(
61           fname=path,
62           origin=origin_folder + "mnist.npz",
63           file_hash=(  # noqa: E501
64               "731c5ac602752760c8e48fbffcf8c3b850d9dc2a2aedcf2cc48468fc17b673d1"
65           ),
66       )
67       with np.load(path, allow_pickle=True) as f:
68           x_train, y_train = f["x_train"], f["y_train"]
69           x_test, y_test = f["x_test"], f["y_test"]
70   
71           return (x_train, y_train), (x_test, y_test)
