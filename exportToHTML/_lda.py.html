44   def _update_doc_distribution(
45       X,
46       exp_topic_word_distr,
47       doc_topic_prior,
48       max_doc_update_iter,
49       mean_change_tol,
50       cal_sstats,
51       random_state,
52   ):
53       """E-step: update document-topic distribution. 
54    
55       Parameters 
56       ---------- 
57       X : {array-like, sparse matrix} of shape (n_samples, n_features) 
58           Document word matrix. 
59    
60       exp_topic_word_distr : ndarray of shape (n_topics, n_features) 
61           Exponential value of expectation of log topic word distribution. 
62           In the literature, this is `exp(E[log(beta)])`. 
63    
64       doc_topic_prior : float 
65           Prior of document topic distribution `theta`. 
66    
67       max_doc_update_iter : int 
68           Max number of iterations for updating document topic distribution in 
69           the E-step. 
70    
71       mean_change_tol : float 
72           Stopping tolerance for updating document topic distribution in E-step. 
73    
74       cal_sstats : bool 
75           Parameter that indicate to calculate sufficient statistics or not. 
76           Set `cal_sstats` to `True` when we need to run M-step. 
77    
78       random_state : RandomState instance or None 
79           Parameter that indicate how to initialize document topic distribution. 
80           Set `random_state` to None will initialize document topic distribution 
81           to a constant number. 
82    
83       Returns 
84       ------- 
85       (doc_topic_distr, suff_stats) : 
86           `doc_topic_distr` is unnormalized topic distribution for each document. 
87           In the literature, this is `gamma`. we can calculate `E[log(theta)]` 
88           from it. 
89           `suff_stats` is expected sufficient statistics for the M-step. 
90               When `cal_sstats == False`, this will be None. 
91    
92       """
93       is_sparse_x = sp.issparse(X)
94       n_samples, n_features = X.shape
95       n_topics = exp_topic_word_distr.shape[0]
96   
97       if random_state:
98           doc_topic_distr = random_state.gamma(100.0, 0.01, (n_samples, n_topics)).astype(
99               X.dtype, copy=False
100          )
101      else:
102          doc_topic_distr = np.ones((n_samples, n_topics), dtype=X.dtype)
103  
104      # In the literature, this is `exp(E[log(theta)])`
105      exp_doc_topic = np.exp(_dirichlet_expectation_2d(doc_topic_distr))
106  
107      # diff on `component_` (only calculate it when `cal_diff` is True)
108      suff_stats = (
109          np.zeros(exp_topic_word_distr.shape, dtype=X.dtype) if cal_sstats else None
110      )
111  
112      if is_sparse_x:
113          X_data = X.data
114          X_indices = X.indices
115          X_indptr = X.indptr
116  
117      # These cython functions are called in a nested loop on usually very small arrays
118      # (length=n_topics). In that case, finding the appropriate signature of the
119      # fused-typed function can be more costly than its execution, hence the dispatch
120      # is done outside of the loop.
121      ctype = "float" if X.dtype == np.float32 else "double"
122      mean_change = cy_mean_change[ctype]
123      dirichlet_expectation_1d = cy_dirichlet_expectation_1d[ctype]
124      eps = np.finfo(X.dtype).eps
125  
126      for idx_d in range(n_samples):
127          if is_sparse_x:
128              ids = X_indices[X_indptr[idx_d] : X_indptr[idx_d + 1]]
129              cnts = X_data[X_indptr[idx_d] : X_indptr[idx_d + 1]]
130          else:
131              ids = np.nonzero(X[idx_d, :])[0]
132              cnts = X[idx_d, ids]
133  
134          doc_topic_d = doc_topic_distr[idx_d, :]
135          # The next one is a copy, since the inner loop overwrites it.
136          exp_doc_topic_d = exp_doc_topic[idx_d, :].copy()
137          exp_topic_word_d = exp_topic_word_distr[:, ids]
138  
139          # Iterate between `doc_topic_d` and `norm_phi` until convergence
140          for _ in range(0, max_doc_update_iter):
141              last_d = doc_topic_d
142  
143              # The optimal phi_{dwk} is proportional to
144              # exp(E[log(theta_{dk})]) * exp(E[log(beta_{dw})]).
145              norm_phi = np.dot(exp_doc_topic_d, exp_topic_word_d) + eps
146  
147              doc_topic_d = exp_doc_topic_d * np.dot(cnts / norm_phi, exp_topic_word_d.T)
148              # Note: adds doc_topic_prior to doc_topic_d, in-place.
149              dirichlet_expectation_1d(doc_topic_d, doc_topic_prior, exp_doc_topic_d)
150  
151              if mean_change(last_d, doc_topic_d) < mean_change_tol:
152                  break
153          doc_topic_distr[idx_d, :] = doc_topic_d
154  
155          # Contribution of document d to the expected sufficient
156          # statistics for the M step.
157          if cal_sstats:
158              norm_phi = np.dot(exp_doc_topic_d, exp_topic_word_d) + eps
159              suff_stats[:, ids] += np.outer(exp_doc_topic_d, cnts / norm_phi)
160  
161      return (doc_topic_distr, suff_stats)

164  class LatentDirichletAllocation(
165      ClassNamePrefixFeaturesOutMixin, TransformerMixin, BaseEstimator
166  ):
167      """Latent Dirichlet Allocation with online variational Bayes algorithm. 
168   
169      The implementation is based on [1]_ and [2]_. 
170   
171      .. versionadded:: 0.17 
172   
173      Read more in the :ref:`User Guide <LatentDirichletAllocation>`. 
174   
175      Parameters 
176      ---------- 
177      n_components : int, default=10 
178          Number of topics. 
179   
180          .. versionchanged:: 0.19 
181              ``n_topics`` was renamed to ``n_components`` 
182   
183      doc_topic_prior : float, default=None 
184          Prior of document topic distribution `theta`. If the value is None, 
185          defaults to `1 / n_components`. 
186          In [1]_, this is called `alpha`. 
187   
188      topic_word_prior : float, default=None 
189          Prior of topic word distribution `beta`. If the value is None, defaults 
190          to `1 / n_components`. 
191          In [1]_, this is called `eta`. 
192   
193      learning_method : {'batch', 'online'}, default='batch' 
194          Method used to update `_component`. Only used in :meth:`fit` method. 
195          In general, if the data size is large, the online update will be much 
196          faster than the batch update. 
197   
198          Valid options: 
199   
200          - 'batch': Batch variational Bayes method. Use all training data in each EM 
201            update. Old `components_` will be overwritten in each iteration. 
202          - 'online': Online variational Bayes method. In each EM update, use mini-batch 
203            of training data to update the ``components_`` variable incrementally. The 
204            learning rate is controlled by the ``learning_decay`` and the 
205            ``learning_offset`` parameters. 
206   
207          .. versionchanged:: 0.20 
208              The default learning method is now ``"batch"``. 
209   
210      learning_decay : float, default=0.7 
211          It is a parameter that control learning rate in the online learning 
212          method. The value should be set between (0.5, 1.0] to guarantee 
213          asymptotic convergence. When the value is 0.0 and batch_size is 
214          ``n_samples``, the update method is same as batch learning. In the 
215          literature, this is called kappa. 
216   
217      learning_offset : float, default=10.0 
218          A (positive) parameter that downweights early iterations in online 
219          learning.  It should be greater than 1.0. In the literature, this is 
220          called tau_0. 
221   
222      max_iter : int, default=10 
223          The maximum number of passes over the training data (aka epochs). 
224          It only impacts the behavior in the :meth:`fit` method, and not the 
225          :meth:`partial_fit` method. 
226   
227      batch_size : int, default=128 
228          Number of documents to use in each EM iteration. Only used in online 
229          learning. 
230   
231      evaluate_every : int, default=-1 
232          How often to evaluate perplexity. Only used in `fit` method. 
233          set it to 0 or negative number to not evaluate perplexity in 
234          training at all. Evaluating perplexity can help you check convergence 
235          in training process, but it will also increase total training time. 
236          Evaluating perplexity in every iteration might increase training time 
237          up to two-fold. 
238   
239      total_samples : int, default=1e6 
240          Total number of documents. Only used in the :meth:`partial_fit` method. 
241   
242      perp_tol : float, default=1e-1 
243          Perplexity tolerance. Only used when ``evaluate_every`` is greater than 0. 
244   
245      mean_change_tol : float, default=1e-3 
246          Stopping tolerance for updating document topic distribution in E-step. 
247   
248      max_doc_update_iter : int, default=100 
249          Max number of iterations for updating document topic distribution in 
250          the E-step. 
251   
252      n_jobs : int, default=None 
253          The number of jobs to use in the E-step. 
254          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
255          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
256          for more details. 
257   
258      verbose : int, default=0 
259          Verbosity level. 
260   
261      random_state : int, RandomState instance or None, default=None 
262          Pass an int for reproducible results across multiple function calls. 
263          See :term:`Glossary <random_state>`. 
264   
265      Attributes 
266      ---------- 
267      components_ : ndarray of shape (n_components, n_features) 
268          Variational parameters for topic word distribution. Since the complete 
269          conditional for topic word distribution is a Dirichlet, 
270          ``components_[i, j]`` can be viewed as pseudocount that represents the 
271          number of times word `j` was assigned to topic `i`. 
272          It can also be viewed as distribution over the words for each topic 
273          after normalization: 
274          ``model.components_ / model.components_.sum(axis=1)[:, np.newaxis]``. 
275   
276      exp_dirichlet_component_ : ndarray of shape (n_components, n_features) 
277          Exponential value of expectation of log topic word distribution. 
278          In the literature, this is `exp(E[log(beta)])`. 
279   
280      n_batch_iter_ : int 
281          Number of iterations of the EM step. 
282   
283      n_features_in_ : int 
284          Number of features seen during :term:`fit`. 
285   
286          .. versionadded:: 0.24 
287   
288      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
289          Names of features seen during :term:`fit`. Defined only when `X` 
290          has feature names that are all strings. 
291   
292          .. versionadded:: 1.0 
293   
294      n_iter_ : int 
295          Number of passes over the dataset. 
296   
297      bound_ : float 
298          Final perplexity score on training set. 
299   
300      doc_topic_prior_ : float 
301          Prior of document topic distribution `theta`. If the value is None, 
302          it is `1 / n_components`. 
303   
304      random_state_ : RandomState instance 
305          RandomState instance that is generated either from a seed, the random 
306          number generator or by `np.random`. 
307   
308      topic_word_prior_ : float 
309          Prior of topic word distribution `beta`. If the value is None, it is 
310          `1 / n_components`. 
311   
312      See Also 
313      -------- 
314      sklearn.discriminant_analysis.LinearDiscriminantAnalysis: 
315          A classifier with a linear decision boundary, generated by fitting 
316          class conditional densities to the data and using Bayes' rule. 
317   
318      References 
319      ---------- 
320      .. [1] "Online Learning for Latent Dirichlet Allocation", Matthew D. 
321             Hoffman, David M. Blei, Francis Bach, 2010 
322             https://github.com/blei-lab/onlineldavb 
323   
324      .. [2] "Stochastic Variational Inference", Matthew D. Hoffman, 
325             David M. Blei, Chong Wang, John Paisley, 2013 
326   
327      Examples 
328      -------- 
329      >>> from sklearn.decomposition import LatentDirichletAllocation 
330      >>> from sklearn.datasets import make_multilabel_classification 
331      >>> # This produces a feature matrix of token counts, similar to what 
332      >>> # CountVectorizer would produce on text. 
333      >>> X, _ = make_multilabel_classification(random_state=0) 
334      >>> lda = LatentDirichletAllocation(n_components=5, 
335      ...     random_state=0) 
336      >>> lda.fit(X) 
337      LatentDirichletAllocation(...) 
338      >>> # get topics for some given samples: 
339      >>> lda.transform(X[-2:]) 
340      array([[0.00360392, 0.25499205, 0.0036211 , 0.64236448, 0.09541846], 
341             [0.15297572, 0.00362644, 0.44412786, 0.39568399, 0.003586  ]]) 
342      """

400      def _init_latent_vars(self, n_features, dtype=np.float64):
401          """Initialize latent variables."""
402  
403          self.random_state_ = check_random_state(self.random_state)
404          self.n_batch_iter_ = 1
405          self.n_iter_ = 0
406  
407          if self.doc_topic_prior is None:
408              self.doc_topic_prior_ = 1.0 / self.n_components
409          else:
410              self.doc_topic_prior_ = self.doc_topic_prior
411  
412          if self.topic_word_prior is None:
413              self.topic_word_prior_ = 1.0 / self.n_components
414          else:
415              self.topic_word_prior_ = self.topic_word_prior
416  
417          init_gamma = 100.0
418          init_var = 1.0 / init_gamma
419          # In the literature, this is called `lambda`
420          self.components_ = self.random_state_.gamma(
421              init_gamma, init_var, (self.n_components, n_features)
422          ).astype(dtype, copy=False)
423  
424          # In the literature, this is `exp(E[log(beta)])`
425          self.exp_dirichlet_component_ = np.exp(
426              _dirichlet_expectation_2d(self.components_)
427          )
428  
429      def _e_step(self, X, cal_sstats, random_init, parallel=None):
430          """E-step in EM update. 
431   
432          Parameters 
433          ---------- 
434          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
435              Document word matrix. 
436   
437          cal_sstats : bool 
438              Parameter that indicate whether to calculate sufficient statistics 
439              or not. Set ``cal_sstats`` to True when we need to run M-step. 
440   
441          random_init : bool 
442              Parameter that indicate whether to initialize document topic 
443              distribution randomly in the E-step. Set it to True in training 
444              steps. 
445   
446          parallel : joblib.Parallel, default=None 
447              Pre-initialized instance of joblib.Parallel. 
448   
449          Returns 
450          ------- 
451          (doc_topic_distr, suff_stats) : 
452              `doc_topic_distr` is unnormalized topic distribution for each 
453              document. In the literature, this is called `gamma`. 
454              `suff_stats` is expected sufficient statistics for the M-step. 
455              When `cal_sstats == False`, it will be None. 
456   
457          """
458  
459          # Run e-step in parallel
460          random_state = self.random_state_ if random_init else None
461  
462          # TODO: make Parallel._effective_n_jobs public instead?
463          n_jobs = effective_n_jobs(self.n_jobs)
464          if parallel is None:
465              parallel = Parallel(n_jobs=n_jobs, verbose=max(0, self.verbose - 1))
466          results = parallel(
467              delayed(_update_doc_distribution)(
468                  X[idx_slice, :],
469                  self.exp_dirichlet_component_,
470                  self.doc_topic_prior_,
471                  self.max_doc_update_iter,
472                  self.mean_change_tol,
473                  cal_sstats,
474                  random_state,
475              )
476              for idx_slice in gen_even_slices(X.shape[0], n_jobs)
477          )
478  
479          # merge result
480          doc_topics, sstats_list = zip(*results)
481          doc_topic_distr = np.vstack(doc_topics)
482  
483          if cal_sstats:
484              # This step finishes computing the sufficient statistics for the
485              # M-step.
486              suff_stats = np.zeros(self.components_.shape, dtype=self.components_.dtype)
487              for sstats in sstats_list:
488                  suff_stats += sstats
489              suff_stats *= self.exp_dirichlet_component_
490          else:
491              suff_stats = None
492  
493          return (doc_topic_distr, suff_stats)
494  
495      def _em_step(self, X, total_samples, batch_update, parallel=None):
496          """EM update for 1 iteration. 
497   
498          update `_component` by batch VB or online VB. 
499   
500          Parameters 
501          ---------- 
502          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
503              Document word matrix. 
504   
505          total_samples : int 
506              Total number of documents. It is only used when 
507              batch_update is `False`. 
508   
509          batch_update : bool 
510              Parameter that controls updating method. 
511              `True` for batch learning, `False` for online learning. 
512   
513          parallel : joblib.Parallel, default=None 
514              Pre-initialized instance of joblib.Parallel 
515   
516          Returns 
517          ------- 
518          doc_topic_distr : ndarray of shape (n_samples, n_components) 
519              Unnormalized document topic distribution. 
520          """
521  
522          # E-step
523          _, suff_stats = self._e_step(
524              X, cal_sstats=True, random_init=True, parallel=parallel
525          )
526  
527          # M-step
528          if batch_update:
529              self.components_ = self.topic_word_prior_ + suff_stats
530          else:
531              # online update
532              # In the literature, the weight is `rho`
533              weight = np.power(
534                  self.learning_offset + self.n_batch_iter_, -self.learning_decay
535              )
536              doc_ratio = float(total_samples) / X.shape[0]
537              self.components_ *= 1 - weight
538              self.components_ += weight * (
539                  self.topic_word_prior_ + doc_ratio * suff_stats
540              )
541  
542          # update `component_` related variables
543          self.exp_dirichlet_component_ = np.exp(
544              _dirichlet_expectation_2d(self.components_)
545          )
546          self.n_batch_iter_ += 1
547          return

579      @_fit_context(prefer_skip_nested_validation=True)
580      def partial_fit(self, X, y=None):
581          """Online VB with Mini-Batch update. 
582   
583          Parameters 
584          ---------- 
585          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
586              Document word matrix. 
587   
588          y : Ignored 
589              Not used, present here for API consistency by convention. 
590   
591          Returns 
592          ------- 
593          self 
594              Partially fitted estimator. 
595          """
596          first_time = not hasattr(self, "components_")
597  
598          X = self._check_non_neg_array(
599              X, reset_n_features=first_time, whom="LatentDirichletAllocation.partial_fit"
600          )
601          n_samples, n_features = X.shape
602          batch_size = self.batch_size
603  
604          # initialize parameters or check
605          if first_time:
606              self._init_latent_vars(n_features, dtype=X.dtype)
607  
608          if n_features != self.components_.shape[1]:
609              raise ValueError(
610                  "The provided data has %d dimensions while "
611                  "the model was trained with feature size %d."
612                  % (n_features, self.components_.shape[1])
613              )
614  
615          n_jobs = effective_n_jobs(self.n_jobs)
616          with Parallel(n_jobs=n_jobs, verbose=max(0, self.verbose - 1)) as parallel:
617              for idx_slice in gen_batches(n_samples, batch_size):
618                  self._em_step(
619                      X[idx_slice, :],
620                      total_samples=self.total_samples,
621                      batch_update=False,
622                      parallel=parallel,
623                  )
624  
625          return self

627      @_fit_context(prefer_skip_nested_validation=True)
628      def fit(self, X, y=None):
629          """Learn model for the data X with variational Bayes method. 
630   
631          When `learning_method` is 'online', use mini-batch update. 
632          Otherwise, use batch update. 
633   
634          Parameters 
635          ---------- 
636          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
637              Document word matrix. 
638   
639          y : Ignored 
640              Not used, present here for API consistency by convention. 
641   
642          Returns 
643          ------- 
644          self 
645              Fitted estimator. 
646          """
647          X = self._check_non_neg_array(
648              X, reset_n_features=True, whom="LatentDirichletAllocation.fit"
649          )
650          n_samples, n_features = X.shape
651          max_iter = self.max_iter
652          evaluate_every = self.evaluate_every
653          learning_method = self.learning_method
654  
655          batch_size = self.batch_size
656  
657          # initialize parameters
658          self._init_latent_vars(n_features, dtype=X.dtype)
659          # change to perplexity later
660          last_bound = None
661          n_jobs = effective_n_jobs(self.n_jobs)
662          with Parallel(n_jobs=n_jobs, verbose=max(0, self.verbose - 1)) as parallel:
663              for i in range(max_iter):
664                  if learning_method == "online":
665                      for idx_slice in gen_batches(n_samples, batch_size):
666                          self._em_step(
667                              X[idx_slice, :],
668                              total_samples=n_samples,
669                              batch_update=False,
670                              parallel=parallel,
671                          )
672                  else:
673                      # batch update
674                      self._em_step(
675                          X, total_samples=n_samples, batch_update=True, parallel=parallel
676                      )
677  
678                  # check perplexity
679                  if evaluate_every > 0 and (i + 1) % evaluate_every == 0:
680                      doc_topics_distr, _ = self._e_step(
681                          X, cal_sstats=False, random_init=False, parallel=parallel
682                      )
683                      bound = self._perplexity_precomp_distr(
684                          X, doc_topics_distr, sub_sampling=False
685                      )
686                      if self.verbose:
687                          print(
688                              "iteration: %d of max_iter: %d, perplexity: %.4f"
689                              % (i + 1, max_iter, bound)
690                          )
691  
692                      if last_bound and abs(last_bound - bound) < self.perp_tol:
693                          break
694                      last_bound = bound
695  
696                  elif self.verbose:
697                      print("iteration: %d of max_iter: %d" % (i + 1, max_iter))
698                  self.n_iter_ += 1
699  
700          # calculate final perplexity value on train set
701          doc_topics_distr, _ = self._e_step(
702              X, cal_sstats=False, random_init=False, parallel=parallel
703          )
704          self.bound_ = self._perplexity_precomp_distr(
705              X, doc_topics_distr, sub_sampling=False
706          )
707  
708          return self
709  
710      def _unnormalized_transform(self, X):
711          """Transform data X according to fitted model. 
712   
713          Parameters 
714          ---------- 
715          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
716              Document word matrix. 
717   
718          Returns 
719          ------- 
720          doc_topic_distr : ndarray of shape (n_samples, n_components) 
721              Document topic distribution for X. 
722          """
723          doc_topic_distr, _ = self._e_step(X, cal_sstats=False, random_init=False)
724  
725          return doc_topic_distr

727      def transform(self, X, *, normalize=True):
728          """Transform data X according to the fitted model. 
729   
730          .. versionchanged:: 0.18 
731              `doc_topic_distr` is now normalized. 
732   
733          Parameters 
734          ---------- 
735          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
736              Document word matrix. 
737   
738          normalize : bool, default=True 
739              Whether to normalize the document topic distribution. 
740   
741          Returns 
742          ------- 
743          doc_topic_distr : ndarray of shape (n_samples, n_components) 
744              Document topic distribution for X. 
745          """
746          check_is_fitted(self)
747          X = self._check_non_neg_array(
748              X, reset_n_features=False, whom="LatentDirichletAllocation.transform"
749          )
750          doc_topic_distr = self._unnormalized_transform(X)
751          if normalize:
752              doc_topic_distr /= doc_topic_distr.sum(axis=1)[:, np.newaxis]
753          return doc_topic_distr

755      def fit_transform(self, X, y=None, *, normalize=True):
756          """ 
757          Fit to data, then transform it. 
758   
759          Fits transformer to `X` and `y` and returns a transformed version of `X`. 
760   
761          Parameters 
762          ---------- 
763          X : array-like of shape (n_samples, n_features) 
764              Input samples. 
765   
766          y :  array-like of shape (n_samples,) or (n_samples, n_outputs), \ 
767                  default=None 
768              Target values (None for unsupervised transformations). 
769   
770          normalize : bool, default=True 
771              Whether to normalize the document topic distribution in `transform`. 
772   
773          Returns 
774          ------- 
775          X_new : ndarray array of shape (n_samples, n_features_new) 
776              Transformed array. 
777          """
778          return self.fit(X, y).transform(X, normalize=normalize)

780      def _approx_bound(self, X, doc_topic_distr, sub_sampling):
781          """Estimate the variational bound. 
782   
783          Estimate the variational bound over "all documents" using only the 
784          documents passed in as X. Since log-likelihood of each word cannot 
785          be computed directly, we use this bound to estimate it. 
786   
787          Parameters 
788          ---------- 
789          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
790              Document word matrix. 
791   
792          doc_topic_distr : ndarray of shape (n_samples, n_components) 
793              Document topic distribution. In the literature, this is called 
794              gamma. 
795   
796          sub_sampling : bool, default=False 
797              Compensate for subsampling of documents. 
798              It is used in calculate bound in online learning. 
799   
800          Returns 
801          ------- 
802          score : float 
803   
804          """
805  
806          def _loglikelihood(prior, distr, dirichlet_distr, size):
807              # calculate log-likelihood
808              score = np.sum((prior - distr) * dirichlet_distr)
809              score += np.sum(gammaln(distr) - gammaln(prior))
810              score += np.sum(gammaln(prior * size) - gammaln(np.sum(distr, 1)))
811              return score
812  
813          is_sparse_x = sp.issparse(X)
814          n_samples, n_components = doc_topic_distr.shape
815          n_features = self.components_.shape[1]
816          score = 0
817  
818          dirichlet_doc_topic = _dirichlet_expectation_2d(doc_topic_distr)
819          dirichlet_component_ = _dirichlet_expectation_2d(self.components_)
820          doc_topic_prior = self.doc_topic_prior_
821          topic_word_prior = self.topic_word_prior_
822  
823          if is_sparse_x:
824              X_data = X.data
825              X_indices = X.indices
826              X_indptr = X.indptr
827  
828          # E[log p(docs | theta, beta)]
829          for idx_d in range(0, n_samples):
830              if is_sparse_x:
831                  ids = X_indices[X_indptr[idx_d] : X_indptr[idx_d + 1]]
832                  cnts = X_data[X_indptr[idx_d] : X_indptr[idx_d + 1]]
833              else:
834                  ids = np.nonzero(X[idx_d, :])[0]
835                  cnts = X[idx_d, ids]
836              temp = (
837                  dirichlet_doc_topic[idx_d, :, np.newaxis] + dirichlet_component_[:, ids]
838              )
839              norm_phi = logsumexp(temp, axis=0)
840              score += np.dot(cnts, norm_phi)
841  
842          # compute E[log p(theta | alpha) - log q(theta | gamma)]
843          score += _loglikelihood(
844              doc_topic_prior, doc_topic_distr, dirichlet_doc_topic, self.n_components
845          )
846  
847          # Compensate for the subsampling of the population of documents
848          if sub_sampling:
849              doc_ratio = float(self.total_samples) / n_samples
850              score *= doc_ratio
851  
852          # E[log p(beta | eta) - log q (beta | lambda)]
853          score += _loglikelihood(
854              topic_word_prior, self.components_, dirichlet_component_, n_features
855          )
856  
857          return score
858  
859      def score(self, X, y=None):
860          """Calculate approximate log-likelihood as score. 
861   
862          Parameters 
863          ---------- 
864          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
865              Document word matrix. 
866   
867          y : Ignored 
868              Not used, present here for API consistency by convention. 
869   
870          Returns 
871          ------- 
872          score : float 
873              Use approximate bound as score. 
874          """
875          check_is_fitted(self)
876          X = self._check_non_neg_array(
877              X, reset_n_features=False, whom="LatentDirichletAllocation.score"
878          )
879  
880          doc_topic_distr = self._unnormalized_transform(X)
881          score = self._approx_bound(X, doc_topic_distr, sub_sampling=False)
882          return score

884      def _perplexity_precomp_distr(self, X, doc_topic_distr=None, sub_sampling=False):
885          """Calculate approximate perplexity for data X with ability to accept 
886          precomputed doc_topic_distr 
887   
888          Perplexity is defined as exp(-1. * log-likelihood per word) 
889   
890          Parameters 
891          ---------- 
892          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
893              Document word matrix. 
894   
895          doc_topic_distr : ndarray of shape (n_samples, n_components), \ 
896                  default=None 
897              Document topic distribution. 
898              If it is None, it will be generated by applying transform on X. 
899   
900          Returns 
901          ------- 
902          score : float 
903              Perplexity score. 
904          """
905          if doc_topic_distr is None:
906              doc_topic_distr = self._unnormalized_transform(X)
907          else:
908              n_samples, n_components = doc_topic_distr.shape
909              if n_samples != X.shape[0]:
910                  raise ValueError(
911                      "Number of samples in X and doc_topic_distr do not match."
912                  )
913  
914              if n_components != self.n_components:
915                  raise ValueError("Number of topics does not match.")
916  
917          current_samples = X.shape[0]
918          bound = self._approx_bound(X, doc_topic_distr, sub_sampling)
919  
920          if sub_sampling:
921              word_cnt = X.sum() * (float(self.total_samples) / current_samples)
922          else:
923              word_cnt = X.sum()
924          perword_bound = bound / word_cnt
925  
926          return np.exp(-1.0 * perword_bound)
927  
928      def perplexity(self, X, sub_sampling=False):
929          """Calculate approximate perplexity for data X. 
930   
931          Perplexity is defined as exp(-1. * log-likelihood per word) 
932   
933          .. versionchanged:: 0.19 
934             *doc_topic_distr* argument has been deprecated and is ignored 
935             because user no longer has access to unnormalized distribution 
936   
937          Parameters 
938          ---------- 
939          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
940              Document word matrix. 
941   
942          sub_sampling : bool 
943              Do sub-sampling or not. 
944   
945          Returns 
946          ------- 
947          score : float 
948              Perplexity score. 
949          """
950          check_is_fitted(self)
951          X = self._check_non_neg_array(
952              X, reset_n_features=True, whom="LatentDirichletAllocation.perplexity"
953          )
954          return self._perplexity_precomp_distr(X, sub_sampling=sub_sampling)
