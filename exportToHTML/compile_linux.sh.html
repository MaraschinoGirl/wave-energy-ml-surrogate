compile_linux.sh 
1    set -e
2    
3    ARCH="$(uname -m)"
4    case $ARCH in
5        i*86) SUFFIX=x86;;
6        x86_64*) SUFFIX=amd64;;
7        *) echo >&2 "unsupported: $ARCH"; exit 1;;
8    esac
9    
10   SRC="$(dirname "$0")/.."
11   g++ -std=c++11 -shared -fPIC -O2 -D_FORTIFY_SOURCE=2 -nostartfiles --stack-protector-strong $SRC/linux_and_mac/attach.cpp -o $SRC/attach_linux_$SUFFIX.so
12
