3    from keras.src import backend

6    from keras.src.utils import io_utils

9    @keras_export("keras.callbacks.LearningRateScheduler")
10   class LearningRateScheduler(Callback):
11       """Learning rate scheduler. 
12    
13       At the beginning of every epoch, this callback gets the updated learning 
14       rate value from `schedule` function provided at `__init__`, with the current 
15       epoch and current learning rate, and applies the updated learning rate on 
16       the optimizer. 
17    
18       Args: 
19           schedule: A function that takes an epoch index (integer, indexed from 0) 
20               and current learning rate (float) as inputs and returns a new 
21               learning rate as output (float). 
22           verbose: Integer. 0: quiet, 1: log update messages. 
23    
24       Example: 
25    
26       >>> # This function keeps the initial learning rate for the first ten epochs 
27       >>> # and decreases it exponentially after that. 
28       >>> def scheduler(epoch, lr): 
29       ...     if epoch < 10: 
30       ...         return lr 
31       ...     else: 
32       ...         return lr * ops.exp(-0.1) 
33       >>> 
34       >>> model = keras.models.Sequential([keras.layers.Dense(10)]) 
35       >>> model.compile(keras.optimizers.SGD(), loss='mse') 
36       >>> round(model.optimizer.learning_rate, 5) 
37       0.01 
38    
39       >>> callback = keras.callbacks.LearningRateScheduler(scheduler) 
40       >>> history = model.fit(np.arange(100).reshape(5, 20), np.zeros(5), 
41       ...                     epochs=15, callbacks=[callback], verbose=0) 
42       >>> round(model.optimizer.learning_rate, 5) 
43       0.00607 
44    
45       """
46   
47       def __init__(self, schedule, verbose=0):
48           super().__init__()
49           self.schedule = schedule
50           self.verbose = verbose
51   
52       def on_epoch_begin(self, epoch, logs=None):
53           if not hasattr(self.model.optimizer, "learning_rate"):
54               raise ValueError('Optimizer must have a "learning_rate" attribute.')
55   
56           try:  # new API
57               learning_rate = float(
58                   backend.convert_to_numpy(self.model.optimizer.learning_rate)
59               )
60               learning_rate = self.schedule(epoch, learning_rate)
61           except TypeError:  # Support for old API for backward compatibility
62               learning_rate = self.schedule(epoch)
63   
64           if not isinstance(learning_rate, (float, np.float32, np.float64)):
65               raise ValueError(
66                   "The output of the `schedule` function should be a float. "
67                   f"Got: {learning_rate}"
68               )
69   
70           self.model.optimizer.learning_rate = learning_rate
71           if self.verbose > 0:
72               io_utils.print_msg(
73                   f"\nEpoch {epoch + 1}: LearningRateScheduler setting learning "
74                   f"rate to {learning_rate}."
75               )
76   
77       def on_epoch_end(self, epoch, logs=None):
78           logs = logs or {}
79           logs["learning_rate"] = float(
80               backend.convert_to_numpy(self.model.optimizer.learning_rate)
81           )
