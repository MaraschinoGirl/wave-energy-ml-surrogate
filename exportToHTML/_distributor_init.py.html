2    '''Helper to preload vcomp140.dll and msvcp140.dll to prevent 
3    "not found" errors. 
4     
5    Once vcomp140.dll and msvcp140.dll are 
6    preloaded, the namespace is made available to any subsequent 
7    vcomp140.dll and msvcp140.dll. This is 
8    created as part of the scripts that build the wheel. 
9    '''
10   
11   
12   import os
13   import os.path as op
14   from ctypes import WinDLL
15   
16   
17   if os.name == "nt":
18       libs_path = op.join(op.dirname(__file__), ".libs")
19       vcomp140_dll_filename = op.join(libs_path, "vcomp140.dll")
20       msvcp140_dll_filename = op.join(libs_path, "msvcp140.dll")
21       WinDLL(op.abspath(vcomp140_dll_filename))
22       WinDLL(op.abspath(msvcp140_dll_filename))
