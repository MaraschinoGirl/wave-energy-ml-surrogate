12   class GenericFrame(window.MDIChildWnd):
13       def OnCreateClient(self, cp, context):
14           # handlers for toolbar buttons
15           self.HookCommand(self.OnPrevious, 401)
16           self.HookCommand(self.OnNext, 402)
17           # It's not necessary for us to hook both of these - the
18           # common controls should fall-back all by themselves.
19           # Indeed, given we hook TTN_NEEDTEXTW, commctrl.TTN_NEEDTEXTA
20           # will not be called.
21           self.HookNotify(self.GetTTText, commctrl.TTN_NEEDTEXT)
22           self.HookNotify(self.GetTTText, commctrl.TTN_NEEDTEXTW)
23   
24           #       parent = win32ui.GetMainFrame()
25           parent = self
26           style = (
27               win32con.WS_CHILD
28               | win32con.WS_VISIBLE
29               | afxres.CBRS_SIZE_DYNAMIC
30               | afxres.CBRS_TOP
31               | afxres.CBRS_TOOLTIPS
32               | afxres.CBRS_FLYBY
33           )
34   
35           buttons = (win32ui.ID_APP_ABOUT, win32ui.ID_VIEW_INTERACTIVE)
36           bitmap = win32ui.IDB_BROWSER_HIER
37           tbid = 0xE840
38           self.toolbar = tb = win32ui.CreateToolBar(parent, style, tbid)
39           tb.LoadBitmap(bitmap)
40           tb.SetButtons(buttons)
41   
42           tb.EnableDocking(afxres.CBRS_ALIGN_ANY)
43           tb.SetWindowText("Test")
44           parent.EnableDocking(afxres.CBRS_ALIGN_ANY)
45           parent.DockControlBar(tb)
46           parent.LoadBarState("ToolbarTest")
47           window.MDIChildWnd.OnCreateClient(self, cp, context)
48           return 1
49   
50       def OnDestroy(self, msg):
51           self.SaveBarState("ToolbarTest")
52   
53       def GetTTText(self, std, extra):
54           (hwndFrom, idFrom, code) = std
55           text, hinst, flags = extra
56           if flags & commctrl.TTF_IDISHWND:
57               return  # Not handled
58           if idFrom == win32ui.ID_APP_ABOUT:
59               # our 'extra' return value needs to be the following
60               # entries from a NMTTDISPINFO[W] struct:
61               # (szText, hinst, uFlags).  None means 'don't change
62               # the value'
63               return 0, ("It works!", None, None)
64           return None  # not handled.
65   
66       def GetMessageString(self, id):
67           if id == win32ui.ID_APP_ABOUT:
68               return "Dialog Test\nTest"
69           else:
70               return self._obj_.GetMessageString(id)
71   
72       def OnSize(self, params):
73           print("OnSize called with ", params)
74   
75       def OnNext(self, id, cmd):
76           print("OnNext called")
77   
78       def OnPrevious(self, id, cmd):
79           print("OnPrevious called")
