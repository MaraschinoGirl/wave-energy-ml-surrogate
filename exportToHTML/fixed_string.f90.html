1    function sint(s) result(i)
2       implicit none
3       character(len=*) :: s
4       integer :: j, i
5       i = 0
6       do j=len(s), 1, -1
7        if (.not.((i.eq.0).and.(s(j:j).eq.' '))) then
8          i = i + ichar(s(j:j)) * 10 ** (j - 1)
9        endif
10      end do
11      return
12    end function sint
