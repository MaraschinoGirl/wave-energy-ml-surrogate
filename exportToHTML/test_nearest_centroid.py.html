37   @pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
38   def test_classification_toy(csr_container):
39       # Check classification on a toy dataset, including sparse versions.
40       X_csr = csr_container(X)
41       T_csr = csr_container(T)
42   
43       # Check classification on a toy dataset, including sparse versions.
44       clf = NearestCentroid()
45       clf.fit(X, y)
46       assert_array_equal(clf.predict(T), true_result)
47       assert_array_almost_equal(clf.decision_function(T), true_discriminant_scores)
48       assert_array_almost_equal(clf.predict_proba(T), true_proba)
49   
50       # Test uniform priors
51       clf = NearestCentroid(priors="uniform")
52       clf.fit(X, y)
53       assert_array_equal(clf.predict(T), true_result)
54       assert_array_almost_equal(clf.decision_function(T), true_discriminant_scores)
55       assert_array_almost_equal(clf.predict_proba(T), true_proba)
56   
57       clf = NearestCentroid(priors="empirical")
58       clf.fit(X, y)
59       assert_array_equal(clf.predict(T), true_result)
60       assert_array_almost_equal(clf.decision_function(T), true_discriminant_scores)
61       assert_array_almost_equal(clf.predict_proba(T), true_proba)
62   
63       # Test custom priors
64       clf = NearestCentroid(priors=[0.25, 0.75])
65       clf.fit(X, y)
66       assert_array_equal(clf.predict(T), true_result_prior1)
67   
68       # Same test, but with a sparse matrix to fit and test.
69       clf = NearestCentroid()
70       clf.fit(X_csr, y)
71       assert_array_equal(clf.predict(T_csr), true_result)
72   
73       # Fit with sparse, test with non-sparse
74       clf = NearestCentroid()
75       clf.fit(X_csr, y)
76       assert_array_equal(clf.predict(T), true_result)
77   
78       # Fit with non-sparse, test with sparse
79       clf = NearestCentroid()
80       clf.fit(X, y)
81       assert_array_equal(clf.predict(T_csr), true_result)
82   
83       # Fit and predict with non-CSR sparse matrices
84       clf = NearestCentroid()
85       clf.fit(X_csr.tocoo(), y)
86       assert_array_equal(clf.predict(T_csr.tolil()), true_result)

89   def test_iris():
90       # Check consistency on dataset iris.
91       for metric in ("euclidean", "manhattan"):
92           clf = NearestCentroid(metric=metric).fit(iris.data, iris.target)
93           score = np.mean(clf.predict(iris.data) == iris.target)
94           assert score > 0.9, "Failed with score = " + str(score)

97   def test_iris_shrinkage():
98       # Check consistency on dataset iris, when using shrinkage.
99       for metric in ("euclidean", "manhattan"):
100          for shrink_threshold in [None, 0.1, 0.5]:
101              clf = NearestCentroid(metric=metric, shrink_threshold=shrink_threshold)
102              clf = clf.fit(iris.data, iris.target)
103              score = np.mean(clf.predict(iris.data) == iris.target)
104              assert score > 0.8, "Failed with score = " + str(score)

107  def test_pickle():
108      import pickle
109  
110      # classification
111      obj = NearestCentroid()
112      obj.fit(iris.data, iris.target)
113      score = obj.score(iris.data, iris.target)
114      s = pickle.dumps(obj)
115  
116      obj2 = pickle.loads(s)
117      assert type(obj2) == obj.__class__
118      score2 = obj2.score(iris.data, iris.target)
119      assert_array_equal(
120          score,
121          score2,
122          "Failed to generate same score after pickling (classification).",
123      )

126  def test_shrinkage_correct():
127      # Ensure that the shrinking is correct.
128      # The expected result is calculated by R (pamr),
129      # which is implemented by the author of the original paper.
130      # (One need to modify the code to output the new centroid in pamr.predict)
131  
132      X = np.array([[0, 1], [1, 0], [1, 1], [2, 0], [6, 8]])
133      y = np.array([1, 1, 2, 2, 2])
134      clf = NearestCentroid(shrink_threshold=0.1)
135      clf.fit(X, y)
136      expected_result = np.array([[0.7787310, 0.8545292], [2.814179, 2.763647]])
137      np.testing.assert_array_almost_equal(clf.centroids_, expected_result)

140  def test_shrinkage_threshold_decoded_y():
141      clf = NearestCentroid(shrink_threshold=0.01)
142      y_ind = np.asarray(y)
143      y_ind[y_ind == -1] = 0
144      clf.fit(X, y_ind)
145      centroid_encoded = clf.centroids_
146      clf.fit(X, y)
147      assert_array_equal(centroid_encoded, clf.centroids_)

150  def test_predict_translated_data():
151      # Test that NearestCentroid gives same results on translated data
152  
153      rng = np.random.RandomState(0)
154      X = rng.rand(50, 50)
155      y = rng.randint(0, 3, 50)
156      noise = rng.rand(50)
157      clf = NearestCentroid(shrink_threshold=0.1)
158      clf.fit(X, y)
159      y_init = clf.predict(X)
160      clf = NearestCentroid(shrink_threshold=0.1)
161      X_noise = X + noise
162      clf.fit(X_noise, y)
163      y_translate = clf.predict(X_noise)
164      assert_array_equal(y_init, y_translate)

201  def test_warn_non_normalized_priors():
202      """Check that we raise a warning and normalize the user-defined priors when they 
203      don't sum to 1. 
204      """
205      priors = [2, 4]
206      clf = NearestCentroid(priors=priors)
207      with pytest.warns(
208          UserWarning,
209          match="The priors do not sum to 1. Normalizing such that it sums to one.",
210      ):
211          clf.fit(X, y)
212  
213      assert_allclose(clf.class_prior_, np.asarray(priors) / np.asarray(priors).sum())
