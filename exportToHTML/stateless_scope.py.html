1    from keras.src.api_export import keras_export
2    from keras.src.backend.common import global_state
3    
4    
5    @keras_export("keras.StatelessScope")
6    class StatelessScope:
7        """Scope to prevent any update to Keras Variables. 
8     
9        The values of variables to be used inside the scope 
10       should be passed via the `state_mapping` argument, a 
11       list of tuples `(k, v)` where `k` is a `Variable` 
12       and `v` is the intended value for this variable 
13       (a backend tensor). 
14    
15       Updated values can be collected on scope exit via 
16       `value = scope.get_current_value(variable)`. No updates 
17       will be applied in-place to any variables for the duration 
18       of the scope. 
19    
20       Example: 
21    
22       ```python 
23       state_mapping = [(k, ops.ones(k.shape, k.dtype)) for k in model.weights] 
24       with keras.StatelessScope(state_mapping) as scope: 
25           outputs = model.some_function(inputs) 
26    
27       # All model variables remain unchanged. Their new values can be 
28       # collected via: 
29       for k in model.weights: 
30           new_value = scope.get_current_value(k) 
31           print(f"New value for {k}: {new_value}) 
32       ``` 
33       """
34   
35       def __init__(
36           self,
37           state_mapping=None,
38           collect_losses=False,
39           initialize_variables=True,
40       ):
41           from keras.src import backend
42           from keras.src.backend.common.variables import Variable
43   
44           self.collect_losses = collect_losses
45           self.initialize_variables = initialize_variables
46           self.losses = []
47           self.state_mapping = {}
48           state_mapping = state_mapping or {}
49           for k, v in state_mapping:
50               if not isinstance(k, Variable):
51                   raise ValueError(
52                       "Invalid reference variable in StatelessScope: "
53                       "all keys in argument `mapping` must be Variable "
54                       f"instances. Received instead: {k}"
55                   )
56               if isinstance(v, Variable):
57                   v = backend.cast(v.value, dtype=k.dtype)
58               else:
59                   v = backend.convert_to_tensor(v, dtype=k.dtype)
60               if k.shape != v.shape:
61                   raise ValueError(
62                       "Invalid variable value in StatelessScope: "
63                       "all values in argument `mapping` must be tensors with "
64                       "a shape that matches the corresponding variable shape. "
65                       f"For variable {k}, received invalid value {v} with shape "
66                       f"{v.shape}."
67                   )
68               self.state_mapping[id(k)] = v
69   
70       def __enter__(self):
71           self.original_scope = get_stateless_scope()
72           global_state.set_global_attribute("stateless_scope", self)
73           return self
74   
75       def add_loss(self, loss):
76           self.losses.append(loss)
77   
78       def add_update(self, update):
79           variable, value = update
80           self.state_mapping[id(variable)] = value
81   
82       def get_current_value(self, variable):
83           return self.state_mapping.get(id(variable), None)
84   
85       def __exit__(self, *args, **kwargs):
86           global_state.set_global_attribute(
87               "stateless_scope", self.original_scope
88           )
89           if self.original_scope is None and self.initialize_variables:
90               # We're back in eager scope;
91               # if any variables were created within the stateless
92               # scope, we initialize them here.
93               from keras.src.backend.common.variables import (
94                   initialize_all_variables,
95               )
96   
97               initialize_all_variables()
98   
99   
100  def in_stateless_scope():
101      return global_state.get_global_attribute("stateless_scope") is not None
102  
103  
104  def get_stateless_scope():
105      return global_state.get_global_attribute("stateless_scope")
