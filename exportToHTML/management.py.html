45   class PtyWithClients:
46       """A pty object with associated clients."""
47   
48       term_name: str | None
49   
50       def __init__(self, argv: Any, env: dict[str, str] | None = None, cwd: str | None = None):
51           """Initialize the pty."""
52           self.clients: list[Any] = []
53           # Use read_buffer to store historical messages for reconnection
54           self.read_buffer: deque[str] = deque([], maxlen=1000)
55           kwargs = {"argv": argv, "env": env or [], "cwd": cwd}
56           if preexec_fn is not None:
57               kwargs["preexec_fn"] = preexec_fn
58           self.ptyproc = PtyProcessUnicode.spawn(**kwargs)
59           # The output might not be strictly UTF-8 encoded, so
60           # we replace the inner decoder of PtyProcessUnicode
61           # to allow non-strict decode.
62           self.ptyproc.decoder = codecs.getincrementaldecoder("utf-8")(errors="replace")
63   
64       def resize_to_smallest(self) -> None:
65           """Set the terminal size to that of the smallest client dimensions. 
66    
67           A terminal not using the full space available is much nicer than a 
68           terminal trying to use more than the available space, so we keep it 
69           sized to the smallest client. 
70           """
71           minrows = mincols = 10001
72           for client in self.clients:
73               rows, cols = client.size
74               if rows is not None and rows < minrows:
75                   minrows = rows
76               if cols is not None and cols < mincols:
77                   mincols = cols
78   
79           if minrows == 10001 or mincols == 10001:
80               return
81   
82           rows, cols = self.ptyproc.getwinsize()
83           if (rows, cols) != (minrows, mincols):
84               self.ptyproc.setwinsize(minrows, mincols)
85   
86       def kill(self, sig: int = signal.SIGTERM) -> None:
87           """Send a signal to the process in the pty"""
88           self.ptyproc.kill(sig)
89   
90       def killpg(self, sig: int = signal.SIGTERM) -> Any:
91           """Send a signal to the process group of the process in the pty"""
92           if os.name == "nt":
93               return self.ptyproc.kill(sig)
94           pgid = os.getpgid(self.ptyproc.pid)
95           os.killpg(pgid, sig)
96           return None
97   
98       async def terminate(self, force: bool = False) -> bool:
99           """This forces a child process to terminate. It starts nicely with 
100          SIGHUP and SIGINT. If "force" is True then moves onto SIGKILL. This 
101          returns True if the child was terminated. This returns False if the 
102          child could not be terminated."""
103          if os.name == "nt":
104              signals = [signal.SIGINT, signal.SIGTERM]
105          else:
106              signals = [signal.SIGHUP, signal.SIGCONT, signal.SIGINT, signal.SIGTERM]
107  
108          _ = IOLoop.current()
109  
110          def sleep() -> Coroutine[Any, Any, None]:
111              """Sleep to allow the terminal to exit gracefully."""
112              return asyncio.sleep(self.ptyproc.delayafterterminate)
113  
114          if not self.ptyproc.isalive():
115              return True
116          try:
117              for sig in signals:
118                  self.kill(sig)
119                  await sleep()
120                  if not self.ptyproc.isalive():
121                      return True
122              if force:
123                  self.kill(signal.SIGKILL)
124                  await sleep()
125                  return bool(not self.ptyproc.isalive())
126              return False
127          except OSError:
128              # I think there are kernel timing issues that sometimes cause
129              # this to happen. I think isalive() reports True, but the
130              # process is dead to the kernel.
131              # Make one last attempt to see if the kernel is up to date.
132              await sleep()
133              return bool(not self.ptyproc.isalive())

136  def _update_removing(target: Any, changes: Any) -> None:
137      """Like dict.update(), but remove keys where the value is None."""
138      for k, v in changes.items():
139          if v is None:
140              target.pop(k, None)
141          else:
142              target[k] = v

145  def _poll(fd: int, timeout: float = 0.1) -> list[tuple[int, int]]:
146      """Poll using poll() on posix systems and select() elsewhere (e.g., Windows)"""
147      if os.name == "posix":
148          poller = select.poll()
149          poller.register(
150              fd, select.POLLIN | select.POLLPRI | select.POLLHUP | select.POLLERR
151          )  # read-only
152          return poller.poll(timeout * 1000)  # milliseconds
153      # poll() not supported on Windows
154      r, _, _ = select.select([fd], [], [], timeout)
155      return r

158  class TermManagerBase:
159      """Base class for a terminal manager."""
160  
161      def __init__(
162          self,
163          shell_command: str,
164          server_url: str = "",
165          term_settings: Any = None,
166          extra_env: Any = None,
167          ioloop: Any = None,
168          blocking_io_executor: Any = None,
169      ):
170          """Initialize the manager."""
171          self.shell_command = shell_command
172          self.server_url = server_url
173          self.term_settings = term_settings or {}
174          self.extra_env = extra_env
175          self.log = logging.getLogger(__name__)
176  
177          self.ptys_by_fd: dict[int, PtyWithClients] = {}
178  
179          if blocking_io_executor is None:
180              self._blocking_io_executor_is_external = False
181              self.blocking_io_executor = futures.ThreadPoolExecutor(max_workers=1)
182          else:
183              self._blocking_io_executor_is_external = True
184              self.blocking_io_executor = blocking_io_executor
185  
186          if ioloop is not None:
187              warnings.warn(
188                  f"Setting {self.__class__.__name__}.ioloop is deprecated and ignored",
189                  DeprecationWarning,
190                  stacklevel=2,
191              )
192  
193      def make_term_env(
194          self,
195          height: int = 25,
196          width: int = 80,
197          winheight: int = 0,
198          winwidth: int = 0,
199          **kwargs: Any,
200      ) -> dict[str, str]:
201          """Build the environment variables for the process in the terminal."""
202          env = os.environ.copy()
203          # ignore any previously set TERM
204          # TERM is set according to xterm.js capabilities
205          env["TERM"] = self.term_settings.get("type", DEFAULT_TERM_TYPE)
206          dimensions = "%dx%d" % (width, height)
207          if winwidth and winheight:
208              dimensions += ";%dx%d" % (winwidth, winheight)
209          env[ENV_PREFIX + "DIMENSIONS"] = dimensions
210          env["COLUMNS"] = str(width)
211          env["LINES"] = str(height)
212  
213          if self.server_url:
214              env[ENV_PREFIX + "URL"] = self.server_url
215  
216          if self.extra_env:
217              _update_removing(env, self.extra_env)
218  
219          term_env = kwargs.get("extra_env", {})
220          if term_env and isinstance(term_env, dict):
221              _update_removing(env, term_env)
222  
223          return env
224  
225      def new_terminal(self, **kwargs: Any) -> PtyWithClients:
226          """Make a new terminal, return a :class:`PtyWithClients` instance."""
227          options = self.term_settings.copy()
228          options["shell_command"] = self.shell_command
229          options.update(kwargs)
230          argv = options["shell_command"]
231          env = self.make_term_env(**options)
232          cwd = options.get("cwd", None)
233          return PtyWithClients(argv, env, cwd)
234  
235      def start_reading(self, ptywclients: PtyWithClients) -> None:
236          """Connect a terminal to the tornado event loop to read data from it."""
237          fd = ptywclients.ptyproc.fd
238          self.ptys_by_fd[fd] = ptywclients
239          loop = IOLoop.current()
240          loop.add_handler(fd, self.pty_read, loop.READ)
241  
242      def on_eof(self, ptywclients: PtyWithClients) -> None:
243          """Called when the pty has closed."""
244          # Stop trying to read from that terminal
245          fd = ptywclients.ptyproc.fd
246          self.log.info("EOF on FD %d; stopping reading", fd)
247          del self.ptys_by_fd[fd]
248          IOLoop.current().remove_handler(fd)
249  
250          # This closes the fd, and should result in the process being reaped.
251          ptywclients.ptyproc.close()
252  
253      def pty_read(self, fd: int, events: Any = None) -> None:
254          """Called by the event loop when there is pty data ready to read."""
255          # prevent blocking on fd
256          if not _poll(fd, timeout=0.1):  # 100ms
257              self.log.debug("Spurious pty_read() on fd %s", fd)
258              return
259          ptywclients = self.ptys_by_fd[fd]
260          try:
261              self.pre_pty_read_hook(ptywclients)
262              s = ptywclients.ptyproc.read(65536)
263              ptywclients.read_buffer.append(s)
264              for client in ptywclients.clients:
265                  client.on_pty_read(s)
266          except EOFError:
267              self.on_eof(ptywclients)
268              for client in ptywclients.clients:
269                  client.on_pty_died()
270  
271      def pre_pty_read_hook(self, ptywclients: PtyWithClients) -> None:
272          """Hook before pty read, subclass can patch something into ptywclients when pty_read"""
273  
274      def get_terminal(self, url_component: Any = None) -> PtyWithClients:
275          """Override in a subclass to give a terminal to a new websocket connection 
276   
277          The :class:`TermSocket` handler works with zero or one URL components 
278          (capturing groups in the URL spec regex). If it receives one, it is 
279          passed as the ``url_component`` parameter; otherwise, this is None. 
280          """
281          raise NotImplementedError
282  
283      def client_disconnected(self, websocket: Any) -> None:
284          """Override this to e.g. kill terminals on client disconnection."""
285  
286      async def shutdown(self) -> None:
287          """Shutdown the manager."""
288          await self.kill_all()
289          if not self._blocking_io_executor_is_external:
290              self.blocking_io_executor.shutdown(wait=False, cancel_futures=True)  # type:ignore[call-arg]
291  
292      async def kill_all(self) -> None:
293          """Kill all terminals."""
294          futures = []
295          for term in self.ptys_by_fd.values():
296              futures.append(term.terminate(force=True))
297          # wait for futures to finish
298          if futures:
299              await asyncio.gather(*futures)

302  class SingleTermManager(TermManagerBase):
303      """All connections to the websocket share a common terminal."""
304  
305      def __init__(self, **kwargs: Any) -> None:
306          """Initialize the manager."""
307          super().__init__(**kwargs)
308          self.terminal: PtyWithClients | None = None
309  
310      def get_terminal(self, url_component: Any = None) -> PtyWithClients:
311          """ "Get the singleton terminal."""
312          if self.terminal is None:
313              self.terminal = self.new_terminal()
314              self.start_reading(self.terminal)
315          return self.terminal
316  
317      async def kill_all(self) -> None:
318          """Kill the singletone terminal."""
319          await super().kill_all()
320          self.terminal = None

323  class MaxTerminalsReached(Exception):
324      """An error raised when we exceed the max number of terminals."""
325  
326      def __init__(self, max_terminals: int) -> None:
327          """Initialize the error."""
328          self.max_terminals = max_terminals
329  
330      def __str__(self) -> str:
331          """The string representation of the error."""
332          return "Cannot create more than %d terminals" % self.max_terminals

335  class UniqueTermManager(TermManagerBase):
336      """Give each websocket a unique terminal to use."""
337  
338      def __init__(self, max_terminals: int | None = None, **kwargs: Any) -> None:
339          """Initialize the manager."""
340          super().__init__(**kwargs)
341          self.max_terminals = max_terminals
342  
343      def get_terminal(self, url_component: Any = None) -> PtyWithClients:
344          """Get a terminal from the manager."""
345          if self.max_terminals and len(self.ptys_by_fd) >= self.max_terminals:
346              raise MaxTerminalsReached(self.max_terminals)
347  
348          term = self.new_terminal()
349          self.start_reading(term)
350          return term
351  
352      def client_disconnected(self, websocket: TermSocket) -> None:
353          """Send terminal SIGHUP when client disconnects."""
354          self.log.info("Websocket closed, sending SIGHUP to terminal.")
355          if websocket.terminal:
356              if os.name == "nt":
357                  websocket.terminal.kill()
358                  # Immediately call the pty reader to process
359                  # the eof and free up space
360                  self.pty_read(websocket.terminal.ptyproc.fd)
361                  return
362              websocket.terminal.killpg(signal.SIGHUP)

365  class NamedTermManager(TermManagerBase):
366      """Share terminals between websockets connected to the same endpoint."""
367  
368      def __init__(self, max_terminals: Any = None, **kwargs: Any) -> None:
369          """Initialize the manager."""
370          super().__init__(**kwargs)
371          self.max_terminals = max_terminals
372          self.terminals: dict[str, PtyWithClients] = {}
373  
374      def get_terminal(self, term_name: str) -> PtyWithClients:  # type:ignore[override]
375          """Get or create a terminal by name."""
376          assert term_name is not None
377  
378          if term_name in self.terminals:
379              return self.terminals[term_name]
380  
381          if self.max_terminals and len(self.terminals) >= self.max_terminals:
382              raise MaxTerminalsReached(self.max_terminals)
383  
384          # Create new terminal
385          self.log.info("New terminal with specified name: %s", term_name)
386          term = self.new_terminal()
387          term.term_name = term_name
388          self.terminals[term_name] = term
389          self.start_reading(term)
390          return term
391  
392      name_template = "%d"
393  
394      def _next_available_name(self) -> str | None:
395          for n in itertools.count(start=1):
396              name = self.name_template % n
397              if name not in self.terminals:
398                  return name
399          return None
400  
401      def new_named_terminal(self, **kwargs: Any) -> tuple[str, PtyWithClients]:
402          """Create a new named terminal with an automatic name."""
403          name = kwargs["name"] if "name" in kwargs else self._next_available_name()
404          term = self.new_terminal(**kwargs)
405          self.log.info("New terminal with automatic name: %s", name)
406          term.term_name = name
407          self.terminals[name] = term
408          self.start_reading(term)
409          return name, term
410  
411      def kill(self, name: str, sig: int = signal.SIGTERM) -> None:
412          """Kill a terminal by name."""
413          term = self.terminals[name]
414          term.kill(sig)  # This should lead to an EOF
415  
416      async def terminate(self, name: str, force: bool = False) -> None:
417          """Terminate a terminal by name."""
418          term = self.terminals[name]
419          await term.terminate(force=force)
420  
421      def on_eof(self, ptywclients: PtyWithClients) -> None:
422          """Handle end of file for a pty with clients."""
423          super().on_eof(ptywclients)
424          name = ptywclients.term_name
425          self.log.info("Terminal %s closed", name)
426          assert name is not None
427          self.terminals.pop(name, None)
428  
429      async def kill_all(self) -> None:
430          """Kill all terminals."""
431          await super().kill_all()
432          self.terminals = {}
