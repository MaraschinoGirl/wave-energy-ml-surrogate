63   @pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
64   @pytest.mark.parametrize("method", ["sigmoid", "isotonic"])
65   @pytest.mark.parametrize("ensemble", [True, False])
66   def test_calibration(data, method, csr_container, ensemble):
67       # Test calibration objects with isotonic and sigmoid
68       n_samples = N_SAMPLES // 2
69       X, y = data
70       sample_weight = np.random.RandomState(seed=42).uniform(size=y.size)
71   
72       X -= X.min()  # MultinomialNB only allows positive X
73   
74       # split train and test
75       X_train, y_train, sw_train = X[:n_samples], y[:n_samples], sample_weight[:n_samples]
76       X_test, y_test = X[n_samples:], y[n_samples:]
77   
78       # Naive-Bayes
79       clf = MultinomialNB().fit(X_train, y_train, sample_weight=sw_train)
80       prob_pos_clf = clf.predict_proba(X_test)[:, 1]
81   
82       cal_clf = CalibratedClassifierCV(clf, cv=y.size + 1, ensemble=ensemble)
83       with pytest.raises(ValueError):
84           cal_clf.fit(X, y)
85   
86       # Naive Bayes with calibration
87       for this_X_train, this_X_test in [
88           (X_train, X_test),
89           (csr_container(X_train), csr_container(X_test)),
90       ]:
91           cal_clf = CalibratedClassifierCV(clf, method=method, cv=5, ensemble=ensemble)
92           # Note that this fit overwrites the fit on the entire training
93           # set
94           cal_clf.fit(this_X_train, y_train, sample_weight=sw_train)
95           prob_pos_cal_clf = cal_clf.predict_proba(this_X_test)[:, 1]
96   
97           # Check that brier score has improved after calibration
98           assert brier_score_loss(y_test, prob_pos_clf) > brier_score_loss(
99               y_test, prob_pos_cal_clf
100          )
101  
102          # Check invariance against relabeling [0, 1] -> [1, 2]
103          cal_clf.fit(this_X_train, y_train + 1, sample_weight=sw_train)
104          prob_pos_cal_clf_relabeled = cal_clf.predict_proba(this_X_test)[:, 1]
105          assert_array_almost_equal(prob_pos_cal_clf, prob_pos_cal_clf_relabeled)
106  
107          # Check invariance against relabeling [0, 1] -> [-1, 1]
108          cal_clf.fit(this_X_train, 2 * y_train - 1, sample_weight=sw_train)
109          prob_pos_cal_clf_relabeled = cal_clf.predict_proba(this_X_test)[:, 1]
110          assert_array_almost_equal(prob_pos_cal_clf, prob_pos_cal_clf_relabeled)
111  
112          # Check invariance against relabeling [0, 1] -> [1, 0]
113          cal_clf.fit(this_X_train, (y_train + 1) % 2, sample_weight=sw_train)
114          prob_pos_cal_clf_relabeled = cal_clf.predict_proba(this_X_test)[:, 1]
115          if method == "sigmoid":
116              assert_array_almost_equal(prob_pos_cal_clf, 1 - prob_pos_cal_clf_relabeled)
117          else:
118              # Isotonic calibration is not invariant against relabeling
119              # but should improve in both cases
120              assert brier_score_loss(y_test, prob_pos_clf) > brier_score_loss(
121                  (y_test + 1) % 2, prob_pos_cal_clf_relabeled
122              )

167  def test_sample_weight(data, method, ensemble):
168      n_samples = N_SAMPLES // 2
169      X, y = data
170  
171      sample_weight = np.random.RandomState(seed=42).uniform(size=len(y))
172      X_train, y_train, sw_train = X[:n_samples], y[:n_samples], sample_weight[:n_samples]
173      X_test = X[n_samples:]
174  
175      estimator = LinearSVC(random_state=42)
176      calibrated_clf = CalibratedClassifierCV(estimator, method=method, ensemble=ensemble)
177      calibrated_clf.fit(X_train, y_train, sample_weight=sw_train)
178      probs_with_sw = calibrated_clf.predict_proba(X_test)
179  
180      # As the weights are used for the calibration, they should still yield
181      # different predictions
182      calibrated_clf.fit(X_train, y_train)
183      probs_without_sw = calibrated_clf.predict_proba(X_test)
184  
185      diff = np.linalg.norm(probs_with_sw - probs_without_sw)
186      assert diff > 0.1

213  @pytest.mark.parametrize("method", ["sigmoid", "isotonic"])
214  @pytest.mark.parametrize("ensemble", [True, False])
215  # increase the number of RNG seeds to assess the statistical stability of this
216  # test:
217  @pytest.mark.parametrize("seed", range(2))
218  def test_calibration_multiclass(method, ensemble, seed):
219      def multiclass_brier(y_true, proba_pred, n_classes):
220          Y_onehot = np.eye(n_classes)[y_true]
221          return np.sum((Y_onehot - proba_pred) ** 2) / Y_onehot.shape[0]
222  
223      # Test calibration for multiclass with classifier that implements
224      # only decision function.
225      clf = LinearSVC(random_state=7)
226      X, y = make_blobs(
227          n_samples=500, n_features=100, random_state=seed, centers=10, cluster_std=15.0
228      )
229  
230      # Use an unbalanced dataset by collapsing 8 clusters into one class
231      # to make the naive calibration based on a softmax more unlikely
232      # to work.
233      y[y > 2] = 2
234      n_classes = np.unique(y).shape[0]
235      X_train, y_train = X[::2], y[::2]
236      X_test, y_test = X[1::2], y[1::2]
237  
238      clf.fit(X_train, y_train)
239  
240      cal_clf = CalibratedClassifierCV(clf, method=method, cv=5, ensemble=ensemble)
241      cal_clf.fit(X_train, y_train)
242      probas = cal_clf.predict_proba(X_test)
243      # Check probabilities sum to 1
244      assert_allclose(np.sum(probas, axis=1), np.ones(len(X_test)))
245  
246      # Check that the dataset is not too trivial, otherwise it's hard
247      # to get interesting calibration data during the internal
248      # cross-validation loop.
249      assert 0.65 < clf.score(X_test, y_test) < 0.95
250  
251      # Check that the accuracy of the calibrated model is never degraded
252      # too much compared to the original classifier.
253      assert cal_clf.score(X_test, y_test) > 0.95 * clf.score(X_test, y_test)
254  
255      # Check that Brier loss of calibrated classifier is smaller than
256      # loss obtained by naively turning OvR decision function to
257      # probabilities via a softmax
258      uncalibrated_brier = multiclass_brier(
259          y_test, softmax(clf.decision_function(X_test)), n_classes=n_classes
260      )
261      calibrated_brier = multiclass_brier(y_test, probas, n_classes=n_classes)
262  
263      assert calibrated_brier < 1.1 * uncalibrated_brier
264  
265      # Test that calibration of a multiclass classifier decreases log-loss
266      # for RandomForestClassifier
267      clf = RandomForestClassifier(n_estimators=30, random_state=42)
268      clf.fit(X_train, y_train)
269      clf_probs = clf.predict_proba(X_test)
270      uncalibrated_brier = multiclass_brier(y_test, clf_probs, n_classes=n_classes)
271  
272      cal_clf = CalibratedClassifierCV(clf, method=method, cv=5, ensemble=ensemble)
273      cal_clf.fit(X_train, y_train)
274      cal_clf_probs = cal_clf.predict_proba(X_test)
275      calibrated_brier = multiclass_brier(y_test, cal_clf_probs, n_classes=n_classes)
276      assert calibrated_brier < 1.1 * uncalibrated_brier

279  def test_calibration_zero_probability():
280      # Test an edge case where _CalibratedClassifier avoids numerical errors
281      # in the multiclass normalization step if all the calibrators output
282      # are zero all at once for a given sample and instead fallback to uniform
283      # probabilities.
284      class ZeroCalibrator:
285          # This function is called from _CalibratedClassifier.predict_proba.
286          def predict(self, X):
287              return np.zeros(X.shape[0])
288  
289      X, y = make_blobs(
290          n_samples=50, n_features=10, random_state=7, centers=10, cluster_std=15.0
291      )
292      clf = DummyClassifier().fit(X, y)
293      calibrator = ZeroCalibrator()
294      cal_clf = _CalibratedClassifier(
295          estimator=clf, calibrators=[calibrator], classes=clf.classes_
296      )
297  
298      probas = cal_clf.predict_proba(X)
299  
300      # Check that all probabilities are uniformly 1. / clf.n_classes_
301      assert_allclose(probas, 1.0 / clf.n_classes_)

305  @pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
306  def test_calibration_prefit(csr_container):
307      """Test calibration for prefitted classifiers"""
308      # TODO(1.8): Remove cv="prefit" options here and the @ignore_warnings of the test
309      n_samples = 50
310      X, y = make_classification(n_samples=3 * n_samples, n_features=6, random_state=42)
311      sample_weight = np.random.RandomState(seed=42).uniform(size=y.size)
312  
313      X -= X.min()  # MultinomialNB only allows positive X
314  
315      # split train and test
316      X_train, y_train, sw_train = X[:n_samples], y[:n_samples], sample_weight[:n_samples]
317      X_calib, y_calib, sw_calib = (
318          X[n_samples : 2 * n_samples],
319          y[n_samples : 2 * n_samples],
320          sample_weight[n_samples : 2 * n_samples],
321      )
322      X_test, y_test = X[2 * n_samples :], y[2 * n_samples :]
323  
324      # Naive-Bayes
325      clf = MultinomialNB()
326      # Check error if clf not prefit
327      unfit_clf = CalibratedClassifierCV(clf, cv="prefit")
328      with pytest.raises(NotFittedError):
329          unfit_clf.fit(X_calib, y_calib)
330  
331      clf.fit(X_train, y_train, sw_train)
332      prob_pos_clf = clf.predict_proba(X_test)[:, 1]
333  
334      # Naive Bayes with calibration
335      for this_X_calib, this_X_test in [
336          (X_calib, X_test),
337          (csr_container(X_calib), csr_container(X_test)),
338      ]:
339          for method in ["isotonic", "sigmoid"]:
340              cal_clf_prefit = CalibratedClassifierCV(clf, method=method, cv="prefit")
341              cal_clf_frozen = CalibratedClassifierCV(FrozenEstimator(clf), method=method)
342  
343              for sw in [sw_calib, None]:
344                  cal_clf_prefit.fit(this_X_calib, y_calib, sample_weight=sw)
345                  cal_clf_frozen.fit(this_X_calib, y_calib, sample_weight=sw)
346  
347                  y_prob_prefit = cal_clf_prefit.predict_proba(this_X_test)
348                  y_prob_frozen = cal_clf_frozen.predict_proba(this_X_test)
349                  y_pred_prefit = cal_clf_prefit.predict(this_X_test)
350                  y_pred_frozen = cal_clf_frozen.predict(this_X_test)
351                  prob_pos_cal_clf_prefit = y_prob_prefit[:, 1]
352                  prob_pos_cal_clf_frozen = y_prob_frozen[:, 1]
353                  assert_array_equal(y_pred_prefit, y_pred_frozen)
354                  assert_array_equal(
355                      y_pred_prefit, np.array([0, 1])[np.argmax(y_prob_prefit, axis=1)]
356                  )
357                  assert brier_score_loss(y_test, prob_pos_clf) > brier_score_loss(
358                      y_test, prob_pos_cal_clf_frozen
359                  )

362  @pytest.mark.parametrize("method", ["sigmoid", "isotonic"])
363  def test_calibration_ensemble_false(data, method):
364      # Test that `ensemble=False` is the same as using predictions from
365      # `cross_val_predict` to train calibrator.
366      X, y = data
367      clf = LinearSVC(random_state=7)
368  
369      cal_clf = CalibratedClassifierCV(clf, method=method, cv=3, ensemble=False)
370      cal_clf.fit(X, y)
371      cal_probas = cal_clf.predict_proba(X)
372  
373      # Get probas manually
374      unbiased_preds = cross_val_predict(clf, X, y, cv=3, method="decision_function")
375      if method == "isotonic":
376          calibrator = IsotonicRegression(out_of_bounds="clip")
377      else:
378          calibrator = _SigmoidCalibration()
379      calibrator.fit(unbiased_preds, y)
380      # Use `clf` fit on all data
381      clf.fit(X, y)
382      clf_df = clf.decision_function(X)
383      manual_probas = calibrator.predict(clf_df)
384      assert_allclose(cal_probas[:, 1], manual_probas)

387  def test_sigmoid_calibration():
388      """Test calibration values with Platt sigmoid model"""
389      exF = np.array([5, -4, 1.0])
390      exY = np.array([1, -1, -1])

404  def test_calibration_curve():
405      """Check calibration_curve function"""
406      y_true = np.array([0, 0, 0, 1, 1, 1])
407      y_pred = np.array([0.0, 0.1, 0.2, 0.8, 0.9, 1.0])
408      prob_true, prob_pred = calibration_curve(y_true, y_pred, n_bins=2)
409      assert len(prob_true) == len(prob_pred)
410      assert len(prob_true) == 2
411      assert_almost_equal(prob_true, [0, 1])
412      assert_almost_equal(prob_pred, [0.1, 0.9])
413  
414      # Probabilities outside [0, 1] should not be accepted at all.
415      with pytest.raises(ValueError):
416          calibration_curve([1], [-0.1])
417  
418      # test that quantiles work as expected
419      y_true2 = np.array([0, 0, 0, 0, 1, 1])
420      y_pred2 = np.array([0.0, 0.1, 0.2, 0.5, 0.9, 1.0])
421      prob_true_quantile, prob_pred_quantile = calibration_curve(
422          y_true2, y_pred2, n_bins=2, strategy="quantile"
423      )
424  
425      assert len(prob_true_quantile) == len(prob_pred_quantile)
426      assert len(prob_true_quantile) == 2
427      assert_almost_equal(prob_true_quantile, [0, 2 / 3])
428      assert_almost_equal(prob_pred_quantile, [0.1, 0.8])
429  
430      # Check that error is raised when invalid strategy is selected
431      with pytest.raises(ValueError):
432          calibration_curve(y_true2, y_pred2, strategy="percentile")

465  @pytest.mark.parametrize("ensemble", [True, False])
466  def test_calibration_less_classes(ensemble):
467      # Test to check calibration works fine when train set in a test-train
468      # split does not contain all classes
469      # In 1st split, train is missing class 0
470      # In 3rd split, train is missing class 3
471      X = np.random.randn(12, 5)
472      y = [0, 0, 0, 1] + [1, 1, 2, 2] + [2, 3, 3, 3]
473      clf = DecisionTreeClassifier(random_state=7)
474      cal_clf = CalibratedClassifierCV(
475          clf, method="sigmoid", cv=KFold(3), ensemble=ensemble
476      )
477      cal_clf.fit(X, y)
478  
479      if ensemble:
480          classes = np.arange(4)
481          for calib_i, class_i in zip([0, 2], [0, 3]):
482              proba = cal_clf.calibrated_classifiers_[calib_i].predict_proba(X)
483              # Check that the unobserved class has proba=0
484              assert_array_equal(proba[:, class_i], np.zeros(len(y)))
485              # Check for all other classes proba>0
486              assert np.all(proba[:, classes != class_i] > 0)
487  
488      # When `ensemble=False`, `cross_val_predict` is used to compute predictions
489      # to fit only one `calibrated_classifiers_`
490      else:
491          proba = cal_clf.calibrated_classifiers_[0].predict_proba(X)
492          assert_array_almost_equal(proba.sum(axis=1), np.ones(proba.shape[0]))

869      """Check that passing repeating twice the dataset `X` is equivalent to 
870      passing a `sample_weight` with a factor 2."""
871      X, y = load_iris(return_X_y=True)
872      # Scale the data to avoid any convergence issue
873      X = StandardScaler().fit_transform(X)
874      # Only use 2 classes

877  
878      # Interlace the data such that a 2-fold cross-validation will be equivalent
879      # to using the original dataset with a sample weights of 2
880      X_twice = np.zeros((X.shape[0] * 2, X.shape[1]), dtype=X.dtype)
881      X_twice[::2, :] = X
882      X_twice[1::2, :] = X
883      y_twice = np.zeros(y.shape[0] * 2, dtype=y.dtype)
884      y_twice[::2] = y
885      y_twice[1::2] = y
886  
887      estimator = LogisticRegression()
888      calibrated_clf_without_weights = CalibratedClassifierCV(
889          estimator,
890          method=method,
891          ensemble=ensemble,
892          cv=2,
893      )
894      calibrated_clf_with_weights = clone(calibrated_clf_without_weights)
895  
896      calibrated_clf_with_weights.fit(X, y, sample_weight=sample_weight)
897      calibrated_clf_without_weights.fit(X_twice, y_twice)
898  
899      # Check that the underlying fitted estimators have the same coefficients
900      for est_with_weights, est_without_weights in zip(
901          calibrated_clf_with_weights.calibrated_classifiers_,
902          calibrated_clf_without_weights.calibrated_classifiers_,
903      ):
904          assert_allclose(

906              est_without_weights.estimator.coef_,
907          )
908  
909      # Check that the predictions are the same
910      y_pred_with_weights = calibrated_clf_with_weights.predict_proba(X)
911      y_pred_without_weights = calibrated_clf_without_weights.predict_proba(X)
912  
913      assert_allclose(y_pred_with_weights, y_pred_without_weights)

1040     random_state = np.random.RandomState(seed=global_random_seed)
1041     n = 100
1042     y = random_state.randint(0, 2, size=n)
1043 
1044     # Check that for small enough predictions ranging from -2 to 2, the
1045     # threshold value has no impact on the outcome
1046     predictions_small = random_state.uniform(low=-2, high=2, size=100)
1047 
1048     # Using a threshold lower than the maximum absolute value of the
1049     # predictions enables internal re-scaling by max(abs(predictions_small)).
1050     threshold_1 = 0.1
1051     a1, b1 = _sigmoid_calibration(
1052         predictions=predictions_small,
1053         y=y,
1054         max_abs_prediction_threshold=threshold_1,
1055     )
1056 
1057     # Using a larger threshold disables rescaling.
1058     threshold_2 = 10
1059     a2, b2 = _sigmoid_calibration(
1060         predictions=predictions_small,
1061         y=y,
1062         max_abs_prediction_threshold=threshold_2,
1063     )
1064 
1065     # Using default threshold of 30 also disables the scaling.
1066     a3, b3 = _sigmoid_calibration(
1067         predictions=predictions_small,
1068         y=y,
1069     )
1070 
1071     # Depends on the tolerance of the underlying quasy-newton solver which is
1072     # not too strict by default.
1073     atol = 1e-6
1074     assert_allclose(a1, a2, atol=atol)
1075     assert_allclose(a2, a3, atol=atol)
1076     assert_allclose(b1, b2, atol=atol)
