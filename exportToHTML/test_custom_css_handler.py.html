4    import os
5    
6    import pytest
7    
8    CUSTOM_CSS = """body #top-panel-wrapper, 
9    #jp-top-bar { 
10     background-color: #aecad4 !important; 
11   } 
12    
13   body h1 { 
14     font-size: 22px; 
15     margin-bottom: 40px; 
16     color: #10929e; 
17     text-decoration: underline; 
18   }"""
19   
20   
21   @pytest.fixture
22   def jp_server_config(jp_server_config, tmp_path):
23       config = jp_server_config.copy()
24       config["LabApp"]["custom_css"] = True
25       return config
26   
27   
28   async def test_CustomCssHandler(tmp_path, jp_serverapp, labserverapp, jp_fetch):
29       custom_path = tmp_path / "config" / "custom"
30       # Check we are placing the custom.css file in the appropriate folder
31       assert str(custom_path) in jp_serverapp.web_app.settings["static_custom_path"]
32       custom_path.mkdir(parents=True, exist_ok=True)
33       (custom_path / "custom.css").write_text(CUSTOM_CSS)
34       response = await jp_fetch("custom", "custom.css", method="GET")
35   
36       assert response.code == 200
37       assert response.body.decode().replace(os.linesep, "\n") == CUSTOM_CSS
