32   def _graph_connected_component(graph, node_id):
33       """Find the largest graph connected components that contains one 
34       given node. 
35    
36       Parameters 
37       ---------- 
38       graph : array-like of shape (n_samples, n_samples) 
39           Adjacency matrix of the graph, non-zero weight means an edge 
40           between the nodes. 
41    
42       node_id : int 
43           The index of the query node of the graph. 
44    
45       Returns 
46       ------- 
47       connected_components_matrix : array-like of shape (n_samples,) 
48           An array of bool value indicating the indexes of the nodes 
49           belonging to the largest connected components of the given query 
50           node. 
51       """
52       n_node = graph.shape[0]
53       if sparse.issparse(graph):
54           # speed up row-wise access to boolean connection mask
55           graph = graph.tocsr()
56       connected_nodes = np.zeros(n_node, dtype=bool)
57       nodes_to_explore = np.zeros(n_node, dtype=bool)
58       nodes_to_explore[node_id] = True
59       for _ in range(n_node):
60           last_num_component = connected_nodes.sum()
61           np.logical_or(connected_nodes, nodes_to_explore, out=connected_nodes)
62           if last_num_component >= connected_nodes.sum():
63               break
64           indices = np.where(nodes_to_explore)[0]
65           nodes_to_explore.fill(False)
66           for i in indices:
67               if sparse.issparse(graph):
68                   # scipy not yet implemented 1D sparse slices; can be changed back to
69                   # `neighbors = graph[i].toarray().ravel()` once implemented
70                   neighbors = graph[[i], :].toarray().ravel()
71               else:
72                   neighbors = graph[i]
73               np.logical_or(nodes_to_explore, neighbors, out=nodes_to_explore)
74       return connected_nodes

77   def _graph_is_connected(graph):
78       """Return whether the graph is connected (True) or Not (False). 
79    
80       Parameters 
81       ---------- 
82       graph : {array-like, sparse matrix} of shape (n_samples, n_samples) 
83           Adjacency matrix of the graph, non-zero weight means an edge 
84           between the nodes. 
85    
86       Returns 
87       ------- 
88       is_connected : bool 
89           True means the graph is fully connected and False means not. 
90       """
91       if sparse.issparse(graph):
92           # Before Scipy 1.11.3, `connected_components` only supports 32-bit indices.
93           # PR: https://github.com/scipy/scipy/pull/18913
94           # First integration in 1.11.3: https://github.com/scipy/scipy/pull/19279
95           # TODO(jjerphan): Once SciPy 1.11.3 is the minimum supported version, use
96           # `accept_large_sparse=True`.
97           accept_large_sparse = sp_version >= parse_version("1.11.3")
98           graph = check_array(
99               graph, accept_sparse=True, accept_large_sparse=accept_large_sparse
100          )
101          # sparse graph, find all the connected components
102          n_connected_components, _ = connected_components(graph)
103          return n_connected_components == 1
104      else:
105          # dense graph, find all connected components start from node 0
106          return _graph_connected_component(graph, 0).sum() == graph.shape[0]

109  def _set_diag(laplacian, value, norm_laplacian):
110      """Set the diagonal of the laplacian matrix and convert it to a 
111      sparse format well suited for eigenvalue decomposition. 
112   
113      Parameters 
114      ---------- 
115      laplacian : {ndarray, sparse matrix} 
116          The graph laplacian. 
117   
118      value : float 
119          The value of the diagonal. 
120   
121      norm_laplacian : bool 
122          Whether the value of the diagonal should be changed or not. 
123   
124      Returns 
125      ------- 
126      laplacian : {array, sparse matrix} 
127          An array of matrix in a form that is well suited to fast 
128          eigenvalue decomposition, depending on the band width of the 
129          matrix. 
130      """
131      n_nodes = laplacian.shape[0]
132      # We need all entries in the diagonal to values
133      if not sparse.issparse(laplacian):
134          if norm_laplacian:
135              laplacian.flat[:: n_nodes + 1] = value
136      else:
137          laplacian = laplacian.tocoo()
138          if norm_laplacian:
139              diag_idx = laplacian.row == laplacian.col
140              laplacian.data[diag_idx] = value
141          # If the matrix has a small number of diagonals (as in the
142          # case of structured matrices coming from images), the
143          # dia format might be best suited for matvec products:
144          n_diags = np.unique(laplacian.row - laplacian.col).size
145          if n_diags <= 7:
146              # 3 or less outer diagonals on each side
147              laplacian = laplacian.todia()
148          else:
149              # csr has the fastest matvec and is thus best suited to
150              # arpack
151              laplacian = laplacian.tocsr()
152      return laplacian

155  @validate_params(
156      {
157          "adjacency": ["array-like", "sparse matrix"],
158          "n_components": [Interval(Integral, 1, None, closed="left")],
159          "eigen_solver": [StrOptions({"arpack", "lobpcg", "amg"}), None],
160          "random_state": ["random_state"],
161          "eigen_tol": [Interval(Real, 0, None, closed="left"), StrOptions({"auto"})],
162          "norm_laplacian": ["boolean"],
163          "drop_first": ["boolean"],
164      },
165      prefer_skip_nested_validation=True,
166  )
167  def spectral_embedding(
168      adjacency,
169      *,
170      n_components=8,
171      eigen_solver=None,
172      random_state=None,
173      eigen_tol="auto",
174      norm_laplacian=True,
175      drop_first=True,
176  ):
177      """Project the sample on the first eigenvectors of the graph Laplacian. 
178   
179      The adjacency matrix is used to compute a normalized graph Laplacian 
180      whose spectrum (especially the eigenvectors associated to the 
181      smallest eigenvalues) has an interpretation in terms of minimal 
182      number of cuts necessary to split the graph into comparably sized 
183      components. 
184   
185      This embedding can also 'work' even if the ``adjacency`` variable is 
186      not strictly the adjacency matrix of a graph but more generally 
187      an affinity or similarity matrix between samples (for instance the 
188      heat kernel of a euclidean distance matrix or a k-NN matrix). 
189   
190      However care must taken to always make the affinity matrix symmetric 
191      so that the eigenvector decomposition works as expected. 
192   
193      Note : Laplacian Eigenmaps is the actual algorithm implemented here. 
194   
195      Read more in the :ref:`User Guide <spectral_embedding>`. 
196   
197      Parameters 
198      ---------- 
199      adjacency : {array-like, sparse graph} of shape (n_samples, n_samples) 
200          The adjacency matrix of the graph to embed. 
201   
202      n_components : int, default=8 
203          The dimension of the projection subspace. 
204   
205      eigen_solver : {'arpack', 'lobpcg', 'amg'}, default=None 
206          The eigenvalue decomposition strategy to use. AMG requires pyamg 
207          to be installed. It can be faster on very large, sparse problems, 
208          but may also lead to instabilities. If None, then ``'arpack'`` is 
209          used. 
210   
211      random_state : int, RandomState instance or None, default=None 
212          A pseudo random number generator used for the initialization 
213          of the lobpcg eigen vectors decomposition when `eigen_solver == 
214          'amg'`, and for the K-Means initialization. Use an int to make 
215          the results deterministic across calls (See 
216          :term:`Glossary <random_state>`). 
217   
218          .. note:: 
219              When using `eigen_solver == 'amg'`, 
220              it is necessary to also fix the global numpy seed with 
221              `np.random.seed(int)` to get deterministic results. See 
222              https://github.com/pyamg/pyamg/issues/139 for further 
223              information. 
224   
225      eigen_tol : float, default="auto" 
226          Stopping criterion for eigendecomposition of the Laplacian matrix. 
227          If `eigen_tol="auto"` then the passed tolerance will depend on the 
228          `eigen_solver`: 
229   
230          - If `eigen_solver="arpack"`, then `eigen_tol=0.0`; 
231          - If `eigen_solver="lobpcg"` or `eigen_solver="amg"`, then 
232            `eigen_tol=None` which configures the underlying `lobpcg` solver to 
233            automatically resolve the value according to their heuristics. See, 
234            :func:`scipy.sparse.linalg.lobpcg` for details. 
235   
236          Note that when using `eigen_solver="amg"` values of `tol<1e-5` may lead 
237          to convergence issues and should be avoided. 
238   
239          .. versionadded:: 1.2 
240             Added 'auto' option. 
241   
242      norm_laplacian : bool, default=True 
243          If True, then compute symmetric normalized Laplacian. 
244   
245      drop_first : bool, default=True 
246          Whether to drop the first eigenvector. For spectral embedding, this 
247          should be True as the first eigenvector should be constant vector for 
248          connected graph, but for spectral clustering, this should be kept as 
249          False to retain the first eigenvector. 
250   
251      Returns 
252      ------- 
253      embedding : ndarray of shape (n_samples, n_components) 
254          The reduced samples. 
255   
256      Notes 
257      ----- 
258      Spectral Embedding (Laplacian Eigenmaps) is most useful when the graph 
259      has one connected component. If there graph has many components, the first 
260      few eigenvectors will simply uncover the connected components of the graph. 
261   
262      References 
263      ---------- 
264      * https://en.wikipedia.org/wiki/LOBPCG 
265   
266      * :doi:`"Toward the Optimal Preconditioned Eigensolver: Locally Optimal 
267        Block Preconditioned Conjugate Gradient Method", 
268        Andrew V. Knyazev 
269        <10.1137/S1064827500366124>` 
270   
271      Examples 
272      -------- 
273      >>> from sklearn.datasets import load_digits 
274      >>> from sklearn.neighbors import kneighbors_graph 
275      >>> from sklearn.manifold import spectral_embedding 
276      >>> X, _ = load_digits(return_X_y=True) 
277      >>> X = X[:100] 
278      >>> affinity_matrix = kneighbors_graph( 
279      ...     X, n_neighbors=int(X.shape[0] / 10), include_self=True 
280      ... ) 
281      >>> # make the matrix symmetric 
282      >>> affinity_matrix = 0.5 * (affinity_matrix + affinity_matrix.T) 
283      >>> embedding = spectral_embedding(affinity_matrix, n_components=2, random_state=42) 
284      >>> embedding.shape 
285      (100, 2) 
286      """
287      random_state = check_random_state(random_state)
288  
289      return _spectral_embedding(
290          adjacency,
291          n_components=n_components,
292          eigen_solver=eigen_solver,
293          random_state=random_state,
294          eigen_tol=eigen_tol,
295          norm_laplacian=norm_laplacian,
296          drop_first=drop_first,
297      )

300  def _spectral_embedding(
301      adjacency,
302      *,
303      n_components=8,
304      eigen_solver=None,
305      random_state=None,
306      eigen_tol="auto",
307      norm_laplacian=True,
308      drop_first=True,
309  ):
310      adjacency = check_symmetric(adjacency)
311  
312      if eigen_solver == "amg":
313          try:
314              from pyamg import smoothed_aggregation_solver
315          except ImportError as e:
316              raise ValueError(
317                  "The eigen_solver was set to 'amg', but pyamg is not available."
318              ) from e
319  
320      if eigen_solver is None:
321          eigen_solver = "arpack"
322  
323      n_nodes = adjacency.shape[0]
324      # Whether to drop the first eigenvector
325      if drop_first:
326          n_components = n_components + 1
327  
328      if not _graph_is_connected(adjacency):
329          warnings.warn(
330              "Graph is not fully connected, spectral embedding may not work as expected."
331          )
332  
333      laplacian, dd = csgraph_laplacian(
334          adjacency, normed=norm_laplacian, return_diag=True
335      )
336      if (
337          eigen_solver == "arpack"
338          or eigen_solver != "lobpcg"
339          and (not sparse.issparse(laplacian) or n_nodes < 5 * n_components)
340      ):
341          # lobpcg used with eigen_solver='amg' has bugs for low number of nodes
342          # for details see the source code in scipy:
343          # https://github.com/scipy/scipy/blob/v0.11.0/scipy/sparse/linalg/eigen
344          # /lobpcg/lobpcg.py#L237
345          # or matlab:
346          # https://www.mathworks.com/matlabcentral/fileexchange/48-lobpcg-m
347          laplacian = _set_diag(laplacian, 1, norm_laplacian)
348  
349          # Here we'll use shift-invert mode for fast eigenvalues
350          # (see https://docs.scipy.org/doc/scipy/reference/tutorial/arpack.html
351          #  for a short explanation of what this means)
352          # Because the normalized Laplacian has eigenvalues between 0 and 2,
353          # I - L has eigenvalues between -1 and 1.  ARPACK is most efficient
354          # when finding eigenvalues of largest magnitude (keyword which='LM')
355          # and when these eigenvalues are very large compared to the rest.
356          # For very large, very sparse graphs, I - L can have many, many
357          # eigenvalues very near 1.0.  This leads to slow convergence.  So
358          # instead, we'll use ARPACK's shift-invert mode, asking for the
359          # eigenvalues near 1.0.  This effectively spreads-out the spectrum
360          # near 1.0 and leads to much faster convergence: potentially an
361          # orders-of-magnitude speedup over simply using keyword which='LA'
362          # in standard mode.
363          try:
364              # We are computing the opposite of the laplacian inplace so as
365              # to spare a memory allocation of a possibly very large array
366              tol = 0 if eigen_tol == "auto" else eigen_tol
367              laplacian *= -1
368              v0 = _init_arpack_v0(laplacian.shape[0], random_state)
369              laplacian = check_array(
370                  laplacian, accept_sparse="csr", accept_large_sparse=False
371              )
372              _, diffusion_map = eigsh(
373                  laplacian, k=n_components, sigma=1.0, which="LM", tol=tol, v0=v0
374              )
375              embedding = diffusion_map.T[n_components::-1]
376              if norm_laplacian:
377                  # recover u = D^-1/2 x from the eigenvector output x
378                  embedding = embedding / dd
379          except RuntimeError:
380              # When submatrices are exactly singular, an LU decomposition
381              # in arpack fails. We fallback to lobpcg
382              eigen_solver = "lobpcg"
383              # Revert the laplacian to its opposite to have lobpcg work
384              laplacian *= -1
385  
386      elif eigen_solver == "amg":
387          # Use AMG to get a preconditioner and speed up the eigenvalue
388          # problem.
389          if not sparse.issparse(laplacian):
390              warnings.warn("AMG works better for sparse matrices")
391          laplacian = check_array(
392              laplacian, dtype=[np.float64, np.float32], accept_sparse=True
393          )
394          laplacian = _set_diag(laplacian, 1, norm_laplacian)
395  
396          # The Laplacian matrix is always singular, having at least one zero
397          # eigenvalue, corresponding to the trivial eigenvector, which is a
398          # constant. Using a singular matrix for preconditioning may result in
399          # random failures in LOBPCG and is not supported by the existing
400          # theory:
401          #     see https://doi.org/10.1007/s10208-015-9297-1
402          # Shift the Laplacian so its diagononal is not all ones. The shift
403          # does change the eigenpairs however, so we'll feed the shifted
404          # matrix to the solver and afterward set it back to the original.
405          diag_shift = 1e-5 * sparse.eye(laplacian.shape[0])
406          laplacian += diag_shift
407          if hasattr(sparse, "csr_array") and isinstance(laplacian, sparse.csr_array):
408              # `pyamg` does not work with `csr_array` and we need to convert it to a
409              # `csr_matrix` object.
410              laplacian = sparse.csr_matrix(laplacian)
411          ml = smoothed_aggregation_solver(check_array(laplacian, accept_sparse="csr"))
412          laplacian -= diag_shift
413  
414          M = ml.aspreconditioner()
415          # Create initial approximation X to eigenvectors
416          X = random_state.standard_normal(size=(laplacian.shape[0], n_components + 1))
417          X[:, 0] = dd.ravel()
418          X = X.astype(laplacian.dtype)
419  
420          tol = None if eigen_tol == "auto" else eigen_tol
421          _, diffusion_map = lobpcg(laplacian, X, M=M, tol=tol, largest=False)
422          embedding = diffusion_map.T
423          if norm_laplacian:
424              # recover u = D^-1/2 x from the eigenvector output x
425              embedding = embedding / dd
426          if embedding.shape[0] == 1:
427              raise ValueError
428  
429      if eigen_solver == "lobpcg":
430          laplacian = check_array(
431              laplacian, dtype=[np.float64, np.float32], accept_sparse=True
432          )
433          if n_nodes < 5 * n_components + 1:
434              # see note above under arpack why lobpcg has problems with small
435              # number of nodes
436              # lobpcg will fallback to eigh, so we short circuit it
437              if sparse.issparse(laplacian):
438                  laplacian = laplacian.toarray()
439              _, diffusion_map = eigh(laplacian, check_finite=False)
440              embedding = diffusion_map.T[:n_components]
441              if norm_laplacian:
442                  # recover u = D^-1/2 x from the eigenvector output x
443                  embedding = embedding / dd
444          else:
445              laplacian = _set_diag(laplacian, 1, norm_laplacian)
446              # We increase the number of eigenvectors requested, as lobpcg
447              # doesn't behave well in low dimension and create initial
448              # approximation X to eigenvectors
449              X = random_state.standard_normal(
450                  size=(laplacian.shape[0], n_components + 1)
451              )
452              X[:, 0] = dd.ravel()
453              X = X.astype(laplacian.dtype)
454              tol = None if eigen_tol == "auto" else eigen_tol
455              _, diffusion_map = lobpcg(
456                  laplacian, X, tol=tol, largest=False, maxiter=2000
457              )
458              embedding = diffusion_map.T[:n_components]
459              if norm_laplacian:
460                  # recover u = D^-1/2 x from the eigenvector output x
461                  embedding = embedding / dd
462              if embedding.shape[0] == 1:
463                  raise ValueError
464  
465      embedding = _deterministic_vector_sign_flip(embedding)
466      if drop_first:
467          return embedding[1:n_components].T
468      else:
469          return embedding[:n_components].T

472  class SpectralEmbedding(BaseEstimator):
473      """Spectral embedding for non-linear dimensionality reduction. 
474   
475      Forms an affinity matrix given by the specified function and 
476      applies spectral decomposition to the corresponding graph laplacian. 
477      The resulting transformation is given by the value of the 
478      eigenvectors for each data point. 
479   
480      Note : Laplacian Eigenmaps is the actual algorithm implemented here. 
481   
482      Read more in the :ref:`User Guide <spectral_embedding>`. 
483   
484      Parameters 
485      ---------- 
486      n_components : int, default=2 
487          The dimension of the projected subspace. 
488   
489      affinity : {'nearest_neighbors', 'rbf', 'precomputed', \ 
490                  'precomputed_nearest_neighbors'} or callable, \ 
491                  default='nearest_neighbors' 
492          How to construct the affinity matrix. 
493           - 'nearest_neighbors' : construct the affinity matrix by computing a 
494             graph of nearest neighbors. 
495           - 'rbf' : construct the affinity matrix by computing a radial basis 
496             function (RBF) kernel. 
497           - 'precomputed' : interpret ``X`` as a precomputed affinity matrix. 
498           - 'precomputed_nearest_neighbors' : interpret ``X`` as a sparse graph 
499             of precomputed nearest neighbors, and constructs the affinity matrix 
500             by selecting the ``n_neighbors`` nearest neighbors. 
501           - callable : use passed in function as affinity 
502             the function takes in data matrix (n_samples, n_features) 
503             and return affinity matrix (n_samples, n_samples). 
504   
505      gamma : float, default=None 
506          Kernel coefficient for rbf kernel. If None, gamma will be set to 
507          1/n_features. 
508   
509      random_state : int, RandomState instance or None, default=None 
510          A pseudo random number generator used for the initialization 
511          of the lobpcg eigen vectors decomposition when `eigen_solver == 
512          'amg'`, and for the K-Means initialization. Use an int to make 
513          the results deterministic across calls (See 
514          :term:`Glossary <random_state>`). 
515   
516          .. note:: 
517              When using `eigen_solver == 'amg'`, 
518              it is necessary to also fix the global numpy seed with 
519              `np.random.seed(int)` to get deterministic results. See 
520              https://github.com/pyamg/pyamg/issues/139 for further 
521              information. 
522   
523      eigen_solver : {'arpack', 'lobpcg', 'amg'}, default=None 
524          The eigenvalue decomposition strategy to use. AMG requires pyamg 
525          to be installed. It can be faster on very large, sparse problems. 
526          If None, then ``'arpack'`` is used. 
527   
528      eigen_tol : float, default="auto" 
529          Stopping criterion for eigendecomposition of the Laplacian matrix. 
530          If `eigen_tol="auto"` then the passed tolerance will depend on the 
531          `eigen_solver`: 
532   
533          - If `eigen_solver="arpack"`, then `eigen_tol=0.0`; 
534          - If `eigen_solver="lobpcg"` or `eigen_solver="amg"`, then 
535            `eigen_tol=None` which configures the underlying `lobpcg` solver to 
536            automatically resolve the value according to their heuristics. See, 
537            :func:`scipy.sparse.linalg.lobpcg` for details. 
538   
539          Note that when using `eigen_solver="lobpcg"` or `eigen_solver="amg"` 
540          values of `tol<1e-5` may lead to convergence issues and should be 
541          avoided. 
542   
543          .. versionadded:: 1.2 
544   
545      n_neighbors : int, default=None 
546          Number of nearest neighbors for nearest_neighbors graph building. 
547          If None, n_neighbors will be set to max(n_samples/10, 1). 
548   
549      n_jobs : int, default=None 
550          The number of parallel jobs to run. 
551          ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context. 
552          ``-1`` means using all processors. See :term:`Glossary <n_jobs>` 
553          for more details. 
554   
555      Attributes 
556      ---------- 
557      embedding_ : ndarray of shape (n_samples, n_components) 
558          Spectral embedding of the training matrix. 
559   
560      affinity_matrix_ : ndarray of shape (n_samples, n_samples) 
561          Affinity_matrix constructed from samples or precomputed. 
562   
563      n_features_in_ : int 
564          Number of features seen during :term:`fit`. 
565   
566          .. versionadded:: 0.24 
567   
568      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
569          Names of features seen during :term:`fit`. Defined only when `X` 
570          has feature names that are all strings. 
571   
572          .. versionadded:: 1.0 
573   
574      n_neighbors_ : int 
575          Number of nearest neighbors effectively used. 
576   
577      See Also 
578      -------- 
579      Isomap : Non-linear dimensionality reduction through Isometric Mapping. 
580   
581      References 
582      ---------- 
583   
584      - :doi:`A Tutorial on Spectral Clustering, 2007 
585        Ulrike von Luxburg 
586        <10.1007/s11222-007-9033-z>` 
587   
588      - `On Spectral Clustering: Analysis and an algorithm, 2001 
589        Andrew Y. Ng, Michael I. Jordan, Yair Weiss 
590        <https://citeseerx.ist.psu.edu/doc_view/pid/796c5d6336fc52aa84db575fb821c78918b65f58>`_ 
591   
592      - :doi:`Normalized cuts and image segmentation, 2000 
593        Jianbo Shi, Jitendra Malik 
594        <10.1109/34.868688>` 
595   
596      Examples 
597      -------- 
598      >>> from sklearn.datasets import load_digits 
599      >>> from sklearn.manifold import SpectralEmbedding 
600      >>> X, _ = load_digits(return_X_y=True) 
601      >>> X.shape 
602      (1797, 64) 
603      >>> embedding = SpectralEmbedding(n_components=2) 
604      >>> X_transformed = embedding.fit_transform(X[:100]) 
605      >>> X_transformed.shape 
606      (100, 2) 
607      """
608  
609      _parameter_constraints: dict = {
610          "n_components": [Interval(Integral, 1, None, closed="left")],
611          "affinity": [
612              StrOptions(
613                  {
614                      "nearest_neighbors",
615                      "rbf",
616                      "precomputed",
617                      "precomputed_nearest_neighbors",
618                  },
619              ),
620              callable,

660      def _get_affinity_matrix(self, X, Y=None):
661          """Calculate the affinity matrix from data 
662          Parameters 
663          ---------- 
664          X : array-like of shape (n_samples, n_features) 
665              Training vector, where `n_samples` is the number of samples 
666              and `n_features` is the number of features. 
667   
668              If affinity is "precomputed" 
669              X : array-like of shape (n_samples, n_samples), 
670              Interpret X as precomputed adjacency graph computed from 
671              samples. 
672   
673          Y: Ignored 
674   
675          Returns 
676          ------- 
677          affinity_matrix of shape (n_samples, n_samples) 
678          """
679          if self.affinity == "precomputed":
680              self.affinity_matrix_ = X
681              return self.affinity_matrix_
682          if self.affinity == "precomputed_nearest_neighbors":
683              estimator = NearestNeighbors(
684                  n_neighbors=self.n_neighbors, n_jobs=self.n_jobs, metric="precomputed"
685              ).fit(X)
686              connectivity = estimator.kneighbors_graph(X=X, mode="connectivity")
687              self.affinity_matrix_ = 0.5 * (connectivity + connectivity.T)
688              return self.affinity_matrix_
689          if self.affinity == "nearest_neighbors":
690              if sparse.issparse(X):
691                  warnings.warn(
692                      "Nearest neighbors affinity currently does "
693                      "not support sparse input, falling back to "
694                      "rbf affinity"
695                  )
696                  self.affinity = "rbf"
697              else:
698                  self.n_neighbors_ = (
699                      self.n_neighbors
700                      if self.n_neighbors is not None
701                      else max(int(X.shape[0] / 10), 1)
702                  )
703                  self.affinity_matrix_ = kneighbors_graph(
704                      X, self.n_neighbors_, include_self=True, n_jobs=self.n_jobs
705                  )
706                  # currently only symmetric affinity_matrix supported
707                  self.affinity_matrix_ = 0.5 * (
708                      self.affinity_matrix_ + self.affinity_matrix_.T
709                  )
710                  return self.affinity_matrix_
711          if self.affinity == "rbf":
712              self.gamma_ = self.gamma if self.gamma is not None else 1.0 / X.shape[1]
713              self.affinity_matrix_ = rbf_kernel(X, gamma=self.gamma_)
714              return self.affinity_matrix_
715          self.affinity_matrix_ = self.affinity(X)
716          return self.affinity_matrix_

718      @_fit_context(prefer_skip_nested_validation=True)
719      def fit(self, X, y=None):
720          """Fit the model from data in X. 
721   
722          Parameters 
723          ---------- 
724          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
725              Training vector, where `n_samples` is the number of samples 
726              and `n_features` is the number of features. 
727   
728              If affinity is "precomputed" 
729              X : {array-like, sparse matrix}, shape (n_samples, n_samples), 
730              Interpret X as precomputed adjacency graph computed from 
731              samples. 
732   
733          y : Ignored 
734              Not used, present for API consistency by convention. 
735   
736          Returns 
737          ------- 
738          self : object 
739              Returns the instance itself. 
740          """
741          X = validate_data(self, X, accept_sparse="csr", ensure_min_samples=2)
742  
743          random_state = check_random_state(self.random_state)
744  
745          affinity_matrix = self._get_affinity_matrix(X)
746          self.embedding_ = _spectral_embedding(
747              affinity_matrix,
748              n_components=self.n_components,
749              eigen_solver=self.eigen_solver,
750              eigen_tol=self.eigen_tol,
751              random_state=random_state,
752          )
753          return self

755      def fit_transform(self, X, y=None):
756          """Fit the model from data in X and transform X. 
757   
758          Parameters 
759          ---------- 
760          X : {array-like, sparse matrix} of shape (n_samples, n_features) 
761              Training vector, where `n_samples` is the number of samples

764              If affinity is "precomputed"
