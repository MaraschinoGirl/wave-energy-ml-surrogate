14   def GetDefaultProfileName():
15       import win32api
16       import win32con
17   
18       try:
19           key = win32api.RegOpenKey(
20               win32con.HKEY_CURRENT_USER,
21               "Software\\Microsoft\\Windows NT\\CurrentVersion\\Windows Messaging Subsystem\\Profiles",
22           )
23           try:
24               return win32api.RegQueryValueEx(key, "DefaultProfile")[0]
25           finally:
26               key.Close()
27       except win32api.error:
28           return None

34   def DumpFolder(folder, indent=0):
35       print(" " * indent, folder.Name)
36       folders = folder.Folders
37       folder = folders.GetFirst()
38       while folder:
39           DumpFolder(folder, indent + 1)
40           folder = folders.GetNext()

43   def DumpFolders(session):
44       try:
45           infostores = session.InfoStores
46       except AttributeError:
47           # later outlook?
48           store = session.DefaultStore
49           folder = store.GetRootFolder()
50           DumpFolder(folder)
51           return
52   
53       print(infostores)
54       print("There are %d infostores" % infostores.Count)
55       for i in range(infostores.Count):
56           infostore = infostores[i + 1]
57           print("Infostore = ", infostore.Name)
58           try:
59               folder = infostore.RootFolder
60           except pythoncom.com_error as details:
61               hr, msg, exc, arg = details
62               # -2147221219 == MAPI_E_FAILONEPROVIDER - a single provider temporarily not available.
63               if exc and exc[-1] == -2147221219:
64                   print("This info store is currently not available")
65                   continue
66           DumpFolder(folder)
