4    import numpy as np
5    
6    from .extmath import stable_cumsum
7    
8    
9    def _weighted_percentile(array, sample_weight, percentile=50):
10       """Compute weighted percentile 
11    
12       Computes lower weighted percentile. If `array` is a 2D array, the 
13       `percentile` is computed along the axis 0. 
14    
15           .. versionchanged:: 0.24 
16               Accepts 2D `array`. 
17    
18       Parameters 
19       ---------- 
20       array : 1D or 2D array 
21           Values to take the weighted percentile of. 
22    
23       sample_weight: 1D or 2D array 
24           Weights for each value in `array`. Must be same shape as `array` or 
25           of shape `(array.shape[0],)`. 
26    
27       percentile: int or float, default=50 
28           Percentile to compute. Must be value between 0 and 100. 
29    
30       Returns 
31       ------- 
32       percentile : int if `array` 1D, ndarray if `array` 2D 
33           Weighted percentile. 
34       """
35       n_dim = array.ndim
36       if n_dim == 0:
37           return array[()]
38       if array.ndim == 1:
39           array = array.reshape((-1, 1))
40       # When sample_weight 1D, repeat for each array.shape[1]
41       if array.shape != sample_weight.shape and array.shape[0] == sample_weight.shape[0]:
42           sample_weight = np.tile(sample_weight, (array.shape[1], 1)).T
43       sorted_idx = np.argsort(array, axis=0)
44       sorted_weights = np.take_along_axis(sample_weight, sorted_idx, axis=0)
45   
46       # Find index of median prediction for each sample
47       weight_cdf = stable_cumsum(sorted_weights, axis=0)
48       adjusted_percentile = percentile / 100 * weight_cdf[-1]
49   
50       # For percentile=0, ignore leading observations with sample_weight=0. GH20528
51       mask = adjusted_percentile == 0
52       adjusted_percentile[mask] = np.nextafter(
53           adjusted_percentile[mask], adjusted_percentile[mask] + 1
54       )
55   
56       percentile_idx = np.array(
57           [
58               np.searchsorted(weight_cdf[:, i], adjusted_percentile[i])
59               for i in range(weight_cdf.shape[1])
60           ]
61       )
62       percentile_idx = np.array(percentile_idx)
63       # In rare cases, percentile_idx equals to sorted_idx.shape[0]
64       max_idx = sorted_idx.shape[0] - 1
65       percentile_idx = np.apply_along_axis(
66           lambda x: np.clip(x, 0, max_idx), axis=0, arr=percentile_idx
67       )
68   
69       col_index = np.arange(array.shape[1])
70       percentile_in_sorted = sorted_idx[percentile_idx, col_index]
71       percentile = array[percentile_in_sorted, col_index]
72       return percentile[0] if n_dim == 1 else percentile
