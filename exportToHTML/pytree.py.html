15   r"""Utilities for working with ``PyTree``\s. 
16    
17   The :mod:`optree.pytree` namespace contains aliases of ``optree.tree_*`` utilities. 
18    
19   >>> import optree.pytree as pytree 
20   >>> tree = {'b': (2, [3, 4]), 'a': 1, 'c': None, 'd': 5} 
21   >>> leaves, treespec = pytree.flatten(tree) 
22   >>> leaves, treespec  # doctest: +IGNORE_WHITESPACE 
23   ( 
24       [1, 2, 3, 4, 5], 
25       PyTreeSpec({'a': *, 'b': (*, [*, *]), 'c': None, 'd': *}) 
26   ) 
27   >>> tree == pytree.unflatten(treespec, leaves) 
28   True 
29    
30   .. versionadded:: 0.14.1 
31   """
