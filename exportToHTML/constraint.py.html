30   class LinearConstraint(object):
31       """A linear constraint in matrix form. 
32    
33       This object represents a linear constraint of the form `Ax = b`. 
34    
35       Usually you won't be constructing these by hand, but instead get them as 
36       the return value from :meth:`DesignInfo.linear_constraint`. 
37    
38       .. attribute:: coefs 
39    
40          A 2-dimensional ndarray with float dtype, representing `A`. 
41    
42       .. attribute:: constants 
43    
44          A 2-dimensional single-column ndarray with float dtype, representing 
45          `b`. 
46    
47       .. attribute:: variable_names 
48    
49          A list of strings giving the names of the variables being 
50          constrained. (Used only for consistency checking.) 
51       """
52   
53       def __init__(self, variable_names, coefs, constants=None):
54           self.variable_names = list(variable_names)
55           self.coefs = np.atleast_2d(np.asarray(coefs, dtype=float))
56           if constants is None:
57               constants = np.zeros(self.coefs.shape[0], dtype=float)
58           constants = np.asarray(constants, dtype=float)
59           self.constants = atleast_2d_column_default(constants)
60           if self.constants.ndim != 2 or self.constants.shape[1] != 1:
61               raise ValueError("constants is not (convertible to) a column matrix")
62           if self.coefs.ndim != 2 or self.coefs.shape[1] != len(variable_names):
63               raise ValueError("wrong shape for coefs")
64           if self.coefs.shape[0] == 0:
65               raise ValueError("must have at least one row in constraint matrix")
66           if self.coefs.shape[0] != self.constants.shape[0]:
67               raise ValueError("shape mismatch between coefs and constants")
68   
69       __repr__ = repr_pretty_delegate
70   
71       def _repr_pretty_(self, p, cycle):
72           assert not cycle
73           return repr_pretty_impl(
74               p, self, [self.variable_names, self.coefs, self.constants]
75           )
76   
77       __getstate__ = no_pickling

79       @classmethod
80       def combine(cls, constraints):
81           """Create a new LinearConstraint by ANDing together several existing 
82           LinearConstraints. 
83    
84           :arg constraints: An iterable of LinearConstraint objects. Their 
85             :attr:`variable_names` attributes must all match. 
86           :returns: A new LinearConstraint object. 
87           """
88           if not constraints:
89               raise ValueError("no constraints specified")
90           variable_names = constraints[0].variable_names
91           for constraint in constraints:
92               if constraint.variable_names != variable_names:
93                   raise ValueError("variable names don't match")
94           coefs = np.vstack([c.coefs for c in constraints])
95           constants = np.vstack([c.constants for c in constraints])
96           return cls(variable_names, coefs, constants)

175  def _token_maker(type, string):
176      def make_token(scanner, token_string):
177          if type == "__OP__":
178              actual_type = token_string
179          else:
180              actual_type = type
181          return Token(actual_type, Origin(string, *scanner.match.span()), token_string)
182  
183      return make_token

186  def _tokenize_constraint(string, variable_names):
187      lparen_re = r"\("
188      rparen_re = r"\)"
189      op_re = "|".join([re.escape(op.token_type) for op in _ops])
190      num_re = r"[-+]?[0-9]*\.?[0-9]+([eE][-+]?[0-9]+)?"
191      whitespace_re = r"\s+"
192  
193      # Prefer long matches:
194      variable_names = sorted(variable_names, key=len, reverse=True)
195      variable_re = "|".join([re.escape(n) for n in variable_names])
196  
197      lexicon = [
198          (lparen_re, _token_maker(Token.LPAREN, string)),
199          (rparen_re, _token_maker(Token.RPAREN, string)),
200          (op_re, _token_maker("__OP__", string)),
201          (variable_re, _token_maker("VARIABLE", string)),
202          (num_re, _token_maker("NUMBER", string)),
203          (whitespace_re, None),
204      ]
205  
206      scanner = re.Scanner(lexicon)
207      tokens, leftover = scanner.scan(string)
208      if leftover:
209          offset = len(string) - len(leftover)
210          raise PatsyError(
211              "unrecognized token in constraint", Origin(string, offset, offset + 1)
212          )
213  
214      return tokens

261  class _EvalConstraint(object):
262      def __init__(self, variable_names):
263          self._variable_names = variable_names
264          self._N = len(variable_names)
265  
266          self._dispatch = {
267              ("VARIABLE", 0): self._eval_variable,
268              ("NUMBER", 0): self._eval_number,
269              ("+", 1): self._eval_unary_plus,
270              ("-", 1): self._eval_unary_minus,
271              ("+", 2): self._eval_binary_plus,
272              ("-", 2): self._eval_binary_minus,
273              ("*", 2): self._eval_binary_multiply,
274              ("/", 2): self._eval_binary_div,
275              ("=", 2): self._eval_binary_eq,
276              (",", 2): self._eval_binary_comma,
277          }
278  
279      # General scheme: there are 2 types we deal with:
280      #   - linear combinations ("lincomb"s) of variables and constants,
281      #     represented as ndarrays with size N+1
282      #     The last entry is the constant, so [10, 20, 30] means 10x + 20y +
283      #     30.
284      #   - LinearConstraint objects
285  
286      def is_constant(self, coefs):
287          return np.all(coefs[: self._N] == 0)
288  
289      def _eval_variable(self, tree):
290          var = tree.token.extra
291          coefs = np.zeros((self._N + 1,), dtype=float)
292          coefs[self._variable_names.index(var)] = 1
293          return coefs
294  
295      def _eval_number(self, tree):
296          coefs = np.zeros((self._N + 1,), dtype=float)
297          coefs[-1] = float(tree.token.extra)
298          return coefs
299  
300      def _eval_unary_plus(self, tree):
301          return self.eval(tree.args[0])
302  
303      def _eval_unary_minus(self, tree):
304          return -1 * self.eval(tree.args[0])
305  
306      def _eval_binary_plus(self, tree):
307          return self.eval(tree.args[0]) + self.eval(tree.args[1])
308  
309      def _eval_binary_minus(self, tree):
310          return self.eval(tree.args[0]) - self.eval(tree.args[1])
311  
312      def _eval_binary_div(self, tree):
313          left = self.eval(tree.args[0])
314          right = self.eval(tree.args[1])
315          if not self.is_constant(right):
316              raise PatsyError(
317                  "Can't divide by a variable in a linear " "constraint", tree.args[1]
318              )
319          return left / right[-1]
320  
321      def _eval_binary_multiply(self, tree):
322          left = self.eval(tree.args[0])
323          right = self.eval(tree.args[1])
324          if self.is_constant(left):
325              return left[-1] * right
326          elif self.is_constant(right):
327              return left * right[-1]
328          else:
329              raise PatsyError(
330                  "Can't multiply one variable by another " "in a linear constraint", tree
331              )
332  
333      def _eval_binary_eq(self, tree):
334          # Handle "a1 = a2 = a3", which is parsed as "(a1 = a2) = a3"
335          args = list(tree.args)
336          constraints = []
337          for i, arg in enumerate(args):
338              if arg.type == "=":
339                  constraints.append(self.eval(arg, constraint=True))
340                  # make our left argument be their right argument, or
341                  # vice-versa
342                  args[i] = arg.args[1 - i]
343          left = self.eval(args[0])
344          right = self.eval(args[1])
345          coefs = left[: self._N] - right[: self._N]
346          if np.all(coefs == 0):
347              raise PatsyError("no variables appear in constraint", tree)
348          constant = -left[-1] + right[-1]
349          constraint = LinearConstraint(self._variable_names, coefs, constant)
350          constraints.append(constraint)
351          return LinearConstraint.combine(constraints)
352  
353      def _eval_binary_comma(self, tree):
354          left = self.eval(tree.args[0], constraint=True)
355          right = self.eval(tree.args[1], constraint=True)
356          return LinearConstraint.combine([left, right])
357  
358      def eval(self, tree, constraint=False):
359          key = (tree.type, len(tree.args))
360          assert key in self._dispatch
361          val = self._dispatch[key](tree)
362          if constraint:
363              # Force it to be a constraint
364              if isinstance(val, LinearConstraint):
365                  return val
366              else:
367                  assert val.size == self._N + 1
368                  if np.all(val[: self._N] == 0):
369                      raise PatsyError("term is constant, with no variables", tree)
370                  return LinearConstraint(self._variable_names, val[: self._N], -val[-1])
371          else:
372              # Force it to *not* be a constraint
373              if isinstance(val, LinearConstraint):
374                  raise PatsyError("unexpected constraint object", tree)
375              return val

378  def linear_constraint(constraint_like, variable_names):
379      """This is the internal interface implementing 
380      DesignInfo.linear_constraint, see there for docs."""
381      if isinstance(constraint_like, LinearConstraint):
382          if constraint_like.variable_names != variable_names:
383              raise ValueError(
384                  "LinearConstraint has wrong variable_names "
385                  "(got %r, expected %r)"
386                  % (constraint_like.variable_names, variable_names)
387              )
388          return constraint_like
389  
390      if isinstance(constraint_like, Mapping):
391          # Simple conjunction-of-equality constraints can be specified as
392          # dicts. {"x": 1, "y": 2} -> tests x = 1 and y = 2. Keys can be
393          # either variable names, or variable indices.
394          coefs = np.zeros((len(constraint_like), len(variable_names)), dtype=float)
395          constants = np.zeros(len(constraint_like))
396          used = set()
397          for i, (name, value) in enumerate(constraint_like.items()):
398              if name in variable_names:
399                  idx = variable_names.index(name)
400              elif isinstance(name, int):
401                  idx = name
402              else:
403                  raise ValueError("unrecognized variable name/index %r" % (name,))
404              if idx in used:
405                  raise ValueError("duplicated constraint on %r" % (variable_names[idx],))
406              used.add(idx)
407              coefs[i, idx] = 1
408              constants[i] = value
409          return LinearConstraint(variable_names, coefs, constants)
410  
411      if isinstance(constraint_like, str):
412          constraint_like = [constraint_like]
413          # fall-through
414  
415      if (
416          isinstance(constraint_like, list)
417          and constraint_like
418          and isinstance(constraint_like[0], str)
419      ):
420          constraints = []
421          for code in constraint_like:
422              if not isinstance(code, str):
423                  raise ValueError("expected a string, not %r" % (code,))
424              tree = parse_constraint(code, variable_names)
425              evaluator = _EvalConstraint(variable_names)
426              constraints.append(evaluator.eval(tree, constraint=True))
427          return LinearConstraint.combine(constraints)
428  
429      if isinstance(constraint_like, tuple):
430          if len(constraint_like) != 2:
431              raise ValueError("constraint tuple must have length 2")
432          coef, constants = constraint_like
433          return LinearConstraint(variable_names, coef, constants)
434  
435      # assume a raw ndarray
436      coefs = np.asarray(constraint_like, dtype=float)
437      return LinearConstraint(variable_names, coefs)
