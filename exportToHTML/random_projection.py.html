random_projection.py 
1    """Random projection transformers. 
2     
3    Random projections are a simple and computationally efficient way to 
4    reduce the dimensionality of the data by trading a controlled amount 
5    of accuracy (as additional variance) for faster processing times and 
6    smaller model sizes. 
7     
8    The dimensions and distribution of random projections matrices are 
9    controlled so as to preserve the pairwise distances between any two 
10   samples of the dataset. 
11    
12   The main theoretical result behind the efficiency of random projection is the 
13   `Johnson-Lindenstrauss lemma (quoting Wikipedia) 
14   <https://en.wikipedia.org/wiki/Johnson%E2%80%93Lindenstrauss_lemma>`_: 
15    
16     In mathematics, the Johnson-Lindenstrauss lemma is a result 
17     concerning low-distortion embeddings of points from high-dimensional 
18     into low-dimensional Euclidean space. The lemma states that a small set 
19     of points in a high-dimensional space can be embedded into a space of 
20     much lower dimension in such a way that distances between the points are 
21     nearly preserved. The map used for the embedding is at least Lipschitz, 
22     and can even be taken to be an orthogonal projection. 
23   """
24

63   def johnson_lindenstrauss_min_dim(n_samples, *, eps=0.1):
64       """Find a 'safe' number of components to randomly project to. 
65    
66       The distortion introduced by a random projection `p` only changes the 
67       distance between two points by a factor (1 +- eps) in a euclidean space 
68       with good probability. The projection `p` is an eps-embedding as defined 
69       by: 
70    
71       .. code-block:: text 
72    
73         (1 - eps) ||u - v||^2 < ||p(u) - p(v)||^2 < (1 + eps) ||u - v||^2 
74    
75       Where u and v are any rows taken from a dataset of shape (n_samples, 
76       n_features), eps is in ]0, 1[ and p is a projection by a random Gaussian 
77       N(0, 1) matrix of shape (n_components, n_features) (or a sparse 
78       Achlioptas matrix). 
79    
80       The minimum number of components to guarantee the eps-embedding is 
81       given by: 
82    
83       .. code-block:: text 
84    
85         n_components >= 4 log(n_samples) / (eps^2 / 2 - eps^3 / 3) 
86    
87       Note that the number of dimensions is independent of the original 
88       number of features but instead depends on the size of the dataset: 
89       the larger the dataset, the higher is the minimal dimensionality of 
90       an eps-embedding. 
91    
92       Read more in the :ref:`User Guide <johnson_lindenstrauss>`. 
93    
94       Parameters 
95       ---------- 
96       n_samples : int or array-like of int 
97           Number of samples that should be an integer greater than 0. If an array 
98           is given, it will compute a safe number of components array-wise. 
99    
100      eps : float or array-like of shape (n_components,), dtype=float, \ 
101              default=0.1 
102          Maximum distortion rate in the range (0, 1) as defined by the 
103          Johnson-Lindenstrauss lemma. If an array is given, it will compute a 
104          safe number of components array-wise. 
105   
106      Returns 
107      ------- 
108      n_components : int or ndarray of int 
109          The minimal number of components to guarantee with good probability 
110          an eps-embedding with n_samples. 
111   
112      References 
113      ---------- 
114   
115      .. [1] https://en.wikipedia.org/wiki/Johnson%E2%80%93Lindenstrauss_lemma 
116   
117      .. [2] `Sanjoy Dasgupta and Anupam Gupta, 1999, 
118             "An elementary proof of the Johnson-Lindenstrauss Lemma." 
119             <https://citeseerx.ist.psu.edu/doc_view/pid/95cd464d27c25c9c8690b378b894d337cdf021f9>`_ 
120   
121      Examples 
122      -------- 
123      >>> from sklearn.random_projection import johnson_lindenstrauss_min_dim 
124      >>> johnson_lindenstrauss_min_dim(1e6, eps=0.5) 
125      np.int64(663) 
126   
127      >>> johnson_lindenstrauss_min_dim(1e6, eps=[0.5, 0.1, 0.01]) 
128      array([    663,   11841, 1112658]) 
129   
130      >>> johnson_lindenstrauss_min_dim([1e4, 1e5, 1e6], eps=0.1) 
131      array([ 7894,  9868, 11841]) 
132      """
133      eps = np.asarray(eps)
134      n_samples = np.asarray(n_samples)
135  
136      if np.any(eps <= 0.0) or np.any(eps >= 1):
137          raise ValueError("The JL bound is defined for eps in ]0, 1[, got %r" % eps)
138  
139      if np.any(n_samples <= 0):
140          raise ValueError(
141              "The JL bound is defined for n_samples greater than zero, got %r"
142              % n_samples
143          )
144  
145      denominator = (eps**2 / 2) - (eps**3 / 3)
146      return (4 * np.log(n_samples) / denominator).astype(np.int64)

149  def _check_density(density, n_features):
150      """Factorize density check according to Li et al."""
151      if density == "auto":
152          density = 1 / np.sqrt(n_features)
153  
154      elif density <= 0 or density > 1:
155          raise ValueError("Expected density in range ]0, 1], got: %r" % density)
156      return density
157

159  def _check_input_size(n_components, n_features):
160      """Factorize argument checking for random matrix generation."""
161      if n_components <= 0:
162          raise ValueError(
163              "n_components must be strictly positive, got %d" % n_components
164          )
165      if n_features <= 0:
166          raise ValueError("n_features must be strictly positive, got %d" % n_features)

169  def _gaussian_random_matrix(n_components, n_features, random_state=None):
170      """Generate a dense Gaussian random matrix. 
171   
172      The components of the random matrix are drawn from 
173   
174          N(0, 1.0 / n_components). 
175   
176      Read more in the :ref:`User Guide <gaussian_random_matrix>`. 
177   
178      Parameters 
179      ---------- 
180      n_components : int, 
181          Dimensionality of the target projection space. 
182   
183      n_features : int, 
184          Dimensionality of the original source space. 
185   
186      random_state : int, RandomState instance or None, default=None 
187          Controls the pseudo random number generator used to generate the matrix 
188          at fit time. 
189          Pass an int for reproducible output across multiple function calls. 
190          See :term:`Glossary <random_state>`. 
191   
192      Returns 
193      ------- 
194      components : ndarray of shape (n_components, n_features) 
195          The generated Gaussian random matrix. 
196   
197      See Also 
198      -------- 
199      GaussianRandomProjection 
200      """
201      _check_input_size(n_components, n_features)
202      rng = check_random_state(random_state)
203      components = rng.normal(
204          loc=0.0, scale=1.0 / np.sqrt(n_components), size=(n_components, n_features)
205      )
206      return components

209  def _sparse_random_matrix(n_components, n_features, density="auto", random_state=None):
210      """Generalized Achlioptas random sparse matrix for random projection. 
211   
212      Setting density to 1 / 3 will yield the original matrix by Dimitris 
213      Achlioptas while setting a lower value will yield the generalization 
214      by Ping Li et al. 
215   
216      If we note :math:`s = 1 / density`, the components of the random matrix are 
217      drawn from: 
218   
219        - -sqrt(s) / sqrt(n_components)   with probability 1 / 2s 
220        -  0                              with probability 1 - 1 / s 
221        - +sqrt(s) / sqrt(n_components)   with probability 1 / 2s 
222   
223      Read more in the :ref:`User Guide <sparse_random_matrix>`. 
224   
225      Parameters 
226      ---------- 
227      n_components : int, 
228          Dimensionality of the target projection space. 
229   
230      n_features : int, 
231          Dimensionality of the original source space. 
232   
233      density : float or 'auto', default='auto' 
234          Ratio of non-zero component in the random projection matrix in the 
235          range `(0, 1]` 
236   
237          If density = 'auto', the value is set to the minimum density 
238          as recommended by Ping Li et al.: 1 / sqrt(n_features). 
239   
240          Use density = 1 / 3.0 if you want to reproduce the results from 
241          Achlioptas, 2001. 
242   
243      random_state : int, RandomState instance or None, default=None 
244          Controls the pseudo random number generator used to generate the matrix 
245          at fit time. 
246          Pass an int for reproducible output across multiple function calls. 
247          See :term:`Glossary <random_state>`. 
248   
249      Returns 
250      ------- 
251      components : {ndarray, sparse matrix} of shape (n_components, n_features) 
252          The generated Gaussian random matrix. Sparse matrix will be of CSR 
253          format. 
254   
255      See Also 
256      -------- 
257      SparseRandomProjection 
258   
259      References 
260      ---------- 
261   
262      .. [1] Ping Li, T. Hastie and K. W. Church, 2006, 
263             "Very Sparse Random Projections". 
264             https://web.stanford.edu/~hastie/Papers/Ping/KDD06_rp.pdf 
265   
266      .. [2] D. Achlioptas, 2001, "Database-friendly random projections", 
267             https://cgi.di.uoa.gr/~optas/papers/jl.pdf 
268   
269      """
270      _check_input_size(n_components, n_features)
271      density = _check_density(density, n_features)
272      rng = check_random_state(random_state)
273  
274      if density == 1:
275          # skip index generation if totally dense
276          components = rng.binomial(1, 0.5, (n_components, n_features)) * 2 - 1
277          return 1 / np.sqrt(n_components) * components
278  
279      else:
280          # Generate location of non zero elements
281          indices = []
282          offset = 0
283          indptr = [offset]
284          for _ in range(n_components):
285              # find the indices of the non-zero components for row i
286              n_nonzero_i = rng.binomial(n_features, density)
287              indices_i = sample_without_replacement(
288                  n_features, n_nonzero_i, random_state=rng
289              )
290              indices.append(indices_i)
291              offset += n_nonzero_i
292              indptr.append(offset)
293  
294          indices = np.concatenate(indices)
295  
296          # Among non zero components the probability of the sign is 50%/50%
297          data = rng.binomial(1, 0.5, size=np.size(indices)) * 2 - 1
298  
299          # build the CSR structure by concatenating the rows
300          components = sp.csr_matrix(
301              (data, indices, indptr), shape=(n_components, n_features)
302          )
303  
304          return np.sqrt(1 / density) / np.sqrt(n_components) * components

307  class BaseRandomProjection(
308      ClassNamePrefixFeaturesOutMixin, TransformerMixin, BaseEstimator, metaclass=ABCMeta
309  ):
310      """Base class for random projections. 
311   
312      Warning: This class should not be used directly. 
313      Use derived classes instead. 
314      """
315  
316      _parameter_constraints: dict = {
317          "n_components": [
318              Interval(Integral, 1, None, closed="left"),
319              StrOptions({"auto"}),
320          ],
321          "eps": [Interval(Real, 0, None, closed="neither")],
322          "compute_inverse_components": ["boolean"],
323          "random_state": ["random_state"],
324      }
325  
326      @abstractmethod
327      def __init__(
328          self,
329          n_components="auto",
330          *,
331          eps=0.1,
332          compute_inverse_components=False,
333          random_state=None,
334      ):
335          self.n_components = n_components
336          self.eps = eps
337          self.compute_inverse_components = compute_inverse_components
338          self.random_state = random_state
339  
340      @abstractmethod
341      def _make_random_matrix(self, n_components, n_features):
342          """Generate the random projection matrix. 
343   
344          Parameters 
345          ---------- 
346          n_components : int, 
347              Dimensionality of the target projection space. 
348   
349          n_features : int, 
350              Dimensionality of the original source space. 
351   
352          Returns 
353          ------- 
354          components : {ndarray, sparse matrix} of shape (n_components, n_features) 
355              The generated random matrix. Sparse matrix will be of CSR format. 
356   
357          """
358  
359      def _compute_inverse_components(self):
360          """Compute the pseudo-inverse of the (densified) components."""
361          components = self.components_
362          if sp.issparse(components):
363              components = components.toarray()
364          return linalg.pinv(components, check_finite=False)
365  
366      @_fit_context(prefer_skip_nested_validation=True)
367      def fit(self, X, y=None):
368          """Generate a sparse random projection matrix. 
369   
370          Parameters 
371          ---------- 
372          X : {ndarray, sparse matrix} of shape (n_samples, n_features) 
373              Training set: only the shape is used to find optimal random 
374              matrix dimensions based on the theory referenced in the 
375              afore mentioned papers. 
376   
377          y : Ignored 
378              Not used, present here for API consistency by convention. 
379   
380          Returns 
381          ------- 
382          self : object 
383              BaseRandomProjection class instance. 
384          """
385          X = validate_data(
386              self, X, accept_sparse=["csr", "csc"], dtype=[np.float64, np.float32]
387          )
388  
389          n_samples, n_features = X.shape
390  
391          if self.n_components == "auto":
392              self.n_components_ = johnson_lindenstrauss_min_dim(
393                  n_samples=n_samples, eps=self.eps
394              )
395  
396              if self.n_components_ <= 0:
397                  raise ValueError(
398                      "eps=%f and n_samples=%d lead to a target dimension of "
399                      "%d which is invalid" % (self.eps, n_samples, self.n_components_)
400                  )
401  
402              elif self.n_components_ > n_features:
403                  raise ValueError(
404                      "eps=%f and n_samples=%d lead to a target dimension of "
405                      "%d which is larger than the original space with "
406                      "n_features=%d"
407                      % (self.eps, n_samples, self.n_components_, n_features)
408                  )
409          else:
410              if self.n_components > n_features:
411                  warnings.warn(
412                      "The number of components is higher than the number of"
413                      " features: n_features < n_components (%s < %s)."
414                      "The dimensionality of the problem will not be reduced."
415                      % (n_features, self.n_components),
416                      DataDimensionalityWarning,
417                  )
418  
419              self.n_components_ = self.n_components
420  
421          # Generate a projection matrix of size [n_components, n_features]
422          self.components_ = self._make_random_matrix(
423              self.n_components_, n_features
424          ).astype(X.dtype, copy=False)
425  
426          if self.compute_inverse_components:
427              self.inverse_components_ = self._compute_inverse_components()
428  
429          # Required by ClassNamePrefixFeaturesOutMixin.get_feature_names_out.
430          self._n_features_out = self.n_components
431  
432          return self
433  
434      def inverse_transform(self, X):
435          """Project data back to its original space. 
436   
437          Returns an array X_original whose transform would be X. Note that even 
438          if X is sparse, X_original is dense: this may use a lot of RAM. 
439   
440          If `compute_inverse_components` is False, the inverse of the components is 
441          computed during each call to `inverse_transform` which can be costly. 
442   
443          Parameters 
444          ---------- 
445          X : {array-like, sparse matrix} of shape (n_samples, n_components) 
446              Data to be transformed back. 
447   
448          Returns 
449          ------- 
450          X_original : ndarray of shape (n_samples, n_features) 
451              Reconstructed data. 
452          """
453          check_is_fitted(self)
454  
455          X = check_array(X, dtype=[np.float64, np.float32], accept_sparse=("csr", "csc"))
456  
457          if self.compute_inverse_components:
458              return X @ self.inverse_components_.T
459  
460          inverse_components = self._compute_inverse_components()
461          return X @ inverse_components.T
462  
463      def __sklearn_tags__(self):
464          tags = super().__sklearn_tags__()
465          tags.transformer_tags.preserves_dtype = ["float64", "float32"]
466          tags.input_tags.sparse = True
467          return tags

470  class GaussianRandomProjection(BaseRandomProjection):
471      """Reduce dimensionality through Gaussian random projection. 
472   
473      The components of the random matrix are drawn from N(0, 1 / n_components). 
474   
475      Read more in the :ref:`User Guide <gaussian_random_matrix>`. 
476   
477      .. versionadded:: 0.13 
478   
479      Parameters 
480      ---------- 
481      n_components : int or 'auto', default='auto' 
482          Dimensionality of the target projection space. 
483   
484          n_components can be automatically adjusted according to the 
485          number of samples in the dataset and the bound given by the 
486          Johnson-Lindenstrauss lemma. In that case the quality of the 
487          embedding is controlled by the ``eps`` parameter. 
488   
489          It should be noted that Johnson-Lindenstrauss lemma can yield 
490          very conservative estimated of the required number of components 
491          as it makes no assumption on the structure of the dataset. 
492   
493      eps : float, default=0.1 
494          Parameter to control the quality of the embedding according to 
495          the Johnson-Lindenstrauss lemma when `n_components` is set to 
496          'auto'. The value should be strictly positive. 
497   
498          Smaller values lead to better embedding and higher number of 
499          dimensions (n_components) in the target projection space. 
500   
501      compute_inverse_components : bool, default=False 
502          Learn the inverse transform by computing the pseudo-inverse of the 
503          components during fit. Note that computing the pseudo-inverse does not 
504          scale well to large matrices. 
505   
506      random_state : int, RandomState instance or None, default=None 
507          Controls the pseudo random number generator used to generate the 
508          projection matrix at fit time. 
509          Pass an int for reproducible output across multiple function calls. 
510          See :term:`Glossary <random_state>`. 
511   
512      Attributes 
513      ---------- 
514      n_components_ : int 
515          Concrete number of components computed when n_components="auto". 
516   
517      components_ : ndarray of shape (n_components, n_features) 
518          Random matrix used for the projection. 
519   
520      inverse_components_ : ndarray of shape (n_features, n_components) 
521          Pseudo-inverse of the components, only computed if 
522          `compute_inverse_components` is True. 
523   
524          .. versionadded:: 1.1 
525   
526      n_features_in_ : int 
527          Number of features seen during :term:`fit`. 
528   
529          .. versionadded:: 0.24 
530   
531      feature_names_in_ : ndarray of shape (`n_features_in_`,) 
532          Names of features seen during :term:`fit`. Defined only when `X` 
533          has feature names that are all strings. 
534   
535          .. versionadded:: 1.0 
536   
537      See Also 
538      -------- 
539      SparseRandomProjection : Reduce dimensionality through sparse 
540          random projection. 
541   
542      Examples 
543      -------- 
544      >>> import numpy as np 
545      >>> from sklearn.random_projection import GaussianRandomProjection 
546      >>> rng = np.random.RandomState(42) 
547      >>> X = rng.rand(25, 3000) 
548      >>> transformer = GaussianRandomProjection(random_state=rng) 
549      >>> X_new = transformer.fit_transform(X) 
550      >>> X_new.shape 
551      (25, 2759) 
552      """
553  
554      def __init__(
555          self,
556          n_components="auto",
557          *,
558          eps=0.1,
559          compute_inverse_components=False,
560          random_state=None,
561      ):
562          super().__init__(
563              n_components=n_components,
564              eps=eps,
565              compute_inverse_components=compute_inverse_components,
566              random_state=random_state,
567          )
568  
569      def _make_random_matrix(self, n_components, n_features):
570          """Generate the random projection matrix. 
571   
572          Parameters 
573          ---------- 
574          n_components : int, 
575              Dimensionality of the target projection space. 
576   
577          n_features : int, 
578              Dimensionality of the original source space. 
579   
580          Returns 
581          ------- 
582          components : ndarray of shape (n_components, n_features) 
583              The generated random matrix. 
584          """
585          random_state = check_random_state(self.random_state)
586          return _gaussian_random_matrix(
587              n_components, n_features, random_state=random_state
588          )
589  
590      def transform(self, X):
591          """Project the data by using matrix product with the random matrix. 
592   
593          Parameters 
594          ---------- 
595          X : {ndarray, sparse matrix} of shape (n_samples, n_features) 
596              The input data to project into a smaller dimensional space. 
597   
598          Returns 
599          ------- 
600          X_new : ndarray of shape (n_samples, n_components) 
601              Projected array. 
602          """
603          check_is_fitted(self)
604          X = validate_data(
605              self,
606              X,
607              accept_sparse=["csr", "csc"],
608              reset=False,
609              dtype=[np.float64, np.float32],
610          )
611  
612          return X @ self.components_.T

615  class SparseRandomProjection(BaseRandomProjection):
616      """Reduce dimensionality through sparse random projection. 
617   
618      Sparse random matrix is an alternative to dense random 
619      projection matrix that guarantees similar embedding quality while being 
620      much more memory efficient and allowing faster computation of the 
621      projected data. 
622   
623      If we note `s = 1 / density` the components of the random matrix are 
624      drawn from: 
625   
626      .. code-block:: text 
627   
628        -sqrt(s) / sqrt(n_components)   with probability 1 / 2s 
629         0                              with probability 1 - 1 / s 
630        +sqrt(s) / sqrt(n_components)   with probability 1 / 2s 
631   
632      Read more in the :ref:`User Guide <sparse_random_matrix>`. 
633   
634      .. versionadded:: 0.13 
635   
636      Parameters 
637      ---------- 
638      n_components : int or 'auto', default='auto' 
639          Dimensionality of the target projection space. 
640   
641          n_components can be automatically adjusted according to the 
642          number of samples in the dataset and the bound given by the 
643          Johnson-Lindenstrauss lemma. In that case the quality of the 
644          embedding is controlled by the ``eps`` parameter. 
645   
646          It should be noted that Johnson-Lindenstrauss lemma can yield 
647          very conservative estimated of the required number of components 
648          as it makes no assumption on the structure of the dataset. 
649   
650      density : float or 'auto', default='auto' 
651          Ratio in the range (0, 1] of non-zero component in the random 
652          projection matrix. 
653   
654          If density = 'auto', the value is set to the minimum density 
655          as recommended by Ping Li et al.: 1 / sqrt(n_features). 
656   
657          Use density = 1 / 3.0 if you want to reproduce the results from 
658          Achlioptas, 2001. 
659   
660      eps : float, default=0.1 
661          Parameter to control the quality of the embedding according to 
662          the Johnson-Lindenstrauss lemma when n_components is set to 
663          'auto'. This value should be strictly positive. 
664   
665          Smaller values lead to better embedding and higher number of 
666          dimensions (n_components) in the target projection space. 
667   
668      dense_output : bool, default=False 
669          If True, ensure that the output of the random projection is a 
670          dense numpy array even if the input and random projection matrix 
671          are both sparse. In practice, if the number of components is 
672          small the number of zero components in the projected data will 
673          be very small and it will be more CPU and memory efficient to 
674          use a dense representation. 
675   
676          If False, the projected data uses a sparse representation if 
677          the input is sparse. 
678   
679      compute_inverse_components : bool, default=False 
680          Learn the inverse transform by computing the pseudo-inverse of the 
681          components during fit. Note that the pseudo-inverse is always a dense 
682          array, even if the training data was sparse. This means that it might be 
683          necessary to call `inverse_transform` on a small batch of samples at a 
684          time to avoid exhausting the available memory on the host. Moreover, 
685          computing the pseudo-inverse does not scale well to large matrices. 
686   
687      random_state : int, RandomState instance or None, default=None 
688          Controls the pseudo random number generator used to generate the 
689          projection matrix at fit time. 
690          Pass an int for reproducible output across multiple function calls. 
691          See :term:`Glossary <random_state>`. 
692   
693      Attributes 
694      ---------- 
695      n_components_ : int 
696          Concrete number of components computed when n_components="auto". 
697

778      def _make_random_matrix(self, n_components, n_features):
779          """Generate the random projection matrix 
780   
781          Parameters 
782          ---------- 
783          n_components : int 
784              Dimensionality of the target projection space. 
785   
786          n_features : int 
787              Dimensionality of the original source space. 
788   
789          Returns 
790          ------- 
791          components : sparse matrix of shape (n_components, n_features) 
792              The generated random matrix in CSR format. 
793   
794          """
795          random_state = check_random_state(self.random_state)
796          self.density_ = _check_density(self.density, n_features)
797          return _sparse_random_matrix(
798              n_components, n_features, density=self.density_, random_state=random_state
799          )

801      def transform(self, X):
802          """Project the data by using matrix product with the random matrix. 
803   
804          Parameters 
805          ---------- 
806          X : {ndarray, sparse matrix} of shape (n_samples, n_features) 
807              The input data to project into a smaller dimensional space. 
808   
809          Returns 
810          ------- 
811          X_new : {ndarray, sparse matrix} of shape (n_samples, n_components) 
812              Projected array. It is a sparse matrix only when the input is sparse and 
813              `dense_output = False`. 
814          """
815          check_is_fitted(self)
816          X = validate_data(
817              self,
818              X,
819              accept_sparse=["csr", "csc"],
820              reset=False,
821              dtype=[np.float64, np.float32],
822          )
823  
824          return safe_sparse_dot(X, self.components_.T, dense_output=self.dense_output)
