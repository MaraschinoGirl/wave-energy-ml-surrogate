6            def pydev_debugger_exc_info():
7                type, value, traceback = fun()
8                if type == ImportError:
9                    # we should not show frame added by plugin_import call
10                   if traceback and hasattr(traceback, "tb_next"):
11                       return type, value, traceback.tb_next
12               return type, value, traceback

14           return pydev_debugger_exc_info
