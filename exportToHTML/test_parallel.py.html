22   @pytest.mark.parametrize("n_jobs", [1, 2])
23   @pytest.mark.parametrize("backend", ["loky", "threading", "multiprocessing"])
24   def test_configuration_passes_through_to_joblib(n_jobs, backend):
25       # Tests that the global global configuration is passed to joblib jobs
26   
27       with config_context(working_memory=123):
28           results = Parallel(n_jobs=n_jobs, backend=backend)(
29               delayed(get_working_memory)() for _ in range(2)
30           )

32       assert_array_equal(results, [123] * 2)

35   def test_parallel_delayed_warnings():
36       """Informative warnings should be raised when mixing sklearn and joblib API"""
37       # We should issue a warning when one wants to use sklearn.utils.fixes.Parallel
38       # with joblib.delayed. The config will not be propagated to the workers.
39       warn_msg = "`sklearn.utils.parallel.Parallel` needs to be used in conjunction"
40       with pytest.warns(UserWarning, match=warn_msg) as records:
41           Parallel()(joblib.delayed(time.sleep)(0) for _ in range(10))
42       assert len(records) == 10
43   
44       # We should issue a warning if one wants to use sklearn.utils.fixes.delayed with
45       # joblib.Parallel
46       warn_msg = (
47           "`sklearn.utils.parallel.delayed` should be used with "
48           "`sklearn.utils.parallel.Parallel` to make it possible to propagate"
49       )
50       with pytest.warns(UserWarning, match=warn_msg) as records:
51           joblib.Parallel()(delayed(time.sleep)(0) for _ in range(10))
52       assert len(records) == 10

55   @pytest.mark.parametrize("n_jobs", [1, 2])
56   def test_dispatch_config_parallel(n_jobs):
57       """Check that we properly dispatch the configuration in parallel processing. 
58    
59       Non-regression test for: 
60       https://github.com/scikit-learn/scikit-learn/issues/25239 
61       """
62       pd = pytest.importorskip("pandas")
63       iris = load_iris(as_frame=True)
64   
65       class TransformerRequiredDataFrame(StandardScaler):
66           def fit(self, X, y=None):
67               assert isinstance(X, pd.DataFrame), "X should be a DataFrame"
68               return super().fit(X, y)
69   
70           def transform(self, X, y=None):
71               assert isinstance(X, pd.DataFrame), "X should be a DataFrame"
72               return super().transform(X, y)
73   
74       dropper = make_column_transformer(
75           ("drop", [0]),
76           remainder="passthrough",
77           n_jobs=n_jobs,
78       )
79       param_grid = {"randomforestclassifier__max_depth": [1, 2, 3]}
80       search_cv = GridSearchCV(
81           make_pipeline(
82               dropper,
83               TransformerRequiredDataFrame(),
84               RandomForestClassifier(n_estimators=5, n_jobs=n_jobs),
85           ),
86           param_grid,
87           cv=5,
88           n_jobs=n_jobs,
89           error_score="raise",  # this search should not fail
90       )
91   
92       # make sure that `fit` would fail in case we don't request dataframe
93       with pytest.raises(AssertionError, match="X should be a DataFrame"):
94           search_cv.fit(iris.data, iris.target)
95   
96       with config_context(transform_output="pandas"):
97           # we expect each intermediate steps to output a DataFrame
98           search_cv.fit(iris.data, iris.target)
99   
100      assert not np.isnan(search_cv.cv_results_["mean_test_score"]).any()
