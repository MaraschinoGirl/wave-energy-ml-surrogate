10   def _minimize_cobyqa(fun, x0, args=(), bounds=None, constraints=(),
11                        callback=None, disp=False, maxfev=None, maxiter=None,
12                        f_target=-np.inf, feasibility_tol=1e-8,
13                        initial_tr_radius=1.0, final_tr_radius=1e-6, scale=False,
14                        **unknown_options):
15       """ 
16       Minimize a scalar function of one or more variables using the 
17       Constrained Optimization BY Quadratic Approximations (COBYQA) algorithm [1]_. 
18    
19       .. versionadded:: 1.14.0 
20    
21       Options 
22       ------- 
23       disp : bool 
24           Set to True to print information about the optimization procedure. 
25           Default is ``False``. 
26       maxfev : int 
27           Maximum number of function evaluations. Default is ``500 * n``, where 
28           ``n`` is the number of variables. 
29       maxiter : int 
30           Maximum number of iterations. Default is ``1000 * n``, where ``n`` is 
31           the number of variables. 
32       f_target : float 
33           Target value for the objective function. The optimization procedure is 
34           terminated when the objective function value of a feasible point (see 
35           `feasibility_tol` below) is less than or equal to this target. Default 
36           is ``-numpy.inf``. 
37       feasibility_tol : float 
38           Absolute tolerance for the constraint violation. Default is ``1e-8``. 
39       initial_tr_radius : float 
40           Initial trust-region radius. Typically, this value should be in the 
41           order of one tenth of the greatest expected change to the variables. 
42           Default is ``1.0``. 
43       final_tr_radius : float 
44           Final trust-region radius. It should indicate the accuracy required in 
45           the final values of the variables. If provided, this option overrides 
46           the value of `tol` in the `minimize` function. Default is ``1e-6``. 
47       scale : bool 
48           Set to True to scale the variables according to the bounds. If True and 
49           if all the lower and upper bounds are finite, the variables are scaled 
50           to be within the range :math:`[-1, 1]`. If any of the lower or upper 
51           bounds is infinite, the variables are not scaled. Default is ``False``. 
52    
53       References 
54       ---------- 
55       .. [1] COBYQA 
56              https://www.cobyqa.com/stable/ 
57       """
58       from .._lib.cobyqa import minimize  # import here to avoid circular imports
59   
60       _check_unknown_options(unknown_options)
61       options = {
62           'disp': bool(disp),
63           'maxfev': int(maxfev) if maxfev is not None else 500 * len(x0),
64           'maxiter': int(maxiter) if maxiter is not None else 1000 * len(x0),
65           'target': float(f_target),
66           'feasibility_tol': float(feasibility_tol),
67           'radius_init': float(initial_tr_radius),
68           'radius_final': float(final_tr_radius),
69           'scale': bool(scale),
70       }
71       with COBYQA_LOCK:
72           return minimize(fun, x0, args, bounds, constraints, callback, options)
