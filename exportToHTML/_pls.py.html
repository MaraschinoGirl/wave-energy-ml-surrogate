41   def _pinv2_old(a):
42       # Used previous scipy pinv2 that was updated in:
43       # https://github.com/scipy/scipy/pull/10067
44       # We can not set `cond` or `rcond` for pinv2 in scipy >= 1.3 to keep the
45       # same behavior of pinv2 for scipy < 1.3, because the condition used to
46       # determine the rank is dependent on the output of svd.
47       u, s, vh = svd(a, full_matrices=False, check_finite=False)
48   
49       t = u.dtype.char.lower()
50       factor = {"f": 1e3, "d": 1e6}
51       cond = np.max(s) * factor[t] * np.finfo(t).eps
52       rank = np.sum(s > cond)
53   
54       u = u[:, :rank]
55       u /= s[:rank]
56       return np.transpose(np.conjugate(np.dot(u, vh[:rank])))

59   def _get_first_singular_vectors_power_method(
60       X, Y, mode="A", max_iter=500, tol=1e-06, norm_y_weights=False
61   ):
62       """Return the first left and right singular vectors of X'Y. 
63    
64       Provides an alternative to the svd(X'Y) and uses the power method instead. 
65       With norm_y_weights to True and in mode A, this corresponds to the 
66       algorithm section 11.3 of the Wegelin's review, except this starts at the 
67       "update saliences" part. 
68       """
69   
70       eps = np.finfo(X.dtype).eps
71       try:
72           y_score = next(col for col in Y.T if np.any(np.abs(col) > eps))
73       except StopIteration as e:
74           raise StopIteration("y residual is constant") from e
75   
76       x_weights_old = 100  # init to big value for first convergence check
77   
78       if mode == "B":
79           # Precompute pseudo inverse matrices
80           # Basically: X_pinv = (X.T X)^-1 X.T
81           # Which requires inverting a (n_features, n_features) matrix.
82           # As a result, and as detailed in the Wegelin's review, CCA (i.e. mode
83           # B) will be unstable if n_features > n_samples or n_targets >
84           # n_samples
85           X_pinv, Y_pinv = _pinv2_old(X), _pinv2_old(Y)
86   
87       for i in range(max_iter):
88           if mode == "B":
89               x_weights = np.dot(X_pinv, y_score)
90           else:
91               x_weights = np.dot(X.T, y_score) / np.dot(y_score, y_score)
92   
93           x_weights /= np.sqrt(np.dot(x_weights, x_weights)) + eps
94           x_score = np.dot(X, x_weights)
95   
96           if mode == "B":
97               y_weights = np.dot(Y_pinv, x_score)
98           else:
99               y_weights = np.dot(Y.T, x_score) / np.dot(x_score.T, x_score)
100  
101          if norm_y_weights:
102              y_weights /= np.sqrt(np.dot(y_weights, y_weights)) + eps
103  
104          y_score = np.dot(Y, y_weights) / (np.dot(y_weights, y_weights) + eps)
105  
106          x_weights_diff = x_weights - x_weights_old
107          if np.dot(x_weights_diff, x_weights_diff) < tol or Y.shape[1] == 1:
108              break
109          x_weights_old = x_weights
110  
111      n_iter = i + 1
112      if n_iter == max_iter:
113          warnings.warn("Maximum number of iterations reached", ConvergenceWarning)
114  
115      return x_weights, y_weights, n_iter

118  def _get_first_singular_vectors_svd(X, Y):
119      """Return the first left and right singular vectors of X'Y. 
120   
121      Here the whole SVD is computed. 
122      """
123      C = np.dot(X.T, Y)
124      U, _, Vt = svd(C, full_matrices=False)
125      return U[:, 0], Vt[0, :]

128  def _center_scale_xy(X, Y, scale=True):
129      """Center X, Y and scale if the scale parameter==True 
130   
131      Returns 
132      ------- 
133          X, Y, x_mean, y_mean, x_std, y_std 
134      """
135      # center
136      x_mean = X.mean(axis=0)
137      X -= x_mean
138      y_mean = Y.mean(axis=0)
139      Y -= y_mean
140      # scale
141      if scale:
142          x_std = X.std(axis=0, ddof=1)
143          x_std[x_std == 0.0] = 1.0
144          X /= x_std
145          y_std = Y.std(axis=0, ddof=1)
146          y_std[y_std == 0.0] = 1.0
147          Y /= y_std
148      else:
149          x_std = np.ones(X.shape[1])
150          y_std = np.ones(Y.shape[1])
151      return X, Y, x_mean, y_mean, x_std, y_std

154  def _svd_flip_1d(u, v):
155      """Same as svd_flip but works on 1d arrays, and is inplace"""
156      # svd_flip would force us to convert to 2d array and would also return 2d
157      # arrays. We don't want that.
158      biggest_abs_val_idx = np.argmax(np.abs(u))
159      sign = np.sign(u[biggest_abs_val_idx])
160      u *= sign
161      v *= sign

186  class _PLS(
187      ClassNamePrefixFeaturesOutMixin,
188      TransformerMixin,
189      RegressorMixin,
190      MultiOutputMixin,
191      BaseEstimator,
192      metaclass=ABCMeta,
193  ):
194      """Partial Least Squares (PLS) 
195   
196      This class implements the generic PLS algorithm. 
197   
198      Main ref: Wegelin, a survey of Partial Least Squares (PLS) methods, 
199      with emphasis on the two-block case 
200      https://stat.uw.edu/sites/default/files/files/reports/2000/tr371.pdf 
201      """
202  
203      _parameter_constraints: dict = {
204          "n_components": [Interval(Integral, 1, None, closed="left")],
205          "scale": ["boolean"],
206          "deflation_mode": [StrOptions({"regression", "canonical"})],
207          "mode": [StrOptions({"A", "B"})],
208          "algorithm": [StrOptions({"svd", "nipals"})],
209          "max_iter": [Interval(Integral, 1, None, closed="left")],
210          "tol": [Interval(Real, 0, None, closed="left")],
211          "copy": ["boolean"],
212      }
213  
214      @abstractmethod
215      def __init__(
216          self,
217          n_components=2,
218          *,
219          scale=True,
220          deflation_mode="regression",
221          mode="A",
222          algorithm="nipals",
223          max_iter=500,
224          tol=1e-06,
225          copy=True,
226      ):
227          self.n_components = n_components
228          self.deflation_mode = deflation_mode
229          self.mode = mode
230          self.scale = scale
231          self.algorithm = algorithm
232          self.max_iter = max_iter
233          self.tol = tol
234          self.copy = copy
235  
236      @_fit_context(prefer_skip_nested_validation=True)
237      def fit(self, X, y=None, Y=None):
238          """Fit model to data. 
239   
240          Parameters 
241          ---------- 
242          X : array-like of shape (n_samples, n_features) 
243              Training vectors, where `n_samples` is the number of samples and 
244              `n_features` is the number of predictors. 
245   
246          y : array-like of shape (n_samples,) or (n_samples, n_targets) 
247              Target vectors, where `n_samples` is the number of samples and 
248              `n_targets` is the number of response variables. 
249   
250          Y : array-like of shape (n_samples,) or (n_samples, n_targets) 
251              Target vectors, where `n_samples` is the number of samples and 
252              `n_targets` is the number of response variables. 
253   
254              .. deprecated:: 1.5 
255                 `Y` is deprecated in 1.5 and will be removed in 1.7. Use `y` instead. 
256   
257          Returns 
258          ------- 
259          self : object 
260              Fitted model. 
261          """
262          y = _deprecate_Y_when_required(y, Y)
263  
264          check_consistent_length(X, y)
265          X = validate_data(
266              self,
267              X,
268              dtype=np.float64,
269              force_writeable=True,
270              copy=self.copy,
271              ensure_min_samples=2,
272          )
273          y = check_array(
274              y,
275              input_name="y",
276              dtype=np.float64,
277              force_writeable=True,
278              copy=self.copy,
279              ensure_2d=False,
280          )
281          if y.ndim == 1:
282              self._predict_1d = True
283              y = y.reshape(-1, 1)
284          else:
285              self._predict_1d = False
286  
287          n = X.shape[0]
288          p = X.shape[1]
289          q = y.shape[1]
290  
291          n_components = self.n_components
292          # With PLSRegression n_components is bounded by the rank of (X.T X) see
293          # Wegelin page 25. With CCA and PLSCanonical, n_components is bounded
294          # by the rank of X and the rank of Y: see Wegelin page 12
295          rank_upper_bound = (
296              min(n, p) if self.deflation_mode == "regression" else min(n, p, q)
297          )
298          if n_components > rank_upper_bound:
299              raise ValueError(
300                  f"`n_components` upper bound is {rank_upper_bound}. "
301                  f"Got {n_components} instead. Reduce `n_components`."
302              )
303  
304          self._norm_y_weights = self.deflation_mode == "canonical"  # 1.1
305          norm_y_weights = self._norm_y_weights
306  
307          # Scale (in place)
308          Xk, yk, self._x_mean, self._y_mean, self._x_std, self._y_std = _center_scale_xy(
309              X, y, self.scale
310          )
311  
312          self.x_weights_ = np.zeros((p, n_components))  # U
313          self.y_weights_ = np.zeros((q, n_components))  # V
314          self._x_scores = np.zeros((n, n_components))  # Xi
315          self._y_scores = np.zeros((n, n_components))  # Omega
316          self.x_loadings_ = np.zeros((p, n_components))  # Gamma
317          self.y_loadings_ = np.zeros((q, n_components))  # Delta
318          self.n_iter_ = []
319  
320          # This whole thing corresponds to the algorithm in section 4.1 of the
321          # review from Wegelin. See above for a notation mapping from code to
322          # paper.
323          y_eps = np.finfo(yk.dtype).eps
324          for k in range(n_components):
325              # Find first left and right singular vectors of the X.T.dot(Y)
326              # cross-covariance matrix.
327              if self.algorithm == "nipals":
328                  # Replace columns that are all close to zero with zeros
329                  yk_mask = np.all(np.abs(yk) < 10 * y_eps, axis=0)
330                  yk[:, yk_mask] = 0.0
331  
332                  try:
333                      (
334                          x_weights,
335                          y_weights,
336                          n_iter_,
337                      ) = _get_first_singular_vectors_power_method(
338                          Xk,
339                          yk,
340                          mode=self.mode,
341                          max_iter=self.max_iter,
342                          tol=self.tol,
343                          norm_y_weights=norm_y_weights,
344                      )
345                  except StopIteration as e:
346                      if str(e) != "y residual is constant":
347                          raise
348                      warnings.warn(f"y residual is constant at iteration {k}")
349                      break
350  
351                  self.n_iter_.append(n_iter_)
352  
353              elif self.algorithm == "svd":
354                  x_weights, y_weights = _get_first_singular_vectors_svd(Xk, yk)
355  
356              # inplace sign flip for consistency across solvers and archs
357              _svd_flip_1d(x_weights, y_weights)
358  
359              # compute scores, i.e. the projections of X and Y
360              x_scores = np.dot(Xk, x_weights)
361              if norm_y_weights:
362                  y_ss = 1
363              else:
364                  y_ss = np.dot(y_weights, y_weights)
365              y_scores = np.dot(yk, y_weights) / y_ss
366  
367              # Deflation: subtract rank-one approx to obtain Xk+1 and Yk+1
368              x_loadings = np.dot(x_scores, Xk) / np.dot(x_scores, x_scores)
369              Xk -= np.outer(x_scores, x_loadings)
370  
371              if self.deflation_mode == "canonical":
372                  # regress Yk on y_score
373                  y_loadings = np.dot(y_scores, yk) / np.dot(y_scores, y_scores)
374                  yk -= np.outer(y_scores, y_loadings)
375              if self.deflation_mode == "regression":
376                  # regress Yk on x_score
377                  y_loadings = np.dot(x_scores, yk) / np.dot(x_scores, x_scores)
378                  yk -= np.outer(x_scores, y_loadings)
379  
380              self.x_weights_[:, k] = x_weights
381              self.y_weights_[:, k] = y_weights
382              self._x_scores[:, k] = x_scores
383              self._y_scores[:, k] = y_scores
384              self.x_loadings_[:, k] = x_loadings
385              self.y_loadings_[:, k] = y_loadings
386  
387          # X was approximated as Xi . Gamma.T + X_(R+1)
388          # Xi . Gamma.T is a sum of n_components rank-1 matrices. X_(R+1) is
389          # whatever is left to fully reconstruct X, and can be 0 if X is of rank
390          # n_components.
391          # Similarly, y was approximated as Omega . Delta.T + y_(R+1)
392  
393          # Compute transformation matrices (rotations_). See User Guide.
394          self.x_rotations_ = np.dot(
395              self.x_weights_,
396              pinv2(np.dot(self.x_loadings_.T, self.x_weights_), check_finite=False),
397          )
398          self.y_rotations_ = np.dot(
399              self.y_weights_,
400              pinv2(np.dot(self.y_loadings_.T, self.y_weights_), check_finite=False),
401          )
402          self.coef_ = np.dot(self.x_rotations_, self.y_loadings_.T)
403          self.coef_ = (self.coef_ * self._y_std).T / self._x_std
404          self.intercept_ = self._y_mean
405          self._n_features_out = self.x_rotations_.shape[1]
406          return self
407  
408      def transform(self, X, y=None, Y=None, copy=True):
409          """Apply the dimension reduction. 
410   
411          Parameters 
412          ---------- 
413          X : array-like of shape (n_samples, n_features) 
414              Samples to transform. 
415   
416          y : array-like of shape (n_samples, n_targets), default=None 
417              Target vectors. 
418   
419          Y : array-like of shape (n_samples, n_targets), default=None 
420              Target vectors. 
421   
422              .. deprecated:: 1.5 
423                 `Y` is deprecated in 1.5 and will be removed in 1.7. Use `y` instead. 
424   
425          copy : bool, default=True 
426              Whether to copy `X` and `Y`, or perform in-place normalization. 
427   
428          Returns 
429          ------- 
430          x_scores, y_scores : array-like or tuple of array-like 
431              Return `x_scores` if `Y` is not given, `(x_scores, y_scores)` otherwise. 
432          """
433          y = _deprecate_Y_when_optional(y, Y)
434  
435          check_is_fitted(self)
436          X = validate_data(self, X, copy=copy, dtype=FLOAT_DTYPES, reset=False)
437          # Normalize
438          X -= self._x_mean
439          X /= self._x_std
440          # Apply rotation
441          x_scores = np.dot(X, self.x_rotations_)
442          if y is not None:
443              y = check_array(
444                  y, input_name="y", ensure_2d=False, copy=copy, dtype=FLOAT_DTYPES
445              )
446              if y.ndim == 1:
447                  y = y.reshape(-1, 1)
448              y -= self._y_mean
449              y /= self._y_std
450              y_scores = np.dot(y, self.y_rotations_)
451              return x_scores, y_scores
452  
453          return x_scores
454  
455      def inverse_transform(self, X, y=None, Y=None):
456          """Transform data back to its original space. 
457   
458          Parameters 
459          ---------- 
460          X : array-like of shape (n_samples, n_components) 
461              New data, where `n_samples` is the number of samples 
462              and `n_components` is the number of pls components. 
463   
464          y : array-like of shape (n_samples,) or (n_samples, n_components) 
465              New target, where `n_samples` is the number of samples 
466              and `n_components` is the number of pls components. 
467   
468          Y : array-like of shape (n_samples, n_components) 
469              New target, where `n_samples` is the number of samples 
470              and `n_components` is the number of pls components. 
471   
472              .. deprecated:: 1.5 
473                 `Y` is deprecated in 1.5 and will be removed in 1.7. Use `y` instead. 
474   
475          Returns 
476          ------- 
477          X_reconstructed : ndarray of shape (n_samples, n_features) 
478              Return the reconstructed `X` data. 
479   
480          y_reconstructed : ndarray of shape (n_samples, n_targets) 
481              Return the reconstructed `X` target. Only returned when `y` is given. 
482   
483          Notes 
484          ----- 
485          This transformation will only be exact if `n_components=n_features`. 
486          """
487          y = _deprecate_Y_when_optional(y, Y)
488  
489          check_is_fitted(self)
490          X = check_array(X, input_name="X", dtype=FLOAT_DTYPES)
491          # From pls space to original space
492          X_reconstructed = np.matmul(X, self.x_loadings_.T)
493          # Denormalize
494          X_reconstructed *= self._x_std
495          X_reconstructed += self._x_mean
496  
497          if y is not None:
498              y = check_array(y, input_name="y", dtype=FLOAT_DTYPES)
499              # From pls space to original space
500              y_reconstructed = np.matmul(y, self.y_loadings_.T)
501              # Denormalize
502              y_reconstructed *= self._y_std
503              y_reconstructed += self._y_mean
504              return X_reconstructed, y_reconstructed
505  
506          return X_reconstructed
507  
508      def predict(self, X, copy=True):
509          """Predict targets of given samples. 
510   
511          Parameters 
512          ---------- 
513          X : array-like of shape (n_samples, n_features) 
514              Samples. 
515   
516          copy : bool, default=True 
517              Whether to copy `X` and `Y`, or perform in-place normalization. 
518   
519          Returns 
520          ------- 
521          y_pred : ndarray of shape (n_samples,) or (n_samples, n_targets) 
522              Returns predicted values. 
523   
524          Notes 
525          ----- 
526          This call requires the estimation of a matrix of shape 
527          `(n_features, n_targets)`, which may be an issue in high dimensional 
528          space. 
529          """
530          check_is_fitted(self)
531          X = validate_data(self, X, copy=copy, dtype=FLOAT_DTYPES, reset=False)
532          # Only center X but do not scale it since the coefficients are already scaled
533          X -= self._x_mean
534          Ypred = X @ self.coef_.T + self.intercept_
535          return Ypred.ravel() if self._predict_1d else Ypred
536  
537      def fit_transform(self, X, y=None):
538          """Learn and apply the dimension reduction on the train data. 
539   
540          Parameters 
541          ---------- 
542          X : array-like of shape (n_samples, n_features) 
543              Training vectors, where `n_samples` is the number of samples and 
544              `n_features` is the number of predictors. 
545   
546          y : array-like of shape (n_samples, n_targets), default=None 
547              Target vectors, where `n_samples` is the number of samples and 
548              `n_targets` is the number of response variables. 
549   
550          Returns 
551          ------- 
552          self : ndarray of shape (n_samples, n_components) 
553              Return `x_scores` if `Y` is not given, `(x_scores, y_scores)` otherwise. 
554          """
555          return self.fit(X, y).transform(X, y)

564  class PLSRegression(_PLS):
565      """PLS regression. 
566   
567      PLSRegression is also known as PLS2 or PLS1, depending on the number of 
568      targets. 
569   
570      For a comparison between other cross decomposition algorithms, see 
571      :ref:`sphx_glr_auto_examples_cross_decomposition_plot_compare_cross_decomposition.py`. 
572   
573      Read more in the :ref:`User Guide <cross_decomposition>`. 
574   
575      .. versionadded:: 0.8 
576   
577      Parameters 
578      ---------- 
579      n_components : int, default=2 
580          Number of components to keep. Should be in `[1, n_features]`. 
581   
582      scale : bool, default=True 
583          Whether to scale `X` and `Y`. 
584   
585      max_iter : int, default=500 
586          The maximum number of iterations of the power method when 
587          `algorithm='nipals'`. Ignored otherwise. 
588   
589      tol : float, default=1e-06 
590          The tolerance used as convergence criteria in the power method: the 
591          algorithm stops whenever the squared norm of `u_i - u_{i-1}` is less 
592          than `tol`, where `u` corresponds to the left singular vector. 
593   
594      copy : bool, default=True 
595          Whether to copy `X` and `Y` in :term:`fit` before applying centering, 
596          and potentially scaling. If `False`, these operations will be done 
597          inplace, modifying both arrays. 
598   
599      Attributes 
600      ---------- 
601      x_weights_ : ndarray of shape (n_features, n_components) 
602          The left singular vectors of the cross-covariance matrices of each 
603          iteration. 
604   
605      y_weights_ : ndarray of shape (n_targets, n_components) 
606          The right singular vectors of the cross-covariance matrices of each 
607          iteration. 
608   
609      x_loadings_ : ndarray of shape (n_features, n_components) 
610          The loadings of `X`. 
611   
612      y_loadings_ : ndarray of shape (n_targets, n_components) 
613          The loadings of `Y`. 
614   
615      x_scores_ : ndarray of shape (n_samples, n_components) 
616          The transformed training samples. 
617   
618      y_scores_ : ndarray of shape (n_samples, n_components) 
619          The transformed training targets. 
620   
621      x_rotations_ : ndarray of shape (n_features, n_components) 
622          The projection matrix used to transform `X`. 
623   
624      y_rotations_ : ndarray of shape (n_targets, n_components) 
625          The projection matrix used to transform `Y`. 
626   
627      coef_ : ndarray of shape (n_target, n_features)

792          `Y = X @ coef_.T + intercept_`.

820      >>> plsca.fit(X, y)

825      _parameter_constraints: dict = {**_PLS._parameter_constraints}

969  class PLSSVD(ClassNamePrefixFeaturesOutMixin, TransformerMixin, BaseEstimator):
970      """Partial Least Square SVD. 
971   
972      This transformer simply performs a SVD on the cross-covariance matrix 
973      `X'Y`. It is able to project both the training data `X` and the targets 
974      `Y`. The training data `X` is projected on the left singular vectors, while 
975      the targets are projected on the right singular vectors. 
976   
977      Read more in the :ref:`User Guide <cross_decomposition>`. 
978   
979      .. versionadded:: 0.8 
980   
981      Parameters 
982      ---------- 
983      n_components : int, default=2 
984          The number of components to keep. Should be in `[1, 
985          min(n_samples, n_features, n_targets)]`. 
986   
987      scale : bool, default=True 
988          Whether to scale `X` and `Y`. 
989   
990      copy : bool, default=True 
991          Whether to copy `X` and `Y` in fit before applying centering, and 
992          potentially scaling. If `False`, these operations will be done inplace, 
993          modifying both arrays. 
994   
995      Attributes 
996      ---------- 
997      x_weights_ : ndarray of shape (n_features, n_components) 
998          The left singular vectors of the SVD of the cross-covariance matrix. 
999          Used to project `X` in :meth:`transform`. 
1000  
1001     y_weights_ : ndarray of (n_targets, n_components) 
1002         The right singular vectors of the SVD of the cross-covariance matrix. 
1003         Used to project `X` in :meth:`transform`. 
1004  
1005     n_features_in_ : int 
1006         Number of features seen during :term:`fit`. 
1007  
1008     feature_names_in_ : ndarray of shape (`n_features_in_`,) 
1009         Names of features seen during :term:`fit`. Defined only when `X` 
1010         has feature names that are all strings. 
1011  
1012         .. versionadded:: 1.0 
1013  
1014     See Also 
1015     -------- 
1016     PLSCanonical : Partial Least Squares transformer and regressor. 
1017     CCA : Canonical Correlation Analysis. 
1018  
1019     Examples 
1020     -------- 
1021     >>> import numpy as np 
1022     >>> from sklearn.cross_decomposition import PLSSVD 
1023     >>> X = np.array([[0., 0., 1.], 
1024     ...               [1., 0., 0.], 
1025     ...               [2., 2., 2.], 
1026     ...               [2., 5., 4.]]) 
1027     >>> y = np.array([[0.1, -0.2], 
1028     ...               [0.9, 1.1], 
1029     ...               [6.2, 5.9], 
1030     ...               [11.9, 12.3]]) 
1031     >>> pls = PLSSVD(n_components=2).fit(X, y) 
1032     >>> X_c, y_c = pls.transform(X, y) 
1033     >>> X_c.shape, y_c.shape 
1034     ((4, 2), (4, 2)) 
1035     """
1036 
1037     _parameter_constraints: dict = {
1038         "n_components": [Interval(Integral, 1, None, closed="left")],
1039         "scale": ["boolean"],
1040         "copy": ["boolean"],
1041     }
1042 
1043     def __init__(self, n_components=2, *, scale=True, copy=True):
1044         self.n_components = n_components
1045         self.scale = scale
1046         self.copy = copy
1047 
1048     @_fit_context(prefer_skip_nested_validation=True)
1049     def fit(self, X, y=None, Y=None):
1050         """Fit model to data. 
1051  
1052         Parameters 
1053         ---------- 
1054         X : array-like of shape (n_samples, n_features) 
1055             Training samples. 
1056  
1057         y : array-like of shape (n_samples,) or (n_samples, n_targets) 
1058             Targets. 
1059  
1060         Y : array-like of shape (n_samples,) or (n_samples, n_targets) 
1061             Targets. 
1062  
1063             .. deprecated:: 1.5 
1064                `Y` is deprecated in 1.5 and will be removed in 1.7. Use `y` instead. 
1065  
1066         Returns 
1067         ------- 
1068         self : object 
1069             Fitted estimator. 
1070         """
1071         y = _deprecate_Y_when_required(y, Y)
1072         check_consistent_length(X, y)
1073         X = validate_data(
1074             self,
1075             X,
1076             dtype=np.float64,
1077             force_writeable=True,
1078             copy=self.copy,
1079             ensure_min_samples=2,
1080         )
1081         y = check_array(
1082             y,
1083             input_name="y",
1084             dtype=np.float64,
1085             force_writeable=True,
1086             copy=self.copy,
1087             ensure_2d=False,
1088         )
1089         if y.ndim == 1:
1090             y = y.reshape(-1, 1)
1091 
1092         # we'll compute the SVD of the cross-covariance matrix = X.T.dot(y)
1093         # This matrix rank is at most min(n_samples, n_features, n_targets) so
1094         # n_components cannot be bigger than that.
1095         n_components = self.n_components
1096         rank_upper_bound = min(X.shape[0], X.shape[1], y.shape[1])
1097         if n_components > rank_upper_bound:
1098             raise ValueError(
1099                 f"`n_components` upper bound is {rank_upper_bound}. "
1100                 f"Got {n_components} instead. Reduce `n_components`."
1101             )
1102 
1103         X, y, self._x_mean, self._y_mean, self._x_std, self._y_std = _center_scale_xy(
1104             X, y, self.scale
1105         )
1106 
1107         # Compute SVD of cross-covariance matrix
1108         C = np.dot(X.T, y)
1109         U, s, Vt = svd(C, full_matrices=False)
1110         U = U[:, :n_components]
1111         Vt = Vt[:n_components]
1112         U, Vt = svd_flip(U, Vt)
1113         V = Vt.T
1114 
1115         self.x_weights_ = U
1116         self.y_weights_ = V
1117         self._n_features_out = self.x_weights_.shape[1]
1118         return self
1119 
1120     def transform(self, X, y=None, Y=None):
1121         """ 
1122         Apply the dimensionality reduction. 
1123  
1124         Parameters 
1125         ---------- 
1126         X : array-like of shape (n_samples, n_features) 
1127             Samples to be transformed. 
1128  
1129         y : array-like of shape (n_samples,) or (n_samples, n_targets), \ 
1130                 default=None 
1131             Targets. 
1132  
1133         Y : array-like of shape (n_samples,) or (n_samples, n_targets), \ 
1134                 default=None 
1135             Targets. 
1136  
1137             .. deprecated:: 1.5 
1138                `Y` is deprecated in 1.5 and will be removed in 1.7. Use `y` instead. 
1139  
1140         Returns 
1141         ------- 
1142         x_scores : array-like or tuple of array-like 
1143             The transformed data `X_transformed` if `Y is not None`, 
1144             `(X_transformed, Y_transformed)` otherwise. 
1145         """
1146         y = _deprecate_Y_when_optional(y, Y)
1147         check_is_fitted(self)
1148         X = validate_data(self, X, dtype=np.float64, reset=False)
1149         Xr = (X - self._x_mean) / self._x_std
1150         x_scores = np.dot(Xr, self.x_weights_)
1151         if y is not None:
1152             y = check_array(y, input_name="y", ensure_2d=False, dtype=np.float64)
1153             if y.ndim == 1:
1154                 y = y.reshape(-1, 1)
1155             yr = (y - self._y_mean) / self._y_std
1156             y_scores = np.dot(yr, self.y_weights_)
1157             return x_scores, y_scores
1158         return x_scores
1159 
1160     def fit_transform(self, X, y=None):
1161         """Learn and apply the dimensionality reduction. 
1162  
1163         Parameters 
1164         ---------- 
1165         X : array-like of shape (n_samples, n_features) 
1166             Training samples. 
1167  
1168         y : array-like of shape (n_samples,) or (n_samples, n_targets), \ 
1169                 default=None 
1170             Targets. 
1171  
1172         Returns 
1173         ------- 
1174         out : array-like or tuple of array-like 
1175             The transformed data `X_transformed` if `Y is not None`, 
1176             `(X_transformed, Y_transformed)` otherwise. 
1177         """
1178         return self.fit(X, y).transform(X, y)
