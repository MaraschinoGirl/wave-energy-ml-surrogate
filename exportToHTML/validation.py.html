32   # case that we merge a new public function without kwarg only by mistake, which would

34   def _deprecate_positional_args(func=None, *, version="1.3"):
35       """Decorator for methods that issues warnings for positional arguments. 
36    
37       Using the keyword-only argument syntax in pep 3102, arguments after the 
38       * will issue a warning when passed as a positional argument. 
39    
40       Parameters 
41       ---------- 
42       func : callable, default=None 
43           Function to check arguments on. 
44       version : callable, default="1.3" 
45           The version when positional arguments will result in error. 
46       """
47   
48       def _inner_deprecate_positional_args(f):
49           sig = signature(f)
50           kwonly_args = []
51           all_args = []
52   
53           for name, param in sig.parameters.items():
54               if param.kind == Parameter.POSITIONAL_OR_KEYWORD:
55                   all_args.append(name)
56               elif param.kind == Parameter.KEYWORD_ONLY:
57                   kwonly_args.append(name)
58   
59           @wraps(f)
60           def inner_f(*args, **kwargs):
61               extra_args = len(args) - len(all_args)
62               if extra_args <= 0:
63                   return f(*args, **kwargs)
64   
65               # extra_args > 0
66               args_msg = [
67                   "{}={}".format(name, arg)
68                   for name, arg in zip(kwonly_args[:extra_args], args[-extra_args:])
69               ]
70               args_msg = ", ".join(args_msg)
71               warnings.warn(
72                   (
73                       f"Pass {args_msg} as keyword args. From version "
74                       f"{version} passing these as positional arguments "
75                       "will result in an error"
76                   ),
77                   FutureWarning,
78               )
79               kwargs.update(zip(sig.parameters, args))
80               return f(**kwargs)
81   
82           return inner_f
83   
84       if func is not None:
85           return _inner_deprecate_positional_args(func)
86   
87       return _inner_deprecate_positional_args

90   def _assert_all_finite(
91       X, allow_nan=False, msg_dtype=None, estimator_name=None, input_name=""
92   ):
93       """Like assert_all_finite, but only for ndarray."""
94   
95       xp, is_array_api = get_namespace(X)
96   
97       if _get_config()["assume_finite"]:
98           return
99   
100      X = xp.asarray(X)
101  
102      # for object dtype data, we only check for NaNs (GH-13254)
103      if not is_array_api and X.dtype == np.dtype("object") and not allow_nan:
104          if _object_dtype_isnan(X).any():
105              raise ValueError("Input contains NaN")
106  
107      # We need only consider float arrays, hence can early return for all else.
108      if not xp.isdtype(X.dtype, ("real floating", "complex floating")):
109          return
110  
111      # First try an O(n) time, O(1) space solution for the common case that
112      # everything is finite; fall back to O(n) space `np.isinf/isnan` or custom
113      # Cython implementation to prevent false positives and provide a detailed
114      # error message.
115      with np.errstate(over="ignore"):
116          first_pass_isfinite = xp.isfinite(xp.sum(X))
117      if first_pass_isfinite:
118          return
119  
120      _assert_all_finite_element_wise(
121          X,
122          xp=xp,
123          allow_nan=allow_nan,
124          msg_dtype=msg_dtype,
125          estimator_name=estimator_name,
126          input_name=input_name,
127      )

130  def _assert_all_finite_element_wise(
131      X, *, xp, allow_nan, msg_dtype=None, estimator_name=None, input_name=""
132  ):
133      # Cython implementation doesn't support FP16 or complex numbers
134      use_cython = (
135          xp is np and X.data.contiguous and X.dtype.type in {np.float32, np.float64}
136      )
137      if use_cython:
138          out = cy_isfinite(X.reshape(-1), allow_nan=allow_nan)
139          has_nan_error = False if allow_nan else out == FiniteStatus.has_nan
140          has_inf = out == FiniteStatus.has_infinite
141      else:
142          has_inf = xp.any(xp.isinf(X))
143          has_nan_error = False if allow_nan else xp.any(xp.isnan(X))
144      if has_inf or has_nan_error:
145          if has_nan_error:
146              type_err = "NaN"
147          else:
148              msg_dtype = msg_dtype if msg_dtype is not None else X.dtype
149              type_err = f"infinity or a value too large for {msg_dtype!r}"
150          padded_input_name = input_name + " " if input_name else ""
151          msg_err = f"Input {padded_input_name}contains {type_err}."
152          if estimator_name and input_name == "X" and has_nan_error:
153              # Improve the error message on how to handle missing values in
154              # scikit-learn.
155              msg_err += (
156                  f"\n{estimator_name} does not accept missing values"
157                  " encoded as NaN natively. For supervised learning, you might want"
158                  " to consider sklearn.ensemble.HistGradientBoostingClassifier and"
159                  " Regressor which accept missing values encoded as NaNs natively."
160                  " Alternatively, it is possible to preprocess the data, for"
161                  " instance by using an imputer transformer in a pipeline or drop"
162                  " samples with missing values. See"
163                  " https://scikit-learn.org/stable/modules/impute.html"
164                  " You can find a list of all estimators that handle NaN values"
165                  " at the following page:"
166                  " https://scikit-learn.org/stable/modules/impute.html"
167                  "#estimators-that-handle-nan-values"
168              )
169          raise ValueError(msg_err)

176      estimator_name=None,
177      input_name="",
178  ):
179      """Throw a ValueError if X contains NaN or infinity. 
180   
181      Parameters 
182      ----------

214          input_name=input_name,

218  def as_float_array(
219      X, *, copy=True, force_all_finite="deprecated", ensure_all_finite=None
220  ):
221      """Convert an array-like to an array of floats. 
222   
223      The new dtype will be np.float32 or np.float64, depending on the original 
224      type. The function can create a copy or modify the argument depending 
225      on the argument copy. 
226   
227      Parameters 
228      ---------- 
229      X : {array-like, sparse matrix} 
230          The input data. 
231   
232      copy : bool, default=True 
233          If True, a copy of X will be created. If False, a copy may still be 
234          returned if X's dtype is not a floating point type. 
235   
236      force_all_finite : bool or 'allow-nan', default=True 
237          Whether to raise an error on np.inf, np.nan, pd.NA in X. The 
238          possibilities are: 
239   
240          - True: Force all values of X to be finite. 
241          - False: accepts np.inf, np.nan, pd.NA in X. 
242          - 'allow-nan': accepts only np.nan and pd.NA values in X. Values cannot 
243            be infinite. 
244   
245          .. versionadded:: 0.20 
246             ``force_all_finite`` accepts the string ``'allow-nan'``. 
247   
248          .. versionchanged:: 0.23 
249             Accepts `pd.NA` and converts it into `np.nan` 
250   
251          .. deprecated:: 1.6 
252             `force_all_finite` was renamed to `ensure_all_finite` and will be removed 
253             in 1.8. 
254   
255      ensure_all_finite : bool or 'allow-nan', default=True 
256          Whether to raise an error on np.inf, np.nan, pd.NA in X. The 
257          possibilities are: 
258   
259          - True: Force all values of X to be finite. 
260          - False: accepts np.inf, np.nan, pd.NA in X. 
261          - 'allow-nan': accepts only np.nan and pd.NA values in X. Values cannot 
262            be infinite. 
263   
264          .. versionadded:: 1.6 
265             `force_all_finite` was renamed to `ensure_all_finite`. 
266   
267      Returns 
268      ------- 
269      XT : {ndarray, sparse matrix} 
270          An array of type float. 
271   
272      Examples 
273      -------- 
274      >>> from sklearn.utils import as_float_array 
275      >>> import numpy as np 
276      >>> array = np.array([0, 0, 1, 2, 2], dtype=np.int64) 
277      >>> as_float_array(array) 
278      array([0., 0., 1., 2., 2.]) 
279      """
280      ensure_all_finite = _deprecate_force_all_finite(force_all_finite, ensure_all_finite)
281  
282      if isinstance(X, np.matrix) or (
283          not isinstance(X, np.ndarray) and not sp.issparse(X)
284      ):
285          return check_array(
286              X,
287              accept_sparse=["csr", "csc", "coo"],
288              dtype=np.float64,
289              copy=copy,
290              ensure_all_finite=ensure_all_finite,
291              ensure_2d=False,
292          )
293      elif sp.issparse(X) and X.dtype in [np.float32, np.float64]:
294          return X.copy() if copy else X
295      elif X.dtype in [np.float32, np.float64]:  # is numpy array
296          return X.copy("F" if X.flags["F_CONTIGUOUS"] else "C") if copy else X
297      else:
298          if X.dtype.kind in "uib" and X.dtype.itemsize <= 4:
299              return_dtype = np.float32
300          else:
301              return_dtype = np.float64
302          return X.astype(return_dtype)

328  def _num_features(X):
329      """Return the number of features in an array-like X. 
330   
331      This helper function tries hard to avoid to materialize an array version 
332      of X unless necessary. For instance, if X is a list of lists, 
333      this function will return the length of the first element, assuming 
334      that subsequent elements are all lists of the same length without 
335      checking. 
336      Parameters 
337      ---------- 
338      X : array-like 
339          array-like to get the number of features. 
340   
341      Returns 
342      ------- 
343      features : int 
344          Number of features 
345      """
346      type_ = type(X)
347      if type_.__module__ == "builtins":
348          type_name = type_.__qualname__
349      else:
350          type_name = f"{type_.__module__}.{type_.__qualname__}"
351      message = f"Unable to find the number of features from X of type {type_name}"
352      if not hasattr(X, "__len__") and not hasattr(X, "shape"):
353          if not hasattr(X, "__array__"):
354              raise TypeError(message)
355          # Only convert X to a numpy array if there is no cheaper, heuristic
356          # option.
357          X = np.asarray(X)
358  
359      if hasattr(X, "shape"):
360          if not hasattr(X.shape, "__len__") or len(X.shape) <= 1:
361              message += f" with shape {X.shape}"
362              raise TypeError(message)
363          return X.shape[1]
364  
365      first_sample = X[0]
366  
367      # Do not consider an array-like of strings or dicts to be a 2D array
368      if isinstance(first_sample, (str, bytes, dict)):
369          message += f" where the samples are of type {type(first_sample).__qualname__}"
370          raise TypeError(message)
371  
372      try:
373          # If X is a list of lists, for instance, we assume that all nested
374          # lists have the same length without checking or converting to
375          # a numpy array to keep this function call as cheap as possible.
376          return len(first_sample)
377      except Exception as err:
378          raise TypeError(message) from err

381  def _num_samples(x):
382      """Return number of samples in array-like x."""
383      message = "Expected sequence or array-like, got %s" % type(x)
384      if hasattr(x, "fit") and callable(x.fit):
385          # Don't get num_samples from an ensembles length!
386          raise TypeError(message)
387  
388      if _use_interchange_protocol(x):
389          return x.__dataframe__().num_rows()
390  
391      if not hasattr(x, "__len__") and not hasattr(x, "shape"):
392          if hasattr(x, "__array__"):
393              x = np.asarray(x)
394          else:
395              raise TypeError(message)
396  
397      if hasattr(x, "shape") and x.shape is not None:
398          if len(x.shape) == 0:
399              raise TypeError(
400                  "Input should have at least 1 dimension i.e. satisfy "
401                  f"`len(x.shape) > 0`, got scalar `{x!r}` instead."
402              )
403          # Check that shape is returning an integer or default to len
404          # Dask dataframes may not return numeric shape[0] value
405          if isinstance(x.shape[0], numbers.Integral):
406              return x.shape[0]
407  
408      try:
409          return len(x)
410      except TypeError as type_error:
411          raise TypeError(message) from type_error

414  def check_memory(memory):
415      """Check that ``memory`` is joblib.Memory-like. 
416   
417      joblib.Memory-like means that ``memory`` can be converted into a 
418      joblib.Memory instance (typically a str denoting the ``location``) 
419      or has the same interface (has a ``cache`` method). 
420   
421      Parameters 
422      ---------- 
423      memory : None, str or object with the joblib.Memory interface 
424          - If string, the location where to create the `joblib.Memory` interface. 
425          - If None, no caching is done and the Memory object is completely transparent. 
426   
427      Returns 
428      ------- 
429      memory : object with the joblib.Memory interface 
430          A correct joblib.Memory object. 
431   
432      Raises 
433      ------ 
434      ValueError 
435          If ``memory`` is not joblib.Memory-like. 
436   
437      Examples 
438      -------- 
439      >>> from sklearn.utils.validation import check_memory 
440      >>> check_memory("caching_dir") 
441      Memory(location=caching_dir/joblib) 
442      """
443      if memory is None or isinstance(memory, str):
444          memory = joblib.Memory(location=memory, verbose=0)
445      elif not hasattr(memory, "cache"):
446          raise ValueError(
447              "'memory' should be None, a string or have the same"
448              " interface as joblib.Memory."
449              " Got memory='{}' instead.".format(memory)
450          )
451      return memory

454  def check_consistent_length(*arrays):
455      """Check that all arrays have consistent first dimensions. 
456   
457      Checks whether all objects in arrays have the same shape or length. 
458   
459      Parameters 
460      ---------- 
461      *arrays : list or tuple of input objects. 
462          Objects that will be checked for consistent length. 
463   
464      Examples 
465      -------- 
466      >>> from sklearn.utils.validation import check_consistent_length 
467      >>> a = [1, 2, 3] 
468      >>> b = [2, 3, 4] 
469      >>> check_consistent_length(a, b) 
470      """
471  
472      lengths = [_num_samples(X) for X in arrays if X is not None]
473      uniques = np.unique(lengths)
474      if len(uniques) > 1:
475          raise ValueError(
476              "Found input variables with inconsistent numbers of samples: %r"
477              % [int(l) for l in lengths]
478          )

481  def _make_indexable(iterable):
482      """Ensure iterable supports indexing or convert to an indexable variant. 
483   
484      Convert sparse matrices to csr and other non-indexable iterable to arrays. 
485      Let `None` and indexable objects (e.g. pandas dataframes) pass unchanged. 
486   
487      Parameters 
488      ---------- 
489      iterable : {list, dataframe, ndarray, sparse matrix} or None 
490          Object to be converted to an indexable iterable. 
491      """
492      if sp.issparse(iterable):
493          return iterable.tocsr()
494      elif hasattr(iterable, "__getitem__") or hasattr(iterable, "iloc"):
495          return iterable
496      elif iterable is None:
497          return iterable
498      return np.array(iterable)

501  def indexable(*iterables):
502      """Make arrays indexable for cross-validation. 
503   
504      Checks consistent length, passes through None, and ensures that everything 
505      can be indexed by converting sparse matrices to csr and converting 
506      non-iterable objects to arrays. 
507   
508      Parameters 
509      ---------- 
510      *iterables : {lists, dataframes, ndarrays, sparse matrices} 
511          List of objects to ensure sliceability. 
512   
513      Returns 
514      ------- 
515      result : list of {ndarray, sparse matrix, dataframe} or None 
516          Returns a list containing indexable arrays (i.e. NumPy array, 
517          sparse matrix, or dataframe) or `None`. 
518   
519      Examples 
520      -------- 
521      >>> from sklearn.utils import indexable 
522      >>> from scipy.sparse import csr_matrix 
523      >>> import numpy as np 
524      >>> iterables = [ 
525      ...     [1, 2, 3], np.array([2, 3, 4]), None, csr_matrix([[5], [6], [7]]) 
526      ... ] 
527      >>> indexable(*iterables) 
528      [[1, 2, 3], array([2, 3, 4]), None, <...Sparse...dtype 'int64'...shape (3, 1)>] 
529      """
530  
531      result = [_make_indexable(X) for X in iterables]
532      check_consistent_length(*result)
533      return result

536  def _ensure_sparse_format(
537      sparse_container,
538      accept_sparse,
539      dtype,
540      copy,
541      ensure_all_finite,
542      accept_large_sparse,
543      estimator_name=None,
544      input_name="",
545  ):
546      """Convert a sparse container to a given format. 
547   
548      Checks the sparse format of `sparse_container` and converts if necessary. 
549   
550      Parameters 
551      ---------- 
552      sparse_container : sparse matrix or array 
553          Input to validate and convert. 
554   
555      accept_sparse : str, bool or list/tuple of str 
556          String[s] representing allowed sparse matrix formats ('csc', 
557          'csr', 'coo', 'dok', 'bsr', 'lil', 'dia'). If the input is sparse but 
558          not in the allowed format, it will be converted to the first listed 
559          format. True allows the input to be any format. False means 
560          that a sparse matrix input will raise an error. 
561   
562      dtype : str, type or None 
563          Data type of result. If None, the dtype of the input is preserved. 
564   
565      copy : bool 
566          Whether a forced copy will be triggered. If copy=False, a copy might 
567          be triggered by a conversion. 
568   
569      ensure_all_finite : bool or 'allow-nan' 
570          Whether to raise an error on np.inf, np.nan, pd.NA in X. The 
571          possibilities are: 
572   
573          - True: Force all values of X to be finite. 
574          - False: accepts np.inf, np.nan, pd.NA in X. 
575          - 'allow-nan': accepts only np.nan and pd.NA values in X. Values cannot 
576            be infinite. 
577   
578          .. versionadded:: 0.20 
579             ``ensure_all_finite`` accepts the string ``'allow-nan'``. 
580   
581          .. versionchanged:: 0.23 
582             Accepts `pd.NA` and converts it into `np.nan` 
583   
584   
585      estimator_name : str, default=None 
586          The estimator name, used to construct the error message. 
587   
588      input_name : str, default="" 
589          The data name used to construct the error message. In particular 
590          if `input_name` is "X" and the data has NaN values and 
591          allow_nan is False, the error message will link to the imputer 
592          documentation. 
593   
594      Returns 
595      ------- 
596      sparse_container_converted : sparse matrix or array 
597          Sparse container (matrix/array) that is ensured to have an allowed type. 
598      """
599      if dtype is None:
600          dtype = sparse_container.dtype
601  
602      changed_format = False
603      sparse_container_type_name = type(sparse_container).__name__
604  
605      if isinstance(accept_sparse, str):
606          accept_sparse = [accept_sparse]
607  
608      # Indices dtype validation
609      _check_large_sparse(sparse_container, accept_large_sparse)
610  
611      if accept_sparse is False:
612          padded_input = " for " + input_name if input_name else ""
613          raise TypeError(
614              f"Sparse data was passed{padded_input}, but dense data is required. "
615              "Use '.toarray()' to convert to a dense numpy array."
616          )
617      elif isinstance(accept_sparse, (list, tuple)):
618          if len(accept_sparse) == 0:
619              raise ValueError(
620                  "When providing 'accept_sparse' as a tuple or list, it must contain at "
621                  "least one string value."
622              )
623          # ensure correct sparse format
624          if sparse_container.format not in accept_sparse:
625              # create new with correct sparse
626              sparse_container = sparse_container.asformat(accept_sparse[0])
627              changed_format = True
628      elif accept_sparse is not True:
629          # any other type
630          raise ValueError(
631              "Parameter 'accept_sparse' should be a string, boolean or list of strings."
632              f" You provided 'accept_sparse={accept_sparse}'."
633          )
634  
635      if dtype != sparse_container.dtype:
636          # convert dtype
637          sparse_container = sparse_container.astype(dtype)
638      elif copy and not changed_format:
639          # force copy
640          sparse_container = sparse_container.copy()
641  
642      if ensure_all_finite:
643          if not hasattr(sparse_container, "data"):
644              warnings.warn(
645                  f"Can't check {sparse_container.format} sparse matrix for nan or inf.",
646                  stacklevel=2,
647              )
648          else:
649              _assert_all_finite(
650                  sparse_container.data,
651                  allow_nan=ensure_all_finite == "allow-nan",
652                  estimator_name=estimator_name,
653                  input_name=input_name,
654              )
655  
656      # TODO: Remove when the minimum version of SciPy supported is 1.12
657      # With SciPy sparse arrays, conversion from DIA format to COO, CSR, or BSR
658      # triggers the use of `np.int64` indices even if the data is such that it could
659      # be more efficiently represented with `np.int32` indices.
660      # https://github.com/scipy/scipy/issues/19245 Since not all scikit-learn
661      # algorithms support large indices, the following code downcasts to `np.int32`
662      # indices when it's safe to do so.
663      if changed_format:
664          # accept_sparse is specified to a specific format and a conversion occurred
665          requested_sparse_format = accept_sparse[0]

692  def _pandas_dtype_needs_early_conversion(pd_dtype):
693      """Return True if pandas extension pd_dtype need to be converted early."""
694      # Check these early for pandas versions without extension dtypes
695      from pandas import SparseDtype
696      from pandas.api.types import (
697          is_bool_dtype,
698          is_float_dtype,
699          is_integer_dtype,
700      )
701  
702      if is_bool_dtype(pd_dtype):
703          # bool and extension booleans need early conversion because __array__
704          # converts mixed dtype dataframes into object dtypes
705          return True
706  
707      if isinstance(pd_dtype, SparseDtype):
708          # Sparse arrays will be converted later in `check_array`
709          return False
710  
711      try:
712          from pandas.api.types import is_extension_array_dtype
713      except ImportError:
714          return False
715  
716      if isinstance(pd_dtype, SparseDtype) or not is_extension_array_dtype(pd_dtype):
717          # Sparse arrays will be converted later in `check_array`
718          # Only handle extension arrays for integer and floats
719          return False
720      elif is_float_dtype(pd_dtype):
721          # Float ndarrays can normally support nans. They need to be converted
722          # first to map pd.NA to np.nan
723          return True
724      elif is_integer_dtype(pd_dtype):
725          # XXX: Warn when converting from a high integer to a float
726          return True
727  
728      return False

736  def check_array(
737      array,
738      accept_sparse=False,
739      *,
740      accept_large_sparse=True,
741      dtype="numeric",
742      order=None,
743      copy=False,
744      force_writeable=False,
745      force_all_finite="deprecated",
746      ensure_all_finite=None,
747      ensure_non_negative=False,
748      ensure_2d=True,
749      allow_nd=False,
750      ensure_min_samples=1,
751      ensure_min_features=1,
752      estimator=None,
753      input_name="",
754  ):
755      """Input validation on an array, list, sparse matrix or similar. 
756   
757      By default, the input is checked to be a non-empty 2D array containing 
758      only finite values. If the dtype of the array is object, attempt 
759      converting to float, raising on failure. 
760   
761      Parameters 
762      ---------- 
763      array : object 
764          Input object to check / convert. 
765   
766      accept_sparse : str, bool or list/tuple of str, default=False 
767          String[s] representing allowed sparse matrix formats, such as 'csc', 
768          'csr', etc. If the input is sparse but not in the allowed format, 
769          it will be converted to the first listed format. True allows the input 
770          to be any format. False means that a sparse matrix input will 
771          raise an error. 
772   
773      accept_large_sparse : bool, default=True 
774          If a CSR, CSC, COO or BSR sparse matrix is supplied and accepted by 
775          accept_sparse, accept_large_sparse=False will cause it to be accepted 
776          only if its indices are stored with a 32-bit dtype. 
777   
778          .. versionadded:: 0.20 
779   
780      dtype : 'numeric', type, list of type or None, default='numeric' 
781          Data type of result. If None, the dtype of the input is preserved. 
782          If "numeric", dtype is preserved unless array.dtype is object. 
783          If dtype is a list of types, conversion on the first type is only 
784          performed if the dtype of the input is not in the list. 
785   
786      order : {'F', 'C'} or None, default=None 
787          Whether an array will be forced to be fortran or c-style. 
788          When order is None (default), then if copy=False, nothing is ensured 
789          about the memory layout of the output array; otherwise (copy=True) 
790          the memory layout of the returned array is kept as close as possible 
791          to the original array. 
792   
793      copy : bool, default=False 
794          Whether a forced copy will be triggered. If copy=False, a copy might 
795          be triggered by a conversion. 
796   
797      force_writeable : bool, default=False 
798          Whether to force the output array to be writeable. If True, the returned array 
799          is guaranteed to be writeable, which may require a copy. Otherwise the 
800          writeability of the input array is preserved. 
801   
802          .. versionadded:: 1.6 
803   
804      force_all_finite : bool or 'allow-nan', default=True 
805          Whether to raise an error on np.inf, np.nan, pd.NA in array. The 
806          possibilities are: 
807   
808          - True: Force all values of array to be finite. 
809          - False: accepts np.inf, np.nan, pd.NA in array. 
810          - 'allow-nan': accepts only np.nan and pd.NA values in array. Values 
811            cannot be infinite. 
812   
813          .. versionadded:: 0.20 
814             ``force_all_finite`` accepts the string ``'allow-nan'``. 
815   
816          .. versionchanged:: 0.23 
817             Accepts `pd.NA` and converts it into `np.nan` 
818   
819          .. deprecated:: 1.6 
820             `force_all_finite` was renamed to `ensure_all_finite` and will be removed 
821             in 1.8. 
822   
823      ensure_all_finite : bool or 'allow-nan', default=True 
824          Whether to raise an error on np.inf, np.nan, pd.NA in array. The 
825          possibilities are: 
826   
827          - True: Force all values of array to be finite. 
828          - False: accepts np.inf, np.nan, pd.NA in array. 
829          - 'allow-nan': accepts only np.nan and pd.NA values in array. Values 
830            cannot be infinite. 
831   
832          .. versionadded:: 1.6 
833             `force_all_finite` was renamed to `ensure_all_finite`. 
834   
835      ensure_non_negative : bool, default=False 
836          Make sure the array has only non-negative values. If True, an array that 
837          contains negative values will raise a ValueError. 
838   
839          .. versionadded:: 1.6 
840   
841      ensure_2d : bool, default=True 
842          Whether to raise a value error if array is not 2D. 
843   
844      allow_nd : bool, default=False 
845          Whether to allow array.ndim > 2. 
846   
847      ensure_min_samples : int, default=1 
848          Make sure that the array has a minimum number of samples in its first 
849          axis (rows for a 2D array). Setting to 0 disables this check. 
850   
851      ensure_min_features : int, default=1 
852          Make sure that the 2D array has some minimum number of features 
853          (columns). The default value of 1 rejects empty datasets. 
854          This check is only enforced when the input data has effectively 2 
855          dimensions or is originally 1D and ``ensure_2d`` is True. Setting to 0 
856          disables this check. 
857   
858      estimator : str or estimator instance, default=None 
859          If passed, include the name of the estimator in warning messages. 
860   
861      input_name : str, default="" 
862          The data name used to construct the error message. In particular 
863          if `input_name` is "X" and the data has NaN values and 
864          allow_nan is False, the error message will link to the imputer 
865          documentation. 
866   
867          .. versionadded:: 1.1.0 
868   
869      Returns 
870      ------- 
871      array_converted : object 
872          The converted and validated array. 
873   
874      Examples 
875      -------- 
876      >>> from sklearn.utils.validation import check_array 
877      >>> X = [[1, 2, 3], [4, 5, 6]] 
878      >>> X_checked = check_array(X) 
879      >>> X_checked 
880      array([[1, 2, 3], [4, 5, 6]]) 
881      """
882      ensure_all_finite = _deprecate_force_all_finite(force_all_finite, ensure_all_finite)
883  
884      if isinstance(array, np.matrix):
885          raise TypeError(
886              "np.matrix is not supported. Please convert to a numpy array with "
887              "np.asarray. For more information see: "
888              "https://numpy.org/doc/stable/reference/generated/numpy.matrix.html"
889          )
890  
891      xp, is_array_api_compliant = get_namespace(array)
892  
893      # store reference to original array to check if copy is needed when
894      # function returns
895      array_orig = array
896  
897      # store whether originally we wanted numeric dtype
898      dtype_numeric = isinstance(dtype, str) and dtype == "numeric"
899  
900      dtype_orig = getattr(array, "dtype", None)
901      if not is_array_api_compliant and not hasattr(dtype_orig, "kind"):
902          # not a data type (e.g. a column named dtype in a pandas DataFrame)
903          dtype_orig = None
904  
905      # check if the object contains several dtypes (typically a pandas
906      # DataFrame), and store them. If not, store None.
907      dtypes_orig = None
908      pandas_requires_conversion = False
909      # track if we have a Series-like object to raise a better error message
910      type_if_series = None
911      if hasattr(array, "dtypes") and hasattr(array.dtypes, "__array__"):
912          # throw warning if columns are sparse. If all columns are sparse, then
913          # array.sparse exists and sparsity will be preserved (later).
914          with suppress(ImportError):
915              from pandas import SparseDtype
916  
917              def is_sparse(dtype):
918                  return isinstance(dtype, SparseDtype)
919  
920              if not hasattr(array, "sparse") and array.dtypes.apply(is_sparse).any():
921                  warnings.warn(
922                      "pandas.DataFrame with sparse columns found."
923                      "It will be converted to a dense numpy array."
924                  )
925  
926          dtypes_orig = list(array.dtypes)
927          pandas_requires_conversion = any(
928              _pandas_dtype_needs_early_conversion(i) for i in dtypes_orig
929          )
930          if all(isinstance(dtype_iter, np.dtype) for dtype_iter in dtypes_orig):
931              dtype_orig = np.result_type(*dtypes_orig)
932          elif pandas_requires_conversion and any(d == object for d in dtypes_orig):
933              # Force object if any of the dtypes is an object
934              dtype_orig = object
935  
936      elif (_is_extension_array_dtype(array) or hasattr(array, "iloc")) and hasattr(
937          array, "dtype"
938      ):
939          # array is a pandas series
940          type_if_series = type(array)
941          pandas_requires_conversion = _pandas_dtype_needs_early_conversion(array.dtype)
942          if isinstance(array.dtype, np.dtype):
943              dtype_orig = array.dtype
944          else:
945              # Set to None to let array.astype work out the best dtype
946              dtype_orig = None
947  
948      if dtype_numeric:
949          if (
950              dtype_orig is not None
951              and hasattr(dtype_orig, "kind")
952              and dtype_orig.kind == "O"
953          ):
954              # if input is object, convert to float.
955              dtype = xp.float64
956          else:
957              dtype = None
958  
959      if isinstance(dtype, (list, tuple)):
960          if dtype_orig is not None and dtype_orig in dtype:
961              # no dtype conversion required
962              dtype = None
963          else:
964              # dtype conversion required. Let's select the first element of the
965              # list of accepted types.
966              dtype = dtype[0]
967  
968      if pandas_requires_conversion:
969          # pandas dataframe requires conversion earlier to handle extension dtypes with
970          # nans
971          # Use the original dtype for conversion if dtype is None
972          new_dtype = dtype_orig if dtype is None else dtype
973          array = array.astype(new_dtype)
974          # Since we converted here, we do not need to convert again later
975          dtype = None
976  
977      if ensure_all_finite not in (True, False, "allow-nan"):
978          raise ValueError(
979              "ensure_all_finite should be a bool or 'allow-nan'. Got "
980              f"{ensure_all_finite!r} instead."
981          )
982  
983      if dtype is not None and _is_numpy_namespace(xp):
984          # convert to dtype object to conform to Array API to be use `xp.isdtype` later
985          dtype = np.dtype(dtype)
986  
987      estimator_name = _check_estimator_name(estimator)
988      context = " by %s" % estimator_name if estimator is not None else ""
989  
990      # When all dataframe columns are sparse, convert to a sparse array
991      if hasattr(array, "sparse") and array.ndim > 1:
992          with suppress(ImportError):
993              from pandas import SparseDtype  # noqa: F811
994  
995              def is_sparse(dtype):
996                  return isinstance(dtype, SparseDtype)
997  
998              if array.dtypes.apply(is_sparse).all():
999                  # DataFrame.sparse only supports `to_coo`
1000                 array = array.sparse.to_coo()
1001                 if array.dtype == np.dtype("object"):
1002                     unique_dtypes = set([dt.subtype.name for dt in array_orig.dtypes])
1003                     if len(unique_dtypes) > 1:
1004                         raise ValueError(
1005                             "Pandas DataFrame with mixed sparse extension arrays "
1006                             "generated a sparse matrix with object dtype which "
1007                             "can not be converted to a scipy sparse matrix."
1008                             "Sparse extension arrays should all have the same "
1009                             "numeric type."
1010                         )
1011 
1012     if sp.issparse(array):
1013         _ensure_no_complex_data(array)
1014         array = _ensure_sparse_format(
1015             array,
1016             accept_sparse=accept_sparse,
1017             dtype=dtype,
1018             copy=copy,
1019             ensure_all_finite=ensure_all_finite,
1020             accept_large_sparse=accept_large_sparse,
1021             estimator_name=estimator_name,
1022             input_name=input_name,
1023         )
1024         if ensure_2d and array.ndim < 2:
1025             raise ValueError(
1026                 f"Expected 2D input, got input with shape {array.shape}.\n"
1027                 "Reshape your data either using array.reshape(-1, 1) if "
1028                 "your data has a single feature or array.reshape(1, -1) "
1029                 "if it contains a single sample."
1030             )
1031     else:
1032         # If np.array(..) gives ComplexWarning, then we convert the warning
1033         # to an error. This is needed because specifying a non complex
1034         # dtype to the function converts complex to real dtype,
1035         # thereby passing the test made in the lines following the scope
1036         # of warnings context manager.
1037         with warnings.catch_warnings():
1038             try:
1039                 warnings.simplefilter("error", ComplexWarning)
1040                 if dtype is not None and xp.isdtype(dtype, "integral"):
1041                     # Conversion float -> int should not contain NaN or
1042                     # inf (numpy#14412). We cannot use casting='safe' because
1043                     # then conversion float -> int would be disallowed.
1044                     array = _asarray_with_order(array, order=order, xp=xp)
1045                     if xp.isdtype(array.dtype, ("real floating", "complex floating")):
1046                         _assert_all_finite(
1047                             array,
1048                             allow_nan=False,
1049                             msg_dtype=dtype,
1050                             estimator_name=estimator_name,
1051                             input_name=input_name,
1052                         )
1053                     array = xp.astype(array, dtype, copy=False)
1054                 else:
1055                     array = _asarray_with_order(array, order=order, dtype=dtype, xp=xp)
1056             except ComplexWarning as complex_warning:
1057                 raise ValueError(
1058                     "Complex data not supported\n{}\n".format(array)
1059                 ) from complex_warning
1060 
1061         # It is possible that the np.array(..) gave no warning. This happens
1062         # when no dtype conversion happened, for example dtype = None. The
1063         # result is that np.array(..) produces an array of complex dtype
1064         # and we need to catch and raise exception for such cases.
1065         _ensure_no_complex_data(array)
1066 
1067         if ensure_2d:
1068             # If input is scalar raise error
1069             if array.ndim == 0:
1070                 raise ValueError(
1071                     "Expected 2D array, got scalar array instead:\narray={}.\n"
1072                     "Reshape your data either using array.reshape(-1, 1) if "
1073                     "your data has a single feature or array.reshape(1, -1) "
1074                     "if it contains a single sample.".format(array)
1075                 )
1076             # If input is 1D raise error
1077             if array.ndim == 1:
1078                 # If input is a Series-like object (eg. pandas Series or polars Series)
1079                 if type_if_series is not None:
1080                     msg = (
1081                         f"Expected a 2-dimensional container but got {type_if_series} "
1082                         "instead. Pass a DataFrame containing a single row (i.e. "
1083                         "single sample) or a single column (i.e. single feature) "
1084                         "instead."
1085                     )
1086                 else:
1087                     msg = (
1088                         f"Expected 2D array, got 1D array instead:\narray={array}.\n"
1089                         "Reshape your data either using array.reshape(-1, 1) if "
1090                         "your data has a single feature or array.reshape(1, -1) "
1091                         "if it contains a single sample."
1092                     )
1093                 raise ValueError(msg)
1094 
1095         if dtype_numeric and hasattr(array.dtype, "kind") and array.dtype.kind in "USV":
1096             raise ValueError(
1097                 "dtype='numeric' is not compatible with arrays of bytes/strings."
1098                 "Convert your data to numeric values explicitly instead."
1099             )
1100         if not allow_nd and array.ndim >= 3:
1101             raise ValueError(
1102                 "Found array with dim %d. %s expected <= 2."
1103                 % (array.ndim, estimator_name)
1104             )
1105 
1106         if ensure_all_finite:
1107             _assert_all_finite(
1108                 array,
1109                 input_name=input_name,
1110                 estimator_name=estimator_name,
1111                 allow_nan=ensure_all_finite == "allow-nan",
1112             )
1113 
1114         if copy:
1115             if _is_numpy_namespace(xp):
1116                 # only make a copy if `array` and `array_orig` may share memory`
1117                 if np.may_share_memory(array, array_orig):
1118                     array = _asarray_with_order(
1119                         array, dtype=dtype, order=order, copy=True, xp=xp
1120                     )
1121             else:
1122                 # always make a copy for non-numpy arrays
1123                 array = _asarray_with_order(
1124                     array, dtype=dtype, order=order, copy=True, xp=xp
1125                 )
1126 
1127     if ensure_min_samples > 0:
1128         n_samples = _num_samples(array)
1129         if n_samples < ensure_min_samples:
1130             raise ValueError(
1131                 "Found array with %d sample(s) (shape=%s) while a"
1132                 " minimum of %d is required%s."
1133                 % (n_samples, array.shape, ensure_min_samples, context)
1134             )
1135 
1136     if ensure_min_features > 0 and array.ndim == 2:
1137         n_features = array.shape[1]
1138         if n_features < ensure_min_features:
1139             raise ValueError(
1140                 "Found array with %d feature(s) (shape=%s) while"
1141                 " a minimum of %d is required%s."
1142                 % (n_features, array.shape, ensure_min_features, context)
1143             )
1144 
1145     if ensure_non_negative:
1146         whom = input_name
1147         if estimator_name:
1148             whom += f" in {estimator_name}"
1149         check_non_negative(array, whom)
1150 
1151     if force_writeable:
1152         # By default, array.copy() creates a C-ordered copy. We set order=K to
1153         # preserve the order of the array.
1154         copy_params = {"order": "K"} if not sp.issparse(array) else {}
1155 
1156         array_data = array.data if sp.issparse(array) else array
1157         flags = getattr(array_data, "flags", None)
1158         if not getattr(flags, "writeable", True):
1159             # This situation can only happen when copy=False, the array is read-only and
1160             # a writeable output is requested. This is an ambiguous setting so we chose
1161             # to always (except for one specific setting, see below) make a copy to
1162             # ensure that the output is writeable, even if avoidable, to not overwrite
1163             # the user's data by surprise.
1164 
1165             if _is_pandas_df_or_series(array_orig):
1166                 try:
1167                     # In pandas >= 3, np.asarray(df), called earlier in check_array,
1168                     # returns a read-only intermediate array. It can be made writeable
1169                     # safely without copy because if the original DataFrame was backed
1170                     # by a read-only array, trying to change the flag would raise an
1171                     # error, in which case we make a copy.
1172                     array_data.flags.writeable = True
1173                 except ValueError:
1174                     array = array.copy(**copy_params)
1175             else:
1176                 array = array.copy(**copy_params)
1177 
1178     return array

1195                     "Only sparse matrices with 32-bit integer indices are accepted."
1196                     f" Got {indices_datatype} indices. Please do report a minimal"
1197                     " reproducer on scikit-learn issue tracker so that support for"
1198                     " your use-case can be studied by maintainers. See:"
1199                     " https://scikit-learn.org/dev/developers/minimal_reproducer.html"
1200                 )

1557 def check_symmetric(array, *, tol=1e-10, raise_warning=True, raise_exception=False):
1558     """Make sure that array is 2D, square and symmetric. 
1559  
1560     If the array is not symmetric, then a symmetrized version is returned. 
1561     Optionally, a warning or exception is raised if the matrix is not 
1562     symmetric. 
1563  
1564     Parameters 
1565     ---------- 
1566     array : {ndarray, sparse matrix} 
1567         Input object to check / convert. Must be two-dimensional and square, 
1568         otherwise a ValueError will be raised. 
1569  
1570     tol : float, default=1e-10 
1571         Absolute tolerance for equivalence of arrays. Default = 1E-10. 
1572  
1573     raise_warning : bool, default=True 
1574         If True then raise a warning if conversion is required. 
1575  
1576     raise_exception : bool, default=False 
1577         If True then raise an exception if array is not symmetric. 
1578  
1579     Returns 
1580     ------- 
1581     array_sym : {ndarray, sparse matrix} 
1582         Symmetrized version of the input array, i.e. the average of array 
1583         and array.transpose(). If sparse, then duplicate entries are first 
1584         summed and zeros are eliminated. 
1585  
1586     Examples 
1587     -------- 
1588     >>> import numpy as np 
1589     >>> from sklearn.utils.validation import check_symmetric 
1590     >>> symmetric_array = np.array([[0, 1, 2], [1, 0, 1], [2, 1, 0]]) 
1591     >>> check_symmetric(symmetric_array) 
1592     array([[0, 1, 2], 
1593            [1, 0, 1], 
1594            [2, 1, 0]]) 
1595     >>> from scipy.sparse import csr_matrix 
1596     >>> sparse_symmetric_array = csr_matrix(symmetric_array) 
1597     >>> check_symmetric(sparse_symmetric_array) 
1598     <Compressed Sparse Row sparse matrix of dtype 'int64' 
1599         with 6 stored elements and shape (3, 3)> 
1600     """
1601     if (array.ndim != 2) or (array.shape[0] != array.shape[1]):
1602         raise ValueError(
1603             "array must be 2-dimensional and square. shape = {0}".format(array.shape)
1604         )
1605 
1606     if sp.issparse(array):
1607         diff = array - array.T
1608         # only csr, csc, and coo have `data` attribute
1609         if diff.format not in ["csr", "csc", "coo"]:
1610             diff = diff.tocsr()
1611         symmetric = np.all(abs(diff.data) < tol)
1612     else:
1613         symmetric = np.allclose(array, array.T, atol=tol)
1614 
1615     if not symmetric:
1616         if raise_exception:
1617             raise ValueError("Array must be symmetric")
1618         if raise_warning:
1619             warnings.warn(
1620                 (
1621                     "Array is not symmetric, and will be converted "
1622                     "to symmetric by average with its transpose."
1623                 ),
1624                 stacklevel=2,
1625             )
1626         if sp.issparse(array):
1627             conversion = "to" + array.format
1628             array = getattr(0.5 * (array + array.T), conversion)()
1629         else:
1630             array = 0.5 * (array + array.T)
1631 
1632     return array

1673 def check_is_fitted(estimator, attributes=None, *, msg=None, all_or_any=all):
1674     """Perform is_fitted validation for estimator. 
1675  
1676     Checks if the estimator is fitted by verifying the presence of 
1677     fitted attributes (ending with a trailing underscore) and otherwise 
1678     raises a :class:`~sklearn.exceptions.NotFittedError` with the given message. 
1679

1681     can define a ``__sklearn_is_fitted__`` method returning a boolean to 
1682     specify if the estimator is fitted or not. See 
1683     :ref:`sphx_glr_auto_examples_developing_estimators_sklearn_is_fitted.py`

1691     Parameters 
1692     ---------- 
1693     estimator : estimator instance 
1694         Estimator instance for which the check is performed.

1698         Eg.: ``["coef_", "estimator_", ...], "coef_"`` 
1699

1701         attribute that ends with a underscore and does not start with double 
1702         underscore. 
1703  
1704     msg : str, default=None 
1705         The default error message is, "This %(name)s instance is not fitted 
1706         yet. Call 'fit' with appropriate arguments before using this 
1707         estimator." 
1708  
1709         For custom messages if "%(name)s" is present in the message string, 
1710         it is substituted for the estimator name. 
1711  
1712         Eg. : "Estimator, %(name)s, must be fitted before sparsifying". 
1713  
1714     all_or_any : callable, {all, any}, default=all 
1715         Specify whether all or any of the given attributes must exist. 
1716  
1717     Raises 
1718     ------ 
1719     TypeError 
1720         If the estimator is a class or not an estimator instance 
1721  
1722     NotFittedError 
1723         If the attributes are not found. 
1724  
1725     Examples 
1726     -------- 
1727     >>> from sklearn.linear_model import LogisticRegression 
1728     >>> from sklearn.utils.validation import check_is_fitted 
1729     >>> from sklearn.exceptions import NotFittedError 
1730     >>> lr = LogisticRegression()

1732     ...     check_is_fitted(lr) 
1733     ... except NotFittedError as exc:

1760 def _estimator_has(attr, *, delegates=("estimator_", "estimator")):
1761     """Check if we can delegate a method to the underlying estimator. 
1762  
1763     We check the `delegates` in the order they are passed. By default, we first check 
1764     the fitted estimator if available, otherwise we check the unfitted estimator. 
1765  
1766     Parameters 
1767     ---------- 
1768     attr : str 
1769         Name of the attribute the delegate might or might not have. 
1770  
1771     delegates: tuple of str, default=("estimator_", "estimator") 
1772         A tuple of sub-estimator(s) to check if we can delegate the `attr` method. 
1773  
1774     Returns 
1775     ------- 
1776     check : function 
1777         Function to check if the delegate has the attribute. 
1778  
1779     Raises 
1780     ------ 
1781     ValueError 
1782         Raised when none of the delegates are present in the object. 
1783     """
1784 
1785     def check(self):
1786         for delegate in delegates:
1787             # In meta estimators with multiple sub estimators,
1788             # only the attribute of the first sub estimator is checked,
1789             # assuming uniformity across all sub estimators.
1790             if hasattr(self, delegate):
1791                 delegator = getattr(self, delegate)
1792                 if isinstance(delegator, Sequence):
1793                     return getattr(delegator[0], attr)
1794                 else:
1795                     return getattr(delegator, attr)
1796 
1797         raise ValueError(f"None of the delegates {delegates} are present in the class.")
1798 
1799     return check

1840

1857         The maximum valid value the parameter can take. If None (default) it

1867           It is equivalent to the interval `( min_val, max_val ]`. 
1868         - `"both"`: `min_val` and `max_val` are included in the valid interval. 
1869           It is equivalent to the interval `[ min_val, max_val ]`. 
1870         - `"neither"`: neither `min_val` nor `max_val` are included in the 
1871           valid interval. It is equivalent to the interval `( min_val, max_val )`.

1873     Returns 
1874     ------- 
1875     x : numbers.Number 
1876         The validated number. 
1877  
1878     Raises 
1879     ------

1896         module = t.__module__
1897         qualname = t.__qualname__
1898         if module == "builtins":
1899             return qualname
1900         elif t == numbers.Real:

1958 def _check_psd_eigenvalues(lambdas, enable_warnings=False):
1959     """Check the eigenvalues of a positive semidefinite (PSD) matrix. 
1960  
1961     Checks the provided array of PSD matrix eigenvalues for numerical or 
1962     conditioning issues and returns a fixed validated version. This method 
1963     should typically be used if the PSD matrix is user-provided (e.g. a 
1964     Gram matrix) or computed using a user-provided dissimilarity metric 
1965     (e.g. kernel function), or if the decomposition process uses approximation 
1966     methods (randomized SVD, etc.). 
1967  
1968     It checks for three things: 
1969  
1970     - that there are no significant imaginary parts in eigenvalues (more than 
1971       1e-5 times the maximum real part). If this check fails, it raises a 
1972       ``ValueError``. Otherwise all non-significant imaginary parts that may 
1973       remain are set to zero. This operation is traced with a 
1974       ``PositiveSpectrumWarning`` when ``enable_warnings=True``. 
1975  
1976     - that eigenvalues are not all negative. If this check fails, it raises a 
1977       ``ValueError`` 
1978  
1979     - that there are no significant negative eigenvalues with absolute value 
1980       more than 1e-10 (1e-6) and more than 1e-5 (5e-3) times the largest 
1981       positive eigenvalue in double (simple) precision. If this check fails, 
1982       it raises a ``ValueError``. Otherwise all negative eigenvalues that may 
1983       remain are set to zero. This operation is traced with a 
1984       ``PositiveSpectrumWarning`` when ``enable_warnings=True``. 
1985  
1986     Finally, all the positive eigenvalues that are too small (with a value 
1987     smaller than the maximum eigenvalue multiplied by 1e-12 (2e-7)) are set to 
1988     zero. This operation is traced with a ``PositiveSpectrumWarning`` when 
1989     ``enable_warnings=True``. 
1990  
1991     Parameters 
1992     ---------- 
1993     lambdas : array-like of shape (n_eigenvalues,) 
1994         Array of eigenvalues to check / fix. 
1995  
1996     enable_warnings : bool, default=False 
1997         When this is set to ``True``, a ``PositiveSpectrumWarning`` will be 
1998         raised when there are imaginary parts, negative eigenvalues, or 
1999         extremely small non-zero eigenvalues. Otherwise no warning will be 
2000         raised. In both cases, imaginary parts, negative eigenvalues, and 
2001         extremely small non-zero eigenvalues will be set to zero. 
2002  
2003     Returns 
2004     ------- 
2005     lambdas_fixed : ndarray of shape (n_eigenvalues,) 
2006         A fixed validated copy of the array of eigenvalues. 
2007  
2008     Examples 
2009     -------- 
2010     >>> from sklearn.utils.validation import _check_psd_eigenvalues 
2011     >>> _check_psd_eigenvalues([1, 2])      # nominal case 
2012     array([1, 2]) 
2013     >>> _check_psd_eigenvalues([5, 5j])     # significant imag part 
2014     Traceback (most recent call last): 
2015         ... 
2016     ValueError: There are significant imaginary parts in eigenvalues (1 
2017         of the maximum real part). Either the matrix is not PSD, or there was 
2018         an issue while computing the eigendecomposition of the matrix. 
2019     >>> _check_psd_eigenvalues([5, 5e-5j])  # insignificant imag part 
2020     array([5., 0.]) 
2021     >>> _check_psd_eigenvalues([-5, -1])    # all negative 
2022     Traceback (most recent call last): 
2023         ... 
2024     ValueError: All eigenvalues are negative (maximum is -1). Either the 
2025         matrix is not PSD, or there was an issue while computing the 
2026         eigendecomposition of the matrix. 
2027     >>> _check_psd_eigenvalues([5, -1])     # significant negative 
2028     Traceback (most recent call last): 
2029         ... 
2030     ValueError: There are significant negative eigenvalues (0.2 of the 
2031         maximum positive). Either the matrix is not PSD, or there was an issue 
2032         while computing the eigendecomposition of the matrix. 
2033     >>> _check_psd_eigenvalues([5, -5e-5])  # insignificant negative 
2034     array([5., 0.]) 
2035     >>> _check_psd_eigenvalues([5, 4e-12])  # bad conditioning (too small) 
2036     array([5., 0.]) 
2037  
2038     """
2039 
2040     lambdas = np.array(lambdas)
2041     is_double_precision = lambdas.dtype == np.float64
2042 
2043     # note: the minimum value available is
2044     #  - single-precision: np.finfo('float32').eps = 1.2e-07
2045     #  - double-precision: np.finfo('float64').eps = 2.2e-16
2046 
2047     # the various thresholds used for validation
2048     # we may wish to change the value according to precision.
2049     significant_imag_ratio = 1e-5
2050     significant_neg_ratio = 1e-5 if is_double_precision else 5e-3
2051     significant_neg_value = 1e-10 if is_double_precision else 1e-6
2052     small_pos_ratio = 1e-12 if is_double_precision else 2e-7
2053 
2054     # Check that there are no significant imaginary parts
2055     if not np.isreal(lambdas).all():
2056         max_imag_abs = np.abs(np.imag(lambdas)).max()
2057         max_real_abs = np.abs(np.real(lambdas)).max()
2058         if max_imag_abs > significant_imag_ratio * max_real_abs:
2059             raise ValueError(
2060                 "There are significant imaginary parts in eigenvalues (%g "
2061                 "of the maximum real part). Either the matrix is not PSD, or "
2062                 "there was an issue while computing the eigendecomposition "
2063                 "of the matrix." % (max_imag_abs / max_real_abs)
2064             )
2065 
2066         # warn about imaginary parts being removed
2067         if enable_warnings:
2068             warnings.warn(
2069                 "There are imaginary parts in eigenvalues (%g "
2070                 "of the maximum real part). Either the matrix is not"
2071                 " PSD, or there was an issue while computing the "
2072                 "eigendecomposition of the matrix. Only the real "
2073                 "parts will be kept." % (max_imag_abs / max_real_abs),
2074                 PositiveSpectrumWarning,
2075             )
2076 
2077     # Remove all imaginary parts (even if zero)
2078     lambdas = np.real(lambdas)
2079 
2080     # Check that there are no significant negative eigenvalues
2081     max_eig = lambdas.max()
2082     if max_eig < 0:
2083         raise ValueError(
2084             "All eigenvalues are negative (maximum is %g). "
2085             "Either the matrix is not PSD, or there was an "
2086             "issue while computing the eigendecomposition of "
2087             "the matrix." % max_eig
2088         )
2089 
2090     else:
2091         min_eig = lambdas.min()
2092         if (
2093             min_eig < -significant_neg_ratio * max_eig
2094             and min_eig < -significant_neg_value
2095         ):
2096             raise ValueError(
2097                 "There are significant negative eigenvalues (%g"
2098                 " of the maximum positive). Either the matrix is "
2099                 "not PSD, or there was an issue while computing "
2100                 "the eigendecomposition of the matrix." % (-min_eig / max_eig)
2101             )
2102         elif min_eig < 0:
2103             # Remove all negative values and warn about it
2104             if enable_warnings:
2105                 warnings.warn(
2106                     "There are negative eigenvalues (%g of the "
2107                     "maximum positive). Either the matrix is not "
2108                     "PSD, or there was an issue while computing the"
2109                     " eigendecomposition of the matrix. Negative "
2110                     "eigenvalues will be replaced with 0." % (-min_eig / max_eig),
2111                     PositiveSpectrumWarning,
2112                 )
2113             lambdas[lambdas < 0] = 0
2114 
2115     # Check for conditioning (small positive non-zeros)
2116     too_small_lambdas = (0 < lambdas) & (lambdas < small_pos_ratio * max_eig)
2117     if too_small_lambdas.any():
2118         if enable_warnings:
2119             warnings.warn(
2120                 "Badly conditioned PSD matrix spectrum: the largest "
2121                 "eigenvalue is more than %g times the smallest. "
2122                 "Small eigenvalues will be replaced with 0."
2123                 "" % (1 / small_pos_ratio),
2124                 PositiveSpectrumWarning,
2125             )
2126         lambdas[too_small_lambdas] = 0
2127 
2128     return lambdas

2206 def _allclose_dense_sparse(x, y, rtol=1e-7, atol=1e-9):
2207     """Check allclose for sparse and dense data. 
2208  
2209     Both x and y need to be either sparse or dense, they 
2210     can't be mixed. 
2211  
2212     Parameters 
2213     ---------- 
2214     x : {array-like, sparse matrix} 
2215         First array to compare. 
2216  
2217     y : {array-like, sparse matrix} 
2218         Second array to compare. 
2219  
2220     rtol : float, default=1e-7 
2221         Relative tolerance; see numpy.allclose. 
2222  
2223     atol : float, default=1e-9

2225         more tolerant than the default for numpy.testing.assert_allclose, where 
2226         atol=0.

2230         y = y.tocsr()

2234             np.array_equal(x.indices, y.indices)
2235             and np.array_equal(x.indptr, y.indptr)
2236             and np.allclose(x.data, y.data, rtol=rtol, atol=atol)
2237         )
2238     elif not sp.issparse(x) and not sp.issparse(y):
2239         return np.allclose(x, y, rtol=rtol, atol=atol)

2241         "Can only compare two sparse matrices, not a sparse matrix and an array"

2370 def _get_feature_names(X):

2538             corresponding feature is required to be monotonically decreasing. 
2539         - If dict, then it the keys should be the feature names occurring in 
2540             `estimator.feature_names_in_` and the values should be -1, 0 or 1. 
2541         - If None, then an array of 0s will be allocated. 
2542  
2543     Returns 
2544     ------- 
2545     monotonic_cst : ndarray of int 
2546         Monotonic constraints for each feature. 
2547     """
2548     original_monotonic_cst = monotonic_cst
2549     if monotonic_cst is None or isinstance(monotonic_cst, dict):
2550         monotonic_cst = np.full(
2551             shape=estimator.n_features_in_,
2552             fill_value=0,
2553             dtype=np.int8,
2554         )
2555         if isinstance(original_monotonic_cst, dict):
2556             if not hasattr(estimator, "feature_names_in_"):
2557                 raise ValueError(
2558                     f"{estimator.__class__.__name__} was not fitted on data "
2559                     "with feature names. Pass monotonic_cst as an integer "
2560                     "array instead."
2561                 )
2562             unexpected_feature_names = list(
2563                 set(original_monotonic_cst) - set(estimator.feature_names_in_)
2564             )
2565             unexpected_feature_names.sort()  # deterministic error message
2566             n_unexpeced = len(unexpected_feature_names)
2567             if unexpected_feature_names:
2568                 if len(unexpected_feature_names) > 5:
2569                     unexpected_feature_names = unexpected_feature_names[:5]
2570                     unexpected_feature_names.append("...")
2571                 raise ValueError(
2572                     f"monotonic_cst contains {n_unexpeced} unexpected feature "
2573                     f"names: {unexpected_feature_names}."
2574                 )
2575             for feature_idx, feature_name in enumerate(estimator.feature_names_in_):
2576                 if feature_name in original_monotonic_cst:
2577                     cst = original_monotonic_cst[feature_name]
2578                     if cst not in [-1, 0, 1]:
2579                         raise ValueError(
2580                             f"monotonic_cst['{feature_name}'] must be either "
2581                             f"-1, 0 or 1. Got {cst!r}."
2582                         )
2583                     monotonic_cst[feature_idx] = cst
2584     else:
2585         unexpected_cst = np.setdiff1d(monotonic_cst, [-1, 0, 1])
2586         if unexpected_cst.shape[0]:
2587             raise ValueError(
2588                 "monotonic_cst must be an array-like of -1, 0 or 1. Observed "
2589                 f"values: {unexpected_cst.tolist()}."
2590             )
2591 
2592         monotonic_cst = np.asarray(monotonic_cst, dtype=np.int8)
2593         if monotonic_cst.shape[0] != estimator.n_features_in_:
2594             raise ValueError(
2595                 f"monotonic_cst has shape {monotonic_cst.shape} but the input data "
2596                 f"X has {estimator.n_features_in_} features."
2597             )
2598     return monotonic_cst

2601 def _check_pos_label_consistency(pos_label, y_true):
2602     """Check if `pos_label` need to be specified or not. 
2603  
2604     In binary classification, we fix `pos_label=1` if the labels are in the set 
2605     {-1, 1} or {0, 1}. Otherwise, we raise an error asking to specify the 
2606     `pos_label` parameters. 
2607  
2608     Parameters 
2609     ---------- 
2610     pos_label : int, float, bool, str or None 
2611         The positive label. 
2612     y_true : ndarray of shape (n_samples,) 
2613         The target vector. 
2614  
2615     Returns 
2616     ------- 
2617     pos_label : int, float, bool or str 
2618         If `pos_label` can be inferred, it will be returned. 
2619  
2620     Raises 
2621     ------ 
2622     ValueError 
2623         In the case that `y_true` does not have label in {-1, 1} or {0, 1}, 
2624         it will raise a `ValueError`. 
2625     """
2626     # ensure binary classification if pos_label is not specified
2627     # classes.dtype.kind in ('O', 'U', 'S') is required to avoid
2628     # triggering a FutureWarning by calling np.array_equal(a, b)
2629     # when elements in the two arrays are not comparable.
2630     if pos_label is None:
2631         # Compute classes only if pos_label is not specified:
2632         classes = np.unique(y_true)
2633         if classes.dtype.kind in "OUS" or not (
2634             np.array_equal(classes, [0, 1])
2635             or np.array_equal(classes, [-1, 1])
2636             or np.array_equal(classes, [0])
2637             or np.array_equal(classes, [-1])
2638             or np.array_equal(classes, [1])
2639         ):
2640             classes_repr = ", ".join([repr(c) for c in classes.tolist()])
2641             raise ValueError(
2642                 f"y_true takes value in {{{classes_repr}}} and pos_label is not "
2643                 "specified: either make y_true take value in {0, 1} or "
2644                 "{-1, 1} or pass pos_label explicitly."
2645             )
2646         pos_label = 1
2647 
2648     return pos_label

2651 def _to_object_array(sequence):
2652     """Convert sequence to a 1-D NumPy array of object dtype. 
2653  
2654     numpy.array constructor has a similar use but it's output 
2655     is ambiguous. It can be 1-D NumPy array of object dtype if 
2656     the input is a ragged array, but if the input is a list of 
2657     equal length arrays, then the output is a 2D numpy.array. 
2658     _to_object_array solves this ambiguity by guarantying that 
2659     the output is a 1-D NumPy array of objects for any input. 
2660  
2661     Parameters 
2662     ---------- 
2663     sequence : array-like of shape (n_elements,) 
2664         The sequence to be converted. 
2665  
2666     Returns 
2667     ------- 
2668     out : ndarray of shape (n_elements,), dtype=object 
2669         The converted sequence into a 1-D NumPy array of object dtype. 
2670  
2671     Examples 
2672     -------- 
2673     >>> import numpy as np 
2674     >>> from sklearn.utils.validation import _to_object_array 
2675     >>> _to_object_array([np.array([0]), np.array([1])]) 
2676     array([array([0]), array([1])], dtype=object) 
2677     >>> _to_object_array([np.array([0]), np.array([1, 2])]) 
2678     array([array([0]), array([1, 2])], dtype=object) 
2679     >>> _to_object_array([np.array([0]), np.array([1, 2])]) 
2680     array([array([0]), array([1, 2])], dtype=object) 
2681     """
2682     out = np.empty(len(sequence), dtype=object)
2683     out[:] = sequence
2684     return out

2687 def _check_feature_names(estimator, X, *, reset):
2688     """Set or check the `feature_names_in_` attribute of an estimator. 
2689  
2690     .. versionadded:: 1.0 
2691  
2692     .. versionchanged:: 1.6 
2693         Moved from :class:`~sklearn.base.BaseEstimator` to 
2694         :mod:`sklearn.utils.validation`. 
2695  
2696     Parameters 
2697     ---------- 
2698     estimator : estimator instance 
2699         The estimator to validate the input for. 
2700  
2701     X : {ndarray, dataframe} of shape (n_samples, n_features) 
2702         The input samples. 
2703  
2704     reset : bool 
2705         Whether to reset the `feature_names_in_` attribute. 
2706         If False, the input will be checked for consistency with 
2707         feature names of data provided when reset was last True. 
2708         .. note:: 
2709            It is recommended to call `reset=True` in `fit` and in the first 
2710            call to `partial_fit`. All other methods that validate `X` 
2711            should set `reset=False`. 
2712     """
2713 
2714     if reset:
2715         feature_names_in = _get_feature_names(X)
2716         if feature_names_in is not None:
2717             estimator.feature_names_in_ = feature_names_in
2718         elif hasattr(estimator, "feature_names_in_"):
2719             # Delete the attribute when the estimator is fitted on a new dataset
2720             # that has no feature names.
2721             delattr(estimator, "feature_names_in_")
2722         return
2723 
2724     fitted_feature_names = getattr(estimator, "feature_names_in_", None)
2725     X_feature_names = _get_feature_names(X)
2726 
2727     if fitted_feature_names is None and X_feature_names is None:
2728         # no feature names seen in fit and in X
2729         return
2730 
2731     if X_feature_names is not None and fitted_feature_names is None:
2732         warnings.warn(
2733             f"X has feature names, but {estimator.__class__.__name__} was fitted "
2734             "without feature names"
2735         )
2736         return
2737 
2738     if X_feature_names is None and fitted_feature_names is not None:
2739         warnings.warn(
2740             "X does not have valid feature names, but"
2741             f" {estimator.__class__.__name__} was fitted with feature names"
2742         )
2743         return
2744 
2745     # validate the feature names against the `feature_names_in_` attribute
2746     if len(fitted_feature_names) != len(X_feature_names) or np.any(
2747         fitted_feature_names != X_feature_names
2748     ):
2749         message = "The feature names should match those that were passed during fit.\n"
2750         fitted_feature_names_set = set(fitted_feature_names)
2751         X_feature_names_set = set(X_feature_names)
2752 
2753         unexpected_names = sorted(X_feature_names_set - fitted_feature_names_set)
2754         missing_names = sorted(fitted_feature_names_set - X_feature_names_set)
2755 
2756         def add_names(names):
2757             output = ""
2758             max_n_names = 5
2759             for i, name in enumerate(names):
2760                 if i >= max_n_names:
2761                     output += "- ...\n"
2762                     break
2763                 output += f"- {name}\n"
2764             return output
2765 
2766         if unexpected_names:
2767             message += "Feature names unseen at fit time:\n"
2768             message += add_names(unexpected_names)
2769 
2770         if missing_names:
2771             message += "Feature names seen at fit time, yet now missing:\n"
2772             message += add_names(missing_names)
2773 
2774         if not missing_names and not unexpected_names:
2775             message += "Feature names must be in the same order as they were in fit.\n"
2776 
2777         raise ValueError(message)

2780 def _check_n_features(estimator, X, reset):
2781     """Set the `n_features_in_` attribute, or check against it on an estimator. 
2782  
2783     .. versionchanged:: 1.6 
2784         Moved from :class:`~sklearn.base.BaseEstimator` to 
2785         :mod:`~sklearn.utils.validation`. 
2786  
2787     Parameters 
2788     ---------- 
2789     estimator : estimator instance 
2790         The estimator to validate the input for. 
2791  
2792     X : {ndarray, sparse matrix} of shape (n_samples, n_features) 
2793         The input samples. 
2794  
2795     reset : bool 
2796         If True, the `n_features_in_` attribute is set to `X.shape[1]`. 
2797         If False and the attribute exists, then check that it is equal to 
2798         `X.shape[1]`. If False and the attribute does *not* exist, then 
2799         the check is skipped. 
2800         .. note:: 
2801            It is recommended to call reset=True in `fit` and in the first 
2802            call to `partial_fit`. All other methods that validate `X` 
2803            should set `reset=False`. 
2804     """
2805     try:
2806         n_features = _num_features(X)
2807     except TypeError as e:
2808         if not reset and hasattr(estimator, "n_features_in_"):
2809             raise ValueError(
2810                 "X does not contain any features, but "
2811                 f"{estimator.__class__.__name__} is expecting "
2812                 f"{estimator.n_features_in_} features"
2813             ) from e
2814         # If the number of features is not defined and reset=True,
2815         # then we skip this check
2816         return
2817 
2818     if reset:
2819         estimator.n_features_in_ = n_features
2820         return
2821 
2822     if not hasattr(estimator, "n_features_in_"):
2823         # Skip this check if the expected number of expected input features
2824         # was not recorded by calling fit first. This is typically the case
2825         # for stateless transformers.
2826         return
2827 
2828     if n_features != estimator.n_features_in_:
2829         raise ValueError(
2830             f"X has {n_features} features, but {estimator.__class__.__name__} "
2831             f"is expecting {estimator.n_features_in_} features as input."
2832         )

2835 def validate_data(
2836     _estimator,
2837     /,
2838     X="no_validation",
2839     y="no_validation",
2840     reset=True,
2841     validate_separately=False,
2842     skip_check_array=False,
2843     **check_params,
2844 ):
2845     """Validate input data and set or check feature names and counts of the input. 
2846  
2847     This helper function should be used in an estimator that requires input 
2848     validation. This mutates the estimator and sets the `n_features_in_` and 
2849     `feature_names_in_` attributes if `reset=True`. 
2850  
2851     .. versionadded:: 1.6 
2852  
2853     Parameters 
2854     ---------- 
2855     _estimator : estimator instance 
2856         The estimator to validate the input for. 
2857  
2858     X : {array-like, sparse matrix, dataframe} of shape \ 
2859             (n_samples, n_features), default='no validation' 
2860         The input samples. 
2861         If `'no_validation'`, no validation is performed on `X`. This is 
2862         useful for meta-estimator which can delegate input validation to 
2863         their underlying estimator(s). In that case `y` must be passed and 
2864         the only accepted `check_params` are `multi_output` and 
2865         `y_numeric`. 
2866  
2867     y : array-like of shape (n_samples,), default='no_validation' 
2868         The targets. 
2869  
2870         - If `None`, :func:`~sklearn.utils.check_array` is called on `X`. If 
2871           the estimator's `requires_y` tag is True, then an error will be raised. 
2872         - If `'no_validation'`, :func:`~sklearn.utils.check_array` is called 
2873           on `X` and the estimator's `requires_y` tag is ignored. This is a default 
2874           placeholder and is never meant to be explicitly set. In that case `X` must be 
2875           passed. 
2876         - Otherwise, only `y` with `_check_y` or both `X` and `y` are checked with 
2877           either :func:`~sklearn.utils.check_array` or 
2878           :func:`~sklearn.utils.check_X_y` depending on `validate_separately`. 
2879  
2880     reset : bool, default=True 
2881         Whether to reset the `n_features_in_` attribute. 
2882         If False, the input will be checked for consistency with data 
2883         provided when reset was last True. 
2884  
2885         .. note:: 
2886  
2887            It is recommended to call `reset=True` in `fit` and in the first 
2888            call to `partial_fit`. All other methods that validate `X` 
2889            should set `reset=False`. 
2890  
2891     validate_separately : False or tuple of dicts, default=False 
2892         Only used if `y` is not `None`. 
2893         If `False`, call :func:`~sklearn.utils.check_X_y`. Else, it must be a tuple of 
2894         kwargs to be used for calling :func:`~sklearn.utils.check_array` on `X` and `y` 
2895         respectively. 
2896  
2897         `estimator=self` is automatically added to these dicts to generate 
2898         more informative error message in case of invalid input data. 
2899  
2900     skip_check_array : bool, default=False 
2901         If `True`, `X` and `y` are unchanged and only `feature_names_in_` and 
2902         `n_features_in_` are checked. Otherwise, :func:`~sklearn.utils.check_array` 
2903         is called on `X` and `y`. 
2904  
2905     **check_params : kwargs 
2906         Parameters passed to :func:`~sklearn.utils.check_array` or 
2907         :func:`~sklearn.utils.check_X_y`. Ignored if validate_separately 
2908         is not False. 
2909  
2910         `estimator=self` is automatically added to these params to generate 
2911         more informative error message in case of invalid input data. 
2912  
2913     Returns 
2914     ------- 
2915     out : {ndarray, sparse matrix} or tuple of these 
2916         The validated input. A tuple is returned if both `X` and `y` are 
2917         validated. 
2918     """
2919     _check_feature_names(_estimator, X, reset=reset)
2920     tags = get_tags(_estimator)
2921     if y is None and tags.target_tags.required:
2922         raise ValueError(
2923             f"This {_estimator.__class__.__name__} estimator "
2924             "requires y to be passed, but the target y is None."
2925         )
2926 
2927     no_val_X = isinstance(X, str) and X == "no_validation"
2928     no_val_y = y is None or isinstance(y, str) and y == "no_validation"
2929 
2930     if no_val_X and no_val_y:
2931         raise ValueError("Validation should be done on X, y or both.")
2932 
2933     default_check_params = {"estimator": _estimator}
2934     check_params = {**default_check_params, **check_params}
2935 
2936     if skip_check_array:
2937         if not no_val_X and no_val_y:
2938             out = X
2939         elif no_val_X and not no_val_y:
2940             out = y
2941         else:
2942             out = X, y
2943     elif not no_val_X and no_val_y:
2944         out = check_array(X, input_name="X", **check_params)
2945     elif no_val_X and not no_val_y:
2946         out = _check_y(y, **check_params)
2947     else:
2948         if validate_separately:
2949             # We need this because some estimators validate X and y
2950             # separately, and in general, separately calling check_array()
2951             # on X and y isn't equivalent to just calling check_X_y()
2952             # :(
2953             check_X_params, check_y_params = validate_separately
2954             if "estimator" not in check_X_params:
2955                 check_X_params = {**default_check_params, **check_X_params}
2956             X = check_array(X, input_name="X", **check_X_params)
2957             if "estimator" not in check_y_params:
2958                 check_y_params = {**default_check_params, **check_y_params}
2959             y = check_array(y, input_name="y", **check_y_params)
2960         else:
2961             X, y = check_X_y(X, y, **check_params)
2962         out = X, y
2963 
2964     if not no_val_X and check_params.get("ensure_2d", True):
2965         _check_n_features(_estimator, X, reset=reset)
2966 
2967     return out
