1    # This demo uses the Internet Explorer Web Browser control.

6    import sys

8    import regutil
9    import win32api
10   import win32con
11   import win32ui
12   from pywin.mfc import activex, window
13   from win32com.client import gencache
14   
15   WebBrowserModule = gencache.EnsureModule(
16       "{EAB22AC0-30C1-11CF-A7EB-0000C05BAE0B}", 0, 1, 1
17   )
18   if WebBrowserModule is None:
19       raise ImportError("Internet Explorer does not appear to be installed.")

22   class MyWebBrowser(activex.Control, WebBrowserModule.WebBrowser):
23       def OnBeforeNavigate2(
24           self, pDisp, URL, Flags, TargetFrameName, PostData, Headers, Cancel
25       ):
26           self.GetParent().OnNavigate(URL)
27           # print("BeforeNavigate2", pDisp, URL, Flags, TargetFrameName, PostData, Headers, Cancel)

30   class BrowserFrame(window.MDIChildWnd):
31       def __init__(self, url=None):
32           if url is None:
33               self.url = regutil.GetRegisteredHelpFile("Main Python Documentation")
34               if self.url is None:
35                   self.url = "https://www.python.org"
36           else:
37               self.url = url
38           pass  # Don't call base class doc/view version...
39   
40       def Create(self, title, rect=None, parent=None):
41           style = win32con.WS_CHILD | win32con.WS_VISIBLE | win32con.WS_OVERLAPPEDWINDOW
42           self._obj_ = win32ui.CreateMDIChild()
43           self._obj_.AttachObject(self)
44           self._obj_.CreateWindow(None, title, style, rect, parent)
45           rect = self.GetClientRect()
46           rect = (0, 0, rect[2] - rect[0], rect[3] - rect[1])
47           self.ocx = MyWebBrowser()
48           self.ocx.CreateControl(
49               "Web Browser", win32con.WS_VISIBLE | win32con.WS_CHILD, rect, self, 1000
50           )
51           self.ocx.Navigate(self.url)
52           self.HookMessage(self.OnSize, win32con.WM_SIZE)
53   
54       def OnSize(self, params):
55           rect = self.GetClientRect()
56           rect = (0, 0, rect[2] - rect[0], rect[3] - rect[1])
57           self.ocx.SetWindowPos(0, rect, 0)
58   
59       def OnNavigate(self, url):
60           title = f"Web Browser - {url}"
61           self.SetWindowText(title)
62   
63   
64   def Demo(url=None):
65       if url is None and len(sys.argv) > 1:
66           url = win32api.GetFullPathName(sys.argv[1])
67       f = BrowserFrame(url)
68       f.Create("Web Browser")

71   if __name__ == "__main__":
72       Demo()
