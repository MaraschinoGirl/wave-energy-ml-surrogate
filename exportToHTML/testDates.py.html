1    import unittest

4    import pywintypes

11   # A COM object so we can pass dates to and from the COM boundary.
12   class Tester:
13       _public_methods_ = ["TestDate"]
14   
15       def TestDate(self, d):
16           assert isinstance(d, datetime)
17           return d
18   
19   
20   def test_ob():
21       return win32com.client.Dispatch(win32com.server.util.wrap(Tester()))
22   
23   
24   class TestCase(win32com.test.util.TestCase):
25       def check(self, d, expected=None):
26           if not issubclass(pywintypes.TimeType, datetime):
27               self.skipTest("this is testing pywintypes and datetime")
28           got = test_ob().TestDate(d)
29           self.assertEqual(got, expected or d)
30   
31       def testUTC(self):

33               datetime(
