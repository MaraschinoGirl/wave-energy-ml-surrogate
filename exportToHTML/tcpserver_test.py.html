72       @gen_test
73       def test_stop_in_callback(self):
74           # Issue #2069: calling server.stop() in a loop callback should not
75           # raise EBADF when the loop handles other server connection
76           # requests in the same loop iteration
77   
78           class TestServer(TCPServer):
79               @gen.coroutine
80               def handle_stream(self, stream, address):
81                   server.stop()  # type: ignore
82                   yield stream.read_until_close()
83   
84           sock, port = bind_unused_port()
85           server = TestServer()
86           server.add_socket(sock)
87           server_addr = ("localhost", port)
88           N = 40
89           clients = [IOStream(socket.socket()) for i in range(N)]
90           connected_clients = []
91   
92           @gen.coroutine
93           def connect(c):
94               try:
95                   yield c.connect(server_addr)
96               except EnvironmentError:
97                   pass
98               else:
99                   connected_clients.append(c)
100  
101          yield [connect(c) for c in clients]
102  
103          self.assertGreater(len(connected_clients), 0, "all clients failed connecting")
104          try:
105              if len(connected_clients) == N:
106                  # Ideally we'd make the test deterministic, but we're testing
107                  # for a race condition in combination with the system's TCP stack...
108                  self.skipTest(
109                      "at least one client should fail connecting "
110                      "for the test to be meaningful"
111                  )
112          finally:
113              for c in connected_clients:
114                  c.close()
115

120  class TestMultiprocess(unittest.TestCase):
121      # These tests verify that the two multiprocess examples from the
122      # TCPServer docs work. Both tests start a server with three worker
123      # processes, each of which prints its task id to stdout (a single
124      # byte, so we don't have to worry about atomicity of the shared
125      # stdout stream) and then exits.
126      def run_subproc(self, code: str) -> Tuple[str, str]:
127          try:
128              result = subprocess.run(
129                  [sys.executable, "-Werror::DeprecationWarning"],
130                  capture_output=True,
131                  input=code,
132                  encoding="utf8",
133                  check=True,
134              )
135          except subprocess.CalledProcessError as e:
136              raise RuntimeError(
137                  f"Process returned {e.returncode} stdout={e.stdout} stderr={e.stderr}"
138              ) from e
139          return result.stdout, result.stderr

141      def test_listen_single(self):
142          # As a sanity check, run the single-process version through this test
143          # harness too.
144          code = textwrap.dedent(
145              """ 
146              import asyncio 
147              from tornado.tcpserver import TCPServer 
148   
149              async def main(): 
150                  server = TCPServer() 
151                  server.listen(0, address='127.0.0.1') 
152   
153              asyncio.run(main()) 
154              print('012', end='') 
155          """
156          )
157          out, err = self.run_subproc(code)
158          self.assertEqual("".join(sorted(out)), "012")
