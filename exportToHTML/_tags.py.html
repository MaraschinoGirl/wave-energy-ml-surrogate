14   @dataclass(**_dataclass_args())
15   class InputTags:
16       """Tags for the input data. 
17    
18       Parameters 
19       ---------- 
20       one_d_array : bool, default=False 
21           Whether the input can be a 1D array. 
22    
23       two_d_array : bool, default=True 
24           Whether the input can be a 2D array. Note that most common 
25           tests currently run only if this flag is set to ``True``. 
26    
27       three_d_array : bool, default=False 
28           Whether the input can be a 3D array. 
29    
30       sparse : bool, default=False 
31           Whether the input can be a sparse matrix. 
32    
33       categorical : bool, default=False 
34           Whether the input can be categorical. 
35    
36       string : bool, default=False 
37           Whether the input can be an array-like of strings. 
38    
39       dict : bool, default=False 
40           Whether the input can be a dictionary. 
41    
42       positive_only : bool, default=False 
43           Whether the estimator requires positive X. 
44    
45       allow_nan : bool, default=False 
46           Whether the estimator supports data with missing values encoded as `np.nan`. 
47    
48       pairwise : bool, default=False 
49           This boolean attribute indicates whether the data (`X`), 
50           :term:`fit` and similar methods consists of pairwise measures 
51           over samples rather than a feature representation for each 
52           sample.  It is usually `True` where an estimator has a 
53           `metric` or `affinity` or `kernel` parameter with value 
54           'precomputed'. Its primary purpose is to support a 
55           :term:`meta-estimator` or a cross validation procedure that 
56           extracts a sub-sample of data intended for a pairwise 
57           estimator, where the data needs to be indexed on both axes. 
58           Specifically, this tag is used by 
59           `sklearn.utils.metaestimators._safe_split` to slice rows and 
60           columns. 
61    
62           Note that if setting this tag to ``True`` means the estimator can take only 
63           positive values, the `positive_only` tag must reflect it and also be set to 
64           ``True``. 
65       """
66   
67       one_d_array: bool = False
68       two_d_array: bool = True
69       three_d_array: bool = False
70       sparse: bool = False
71       categorical: bool = False
72       string: bool = False
73       dict: bool = False
74       positive_only: bool = False
75       allow_nan: bool = False
76       pairwise: bool = False

81       """Tags for the target data. 
82    
83       Parameters 
84       ---------- 
85       required : bool 
86           Whether the estimator requires y to be passed to `fit`, 
87           `fit_predict` or `fit_transform` methods. The tag is ``True`` 
88           for estimators inheriting from `~sklearn.base.RegressorMixin` 
89           and `~sklearn.base.ClassifierMixin`. 
90    
91       one_d_labels : bool, default=False 
92           Whether the input is a 1D labels (y). 
93    
94       two_d_labels : bool, default=False 
95           Whether the input is a 2D labels (y). 
96    
97       positive_only : bool, default=False 
98           Whether the estimator requires a positive y (only applicable 
99           for regression). 
100   
101      multi_output : bool, default=False 
102          Whether a regressor supports multi-target outputs or a classifier supports 
103          multi-class multi-output. 
104   
105          See :term:`multi-output` in the glossary. 
106   
107      single_output : bool, default=True 
108          Whether the target can be single-output. This can be ``False`` if the 
109          estimator supports only multi-output cases. 
110      """
111  
112      required: bool
113      one_d_labels: bool = False
114      two_d_labels: bool = False
115      positive_only: bool = False
116      multi_output: bool = False
117      single_output: bool = True

120  @dataclass(**_dataclass_args())
121  class TransformerTags:
122      """Tags for the transformer. 
123   
124      Parameters 
125      ---------- 
126      preserves_dtype : list[str], default=["float64"] 
127          Applies only on transformers. It corresponds to the data types 
128          which will be preserved such that `X_trans.dtype` is the same 
129          as `X.dtype` after calling `transformer.transform(X)`. If this 
130          list is empty, then the transformer is not expected to 
131          preserve the data type. The first value in the list is 
132          considered as the default data type, corresponding to the data 
133          type of the output when the input data type is not going to be 
134          preserved. 
135      """
136  
137      preserves_dtype: list[str] = field(default_factory=lambda: ["float64"])

141  class ClassifierTags:
142      """Tags for the classifier. 
143   
144      Parameters 
145      ---------- 
146      poor_score : bool, default=False 
147          Whether the estimator fails to provide a "reasonable" test-set 
148          score, which currently for classification is an accuracy of 
149          0.83 on ``make_blobs(n_samples=300, random_state=0)``. The 
150          datasets and values are based on current estimators in scikit-learn 
151          and might be replaced by something more systematic. 
152   
153      multi_class : bool, default=True 
154          Whether the classifier can handle multi-class 
155          classification. Note that all classifiers support binary 
156          classification. Therefore this flag indicates whether the 
157          classifier is a binary-classifier-only or not. 
158   
159          See :term:`multi-class` in the glossary. 
160   
161      multi_label : bool, default=False 
162          Whether the classifier supports multi-label output: a data point can 
163          be predicted to belong to a variable number of classes. 
164   
165          See :term:`multi-label` in the glossary. 
166      """
167  
168      poor_score: bool = False
169      multi_class: bool = True
170      multi_label: bool = False

192  class Tags:
193      """Tags for the estimator. 
194   
195      See :ref:`estimator_tags` for more information. 
196   
197      Parameters 
198      ---------- 
199      estimator_type : str or None 
200          The type of the estimator. Can be one of: 
201          - "classifier" 
202          - "regressor" 
203          - "transformer" 
204          - "clusterer" 
205          - "outlier_detector" 
206          - "density_estimator" 
207   
208      target_tags : :class:`TargetTags` 
209          The target(y) tags. 
210   
211      transformer_tags : :class:`TransformerTags` or None 
212          The transformer tags. 
213   
214      classifier_tags : :class:`ClassifierTags` or None 
215          The classifier tags. 
216   
217      regressor_tags : :class:`RegressorTags` or None 
218          The regressor tags. 
219   
220      array_api_support : bool, default=False 
221          Whether the estimator supports Array API compatible inputs. 
222   
223      no_validation : bool, default=False 
224          Whether the estimator skips input-validation. This is only meant for 
225          stateless and dummy transformers! 
226   
227      non_deterministic : bool, default=False 
228          Whether the estimator is not deterministic given a fixed ``random_state``. 
229   
230      requires_fit : bool, default=True 
231          Whether the estimator requires to be fitted before calling one of 
232          `transform`, `predict`, `predict_proba`, or `decision_function`. 
233   
234      _skip_test : bool, default=False 
235          Whether to skip common tests entirely. Don't use this unless 
236          you have a *very good* reason. 
237   
238      input_tags : :class:`InputTags` 
239          The input data(X) tags. 
240      """
241  
242      estimator_type: str | None
243      target_tags: TargetTags
244      transformer_tags: TransformerTags | None = None
245      classifier_tags: ClassifierTags | None = None
246      regressor_tags: RegressorTags | None = None
247      array_api_support: bool = False
248      no_validation: bool = False
249      non_deterministic: bool = False
250      requires_fit: bool = True
251      _skip_test: bool = False
252      input_tags: InputTags = field(default_factory=InputTags)

256  def default_tags(estimator) -> Tags:
257      """Get the default tags for an estimator. 
258   
259      This ignores any ``__sklearn_tags__`` method that the estimator may have. 
260   
261      If the estimator is a classifier or a regressor, ``target_tags.required`` 
262      will be set to ``True``, otherwise it will be set to ``False``. 
263   
264      ``transformer_tags`` will be set to :class:`~.sklearn.utils. TransformerTags` if the 
265      estimator has a ``transform`` or ``fit_transform`` method, otherwise it will be set 
266      to ``None``. 
267   
268      ``classifier_tags`` will be set to :class:`~.sklearn.utils.ClassifierTags` if the 
269      estimator is a classifier, otherwise it will be set to ``None``. 
270      a classifier, otherwise it will be set to ``None``. 
271   
272      ``regressor_tags`` will be set to :class:`~.sklearn.utils.RegressorTags` if the 
273      estimator is a regressor, otherwise it will be set to ``None``. 
274   
275      Parameters 
276      ---------- 
277      estimator : estimator object 
278          The estimator for which to get the default tags. 
279   
280      Returns 
281      ------- 
282      tags : Tags 
283          The default tags for the estimator. 
284      """
285      est_is_classifier = getattr(estimator, "_estimator_type", None) == "classifier"
286      est_is_regressor = getattr(estimator, "_estimator_type", None) == "regressor"
287      target_required = est_is_classifier or est_is_regressor
288  
289      return Tags(
290          estimator_type=getattr(estimator, "_estimator_type", None),
291          target_tags=TargetTags(required=target_required),
292          transformer_tags=(
293              TransformerTags()
294              if hasattr(estimator, "transform") or hasattr(estimator, "fit_transform")
295              else None
296          ),
297          classifier_tags=ClassifierTags() if est_is_classifier else None,
298          regressor_tags=RegressorTags() if est_is_regressor else None,
299      )

303  def _find_tags_provider(estimator, warn=True):
304      """Find the tags provider for an estimator. 
305   
306      Parameters 
307      ---------- 
308      estimator : estimator object 
309          The estimator to find the tags provider for. 
310   
311      warn : bool, default=True 
312          Whether to warn if the tags provider is not found. 
313   
314      Returns 
315      ------- 
316      tag_provider : str 
317          The tags provider for the estimator. Can be one of: 
318          - "_get_tags": to use the old tags infrastructure 
319          - "__sklearn_tags__": to use the new tags infrastructure 
320      """
321      mro_model = type(estimator).mro()
322      tags_mro = OrderedDict()
323      for klass in mro_model:
324          tags_provider = []
325          if "_more_tags" in vars(klass):
326              tags_provider.append("_more_tags")
327          if "_get_tags" in vars(klass):
328              tags_provider.append("_get_tags")
329          if "__sklearn_tags__" in vars(klass):
330              tags_provider.append("__sklearn_tags__")
331          tags_mro[klass.__name__] = tags_provider
332  
333      all_providers = set(chain.from_iterable(tags_mro.values()))
334      if "__sklearn_tags__" not in all_providers:
335          # default on the old tags infrastructure
336          return "_get_tags"
337  
338      tag_provider = "__sklearn_tags__"
339      for klass in tags_mro:
340          has_get_or_more_tags = any(
341              provider in tags_mro[klass] for provider in ("_get_tags", "_more_tags")
342          )
343          has_sklearn_tags = "__sklearn_tags__" in tags_mro[klass]
344  
345          if tags_mro[klass] and tag_provider == "__sklearn_tags__":  # is it empty
346              if has_get_or_more_tags and not has_sklearn_tags:
347                  # Case where a class does not implement __sklearn_tags__ and we fallback
348                  # to _get_tags. We should therefore warn for implementing
349                  # __sklearn_tags__.
350                  tag_provider = "_get_tags"
351                  break
352  
353      if warn and tag_provider == "_get_tags":
354          warnings.warn(
355              f"The {estimator.__class__.__name__} or classes from which it inherits "
356              "use `_get_tags` and `_more_tags`. Please define the "
357              "`__sklearn_tags__` method, or inherit from `sklearn.base.BaseEstimator` "
358              "and/or other appropriate mixins such as `sklearn.base.TransformerMixin`, "
359              "`sklearn.base.ClassifierMixin`, `sklearn.base.RegressorMixin`, and "
360              "`sklearn.base.OutlierMixin`. From scikit-learn 1.7, not defining "
361              "`__sklearn_tags__` will raise an error.",
362              category=DeprecationWarning,
363          )
364      return tag_provider

367  def get_tags(estimator) -> Tags:
368      """Get estimator tags. 
369   
370      :class:`~sklearn.BaseEstimator` provides the estimator tags machinery. 
371      However, if an estimator does not inherit from this base class, we should 
372      fall-back to the default tags. 
373   
374      For scikit-learn built-in estimators, we should still rely on 
375      `self.__sklearn_tags__()`. `get_tags(est)` should be used when we 
376      are not sure where `est` comes from: typically 
377      `get_tags(self.estimator)` where `self` is a meta-estimator, or in 
378      the common checks. 
379   
380      .. versionadded:: 1.6 
381   
382      Parameters 
383      ---------- 
384      estimator : estimator object 
385          The estimator from which to get the tag. 
386   
387      Returns 
388      ------- 
389      tags : :class:`~.sklearn.utils.Tags` 
390          The estimator tags. 
391      """
392  
393      tag_provider = _find_tags_provider(estimator)
394  
395      if tag_provider == "__sklearn_tags__":
396          # TODO(1.7): turn the warning into an error
397          try:
398              tags = estimator.__sklearn_tags__()
399          except AttributeError as exc:
400              if str(exc) == "'super' object has no attribute '__sklearn_tags__'":
401                  # workaround the regression reported in
402                  # https://github.com/scikit-learn/scikit-learn/issues/30479
403                  # `__sklearn_tags__` is implemented by calling
404                  # `super().__sklearn_tags__()` but there is no `__sklearn_tags__`
405                  # method in the base class.
406                  warnings.warn(
407                      f"The following error was raised: {str(exc)}. It seems that "
408                      "there are no classes that implement `__sklearn_tags__` "
409                      "in the MRO and/or all classes in the MRO call "
410                      "`super().__sklearn_tags__()`. Make sure to inherit from "
411                      "`BaseEstimator` which implements `__sklearn_tags__` (or "
412                      "alternatively define `__sklearn_tags__` but we don't recommend "
413                      "this approach). Note that `BaseEstimator` needs to be on the "
414                      "right side of other Mixins in the inheritance order. The "
415                      "default are now used instead since retrieving tags failed. "
416                      "This warning will be replaced by an error in 1.7.",
417                      category=DeprecationWarning,
418                  )
419                  tags = default_tags(estimator)
420              else:
421                  raise
422      else:
423          # TODO(1.7): Remove this branch of the code
424          # Let's go through the MRO and patch each class implementing _more_tags
425          sklearn_tags_provider = {}
426          more_tags_provider = {}
427          class_order = []
428          for klass in reversed(type(estimator).mro()):
429              if "__sklearn_tags__" in vars(klass):
430                  sklearn_tags_provider[klass] = klass.__sklearn_tags__(estimator)  # type: ignore[attr-defined]
431                  class_order.append(klass)
432              elif "_more_tags" in vars(klass):
433                  more_tags_provider[klass] = klass._more_tags(estimator)  # type: ignore[attr-defined]
434                  class_order.append(klass)
435  
436          # Find differences between consecutive in the case of __sklearn_tags__
437          # inheritance
438          sklearn_tags_diff = {}
439          items = list(sklearn_tags_provider.items())
440          for current_item, next_item in zip(items[:-1], items[1:]):
441              current_name, current_tags = current_item
442              next_name, next_tags = next_item
443              current_tags = _to_old_tags(current_tags)
444              next_tags = _to_old_tags(next_tags)
445  
446              # Compare tags and store differences
447              diff = {}
448              for key in current_tags:
449                  if current_tags[key] != next_tags[key]:
450                      diff[key] = next_tags[key]
451  
452              sklearn_tags_diff[next_name] = diff
453  
454          tags = {}
455          for klass in class_order:
456              if klass in sklearn_tags_diff:
457                  tags.update(sklearn_tags_diff[klass])
458              elif klass in more_tags_provider:
459                  tags.update(more_tags_provider[klass])
460  
461          tags = _to_new_tags(
462              {**_to_old_tags(default_tags(estimator)), **tags}, estimator
463          )
464  
465      return tags

469  def _safe_tags(estimator, key=None):
470      warnings.warn(
471          "The `_safe_tags` function is deprecated in 1.6 and will be removed in "
472          "1.7. Use the public `get_tags` function instead and make sure to implement "
473          "the `__sklearn_tags__` method.",
474          category=DeprecationWarning,
475      )
476      tags = _to_old_tags(get_tags(estimator))
477  
478      if key is not None:
479          if key not in tags:
480              raise ValueError(
481                  f"The key {key} is not defined for the class "
482                  f"{estimator.__class__.__name__}."
483              )
484          return tags[key]
485      return tags

489  def _to_new_tags(old_tags, estimator=None):
490      """Utility function convert old tags (dictionary) to new tags (dataclass)."""
491      input_tags = InputTags(
492          one_d_array="1darray" in old_tags["X_types"],
493          two_d_array="2darray" in old_tags["X_types"],
494          three_d_array="3darray" in old_tags["X_types"],
495          sparse="sparse" in old_tags["X_types"],
496          categorical="categorical" in old_tags["X_types"],
497          string="string" in old_tags["X_types"],
498          dict="dict" in old_tags["X_types"],
499          positive_only=old_tags["requires_positive_X"],
500          allow_nan=old_tags["allow_nan"],
501          pairwise=old_tags["pairwise"],
502      )
503      target_tags = TargetTags(
504          required=old_tags["requires_y"],
505          one_d_labels="1dlabels" in old_tags["X_types"],
506          two_d_labels="2dlabels" in old_tags["X_types"],
507          positive_only=old_tags["requires_positive_y"],
508          multi_output=old_tags["multioutput"] or old_tags["multioutput_only"],
509          single_output=not old_tags["multioutput_only"],
510      )
511      if estimator is not None and (
512          hasattr(estimator, "transform") or hasattr(estimator, "fit_transform")
513      ):
514          transformer_tags = TransformerTags(
515              preserves_dtype=old_tags["preserves_dtype"],
516          )
517      else:
518          transformer_tags = None
519      estimator_type = getattr(estimator, "_estimator_type", None)
520      if estimator_type == "classifier":
521          classifier_tags = ClassifierTags(
522              poor_score=old_tags["poor_score"],
523              multi_class=not old_tags["binary_only"],
524              multi_label=old_tags["multilabel"],
525          )
526      else:
527          classifier_tags = None
528      if estimator_type == "regressor":
529          regressor_tags = RegressorTags(
530              poor_score=old_tags["poor_score"],
531          )
532      else:
533          regressor_tags = None
534      return Tags(
535          estimator_type=estimator_type,
536          target_tags=target_tags,
537          transformer_tags=transformer_tags,
538          classifier_tags=classifier_tags,
539          regressor_tags=regressor_tags,
540          input_tags=input_tags,
541          array_api_support=old_tags["array_api_support"],
542          no_validation=old_tags["no_validation"],
543          non_deterministic=old_tags["non_deterministic"],
544          requires_fit=old_tags["requires_fit"],
545          _skip_test=old_tags["_skip_test"],
546      )
547

550  def _to_old_tags(new_tags):
551      """Utility function convert old tags (dictionary) to new tags (dataclass)."""
552      if new_tags.classifier_tags:
553          binary_only = not new_tags.classifier_tags.multi_class
554          multilabel = new_tags.classifier_tags.multi_label
555          poor_score_clf = new_tags.classifier_tags.poor_score
556      else:
557          binary_only = False
558          multilabel = False
559          poor_score_clf = False
560  
561      if new_tags.regressor_tags:
562          poor_score_reg = new_tags.regressor_tags.poor_score
563      else:
564          poor_score_reg = False
565  
566      if new_tags.transformer_tags:
567          preserves_dtype = new_tags.transformer_tags.preserves_dtype
568      else:
569          preserves_dtype = ["float64"]
570  
571      tags = {
572          "allow_nan": new_tags.input_tags.allow_nan,
573          "array_api_support": new_tags.array_api_support,
574          "binary_only": binary_only,
575          "multilabel": multilabel,
576          "multioutput": new_tags.target_tags.multi_output,
577          "multioutput_only": (
578              not new_tags.target_tags.single_output and new_tags.target_tags.multi_output
579          ),
580          "no_validation": new_tags.no_validation,
581          "non_deterministic": new_tags.non_deterministic,
582          "pairwise": new_tags.input_tags.pairwise,
583          "preserves_dtype": preserves_dtype,
584          "poor_score": poor_score_clf or poor_score_reg,
585          "requires_fit": new_tags.requires_fit,
586          "requires_positive_X": new_tags.input_tags.positive_only,
587          "requires_y": new_tags.target_tags.required,
588          "requires_positive_y": new_tags.target_tags.positive_only,
589          "_skip_test": new_tags._skip_test,
590          "stateless": new_tags.requires_fit,
591      }
592      X_types = []
593      if new_tags.input_tags.one_d_array:
594          X_types.append("1darray")
595      if new_tags.input_tags.two_d_array:
596          X_types.append("2darray")
597      if new_tags.input_tags.three_d_array:
598          X_types.append("3darray")
599      if new_tags.input_tags.sparse:
600          X_types.append("sparse")
601      if new_tags.input_tags.categorical:
602          X_types.append("categorical")
603      if new_tags.input_tags.string:
604          X_types.append("string")
605      if new_tags.input_tags.dict:
606          X_types.append("dict")
607      if new_tags.target_tags.one_d_labels:
608          X_types.append("1dlabels")
609      if new_tags.target_tags.two_d_labels:
610          X_types.append("2dlabels")
611      tags["X_types"] = X_types
612      return tags
