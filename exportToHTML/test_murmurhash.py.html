10   def test_mmhash3_int():
11       assert murmurhash3_32(3) == 847579505
12       assert murmurhash3_32(3, seed=0) == 847579505
13       assert murmurhash3_32(3, seed=42) == -1823081949
14   
15       assert murmurhash3_32(3, positive=False) == 847579505
16       assert murmurhash3_32(3, seed=0, positive=False) == 847579505
17       assert murmurhash3_32(3, seed=42, positive=False) == -1823081949
18   
19       assert murmurhash3_32(3, positive=True) == 847579505
20       assert murmurhash3_32(3, seed=0, positive=True) == 847579505
21       assert murmurhash3_32(3, seed=42, positive=True) == 2471885347

24   def test_mmhash3_int_array():
25       rng = np.random.RandomState(42)
26       keys = rng.randint(-5342534, 345345, size=3 * 2 * 1).astype(np.int32)
27       keys = keys.reshape((3, 2, 1))
28   
29       for seed in [0, 42]:
30           expected = np.array([murmurhash3_32(int(k), seed) for k in keys.flat])
31           expected = expected.reshape(keys.shape)
32           assert_array_equal(murmurhash3_32(keys, seed), expected)
33   
34       for seed in [0, 42]:
35           expected = np.array([murmurhash3_32(k, seed, positive=True) for k in keys.flat])
36           expected = expected.reshape(keys.shape)
37           assert_array_equal(murmurhash3_32(keys, seed, positive=True), expected)

40   def test_mmhash3_bytes():
41       assert murmurhash3_32(b"foo", 0) == -156908512
42       assert murmurhash3_32(b"foo", 42) == -1322301282

56   def test_no_collision_on_byte_range():
57       previous_hashes = set()
58       for i in range(100):
59           h = murmurhash3_32(" " * i, 0)
60           assert h not in previous_hashes, "Found collision on growing empty string"

63   def test_uniform_distribution():
64       n_bins, n_samples = 10, 100000
65       bins = np.zeros(n_bins, dtype=np.float64)
66   
67       for i in range(n_samples):
68           bins[murmurhash3_32(i, positive=True) % n_bins] += 1
69   
70       means = bins / n_samples
71       expected = np.full(n_bins, 1.0 / n_bins)
72   
73       assert_array_almost_equal(means / expected, np.ones(n_bins), 2)
