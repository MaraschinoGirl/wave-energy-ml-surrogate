39   class _HashablePartialShim:
40       """Object that delegates :meth:`__call__`, :meth:`__eq__`, and :meth:`__hash__` to another object."""
41   
42       __slots__: ClassVar[tuple[str, ...]] = ('args', 'func', 'keywords', 'partial_func')
43   
44       func: Callable[..., Any]
45       args: tuple[Any, ...]
46       keywords: dict[str, Any]
47   
48       def __init__(self, partial_func: functools.partial, /) -> None:
49           self.partial_func: functools.partial = partial_func
50   
51       def __call__(self, /, *args: Any, **kwargs: Any) -> Any:
52           return self.partial_func(*args, **kwargs)
53   
54       def __eq__(self, other: object, /) -> bool:
55           if isinstance(other, _HashablePartialShim):
56               return self.partial_func == other.partial_func
57           return self.partial_func == other
58   
59       def __hash__(self, /) -> int:
60           return hash(self.partial_func)
61   
62       def __repr__(self, /) -> str:
63           return repr(self.partial_func)

66   # pylint: disable-next=protected-access
67   @registry.register_pytree_node_class(namespace=registry.__GLOBAL_NAMESPACE)
68   class partial(  # noqa: N801 # pylint: disable=invalid-name,too-few-public-methods
69       functools.partial,
70       CustomTreeNode[T],
71   ):
72       """A version of :func:`functools.partial` that works in pytrees. 
73    
74       Use it for partial function evaluation in a way that is compatible with transformations, 
75       e.g., ``partial(func, *args, **kwargs)``. 
76    
77       (You need to explicitly opt-in to this behavior because we did not want to give 
78       :func:`functools.partial` different semantics than normal function closures.) 
79    
80       For example, here is a basic usage of :class:`partial` in a manner similar to 
81       :func:`functools.partial`: 
82    
83       >>> import operator 
84       >>> import torch 
85       >>> add_one = partial(operator.add, torch.ones(())) 
86       >>> add_one(torch.tensor([[1, 2], [3, 4]])) 
87       tensor([[2., 3.], 
88               [4., 5.]]) 
89    
90       Pytree compatibility means that the resulting partial function can be passed as an argument 
91       within tree-map functions, which is not possible with a standard :func:`functools.partial` 
92       function: 
93    
94       >>> def call_func_on_cuda(f, *args, **kwargs): 
95       ...     f, args, kwargs = tree_map(lambda t: t.cuda(), (f, args, kwargs)) 
96       ...     return f(*args, **kwargs) 
97       ... 
98       >>> # doctest: +SKIP 
99       >>> tree_map(lambda t: t.cuda(), add_one) 
100      optree.functools.partial(<built-in function add>, tensor(1., device='cuda:0')) 
101      >>> call_func_on_cuda(add_one, torch.tensor([[1, 2], [3, 4]])) 
102      tensor([[2., 3.], 
103              [4., 5.]], device='cuda:0') 
104   
105      Passing zero arguments to :class:`partial` effectively wraps the original function, making it a 
106      valid argument in tree-map functions: 
107   
108      >>> # doctest: +SKIP 
109      >>> call_func_on_cuda(partial(torch.add), torch.tensor(1), torch.tensor(2)) 
110      tensor(3, device='cuda:0') 
111   
112      Had we passed :func:`operator.add` to ``call_func_on_cuda`` directly, it would have resulted in 
113      a :class:`TypeError` or :class:`AttributeError`. 
114      """
115  
116      __slots__: ClassVar[tuple[()]] = ()
117  
118      func: Callable[..., Any]
119      args: tuple[T, ...]
120      keywords: dict[str, T]
121  
122      TREE_PATH_ENTRY_TYPE: ClassVar[type[PyTreeEntry]] = GetAttrEntry
123  
124      def __new__(cls, func: Callable[..., Any], /, *args: T, **keywords: T) -> Self:
125          """Create a new :class:`partial` instance."""
126          # In Python 3.10+, if func is itself a functools.partial instance, functools.partial.__new__
127          # would merge the arguments of this partial instance with the arguments of the func. We box
128          # func in a class that does not (yet) have a `func` attribute to defeat this optimization,
129          # since we care exactly which arguments are considered part of the pytree.
130          if isinstance(func, functools.partial):
131              original_func = func
132              func = _HashablePartialShim(original_func)
133              assert not hasattr(func, 'func'), 'shimmed function should not have a `func` attribute'
134              out = super().__new__(cls, func, *args, **keywords)
135              func.func = original_func.func
136              func.args = original_func.args
137              func.keywords = original_func.keywords
138              return out
139  
140          return super().__new__(cls, func, *args, **keywords)
141  
142      def __repr__(self, /) -> str:
143          """Return a string representation of the :class:`partial` instance."""
144          args = [repr(self.func)]
145          args.extend(repr(x) for x in self.args)
146          args.extend(f'{k}={v!r}' for (k, v) in self.keywords.items())
147          return f'{self.__class__.__module__}.{self.__class__.__qualname__}({", ".join(args)})'
148  
149      def tree_flatten(  # type: ignore[override]
150          self,
151          /,
152      ) -> tuple[
153          tuple[tuple[T, ...], dict[str, T]],
154          Callable[..., Any],
155          tuple[str, str],
156      ]:
157          """Flatten the :class:`partial` instance to children and metadata."""
158          return (self.args, self.keywords), self.func, ('args', 'keywords')
159  
160      @classmethod
161      def tree_unflatten(  # type: ignore[override]
162          cls,
163          metadata: Callable[..., Any],
164          children: tuple[tuple[T, ...], dict[str, T]],
165          /,
166      ) -> Self:
167          """Unflatten the children and metadata into a :class:`partial` instance."""
168          args, keywords = children
169          return cls(metadata, *args, **keywords)
