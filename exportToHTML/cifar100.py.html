1    """CIFAR100 small images classification dataset."""
2    
3    import os
4    
5    import numpy as np
6    
7    from keras.src import backend
8    from keras.src.api_export import keras_export
9    from keras.src.datasets.cifar import load_batch
10   from keras.src.utils.file_utils import get_file
11   
12   
13   @keras_export("keras.datasets.cifar100.load_data")
14   def load_data(label_mode="fine"):
15       """Loads the CIFAR100 dataset. 
16    
17       This is a dataset of 50,000 32x32 color training images and 
18       10,000 test images, labeled over 100 fine-grained classes that are 
19       grouped into 20 coarse-grained classes. See more info at the 
20       [CIFAR homepage](https://www.cs.toronto.edu/~kriz/cifar.html). 
21    
22       Args: 
23           label_mode: one of `"fine"`, `"coarse"`. 
24               If it is `"fine"`, the category labels 
25               are the fine-grained labels, and if it is `"coarse"`, 
26               the output labels are the coarse-grained superclasses. 
27    
28       Returns: 
29           Tuple of NumPy arrays: `(x_train, y_train), (x_test, y_test)`. 
30    
31       **`x_train`**: `uint8` NumPy array of grayscale image data with shapes 
32         `(50000, 32, 32, 3)`, containing the training data. Pixel values range 
33         from 0 to 255. 
34    
35       **`y_train`**: `uint8` NumPy array of labels (integers in range 0-99) 
36         with shape `(50000, 1)` for the training data. 
37    
38       **`x_test`**: `uint8` NumPy array of grayscale image data with shapes 
39         `(10000, 32, 32, 3)`, containing the test data. Pixel values range 
40         from 0 to 255. 
41    
42       **`y_test`**: `uint8` NumPy array of labels (integers in range 0-99) 
43         with shape `(10000, 1)` for the test data. 
44    
45       Example: 
46    
47       ```python 
48       (x_train, y_train), (x_test, y_test) = keras.datasets.cifar100.load_data() 
49       assert x_train.shape == (50000, 32, 32, 3) 
50       assert x_test.shape == (10000, 32, 32, 3) 
51       assert y_train.shape == (50000, 1) 
52       assert y_test.shape == (10000, 1) 
53       ``` 
54       """
55       if label_mode not in ["fine", "coarse"]:
56           raise ValueError(
57               '`label_mode` must be one of `"fine"`, `"coarse"`. '
58               f"Received: label_mode={label_mode}."
59           )
60   
61       dirname = "cifar-100-python-target"
62       origin = "https://www.cs.toronto.edu/~kriz/cifar-100-python.tar.gz"
63       path = get_file(
64           fname=dirname,
65           origin=origin,
66           extract=True,
67           file_hash=(  # noqa: E501
68               "85cd44d02ba6437773c5bbd22e183051d648de2e7d6b014e1ef29b855ba677a7"
69           ),
70       )
71   
72       path = os.path.join(path, "cifar-100-python")
73       fpath = os.path.join(path, "train")
74       x_train, y_train = load_batch(fpath, label_key=label_mode + "_labels")
75   
76       fpath = os.path.join(path, "test")
77       x_test, y_test = load_batch(fpath, label_key=label_mode + "_labels")
78   
79       y_train = np.reshape(y_train, (len(y_train), 1))
80       y_test = np.reshape(y_test, (len(y_test), 1))
81   
82       if backend.image_data_format() == "channels_last":
83           x_train = x_train.transpose(0, 2, 3, 1)
84           x_test = x_test.transpose(0, 2, 3, 1)
85   
86       return (x_train, y_train), (x_test, y_test)
