16   import unittest
17   
18   from tornado.testing import AsyncTestCase, gen_test
19   
20   try:
21       from twisted.internet.defer import inlineCallbacks  # type: ignore
22   
23       have_twisted = True
24   except ImportError:
25       have_twisted = False
26   else:
27       # Not used directly but needed for `yield deferred` to work.
28       import tornado.platform.twisted  # noqa: F401
29   
30   skipIfNoTwisted = unittest.skipUnless(have_twisted, "twisted module not present")
31   
32   
33   @skipIfNoTwisted
34   class ConvertDeferredTest(AsyncTestCase):
35       @gen_test
36       def test_success(self):
37           @inlineCallbacks
38           def fn():
39               if False:
40                   # inlineCallbacks doesn't work with regular functions;
41                   # must have a yield even if it's unreachable.
42                   yield
43               return 42
44   
45           res = yield fn()
46           self.assertEqual(res, 42)
47   
48       @gen_test
49       def test_failure(self):
50           @inlineCallbacks
51           def fn():
52               if False:
53                   yield
54               1 / 0
55   
56           with self.assertRaises(ZeroDivisionError):
57               yield fn()
58   
59   
60   if __name__ == "__main__":
61       unittest.main()
