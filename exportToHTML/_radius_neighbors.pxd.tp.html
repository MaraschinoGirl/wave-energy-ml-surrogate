35       """float{{name_suffix}} implementation of the RadiusNeighbors."""

37       cdef:

40           # DistanceMetric{{name_suffix}} compute rank-preserving surrogate distance via rdist
41           # which are proxies necessitating less computations.
42           # We get the equivalent for the radius to be able to compare it against
43           # vectors' rank-preserving surrogate distances.
44           float64_t r_radius
45   
46           # Neighbors indices and distances are returned as np.ndarrays of np.ndarrays.
47           #
48           # For this implementation, we want resizable buffers which we will wrap
49           # into numpy arrays at the end. std::vector comes as a handy container
50           # for interacting efficiently with resizable buffers.
51           #
52           # Though it is possible to access their buffer address with
53           # std::vector::data, they can't be stolen: buffers lifetime
54           # is tied to their std::vector and are deallocated when
55           # std::vectors are.
56           #
57           # To solve this, we dynamically allocate std::vectors and then
58           # encapsulate them in a StdVectorSentinel responsible for
59           # freeing them when the associated np.ndarray is freed.
60           #
61           # Shared pointers (defined via shared_ptr) are use for safer memory management.
62           # Unique pointers (defined via unique_ptr) can't be used as datastructures
63           # are shared across threads for parallel_on_X; see _parallel_on_X_init_chunk.
64           shared_ptr[vector[vector[intp_t]]] neigh_indices
65           shared_ptr[vector[vector[float64_t]]] neigh_distances
66   
67           # Used as array of pointers to private datastructures used in threads.
68           vector[shared_ptr[vector[vector[intp_t]]]] neigh_indices_chunks
69           vector[shared_ptr[vector[vector[float64_t]]]] neigh_distances_chunks
70   
71           bint sort_results
