1    # ModuleSpec, ModuleType, Loader are part of a dependency cycle.
2    # They are officially defined/exported in other places:
3    #
4    # - ModuleType in types
5    # - Loader in importlib.abc
6    # - ModuleSpec in importlib.machinery (3.4 and later only)
7    #
8    # _Loader is the PEP-451-defined interface for a loader type/object.
9    
10   from abc import ABCMeta
11   from typing import Any, Dict, List, Optional, Protocol
12   
13   class _Loader(Protocol):
14       def load_module(self, fullname: str) -> ModuleType: ...
15   
16   class ModuleSpec:
17       def __init__(
18           self,
19           name: str,
20           loader: Optional[Loader],
21           *,
22           origin: Optional[str] = ...,
23           loader_state: Any = ...,
24           is_package: Optional[bool] = ...,
25       ) -> None: ...
26       name: str
27       loader: Optional[_Loader]
28       origin: Optional[str]
29       submodule_search_locations: Optional[List[str]]
30       loader_state: Any
31       cached: Optional[str]
32       parent: Optional[str]
33       has_location: bool
34   
35   class ModuleType:
36       __name__: str
37       __file__: str
38       __dict__: Dict[str, Any]
39       __loader__: Optional[_Loader]
40       __package__: Optional[str]
41       __spec__: Optional[ModuleSpec]
42       def __init__(self, name: str, doc: Optional[str] = ...) -> None: ...
43   
44   class Loader(metaclass=ABCMeta):
45       def load_module(self, fullname: str) -> ModuleType: ...
46       def module_repr(self, module: ModuleType) -> str: ...
47       def create_module(self, spec: ModuleSpec) -> Optional[ModuleType]: ...
48       # Not defined on the actual class for backwards-compatibility reasons,
