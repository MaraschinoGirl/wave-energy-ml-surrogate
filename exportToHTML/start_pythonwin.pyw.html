1    # A Python file that can be used to start Pythonwin, instead of using
2    # pythonwin.exe

4    import sys

10   assert pywin.framework.intpyapp  # not unused
11   # Pretend this script doesn't exist, or pythonwin tries to edit it
12   sys.argv[:] = sys.argv[1:] or [""]  # like PySys_SetArgv(Ex)
13   if sys.path[0] not in ("", ".", os.getcwd()):
14       sys.path.insert(0, os.getcwd())
15   # And bootstrap the app.
16   app = win32ui.GetApp()
17   if not app.InitInstance():
18       # Run when not already handled by DDE
