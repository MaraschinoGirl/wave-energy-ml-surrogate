18   def _extract_image_batch(images, num_images, batch_size):
19       """Extracts a batch of images for plotting. 
20    
21       Args: 
22           images: The 4D tensor or NumPy array of images. 
23           num_images: The number of images to extract. 
24           batch_size: The original batch size of the images. 
25    
26       Returns: 
27           A 4D tensor or NumPy array containing the extracted images. 
28    
29       Raises: 
30           ValueError: If `images` is not a 4D tensor/array. 
31       """
32   
33       if len(ops.shape(images)) != 4:
34           raise ValueError(
35               "`plot_images_gallery()` requires you to "
36               "batch your `np.array` samples together."
37           )
38       num_samples = min(num_images, batch_size)
39       sample = images[:num_samples, ...]
40   
41       return sample

44   @keras_export("keras.visualization.plot_image_gallery")
45   def plot_image_gallery(
46       images,
47       y_true=None,
48       y_pred=None,
49       label_map=None,
50       rows=None,
51       cols=None,
52       value_range=(0, 255),
53       scale=2,
54       path=None,
55       show=None,
56       transparent=True,
57       dpi=60,
58       legend_handles=None,
59       data_format=None,
60   ):
61       """Displays a gallery of images with optional labels and predictions. 
62    
63       Args: 
64           images: A 4D tensor or NumPy array of images. Shape should be 
65              `(batch_size, height, width, channels)`. 
66           y_true: A 1D tensor or NumPy array of true labels (class indices). 
67              Defaults to `None`. 
68           y_pred: A 1D tensor or NumPy array of predicted labels (class indices). 
69              Defaults to `None`. 
70           label_map: A dictionary mapping class indices to class names. 
71               Required if `y_true` or `y_pred` are provided. 
72              Defaults to `None`. 
73           value_range: A tuple specifying the value range of the images 
74               (e.g., `(0, 255)` or `(0, 1)`). Defaults to `(0, 255)`. 
75           rows: The number of rows in the gallery. If `None`, it's calculated 
76               based on the number of images and `cols`. Defaults to `None`. 
77           cols: The number of columns in the gallery. If `None`, it's calculated 
78               based on the number of images and `rows`. Defaults to `None`. 
79           scale: A float controlling the size of the displayed images. The images 
80               are scaled by this factor. Defaults to `2`. 
81           path: The path to save the generated gallery image. If `None`, the 
82               image is displayed using `plt.show()`. Defaults to `None`. 
83           show: Whether to display the image using `plt.show()`. If `True`, the 
84               image is displayed. If `False`, the image is not displayed. 
85               Ignored if `path` is not `None`. Defaults to `True` if `path` 
86               is `None`, `False` otherwise. 
87           transparent:  A boolean, whether to save the figure with a transparent 
88               background. Defaults to `True`. 
89           dpi: The DPI (dots per inch) for saving the figure. Defaults to 60. 
90           legend_handles: A list of matplotlib `Patch` objects to use as legend 
91               handles. Defaults to `None`. 
92           data_format: The image data format `"channels_last"` or 
93               `"channels_first"`. Defaults to the Keras backend data format. 
94    
95       Raises: 
96           ValueError: If both `path` and `show` are set to non-`None` values, 
97               if `images` is not a 4D tensor or array, or if `y_true` or `y_pred` 
98               are provided without a `label_map`. 
99           ImportError: if matplotlib is not installed. 
100      """
101      if plt is None:
102          raise ImportError(
103              "The `plot_image_gallery` function requires the `matplotlib` "
104              "package. Please install it with `pip install matplotlib`."
105          )
106  
107      if path is not None and show:
108          raise ValueError(
109              "plot_gallery() expects either `path` to be set, or `show` "
110              "to be true."
111          )
112  
113      if (y_true is not None or y_pred is not None) and label_map is None:
114          raise ValueError(
115              "If `y_true` or `y_pred` are provided, a `label_map` must also be"
116              " provided."
117          )
118  
119      show = show if show is not None else (path is None)
120      data_format = data_format or backend.image_data_format()
121  
122      batch_size = ops.shape(images)[0] if len(ops.shape(images)) == 4 else 1
123  
124      rows = rows or int(math.ceil(math.sqrt(batch_size)))
125      cols = cols or int(math.ceil(batch_size // rows))
126      num_images = rows * cols
127  
128      images = _extract_image_batch(images, num_images, batch_size)
129      if (
130          data_format == "channels_first"
131      ):  # Ensure correct data format for plotting
132          images = ops.transpose(images, (0, 2, 3, 1))
133  
134      # Generate subplots
135      fig, axes = plt.subplots(
136          nrows=rows,
137          ncols=cols,
138          figsize=(cols * scale, rows * scale),
139          frameon=False,
140          layout="tight",
141          squeeze=True,
142          sharex="row",
143          sharey="col",
144      )
145      fig.subplots_adjust(wspace=0, hspace=0)
146  
147      if isinstance(axes, np.ndarray) and len(axes.shape) == 1:
148          expand_axis = 0 if rows == 1 else -1
149          axes = np.expand_dims(axes, expand_axis)
150  
151      if legend_handles is not None:
152          fig.legend(handles=legend_handles, loc="lower center")
153  
154      images = BaseImagePreprocessingLayer()._transform_value_range(
155          images=images, original_range=value_range, target_range=(0, 255)
156      )
157  
158      images = ops.convert_to_numpy(images)
159      if data_format == "channels_first":
160          images = images.transpose(0, 2, 3, 1)
161  
162      if y_true is not None:
163          y_true = ops.convert_to_numpy(y_true)
164      if y_pred is not None:
165          y_pred = ops.convert_to_numpy(y_pred)
166  
167      for row in range(rows):
168          for col in range(cols):
169              index = row * cols + col
170              current_axis = (
171                  axes[row, col] if isinstance(axes, np.ndarray) else axes
172              )
173              current_axis.imshow(images[index].astype("uint8"))
174              current_axis.margins(x=0, y=0)
175              current_axis.axis("off")
176              title_parts = []
177              if y_true is not None and index < len(y_true):
178                  title_parts.append(
179                      f"Label: {label_map.get(y_true[index], 'Unknown')}"
180                  )
181              if y_pred is not None and index < len(y_pred):
182                  title_parts.append(
183                      f"Pred: {label_map.get(y_pred[index], 'Unknown')}"
184                  )
185  
186              if title_parts:
187                  current_axis.set_title("  ".join(title_parts), fontsize=8)
188  
189      if path is not None:
190          plt.savefig(
191              fname=path,
192              pad_inches=0,
193              bbox_inches="tight",
194              transparent=transparent,
195              dpi=dpi,
196          )
197          plt.close()
198      elif show:
199          plt.show()
200          plt.close()
