60    
61       GH 33987 
62       """

64       df1 = DataFrame(

66               "int": [1, 3],
67               "float": [2.0, np.nan],
68               "str": ["t", "s"],

71       )

74   
75       if format == "csv":

117  def assert_equal_zip_safe(result: bytes, expected: bytes, compression: str):
118      """ 
119      For zip compression, only compare the CRC-32 checksum of the file contents 
120      to avoid checking the time-dependent last-modified timestamp which 
121      in some CI builds is off-by-one 
122   
123      See https://en.wikipedia.org/wiki/ZIP_(file_format)#File_headers 
124      """
125      if compression == "zip":
126          # Only compare the CRC checksum of the file contents
127          with zipfile.ZipFile(BytesIO(result)) as exp, zipfile.ZipFile(
128              BytesIO(expected)
129          ) as res:
130              for res_info, exp_info in zip(res.infolist(), exp.infolist()):
131                  assert res_info.CRC == exp_info.CRC
132      elif compression == "tar":
133          with tarfile.open(fileobj=BytesIO(result)) as tar_exp, tarfile.open(
134              fileobj=BytesIO(expected)
135          ) as tar_res:
136              for tar_res_info, tar_exp_info in zip(
137                  tar_res.getmembers(), tar_exp.getmembers()
138              ):
139                  actual_file = tar_res.extractfile(tar_res_info)
140                  expected_file = tar_exp.extractfile(tar_exp_info)
141                  assert (actual_file is None) == (expected_file is None)
142                  if actual_file is not None and expected_file is not None:
143                      assert actual_file.read() == expected_file.read()
144      else:
145          assert result == expected

148  @pytest.mark.parametrize("encoding", ["utf-8", "cp1251"])
149  def test_to_csv_compression_encoding_gcs(
150      gcs_buffer, compression_only, encoding, compression_to_extension
151  ):
152      """ 
153      Compression and encoding should with GCS. 
154   
155      GH 35677 (to_csv, compression), GH 26124 (to_csv, encoding), and 
156      GH 32392 (read_csv, encoding) 
157      """
158      df = DataFrame(
159          1.1 * np.arange(120).reshape((30, 4)),
160          columns=Index(list("ABCD"), dtype=object),
161          index=Index([f"i-{i}" for i in range(30)], dtype=object),
162      )
163  
164      # reference of compressed and encoded file
165      compression = {"method": compression_only}
166      if compression_only == "gzip":
167          compression["mtime"] = 1  # be reproducible
168      buffer = BytesIO()
169      df.to_csv(buffer, compression=compression, encoding=encoding, mode="wb")
170  
171      # write compressed file with explicit compression
172      path_gcs = "gs://test/test.csv"
173      df.to_csv(path_gcs, compression=compression, encoding=encoding)
174      res = gcs_buffer.getvalue()
175      expected = buffer.getvalue()
176      assert_equal_zip_safe(res, expected, compression_only)
177  
178      read_df = read_csv(
179          path_gcs, index_col=0, compression=compression_only, encoding=encoding
180      )
181      tm.assert_frame_equal(df, read_df)
182  
183      # write compressed file with implicit compression
184      file_ext = compression_to_extension[compression_only]
185      compression["method"] = "infer"
186      path_gcs += f".{file_ext}"
187      df.to_csv(path_gcs, compression=compression, encoding=encoding)
188  
189      res = gcs_buffer.getvalue()
190      expected = buffer.getvalue()
191      assert_equal_zip_safe(res, expected, compression_only)
192  
193      read_df = read_csv(path_gcs, index_col=0, compression="infer", encoding=encoding)
194      tm.assert_frame_equal(df, read_df)
