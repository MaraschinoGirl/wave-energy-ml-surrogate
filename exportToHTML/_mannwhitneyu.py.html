10   def _broadcast_concatenate(x, y, axis):
11       '''Broadcast then concatenate arrays, leaving concatenation axis last'''
12       x = np.moveaxis(x, axis, -1)
13       y = np.moveaxis(y, axis, -1)
14       z = np.broadcast(x[..., 0], y[..., 0])
15       x = np.broadcast_to(x, z.shape + (x.shape[-1],))
16       y = np.broadcast_to(y, z.shape + (y.shape[-1],))
17       z = np.concatenate((x, y), axis=-1)
18       return x, y, z

21   class _MWU:
22       '''Distribution of MWU statistic under the null hypothesis'''
23   
24       def __init__(self, n1, n2):
25           self._reset(n1, n2)
26   
27       def set_shapes(self, n1, n2):
28           n1, n2 = min(n1, n2), max(n1, n2)
29           if (n1, n2) == (self.n1, self.n2):
30               return
31   
32           self.n1 = n1
33           self.n2 = n2
34           self.s_array = np.zeros(0, dtype=int)
35           self.configurations = np.zeros(0, dtype=np.uint64)
36   
37       def reset(self):
38           self._reset(self.n1, self.n2)
39   
40       def _reset(self, n1, n2):
41           self.n1 = None
42           self.n2 = None
43           self.set_shapes(n1, n2)
44   
45       def pmf(self, k):
46   
47           # In practice, `pmf` is never called with k > m*n/2.
48           # If it were, we'd exploit symmetry here:
49           # k = np.array(k, copy=True)
50           # k2 = m*n - k
51           # i = k2 < k
52           # k[i] = k2[i]
53   
54           pmfs = self.build_u_freqs_array(np.max(k))
55           return pmfs[k]
56   
57       def cdf(self, k):
58           '''Cumulative distribution function'''
59   
60           # In practice, `cdf` is never called with k > m*n/2.
61           # If it were, we'd exploit symmetry here rather than in `sf`
62           pmfs = self.build_u_freqs_array(np.max(k))
63           cdfs = np.cumsum(pmfs)
64           return cdfs[k]
65   
66       def sf(self, k):
67           '''Survival function'''
68           # Note that both CDF and SF include the PMF at k. The p-value is
69           # calculated from the SF and should include the mass at k, so this
70           # is desirable
71   
72           # Use the fact that the distribution is symmetric and sum from the left
73           kc = np.asarray(self.n1*self.n2 - k)  # complement of k
74           i = k < kc
75           if np.any(i):
76               kc[i] = k[i]
77               cdfs = np.asarray(self.cdf(kc))
78               cdfs[i] = 1. - cdfs[i] + self.pmf(kc[i])
79           else:
80               cdfs = np.asarray(self.cdf(kc))
81           return cdfs[()]
82   
83       # build_sigma_array and build_u_freqs_array adapted from code
84       # by @toobaz with permission. Thanks to @andreasloe for the suggestion.
85       # See https://github.com/scipy/scipy/pull/4933#issuecomment-1898082691
86       def build_sigma_array(self, a):
87           n1, n2 = self.n1, self.n2
88           if a + 1 <= self.s_array.size:
89               return self.s_array[1:a+1]
90   
91           s_array = np.zeros(a + 1, dtype=int)
92   
93           for d in np.arange(1, n1 + 1):
94               # All multiples of d, except 0:
95               indices = np.arange(d, a + 1, d)
96               # \epsilon_d = 1:
97               s_array[indices] += d
98   
99           for d in np.arange(n2 + 1, n2 + n1 + 1):
100              # All multiples of d, except 0:
101              indices = np.arange(d, a + 1, d)
102              # \epsilon_d = -1:
103              s_array[indices] -= d
104  
105          # We don't need 0:
106          self.s_array = s_array
107          return s_array[1:]
108  
109      def build_u_freqs_array(self, maxu):
110          """ 
111          Build all the array of frequencies for u from 0 to maxu. 
112          Assumptions: 
113            n1 <= n2 
114            maxu <= n1 * n2 / 2 
115          """
116          n1, n2 = self.n1, self.n2
117          total = special.binom(n1 + n2, n1)
118  
119          if maxu + 1 <= self.configurations.size:
120              return self.configurations[:maxu + 1] / total
121  
122          s_array = self.build_sigma_array(maxu)
123  
124          # Start working with ints, for maximum precision and efficiency:
125          configurations = np.zeros(maxu + 1, dtype=np.uint64)
126          configurations_is_uint = True
127          uint_max = np.iinfo(np.uint64).max
128          # How many ways to have U=0? 1
129          configurations[0] = 1
130  
131          for u in np.arange(1, maxu + 1):
132              coeffs = s_array[u - 1::-1]
133              new_val = np.dot(configurations[:u], coeffs) / u
134              if new_val > uint_max and configurations_is_uint:
135                  # OK, we got into numbers too big for uint64.
136                  # So now we start working with floats.
137                  # By doing this since the beginning, we would have lost precision.
138                  # (And working on python long ints would be unbearably slow)
139                  configurations = configurations.astype(float)
140                  configurations_is_uint = False
141              configurations[u] = new_val
142  
143          self.configurations = configurations
144          return configurations / total

153  def _get_mwu_z(U, n1, n2, t, axis=0, continuity=True):
154      '''Standardized MWU statistic'''
155      # Follows mannwhitneyu [2]
156      mu = n1 * n2 / 2
157      n = n1 + n2
158  
159      # Tie correction according to [2], "Normal approximation and tie correction"
160      # "A more computationally-efficient form..."
161      tie_term = (t**3 - t).sum(axis=-1)
162      s = np.sqrt(n1*n2/12 * ((n + 1) - tie_term/(n*(n-1))))
163  
164      numerator = U - mu
165  
166      # Continuity correction.
167      # Because SF is always used to calculate the p-value, we can always
168      # _subtract_ 0.5 for the continuity correction. This always increases the
169      # p-value to account for the rest of the probability mass _at_ q = U.
170      if continuity:
171          numerator -= 0.5
172  
173      # no problem evaluating the norm SF at an infinity
174      with np.errstate(divide='ignore', invalid='ignore'):
175          z = numerator / s
176      return z

179  def _mwu_input_validation(x, y, use_continuity, alternative, axis, method):
180      ''' Input validation and standardization for mannwhitneyu '''
181      # Would use np.asarray_chkfinite, but infs are OK
182      x, y = np.atleast_1d(x), np.atleast_1d(y)
183      if np.isnan(x).any() or np.isnan(y).any():
184          raise ValueError('`x` and `y` must not contain NaNs.')
185      if np.size(x) == 0 or np.size(y) == 0:
186          raise ValueError('`x` and `y` must be of nonzero size.')
187  
188      bools = {True, False}
189      if use_continuity not in bools:
190          raise ValueError(f'`use_continuity` must be one of {bools}.')
191  
192      alternatives = {"two-sided", "less", "greater"}
193      alternative = alternative.lower()
194      if alternative not in alternatives:
195          raise ValueError(f'`alternative` must be one of {alternatives}.')
196  
197      axis_int = int(axis)
198      if axis != axis_int:
199          raise ValueError('`axis` must be an integer.')
200  
201      if not isinstance(method, stats.PermutationMethod):
202          methods = {"asymptotic", "exact", "auto"}
203          method = method.lower()
204          if method not in methods:
205              raise ValueError(f'`method` must be one of {methods}.')
206  
207      return x, y, use_continuity, alternative, axis_int, method

210  def _mwu_choose_method(n1, n2, ties):
211      """Choose method 'asymptotic' or 'exact' depending on input size, ties"""
212  
213      # if both inputs are large, asymptotic is OK
214      if n1 > 8 and n2 > 8:
215          return "asymptotic"
216  
217      # if there are any ties, asymptotic is preferred
218      if ties:
219          return "asymptotic"
220  
221      return "exact"

227  @_axis_nan_policy_factory(MannwhitneyuResult, n_samples=2)
228  def mannwhitneyu(x, y, use_continuity=True, alternative="two-sided",
229                   axis=0, method="auto"):
230      r'''Perform the Mann-Whitney U rank test on two independent samples. 
231   
232      The Mann-Whitney U test is a nonparametric test of the null hypothesis 
233      that the distribution underlying sample `x` is the same as the 
234      distribution underlying sample `y`. It is often used as a test of 
235      difference in location between distributions. 
236   
237      Parameters 
238      ---------- 
239      x, y : array-like 
240          N-d arrays of samples. The arrays must be broadcastable except along 
241          the dimension given by `axis`. 
242      use_continuity : bool, optional 
243              Whether a continuity correction (1/2) should be applied. 
244              Default is True when `method` is ``'asymptotic'``; has no effect 
245              otherwise. 
246      alternative : {'two-sided', 'less', 'greater'}, optional 
247          Defines the alternative hypothesis. Default is 'two-sided'. 
248          Let *SX(u)* and *SY(u)* be the survival functions of the 
249          distributions underlying `x` and `y`, respectively. Then the following 
250          alternative hypotheses are available: 
251   
252          * 'two-sided': the distributions are not equal, i.e. *SX(u) ≠ SY(u)* for 
253            at least one *u*. 
254          * 'less': the distribution underlying `x` is stochastically less 
255            than the distribution underlying `y`, i.e. *SX(u) < SY(u)* for all *u*. 
256          * 'greater': the distribution underlying `x` is stochastically greater 
257            than the distribution underlying `y`, i.e. *SX(u) > SY(u)* for all *u*. 
258   
259          Under a more restrictive set of assumptions, the alternative hypotheses 
260          can be expressed in terms of the locations of the distributions; 
261          see [5]_ section 5.1. 
262      axis : int, optional 
263          Axis along which to perform the test. Default is 0. 
264      method : {'auto', 'asymptotic', 'exact'} or `PermutationMethod` instance, optional 
265          Selects the method used to calculate the *p*-value. 
266          Default is 'auto'. The following options are available. 
267   
268          * ``'asymptotic'``: compares the standardized test statistic 
269            against the normal distribution, correcting for ties. 
270          * ``'exact'``: computes the exact *p*-value by comparing the observed 
271            :math:`U` statistic against the exact distribution of the :math:`U` 
272            statistic under the null hypothesis. No correction is made for ties. 
273          * ``'auto'``: chooses ``'exact'`` when the size of one of the samples 
274            is less than or equal to 8 and there are no ties; 
275            chooses ``'asymptotic'`` otherwise. 
276          * `PermutationMethod` instance. In this case, the p-value 
277            is computed using `permutation_test` with the provided 
278            configuration options and other appropriate settings. 
279   
280      Returns 
281      ------- 
282      res : MannwhitneyuResult 
283          An object containing attributes: 
284   
285          statistic : float 
286              The Mann-Whitney U statistic corresponding with sample `x`. See 
287              Notes for the test statistic corresponding with sample `y`. 
288          pvalue : float 
289              The associated *p*-value for the chosen `alternative`. 
290   
291      Notes 
292      ----- 
293      If ``U1`` is the statistic corresponding with sample `x`, then the 
294      statistic corresponding with sample `y` is 
295      ``U2 = x.shape[axis] * y.shape[axis] - U1``. 
296   
297      `mannwhitneyu` is for independent samples. For related / paired samples, 
298      consider `scipy.stats.wilcoxon`. 
299   
300      `method` ``'exact'`` is recommended when there are no ties and when either 
301      sample size is less than 8 [1]_. The implementation follows the algorithm 
302      reported in [3]_. 
303      Note that the exact method is *not* corrected for ties, but 
304      `mannwhitneyu` will not raise errors or warnings if there are ties in the 
305      data. If there are ties and either samples is small (fewer than ~10 
306      observations), consider passing an instance of `PermutationMethod` 
307      as the `method` to perform a permutation test. 
308   
309      The Mann-Whitney U test is a non-parametric version of the t-test for 
310      independent samples. When the means of samples from the populations 
311      are normally distributed, consider `scipy.stats.ttest_ind`. 
312   
313      See Also 
314      -------- 
315      scipy.stats.wilcoxon, scipy.stats.ranksums, scipy.stats.ttest_ind 
316   
317      References 
318      ---------- 
319      .. [1] H.B. Mann and D.R. Whitney, "On a test of whether one of two random 
320             variables is stochastically larger than the other", The Annals of 
321             Mathematical Statistics, Vol. 18, pp. 50-60, 1947. 
322      .. [2] Mann-Whitney U Test, Wikipedia, 
323             http://en.wikipedia.org/wiki/Mann-Whitney_U_test 
324      .. [3] Andreas Löffler, 
325             "Über eine Partition der nat. Zahlen und ihr Anwendung beim U-Test", 
326             Wiss. Z. Univ. Halle, XXXII'83 pp. 87-89. 
327      .. [4] Rosie Shier, "Statistics: 2.3 The Mann-Whitney U Test", Mathematics 
328             Learning Support Centre, 2004. 
329      .. [5] Michael P. Fay and Michael A. Proschan. "Wilcoxon-Mann-Whitney 
330             or t-test? On assumptions for hypothesis tests and multiple \ 
331             interpretations of decision rules." Statistics surveys, Vol. 4, pp. 
332             1-39, 2010. https://www.ncbi.nlm.nih.gov/pmc/articles/PMC2857732/ 
333   
334      Examples 
335      -------- 
336      We follow the example from [4]_: nine randomly sampled young adults were 
337      diagnosed with type II diabetes at the ages below. 
338   
339      >>> males = [19, 22, 16, 29, 24] 
340      >>> females = [20, 11, 17, 12] 
341   
342      We use the Mann-Whitney U test to assess whether there is a statistically 
343      significant difference in the diagnosis age of males and females. 
344      The null hypothesis is that the distribution of male diagnosis ages is 
345      the same as the distribution of female diagnosis ages. We decide 
346      that a confidence level of 95% is required to reject the null hypothesis 
347      in favor of the alternative that the distributions are different. 
348      Since the number of samples is very small and there are no ties in the 
349      data, we can compare the observed test statistic against the *exact* 
350      distribution of the test statistic under the null hypothesis. 
351   
352      >>> from scipy.stats import mannwhitneyu 
353      >>> U1, p = mannwhitneyu(males, females, method="exact") 
354      >>> print(U1) 
355      17.0 
356   
357      `mannwhitneyu` always reports the statistic associated with the first 
358      sample, which, in this case, is males. This agrees with :math:`U_M = 17` 
359      reported in [4]_. The statistic associated with the second statistic 
360      can be calculated: 
361   
362      >>> nx, ny = len(males), len(females) 
363      >>> U2 = nx*ny - U1 
364      >>> print(U2) 
365      3.0 
366   
367      This agrees with :math:`U_F = 3` reported in [4]_. The two-sided 
368      *p*-value can be calculated from either statistic, and the value produced 
369      by `mannwhitneyu` agrees with :math:`p = 0.11` reported in [4]_. 
370   
371      >>> print(p) 
372      0.1111111111111111 
373   
374      The exact distribution of the test statistic is asymptotically normal, so 
375      the example continues by comparing the exact *p*-value against the 
376      *p*-value produced using the normal approximation. 
377   
378      >>> _, pnorm = mannwhitneyu(males, females, method="asymptotic") 
379      >>> print(pnorm) 
380      0.11134688653314041 
381   
382      Here `mannwhitneyu`'s reported *p*-value appears to conflict with the 
383      value :math:`p = 0.09` given in [4]_. The reason is that [4]_ 
384      does not apply the continuity correction performed by `mannwhitneyu`; 
385      `mannwhitneyu` reduces the distance between the test statistic and the 
386      mean :math:`\mu = n_x n_y / 2` by 0.5 to correct for the fact that the 
387      discrete statistic is being compared against a continuous distribution. 
388      Here, the :math:`U` statistic used is less than the mean, so we reduce 
389      the distance by adding 0.5 in the numerator. 
390   
391      >>> import numpy as np 
392      >>> from scipy.stats import norm 
393      >>> U = min(U1, U2) 
394      >>> N = nx + ny 
395      >>> z = (U - nx*ny/2 + 0.5) / np.sqrt(nx*ny * (N + 1)/ 12) 
396      >>> p = 2 * norm.cdf(z)  # use CDF to get p-value from smaller statistic 
397      >>> print(p) 
398      0.11134688653314041 
399   
400      If desired, we can disable the continuity correction to get a result 
401      that agrees with that reported in [4]_. 
402   
403      >>> _, pnorm = mannwhitneyu(males, females, use_continuity=False, 
404      ...                         method="asymptotic") 
405      >>> print(pnorm) 
406      0.0864107329737 
407   
408      Regardless of whether we perform an exact or asymptotic test, the 
409      probability of the test statistic being as extreme or more extreme by 
410      chance exceeds 5%, so we do not consider the results statistically 
411      significant. 
412   
413      Suppose that, before seeing the data, we had hypothesized that females 
414      would tend to be diagnosed at a younger age than males. 
415      In that case, it would be natural to provide the female ages as the 
416      first input, and we would have performed a one-sided test using 
417      ``alternative = 'less'``: females are diagnosed at an age that is 
418      stochastically less than that of males. 
419   
420      >>> res = mannwhitneyu(females, males, alternative="less", method="exact") 
421      >>> print(res) 
422      MannwhitneyuResult(statistic=3.0, pvalue=0.05555555555555555) 
423   
424      Again, the probability of getting a sufficiently low value of the 
425      test statistic by chance under the null hypothesis is greater than 5%, 
426      so we do not reject the null hypothesis in favor of our alternative. 
427   
428      If it is reasonable to assume that the means of samples from the 
429      populations are normally distributed, we could have used a t-test to 
430      perform the analysis. 
431   
432      >>> from scipy.stats import ttest_ind 
433      >>> res = ttest_ind(females, males, alternative="less") 
434      >>> print(res) 
435      TtestResult(statistic=-2.239334696520584, 
436                  pvalue=0.030068441095757924, 
437                  df=7.0) 
438   
439      Under this assumption, the *p*-value would be low enough to reject the 
440      null hypothesis in favor of the alternative. 
441   
442      '''
443  
444      x, y, use_continuity, alternative, axis_int, method = (
445          _mwu_input_validation(x, y, use_continuity, alternative, axis, method))
446  
447      x, y, xy = _broadcast_concatenate(x, y, axis)
448  
449      n1, n2 = x.shape[-1], y.shape[-1]
450  
451      # Follows [2]
452      ranks, t = _rankdata(xy, 'average', return_ties=True)  # method 2, step 1
453      R1 = ranks[..., :n1].sum(axis=-1)                      # method 2, step 2
454      U1 = R1 - n1*(n1+1)/2                                  # method 2, step 3
455      U2 = n1 * n2 - U1                                      # as U1 + U2 = n1 * n2
456  
457      if alternative == "greater":
458          U, f = U1, 1  # U is the statistic to use for p-value, f is a factor
459      elif alternative == "less":
460          U, f = U2, 1  # Due to symmetry, use SF of U2 rather than CDF of U1
461      else:
462          U, f = np.maximum(U1, U2), 2  # multiply SF by two for two-sided test
463  
464      if method == "auto":
465          method = _mwu_choose_method(n1, n2, np.any(t > 1))
466  
467      if method == "exact":
468          if not hasattr(_mwu_state, 's'):
469              _mwu_state.s = _MWU(0, 0)
470          _mwu_state.s.set_shapes(n1, n2)
471          p = _mwu_state.s.sf(U.astype(int))
472      elif method == "asymptotic":
473          z = _get_mwu_z(U, n1, n2, t, continuity=use_continuity)
474          p = stats.norm.sf(z)
475      else:  # `PermutationMethod` instance (already validated)
476          def statistic(x, y, axis):
477              return mannwhitneyu(x, y, use_continuity=use_continuity,
478                                  alternative=alternative, axis=axis,
479                                  method="asymptotic").statistic
480  
481          res = stats.permutation_test((x, y), statistic, axis=axis,
482                                       **method._asdict(), alternative=alternative)
483          p = res.pvalue
484          f = 1
485  
486      p *= f
487  
488      # Ensure that test statistic is not greater than 1
489      # This could happen for exact test when U = m*n/2
490      p = np.clip(p, 0, 1)
491  
492      return MannwhitneyuResult(U1, p)
