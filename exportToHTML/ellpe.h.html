5    /*                                                     ellpe.c 
6     * 
7     *     Complete elliptic integral of the second kind 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double m, y, ellpe(); 
14    * 
15    * y = ellpe( m ); 
16    * 
17    * 
18    * 
19    * DESCRIPTION: 
20    * 
21    * Approximates the integral 
22    * 
23    * 
24    *            pi/2 
25    *             - 
26    *            | |                 2 
27    * E(m)  =    |    sqrt( 1 - m sin t ) dt 
28    *          | | 
29    *           - 
30    *            0 
31    * 
32    * Where m = 1 - m1, using the approximation 
33    * 
34    *      P(x)  -  x log x Q(x). 
35    * 
36    * Though there are no singularities, the argument m1 is used 
37    * internally rather than m for compatibility with ellpk(). 
38    * 
39    * E(1) = 1; E(0) = pi/2. 
40    * 
41    * 
42    * ACCURACY: 
43    * 
44    *                      Relative error: 
45    * arithmetic   domain     # trials      peak         rms 
46    *    IEEE       0, 1       10000       2.1e-16     7.3e-17 
47    * 
48    * 
49    * ERROR MESSAGES: 
50    * 
51    *   message         condition      value returned 
52    * ellpe domain      x<0, x>1            0.0 
53    * 
54    */

92       XSF_HOST_DEVICE inline double ellpe(double x) {
93           x = 1.0 - x;
94           if (x <= 0.0) {
95               if (x == 0.0)
96                   return (1.0);
97               set_error("ellpe", SF_ERROR_DOMAIN, NULL);
98               return (std::numeric_limits<double>::quiet_NaN());
99           }
100          if (x > 1.0) {
101              return ellpe(1.0 - 1 / x) * std::sqrt(x);
102          }
103          return (polevl(x, detail::ellpe_P, 10) - std::log(x) * (x * polevl(x, detail::ellpe_Q, 9)));
104      }
