1    import collections
2    import csv
3    
4    import numpy as np
5    
6    from keras.src.api_export import keras_export
7    from keras.src.callbacks.callback import Callback
8    from keras.src.utils import file_utils
9    
10   
11   @keras_export("keras.callbacks.CSVLogger")
12   class CSVLogger(Callback):
13       """Callback that streams epoch results to a CSV file. 
14    
15       Supports all values that can be represented as a string, 
16       including 1D iterables such as `np.ndarray`. 
17    
18       Args: 
19           filename: Filename of the CSV file, e.g. `'run/log.csv'`. 
20           separator: String used to separate elements in the CSV file. 
21           append: Boolean. True: append if file exists (useful for continuing 
22               training). False: overwrite existing file. 
23    
24       Example: 
25    
26       ```python 
27       csv_logger = CSVLogger('training.log') 
28       model.fit(X_train, Y_train, callbacks=[csv_logger]) 
29       ``` 
30       """
31   
32       def __init__(self, filename, separator=",", append=False):
33           super().__init__()
34           self.sep = separator
35           self.filename = file_utils.path_to_string(filename)
36           self.append = append
37           self.writer = None
38           self.keys = None
39           self.append_header = True
40   
41       def on_train_begin(self, logs=None):
42           if self.append:
43               if file_utils.exists(self.filename):
44                   with file_utils.File(self.filename, "r") as f:
45                       self.append_header = not bool(len(f.readline()))
46               mode = "a"
47           else:
48               mode = "w"
49           self.csv_file = file_utils.File(self.filename, mode)
50   
51       def on_epoch_end(self, epoch, logs=None):
52           logs = logs or {}
53   
54           def handle_value(k):
55               is_zero_dim_ndarray = isinstance(k, np.ndarray) and k.ndim == 0
56               if isinstance(k, str):
57                   return k
58               elif (
59                   isinstance(k, collections.abc.Iterable)
60                   and not is_zero_dim_ndarray
61               ):
62                   return f'"[{", ".join(map(str, k))}]"'
63               else:
64                   return k
65   
66           if self.keys is None:
67               self.keys = sorted(logs.keys())
68               # When validation_freq > 1, `val_` keys are not in first epoch logs
69               # Add the `val_` keys so that its part of the fieldnames of writer.
70               val_keys_found = False
71               for key in self.keys:
72                   if key.startswith("val_"):
73                       val_keys_found = True
74                       break
75               if not val_keys_found:
76                   self.keys.extend(["val_" + k for k in self.keys])
77   
78           if not self.writer:
79   
80               class CustomDialect(csv.excel):
81                   delimiter = self.sep
82   
83               fieldnames = ["epoch"] + self.keys
84   
85               self.writer = csv.DictWriter(
86                   self.csv_file, fieldnames=fieldnames, dialect=CustomDialect
87               )
88               if self.append_header:
89                   self.writer.writeheader()
90   
91           row_dict = collections.OrderedDict({"epoch": epoch})
92           row_dict.update(
93               (key, handle_value(logs.get(key, "NA"))) for key in self.keys
94           )
95           self.writer.writerow(row_dict)
96           self.csv_file.flush()
97   
98       def on_train_end(self, logs=None):
99           self.csv_file.close()
100          self.writer = None
