1    """Test the old numpy pickler, compatibility version."""
2    
3    # numpy_pickle is not a drop-in replacement of pickle, as it takes
4    # filenames instead of open files as arguments.
5    from joblib import numpy_pickle_compat
6    
7    
8    def test_z_file(tmpdir):
9        # Test saving and loading data with Zfiles.
10       filename = tmpdir.join('test.pkl').strpath
11       data = numpy_pickle_compat.asbytes('Foo, \n Bar, baz, \n\nfoobar')
12       with open(filename, 'wb') as f:
13           numpy_pickle_compat.write_zfile(f, data)
14       with open(filename, 'rb') as f:
15           data_read = numpy_pickle_compat.read_zfile(f)
16       assert data == data_read
