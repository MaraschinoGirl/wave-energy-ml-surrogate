12   def delete_old_compiled_extensions():
13       import _pydevd_bundle
14   
15       cython_extensions_dir = os.path.dirname(os.path.dirname(_pydevd_bundle.__file__))
16       _pydevd_bundle_ext_dir = os.path.dirname(_pydevd_bundle.__file__)
17       _pydevd_frame_eval_ext_dir = os.path.join(cython_extensions_dir, "_pydevd_frame_eval_ext")
18       try:
19           import shutil
20   
21           for file in os.listdir(_pydevd_bundle_ext_dir):
22               if file.startswith("pydevd") and file.endswith(".so"):
23                   os.remove(os.path.join(_pydevd_bundle_ext_dir, file))
24           for file in os.listdir(_pydevd_frame_eval_ext_dir):
25               if file.startswith("pydevd") and file.endswith(".so"):
26                   os.remove(os.path.join(_pydevd_frame_eval_ext_dir, file))
27           build_dir = os.path.join(cython_extensions_dir, "build")
28           if os.path.exists(build_dir):
29               shutil.rmtree(os.path.join(cython_extensions_dir, "build"))
30       except OSError:
31           pydev_log.error_once(
32               "warning: failed to delete old cython speedups. Please delete all *.so files from the directories "
33               '"%s" and "%s"' % (_pydevd_bundle_ext_dir, _pydevd_frame_eval_ext_dir)
34           )
