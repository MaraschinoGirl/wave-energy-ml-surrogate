18   def balanced(**kwargs):
19       """balanced(factor_name=num_levels, [factor_name=num_levels, ..., repeat=1]) 
20    
21       Create simple balanced factorial designs for testing. 
22    
23       Given some factor names and the number of desired levels for each, 
24       generates a balanced factorial design in the form of a data 
25       dictionary. For example: 
26    
27       .. ipython:: 
28    
29          In [1]: balanced(a=2, b=3) 
30          Out[1]: 
31          {'a': ['a1', 'a1', 'a1', 'a2', 'a2', 'a2'], 
32           'b': ['b1', 'b2', 'b3', 'b1', 'b2', 'b3']} 
33    
34       By default it produces exactly one instance of each combination of levels, 
35       but if you want multiple replicates this can be accomplished via the 
36       `repeat` argument: 
37    
38       .. ipython:: 
39    
40          In [2]: balanced(a=2, b=2, repeat=2) 
41          Out[2]: 
42          {'a': ['a1', 'a1', 'a2', 'a2', 'a1', 'a1', 'a2', 'a2'], 
43           'b': ['b1', 'b2', 'b1', 'b2', 'b1', 'b2', 'b1', 'b2']} 
44       """
45       repeat = kwargs.pop("repeat", 1)
46       levels = []
47       names = sorted(kwargs)
48       for name in names:
49           level_count = kwargs[name]
50           levels.append(["%s%s" % (name, i) for i in range(1, level_count + 1)])
51       # zip(*...) does an "unzip"
52       values = zip(*itertools.product(*levels))
53       data = {}
54       for name, value in zip(names, values):
55           data[name] = list(value) * repeat
56       return data

94   def demo_data(*names, **kwargs):
95       """demo_data(*names, nlevels=2, min_rows=5) 
96    
97       Create simple categorical/numerical demo data. 
98    
99       Pass in a set of variable names, and this function will return a simple 
100      data set using those variable names. 
101   
102      Names whose first letter falls in the range "a" through "m" will be made 
103      categorical (with `nlevels` levels). Those that start with a "p" through 
104      "z" are numerical. 
105   
106      We attempt to produce a balanced design on the categorical variables, 
107      repeating as necessary to generate at least `min_rows` data 
108      points. Categorical variables are returned as a list of strings. 
109   
110      Numerical data is generated by sampling from a normal distribution. A 
111      fixed random seed is used, so that identical calls to demo_data() will 
112      produce identical results. Numerical data is returned in a numpy array. 
113   
114      Example: 
115   
116      .. ipython: 
117   
118         In [1]: patsy.demo_data("a", "b", "x", "y") 
119         Out[1]: 
120         {'a': ['a1', 'a1', 'a2', 'a2', 'a1', 'a1', 'a2', 'a2'], 
121          'b': ['b1', 'b2', 'b1', 'b2', 'b1', 'b2', 'b1', 'b2'], 
122          'x': array([ 1.76405235,  0.40015721,  0.97873798,  2.2408932 , 
123                       1.86755799, -0.97727788,  0.95008842, -0.15135721]), 
124          'y': array([-0.10321885,  0.4105985 ,  0.14404357,  1.45427351, 
125                       0.76103773,  0.12167502,  0.44386323,  0.33367433])} 
126      """
127      nlevels = kwargs.pop("nlevels", 2)
128      min_rows = kwargs.pop("min_rows", 5)
129      if kwargs:
130          raise TypeError("unexpected keyword arguments %r" % (kwargs,))
131      numerical = set()
132      categorical = {}
133      for name in names:
134          if name[0] in "abcdefghijklmn":
135              categorical[name] = nlevels
136          elif name[0] in "pqrstuvwxyz":
137              numerical.add(name)
138          else:
139              raise PatsyError("bad name %r" % (name,))
140      balanced_design_size = np.prod(list(categorical.values()), dtype=int)
141      repeat = int(np.ceil(min_rows * 1.0 / balanced_design_size))
142      num_rows = repeat * balanced_design_size
143      data = balanced(repeat=repeat, **categorical)
144      r = np.random.RandomState(0)
145      for name in sorted(numerical):
146          data[name] = r.normal(size=num_rows)
147      return data

150  def test_demo_data():
151      d1 = demo_data("a", "b", "x")
152      assert sorted(d1.keys()) == ["a", "b", "x"]
153      assert d1["a"] == ["a1", "a1", "a2", "a2", "a1", "a1", "a2", "a2"]
154      assert d1["b"] == ["b1", "b2", "b1", "b2", "b1", "b2", "b1", "b2"]
155      assert d1["x"].dtype == np.dtype(float)
156      assert d1["x"].shape == (8,)
157  
158      d2 = demo_data("x", "y")
159      assert sorted(d2.keys()) == ["x", "y"]
160      assert len(d2["x"]) == len(d2["y"]) == 5
161  
162      assert len(demo_data("x", min_rows=10)["x"]) == 10
163      assert len(demo_data("a", "b", "x", min_rows=10)["x"]) == 12
164      assert len(demo_data("a", "b", "x", min_rows=10, nlevels=3)["x"]) == 18

172  class LookupFactor(object):
173      """A simple factor class that simply looks up a named entry in the given 
174      data. 
175   
176      Useful for programatically constructing formulas, and as a simple example 
177      of the factor protocol.  For details see 
178      :ref:`expert-model-specification`. 
179   
180      Example:: 
181   
182        dmatrix(ModelDesc([], [Term([LookupFactor("x")])]), {"x": [1, 2, 3]}) 
183   
184      :arg varname: The name of this variable; used as a lookup key in the 
185        passed in data dictionary/DataFrame/whatever. 
186      :arg force_categorical: If True, then treat this factor as 
187        categorical. (Equivalent to using :func:`C` in a regular formula, but 
188        of course you can't do that with a :class:`LookupFactor`. 
189      :arg contrast: If given, the contrast to use; see :func:`C`. (Requires 
190        ``force_categorical=True``.) 
191      :arg levels: If given, the categorical levels; see :func:`C`. (Requires 
192        ``force_categorical=True``.) 
193      :arg origin: Either ``None``, or the :class:`Origin` of this factor for use 
194        in error reporting. 
195   
196      .. versionadded:: 0.2.0 
197         The ``force_categorical`` and related arguments. 
198      """
199  
200      def __init__(
201          self, varname, force_categorical=False, contrast=None, levels=None, origin=None
202      ):
203          self._varname = varname
204          self._force_categorical = force_categorical
205          self._contrast = contrast
206          self._levels = levels
207          self.origin = origin
208          if not self._force_categorical:
209              if contrast is not None:
210                  raise ValueError("contrast= requires force_categorical=True")
211              if levels is not None:
212                  raise ValueError("levels= requires force_categorical=True")
213  
214      def name(self):
215          return self._varname
216  
217      def __repr__(self):
218          return "%s(%r)" % (self.__class__.__name__, self._varname)
219  
220      def __eq__(self, other):
221          return (
222              isinstance(other, LookupFactor)
223              and self._varname == other._varname
224              and self._force_categorical == other._force_categorical
225              and self._contrast == other._contrast
226              and self._levels == other._levels
227          )
228  
229      def __ne__(self, other):
230          return not self == other
231  
232      def __hash__(self):
233          return hash(
234              (
235                  LookupFactor,
236                  self._varname,
237                  self._force_categorical,
238                  self._contrast,
239                  self._levels,
240              )
241          )
242  
243      def memorize_passes_needed(self, state, eval_env):
244          return 0
245  
246      def memorize_chunk(self, state, which_pass, data):  # pragma: no cover
247          assert False
248  
249      def memorize_finish(self, state, which_pass):  # pragma: no cover
250          assert False
251  
252      def eval(self, memorize_state, data):
253          value = data[self._varname]
254          if self._force_categorical:
255              value = C(value, contrast=self._contrast, levels=self._levels)
256          return value
257  
258      __getstate__ = no_pickling

261  def test_LookupFactor():
262      l_a = LookupFactor("a")
263      assert l_a.name() == "a"
264      assert l_a == LookupFactor("a")
265      assert l_a != LookupFactor("b")
266      assert hash(l_a) == hash(LookupFactor("a"))
267      assert hash(l_a) != hash(LookupFactor("b"))
268      assert l_a.eval({}, {"a": 1}) == 1
269      assert l_a.eval({}, {"a": 2}) == 2
270      assert repr(l_a) == "LookupFactor('a')"
271      assert l_a.origin is None
272      l_with_origin = LookupFactor("b", origin="asdf")
273      assert l_with_origin.origin == "asdf"
274  
275      l_c = LookupFactor("c", force_categorical=True, contrast="CONTRAST", levels=(1, 2))
276      box = l_c.eval({}, {"c": [1, 1, 2]})
277      assert box.data == [1, 1, 2]
278      assert box.contrast == "CONTRAST"
279      assert box.levels == (1, 2)
